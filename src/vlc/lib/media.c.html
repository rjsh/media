101/**************************************************************************
102 * input_item_subitem_added (Private) (vlc event Callback)
103 **************************************************************************/
104static void input_item_subitem_added( const vlc_event_t *p_event,
105 void * user_data )
106{
107 libvlc_media_t * p_md = user_data;
108 libvlc_media_t * p_md_child;
109 libvlc_event_t event;
110
111 p_md_child = libvlc_media_new_from_input_item(
112 p_md->p_libvlc_instance,
113 p_event->u.input_item_subitem_added.p_new_child );
114
115 /* Add this to our media list */
116 if( p_md->p_subitems == NULL )
117 {
118 p_md->p_subitems = libvlc_media_list_new( p_md->p_libvlc_instance );
119 if( unlikely(p_md->p_subitems == NULL) )
120 abort();
121 libvlc_media_list_set_media( p_md->p_subitems, p_md );
122 }
123 libvlc_media_list_add_media( p_md->p_subitems, p_md_child );
124
125 /* Construct the event */
126 event.type = libvlc_MediaSubItemAdded;
127 event.u.media_subitem_added.new_child = p_md_child;
128
129 /* Send the event */
130 libvlc_event_send( p_md->p_event_manager, &event );
131 libvlc_media_release( p_md_child );
132}

188/**************************************************************************
189 * input_item_preparsed_changed (Private) (vlc event Callback)
190 **************************************************************************/
191static void input_item_preparsed_changed(const vlc_event_t *p_event,
192 void * user_data)
193{
194 libvlc_media_t *media = user_data;
195 libvlc_event_t event;
196
197 /* Eventually notify libvlc_media_parse() */
198 vlc_mutex_lock(&media->parsed_lock);
199 media->is_parsed = true;
200 vlc_cond_broadcast(&media->parsed_cond);
201 vlc_mutex_unlock(&media->parsed_lock);
202
203
204 /* Construct the event */
205 event.type = libvlc_MediaParsedChanged;
206 event.u.media_parsed_changed.new_status =
207 p_event->u.input_item_preparsed_changed.new_status;
208
209 /* Send the event */
210 libvlc_event_send(media->p_event_manager, &event);
211}

213/**************************************************************************
214 * Install event handler (Private)
215 **************************************************************************/
216static void install_input_item_observer( libvlc_media_t *p_md )
217{
218 vlc_event_attach( &p_md->p_input_item->event_manager,
219 vlc_InputItemSubItemAdded,
220 input_item_subitem_added,
221 p_md );
222 vlc_event_attach( &p_md->p_input_item->event_manager,
223 vlc_InputItemMetaChanged,
224 input_item_meta_changed,
225 p_md );
226 vlc_event_attach( &p_md->p_input_item->event_manager,
227 vlc_InputItemDurationChanged,
228 input_item_duration_changed,
229 p_md );
230 vlc_event_attach( &p_md->p_input_item->event_manager,
231 vlc_InputItemPreparsedChanged,
232 input_item_preparsed_changed,
233 p_md );
234 vlc_event_attach( &p_md->p_input_item->event_manager,
235 vlc_InputItemSubItemTreeAdded,
236 input_item_subitemtree_added,
237 p_md );
238}

267/**************************************************************************
268 * Create a new media descriptor object from an input_item
269 * (libvlc internal)
270 * That's the generic constructor
271 **************************************************************************/
272libvlc_media_t * libvlc_media_new_from_input_item(
273 libvlc_instance_t *p_instance,
274 input_item_t *p_input_item )
275{
276 libvlc_media_t * p_md;
277
278 if (!p_input_item)
279 {
280 libvlc_printerr( "No input item given" );
281 return NULL;
282 }
283
284 p_md = calloc( 1, sizeof(libvlc_media_t) );
285 if( !p_md )
286 {
287 libvlc_printerr( "Not enough memory" );
288 return NULL;
289 }
290
291 p_md->p_libvlc_instance = p_instance;
292 p_md->p_input_item = p_input_item;
293 p_md->i_refcount = 1;
294
295 vlc_cond_init(&p_md->parsed_cond);
296 vlc_mutex_init(&p_md->parsed_lock);
297
298 p_md->state = libvlc_NothingSpecial;
299
300 /* A media descriptor can be a playlist. When you open a playlist
301 * It can give a bunch of item to read. */
302 p_md->p_subitems = NULL;
303
304 p_md->p_event_manager = libvlc_event_manager_new( p_md, p_instance );
305 if( unlikely(p_md->p_event_manager == NULL) )
306 {
307 free(p_md);
308 return NULL;
309 }
310
311 libvlc_event_manager_t *em = p_md->p_event_manager;
312 libvlc_event_manager_register_event_type(em, libvlc_MediaMetaChanged);
313 libvlc_event_manager_register_event_type(em, libvlc_MediaSubItemAdded);
314 libvlc_event_manager_register_event_type(em, libvlc_MediaFreed);
315 libvlc_event_manager_register_event_type(em, libvlc_MediaDurationChanged);
316 libvlc_event_manager_register_event_type(em, libvlc_MediaStateChanged);
317 libvlc_event_manager_register_event_type(em, libvlc_MediaParsedChanged);
318 libvlc_event_manager_register_event_type(em, libvlc_MediaSubItemTreeAdded);
319
320 vlc_gc_incref( p_md->p_input_item );
321
322 install_input_item_observer( p_md );
323
324 return p_md;
325}

327/**************************************************************************
328 * Create a new media descriptor object
329 **************************************************************************/
330libvlc_media_t *libvlc_media_new_location( libvlc_instance_t *p_instance,
331 const char * psz_mrl )
332{
333 input_item_t * p_input_item;
334 libvlc_media_t * p_md;
335
336 p_input_item = input_item_New( psz_mrl, NULL );
337
338 if (!p_input_item)
339 {
340 libvlc_printerr( "Not enough memory" );
341 return NULL;
342 }
343
344 p_md = libvlc_media_new_from_input_item( p_instance, p_input_item );
345
346 /* The p_input_item is retained in libvlc_media_new_from_input_item */
347 vlc_gc_decref( p_input_item );
348
349 return p_md;
350}

352libvlc_media_t *libvlc_media_new_path( libvlc_instance_t *p_instance,
353 const char *path )
354{
355 char *mrl = vlc_path2uri( path, NULL );
356 if( unlikely(mrl == NULL) )
357 {
358 libvlc_printerr( "%s", vlc_strerror_c(errno) );
359 return NULL;
360 }
361
362 libvlc_media_t *m = libvlc_media_new_location( p_instance, mrl );
363 free( mrl );
364 return m;
365}

369 char mrl[16];
370 snprintf( mrl, sizeof(mrl), "fd://%d", fd );

424/**************************************************************************
425 * Delete a media descriptor object
426 **************************************************************************/
427void libvlc_media_release( libvlc_media_t *p_md )
428{
429 if (!p_md)
430 return;
431
432 p_md->i_refcount--;
433
434 if( p_md->i_refcount > 0 )
435 return;
436
437 if( p_md->p_subitems )
438 libvlc_media_list_release( p_md->p_subitems );
439
440 uninstall_input_item_observer( p_md );
441 vlc_gc_decref( p_md->p_input_item );
442
443 vlc_cond_destroy( &p_md->parsed_cond );
444 vlc_mutex_destroy( &p_md->parsed_lock );
445
446 /* Construct the event */
447 libvlc_event_t event;
448 event.type = libvlc_MediaFreed;
449 event.u.media_freed.md = p_md;
450
451 /* Send the event */
452 libvlc_event_send( p_md->p_event_manager, &event );
453
454 libvlc_event_manager_release( p_md->p_event_manager );
455
456 free( p_md );
457}

610/**************************************************************************
611 * Get duration of media object (in ms)
612 **************************************************************************/
613int64_t
614libvlc_media_get_duration( libvlc_media_t * p_md )
615{
616 assert( p_md );
617
618 if( !p_md->p_input_item )
619 {
620 libvlc_printerr( "No input item" );
621 return -1;
622 }
623
624 if (!input_item_IsPreparsed( p_md->p_input_item ))
625 return -1;
626
627 return from_mtime(input_item_GetDuration( p_md->p_input_item ));
628}

630static int media_parse(libvlc_media_t *media)
631{
632 libvlc_int_t *libvlc = media->p_libvlc_instance->p_libvlc_int;
633 input_item_t *item = media->p_input_item;
634
635 /* TODO: Fetch art on need basis. But how not to break compatibility? */
636 libvlc_ArtRequest(libvlc, item, META_REQUEST_OPTION_NONE);
637 return libvlc_MetaRequest(libvlc, item, META_REQUEST_OPTION_NONE);

640/**************************************************************************
641 * Parse the media and wait.
642 **************************************************************************/
643void
644libvlc_media_parse(libvlc_media_t *media)
645{
646 vlc_mutex_lock(&media->parsed_lock);
647 if (!media->has_asked_preparse)
648 {
649 media->has_asked_preparse = true;
650 vlc_mutex_unlock(&media->parsed_lock);
651
652 if (media_parse(media))
653 /* Parse failed: do not wait! */
654 return;
655 vlc_mutex_lock(&media->parsed_lock);
656 }
657
658 while (!media->is_parsed)
659 vlc_cond_wait(&media->parsed_cond, &media->parsed_lock);
660 vlc_mutex_unlock(&media->parsed_lock);
661}
662
663/**************************************************************************
664 * Parse the media but do not wait.
665 **************************************************************************/
666void
667libvlc_media_parse_async(libvlc_media_t *media)
668{
669 bool needed;
670
671 vlc_mutex_lock(&media->parsed_lock);
672 needed = !media->has_asked_preparse;
673 media->has_asked_preparse = true;
674 vlc_mutex_unlock(&media->parsed_lock);
675
676 if (needed)
677 media_parse(media);
678}

718/**************************************************************************
719 * Get media descriptor's elementary streams description
720 **************************************************************************/
721int
722libvlc_media_get_tracks_info( libvlc_media_t *p_md, libvlc_media_track_info_t ** pp_es )
723{
724 assert( p_md );
725
726 input_item_t *p_input_item = p_md->p_input_item;
727 vlc_mutex_lock( &p_input_item->lock );
728
729 const int i_es = p_input_item->i_es;
730 *pp_es = (i_es > 0) ? malloc( i_es * sizeof(libvlc_media_track_info_t) ) : NULL;
731
732 if( !*pp_es ) /* no ES, or OOM */
733 {
734 vlc_mutex_unlock( &p_input_item->lock );
735 return 0;
736 }
737
738 /* Fill array */
739 for( int i = 0; i < i_es; i++ )
740 {
741 libvlc_media_track_info_t *p_mes = *pp_es+i;
742 const es_format_t *p_es = p_input_item->es[i];
743
744 p_mes->i_codec = p_es->i_codec;
745 p_mes->i_id = p_es->i_id;
746
747 p_mes->i_profile = p_es->i_profile;
748 p_mes->i_level = p_es->i_level;
749
750 switch(p_es->i_cat)
751 {
752 case UNKNOWN_ES:
753 default:
754 p_mes->i_type = libvlc_track_unknown;
755 break;
756 case VIDEO_ES:
757 p_mes->i_type = libvlc_track_video;
758 p_mes->u.video.i_height = p_es->video.i_height;
759 p_mes->u.video.i_width = p_es->video.i_width;
760 break;
761 case AUDIO_ES:
762 p_mes->i_type = libvlc_track_audio;
763 p_mes->u.audio.i_channels = p_es->audio.i_channels;
764 p_mes->u.audio.i_rate = p_es->audio.i_rate;
765 break;
766 case SPU_ES:
767 p_mes->i_type = libvlc_track_text;
768 break;
769 }
770 }
771
772 vlc_mutex_unlock( &p_input_item->lock );
773 return i_es;
774}

776unsigned
777libvlc_media_tracks_get( libvlc_media_t *p_md, libvlc_media_track_t *** pp_es )
778{
779 assert( p_md );
780
781 input_item_t *p_input_item = p_md->p_input_item;
782 vlc_mutex_lock( &p_input_item->lock );
783
784 const int i_es = p_input_item->i_es;
785 *pp_es = (i_es > 0) ? calloc( i_es, sizeof(**pp_es) ) : NULL;
786
787 if( !*pp_es ) /* no ES, or OOM */
788 {
789 vlc_mutex_unlock( &p_input_item->lock );
790 return 0;
791 }
792
793 /* Fill array */
794 for( int i = 0; i < i_es; i++ )
795 {
796 libvlc_media_track_t *p_mes = calloc( 1, sizeof(*p_mes) );
797 if ( p_mes )
798 {
799 p_mes->audio = malloc( __MAX(__MAX(sizeof(*p_mes->audio),
800 sizeof(*p_mes->video)),
801 sizeof(*p_mes->subtitle)) );
802 }
803 if ( !p_mes || !p_mes->audio )
804 {
805 libvlc_media_tracks_release( *pp_es, i_es );
806 *pp_es = NULL;
807 free( p_mes );
808 vlc_mutex_unlock( &p_input_item->lock );
809 return 0;
810 }
811 (*pp_es)[i] = p_mes;
812
813 const es_format_t *p_es = p_input_item->es[i];
814
815 p_mes->i_codec = p_es->i_codec;
816 p_mes->i_original_fourcc = p_es->i_original_fourcc;
817 p_mes->i_id = p_es->i_id;
818
819 p_mes->i_profile = p_es->i_profile;
820 p_mes->i_level = p_es->i_level;
821
822 p_mes->i_bitrate = p_es->i_bitrate;
823 p_mes->psz_language = p_es->psz_language != NULL ? strdup(p_es->psz_language) : NULL;
824 p_mes->psz_description = p_es->psz_description != NULL ? strdup(p_es->psz_description) : NULL;
825
826 switch(p_es->i_cat)
827 {
828 case UNKNOWN_ES:
829 default:
830 p_mes->i_type = libvlc_track_unknown;
831 break;
832 case VIDEO_ES:
833 p_mes->i_type = libvlc_track_video;
834 p_mes->video->i_height = p_es->video.i_height;
835 p_mes->video->i_width = p_es->video.i_width;
836 p_mes->video->i_sar_num = p_es->video.i_sar_num;
837 p_mes->video->i_sar_den = p_es->video.i_sar_den;
838 p_mes->video->i_frame_rate_num = p_es->video.i_frame_rate;
839 p_mes->video->i_frame_rate_den = p_es->video.i_frame_rate_base;
840 break;
841 case AUDIO_ES:
842 p_mes->i_type = libvlc_track_audio;
843 p_mes->audio->i_channels = p_es->audio.i_channels;
844 p_mes->audio->i_rate = p_es->audio.i_rate;
845 break;
846 case SPU_ES:
847 p_mes->i_type = libvlc_track_text;
848 p_mes->subtitle->psz_encoding = p_es->subs.psz_encoding != NULL ?
849 strdup(p_es->subs.psz_encoding) : NULL;
850 break;
851 }
852 }
853
854 vlc_mutex_unlock( &p_input_item->lock );
855 return i_es;
856}

859/**************************************************************************
860 * Release media descriptor's elementary streams description array
861 **************************************************************************/
862void libvlc_media_tracks_release( libvlc_media_track_t **p_tracks, unsigned i_count )
863{
864 for( unsigned i = 0; i < i_count; ++i )
865 {
866 if ( !p_tracks[i] )
867 continue;
868 free( p_tracks[i]->psz_language );
869 free( p_tracks[i]->psz_description );
870 switch( p_tracks[i]->i_type )
871 {
872 case libvlc_track_audio:
873 break;
874 case libvlc_track_video:
875 break;
876 case libvlc_track_text:
877 free( p_tracks[i]->subtitle->psz_encoding );
878 break;
879 case libvlc_track_unknown:
880 default:
881 break;
882 }
883 free( p_tracks[i]->audio );
884 free( p_tracks[i] );
885 }
886 free( p_tracks );
887}
