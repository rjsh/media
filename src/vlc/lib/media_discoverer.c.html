56/**************************************************************************
57 * services_discovery_item_added (Private) (VLC event callback)
58 **************************************************************************/
59
60static void services_discovery_item_added( const vlc_event_t * p_event,
61 void * user_data )
62{
63 input_item_t * p_item = p_event->u.services_discovery_item_added.p_new_item;
64 const char * psz_cat = p_event->u.services_discovery_item_added.psz_category;
65 libvlc_media_t * p_md;
66 libvlc_media_discoverer_t * p_mdis = user_data;
67 libvlc_media_list_t * p_mlist = p_mdis->p_mlist;
68
69 p_md = libvlc_media_new_from_input_item( p_mdis->p_libvlc_instance,
70 p_item );
71
72 /* If we have a category, that mean we have to group the items having
73 * that category in a media_list. */
74 if( psz_cat )
75 {
76 p_mlist = vlc_dictionary_value_for_key( &p_mdis->catname_to_submedialist, psz_cat );
77
78 if( p_mlist == kVLCDictionaryNotFound )
79 {
80 libvlc_media_t * p_catmd;
81 p_catmd = libvlc_media_new_as_node( p_mdis->p_libvlc_instance, psz_cat );
82 p_mlist = libvlc_media_subitems( p_catmd );
83 p_mlist->b_read_only = true;
84
85 /* Insert the newly created mlist in our dictionary */
86 vlc_dictionary_insert( &p_mdis->catname_to_submedialist, psz_cat, p_mlist );
87
88 /* Insert the md into the root list */
89 libvlc_media_list_lock( p_mdis->p_mlist );
90 _libvlc_media_list_add_media( p_mdis->p_mlist, p_catmd );
91 libvlc_media_list_unlock( p_mdis->p_mlist );
92
93 /* We don't release the mlist cause the dictionary
94 * doesn't retain the object. But we release the md. */
95 libvlc_media_release( p_catmd );
96 }
97 }
98
99 libvlc_media_list_lock( p_mlist );
100 _libvlc_media_list_add_media( p_mlist, p_md );
101 libvlc_media_list_unlock( p_mlist );
102
103 libvlc_media_release( p_md );
104}

106/**************************************************************************
107 * services_discovery_item_removed (Private) (VLC event callback)
108 **************************************************************************/
109
110static void services_discovery_item_removed( const vlc_event_t * p_event,
111 void * user_data )
112{
113 input_item_t * p_item = p_event->u.services_discovery_item_added.p_new_item;
114 libvlc_media_t * p_md;
115 libvlc_media_discoverer_t * p_mdis = user_data;
116
117 int i, count = libvlc_media_list_count( p_mdis->p_mlist );
118 libvlc_media_list_lock( p_mdis->p_mlist );
119 for( i = 0; i < count; i++ )
120 {
121 p_md = libvlc_media_list_item_at_index( p_mdis->p_mlist, i );
122 if( p_md->p_input_item == p_item )
123 {
124 _libvlc_media_list_remove_index( p_mdis->p_mlist, i );
125 break;
126 }
127 }
128 libvlc_media_list_unlock( p_mdis->p_mlist );
129}

131/**************************************************************************
132 * services_discovery_removeall (Private) (VLC event callback)
133 **************************************************************************/
134static void services_discovery_removeall( const vlc_event_t * p_event,
135 void * user_data )
136{
137 VLC_UNUSED(p_event);
138 libvlc_media_discoverer_t * p_mdis = user_data;
139
140 libvlc_media_list_lock( p_mdis->p_mlist );
141 for( int i = 0; i < libvlc_media_list_count( p_mdis->p_mlist ); i++ )
142 {
143 _libvlc_media_list_remove_index( p_mdis->p_mlist, i );
144 }
145 libvlc_media_list_unlock( p_mdis->p_mlist );
146}

148/**************************************************************************

150 **************************************************************************/
151
152static void services_discovery_started( const vlc_event_t * p_event,
153 void * user_data )
154{
155 VLC_UNUSED(p_event);
156 libvlc_media_discoverer_t * p_mdis = user_data;

182/**************************************************************************
183 * new (Public)
184 *
185 * Init an object.
186 **************************************************************************/
187libvlc_media_discoverer_t *
188libvlc_media_discoverer_new_from_name( libvlc_instance_t * p_inst,
189 const char * psz_name )
190{
191 /* podcast SD is a hack and only works with custom playlist callbacks. */
192 if( !strncasecmp( psz_name, "podcast", 7 ) )
193 return NULL;
194
195 libvlc_media_discoverer_t *p_mdis = malloc(sizeof(*p_mdis));
196 if( unlikely(!p_mdis) )
197 {
198 libvlc_printerr( "Not enough memory" );
199 return NULL;
200 }
201
202 p_mdis->p_libvlc_instance = p_inst;
203 p_mdis->p_mlist = libvlc_media_list_new( p_inst );
204 p_mdis->p_mlist->b_read_only = true;
205 p_mdis->running = false;
206
207 vlc_dictionary_init( &p_mdis->catname_to_submedialist, 0 );
208
209 p_mdis->p_event_manager = libvlc_event_manager_new( p_mdis, p_inst );
210 if( unlikely(p_mdis->p_event_manager == NULL) )
211 {
212 free( p_mdis );
213 return NULL;
214 }
215
216 libvlc_event_manager_register_event_type( p_mdis->p_event_manager,
217 libvlc_MediaDiscovererStarted );
218 libvlc_event_manager_register_event_type( p_mdis->p_event_manager,
219 libvlc_MediaDiscovererEnded );
220
221 p_mdis->p_sd = vlc_sd_Create( (vlc_object_t*)p_inst->p_libvlc_int,
222 psz_name );
223 if( unlikely(p_mdis->p_sd == NULL) )
224 {
225 libvlc_printerr( "%s: no such discovery module found", psz_name );
226 libvlc_media_list_release( p_mdis->p_mlist );
227 libvlc_event_manager_release( p_mdis->p_event_manager );
228 free( p_mdis );
229 return NULL;
230 }
231
232 vlc_event_attach( services_discovery_EventManager( p_mdis->p_sd ),
233 vlc_ServicesDiscoveryItemAdded,
234 services_discovery_item_added,
235 p_mdis );
236 vlc_event_attach( services_discovery_EventManager( p_mdis->p_sd ),
237 vlc_ServicesDiscoveryItemRemoved,
238 services_discovery_item_removed,
239 p_mdis );
240 vlc_event_attach( services_discovery_EventManager( p_mdis->p_sd ),
241 vlc_ServicesDiscoveryStarted,
242 services_discovery_started,
243 p_mdis );
244 vlc_event_attach( services_discovery_EventManager( p_mdis->p_sd ),
245 vlc_ServicesDiscoveryEnded,
246 services_discovery_ended,
247 p_mdis );
248 vlc_event_attach( services_discovery_EventManager( p_mdis->p_sd ),
249 vlc_ServicesDiscoveryItemRemoveAll,
250 services_discovery_removeall,
251 p_mdis );
252
253 /* Here we go */
254 if( !vlc_sd_Start( p_mdis->p_sd ) )
255 {
256 libvlc_printerr( "%s: internal module error",
257 p_mdis->p_sd->psz_name );
258 libvlc_media_list_release( p_mdis->p_mlist );
259 libvlc_event_manager_release( p_mdis->p_event_manager );
260 free( p_mdis );
261 return NULL;
262 }
263
264 return p_mdis;
265}

267/**************************************************************************
268 * release (Public)
269 **************************************************************************/
270void
271libvlc_media_discoverer_release( libvlc_media_discoverer_t * p_mdis )
272{
273 int i;
274
275 vlc_event_detach( services_discovery_EventManager( p_mdis->p_sd ),
276 vlc_ServicesDiscoveryItemAdded,
277 services_discovery_item_added,
278 p_mdis );
279 vlc_event_detach( services_discovery_EventManager( p_mdis->p_sd ),
280 vlc_ServicesDiscoveryItemRemoved,
281 services_discovery_item_removed,
282 p_mdis );
283 vlc_event_detach( services_discovery_EventManager( p_mdis->p_sd ),
284 vlc_ServicesDiscoveryStarted,
285 services_discovery_started,
286 p_mdis );
287 vlc_event_detach( services_discovery_EventManager( p_mdis->p_sd ),
288 vlc_ServicesDiscoveryEnded,
289 services_discovery_ended,
290 p_mdis );
291 vlc_event_detach( services_discovery_EventManager( p_mdis->p_sd ),
292 vlc_ServicesDiscoveryItemRemoveAll,
293 services_discovery_removeall,
294 p_mdis );
295
296 libvlc_media_list_release( p_mdis->p_mlist );
297
298 vlc_sd_StopAndDestroy( p_mdis->p_sd );
299
300 /* Free catname_to_submedialist and all the mlist */
301 char ** all_keys = vlc_dictionary_all_keys( &p_mdis->catname_to_submedialist );
302 for( i = 0; all_keys[i]; i++ )
303 {
304 libvlc_media_list_t * p_catmlist = vlc_dictionary_value_for_key( &p_mdis->catname_to_submedialist, all_keys[i] );
305 libvlc_media_list_release( p_catmlist );
306 free( all_keys[i] );
307 }
308 free( all_keys );
309
310 vlc_dictionary_clear( &p_mdis->catname_to_submedialist, NULL, NULL );
311 libvlc_event_manager_release( p_mdis->p_event_manager );
312
313 free( p_mdis );
314}
