38struct libvlc_media_list_t
39{
40 libvlc_event_manager_t * p_event_manager;
41 libvlc_instance_t * p_libvlc_instance;
42 int i_refcount;
43 vlc_mutex_t object_lock;
44 vlc_mutex_t refcount_lock;
45 libvlc_media_t * p_md; /* The media from which the
46 * mlist comes, if any. */
47 vlc_array_t items;
48
49 /* This indicates if this media list is read-only
50 * from a user point of view */
51 bool b_read_only;
52};

64 libvlc_media_list_t * p_mlist, int index );
