108/**
109 * A LibVLC event
110 */
111typedef struct libvlc_event_t
112{
113 int type; /**< Event type (see @ref libvlc_event_e) */
114 void *p_obj; /**< Object emitting the event */
115 union
116 {
117 /* media descriptor */
118 struct
119 {
120 libvlc_meta_t meta_type;
121 } media_meta_changed;
122 struct
123 {
124 libvlc_media_t * new_child;
125 } media_subitem_added;
126 struct
127 {
128 int64_t new_duration;
129 } media_duration_changed;
130 struct
131 {
132 int new_status;
133 } media_parsed_changed;
134 struct
135 {
136 libvlc_media_t * md;
137 } media_freed;
138 struct
139 {
140 libvlc_state_t new_state;
141 } media_state_changed;
142 struct
143 {
144 libvlc_media_t * item;
145 } media_subitemtree_added;
146
147 /* media instance */
148 struct
149 {
150 float new_cache;
151 } media_player_buffering;
152 struct
153 {
154 float new_position;
155 } media_player_position_changed;
156 struct
157 {
158 libvlc_time_t new_time;
159 } media_player_time_changed;
160 struct
161 {
162 int new_title;
163 } media_player_title_changed;
164 struct
165 {
166 int new_seekable;
167 } media_player_seekable_changed;
168 struct
169 {
170 int new_pausable;
171 } media_player_pausable_changed;
172 struct
173 {
174 int new_scrambled;
175 } media_player_scrambled_changed;
176 struct
177 {
178 int new_count;
179 } media_player_vout;
180
181 /* media list */
182 struct
183 {
184 libvlc_media_t * item;
185 int index;
186 } media_list_item_added;
187 struct
188 {
189 libvlc_media_t * item;
190 int index;
191 } media_list_will_add_item;
192 struct
193 {
194 libvlc_media_t * item;
195 int index;
196 } media_list_item_deleted;
197 struct
198 {
199 libvlc_media_t * item;
200 int index;
201 } media_list_will_delete_item;
202
203 /* media list player */
204 struct
205 {
206 libvlc_media_t * item;
207 } media_list_player_next_item_set;
208
209 /* snapshot taken */
210 struct
211 {
212 char* psz_filename ;
213 } media_player_snapshot_taken ;
214
215 /* Length changed */
216 struct
217 {
218 libvlc_time_t new_length;
219 } media_player_length_changed;
220
221 /* VLM media */
222 struct
223 {
224 const char * psz_media_name;
225 const char * psz_instance_name;
226 } vlm_media_event;
227
228 /* Extra MediaPlayer */
229 struct
230 {
231 libvlc_media_t * new_media;
232 } media_player_media_changed;
233 } u; /**< Type-dependent event description */
234} libvlc_event_t;
235
