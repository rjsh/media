143/**
144 * High precision date or time interval
145 *
146 * Store a high precision date or time interval. The maximum precision is the
147 * microsecond, and a 64 bits integer is used to avoid overflows (maximum
148 * time interval is then 292271 years, which should be long enough for any
149 * video). Dates are stored as microseconds since a common date (usually the
150 * epoch). Note that date and time intervals can be manipulated using regular
151 * arithmetic operators, and that no special functions are required.
152 */
153typedef int64_t mtime_t;

155/**
156 * The vlc_fourcc_t type.
157 *
158 * See http://www.webartz.com/fourcc/ for a very detailed list.
159 */
160typedef uint32_t vlc_fourcc_t;
161
162#ifdef WORDS_BIGENDIAN
163# define VLC_FOURCC( a, b, c, d ) \
164 ( ((uint32_t)d) | ( ((uint32_t)c) << 8 ) \
165 | ( ((uint32_t)b) << 16 ) | ( ((uint32_t)a) << 24 ) )
166# define VLC_TWOCC( a, b ) \
167 ( (uint16_t)(b) | ( (uint16_t)(a) << 8 ) )
168
169#else
170# define VLC_FOURCC( a, b, c, d ) \
171 ( ((uint32_t)a) | ( ((uint32_t)b) << 8 ) \
172 | ( ((uint32_t)c) << 16 ) | ( ((uint32_t)d) << 24 ) )
173# define VLC_TWOCC( a, b ) \
174 ( (uint16_t)(a) | ( (uint16_t)(b) << 8 ) )
175
176#endif
177
178/**
179 * Translate a vlc_fourcc into its string representation. This function
180 * assumes there is enough room in psz_fourcc to store 4 characters in.
181 *
182 * \param fcc a vlc_fourcc_t
183 * \param psz_fourcc string to store string representation of vlc_fourcc in
184 */
185static inline void vlc_fourcc_to_char( vlc_fourcc_t fcc, char *psz_fourcc )
186{
187 memcpy( psz_fourcc, &fcc, 4 );
188}
189
190#define vlc_fourcc_to_char( a, b ) \
191 vlc_fourcc_to_char( (vlc_fourcc_t)(a), (char *)(b) )

484static inline int64_t GCD ( int64_t a, int64_t b )
485{
486 while( b )
487 {
488 int64_t c = a % b;
489 a = b;
490 b = c;
491 }

493}

503/** Count leading zeroes */
504VLC_USED
505static inline unsigned clz (unsigned x)
506{
507#if VLC_GCC_VERSION(3,4)
508 return __builtin_clz (x);
509#else
510 unsigned i = sizeof (x) * 8;
511
512 while (x)
513 {
514 x >>= 1;
515 i--;
516 }
517 return i;
518#endif
519}
520
521#define clz8( x ) (clz(x) - ((sizeof(unsigned) - sizeof (uint8_t)) * 8))
522#define clz16( x ) (clz(x) - ((sizeof(unsigned) - sizeof (uint16_t)) * 8))
523/* XXX: this assumes that int is 32-bits or more */
524#define clz32( x ) (clz(x) - ((sizeof(unsigned) - sizeof (uint32_t)) * 8))

544/** Bit weight */
545VLC_USED
546static inline unsigned popcount (unsigned x)
547{
548#if VLC_GCC_VERSION(3,4)
549 return __builtin_popcount (x);
550#else
551 unsigned count = 0;
552 while (x)
553 {
554 count += x & 1;
555 x = x >> 1;
556 }
557 return count;
558#endif
559}

561VLC_USED
562static inline unsigned parity (unsigned x)
563{
564#if VLC_GCC_VERSION(3,4)
565 return __builtin_parity (x);
566#else
567 for (unsigned i = 4 * sizeof (x); i > 0; i /= 2)
568 x ^= x >> i;
569 return x & 1;
570#endif
571}

579/** Byte swap (16 bits) */
580VLC_USED
581static inline uint16_t bswap16 (uint16_t x)
582{
583 return (x << 8) | (x >> 8);
584}
585
586/** Byte swap (32 bits) */
587VLC_USED
588static inline uint32_t bswap32 (uint32_t x)
589{
590#if VLC_GCC_VERSION(4,3) || defined(__clang__)
591 return __builtin_bswap32 (x);
592#else
593 return ((x & 0x000000FF) << 24)
594 | ((x & 0x0000FF00) << 8)
595 | ((x & 0x00FF0000) >> 8)
596 | ((x & 0xFF000000) >> 24);
597#endif
598}
599
600/** Byte swap (64 bits) */
601VLC_USED
602static inline uint64_t bswap64 (uint64_t x)
603{
604#if VLC_GCC_VERSION(4,3) || defined(__clang__)
605 return __builtin_bswap64 (x);
606#elif !defined (__cplusplus)
607 return ((x & 0x00000000000000FF) << 56)
608 | ((x & 0x000000000000FF00) << 40)
609 | ((x & 0x0000000000FF0000) << 24)
610 | ((x & 0x00000000FF000000) << 8)
611 | ((x & 0x000000FF00000000) >> 8)
612 | ((x & 0x0000FF0000000000) >> 24)
613 | ((x & 0x00FF000000000000) >> 40)
614 | ((x & 0xFF00000000000000) >> 56);
615#else
616 return ((x & 0x00000000000000FFULL) << 56)
617 | ((x & 0x000000000000FF00ULL) << 40)
618 | ((x & 0x0000000000FF0000ULL) << 24)
619 | ((x & 0x00000000FF000000ULL) << 8)
620 | ((x & 0x000000FF00000000ULL) >> 8)
621 | ((x & 0x0000FF0000000000ULL) >> 24)
622 | ((x & 0x00FF000000000000ULL) >> 40)
623 | ((x & 0xFF00000000000000ULL) >> 56);
624#endif
625}

815static inline void *vlc_memalign(size_t align, size_t size)
816{
817 long diff;
818 void *ptr;
819
820 ptr = malloc(size+align);
821 if(!ptr)
822 return ptr;
823 diff = ((-(long)ptr - 1)&(align-1)) + 1;
824 ptr = (char*)ptr + diff;
825 ((char*)ptr)[-1]= diff;
826 return ptr;
827}

832 free((char*)ptr - ((char*)ptr)[-1]);

836{
