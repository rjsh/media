428static inline void wf_tag_to_fourcc( uint16_t i_tag, vlc_fourcc_t *fcc,
429 const char **ppsz_name )
430{
431 int i;
432 for( i = 0; wave_format_tag_to_fourcc[i].i_tag != 0; i++ )
433 {
434 if( wave_format_tag_to_fourcc[i].i_tag == i_tag ) break;
435 }
436 if( fcc ) *fcc = wave_format_tag_to_fourcc[i].i_fourcc;
437 if( ppsz_name ) *ppsz_name = wave_format_tag_to_fourcc[i].psz_name;
438}

467/* compares two GUIDs, returns 1 if identical, 0 otherwise */
468static inline int guidcmp( const GUID *s1, const GUID *s2 )
469{
470 return( s1->Data1 == s2->Data1 && s1->Data2 == s2->Data2 &&
471 s1->Data3 == s2->Data3 && !memcmp( s1->Data4, s2->Data4, 8 ) );
472}

474static inline void sf_tag_to_fourcc( GUID *guid_tag,
475 vlc_fourcc_t *fcc, const char **ppsz_name )
476{
477 int i;
478
479 for( i = 0; !guidcmp( &sub_format_tag_to_fourcc[i].guid_tag,
480 &KSDATAFORMAT_SUBTYPE_UNKNOWN ); i++ )
481 {
482 if( guidcmp( &sub_format_tag_to_fourcc[i].guid_tag, guid_tag ) ) break;
