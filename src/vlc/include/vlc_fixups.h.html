219/* locale.h */
220#ifndef HAVE_USELOCALE
221#define LC_ALL_MASK 0
222#define LC_NUMERIC_MASK 0
223#define LC_MESSAGES_MASK 0
224#define LC_GLOBAL_LOCALE ((locale_t)(uintptr_t)1)
225typedef void *locale_t;
226static inline locale_t uselocale(locale_t loc)
227{
228 (void)loc;
229 return NULL;
230}
231static inline void freelocale(locale_t loc)
232{
233 (void)loc;
234}
235static inline locale_t newlocale(int mask, const char * locale, locale_t base)
236{
237 (void)mask; (void)locale; (void)base;
238 return NULL;
239}
240#endif
