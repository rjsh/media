31/* Buffers which arrive in advance of more than AOUT_MAX_ADVANCE_TIME
32 * will be considered as bogus and be trashed */
33#define AOUT_MAX_ADVANCE_TIME (AOUT_MAX_PREPARE_TIME + CLOCK_FREQ)
34
35/* Buffers which arrive in advance of more than AOUT_MAX_PREPARE_TIME
36 * will cause the calling thread to sleep */
37#define AOUT_MAX_PREPARE_TIME (2 * CLOCK_FREQ)
38
39/* Buffers which arrive after pts - AOUT_MIN_PREPARE_TIME will be trashed
40 * to avoid too heavy resampling */
41#define AOUT_MIN_PREPARE_TIME AOUT_MAX_PTS_ADVANCE
42
43/* Tolerance values from EBU Recommendation 37 */
44/** Maximum advance of actual audio playback time to coded PTS,
45 * above which downsampling will be performed */
46#define AOUT_MAX_PTS_ADVANCE (CLOCK_FREQ / 25)
47
48/** Maximum delay of actual audio playback time from coded PTS,
49 * above which upsampling will be performed */
50#define AOUT_MAX_PTS_DELAY (3 * CLOCK_FREQ / 50)
51
52/* Max acceptable resampling (in %) */
53#define AOUT_MAX_RESAMPLING 10

57#define AOUT_FMTS_IDENTICAL( p_first, p_second ) ( \
58 ((p_first)->i_format == (p_second)->i_format) \
59 && AOUT_FMTS_SIMILAR(p_first, p_second) )
60
61/* Check if i_rate == i_rate and i_channels == i_channels */
62#define AOUT_FMTS_SIMILAR( p_first, p_second ) ( \
63 ((p_first)->i_rate == (p_second)->i_rate) \
64 && ((p_first)->i_physical_channels == (p_second)->i_physical_channels)\
65 && ((p_first)->i_original_channels == (p_second)->i_original_channels) )
66

68 (aout_BitsPerSample((p_format)->i_format) != 0)

100/** Audio output object */
101struct audio_output
102{
103 VLC_COMMON_MEMBERS
104
105 struct aout_sys_t *sys; /**< Private data for callbacks */
106
107 int (*start)(audio_output_t *, audio_sample_format_t *fmt);
108 /**< Starts a new stream (mandatory, cannot be NULL).
109 * \param fmt input stream sample format upon entry,
110 * output stream sample format upon return [IN/OUT]
111 * \return VLC_SUCCESS on success, non-zero on failure
112 * \note No other stream may be already started when called.
113 */
114 void (*stop)(audio_output_t *);
115 /**< Stops the existing stream (optional, may be NULL).
116 * \note A stream must have been started when called.
117 */
118 int (*time_get)(audio_output_t *, mtime_t *delay);
119 /**< Estimates playback buffer latency (optional, may be NULL).
120 * \param delay pointer to the delay until the next sample to be written
121 * to the playback buffer is rendered [OUT]
122 * \return 0 on success, non-zero on failure or lack of data
123 * \note A stream must have been started when called.
124 */
125 void (*play)(audio_output_t *, block_t *);
126 /**< Queues a block of samples for playback (mandatory, cannot be NULL).
127 * \note A stream must have been started when called.
128 */
129 void (*pause)( audio_output_t *, bool pause, mtime_t date);
130 /**< Pauses or resumes playback (optional, may be NULL).
131 * \param pause pause if true, resume from pause if false
132 * \param date timestamp when the pause or resume was requested
133 * \note A stream must have been started when called.
134 */
135 void (*flush)( audio_output_t *, bool wait);
136 /**< Flushes or drains the playback buffers (mandatory, cannot be NULL).
137 * \param wait true to wait for playback of pending buffers (drain),
138 * false to discard pending buffers (flush)
139 * \note A stream must have been started when called.
140 */
141 int (*volume_set)(audio_output_t *, float volume);
142 /**< Changes playback volume (optional, may be NULL).
143 * \param volume requested volume (0. = mute, 1. = nominal)
144 * \note The volume is always a positive number.
145 * \warning A stream may or may not have been started when called.
146 */
147 int (*mute_set)(audio_output_t *, bool mute);
148 /**< Changes muting (optinal, may be NULL).
149 * \param mute true to mute, false to unmute
150 * \warning A stream may or may not have been started when called.
151 */
152 int (*device_select)(audio_output_t *, const char *id);
153 /**< Selects an audio output device (optional, may be NULL).
154 * \param id nul-terminated device unique identifier.
155 * \return 0 on success, non-zero on failure.
156 * \warning A stream may or may not have been started when called.
157 */
158 struct {
159 void (*volume_report)(audio_output_t *, float);
160 void (*mute_report)(audio_output_t *, bool);
161 void (*policy_report)(audio_output_t *, bool);
162 void (*device_report)(audio_output_t *, const char *);
163 void (*hotplug_report)(audio_output_t *, const char *, const char *);
164 int (*gain_request)(audio_output_t *, float);
165 void (*restart_request)(audio_output_t *, unsigned);
166 } event;
167};

188/**
189 * This function computes the reordering needed to go from pi_chan_order_in to
190 * pi_chan_order_out.
191 * If pi_chan_order_in or pi_chan_order_out is NULL, it will assume that vlc
192 * internal (WG4) order is requested.
193 */
194VLC_API unsigned aout_CheckChannelReorder( const uint32_t *, const uint32_t *,
195 uint32_t mask, uint8_t *table );
196VLC_API void aout_ChannelReorder(void *, size_t, unsigned, const uint8_t *, vlc_fourcc_t);
197
198VLC_API void aout_Interleave(void *dst, const void *const *planes,
199 unsigned samples, unsigned channels,
200 vlc_fourcc_t fourcc);
201VLC_API void aout_Deinterleave(void *dst, const void *src, unsigned samples,
202 unsigned channels, vlc_fourcc_t fourcc);
203
204/**
205 * This function will compute the extraction parameter into pi_selection to go
206 * from i_channels with their type given by pi_order_src[] into the order
207 * describe by pi_order_dst.
208 * It will also set :
209 * - *pi_channels as the number of channels that will be extracted which is
210 * lower (in case of non understood channels type) or equal to i_channels.
211 * - the layout of the channels (*pi_layout).
212 *
213 * It will return true if channel extraction is really needed, in which case
214 * aout_ChannelExtract must be used
215 *
216 * XXX It must be used when the source may have channel type not understood
217 * by VLC. In this case the channel type pi_order_src[] must be set to 0.
218 * XXX It must also be used if multiple channels have the same type.
219 */
220VLC_API bool aout_CheckChannelExtraction( int *pi_selection, uint32_t *pi_layout, int *pi_channels, const uint32_t pi_order_dst[AOUT_CHAN_MAX], const uint32_t *pi_order_src, int i_channels );
221
222/**
223 * Do the actual channels extraction using the parameters created by
224 * aout_CheckChannelExtraction.
225 *
226 * XXX this function does not work in place (p_dst and p_src must not overlap).
227 * XXX Only 8, 16, 24, 32, 64 bits per sample are supported.
228 */
229VLC_API void aout_ChannelExtract( void *p_dst, int i_dst_channels, const void *p_src, int i_src_channels, int i_sample_count, const int *pi_selection, int i_bits_per_sample );

311static inline int aout_ChannelsRestart (vlc_object_t *obj, const char *varname,
312 vlc_value_t oldval, vlc_value_t newval, void *data)
313{
314 audio_output_t *aout = (audio_output_t *)obj;
315 (void)varname; (void)oldval; (void)newval; (void)data;
316
317 aout_RestartRequest (aout, AOUT_RESTART_OUTPUT);
