24#ifndef VLC_MD5_H
25# define VLC_MD5_H
26
27/**
28 * \file
29 * This file defines functions and structures to compute MD5 digests
30 */
31
32struct md5_s
33{
34 uint32_t A, B, C, D; /* chaining variables */
35 uint32_t nblocks;
36 uint8_t buf[64];
37 int count;
38};
39
40VLC_API void InitMD5( struct md5_s * );
41VLC_API void AddMD5( struct md5_s *, const void *, size_t );
42VLC_API void EndMD5( struct md5_s * );
43
44/**
45 * Returns a char representation of the md5 hash, as shown by UNIX md5 or
46 * md5sum tools.
47 */
48static inline char * psz_md5_hash( struct md5_s *md5_s )
49{
50 char *psz = malloc( 33 ); /* md5 string is 32 bytes + NULL character */
51 if( likely(psz) )
52 {
53 for( int i = 0; i < 16; i++ )
54 sprintf( &psz[2*i], "%02"PRIx8, md5_s->buf[i] );
55 }
56 return psz;
57}
