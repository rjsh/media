34#include <vlc_es.h>

37/** Description of a planar graphic field */
38typedef struct plane_t
39{
40 uint8_t *p_pixels; /**< Start of the plane's data */
41
42 /* Variables used for fast memcpy operations */
43 int i_lines; /**< Number of lines, including margins */
44 int i_pitch; /**< Number of bytes in a line, including margins */
45
46 /** Size of a macropixel, defaults to 1 */
47 int i_pixel_pitch;
48
49 /* Variables used for pictures with margins */
50 int i_visible_lines; /**< How many visible lines are there ? */
51 int i_visible_pitch; /**< How many visible pixels are there ? */
52
53} plane_t;
54
55/**
56 * Maximum number of plane for a picture
57 */
58#define PICTURE_PLANE_MAX (VOUT_MAX_PLANES)
59
60
61/**
62 * A private definition to help overloading picture release
63 */
64typedef struct picture_gc_sys_t picture_gc_sys_t;
65
66/**
67 * Video picture
68 */
69struct picture_t
70{
71 /**
72 * The properties of the picture
73 */
74 video_frame_format_t format;
75
76 plane_t p[PICTURE_PLANE_MAX]; /**< description of the planes */
77 int i_planes; /**< number of allocated planes */
78
79 /** \name Picture management properties
80 * These properties can be modified using the video output thread API,
81 * but should never be written directly */
82 /**@{*/
83 mtime_t date; /**< display date */
84 bool b_force;
85 /**@}*/
86
87 /** \name Picture dynamic properties
88 * Those properties can be changed by the decoder
89 * @{
90 */
91 bool b_progressive; /**< is it a progressive frame ? */
92 bool b_top_field_first; /**< which field is first */
93 unsigned int i_nb_fields; /**< # of displayed fields */
94 void * context; /**< video format-specific data pointer,
95 * must point to a (void (*)(void*)) pointer to free the context */
96 /**@}*/
97
98 /** Private data - the video output plugin might want to put stuff here to
99 * keep track of the picture */
100 picture_sys_t * p_sys;
101
102 /** This way the picture_Release can be overloaded */
103 struct
104 {
105 atomic_uintptr_t refcount;
106 void (*pf_destroy)( picture_t * );
107 picture_gc_sys_t *p_sys;
108 } gc;
109
110 /** Next picture in a FIFO a pictures */
111 struct picture_t *p_next;
112};
113
114/**
115 * This function will create a new picture.
116 * The picture created will implement a default release management compatible
117 * with picture_Hold and picture_Release. This default management will release
118 * p_sys, gc.p_sys fields if non NULL.
119 */
120VLC_API picture_t * picture_New( vlc_fourcc_t i_chroma, int i_width, int i_height, int i_sar_num, int i_sar_den ) VLC_USED;
121
122/**
123 * This function will create a new picture using the given format.
124 *
125 * When possible, it is preferred to use this function over picture_New
126 * as more information about the format is kept.
127 */
128VLC_API picture_t * picture_NewFromFormat( const video_format_t *p_fmt ) VLC_USED;
129
130/**
131 * Resource for a picture.
132 */
133typedef struct
134{
135 picture_sys_t *p_sys;
136 void (*pf_destroy)(picture_t *);
137
138 /* Plane resources
139 * XXX all fields MUST be set to the right value.
140 */
141 struct
142 {
143 uint8_t *p_pixels; /**< Start of the plane's data */
144 int i_lines; /**< Number of lines, including margins */
145 int i_pitch; /**< Number of bytes in a line, including margins */
146 } p[PICTURE_PLANE_MAX];
147
148} picture_resource_t;
149
150/**
151 * This function will create a new picture using the provided resource.
152 *
153 * If the resource is NULL then a plain picture_NewFromFormat is returned.
154 */
155VLC_API picture_t * picture_NewFromResource( const video_format_t *, const picture_resource_t * ) VLC_USED;
156
157/**
158 * This function will increase the picture reference count.
159 * It will not have any effect on picture obtained from vout
160 *
161 * It returns the given picture for convenience.
162 */
163VLC_API picture_t *picture_Hold( picture_t *p_picture );
164
165/**
166 * This function will release a picture.
167 * It will not have any effect on picture obtained from vout
168 */
169VLC_API void picture_Release( picture_t *p_picture );
170
171/**
172 * This function will return true if you are not the only owner of the
173 * picture.
174 *
175 * It is only valid if it is created using picture_New.
176 */
177VLC_API bool picture_IsReferenced( picture_t *p_picture );
178
179/**
180 * This function will copy all picture dynamic properties.
181 */
182VLC_API void picture_CopyProperties( picture_t *p_dst, const picture_t *p_src );
183
184/**
185 * This function will reset a picture information (properties and quantizers).
186 * It is sometimes useful for reusing pictures (like from a pool).
187 */
188VLC_API void picture_Reset( picture_t * );
189
190/**
191 * This function will copy the picture pixels.
192 * You can safely copy between pictures that do not have the same size,
193 * only the compatible(smaller) part will be copied.
194 */
195VLC_API void picture_CopyPixels( picture_t *p_dst, const picture_t *p_src );
196VLC_API void plane_CopyPixels( plane_t *p_dst, const plane_t *p_src );
197
198/**
199 * This function will copy both picture dynamic properties and pixels.
200 * You have to notice that sometime a simple picture_Hold may do what
201 * you want without the copy overhead.
202 * Provided for convenience.
203 *
204 * \param p_dst pointer to the destination picture.
205 * \param p_src pointer to the source picture.
206 */
207VLC_API void picture_Copy( picture_t *p_dst, const picture_t *p_src );
208
209/**
210 * This function will export a picture to an encoded bitstream.
211 *
212 * pp_image will contain the encoded bitstream in psz_format format.
213 *
214 * p_fmt can be NULL otherwise it will be set with the format used for the
215 * picture before encoding.
216 *
217 * i_override_width/height allow to override the width and/or the height of the
218 * picture to be encoded:
219 * - if strictly lower than 0, the original dimension will be used.
220 * - if equal to 0, it will be deduced from the other dimension which must be
221 * different to 0.
222 * - if strictly higher than 0, it will override the dimension.
223 * If at most one of them is > 0 then the picture aspect ratio will be kept.
224 */
225VLC_API int picture_Export( vlc_object_t *p_obj, block_t **pp_image, video_format_t *p_fmt, picture_t *p_picture, vlc_fourcc_t i_format, int i_override_width, int i_override_height );
226
227/**
228 * This function will setup all fields of a picture_t without allocating any
229 * memory.
230 * XXX The memory must already be initialized.
231 * It does not need to be released.
232 *
233 * It will return VLC_EGENERIC if the core does not understand the requested
234 * format.
235 *
236 * It can be useful to get the properties of planes.
237 */
238VLC_API int picture_Setup( picture_t *, const video_format_t * );
239
240
241/**
242 * This function will blend a given subpicture onto a picture.
243 *
244 * The subpicture and all its region must:
245 * - be absolute.
246 * - not be ephemere.
247 * - not have the fade flag.
248 * - contains only picture (no text rendering).
249 * \return the number of region(s) succesfully blent
250 */
251VLC_API unsigned picture_BlendSubpicture( picture_t *, filter_t *p_blend, subpicture_t * );
