24#ifndef VLC_GL_H
25#define VLC_GL_H 1
26
27/**
28 * \file
29 * This file defines GL structures and functions.
30 */
31
32struct vout_window_t;
33
34/**
35 * A VLC GL context (and its underlying surface)
36 */
37typedef struct vlc_gl_t vlc_gl_t;
38
39struct vlc_gl_t
40{
41 VLC_COMMON_MEMBERS
42
43 struct vout_window_t *surface;
44 module_t *module;
45 void *sys;
46
47 int (*makeCurrent)(vlc_gl_t *);
48 void (*releaseCurrent)(vlc_gl_t *);
49 void (*swap)(vlc_gl_t *);
50 int (*lock)(vlc_gl_t *);
51 void (*unlock)(vlc_gl_t *);
52 void*(*getProcAddress)(vlc_gl_t *, const char *);
53};
54
55enum {
56 VLC_OPENGL,
57 VLC_OPENGL_ES,
58 VLC_OPENGL_ES2,
59};
60
61VLC_API vlc_gl_t *vlc_gl_Create(struct vout_window_t *, unsigned, const char *) VLC_USED;
62VLC_API void vlc_gl_Destroy(vlc_gl_t *);
63
64static inline int vlc_gl_MakeCurrent(vlc_gl_t *gl)
65{
66 return gl->makeCurrent(gl);
67}
68
69static inline void vlc_gl_ReleaseCurrent(vlc_gl_t *gl)
70{
71 gl->releaseCurrent(gl);
72}
73
74static inline int vlc_gl_Lock(vlc_gl_t *gl)
75{
76 return (gl->lock != NULL) ? gl->lock(gl) : VLC_SUCCESS;
77}
78
79static inline void vlc_gl_Unlock(vlc_gl_t *gl)
80{
81 if (gl->unlock != NULL)
82 gl->unlock(gl);
83}
84
85static inline void vlc_gl_Swap(vlc_gl_t *gl)
86{
87 gl->swap(gl);
88}
89
90static inline void *vlc_gl_GetProcAddress(vlc_gl_t *gl, const char *name)
91{
92 return (gl->getProcAddress != NULL) ? gl->getProcAddress(gl, name) : NULL;
93}
94
95#endif /* VLC_GL_H */
