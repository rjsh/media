96static inline bool sout_AccessOutCanControlPace( sout_access_out_t *p_ao )
97{
98 bool b;
99 if( sout_AccessOutControl( p_ao, ACCESS_OUT_CONTROLS_PACE, &b ) )
100 return true;
101 return b;
102}

104/** Muxer structure */
105struct sout_mux_t
106{
107 VLC_COMMON_MEMBERS
108 module_t *p_module;
109
110 sout_instance_t *p_sout;
111
112 char *psz_mux;
113 config_chain_t *p_cfg;
114
115 sout_access_out_t *p_access;
116
117 int (*pf_addstream)( sout_mux_t *, sout_input_t * );
118 int (*pf_delstream)( sout_mux_t *, sout_input_t * );
119 int (*pf_mux) ( sout_mux_t * );
120 int (*pf_control) ( sout_mux_t *, int, va_list );
121
122 /* here are all inputs accepted by muxer */
123 int i_nb_inputs;
124 sout_input_t **pp_inputs;
125
126 /* mux private */
127 sout_mux_sys_t *p_sys;
128
129 /* XXX private to stream_output.c */
130 /* if muxer doesn't support adding stream at any time then we first wait
131 * for stream then we refuse all stream and start muxing */
132 bool b_add_stream_any_time;
133 bool b_waiting_stream;
134 /* we wait one second after first stream added */
135 mtime_t i_add_stream_start;
136};
137

143 MUX_GET_ADD_STREAM_WAIT, /* arg1= bool *, res=cannot fail */
144 MUX_GET_MIME, /* arg1= char ** res=can fail */
145};

163static inline int sout_MuxControl( sout_mux_t *p_mux, int i_query, ... )
164{
165 va_list args;
166 int i_result;
167
168 va_start( args, i_query );
169 i_result = p_mux->pf_control( p_mux, i_query, args );
170 va_end( args );
171 return i_result;
172}
