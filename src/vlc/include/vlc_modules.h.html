68VLC_USED static inline module_t *module_get_main (void)
69{
70 return module_find ("core");
71}
72#define module_get_main(a) module_get_main()
73
74VLC_USED static inline bool module_is_main( const module_t * p_module )
75{
76 return !strcmp( module_get_object( p_module ), "core" );
77}
