170/**
171 * String uuid to binary uuid helpers
172 */
173static inline bool addons_uuid_read( const char *psz_uuid, addon_uuid_t *p_uuid )
174{
175 if ( !psz_uuid ) return false;
176 if ( strlen( psz_uuid ) < ADDON_UUID_PSZ_SIZE ) return false;
177
178 int i = 0, j = 0;
179 while ( i<ADDON_UUID_PSZ_SIZE )
180 {
181 if ( *( psz_uuid + i ) == '-' )
182 i++;
183 int v;
184 sscanf( psz_uuid + i, "%02x", &v );
185 (*p_uuid)[j++] = v & 0xFF;
186 i+=2;
187 }
188
189 return true;
190}
191
192static inline char * addons_uuid_to_psz( const addon_uuid_t * p_uuid )
193{
194 char *psz = (char*) calloc( ADDON_UUID_PSZ_SIZE + 1 , sizeof(char) );
195 if ( psz )
196 {
197 int i=0;
198 char *p = psz;
199 while ( i < ADDON_UUID_SIZE )
200 {
201 if ( i == 4 || i== 7 || i== 9 || i== 11 )
202 *p++ = '-';
203 int v = 0xFF & (*p_uuid)[i];
204 sprintf( p, "%02x", v );
205 p += 2;
206 i++;
207 }
208 }
209 return psz;
210}
