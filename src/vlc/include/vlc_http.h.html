36/* RFC 2617: Basic and Digest Access Authentication */
37typedef struct http_auth_t
38{
39 char *psz_realm;
40 char *psz_domain;
41 char *psz_nonce;
42 char *psz_opaque;
43 char *psz_stale;
44 char *psz_algorithm;
45 char *psz_qop;
46 int i_nonce;
47 char *psz_cnonce;
48 char *psz_HA1; /* stored H(A1) value if algorithm = "MD5-sess" */
49} http_auth_t;
50
51
52VLC_API void http_auth_Init( http_auth_t * );
53VLC_API void http_auth_Reset( http_auth_t * );
54VLC_API void http_auth_ParseWwwAuthenticateHeader
55 ( vlc_object_t *, http_auth_t * ,
56 const char * );
57VLC_API int http_auth_ParseAuthenticationInfoHeader
58 ( vlc_object_t *, http_auth_t *,
59 const char *, const char *,
60 const char *, const char *,
61 const char * );
62VLC_API char *http_auth_FormatAuthorizationHeader
63 ( vlc_object_t *, http_auth_t *,
64 const char *, const char *,
65 const char *, const char * ) VLC_USED;
66
67#endif /* VLC_HTTP_H */
