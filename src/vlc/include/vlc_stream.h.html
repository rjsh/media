85/**
86 * Possible commands to send to stream_Control() and stream_vaControl()
87 */
88enum stream_query_e
89{
90 /* capabilities */
91 STREAM_CAN_SEEK, /**< arg1= bool * res=cannot fail*/
92 STREAM_CAN_FASTSEEK, /**< arg1= bool * res=cannot fail*/
93 STREAM_CAN_PAUSE, /**< arg1= bool * res=cannot fail*/
94 STREAM_CAN_CONTROL_PACE, /**< arg1= bool * res=cannot fail*/
95
96 /* */
97 STREAM_SET_POSITION, /**< arg1= uint64_t res=can fail */
98 STREAM_GET_POSITION, /**< arg1= uint64_t * res=cannot fail*/
99
100 STREAM_GET_SIZE, /**< arg1= uint64_t * res=cannot fail (0 if no sense)*/
101
102 /* You should update size of source if any and then update size
103 * FIXME find a way to avoid it */
104 STREAM_UPDATE_SIZE,
105
106 /* */
107 STREAM_GET_PTS_DELAY = 0x101,/**< arg1= int64_t* res=cannot fail */
108 STREAM_GET_TITLE_INFO, /**< arg1=input_title_t*** arg2=int* res=can fail */
109 STREAM_GET_TITLE, /**< arg1=unsigned * res=can fail */
110 STREAM_GET_SEEKPOINT, /**< arg1=unsigned * res=can fail */
111 STREAM_GET_META, /**< arg1= vlc_meta_t ** res=can fail */
112 STREAM_GET_CONTENT_TYPE, /**< arg1= char ** res=can fail */
113 STREAM_GET_SIGNAL, /**< arg1=double *pf_quality, arg2=double *pf_strength res=can fail */
114
115 STREAM_SET_PAUSE_STATE = 0x200, /**< arg1= bool res=can fail */
116 STREAM_SET_TITLE, /**< arg1= int res=can fail */
117 STREAM_SET_SEEKPOINT, /**< arg1= int res=can fail */
118
119 /* XXX only data read through stream_Read/Block will be recorded */
120 STREAM_SET_RECORD_STATE, /**< arg1=bool, arg2=const char *psz_ext (if arg1 is true) res=can fail */
121
122 STREAM_SET_PRIVATE_ID_STATE = 0x1000, /* arg1= int i_private_data, bool b_selected res=can fail */
123 STREAM_SET_PRIVATE_ID_CA, /* arg1= int i_program_number, uint16_t i_vpid, uint16_t i_apid1, uint16_t i_apid2, uint16_t i_apid3, uint8_t i_length, uint8_t *p_data */
124 STREAM_GET_PRIVATE_ID_STATE, /* arg1=int i_private_data arg2=bool * res=can fail */
125};

136/**
137 * Get the current position in a stream
138 */
139static inline int64_t stream_Tell( stream_t *s )
140{
141 uint64_t i_pos;
142 stream_Control( s, STREAM_GET_POSITION, &i_pos );
143 if( i_pos >> 62 )
144 return (int64_t)1 << 62;
145 return i_pos;
146}
147
148/**
149 * Get the size of the stream.
150 */
151static inline int64_t stream_Size( stream_t *s )
152{
153 uint64_t i_pos;
154 stream_Control( s, STREAM_GET_SIZE, &i_pos );
155 if( i_pos >> 62 )
156 return (int64_t)1 << 62;
157 return i_pos;
158}
159
160static inline int stream_Seek( stream_t *s, uint64_t i_pos )
161{
162 return stream_Control( s, STREAM_SET_POSITION, i_pos );
163}
164
165/**
166 * Get the Content-Type of a stream, or NULL if unknown.
167 * Result must be free()'d.
168 */
169static inline char *stream_ContentType( stream_t *s )
170{
171 char *res;
172 if( stream_Control( s, STREAM_GET_CONTENT_TYPE, &res ) )
173 return NULL;
174 return res;
175}

194static inline int stream_DemuxControl( stream_t *s, int query, ... )
195{
196 va_list ap;
197 int ret;
198
199 va_start( ap, query );
200 ret = stream_DemuxControlVa( s, query, ap );
201 va_end( ap );
202 return ret;
203}
