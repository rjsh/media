41struct config_category_t
42{
43 int i_id;
44 const char *psz_name;
45 const char *psz_help;
46};
47
48typedef union
49{
50 char *psz;
51 int64_t i;
52 float f;
53} module_value_t;
54
55typedef int (*vlc_string_list_cb)(vlc_object_t *, const char *,
56 char ***, char ***);
57typedef int (*vlc_integer_list_cb)(vlc_object_t *, const char *,
58 int64_t **, char ***);
59
60struct module_config_t
61{
62 uint8_t i_type; /* Configuration type */
63 char i_short; /* Optional short option name */
64 unsigned b_advanced:1; /* Advanced option */
65 unsigned b_internal:1; /* Hidden from prefs and help */
66 unsigned b_unsaveable:1; /* Not stored in configuration */
67 unsigned b_safe:1; /* Safe in web plugins and playlists */
68 unsigned b_removed:1; /* Deprecated */
69
70 char *psz_type; /* Configuration subtype */
71 char *psz_name; /* Option name */
72 char *psz_text; /* Short comment on the configuration option */
73 char *psz_longtext; /* Long comment on the configuration option */
74
75 module_value_t value; /* Option value */
76 module_value_t orig;
77 module_value_t min;
78 module_value_t max;
79
80 /* Values list */
81 uint16_t list_count; /* Options list size */
82 union
83 {
84 char **psz; /* List of possible values for the option */
85 int *i;
86 vlc_string_list_cb psz_cb;
87 vlc_integer_list_cb i_cb;
88 } list;
89 char **list_text; /* Friendly names for list values */
90};

153/****************************************************************************
154 * config_chain_t:
155 ****************************************************************************/
156struct config_chain_t
157{
158 config_chain_t *p_next; /**< Pointer on the next config_chain_t element */
159
160 char *psz_name; /**< Option name */
161 char *psz_value; /**< Option value */
162};
163
164/**
165 * This function will
166 * - create all options in the array ppsz_options (var_Create).
167 * - parse the given linked list of config_chain_t and set the value (var_Set).
168 *
169 * The option names will be created by adding the psz_prefix prefix.
170 */
171VLC_API void config_ChainParse( vlc_object_t *, const char *psz_prefix, const char *const *ppsz_options, config_chain_t * );
172#define config_ChainParse( a, b, c, d ) config_ChainParse( VLC_OBJECT(a), b, c, d )
173
174/**
175 * This function will parse a configuration string (psz_opts) and
176 * - set all options for this module in a chained list (*pp_cfg)
177 * - returns a pointer on the next module if any.
178 *
179 * The string format is
180 * module{option=*,option=*}
181 *
182 * The options values are unescaped using config_StringUnescape.
183 */
184VLC_API const char *config_ChainParseOptions( config_chain_t **pp_cfg, const char *ppsz_opts );
185
186/**
187 * This function will parse a configuration string (psz_string) and
188 * - set the module name (*ppsz_name)
189 * - set all options for this module in a chained list (*pp_cfg)
190 * - returns a pointer on the next module if any.
191 *
192 * The string format is
193 * module{option=*,option=*}[:modulenext{option=*,...}]
194 *
195 * The options values are unescaped using config_StringUnescape.
196 */
197VLC_API char *config_ChainCreate( char **ppsz_name, config_chain_t **pp_cfg, const char *psz_string ) VLC_USED VLC_MALLOC;
198
199/**
200 * This function will release a linked list of config_chain_t
201 * (Including the head)
202 */
203VLC_API void config_ChainDestroy( config_chain_t * );
204
205/**
206 * This function will duplicate a linked list of config_chain_t
207 */
208VLC_API config_chain_t * config_ChainDuplicate( const config_chain_t * ) VLC_USED VLC_MALLOC;
209
210/**
211 * This function will unescape a string in place and will return a pointer on
212 * the given string.
213 * No memory is allocated by it (unlike config_StringEscape).
214 * If NULL is given as parameter nothing will be done (NULL will be returned).
215 *
216 * The following sequences will be unescaped (only one time):
217 * \\ \' and \"
218 */
219VLC_API char * config_StringUnescape( char *psz_string );
220
221/**
222 * This function will escape a string that can be unescaped by
223 * config_StringUnescape.
224 * The returned value is allocated by it. You have to free it once you
225 * do not need it anymore (unlike config_StringUnescape).
226 * If NULL is given as parameter nothing will be done (NULL will be returned).
227 *
228 * The escaped characters are ' " and \
229 */
230VLC_API char * config_StringEscape( const char *psz_string ) VLC_USED VLC_MALLOC;
