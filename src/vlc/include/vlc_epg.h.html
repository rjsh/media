24#ifndef VLC_EPG_H
25#define VLC_EPG_H 1
26
27/**
28 * \file
29 * This file defines functions and structures for storing dvb epg information
30 */
31
32typedef struct
33{
34 int64_t i_start; /* Interpreted as a value return by time() */
35 int i_duration; /* Duration of the event in second */
36
37 char *psz_name;
38 char *psz_short_description;
39 char *psz_description;
40
41 uint8_t i_rating; /* Parental control, set to 0 when undefined */
42} vlc_epg_event_t;
43
44typedef struct
45{
46 char *psz_name;
47 vlc_epg_event_t *p_current; /* Can be null or should be the same than one of pp_event entry */
48
49 int i_event;
50 vlc_epg_event_t **pp_event;
51} vlc_epg_t;
52
53/**
54 * It initializes a vlc_epg_t.
55 *
56 * You must call vlc_epg_Clean to release the associated resource.
57 */
58VLC_API void vlc_epg_Init(vlc_epg_t *p_epg, const char *psz_name);
59
60/**
61 * It releases all resources associated to a vlc_epg_t
62 */
63VLC_API void vlc_epg_Clean(vlc_epg_t *p_epg);
64
65/**
66 * It creates and appends a new vlc_epg_event_t to a vlc_epg_t.
67 *
68 * \see vlc_epg_t for the definitions of the parameters.
69 */
70VLC_API void vlc_epg_AddEvent(vlc_epg_t *p_epg, int64_t i_start, int i_duration, const char *psz_name, const char *psz_short_description, const char *psz_description, uint8_t i_rating );
71
72/**
73 * It creates a new vlc_epg_t*
74 *
75 * You must call vlc_epg_Delete to release the associated resource.
76 */
77VLC_API vlc_epg_t * vlc_epg_New(const char *psz_name) VLC_USED;
78
79/**
80 * It releases a vlc_epg_t*.
81 */
82VLC_API void vlc_epg_Delete(vlc_epg_t *p_epg);
83
84/**
85 * It set the current event of a vlc_epg_t given a start time
86 */
87VLC_API void vlc_epg_SetCurrent(vlc_epg_t *p_epg, int64_t i_start);
88
89/**
90 * It merges all the event of \p p_src and \p p_dst into \p p_dst.
91 *
92 * \p p_src is not modified.
93 */
94VLC_API void vlc_epg_Merge(vlc_epg_t *p_dst, const vlc_epg_t *p_src);
95
96#endif
