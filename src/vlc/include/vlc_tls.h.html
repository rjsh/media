22#ifndef VLC_TLS_H
23# define VLC_TLS_H
24
25/**
26 * \file
27 * This file defines Transport Layer Security API (TLS) in vlc
28 */
29
30# include <vlc_network.h>
31
32typedef struct vlc_tls vlc_tls_t;
33typedef struct vlc_tls_sys vlc_tls_sys_t;
34typedef struct vlc_tls_creds vlc_tls_creds_t;
35typedef struct vlc_tls_creds_sys vlc_tls_creds_sys_t;
36
37/** TLS session */
38struct vlc_tls
39{
40 VLC_COMMON_MEMBERS
41
42 vlc_tls_sys_t *sys;
43
44 struct virtual_socket_t sock;
45 int (*handshake) (vlc_tls_t *, const char *host, const char *service);
46};
47
48VLC_API vlc_tls_t *vlc_tls_ClientSessionCreate (vlc_tls_creds_t *, int fd,
49 const char *host, const char *service);
50vlc_tls_t *vlc_tls_SessionCreate (vlc_tls_creds_t *, int fd, const char *host);
51int vlc_tls_SessionHandshake (vlc_tls_t *, const char *host, const char *serv);
52VLC_API void vlc_tls_SessionDelete (vlc_tls_t *);
53
54/* NOTE: It is assumed that a->sock.p_sys = a */
55# define tls_Send( a, b, c ) (((vlc_tls_t *)a)->sock.pf_send (a, b, c))
56
57# define tls_Recv( a, b, c ) (((vlc_tls_t *)a)->sock.pf_recv (a, b, c))
58
59
60/** TLS credentials (certificate, private and trust settings) */
61struct vlc_tls_creds
62{
63 VLC_COMMON_MEMBERS
64
65 module_t *module;
66 vlc_tls_creds_sys_t *sys;
67
68 int (*add_CA) (vlc_tls_creds_t *, const char *path);
69 int (*add_CRL) (vlc_tls_creds_t *, const char *path);
70
71 int (*open) (vlc_tls_creds_t *, vlc_tls_t *, int fd, const char *host);
72 void (*close) (vlc_tls_creds_t *, vlc_tls_t *);
73};
74
75VLC_API vlc_tls_creds_t *vlc_tls_ClientCreate (vlc_object_t *);
76vlc_tls_creds_t *vlc_tls_ServerCreate (vlc_object_t *,
77 const char *cert, const char *key);
78VLC_API void vlc_tls_Delete (vlc_tls_creds_t *);
79int vlc_tls_ServerAddCA (vlc_tls_creds_t *srv, const char *path);
80int vlc_tls_ServerAddCRL (vlc_tls_creds_t *srv, const char *path);
81
82#endif
