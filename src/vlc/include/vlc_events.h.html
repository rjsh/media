40/*
41 **** Background
42 *
43 * This implements a way to send and receive event for an object (which can be
44 * a simple C struct or less).
45 *
46 * This is in direct concurrency with the Variable based Callback
47 * (see src/misc/variables.c).
48 *
49 * It has the following advantages over Variable based Callback:
50 * - No need to implement the whole VLC_COMMON_MEMBERS in the object,
51 * thus it reduce it size. This is especially true for input_item_t which
52 * doesn't have VLC_COMMON_MEMBERS. This is the first reason of existence of
53 * this implementation.
54 * - Libvlc can easily be based upon that.
55 * - Existing event are clearly declared (in include/vlc_events.h)
56 *
57 *
58 **** Example usage
59 *
60 * (vlc_cool_object_t doesn't need to have the VLC_COMMON_MEMBERS.)
61 *
62 * struct vlc_cool_object_t
63 * {
64 * ...
65 * vlc_event_manager_t p_event_manager;
66 * ...
67 * }
68 *
69 * vlc_my_cool_object_new()
70 * {
71 * ...
72 * vlc_event_manager_init( &p_self->p_event_manager, p_self, p_a_libvlc_object );
73 * vlc_event_manager_register_event_type(p_self->p_event_manager,
74 * vlc_MyCoolObjectDidSomething, p_e)
75 * ...
76 * }
77 *
78 * vlc_my_cool_object_release()
79 * {
80 * ...
81 * vlc_event_manager_fini( &p_self->p_event_manager );
82 * ...
83 * }
84 *
85 * vlc_my_cool_object_do_something()
86 * {
87 * ...
88 * vlc_event_t event;
89 * event.type = vlc_MyCoolObjectDidSomething;
90 * event.u.my_cool_object_did_something.what_it_did = kSomething;
91 * vlc_event_send( &p_self->p_event_manager, &event );
92 * }
93 * */
94
95 /*****************************************************************************
96 * Event Type
97 *****************************************************************************/
98
99/* Private structure defined in misc/events.c */
100struct vlc_event_listeners_group_t;
101
102/* Event manager type */
103typedef struct vlc_event_manager_t
104{
105 void * p_obj;
106 vlc_mutex_t object_lock;
107 vlc_mutex_t event_sending_lock;
108 DECL_ARRAY(struct vlc_event_listeners_group_t *) listeners_groups;
109} vlc_event_manager_t;
110
111/* List of event */
112typedef enum vlc_event_type_t {
113 /* Input (thread) events */
114 vlc_InputStateChanged,
115 vlc_InputSelectedStreamChanged,
116
117 /* Input item events */
118 vlc_InputItemMetaChanged,
119 vlc_InputItemSubItemAdded,
120 vlc_InputItemSubItemTreeAdded,
121 vlc_InputItemDurationChanged,
122 vlc_InputItemPreparsedChanged,
123 vlc_InputItemNameChanged,
124 vlc_InputItemInfoChanged,
125 vlc_InputItemErrorWhenReadingChanged,
126
127 /* Service Discovery event */
128 vlc_ServicesDiscoveryItemAdded,
129 vlc_ServicesDiscoveryItemRemoved,
130 vlc_ServicesDiscoveryItemRemoveAll,
131 vlc_ServicesDiscoveryStarted,
132 vlc_ServicesDiscoveryEnded,
133
134 /* Addons Manager events */
135 vlc_AddonFound,
136 vlc_AddonsDiscoveryEnded,
137 vlc_AddonChanged
138} vlc_event_type_t;
139
140/* Event definition */
141typedef struct vlc_event_t
142{
143 vlc_event_type_t type;
144 void * p_obj; /* Sender object, automatically filled by vlc_event_send() */
145 union vlc_event_type_specific
146 {
147 /* Input (thread) events */
148 struct vlc_input_state_changed
149 {
150 int new_state;
151 } input_state_changed;
152 struct vlc_input_selected_stream_changed
153 {
154 void * unused;
155 } input_selected_stream_changed;
156
157 /* Input item events */
158 struct vlc_input_item_meta_changed
159 {
160 vlc_meta_type_t meta_type;
161 } input_item_meta_changed;
162 struct vlc_input_item_subitem_added
163 {
164 input_item_t * p_new_child;
165 } input_item_subitem_added;
166 struct vlc_input_item_subitem_tree_added
167 {
168 input_item_node_t * p_root;
169 } input_item_subitem_tree_added;
170 struct vlc_input_item_duration_changed
171 {
172 mtime_t new_duration;
173 } input_item_duration_changed;
174 struct vlc_input_item_preparsed_changed
175 {
176 int new_status;
177 } input_item_preparsed_changed;
178 struct vlc_input_item_name_changed
179 {
180 const char * new_name;
181 } input_item_name_changed;
182 struct vlc_input_item_info_changed
183 {
184 void * unused;
185 } input_item_info_changed;
186 struct input_item_error_when_reading_changed
187 {
188 bool new_value;
189 } input_item_error_when_reading_changed;
190
191 /* Service discovery events */
192 struct vlc_services_discovery_item_added
193 {
194 input_item_t * p_new_item;
195 const char * psz_category;
196 } services_discovery_item_added;
197 struct vlc_services_discovery_item_removed
198 {
199 input_item_t * p_item;
200 } services_discovery_item_removed;
201 struct vlc_services_discovery_started
202 {
203 void * unused;
204 } services_discovery_started;
205 struct vlc_services_discovery_ended
206 {
207 void * unused;
208 } services_discovery_ended;
209
210 /* Addons */
211 struct vlc_addon_generic_event
212 {
213 addon_entry_t * p_entry;
214 } addon_generic_event;
215 } u;
216} vlc_event_t;
217
218/* Event callback type */
219typedef void ( *vlc_event_callback_t )( const vlc_event_t *, void * );

225/*
226 * p_obj points to the object that owns the event manager, and from
227 * which events are sent
228 */
229VLC_API int vlc_event_manager_init( vlc_event_manager_t * p_em, void * p_obj );
230
231/*
232 * Destroy
233 */
234VLC_API void vlc_event_manager_fini( vlc_event_manager_t * p_em );
235
236/*
237 * Tells a specific event manager that it will handle event_type object
238 */
239VLC_API int vlc_event_manager_register_event_type( vlc_event_manager_t * p_em,
240 vlc_event_type_t );
241
242/*
243 * Send an event to the listener attached to this p_em.
244 */
245VLC_API void vlc_event_send( vlc_event_manager_t * p_em, vlc_event_t * );
246
247/*
248 * Add a callback for an event.
249 */
250VLC_API int vlc_event_attach( vlc_event_manager_t * p_event_manager,
251 vlc_event_type_t event_type,
252 vlc_event_callback_t pf_callback,
253 void *p_user_data );
254
255/*
256 * Remove a callback for an event.
257 */
258VLC_API void vlc_event_detach( vlc_event_manager_t *p_event_manager,
259 vlc_event_type_t event_type,
260 vlc_event_callback_t pf_callback,
261 void *p_user_data );
