25#ifndef VLC_VOUT_WINDOW_H
26#define VLC_VOUT_WINDOW_H 1
27
28/**
29 * \file
30 * This file defines vout windows structures and functions in vlc
31 */
32
33#include <vlc_common.h>
34
35/* */
36typedef struct vout_window_t vout_window_t;
37typedef struct vout_window_sys_t vout_window_sys_t;
38
39
40/**
41 * Window handle type
42 */
43enum {
44 VOUT_WINDOW_TYPE_INVALID=0,
45 VOUT_WINDOW_TYPE_XID,
46 VOUT_WINDOW_TYPE_HWND,
47 VOUT_WINDOW_TYPE_NSOBJECT,
48 VOUT_WINDOW_TYPE_ANDROID_NATIVE,
49};
50
51/**
52 * Control query for vout_window_t
53 */
54enum {
55 VOUT_WINDOW_SET_STATE, /* unsigned state */
56 VOUT_WINDOW_SET_SIZE, /* unsigned i_width, unsigned i_height */
57 VOUT_WINDOW_SET_FULLSCREEN, /* int b_fullscreen */
58};
59
60typedef struct {
61 /* If true, a standalone window is requested */
62 bool is_standalone;
63
64 /* Window handle type */
65 unsigned type;
66
67 /* Window position hint */
68 int x;
69 int y;
70
71 /* Windows size hint */
72 unsigned width;
73 unsigned height;
74
75} vout_window_cfg_t;
76
77/**
78 * FIXME do we need an event system in the window too ?
79 * or the window user will take care of it ?
80 */
81struct vout_window_t {
82 VLC_COMMON_MEMBERS
83
84 unsigned type; /**< Window handle type */
85
86 /* window handle (mandatory)
87 *
88 * It must be filled in the open function.
89 */
90 union {
91 void *hwnd; /* Win32 window handle */
92 uint32_t xid; /* X11 windows ID */
93 void *nsobject; /* Mac OSX view object */
94 void *anativewindow; /* Android native window. */
95 } handle;
96
97 /* display server (mandatory) */
98 union {
99 char *x11; /* X11 display (NULL = use default) */
100 } display;
101
102 /* Control on the module (mandatory)
103 *
104 * Do not use it directly; use vout_window_Control instead.
105 */
106 int (*control)(vout_window_t *, int query, va_list);
107
108 /* Private place holder for the vout_window_t module (optional)
109 *
110 * A module is free to use it as it wishes.
111 */
112 vout_window_sys_t *sys;
113};
114
115/**
116 * Creates a new window.
117 *
118 * @param module plugin name (usually "$window")
119 * @note If you are inside a "vout display", you must use
120 / vout_display_NewWindow() and vout_display_DeleteWindow() instead.
121 * This enables recycling windows.
122 */
123VLC_API vout_window_t * vout_window_New(vlc_object_t *, const char *module, const vout_window_cfg_t *);
124
125/**
126 * Deletes a window created by vout_window_New().
127 *
128 * @note See vout_window_New() about window recycling.
129 */
130VLC_API void vout_window_Delete(vout_window_t *);
131
132
133/**
134 * Reconfigures a window.
135 *
136 * @note The vout_window_* wrappers should be used instead of this function.
137 *
138 * @warning The caller must own the window, as vout_window_t is not thread safe.
139 */
140VLC_API int vout_window_Control(vout_window_t *, int query, ...);
141
142/**
143 * Configures the window manager state for this window.
144 */
145static inline int vout_window_SetState(vout_window_t *window, unsigned state)
146{
147 return vout_window_Control(window, VOUT_WINDOW_SET_STATE, state);
148}
149
150/**
151 * Configures the window display (i.e. inner/useful) size.
152 */
153static inline int vout_window_SetSize(vout_window_t *window,
154 unsigned width, unsigned height)
155{
156 return vout_window_Control(window, VOUT_WINDOW_SET_SIZE, width, height);
157}
158
159/**
160 * Sets fullscreen mode.
161 */
162static inline int vout_window_SetFullScreen(vout_window_t *window, bool full)
163{
164 return vout_window_Control(window, VOUT_WINDOW_SET_FULLSCREEN, full);
165}
166
167#endif /* VLC_VOUT_WINDOW_H */
