195/* Explanation:
196 *
197 * if linking a module statically, we will need:
198 * #define MODULE_FUNC( zog ) module_foo_zog
199 *
200 * this can't easily be done with the C preprocessor, thus a few ugly hacks.

242/*
243 * InitModule: this function is called once and only once, when the module
244 * is looked at for the first time. We get the useful data from it, for
245 * instance the module name, its shortcuts, its capabilities... we also create
246 * a copy of its config because the module can be unloaded at any time.
247 */
248#define vlc_module_begin() \
249EXTERN_SYMBOL DLL_SYMBOL \
250int CDECL_SYMBOL __VLC_SYMBOL(vlc_entry) (vlc_set_cb, void *); \
251EXTERN_SYMBOL DLL_SYMBOL \
252int CDECL_SYMBOL __VLC_SYMBOL(vlc_entry) (vlc_set_cb vlc_set, void *opaque) \
253{ \
254 module_t *module; \
255 module_config_t *config = NULL; \
256 if (vlc_plugin_set (VLC_MODULE_CREATE, &module)) \
257 goto error; \
258 if (vlc_module_set (VLC_MODULE_NAME, (MODULE_STRING))) \
259 goto error;
260
261#define vlc_module_end() \
262 (void) config; \
263 return 0; \
264error: \
265 return -1; \
266} \
267VLC_METADATA_EXPORTS

311/*****************************************************************************
312 * Macros used to build the configuration structure.
313 *
314 * Note that internally we support only 3 types of config data: int, float
315 * and string.
316 * The other types declared here just map to one of these 3 basic types but
317 * have the advantage of also providing very good hints to a configuration
318 * interface so as to make it more user friendly.
319 * The configuration structure also includes category hints. These hints can
320 * provide a configuration interface with some very useful data and again
321 * allow for a more user friendly interface.
322 *****************************************************************************/
