132/* Convert Unicode code point to UTF-8 */
133static char *utf8_cp (uint_fast32_t cp, char *buf)
134{
135 if (cp < (1 << 7))
136 {
137 buf[1] = 0;
138 buf[0] = cp;
139 }
140 else if (cp < (1 << 11))
141 {
142 buf[2] = 0;
143 buf[1] = 0x80 | (cp & 0x3F);
144 cp >>= 6;
145 buf[0] = 0xC0 | cp;
146 }
147 else if (cp < (1 << 16))
148 {
149 buf[3] = 0;
150 buf[2] = 0x80 | (cp & 0x3F);
151 cp >>= 6;
152 buf[1] = 0x80 | (cp & 0x3F);
153 cp >>= 6;
154 buf[0] = 0xE0 | cp;
155 }
156 else if (cp < (1 << 21))
157 {
158 buf[4] = 0;
159 buf[3] = 0x80 | (cp & 0x3F);
160 cp >>= 6;
161 buf[2] = 0x80 | (cp & 0x3F);
162 cp >>= 6;
163 buf[1] = 0x80 | (cp & 0x3F);
164 cp >>= 6;
165 buf[0] = 0xE0 | cp;
166 }
167 else
168 return NULL;
169 return buf;
170}

172/**
173 * Parse a human-readable string representation of a VLC key code.
174 * @note This only works with the American English representation
175 * (a.k.a. C or POSIX), not with the local representation returned from
176 * vlc_keycode2str().
177 * @return a VLC key code, or KEY_UNSET on failure.
178 */
179uint_fast32_t vlc_str2keycode (const char *name)
180{
181 uint_fast32_t mods = 0;
182 uint32_t code;
183
184 for (;;)
185 {
186 size_t len = strcspn (name, "-+");
187 if (len == 0 || name[len] == '\0')
188 break;
189
190 if (len == 4 && !strncasecmp (name, "Ctrl", 4))
191 mods |= KEY_MODIFIER_CTRL;
192 if (len == 3 && !strncasecmp (name, "Alt", 3))
193 mods |= KEY_MODIFIER_ALT;
194 if (len == 5 && !strncasecmp (name, "Shift", 5))
195 mods |= KEY_MODIFIER_SHIFT;
196 if (len == 4 && !strncasecmp (name, "Meta", 4))
197 mods |= KEY_MODIFIER_META;
198 if (len == 7 && !strncasecmp (name, "Command", 7))
199 mods |= KEY_MODIFIER_COMMAND;
200
201 name += len + 1;
202 }
203
204 key_descriptor_t *d = bsearch (name, vlc_keys, KEYS_COUNT,
205 sizeof (vlc_keys[0]), keystrcmp);
206 if (d != NULL)
207 code = d->i_key_code;
208 else
209 if (vlc_towc (name, &code) <= 0)
210 code = KEY_UNSET;
211
212 if (code != KEY_UNSET)
213 code |= mods;
214 return code;
215}

217static char *nooptext (const char *txt)
218{
219 return (char *)txt;
220}

222/**
223 * Format a human-readable and unique representation of a VLC key code
224 * (including modifiers).
225 * @param code key code to translate to a string
226 * @param locale true to get a localized string,
227 * false to get a C string suitable for 'vlcrc'
228 * @return a heap-allocated string, or NULL on error.
229 */
230char *vlc_keycode2str (uint_fast32_t code, bool locale)
231{
232 char *(*tr) (const char *) = locale ? vlc_gettext : nooptext;
233 const char *name;
234 char *str, buf[5];
235 uintptr_t key = code & ~KEY_MODIFIER;
236
237 for (size_t i = 0; i < KEYS_COUNT; i++)
238 if (vlc_keys[i].i_key_code == key)
239 {
240 name = vlc_keys[i].psz_key_string;
241 goto found;
242 }
243
244 if (utf8_cp (key, buf) == NULL)
245 return NULL;
246 name = buf;
247
248found:
249 if (asprintf (&str, "%s%s%s%s%s%s",
250 (code & KEY_MODIFIER_CTRL) ? tr(N_("Ctrl+")) : "",
251 (code & KEY_MODIFIER_ALT) ? tr(N_("Alt+")) : "",
252 (code & KEY_MODIFIER_SHIFT) ? tr(N_("Shift+")) : "",
253 (code & KEY_MODIFIER_META) ? tr(N_("Meta+")) : "",
254 (code & KEY_MODIFIER_COMMAND) ? tr(N_("Command+")) : "",
255 tr(name)) == -1)
256 return NULL;
257 return str;
258}

379struct mapping
380{
381 uint32_t key; ///< Key code
382 vlc_action_t action; ///< Action ID
383};
384
385static int keycmp (const void *a, const void *b)
386{
387 const struct mapping *ka = a, *kb = b;
388
389#if (INT_MAX >= 0x7fffffff)
390 return ka->key - kb->key;
391#else
392 return (ka->key < kb->key) ? -1 : (ka->key > kb->key) ? +1 : 0;
393#endif
394}

396struct vlc_actions
397{
398 void *map; /* Key map */
399 void *global_map; /* Grabbed/global key map */
400 struct hotkey keys[1];
401};
402
403static int vlc_key_to_action (vlc_object_t *obj, const char *varname,
404 vlc_value_t prevkey, vlc_value_t curkey, void *d)
405{
406 void *const *map = d;
407 const struct mapping **pent;
408 uint32_t keycode = curkey.i_int;
409
410 pent = tfind (&keycode, map, keycmp);
411 if (pent == NULL)
412 return VLC_SUCCESS;
413
414 (void) varname;
415 (void) prevkey;
416 return var_SetInteger (obj, "key-action", (*pent)->action);
417}
418
419/**
420 * Adds a mapping from a certain key code to a certain action.
421 */
422static int vlc_AddMapping (void **map, uint32_t keycode, vlc_action_t action)
423{
424 struct mapping *entry = malloc (sizeof (*entry));
425 if (entry == NULL)
426 return ENOMEM;
427 entry->key = keycode;
428 entry->action = action;
429
430 struct mapping **pent = tsearch (entry, map, keycmp);
431 if (unlikely(pent == NULL))
432 return ENOMEM;
433 if (*pent != entry)
434 {
435 free (entry);
436 return EEXIST;
437 }
438 return 0;
439}

441/**
442 * Sets up all key mappings for a given action.
443 * \param map tree (of struct mapping entries) to write mappings to
444 * \param confname VLC configuration item to read mappings from
445 * \param action action ID
446 */
447static void vlc_InitAction (vlc_object_t *obj, void **map,
448 const char *confname, vlc_action_t action)
449{
450 char *keys = var_InheritString (obj, confname);
451 if (keys == NULL)
452 return;
453
454 for (char *buf, *key = strtok_r (keys, "\t", &buf);
455 key != NULL;
456 key = strtok_r (NULL, "\t", &buf))
457 {
458 uint32_t code = vlc_str2keycode (key);
459 if (code == KEY_UNSET)
460 {
461 msg_Warn (obj, "Key \"%s\" unrecognized", key);
462 continue;
463 }
464
465 if (vlc_AddMapping (map, code, action) == EEXIST)
466 msg_Warn (obj, "Key \"%s\" bound to multiple actions", key);
467 }
468 free (keys);
469}

471/**
472 * Initializes the key map from configuration.
473 */
474struct vlc_actions *vlc_InitActions (libvlc_int_t *libvlc)
475{
476 vlc_object_t *obj = VLC_OBJECT(libvlc);
477 struct hotkey *keys;
478 struct vlc_actions *as = malloc (sizeof (*as) + ACTIONS_COUNT * sizeof (*keys));
479
480 if (unlikely(as == NULL))
481 return NULL;
482 as->map = NULL;
483 as->global_map = NULL;
484 keys = as->keys;
485
486 var_Create (obj, "key-pressed", VLC_VAR_INTEGER);
487 var_Create (obj, "global-key-pressed", VLC_VAR_INTEGER);
488 var_Create (obj, "key-action", VLC_VAR_INTEGER);
489
490 /* Initialize from configuration */
491 for (size_t i = 0; i < ACTIONS_COUNT; i++)
492 {
493#ifndef NDEBUG
494 if (i > 0
495 && strcmp (actions[i-1].name, actions[i].name) >= 0)
496 {
497 msg_Err (libvlc, "key-%s and key-%s are not ordered properly",
498 actions[i-1].name, actions[i].name);
499 abort ();
500 }
501#endif
502 keys->psz_action = actions[i].name;
503 keys++;
504
505 char name[12 + MAXACTION];
506
507 snprintf (name, sizeof (name), "global-key-%s", actions[i].name);
508 vlc_InitAction (obj, &as->map, name + 7, actions[i].value);
509 vlc_InitAction (obj, &as->global_map, name, actions[i].value);
510 }
511 keys->psz_action = NULL;
512
513 /* Initialize mouse wheel events */
514 int mousemode = var_InheritInteger (obj, "hotkeys-mousewheel-mode");
515 if (mousemode < 2)
516 {
517 vlc_AddMapping (&as->map,
518 mousemode ? KEY_MOUSEWHEELRIGHT : KEY_MOUSEWHEELUP,
519 ACTIONID_VOL_UP);
520 vlc_AddMapping (&as->map,
521 mousemode ? KEY_MOUSEWHEELLEFT : KEY_MOUSEWHEELDOWN,
522 ACTIONID_VOL_DOWN);
523 vlc_AddMapping (&as->map,
524 mousemode ? KEY_MOUSEWHEELUP : KEY_MOUSEWHEELRIGHT,
525 ACTIONID_JUMP_FORWARD_EXTRASHORT);
526 vlc_AddMapping (&as->map,
527 mousemode ? KEY_MOUSEWHEELDOWN : KEY_MOUSEWHEELLEFT,
528 ACTIONID_JUMP_BACKWARD_EXTRASHORT);
529 }
530
531
532 libvlc->p_hotkeys = as->keys;
533 var_AddCallback (obj, "key-pressed", vlc_key_to_action, &as->map);
534 var_AddCallback (obj, "global-key-pressed", vlc_key_to_action,
535 &as->global_map);
536 return as;
537}
538
539/**
540 * Destroys the key map.
541 */
542void vlc_DeinitActions (libvlc_int_t *libvlc, struct vlc_actions *as)
543{
544 if (unlikely(as == NULL))
545 return;
546
547 var_DelCallback (libvlc, "global-key-pressed", vlc_key_to_action,
548 &as->global_map);
549 var_DelCallback (libvlc, "key-pressed", vlc_key_to_action, &as->map);
550
551 tdestroy (as->global_map, free);
552 tdestroy (as->map, free);
553 free (as);
554 libvlc->p_hotkeys = NULL;
555}

558static int actcmp(const void *key, const void *ent)
559{
560 const struct action *act = ent;
561 return strcmp(key, act->name);
562}

564/**
565 * Get the action ID from the action name in the configuration subsystem.
566 * @return the action ID or ACTIONID_NONE on error.
567 */
568vlc_action_t vlc_GetActionId (const char *name)
569{
570 const struct action *act;
571
572 if (strncmp (name, "key-", 4))
573 return ACTIONID_NONE;
574 name += 4;
575
576 act = bsearch(name, actions, ACTIONS_COUNT, sizeof(*act), actcmp);
577 return (act != NULL) ? act->value : ACTIONID_NONE;
578}
