41/**
42 * Allocate a new picture in the heap.
43 *
44 * This function allocates a fake direct buffer in memory, which can be
45 * used exactly like a video buffer. The video output thread then manages
46 * how it gets displayed.
47 */
48static int AllocatePicture( picture_t *p_pic )
49{
50 /* Calculate how big the new image should be */
51 size_t i_bytes = 0;
52 for( int i = 0; i < p_pic->i_planes; i++ )
53 {
54 const plane_t *p = &p_pic->p[i];
55
56 if( p->i_pitch < 0 || p->i_lines <= 0 ||
57 (size_t)p->i_pitch > (SIZE_MAX - i_bytes)/p->i_lines )
58 {
59 p_pic->i_planes = 0;
60 return VLC_ENOMEM;
61 }
62 i_bytes += p->i_pitch * p->i_lines;
63 }
64
65 uint8_t *p_data = vlc_memalign( 16, i_bytes );
66 if( i_bytes > 0 && p_data == NULL )
67 {
68 p_pic->i_planes = 0;
69 return VLC_EGENERIC;
70 }
71 p_pic->gc.p_sys = (void *)p_data;
72
73 /* Fill the p_pixels field for each plane */
74 p_pic->p[0].p_pixels = p_data;
75 for( int i = 1; i < p_pic->i_planes; i++ )
76 {
77 p_pic->p[i].p_pixels = &p_pic->p[i-1].p_pixels[ p_pic->p[i-1].i_lines *
78 p_pic->p[i-1].i_pitch ];
79 }
80
81 return VLC_SUCCESS;
82}

88static void PictureDestroyContext( picture_t *p_picture )
89{
90 void (**context)( void * ) = p_picture->context;
91 if( context != NULL )
92 {
93 void (*context_destroy)( void * ) = *context;
94 context_destroy( context );
95 p_picture->context = NULL;
96 }
97}

99static void PictureDestroy( picture_t *p_picture )
100{
101 assert( p_picture &&
102 atomic_load( &p_picture->gc.refcount ) == 0 );
103
104 vlc_free( p_picture->gc.p_sys );
105 free( p_picture->p_sys );
106 free( p_picture );
107}

126static int LCM( int a, int b )
127{
128 return a * b / GCD( a, b );

131int picture_Setup( picture_t *p_picture, const video_format_t *restrict fmt )
132{
133 /* Store default values */
134 p_picture->i_planes = 0;
135 for( unsigned i = 0; i < VOUT_MAX_PLANES; i++ )
136 {
137 plane_t *p = &p_picture->p[i];
138 p->p_pixels = NULL;
139 p->i_pixel_pitch = 0;
140 }
141
142 atomic_init( &p_picture->gc.refcount, 0 );
143 p_picture->gc.pf_destroy = NULL;
144 p_picture->gc.p_sys = NULL;
145
146 p_picture->i_nb_fields = 2;
147
148 video_format_Setup( &p_picture->format, fmt->i_chroma, fmt->i_width, fmt->i_height,
149 fmt->i_visible_width, fmt->i_visible_height,
150 fmt->i_sar_num, fmt->i_sar_den );
151
152 const vlc_chroma_description_t *p_dsc =
153 vlc_fourcc_GetChromaDescription( p_picture->format.i_chroma );
154 if( !p_dsc )
155 return VLC_EGENERIC;
156
157 /* We want V (width/height) to respect:
158 (V * p_dsc->p[i].w.i_num) % p_dsc->p[i].w.i_den == 0
159 (V * p_dsc->p[i].w.i_num/p_dsc->p[i].w.i_den * p_dsc->i_pixel_size) % 16 == 0
160 Which is respected if you have
161 V % lcm( p_dsc->p[0..planes].w.i_den * 16) == 0
162 */
163 int i_modulo_w = 1;
164 int i_modulo_h = 1;
165 unsigned int i_ratio_h = 1;
166 for( unsigned i = 0; i < p_dsc->plane_count; i++ )
167 {
168 i_modulo_w = LCM( i_modulo_w, 16 * p_dsc->p[i].w.den );
169 i_modulo_h = LCM( i_modulo_h, 16 * p_dsc->p[i].h.den );
170 if( i_ratio_h < p_dsc->p[i].h.den )
171 i_ratio_h = p_dsc->p[i].h.den;
172 }
173 i_modulo_h = LCM( i_modulo_h, 32 );
174
175 const int i_width_aligned = ( fmt->i_width + i_modulo_w - 1 ) / i_modulo_w * i_modulo_w;
176 const int i_height_aligned = ( fmt->i_height + i_modulo_h - 1 ) / i_modulo_h * i_modulo_h;
177 const int i_height_extra = 2 * i_ratio_h; /* This one is a hack for some ASM functions */
178 for( unsigned i = 0; i < p_dsc->plane_count; i++ )
179 {
180 plane_t *p = &p_picture->p[i];
181
182 p->i_lines = (i_height_aligned + i_height_extra ) * p_dsc->p[i].h.num / p_dsc->p[i].h.den;
183 p->i_visible_lines = fmt->i_visible_height * p_dsc->p[i].h.num / p_dsc->p[i].h.den;
184 p->i_pitch = i_width_aligned * p_dsc->p[i].w.num / p_dsc->p[i].w.den * p_dsc->pixel_size;
185 p->i_visible_pitch = fmt->i_visible_width * p_dsc->p[i].w.num / p_dsc->p[i].w.den * p_dsc->pixel_size;
186 p->i_pixel_pitch = p_dsc->pixel_size;
187
188 assert( (p->i_pitch % 16) == 0 );
189 }
190 p_picture->i_planes = p_dsc->plane_count;
191
192 return VLC_SUCCESS;
193}

198picture_t *picture_NewFromResource( const video_format_t *p_fmt, const picture_resource_t *p_resource )
199{
200 video_format_t fmt = *p_fmt;
201
202 /* It is needed to be sure all information are filled */
203 video_format_Setup( &fmt, p_fmt->i_chroma,
204 p_fmt->i_width, p_fmt->i_height,
205 p_fmt->i_visible_width, p_fmt->i_visible_height,
206 p_fmt->i_sar_num, p_fmt->i_sar_den );
207 if( p_fmt->i_x_offset < p_fmt->i_width &&
208 p_fmt->i_y_offset < p_fmt->i_height &&
209 p_fmt->i_visible_width > 0 && p_fmt->i_x_offset + p_fmt->i_visible_width <= p_fmt->i_width &&
210 p_fmt->i_visible_height > 0 && p_fmt->i_y_offset + p_fmt->i_visible_height <= p_fmt->i_height )
211 video_format_CopyCrop( &fmt, p_fmt );
212
213 /* */
214 picture_t *p_picture = calloc( 1, sizeof(*p_picture) );
215 if( !p_picture )
216 return NULL;
217
218 /* Make sure the real dimensions are a multiple of 16 */
219 if( picture_Setup( p_picture, &fmt ) )
220 {
221 free( p_picture );
222 return NULL;
223 }
224
225 if( p_resource )
226 {
227 p_picture->p_sys = p_resource->p_sys;
228 p_picture->gc.pf_destroy = p_resource->pf_destroy;
229 assert( p_picture->gc.p_sys == NULL );
230
231 for( int i = 0; i < p_picture->i_planes; i++ )
232 {
233 p_picture->p[i].p_pixels = p_resource->p[i].p_pixels;
234 p_picture->p[i].i_lines = p_resource->p[i].i_lines;
235 p_picture->p[i].i_pitch = p_resource->p[i].i_pitch;
236 }
237 }
238 else
239 {
240 if( AllocatePicture( p_picture ) )
241 {
242 free( p_picture );
243 return NULL;
244 }
245 }
246
247 /* */
248 p_picture->format = fmt;
249
250 atomic_init( &p_picture->gc.refcount, 1 );
251 if( p_picture->gc.pf_destroy == NULL )
252 p_picture->gc.pf_destroy = PictureDestroy;
253
254 return p_picture;
255}

257picture_t *picture_NewFromFormat( const video_format_t *p_fmt )
258{
259 return picture_NewFromResource( p_fmt, NULL );

262picture_t *picture_New( vlc_fourcc_t i_chroma, int i_width, int i_height, int i_sar_num, int i_sar_den )
263{
264 video_format_t fmt;
265
266 memset( &fmt, 0, sizeof(fmt) );
267 video_format_Setup( &fmt, i_chroma, i_width, i_height,
268 i_width, i_height, i_sar_num, i_sar_den );
269
270 return picture_NewFromFormat( &fmt );
271}

277picture_t *picture_Hold( picture_t *p_picture )
278{
279 atomic_fetch_add( &p_picture->gc.refcount, 1 );
280 return p_picture;
281}
282
283void picture_Release( picture_t *p_picture )
284{
285 uintptr_t refs = atomic_fetch_sub( &p_picture->gc.refcount, 1 );
286 assert( refs != 0 );
287 if( refs > 1 )
288 return;
289
290 PictureDestroyContext( p_picture );
291 assert( p_picture->gc.pf_destroy != NULL );
292 p_picture->gc.pf_destroy( p_picture );
293}

296{
297 return atomic_load( &p_picture->gc.refcount ) > 1;
298}

303void plane_CopyPixels( plane_t *p_dst, const plane_t *p_src )
304{
305 const unsigned i_width = __MIN( p_dst->i_visible_pitch,
306 p_src->i_visible_pitch );
307 const unsigned i_height = __MIN( p_dst->i_visible_lines,
308 p_src->i_visible_lines );
309
310 /* The 2x visible pitch check does two things:
311 1) Makes field plane_t's work correctly (see the deinterlacer module)
312 2) Moves less data if the pitch and visible pitch differ much.
313 */
314 if( p_src->i_pitch == p_dst->i_pitch &&
315 p_src->i_pitch < 2*p_src->i_visible_pitch )
316 {
317 /* There are margins, but with the same width : perfect ! */
318 memcpy( p_dst->p_pixels, p_src->p_pixels,
319 p_src->i_pitch * i_height );
320 }
321 else
322 {
323 /* We need to proceed line by line */
324 uint8_t *p_in = p_src->p_pixels;
325 uint8_t *p_out = p_dst->p_pixels;
326 int i_line;
327
328 assert( p_in );
329 assert( p_out );
330
331 for( i_line = i_height; i_line--; )
332 {
333 memcpy( p_out, p_in, i_width );
334 p_in += p_src->i_pitch;
335 p_out += p_dst->i_pitch;
336 }
337 }
338}

340void picture_CopyProperties( picture_t *p_dst, const picture_t *p_src )
341{
342 p_dst->date = p_src->date;
343 p_dst->b_force = p_src->b_force;
344
345 p_dst->b_progressive = p_src->b_progressive;
346 p_dst->i_nb_fields = p_src->i_nb_fields;
347 p_dst->b_top_field_first = p_src->b_top_field_first;
348}
349
350void picture_CopyPixels( picture_t *p_dst, const picture_t *p_src )
351{
352 int i;
353
354 for( i = 0; i < p_src->i_planes ; i++ )
355 plane_CopyPixels( p_dst->p+i, p_src->p+i );
356}
357
358void picture_Copy( picture_t *p_dst, const picture_t *p_src )
359{
360 picture_CopyPixels( p_dst, p_src );
361 picture_CopyProperties( p_dst, p_src );
362}

368int picture_Export( vlc_object_t *p_obj,
369 block_t **pp_image,
370 video_format_t *p_fmt,
371 picture_t *p_picture,
372 vlc_fourcc_t i_format,
373 int i_override_width, int i_override_height )
374{
375 /* */
376 video_format_t fmt_in = p_picture->format;
377 if( fmt_in.i_sar_num <= 0 || fmt_in.i_sar_den <= 0 )
378 {
379 fmt_in.i_sar_num =
380 fmt_in.i_sar_den = 1;
381 }
382
383 /* */
384 video_format_t fmt_out;
385 memset( &fmt_out, 0, sizeof(fmt_out) );
386 fmt_out.i_sar_num =
387 fmt_out.i_sar_den = 1;
388 fmt_out.i_chroma = i_format;
389
390 /* compute original width/height */
391 unsigned int i_original_width;
392 unsigned int i_original_height;
393 if( fmt_in.i_sar_num >= fmt_in.i_sar_den )
394 {
395 i_original_width = (int64_t)fmt_in.i_width * fmt_in.i_sar_num / fmt_in.i_sar_den;
396 i_original_height = fmt_in.i_height;
397 }
398 else
399 {
400 i_original_width = fmt_in.i_width;
401 i_original_height = (int64_t)fmt_in.i_height * fmt_in.i_sar_den / fmt_in.i_sar_num;
402 }
403
404 /* */
405 fmt_out.i_width = ( i_override_width < 0 ) ?
406 i_original_width : (unsigned)i_override_width;
407 fmt_out.i_height = ( i_override_height < 0 ) ?
408 i_original_height : (unsigned)i_override_height;
409
410 /* scale if only one direction is provided */
411 if( fmt_out.i_height == 0 && fmt_out.i_width > 0 )
412 {
413 fmt_out.i_height = fmt_in.i_height * fmt_out.i_width
414 * fmt_in.i_sar_den / fmt_in.i_width / fmt_in.i_sar_num;
415 }
416 else if( fmt_out.i_width == 0 && fmt_out.i_height > 0 )
417 {
418 fmt_out.i_width = fmt_in.i_width * fmt_out.i_height
419 * fmt_in.i_sar_num / fmt_in.i_height / fmt_in.i_sar_den;
420 }
421
422 image_handler_t *p_image = image_HandlerCreate( p_obj );
423
424 block_t *p_block = image_Write( p_image, p_picture, &fmt_in, &fmt_out );
425
426 image_HandlerDelete( p_image );
427
428 if( !p_block )
429 return VLC_EGENERIC;
430
431 p_block->i_pts =
432 p_block->i_dts = p_picture->date;
433
434 if( p_fmt )
435 *p_fmt = fmt_out;
436 *pp_image = p_block;
437
438 return VLC_SUCCESS;
439}

441unsigned picture_BlendSubpicture(picture_t *dst,
442 filter_t *blend, subpicture_t *src)
443{
444 unsigned done = 0;
445
446 assert(src && !src->b_fade && src->b_absolute);
447
448 for (subpicture_region_t *r = src->p_region; r != NULL; r = r->p_next) {
449 assert(r->p_picture && r->i_align == 0);
450 if (filter_ConfigureBlend(blend, dst->format.i_width,
451 dst->format.i_height, &r->fmt)
452 || filter_Blend(blend, dst, r->i_x, r->i_y, r->p_picture,
453 src->i_alpha * r->i_alpha / 255))
454 msg_Err(blend, "blending %4.4s to %4.4s failed",
455 (char *)&blend->fmt_in.video.i_chroma,
456 (char *)&blend->fmt_out.video.i_chroma );
457 else
458 done++;
459 }
460 return done;
461}
