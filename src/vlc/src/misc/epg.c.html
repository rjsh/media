35void vlc_epg_Init( vlc_epg_t *p_epg, const char *psz_name )
36{
37 p_epg->psz_name = psz_name ? strdup( psz_name ) : NULL;
38 p_epg->p_current = NULL;
39 TAB_INIT( p_epg->i_event, p_epg->pp_event );
40}
41
42void vlc_epg_Clean( vlc_epg_t *p_epg )
43{
44 int i;
45 for( i = 0; i < p_epg->i_event; i++ )
46 {
47 vlc_epg_event_t *p_evt = p_epg->pp_event[i];
48 free( p_evt->psz_name );
49 free( p_evt->psz_short_description );
50 free( p_evt->psz_description );
51 free( p_evt );
52 }
53 TAB_CLEAN( p_epg->i_event, p_epg->pp_event );
54 free( p_epg->psz_name );
55}
56
57void vlc_epg_AddEvent( vlc_epg_t *p_epg, int64_t i_start, int i_duration,
58 const char *psz_name, const char *psz_short_description,
59 const char *psz_description, uint8_t i_rating )
60{
61 vlc_epg_event_t *p_evt = malloc( sizeof(*p_evt) );
62 if( !p_evt )
63 return;
64 p_evt->i_start = i_start;
65 p_evt->i_duration = i_duration;
66 p_evt->psz_name = psz_name ? strdup( psz_name ) : NULL;
67 p_evt->psz_short_description = psz_short_description ? strdup( psz_short_description ) : NULL;
68 p_evt->psz_description = psz_description ? strdup( psz_description ) : NULL;
69 p_evt->i_rating = i_rating;
70 TAB_APPEND( p_epg->i_event, p_epg->pp_event, p_evt );
71}
72
73vlc_epg_t *vlc_epg_New( const char *psz_name )
74{
75 vlc_epg_t *p_epg = malloc( sizeof(*p_epg) );
76 if( p_epg )
77 vlc_epg_Init( p_epg, psz_name );
78 return p_epg;
79}
80
81void vlc_epg_Delete( vlc_epg_t *p_epg )
82{
83 vlc_epg_Clean( p_epg );
84 free( p_epg );
85}

87void vlc_epg_SetCurrent( vlc_epg_t *p_epg, int64_t i_start )
88{
89 int i;
90 p_epg->p_current = NULL;
91 if( i_start < 0 )
92 return;
93
94 for( i = 0; i < p_epg->i_event; i++ )
95 {
96 if( p_epg->pp_event[i]->i_start == i_start )
97 {
98 p_epg->p_current = p_epg->pp_event[i];
99 break;
100 }
101 }
102}

104void vlc_epg_Merge( vlc_epg_t *p_dst, const vlc_epg_t *p_src )
105{
106 int i;
107
108 /* Add new event */
109 for( i = 0; i < p_src->i_event; i++ )
110 {
111 vlc_epg_event_t *p_evt = p_src->pp_event[i];
112 bool b_add = true;
113 int j;
114
115 for( j = 0; j < p_dst->i_event; j++ )
116 {
117 if( p_dst->pp_event[j]->i_start == p_evt->i_start && p_dst->pp_event[j]->i_duration == p_evt->i_duration )
118 {
119 b_add = false;
120 break;
121 }
122 if( p_dst->pp_event[j]->i_start > p_evt->i_start )
123 break;
124 }
125 if( b_add )
126 {
127 vlc_epg_event_t *p_copy = calloc( 1, sizeof(vlc_epg_event_t) );
128 if( !p_copy )
129 break;
130 p_copy->i_start = p_evt->i_start;
131 p_copy->i_duration = p_evt->i_duration;
132 p_copy->psz_name = p_evt->psz_name ? strdup( p_evt->psz_name ) : NULL;
133 p_copy->psz_short_description = p_evt->psz_short_description ? strdup( p_evt->psz_short_description ) : NULL;
134 p_copy->psz_description = p_evt->psz_description ? strdup( p_evt->psz_description ) : NULL;
135 p_copy->i_rating = p_evt->i_rating;
136 TAB_INSERT( p_dst->i_event, p_dst->pp_event, p_copy, j );
137 }
138 }
139 /* Update current */
140 if( p_src->p_current )
141 vlc_epg_SetCurrent( p_dst, p_src->p_current->i_start );
142
143 /* Keep only 1 old event */
144 if( p_dst->p_current )
145 {
146 while( p_dst->i_event > 1 && p_dst->pp_event[0] != p_dst->p_current && p_dst->pp_event[1] != p_dst->p_current )
147 TAB_REMOVE( p_dst->i_event, p_dst->pp_event, p_dst->pp_event[0] );
148 }
149}
