85/**
86 * vlc_meta contructor.
87 * vlc_meta_Delete() will free the returned pointer.
88 */
89vlc_meta_t *vlc_meta_New( void )
90{
91 vlc_meta_t *m = (vlc_meta_t*)malloc( sizeof(*m) );
92 if( !m )
93 return NULL;
94 memset( m->ppsz_meta, 0, sizeof(m->ppsz_meta) );
95 m->i_status = 0;
96 vlc_dictionary_init( &m->extra_tags, 0 );
97 return m;
98}
99
100/* Free a dictonary key allocated by strdup() in vlc_meta_AddExtra() */
101static void vlc_meta_FreeExtraKey( void *p_data, void *p_obj )
102{
103 VLC_UNUSED( p_obj );
104 free( p_data );
105}
106
107void vlc_meta_Delete( vlc_meta_t *m )
108{
109 int i;
110 for( i = 0; i < VLC_META_TYPE_COUNT ; i++ )
111 free( m->ppsz_meta[i] );
112 vlc_dictionary_clear( &m->extra_tags, vlc_meta_FreeExtraKey, NULL );
113 free( m );
114}

122void vlc_meta_Set( vlc_meta_t *p_meta, vlc_meta_type_t meta_type, const char *psz_val )
123{
124 free( p_meta->ppsz_meta[meta_type] );
125 assert( psz_val == NULL || IsUTF8( psz_val ) );
126 p_meta->ppsz_meta[meta_type] = psz_val ? strdup( psz_val ) : NULL;
127}

129const char *vlc_meta_Get( const vlc_meta_t *p_meta, vlc_meta_type_t meta_type )
130{
131 return p_meta->ppsz_meta[meta_type];
132}

134void vlc_meta_AddExtra( vlc_meta_t *m, const char *psz_name, const char *psz_value )
135{
136 char *psz_oldvalue = (char *)vlc_dictionary_value_for_key( &m->extra_tags, psz_name );
137 if( psz_oldvalue != kVLCDictionaryNotFound )
138 vlc_dictionary_remove_value_for_key( &m->extra_tags, psz_name,
139 vlc_meta_FreeExtraKey, NULL );
140 vlc_dictionary_insert( &m->extra_tags, psz_name, strdup(psz_value) );
141}
142
143const char * vlc_meta_GetExtra( const vlc_meta_t *m, const char *psz_name )
144{
145 return (char *)vlc_dictionary_value_for_key(&m->extra_tags, psz_name);
146}

172/**
173 * Merging meta
174 */
175void vlc_meta_Merge( vlc_meta_t *dst, const vlc_meta_t *src )
176{
177 char **ppsz_all_keys;
178 int i;
179
180 if( !dst || !src )
181 return;
182
183 for( i = 0; i < VLC_META_TYPE_COUNT; i++ )
184 {
185 if( src->ppsz_meta[i] )
186 {
187 free( dst->ppsz_meta[i] );
188 dst->ppsz_meta[i] = strdup( src->ppsz_meta[i] );
189 }
190 }
191
192 /* XXX: If speed up are needed, it is possible */
193 ppsz_all_keys = vlc_dictionary_all_keys( &src->extra_tags );
194 for( i = 0; ppsz_all_keys && ppsz_all_keys[i]; i++ )
195 {
196 /* Always try to remove the previous value */
197 vlc_dictionary_remove_value_for_key( &dst->extra_tags, ppsz_all_keys[i], vlc_meta_FreeExtraKey, NULL );
198
199 void *p_value = vlc_dictionary_value_for_key( &src->extra_tags, ppsz_all_keys[i] );
200 vlc_dictionary_insert( &dst->extra_tags, ppsz_all_keys[i], strdup( (const char*)p_value ) );
201 free( ppsz_all_keys[i] );
202 }
203 free( ppsz_all_keys );
204}

207void input_ExtractAttachmentAndCacheArt( input_thread_t *p_input )
208{
209 input_item_t *p_item = p_input->p->p_item;
210
211 /* */
212 char *psz_arturl = input_item_GetArtURL( p_item );
213 if( !psz_arturl || strncmp( psz_arturl, "attachment://", strlen("attachment://") ) )
214 {
215 msg_Err( p_input, "internal input error with input_ExtractAttachmentAndCacheArt" );
216 free( psz_arturl );
217 return;
218 }
219
220 if( input_item_IsArtFetched( p_item ) )
221 {
222 /* XXX Weird, we should not have end up with attachment:// art url unless there is a race
223 * condition */
224 msg_Warn( p_input, "internal input error with input_ExtractAttachmentAndCacheArt" );
225 playlist_FindArtInCache( p_item );
226 goto exit;
227 }
228
229 /* */
230 input_attachment_t *p_attachment = NULL;
231
232 vlc_mutex_lock( &p_item->lock );
233 for( int i_idx = 0; i_idx < p_input->p->i_attachment; i_idx++ )
234 {
235 if( !strcmp( p_input->p->attachment[i_idx]->psz_name,
236 &psz_arturl[strlen("attachment://")] ) )
237 {
238 p_attachment = vlc_input_attachment_Duplicate( p_input->p->attachment[i_idx] );
239 break;
240 }
241 }
242 vlc_mutex_unlock( &p_item->lock );
243
244 if( !p_attachment || p_attachment->i_data <= 0 )
245 {
246 if( p_attachment )
247 vlc_input_attachment_Delete( p_attachment );
248 msg_Warn( p_input, "internal input error with input_ExtractAttachmentAndCacheArt" );
249 goto exit;
250 }
251
252 /* */
253 const char *psz_type = NULL;
254 if( !strcmp( p_attachment->psz_mime, "image/jpeg" ) )
255 psz_type = ".jpg";
256 else if( !strcmp( p_attachment->psz_mime, "image/png" ) )
257 psz_type = ".png";
258
259 /* */
260 playlist_SaveArt( VLC_OBJECT(p_input), p_item,
261 p_attachment->p_data, p_attachment->i_data, psz_type );
262
263 vlc_input_attachment_Delete( p_attachment );
264
265exit:
266 free( psz_arturl );
267}

269int input_item_WriteMeta( vlc_object_t *obj, input_item_t *p_item )
270{
271 meta_export_t *p_export =
272 vlc_custom_create( obj, sizeof( *p_export ), "meta writer" );
273 if( p_export == NULL )
274 return VLC_ENOMEM;
275 p_export->p_item = p_item;
276
277 int type;
278 vlc_mutex_lock( &p_item->lock );
279 type = p_item->i_type;
280 vlc_mutex_unlock( &p_item->lock );
281 if( type != ITEM_TYPE_FILE )
282 goto error;
283
284 char *psz_uri = input_item_GetURI( p_item );
285 p_export->psz_file = make_path( psz_uri );
286 if( p_export->psz_file == NULL )
287 msg_Err( p_export, "cannot write meta to remote media %s", psz_uri );
288 free( psz_uri );
289 if( p_export->psz_file == NULL )
290 goto error;
291
292 module_t *p_mod = module_need( p_export, "meta writer", NULL, false );
293 if( p_mod )
294 module_unneed( p_export, p_mod );
295 vlc_object_release( p_export );
296 return VLC_SUCCESS;
297
298error:
299 vlc_object_release( p_export );
300 return VLC_EGENERIC;
301}

303void vlc_audio_replay_gain_MergeFromMeta( audio_replay_gain_t *p_dst,
304 const vlc_meta_t *p_meta )
305{
306 const char * psz_value;
307
308 if( !p_meta )
309 return;
310
311 if( (psz_value = vlc_meta_GetExtra(p_meta, "REPLAYGAIN_TRACK_GAIN")) ||
312 (psz_value = vlc_meta_GetExtra(p_meta, "RG_RADIO")) )
313 {
314 p_dst->pb_gain[AUDIO_REPLAY_GAIN_TRACK] = true;
315 p_dst->pf_gain[AUDIO_REPLAY_GAIN_TRACK] = us_atof( psz_value );
316 }
317
318 if( (psz_value = vlc_meta_GetExtra(p_meta, "REPLAYGAIN_TRACK_PEAK" )) ||
319 (psz_value = vlc_meta_GetExtra(p_meta, "RG_PEAK" )) )
320 {
321 p_dst->pb_peak[AUDIO_REPLAY_GAIN_TRACK] = true;
322 p_dst->pf_peak[AUDIO_REPLAY_GAIN_TRACK] = us_atof( psz_value );
323 }
324
325 if( (psz_value = vlc_meta_GetExtra(p_meta, "REPLAYGAIN_ALBUM_GAIN" )) ||
326 (psz_value = vlc_meta_GetExtra(p_meta, "RG_AUDIOPHILE" )) )
327 {
328 p_dst->pb_gain[AUDIO_REPLAY_GAIN_ALBUM] = true;
329 p_dst->pf_gain[AUDIO_REPLAY_GAIN_ALBUM] = us_atof( psz_value );
330 }
331
332 if( (psz_value = vlc_meta_GetExtra(p_meta, "REPLAYGAIN_ALBUM_PEAK" )) )
333 {
334 p_dst->pb_peak[AUDIO_REPLAY_GAIN_ALBUM] = true;
335 p_dst->pf_peak[AUDIO_REPLAY_GAIN_ALBUM] = us_atof( psz_value );
336 }
