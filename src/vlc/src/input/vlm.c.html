68static int InputEventPreparse( vlc_object_t *p_this, char const *psz_cmd,
69 vlc_value_t oldval, vlc_value_t newval, void *p_data )
70{
71 VLC_UNUSED(p_this); VLC_UNUSED(psz_cmd); VLC_UNUSED(oldval);
72 preparse_data_t *p_pre = p_data;
73
74 if( newval.i_int == INPUT_EVENT_DEAD ||
75 ( p_pre->b_mux && newval.i_int == INPUT_EVENT_ITEM_META ) )
76 vlc_sem_post( p_pre->p_sem );
77
78 return VLC_SUCCESS;
79}

81static int InputEvent( vlc_object_t *p_this, char const *psz_cmd,
82 vlc_value_t oldval, vlc_value_t newval,
83 void *p_data )
84{
85 VLC_UNUSED(psz_cmd);
86 VLC_UNUSED(oldval);
87 input_thread_t *p_input = (input_thread_t *)p_this;
88 vlm_t *p_vlm = libvlc_priv( p_input->p_libvlc )->p_vlm;
89 assert( p_vlm );
90 vlm_media_sys_t *p_media = p_data;
91 const char *psz_instance_name = NULL;
92
93 if( newval.i_int == INPUT_EVENT_STATE )
94 {
95 for( int i = 0; i < p_media->i_instance; i++ )
96 {
97 if( p_media->instance[i]->p_input == p_input )
98 {
99 psz_instance_name = p_media->instance[i]->psz_name;
100 break;
101 }
102 }
103 vlm_SendEventMediaInstanceState( p_vlm, p_media->cfg.id, p_media->cfg.psz_name, psz_instance_name, var_GetInteger( p_input, "state" ) );
104
105 vlc_mutex_lock( &p_vlm->lock_manage );
106 p_vlm->input_state_changed = true;
107 vlc_cond_signal( &p_vlm->wait_manage );
108 vlc_mutex_unlock( &p_vlm->lock_manage );
109 }
110 return VLC_SUCCESS;
111}

116/*****************************************************************************
117 * vlm_New:
118 *****************************************************************************/
119vlm_t *vlm_New ( vlc_object_t *p_this )
120{
121 vlm_t *p_vlm = NULL, **pp_vlm = &(libvlc_priv (p_this->p_libvlc)->p_vlm);
122 char *psz_vlmconf;
123
124 /* Avoid multiple creation */
125 vlc_mutex_lock( &vlm_mutex );
126
127 p_vlm = *pp_vlm;
128 if( p_vlm )
129 { /* VLM already exists */
130 if( likely( p_vlm->users < UINT_MAX ) )
131 p_vlm->users++;
132 else
133 p_vlm = NULL;
134 vlc_mutex_unlock( &vlm_mutex );
135 return p_vlm;
136 }
137
138 msg_Dbg( p_this, "creating VLM" );
139
140 p_vlm = vlc_custom_create( p_this->p_libvlc, sizeof( *p_vlm ),
141 "vlm daemon" );
142 if( !p_vlm )
143 {
144 vlc_mutex_unlock( &vlm_mutex );
145 return NULL;
146 }
147
148 vlc_mutex_init( &p_vlm->lock );
149 vlc_mutex_init( &p_vlm->lock_manage );
150 vlc_cond_init_daytime( &p_vlm->wait_manage );
151 p_vlm->users = 1;
152 p_vlm->input_state_changed = false;
153 p_vlm->i_id = 1;
154 TAB_INIT( p_vlm->i_media, p_vlm->media );
155 TAB_INIT( p_vlm->i_schedule, p_vlm->schedule );
156 p_vlm->p_vod = NULL;
157 var_Create( p_vlm, "intf-event", VLC_VAR_ADDRESS );
158
159 if( vlc_clone( &p_vlm->thread, Manage, p_vlm, VLC_THREAD_PRIORITY_LOW ) )
160 {
161 vlc_cond_destroy( &p_vlm->wait_manage );
162 vlc_mutex_destroy( &p_vlm->lock );
163 vlc_mutex_destroy( &p_vlm->lock_manage );
164 vlc_object_release( p_vlm );
165 vlc_mutex_unlock( &vlm_mutex );
166 return NULL;
167 }
168
169 *pp_vlm = p_vlm; /* for future reference */
170
171 /* Load our configuration file */
172 psz_vlmconf = var_CreateGetString( p_vlm, "vlm-conf" );
173 if( psz_vlmconf && *psz_vlmconf )
174 {
175 vlm_message_t *p_message = NULL;
176 char *psz_buffer = NULL;
177
178 msg_Dbg( p_this, "loading VLM configuration" );
179 if( asprintf(&psz_buffer, "load %s", psz_vlmconf ) != -1 )
180 {
181 msg_Dbg( p_this, "%s", psz_buffer );
182 if( vlm_ExecuteCommand( p_vlm, psz_buffer, &p_message ) )
183 msg_Warn( p_this, "error while loading the configuration file" );
184
185 vlm_MessageDelete( p_message );
186 free( psz_buffer );
187 }
188 }
189 free( psz_vlmconf );
190
191 vlc_mutex_unlock( &vlm_mutex );
192
193 return p_vlm;
194}

196/*****************************************************************************
197 * vlm_Delete:
198 *****************************************************************************/
199void vlm_Delete( vlm_t *p_vlm )
200{
201 /* vlm_Delete() is serialized against itself, and against vlm_New().
202 * This mutex protects libvlc_priv->p_vlm and p_vlm->users. */
203 vlc_mutex_lock( &vlm_mutex );
204 assert( p_vlm->users > 0 );
205 if( --p_vlm->users == 0 )
206 assert( libvlc_priv(p_vlm->p_libvlc)->p_vlm == p_vlm );
207 else
208 p_vlm = NULL;
209
210 if( p_vlm == NULL )
211 {
212 vlc_mutex_unlock( &vlm_mutex );
213 return;
214 }
215
216 /* Destroy and release VLM */
217 vlc_mutex_lock( &p_vlm->lock );
218 vlm_ControlInternal( p_vlm, VLM_CLEAR_MEDIAS );
219 TAB_CLEAN( p_vlm->i_media, p_vlm->media );
220
221 vlm_ControlInternal( p_vlm, VLM_CLEAR_SCHEDULES );
222 TAB_CLEAN( p_vlm->i_schedule, p_vlm->schedule );
223 vlc_mutex_unlock( &p_vlm->lock );
224
225 vlc_cancel( p_vlm->thread );
226
227 if( p_vlm->p_vod )
228 {
229 module_unneed( p_vlm->p_vod, p_vlm->p_vod->p_module );
230 vlc_object_release( p_vlm->p_vod );
231 }
232
233 libvlc_priv(p_vlm->p_libvlc)->p_vlm = NULL;
234 vlc_mutex_unlock( &vlm_mutex );
235
236 vlc_join( p_vlm->thread, NULL );
237
238 vlc_cond_destroy( &p_vlm->wait_manage );
239 vlc_mutex_destroy( &p_vlm->lock );
240 vlc_mutex_destroy( &p_vlm->lock_manage );
241 vlc_object_release( p_vlm );
242}

246 *****************************************************************************/
247int vlm_ExecuteCommand( vlm_t *p_vlm, const char *psz_command,
248 vlm_message_t **pp_message)
249{
250 int i_result;
251
252 vlc_mutex_lock( &p_vlm->lock );
253 i_result = ExecuteCommand( p_vlm, psz_command, pp_message );
254 vlc_mutex_unlock( &p_vlm->lock );
255
256 return i_result;
257}

260int64_t vlm_Date(void)
261{
262 struct timeval tv;
263
264 (void)gettimeofday( &tv, NULL );
265 return tv.tv_sec * INT64_C(1000000) + tv.tv_usec;
266}

270 *
271 *****************************************************************************/
272static int vlm_MediaVodControl( void *p_private, vod_media_t *p_vod_media,
273 const char *psz_id, int i_query, va_list args )
274{
275 vlm_t *vlm = (vlm_t *)p_private;
276 int i, i_ret;
277 const char *psz;
278 int64_t id;
279
280 if( !p_private || !p_vod_media )
281 return VLC_EGENERIC;
282
283 vlc_mutex_lock( &vlm->lock );
284
285 /* Find media id */
286 for( i = 0, id = -1; i < vlm->i_media; i++ )
287 {
288 if( p_vod_media == vlm->media[i]->vod.p_media )
289 {
290 id = vlm->media[i]->cfg.id;
291 break;
292 }
293 }
294 if( id == -1 )
295 {
296 vlc_mutex_unlock( &vlm->lock );
297 return VLC_EGENERIC;
298 }
299
300 switch( i_query )
301 {
302 case VOD_MEDIA_PLAY:
303 {
304 psz = (const char *)va_arg( args, const char * );
305 int64_t *i_time = (int64_t *)va_arg( args, int64_t *);
306 bool b_retry = false;
307 if (*i_time < 0)
308 {
309 /* No start time requested: return the current NPT */
310 i_ret = vlm_ControlInternal( vlm, VLM_GET_MEDIA_INSTANCE_TIME, id, psz_id, i_time );
311 /* The instance is not running yet, it will start at 0 */
312 if (i_ret)
313 *i_time = 0;
314 }
315 else
316 {
317 /* We want to seek before unpausing, but it won't
318 * work if the instance is not running yet. */
319 b_retry = vlm_ControlInternal( vlm, VLM_SET_MEDIA_INSTANCE_TIME, id, psz_id, *i_time );
320 }
321
322 i_ret = vlm_ControlInternal( vlm, VLM_START_MEDIA_VOD_INSTANCE, id, psz_id, 0, psz );
323
324 if (!i_ret && b_retry)
325 i_ret = vlm_ControlInternal( vlm, VLM_SET_MEDIA_INSTANCE_TIME, id, psz_id, *i_time );
326 break;
327 }
328
329 case VOD_MEDIA_PAUSE:
330 {
331 int64_t *i_time = (int64_t *)va_arg( args, int64_t *);
332 i_ret = vlm_ControlInternal( vlm, VLM_PAUSE_MEDIA_INSTANCE, id, psz_id );
333 if (!i_ret)
334 i_ret = vlm_ControlInternal( vlm, VLM_GET_MEDIA_INSTANCE_TIME, id, psz_id, i_time );
335 break;
336 }
337
338 case VOD_MEDIA_STOP:
339 i_ret = vlm_ControlInternal( vlm, VLM_STOP_MEDIA_INSTANCE, id, psz_id );
340 break;
341
342 case VOD_MEDIA_SEEK:
343 {
344 int64_t i_time = (int64_t)va_arg( args, int64_t );
345 i_ret = vlm_ControlInternal( vlm, VLM_SET_MEDIA_INSTANCE_TIME, id, psz_id, i_time );
346 break;
347 }
348
349 case VOD_MEDIA_REWIND:
350 {
351 double d_scale = (double)va_arg( args, double );
352 double d_position;
353
354 vlm_ControlInternal( vlm, VLM_GET_MEDIA_INSTANCE_POSITION, id, psz_id, &d_position );
355 d_position -= (d_scale / 1000.0);
356 if( d_position < 0.0 )
357 d_position = 0.0;
358 i_ret = vlm_ControlInternal( vlm, VLM_SET_MEDIA_INSTANCE_POSITION, id, psz_id, d_position );
359 break;
360 }
361
362 case VOD_MEDIA_FORWARD:
363 {
364 double d_scale = (double)va_arg( args, double );
365 double d_position;
366
367 vlm_ControlInternal( vlm, VLM_GET_MEDIA_INSTANCE_POSITION, id, psz_id, &d_position );
368 d_position += (d_scale / 1000.0);
369 if( d_position > 1.0 )
370 d_position = 1.0;
371 i_ret = vlm_ControlInternal( vlm, VLM_SET_MEDIA_INSTANCE_POSITION, id, psz_id, d_position );
372 break;
373 }
374
375 default:
376 i_ret = VLC_EGENERIC;
377 break;
378 }
379
380 vlc_mutex_unlock( &vlm->lock );
381
382 return i_ret;
383}

386/*****************************************************************************
387 * Manage:
388 *****************************************************************************/
389static void* Manage( void* p_object )
390{
391 vlm_t *vlm = (vlm_t*)p_object;
392 int i, j;
393 mtime_t i_lastcheck;
394 mtime_t i_time;
395 mtime_t i_nextschedule = 0;
396
397 i_lastcheck = vlm_Date();
398
399 for( ;; )
400 {
401 char **ppsz_scheduled_commands = NULL;
402 int i_scheduled_commands = 0;
403 bool scheduled_command = false;
404
405 vlc_mutex_lock( &vlm->lock_manage );
406 mutex_cleanup_push( &vlm->lock_manage );
407 while( !vlm->input_state_changed && !scheduled_command )
408 {
409 if( i_nextschedule )
410 scheduled_command = vlc_cond_timedwait( &vlm->wait_manage, &vlm->lock_manage, i_nextschedule ) != 0;
411 else
412 vlc_cond_wait( &vlm->wait_manage, &vlm->lock_manage );
413 }
414 vlm->input_state_changed = false;
415 vlc_cleanup_run( );
416
417 int canc = vlc_savecancel ();
418 /* destroy the inputs that wants to die, and launch the next input */
419 vlc_mutex_lock( &vlm->lock );
420 for( i = 0; i < vlm->i_media; i++ )
421 {
422 vlm_media_sys_t *p_media = vlm->media[i];
423
424 for( j = 0; j < p_media->i_instance; )
425 {
426 vlm_media_instance_sys_t *p_instance = p_media->instance[j];
427
428 if( p_instance->p_input && ( p_instance->p_input->b_eof || p_instance->p_input->b_error ) )
429 {
430 int i_new_input_index;
431
432 /* */
433 i_new_input_index = p_instance->i_index + 1;
434 if( !p_media->cfg.b_vod && p_media->cfg.broadcast.b_loop && i_new_input_index >= p_media->cfg.i_input )
435 i_new_input_index = 0;
436
437 /* FIXME implement multiple input with VOD */
438 if( p_media->cfg.b_vod || i_new_input_index >= p_media->cfg.i_input )
439 vlm_ControlInternal( vlm, VLM_STOP_MEDIA_INSTANCE, p_media->cfg.id, p_instance->psz_name );
440 else
441 vlm_ControlInternal( vlm, VLM_START_MEDIA_BROADCAST_INSTANCE, p_media->cfg.id, p_instance->psz_name, i_new_input_index );
442
443 j = 0;
444 }
445 else
446 {
447 j++;
448 }
449 }
450 }
451
452 /* scheduling */
453 i_time = vlm_Date();
454 i_nextschedule = 0;
455
456 for( i = 0; i < vlm->i_schedule; i++ )
457 {
458 mtime_t i_real_date = vlm->schedule[i]->i_date;
459
460 if( vlm->schedule[i]->b_enabled )
461 {
462 if( vlm->schedule[i]->i_date == 0 ) // now !
463 {
464 vlm->schedule[i]->i_date = (i_time / 1000000) * 1000000 ;
465 i_real_date = i_time;
466 }
467 else if( vlm->schedule[i]->i_period != 0 )
468 {
469 int j = 0;
470 while( vlm->schedule[i]->i_date + j *
471 vlm->schedule[i]->i_period <= i_lastcheck &&
472 ( vlm->schedule[i]->i_repeat > j ||
473 vlm->schedule[i]->i_repeat == -1 ) )
474 {
475 j++;
476 }
477
478 i_real_date = vlm->schedule[i]->i_date + j *
479 vlm->schedule[i]->i_period;
480 }
481
482 if( i_real_date <= i_time )
483 {
484 if( i_real_date > i_lastcheck )
485 {
486 for( j = 0; j < vlm->schedule[i]->i_command; j++ )
487 {
488 TAB_APPEND( i_scheduled_commands,
489 ppsz_scheduled_commands,
490 strdup(vlm->schedule[i]->command[j] ) );
491 }
492 }
493 }
494 else if( i_nextschedule == 0 || i_real_date < i_nextschedule )
495 {
496 i_nextschedule = i_real_date;
497 }
498 }
499 }
500
501 while( i_scheduled_commands )
502 {
503 vlm_message_t *message = NULL;
504 char *psz_command = ppsz_scheduled_commands[0];
505 ExecuteCommand( vlm, psz_command,&message );
506
507 /* for now, drop the message */
508 vlm_MessageDelete( message );
509 TAB_REMOVE( i_scheduled_commands,
510 ppsz_scheduled_commands,
511 psz_command );
512 free( psz_command );
513 }
514
515 i_lastcheck = i_time;
516 vlc_mutex_unlock( &vlm->lock );
517 vlc_restorecancel (canc);
518 }
519
520 return NULL;
521}

585/* Called after a media description is changed/added */
586static int vlm_OnMediaUpdate( vlm_t *p_vlm, vlm_media_sys_t *p_media )
587{
588 vlm_media_t *p_cfg = &p_media->cfg;
589 /* Check if we need to create/delete a vod media */
590 if( p_cfg->b_vod && p_vlm->p_vod )
591 {
592 if( !p_cfg->b_enabled && p_media->vod.p_media )
593 {
594 p_vlm->p_vod->pf_media_del( p_vlm->p_vod, p_media->vod.p_media );
595 p_media->vod.p_media = NULL;
596 }
597 else if( p_cfg->b_enabled && !p_media->vod.p_media && p_cfg->i_input )
598 {
599 /* Pre-parse the input */
600 input_thread_t *p_input;
601 char *psz_output;
602 char *psz_header;
603 char *psz_dup;
604 int i;
605
606 vlc_gc_decref( p_media->vod.p_item );
607
608 if( strstr( p_cfg->ppsz_input[0], "://" ) == NULL )
609 {
610 char *psz_uri = vlc_path2uri( p_cfg->ppsz_input[0], NULL );
611 p_media->vod.p_item = input_item_New( psz_uri,
612 p_cfg->psz_name );
613 free( psz_uri );
614 }
615 else
616 p_media->vod.p_item = input_item_New( p_cfg->ppsz_input[0],
617 p_cfg->psz_name );
618
619 if( p_cfg->psz_output )
620 {
621 if( asprintf( &psz_output, "%s:description", p_cfg->psz_output ) == -1 )
622 psz_output = NULL;
623 }
624 else
625 psz_output = strdup( "#description" );
626
627 if( psz_output && asprintf( &psz_dup, "sout=%s", psz_output ) != -1 )
628 {
629 input_item_AddOption( p_media->vod.p_item, psz_dup, VLC_INPUT_OPTION_TRUSTED );
630 free( psz_dup );
631 }
632 free( psz_output );
633
634 for( i = 0; i < p_cfg->i_option; i++ )
635 input_item_AddOption( p_media->vod.p_item,
636 p_cfg->ppsz_option[i], VLC_INPUT_OPTION_TRUSTED );
637
638 if( asprintf( &psz_header, _("Media: %s"), p_cfg->psz_name ) == -1 )
639 psz_header = NULL;
640
641 sout_description_data_t data;
642 TAB_INIT(data.i_es, data.es);
643
644 p_input = input_Create( p_vlm->p_vod, p_media->vod.p_item, psz_header, NULL );
645 if( p_input )
646 {
647 vlc_sem_t sem_preparse;
648 vlc_sem_init( &sem_preparse, 0 );
649
650 preparse_data_t preparse = { .p_sem = &sem_preparse,
651 .b_mux = (p_cfg->vod.psz_mux != NULL) };
652 var_AddCallback( p_input, "intf-event", InputEventPreparse,
653 &preparse );
654
655 data.sem = &sem_preparse;
656 var_Create( p_input, "sout-description-data", VLC_VAR_ADDRESS );
657 var_SetAddress( p_input, "sout-description-data", &data );
658
659 if( !input_Start( p_input ) )
660 vlc_sem_wait( &sem_preparse );
661
662 var_DelCallback( p_input, "intf-event", InputEventPreparse,
663 &preparse );
664
665 input_Stop( p_input, true );
666 input_Close( p_input );
667 vlc_sem_destroy( &sem_preparse );
668 }

925static int vlm_ControlMediaInstanceStart( vlm_t *p_vlm, int64_t id, const char *psz_id, int i_input_index, const char *psz_vod_output )
926{
927 vlm_media_sys_t *p_media = vlm_ControlMediaGetById( p_vlm, id );
928 vlm_media_instance_sys_t *p_instance;
929 char *psz_log;
930
931 if( !p_media || !p_media->cfg.b_enabled || p_media->cfg.i_input <= 0 )
932 return VLC_EGENERIC;
933
934 /* TODO support multiple input for VOD with sout-keep ? */
935
936 if( ( p_media->cfg.b_vod && !psz_vod_output ) || ( !p_media->cfg.b_vod && psz_vod_output ) )
937 return VLC_EGENERIC;
938
939 if( i_input_index < 0 || i_input_index >= p_media->cfg.i_input )
940 return VLC_EGENERIC;
941
942 p_instance = vlm_ControlMediaInstanceGetByName( p_media, psz_id );
943 if( !p_instance )
944 {
945 vlm_media_t *p_cfg = &p_media->cfg;
946 int i;
947
948 p_instance = vlm_MediaInstanceNew( p_vlm, psz_id );
949 if( !p_instance )
950 return VLC_ENOMEM;
951
952 if ( p_cfg->b_vod )
953 {
954 var_Create( p_instance->p_parent, "vod-media", VLC_VAR_ADDRESS );
955 var_SetAddress( p_instance->p_parent, "vod-media",
956 p_media->vod.p_media );
957 var_Create( p_instance->p_parent, "vod-session", VLC_VAR_STRING );
958 var_SetString( p_instance->p_parent, "vod-session", psz_id );
959 }
960
961 if( p_cfg->psz_output != NULL || psz_vod_output != NULL )
962 {
963 char *psz_buffer;
964 if( asprintf( &psz_buffer, "sout=%s%s%s",
965 p_cfg->psz_output ? p_cfg->psz_output : "",
966 (p_cfg->psz_output && psz_vod_output) ? ":" : psz_vod_output ? "#" : "",
967 psz_vod_output ? psz_vod_output : "" ) != -1 )
968 {
969 input_item_AddOption( p_instance->p_item, psz_buffer, VLC_INPUT_OPTION_TRUSTED );
970 free( psz_buffer );
971 }
972 }
973
974 for( i = 0; i < p_cfg->i_option; i++ )
975 {
976 if( !strcmp( p_cfg->ppsz_option[i], "sout-keep" ) )
977 p_instance->b_sout_keep = true;
978 else if( !strcmp( p_cfg->ppsz_option[i], "nosout-keep" ) || !strcmp( p_cfg->ppsz_option[i], "no-sout-keep" ) )
979 p_instance->b_sout_keep = false;
980 else
981 input_item_AddOption( p_instance->p_item, p_cfg->ppsz_option[i], VLC_INPUT_OPTION_TRUSTED );
982 }
983 TAB_APPEND( p_media->i_instance, p_media->instance, p_instance );
984 }
985
986 /* Stop old instance */
987 input_thread_t *p_input = p_instance->p_input;
988 if( p_input )
989 {
990 if( p_instance->i_index == i_input_index &&
991 !p_input->b_eof && !p_input->b_error )
992 {
993 if( var_GetInteger( p_input, "state" ) == PAUSE_S )
994 var_SetInteger( p_input, "state", PLAYING_S );
995 return VLC_SUCCESS;
996 }
997
998
999 input_Stop( p_input, true );
1000 input_Join( p_input );
1001 var_DelCallback( p_instance->p_input, "intf-event", InputEvent, p_media );
1002 input_Release( p_input );
1003
1004 if( !p_instance->b_sout_keep )
1005 input_resource_TerminateSout( p_instance->p_input_resource );
1006 input_resource_TerminateVout( p_instance->p_input_resource );
1007
1008 vlm_SendEventMediaInstanceStopped( p_vlm, id, p_media->cfg.psz_name );
1009 }
1010
1011 /* Start new one */
1012 p_instance->i_index = i_input_index;
1013 if( strstr( p_media->cfg.ppsz_input[p_instance->i_index], "://" ) == NULL )
1014 {
1015 char *psz_uri = vlc_path2uri(
1016 p_media->cfg.ppsz_input[p_instance->i_index], NULL );
1017 input_item_SetURI( p_instance->p_item, psz_uri ) ;
1018 free( psz_uri );
1019 }
1020 else
1021 input_item_SetURI( p_instance->p_item, p_media->cfg.ppsz_input[p_instance->i_index] ) ;
1022
1023 if( asprintf( &psz_log, _("Media: %s"), p_media->cfg.psz_name ) != -1 )
1024 {
1025 p_instance->p_input = input_Create( p_instance->p_parent,
1026 p_instance->p_item, psz_log,
1027 p_instance->p_input_resource );
1028 if( p_instance->p_input )
1029 {
1030 var_AddCallback( p_instance->p_input, "intf-event", InputEvent, p_media );
1031
1032 if( input_Start( p_instance->p_input ) != VLC_SUCCESS )
1033 {
1034 var_DelCallback( p_instance->p_input, "intf-event", InputEvent, p_media );
1035 vlc_object_release( p_instance->p_input );
1036 p_instance->p_input = NULL;
1037 }
1038 }
1039
1040 if( !p_instance->p_input )
1041 {
1042 vlm_MediaInstanceDelete( p_vlm, id, p_instance, p_media );
1043 }
1044 else
1045 {
1046 vlm_SendEventMediaInstanceStarted( p_vlm, id, p_media->cfg.psz_name );
1047 }
1048 free( psz_log );
1049 }
1050
1051 return VLC_SUCCESS;
1052}

1199 int64_t i_i64;
1200 double *pd_double;
1201 double d_double;
1202
1203 switch( i_query )
1204 {
1205 /* Media control */
1206 case VLM_GET_MEDIAS:
1207 ppp_dsc = (vlm_media_t ***)va_arg( args, vlm_media_t *** );
1208 pi_int = (int *)va_arg( args, int * );

1221 return vlm_ControlMediaAdd( p_vlm, p_dsc, p_id );
1222
1223 case VLM_DEL_MEDIA:
1224 id = (int64_t)va_arg( args, int64_t );
1225 return vlm_ControlMediaDel( p_vlm, id );
1226
1227 case VLM_GET_MEDIA:
1228 id = (int64_t)va_arg( args, int64_t );
1229 pp_dsc = (vlm_media_t **)va_arg( args, vlm_media_t ** );
