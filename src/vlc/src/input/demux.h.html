39static inline int demux_Demux( demux_t *p_demux )
40{
41 if( !p_demux->pf_demux )
42 return 1;
43
44 return p_demux->pf_demux( p_demux );
45}
46static inline int demux_vaControl( demux_t *p_demux, int i_query, va_list args )
47{
48 return p_demux->pf_control( p_demux, i_query, args );
49}
50static inline int demux_Control( demux_t *p_demux, int i_query, ... )
51{
52 va_list args;
53 int i_result;
54
55 va_start( args, i_query );
56 i_result = demux_vaControl( p_demux, i_query, args );
57 va_end( args );
58 return i_result;
59}
