46struct input_resource_t
47{
48 atomic_uint refs;
49
50 vlc_object_t *p_parent;
51
52 /* This lock is used to serialize request and protect
53 * our variables */
54 vlc_mutex_t lock;
55
56 /* */
57 input_thread_t *p_input;
58
59 sout_instance_t *p_sout;
60 vout_thread_t *p_vout_free;
61
62 /* This lock is used to protect vout resources access (for hold)
63 * It is a special case because of embed video (possible deadlock
64 * between vout window request and vout holds in some(qt4) interface)
65 */
66 vlc_mutex_t lock_hold;
67
68 /* You need lock+lock_hold to write to the following variables and
69 * only lock or lock_hold to read them */
70
71 vout_thread_t **pp_vout;
72 int i_vout;
73
74 bool b_aout_busy;
75 audio_output_t *p_aout;
76};

79static void DestroySout( input_resource_t *p_resource )
80{
81#ifdef ENABLE_SOUT
82 if( p_resource->p_sout )
83 sout_DeleteInstance( p_resource->p_sout );
84#endif
85 p_resource->p_sout = NULL;
86}

88static sout_instance_t *RequestSout( input_resource_t *p_resource,
89 sout_instance_t *p_sout, const char *psz_sout )
90{
91#ifdef ENABLE_SOUT
92 if( !p_sout && !psz_sout )
93 {
94 if( p_resource->p_sout )
95 msg_Dbg( p_resource->p_sout, "destroying useless sout" );
96 DestroySout( p_resource );
97 return NULL;
98 }
99
100 assert( !p_sout || ( !p_resource->p_sout && !psz_sout ) );
101
102 /* Check the validity of the sout */
103 if( p_resource->p_sout &&
104 strcmp( p_resource->p_sout->psz_sout, psz_sout ) )
105 {
106 msg_Dbg( p_resource->p_parent, "destroying unusable sout" );
107 DestroySout( p_resource );
108 }
109
110 if( psz_sout )
111 {
112 if( p_resource->p_sout )
113 {
114 /* Reuse it */
115 msg_Dbg( p_resource->p_parent, "reusing sout" );
116 msg_Dbg( p_resource->p_parent, "you probably want to use gather stream_out" );
117 }
118 else
119 {
120 /* Create a new one */
121 p_resource->p_sout = sout_NewInstance( p_resource->p_parent, psz_sout );
122 }
123
124 p_sout = p_resource->p_sout;
125 p_resource->p_sout = NULL;
126
127 return p_sout;
128 }
129 else
130 {
131 p_resource->p_sout = p_sout;
132 return NULL;
133 }
134#else
135 VLC_UNUSED (p_resource); VLC_UNUSED (p_sout); VLC_UNUSED (psz_sout);
136 return NULL;
137#endif
138}

141static void DestroyVout( input_resource_t *p_resource )
142{
143 assert( p_resource->i_vout == 0 );

151static void DisplayVoutTitle( input_resource_t *p_resource,
152 vout_thread_t *p_vout )
153{
154 if( p_resource->p_input == NULL )
155 return;
156
157 /* TODO display the title only one time for the same input ? */
158
159 input_item_t *p_item = input_GetItem( p_resource->p_input );
160
161 char *psz_nowplaying = input_item_GetNowPlaying( p_item );
162 if( psz_nowplaying && *psz_nowplaying )
163 {
164 vout_DisplayTitle( p_vout, psz_nowplaying );
165 }
166 else
167 {
168 char *psz_artist = input_item_GetArtist( p_item );
169 char *psz_name = input_item_GetTitle( p_item );
170
171 if( !psz_name || *psz_name == '\0' )
172 {
173 free( psz_name );
174 psz_name = input_item_GetName( p_item );
175 }
176 if( psz_artist && *psz_artist )
177 {
178 char *psz_string;
179 if( asprintf( &psz_string, "%s - %s", psz_name, psz_artist ) != -1 )
180 {
181 vout_DisplayTitle( p_vout, psz_string );
182 free( psz_string );
183 }
184 }
185 else if( psz_name )
186 {
187 vout_DisplayTitle( p_vout, psz_name );
188 }
189 free( psz_name );
190 free( psz_artist );
191 }
192 free( psz_nowplaying );
193}
194static vout_thread_t *RequestVout( input_resource_t *p_resource,
195 vout_thread_t *p_vout,
196 video_format_t *p_fmt, unsigned dpb_size,
197 bool b_recycle )
198{
199 vlc_assert_locked( &p_resource->lock );
200
201 if( !p_vout && !p_fmt )
202 {
203 if( p_resource->p_vout_free )
204 {
205 msg_Dbg( p_resource->p_vout_free, "destroying useless vout" );
206 vout_CloseAndRelease( p_resource->p_vout_free );
207 p_resource->p_vout_free = NULL;
208 }
209 return NULL;
210 }
211
212 if( p_fmt )
213 {
214 /* */
215 if( !p_vout && p_resource->p_vout_free )
216 {
217 msg_Dbg( p_resource->p_parent, "trying to reuse free vout" );
218 p_vout = p_resource->p_vout_free;
219
220 p_resource->p_vout_free = NULL;
221 }
222 else if( p_vout )
223 {
224 assert( p_vout != p_resource->p_vout_free );
225
226 vlc_mutex_lock( &p_resource->lock_hold );
227 TAB_REMOVE( p_resource->i_vout, p_resource->pp_vout, p_vout );
228 vlc_mutex_unlock( &p_resource->lock_hold );
229 }
230
231 /* */
232 vout_configuration_t cfg = {
233 .vout = p_vout,
234 .input = VLC_OBJECT(p_resource->p_input),
235 .change_fmt = true,
236 .fmt = p_fmt,
237 .dpb_size = dpb_size,
238 };
239 p_vout = vout_Request( p_resource->p_parent, &cfg );
240 if( !p_vout )
241 return NULL;
242
243 DisplayVoutTitle( p_resource, p_vout );
244
245 vlc_mutex_lock( &p_resource->lock_hold );
246 TAB_APPEND( p_resource->i_vout, p_resource->pp_vout, p_vout );
247 vlc_mutex_unlock( &p_resource->lock_hold );
248
249 return p_vout;
250 }
251 else
252 {
253 assert( p_vout );
254
255 vlc_mutex_lock( &p_resource->lock_hold );
256 TAB_REMOVE( p_resource->i_vout, p_resource->pp_vout, p_vout );
257 const int i_vout_active = p_resource->i_vout;
258 vlc_mutex_unlock( &p_resource->lock_hold );
259
260 if( p_resource->p_vout_free || i_vout_active > 0 || !b_recycle )
261 {
262 if( b_recycle )
263 msg_Dbg( p_resource->p_parent, "destroying vout (already one saved or active)" );
264 vout_CloseAndRelease( p_vout );
265 }
266 else
267 {
268 msg_Dbg( p_resource->p_parent, "saving a free vout" );
269 vout_Flush( p_vout, 1 );
270 vout_FlushSubpictureChannel( p_vout, -1 );
271
272 vout_configuration_t cfg = {
273 .vout = p_vout,
274 .input = NULL,
275 .change_fmt = false,
276 .fmt = NULL,
277 .dpb_size = 0,
278 };
279 p_resource->p_vout_free = vout_Request( p_resource->p_parent, &cfg );
280 }
281 return NULL;
282 }

284static vout_thread_t *HoldVout( input_resource_t *p_resource )
285{
286 /* TODO FIXME: p_resource->pp_vout order is NOT stable */
287 vlc_mutex_lock( &p_resource->lock_hold );
288
289 vout_thread_t *p_vout = p_resource->i_vout > 0 ? p_resource->pp_vout[0] : NULL;
290 if( p_vout )
291 vlc_object_hold( p_vout );
292
293 vlc_mutex_unlock( &p_resource->lock_hold );
294
295 return p_vout;
296}

298static void HoldVouts( input_resource_t *p_resource, vout_thread_t ***ppp_vout,
299 size_t *pi_vout )
300{
301 vout_thread_t **pp_vout;
302
303 *pi_vout = 0;
304 *ppp_vout = NULL;
305
306 vlc_mutex_lock( &p_resource->lock_hold );
307
308 if( p_resource->i_vout <= 0 )
309 goto exit;
310
311 pp_vout = malloc( p_resource->i_vout * sizeof(*pp_vout) );
312 if( !pp_vout )
313 goto exit;
314
315 *ppp_vout = pp_vout;
316 *pi_vout = p_resource->i_vout;
317
318 for( int i = 0; i < p_resource->i_vout; i++ )
319 {
320 pp_vout[i] = p_resource->pp_vout[i];
321 vlc_object_hold( pp_vout[i] );
322 }
323
324exit:
325 vlc_mutex_unlock( &p_resource->lock_hold );
326}

328/* Audio output */
329audio_output_t *input_resource_GetAout( input_resource_t *p_resource )
330{
331 audio_output_t *p_aout;
332
333 vlc_mutex_lock( &p_resource->lock_hold );
334 p_aout = p_resource->p_aout;
335
336 if( p_aout == NULL || p_resource->b_aout_busy )
337 {
338 msg_Dbg( p_resource->p_parent, "creating audio output" );
339 vlc_mutex_unlock( &p_resource->lock_hold );
340
341 p_aout = aout_New( p_resource->p_parent );
342 if( p_aout == NULL )
343 return NULL; /* failed */
344
345 vlc_mutex_lock( &p_resource->lock_hold );
346 if( p_resource->p_aout == NULL )
347 p_resource->p_aout = p_aout;
348 }
349 else
350 msg_Dbg( p_resource->p_parent, "reusing audio output" );
351
352 if( p_resource->p_aout == p_aout )
353 {
354 assert( !p_resource->b_aout_busy );
355 p_resource->b_aout_busy = true;
356 }
357 vlc_mutex_unlock( &p_resource->lock_hold );
358 return p_aout;
359}

361void input_resource_PutAout( input_resource_t *p_resource,
362 audio_output_t *p_aout )
363{
364 assert( p_aout != NULL );
365
366 vlc_mutex_lock( &p_resource->lock_hold );
367 if( p_aout == p_resource->p_aout )
368 {
369 assert( p_resource->b_aout_busy );
370 p_resource->b_aout_busy = false;
371 msg_Dbg( p_resource->p_parent, "keeping audio output" );
372 p_aout = NULL;
373 }
374 else
375 msg_Dbg( p_resource->p_parent, "destroying extra audio output" );
376 vlc_mutex_unlock( &p_resource->lock_hold );
377
378 if( p_aout != NULL )
379 aout_Destroy( p_aout );
380}

382audio_output_t *input_resource_HoldAout( input_resource_t *p_resource )
383{
384 audio_output_t *p_aout;
385
386 vlc_mutex_lock( &p_resource->lock_hold );
387 p_aout = p_resource->p_aout;
388 if( p_aout != NULL )
389 vlc_object_hold( p_aout );
390 vlc_mutex_unlock( &p_resource->lock_hold );
391
392 return p_aout;
393}

395void input_resource_ResetAout( input_resource_t *p_resource )
396{
397 audio_output_t *p_aout = NULL;
398
399 vlc_mutex_lock( &p_resource->lock_hold );
400 if( !p_resource->b_aout_busy )
401 p_aout = p_resource->p_aout;
402
403 p_resource->p_aout = NULL;
404 p_resource->b_aout_busy = false;
405 vlc_mutex_unlock( &p_resource->lock_hold );
406
407 if( p_aout != NULL )
408 aout_Destroy( p_aout );
409}

411/* Common */
412input_resource_t *input_resource_New( vlc_object_t *p_parent )
413{
414 input_resource_t *p_resource = calloc( 1, sizeof(*p_resource) );
415 if( !p_resource )
416 return NULL;
417
418 atomic_init( &p_resource->refs, 1 );
419 p_resource->p_parent = p_parent;
420 vlc_mutex_init( &p_resource->lock );
421 vlc_mutex_init( &p_resource->lock_hold );
422 return p_resource;
423}
424
425void input_resource_Release( input_resource_t *p_resource )
426{
427 if( atomic_fetch_sub( &p_resource->refs, 1 ) != 1 )
428 return;
429
430 DestroySout( p_resource );
431 DestroyVout( p_resource );
432 if( p_resource->p_aout != NULL )
433 aout_Destroy( p_resource->p_aout );
434
435 vlc_mutex_destroy( &p_resource->lock_hold );
436 vlc_mutex_destroy( &p_resource->lock );
437 free( p_resource );
438}
439
440input_resource_t *input_resource_Hold( input_resource_t *p_resource )
441{
442 atomic_fetch_add( &p_resource->refs, 1 );
443 return p_resource;
444}

446void input_resource_SetInput( input_resource_t *p_resource, input_thread_t *p_input )
447{
448 vlc_mutex_lock( &p_resource->lock );
449
450 if( p_resource->p_input && !p_input )
451 assert( p_resource->i_vout == 0 );
452
453 /* */
454 p_resource->p_input = p_input;
455
456 vlc_mutex_unlock( &p_resource->lock );
457}

481void input_resource_TerminateVout( input_resource_t *p_resource )
482{
483 input_resource_RequestVout( p_resource, NULL, NULL, 0, false );

485bool input_resource_HasVout( input_resource_t *p_resource )
486{
487 vlc_mutex_lock( &p_resource->lock );
488 assert( !p_resource->p_input );
489 const bool b_vout = p_resource->p_vout_free != NULL;
490 vlc_mutex_unlock( &p_resource->lock );
491
492 return b_vout;
493}
