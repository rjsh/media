24#ifndef LIBVLC_INPUT_RESOURCE_H
25#define LIBVLC_INPUT_RESOURCE_H 1
26
27#include <vlc_common.h>
28
29/**
30 * This function set the associated input.
31 */
32void input_resource_SetInput( input_resource_t *, input_thread_t * );
33
34/**
35 * This function handles sout request.
36 */
37sout_instance_t *input_resource_RequestSout( input_resource_t *, sout_instance_t *, const char *psz_sout );
38
39/**
40 * This function handles vout request.
41 */
42vout_thread_t *input_resource_RequestVout( input_resource_t *, vout_thread_t *, video_format_t *, unsigned dpb_size, bool b_recycle );
43
44/**
45 * This function returns one of the current vout if any.
46 *
47 * You must call vlc_object_release on the value returned (if non NULL).
48 */
49vout_thread_t *input_resource_HoldVout( input_resource_t * );
50
51/**
52 * This function returns all current vouts if any.
53 *
54 * You must call vlc_object_release on all values returned (if non NULL).
55 */
56void input_resource_HoldVouts( input_resource_t *, vout_thread_t ***, size_t * );
57
58/**
59 * This function releases all resources (object).
60 */
61void input_resource_Terminate( input_resource_t * );
62
63/**
64 * This function holds the input_resource_t itself
65 */
66input_resource_t *input_resource_Hold( input_resource_t * );
67
68#endif
