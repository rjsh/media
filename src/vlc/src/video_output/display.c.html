54static picture_t *VideoBufferNew(filter_t *filter)
55{
56 vout_display_t *vd = (vout_display_t*)filter->p_owner;
57 const video_format_t *fmt = &filter->fmt_out.video;
58
59 assert(vd->fmt.i_chroma == fmt->i_chroma &&
60 vd->fmt.i_width == fmt->i_width &&
61 vd->fmt.i_height == fmt->i_height);
62
63 picture_pool_t *pool = vout_display_Pool(vd, 3);
64 if (!pool)
65 return NULL;
66 return picture_pool_Get(pool);
67}

69{
70 VLC_UNUSED(filter);
71 picture_Release(picture);
72}

93/**
94 * It creates a new vout_display_t using the given configuration.
95 */
96static vout_display_t *vout_display_New(vlc_object_t *obj,
97 const char *module, bool load_module,
98 const video_format_t *fmt,
99 const vout_display_cfg_t *cfg,
100 vout_display_owner_t *owner)
101{
102 /* */
103 vout_display_t *vd = vlc_custom_create(obj, sizeof(*vd), "vout display" );
104
105 /* */
106 video_format_Copy(&vd->source, fmt);
107
108 /* Picture buffer does not have the concept of aspect ratio */
109 video_format_Copy(&vd->fmt, fmt);
110 vd->fmt.i_sar_num = 0;
111 vd->fmt.i_sar_den = 0;
112
113 vd->info.is_slow = false;
114 vd->info.has_double_click = false;
115 vd->info.has_hide_mouse = false;
116 vd->info.has_pictures_invalid = false;
117 vd->info.has_event_thread = false;
118 vd->info.subpicture_chromas = NULL;
119
120 vd->cfg = cfg;
121 vd->pool = NULL;
122 vd->prepare = NULL;
123 vd->display = NULL;
124 vd->control = NULL;
125 vd->manage = NULL;
126 vd->sys = NULL;
127
128 vd->owner = *owner;
129
130 if (load_module) {
131 vd->module = module_need(vd, "vout display", module, module && *module != '\0');
132 if (!vd->module) {
133 vlc_object_release(vd);
134 return NULL;
135 }
136 } else {
137 vd->module = NULL;
138 }
139 return vd;
140}

175void vout_display_GetDefaultDisplaySize(unsigned *width, unsigned *height,
176 const video_format_t *source,
177 const vout_display_cfg_t *cfg)
178{
179 if (cfg->display.width > 0 && cfg->display.height > 0) {
180 *width = cfg->display.width;
181 *height = cfg->display.height;
182 } else if (cfg->display.width > 0) {
183 *width = cfg->display.width;
184 *height = (int64_t)source->i_visible_height * source->i_sar_den * cfg->display.width * cfg->display.sar.num /
185 source->i_visible_width / source->i_sar_num / cfg->display.sar.den;
186 } else if (cfg->display.height > 0) {
187 *width = (int64_t)source->i_visible_width * source->i_sar_num * cfg->display.height * cfg->display.sar.den /
188 source->i_visible_height / source->i_sar_den / cfg->display.sar.num;
189 *height = cfg->display.height;
190 } else if (source->i_sar_num >= source->i_sar_den) {
191 *width = (int64_t)source->i_visible_width * source->i_sar_num * cfg->display.sar.den / source->i_sar_den / cfg->display.sar.num;
192 *height = source->i_visible_height;
193 } else {
194 *width = source->i_visible_width;
195 *height = (int64_t)source->i_visible_height * source->i_sar_den * cfg->display.sar.num / source->i_sar_num / cfg->display.sar.den;
196 }
197
198 *width = *width * cfg->zoom.num / cfg->zoom.den;
199 *height = *height * cfg->zoom.num / cfg->zoom.den;
200
201 if (ORIENT_IS_SWAP(source->orientation)) {
202
203 unsigned store = *width;
204 *width = *height;
205 *height = store;
206 }
207}

209/* */
210void vout_display_PlacePicture(vout_display_place_t *place,
211 const video_format_t *source,
212 const vout_display_cfg_t *cfg,
213 bool do_clipping)
214{
215 /* */
216 memset(place, 0, sizeof(*place));
217 if (cfg->display.width <= 0 || cfg->display.height <= 0)
218 return;
219
220 /* */
221 unsigned display_width;
222 unsigned display_height;
223
224 video_format_t source_rot;
225 video_format_ApplyRotation(&source_rot, source);
226 source = &source_rot;
227
228 if (cfg->is_display_filled) {
229 display_width = cfg->display.width;
230 display_height = cfg->display.height;
231 } else {
232 vout_display_cfg_t cfg_tmp = *cfg;
233
234 cfg_tmp.display.width = 0;
235 cfg_tmp.display.height = 0;
236 vout_display_GetDefaultDisplaySize(&display_width, &display_height,
237 source, &cfg_tmp);
238
239 if (do_clipping) {
240 display_width = __MIN(display_width, cfg->display.width);
241 display_height = __MIN(display_height, cfg->display.height);
242 }
243 }
244
245 const unsigned width = source->i_visible_width;
246 const unsigned height = source->i_visible_height;
247 /* Compute the height if we use the width to fill up display_width */
248 const int64_t scaled_height = (int64_t)height * display_width * cfg->display.sar.num * source->i_sar_den / width / source->i_sar_num / cfg->display.sar.den;
249 /* And the same but switching width/height */
250 const int64_t scaled_width = (int64_t)width * display_height * cfg->display.sar.den * source->i_sar_num / height / source->i_sar_den / cfg->display.sar.num;
251
252 /* We keep the solution that avoid filling outside the display */
253 if (scaled_width <= cfg->display.width) {
254 place->width = scaled_width;
255 place->height = display_height;
256 } else {
257 place->width = display_width;
258 place->height = scaled_height;
259 }
260
261 /* Compute position */
262 switch (cfg->align.horizontal) {
263 case VOUT_DISPLAY_ALIGN_LEFT:
264 place->x = 0;
265 break;
266 case VOUT_DISPLAY_ALIGN_RIGHT:
267 place->x = cfg->display.width - place->width;
268 break;
269 default:
270 place->x = ((int)cfg->display.width - (int)place->width) / 2;
271 break;
272 }
273
274 switch (cfg->align.vertical) {
275 case VOUT_DISPLAY_ALIGN_TOP:
276 place->y = 0;
277 break;
278 case VOUT_DISPLAY_ALIGN_BOTTOM:
279 place->y = cfg->display.height - place->height;
280 break;
281 default:
282 place->y = ((int)cfg->display.height - (int)place->height) / 2;
283 break;
284 }
285}

287void vout_display_SendMouseMovedDisplayCoordinates(vout_display_t *vd, video_orientation_t orient_display, int m_x, int m_y, vout_display_place_t *place)
288{
289 video_format_t source_rot = vd->source;
290 video_format_TransformTo(&source_rot, orient_display);
291
292 if (place->width > 0 && place->height > 0) {
293
294 int x = (int)(source_rot.i_x_offset +
295 (int64_t)(m_x - place->x) * source_rot.i_visible_width / place->width);
296 int y = (int)(source_rot.i_y_offset +
297 (int64_t)(m_y - place->y) * source_rot.i_visible_height/ place->height);
298
299 video_transform_t transform = video_format_GetTransform(vd->source.orientation, orient_display);
300
301 int store;
302
303 switch (transform) {
304
305 case TRANSFORM_R90:
306 store = x;
307 x = y;
308 y = vd->source.i_visible_height - store;
309 break;
310 case TRANSFORM_R180:
311 x = vd->source.i_visible_width - x;
312 y = vd->source.i_visible_height - y;
313 break;
314 case TRANSFORM_R270:
315 store = x;
316 x = vd->source.i_visible_width - y;
317 y = store;
318 break;
319 case TRANSFORM_HFLIP:
320 x = vd->source.i_visible_width - x;
321 break;
322 case TRANSFORM_VFLIP:
323 y = vd->source.i_visible_height - y;
324 break;
325 case TRANSFORM_TRANSPOSE:
326 store = x;
327 x = y;
328 y = store;
329 break;
330 case TRANSFORM_ANTI_TRANSPOSE:
331 store = x;
332 x = vd->source.i_visible_width - y;
333 y = vd->source.i_visible_height - store;
334 break;
335 default:
336 break;
337 }
338
339 vout_display_SendEventMouseMoved (vd, x, y);
340 }

441static void VoutDisplayCreateRender(vout_display_t *vd)
442{
443 vout_display_owner_sys_t *osys = vd->owner.sys;
444
445 osys->filters = NULL;
446
447 video_format_t v_src = vd->source;
448 v_src.i_sar_num = 0;
449 v_src.i_sar_den = 0;
450
451 video_format_t v_dst = vd->fmt;
452 v_dst.i_sar_num = 0;
453 v_dst.i_sar_den = 0;
454
455 video_format_t v_dst_cmp = v_dst;
456 if ((v_src.i_chroma == VLC_CODEC_J420 && v_dst.i_chroma == VLC_CODEC_I420) ||
457 (v_src.i_chroma == VLC_CODEC_J422 && v_dst.i_chroma == VLC_CODEC_I422) ||
458 (v_src.i_chroma == VLC_CODEC_J440 && v_dst.i_chroma == VLC_CODEC_I440) ||
459 (v_src.i_chroma == VLC_CODEC_J444 && v_dst.i_chroma == VLC_CODEC_I444))
460 v_dst_cmp.i_chroma = v_src.i_chroma;
461
462 const bool convert = memcmp(&v_src, &v_dst_cmp, sizeof(v_src)) != 0;
463 if (!convert)
464 return;
465
466 msg_Dbg(vd, "A filter to adapt decoder to display is needed");
467
468 osys->filters = filter_chain_New(vd, "video filter2", false,
469 FilterAllocationInit,
470 FilterAllocationClean, vd);
471 assert(osys->filters); /* TODO critical */
472
473 /* */
474 es_format_t src;
475 es_format_InitFromVideo(&src, &v_src);
476
477 /* */
478 es_format_t dst;
479
480 filter_t *filter;
481 for (int i = 0; i < 1 + (v_dst_cmp.i_chroma != v_dst.i_chroma); i++) {
482
483 es_format_InitFromVideo(&dst, i == 0 ? &v_dst : &v_dst_cmp);
484
485 filter_chain_Reset(osys->filters, &src, &dst);
486 filter = filter_chain_AppendFilter(osys->filters,
487 NULL, NULL, &src, &dst);
488 if (filter)
489 break;
490 }
491 if (!filter)
492 msg_Err(vd, "Failed to adapt decoder format to display");
493}

508static void VoutDisplayEventMouse(vout_display_t *vd, int event, va_list args)
509{
510 vout_display_owner_sys_t *osys = vd->owner.sys;
511
512 vlc_mutex_lock(&osys->lock);
513
514 /* */
515 vlc_mouse_t m = osys->mouse.state;
516 bool is_ignored = false;
517
518 switch (event) {
519 case VOUT_DISPLAY_EVENT_MOUSE_STATE: {
520 const int x = (int)va_arg(args, int);
521 const int y = (int)va_arg(args, int);
522 const int button_mask = (int)va_arg(args, int);
523
524 vlc_mouse_Init(&m);
525 m.i_x = x;
526 m.i_y = y;
527 m.i_pressed = button_mask;
528 break;
529 }
530 case VOUT_DISPLAY_EVENT_MOUSE_MOVED: {
531 const int x = (int)va_arg(args, int);
532 const int y = (int)va_arg(args, int);
533
534 //msg_Dbg(vd, "VoutDisplayEvent 'mouse' @%d,%d", x, y);
535
536 m.i_x = x;
537 m.i_y = y;
538 m.b_double_click = false;
539 break;
540 }
541 case VOUT_DISPLAY_EVENT_MOUSE_PRESSED:
542 case VOUT_DISPLAY_EVENT_MOUSE_RELEASED: {
543 const int button = (int)va_arg(args, int);
544 const int button_mask = 1 << button;
545
546 /* Ignore inconsistent event */
547 if ((event == VOUT_DISPLAY_EVENT_MOUSE_PRESSED && (osys->mouse.state.i_pressed & button_mask)) ||
548 (event == VOUT_DISPLAY_EVENT_MOUSE_RELEASED && !(osys->mouse.state.i_pressed & button_mask))) {
549 is_ignored = true;
550 break;
551 }
552
553 /* */
554 msg_Dbg(vd, "VoutDisplayEvent 'mouse button' %d t=%d", button, event);
555
556 m.b_double_click = false;
557 if (event == VOUT_DISPLAY_EVENT_MOUSE_PRESSED)
558 m.i_pressed |= button_mask;
559 else
560 m.i_pressed &= ~button_mask;
561 break;
562 }
563 case VOUT_DISPLAY_EVENT_MOUSE_DOUBLE_CLICK:
564 msg_Dbg(vd, "VoutDisplayEvent 'double click'");
565
566 m.b_double_click = true;
567 break;
568 default:
569 assert(0);
570 }
571
572 if (is_ignored) {
573 vlc_mutex_unlock(&osys->lock);
574 return;
575 }
576
577 /* Emulate double-click if needed */
578 if (!vd->info.has_double_click &&
579 vlc_mouse_HasPressed(&osys->mouse.state, &m, MOUSE_BUTTON_LEFT)) {
580 const mtime_t i_date = mdate();
581
582 if (i_date - osys->mouse.last_pressed < osys->mouse.double_click_timeout ) {
583 m.b_double_click = true;
584 osys->mouse.last_pressed = 0;
585 } else {
586 osys->mouse.last_pressed = mdate();
587 }
588 }
589
590 /* */
591 osys->mouse.state = m;
592
593 /* */
594 osys->mouse.ch_activity = true;
595 if (!vd->info.has_hide_mouse)
596 osys->mouse.last_moved = mdate();
597
598 /* */
599 vout_SendEventMouseVisible(osys->vout);
600#ifdef ALLOW_DUMMY_VOUT
601 DummyVoutSendDisplayEventMouse(osys->vout, &osys->vout_mouse, &m);
602#else
603 vout_SendDisplayEventMouse(osys->vout, &m);
604#endif
605 vlc_mutex_unlock(&osys->lock);
606}

627static void VoutDisplayEventKey(vout_display_t *vd, int key)
628{
629 vout_display_owner_sys_t *osys = vd->owner.sys;
630
631 if (!osys->event.fifo) {
632 osys->event.fifo = block_FifoNew();
633 if (!osys->event.fifo)
634 return;
635 if (vlc_clone(&osys->event.thread, VoutDisplayEventKeyDispatch,
636 osys, VLC_THREAD_PRIORITY_LOW)) {
637 block_FifoRelease(osys->event.fifo);
638 osys->event.fifo = NULL;
639 return;
640 }
641 }
642 block_t *event = block_Alloc(sizeof(key));
643 if (event) {
644 memcpy(event->p_buffer, &key, sizeof(key));
645 block_FifoPut(osys->event.fifo, event);
646 }
647}

649static void VoutDisplayEvent(vout_display_t *vd, int event, va_list args)
650{
651 vout_display_owner_sys_t *osys = vd->owner.sys;
652
653 switch (event) {
654 case VOUT_DISPLAY_EVENT_CLOSE: {
655 msg_Dbg(vd, "VoutDisplayEvent 'close'");
656 vout_SendEventClose(osys->vout);
657 break;
658 }
659 case VOUT_DISPLAY_EVENT_KEY: {
660 const int key = (int)va_arg(args, int);
661 msg_Dbg(vd, "VoutDisplayEvent 'key' 0x%2.2x", key);
662 if (vd->info.has_event_thread)
663 vout_SendEventKey(osys->vout, key);
664 else
665 VoutDisplayEventKey(vd, key);
666 break;
667 }
668 case VOUT_DISPLAY_EVENT_MOUSE_STATE:
669 case VOUT_DISPLAY_EVENT_MOUSE_MOVED:
670 case VOUT_DISPLAY_EVENT_MOUSE_PRESSED:
671 case VOUT_DISPLAY_EVENT_MOUSE_RELEASED:
672 case VOUT_DISPLAY_EVENT_MOUSE_DOUBLE_CLICK:
673 VoutDisplayEventMouse(vd, event, args);
674 break;
675
676 case VOUT_DISPLAY_EVENT_FULLSCREEN: {
677 const int is_fullscreen = (int)va_arg(args, int);
678
679 msg_Dbg(vd, "VoutDisplayEvent 'fullscreen' %d", is_fullscreen);
680
681 vlc_mutex_lock(&osys->lock);
682 if (!is_fullscreen != !osys->is_fullscreen) {
683 osys->ch_fullscreen = true;
684 osys->is_fullscreen = is_fullscreen;
685 }
686 vlc_mutex_unlock(&osys->lock);
687 break;
688 }
689
690 case VOUT_DISPLAY_EVENT_WINDOW_STATE: {
691 const unsigned state = va_arg(args, unsigned);
692
693 msg_Dbg(vd, "VoutDisplayEvent 'window state' %u", state);
694
695 vlc_mutex_lock(&osys->lock);
696 if (state != osys->wm_state) {
697 osys->ch_wm_state = true;
698 osys->wm_state = state;
699 }
700 vlc_mutex_unlock(&osys->lock);
701 break;
702 }
703
704 case VOUT_DISPLAY_EVENT_DISPLAY_SIZE: {
705 const int width = (int)va_arg(args, int);
706 const int height = (int)va_arg(args, int);
707 const bool is_fullscreen = (bool)va_arg(args, int);
708 msg_Dbg(vd, "VoutDisplayEvent 'resize' %dx%d %s",
709 width, height, is_fullscreen ? "fullscreen" : "window");
710
711 /* */
712 vlc_mutex_lock(&osys->lock);
713
714 osys->ch_display_size = true;
715 osys->display_width = width;
716 osys->display_height = height;
717 osys->display_is_fullscreen = is_fullscreen;
718 osys->display_is_forced = false;
719
720 vlc_mutex_unlock(&osys->lock);
721 break;
722 }
723
724 case VOUT_DISPLAY_EVENT_PICTURES_INVALID: {
725 msg_Warn(vd, "VoutDisplayEvent 'pictures invalid'");
726
727 /* */
728 assert(vd->info.has_pictures_invalid);
729
730 vlc_mutex_lock(&osys->lock);

805static void VoutDisplayCropRatio(int *left, int *top, int *right, int *bottom,
806 const video_format_t *source,
807 unsigned num, unsigned den)
808{
809 unsigned scaled_width = (uint64_t)source->i_visible_height * num * source->i_sar_den / den / source->i_sar_num;
810 unsigned scaled_height = (uint64_t)source->i_visible_width * den * source->i_sar_num / num / source->i_sar_den;
811
812 if (scaled_width < source->i_visible_width) {
813 *left = (source->i_visible_width - scaled_width) / 2;
814 *top = 0;
815 *right = *left + scaled_width;
816 *bottom = *top + source->i_visible_height;
817 } else {
818 *left = 0;
819 *top = (source->i_visible_height - scaled_height) / 2;
820 *right = *left + source->i_visible_width;
821 *bottom = *top + scaled_height;
822 }
823}

825bool vout_ManageDisplay(vout_display_t *vd, bool allow_reset_pictures)
826{
827 vout_display_owner_sys_t *osys = vd->owner.sys;
828
829 vout_display_Manage(vd);
830
831 /* Handle mouse timeout */
832 const mtime_t date = mdate();
833 bool hide_mouse = false;
834
835 vlc_mutex_lock(&osys->lock);
836
837 if (!osys->mouse.is_hidden &&
838 osys->mouse.last_moved + osys->mouse.hide_timeout < date) {
839 osys->mouse.is_hidden = hide_mouse = true;
840 } else if (osys->mouse.ch_activity) {
841 osys->mouse.is_hidden = false;
842 }
843 osys->mouse.ch_activity = false;
844 vlc_mutex_unlock(&osys->lock);
845
846 if (hide_mouse) {
847 if (!vd->info.has_hide_mouse) {
848 msg_Dbg(vd, "auto hiding mouse cursor");
849 vout_display_Control(vd, VOUT_DISPLAY_HIDE_MOUSE);
850 }
851 vout_SendEventMouseHidden(osys->vout);
852 }
853
854 bool reset_render = false;
855 for (;;) {
856
857 vlc_mutex_lock(&osys->lock);
858
859 bool ch_fullscreen = osys->ch_fullscreen;
860 bool is_fullscreen = osys->is_fullscreen;
861 osys->ch_fullscreen = false;
862
863 bool ch_wm_state = osys->ch_wm_state;
864 unsigned wm_state = osys->wm_state;
865 osys->ch_wm_state = false;
866
867 bool ch_display_size = osys->ch_display_size;
868 int display_width = osys->display_width;
869 int display_height = osys->display_height;
870 bool display_is_fullscreen = osys->display_is_fullscreen;
871 bool display_is_forced = osys->display_is_forced;
872 osys->ch_display_size = false;
873
874 bool reset_pictures;
875 if (allow_reset_pictures) {
876 reset_pictures = osys->reset_pictures;
877 osys->reset_pictures = false;
878 } else {
879 reset_pictures = false;
880 }
881
882 vlc_mutex_unlock(&osys->lock);
883
884 if (!ch_fullscreen &&
885 !ch_display_size &&
886 !reset_pictures &&
887 !osys->ch_display_filled &&
888 !osys->ch_zoom &&
889 !ch_wm_state &&
890 !osys->ch_sar &&
891 !osys->ch_crop) {
892
893 if (!osys->cfg.is_fullscreen && osys->fit_window != 0) {
894 VoutDisplayFitWindow(vd, osys->fit_window == -1);
895 osys->fit_window = 0;
896 continue;
897 }
898 break;
899 }
900
901 /* */
902 if (ch_fullscreen) {
903 vout_display_cfg_t cfg = osys->cfg;
904
905 cfg.is_fullscreen = is_fullscreen;
906 cfg.display.width = cfg.is_fullscreen ? 0 : osys->width_saved;
907 cfg.display.height = cfg.is_fullscreen ? 0 : osys->height_saved;
908
909 if (vout_display_Control(vd, VOUT_DISPLAY_CHANGE_FULLSCREEN, &cfg)) {
910 msg_Err(vd, "Failed to set fullscreen");
911 is_fullscreen = osys->cfg.is_fullscreen;
912 } else if (!is_fullscreen) {
913 vout_display_Control(vd, VOUT_DISPLAY_CHANGE_DISPLAY_SIZE, &cfg, true);
914 }
915 osys->cfg.is_fullscreen = is_fullscreen;
916
917 /* */
918 vout_SendEventFullscreen(osys->vout, osys->cfg.is_fullscreen);
919 }
920
921 /* */
922 if (ch_display_size) {
923 vout_display_cfg_t cfg = osys->cfg;
924 cfg.display.width = display_width;
925 cfg.display.height = display_height;
926
927 if (!cfg.is_fullscreen != !display_is_fullscreen ||
928 vout_display_Control(vd, VOUT_DISPLAY_CHANGE_DISPLAY_SIZE, &cfg, display_is_forced)) {
929 if (!display_is_forced)
930 msg_Err(vd, "Failed to resize display");
931
932 /* We ignore the resized */
933 display_width = osys->cfg.display.width;
934 display_height = osys->cfg.display.height;
935 }
936 osys->cfg.display.width = display_width;
937 osys->cfg.display.height = display_height;
938
939 if (!display_is_fullscreen) {
940 osys->width_saved = display_width;
941 osys->height_saved = display_height;
942 }
943 }
944 /* */
945 if (osys->ch_display_filled) {
946 vout_display_cfg_t cfg = osys->cfg;
947
948 cfg.is_display_filled = osys->is_display_filled;
949
950 if (vout_display_Control(vd, VOUT_DISPLAY_CHANGE_DISPLAY_FILLED, &cfg)) {
951 msg_Err(vd, "Failed to change display filled state");
952 osys->is_display_filled = osys->cfg.is_display_filled;
953 }
954 osys->cfg.is_display_filled = osys->is_display_filled;
955 osys->ch_display_filled = false;
956
957 vout_SendEventDisplayFilled(osys->vout, osys->cfg.is_display_filled);
958 }
959 /* */
960 if (osys->ch_zoom) {
961 vout_display_cfg_t cfg = osys->cfg;
962
963 cfg.zoom.num = osys->zoom.num;
964 cfg.zoom.den = osys->zoom.den;
965
966 if (10 * cfg.zoom.num <= cfg.zoom.den) {
967 cfg.zoom.num = 1;
968 cfg.zoom.den = 10;
969 } else if (cfg.zoom.num >= 10 * cfg.zoom.den) {
970 cfg.zoom.num = 10;
971 cfg.zoom.den = 1;
972 }
973
974 if (vout_display_Control(vd, VOUT_DISPLAY_CHANGE_ZOOM, &cfg)) {
975 msg_Err(vd, "Failed to change zoom");
976 osys->zoom.num = osys->cfg.zoom.num;
977 osys->zoom.den = osys->cfg.zoom.den;
978 } else {
979 osys->fit_window = -1;
980 }
981
982 osys->cfg.zoom.num = osys->zoom.num;
983 osys->cfg.zoom.den = osys->zoom.den;
984 osys->ch_zoom = false;
985
986 vout_SendEventZoom(osys->vout, osys->cfg.zoom.num, osys->cfg.zoom.den);
987 }
988 /* */
989 if (ch_wm_state) {
990 if (vout_display_Control(vd, VOUT_DISPLAY_CHANGE_WINDOW_STATE, wm_state)) {
991 msg_Err(vd, "Failed to set on top");
992 wm_state = osys->wm_state;
993 }
994 osys->wm_state_initial = wm_state;
995
996 /* */
997 vout_SendEventOnTop(osys->vout, osys->wm_state_initial);
998 }
999 /* */
1000 if (osys->ch_sar) {
1001 video_format_t source = vd->source;
1002
1003 if (osys->sar.num > 0 && osys->sar.den > 0) {
1004 source.i_sar_num = osys->sar.num;
1005 source.i_sar_den = osys->sar.den;
1006 } else {
1007 source.i_sar_num = osys->source.i_sar_num;
1008 source.i_sar_den = osys->source.i_sar_den;
1009 }
1010
1011 if (vout_display_Control(vd, VOUT_DISPLAY_CHANGE_SOURCE_ASPECT, &source)) {
1012 /* There nothing much we can do. The only reason a vout display
1013 * does not support it is because it need the core to add black border
1014 * to the video for it.
1015 * TODO add black borders ?
1016 */
1017 msg_Err(vd, "Failed to change source AR");
1018 source = vd->source;
1019 } else if (!osys->fit_window) {
1020 osys->fit_window = 1;
1021 }
1022 vd->source = source;
1023 osys->sar.num = source.i_sar_num;
1024 osys->sar.den = source.i_sar_den;
1025 osys->ch_sar = false;
1026
1027 /* */
1028 if (osys->sar.num == osys->source.i_sar_num &&
1029 osys->sar.den == osys->source.i_sar_den)
1030 {
1031 vout_SendEventSourceAspect(osys->vout, 0, 0);
1032 }
1033 else
1034 {
1035 unsigned dar_num, dar_den;
1036 vlc_ureduce( &dar_num, &dar_den,
1037 osys->sar.num * vd->source.i_visible_width,
1038 osys->sar.den * vd->source.i_visible_height,
1039 65536);
1040 vout_SendEventSourceAspect(osys->vout, dar_num, dar_den);
1041 }
1042 /* If a crop ratio is requested, recompute the parameters */
1043 if (osys->crop.num > 0 && osys->crop.den > 0)
1044 osys->ch_crop = true;
1045 }
1046 /* */
1047 if (osys->ch_crop) {
1048 video_format_t source = vd->source;
1049
1050 unsigned crop_num = osys->crop.num;
1051 unsigned crop_den = osys->crop.den;
1052 if (crop_num > 0 && crop_den > 0) {
1053 video_format_t fmt = osys->source;
1054 fmt.i_sar_num = source.i_sar_num;
1055 fmt.i_sar_den = source.i_sar_den;
1056 VoutDisplayCropRatio(&osys->crop.left, &osys->crop.top,
1057 &osys->crop.right, &osys->crop.bottom,
1058 &fmt, crop_num, crop_den);
1059 }
1060 const int right_max = osys->source.i_x_offset + osys->source.i_visible_width;
1061 const int bottom_max = osys->source.i_y_offset + osys->source.i_visible_height;
1062 int left = VLC_CLIP((int)osys->source.i_x_offset + osys->crop.left,
1063 0, right_max - 1);
1064 int top = VLC_CLIP((int)osys->source.i_y_offset + osys->crop.top,
1065 0, bottom_max - 1);
1066 int right, bottom;
1067 if (osys->crop.right <= 0)
1068 right = (int)(osys->source.i_x_offset + osys->source.i_visible_width) + osys->crop.right;
1069 else
1070 right = (int)osys->source.i_x_offset + osys->crop.right;
1071 right = VLC_CLIP(right, left + 1, right_max);
1072 if (osys->crop.bottom <= 0)
1073 bottom = (int)(osys->source.i_y_offset + osys->source.i_visible_height) + osys->crop.bottom;
1074 else
1075 bottom = (int)osys->source.i_y_offset + osys->crop.bottom;
1076 bottom = VLC_CLIP(bottom, top + 1, bottom_max);
1077
1078 source.i_x_offset = left;
1079 source.i_y_offset = top;
1080 source.i_visible_width = right - left;
1081 source.i_visible_height = bottom - top;
1082 video_format_Print(VLC_OBJECT(vd), "SOURCE ", &osys->source);
1083 video_format_Print(VLC_OBJECT(vd), "CROPPED", &source);
1084 if (vout_display_Control(vd, VOUT_DISPLAY_CHANGE_SOURCE_CROP, &source)) {
1085 msg_Err(vd, "Failed to change source crop TODO implement crop at core");
1086
1087 source = vd->source;
1088 crop_num = osys->crop_saved.num;
1089 crop_den = osys->crop_saved.den;
1090 /* FIXME implement cropping in the core if not supported by the
1091 * vout module (easy)
1092 */
1093 } else if (!osys->fit_window) {
1094 osys->fit_window = 1;
1095 }
1096 vd->source = source;
1097 osys->crop.left = source.i_x_offset - osys->source.i_x_offset;
1098 osys->crop.top = source.i_y_offset - osys->source.i_y_offset;
1099 /* FIXME for right/bottom we should keep the 'type' border vs window */
1100 osys->crop.right = (source.i_x_offset + source.i_visible_width) -
1101 (osys->source.i_x_offset + osys->source.i_visible_width);
1102 osys->crop.bottom = (source.i_y_offset + source.i_visible_height) -
1103 (osys->source.i_y_offset + osys->source.i_visible_height);
1104 osys->crop.num = crop_num;
1105 osys->crop.den = crop_den;
1106 osys->ch_crop = false;
1107
1108 vout_SendEventSourceCrop(osys->vout,
1109 osys->crop.num, osys->crop.den,
1110 osys->crop.left, osys->crop.top,
1111 -osys->crop.right, -osys->crop.bottom);
1112 }
1113
1114 /* */
1115 if (reset_pictures) {
1116 if (vout_display_Control(vd, VOUT_DISPLAY_RESET_PICTURES)) {
1117 /* FIXME what to do here ? */
1118 msg_Err(vd, "Failed to reset pictures (probably fatal)");
1119 }
1120 reset_render = true;
1121 }
1122 }
1123 if (reset_render)
1124 VoutDisplayResetRender(vd);
1125
1126 return reset_render;
1127}

1292static vout_display_t *DisplayNew(vout_thread_t *vout,
1293 const video_format_t *source,
1294 const vout_display_state_t *state,

1297 mtime_t double_click_timeout,
1298 mtime_t hide_timeout,
1299 const vout_display_owner_t *owner_ptr)
1300{
1301 /* */
1302 vout_display_owner_sys_t *osys = calloc(1, sizeof(*osys));
1303 vout_display_cfg_t *cfg = &osys->cfg;
1304
1305 *cfg = state->cfg;

1316 vlc_mutex_init(&osys->lock);
1317
1318 vlc_mouse_Init(&osys->mouse.state);
1319 osys->mouse.last_moved = mdate();
1320 osys->mouse.double_click_timeout = double_click_timeout;
1321 osys->mouse.hide_timeout = hide_timeout;

1323 osys->display_width = cfg->display.width;
1324 osys->display_height = cfg->display.height;
1325 osys->is_display_filled = cfg->is_display_filled;
