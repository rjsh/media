24#ifndef LIBVLC_VOUT_STATISTIC_H
25# define LIBVLC_VOUT_STATISTIC_H
26# include <vlc_atomic.h>
27
28/* NOTE: Both statistics are atomic on their own, so one might be older than
29 * the other one. Currently, only one of them is updated at a time, so this
30 * is a non-issue. */
31typedef struct {
32 atomic_uint displayed;
33 atomic_uint lost;
34} vout_statistic_t;
35
36static inline void vout_statistic_Init(vout_statistic_t *stat)
37{
38 atomic_init(&stat->displayed, 0);
39 atomic_init(&stat->lost, 0);
40}
41
42static inline void vout_statistic_Clean(vout_statistic_t *stat)
43{
44 (void) stat;
45}
46
47static inline void vout_statistic_GetReset(vout_statistic_t *stat, int *displayed, int *lost)
48{
49 *displayed = atomic_exchange(&stat->displayed, 0);
50 *lost = atomic_exchange(&stat->lost, 0);
51}
52
53static inline void vout_statistic_AddDisplayed(vout_statistic_t *stat,
54 int displayed)
55{
56 atomic_fetch_add(&stat->displayed, displayed);
57}
58
59static inline void vout_statistic_AddLost(vout_statistic_t *stat, int lost)
60{
61 atomic_fetch_add(&stat->lost, lost);
62}
63
64#endif
