99/**
100 * Creates a VLC object.
101 *
102 * Note that because the object name pointer must remain valid, potentially
103 * even after the destruction of the object (through the message queues), this
104 * function CANNOT be exported to plugins as is. In this case, the old
105 * vlc_object_create() must be used instead.
106 *
107 * @param p_this an existing VLC object
108 * @param i_size byte size of the object structure
109 * @param psz_type object type name
110 * @return the created object, or NULL.
111 */
112extern void *
113vlc_custom_create (vlc_object_t *p_this, size_t i_size, const char *psz_type);
114#define vlc_custom_create(o, s, n) \
115 vlc_custom_create(VLC_OBJECT(o), s, n)

142/**
143 * Private LibVLC instance data.
144 */
145typedef struct libvlc_priv_t
146{
147 libvlc_int_t public_data;
148
149 /* Logging */
150 struct
151 {
152 void (*cb) (void *, int, const vlc_log_t *, const char *, va_list);
153 void *opaque;
154 signed char verbose;
155 vlc_rwlock_t lock;
156 } log;
157 bool b_stats; ///< Whether to collect stats
158
159 /* Singleton objects */
160 vlm_t *p_vlm; ///< the VLM singleton (or NULL)
161 vlc_object_t *p_dialog_provider; ///< dialog provider
162#ifdef ENABLE_SOUT
163 sap_handler_t *p_sap; ///< SAP SDP advertiser
164#endif
165 struct playlist_t *playlist; ///< Playlist for interfaces
166 struct playlist_preparser_t *parser; ///< Input item meta data handler
167 struct vlc_actions *actions; ///< Hotkeys handler
168
169 /* Objects tree */
170 vlc_mutex_t structure_lock;
171
172 /* Exit callback */
173 vlc_exit_t exit;
174} libvlc_priv_t;
175
176static inline libvlc_priv_t *libvlc_priv (libvlc_int_t *libvlc)
177{
178 return (libvlc_priv_t *)libvlc;
179}
