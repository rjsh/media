48/**
49 * Opens a system file handle.
50 *
51 * @param filename file path to open (with UTF-8 encoding)
52 * @param flags open() flags, see the C library open() documentation
53 * @return a file handle on success, -1 on error (see errno).
54 * @note Contrary to standard open(), this function returns file handles
55 * with the close-on-exec flag enabled.
56 */
57int vlc_open (const char *filename, int flags, ...)
58{
59 unsigned int mode = 0;
60 va_list ap;
61
62 va_start (ap, flags);
63 if (flags & O_CREAT)
64 mode = va_arg (ap, unsigned int);
65 va_end (ap);
66
67#ifdef O_CLOEXEC
68 flags |= O_CLOEXEC;
69#endif
70
71 int fd = open (filename, flags, mode);
72 if (fd != -1)
73 fcntl (fd, F_SETFD, FD_CLOEXEC);
74 return fd;
75}

77/**
78 * Opens a system file handle relative to an existing directory handle.
79 *
80 * @param dir directory file descriptor
81 * @param filename file path to open (with UTF-8 encoding)
82 * @param flags open() flags, see the C library open() documentation
83 * @return a file handle on success, -1 on error (see errno).
84 * @note Contrary to standard open(), this function returns file handles
85 * with the close-on-exec flag enabled.
86 */
87int vlc_openat (int dir, const char *filename, int flags, ...)
88{
89 unsigned int mode = 0;
90 va_list ap;
91
92 va_start (ap, flags);
93 if (flags & O_CREAT)
94 mode = va_arg (ap, unsigned int);
95 va_end (ap);
96
97#ifdef O_CLOEXEC
98 flags |= O_CLOEXEC;
99#endif
100
101#ifdef HAVE_OPENAT
102 int fd = openat (dir, filename, flags, mode);
103 if (fd != -1)
104 fcntl (fd, F_SETFD, FD_CLOEXEC);
105#else
106 VLC_UNUSED (dir);
107 VLC_UNUSED (filename);
108 VLC_UNUSED (mode);
109
110 int fd = -1;
111 errno = ENOSYS;
112#endif
113 return fd;
114}

142/**
143 * Reads the next file name from an open directory.
144 *
145 * @param dir directory handle as returned by vlc_opendir()
146 * (must not be used by another thread concurrently)
147 *
148 * @return a UTF-8 string of the directory entry. The string is valid until
149 * the next call to vlc_readdir() or closedir() on the handle.
150 * If there are no more entries in the directory, NULL is returned.
151 * If an error occurs, errno is set and NULL is returned.
152 */
153char *vlc_readdir( DIR *dir )
154{
155 struct dirent *ent = readdir (dir);
156 return (ent != NULL) ? ent->d_name : NULL;
157}

206/**
207 * Determines the current working directory.
208 *
209 * @return the current working directory (must be free()'d)
210 * or NULL on error
211 */
212char *vlc_getcwd (void)
213{
214 /* Try $PWD */
215 const char *pwd = getenv ("PWD");
216 if (pwd != NULL)
217 {
218 struct stat s1, s2;
219 /* Make sure $PWD is correct */
220 if (stat (pwd, &s1) == 0 && stat (".", &s2) == 0
221 && s1.st_dev == s2.st_dev && s1.st_ino == s2.st_ino)
222 return strdup (pwd);
223 }
224
225 /* Otherwise iterate getcwd() until the buffer is big enough */
226 long path_max = pathconf (".", _PC_PATH_MAX);
227 size_t size = (path_max == -1 || path_max > 4096) ? 4096 : path_max;
228
229 for (;; size *= 2)
230 {
231 char *buf = malloc (size);
232 if (unlikely(buf == NULL))
233 break;
234
235 if (getcwd (buf, size) != NULL)
236 return buf;
237 free (buf);
238
239 if (errno != ERANGE)
240 break;
241 }
242 return NULL;
243}

245/**
246 * Duplicates a file descriptor. The new file descriptor has the close-on-exec
247 * descriptor flag set.
248 * @return a new file descriptor or -1
249 */
250int vlc_dup (int oldfd)
251{
252 int newfd;
253
254#ifdef F_DUPFD_CLOEXEC
255 newfd = fcntl (oldfd, F_DUPFD_CLOEXEC, 0);
256 if (unlikely(newfd == -1 && errno == EINVAL))
257#endif
258 {
259 newfd = dup (oldfd);
260 if (likely(newfd != -1))
261 fcntl (newfd, F_SETFD, FD_CLOEXEC);
262 }
263 return newfd;
264}

266/**
267 * Creates a pipe (see "man pipe" for further reference).
268 */
269int vlc_pipe (int fds[2])
270{
271#ifdef HAVE_PIPE2
272 if (pipe2 (fds, O_CLOEXEC) == 0)
273 return 0;
274 if (errno != ENOSYS)
275 return -1;
276#endif
277
278 if (pipe (fds))
279 return -1;
280
281 fcntl (fds[0], F_SETFD, FD_CLOEXEC);
282 fcntl (fds[1], F_SETFD, FD_CLOEXEC);
283 return 0;
284}

288/**
289 * Creates a socket file descriptor. The new file descriptor has the
290 * close-on-exec flag set.
291 * @param pf protocol family
292 * @param type socket type
293 * @param proto network protocol
294 * @param nonblock true to create a non-blocking socket
295 * @return a new file descriptor or -1
296 */
297int vlc_socket (int pf, int type, int proto, bool nonblock)
298{
299 int fd;
300
301#ifdef SOCK_CLOEXEC
302 type |= SOCK_CLOEXEC;
303 if (nonblock)
304 type |= SOCK_NONBLOCK;
305 fd = socket (pf, type, proto);
306 if (fd != -1 || errno != EINVAL)
307 return fd;
308
309 type &= ~(SOCK_CLOEXEC|SOCK_NONBLOCK);
310#endif
311
312 fd = socket (pf, type, proto);
313 if (fd == -1)
314 return -1;
315
316 fcntl (fd, F_SETFD, FD_CLOEXEC);
317 if (nonblock)
318 fcntl (fd, F_SETFL, fcntl (fd, F_GETFL, 0) | O_NONBLOCK);
319 return fd;
320}

322/**
323 * Accepts an inbound connection request on a listening socket.
324 * The new file descriptor has the close-on-exec flag set.
325 * @param lfd listening socket file descriptor
326 * @param addr pointer to the peer address or NULL [OUT]
327 * @param alen pointer to the length of the peer address or NULL [OUT]
328 * @param nonblock whether to put the new socket in non-blocking mode
329 * @return a new file descriptor, or -1 on error.
330 */
331int vlc_accept (int lfd, struct sockaddr *addr, socklen_t *alen, bool nonblock)
332{
333#ifdef HAVE_ACCEPT4
334 int flags = SOCK_CLOEXEC;
335 if (nonblock)
336 flags |= SOCK_NONBLOCK;
337
338 do
339 {
340 int fd = accept4 (lfd, addr, alen, flags);
341 if (fd != -1)
342 return fd;
343 }
344 while (errno == EINTR);
345
346 if (errno != ENOSYS)
347 return -1;
348#endif
349
350 do
351 {
352 int fd = accept (lfd, addr, alen);
353 if (fd != -1)
354 {
355 fcntl (fd, F_SETFD, FD_CLOEXEC);
356 if (nonblock)
357 fcntl (fd, F_SETFL, fcntl (fd, F_GETFL, 0) | O_NONBLOCK);
358 return fd;
359 }
360 }
361 while (errno == EINTR);
362
363 return -1;
364}
