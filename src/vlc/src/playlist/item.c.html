48/*****************************************************************************
49 * An input item has gained subitems (Event Callback)
50 *****************************************************************************/
51
52static void input_item_add_subitem_tree ( const vlc_event_t * p_event,
53 void * user_data )
54{
55 input_item_t *p_input = p_event->p_obj;
56 playlist_t *p_playlist = (( playlist_item_t* ) user_data)->p_playlist;
57 input_item_node_t *p_new_root = p_event->u.input_item_subitem_tree_added.p_root;
58
59 PL_LOCK;
60
61 playlist_item_t *p_item =
62 playlist_ItemGetByInput( p_playlist, p_input );
63
64 assert( p_item != NULL );
65
66 bool b_current = get_current_status_item( p_playlist ) == p_item;
67 bool b_autostart = var_GetBool( p_playlist, "playlist-autostart" );
68 bool b_stop = p_item->i_flags & PLAYLIST_SUBITEM_STOP_FLAG;
69 bool b_flat = false;
70
71 p_item->i_flags &= ~PLAYLIST_SUBITEM_STOP_FLAG;
72
73 /* We will have to flatten the tree out if we are in "the playlist" node and
74 the user setting demands flat playlist */
75
76 if( !pl_priv(p_playlist)->b_tree ) {
77 playlist_item_t *p_up = p_item;
78 while( p_up->p_parent )
79 {
80 if( p_up->p_parent == p_playlist->p_playing )
81 {
82 b_flat = true;
83 break;
84 }
85 p_up = p_up->p_parent;
86 }
87 }
88
89 int pos = 0;
90
91 /* If we have to flatten out, then take the item's position in the parent as
92 insertion point and delete the item */
93
94 if( b_flat )
95 {
96 playlist_item_t *p_parent = p_item->p_parent;
97 assert( p_parent != NULL );
98
99 int i;
100 for( i = 0; i < p_parent->i_children; i++ )
101 {
102 if( p_parent->pp_children[i] == p_item )
103 {
104 pos = i;
105 break;
106 }
107 }
108 assert( i < p_parent->i_children );
109
110 playlist_DeleteItem( p_playlist, p_item, true );
111
112 p_item = p_parent;
113 }
114 else
115 {
116 pos = p_item->i_children >= 0 ? p_item->i_children : 0;
117 }
118
119 /* At this point:
120 "p_item" is the node where sub-items should be inserted,
121 "pos" is the insertion position in that node */
122
123 int last_pos = playlist_InsertInputItemTree( p_playlist,
124 p_item,
125 p_new_root,
126 pos,
127 b_flat );
128
129 if( !b_flat ) var_SetInteger( p_playlist, "leaf-to-parent", p_item->i_id );
130
131 //control playback only if it was the current playing item that got subitems
132 if( b_current )
133 {
134 if( last_pos == pos || ( b_stop && !b_flat ) || !b_autostart )
135 {
136 /* We stop, either because no sub-item was actually created, or some
137 flags/settings want us to do so at this point */
138 PL_UNLOCK;
139 playlist_Stop( p_playlist );
140 return;
141 }
142 else
143 {
144 /* Continue to play, either random or the first new item */
145 playlist_item_t *p_play_item;
146
147 if( var_GetBool( p_playlist, "random" ) )
148 {
149 p_play_item = NULL;
150 }
151 else
152 {
153 p_play_item = p_item->pp_children[pos];
154 /* NOTE: this is a work around the general bug:
155 if node-to-be-played contains sub-nodes, then second instead
156 of first leaf starts playing (only in case the leafs have just
157 been instered and playlist has not yet been rebuilt.)
158 */
159 while( p_play_item->i_children > 0 )
160 p_play_item = p_play_item->pp_children[0];
161 }
162
163 playlist_Control( p_playlist, PLAYLIST_VIEWPLAY,
164 pl_Locked,
165 get_current_status_node( p_playlist ),
166 p_play_item );
167 }
168 }
169
170 PL_UNLOCK;
171}

183/*****************************************************************************
184 * Listen to vlc_InputItemAddSubItem event
185 *****************************************************************************/
186static void install_input_item_observer( playlist_item_t * p_item )
187{
188 vlc_event_manager_t * p_em = &p_item->p_input->event_manager;
189 vlc_event_attach( p_em, vlc_InputItemSubItemTreeAdded,
190 input_item_add_subitem_tree, p_item );
191 vlc_event_attach( p_em, vlc_InputItemDurationChanged,
192 input_item_changed, p_item );
193 vlc_event_attach( p_em, vlc_InputItemMetaChanged,
194 input_item_changed, p_item );
195 vlc_event_attach( p_em, vlc_InputItemNameChanged,
196 input_item_changed, p_item );
197 vlc_event_attach( p_em, vlc_InputItemInfoChanged,
198 input_item_changed, p_item );
199 vlc_event_attach( p_em, vlc_InputItemErrorWhenReadingChanged,
200 input_item_changed, p_item );
201}
202
203static void uninstall_input_item_observer( playlist_item_t * p_item )
204{
205 vlc_event_manager_t * p_em = &p_item->p_input->event_manager;
206 vlc_event_detach( p_em, vlc_InputItemSubItemTreeAdded,
207 input_item_add_subitem_tree, p_item );
208 vlc_event_detach( p_em, vlc_InputItemMetaChanged,
209 input_item_changed, p_item );
210 vlc_event_detach( p_em, vlc_InputItemDurationChanged,
211 input_item_changed, p_item );
212 vlc_event_detach( p_em, vlc_InputItemNameChanged,
213 input_item_changed, p_item );
214 vlc_event_detach( p_em, vlc_InputItemInfoChanged,
215 input_item_changed, p_item );
216 vlc_event_detach( p_em, vlc_InputItemErrorWhenReadingChanged,
217 input_item_changed, p_item );
218}

220/*****************************************************************************
221 * Playlist item creation
222 *****************************************************************************/
223playlist_item_t *playlist_ItemNewFromInput( playlist_t *p_playlist,
224 input_item_t *p_input )
225{
226 playlist_item_t* p_item = malloc( sizeof( playlist_item_t ) );
227 if( !p_item )
228 return NULL;
229
230 assert( p_input );
231
232 p_item->p_input = p_input;
233 vlc_gc_incref( p_item->p_input );
234
235 p_item->i_id = ++pl_priv(p_playlist)->i_last_playlist_id;
236
237 p_item->p_parent = NULL;
238 p_item->i_children = -1;
239 p_item->pp_children = NULL;
240 p_item->i_flags = 0;
241 p_item->p_playlist = p_playlist;
242
243 install_input_item_observer( p_item );
244
245 return p_item;
246}

252/**
253 * Release an item
254 *
255 * \param p_item item to delete
256 * \return VLC_SUCCESS
257*/
258int playlist_ItemRelease( playlist_item_t *p_item )
259{
260 /* For the assert */
261 playlist_t *p_playlist = p_item->p_playlist;
262 PL_ASSERT_LOCKED;
263
264 /* Surprise, we can't actually do more because we
265 * don't do refcounting, or eauivalent.
266 * Because item are not only accessed by their id
267 * using playlist_item outside the PL_LOCK isn't safe.
268 * Most of the modules does that.
269 *
270 * Who wants to add proper memory management? */
271 uninstall_input_item_observer( p_item );
272 ARRAY_APPEND( pl_priv(p_playlist)->items_to_delete, p_item);
273 return VLC_SUCCESS;
274}

277 * Delete input item
278 *
279 * Remove an input item when it appears from a root playlist item
280 * \param p_playlist playlist object
281 * \param p_input the input to delete
282 * \param p_root root playlist item
283 * \param b_do_stop must stop or not the playlist
284 * \return VLC_SUCCESS or VLC_EGENERIC
285*/
286static int DeleteFromInput( playlist_t *p_playlist, input_item_t *p_input,
287 playlist_item_t *p_root, bool b_do_stop )
288{
289 PL_ASSERT_LOCKED;
290 playlist_item_t *p_item = playlist_ItemFindFromInputAndRoot(
291 p_playlist, p_input, p_root, false );
292 if( !p_item ) return VLC_EGENERIC;
293 return playlist_DeleteItem( p_playlist, p_item, b_do_stop );
294}

433/**
434 * Add an input item to the playlist node
435 *
436 * \param p_playlist the playlist to add into
437 * \param p_input the input item to add
438 * \param i_mode the mode used when adding
439 * \param i_pos the position in the playlist where to add. If this is
440 * PLAYLIST_END the item will be added at the end of the playlist
441 * regardless of its size
442 * \param b_playlist TRUE for playlist, FALSE for media library
443 * \param b_locked TRUE if the playlist is locked
444 * \return VLC_SUCCESS or VLC_ENOMEM or VLC_EGENERIC
445*/
446int playlist_AddInput( playlist_t* p_playlist, input_item_t *p_input,
447 int i_mode, int i_pos, bool b_playlist,
448 bool b_locked )
449{
450 playlist_item_t *p_item;
451
452 if( !pl_priv(p_playlist)->b_doing_ml )
453 PL_DEBUG( "adding item `%s' ( %s )", p_input->psz_name,
454 p_input->psz_uri );
455
456 PL_LOCK_IF( !b_locked );
457
458 p_item = playlist_ItemNewFromInput( p_playlist, p_input );
459 if( p_item == NULL )
460 {
461 PL_UNLOCK_IF( !b_locked );
462 return VLC_ENOMEM;
463 }
464 AddItem( p_playlist, p_item,
465 b_playlist ? p_playlist->p_playing :
466 p_playlist->p_media_library , i_mode, i_pos );
467
468 GoAndPreparse( p_playlist, i_mode, p_item );
469
470 PL_UNLOCK_IF( !b_locked );
471 return VLC_SUCCESS;
472}

474/**
475 * Add an input item to a given node
476 *
477 * \param p_playlist the playlist to add into
478 * \param p_input the input item to add
479 * \param p_parent the parent item to add into
480 * \param i_mode the mode used when addin
481 * \param i_pos the position in the playlist where to add. If this is
482 * PLAYLIST_END the item will be added at the end of the playlist
483 * regardless of its size
484 * \param b_locked TRUE if the playlist is locked
485 * \return the new playlist item
486 */
487playlist_item_t * playlist_NodeAddInput( playlist_t *p_playlist,
488 input_item_t *p_input,
489 playlist_item_t *p_parent,
490 int i_mode, int i_pos,
491 bool b_locked )
492{
493 playlist_item_t *p_item;
494 assert( p_input );
495 assert( p_parent && p_parent->i_children != -1 );
496
497 PL_LOCK_IF( !b_locked );
498
499 p_item = playlist_ItemNewFromInput( p_playlist, p_input );
500 if( p_item == NULL )
501 goto end;
502 AddItem( p_playlist, p_item, p_parent, i_mode, i_pos );
503
504 GoAndPreparse( p_playlist, i_mode, p_item );
505
506end:
507 PL_UNLOCK_IF( !b_locked );
508
509 return p_item;
510}

512/**
513 * Copy an item (and all its children, if any) into another node
514 *
515 * \param p_playlist the playlist to operate on
516 * \param p_item the playlist item to copy
517 * \param p_parent the parent item to copy into
518 * \param i_pos the position in the parent item for the new copy;
519 * if this is PLAYLIST_END, the copy is appended after all
520 * parent's children
521 * \return the position in parent item just behind the last new item inserted
522 */
523int playlist_NodeAddCopy (
524 playlist_t *p_playlist, playlist_item_t *p_item,
525 playlist_item_t *p_parent, int i_pos )
526{
527 PL_ASSERT_LOCKED;
528 assert( p_parent != NULL && p_item != NULL );
529 assert( p_parent->i_children > -1 );
530
531 if( i_pos == PLAYLIST_END ) i_pos = p_parent->i_children;
532
533 bool b_flat = false;
534
535 playlist_item_t *p_up = p_parent;
536 while( p_up )
537 {
538 if( p_up == p_playlist->p_playing )
539 if( !pl_priv(p_playlist)->b_tree ) b_flat = true;
540 if( p_up == p_item )
541 /* TODO: We don't support copying a node into itself (yet),
542 because we insert items as we copy. Instead, we should copy
543 all items first and then insert. */
544 return i_pos;
545 p_up = p_up->p_parent;
546 }
547
548 return RecursiveInsertCopy( p_playlist, p_item, p_parent, i_pos, b_flat );
549}

551/**
552 * Insert a tree of input items into a given playlist node
553 *
554 * \param p_playlist the playlist to insert into
555 * \param p_parent the receiving playlist node (can be an item)
556 * \param p_node the root of input item tree,
557 only it's contents will be inserted
558 * \param i_pos the position in the playlist where to insert. If this is
559 * PLAYLIST_END the items will be added at the end of the playlist
560 * regardless of its size
561 * \param b_flat TRUE if the new tree contents should be flattened into a list
562 * \return the first new leaf inserted (in playing order)
563 */
564int playlist_InsertInputItemTree (
565 playlist_t *p_playlist, playlist_item_t *p_parent,
566 input_item_node_t *p_node, int i_pos, bool b_flat )
567{
568 playlist_item_t *p_first_leaf = NULL;
569 return RecursiveAddIntoParent ( p_playlist, p_parent, p_node, i_pos, b_flat, &p_first_leaf );
570}

577/**
578 * Find an item within a root, given its input id.
579 *
580 * \param p_playlist the playlist object
581 * \param p_item the input item
582 * \param p_root root playlist item
583 * \param b_items_only TRUE if we want the item himself
584 * \return the first found item, or NULL if not found
585 */
586playlist_item_t *playlist_ItemFindFromInputAndRoot( playlist_t *p_playlist,
587 input_item_t *p_item,
588 playlist_item_t *p_root,
589 bool b_items_only )
590{
591 int i;
592 for( i = 0 ; i< p_root->i_children ; i++ )
593 {
594 if( ( b_items_only ? p_root->pp_children[i]->i_children == -1 : 1 ) &&
595 p_root->pp_children[i]->p_input == p_item )
596 {
597 return p_root->pp_children[i];
598 }
599 else if( p_root->pp_children[i]->i_children >= 0 )
600 {
601 playlist_item_t *p_search =
602 playlist_ItemFindFromInputAndRoot( p_playlist, p_item,
603 p_root->pp_children[i],
604 b_items_only );
605 if( p_search ) return p_search;
606 }
607 }
608 return NULL;
609}

612static int ItemIndex ( playlist_item_t *p_item )
613{
614 for( int i = 0; i < p_item->p_parent->i_children; i++ )
615 if( p_item->p_parent->pp_children[i] == p_item ) return i;
616 return -1;
617}

619/**
620 * Moves an item
621 *
622 * This function must be entered with the playlist lock
623 *
624 * \param p_playlist the playlist
625 * \param p_item the item to move
626 * \param p_node the new parent of the item
627 * \param i_newpos the new position under this new parent
628 * \return VLC_SUCCESS or an error
629 */
630int playlist_TreeMove( playlist_t * p_playlist, playlist_item_t *p_item,
631 playlist_item_t *p_node, int i_newpos )
632{
633 PL_ASSERT_LOCKED;
634
635 if( p_node->i_children == -1 ) return VLC_EGENERIC;
636
637 playlist_item_t *p_detach = p_item->p_parent;
638 int i_index = ItemIndex( p_item );
639
640 REMOVE_ELEM( p_detach->pp_children, p_detach->i_children, i_index );
641
642 if( p_detach == p_node && i_index < i_newpos )
643 i_newpos--;
644
645 INSERT_ELEM( p_node->pp_children, p_node->i_children, i_newpos, p_item );
646 p_item->p_parent = p_node;
647
648 pl_priv( p_playlist )->b_reset_currently_playing = true;
649 vlc_cond_signal( &pl_priv( p_playlist )->signal );
650 return VLC_SUCCESS;
651}

653/**
654 * Moves an array of items
655 *
656 * This function must be entered with the playlist lock
657 *
658 * \param p_playlist the playlist
659 * \param i_items the number of indexes to move
660 * \param pp_items the array of indexes to move
661 * \param p_node the target node
662 * \param i_newpos the target position under this node
663 * \return VLC_SUCCESS or an error
664 */
665int playlist_TreeMoveMany( playlist_t *p_playlist,
666 int i_items, playlist_item_t **pp_items,
667 playlist_item_t *p_node, int i_newpos )
668{
669 PL_ASSERT_LOCKED;
670
671 if ( p_node->i_children == -1 ) return VLC_EGENERIC;
672
673 int i;
674 for( i = 0; i < i_items; i++ )
675 {
676 playlist_item_t *p_item = pp_items[i];
677 int i_index = ItemIndex( p_item );
678 playlist_item_t *p_parent = p_item->p_parent;
679 REMOVE_ELEM( p_parent->pp_children, p_parent->i_children, i_index );
680 if ( p_parent == p_node && i_index < i_newpos ) i_newpos--;
681 }
682 for( i = i_items - 1; i >= 0; i-- )
683 {
684 playlist_item_t *p_item = pp_items[i];
685 INSERT_ELEM( p_node->pp_children, p_node->i_children, i_newpos, p_item );
686 p_item->p_parent = p_node;
687 }
688
689 pl_priv( p_playlist )->b_reset_currently_playing = true;
690 vlc_cond_signal( &pl_priv( p_playlist )->signal );
691 return VLC_SUCCESS;
692}

721 * Get the duration of all items in a node.
722 */
723mtime_t playlist_GetNodeDuration( playlist_item_t* node )
724{
725 /* For the assert */
726 playlist_t *p_playlist = node->p_playlist;
727 PL_ASSERT_LOCKED;

730

732 for( int i = 0; i < node->i_children; i++ )

748/* Enqueue an item for preparsing, and play it, if needed */
749static void GoAndPreparse( playlist_t *p_playlist, int i_mode,
750 playlist_item_t *p_item )
751{
752 playlist_private_t *sys = pl_priv(p_playlist);
753
754 PL_ASSERT_LOCKED;
755 if( (i_mode & PLAYLIST_GO ) )
756 {
757 sys->request.b_request = true;
758 sys->request.i_skip = 0;
759 sys->request.p_item = p_item;
760 if( sys->p_input != NULL )
761 input_Stop( sys->p_input, true );
762 sys->request.i_status = PLAYLIST_RUNNING;
763 vlc_cond_signal( &sys->signal );
764 }
765 /* Preparse if no artist/album info, and hasn't been preparsed allready
766 and if user has some preparsing option (auto-preparse variable)
767 enabled*/
768 char *psz_artist = input_item_GetArtist( p_item->p_input );
769 char *psz_album = input_item_GetAlbum( p_item->p_input );
770 if( sys->p_preparser != NULL && !input_item_IsPreparsed( p_item->p_input )
771 && (EMPTY_STR(psz_artist) || EMPTY_STR(psz_album)) )
772 playlist_preparser_Push( sys->p_preparser, p_item->p_input, 0 );
773 free( psz_artist );
774 free( psz_album );
775}

795/* Actually convert an item to a node */
796static void ChangeToNode( playlist_t *p_playlist, playlist_item_t *p_item )
797{
798 int i;
799 if( p_item->i_children != -1 ) return;
800
801 p_item->i_children = 0;
802
803 input_item_t *p_input = p_item->p_input;
804 vlc_mutex_lock( &p_input->lock );
805 p_input->i_type = ITEM_TYPE_NODE;
806 vlc_mutex_unlock( &p_input->lock );
807
808 var_SetAddress( p_playlist, "item-change", p_item->p_input );
809
810 /* Remove it from the array of available items */
811 ARRAY_BSEARCH( p_playlist->items,->i_id, int, p_item->i_id, i );
812 if( i != -1 )
813 ARRAY_REMOVE( p_playlist->items, i );
814}

824static int RecursiveAddIntoParent (
825 playlist_t *p_playlist, playlist_item_t *p_parent,
826 input_item_node_t *p_node, int i_pos, bool b_flat,
827 playlist_item_t **pp_first_leaf )
828{
829 *pp_first_leaf = NULL;
830
831 if( p_parent->i_children == -1 ) ChangeToNode( p_playlist, p_parent );
832
833 if( i_pos == PLAYLIST_END ) i_pos = p_parent->i_children;
834
835 for( int i = 0; i < p_node->i_children; i++ )
836 {
837 input_item_node_t *p_child_node = p_node->pp_children[i];
838
839 playlist_item_t *p_new_item = NULL;
840 bool b_children = p_child_node->i_children > 0;
841
842 //Create the playlist item represented by input node, if allowed.
843 if( !(b_flat && b_children) )
844 {
845 p_new_item = playlist_NodeAddInput( p_playlist,
846 p_child_node->p_item,
847 p_parent,
848 PLAYLIST_INSERT, i_pos,
849 pl_Locked );
850 if( !p_new_item ) return i_pos;
851
852 i_pos++;
853 }
854 //Recurse if any children
855 if( b_children )
856 {
857 //Substitute p_new_item for first child leaf
858 //(If flat, continue counting from current position)
859 int i_last_pos = RecursiveAddIntoParent(
860 p_playlist,
861 p_new_item ? p_new_item : p_parent,
862 p_child_node,
863 ( b_flat ? i_pos : 0 ),
864 b_flat,
865 &p_new_item );
866 //If flat, take position after recursion as current position
867 if( b_flat ) i_pos = i_last_pos;
868 }
869
870 assert( p_new_item != NULL );
871 if( i == 0 ) *pp_first_leaf = p_new_item;
872 }
873 return i_pos;
874}

876static int RecursiveInsertCopy (
877 playlist_t *p_playlist, playlist_item_t *p_item,
878 playlist_item_t *p_parent, int i_pos, bool b_flat )
879{
880 PL_ASSERT_LOCKED;
881 assert( p_parent != NULL && p_item != NULL );
882
883 if( p_item == p_parent ) return i_pos;
884
885 input_item_t *p_input = p_item->p_input;
886
887 if( !(p_item->i_children != -1 && b_flat) )
888 {
889 input_item_t *p_new_input = input_item_Copy( p_input );
890 if( !p_new_input ) return i_pos;
891
892 playlist_item_t *p_new_item = NULL;
893 if( p_item->i_children == -1 )
894 p_new_item = playlist_NodeAddInput( p_playlist, p_new_input,
895 p_parent, PLAYLIST_INSERT, i_pos,
896 pl_Locked );
897 else
898 p_new_item = playlist_NodeCreate( p_playlist, NULL,
899 p_parent, i_pos, 0, p_new_input );
900 vlc_gc_decref( p_new_input );
901 if( !p_new_item ) return i_pos;
902
903 i_pos++;
904
905 if( p_new_item->i_children != -1 )
906 p_parent = p_new_item;
907 }
908
909 for( int i = 0; i < p_item->i_children; i++ )
910 {
911 if( b_flat )
912 i_pos = RecursiveInsertCopy( p_playlist, p_item->pp_children[i],
913 p_parent, i_pos, true );
914 else
915 RecursiveInsertCopy( p_playlist, p_item->pp_children[i],
916 p_parent, p_parent->i_children, false );
917 }
918
919 return i_pos;
920}
