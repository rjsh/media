97static void aout_HotplugNotify (audio_output_t *aout,
98 const char *id, const char *name)
99{
100 aout_owner_t *owner = aout_owner (aout);
101 aout_dev_t *dev, **pp = &owner->dev.list;
102
103 vlc_mutex_lock (&owner->dev.lock);
104 while ((dev = *pp) != NULL)
105 {
106 if (!strcmp (id, dev->id))
107 break;
108 pp = &dev->next;
109 }
110
111 if (name != NULL)
112 {
113 if (dev == NULL) /* Added device */
114 {
115 dev = malloc (sizeof (*dev) + strlen (id));
116 if (unlikely(dev == NULL))
117 goto out;
118 dev->next = NULL;
119 strcpy (dev->id, id);
120 *pp = dev;
121 owner->dev.count++;
122 }
123 else /* Modified device */
124 free (dev->name);
125 dev->name = strdup (name);
126 }
127 else
128 {
129 if (dev != NULL) /* Removed device */
130 {
131 owner->dev.count--;
132 *pp = dev->next;
133 free (dev->name);
134 free (dev);
135 }
136 }
137out:
138 vlc_mutex_unlock (&owner->dev.lock);
139}

156static int FilterCallback (vlc_object_t *obj, const char *var,
157 vlc_value_t prev, vlc_value_t cur, void *data)
158{
159 if (strcmp(prev.psz_string, cur.psz_string))

166/**
167 * Creates an audio output object and initializes an output module.
168 */
169audio_output_t *aout_New (vlc_object_t *parent)
170{
171 vlc_value_t val, text;
172
173 audio_output_t *aout = vlc_custom_create (parent, sizeof (aout_instance_t),
174 "audio output");
175 if (unlikely(aout == NULL))
176 return NULL;
177
178 aout_owner_t *owner = aout_owner (aout);
179
180 vlc_mutex_init (&owner->lock);
181 vlc_mutex_init (&owner->req.lock);
182 vlc_mutex_init (&owner->dev.lock);
183 owner->req.device = (char *)unset_str;
184 owner->req.volume = -1.f;
185 owner->req.mute = -1;
186
187 vlc_object_set_destructor (aout, aout_Destructor);
188
189 /* Audio output module callbacks */
190 var_Create (aout, "volume", VLC_VAR_FLOAT);
191 var_AddCallback (aout, "volume", var_Copy, parent);
192 var_Create (aout, "mute", VLC_VAR_BOOL | VLC_VAR_DOINHERIT);
193 var_AddCallback (aout, "mute", var_Copy, parent);
194 var_Create (aout, "device", VLC_VAR_STRING);
195
196 aout->event.volume_report = aout_VolumeNotify;
197 aout->event.mute_report = aout_MuteNotify;
198 aout->event.policy_report = aout_PolicyNotify;
199 aout->event.device_report = aout_DeviceNotify;
200 aout->event.hotplug_report = aout_HotplugNotify;
201 aout->event.gain_request = aout_GainNotify;
202 aout->event.restart_request = aout_RestartNotify;
203
204 /* Audio output module initialization */
205 aout->start = NULL;
206 aout->stop = NULL;
207 aout->volume_set = NULL;
208 aout->mute_set = NULL;
209 aout->device_select = NULL;
210 owner->module = module_need (aout, "audio output", "$aout", false);
211 if (owner->module == NULL)
212 {
213 msg_Err (aout, "no suitable audio output module");
214 vlc_object_release (aout);
215 return NULL;
216 }
217
218 /*
219 * Persistent audio output variables
220 */
221 module_config_t *cfg;
222 char *str;
223
224 /* Visualizations */
225 var_Create (aout, "visual", VLC_VAR_STRING | VLC_VAR_HASCHOICE);
226 text.psz_string = _("Visualizations");
227 var_Change (aout, "visual", VLC_VAR_SETTEXT, &text, NULL);
228 val.psz_string = (char *)"";
229 text.psz_string = _("Disable");
230 var_Change (aout, "visual", VLC_VAR_ADDCHOICE, &val, &text);
231 val.psz_string = (char *)"spectrometer";
232 text.psz_string = _("Spectrometer");
233 var_Change (aout, "visual", VLC_VAR_ADDCHOICE, &val, &text);
234 val.psz_string = (char *)"scope";
235 text.psz_string = _("Scope");
236 var_Change (aout, "visual", VLC_VAR_ADDCHOICE, &val, &text);
237 val.psz_string = (char *)"spectrum";
238 text.psz_string = _("Spectrum");
239 var_Change (aout, "visual", VLC_VAR_ADDCHOICE, &val, &text);
240 val.psz_string = (char *)"vuMeter";
241 text.psz_string = _("Vu meter");
242 var_Change (aout, "visual", VLC_VAR_ADDCHOICE, &val, &text);
243 /* Look for goom plugin */
244 if (module_exists ("goom"))
245 {
246 val.psz_string = (char *)"goom";
247 text.psz_string = (char *)"Goom";
248 var_Change (aout, "visual", VLC_VAR_ADDCHOICE, &val, &text);
249 }
250 /* Look for libprojectM plugin */
251 if (module_exists ("projectm"))
252 {
253 val.psz_string = (char *)"projectm";
254 text.psz_string = (char*)"projectM";
255 var_Change (aout, "visual", VLC_VAR_ADDCHOICE, &val, &text);
256 }
257 /* Look for VSXu plugin */
258 if (module_exists ("vsxu"))
259 {
260 val.psz_string = (char *)"vsxu";
261 text.psz_string = (char*)"Vovoid VSXu";
262 var_Change (aout, "visual", VLC_VAR_ADDCHOICE, &val, &text);
263 }
264 /* Look for glspectrum plugin */
265 if (module_exists ("glspectrum"))
266 {
267 val.psz_string = (char *)"glspectrum";
268 text.psz_string = (char*)"3D spectrum";
269 var_Change (aout, "visual", VLC_VAR_ADDCHOICE, &val, &text);
270 }
271 str = var_GetNonEmptyString (aout, "effect-list");
272 if (str != NULL)
273 {
274 var_SetString (aout, "visual", str);
275 free (str);
276 }
277
278 var_Create (aout, "audio-filter", VLC_VAR_STRING | VLC_VAR_DOINHERIT);
279 var_AddCallback (aout, "audio-filter", FilterCallback, NULL);
280 text.psz_string = _("Audio filters");
281 var_Change (aout, "audio-filter", VLC_VAR_SETTEXT, &text, NULL);
282
283
284 var_Create (aout, "audio-visual", VLC_VAR_STRING | VLC_VAR_DOINHERIT);
285 text.psz_string = _("Audio visualizations");
286 var_Change (aout, "audio-visual", VLC_VAR_SETTEXT, &text, NULL);
287
288 /* Replay gain */
289 var_Create (aout, "audio-replay-gain-mode",
290 VLC_VAR_STRING | VLC_VAR_DOINHERIT );
291 text.psz_string = _("Replay gain");
292 var_Change (aout, "audio-replay-gain-mode", VLC_VAR_SETTEXT, &text, NULL);
293 cfg = config_FindConfig (VLC_OBJECT(aout), "audio-replay-gain-mode");
294 if (likely(cfg != NULL))
295 for (unsigned i = 0; i < cfg->list_count; i++)
296 {
297 val.psz_string = cfg->list.psz[i];
298 text.psz_string = vlc_gettext(cfg->list_text[i]);
299 var_Change (aout, "audio-replay-gain-mode", VLC_VAR_ADDCHOICE,
300 &val, &text);
301 }
302
303 /* Equalizer */
304 var_Create (aout, "equalizer-preamp", VLC_VAR_FLOAT | VLC_VAR_DOINHERIT);
305 var_Create (aout, "equalizer-bands", VLC_VAR_STRING | VLC_VAR_DOINHERIT);
306 var_Create (aout, "equalizer-preset", VLC_VAR_STRING | VLC_VAR_DOINHERIT);
307
308 return aout;
309}

311/**
312 * Deinitializes an audio output module and destroys an audio output object.
313 */
314void aout_Destroy (audio_output_t *aout)
315{
316 aout_owner_t *owner = aout_owner (aout);
317
318 aout_OutputLock (aout);
319 module_unneed (aout, owner->module);
320 /* Protect against late call from intf.c */
321 aout->volume_set = NULL;
322 aout->mute_set = NULL;
323 aout->device_select = NULL;
324 aout_OutputUnlock (aout);
325
326 var_DelCallback (aout, "audio-filter", FilterCallback, NULL);
327 var_DelCallback (aout, "mute", var_Copy, aout->p_parent);
328 var_SetFloat (aout, "volume", -1.f);
329 var_DelCallback (aout, "volume", var_Copy, aout->p_parent);
330 vlc_object_release (aout);
331}

333/**
334 * Destroys the audio output lock used (asynchronously) by interface functions.
335 */
336static void aout_Destructor (vlc_object_t *obj)
337{
338 audio_output_t *aout = (audio_output_t *)obj;
339 aout_owner_t *owner = aout_owner (aout);
340
341 vlc_mutex_destroy (&owner->dev.lock);
342 for (aout_dev_t *dev = owner->dev.list, *next; dev != NULL; dev = next)
343 {
344 next = dev->next;
345 free (dev->name);
346 free (dev);
347 }
348
349 assert (owner->req.device == unset_str);
350 vlc_mutex_destroy (&owner->req.lock);
351 vlc_mutex_destroy (&owner->lock);
352}

354/**
355 * Starts an audio output stream.
356 * \param fmt audio output stream format [IN/OUT]
357 * \warning The caller must hold the audio output lock.
358 */
359int aout_OutputNew (audio_output_t *aout, audio_sample_format_t *restrict fmt)
360{
361 aout_OutputAssertLocked (aout);
362
363 /* Ideally, the audio filters would be created before the audio output,
364 * and the ideal audio format would be the output of the filters chain.
365 * But that scheme would not really play well with digital pass-through. */
366 if (AOUT_FMT_LINEAR(fmt))
367 { /* Try to stay in integer domain if possible for no/slow FPU. */
368 fmt->i_format = (fmt->i_bitspersample > 16) ? VLC_CODEC_FL32
369 : VLC_CODEC_S16N;
370 aout_FormatPrepare (fmt);
371 }
372
373 if (aout->start (aout, fmt))
374 {
375 msg_Err (aout, "module not functional");
376 return -1;
377 }
378
379 if (!var_Type (aout, "stereo-mode"))
380 {
381 var_Create (aout, "stereo-mode",
382 VLC_VAR_INTEGER | VLC_VAR_HASCHOICE | VLC_VAR_DOINHERIT);
383
384 vlc_value_t txt;
385 txt.psz_string = _("Stereo audio mode");
386 var_Change (aout, "stereo-mode", VLC_VAR_SETTEXT, &txt, NULL);
387 }
388
389 /* The user may have selected a different channels configuration. */
390 var_AddCallback (aout, "stereo-mode", aout_ChannelsRestart, NULL);
391 switch (var_GetInteger (aout, "stereo-mode"))
392 {
393 case AOUT_VAR_CHAN_RSTEREO:
394 fmt->i_original_channels |= AOUT_CHAN_REVERSESTEREO;
395 break;
396 case AOUT_VAR_CHAN_STEREO:
397 fmt->i_original_channels = AOUT_CHANS_STEREO;
398 break;
399 case AOUT_VAR_CHAN_LEFT:
400 fmt->i_original_channels = AOUT_CHAN_LEFT;
401 break;
402 case AOUT_VAR_CHAN_RIGHT:
403 fmt->i_original_channels = AOUT_CHAN_RIGHT;
404 break;
405 case AOUT_VAR_CHAN_DOLBYS:
406 fmt->i_original_channels = AOUT_CHANS_STEREO|AOUT_CHAN_DOLBYSTEREO;
407 break;
408 default:
409 {
410 if ((fmt->i_original_channels & AOUT_CHAN_PHYSMASK)
411 != AOUT_CHANS_STEREO)
412 break;
413
414 vlc_value_t val, txt;
415 val.i_int = 0;
416 var_Change (aout, "stereo-mode", VLC_VAR_DELCHOICE, &val, NULL);
417 if (fmt->i_original_channels & AOUT_CHAN_DOLBYSTEREO)
418 {
419 val.i_int = AOUT_VAR_CHAN_DOLBYS;
420 txt.psz_string = _("Dolby Surround");
421 }
422 else
423 {
424 val.i_int = AOUT_VAR_CHAN_STEREO;
425 txt.psz_string = _("Stereo");
426 }
427 var_Change (aout, "stereo-mode", VLC_VAR_ADDCHOICE, &val, &txt);
428 var_Change (aout, "stereo-mode", VLC_VAR_SETVALUE, &val, NULL);
429 val.i_int = AOUT_VAR_CHAN_LEFT;
430 txt.psz_string = _("Left");
431 var_Change (aout, "stereo-mode", VLC_VAR_ADDCHOICE, &val, &txt);
432 if (fmt->i_original_channels & AOUT_CHAN_DUALMONO)
433 { /* Go directly to the left channel. */
434 fmt->i_original_channels = AOUT_CHAN_LEFT;
435 var_Change (aout, "stereo-mode", VLC_VAR_SETVALUE, &val, NULL);
436 }
437 val.i_int = AOUT_VAR_CHAN_RIGHT;
438 txt.psz_string = _("Right");
439 var_Change (aout, "stereo-mode", VLC_VAR_ADDCHOICE, &val, &txt);
440 val.i_int = AOUT_VAR_CHAN_RSTEREO;
441 txt.psz_string = _("Reverse stereo");
442 var_Change (aout, "stereo-mode", VLC_VAR_ADDCHOICE, &val, &txt);
443 }
444 }
445
446 aout_FormatPrepare (fmt);
447 aout_FormatPrint (aout, "output", fmt);
448 return 0;
449}

546 aout_owner_t *owner = aout_owner (aout);

550
551void aout_OutputUnlock (audio_output_t *aout)
552{
553 aout_owner_t *owner = aout_owner (aout);
554
555 vlc_assert_locked (&owner->lock);
556 vlc_mutex_lock (&owner->req.lock);
557
558 if (owner->req.device != unset_str)
559 {
560 aout_OutputDeviceSet (aout, owner->req.device);
561 free (owner->req.device);
562 owner->req.device = (char *)unset_str;
563 }
564
565 if (owner->req.volume >= 0.f)
566 {
567 aout_OutputVolumeSet (aout, owner->req.volume);
568 owner->req.volume = -1.f;
569 }
570
571 if (owner->req.mute >= 0)
572 {
573 aout_OutputMuteSet (aout, owner->req.mute);
574 owner->req.mute = -1;
575 }
576
577 vlc_mutex_unlock (&owner->lock);
578 /* If another thread is blocked waiting for owner->req.lock at this point,
579 * this aout_OutputUnlock() call will not see and apply its change request.
580 * The other thread will need to apply the change request itself, which
581 * implies it is able to (try-)lock owner->lock. Therefore this thread must
582 * release owner->lock _before_ owner->req.lock. Do not reorder!!! */
583 vlc_mutex_unlock (&owner->req.lock);
584}

651/**
652 * Selects an audio output device.
653 * \param id device ID to select, or NULL for the default device
654 * \return zero on success, non-zero on error (TODO).
655 */
656int aout_DeviceSet (audio_output_t *aout, const char *id)
657{
658 aout_owner_t *owner = aout_owner (aout);
659
660 char *dev = NULL;
661 if (id != NULL)
662 {
663 dev = strdup (id);
664 if (unlikely(dev == NULL))
665 return -1;
666 }
667
668 vlc_mutex_lock (&owner->req.lock);
669 if (owner->req.device != unset_str)
670 free (owner->req.device);
671 owner->req.device = dev;
672 vlc_mutex_unlock (&owner->req.lock);
673
674 if (aout_OutputTryLock (aout) == 0)
675 aout_OutputUnlock (aout);
676 return 0;
677}

679/**
680 * Enumerates possible audio output devices.
681 *
682 * The function will heap-allocate two tables of heap-allocated strings;
683 * the caller is responsible for freeing all strings and both tables.
684 *
685 * \param ids pointer to a table of device identifiers [OUT]
686 * \param names pointer to a table of device human-readable descriptions [OUT]
687 * \return the number of devices, or negative on error.
688 * \note In case of error, *ids and *names are undefined.
689 */
690int aout_DevicesList (audio_output_t *aout, char ***ids, char ***names)
691{
692 aout_owner_t *owner = aout_owner (aout);
693 char **tabid, **tabname;
694 unsigned count;
695
696 vlc_mutex_lock (&owner->dev.lock);
697 count = owner->dev.count;
698 tabid = xmalloc (sizeof (*tabid) * count);
699 tabname = xmalloc (sizeof (*tabname) * count);
700 *ids = tabid;
701 *names = tabname;
702 for (aout_dev_t *dev = owner->dev.list; dev != NULL; dev = dev->next)
703 {
704 *(tabid++) = xstrdup (dev->id);
705 *(tabname++) = xstrdup (dev->name);
706 }
707 vlc_mutex_unlock (&owner->dev.lock);
708
709 return count;
710}
