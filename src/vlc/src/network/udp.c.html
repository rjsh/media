91/* */
92static int net_SetupDgramSocket (vlc_object_t *p_obj, int fd,
93 const struct addrinfo *ptr)
94{
95#ifdef SO_REUSEPORT
96 setsockopt (fd, SOL_SOCKET, SO_REUSEPORT, &(int){ 1 }, sizeof (int));
97#endif
98
99#if defined (_WIN32)
100 if (net_SockAddrIsMulticast (ptr->ai_addr, ptr->ai_addrlen)
101 && (sizeof (struct sockaddr_storage) >= ptr->ai_addrlen))
102 {
103 // This works for IPv4 too - don't worry!
104 struct sockaddr_in6 dumb =
105 {
106 .sin6_family = ptr->ai_addr->sa_family,
107 .sin6_port = ((struct sockaddr_in *)(ptr->ai_addr))->sin_port
108 };
109
110 bind (fd, (struct sockaddr *)&dumb, ptr->ai_addrlen);
111 }
112 else
113#endif
114 if (bind (fd, ptr->ai_addr, ptr->ai_addrlen))
115 {
116 msg_Err( p_obj, "socket bind error: %s", vlc_strerror_c(net_errno) );
117 net_Close (fd);
118 return -1;
119 }
120 return fd;
121}

123/* */
124static int net_ListenSingle (vlc_object_t *obj, const char *host, int port,
125 int protocol)
126{
127 struct addrinfo hints = {
128 .ai_socktype = SOCK_DGRAM,
129 .ai_protocol = protocol,
130 .ai_flags = AI_PASSIVE | AI_NUMERICSERV | AI_IDN,
131 }, *res;
132
133 if (host && !*host)
134 host = NULL;
135
136 msg_Dbg (obj, "net: opening %s datagram port %d",
137 host ? host : "any", port);
138
139 int val = vlc_getaddrinfo (host, port, &hints, &res);
140 if (val)
141 {
142 msg_Err (obj, "Cannot resolve %s port %d : %s", host, port,
143 gai_strerror (val));
144 return -1;
145 }
146
147 val = -1;
148
149 for (const struct addrinfo *ptr = res; ptr != NULL; ptr = ptr->ai_next)
150 {
151 int fd = net_Socket (obj, ptr->ai_family, ptr->ai_socktype,
152 ptr->ai_protocol);
153 if (fd == -1)
154 {
155 msg_Dbg (obj, "socket error: %s", vlc_strerror_c(net_errno));
156 continue;
157 }
158
159#ifdef IPV6_V6ONLY
160 /* Try dual-mode IPv6 if available. */
161 if (ptr->ai_family == AF_INET6)
162 setsockopt (fd, SOL_IPV6, IPV6_V6ONLY, &(int){ 0 }, sizeof (int));
163#endif
164 fd = net_SetupDgramSocket( obj, fd, ptr );
165 if( fd == -1 )
166 continue;
167
168 if (net_SockAddrIsMulticast (ptr->ai_addr, ptr->ai_addrlen)
169 && net_Subscribe (obj, fd, ptr->ai_addr, ptr->ai_addrlen))
170 {
171 net_Close (fd);
172 continue;
173 }
174
175 val = fd;
176 break;
177 }
178
179 freeaddrinfo (res);
180 return val;
181}

184static int net_SetMcastHopLimit( vlc_object_t *p_this,
185 int fd, int family, int hlim )
186{
187 int proto, cmd;
188
189 /* There is some confusion in the world whether IP_MULTICAST_TTL
190 * takes a byte or an int as an argument.
191 * BSD seems to indicate byte so we are going with that and use
192 * int as a fallback to be safe */
193 switch( family )
194 {
195#ifdef IP_MULTICAST_TTL
196 case AF_INET:
197 proto = SOL_IP;
198 cmd = IP_MULTICAST_TTL;
199 break;
200#endif
201
202#ifdef IPV6_MULTICAST_HOPS
203 case AF_INET6:
204 proto = SOL_IPV6;
205 cmd = IPV6_MULTICAST_HOPS;
206 break;
207#endif
208
209 default:
210 errno = EAFNOSUPPORT;
211 msg_Warn( p_this, "%s", vlc_strerror_c(EAFNOSUPPORT) );
212 return VLC_EGENERIC;
213 }
214
215 if( setsockopt( fd, proto, cmd, &hlim, sizeof( hlim ) ) < 0 )
216 {
217 /* BSD compatibility */
218 unsigned char buf;
219
220 msg_Dbg( p_this, "cannot set hop limit (%d): %s", hlim,
221 vlc_strerror_c(net_errno) );
222 buf = (unsigned char)(( hlim > 255 ) ? 255 : hlim);
223 if( setsockopt( fd, proto, cmd, &buf, sizeof( buf ) ) )
224 {
225 msg_Err( p_this, "cannot set hop limit (%d): %s", hlim,
226 vlc_strerror_c(net_errno) );
227 return VLC_EGENERIC;
228 }
229 }
230
231 return VLC_SUCCESS;
232}

235static int net_SetMcastOut (vlc_object_t *p_this, int fd, int family,
236 const char *iface)
237{
238 int scope = if_nametoindex (iface);
239 if (scope == 0)
240 {
241 msg_Err (p_this, "invalid multicast interface: %s", iface);
242 return -1;
243 }
244
245 switch (family)
246 {
247#ifdef IPV6_MULTICAST_IF
248 case AF_INET6:
249 if (setsockopt (fd, SOL_IPV6, IPV6_MULTICAST_IF,
250 &scope, sizeof (scope)) == 0)
251 return 0;
252 break;
253#endif
254
255#ifdef __linux__
256 case AF_INET:
257 {
258 struct ip_mreqn req = { .imr_ifindex = scope };
259 if (setsockopt (fd, SOL_IP, IP_MULTICAST_IF,
260 &req, sizeof (req)) == 0)
261 return 0;
262 break;
263 }
264#endif
265 default:
266 errno = EAFNOSUPPORT;
267 }
268 msg_Err (p_this, "cannot force multicast interface %s: %s", iface,
269 vlc_strerror_c(errno));
270 return -1;
271}

274static unsigned var_GetIfIndex (vlc_object_t *obj)
275{
276 char *ifname = var_InheritString (obj, "miface");
277 if (ifname == NULL)
278 return 0;
279
280 unsigned ifindex = if_nametoindex (ifname);
281 if (ifindex == 0)
282 msg_Err (obj, "invalid multicast interface: %s", ifname);
283 free (ifname);
284 return ifindex;
285}

288/**
289 * IP-agnostic multicast join,
290 * with fallback to old APIs, and fallback from SSM to ASM.
291 */
292static int
293net_SourceSubscribe (vlc_object_t *obj, int fd,
294 const struct sockaddr *src, socklen_t srclen,
295 const struct sockaddr *grp, socklen_t grplen)
296{
297#ifdef MCAST_JOIN_SOURCE_GROUP
298 /* Agnostic SSM multicast join */
299 int level;
300 struct group_source_req gsr;
301
302 memset (&gsr, 0, sizeof (gsr));
303 gsr.gsr_interface = var_GetIfIndex (obj);
304
305 switch (grp->sa_family)
306 {
307#ifdef AF_INET6
308 case AF_INET6:
309 {
310 const struct sockaddr_in6 *g6 = (const struct sockaddr_in6 *)grp;
311
312 level = SOL_IPV6;
313 assert (grplen >= sizeof (struct sockaddr_in6));
314 if (g6->sin6_scope_id != 0)
315 gsr.gsr_interface = g6->sin6_scope_id;
316 break;
317 }
318#endif
319 case AF_INET:
320 level = SOL_IP;
321 break;
322 default:
323 errno = EAFNOSUPPORT;
324 return -1;
325 }
326
327 assert (grplen <= sizeof (gsr.gsr_group));
328 memcpy (&gsr.gsr_source, src, srclen);
329 assert (srclen <= sizeof (gsr.gsr_source));
330 memcpy (&gsr.gsr_group, grp, grplen);
331 if (setsockopt (fd, level, MCAST_JOIN_SOURCE_GROUP,
332 &gsr, sizeof (gsr)) == 0)
333 return 0;
334
335#else
336 if (src->sa_family != grp->sa_family)
337 {
338 errno = EAFNOSUPPORT;
339 return -1;
340 }
341
342 switch (grp->sa_family)
343 {
344# ifdef IP_ADD_SOURCE_MEMBERSHIP
345 /* IPv4-specific API */
346 case AF_INET:
347 {
348 struct ip_mreq_source imr;
349
350 memset (&imr, 0, sizeof (imr));
351 assert (grplen >= sizeof (struct sockaddr_in));
352 imr.imr_multiaddr = ((const struct sockaddr_in *)grp)->sin_addr;
353 assert (srclen >= sizeof (struct sockaddr_in));
354 imr.imr_sourceaddr = ((const struct sockaddr_in *)src)->sin_addr;
355 if (setsockopt (fd, SOL_IP, IP_ADD_SOURCE_MEMBERSHIP,
356 &imr, sizeof (imr)) == 0)
357 return 0;
358 break;
359 }
360# endif
361 default:
362 errno = EAFNOSUPPORT;
363 }
364
365#endif
366 msg_Err (obj, "cannot join source multicast group: %s",
367 vlc_strerror_c(net_errno));
368 msg_Warn (obj, "trying ASM instead of SSM...");
369 return net_Subscribe (obj, fd, grp, grplen);
370}

373int net_Subscribe (vlc_object_t *obj, int fd,
374 const struct sockaddr *grp, socklen_t grplen)
375{
376/* MCAST_JOIN_GROUP was introduced to OS X in v10.7, but it doesn't work,
377 * so ignore it to use the same code as on 10.5 or 10.6 */
378#if defined (MCAST_JOIN_GROUP) && !defined (__APPLE__)
379 /* Agnostic SSM multicast join */
380 int level;
381 struct group_req gr;
382
383 memset (&gr, 0, sizeof (gr));
384 gr.gr_interface = var_GetIfIndex (obj);
385
386 switch (grp->sa_family)
387 {
388#ifdef AF_INET6
389 case AF_INET6:
390 {
391 const struct sockaddr_in6 *g6 = (const struct sockaddr_in6 *)grp;
392
393 level = SOL_IPV6;
394 assert (grplen >= sizeof (struct sockaddr_in6));
395 if (g6->sin6_scope_id != 0)
396 gr.gr_interface = g6->sin6_scope_id;
397 break;
398 }
399#endif
400 case AF_INET:
401 level = SOL_IP;
402 break;
403 default:
404 errno = EAFNOSUPPORT;
405 return -1;
406 }
407
408 assert (grplen <= sizeof (gr.gr_group));
409 memcpy (&gr.gr_group, grp, grplen);
410 if (setsockopt (fd, level, MCAST_JOIN_GROUP, &gr, sizeof (gr)) == 0)
411 return 0;
412
413#else
414 switch (grp->sa_family)
415 {
416# ifdef IPV6_JOIN_GROUP
417 case AF_INET6:
418 {
419 struct ipv6_mreq ipv6mr;
420 const struct sockaddr_in6 *g6 = (const struct sockaddr_in6 *)grp;
421
422 memset (&ipv6mr, 0, sizeof (ipv6mr));
423 assert (grplen >= sizeof (struct sockaddr_in6));
424 ipv6mr.ipv6mr_multiaddr = g6->sin6_addr;
425 ipv6mr.ipv6mr_interface = g6->sin6_scope_id;
426 if (!setsockopt (fd, SOL_IPV6, IPV6_JOIN_GROUP,
427 &ipv6mr, sizeof (ipv6mr)))
428 return 0;
429 break;
430 }
431# endif
432# ifdef IP_ADD_MEMBERSHIP
433 case AF_INET:
434 {
435 struct ip_mreq imr;
436
437 memset (&imr, 0, sizeof (imr));
438 assert (grplen >= sizeof (struct sockaddr_in));
439 imr.imr_multiaddr = ((const struct sockaddr_in *)grp)->sin_addr;
440 if (setsockopt (fd, SOL_IP, IP_ADD_MEMBERSHIP,
441 &imr, sizeof (imr)) == 0)
442 return 0;
443 break;
444 }
445# endif
446 default:
447 errno = EAFNOSUPPORT;
448 }
449
450#endif
451 msg_Err (obj, "cannot join multicast group: %s",
452 vlc_strerror_c(net_errno));
453 return -1;
454}

457static int net_SetDSCP( int fd, uint8_t dscp )
458{
459 struct sockaddr_storage addr;
460 if( getsockname( fd, (struct sockaddr *)&addr, &(socklen_t){ sizeof (addr) }) )
461 return -1;
462
463 int level, cmd;
464
465 switch( addr.ss_family )
466 {
467#ifdef IPV6_TCLASS
468 case AF_INET6:
469 level = SOL_IPV6;
470 cmd = IPV6_TCLASS;
471 break;
472#endif
473
474 case AF_INET:
475 level = SOL_IP;
476 cmd = IP_TOS;
477 break;
478
479 default:
480#ifdef ENOPROTOOPT
481 errno = ENOPROTOOPT;
482#endif
483 return -1;
484 }
485
486 return setsockopt( fd, level, cmd, &(int){ dscp }, sizeof (int));
487}

489#undef net_ConnectDgram
490/*****************************************************************************
491 * net_ConnectDgram:
492 *****************************************************************************
493 * Open a datagram socket to send data to a defined destination, with an
494 * optional hop limit.
495 *****************************************************************************/
496int net_ConnectDgram( vlc_object_t *p_this, const char *psz_host, int i_port,
497 int i_hlim, int proto )
498{
499 struct addrinfo hints = {
500 .ai_socktype = SOCK_DGRAM,
501 .ai_protocol = proto,
502 .ai_flags = AI_NUMERICSERV | AI_IDN,
503 }, *res;
504 int i_handle = -1;
505 bool b_unreach = false;
506
507 if( i_hlim < 0 )
508 i_hlim = var_InheritInteger( p_this, "ttl" );
509
510 msg_Dbg( p_this, "net: connecting to [%s]:%d", psz_host, i_port );
511
512 int val = vlc_getaddrinfo (psz_host, i_port, &hints, &res);
513 if (val)
514 {
515 msg_Err (p_this, "cannot resolve [%s]:%d : %s", psz_host, i_port,
516 gai_strerror (val));
517 return -1;
518 }
519
520 for (struct addrinfo *ptr = res; ptr != NULL; ptr = ptr->ai_next)
521 {
522 char *str;
523 int fd = net_Socket (p_this, ptr->ai_family, ptr->ai_socktype,
524 ptr->ai_protocol);
525 if (fd == -1)
526 continue;
527
528 /* Allow broadcast sending */
529 setsockopt (fd, SOL_SOCKET, SO_BROADCAST, &(int){ 1 }, sizeof (int));
530
531 if( i_hlim >= 0 )
532 net_SetMcastHopLimit( p_this, fd, ptr->ai_family, i_hlim );
533
534 str = var_InheritString (p_this, "miface");
535 if (str != NULL)
536 {
537 net_SetMcastOut (p_this, fd, ptr->ai_family, str);
538 free (str);
539 }
540
541 net_SetDSCP (fd, var_InheritInteger (p_this, "dscp"));
542
543 if( connect( fd, ptr->ai_addr, ptr->ai_addrlen ) == 0 )
544 {
545 /* success */
546 i_handle = fd;
547 break;
548 }
549
550#if defined( _WIN32 )
551 if( WSAGetLastError () == WSAENETUNREACH )
552#else
553 if( errno == ENETUNREACH )
554#endif
555 b_unreach = true;
556 else
557 msg_Warn( p_this, "%s port %d : %s", psz_host, i_port,
558 vlc_strerror_c(errno) );
559 net_Close( fd );
560 }
561
562 freeaddrinfo( res );
563
564 if( i_handle == -1 )
565 {
566 if( b_unreach )
567 msg_Err( p_this, "Host %s port %d is unreachable", psz_host,
568 i_port );
569 return -1;
570 }
571
572 return i_handle;
573}

575#undef net_OpenDgram
576/*****************************************************************************
577 * net_OpenDgram:
578 *****************************************************************************
579 * OpenDgram a datagram socket and return a handle
580 *****************************************************************************/
581int net_OpenDgram( vlc_object_t *obj, const char *psz_bind, int i_bind,
582 const char *psz_server, int i_server, int protocol )
583{
584 if ((psz_server == NULL) || (psz_server[0] == '\0'))
585 return net_ListenSingle (obj, psz_bind, i_bind, protocol);
586
587 msg_Dbg (obj, "net: connecting to [%s]:%d from [%s]:%d",
588 psz_server, i_server, psz_bind, i_bind);
589
590 struct addrinfo hints = {
591 .ai_socktype = SOCK_DGRAM,
592 .ai_protocol = protocol,
593 .ai_flags = AI_NUMERICSERV | AI_IDN,
594 }, *loc, *rem;
595
596 int val = vlc_getaddrinfo (psz_server, i_server, &hints, &rem);
597 if (val)
598 {
599 msg_Err (obj, "cannot resolve %s port %d : %s", psz_bind, i_bind,
600 gai_strerror (val));
601 return -1;
602 }
603
604 hints.ai_flags |= AI_PASSIVE;
605 val = vlc_getaddrinfo (psz_bind, i_bind, &hints, &loc);
606 if (val)
607 {
608 msg_Err (obj, "cannot resolve %s port %d : %s", psz_bind, i_bind,
609 gai_strerror (val));
610 freeaddrinfo (rem);
611 return -1;
612 }
613
614 val = -1;
615 for (struct addrinfo *ptr = loc; ptr != NULL; ptr = ptr->ai_next)
616 {
617 int fd = net_Socket (obj, ptr->ai_family, ptr->ai_socktype,
618 ptr->ai_protocol);
619 if (fd == -1)
620 continue; // usually, address family not supported
621
622 fd = net_SetupDgramSocket( obj, fd, ptr );
623 if( fd == -1 )
624 continue;
625
626 for (struct addrinfo *ptr2 = rem; ptr2 != NULL; ptr2 = ptr2->ai_next)
627 {
628 if ((ptr2->ai_family != ptr->ai_family)
629 || (ptr2->ai_socktype != ptr->ai_socktype)
630 || (ptr2->ai_protocol != ptr->ai_protocol))
631 continue;
632
633 if (net_SockAddrIsMulticast (ptr->ai_addr, ptr->ai_addrlen)
634 ? net_SourceSubscribe (obj, fd,
635 ptr2->ai_addr, ptr2->ai_addrlen,
636 ptr->ai_addr, ptr->ai_addrlen)
637 : connect (fd, ptr2->ai_addr, ptr2->ai_addrlen))
638 {
639 msg_Err (obj, "cannot connect to %s port %d: %s",
640 psz_server, i_server, vlc_strerror_c(net_errno));
641 continue;
642 }
643 val = fd;
644 break;
645 }
646
647 if (val != -1)
648 break;
649
650 net_Close (fd);
651 }
652
653 freeaddrinfo (rem);
654 freeaddrinfo (loc);
655 return val;
656}

659/**
660 * net_SetCSCov:
661 * Sets the send and receive checksum coverage of a socket:
662 * @param fd socket
663 * @param sendcov payload coverage of sent packets (bytes), -1 for full
664 * @param recvcov minimum payload coverage of received packets, -1 for full
665 */
666int net_SetCSCov (int fd, int sendcov, int recvcov)
667{
668 int type;
669
670 if (getsockopt (fd, SOL_SOCKET, SO_TYPE,
671 &type, &(socklen_t){ sizeof (type) }))
672 return VLC_EGENERIC;
673
674 switch (type)
675 {
676#ifdef UDPLITE_RECV_CSCOV
677 case SOCK_DGRAM: /* UDP-Lite */
678 if (sendcov == -1)
679 sendcov = 0;
680 else
681 sendcov += 8; /* partial */
682 if (setsockopt (fd, SOL_UDPLITE, UDPLITE_SEND_CSCOV, &sendcov,
683 sizeof (sendcov)))
684 return VLC_EGENERIC;
685
686 if (recvcov == -1)
687 recvcov = 0;
688 else
689 recvcov += 8;
690 if (setsockopt (fd, SOL_UDPLITE, UDPLITE_RECV_CSCOV,
691 &recvcov, sizeof (recvcov)))
692 return VLC_EGENERIC;
693
694 return VLC_SUCCESS;
695#endif
696#ifdef DCCP_SOCKOPT_SEND_CSCOV
697 case SOCK_DCCP: /* DCCP and its ill-named socket type */
698 if ((sendcov == -1) || (sendcov > 56))
699 sendcov = 0;
700 else
701 sendcov = (sendcov + 3) / 4;
702 if (setsockopt (fd, SOL_DCCP, DCCP_SOCKOPT_SEND_CSCOV,
703 &sendcov, sizeof (sendcov)))
704 return VLC_EGENERIC;
705
706 if ((recvcov == -1) || (recvcov > 56))
707 recvcov = 0;
708 else
709 recvcov = (recvcov + 3) / 4;
710 if (setsockopt (fd, SOL_DCCP, DCCP_SOCKOPT_RECV_CSCOV,
711 &recvcov, sizeof (recvcov)))
712 return VLC_EGENERIC;
713
714 return VLC_SUCCESS;
715#endif
716 }
717#if !defined( UDPLITE_RECV_CSCOV ) && !defined( DCCP_SOCKOPT_SEND_CSCOV )
718 VLC_UNUSED(sendcov);
719 VLC_UNUSED(recvcov);
720#endif
721
722 return VLC_EGENERIC;
723}
