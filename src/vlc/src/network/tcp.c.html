66/*****************************************************************************
67 * net_Connect:
68 *****************************************************************************
69 * Open a network connection.
70 * @return socket handler or -1 on error.
71 *****************************************************************************/
72int net_Connect( vlc_object_t *p_this, const char *psz_host, int i_port,
73 int type, int proto )
74{
75 const char *psz_realhost;
76 char *psz_socks;
77 int i_realport, i_handle = -1;
78
79 int evfd = vlc_object_waitpipe (p_this);
80 if (evfd == -1)
81 return -1;
82
83 psz_socks = var_InheritString( p_this, "socks" );
84 if( psz_socks != NULL )
85 {
86 char *psz = strchr( psz_socks, ':' );
87
88 if( psz )
89 *psz++ = '\0';
90
91 psz_realhost = psz_socks;
92 i_realport = ( psz != NULL ) ? atoi( psz ) : 1080;
93
94 msg_Dbg( p_this, "net: connecting to %s port %d (SOCKS) "
95 "for %s port %d", psz_realhost, i_realport,
96 psz_host, i_port );
97
98 /* We only implement TCP with SOCKS */
99 switch( type )
100 {
101 case 0:
102 type = SOCK_STREAM;
103 case SOCK_STREAM:
104 break;
105 default:
106 msg_Err( p_this, "Socket type not supported through SOCKS" );
107 free( psz_socks );
108 return -1;
109 }
110 switch( proto )
111 {
112 case 0:
113 proto = IPPROTO_TCP;
114 case IPPROTO_TCP:
115 break;
116 default:
117 msg_Err( p_this, "Transport not supported through SOCKS" );
118 free( psz_socks );
119 return -1;
120 }
121 }
122 else
123 {
124 psz_realhost = psz_host;
125 i_realport = i_port;
126
127 msg_Dbg( p_this, "net: connecting to %s port %d", psz_realhost,
128 i_realport );
129 }
130
131 struct addrinfo hints = {
132 .ai_socktype = type,
133 .ai_protocol = proto,
134 .ai_flags = AI_NUMERICSERV | AI_IDN,
135 }, *res;
136
137 int val = vlc_getaddrinfo (psz_realhost, i_realport, &hints, &res);
138
139 if (val)
140 {
141 msg_Err (p_this, "cannot resolve %s port %d : %s", psz_realhost,
142 i_realport, gai_strerror (val));
143 free( psz_socks );
144 return -1;
145 }
146 free( psz_socks );
147
148 int timeout = var_InheritInteger (p_this, "ipv4-timeout");
149 if (timeout < 0)
150 timeout = -1;
151
152 for (struct addrinfo *ptr = res; ptr != NULL; ptr = ptr->ai_next)
153 {
154 int fd = net_Socket( p_this, ptr->ai_family,
155 ptr->ai_socktype, ptr->ai_protocol );
156 if( fd == -1 )
157 {
158 msg_Dbg( p_this, "socket error: %s", vlc_strerror_c(net_errno) );
159 continue;
160 }
161
162 if( connect( fd, ptr->ai_addr, ptr->ai_addrlen ) )
163 {
164 int val;
165
166 if( net_errno != EINPROGRESS && net_errno != EINTR )
167 {
168 msg_Err( p_this, "connection failed: %s",
169 vlc_strerror_c(net_errno) );
170 goto next_ai;
171 }
172
173 struct pollfd ufd[2] = {
174 { .fd = fd, .events = POLLOUT },
175 { .fd = evfd, .events = POLLIN },
176 };
177
178 do
179 /* NOTE: timeout screwed up if we catch a signal (EINTR) */
180 val = poll (ufd, sizeof (ufd) / sizeof (ufd[0]), timeout);
181 while ((val == -1) && (net_errno == EINTR));
182
183 switch (val)
184 {
185 case -1: /* error */
186 msg_Err (p_this, "polling error: %s",
187 vlc_strerror_c(net_errno));
188 goto next_ai;
189
190 case 0: /* timeout */
191 msg_Warn (p_this, "connection timed out");
192 goto next_ai;
193
194 default: /* something happended */
195 if (ufd[1].revents)
196 goto next_ai; /* LibVLC object killed */
197 }
198
199 /* There is NO WAY around checking SO_ERROR.
200 * Don't ifdef it out!!! */
201 if (getsockopt (fd, SOL_SOCKET, SO_ERROR, &val,
202 &(socklen_t){ sizeof (val) }) || val)
203 {
204 msg_Err (p_this, "connection failed: %s",
205 vlc_strerror_c(val));
206 goto next_ai;
207 }
208 }
209
210 msg_Dbg( p_this, "connection succeeded (socket = %d)", fd );
211 i_handle = fd; /* success! */
212 break;
213
214next_ai: /* failure */
215 net_Close( fd );
216 continue;
217 }
218
219 freeaddrinfo( res );
220
221 if( i_handle == -1 )
222 return -1;
223
224 if( psz_socks != NULL )
225 {
226 /* NOTE: psz_socks already free'd! */
227 char *psz_user = var_InheritString( p_this, "socks-user" );
228 char *psz_pwd = var_InheritString( p_this, "socks-pwd" );
229
230 if( SocksHandshakeTCP( p_this, i_handle, 5, psz_user, psz_pwd,
231 psz_host, i_port ) )
232 {
233 msg_Err( p_this, "SOCKS handshake failed" );
234 net_Close( i_handle );
235 i_handle = -1;
236 }
237
238 free( psz_user );
239 free( psz_pwd );
240 }
241
242 return i_handle;
243}

246int net_AcceptSingle (vlc_object_t *obj, int lfd)
247{
248 int fd = vlc_accept (lfd, NULL, NULL, true);
249 if (fd == -1)
250 {
251 if (net_errno != EAGAIN && net_errno != EWOULDBLOCK)
252 msg_Err (obj, "accept failed (from socket %d): %s", lfd,
253 vlc_strerror_c(net_errno));
254 return -1;
255 }
256
257 msg_Dbg (obj, "accepted socket %d (from socket %d)", fd, lfd);
258 setsockopt (fd, SOL_SOCKET, SO_REUSEADDR, &(int){ 1 }, sizeof(int));
259 return fd;
260}

264/**
265 * Accepts an new connection on a set of listening sockets.
266 * If there are no pending connections, this function will wait.
267 * @note If the thread needs to handle events other than incoming connections,
268 * you need to use poll() and net_AcceptSingle() instead.
269 *
270 * @param p_this VLC object for logging and object kill signal
271 * @param pi_fd listening socket set
272 * @return -1 on error (may be transient error due to network issues),
273 * a new socket descriptor on success.
274 */
275int net_Accept (vlc_object_t *p_this, int *pi_fd)
276{
277 int evfd = vlc_object_waitpipe (p_this);
278
279 assert (pi_fd != NULL);
280
281 unsigned n = 0;
282 while (pi_fd[n] != -1)
283 n++;
284 struct pollfd ufd[n + 1];
285
286 /* Initialize file descriptor set */
287 for (unsigned i = 0; i <= n; i++)
288 {
289 ufd[i].fd = (i < n) ? pi_fd[i] : evfd;
290 ufd[i].events = POLLIN;
291 }
292 ufd[n].revents = 0;
293
294 for (;;)
295 {
296 while (poll (ufd, n + (evfd != -1), -1) == -1)
297 {
298 if (net_errno != EINTR)
299 {
300 msg_Err (p_this, "poll error: %s", vlc_strerror_c(net_errno));
301 return -1;
302 }
303 }
304
305 for (unsigned i = 0; i < n; i++)
306 {
307 if (ufd[i].revents == 0)
308 continue;
309
310 int sfd = ufd[i].fd;
311 int fd = net_AcceptSingle (p_this, sfd);
312 if (fd == -1)
313 continue;
314
315 /*
316 * Move listening socket to the end to let the others in the
317 * set a chance next time.
318 */
319 memmove (pi_fd + i, pi_fd + i + 1, n - (i + 1));
320 pi_fd[n - 1] = sfd;
321 return fd;
322 }
323
324 if (ufd[n].revents)
325 {
326 errno = EINTR;
327 break;
328 }
329 }
330 return -1;
331}

334/*****************************************************************************
335 * SocksNegotiate:
336 *****************************************************************************
337 * Negotiate authentication with a SOCKS server.
338 *****************************************************************************/
339static int SocksNegotiate( vlc_object_t *p_obj,
340 int fd, int i_socks_version,
341 const char *psz_socks_user,
342 const char *psz_socks_passwd )
343{
344 uint8_t buffer[128+2*256];
345 int i_len;
346 bool b_auth = false;
347
348 if( i_socks_version != 5 )
349 return VLC_SUCCESS;
350
351 /* We negotiate authentication */
352 buffer[0] = i_socks_version; /* SOCKS version */
353 if( psz_socks_user != NULL && psz_socks_passwd != NULL )
354 {
355 buffer[1] = 2; /* Number of methods */
356 buffer[2] = 0x00; /* - No auth required */
357 buffer[3] = 0x02; /* - USer/Password */
358 i_len = 4;
359 b_auth = true;
360 }
361 else
362 {
363 buffer[1] = 1; /* Number of methods */
364 buffer[2] = 0x00; /* - No auth required */
365 i_len = 3;
366 }
367
368 if( net_Write( p_obj, fd, NULL, buffer, i_len ) != i_len )
369 return VLC_EGENERIC;
370 if( net_Read( p_obj, fd, NULL, buffer, 2, true ) != 2 )
371 return VLC_EGENERIC;
372
373 msg_Dbg( p_obj, "socks: v=%d method=%x", buffer[0], buffer[1] );
374
375 if( buffer[1] == 0x00 )
376 {
377 msg_Dbg( p_obj, "socks: no authentication required" );
378 }
379 else if( buffer[1] == 0x02 )
380 {
381 int i_len1 = __MIN( strlen(psz_socks_user), 255 );
382 int i_len2 = __MIN( strlen(psz_socks_passwd), 255 );
383 msg_Dbg( p_obj, "socks: username/password authentication" );
384
385 /* XXX: we don't support user/pwd > 255 (truncated)*/
386 buffer[0] = i_socks_version; /* Version */
387 buffer[1] = i_len1; /* User length */
388 memcpy( &buffer[2], psz_socks_user, i_len1 );
389 buffer[2+i_len1] = i_len2; /* Password length */
390 memcpy( &buffer[2+i_len1+1], psz_socks_passwd, i_len2 );
391
392 i_len = 3 + i_len1 + i_len2;
393
394 if( net_Write( p_obj, fd, NULL, buffer, i_len ) != i_len )
395 return VLC_EGENERIC;
396
397 if( net_Read( p_obj, fd, NULL, buffer, 2, true ) != 2 )
398 return VLC_EGENERIC;
399
400 msg_Dbg( p_obj, "socks: v=%d status=%x", buffer[0], buffer[1] );
401 if( buffer[1] != 0x00 )
402 {
403 msg_Err( p_obj, "socks: authentication rejected" );
404 return VLC_EGENERIC;
405 }
406 }
407 else
408 {
409 if( b_auth )
410 msg_Err( p_obj, "socks: unsupported authentication method %x",
411 buffer[0] );
412 else
413 msg_Err( p_obj, "socks: authentication needed" );
414 return VLC_EGENERIC;
415 }
416
417 return VLC_SUCCESS;
418}

420/*****************************************************************************
421 * SocksHandshakeTCP:
422 *****************************************************************************
423 * Open a TCP connection using a SOCKS server and return a handle (RFC 1928)
424 *****************************************************************************/
425static int SocksHandshakeTCP( vlc_object_t *p_obj,
426 int fd,
427 int i_socks_version,
428 const char *psz_user, const char *psz_passwd,
429 const char *psz_host, int i_port )
430{
431 uint8_t buffer[128+2*256];
432
433 if( i_socks_version != 4 && i_socks_version != 5 )
434 {
435 msg_Warn( p_obj, "invalid socks protocol version %d", i_socks_version );
436 i_socks_version = 5;
437 }
438
439 if( i_socks_version == 5 &&
440 SocksNegotiate( p_obj, fd, i_socks_version,
441 psz_user, psz_passwd ) )
442 return VLC_EGENERIC;
443
444 if( i_socks_version == 4 )
445 {
446 /* v4 only support ipv4 */
447 static const struct addrinfo hints = {
448 .ai_family = AF_INET,
449 .ai_socktype = SOCK_STREAM,
450 .ai_protocol = IPPROTO_TCP,
451 .ai_flags = AI_IDN,
452 };
453 struct addrinfo *res;
454
455 if (vlc_getaddrinfo (psz_host, 0, &hints, &res))
456 return VLC_EGENERIC;
457
458 buffer[0] = i_socks_version;
459 buffer[1] = 0x01; /* CONNECT */
460 SetWBE( &buffer[2], i_port ); /* Port */
461 memcpy (&buffer[4], /* Address */
462 &((struct sockaddr_in *)(res->ai_addr))->sin_addr, 4);
463 freeaddrinfo (res);
464
465 buffer[8] = 0; /* Empty user id */
466
467 if( net_Write( p_obj, fd, NULL, buffer, 9 ) != 9 )
468 return VLC_EGENERIC;
469 if( net_Read( p_obj, fd, NULL, buffer, 8, true ) != 8 )
470 return VLC_EGENERIC;
471
472 msg_Dbg( p_obj, "socks: v=%d cd=%d",
473 buffer[0], buffer[1] );
474
475 if( buffer[1] != 90 )
476 return VLC_EGENERIC;
477 }
478 else if( i_socks_version == 5 )
479 {
480 int i_hlen = __MIN(strlen( psz_host ), 255);
481 int i_len;
482
483 buffer[0] = i_socks_version; /* Version */
484 buffer[1] = 0x01; /* Cmd: connect */
485 buffer[2] = 0x00; /* Reserved */
486 buffer[3] = 3; /* ATYP: for now domainname */
487
488 buffer[4] = i_hlen;
489 memcpy( &buffer[5], psz_host, i_hlen );
490 SetWBE( &buffer[5+i_hlen], i_port );
491
492 i_len = 5 + i_hlen + 2;
493
494
495 if( net_Write( p_obj, fd, NULL, buffer, i_len ) != i_len )
496 return VLC_EGENERIC;
497
498 /* Read the header */
499 if( net_Read( p_obj, fd, NULL, buffer, 5, true ) != 5 )
500 return VLC_EGENERIC;
501
502 msg_Dbg( p_obj, "socks: v=%d rep=%d atyp=%d",
503 buffer[0], buffer[1], buffer[3] );
504
505 if( buffer[1] != 0x00 )
506 {
507 msg_Err( p_obj, "socks: CONNECT request failed" );
508 return VLC_EGENERIC;
509 }
510
511 /* Read the remaining bytes */
512 if( buffer[3] == 0x01 )
513 i_len = 4-1 + 2;
514 else if( buffer[3] == 0x03 )
515 i_len = buffer[4] + 2;
516 else if( buffer[3] == 0x04 )
517 i_len = 16-1+2;
518 else
519 return VLC_EGENERIC;
520
521 if( net_Read( p_obj, fd, NULL, buffer, i_len, true ) != i_len )
522 return VLC_EGENERIC;
523 }
524
525 return VLC_SUCCESS;
526}

528void net_ListenClose( int *pi_fd )
529{
530 if( pi_fd != NULL )
531 {
532 int *pi;
533
534 for( pi = pi_fd; *pi != -1; pi++ )
535 net_Close( *pi );
536 free( pi_fd );
537 }
