80static int ParseManifest( addons_finder_t *p_finder, addon_entry_t *p_entry,
81 const char *psz_tempfileuri, stream_t *p_stream )
82{
83 int i_num_entries_created = 0;
84 const char *p_node;
85 int i_current_node_type;
86
87 /* attr */
88 const char *attr, *value;
89
90 /* temp reading */
91 const char *psz_filename = NULL;
92 int i_filetype = -1;
93
94 xml_reader_t *p_xml_reader = xml_ReaderCreate( p_finder, p_stream );
95 if( !p_xml_reader ) return 0;
96
97 if( xml_ReaderNextNode( p_xml_reader, &p_node ) != XML_READER_STARTELEM )
98 {
99 msg_Err( p_finder, "invalid xml file" );
100 goto end;
101 }
102
103 if ( strcmp( p_node, "videolan") )
104 {
105 msg_Err( p_finder, "unsupported XML data format" );
106 goto end;
107 }
108
109 while( (i_current_node_type = xml_ReaderNextNode( p_xml_reader, &p_node )) > 0 )
110 {
111 switch( i_current_node_type )
112 {
113 case XML_READER_STARTELEM:
114 {
115 BINDNODE("resource", psz_filename, TYPE_STRING)
116 data_pointer.e_type = TYPE_NONE;
117
118 /*
119 * Manifests are not allowed to update addons properties
120 * such as uuid, score, downloads, ...
121 * On the other hand, repo API must not set files directly.
122 */
123
124 if ( ! strcmp( p_node, "resource" ) )
125 {
126 while( (attr = xml_ReaderNextAttr( p_xml_reader, &value )) )
127 {
128 if ( !strcmp( attr, "type" ) )
129 {
130 i_filetype = ReadType( value );
131 }
132 }
133 }
134 else if ( ! strcmp( p_node, "addon" ) )
135 {
136 while( (attr = xml_ReaderNextAttr( p_xml_reader, &value )) )
137 {
138 if ( !strcmp( attr, "type" ) )
139 {
140 p_entry->e_type = ReadType( value );
141 }
142 }
143 }
144
145 break;
146 }
147 case XML_READER_TEXT:
148 if ( data_pointer.e_type == TYPE_NONE || !p_entry ) break;
149 if ( data_pointer.e_type == TYPE_STRING )
150 *data_pointer.u_data.ppsz = strdup( p_node );
151 else
152 if ( data_pointer.e_type == TYPE_LONG )
153 *data_pointer.u_data.pl = atol( p_node );
154 else
155 if ( data_pointer.e_type == TYPE_INTEGER )
156 *data_pointer.u_data.pi = atoi( p_node );
157 break;
158
159 case XML_READER_ENDELEM:
160
161 if ( ! strcmp( p_node, "resource" ) )
162 {
163 if ( psz_filename && i_filetype >= 0 )
164 {
165 addon_file_t *p_file = malloc( sizeof(addon_file_t) );
166 p_file->e_filetype = i_filetype;
167 p_file->psz_filename = strdup( psz_filename );
168 if ( asprintf( & p_file->psz_download_uri, "%s!/%s",
169 psz_tempfileuri, psz_filename ) > 0 )
170 {
171 ARRAY_APPEND( p_entry->files, p_file );
172 msg_Dbg( p_finder, "manifest lists file %s extractable from %s",
173 psz_filename, p_file->psz_download_uri );
174 i_num_entries_created++;
175 }
176 else
177 {
178 free( p_file->psz_filename );
179 free( p_file );
180 }
181 }
182 /* reset temp */
183 psz_filename = NULL;
184 i_filetype = -1;
185 }
186
187 data_pointer.e_type = TYPE_NONE;
188 break;
189
190 default:
191 break;
192 }
193 }
194
195end:
196 xml_ReaderDelete( p_xml_reader );
197 return i_num_entries_created;
198}

200static int ParseCategoriesInfo( addons_finder_t *p_finder, stream_t *p_stream )
201{
202 int i_num_entries_created = 0;
203
204 const char *p_node;
205 const char *attr, *value;
206 int i_current_node_type;
207 addon_entry_t *p_entry = NULL;
208
209 xml_reader_t *p_xml_reader = xml_ReaderCreate( p_finder, p_stream );
210 if( !p_xml_reader ) return 0;
211
212 if( xml_ReaderNextNode( p_xml_reader, &p_node ) != XML_READER_STARTELEM )
213 {
214 msg_Err( p_finder, "invalid xml file" );
215 goto end;
216 }
217
218 if ( strcmp( p_node, "videolan") )
219 {
220 msg_Err( p_finder, "unsupported XML data format" );
221 goto end;
222 }
223
224 while( (i_current_node_type = xml_ReaderNextNode( p_xml_reader, &p_node )) > 0 )
225 {
226 switch( i_current_node_type )
227 {
228 case XML_READER_STARTELEM:
229 {
230 if ( ! strcmp( p_node, "addon" ) )
231 {
232 if ( p_entry ) /* Unclosed tag */
233 addon_entry_Release( p_entry );
234 p_entry = addon_entry_New();
235 p_entry->psz_source_module = strdup( ADDONS_MODULE_SHORTCUT );
236 p_entry->e_flags = ADDON_MANAGEABLE;
237 p_entry->e_state = ADDON_NOTINSTALLED;
238
239 while( (attr = xml_ReaderNextAttr( p_xml_reader, &value )) )
240 {
241 if ( !strcmp( attr, "type" ) )
242 {
243 p_entry->e_type = ReadType( value );
244 }
245 else if ( !strcmp( attr, "id" ) )
246 {
247 addons_uuid_read( value, & p_entry->uuid );
248 }
249 else if ( !strcmp( attr, "downloads" ) )
250 {
251 p_entry->i_downloads = atoi( value );
252 if ( p_entry->i_downloads < 0 )
253 p_entry->i_downloads = 0;
254 }
255 else if ( !strcmp( attr, "score" ) )
256 {
257 p_entry->i_score = atoi( value );
258 if ( p_entry->i_score < 0 )
259 p_entry->i_score = 0;
260 else if ( p_entry->i_score > ADDON_MAX_SCORE )
261 p_entry->i_score = ADDON_MAX_SCORE;
262 }
263 else if ( !strcmp( attr, "version" ) )
264 {
265 p_entry->psz_version = strdup( value );
266 }
267 }
268
269 break;
270 }
271 if ( !p_entry ) break;
272
273 BINDNODE("name", p_entry->psz_name, TYPE_STRING)
274 BINDNODE("archive", p_entry->psz_archive_uri, TYPE_STRING)
275 BINDNODE("summary", p_entry->psz_summary, TYPE_STRING)
276 BINDNODE("description", p_entry->psz_description, TYPE_STRING)
277 BINDNODE("image", p_entry->psz_image_data, TYPE_STRING)
278 BINDNODE("creator", p_entry->psz_author, TYPE_STRING)
279 BINDNODE("sourceurl", p_entry->psz_source_uri, TYPE_STRING)
280 data_pointer.e_type = TYPE_NONE;
281
282 break;
283 }
284 case XML_READER_TEXT:
285 if ( data_pointer.e_type == TYPE_NONE || !p_entry ) break;
286 if ( data_pointer.e_type == TYPE_STRING )
287 *data_pointer.u_data.ppsz = strdup( p_node );
288 else
289 if ( data_pointer.e_type == TYPE_LONG )
290 *data_pointer.u_data.pl = atol( p_node );
291 else
292 if ( data_pointer.e_type == TYPE_INTEGER )
293 *data_pointer.u_data.pi = atoi( p_node );
294 break;
295
296 case XML_READER_ENDELEM:
297 if ( !p_entry ) break;
298 if ( ! strcmp( p_node, "addon" ) )
299 {
300 /* then append entry */
301 ARRAY_APPEND( p_finder->entries, p_entry );
302 p_entry = NULL;
303 i_num_entries_created++;
304 }
305
306 data_pointer.e_type = TYPE_NONE;
307 break;
308
309 default:
310 break;
311 }
312 }
313
314end:
315 if ( p_entry ) /* Unclosed tag */
316 addon_entry_Release( p_entry );
317 xml_ReaderDelete( p_xml_reader );
318 return i_num_entries_created;
319}

321static int Find( addons_finder_t *p_finder )
322{
323 bool b_done = false;
324
325 while ( !b_done )
326 {
327 char *psz_uri = NULL;
328
329 if ( ! asprintf( &psz_uri, ADDONS_REPO_SCHEMEHOST"/xml" ) ) return VLC_ENOMEM;
330 b_done = true;
331
332 stream_t *p_stream = stream_UrlNew( p_finder, psz_uri );
333 free( psz_uri );
334 if ( !p_stream ) return VLC_EGENERIC;
335
336 if ( ! ParseCategoriesInfo( p_finder, p_stream ) )
337 {
338 /* no more entries have been read: was last page or error */
339 b_done = true;
340 }
341
342 stream_Delete( p_stream );
343 }
344
345 return VLC_SUCCESS;
346}

348static int Retrieve( addons_finder_t *p_finder, addon_entry_t *p_entry )
349{
350 vlc_mutex_lock( &p_entry->lock );
351 if ( !p_entry->psz_archive_uri )
352 {
353 vlc_mutex_unlock( &p_entry->lock );
354 return VLC_EGENERIC;
355 }
356 char *psz_archive_uri = strdup( p_entry->psz_archive_uri );
357 vlc_mutex_unlock( &p_entry->lock );
358 if ( !psz_archive_uri )
359 return VLC_ENOMEM;
360
361 /* get archive and parse manifest */
362 stream_t *p_stream;
363
364 if ( psz_archive_uri[0] == '/' )
365 {
366 /* Relative path */
367 char *psz_uri;
368 if ( ! asprintf( &psz_uri, ADDONS_REPO_SCHEMEHOST"%s", psz_archive_uri ) )
369 {
370 free( psz_archive_uri );
371 return VLC_ENOMEM;
372 }
373 p_stream = stream_UrlNew( p_finder, psz_uri );
374 free( psz_uri );
375 }
376 else
377 {
378 p_stream = stream_UrlNew( p_finder, psz_archive_uri );
379 }
380
381 msg_Dbg( p_finder, "downloading archive %s", psz_archive_uri );
382 free ( psz_archive_uri );
383 if ( !p_stream ) return VLC_EGENERIC;
384
385 /* In case of pf_ reuse */
386 if ( p_finder->p_sys->psz_tempfile )
387 {
388 vlc_unlink( p_finder->p_sys->psz_tempfile );
389 FREENULL( p_finder->p_sys->psz_tempfile );
390 }
391
392 p_finder->p_sys->psz_tempfile = tempnam( NULL, "vlp" );
393 if ( !p_finder->p_sys->psz_tempfile )
394 {
395 msg_Err( p_finder, "Can't create temp storage file" );
396 stream_Delete( p_stream );
397 return VLC_EGENERIC;
398 }
399
400 FILE *p_destfile = vlc_fopen( p_finder->p_sys->psz_tempfile, "w" );
401 if( !p_destfile )
402 {
403 msg_Err( p_finder, "Failed to open addon temp storage file" );
404 FREENULL(p_finder->p_sys->psz_tempfile);
405 stream_Delete( p_stream );
406 return VLC_EGENERIC;
407 }
408
409 char buffer[1<<10];
410 int i_read = 0;
411 while ( ( i_read = stream_Read( p_stream, &buffer, 1<<10 ) ) )
412 {
413 if ( fwrite( &buffer, i_read, 1, p_destfile ) < 1 )
414 {
415 msg_Err( p_finder, "Failed to write to Addon file" );
416 fclose( p_destfile );
417 stream_Delete( p_stream );
418 return VLC_EGENERIC;
419 }
420 }
421 fclose( p_destfile );
422 stream_Delete( p_stream );
423
424 msg_Dbg( p_finder, "Reading manifest from %s", p_finder->p_sys->psz_tempfile );
425
426 char *psz_tempfileuri = vlc_path2uri( p_finder->p_sys->psz_tempfile, "unzip" );
427 if ( !psz_tempfileuri )
428 return VLC_ENOMEM;
429
430 char *psz_manifest_uri;
431 if ( asprintf( &psz_manifest_uri, "%s!/manifest.xml", psz_tempfileuri ) < 1 )
432 {
433 free( psz_tempfileuri );
434 return VLC_ENOMEM;
435 }
436
437 p_stream = stream_UrlNew( p_finder, psz_manifest_uri );
438 free( psz_manifest_uri );
439 if ( !p_stream )
440 {
441 free( psz_tempfileuri );
442 return VLC_EGENERIC;
443 }
444
445 vlc_mutex_lock( &p_entry->lock );
446 int i_ret = ( ParseManifest( p_finder, p_entry, psz_tempfileuri, p_stream ) > 0 )
447 ? VLC_SUCCESS : VLC_EGENERIC;
448 vlc_mutex_unlock( &p_entry->lock );
449 free( psz_tempfileuri );
450 stream_Delete( p_stream );
451
452 return i_ret;
453}

455static int FindDesignated( addons_finder_t *p_finder )
456{
457 char *psz_manifest;
458 const char *psz_path = p_finder->psz_uri + 7; // remove scheme
459
460 if ( asprintf( &psz_manifest, "unzip://%s!/manifest.xml",
461 psz_path ) < 1 )
462 return VLC_ENOMEM;
463
464 stream_t *p_stream = stream_UrlNew( p_finder, psz_manifest );
465 free( psz_manifest );
466 if ( !p_stream ) return VLC_EGENERIC;
467
468 if ( ParseCategoriesInfo( p_finder, p_stream ) )
469 {
470 /* Do archive uri fixup */
471 FOREACH_ARRAY( addon_entry_t *p_entry, p_finder->entries )
472 if ( likely( !p_entry->psz_archive_uri ) )
473 p_entry->psz_archive_uri = strdup( p_finder->psz_uri );
474 FOREACH_END()
475 }
476 else
477 {
478 stream_Delete( p_stream );
479 return VLC_EGENERIC;
480 }
481
482 stream_Delete( p_stream );
483
484 return VLC_SUCCESS;
485}

487static int Open(vlc_object_t *p_this)
488{
489 addons_finder_t *p_finder = (addons_finder_t*) p_this;
490
491 p_finder->p_sys = (addons_finder_sys_t*) malloc(sizeof(addons_finder_sys_t));
492 if ( !p_finder->p_sys )
493 return VLC_ENOMEM;
494 p_finder->p_sys->psz_tempfile = NULL;
495
496 if ( p_finder->psz_uri &&
497 strcmp( "repo://"ADDONS_MODULE_SHORTCUT, p_finder->psz_uri ) &&
498 memcmp( "repo://", p_finder->psz_uri, 8 ) )
499 return VLC_EGENERIC;
500
501 p_finder->pf_find = Find;
502 p_finder->pf_retrieve = Retrieve;
503
504 return VLC_SUCCESS;
505}
506
507static void Close(vlc_object_t *p_this)
508{
509 addons_finder_t *p_finder = (addons_finder_t*) p_this;
510 if ( p_finder->p_sys->psz_tempfile )
511 {
512 vlc_unlink( p_finder->p_sys->psz_tempfile );
513 free( p_finder->p_sys );
514 }
515}
516
517static int OpenDesignated(vlc_object_t *p_this)
518{
519 addons_finder_t *p_finder = (addons_finder_t*) p_this;
520 if ( !p_finder->psz_uri
521 || strncmp( "file://", p_finder->psz_uri, 7 )
522 || strncmp( ".vlp", p_finder->psz_uri + strlen( p_finder->psz_uri ) - 4, 4 )
523 )
524 return VLC_EGENERIC;
525
526 p_finder->pf_find = FindDesignated;
527 p_finder->pf_retrieve = Retrieve;
528
529 return VLC_SUCCESS;
530}
