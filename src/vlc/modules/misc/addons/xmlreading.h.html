21#define BINDNODE(name,target,type)\
22 if ( ! strcmp(p_node, name) )\
23 {\
24 if ( type == TYPE_STRING ) \
25 data_pointer.u_data.ppsz = (char**)&target;\
26 else if ( type == TYPE_LONG ) \
27 data_pointer.u_data.pl = (long*)&target;\
28 else \
29 data_pointer.u_data.pi = (int*)&target;\
30 data_pointer.e_type = type;\
31 } else
32
33struct
34{
35 union
36 {
37 char ** ppsz;
38 int * pi;
39 long * pl;
40 } u_data;
41 enum
42 {
43 TYPE_NONE, TYPE_STRING, TYPE_INTEGER, TYPE_LONG
44 } e_type;
45} data_pointer = { {NULL}, TYPE_NONE };
46
47
48static inline int ReadType( const char *value )
49{
50 if ( !strcmp( value, "playlist" ) )
51 return ADDON_PLAYLIST_PARSER;
52 else if ( !strcmp( value, "skin" ) )
53 return ADDON_SKIN2;
54 else if ( !strcmp( value, "discovery" ) )
55 return ADDON_SERVICE_DISCOVERY;
56 else if ( !strcmp( value, "extension" ) )
57 return ADDON_EXTENSION;
58 else
59 return ADDON_UNKNOWN;
60}
61
62static inline const char * getTypePsz( int i_type )
63{
64 switch( i_type )
65 {
66 case ADDON_PLAYLIST_PARSER:
67 return "playlist";
68 case ADDON_SKIN2:
69 return "skin";
70 case ADDON_SERVICE_DISCOVERY:
71 return "discovery";
72 case ADDON_EXTENSION:
73 return "extension";
74 case ADDON_UNKNOWN:
75 default:
76 return "unknown";
77 }
78}
