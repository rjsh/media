47/*****************************************************************************
48 * Export_M3U: main export function
49 *****************************************************************************/
50static void DoChildren( playlist_export_t *p_export, playlist_item_t *p_root,
51 int (*pf_fprintf) (FILE *, const char *, ...) )
52{
53 /* Write header */
54 fputs( "#EXTM3U\n", p_export->p_file );
55
56 /* Go through the playlist and add items */
57 for( int i = 0; i< p_root->i_children ; i++)
58 {
59 playlist_item_t *p_current = p_root->pp_children[i];
60 assert( p_current );
61
62 if( p_current->i_flags & PLAYLIST_SAVE_FLAG )
63 continue;
64
65 if( p_current->i_children >= 0 )
66 {
67 DoChildren( p_export, p_current, pf_fprintf );
68 continue;
69 }
70
71 /* General info */
72
73 char *psz_uri = input_item_GetURI( p_current->p_input );
74
75 assert( psz_uri );
76
77 char *psz_name = input_item_GetName( p_current->p_input );
78 if( psz_name && strcmp( psz_uri, psz_name ) )
79 {
80 char *psz_artist = input_item_GetArtist( p_current->p_input );
81 if( psz_artist == NULL ) psz_artist = strdup( "" );
82 mtime_t i_duration = input_item_GetDuration( p_current->p_input );
83 if( psz_artist && *psz_artist )
84 {
85 /* write EXTINF with artist */
86 pf_fprintf( p_export->p_file, "#EXTINF:%"PRIu64",%s - %s\n",
87 i_duration / CLOCK_FREQ, psz_artist, psz_name);
88 }
89 else
90 {
91 /* write EXTINF without artist */
92 pf_fprintf( p_export->p_file, "#EXTINF:%"PRIu64",%s\n",
93 i_duration / CLOCK_FREQ, psz_name);
94 }
95 free( psz_artist );
96 }
97 free( psz_name );
98
99 /* VLC specific options */
100 vlc_mutex_lock( &p_current->p_input->lock );
101 for( int j = 0; j < p_current->p_input->i_options; j++ )
102 {
103 pf_fprintf( p_export->p_file, "#EXTVLCOPT:%s\n",
104 p_current->p_input->ppsz_options[j][0] == ':' ?
105 p_current->p_input->ppsz_options[j] + 1 :
106 p_current->p_input->ppsz_options[j] );
107 }
108 vlc_mutex_unlock( &p_current->p_input->lock );
109
110 /* Stupid third party players don't understand file: URIs. */
111 char *psz_path = make_path( psz_uri );
112 if( psz_path != NULL )
113 {
114 free( psz_uri );
115 psz_uri = psz_path;
116 }
117 fprintf( p_export->p_file, "%s\n", psz_uri );
118 free( psz_uri );
119 }
120}
121
122int Export_M3U( vlc_object_t *p_this )
123{
124 playlist_export_t *p_export = (playlist_export_t *)p_this;
125
126 msg_Dbg( p_export, "saving using M3U format");
127
128 DoChildren( p_export, p_export->p_root, utf8_fprintf );
129 return VLC_SUCCESS;
130}
131
132int Export_M3U8( vlc_object_t *p_this )
133{
134 playlist_export_t *p_export = (playlist_export_t *)p_this;
135
136 msg_Dbg( p_export, "saving using M3U8 format");
137
138 DoChildren( p_export, p_export->p_root, fprintf );
139 return VLC_SUCCESS;
140}
