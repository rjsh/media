50typedef unsigned short json_uchar;

52static unsigned char hex_value (json_char c)
53{
54 if (c >= 'A' && c <= 'F')
55 return (c - 'A') + 10;
56
57 if (c >= 'a' && c <= 'f')
58 return (c - 'a') + 10;
59
60 if (c >= '0' && c <= '9')
61 return c - '0';
62
63 return 0xFF;
64}

66typedef struct
67{
68 json_settings settings;
69 int first_pass;
70
71 unsigned long used_memory;
72
73 unsigned int uint_max;
74 unsigned long ulong_max;
75
76} json_state;
77
78static void * json_alloc (json_state * state, unsigned long size, int zero)
79{
80 void * mem;
81
82 if ((state->ulong_max - state->used_memory) < size)
83 return 0;
84
85 if (state->settings.max_memory
86 && (state->used_memory += size) > state->settings.max_memory)
87 {
88 return 0;
89 }
90
91 if (! (mem = zero ? calloc (size, 1) : malloc (size)))
92 return 0;
93
94 return mem;
95}

97static int new_value
98 (json_state * state, json_value ** top, json_value ** root, json_value ** alloc, json_type type)
99{
100 json_value * value;
101 int values_size;
102
103 if (!state->first_pass)
104 {
105 value = *top = *alloc;
106 *alloc = (*alloc)->_reserved.next_alloc;
107
108 if (!*root)
109 *root = value;
110
111 switch (value->type)
112 {
113 case json_array:
114
115 if (! (value->u.array.values = (json_value **) json_alloc
116 (state, value->u.array.length * sizeof (json_value *), 0)) )
117 {
118 return 0;
119 }
120
121 value->u.array.length = 0;
122 break;
123
124 case json_object:
125
126 values_size = sizeof (*value->u.object.values) * value->u.object.length;
127
128 if (! ((*(void **) &value->u.object.values) = json_alloc
129 (state, values_size + ((unsigned long) value->u.object.values), 0)) )
130 {
131 return 0;
132 }
133
134 value->_reserved.object_mem = (*(char **) &value->u.object.values) + values_size;
135
136 value->u.object.length = 0;
137 break;
138
139 case json_string:
140
141 if (! (value->u.string.ptr = (json_char *) json_alloc
142 (state, (value->u.string.length + 1) * sizeof (json_char), 0)) )
143 {
144 return 0;
145 }
146
147 value->u.string.length = 0;
148 break;
149
150 default:
151 break;
152 };
153
154 return 1;
155 }
156
157 value = (json_value *) json_alloc (state, sizeof (json_value), 1);
158
159 if (!value)
160 return 0;
161
162 if (!*root)
163 *root = value;
164
165 value->type = type;
166 value->parent = *top;
167
168 if (*alloc)
169 (*alloc)->_reserved.next_alloc = value;
170
171 *alloc = *top = value;
172
173 return 1;
174}

176#define e_off \
177 ((int) (i - cur_line_begin))
178
179#define whitespace \
180 case '\n': ++ cur_line; cur_line_begin = i; \
181 case ' ': case '\t': case '\r'
182
183#define string_add(b) \
184 do { if (!state.first_pass) string [string_length] = b; ++ string_length; } while (0);
185
186static const long
187 flag_next = 1, flag_reproc = 2, flag_need_comma = 4, flag_seek_value = 8,
188 flag_escaped = 16, flag_string = 32, flag_need_colon = 64, flag_done = 128,
189 flag_num_negative = 256, flag_num_zero = 512, flag_num_e = 1024,
190 flag_num_e_got_sign = 2048, flag_num_e_negative = 4096;
191
192json_value * json_parse_ex (json_settings * settings, const json_char * json, char * error_buf)
193{
194 json_char error [128];
195 unsigned int cur_line;
196 const json_char * cur_line_begin, * i;
197 json_value * top, * root, * alloc = 0;
198 json_state state;
199 long flags;
200 long num_digits = 0, num_fraction = 0, num_e = 0;
201
202 error[0] = '\0';
203
204 memset (&state, 0, sizeof (json_state));
205 memcpy (&state.settings, settings, sizeof (json_settings));
206
207 memset (&state.uint_max, 0xFF, sizeof (state.uint_max));
208 memset (&state.ulong_max, 0xFF, sizeof (state.ulong_max));
209
210 state.uint_max -= 8; /* limit of how much can be added before next check */
211 state.ulong_max -= 8;
212
213 for (state.first_pass = 1; state.first_pass >= 0; -- state.first_pass)
214 {
215 json_uchar uchar;
216 unsigned char uc_b1, uc_b2, uc_b3, uc_b4;
217 json_char * string = NULL;
218 unsigned int string_length = 0;
219
220 top = root = 0;
221 flags = flag_seek_value;
222
223 cur_line = 1;
224 cur_line_begin = json;
225
226 for (i = json ;; ++ i)
227 {
228 json_char b = *i;
229
230 if (flags & flag_done)
231 {
232 if (!b)
233 break;
234
235 switch (b)
236 {
237 whitespace:
238 continue;
239
240 default:
241 sprintf (error, "%d:%d: Trailing garbage: `%c`", cur_line, e_off, b);
242 goto e_failed;
243 };
244 }
245
246 if (flags & flag_string)
247 {
248 if (!b)
249 { sprintf (error, "Unexpected EOF in string (at %d:%d)", cur_line, e_off);
250 goto e_failed;
251 }
252
253 if (string_length > state.uint_max)
254 goto e_overflow;
255
256 if (flags & flag_escaped)
257 {
258 flags &= ~ flag_escaped;
259
260 switch (b)
261 {
262 case 'b': string_add ('\b'); break;
263 case 'f': string_add ('\f'); break;
264 case 'n': string_add ('\n'); break;
265 case 'r': string_add ('\r'); break;
266 case 't': string_add ('\t'); break;
267 case 'u':
268
269 if ((uc_b1 = hex_value (*++ i)) == 0xFF || (uc_b2 = hex_value (*++ i)) == 0xFF
270 || (uc_b3 = hex_value (*++ i)) == 0xFF || (uc_b4 = hex_value (*++ i)) == 0xFF)
271 {
272 sprintf (error, "Invalid character value `%c` (at %d:%d)", b, cur_line, e_off);
273 goto e_failed;
274 }
275
276 uc_b1 = uc_b1 * 16 + uc_b2;
277 uc_b2 = uc_b3 * 16 + uc_b4;
278
279 uchar = ((json_char) uc_b1) * 256 + uc_b2;
280
281 if (sizeof (json_char) >= sizeof (json_uchar) || (uc_b1 == 0 && uc_b2 <= 0x7F))
282 {
283 string_add ((json_char) uchar);
284 break;
285 }
286
287 if (uchar <= 0x7FF)
288 {
289 if (state.first_pass)
290 string_length += 2;
291 else
292 { string [string_length ++] = 0xC0 | ((uc_b2 & 0xC0) >> 6) | ((uc_b1 & 0x7) << 2);
293 string [string_length ++] = 0x80 | (uc_b2 & 0x3F);
294 }
295
296 break;
297 }
298
299 if (state.first_pass)
300 string_length += 3;
301 else
302 { string [string_length ++] = 0xE0 | ((uc_b1 & 0xF0) >> 4);
303 string [string_length ++] = 0x80 | ((uc_b1 & 0xF) << 2) | ((uc_b2 & 0xC0) >> 6);
304 string [string_length ++] = 0x80 | (uc_b2 & 0x3F);
305 }
306
307 break;
308
309 default:
310 string_add (b);
311 };
312
313 continue;
314 }
315
316 if (b == '\\')
317 {
318 flags |= flag_escaped;
319 continue;
320 }
321
322 if (b == '"')
323 {
324 if (!state.first_pass)
325 string [string_length] = 0;
326
327 flags &= ~ flag_string;
328 string = 0;
329
330 switch (top->type)
331 {
332 case json_string:
333
334 top->u.string.length = string_length;
335 flags |= flag_next;
336
337 break;
338
339 case json_object:
340
341 if (state.first_pass)
342 (*(json_char **) &top->u.object.values) += string_length + 1;
343 else
344 {
345 top->u.object.values [top->u.object.length].name
346 = (json_char *) top->_reserved.object_mem;
347
348 (*(json_char **) &top->_reserved.object_mem) += string_length + 1;
349 }
350
351 flags |= flag_seek_value | flag_need_colon;
352 continue;
353
354 default:
355 break;
356 };
357 }
358 else
359 {
360 string_add (b);
361 continue;
362 }
363 }
364
365 if (flags & flag_seek_value)
366 {
367 switch (b)
368 {
369 whitespace:
370 continue;
371
372 case ']':
373
374 if (top->type == json_array)
375 flags = (flags & ~ (flag_need_comma | flag_seek_value)) | flag_next;
376 else if (!state.settings.settings & json_relaxed_commas)
377 { sprintf (error, "%d:%d: Unexpected ]", cur_line, e_off);
378 goto e_failed;
379 }
380
381 break;
382
383 default:
384
385 if (flags & flag_need_comma)
386 {
387 if (b == ',')
388 { flags &= ~ flag_need_comma;
389 continue;
390 }
391 else
392 { sprintf (error, "%d:%d: Expected , before %c", cur_line, e_off, b);
393 goto e_failed;
394 }
395 }
396
397 if (flags & flag_need_colon)
398 {
399 if (b == ':')
400 { flags &= ~ flag_need_colon;
401 continue;
402 }
403 else
404 { sprintf (error, "%d:%d: Expected : before %c", cur_line, e_off, b);
405 goto e_failed;
406 }
407 }
408
409 flags &= ~ flag_seek_value;
410
411 switch (b)
412 {
413 case '{':
414
415 if (!new_value (&state, &top, &root, &alloc, json_object))
416 goto e_alloc_failure;
417
418 continue;
419
420 case '[':
421
422 if (!new_value (&state, &top, &root, &alloc, json_array))
423 goto e_alloc_failure;
424
425 flags |= flag_seek_value;
426 continue;
427
428 case '"':
429
430 if (!new_value (&state, &top, &root, &alloc, json_string))
431 goto e_alloc_failure;
432
433 flags |= flag_string;
434
435 string = top->u.string.ptr;
436 string_length = 0;
437
438 continue;
439
440 case 't':
441
442 if (*(++ i) != 'r' || *(++ i) != 'u' || *(++ i) != 'e')
443 goto e_unknown_value;
444
445 if (!new_value (&state, &top, &root, &alloc, json_boolean))
446 goto e_alloc_failure;
447
448 top->u.boolean = 1;
449
450 flags |= flag_next;
451 break;
452
453 case 'f':
454
455 if (*(++ i) != 'a' || *(++ i) != 'l' || *(++ i) != 's' || *(++ i) != 'e')
456 goto e_unknown_value;
457
458 if (!new_value (&state, &top, &root, &alloc, json_boolean))
459 goto e_alloc_failure;
460
461 flags |= flag_next;
462 break;
463
464 case 'n':
465
466 if (*(++ i) != 'u' || *(++ i) != 'l' || *(++ i) != 'l')
467 goto e_unknown_value;
468
469 if (!new_value (&state, &top, &root, &alloc, json_null))
470 goto e_alloc_failure;
471
472 flags |= flag_next;
473 break;
474
475 default:
476
477 if (isdigit (b) || b == '-')
478 {
479 if (!new_value (&state, &top, &root, &alloc, json_integer))
480 goto e_alloc_failure;
481
482 if (!state.first_pass)
483 {
484 while (isdigit (b) || b == '+' || b == '-'
485 || b == 'e' || b == 'E' || b == '.')
486 {
487 b = *++ i;
488 }
489
490 flags |= flag_next | flag_reproc;
491 break;
492 }
493
494 flags &= ~ (flag_num_negative | flag_num_e |
495 flag_num_e_got_sign | flag_num_e_negative |
496 flag_num_zero);
497
498 num_digits = 0;
499 num_fraction = 0;
500 num_e = 0;
501
502 if (b != '-')
503 {
504 flags |= flag_reproc;
505 break;
506 }
507
508 flags |= flag_num_negative;
509 continue;
510 }
511 else
512 { sprintf (error, "%d:%d: Unexpected %c when seeking value", cur_line, e_off, b);
513 goto e_failed;
514 }
515 };
516 };
517 }
518 else
519 {
520 switch (top->type)
521 {
522 case json_object:
523
524 switch (b)
525 {
526 whitespace:
527 continue;
528
529 case '"':
530
531 if (flags & flag_need_comma && (!state.settings.settings & json_relaxed_commas))
532 {
533 sprintf (error, "%d:%d: Expected , before \"", cur_line, e_off);
534 goto e_failed;
535 }
536
537 flags |= flag_string;
538
539 string = (json_char *) top->_reserved.object_mem;
540 string_length = 0;
541
542 break;
543
544 case '}':
545
546 flags = (flags & ~ flag_need_comma) | flag_next;
547 break;
548
549 case ',':
550
551 if (flags & flag_need_comma)
552 {
553 flags &= ~ flag_need_comma;
554 break;
555 }
556
557 default:
558
559 sprintf (error, "%d:%d: Unexpected `%c` in object", cur_line, e_off, b);
560 goto e_failed;
561 };
562
563 break;
564
565 case json_integer:
566 case json_double:
567
568 if (isdigit (b))
569 {
570 ++ num_digits;
571
572 if (top->type == json_integer || flags & flag_num_e)
573 {
574 if (! (flags & flag_num_e))
575 {
576 if (flags & flag_num_zero)
577 { sprintf (error, "%d:%d: Unexpected `0` before `%c`", cur_line, e_off, b);
578 goto e_failed;
579 }
580
581 if (num_digits == 1 && b == '0')
582 flags |= flag_num_zero;
583 }
584 else
585 {
586 flags |= flag_num_e_got_sign;
587 num_e = (num_e * 10) + (b - '0');
588 continue;
589 }
590
591 top->u.integer = (top->u.integer * 10) + (b - '0');
592 continue;
593 }
594
595 num_fraction = (num_fraction * 10) + (b - '0');
596 continue;
597 }
598
599 if (b == '+' || b == '-')
600 {
601 if ( (flags & flag_num_e) && !(flags & flag_num_e_got_sign))
602 {
603 flags |= flag_num_e_got_sign;
604
605 if (b == '-')
606 flags |= flag_num_e_negative;
607
608 continue;
609 }
610 }
611 else if (b == '.' && top->type == json_integer)
612 {
613 if (!num_digits)
614 { sprintf (error, "%d:%d: Expected digit before `.`", cur_line, e_off);
615 goto e_failed;
616 }
617
618 top->type = json_double;
619 top->u.dbl = (double) top->u.integer;
620
621 num_digits = 0;
622 continue;
623 }
624
625 if (! (flags & flag_num_e))
626 {
627 if (top->type == json_double)
628 {
629 if (!num_digits)
630 { sprintf (error, "%d:%d: Expected digit after `.`", cur_line, e_off);
631 goto e_failed;
632 }
633
634 top->u.dbl += ((double) num_fraction) / (pow ( (double) 10.0, (double) num_digits));
635 }
636
637 if (b == 'e' || b == 'E')
638 {
639 flags |= flag_num_e;
640
641 if (top->type == json_integer)
642 {
643 top->type = json_double;
644 top->u.dbl = (double) top->u.integer;
645 }
646
647 num_digits = 0;
648 flags &= ~ flag_num_zero;
649
650 continue;
651 }
652 }
653 else
654 {
655 if (!num_digits)
656 { sprintf (error, "%d:%d: Expected digit after `e`", cur_line, e_off);
657 goto e_failed;
658 }
659
660 top->u.dbl *= pow (10, (double) (flags & flag_num_e_negative ? - num_e : num_e));
661 }
662
663 if (flags & flag_num_negative)
664 {
665 if (top->type == json_integer)
666 top->u.integer = - top->u.integer;
667 else
668 top->u.dbl = - top->u.dbl;
669 }
670
671 flags |= flag_next | flag_reproc;
672 break;
673
674 default:
675 break;
676 };
677 }
678
679 if (flags & flag_reproc)
680 {
681 flags &= ~ flag_reproc;
682 -- i;
683 }
684
685 if (flags & flag_next)
686 {
687 flags = (flags & ~ flag_next) | flag_need_comma;
688
689 if (!top->parent)
690 {
691 /* root value done */
692
693 flags |= flag_done;
694 continue;
695 }
696
697 if (top->parent->type == json_array)
698 flags |= flag_seek_value;
699
700 if (!state.first_pass)
701 {
702 json_value * parent = top->parent;
703
704 switch (parent->type)
705 {
706 case json_object:
707
708 parent->u.object.values
709 [parent->u.object.length].value = top;
710
711 break;
712
713 case json_array:
714
715 parent->u.array.values
716 [parent->u.array.length] = top;
717
718 break;
719
720 default:
721 break;
722 };
723 }
724
725 if ( (++ top->parent->u.array.length) > state.uint_max)
726 goto e_overflow;
727
728 top = top->parent;
729
730 continue;
731 }
732 }
733
734 alloc = root;
735 }
736
737 return root;
738
739e_unknown_value:
740
741 sprintf (error, "%d:%d: Unknown value", cur_line, e_off);
742 goto e_failed;
743
744e_alloc_failure:
745
746 strcpy (error, "Memory allocation failure");
747 goto e_failed;
748
749e_overflow:
750
751 sprintf (error, "%d:%d: Too long (caught overflow)", cur_line, e_off);
752 goto e_failed;
753
754e_failed:
755
756 if (error_buf)
757 {
758 if (*error)
759 strcpy (error_buf, error);
760 else
761 strcpy (error_buf, "Unknown error");
762 }
763
764 if (state.first_pass)
765 alloc = root;
766
767 while (alloc)
768 {
769 top = alloc->_reserved.next_alloc;
770 free (alloc);
771 alloc = top;
772 }
773
774 if (!state.first_pass)
775 json_value_free (root);
776
777 return 0;
778}
779
780json_value * json_parse (const json_char * json)
781{
782 json_settings settings;
783 memset (&settings, 0, sizeof (json_settings));
784
785 return json_parse_ex (&settings, json, 0);
786}

788void json_value_free (json_value * value)
789{
790 json_value * cur_value;
791
792 if (!value)
793 return;
794
795 value->parent = 0;
796
797 while (value)
798 {
799 switch (value->type)
800 {
801 case json_array:
802
803 if (!value->u.array.length)
804 {
805 free (value->u.array.values);
806 break;
807 }
808
809 value = value->u.array.values [-- value->u.array.length];
810 continue;
811
812 case json_object:
813
814 if (!value->u.object.length)
815 {
816 free (value->u.object.values);
817 break;
818 }
819
820 value = value->u.object.values [-- value->u.object.length].value;
821 continue;
822
823 case json_string:
824
825 free (value->u.string.ptr);
826 break;
827
828 default:
829 break;
830 };
831
832 cur_value = value;
833 value = value->parent;
834 free (cur_value);
835 }
