39enum vlc_inhibit_api
40{
41 FDO_SS, /**< KDE >= 4 and GNOME >= 3.10 */
42 FDO_PM, /**< KDE and GNOME <= 2.26 */
43 MATE, /**< >= 1.0 */
44 GNOME, /**< GNOME 2.26..3.4 */
45};
46
47#define MAX_API (GNOME+1)
48
49/* Currently, all services have identical service and interface names. */
50static const char dbus_service[][40] =
51{
52 [FDO_SS] = "org.freedesktop.ScreenSaver",
53 [FDO_PM] = "org.freedesktop.PowerManagement.Inhibit",
54 [MATE] = "org.mate.SessionManager",
55 [GNOME] = "org.gnome.SessionManager",
56};

58static const char dbus_path[][33] =
59{
60 [FDO_SS] = "/ScreenSaver",
61 [FDO_PM] = "/org/freedesktop/PowerManagement",
62 [MATE] = "/org/mate/SessionManager",
63 [GNOME] = "/org/gnome/SessionManager",
64};

66static const char dbus_method_uninhibit[][10] =
67{
68 [FDO_SS] = "UnInhibit",
69 [FDO_PM] = "UnInhibit",
70 [MATE] = "Uninhibit",
71 [GNOME] = "Uninhibit",
72};

74struct vlc_inhibit_sys
75{
76 DBusConnection *conn;
77 DBusPendingCall *pending;
78 dbus_uint32_t cookie;
79 enum vlc_inhibit_api api;
80};
81
82static void Inhibit(vlc_inhibit_t *ih, unsigned flags)
83{
84 vlc_inhibit_sys_t *sys = ih->p_sys;
85 enum vlc_inhibit_api type = sys->api;
86
87 /* Receive reply from previous request, possibly hours later ;-) */
88 if (sys->pending != NULL)
89 {
90 DBusMessage *reply;
91
92 /* NOTE: Unfortunately, the pending reply cannot simply be cancelled.
93 * Otherwise, the cookie would be lost and inhibition would remain on
94 * (until complete disconnection from the bus). */
95 dbus_pending_call_block(sys->pending);
96 reply = dbus_pending_call_steal_reply(sys->pending);
97 dbus_pending_call_unref(sys->pending);
98 sys->pending = NULL;
99
100 if (reply != NULL)
101 {
102 if (!dbus_message_get_args(reply, NULL,
103 DBUS_TYPE_UINT32, &sys->cookie,
104 DBUS_TYPE_INVALID))
105 sys->cookie = 0;
106 dbus_message_unref(reply);
107 }
108 msg_Dbg(ih, "got cookie %"PRIu32, (uint32_t)sys->cookie);
109 }
110
111 /* FIXME: This check is incorrect if flags change from one non-zero value
112 * to another one. But the D-Bus API cannot currently inhibit suspend
113 * independently from the screensaver. */
114 if (!sys->cookie == !flags)
115 return; /* nothing to do */
116
117 /* Send request */
118 const char *method = flags ? "Inhibit" : dbus_method_uninhibit[type];
119 dbus_bool_t ret;
120
121 DBusMessage *msg = dbus_message_new_method_call(dbus_service[type],
122 dbus_path[type], dbus_service[type], method);
123 if (unlikely(msg == NULL))
124 return;
125
126 if (flags)
127 {
128 const char *app = PACKAGE;
129 const char *reason = _("Playing some media.");
130
131 assert(sys->cookie == 0);
132
133 switch (type)
134 {
135 case MATE:
136 case GNOME:
137 {
138 dbus_uint32_t xid = 0; // FIXME ?
139 dbus_uint32_t gflags = 0xC;
140
141 ret = dbus_message_append_args(msg, DBUS_TYPE_STRING, &app,
142 DBUS_TYPE_UINT32, &xid,
143 DBUS_TYPE_STRING, &reason,
144 DBUS_TYPE_UINT32, &gflags,
145 DBUS_TYPE_INVALID);
146 break;
147 }
148 default:
149 ret = dbus_message_append_args(msg, DBUS_TYPE_STRING, &app,
150 DBUS_TYPE_STRING, &reason,
151 DBUS_TYPE_INVALID);
152 break;
153 }
154
155 if (!ret
156 || !dbus_connection_send_with_reply(sys->conn, msg, &sys->pending, -1))
157 sys->pending = NULL;
158 }
159 else
160 {
161 assert(sys->cookie != 0);
162 if (!dbus_message_append_args(msg, DBUS_TYPE_UINT32, &sys->cookie,
163 DBUS_TYPE_INVALID)
164 || !dbus_connection_send (sys->conn, msg, NULL))
165 sys->cookie = 0;
166 }
167 dbus_connection_flush(sys->conn);
168 dbus_message_unref(msg);
169}

171static void Close(vlc_object_t *obj);

173static int Open (vlc_object_t *obj)
174{
175 vlc_inhibit_t *ih = (vlc_inhibit_t *)obj;
176 vlc_inhibit_sys_t *sys = malloc (sizeof (*sys));
177 if (unlikely(sys == NULL))
178 return VLC_ENOMEM;
179
180 DBusError err;
181 dbus_error_init(&err);
182
183 sys->conn = dbus_bus_get_private (DBUS_BUS_SESSION, &err);
184 if (sys->conn == NULL)
185 {
186 msg_Err(ih, "cannot connect to session bus: %s", err.message);
187 dbus_error_free(&err);
188 free(sys);
189 return VLC_EGENERIC;
190 }
191
192 sys->pending = NULL;
193 sys->cookie = 0;
194 ih->p_sys = sys;
195
196 for (unsigned i = 0; i < MAX_API; i++)
197 {
198 if (dbus_bus_name_has_owner(sys->conn, dbus_service[i], NULL))
199 {
200 msg_Dbg(ih, "found service %s", dbus_service[i]);
201 sys->api = i;
202 ih->inhibit = Inhibit;
203 return VLC_SUCCESS;
204 }
205
206 msg_Dbg(ih, "cannot find service %s", dbus_service[i]);
207 }
208
209 Close(obj);
210 return VLC_EGENERIC;
211}
212
213static void Close (vlc_object_t *obj)
214{
215 vlc_inhibit_t *ih = (vlc_inhibit_t *)obj;
216 vlc_inhibit_sys_t *sys = ih->p_sys;
217
218 if (sys->pending != NULL)
219 {
220 dbus_pending_call_cancel(sys->pending);
221 dbus_pending_call_unref(sys->pending);
222 }
223 dbus_connection_close (sys->conn);
224 dbus_connection_unref (sys->conn);
225 free (sys);
226}
