101static const char *const ppsz_filter_options[] = {
102 "loops", "alpha", "base-image", "base-chroma", "blend-image",
103 "blend-chroma", NULL
104};
105
106/*****************************************************************************
107 * filter_sys_t: filter method descriptor
108 *****************************************************************************/
109struct filter_sys_t
110{
111 bool b_done;
112 int i_loops, i_alpha;
113
114 picture_t *p_base_image;
115 picture_t *p_blend_image;
116
117 vlc_fourcc_t i_base_chroma;
118 vlc_fourcc_t i_blend_chroma;
119};
120
121static int blendbench_LoadImage( vlc_object_t *p_this, picture_t **pp_pic,
122 vlc_fourcc_t i_chroma, char *psz_file, const char *psz_name )
123{
124 image_handler_t *p_image;
125 video_format_t fmt_in, fmt_out;
126
127 memset( &fmt_in, 0, sizeof(video_format_t) );
128 memset( &fmt_out, 0, sizeof(video_format_t) );
129
130 fmt_out.i_chroma = i_chroma;
131 p_image = image_HandlerCreate( p_this );
132 *pp_pic = image_ReadUrl( p_image, psz_file, &fmt_in, &fmt_out );
133 image_HandlerDelete( p_image );
134
135 if( *pp_pic == NULL )
136 {
137 msg_Err( p_this, "Unable to load %s image", psz_name );
138 return VLC_EGENERIC;
139 }
140
141 msg_Dbg( p_this, "%s image has dim %d x %d (Y plane)", psz_name,
142 (*pp_pic)->p[Y_PLANE].i_visible_pitch,
143 (*pp_pic)->p[Y_PLANE].i_visible_lines );
144
145 return VLC_SUCCESS;
146}

148/*****************************************************************************
149 * Create: allocates video thread output method
150 *****************************************************************************/
151static int Create( vlc_object_t *p_this )
152{
153 filter_t *p_filter = (filter_t *)p_this;
154 filter_sys_t *p_sys;
155 char *psz_temp, *psz_cmd;
156 int i_ret;
157
158 /* Allocate structure */
159 p_filter->p_sys = malloc( sizeof( filter_sys_t ) );
160 if( p_filter->p_sys == NULL )
161 return VLC_ENOMEM;
162
163 p_sys = p_filter->p_sys;
164 p_sys->b_done = false;
165
166 p_filter->pf_video_filter = Filter;
167
168 /* needed to get options passed in transcode using the
169 * adjust{name=value} syntax */
170 config_ChainParse( p_filter, CFG_PREFIX, ppsz_filter_options,
171 p_filter->p_cfg );
172
173 p_sys->i_loops = var_CreateGetIntegerCommand( p_filter,
174 CFG_PREFIX "loops" );
175 p_sys->i_alpha = var_CreateGetIntegerCommand( p_filter,
176 CFG_PREFIX "alpha" );
177
178 psz_temp = var_CreateGetStringCommand( p_filter, CFG_PREFIX "base-chroma" );
179 p_sys->i_base_chroma = VLC_FOURCC( psz_temp[0], psz_temp[1],
180 psz_temp[2], psz_temp[3] );
181 psz_cmd = var_CreateGetStringCommand( p_filter, CFG_PREFIX "base-image" );
182 i_ret = blendbench_LoadImage( p_this, &p_sys->p_base_image,
183 p_sys->i_base_chroma, psz_cmd, "Base" );
184 free( psz_temp );
185 free( psz_cmd );
186 if( i_ret != VLC_SUCCESS )
187 {
188 free( p_sys );
189 return i_ret;
190 }
191
192 psz_temp = var_CreateGetStringCommand( p_filter,
193 CFG_PREFIX "blend-chroma" );
194 p_sys->i_blend_chroma = VLC_FOURCC( psz_temp[0], psz_temp[1],
195 psz_temp[2], psz_temp[3] );
196 psz_cmd = var_CreateGetStringCommand( p_filter, CFG_PREFIX "blend-image" );
197 blendbench_LoadImage( p_this, &p_sys->p_blend_image, p_sys->i_blend_chroma,
198 psz_cmd, "Blend" );
199 free( psz_temp );
200 free( psz_cmd );
201
202 return VLC_SUCCESS;
203}

205/*****************************************************************************
206 * Destroy: destroy video thread output method
207 *****************************************************************************/
208static void Destroy( vlc_object_t *p_this )
209{
210 filter_t *p_filter = (filter_t *)p_this;
211 filter_sys_t *p_sys = p_filter->p_sys;
212
213 picture_Release( p_sys->p_base_image );
214 picture_Release( p_sys->p_blend_image );
215}
216
217/*****************************************************************************
218 * Render: displays previously rendered output
219 *****************************************************************************/
220static picture_t *Filter( filter_t *p_filter, picture_t *p_pic )
221{
222 filter_sys_t *p_sys = p_filter->p_sys;
223 filter_t *p_blend;
224
225 if( p_sys->b_done )
226 return p_pic;
227
228 p_blend = vlc_object_create( p_filter, sizeof(filter_t) );
229 if( !p_blend )
230 {
231 picture_Release( p_pic );
232 return NULL;
233 }
234 p_blend->fmt_out.video = p_sys->p_base_image->format;
235 p_blend->fmt_in.video = p_sys->p_blend_image->format;
236 p_blend->p_module = module_need( p_blend, "video blending", NULL, false );
237 if( !p_blend->p_module )
238 {
239 picture_Release( p_pic );
240 vlc_object_release( p_blend );
241 return NULL;
242 }
243
244 mtime_t time = mdate();
245 for( int i_iter = 0; i_iter < p_sys->i_loops; ++i_iter )
246 {
247 p_blend->pf_video_blend( p_blend,
248 p_sys->p_base_image, p_sys->p_blend_image,
249 0, 0, p_sys->i_alpha );
250 }
251 time = mdate() - time;
252
253 msg_Info( p_filter, "Blended %d images in %f sec", p_sys->i_loops,
254 time / 1000000.0f );
255 msg_Info( p_filter, "Speed is: %f images/second, %f pixels/second",
256 (float) p_sys->i_loops / time * 1000000,
257 (float) p_sys->i_loops / time * 1000000 *
258 p_sys->p_blend_image->p[Y_PLANE].i_visible_pitch *
259 p_sys->p_blend_image->p[Y_PLANE].i_visible_lines );
260
261 module_unneed( p_blend, p_blend->p_module );
262
263 vlc_object_release( p_blend );
264
265 p_sys->b_done = true;
266 return p_pic;
267}
