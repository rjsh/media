50/*****************************************************************************
51 * Module descriptor

53#define FACTOR_TEXT N_("Blur factor (1-127)")
54#define FACTOR_LONGTEXT N_("The degree of blurring from 1 to 127.")
55
56#define FILTER_PREFIX "blur-"
57
58vlc_module_begin ()
59 set_shortname( N_("Motion blur") )
60 set_description( N_("Motion blur filter") )
61 set_capability( "video filter2", 0 )
62 set_category( CAT_VIDEO )
63 set_subcategory( SUBCAT_VIDEO_VFILTER )
64
65 add_integer_with_range( FILTER_PREFIX "factor", 80, 1, 127,
66 FACTOR_TEXT, FACTOR_LONGTEXT, false )
67
68 add_shortcut( "blur" )
69
70 set_callbacks( Create, Destroy )
71vlc_module_end ()

73static const char *const ppsz_filter_options[] = {
74 "factor", NULL
75};

77/*****************************************************************************
78 * filter_sys_t
79 *****************************************************************************/
80struct filter_sys_t
81{
82 picture_t *p_tmp;
83 bool b_first;
84 atomic_int i_factor;
85};
86
87/*****************************************************************************
88 * Create
89 *****************************************************************************/
90static int Create( vlc_object_t *p_this )
91{
92 filter_t *p_filter = (filter_t *)p_this;
93
94 const vlc_chroma_description_t *p_chroma =
95 vlc_fourcc_GetChromaDescription( p_filter->fmt_in.video.i_chroma );
96 if( p_chroma == NULL || p_chroma->plane_count == 0 )
97 return VLC_EGENERIC;
98
99 /* Allocate structure */
100 p_filter->p_sys = malloc( sizeof( filter_sys_t ) );
101 if( p_filter->p_sys == NULL )
102 return VLC_ENOMEM;
103
104 p_filter->p_sys->p_tmp = picture_NewFromFormat( &p_filter->fmt_in.video );
105 if( !p_filter->p_sys->p_tmp )
106 {
107 free( p_filter->p_sys );
108 return VLC_ENOMEM;
109 }
110 p_filter->p_sys->b_first = true;
111
112 p_filter->pf_video_filter = Filter;
113
114 config_ChainParse( p_filter, FILTER_PREFIX, ppsz_filter_options,
115 p_filter->p_cfg );
116
117 atomic_init( &p_filter->p_sys->i_factor,
118 var_CreateGetIntegerCommand( p_filter, FILTER_PREFIX "factor" ) );
119 var_AddCallback( p_filter, FILTER_PREFIX "factor",
120 MotionBlurCallback, p_filter->p_sys );
121
122
123 return VLC_SUCCESS;
124}
125
126/*****************************************************************************
127 * Destroy
128 *****************************************************************************/
129static void Destroy( vlc_object_t *p_this )
130{
131 filter_t *p_filter = (filter_t *)p_this;
132
133 var_DelCallback( p_filter, FILTER_PREFIX "factor",
134 MotionBlurCallback, p_filter->p_sys );
135
136 picture_Release( p_filter->p_sys->p_tmp );
137 free( p_filter->p_sys );
138}
139
140/*****************************************************************************
141 * Filter
142 *****************************************************************************/
143static picture_t *Filter( filter_t *p_filter, picture_t *p_pic )
144{
145 picture_t * p_outpic;
146 filter_sys_t *p_sys = p_filter->p_sys;
147
148 if( !p_pic ) return NULL;
149
150 p_outpic = filter_NewPicture( p_filter );
151 if( !p_outpic )
152 {
153 picture_Release( p_pic );
154 return NULL;
155 }
156
157 if( p_sys->b_first )
158 {
159 picture_CopyPixels( p_sys->p_tmp, p_pic );
160 p_sys->b_first = false;
161 }
162
163 /* Get a new picture */
164 RenderBlur( p_sys, p_pic, p_outpic );
165
166 picture_CopyPixels( p_sys->p_tmp, p_outpic );
167
168 return CopyInfoAndRelease( p_outpic, p_pic );
169}
170
171/*****************************************************************************
172 * RenderBlur: renders a blurred picture
173 *****************************************************************************/
174static void RenderBlur( filter_sys_t *p_sys, picture_t *p_newpic,
175 picture_t *p_outpic )
176{
177 int i_plane;
178 const int i_oldfactor = atomic_load( &p_sys->i_factor );
179 int i_newfactor = 128 - i_oldfactor;
180
181 for( i_plane = 0; i_plane < p_outpic->i_planes; i_plane++ )
182 {
183 uint8_t *p_old, *p_new, *p_out, *p_out_end, *p_out_line_end;
184 const int i_visible_pitch = p_outpic->p[i_plane].i_visible_pitch;
185 const int i_visible_lines = p_outpic->p[i_plane].i_visible_lines;
186
187 p_out = p_outpic->p[i_plane].p_pixels;
188 p_new = p_newpic->p[i_plane].p_pixels;
189 p_old = p_sys->p_tmp->p[i_plane].p_pixels;
190 p_out_end = p_out + p_outpic->p[i_plane].i_pitch * i_visible_lines;
191 while ( p_out < p_out_end )
192 {
193 p_out_line_end = p_out + i_visible_pitch;
194
195 while ( p_out < p_out_line_end )
196 {
197 *p_out++ = (((*p_old++) * i_oldfactor) +
198 ((*p_new++) * i_newfactor)) >> 7;
199 }
200
201 p_old += p_sys->p_tmp->p[i_plane].i_pitch - i_visible_pitch;
202 p_new += p_newpic->p[i_plane].i_pitch - i_visible_pitch;
203 p_out += p_outpic->p[i_plane].i_pitch - i_visible_pitch;
204 }
205 }
206}
207
208static int MotionBlurCallback( vlc_object_t *p_this, char const *psz_var,
209 vlc_value_t oldval, vlc_value_t newval,
210 void *p_data )
211{
212 VLC_UNUSED(p_this); VLC_UNUSED(psz_var); VLC_UNUSED(oldval);
213 filter_sys_t *p_sys = (filter_sys_t *)p_data;
214
215 atomic_store( &p_sys->i_factor, VLC_CLIP( newval.i_int, 1, 127 ) );
216 return VLC_SUCCESS;
217}
