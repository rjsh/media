46/* XDeint8x8Detect: detect if a 8x8 block is interlaced.
47 * XXX: It need to access to 8x10
48 * We use more than 8 lines to help with scrolling (text)
49 * (and because XDeint8x8Frame use line 9)
50 * XXX: smooth/uniform area with noise detection doesn't works well
51 * but it's not really a problem because they don't have much details anyway
52 */
53static inline int ssd( int a ) { return a*a; }
54static inline int XDeint8x8DetectC( uint8_t *src, int i_src )
55{
56 int y, x;
57 int ff, fr;
58 int fc;
59
60 /* Detect interlacing */
61 fc = 0;
62 for( y = 0; y < 7; y += 2 )
63 {
64 ff = fr = 0;
65 for( x = 0; x < 8; x++ )
66 {
67 fr += ssd(src[ x] - src[1*i_src+x]) +
68 ssd(src[i_src+x] - src[2*i_src+x]);
69 ff += ssd(src[ x] - src[2*i_src+x]) +
70 ssd(src[i_src+x] - src[3*i_src+x]);
71 }
72 if( ff < 6*fr/8 && fr > 32 )
73 fc++;
74
75 src += 2*i_src;
76 }
77
78 return fc < 1 ? false : true;
79}
80#ifdef CAN_COMPILE_MMXEXT
81VLC_MMX
82static inline int XDeint8x8DetectMMXEXT( uint8_t *src, int i_src )
83{
84
85 int y, x;
86 int32_t ff, fr;
87 int fc;
88
89 /* Detect interlacing */
90 fc = 0;
91 pxor_r2r( mm7, mm7 );
92 for( y = 0; y < 9; y += 2 )
93 {
94 ff = fr = 0;
95 pxor_r2r( mm5, mm5 );
96 pxor_r2r( mm6, mm6 );
97 for( x = 0; x < 8; x+=4 )
98 {
99 movd_m2r( src[ x], mm0 );
100 movd_m2r( src[1*i_src+x], mm1 );
101 movd_m2r( src[2*i_src+x], mm2 );
102 movd_m2r( src[3*i_src+x], mm3 );
103
104 punpcklbw_r2r( mm7, mm0 );
105 punpcklbw_r2r( mm7, mm1 );
106 punpcklbw_r2r( mm7, mm2 );
107 punpcklbw_r2r( mm7, mm3 );
108
109 movq_r2r( mm0, mm4 );
110
111 psubw_r2r( mm1, mm0 );
112 psubw_r2r( mm2, mm4 );
113
114 psubw_r2r( mm1, mm2 );
115 psubw_r2r( mm1, mm3 );
116
117 pmaddwd_r2r( mm0, mm0 );
118 pmaddwd_r2r( mm4, mm4 );
119 pmaddwd_r2r( mm2, mm2 );
120 pmaddwd_r2r( mm3, mm3 );
121 paddd_r2r( mm0, mm2 );
122 paddd_r2r( mm4, mm3 );
123 paddd_r2r( mm2, mm5 );
124 paddd_r2r( mm3, mm6 );
125 }
126
127 movq_r2r( mm5, mm0 );
128 psrlq_i2r( 32, mm0 );
129 paddd_r2r( mm0, mm5 );
130 movd_r2m( mm5, fr );
131
132 movq_r2r( mm6, mm0 );
133 psrlq_i2r( 32, mm0 );
134 paddd_r2r( mm0, mm6 );
135 movd_r2m( mm6, ff );
136
137 if( ff < 6*fr/8 && fr > 32 )
138 fc++;
139
140 src += 2*i_src;
141 }
142 return fc;
143}

146static inline void XDeint8x8MergeC( uint8_t *dst, int i_dst,
147 uint8_t *src1, int i_src1,
148 uint8_t *src2, int i_src2 )
149{
150 int y, x;
151
152 /* Progressive */
153 for( y = 0; y < 8; y += 2 )
154 {
155 memcpy( dst, src1, 8 );
156 dst += i_dst;
157
158 for( x = 0; x < 8; x++ )
159 dst[x] = (src1[x] + 6*src2[x] + src1[i_src1+x] + 4 ) >> 3;
160 dst += i_dst;
161
162 src1 += i_src1;
163 src2 += i_src2;
164 }
165}

167#ifdef CAN_COMPILE_MMXEXT
168VLC_MMX
169static inline void XDeint8x8MergeMMXEXT( uint8_t *dst, int i_dst,
170 uint8_t *src1, int i_src1,
171 uint8_t *src2, int i_src2 )
172{
173 static const uint64_t m_4 = INT64_C(0x0004000400040004);
174 int y, x;
175
176 /* Progressive */
177 pxor_r2r( mm7, mm7 );
178 for( y = 0; y < 8; y += 2 )
179 {
180 for( x = 0; x < 8; x +=4 )
181 {
182 movd_m2r( src1[x], mm0 );
183 movd_r2m( mm0, dst[x] );
184
185 movd_m2r( src2[x], mm1 );
186 movd_m2r( src1[i_src1+x], mm2 );
187
188 punpcklbw_r2r( mm7, mm0 );
189 punpcklbw_r2r( mm7, mm1 );
190 punpcklbw_r2r( mm7, mm2 );
191 paddw_r2r( mm1, mm1 );
192 movq_r2r( mm1, mm3 );
193 paddw_r2r( mm3, mm3 );
194 paddw_r2r( mm2, mm0 );
195 paddw_r2r( mm3, mm1 );
196 paddw_m2r( m_4, mm1 );
197 paddw_r2r( mm1, mm0 );
198 psraw_i2r( 3, mm0 );
199 packuswb_r2r( mm7, mm0 );
200 movd_r2m( mm0, dst[i_dst+x] );
201 }
202 dst += 2*i_dst;
203 src1 += i_src1;
204 src2 += i_src2;
205 }
206}

218/* XDeint8x8FieldE: Stupid deinterlacing (1,0,1) for block that miss a
219 * neighbour
220 * (Use 8x9 pixels)
221 * TODO: a better one for the inner part.
222 */
223static inline void XDeint8x8FieldEC( uint8_t *dst, int i_dst,
224 uint8_t *src, int i_src )
225{
226 int y, x;
227
228 /* Interlaced */
229 for( y = 0; y < 8; y += 2 )
230 {
231 memcpy( dst, src, 8 );
232 dst += i_dst;
233
234 for( x = 0; x < 8; x++ )
235 dst[x] = (src[x] + src[2*i_src+x] ) >> 1;
236 dst += 1*i_dst;
237 src += 2*i_src;
238 }
239}

266/* XDeint8x8Field: Edge oriented interpolation
267 * (Need -4 and +5 pixels H, +1 line)
268 */
269static inline void XDeint8x8FieldC( uint8_t *dst, int i_dst,
270 uint8_t *src, int i_src )
271{
272 int y, x;
273
274 /* Interlaced */
275 for( y = 0; y < 8; y += 2 )
276 {
277 memcpy( dst, src, 8 );
278 dst += i_dst;
279
280 for( x = 0; x < 8; x++ )
281 {
282 uint8_t *src2 = &src[2*i_src];
283 /* I use 8 pixels just to match the MMX version, but it's overkill
284 * 5 would be enough (less isn't good) */
285 const int c0 = abs(src[x-4]-src2[x-2]) + abs(src[x-3]-src2[x-1]) +
286 abs(src[x-2]-src2[x+0]) + abs(src[x-1]-src2[x+1]) +
287 abs(src[x+0]-src2[x+2]) + abs(src[x+1]-src2[x+3]) +
288 abs(src[x+2]-src2[x+4]) + abs(src[x+3]-src2[x+5]);
289
290 const int c1 = abs(src[x-3]-src2[x-3]) + abs(src[x-2]-src2[x-2]) +
291 abs(src[x-1]-src2[x-1]) + abs(src[x+0]-src2[x+0]) +
292 abs(src[x+1]-src2[x+1]) + abs(src[x+2]-src2[x+2]) +
293 abs(src[x+3]-src2[x+3]) + abs(src[x+4]-src2[x+4]);
294
295 const int c2 = abs(src[x-2]-src2[x-4]) + abs(src[x-1]-src2[x-3]) +
296 abs(src[x+0]-src2[x-2]) + abs(src[x+1]-src2[x-1]) +
297 abs(src[x+2]-src2[x+0]) + abs(src[x+3]-src2[x+1]) +
298 abs(src[x+4]-src2[x+2]) + abs(src[x+5]-src2[x+3]);
299
300 if( c0 < c1 && c1 <= c2 )
301 dst[x] = (src[x-1] + src2[x+1]) >> 1;
302 else if( c2 < c1 && c1 <= c0 )
303 dst[x] = (src[x+1] + src2[x-1]) >> 1;
304 else
305 dst[x] = (src[x+0] + src2[x+0]) >> 1;
306 }
307
308 dst += 1*i_dst;
309 src += 2*i_src;
310 }
311}

313#ifdef CAN_COMPILE_MMXEXT
314VLC_MMX
315static inline void XDeint8x8FieldMMXEXT( uint8_t *dst, int i_dst,
316 uint8_t *src, int i_src )
317{
318 int y, x;
319
320 /* Interlaced */
321 for( y = 0; y < 8; y += 2 )
322 {
323 memcpy( dst, src, 8 );
324 dst += i_dst;
325
326 for( x = 0; x < 8; x++ )
327 {
328 uint8_t *src2 = &src[2*i_src];
329 int32_t c0, c1, c2;
330
331 movq_m2r( src[x-2], mm0 );
332 movq_m2r( src[x-3], mm1 );
333 movq_m2r( src[x-4], mm2 );
334
335 psadbw_m2r( src2[x-4], mm0 );
336 psadbw_m2r( src2[x-3], mm1 );
337 psadbw_m2r( src2[x-2], mm2 );
338
339 movd_r2m( mm0, c2 );
340 movd_r2m( mm1, c1 );
341 movd_r2m( mm2, c0 );
342
343 if( c0 < c1 && c1 <= c2 )
344 dst[x] = (src[x-1] + src2[x+1]) >> 1;
345 else if( c2 < c1 && c1 <= c0 )
346 dst[x] = (src[x+1] + src2[x-1]) >> 1;
347 else
348 dst[x] = (src[x+0] + src2[x+0]) >> 1;
349 }
350
351 dst += 1*i_dst;
352 src += 2*i_src;
353 }
354}
355#endif

357/* NxN arbitray size (and then only use pixel in the NxN block)
358 */
359static inline int XDeintNxNDetect( uint8_t *src, int i_src,
360 int i_height, int i_width )
361{
362 int y, x;
363 int ff, fr;
364 int fc;
365
366
367 /* Detect interlacing */
368 /* FIXME way too simple, need to be more like XDeint8x8Detect */
369 ff = fr = 0;
370 fc = 0;
371 for( y = 0; y < i_height - 2; y += 2 )
372 {
373 const uint8_t *s = &src[y*i_src];
374 for( x = 0; x < i_width; x++ )
375 {
376 fr += ssd(s[ x] - s[1*i_src+x]);
377 ff += ssd(s[ x] - s[2*i_src+x]);
378 }
379 if( ff < fr && fr > i_width / 2 )
380 fc++;
381 }
382
383 return fc < 2 ? false : true;
384}

386static inline void XDeintNxNFrame( uint8_t *dst, int i_dst,
387 uint8_t *src, int i_src,
388 int i_width, int i_height )
389{
390 int y, x;
391
392 /* Progressive */
393 for( y = 0; y < i_height; y += 2 )
394 {
395 memcpy( dst, src, i_width );
396 dst += i_dst;
397
398 if( y < i_height - 2 )
399 {
400 for( x = 0; x < i_width; x++ )
401 dst[x] = (src[x] + 2*src[1*i_src+x] + src[2*i_src+x] + 2 ) >> 2;
402 }
403 else
404 {
405 /* Blend last line */
406 for( x = 0; x < i_width; x++ )
407 dst[x] = (src[x] + src[1*i_src+x] ) >> 1;
408 }
409 dst += 1*i_dst;
410 src += 2*i_src;
411 }
412}

442static inline void XDeintNxN( uint8_t *dst, int i_dst, uint8_t *src, int i_src,
443 int i_width, int i_height )
444{
445 if( XDeintNxNDetect( src, i_src, i_width, i_height ) )
446 XDeintNxNField( dst, i_dst, src, i_src, i_width, i_height );
447 else
448 XDeintNxNFrame( dst, i_dst, src, i_src, i_width, i_height );
449}

451static inline int median( int a, int b, int c )
452{
453 int min = a, max =a;
454 if( b < min )
455 min = b;
456 else
457 max = b;
458
459 if( c < min )
460 min = c;
461 else if( c > max )
462 max = c;
463
464 return a + b + c - min - max;
465}

468/* XDeintBand8x8:
469 */
470static inline void XDeintBand8x8C( uint8_t *dst, int i_dst,
471 uint8_t *src, int i_src,
472 const int i_mbx, int i_modx )
473{
474 int x;
475
476 for( x = 0; x < i_mbx; x++ )
477 {
478 int s;
479 if( ( s = XDeint8x8DetectC( src, i_src ) ) )
480 {
481 if( x == 0 || x == i_mbx - 1 )
482 XDeint8x8FieldEC( dst, i_dst, src, i_src );
483 else
484 XDeint8x8FieldC( dst, i_dst, src, i_src );
485 }
486 else
487 {
488 XDeint8x8MergeC( dst, i_dst,
489 &src[0*i_src], 2*i_src,
490 &src[1*i_src], 2*i_src );
491 }
492
493 dst += 8;
494 src += 8;
495 }
496
497 if( i_modx )
498 XDeintNxN( dst, i_dst, src, i_src, i_modx, 8 );
499}

540void RenderX( picture_t *p_outpic, picture_t *p_pic )
541{
542 int i_plane;
543#if defined (CAN_COMPILE_MMXEXT)
544 const bool mmxext = vlc_CPU_MMXEXT();
545#endif
546
547 /* Copy image and skip lines */
548 for( i_plane = 0 ; i_plane < p_pic->i_planes ; i_plane++ )
549 {
550 const int i_mby = ( p_outpic->p[i_plane].i_visible_lines + 7 )/8 - 1;
551 const int i_mbx = p_outpic->p[i_plane].i_visible_pitch/8;
552
553 const int i_mody = p_outpic->p[i_plane].i_visible_lines - 8*i_mby;
554 const int i_modx = p_outpic->p[i_plane].i_visible_pitch - 8*i_mbx;
555
556 const int i_dst = p_outpic->p[i_plane].i_pitch;
557 const int i_src = p_pic->p[i_plane].i_pitch;
558
559 int y, x;
560
561 for( y = 0; y < i_mby; y++ )
562 {
563 uint8_t *dst = &p_outpic->p[i_plane].p_pixels[8*y*i_dst];
564 uint8_t *src = &p_pic->p[i_plane].p_pixels[8*y*i_src];
565
566#ifdef CAN_COMPILE_MMXEXT
567 if( mmxext )
568 XDeintBand8x8MMXEXT( dst, i_dst, src, i_src, i_mbx, i_modx );
569 else
570#endif
571 XDeintBand8x8C( dst, i_dst, src, i_src, i_mbx, i_modx );
572 }
573
574 /* Last line (C only)*/
575 if( i_mody )
576 {
577 uint8_t *dst = &p_outpic->p[i_plane].p_pixels[8*y*i_dst];
578 uint8_t *src = &p_pic->p[i_plane].p_pixels[8*y*i_src];
579
580 for( x = 0; x < i_mbx; x++ )
581 {
582 XDeintNxN( dst, i_dst, src, i_src, 8, i_mody );
583
584 dst += 8;
585 src += 8;
586 }
587
588 if( i_modx )
589 XDeintNxN( dst, i_dst, src, i_src, i_modx, i_mody );
590 }
591 }
592
593#ifdef CAN_COMPILE_MMXEXT
594 if( mmxext )
595 emms();
596#endif
597}
