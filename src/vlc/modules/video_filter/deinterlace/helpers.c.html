50/**
51 * This internal function converts a normal (full frame) plane_t into a
52 * field plane_t.
53 *
54 * Field plane_t's can be used e.g. for a weaving copy operation from two
55 * source frames into one destination frame.
56 *
57 * The pixels themselves will not be touched; only the metadata is generated.
58 * The same pixel data is shared by both the original plane_t and the field
59 * plane_t. Note, however, that the bottom field's data starts from the
60 * second line, so for the bottom field, the actual pixel pointer value
61 * does not exactly match the original plane pixel pointer value. (It points
62 * one line further down.)
63 *
64 * The caller must allocate p_dst (creating a local variable is fine).
65 *
66 * @param p_dst Field plane_t is written here. Must be non-NULL.
67 * @param p_src Original full-frame plane_t. Must be non-NULL.
68 * @param i_field Extract which field? 0 = top field, 1 = bottom field.
69 * @see plane_CopyPixels()
70 * @see ComposeFrame()
71 * @see RenderPhosphor()
72 */
73static void FieldFromPlane( plane_t *p_dst, const plane_t *p_src, int i_field )
74{
75 assert( p_dst != NULL );
76 assert( p_src != NULL );
77 assert( i_field == 0 || i_field == 1 );
78
79 /* Start with a copy of the metadata, and then update it to refer
80 to one field only.
81
82 We utilize the fact that plane_CopyPixels() differentiates between
83 visible_pitch and pitch.
84
85 The other field will be defined as the "margin" by doubling the pitch.
86 The visible pitch will be left as in the original.
87 */
88 (*p_dst) = (*p_src);
89 p_dst->i_lines /= 2;
90 p_dst->i_visible_lines /= 2;
91 p_dst->i_pitch *= 2;
92 /* For the bottom field, skip the first line in the pixel data. */
93 if( i_field == 1 )
94 p_dst->p_pixels += p_src->i_pitch;
95}

97#define T 10
98/**
99 * Internal helper function for EstimateNumBlocksWithMotion():
100 * estimates whether there is motion in the given 8x8 block on one plane
101 * between two images. The block as a whole and its fields are evaluated
102 * separately, and use different motion thresholds.
103 *
104 * This is a low-level function only used by EstimateNumBlocksWithMotion().
105 * There is no need to call this function manually.
106 *
107 * For interpretation of pi_top and pi_bot, it is assumed that the block
108 * starts on an even-numbered line (belonging to the top field).
109 *
110 * The b_mmx parameter avoids the need to call vlc_CPU() separately
111 * for each block.
112 *
113 * @param[in] p_pix_p Base pointer to the block in previous picture
114 * @param[in] p_pix_c Base pointer to the same block in current picture
115 * @param i_pitch_prev i_pitch of previous picture
116 * @param i_pitch_curr i_pitch of current picture
117 * @param[out] pi_top 1 if top field of the block had motion, 0 if no
118 * @param[out] pi_bot 1 if bottom field of the block had motion, 0 if no
119 * @return 1 if the block had motion, 0 if no
120 * @see EstimateNumBlocksWithMotion()
121 */
122static int TestForMotionInBlock( uint8_t *p_pix_p, uint8_t *p_pix_c,
123 int i_pitch_prev, int i_pitch_curr,
124 int* pi_top, int* pi_bot )
125{
126/* Pixel luma/chroma difference threshold to detect motion. */
127
128 int32_t i_motion = 0;
129 int32_t i_top_motion = 0;
130 int32_t i_bot_motion = 0;
131
132 for( int y = 0; y < 8; ++y )
133 {
134 uint8_t *pc = p_pix_c;
135 uint8_t *pp = p_pix_p;
136 int score = 0;
137 for( int x = 0; x < 8; ++x )
138 {
139 int_fast16_t C = abs((*pc) - (*pp));
140 if( C > T )
141 ++score;
142
143 ++pc;
144 ++pp;
145 }
146
147 i_motion += score;
148 if( y % 2 == 0 )
149 i_top_motion += score;
150 else
151 i_bot_motion += score;
152
153 p_pix_c += i_pitch_curr;
154 p_pix_p += i_pitch_prev;
155 }
156
157 /* Field motion thresholds.
158
159 Empirical value - works better in practice than the "4" that
160 would be consistent with the full-block threshold.
161
162 Especially the opening scene of The Third ep. 1 (just after the OP)
163 works better with this. It also fixes some talking scenes in
164 Stellvia ep. 1, where the cadence would otherwise catch on incorrectly,
165 leading to more interlacing artifacts than by just using the emergency
166 mode frame composer.
167 */
168 (*pi_top) = ( i_top_motion >= 8 );
169 (*pi_bot) = ( i_bot_motion >= 8 );
170
171 /* Full-block threshold = (8*8)/8: motion is detected if 1/8 of the block
172 changes "enough". */
173 return (i_motion >= 8);
174}

176#ifdef CAN_COMPILE_MMXEXT
177VLC_MMX
178static int TestForMotionInBlockMMX( uint8_t *p_pix_p, uint8_t *p_pix_c,
179 int i_pitch_prev, int i_pitch_curr,
180 int* pi_top, int* pi_bot )
181{
182 int32_t i_motion = 0;
183 int32_t i_top_motion = 0;
184 int32_t i_bot_motion = 0;
185
186 static const mmx_t bT = { .ub = { T, T, T, T, T, T, T, T } };
187 pxor_r2r( mm6, mm6 ); /* zero, used in psadbw */
188 movq_m2r( bT, mm5 );
189
190 pxor_r2r( mm3, mm3 ); /* score (top field) */
191 pxor_r2r( mm4, mm4 ); /* score (bottom field) */
192 for( int y = 0; y < 8; y+=2 )
193 {
194 /* top field */
195 movq_m2r( *((uint64_t*)p_pix_c), mm0 );
196 movq_m2r( *((uint64_t*)p_pix_p), mm1 );
197 movq_r2r( mm0, mm2 );
198 psubusb_r2r( mm1, mm2 );
199 psubusb_r2r( mm0, mm1 );
200
201 pcmpgtb_r2r( mm5, mm2 );
202 pcmpgtb_r2r( mm5, mm1 );
203 psadbw_r2r( mm6, mm2 );
204 psadbw_r2r( mm6, mm1 );
205
206 paddd_r2r( mm2, mm1 );
207 paddd_r2r( mm1, mm3 ); /* add to top field score */
208
209 p_pix_c += i_pitch_curr;
210 p_pix_p += i_pitch_prev;
211
212 /* bottom field - handling identical to top field, except... */
213 movq_m2r( *((uint64_t*)p_pix_c), mm0 );
214 movq_m2r( *((uint64_t*)p_pix_p), mm1 );
215 movq_r2r( mm0, mm2 );
216 psubusb_r2r( mm1, mm2 );
217 psubusb_r2r( mm0, mm1 );
218
219 pcmpgtb_r2r( mm5, mm2 );
220 pcmpgtb_r2r( mm5, mm1 );
221 psadbw_r2r( mm6, mm2 );
222 psadbw_r2r( mm6, mm1 );
223
224 paddd_r2r( mm2, mm1 );
225 paddd_r2r( mm1, mm4 ); /* ...here we add to bottom field score */
226
227 p_pix_c += i_pitch_curr;
228 p_pix_p += i_pitch_prev;
229 }
230 movq_r2r( mm3, mm7 ); /* score (total) */
231 paddd_r2r( mm4, mm7 );
232 movd_r2m( mm3, i_top_motion );
233 movd_r2m( mm4, i_bot_motion );
234 movd_r2m( mm7, i_motion );
235
236 /* The loop counts actual score * 255. */
237 i_top_motion /= 255;
238 i_bot_motion /= 255;
239 i_motion /= 255;
240
241 emms();
242
243 (*pi_top) = ( i_top_motion >= 8 );
244 (*pi_bot) = ( i_bot_motion >= 8 );
245 return (i_motion >= 8);
246}

254/* See header for function doc. */
255void ComposeFrame( filter_t *p_filter,
256 picture_t *p_outpic,
257 picture_t *p_inpic_top, picture_t *p_inpic_bottom,
258 compose_chroma_t i_output_chroma, bool swapped_uv_conversion )
259{
260 assert( p_outpic != NULL );
261 assert( p_inpic_top != NULL );
262 assert( p_inpic_bottom != NULL );
263
264 /* Valid 4:2:0 chroma handling modes. */
265 assert( i_output_chroma == CC_ALTLINE ||
266 i_output_chroma == CC_UPCONVERT ||
267 i_output_chroma == CC_SOURCE_TOP ||
268 i_output_chroma == CC_SOURCE_BOTTOM ||
269 i_output_chroma == CC_MERGE );
270
271 const bool b_upconvert_chroma = i_output_chroma == CC_UPCONVERT;
272
273 for( int i_plane = 0 ; i_plane < p_inpic_top->i_planes ; i_plane++ )
274 {
275 bool b_is_chroma_plane = ( i_plane == U_PLANE || i_plane == V_PLANE );
276
277 int i_out_plane;
278 if( b_is_chroma_plane && b_upconvert_chroma && swapped_uv_conversion )
279 {
280 if( i_plane == U_PLANE )
281 i_out_plane = V_PLANE;
282 else /* V_PLANE */
283 i_out_plane = U_PLANE;
284 }
285 else
286 {
287 i_out_plane = i_plane;
288 }
289
290 /* Copy luma or chroma, alternating between input fields. */
291 if( !b_is_chroma_plane || i_output_chroma == CC_ALTLINE )
292 {
293 /* Do an alternating line copy. This is always done for luma,
294 and for 4:2:2 chroma. It can be requested for 4:2:0 chroma
295 using CC_ALTLINE (see function doc).
296
297 Note that when we get here, the number of lines matches
298 in input and output.
299 */
300 plane_t dst_top;
301 plane_t dst_bottom;
302 plane_t src_top;
303 plane_t src_bottom;
304 FieldFromPlane( &dst_top, &p_outpic->p[i_out_plane], 0 );
305 FieldFromPlane( &dst_bottom, &p_outpic->p[i_out_plane], 1 );
306 FieldFromPlane( &src_top, &p_inpic_top->p[i_plane], 0 );
307 FieldFromPlane( &src_bottom, &p_inpic_bottom->p[i_plane], 1 );
308
309 /* Copy each field from the corresponding source. */
310 plane_CopyPixels( &dst_top, &src_top );
311 plane_CopyPixels( &dst_bottom, &src_bottom );
312 }
313 else /* Input 4:2:0, on a chroma plane, and not in altline mode. */
314 {
315 if( i_output_chroma == CC_UPCONVERT )
316 {
317 /* Upconverting copy - use all data from both input fields.
318
319 This produces an output picture with independent chroma
320 for each field. It can be used for general input when
321 the two input frames are different.
322
323 The output is 4:2:2, but the input is 4:2:0. Thus the output
324 has twice the lines of the input, and each full chroma plane
325 in the input corresponds to a field chroma plane in the
326 output.
327 */
328 plane_t dst_top;
329 plane_t dst_bottom;
330 FieldFromPlane( &dst_top, &p_outpic->p[i_out_plane], 0 );
331 FieldFromPlane( &dst_bottom, &p_outpic->p[i_out_plane], 1 );
332
333 /* Copy each field from the corresponding source. */
334 plane_CopyPixels( &dst_top, &p_inpic_top->p[i_plane] );
335 plane_CopyPixels( &dst_bottom, &p_inpic_bottom->p[i_plane] );
336 }
337 else if( i_output_chroma == CC_SOURCE_TOP )
338 {
339 /* Copy chroma of input top field. Ignore chroma of input
340 bottom field. Input and output are both 4:2:0, so we just
341 copy the whole plane. */
342 plane_CopyPixels( &p_outpic->p[i_out_plane],
343 &p_inpic_top->p[i_plane] );
344 }
345 else if( i_output_chroma == CC_SOURCE_BOTTOM )
346 {
347 /* Copy chroma of input bottom field. Ignore chroma of input
348 top field. Input and output are both 4:2:0, so we just
349 copy the whole plane. */
350 plane_CopyPixels( &p_outpic->p[i_out_plane],
351 &p_inpic_bottom->p[i_plane] );
352 }
353 else /* i_output_chroma == CC_MERGE */
354 {
355 /* Average the chroma of the input fields.
356 Input and output are both 4:2:0. */
357 uint8_t *p_in_top, *p_in_bottom, *p_out_end, *p_out;
358 p_in_top = p_inpic_top->p[i_plane].p_pixels;
359 p_in_bottom = p_inpic_bottom->p[i_plane].p_pixels;
360 p_out = p_outpic->p[i_out_plane].p_pixels;
361 p_out_end = p_out + p_outpic->p[i_out_plane].i_pitch
362 * p_outpic->p[i_out_plane].i_visible_lines;
363
364 int w = FFMIN3( p_inpic_top->p[i_plane].i_visible_pitch,
365 p_inpic_bottom->p[i_plane].i_visible_pitch,
366 p_outpic->p[i_plane].i_visible_pitch );
367
368 for( ; p_out < p_out_end ; )
369 {
370 Merge( p_out, p_in_top, p_in_bottom, w );
371 p_out += p_outpic->p[i_out_plane].i_pitch;
372 p_in_top += p_inpic_top->p[i_plane].i_pitch;
373 p_in_bottom += p_inpic_bottom->p[i_plane].i_pitch;
374 }
375 EndMerge();
376 }
377 }
378 }
379}

381/* See header for function doc. */
382int EstimateNumBlocksWithMotion( const picture_t* p_prev,
383 const picture_t* p_curr,
384 int *pi_top, int *pi_bot)
385{
386 assert( p_prev != NULL );
387 assert( p_curr != NULL );
388
389 int i_score_top = 0;
390 int i_score_bot = 0;
391
392 if( p_prev->i_planes != p_curr->i_planes )
393 return -1;
394
395 int (*motion_in_block)(uint8_t *, uint8_t *, int , int, int *, int *) =
396 TestForMotionInBlock;
397 /* We must tell our inline helper whether to use MMX acceleration. */
398#ifdef CAN_COMPILE_MMXEXT
399 if (vlc_CPU_MMXEXT())
400 motion_in_block = TestForMotionInBlockMMX;
401#endif
402
403 int i_score = 0;
404 for( int i_plane = 0 ; i_plane < p_prev->i_planes ; i_plane++ )
405 {
406 /* Sanity check */
407 if( p_prev->p[i_plane].i_visible_lines !=
408 p_curr->p[i_plane].i_visible_lines )
409 return -1;
410
411 const int i_pitch_prev = p_prev->p[i_plane].i_pitch;
412 const int i_pitch_curr = p_curr->p[i_plane].i_pitch;
413
414 /* Last pixels and lines (which do not make whole blocks) are ignored.
415 Shouldn't really matter for our purposes. */
416 const int i_mby = p_prev->p[i_plane].i_visible_lines / 8;
417 const int w = FFMIN( p_prev->p[i_plane].i_visible_pitch,
418 p_curr->p[i_plane].i_visible_pitch );
419 const int i_mbx = w / 8;
420
421 for( int by = 0; by < i_mby; ++by )
422 {
423 uint8_t *p_pix_p = &p_prev->p[i_plane].p_pixels[i_pitch_prev*8*by];
424 uint8_t *p_pix_c = &p_curr->p[i_plane].p_pixels[i_pitch_curr*8*by];
425
426 for( int bx = 0; bx < i_mbx; ++bx )
427 {
428 int i_top_temp, i_bot_temp;
429 i_score += motion_in_block( p_pix_p, p_pix_c,
430 i_pitch_prev, i_pitch_curr,
431 &i_top_temp, &i_bot_temp );
432 i_score_top += i_top_temp;
433 i_score_bot += i_bot_temp;
434
435 p_pix_p += 8;
436 p_pix_c += 8;
437 }
438 }
439 }
440
441 if( pi_top )
442 (*pi_top) = i_score_top;
443 if( pi_bot )
444 (*pi_bot) = i_score_bot;
445
446 return i_score;
447}

449/* Threshold (value from Transcode 1.1.5) */
450#define T 100
451
452#ifdef CAN_COMPILE_MMXEXT
453VLC_MMX
454static int CalculateInterlaceScoreMMX( const picture_t* p_pic_top,
455 const picture_t* p_pic_bot )
456{
457 assert( p_pic_top->i_planes == p_pic_bot->i_planes );
458
459 /* Amount of bits must be known for MMX, thus int32_t.
460 Doesn't hurt the C implementation. */
461 int32_t i_score_mmx = 0; /* this must be divided by 255 when finished */
462 int32_t i_score_c = 0; /* this counts as-is (used for non-MMX parts) */
463
464 pxor_r2r( mm7, mm7 ); /* we will keep score in mm7 */
465
466 for( int i_plane = 0 ; i_plane < p_pic_top->i_planes ; ++i_plane )
467 {
468 /* Sanity check */
469 if( p_pic_top->p[i_plane].i_visible_lines !=
470 p_pic_bot->p[i_plane].i_visible_lines )
471 return -1;
472
473 const int i_lasty = p_pic_top->p[i_plane].i_visible_lines-1;
474 const int w = FFMIN( p_pic_top->p[i_plane].i_visible_pitch,
475 p_pic_bot->p[i_plane].i_visible_pitch );
476 const int wm8 = w % 8; /* remainder */
477 const int w8 = w - wm8; /* part of width that is divisible by 8 */
478
479 /* Current line / neighbouring lines picture pointers */
480 const picture_t *cur = p_pic_bot;
481 const picture_t *ngh = p_pic_top;
482 int wc = cur->p[i_plane].i_pitch;
483 int wn = ngh->p[i_plane].i_pitch;
484
485 /* Transcode 1.1.5 only checks every other line. Checking every line
486 works better for anime, which may contain horizontal,
487 one pixel thick cartoon outlines.
488 */
489 for( int y = 1; y < i_lasty; ++y )
490 {
491 uint8_t *p_c = &cur->p[i_plane].p_pixels[y*wc]; /* this line */
492 uint8_t *p_p = &ngh->p[i_plane].p_pixels[(y-1)*wn]; /* prev line */
493 uint8_t *p_n = &ngh->p[i_plane].p_pixels[(y+1)*wn]; /* next line */
494
495 int x = 0;
496
497 /* Easy-to-read C version further below.
498
499 Assumptions: 0 < T < 127
500 # of pixels < (2^32)/255
501 Note: calculates score * 255
502 */
503 static const mmx_t b0 = { .uq = 0x0000000000000000ULL };
504 static const mmx_t b128 = { .uq = 0x8080808080808080ULL };
505 static const mmx_t bT = { .ub = { T, T, T, T, T, T, T, T } };
506
507 for( ; x < w8; x += 8 )
508 {
509 movq_m2r( *((int64_t*)p_c), mm0 );
510 movq_m2r( *((int64_t*)p_p), mm1 );
511 movq_m2r( *((int64_t*)p_n), mm2 );
512
513 psubb_m2r( b128, mm0 );
514 psubb_m2r( b128, mm1 );
515 psubb_m2r( b128, mm2 );
516
517 psubsb_r2r( mm0, mm1 );
518 psubsb_r2r( mm0, mm2 );
519
520 pxor_r2r( mm3, mm3 );
521 pxor_r2r( mm4, mm4 );
522 pxor_r2r( mm5, mm5 );
523 pxor_r2r( mm6, mm6 );
524
525 punpcklbw_r2r( mm1, mm3 );
526 punpcklbw_r2r( mm2, mm4 );
527 punpckhbw_r2r( mm1, mm5 );
528 punpckhbw_r2r( mm2, mm6 );
529
530 pmulhw_r2r( mm3, mm4 );
531 pmulhw_r2r( mm5, mm6 );
532
533 packsswb_r2r(mm4, mm6);
534 pcmpgtb_m2r( bT, mm6 );
535 psadbw_m2r( b0, mm6 );
536 paddd_r2r( mm6, mm7 );
537
538 p_c += 8;
539 p_p += 8;
540 p_n += 8;
541 }
542
543 for( ; x < w; ++x )
544 {
545 /* Worst case: need 17 bits for "comb". */
546 int_fast32_t C = *p_c;
547 int_fast32_t P = *p_p;
548 int_fast32_t N = *p_n;
549
550 /* Comments in Transcode's filter_ivtc.c attribute this
551 combing metric to Gunnar Thalin.
552
553 The idea is that if the picture is interlaced, both
554 expressions will have the same sign, and this comes
555 up positive. The value T = 100 has been chosen such
556 that a pixel difference of 10 (on average) will
557 trigger the detector.
558 */
559 int_fast32_t comb = (P - C) * (N - C);
560 if( comb > T )
561 ++i_score_c;
562
563 ++p_c;
564 ++p_p;
565 ++p_n;
566 }
567
568 /* Now the other field - swap current and neighbour pictures */
569 const picture_t *tmp = cur;
570 cur = ngh;
571 ngh = tmp;
572 int tmp_pitch = wc;
573 wc = wn;
574 wn = tmp_pitch;
575 }
576 }
577
578 movd_r2m( mm7, i_score_mmx );
579 emms();
580
581 return i_score_mmx/255 + i_score_c;
582}
583#endif

585/* See header for function doc. */
586int CalculateInterlaceScore( const picture_t* p_pic_top,
587 const picture_t* p_pic_bot )
588{
589 /*
590 We use the comb metric from the IVTC filter of Transcode 1.1.5.
591 This was found to work better for the particular purpose of IVTC
592 than RenderX()'s comb metric.
593
594 Note that we *must not* subsample at all in order to catch interlacing
595 in telecined frames with localized motion (e.g. anime with characters
596 talking, where only mouths move and everything else stays still.)
597 */
598
599 assert( p_pic_top != NULL );
600 assert( p_pic_bot != NULL );
601
602 if( p_pic_top->i_planes != p_pic_bot->i_planes )
603 return -1;
604
605#ifdef CAN_COMPILE_MMXEXT
606 if (vlc_CPU_MMXEXT())
607 return CalculateInterlaceScoreMMX( p_pic_top, p_pic_bot );
608#endif
609
610 int32_t i_score = 0;
611
612 for( int i_plane = 0 ; i_plane < p_pic_top->i_planes ; ++i_plane )
613 {
614 /* Sanity check */
615 if( p_pic_top->p[i_plane].i_visible_lines !=
616 p_pic_bot->p[i_plane].i_visible_lines )
617 return -1;
618
619 const int i_lasty = p_pic_top->p[i_plane].i_visible_lines-1;
620 const int w = FFMIN( p_pic_top->p[i_plane].i_visible_pitch,
621 p_pic_bot->p[i_plane].i_visible_pitch );
622
623 /* Current line / neighbouring lines picture pointers */
624 const picture_t *cur = p_pic_bot;
625 const picture_t *ngh = p_pic_top;
626 int wc = cur->p[i_plane].i_pitch;
627 int wn = ngh->p[i_plane].i_pitch;
628
629 /* Transcode 1.1.5 only checks every other line. Checking every line
630 works better for anime, which may contain horizontal,
631 one pixel thick cartoon outlines.
632 */
633 for( int y = 1; y < i_lasty; ++y )
634 {
635 uint8_t *p_c = &cur->p[i_plane].p_pixels[y*wc]; /* this line */
636 uint8_t *p_p = &ngh->p[i_plane].p_pixels[(y-1)*wn]; /* prev line */
637 uint8_t *p_n = &ngh->p[i_plane].p_pixels[(y+1)*wn]; /* next line */
638
639 for( int x = 0; x < w; ++x )
640 {
641 /* Worst case: need 17 bits for "comb". */
642 int_fast32_t C = *p_c;
643 int_fast32_t P = *p_p;
644 int_fast32_t N = *p_n;
645
646 /* Comments in Transcode's filter_ivtc.c attribute this
647 combing metric to Gunnar Thalin.
648
649 The idea is that if the picture is interlaced, both
650 expressions will have the same sign, and this comes
651 up positive. The value T = 100 has been chosen such
652 that a pixel difference of 10 (on average) will
653 trigger the detector.
654 */
655 int_fast32_t comb = (P - C) * (N - C);
656 if( comb > T )
657 ++i_score;
658
659 ++p_c;
660 ++p_p;
661 ++p_n;
662 }
663
664 /* Now the other field - swap current and neighbour pictures */
665 const picture_t *tmp = cur;
666 cur = ngh;
667 ngh = tmp;
668 int tmp_pitch = wc;
669 wc = wn;
670 wn = tmp_pitch;
671 }
672 }
673
674 return i_score;
675}
