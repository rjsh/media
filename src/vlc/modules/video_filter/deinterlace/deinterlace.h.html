68typedef enum { DEINTERLACE_DISCARD, DEINTERLACE_MEAN, DEINTERLACE_BLEND,

73#define METADATA_SIZE (3)
74/**
75 * Metadata history structure, used for framerate doublers.
76 * This is used for computing field duration in Deinterlace().
77 * @see Deinterlace()
78 */
79typedef struct {
80 mtime_t pi_date[METADATA_SIZE];
81 int pi_nb_fields[METADATA_SIZE];
82 bool pb_top_field_first[METADATA_SIZE];
83} metadata_history_t;

85#define HISTORY_SIZE (3)
86#define CUSTOM_PTS -1
87/**
88 * Top-level deinterlace subsystem state.
89 */
90struct filter_sys_t
91{
92 const vlc_chroma_description_t *chroma;
93
94 uint8_t i_mode; /**< Deinterlace mode */
95
96 /* Algorithm behaviour flags */
97 bool b_double_rate; /**< Shall we double the framerate? */
98 bool b_half_height; /**< Shall be divide the height by 2 */
99 bool b_use_frame_history; /**< Use the input frame history buffer? */
100
101 /** Merge routine: C, MMX, SSE, ALTIVEC, NEON, ... */
102 void (*pf_merge) ( void *, const void *, const void *, size_t );
103#if defined (__i386__) || defined (__x86_64__)
104 /** Merge finalization routine for SSE */
105 void (*pf_end_merge) ( void );
106#endif
107
108 /**
109 * Metadata history (PTS, nb_fields, TFF). Used for framerate doublers.
110 * @see metadata_history_t
111 */
112 metadata_history_t meta;
113
114 /** Output frame timing / framerate doubler control
115 (see extra documentation in deinterlace.h) */
116 int i_frame_offset;
117
118 /** Input frame history buffer for algorithms with temporal filtering. */
119 picture_t *pp_history[HISTORY_SIZE];
120
121 /* Algorithm-specific substructures */
122 phosphor_sys_t phosphor; /**< Phosphor algorithm state. */
123 ivtc_sys_t ivtc; /**< IVTC algorithm state. */
124};

130/**
131 * Top-level filtering method.
132 *
133 * Open() sets this up as the processing method (pf_video_filter)
134 * in the filter structure.
135 *
136 * Note that there is no guarantee that the returned picture directly
137 * corresponds to p_pic. The first few times, the filter may not even
138 * return a picture, if it is still filling the history for temporal
139 * filtering (although such filters often return *something* also
140 * while starting up). It should be assumed that N input pictures map to
141 * M output pictures, with no restrictions for N and M (except that there
142 * is not much delay).
143 *
144 * Also, there is no guarantee that the PTS of the frame stays untouched.
145 * In fact, framerate doublers automatically compute the proper PTSs for the
146 * two output frames for each input frame, and IVTC does a nontrivial
147 * framerate conversion (29.97 > 23.976 fps).
148 *
149 * Yadif has an offset of one frame between input and output, but introduces
150 * no delay: the returned frame is the *previous* input frame deinterlaced,
151 * complete with its original PTS.
152 *
153 * Finally, note that returning NULL sometimes can be normal behaviour for some
154 * algorithms (e.g. IVTC).
155 *
156 * Currently:
157 * Most algorithms: 1 -> 1, no offset
158 * All framerate doublers: 1 -> 2, no offset
159 * Yadif: 1 -> 1, offset of one frame
160 * IVTC: 1 -> 1 or 0 (depends on whether a drop was needed)
161 * with an offset of one frame (in most cases)
162 * and framerate conversion.
163 *
164 * @param p_filter The filter instance.
165 * @param p_pic The latest input picture.
166 * @return Deinterlaced picture(s). Linked list of picture_t's or NULL.
167 * @see Open()
168 * @see filter_t
169 * @see filter_sys_t
170 */
171picture_t *Deinterlace( filter_t *p_filter, picture_t *p_pic );
172
173/**
174 * Reads the configuration, sets up and starts the filter.
175 *
176 * Possible reasons for returning VLC_EGENERIC:
177 * - Unsupported input chroma. See IsChromaSupported().
178 * - Caller has set p_filter->b_allow_fmt_out_change to false,
179 * but the algorithm chosen in the configuration
180 * wants to convert the output to a format different
181 * from the input. See SetFilterMethod().
182 *
183 * Open() is atomic: if an error occurs, the state of p_this
184 * is left as it was before the call to this function.
185 *
186 * @param p_this The filter instance as vlc_object_t.
187 * @return VLC error code
188 * @retval VLC_SUCCESS All ok, filter set up and started.
189 * @retval VLC_ENOMEM Memory allocation error, initialization aborted.
190 * @retval VLC_EGENERIC Something went wrong, initialization aborted.
191 * @see IsChromaSupported()
192 * @see SetFilterMethod()
193 */
194int Open( vlc_object_t *p_this );
195
196/**
197 * Resets the filter state, including resetting all algorithm-specific state

199 *

203 * @param p_filter The filter instance.

208 * @see phosphor_sys_t
209 * @see ivtc_sys_t

247/**
248 * \file
249 * Deinterlacer plugin for vlc. Data structures and video filter2 functions.
250 *
251 * Note on i_frame_offset:
252 *
253 * This value indicates the offset between input and output frames in the
254 * currently active deinterlace algorithm. See the rationale below for why
255 * this is needed and how it is used.
256 *
257 * Valid range: 0 <= i_frame_offset < METADATA_SIZE, or
258 * i_frame_offset = CUSTOM_PTS.
259 * The special value CUSTOM_PTS is only allowed
260 * if b_double_rate is false.
261 *
262 * If CUSTOM_PTS is used, the algorithm must compute the outgoing
263 * PTSs itself, and additionally, read the TFF/BFF information
264 * itself (if it needs it) from the incoming frames.
265 *
266 * Meaning of values:
267 * 0 = output frame corresponds to the current input frame
268 * (no frame offset; default if not set),
269 * 1 = output frame corresponds to the previous input frame
270 * (e.g. Yadif and Yadif2x work like this),
271 * ...
272 *
273 * If necessary, i_frame_offset should be updated by the active deinterlace
274 * algorithm to indicate the correct delay for the *next* input frame.
275 * It does not matter at which i_order the algorithm updates this information,
276 * but the new value will only take effect upon the next call to Deinterlace()
277 * (i.e. at the next incoming frame).
278 *
279 * The first-ever frame that arrives to the filter after Open() is always
280 * handled as having i_frame_offset = 0. For the second and all subsequent
281 * frames, each algorithm is responsible for setting the offset correctly.
282 * (The default is 0, so if that is correct, there's no need to do anything.)
283 *
284 * This solution guarantees that i_frame_offset:
285 * 1) is up to date at the start of each frame,
286 * 2) does not change (as far as Deinterlace() is concerned) during
287 * a frame, and
288 * 3) does not need a special API for setting the value at the start of each
289 * input frame, before the algorithm starts rendering the (first) output
290 * frame for that input frame.
291 *
292 * The deinterlace algorithm is allowed to behave differently for different
293 * input frames. This is especially important for startup, when full history
294 * (as defined by each algorithm) is not yet available. During the first-ever
295 * input frame, it is clear that it is the only possible source for
296 * information, so i_frame_offset = 0 is necessarily correct. After that,
297 * what to do is up to each algorithm.
298 *
299 * Having the correct offset at the start of each input frame is critically
300 * important in order to:
301 * 1) Allocate the correct number of output frames for framerate doublers,
302 * and to
303 * 2) Pass correct TFF/BFF information to the algorithm.
304 *
305 * These points are important for proper soft field repeat support. This
306 * feature is used in some streams (especially NTSC) originating from film.
307 * For example, in soft NTSC telecine, the number of fields alternates
308 * as 3,2,3,2,... and the video field dominance flips every two frames (after
309 * every "3"). Also, some streams request an occasional field repeat
310 * (nb_fields = 3), after which the video field dominance flips.
311 * To render such streams correctly, the nb_fields and TFF/BFF information
312 * must be taken from the specific input frame that the algorithm intends
313 * to render.
314 *
315 * Additionally, the output PTS is automatically computed by Deinterlace()
316 * from i_frame_offset and i_order.
317 *
318 * It is possible to use the special value CUSTOM_PTS to indicate that the
319 * algorithm computes the output PTSs itself. In this case, Deinterlace()
320 * will pass them through. This special value is not valid for framerate
321 * doublers, as by definition they are field renderers, so they need to
322 * use the original field timings to work correctly. Basically, this special
323 * value is only intended for algorithms that need to perform nontrivial
324 * framerate conversions (such as IVTC).
325 */
