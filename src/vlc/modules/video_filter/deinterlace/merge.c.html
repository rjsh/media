49void Merge8BitGeneric( void *_p_dest, const void *_p_s1,
50 const void *_p_s2, size_t i_bytes )
51{
52 uint8_t *p_dest = _p_dest;
53 const uint8_t *p_s1 = _p_s1;
54 const uint8_t *p_s2 = _p_s2;
55
56 for( ; i_bytes > 0; i_bytes-- )
57 *p_dest++ = ( *p_s1++ + *p_s2++ ) >> 1;
58}

60void Merge16BitGeneric( void *_p_dest, const void *_p_s1,
61 const void *_p_s2, size_t i_bytes )
62{
63 uint16_t *p_dest = _p_dest;
64 const uint16_t *p_s1 = _p_s1;
65 const uint16_t *p_s2 = _p_s2;
66
67 for( size_t i_words = i_bytes / 2; i_words > 0; i_words-- )
68 *p_dest++ = ( *p_s1++ + *p_s2++ ) >> 1;
69}

71#if defined(CAN_COMPILE_MMXEXT)
72VLC_MMX
73void MergeMMXEXT( void *_p_dest, const void *_p_s1, const void *_p_s2,
74 size_t i_bytes )
75{
76 uint8_t *p_dest = _p_dest;
77 const uint8_t *p_s1 = _p_s1;
78 const uint8_t *p_s2 = _p_s2;
79
80 for( ; i_bytes >= 8; i_bytes -= 8 )
81 {
82 __asm__ __volatile__( "movq %2,%%mm1;"
83 "pavgb %1, %%mm1;"
84 "movq %%mm1, %0" :"=m" (*p_dest):
85 "m" (*p_s1),
86 "m" (*p_s2) : "mm1" );
87 p_dest += 8;
88 p_s1 += 8;
89 p_s2 += 8;
90 }
91
92 for( ; i_bytes > 0; i_bytes-- )
93 *p_dest++ = ( *p_s1++ + *p_s2++ ) >> 1;
94}

125void Merge8BitSSE2( void *_p_dest, const void *_p_s1, const void *_p_s2,
126 size_t i_bytes )
127{
128 uint8_t *p_dest = _p_dest;
129 const uint8_t *p_s1 = _p_s1;
130 const uint8_t *p_s2 = _p_s2;
131
132 for( ; i_bytes > 0 && ((uintptr_t)p_s1 & 15); i_bytes-- )
133 *p_dest++ = ( *p_s1++ + *p_s2++ ) >> 1;
134
135 for( ; i_bytes >= 16; i_bytes -= 16 )
136 {
137 __asm__ __volatile__( "movdqu %2,%%xmm1;"
138 "pavgb %1, %%xmm1;"
139 "movdqu %%xmm1, %0" :"=m" (*p_dest):
140 "m" (*p_s1),
141 "m" (*p_s2) : "xmm1" );
142 p_dest += 16;
143 p_s1 += 16;
144 p_s2 += 16;
145 }
146
147 for( ; i_bytes > 0; i_bytes-- )
148 *p_dest++ = ( *p_s1++ + *p_s2++ ) >> 1;
149}

151VLC_SSE
152void Merge16BitSSE2( void *_p_dest, const void *_p_s1, const void *_p_s2,
153 size_t i_bytes )
154{
155 uint16_t *p_dest = _p_dest;
156 const uint16_t *p_s1 = _p_s1;
157 const uint16_t *p_s2 = _p_s2;
158
159 size_t i_words = i_bytes / 2;
160 for( ; i_words > 0 && ((uintptr_t)p_s1 & 15); i_words-- )
161 *p_dest++ = ( *p_s1++ + *p_s2++ ) >> 1;
162
163 for( ; i_words >= 8; i_words -= 8 )
164 {
165 __asm__ __volatile__( "movdqu %2,%%xmm1;"
166 "pavgw %1, %%xmm1;"
167 "movdqu %%xmm1, %0" :"=m" (*p_dest):
168 "m" (*p_s1),
169 "m" (*p_s2) : "xmm1" );
170 p_dest += 8;
171 p_s1 += 8;
172 p_s2 += 8;
173 }
174
175 for( ; i_words > 0; i_words-- )
176 *p_dest++ = ( *p_s1++ + *p_s2++ ) >> 1;
177}

181#ifdef CAN_COMPILE_C_ALTIVEC
182void MergeAltivec( void *_p_dest, const void *_p_s1,
183 const void *_p_s2, size_t i_bytes )
184{
185 uint8_t *p_dest = _p_dest;
186 const uint8_t *p_s1 = _p_s1;
187 const uint8_t *p_s2 = _p_s2;
188 uint8_t *p_end = p_dest + i_bytes - 15;
189
190 /* Use C until the first 16-bytes aligned destination pixel */
191 while( (uintptr_t)p_dest & 0xF )
192 {
193 *p_dest++ = ( (uint16_t)(*p_s1++) + (uint16_t)(*p_s2++) ) >> 1;
194 }
195
196 if( ( (int)p_s1 & 0xF ) | ( (int)p_s2 & 0xF ) )
197 {
198 /* Unaligned source */
199 vector unsigned char s1v, s2v, destv;
200 vector unsigned char s1oldv, s2oldv, s1newv, s2newv;
201 vector unsigned char perm1v, perm2v;
202
203 perm1v = vec_lvsl( 0, p_s1 );
204 perm2v = vec_lvsl( 0, p_s2 );
205 s1oldv = vec_ld( 0, p_s1 );
206 s2oldv = vec_ld( 0, p_s2 );
207
208 while( p_dest < p_end )
209 {
210 s1newv = vec_ld( 16, p_s1 );
211 s2newv = vec_ld( 16, p_s2 );
212 s1v = vec_perm( s1oldv, s1newv, perm1v );
213 s2v = vec_perm( s2oldv, s2newv, perm2v );
214 s1oldv = s1newv;
215 s2oldv = s2newv;
216 destv = vec_avg( s1v, s2v );
217 vec_st( destv, 0, p_dest );
218
219 p_s1 += 16;
220 p_s2 += 16;
221 p_dest += 16;
222 }
223 }
224 else
225 {
226 /* Aligned source */
227 vector unsigned char s1v, s2v, destv;
228
229 while( p_dest < p_end )
230 {
231 s1v = vec_ld( 0, p_s1 );
232 s2v = vec_ld( 0, p_s2 );
233 destv = vec_avg( s1v, s2v );
234 vec_st( destv, 0, p_dest );
235
236 p_s1 += 16;
237 p_s2 += 16;
238 p_dest += 16;
239 }
240 }
241
242 p_end += 15;
243
244 while( p_dest < p_end )
245 *p_dest++ = ( *p_s1++ + *p_s2++ ) >> 1;
246}
