22/*
23 * The type of an value that fits in an MMX register (note that long
24 * long constant values MUST be suffixed by LL and unsigned long long
25 * values by ULL, lest they be truncated by the compiler)
26 */
27
28#include <stdint.h>
29
30typedef union {
31 int64_t q; /* Quadword (64-bit) value */
32 uint64_t uq; /* Unsigned Quadword */
33 int32_t d[2]; /* 2 Doubleword (32-bit) values */
34 uint32_t ud[2]; /* 2 Unsigned Doubleword */
35 int16_t w[4]; /* 4 Word (16-bit) values */
36 uint16_t uw[4]; /* 4 Unsigned Word */
37 int8_t b[8]; /* 8 Byte (8-bit) values */
38 uint8_t ub[8]; /* 8 Unsigned Byte */
39 float s[2]; /* Single-precision (32-bit) value */
40} ATTR_ALIGN(8) mmx_t; /* On an 8-byte (64-bit) boundary */

43#define mmx_i2r(op,imm,reg) \
44 __asm__ __volatile__ (#op " %0, %%" #reg \
45 : /* nothing */ \
46 : "i" (imm) \
47 : #reg)
48
49#define mmx_m2r(op,mem,reg) \
50 __asm__ __volatile__ (#op " %0, %%" #reg \
51 : /* nothing */ \
52 : "m" (mem) \
53 : #reg)
54
55#define mmx_r2m(op,reg,mem) \
56 __asm__ __volatile__ (#op " %%" #reg ", %0" \
57 : "=m" (mem) \
58 : /* nothing */ \
59 : "memory")
60
61#define mmx_r2r(op,regs,regd) \
62 __asm__ __volatile__ (#op " %%" #regs ", %%" #regd ::: #regd)

203#define mmx_m2ri(op,mem,reg,imm) \
204 __asm__ __volatile__ (#op " %1, %0, %%" #reg \
205 : /* nothing */ \
206 : "X" (mem), "X" (imm) \
207 : #reg)
208#define mmx_r2ri(op,regs,regd,imm) \
209 __asm__ __volatile__ (#op " %0, %%" #regs ", %%" #regd \
210 : /* nothing */ \
211 : "X" (imm) \
212 : #regd)
213
214#define mmx_fetch(mem,hint) \
215 __asm__ __volatile__ ("prefetch" #hint " %0" \
216 : /* nothing */ \
217 : "X" (mem))
