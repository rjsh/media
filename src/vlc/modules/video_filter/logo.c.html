218/**
219 * Common open function
220 */
221static int OpenCommon( vlc_object_t *p_this, bool b_sub )
222{
223 filter_t *p_filter = (filter_t *)p_this;
224 filter_sys_t *p_sys;
225 char *psz_filename;
226
227 /* */
228 if( !b_sub && !es_format_IsSimilar( &p_filter->fmt_in, &p_filter->fmt_out ) )
229 {
230 msg_Err( p_filter, "Input and output format does not match" );
231 return VLC_EGENERIC;
232 }
233
234 /* */
235 p_filter->p_sys = p_sys = malloc( sizeof( *p_sys ) );
236 if( !p_sys )
237 return VLC_ENOMEM;
238
239 /* */
240 p_sys->p_blend = NULL;
241 if( !b_sub )
242 {
243
244 p_sys->p_blend = filter_NewBlend( VLC_OBJECT(p_filter),
245 &p_filter->fmt_in.video );
246 if( !p_sys->p_blend )
247 {
248 free( p_sys );
249 return VLC_EGENERIC;
250 }
251 }
252
253 /* */
254 config_ChainParse( p_filter, CFG_PREFIX, ppsz_filter_options,
255 p_filter->p_cfg );
256
257 /* */
258 logo_list_t *p_list = &p_sys->list;
259
260 psz_filename = var_CreateGetStringCommand( p_filter, "logo-file" );
261 if( !psz_filename )
262 {
263 if( p_sys->p_blend )
264 filter_DeleteBlend( p_sys->p_blend );
265 free( p_sys );
266 return VLC_ENOMEM;
267 }
268 if( *psz_filename == '\0' )
269 msg_Warn( p_this, "no logo file specified" );
270
271 p_list->i_alpha = var_CreateGetIntegerCommand( p_filter, "logo-opacity");
272 p_list->i_alpha = VLC_CLIP( p_list->i_alpha, 0, 255 );
273 p_list->i_delay = var_CreateGetIntegerCommand( p_filter, "logo-delay" );
274 p_list->i_repeat = var_CreateGetIntegerCommand( p_filter, "logo-repeat" );
275
276 p_sys->i_pos = var_CreateGetIntegerCommand( p_filter, "logo-position" );
277 p_sys->i_pos_x = var_CreateGetIntegerCommand( p_filter, "logo-x" );
278 p_sys->i_pos_y = var_CreateGetIntegerCommand( p_filter, "logo-y" );
279
280 /* Ignore aligment if a position is given for video filter */
281 if( !b_sub && p_sys->i_pos_x >= 0 && p_sys->i_pos_y >= 0 )
282 p_sys->i_pos = 0;
283
284 vlc_mutex_init( &p_sys->lock );
285 LogoListLoad( p_this, p_list, psz_filename );
286 p_sys->b_spu_update = true;
287 p_sys->b_mouse_grab = false;
288
289 for( int i = 0; ppsz_filter_callbacks[i]; i++ )
290 var_AddCallback( p_filter, ppsz_filter_callbacks[i],
291 LogoCallback, p_sys );
292
293 /* Misc init */
294 if( b_sub )
295 {
296 p_filter->pf_sub_source = FilterSub;
297 }
298 else
299 {
300 p_filter->pf_video_filter = FilterVideo;
301 p_filter->pf_video_mouse = Mouse;
302 }
303
304 free( psz_filename );
305 return VLC_SUCCESS;
306}

308/**
309 * Common close function
310 */
311static void Close( vlc_object_t *p_this )
312{
313 filter_t *p_filter = (filter_t *)p_this;
314 filter_sys_t *p_sys = p_filter->p_sys;
315
316 for( int i = 0; ppsz_filter_callbacks[i]; i++ )
317 var_DelCallback( p_filter, ppsz_filter_callbacks[i],
318 LogoCallback, p_sys );
319
320 if( p_sys->p_blend )
321 filter_DeleteBlend( p_sys->p_blend );
322
323 vlc_mutex_destroy( &p_sys->lock );
324 LogoListUnload( &p_sys->list );
325 free( p_sys );
326}

328/**
329 * Sub source
330 */
331static subpicture_t *FilterSub( filter_t *p_filter, mtime_t date )
332{
333 filter_sys_t *p_sys = p_filter->p_sys;
334 logo_list_t *p_list = &p_sys->list;
335
336 subpicture_t *p_spu;
337 subpicture_region_t *p_region;
338 video_format_t fmt;
339 picture_t *p_pic;
340 logo_t *p_logo;
341
342 vlc_mutex_lock( &p_sys->lock );
343 /* Basic test: b_spu_update occurs on a dynamic change,
344 & i_next_pic is the general timer, when to
345 look at updating the logo image */
346
347 if( ( !p_sys->b_spu_update && p_list->i_next_pic > date ) ||
348 !p_list->i_repeat )
349 {
350 vlc_mutex_unlock( &p_sys->lock );
351 return NULL;
352 }
353
354 /* adjust index to the next logo */
355 p_logo = LogoListNext( p_list, date );
356 p_sys->b_spu_update = false;
357
358 p_pic = p_logo->p_pic;
359
360 /* Allocate the subpicture internal data. */
361 p_spu = filter_NewSubpicture( p_filter );
362 if( !p_spu )
363 goto exit;
364
365 p_spu->b_absolute = p_sys->b_absolute;
366 p_spu->i_start = date;
367 p_spu->i_stop = 0;
368 p_spu->b_ephemer = true;
369
370 /* Send an empty subpicture to clear the display when needed */
371 if( p_list->i_repeat != -1 && p_list->i_counter == 0 )
372 {
373 p_list->i_repeat--;
374 if( p_list->i_repeat < 0 )
375 goto exit;
376 }
377 if( !p_pic || !p_logo->i_alpha ||
378 ( p_logo->i_alpha == -1 && !p_list->i_alpha ) )
379 goto exit;
380
381 /* Create new SPU region */
382 memset( &fmt, 0, sizeof(video_format_t) );
383 fmt.i_chroma = VLC_CODEC_YUVA;
384 fmt.i_sar_num = fmt.i_sar_den = 1;
385 fmt.i_width = fmt.i_visible_width = p_pic->p[Y_PLANE].i_visible_pitch;
386 fmt.i_height = fmt.i_visible_height = p_pic->p[Y_PLANE].i_visible_lines;
387 fmt.i_x_offset = fmt.i_y_offset = 0;
388 p_region = subpicture_region_New( &fmt );
389 if( !p_region )
390 {
391 msg_Err( p_filter, "cannot allocate SPU region" );
392 p_filter->pf_sub_buffer_del( p_filter, p_spu );
393 p_spu = NULL;
394 goto exit;
395 }
396
397 /* */
398 picture_Copy( p_region->p_picture, p_pic );
399
400 /* where to locate the logo: */
401 if( p_sys->i_pos < 0 )
402 { /* set to an absolute xy */
403 p_region->i_align = SUBPICTURE_ALIGN_RIGHT | SUBPICTURE_ALIGN_TOP;
404 p_spu->b_absolute = true;
405 }
406 else
407 { /* set to one of the 9 relative locations */
408 p_region->i_align = p_sys->i_pos;
409 p_spu->b_absolute = false;
410 }
411
412 p_region->i_x = p_sys->i_pos_x;
413 p_region->i_y = p_sys->i_pos_y;
414
415 p_spu->p_region = p_region;
416
417 p_spu->i_alpha = ( p_logo->i_alpha != -1 ?
418 p_logo->i_alpha : p_list->i_alpha );
419
420exit:
421 vlc_mutex_unlock( &p_sys->lock );
422
423 return p_spu;
424}

426/**
427 * Video filter
428 */
429static picture_t *FilterVideo( filter_t *p_filter, picture_t *p_src )
430{
431 filter_sys_t *p_sys = p_filter->p_sys;
432 logo_list_t *p_list = &p_sys->list;
433
434 picture_t *p_dst = filter_NewPicture( p_filter );
435 if( !p_dst )
436 goto exit;
437
438 picture_Copy( p_dst, p_src );
439
440 /* */
441 vlc_mutex_lock( &p_sys->lock );
442
443 logo_t *p_logo;
444 if( p_list->i_next_pic < p_src->date )
445 p_logo = LogoListNext( p_list, p_src->date );
446 else
447 p_logo = LogoListCurrent( p_list );
448
449 /* */
450 const picture_t *p_pic = p_logo->p_pic;
451 if( p_pic )
452 {
453 const video_format_t *p_fmt = &p_pic->format;
454 const int i_dst_w = p_filter->fmt_out.video.i_visible_width;
455 const int i_dst_h = p_filter->fmt_out.video.i_visible_height;
456
457 if( p_sys->i_pos )
458 {
459 if( p_sys->i_pos & SUBPICTURE_ALIGN_BOTTOM )
460 {
461 p_sys->i_pos_y = i_dst_h - p_fmt->i_visible_height;
462 }
463 else if ( !(p_sys->i_pos & SUBPICTURE_ALIGN_TOP) )
464 {
465 p_sys->i_pos_y = ( i_dst_h - p_fmt->i_visible_height ) / 2;
466 }
467 else
468 {
469 p_sys->i_pos_y = 0;
470 }
471
472 if( p_sys->i_pos & SUBPICTURE_ALIGN_RIGHT )
473 {
474 p_sys->i_pos_x = i_dst_w - p_fmt->i_visible_width;
475 }
476 else if ( !(p_sys->i_pos & SUBPICTURE_ALIGN_LEFT) )
477 {
478 p_sys->i_pos_x = ( i_dst_w - p_fmt->i_visible_width ) / 2;
479 }
480 else
481 {
482 p_sys->i_pos_x = 0;
483 }
484 }
485
486 if( p_sys->i_pos_x < 0 || p_sys->i_pos_y < 0 )
487 {
488 msg_Warn( p_filter,
489 "logo(%ix%i) doesn't fit into video(%ix%i)",
490 p_fmt->i_visible_width, p_fmt->i_visible_height,
491 i_dst_w,i_dst_h );
492 p_sys->i_pos_x = (p_sys->i_pos_x > 0) ? p_sys->i_pos_x : 0;
493 p_sys->i_pos_y = (p_sys->i_pos_y > 0) ? p_sys->i_pos_y : 0;
494 }
495
496 /* */
497 const int i_alpha = p_logo->i_alpha != -1 ? p_logo->i_alpha : p_list->i_alpha;
498 if( filter_ConfigureBlend( p_sys->p_blend, i_dst_w, i_dst_h, p_fmt ) ||
499 filter_Blend( p_sys->p_blend, p_dst, p_sys->i_pos_x, p_sys->i_pos_y,
500 p_pic, i_alpha ) )
501 {
502 msg_Err( p_filter, "failed to blend a picture" );
503 }
504 }
505 vlc_mutex_unlock( &p_sys->lock );
506
507exit:
508 picture_Release( p_src );
509 return p_dst;
510}

512static int Mouse( filter_t *p_filter, vlc_mouse_t *p_mouse,
513 const vlc_mouse_t *p_old, const vlc_mouse_t *p_new )
514{
515 filter_sys_t *p_sys = p_filter->p_sys;
516
517 vlc_mutex_lock( &p_sys->lock );
518 logo_t *p_logo = LogoListCurrent( &p_sys->list );
519 const picture_t *p_pic = p_logo->p_pic;
520
521 if( p_pic )
522 {
523 const video_format_t *p_fmt = &p_pic->format;
524 const int i_logo_w = p_fmt->i_visible_width;
525 const int i_logo_h = p_fmt->i_visible_height;
526
527 /* Check if we are over the logo */
528 const bool b_over = p_new->i_x >= p_sys->i_pos_x &&
529 p_new->i_x < p_sys->i_pos_x + i_logo_w &&
530 p_new->i_y >= p_sys->i_pos_y &&
531 p_new->i_y < p_sys->i_pos_y + i_logo_h;
532
533 if( b_over && vlc_mouse_HasPressed( p_old, p_new, MOUSE_BUTTON_LEFT ) )
534 p_sys->b_mouse_grab = true;
535 else if( vlc_mouse_HasReleased( p_old, p_new, MOUSE_BUTTON_LEFT ) )
536 p_sys->b_mouse_grab = false;
537
538 if( p_sys->b_mouse_grab )
539 {
540 int i_dx, i_dy;
541 vlc_mouse_GetMotion( &i_dx, &i_dy, p_old, p_new );
542 p_sys->i_pos_x = VLC_CLIP( p_sys->i_pos_x + i_dx, 0,
543 p_filter->fmt_in.video.i_width - i_logo_w );
544 p_sys->i_pos_y = VLC_CLIP( p_sys->i_pos_y + i_dy, 0,
545 p_filter->fmt_in.video.i_height - i_logo_h );
546
547 /* object under mouse has moved */
548 var_SetBool( p_filter->p_parent, "mouse-object", true );
549 }
550 else if( b_over )
551 {
552 /* object under mouse stoped moving */
553 var_SetBool( p_filter->p_parent, "mouse-object", false );
554 }
555
556 if( p_sys->b_mouse_grab || b_over )
557 {
558 vlc_mutex_unlock( &p_sys->lock );
559 return VLC_EGENERIC;
560 }
561 }
562 vlc_mutex_unlock( &p_sys->lock );
563
564 *p_mouse = *p_new;
565 return VLC_SUCCESS;
566}

568/*****************************************************************************
569 * Callback to update params on the fly
570 *****************************************************************************/
571static int LogoCallback( vlc_object_t *p_this, char const *psz_var,
572 vlc_value_t oldval, vlc_value_t newval, void *p_data )
573{
574 VLC_UNUSED(oldval);
575 filter_sys_t *p_sys = (filter_sys_t *)p_data;
576 logo_list_t *p_list = &p_sys->list;
577
578 vlc_mutex_lock( &p_sys->lock );
579 if( !strcmp( psz_var, "logo-file" ) )
580 {
581 LogoListUnload( p_list );
582 LogoListLoad( p_this, p_list, newval.psz_string );
583 }
584 else if ( !strcmp( psz_var, "logo-x" ) )
585 {
586 p_sys->i_pos_x = newval.i_int;
587 }
588 else if ( !strcmp( psz_var, "logo-y" ) )
589 {
590 p_sys->i_pos_y = newval.i_int;
591 }
592 else if ( !strcmp( psz_var, "logo-position" ) )
593 {
594 p_sys->i_pos = newval.i_int;
595 }
596 else if ( !strcmp( psz_var, "logo-opacity" ) )
597 {
598 p_list->i_alpha = VLC_CLIP( newval.i_int, 0, 255 );
599 }
600 else if ( !strcmp( psz_var, "logo-repeat" ) )
601 {
602 p_list->i_repeat = newval.i_int;
603 }
604 p_sys->b_spu_update = true;
605 vlc_mutex_unlock( &p_sys->lock );
606
607 return VLC_SUCCESS;
608}

610/**
611 * It loads the logo image into memory.
612 */
613static picture_t *LoadImage( vlc_object_t *p_this, const char *psz_filename )
614{
615 if( !psz_filename )
616 return NULL;
617
618 video_format_t fmt_in;
619 video_format_Init( &fmt_in, 0 );
620
621 video_format_t fmt_out;
622 video_format_Init( &fmt_out, VLC_CODEC_YUVA );
623
624 image_handler_t *p_image = image_HandlerCreate( p_this );
625 if( !p_image )
626 return NULL;
627
628 char *psz_url = vlc_path2uri( psz_filename, NULL );
629 picture_t *p_pic = image_ReadUrl( p_image, psz_url, &fmt_in, &fmt_out );
630 free( psz_url );
631 image_HandlerDelete( p_image );
632
633 return p_pic;
634}

636/**
637 * It loads the logo images into memory.
638 *
639 * Read the logo-file input switch, obtaining a list of images and
640 * associated durations and transparencies. Store the image(s), and
641 * times. An image without a stated time or opacity will use the
642 * logo-delay and logo-opacity values.
643 */
644static void LogoListLoad( vlc_object_t *p_this, logo_list_t *p_logo_list,
645 const char *psz_filename )
646{
647 char *psz_list; /* the list: <logo>[,[<delay>[,[<alpha>]]]][;...] */
648 char *psz_original;
649 unsigned int i;
650 logo_t *p_logo; /* the parsing's result */
651
652 p_logo_list->i_counter = 0;
653 p_logo_list->i_next_pic = 0;
654
655 psz_original = psz_list = strdup( psz_filename );
656 if( !psz_list )
657 abort();
658
659 /* Count the number logos == number of ';' + 1 */
660 p_logo_list->i_count = 1;
661 for( i = 0; i < strlen( psz_list ); i++ )
662 {
663 if( psz_list[i] == ';' )
664 p_logo_list->i_count++;
665 }
666
667 p_logo_list->p_logo =
668 p_logo = calloc( p_logo_list->i_count, sizeof(*p_logo) );
669 if( !p_logo )
670 abort();
671
672 /* Fill the data */
673 for( i = 0; i < p_logo_list->i_count; i++ )
674 {
675 char *p_c = strchr( psz_list, ';' );
676 char *p_c2 = strchr( psz_list, ',' );
677
678 p_logo[i].i_alpha = -1; /* use default settings */
679 p_logo[i].i_delay = -1; /* use default settings */
680
681 if( p_c2 && ( p_c2 < p_c || !p_c ) )
682 {
683 /* <logo>,<delay>[,<alpha>] type */
684 if( p_c2[1] != ',' && p_c2[1] != ';' && p_c2[1] != '\0' )
685 p_logo[i].i_delay = atoi( p_c2+1 );
686 *p_c2 = '\0';
687 if( ( p_c2 = strchr( p_c2+1, ',' ) )
688 && ( p_c2 < p_c || !p_c ) && p_c2[1] != ';' && p_c2[1] != '\0' )
689 p_logo[i].i_alpha = atoi( p_c2 + 1 );
690 }
691 else
692 {
693 /* <logo> type */
694 if( p_c )
695 *p_c = '\0';
696 }
697
698 msg_Dbg( p_this, "logo file name %s, delay %d, alpha %d",
699 psz_list, p_logo[i].i_delay, p_logo[i].i_alpha );
700 p_logo[i].p_pic = LoadImage( p_this, psz_list );
701 if( !p_logo[i].p_pic )
702 {
703 msg_Warn( p_this, "error while loading logo %s, will be skipped",
704 psz_list );
705 }
706
707 if( p_c )
708 psz_list = &p_c[1];
709 }
710
711 /* initialize so that on the first update it will wrap back to 0 */
712 p_logo_list->i_counter = p_logo_list->i_count - 1;
713
714 free( psz_original );
715}

717/**
718 * Unload a list of logo and release associated ressources.
719 */
720static void LogoListUnload( logo_list_t *p_list )
721{
722 for( unsigned i = 0; i < p_list->i_count; i++ )
723 {
724 logo_t *p_logo = &p_list->p_logo[i];

732/**
733 * Go to the next logo and return its pointer.
734 */
735static logo_t *LogoListNext( logo_list_t *p_list, mtime_t i_date )
736{
737 p_list->i_counter = ( p_list->i_counter + 1 ) % p_list->i_count;
738
739 logo_t *p_logo = LogoListCurrent( p_list );
740
741 p_list->i_next_pic = i_date + ( p_logo->i_delay != -1 ?
742 p_logo->i_delay : p_list->i_delay ) * 1000;
743 return p_logo;
744}
