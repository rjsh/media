44/*****************************************************************************
45 * overlay_t: Overlay descriptor
46 *****************************************************************************/
47
48overlay_t *OverlayCreate( void )
49{
50 overlay_t *p_ovl = calloc( 1, sizeof( overlay_t ) );
51 if( p_ovl == NULL )
52 return NULL;
53
54 p_ovl->i_x = p_ovl->i_y = 0;
55 p_ovl->i_alpha = 0xFF;
56 p_ovl->b_active = false;
57 video_format_Setup( &p_ovl->format, VLC_FOURCC( '\0','\0','\0','\0') , 0, 0,
58 0, 0, 1, 1 );
59 p_ovl->p_fontstyle = text_style_New();
60 p_ovl->data.p_text = NULL;
61
62 return p_ovl;
63}
64
65int OverlayDestroy( overlay_t *p_ovl )
66{
67 free( p_ovl->data.p_text );
68 text_style_Delete( p_ovl->p_fontstyle );
69
70 return VLC_SUCCESS;
71}

76static int skip_space( char **psz_command )
77{
78 char *psz_temp = *psz_command;
79
80 while( isspace( (unsigned char)*psz_temp ) )
81 {
82 ++psz_temp;
83 }
84 if( psz_temp == *psz_command )
85 {
86 return VLC_EGENERIC;
87 }
88 *psz_command = psz_temp;
89 return VLC_SUCCESS;
90}
91
92static int parse_digit( char **psz_command, int32_t *value )
93{
94 char *psz_temp;
95 *value = strtol( *psz_command, &psz_temp, 10 );
96 if( psz_temp == *psz_command )
97 {
98 return VLC_EGENERIC;
99 }
100 *psz_command = psz_temp;
101 return VLC_SUCCESS;
102}
103
104static int parse_char( char **psz_command, char **psz_end,
105 int count, char *psz_value )
106{
107 if( *psz_end - *psz_command < count )
108 {
109 return VLC_EGENERIC;
110 }
111 memcpy( psz_value, *psz_command, count );
112 *psz_command += count;
113 return VLC_SUCCESS;
114}

116static int parser_DataSharedMem( char *psz_command,
117 char *psz_end,
118 commandparams_t *p_params )
119{
120 /* Parse: 0 128 128 RGBA 9404459 */
121 skip_space( &psz_command );
122 if( isdigit( (unsigned char)*psz_command ) )
123 {
124 if( parse_digit( &psz_command, &p_params->i_id ) == VLC_EGENERIC )
125 return VLC_EGENERIC;
126 }
127 skip_space( &psz_command );
128 if( isdigit( (unsigned char)*psz_command ) )
129 {
130 if( parse_digit( &psz_command, &p_params->i_width ) == VLC_EGENERIC )
131 return VLC_EGENERIC;
132 }
133 skip_space( &psz_command );
134 if( isdigit( (unsigned char)*psz_command ) )
135 {
136 if( parse_digit( &psz_command, &p_params->i_height ) == VLC_EGENERIC )
137 return VLC_EGENERIC;
138 }
139 skip_space( &psz_command );
140 if( isascii( (unsigned char)*psz_command ) )
141 {
142 if( parse_char( &psz_command, &psz_end, 4, (char*)&p_params->fourcc )
143 == VLC_EGENERIC )
144 return VLC_EGENERIC;
145 }
146 skip_space( &psz_command );
147 if( isdigit( (unsigned char)*psz_command ) )
148 {
149 if( parse_digit( &psz_command, &p_params->i_shmid ) == VLC_EGENERIC )
150 return VLC_EGENERIC;
151 }
152 return VLC_SUCCESS;
153}

240static int parser_SetTextColor( char *psz_command, char *psz_end,
241 commandparams_t *p_params )
242{
243 int r = 0, g = 0, b = 0;
244 VLC_UNUSED(psz_end);
245
246 skip_space( &psz_command );
247 if( isdigit( (unsigned char)*psz_command ) )
248 {
249 if( parse_digit( &psz_command, &p_params->i_id ) == VLC_EGENERIC )
250 return VLC_EGENERIC;
251 }
252 skip_space( &psz_command );
253 if( isdigit( (unsigned char)*psz_command ) )
254 {
255 if( parse_digit( &psz_command, &r ) == VLC_EGENERIC )
256 return VLC_EGENERIC;
257 }
258 skip_space( &psz_command );
259 if( isdigit( (unsigned char)*psz_command ) )
260 {
261 if( parse_digit( &psz_command, &g ) == VLC_EGENERIC )
262 return VLC_EGENERIC;
263 }
264 skip_space( &psz_command );
265 if( isdigit( (unsigned char)*psz_command ) )
266 {
267 if( parse_digit( &psz_command, &b ) == VLC_EGENERIC )
268 return VLC_EGENERIC;
269 }
270 p_params->fontstyle.i_font_color = (r<<16) | (g<<8) | (b<<0);
271 return VLC_SUCCESS;
272}

411static int exec_DataSharedMem( filter_t *p_filter,
412 const commandparams_t *p_params,
413 commandparams_t *p_results )
414{
415#if defined(HAVE_SYS_SHM_H)
416 filter_sys_t *p_sys = (filter_sys_t*) p_filter->p_sys;
417 struct shmid_ds shminfo;
418 overlay_t *p_ovl;
419 size_t i_size;
420
421 VLC_UNUSED(p_results);
422
423 p_ovl = ListGet( &p_sys->overlays, p_params->i_id );
424 if( p_ovl == NULL )
425 {
426 msg_Err( p_filter, "Invalid overlay: %d", p_params->i_id );
427 return VLC_EGENERIC;
428 }
429
430 if( shmctl( p_params->i_shmid, IPC_STAT, &shminfo ) == -1 )
431 {
432 msg_Err( p_filter, "Unable to access shared memory" );
433 return VLC_EGENERIC;
434 }
435 i_size = shminfo.shm_segsz;
436
437 if( p_params->fourcc == VLC_CODEC_TEXT )
438 {
439 char *p_data;
440
441 if( (p_params->i_height != 1) || (p_params->i_width < 1) )
442 {
443 msg_Err( p_filter,
444 "Invalid width and/or height. when specifying text height "
445 "must be 1 and width the number of bytes in the string, "
446 "including the null terminator" );
447 return VLC_EGENERIC;
448 }
449
450 if( (size_t)p_params->i_width > i_size )
451 {
452 msg_Err( p_filter,
453 "Insufficient data in shared memory. need %d, got %zu",
454 p_params->i_width, i_size );
455 return VLC_EGENERIC;
456 }
457
458 p_ovl->data.p_text = malloc( p_params->i_width );
459 if( p_ovl->data.p_text == NULL )
460 {
461 msg_Err( p_filter, "Unable to allocate string storage" );
462 return VLC_ENOMEM;
463 }
464
465 video_format_Setup( &p_ovl->format, VLC_CODEC_TEXT,
466 0, 0, 0, 0, 0, 1 );
467
468 p_data = shmat( p_params->i_shmid, NULL, SHM_RDONLY );
469 if( p_data == NULL )
470 {
471 msg_Err( p_filter, "Unable to attach to shared memory" );
472 free( p_ovl->data.p_text );
473 p_ovl->data.p_text = NULL;
474 return VLC_ENOMEM;
475 }
476 memcpy( p_ovl->data.p_text, p_data, p_params->i_width );
477
478 shmdt( p_data );
479 }
480 else
481 {
482 uint8_t *p_data, *p_in;
483 size_t i_neededsize = 0;
484
485 p_ovl->data.p_pic = picture_New( p_params->fourcc,
486 p_params->i_width, p_params->i_height,
487 1, 1 );
488 if( p_ovl->data.p_pic == NULL )
489 return VLC_ENOMEM;
490
491 p_ovl->format = p_ovl->data.p_pic->format;
492
493 for( size_t i_plane = 0; i_plane < (size_t)p_ovl->data.p_pic->i_planes;
494 ++i_plane )
495 {
496 i_neededsize += p_ovl->data.p_pic->p[i_plane].i_visible_lines *
497 p_ovl->data.p_pic->p[i_plane].i_visible_pitch;
498 }
499
500 if( i_neededsize > i_size )
501 {
502 msg_Err( p_filter,
503 "Insufficient data in shared memory. need %zu, got %zu",
504 i_neededsize, i_size );
505 picture_Release( p_ovl->data.p_pic );
506 p_ovl->data.p_pic = NULL;
507 return VLC_EGENERIC;
508 }
509
510 p_data = shmat( p_params->i_shmid, NULL, SHM_RDONLY );
511 if( p_data == NULL )
512 {
513 msg_Err( p_filter, "Unable to attach to shared memory" );
514 picture_Release( p_ovl->data.p_pic );
515 p_ovl->data.p_pic = NULL;
516 return VLC_ENOMEM;
517 }
518
519 p_in = p_data;
520 for( size_t i_plane = 0; i_plane < (size_t)p_ovl->data.p_pic->i_planes;
521 ++i_plane )
522 {
523 uint8_t *p_out = p_ovl->data.p_pic->p[i_plane].p_pixels;
524 for( size_t i_line = 0;
525 i_line < (size_t)p_ovl->data.p_pic->p[i_plane].i_visible_lines;
526 ++i_line )
527 {
528 memcpy( p_out, p_in,
529 p_ovl->data.p_pic->p[i_plane].i_visible_pitch );
530 p_out += p_ovl->data.p_pic->p[i_plane].i_pitch;
531 p_in += p_ovl->data.p_pic->p[i_plane].i_visible_pitch;
532 }
533 }
534 shmdt( p_data );
535 }
536 p_sys->b_updated = p_ovl->b_active;
537
538 return VLC_SUCCESS;
539#else
540 VLC_UNUSED(p_params);
541 VLC_UNUSED(p_results);
542
543 msg_Err( p_filter, "system doesn't support shared memory" );
544 return VLC_EGENERIC;
545#endif
546}

889void RegisterCommand( filter_t *p_filter )
890{
891 filter_sys_t *p_sys = (filter_sys_t*) p_filter->p_sys;
892 size_t i_index = 0;
893
894 p_sys->i_commands = ARRAY_SIZE(p_commands);
895 p_sys->pp_commands = (commanddesc_t **) calloc( p_sys->i_commands, sizeof(commanddesc_t*) );
896 if( !p_sys->pp_commands ) return;
897 for( i_index = 0; i_index < p_sys->i_commands; i_index ++ )
898 {
899 p_sys->pp_commands[i_index] = (commanddesc_t *) malloc( sizeof(commanddesc_t) );
900 if( !p_sys->pp_commands[i_index] ) return;
901 p_sys->pp_commands[i_index]->psz_command = strdup( p_commands[i_index].psz_command );
902 p_sys->pp_commands[i_index]->b_atomic = p_commands[i_index].b_atomic;
903 p_sys->pp_commands[i_index]->pf_parser = p_commands[i_index].pf_parser;
904 p_sys->pp_commands[i_index]->pf_execute = p_commands[i_index].pf_execute;
905 p_sys->pp_commands[i_index]->pf_unparse = p_commands[i_index].pf_unparse;
906 }
907
908 msg_Dbg( p_filter, "%zu commands are available", p_sys->i_commands );
909 for( size_t i_index = 0; i_index < p_sys->i_commands; i_index++ )
910 msg_Dbg( p_filter, " %s", p_sys->pp_commands[i_index]->psz_command );
911}
912
913void UnregisterCommand( filter_t *p_filter )
914{
915 filter_sys_t *p_sys = (filter_sys_t*) p_filter->p_sys;
916 size_t i_index = 0;
917
918 for( i_index = 0; i_index < p_sys->i_commands; i_index++ )
919 {
920 free( p_sys->pp_commands[i_index]->psz_command );
921 free( p_sys->pp_commands[i_index] );
922 }
923 free( p_sys->pp_commands );
924 p_sys->pp_commands = NULL;
925 p_sys->i_commands = 0;
926}
