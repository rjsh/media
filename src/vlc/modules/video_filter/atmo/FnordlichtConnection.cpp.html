54ATMO_BOOL CFnordlichtConnection::OpenConnection()
55{
56#if defined(_ATMO_VLC_PLUGIN_)
57 char *serdevice = m_pAtmoConfig->getSerialDevice();
58 if ( !serdevice )
59 return ATMO_FALSE;
60#else
61 int portNummer = m_pAtmoConfig->getComport();
62 m_dwLastWin32Error = 0;
63 if ( portNummer < 1 )
64 return ATMO_FALSE; // make no real sense;-)
65#endif
66
67 CloseConnection();
68
69#if !defined(_ATMO_VLC_PLUGIN_)
70 char serdevice[16]; // com4294967295
71 sprintf(serdevice,"com%d",portNummer);
72#endif
73
74#if defined(_WIN32)
75
76 m_hComport = CreateFileA(serdevice,
77 GENERIC_WRITE, 0, NULL,
78 OPEN_EXISTING, 0, NULL);
79 if ( m_hComport == INVALID_HANDLE_VALUE )
80 {
81 // we have a problem here can't open com port...
82 // somebody else may use it?
83 // m_dwLastWin32Error = GetLastError();
84 return ATMO_FALSE;
85 }
86 /* change serial settings (Speed, stopbits etc.) */
87 DCB dcb; // f√ºr comport-parameter
88 dcb.DCBlength = sizeof(DCB);
89 GetCommState(m_hComport, &dcb); // ger current serialport settings
90 dcb.BaudRate = 19200; // set speed
91 dcb.ByteSize = 8; // set databits
92 dcb.Parity = NOPARITY; // set parity
93 dcb.StopBits = ONESTOPBIT; // set one stop bit
94 SetCommState(m_hComport, &dcb); // apply settings
95
96#else
97
98 int bconst = B19200;
99 m_hComport = open(serdevice,O_RDWR |O_NOCTTY);
100 if ( m_hComport < 0 )
101 return ATMO_FALSE;
102
103 struct termios tio;
104 memset(&tio,0,sizeof(tio));
105 tio.c_cflag = (CS8 | CREAD | HUPCL | CLOCAL);
106 tio.c_iflag = (INPCK | BRKINT);
107 cfsetispeed(&tio, bconst);
108 cfsetospeed(&tio, bconst);
109 if( ! tcsetattr(m_hComport, TCSANOW, &tio) )
110 tcflush(m_hComport, TCIOFLUSH);
111 else {
112 // can't change parms
113 close(m_hComport);
114 m_hComport = -1;
115 return false;
116 }
117
118#endif
119
120 // sync fnordlicht
121 if ( sync() )
122 // stop fading on all devices
123 if ( stop(255) )
124 return true; // fnordlicht initialized...
125
126 return false; // something is going wrong...
127}

129void CFnordlichtConnection::CloseConnection()
130{
131 if ( m_hComport != INVALID_HANDLE_VALUE )
132 {
133 reset(255);
134
135#if defined(_WIN32)
136 CloseHandle(m_hComport);
137#else
138 close(m_hComport);
139#endif
140 m_hComport = INVALID_HANDLE_VALUE;
141 }

149/*
150 def fade_rgb(addr, r, g, b, step, delay)
151 $dev.write addr.chr
152 $dev.write "\x01"
153 $dev.write step.chr
154 $dev.write delay.chr
155 $dev.write r.chr
156 $dev.write g.chr
157 $dev.write b.chr
158 $dev.write "\x00\x00\x00\x00\x00"
159 $dev.write "\x00\x00\x00"
160 $dev.flush
161 end
162*/
163ATMO_BOOL CFnordlichtConnection::SendData(pColorPacket data)
164{
165 if ( m_hComport == INVALID_HANDLE_VALUE )
166 return ATMO_FALSE;
167
168 int amount = getAmountFnordlichter();
169 unsigned char buffer[15];
170 memset(&buffer, 0, sizeof(buffer) ); // zero buffer
171 int iBytesWritten;
172
173 Lock();
174
175 buffer[1] = 0x01; // fade to rgb color
176 buffer[2] = 0x80; // in two steps
177 buffer[3] = 0x01; // 1ms pause between steps
178
179 // send all packages to all fnordlicht's
180 for( unsigned char i=0; i < amount; i++ )
181 {
182 int idx;
183 if ( m_ChannelAssignment && i < m_NumAssignedChannels )
184 idx = m_ChannelAssignment[i];
185 else
186 idx = -1; // no channel assigned to fnordlicht[i]
187
188 if( idx >= 0 && idx <= data->numColors )
189 {
190 // fnordlicht address equals to a MoMo Channel
191 buffer[0] = i; // fnordlicht address (0..254, 255 = broadcast)
192 buffer[4] = data->zone[idx].r;
193 buffer[5] = data->zone[idx].g;
194 buffer[6] = data->zone[idx].b;
195 }
196
197#if defined(_WIN32)
198 // send to COM-Port
199 WriteFile( m_hComport, buffer, sizeof(buffer),
200 (DWORD*)&iBytesWritten, NULL );
201#else
202 iBytesWritten = write(m_hComport, buffer, sizeof(buffer));
203 tcflush(m_hComport, TCIOFLUSH);
204 tcdrain(m_hComport); // flush buffer
205#endif
206
207 if (iBytesWritten != sizeof(buffer))
208 {
209 Unlock();
210 return ATMO_FALSE; // shouldn't be...
211 }
212
213 }
214
215 Unlock();
216
217 return ATMO_TRUE;
218}

238/*
239 def sync(addr = 0)
240 1.up to(15) do
241 $dev.write "\e"
242 end
243 $dev.write addr.chr
244 $dev.flush
245 end
246*/
247ATMO_BOOL CFnordlichtConnection::sync(void)
248{
249 if ( m_hComport == INVALID_HANDLE_VALUE )
250 return ATMO_FALSE;
251
252 unsigned char buffer[16];
253
254 int iBytesWritten;
255
256 Lock();
257
258 // fill buffer with 15 escape character
259 memset(&buffer, 0x1b, sizeof(buffer)-1);
260
261 buffer[sizeof(buffer)-1] = 0x00; // append one zero byte
262
263#if defined(_WIN32)
264 // send to COM-Port
265 WriteFile( m_hComport, buffer, sizeof(buffer),
266 (DWORD*)&iBytesWritten, NULL );
267#else
268 iBytesWritten = write(m_hComport, buffer, sizeof(buffer));
269 tcflush(m_hComport, TCIOFLUSH);
270 tcdrain(m_hComport); // flush buffer
271#endif
272
273 Unlock();
274
275 return (iBytesWritten == sizeof(buffer)) ? ATMO_TRUE : ATMO_FALSE;
276
277}

290ATMO_BOOL CFnordlichtConnection::stop(unsigned char addr)
291{
292 if(m_hComport == INVALID_HANDLE_VALUE)
293 return ATMO_FALSE;
294
295 unsigned char buffer[15];
296 memset(&buffer, 0, sizeof(buffer)); // zero buffer
297 int iBytesWritten;
298
299 Lock();
300
301 buffer[0] = addr; // fnordlicht address (255 = broadcast)
302 buffer[1] = 0x08; // stop command
303 buffer[2] = 1; // fading
304
305#if defined(_WIN32)
306 // send to COM-Port
307 WriteFile( m_hComport, buffer, sizeof(buffer),
308 (DWORD*)&iBytesWritten, NULL );
309#else
310 iBytesWritten = write(m_hComport, buffer, sizeof(buffer));
311 tcflush(m_hComport, TCIOFLUSH);
312 tcdrain(m_hComport); // flush buffer
313#endif
314
315 Unlock();
316
317 return (iBytesWritten == sizeof(buffer)) ? ATMO_TRUE : ATMO_FALSE;
318}

322ATMO_BOOL CFnordlichtConnection::reset(unsigned char addr)
323{
324 if(m_hComport == INVALID_HANDLE_VALUE)
325 return ATMO_FALSE;
326
327 stop(255);
328
329 if ( sync() && start_bootloader(addr) )
330 {
331#if defined(_ATMO_VLC_PLUGIN_)
332 do_sleep(200000); // wait 200ms
333#else
334 do_sleep(200); // wait 200ms
335#endif
336 if ( sync() && boot_enter_application(addr) )
337 return ATMO_TRUE;
338 }
339
340 return ATMO_FALSE;
341}
