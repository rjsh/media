37ATMO_BOOL CAtmoClassicConnection::OpenConnection() {
38#if defined(_ATMO_VLC_PLUGIN_)
39 char *serdevice = m_pAtmoConfig->getSerialDevice();
40 if(!serdevice)
41 return ATMO_FALSE;
42#else
43 int portNummer = m_pAtmoConfig->getComport();
44 m_dwLastWin32Error = 0;
45 if(portNummer < 1) return ATMO_FALSE; // make no real sense;-)
46#endif
47
48 CloseConnection();
49
50#if !defined(_ATMO_VLC_PLUGIN_)
51 char serdevice[16]; // com4294967295
52 sprintf(serdevice,"com%d",portNummer);
53#endif
54
55#if defined(_WIN32)
56
57 m_hComport = CreateFileA(serdevice, GENERIC_WRITE, 0, NULL, OPEN_EXISTING, 0, NULL);
58 if(m_hComport == INVALID_HANDLE_VALUE) {
59// we have a problem here can't open com port... somebody else may use it?
60// m_dwLastWin32Error = GetLastError();
61 return ATMO_FALSE;
62 }
63 /* change serial settings (Speed, stopbits etc.) */
64 DCB dcb; // f√ºr comport-parameter
65 dcb.DCBlength = sizeof(DCB);
66 GetCommState (m_hComport, &dcb); // ger current serialport settings
67 dcb.BaudRate = 38400; // set speed
68 dcb.ByteSize = 8; // set databits
69 dcb.Parity = NOPARITY; // set parity
70 dcb.StopBits = ONESTOPBIT; // set one stop bit
71 SetCommState (m_hComport, &dcb); // apply settings
72
73#else
74
75 int bconst = B38400;
76 m_hComport = open(serdevice,O_RDWR |O_NOCTTY);
77 if(m_hComport < 0) {
78 return ATMO_FALSE;
79 }
80
81 struct termios tio;
82 memset(&tio,0,sizeof(tio));
83 tio.c_cflag = (CS8 | CREAD | HUPCL | CLOCAL);
84 tio.c_iflag = (INPCK | BRKINT);
85 cfsetispeed(&tio, bconst);
86 cfsetospeed(&tio, bconst);
87 if(!tcsetattr(m_hComport, TCSANOW, &tio)) {
88 tcflush(m_hComport, TCIOFLUSH);
89 } else {
90 // can't change parms
91 close(m_hComport);
92 m_hComport = -1;
93 return false;
94 }
95
96#endif
97
98 return true;
99}

116ATMO_BOOL CAtmoClassicConnection::HardwareWhiteAdjust(int global_gamma,
117 int global_contrast,
118 int contrast_red,
119 int contrast_green,
120 int contrast_blue,
121 int gamma_red,
122 int gamma_green,
123 int gamma_blue,
124 ATMO_BOOL storeToEeprom) {
125 if(m_hComport == INVALID_HANDLE_VALUE)
126 return ATMO_FALSE;
127
128 DWORD iBytesWritten;
129/*
130[0] = 255
131[1] = 00
132[2] = 00
133[3] = 101
134
135[4] brightness 0..255 ?
136
137[5] Contrast Red 11 .. 100
138[6] Contrast Green 11 .. 100
139[7] Contrast Blue 11 .. 100
140
141[8] Gamma Red 11 .. 35
142[9] Gamma Red 11 .. 35
143[10] Gamma Red 11 .. 35
144
145[11] Globale Contrast 11 .. 100
146
147[12] Store Data: 199 (else 0)
148
149*/
150 unsigned char sendBuffer[16];
151 sendBuffer[0] = 0xFF;
152 sendBuffer[1] = 0x00;
153 sendBuffer[2] = 0x00;
154 sendBuffer[3] = 101;
155
156 sendBuffer[4] = (global_gamma & 255);
157
158 sendBuffer[5] = (contrast_red & 255);
159 sendBuffer[6] = (contrast_green & 255);
160 sendBuffer[7] = (contrast_blue & 255);
161
162 sendBuffer[8] = (gamma_red & 255);
163 sendBuffer[9] = (gamma_green & 255);
164 sendBuffer[10] = (gamma_blue & 255);
165
166 sendBuffer[11] = (global_contrast & 255);
167
168 if(storeToEeprom == ATMO_TRUE)
169 sendBuffer[12] = 199; // store to eeprom!
170 else
171 sendBuffer[12] = 0;
172
173#if defined(_WIN32)
174 WriteFile(m_hComport, sendBuffer, 13, &iBytesWritten, NULL); // send to COM-Port
175#else
176 iBytesWritten = write(m_hComport, sendBuffer, 13);
177 tcdrain(m_hComport);
178#endif
179
180 return (iBytesWritten == 13) ? ATMO_TRUE : ATMO_FALSE;
181}

184ATMO_BOOL CAtmoClassicConnection::SendData(pColorPacket data) {
185 if(m_hComport == INVALID_HANDLE_VALUE)
186 return ATMO_FALSE;
187
188 unsigned char buffer[19];
189 DWORD iBytesWritten;
190
191 buffer[0] = 0xFF; // Start Byte
192 buffer[1] = 0x00; // Start channel 0
193 buffer[2] = 0x00; // Start channel 0
194 buffer[3] = 15; //
195 int iBuffer = 4;
196 int idx;
197
198 Lock();
199
200 for(int i=0; i < 5 ; i++) {
201 if(m_ChannelAssignment && (i < m_NumAssignedChannels))
202 idx = m_ChannelAssignment[i];
203 else
204 idx = -1;
205 if((idx>=0) && (idx<data->numColors)) {
206 buffer[iBuffer++] = data->zone[idx].r;
207 buffer[iBuffer++] = data->zone[idx].g;
208 buffer[iBuffer++] = data->zone[idx].b;
209 } else {
210 buffer[iBuffer++] = 0;
211 buffer[iBuffer++] = 0;
212 buffer[iBuffer++] = 0;
213 }
214 }
215
216#if defined(_WIN32)
217 WriteFile(m_hComport, buffer, 19, &iBytesWritten, NULL); // send to COM-Port
218#else
219 iBytesWritten = write(m_hComport, buffer, 19);
220 tcdrain(m_hComport);
221#endif
222
223 Unlock();
224
225 return (iBytesWritten == 19) ? ATMO_TRUE : ATMO_FALSE;
226}

255 case 2:

258 case 3:

260 break;
261 case 4:
262 sprintf(buf,"Unterer Kanal [%d]",ch);
263 break;
264 default:
265 sprintf(buf,"Kanal [%d]",ch);
266 break;
267 }
268
269 return strdup(buf);
270}
