132void CAtmoPacketQueue::AddPacket(pColorPacket newPacket)
133{
134 pColorPacketItem temp = new ColorPacketItem;
135 temp->packet = newPacket;
136 temp->next = NULL;
137#if defined(_ATMO_VLC_PLUGIN_)
138 temp->tickcount = mdate();
139#else
140 temp->tickcount = GetTickCount();
141#endif
142
143 Lock();
144 if(m_last) {
145 m_last->next = temp;
146 m_last = temp;
147 } else {
148 m_last = temp;
149 m_first = temp;
150 }
151 Unlock();
152 SignalEvent();
153}
154
155pColorPacketItem CAtmoPacketQueue::GetNextPacketContainer()
156{
157 pColorPacketItem temp = NULL;
158
159 Lock();
160 if(m_first) {
161 temp = m_first;
162 m_first = m_first->next;
163 if(!m_first)
164 m_last = NULL;
165 temp->next = NULL;
166 }
167 Unlock();
168
169 return temp;
170}
171
172pColorPacket CAtmoPacketQueue::GetNextPacket()
173{
174 pColorPacketItem item = GetNextPacketContainer();
175 if(item) {
176 pColorPacket temp = item->packet;
177 delete item;
178 return(temp);
179 } else
180 return(NULL);
181}

200#if defined(_ATMO_VLC_PLUGIN_)
201pColorPacket CAtmoPacketQueue::GetNextPacket(mtime_t timecode, ATMO_BOOL withWait, vlc_object_t *p_this, mtime_t &packet_time)
202#else
203pColorPacket CAtmoPacketQueue::GetNextPacket(DWORD timecode, ATMO_BOOL withWait, DWORD &packet_time)
204#endif
205{
206#if !defined(_ATMO_VLC_PLUGIN_)
207 if(timecode & 0x80000000) // GetTickCount - delay < 0 ;-)
208 return NULL;
209#endif
210
211 int timeDiff;
212
213 while(1)
214 {
215 Lock();
216 if(!m_first) {
217 Unlock();
218 break;
219 }
220 timeDiff = m_first->tickcount - timecode;
221 packet_time = m_first->tickcount;
222 Unlock();
223
224 if(timeDiff >= MAX_PACKET_TOO_EARLY) // packet should be process in 35ms or later (usually we are to early for it)
225 {
226 if( !withWait )
227 break;
228 }
229 else
230 {
231 if(timeDiff <= MAX_PACKET_TOO_LATE) {
232 // we are more than -35ms too late for this packet, skip it and throw it away!
233#if defined(_ATMO_VLC_PLUGIN_)
234 msg_Dbg( p_this, "getNextPacket skip late %d ms", timeDiff / 1000 );
235#endif
236 pColorPacket skip = GetNextPacket();
237 delete (char *)skip;
238
239 m_skipcounter++;
240 m_avgDelay += abs(timeDiff);
241
242 continue;
243 }
244 }
245
246 if(withWait && timeDiff > MIN_SLEEP_TIME)
247 {
248 // if this is a sync call, to get in sync with frame source again we wait untils its time!
249#if defined(_ATMO_VLC_PLUGIN_)
250 msg_Dbg( p_this, "getNextPacket Sleep %d ms", timeDiff / 1000 );
251#endif
252 do_sleep( timeDiff );
253
254 m_avgWait += timeDiff;
255 m_waitcounter++;
256 }
257
258 m_framecounter++;
259#if !defined(_ATMO_VLC_PLUGIN_)
260 if(m_StatusMonitor)
261 {
262 if(withWait)
263 m_StatusMonitor->UpdateValues(m_waitcounter, m_skipcounter, m_framecounter, m_nullpackets, m_avgWait, m_avgDelay);
264 }
265#endif
266
267 return GetNextPacket();
268 }
269
270 m_nullpackets++;
271#if !defined(_ATMO_VLC_PLUGIN_)
272 if(m_StatusMonitor)
273 {
274 if(withWait)
275 m_StatusMonitor->UpdateValues(m_waitcounter, m_skipcounter, m_framecounter, m_nullpackets, m_avgWait, m_avgDelay);
276 }
277#endif
278 return NULL;
279}

281ATMO_BOOL CAtmoPacketQueue::WaitForNextPacket(DWORD timeout)
282{
283 UnSignalEvent();
284
285#if !defined(_ATMO_VLC_PLUGIN_)
286
287 return ( WaitForSingleObject( m_hPacketArrivedEvent, timeout ) == WAIT_OBJECT_0 );
288
289#else
290
291 mtime_t maxWait = mdate() + timeout * 1000;
292
293 vlc_mutex_lock( &m_PacketArrivedLock );
294 m_PacketArrived = ATMO_FALSE;
295 while(vlc_cond_timedwait( &m_PacketArrivedCond, &m_PacketArrivedLock, maxWait) == 0)
296 {
297 /*
298 condition was set -> but may be an old signal from previous AddPacket
299 which is still left - so if m_PacketArrived is still false, wait again
300 */
301 if(mdate() >= maxWait)
302 break;
303 if( m_PacketArrived )
304 break;
305 }
306 vlc_mutex_unlock( &m_PacketArrivedLock );
307 return m_PacketArrived;
308
309#endif
310}

312void CAtmoPacketQueue::ClearQueue()
313{
314 pColorPacketItem next;
315
316 Lock();
317
318 while(m_first)
319 {
320 next = m_first->next;
321 delete (char *)(m_first->packet);
322 delete m_first;
323 m_first = next;
324 }
325 m_last = NULL;
326
327 m_waitcounter = 0;
328 m_skipcounter = 0;
329 m_framecounter = 0;
330
331 m_avgWait = 0;
332 m_avgDelay = 0;
333 m_nullpackets = 0;
334
335 Unlock();
336}
