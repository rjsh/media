18CAtmoConnection::CAtmoConnection(CAtmoConfig *cfg)
19{
20 this->m_pAtmoConfig = cfg;
21 m_ChannelAssignment = NULL;
22 m_NumAssignedChannels = 0;
23
24#if defined(_ATMO_VLC_PLUGIN_)
25 vlc_mutex_init( &m_AccessConnection );
26#else
27 InitializeCriticalSection( &m_AccessConnection );
28#endif
29}
30
31CAtmoConnection::~CAtmoConnection(void)
32{
33 if(isOpen())
34 CloseConnection();
35
36#if defined(_ATMO_VLC_PLUGIN_)
37 vlc_mutex_destroy( &m_AccessConnection );
38#else
39 DeleteCriticalSection( &m_AccessConnection );
40#endif
41}
42
43void CAtmoConnection::SetChannelAssignment(CAtmoChannelAssignment *ca)
44{
45 if(ca)
46 {
47 Lock();
48 delete m_ChannelAssignment;
49 m_ChannelAssignment = ca->getMapArrayClone(m_NumAssignedChannels);
50 Unlock();
51 }
52}

55ATMO_BOOL CAtmoConnection::ShowConfigDialog(HINSTANCE hInst, HWND parent, CAtmoConfig *cfg)

63void CAtmoConnection::Lock()
64{
65#if defined(_ATMO_VLC_PLUGIN_)
66 vlc_mutex_lock( &m_AccessConnection );
67#else
68 EnterCriticalSection( &m_AccessConnection );
69#endif
70}
71void CAtmoConnection::Unlock()
72{
73#if defined(_ATMO_VLC_PLUGIN_)
74 vlc_mutex_unlock( &m_AccessConnection );
75#else
76 LeaveCriticalSection( &m_AccessConnection );
77#endif
78}
