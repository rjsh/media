107static const char *const ppsz_filter_options[] = {
108 "croptop", "cropbottom", "cropleft", "cropright",
109 "paddtop", "paddbottom", "paddleft", "paddright",
110 NULL
111};
112
113struct filter_sys_t
114{
115 int i_croptop;
116 int i_cropbottom;
117 int i_cropleft;
118 int i_cropright;
119 int i_paddtop;
120 int i_paddbottom;
121 int i_paddleft;
122 int i_paddright;
123};
124
125/*****************************************************************************
126 * OpenFilter: probe the filter and return score
127 *****************************************************************************/
128static int OpenFilter( vlc_object_t *p_this )
129{
130 filter_t *p_filter = (filter_t*)p_this;
131 filter_sys_t *p_sys;
132
133 if( !p_filter->b_allow_fmt_out_change )
134 {
135 msg_Err( p_filter, "Picture format change isn't allowed" );
136 return VLC_EGENERIC;
137 }
138
139 if( p_filter->fmt_in.video.i_chroma != p_filter->fmt_out.video.i_chroma )
140 {
141 msg_Err( p_filter, "Input and output chromas don't match" );
142 /* In fact we don't really care about this since we're allowed
143 * to change the output format ... FIXME? */
144 return VLC_EGENERIC;
145 }
146
147 const vlc_chroma_description_t *p_chroma =
148 vlc_fourcc_GetChromaDescription( p_filter->fmt_in.video.i_chroma );
149 if( p_chroma == NULL || p_chroma->plane_count == 0 )
150 return VLC_EGENERIC;
151
152 p_filter->p_sys = (filter_sys_t *)malloc( sizeof( filter_sys_t ) );
153 if( !p_filter->p_sys ) return VLC_ENOMEM;
154
155 config_ChainParse( p_filter, CFG_PREFIX, ppsz_filter_options,
156 p_filter->p_cfg );
157
158 p_sys = p_filter->p_sys;
159#define GET_OPTION( name ) \
160 p_sys->i_ ## name = var_CreateGetInteger( p_filter, CFG_PREFIX #name ); \
161 if( p_sys->i_ ## name & 1 ) \
162 msg_Warn( p_filter, "Using even values for `" #name "' is recommended" );
163 GET_OPTION( croptop )
164 GET_OPTION( cropbottom )
165 GET_OPTION( cropleft )
166 GET_OPTION( cropright )
167 GET_OPTION( paddtop )
168 GET_OPTION( paddbottom )
169 GET_OPTION( paddleft )
170 GET_OPTION( paddright )
171
172 p_filter->fmt_out.video.i_height =
173 p_filter->fmt_out.video.i_visible_height =
174 p_filter->fmt_in.video.i_visible_height
175 - p_sys->i_croptop - p_sys->i_cropbottom
176 + p_sys->i_paddtop + p_sys->i_paddbottom;
177
178 p_filter->fmt_out.video.i_width =
179 p_filter->fmt_out.video.i_visible_width =
180 p_filter->fmt_in.video.i_visible_width
181 - p_sys->i_cropleft - p_sys->i_cropright
182 + p_sys->i_paddleft + p_sys->i_paddright;
183
184 p_filter->pf_video_filter = Filter;
185
186 msg_Dbg( p_filter, "Crop: Top: %d, Bottom: %d, Left: %d, Right: %d",
187 p_sys->i_croptop, p_sys->i_cropbottom, p_sys->i_cropleft,
188 p_sys->i_cropright );
189 msg_Dbg( p_filter, "Padd: Top: %d, Bottom: %d, Left: %d, Right: %d",
190 p_sys->i_paddtop, p_sys->i_paddbottom, p_sys->i_paddleft,
191 p_sys->i_paddright );
192 msg_Dbg( p_filter, "%dx%d -> %dx%d",
193 p_filter->fmt_in.video.i_width,
194 p_filter->fmt_in.video.i_height,
195 p_filter->fmt_out.video.i_width,
196 p_filter->fmt_out.video.i_height );
197
198 return VLC_SUCCESS;
199}

201/*****************************************************************************
202 * CloseFilter: clean up the filter

204static void CloseFilter( vlc_object_t *p_this )
205{
206 filter_t *p_filter = (filter_t *)p_this;
207 free( p_filter->p_sys );
208}

210/****************************************************************************
211 * Filter: the whole thing
212 ****************************************************************************/
213static picture_t *Filter( filter_t *p_filter, picture_t *p_pic )
214{
215 filter_sys_t *p_sys = p_filter->p_sys;
216 picture_t *p_outpic;
217 int i_plane;
218 int i_width, i_height, i_xcrop, i_ycrop,
219 i_outwidth, i_outheight, i_xpadd, i_ypadd;
220
221 const int p_padd_color[] = { 0x00, 0x80, 0x80, 0xff };
222
223 if( !p_pic ) return NULL;
224
225 /* Request output picture */
226 p_outpic = filter_NewPicture( p_filter );
227 if( !p_outpic )
228 {
229 picture_Release( p_pic );
230 return NULL;
231 }
232
233 for( i_plane = 0; i_plane < p_pic->i_planes; i_plane++ )
234 /* p_pic and p_outpic have the same chroma/number of planes but that's
235 * about it. */
236 {
237 plane_t *p_plane = p_pic->p+i_plane;
238 plane_t *p_outplane = p_outpic->p+i_plane;
239 uint8_t *p_in = p_plane->p_pixels;
240 uint8_t *p_out = p_outplane->p_pixels;
241 int i_pixel_pitch = p_plane->i_pixel_pitch;
242 int i_padd_color = i_plane > 3 ? p_padd_color[0]
243 : p_padd_color[i_plane];
244
245 /* These assignments assume that the first plane always has
246 * a width and height equal to the picture's */
247 i_width = ( ( p_filter->fmt_in.video.i_visible_width
248 - p_sys->i_cropleft - p_sys->i_cropright )
249 * p_plane->i_visible_pitch )
250 / p_pic->p->i_visible_pitch;
251 i_height = ( ( p_filter->fmt_in.video.i_visible_height
252 - p_sys->i_croptop - p_sys->i_cropbottom )
253 * p_plane->i_visible_lines )
254 / p_pic->p->i_visible_lines;
255 i_xcrop = ( p_sys->i_cropleft * p_plane->i_visible_pitch)
256 / p_pic->p->i_visible_pitch;
257 i_ycrop = ( p_sys->i_croptop * p_plane->i_visible_lines)
258 / p_pic->p->i_visible_lines;
259 i_outwidth = ( p_filter->fmt_out.video.i_visible_width
260 * p_outplane->i_visible_pitch )
261 / p_outpic->p->i_visible_pitch;
262 i_outheight = ( p_filter->fmt_out.video.i_visible_height
263 * p_outplane->i_visible_lines )
264 / p_outpic->p->i_visible_lines;
265 i_xpadd = ( p_sys->i_paddleft * p_outplane->i_visible_pitch )
266 / p_outpic->p->i_visible_pitch;
267 i_ypadd = ( p_sys->i_paddtop * p_outplane->i_visible_lines )
268 / p_outpic->p->i_visible_lines;
269
270 /* Crop the top */
271 p_in += i_ycrop * p_plane->i_pitch;
272
273 /* Padd on the top */
274 memset( p_out, i_padd_color, i_ypadd * p_outplane->i_pitch );
275 p_out += i_ypadd * p_outplane->i_pitch;
276
277 int i_line;
278 for( i_line = 0; i_line < i_height; i_line++ )
279 {
280 uint8_t *p_in_next = p_in + p_plane->i_pitch;
281 uint8_t *p_out_next = p_out + p_outplane->i_pitch;
282
283 /* Crop on the left */
284 p_in += i_xcrop * i_pixel_pitch;
285
286 /* Padd on the left */
287 memset( p_out, i_padd_color, i_xpadd * i_pixel_pitch );
288 p_out += i_xpadd * i_pixel_pitch;
289
290 /* Copy the image and crop on the right */
291 memcpy( p_out, p_in, i_width * i_pixel_pitch );
292 p_out += i_width * i_pixel_pitch;
293 p_in += i_width * i_pixel_pitch;
294
295 /* Padd on the right */
296 memset( p_out, i_padd_color,
297 ( i_outwidth - i_xpadd - i_width ) * i_pixel_pitch );
298
299 /* Got to begining of the next line */
300 p_in = p_in_next;
301 p_out = p_out_next;
302 }
303
304 /* Padd on the bottom */
305 memset( p_out, i_padd_color,
306 ( i_outheight - i_ypadd - i_height ) * p_outplane->i_pitch );
307 }
308
309 return CopyInfoAndRelease( p_outpic, p_pic );
310}
