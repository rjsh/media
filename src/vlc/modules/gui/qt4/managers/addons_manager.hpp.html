38class AddonManagerEvent : public QEvent
39{
40public:
41 static const QEvent::Type AddedEvent;
42 static const QEvent::Type ChangedEvent;
43 static const QEvent::Type DiscoveryEndedEvent;
44
45 AddonManagerEvent( QEvent::Type type, addon_entry_t *_p_entry )
46 : QEvent( type ), p_entry( _p_entry )
47 {
48 addon_entry_Hold( p_entry );
49 }
50 virtual ~AddonManagerEvent()
51 {
52 addon_entry_Release( p_entry );
53 }
54
55 addon_entry_t *entry() const { return p_entry; }
56
57private:
58 addon_entry_t *p_entry;
59};
60
61class AddonsManager : public QObject, public Singleton<AddonsManager>
62{
63 Q_OBJECT
64 friend class Singleton<AddonsManager>;
65
66public:
67 AddonsManager( intf_thread_t * );
68 virtual ~AddonsManager();
69 static void addonsEventsCallback( const vlc_event_t *, void * );
70 void customEvent( QEvent * );
71 void install( QByteArray id );
72 void remove( QByteArray id );
73 static QString getAddonType( int );
74
75signals:
76 void addonAdded( addon_entry_t * );
77 void addonChanged( const addon_entry_t * );
78 void discoveryEnded();
79
80public slots:
81 void findNewAddons();
82 void findDesignatedAddon( QString uri );
83 void findInstalled();
84
85private:
86 addons_manager_t* p_manager;
87};
