86 new DialogHandler (p_intf, this );
87}

118void DialogsProvider::customEvent( QEvent *event )
119{
120 if( event->type() == DialogEvent::DialogEvent_Type )
121 {
122 DialogEvent *de = static_cast<DialogEvent*>(event);
123 switch( de->i_dialog )
124 {
125 case INTF_DIALOG_FILE_SIMPLE:
126 case INTF_DIALOG_FILE:
127 openDialog(); break;
128 case INTF_DIALOG_FILE_GENERIC:
129 openFileGenericDialog( de->p_arg ); break;
130 case INTF_DIALOG_DISC:
131 openDiscDialog(); break;
132 case INTF_DIALOG_NET:
133 openNetDialog(); break;
134 case INTF_DIALOG_SAT:
135 case INTF_DIALOG_CAPTURE:
136 openCaptureDialog(); break;
137 case INTF_DIALOG_DIRECTORY:
138 PLAppendDir(); break;
139 case INTF_DIALOG_PLAYLIST:
140 playlistDialog(); break;
141 case INTF_DIALOG_MESSAGES:
142 messagesDialog(); break;
143 case INTF_DIALOG_FILEINFO:
144 mediaInfoDialog(); break;
145 case INTF_DIALOG_PREFS:
146 prefsDialog(); break;
147 case INTF_DIALOG_BOOKMARKS:
148 bookmarksDialog(); break;
149 case INTF_DIALOG_EXTENDED:
150 extendedDialog(); break;
151#ifdef ENABLE_VLM
152 case INTF_DIALOG_VLM:
153 vlmDialog(); break;
154#endif
155 case INTF_DIALOG_POPUPMENU:
156 VLCMenuBar::PopupMenu( p_intf, (de->i_arg != 0) ); break;
157 case INTF_DIALOG_AUDIOPOPUPMENU:
158 VLCMenuBar::AudioPopupMenu( p_intf, (de->i_arg != 0) ); break;
159 case INTF_DIALOG_VIDEOPOPUPMENU:
160 VLCMenuBar::VideoPopupMenu( p_intf, (de->i_arg != 0) ); break;
161 case INTF_DIALOG_MISCPOPUPMENU:
162 VLCMenuBar::MiscPopupMenu( p_intf, (de->i_arg != 0) ); break;
163 case INTF_DIALOG_WIZARD:
164 case INTF_DIALOG_STREAMWIZARD:
165 openAndStreamingDialogs(); break;
166#ifdef UPDATE_CHECK
167 case INTF_DIALOG_UPDATEVLC:
168 updateDialog(); break;
169#endif
170 case INTF_DIALOG_EXIT:
171 quit(); break;
172 default:
173 msg_Warn( p_intf, "unimplemented dialog" );
174 }
175 }
176}

288/* Generic open file */
289void DialogsProvider::openFileGenericDialog( intf_dialog_args_t *p_arg )
290{
291 if( p_arg == NULL )
292 {
293 msg_Warn( p_intf, "openFileGenericDialog() called with NULL arg" );
294 return;
295 }
296
297 /* Replace the extensions to a Qt format */
298 int i = 0;
299 QString extensions = qfu( p_arg->psz_extensions );
300 while ( ( i = extensions.indexOf( "|", i ) ) != -1 )
301 {
302 if( ( extensions.count( "|" ) % 2 ) == 0 )
303 extensions.replace( i, 1, ");;" );
304 else
305 extensions.replace( i, 1, "(" );
306 }
307 extensions.replace( ";*", " *" );
308 extensions.append( ")" );
309
310 /* Save */
311 if( p_arg->b_save )
312 {
313 QString file = QFileDialog::getSaveFileName( NULL, p_arg->psz_title,
314 p_intf->p_sys->filepath, extensions );
315 if( !file.isEmpty() )
316 {
317 p_arg->i_results = 1;
318 p_arg->psz_results = (char **)malloc( p_arg->i_results * sizeof( char * ) );
319 p_arg->psz_results[0] = strdup( qtu( toNativeSepNoSlash( file ) ) );
320 }
321 else
322 p_arg->i_results = 0;
323 }
324 else /* non-save mode */
325 {
326 QStringList files = QFileDialog::getOpenFileNames( NULL,
327 p_arg->psz_title, p_intf->p_sys->filepath,
328 extensions );
329 p_arg->i_results = files.count();
330 p_arg->psz_results = (char **)malloc( p_arg->i_results * sizeof( char * ) );
331 i = 0;
332 foreach( const QString &file, files )
333 p_arg->psz_results[i++] = strdup( qtu( toNativeSepNoSlash( file ) ) );
334 if(i == 0)
335 p_intf->p_sys->filepath = QString::fromLatin1("");
336 else
337 p_intf->p_sys->filepath = qfu( p_arg->psz_results[i-1] );
338 }
339
340 /* Callback */
341 if( p_arg->pf_callback )
342 p_arg->pf_callback( p_arg );
343
344 /* Clean afterwards */
345 if( p_arg->psz_results )
346 {
347 for( i = 0; i < p_arg->i_results; i++ )
348 free( p_arg->psz_results[i] );
349 free( p_arg->psz_results );
350 }
351 free( p_arg->psz_title );
352 free( p_arg->psz_extensions );
353 free( p_arg );
354}

399/**
400 * Simple open
401 ***/
402QStringList DialogsProvider::showSimpleOpen( const QString& help,
403 int filters,
404 const QString& path )
405{
406 QString fileTypes = "";
407 if( filters & EXT_FILTER_MEDIA ) {
408 ADD_EXT_FILTER( fileTypes, EXTENSIONS_MEDIA );
409 }
410 if( filters & EXT_FILTER_VIDEO ) {
411 ADD_EXT_FILTER( fileTypes, EXTENSIONS_VIDEO );
412 }
413 if( filters & EXT_FILTER_AUDIO ) {
414 ADD_EXT_FILTER( fileTypes, EXTENSIONS_AUDIO );
415 }
416 if( filters & EXT_FILTER_PLAYLIST ) {
417 ADD_EXT_FILTER( fileTypes, EXTENSIONS_PLAYLIST );
418 }
419 if( filters & EXT_FILTER_SUBTITLE ) {
420 ADD_EXT_FILTER( fileTypes, EXTENSIONS_SUBTITLE );
421 }
422 ADD_EXT_FILTER( fileTypes, EXTENSIONS_ALL );
423 fileTypes.replace( ";*", " *");
424
425 QStringList files = QFileDialog::getOpenFileNames( NULL,
426 help.isEmpty() ? qtr(I_OP_SEL_FILES ) : help,
427 path.isEmpty() ? p_intf->p_sys->filepath : path,
428 fileTypes );
429
430 if( !files.isEmpty() ) savedirpathFromFile( files.last() );
431
432 return files;
433}

459/* Url & Clipboard */
460/**
461 * Open a MRL.
462 * If the clipboard contains URLs, the first is automatically 'preselected'.
463 **/
464void DialogsProvider::openUrlDialog()
465{
466 OpenUrlDialog oud( p_intf );
467 if( oud.exec() != QDialog::Accepted )
468 return;
469
470 QString url = oud.url();
471 if( url.isEmpty() )
472 return;
473
474 if( !url.contains( qfu( "://" ) ) )
475 {
476 char *uri = vlc_path2uri( qtu( url ), NULL );
477 if( uri == NULL )
478 return;
479 url = qfu(uri);
480 free( uri );
481 }
482
483 Open::openMRL( p_intf, qtu(url), !oud.shouldEnqueue() );
484}

499QString DialogsProvider::getDirectoryDialog( intf_thread_t *p_intf )
500{
501 QString dir = QFileDialog::getExistingDirectory( NULL,
502 qtr( I_OP_DIR_WINTITLE ), p_intf->p_sys->filepath );
503
504 if( dir.isEmpty() ) return QString();
505
506 p_intf->p_sys->filepath = dir;
507
508 const char *scheme = "directory";
509 if( dir.endsWith( DIR_SEP "VIDEO_TS", Qt::CaseInsensitive ) )
510 scheme = "dvd";
511 else if( dir.endsWith( DIR_SEP "BDMV", Qt::CaseInsensitive ) )
512 {
513 scheme = "bluray";
514 dir.remove( "BDMV" );
515 }
516
517 char *uri = vlc_path2uri( qtu( toNativeSeparators( dir ) ), scheme );
518 if( unlikely(uri == NULL) )
519 return QString();
520
521 dir = qfu( uri );
522 free( uri );
523
524 RecentsMRL::getInstance( p_intf )->addRecent( dir );
525
526 return dir;
527}

552void DialogsProvider::saveAPlaylist(playlist_t *p_playlist, playlist_item_t *p_node)
553{
554 static const struct
555 {
556 char filter_name[14];
557 char filter_patterns[5];
558 char module[12];
559 } types[] = {
560 { N_("XSPF playlist"), "xspf", "export-xspf", },
561 { N_("M3U playlist"), "m3u", "export-m3u", },
562 { N_("M3U8 playlist"), "m3u8", "export-m3u8", },
563 { N_("HTML playlist"), "html", "export-html", },
564 };
565
566 QStringList filters;
567 QString ext = getSettings()->value( "last-playlist-ext" ).toString();
568
569 for( size_t i = 0; i < sizeof (types) / sizeof (types[0]); i++ )
570 {
571 QString tmp = qfu( vlc_gettext( types[i].filter_name ) ) + " (*." + types[i].filter_patterns + ")";
572 if( ext == qfu( types[i].filter_patterns ) )
573 filters.insert( 0, tmp );
574 else
575 filters.append( tmp );
576 }
577
578 QString selected;
579 QString file = QFileDialog::getSaveFileName( NULL,
580 qtr( "Save playlist as..." ),
581 p_intf->p_sys->filepath, filters.join( ";;" ),
582 &selected );
583 const char *psz_selected_module = NULL;
584 const char *psz_last_playlist_ext = NULL;
585
586 if( file.isEmpty() )
587 return;
588
589 /* First test if the file extension is set, and different to selected filter */
590 for( size_t i = 0; i < sizeof (types) / sizeof (types[0]); i++)
591 {
592 if ( file.endsWith( QString( "." ) + qfu( types[i].filter_patterns ) ) )
593 {
594 psz_selected_module = types[i].module;
595 psz_last_playlist_ext = types[i].filter_patterns;
596 break;
597 }
598 }
599
600 /* otherwise apply the selected extension */
601 if ( !psz_last_playlist_ext )
602 {
603 for( size_t i = 0; i < sizeof (types) / sizeof (types[0]); i++)
604 {
605 if ( selected.startsWith( qfu( vlc_gettext( types[i].filter_name ) ) ) )
606 {
607 psz_selected_module = types[i].module;
608 psz_last_playlist_ext = types[i].filter_patterns;
609 /* Fix file extension */
610 file = file.append( QString( "." ) + qfu( psz_last_playlist_ext ) );
611 break;
612 }
613 }
614 }
615
616 if ( psz_selected_module )
617 {
618 playlist_Export( p_playlist, qtu( toNativeSeparators( file ) ),
619 p_node, psz_selected_module );
620 getSettings()->setValue( "last-playlist-ext", psz_last_playlist_ext );
621 }
622}

647void DialogsProvider::streamingDialog( QWidget *parent,
648 const QString& mrl,
649 bool b_transcode_only,
650 QStringList options )
651{
652 QString soutoption;
653
654 /* Stream */
655 if( !b_transcode_only )
656 {
657 SoutDialog *s = new SoutDialog( parent, p_intf, mrl );
658 s->setAttribute( Qt::WA_QuitOnClose, false ); // See #4883
659 if( s->exec() == QDialog::Accepted )
660 {
661 soutoption = s->getMrl();
662 delete s;
663 }
664 else
665 {
666 delete s; return;
667 }
668 } else {
669 /* Convert */
670 ConvertDialog *s = new ConvertDialog( parent, p_intf, mrl );
671 s->setAttribute( Qt::WA_QuitOnClose, false ); // See #4883
672 if( s->exec() == QDialog::Accepted )
673 {
674 soutoption = s->getMrl();
675 delete s;
676 }
677 else
678 {
679 delete s; return;
680 }
681 }
682
683 /* Get SoutMRL */
684 if( !soutoption.isEmpty() )
685 {
686 options += soutoption.split( " :");
687
688 Open::openMRLwithOptions( p_intf, mrl, &options, true, true, _("Streaming") );
689 }

704void DialogsProvider::loadSubtitlesFile()
705{
706 input_thread_t *p_input = THEMIM->getInput();
707 if( !p_input ) return;
708
709 input_item_t *p_item = input_GetItem( p_input );
710 if( !p_item ) return;
711
712 char *path = input_item_GetURI( p_item );
713 char *path2 = NULL;
714 if( path )
715 {
716 path2 = make_path( path );
717 free( path );
718 if( path2 )
719 {
720 char *sep = strrchr( path2, DIR_SEP_CHAR );
721 if( sep ) *sep = '\0';
722 }
723 }
724
725 QStringList qsl = showSimpleOpen( qtr( "Open subtitles..." ),
726 EXT_FILTER_SUBTITLE,
727 qfu( path2 ) );
728 free( path2 );
729 foreach( const QString &qsFile, qsl )
730 {
731 if( input_AddSubtitleOSD( p_input, qtu( toNativeSeparators( qsFile ) ),
732 true, true ) )
733 msg_Warn( p_intf, "unable to load subtitles from '%s'",
734 qtu( qsFile ) );
735 }
736}
