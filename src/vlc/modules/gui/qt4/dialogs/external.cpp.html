41DialogHandler::DialogHandler (intf_thread_t *p_intf, QObject *_parent)
42 : QObject( _parent ), intf (p_intf),
43 critical (VLC_OBJECT(p_intf), "dialog-critical"),
44 login (VLC_OBJECT(p_intf), "dialog-login"),
45 question (VLC_OBJECT(p_intf), "dialog-question"),
46 progressBar (VLC_OBJECT(p_intf), "dialog-progress-bar")
47{
48 var_Create (intf, "dialog-error", VLC_VAR_ADDRESS);
49 var_AddCallback (intf, "dialog-error", error, this);
50 connect (this, SIGNAL(error(const QString &, const QString &)),
51 SLOT(displayError(const QString &, const QString &)));
52
53 critical.addCallback(this, SLOT(displayCritical(void *)),
54 Qt::BlockingQueuedConnection);
55 login.addCallback(this, SLOT(requestLogin(void *)),
56 Qt::BlockingQueuedConnection);
57 question.addCallback(this, SLOT(requestAnswer(void *)),
58 Qt::BlockingQueuedConnection);
59 progressBar.addCallback(this, SLOT(startProgressBar(void *)),
60 Qt::BlockingQueuedConnection);
61
62 dialog_Register (intf);
63}
64
65DialogHandler::~DialogHandler (void)
66{
67 dialog_Unregister (intf);
68
69 var_DelCallback (intf, "dialog-error", error, this);
70 var_Destroy (intf, "dialog-error");
71}

73int DialogHandler::error (vlc_object_t *obj, const char *,
74 vlc_value_t, vlc_value_t value, void *data)
75{
76 const dialog_fatal_t *dialog = (const dialog_fatal_t *)value.p_address;
77 DialogHandler *self = static_cast<DialogHandler *>(data);
78
79 if (var_InheritBool (obj, "qt-error-dialogs"))
80 emit self->error (qfu(dialog->title), qfu(dialog->message));
81 return VLC_SUCCESS;
82}

89void DialogHandler::displayCritical (void *value)

97void DialogHandler::requestLogin (void *value)
98{
99 dialog_login_t *data = (dialog_login_t *)value;
100 QDialog *dialog = new QDialog;
101 QLayout *layout = new QVBoxLayout (dialog);
102
103 dialog->setWindowTitle (qfu(data->title));
104 dialog->setWindowRole ("vlc-login");
105 layout->setMargin (2);
106
107 /* User name and password fields */
108 QWidget *panel = new QWidget (dialog);
109 QGridLayout *grid = new QGridLayout;
110 grid->addWidget (new QLabel (qfu(data->message)), 0, 0, 1, 2);
111
112 QLineEdit *userLine = new QLineEdit;
113 grid->addWidget (new QLabel (qtr("User name")), 1, 0);
114 grid->addWidget (userLine, 1, 1);
115
116 QLineEdit *passLine = new QLineEdit;
117 passLine->setEchoMode (QLineEdit::Password);
118 grid->addWidget (new QLabel (qtr("Password")), 2, 0);
119 grid->addWidget (passLine, 2, 1);
120
121 panel->setLayout (grid);
122 layout->addWidget (panel);
123
124 /* OK, Cancel buttons */
125 QDialogButtonBox *buttonBox = new QDialogButtonBox;
126 QPushButton *okButton = new QPushButton( "&Ok" );
127 QPushButton *cancelButton = new QPushButton( "&Cancel" );
128 buttonBox->addButton( okButton, QDialogButtonBox::AcceptRole );
129 buttonBox->addButton( cancelButton, QDialogButtonBox::RejectRole );
130
131 CONNECT( buttonBox, accepted(), dialog, accept() );
132 CONNECT( buttonBox, rejected(), dialog, reject() );
133 layout->addWidget (buttonBox);
134
135 /* Run the dialog */
136 dialog->setLayout (layout);
137
138 if (dialog->exec ())
139 {
140 *data->username = strdup (qtu(userLine->text ()));
141 *data->password = strdup (qtu(passLine->text ()));
142 }
143 else
144 *data->username = *data->password = NULL;
145
146 delete dialog;
147}

149void DialogHandler::requestAnswer (void *value)
150{
151 dialog_question_t *data = (dialog_question_t *)value;
152
153 QMessageBox *box = new QMessageBox (QMessageBox::Question,
154 qfu(data->title), qfu(data->message));
155 QAbstractButton *yes = (data->yes != NULL)
156 ? box->addButton ("&" + qfu(data->yes), QMessageBox::YesRole) : NULL;
157 QAbstractButton *no = (data->no != NULL)
158 ? box->addButton ("&" + qfu(data->no), QMessageBox::NoRole) : NULL;
159 if (data->cancel != NULL)
160 box->addButton ("&" + qfu(data->cancel), QMessageBox::RejectRole);
161
162 box->exec ();
163
164 int answer;
165 if (box->clickedButton () == yes)
166 answer = 1;
167 else
168 if (box->clickedButton () == no)
169 answer = 2;
170 else
171 answer = 3;
172
173 delete box;
174 data->answer = answer;
175}

178QVLCProgressDialog::QVLCProgressDialog (DialogHandler *parent,
179 struct dialog_progress_bar_t *data)
180 : QProgressDialog (qfu(data->message),
181 data->cancel ? ("&" + qfu(data->cancel)) : 0, 0, 1000),
182 handler (parent),
183 cancelled (false)
184{
185 if (data->cancel)
186 setWindowModality (Qt::ApplicationModal);
187 if (data->title != NULL)
188 setWindowTitle (qfu(data->title));
189
190 setWindowRole ("vlc-progress");
191 setValue( 0 );
192
193 connect (this, SIGNAL(progressed(int)), SLOT(setValue(int)));
194 connect (this, SIGNAL(described(const QString&)),
195 SLOT(setLabelText(const QString&)));
196 connect (this, SIGNAL(canceled(void)), SLOT(saveCancel(void)));
197 connect (this, SIGNAL(released(void)), SLOT(deleteLater(void)));
198
199 data->pf_update = update;
200 data->pf_check = check;
201 data->pf_destroy = destroy;
202 data->p_sys = this;
203}

205void QVLCProgressDialog::update (void *priv, const char *text, float value)
206{
207 QVLCProgressDialog *self = static_cast<QVLCProgressDialog *>(priv);
208
209 if (text != NULL)
210 emit self->described (qfu(text));
211 emit self->progressed ((int)(value * 1000.));
212}

214static QMutex cancel_mutex;
215
216bool QVLCProgressDialog::check (void *priv)
217{
218 QVLCProgressDialog *self = static_cast<QVLCProgressDialog *>(priv);
219 QMutexLocker locker (&cancel_mutex);
220 return self->cancelled;
221}
222
223void QVLCProgressDialog::destroy (void *priv)
224{
225 QVLCProgressDialog *self = static_cast<QVLCProgressDialog *>(priv);
226
227 emit self->released ();
228}
229
230void QVLCProgressDialog::saveCancel (void)
231{
232 QMutexLocker locker (&cancel_mutex);
233 cancelled = true;
234}
