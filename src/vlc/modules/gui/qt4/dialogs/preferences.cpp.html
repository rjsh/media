50PrefsDialog::PrefsDialog( QWidget *parent, intf_thread_t *_p_intf )
51 : QVLCDialog( parent, _p_intf )
52{
53 QGridLayout *main_layout = new QGridLayout( this );
54 setWindowTitle( qtr( "Preferences" ) );
55 setWindowRole( "vlc-preferences" );
56 setWindowModality( Qt::WindowModal );
57
58 /* Whether we want it or not, we need to destroy on close to get
59 consistency when reset */
60 setAttribute( Qt::WA_DeleteOnClose );
61
62 /* Create Panels */
63 simple_tree_panel = new QWidget;
64 simple_tree_panel->setLayout( new QVBoxLayout );
65
66 advanced_tree_panel = new QWidget;
67 advanced_tree_panel->setLayout( new QVBoxLayout );
68
69 /* Choice for types */
70 types = new QGroupBox( qtr("Show settings") );
71 types->setAlignment( Qt::AlignHCenter );
72 QHBoxLayout *types_l = new QHBoxLayout;
73 types_l->setSpacing( 3 ); types_l->setMargin( 3 );
74 simple = new QRadioButton( qtr( "Simple" ), types );
75 simple->setToolTip( qtr( "Switch to simple preferences view" ) );
76 types_l->addWidget( simple );
77 all = new QRadioButton( qtr("All"), types ); types_l->addWidget( all );
78 all->setToolTip( qtr( "Switch to full preferences view" ) );
79 types->setLayout( types_l );
80 simple->setChecked( true );
81
82 /* Tree and panel initialisations */
83 advanced_tree = NULL;
84 tree_filter = NULL;
85 current_filter = NULL;
86 simple_tree = NULL;
87 simple_panels_stack = new QStackedWidget;
88 advanced_panels_stack = new QStackedWidget;
89
90 /* Buttons */
91 QDialogButtonBox *buttonsBox = new QDialogButtonBox();
92 QPushButton *save = new QPushButton( qtr( "&Save" ) );
93 save->setToolTip( qtr( "Save and close the dialog" ) );
94 QPushButton *cancel = new QPushButton( qtr( "&Cancel" ) );
95 QPushButton *reset = new QPushButton( qtr( "&Reset Preferences" ) );
96
97 buttonsBox->addButton( save, QDialogButtonBox::AcceptRole );
98 buttonsBox->addButton( cancel, QDialogButtonBox::RejectRole );
99 buttonsBox->addButton( reset, QDialogButtonBox::ResetRole );
100
101 simple_split_widget = new QWidget();
102 simple_split_widget->setLayout( new QVBoxLayout );
103
104 advanced_split_widget = new QSplitter();
105
106 stack = new QStackedWidget();
107 stack->insertWidget( SIMPLE, simple_split_widget );
108 stack->insertWidget( ADVANCED, advanced_split_widget );
109
110 simple_split_widget->layout()->addWidget( simple_tree_panel );
111 simple_split_widget->layout()->addWidget( simple_panels_stack );
112 simple_split_widget->layout()->setMargin( 0 );
113
114 advanced_split_widget->addWidget( advanced_tree_panel );
115 advanced_split_widget->addWidget( advanced_panels_stack );
116
117 /* Layout */
118 main_layout->addWidget( stack, 0, 0, 3, 3 );
119 main_layout->addWidget( types, 3, 0, 2, 1 );
120 main_layout->addWidget( buttonsBox, 4, 2, 1 ,1 );
121 main_layout->setRowStretch( 2, 4 );
122 main_layout->setMargin( 9 );
123 setLayout( main_layout );
124
125 /* Margins */
126 simple_tree_panel->layout()->setMargin( 1 );
127 simple_panels_stack->layout()->setContentsMargins( 6, 0, 0, 3 );
128
129 for( int i = 0; i < SPrefsMax ; i++ ) simple_panels[i] = NULL;
130
131 if( var_InheritBool( p_intf, "qt-advanced-pref" )
132 || var_InheritBool( p_intf, "advanced" ) )
133 setAdvanced();
134 else
135 setSimple();
136
137 BUTTONACT( save, save() );
138 BUTTONACT( cancel, cancel() );
139 BUTTONACT( reset, reset() );
140
141 BUTTONACT( simple, setSimple() );
142 BUTTONACT( all, setAdvanced() );
143
144 QVLCTools::restoreWidgetPosition( p_intf, "Preferences", this, QSize( 800 , 700 ) );
145}

147void PrefsDialog::setAdvanced()
148{
149 if ( !tree_filter )
150 {
151 tree_filter = new SearchLineEdit( simple_tree_panel );
152 tree_filter->setMinimumHeight( 26 );
153
154 CONNECT( tree_filter, textChanged( const QString & ),
155 this, advancedTreeFilterChanged( const QString & ) );
156
157 advanced_tree_panel->layout()->addWidget( tree_filter );
158
159 current_filter = new QCheckBox( qtr("Only show current") );
160 current_filter->setToolTip(
161 qtr("Only show modules related to current playback") );
162 CONNECT( current_filter, stateChanged(int),
163 this, onlyLoadedToggled() );
164 advanced_tree_panel->layout()->addWidget( current_filter );
165
166 QShortcut *search = new QShortcut( QKeySequence( QKeySequence::Find ), tree_filter );
167 CONNECT( search, activated(), tree_filter, setFocus() );
168 }
169
170 /* If don't have already and advanced TREE, then create it */
171 if( !advanced_tree )
172 {
173 /* Creation */
174 advanced_tree = new PrefsTree( p_intf, simple_tree_panel );
175 /* and connections */
176 CONNECT( advanced_tree,
177 currentItemChanged( QTreeWidgetItem *, QTreeWidgetItem * ),
178 this, changeAdvPanel( QTreeWidgetItem * ) );
179 advanced_tree_panel->layout()->addWidget( advanced_tree );
180 advanced_tree_panel->setSizePolicy( QSizePolicy::Maximum, QSizePolicy::Preferred );
181 }
182
183 /* If no advanced Panel exist, create one, attach it and show it*/
184 if( advanced_panels_stack->count() < 1 )
185 {
186 AdvPrefsPanel *insert = new AdvPrefsPanel( advanced_panels_stack );
187 advanced_panels_stack->insertWidget( 0, insert );
188 }
189
190 /* Select the first Item of the preferences. Maybe you want to select a specified
191 category... */
192 advanced_tree->setCurrentIndex(
193 advanced_tree->model()->index( 0, 0, QModelIndex() ) );
194
195 all->setChecked( true );
196 stack->setCurrentIndex( ADVANCED );
197 setWindowTitle( qtr( "Advanced Preferences" ) );
198}

200void PrefsDialog::setSimple()
201{
202 /* If no simple_tree, create one, connect it */
203 if( !simple_tree )
204 {
205 simple_tree = new SPrefsCatList( p_intf, simple_tree_panel );
206 CONNECT( simple_tree,
207 currentItemChanged( int ),
208 this, changeSimplePanel( int ) );
209 simple_tree_panel->layout()->addWidget( simple_tree );
210 simple_tree_panel->setSizePolicy( QSizePolicy::Fixed, QSizePolicy::Preferred );
211 }
212
213 if( ! simple_panels[SPrefsDefaultCat] )
214 changeSimplePanel( SPrefsDefaultCat );
215
216 simple->setChecked( true );
217 stack->setCurrentIndex( SIMPLE );
218 setWindowTitle( qtr( "Simple Preferences" ) );
219}

221/* Switching from on simple panel to another */
222void PrefsDialog::changeSimplePanel( int number )
223{
224 if( ! simple_panels[number] )
225 {
226 SPrefsPanel *insert = new SPrefsPanel( p_intf, simple_panels_stack, number ) ;
227 simple_panels_stack->insertWidget( number, insert );
228 simple_panels[number] = insert;
229 }
230 simple_panels_stack->setCurrentWidget( simple_panels[number] );
231}
232
233/* Changing from one Advanced Panel to another */
234void PrefsDialog::changeAdvPanel( QTreeWidgetItem *item )
235{
236 if( item == NULL ) return;
237 PrefsItemData *data = item->data( 0, Qt::UserRole ).value<PrefsItemData*>();
238
239 if( !data->panel )
240 {
241 data->panel = new AdvPrefsPanel( p_intf, advanced_panels_stack, data );
242 advanced_panels_stack->insertWidget( advanced_panels_stack->count(),
243 data->panel );
244 }
245 advanced_panels_stack->setCurrentWidget( data->panel );
246}

282/* Actual apply and save for the preferences */
283void PrefsDialog::save()
284{
285 if( simple->isChecked() && simple_tree->isVisible() )
286 {
287 msg_Dbg( p_intf, "Saving the simple preferences" );
288 for( int i = 0 ; i< SPrefsMax; i++ ){
289 if( simple_panels_stack->widget(i) )
290 qobject_cast<SPrefsPanel *>(simple_panels_stack->widget(i))->apply();
291 }
292 }
293 else if( all->isChecked() && advanced_tree->isVisible() )
294 {
295 msg_Dbg( p_intf, "Saving the advanced preferences" );
296 advanced_tree->applyAll();
297 }
298
299 /* Save to file */
300 if( config_SaveConfigFile( p_intf ) != 0 )
301 {
302 ErrorsDialog::getInstance (p_intf)->addError( qtr( "Cannot save Configuration" ),
303 qtr("Preferences file could not be saved") );
304 }
305
306 if( p_intf->p_sys->p_mi )
307 p_intf->p_sys->p_mi->reloadPrefs();
308 accept();
309
310 QVLCTools::saveWidgetPosition( p_intf, "Preferences", this );
311
312}

322/* Reset all the preferences, when you click the button */
323void PrefsDialog::reset()
324{
325 int ret = QMessageBox::question(
326 this,
327 qtr( "Reset Preferences" ),
328 qtr( "Are you sure you want to reset your VLC media player preferences?" ),
329 QMessageBox::Ok | QMessageBox::Cancel,
330 QMessageBox::Ok);
331
332 if( ret == QMessageBox::Ok )
333 {
334 config_ResetAll( p_intf );
335 config_SaveConfigFile( p_intf );
336 getSettings()->clear();
337
338#ifdef _WIN32
339 simple_panels[0]->cleanLang();
340#endif
341
342 accept();
343 }
344}
