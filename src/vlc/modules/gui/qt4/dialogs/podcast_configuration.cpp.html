29PodcastConfigDialog::PodcastConfigDialog( intf_thread_t *_p_intf)
30 : QVLCDialog( (QWidget*)_p_intf->p_sys->p_mi, _p_intf )
31
32{
33 ui.setupUi( this );
34
35 ui.podcastDelete->setToolTip( qtr( "Deletes the selected item" ) );
36 QPushButton *okButton = new QPushButton( qtr( "&Close" ), this );
37 QPushButton *cancelButton = new QPushButton( qtr( "&Cancel" ), this );
38 ui.okCancel->addButton( okButton, QDialogButtonBox::AcceptRole );
39 ui.okCancel->addButton( cancelButton, QDialogButtonBox::RejectRole );
40
41 CONNECT( ui.podcastAdd, clicked(), this, add() );
42 CONNECT( ui.podcastDelete, clicked(), this, remove() );
43 CONNECT( okButton, clicked(), this, close() );
44
45 char *psz_urls = config_GetPsz( p_intf, "podcast-urls" );
46 if( psz_urls )
47 {
48 char *psz_url = psz_urls;
49 while( 1 )
50 {
51 char *psz_tok = strchr( psz_url, '|' );
52 if( psz_tok ) *psz_tok = '\0';
53 ui.podcastList->addItem( psz_url );
54 if( psz_tok ) psz_url = psz_tok+1;
55 else break;
56 }
57 free( psz_urls );
58 }
59}
60
61PodcastConfigDialog::~PodcastConfigDialog()
62{
63}
64
65void PodcastConfigDialog::accept()
66{
67 QString urls = "";
68 for( int i = 0; i < ui.podcastList->count(); i++ )
69 {
70 urls += ui.podcastList->item(i)->text();
71 if( i != ui.podcastList->count()-1 ) urls += "|";
72 }
73 config_PutPsz( p_intf, "podcast-urls", qtu( urls ) );
74
75 if( playlist_IsServicesDiscoveryLoaded( THEPL, "podcast" ) )
76 {
77 var_SetString( THEPL, "podcast-urls", qtu( urls ) );
78 msg_Dbg( p_intf, "You will need to reload the podcast module to take into account deleted podcast urls" );
79 }
80}
81
82void PodcastConfigDialog::add()
83{
84 if( ui.podcastURL->text() != QString( "" ) )
85 {
86 ui.podcastList->addItem( ui.podcastURL->text() );
87 ui.podcastURL->clear();
88 }
89}
90
91void PodcastConfigDialog::remove()
92{
93 delete ui.podcastList->currentItem();
94}
