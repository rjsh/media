149inline void PluginTab::FillTree()
150{
151 size_t count;
152 module_t **p_list = module_list_get( &count );
153
154 for( unsigned int i = 0; i < count; i++ )
155 {
156 module_t *p_module = p_list[i];
157
158 QStringList qs_item;
159 qs_item << qfu( module_get_name( p_module, true ) )
160 << qfu( module_get_capability( p_module ) )
161 << QString::number( module_get_score( p_module ) );
162#ifndef DEBUG
163 if( qs_item.at(1).isEmpty() ) continue;
164#endif
165
166 QTreeWidgetItem *item = new PluginTreeItem( qs_item );
167 treePlugins->addTopLevelItem( item );
168 }
169 module_list_free( p_list );
170}

172void PluginTab::search( const QString& qs )
173{
174 QList<QTreeWidgetItem *> items = treePlugins->findItems( qs, Qt::MatchContains );
175 items += treePlugins->findItems( qs, Qt::MatchContains, 1 );
176
177 QTreeWidgetItem *item = NULL;
178 for( int i = 0; i < treePlugins->topLevelItemCount(); i++ )
179 {
180 item = treePlugins->topLevelItem( i );
181 item->setHidden( !items.contains( item ) );
182 }
183}

201bool PluginTreeItem::operator< ( const QTreeWidgetItem & other ) const
202{
203 int col = treeWidget()->sortColumn();
204 if( col == PluginTab::SCORE )
205 return text( col ).toInt() < other.text( col ).toInt();
206 else if ( col == PluginTab::CAPABILITY )
207 {
208 if ( text( PluginTab::CAPABILITY ) == other.text( PluginTab::CAPABILITY ) )
209 return text( PluginTab::NAME ) < other.text( PluginTab::NAME );
210 else
211 return text( PluginTab::CAPABILITY ) < other.text( PluginTab::CAPABILITY );
212 }
213 return text( col ) < other.text( col );
214}

300static QPixmap hueRotate( QImage image, const QColor &source, const QColor &target )
301{
302 int distance = target.hue() - source.hue();
303 /* must be indexed as we alter palette, not a whole pic */
304 Q_ASSERT( image.colorCount() );
305 if ( target.isValid() )
306 {
307 /* color 1 = transparency */
308 for ( int i=1; i < image.colorCount(); i++ )
309 {
310 QColor color = image.color( i );
311 int newhue = color.hue() + distance;
312 if ( newhue < 0 ) newhue += 255;
313 color.setHsv( newhue, color.saturation(), color.value(), color.alpha() );
314 image.setColor( i, color.rgba() );
315 }
316 }
317 return QPixmap::fromImage( image );
318}

486bool AddonsTab::eventFilter( QObject *obj, QEvent *event )
487{
488 if ( obj != addonsView->viewport() )
489 return false;
490
491 switch( event->type() )
492 {
493 case QEvent::Paint:
494 if ( spinnerAnimation->state() == PixmapAnimator::Running )
495 {
496 QWidget *viewport = qobject_cast<QWidget *>( obj );
497 if ( !viewport ) break;
498 QStylePainter painter( viewport );
499 QPixmap *spinner = spinnerAnimation->getPixmap();
500 QPoint point = viewport->geometry().center();
501 point -= QPoint( spinner->size().width() / 2, spinner->size().height() / 2 );
502 painter.drawPixmap( point, *spinner );
503 QString text = qtr("Retrieving addons...");
504 QSize textsize = fontMetrics().size( 0, text );
505 point = viewport->geometry().center();
506 point -= QPoint( textsize.width() / 2, -spinner->size().height() );
507 painter.drawText( point, text );
508 }
509 else if ( addonsModel->rowCount() == 0 )
510 {
511 QWidget *viewport = qobject_cast<QWidget *>( obj );
512 if ( !viewport ) break;
513 QStylePainter painter( viewport );
514 QString text = qtr("No addons found");
515 QSize size = fontMetrics().size( 0, text );
516 QPoint point = viewport->geometry().center();
517 point -= QPoint( size.width() / 2, size.height() / 2 );
518 painter.drawText( point, text );
519 }
520 break;
521 case QEvent::Show:
522 if ( !b_localdone && addonsView->model()->rowCount() < 1 )
523 {
524 b_localdone = true;
525 AddonsManager *AM = AddonsManager::getInstance( p_intf );
526 AM->findInstalled();
527 }
528 break;
529 case QEvent::DragEnter:
530 {
531 QDragEnterEvent *dragEvent = static_cast<QDragEnterEvent *>(event);
532 if ( !dragEvent ) break;
533 QList<QUrl> urls = dragEvent->mimeData()->urls();
534 if ( dragEvent->proposedAction() != Qt::CopyAction
535 || urls.count() != 1
536 || urls.first().scheme() != "file"
537 || ! urls.first().path().endsWith(".vlp") )
538 return false;
539 dragEvent->acceptProposedAction();
540 return true;
541 }
542 case QEvent::DragMove:
543 {
544 QDragMoveEvent *moveEvent = static_cast<QDragMoveEvent *>(event);
545 if ( !moveEvent ) break;
546 if ( moveEvent->proposedAction() != Qt::CopyAction )
547 return false;
548 moveEvent->acceptProposedAction();
549 return true;
550 }
551 case QEvent::Drop:
552 {
553 QDropEvent *dropEvent = static_cast<QDropEvent *>(event);
554 if ( !dropEvent ) break;
555 if ( dropEvent->proposedAction() != Qt::CopyAction )
556 return false;
557 if ( dropEvent->mimeData()->urls().count() )
558 {
559 AddonsManager *AM = AddonsManager::getInstance( p_intf );
560 AM->findDesignatedAddon( dropEvent->mimeData()->urls().first().toString() );
561 dropEvent->acceptProposedAction();
562 }
563 return true;
564 }
565 default:
566 break;
567 }
568 return false;
569}

601/* Safe copy of the extension_t struct */
602ExtensionListModel::ExtensionCopy::ExtensionCopy( extension_t *p_ext )
603{
604 name = qfu( p_ext->psz_name );
605 description = qfu( p_ext->psz_description );
606 shortdesc = qfu( p_ext->psz_shortdescription );
607 if( description.isEmpty() )
608 description = shortdesc;
609 if( shortdesc.isEmpty() && !description.isEmpty() )
610 shortdesc = description;
611 title = qfu( p_ext->psz_title );
612 author = qfu( p_ext->psz_author );
613 version = qfu( p_ext->psz_version );
614 url = qfu( p_ext->psz_url );
615 icon = loadPixmapFromData( p_ext->p_icondata, p_ext->i_icondata_size );
616}

641 return name;

643 return QVariant();

670
671void ExtensionListModel::updateList()
672{
673 ExtensionCopy *ext;
674
675 // Clear extensions list
676 while( !extensions.isEmpty() )
677 {
678 ext = extensions.takeLast();
679 delete ext;
680 }
681
682 // Find new extensions
683 extensions_manager_t *p_mgr = EM->getManager();
684 if( !p_mgr )
685 return;
686
687 vlc_mutex_lock( &p_mgr->lock );
688 extension_t *p_ext;
689 FOREACH_ARRAY( p_ext, p_mgr->extensions )
690 {
691 ext = new ExtensionCopy( p_ext );
692 extensions.append( ext );
693 }
694 FOREACH_END()
695 vlc_mutex_unlock( &p_mgr->lock );
696 vlc_object_release( p_mgr );
697
698 emit dataChanged( index( 0 ), index( rowCount() - 1 ) );
699}
700
701int ExtensionListModel::rowCount( const QModelIndex& ) const
702{
703 int count = 0;
704 extensions_manager_t *p_mgr = EM->getManager();
705 if( !p_mgr )
706 return 0;
707
708 vlc_mutex_lock( &p_mgr->lock );
709 count = p_mgr->extensions.i_size;
710 vlc_mutex_unlock( &p_mgr->lock );
711 vlc_object_release( p_mgr );
712
713 return count;
714}

716QVariant ExtensionListModel::data( const QModelIndex& index, int role ) const
717{

719 return QVariant();
720
721 ExtensionCopy * extension =

723 if ( !extension )
724 return QVariant();
725 else
726 return extension->data( role );
727}

762QVariant AddonsListModel::Addon::data( int role ) const
763{
764 QVariant returnval;
765
766 vlc_mutex_lock( &p_entry->lock );
767 switch( role )
768 {
769 case Qt::DisplayRole:
770 {
771 returnval = qfu( p_entry->psz_name );
772 break;
773 }
774 case Qt::DecorationRole:
775 if ( p_entry->psz_image_data )
776 {
777 QPixmap pixmap;
778 pixmap.loadFromData( QByteArray::fromBase64( QByteArray( p_entry->psz_image_data ) ),
779 0,
780 Qt::AutoColor
781 );
782 returnval = pixmap;
783 }
784 else if ( p_entry->e_flags & ADDON_BROKEN )
785 returnval = QPixmap( ":/addons/broken" );
786 else
787 returnval = QPixmap( ":/addons/default" );
788 break;
789 case Qt::ToolTipRole:
790 {
791 if ( !( p_entry->e_flags & ADDON_MANAGEABLE ) )
792 {
793 returnval = qtr("This addon has been installed manually. VLC can't manage it by itself.");
794 }
795 break;
796 }
797 case SummaryRole:
798 returnval = qfu( p_entry->psz_summary );
799 break;
800 case DescriptionRole:
801 returnval = qfu( p_entry->psz_description );
802 break;
803 case TypeRole:
804 returnval = QVariant( (int) p_entry->e_type );
805 break;
806 case UUIDRole:
807 returnval = QByteArray( (const char *) p_entry->uuid, (int) sizeof( addon_uuid_t ) );
808 break;
809 case FlagsRole:
810 returnval = QVariant( (int) p_entry->e_flags );
811 break;
812 case StateRole:
813 returnval = QVariant( (int) p_entry->e_state );
814 break;
815 case DownloadsCountRole:
816 returnval = QVariant( (double) p_entry->i_downloads );
817 break;
818 case ScoreRole:
819 returnval = QVariant( (int) p_entry->i_score );
820 break;
821 case VersionRole:
822 returnval = QVariant( p_entry->psz_version );
823 break;
824 case AuthorRole:
825 returnval = qfu( p_entry->psz_author );
826 break;
827 case LinkRole:
828 returnval = qfu( p_entry->psz_source_uri );
829 break;
830 case FilenameRole:
831 {
832 QList<QString> list;
833 FOREACH_ARRAY( addon_file_t *p_file, p_entry->files )
834 list << qfu( p_file->psz_filename );
835 FOREACH_END();
836 returnval = QVariant( list );
837 break;
838 }
839 default:
840 break;
841 }
842 vlc_mutex_unlock( &p_entry->lock );
843
844 return returnval;
845}

1095void AddonItemDelegate::paint( QPainter *painter,
1096 const QStyleOptionViewItem &option,
1097 const QModelIndex &index ) const
1098{
1099 QStyleOptionViewItemV4 newopt = option;
1100 int i_state = index.data( AddonsListModel::StateRole ).toInt();
1101 int i_type = index.data( AddonsListModel::TypeRole ).toInt();
1102
1103 /* Draw Background gradient by addon type */
1104 QColor backgroundColor = AddonsListModel::getColorByAddonType( i_type );
1105
1106 if ( backgroundColor.isValid() )
1107 {
1108 painter->save();
1109 int i_corner = qMin( (int)(option.rect.width() * .05), 30 );
1110 QLinearGradient gradient(
1111 QPoint( option.rect.right() - i_corner, option.rect.bottom() - i_corner ),
1112 option.rect.bottomRight() );
1113 gradient.setColorAt( 0, Qt::transparent );
1114 gradient.setColorAt( 1.0, backgroundColor );
1115 painter->fillRect( option.rect, gradient );
1116 painter->restore();
1117 }
1118
1119 /* Draw base info from parent */
1120 ExtensionItemDelegate::paint( painter, newopt, index );
1121
1122 initStyleOption( &newopt, index );
1123
1124 painter->save();
1125 painter->setRenderHint( QPainter::TextAntialiasing );
1126
1127 /* Addon status */
1128 if ( i_state == ADDON_INSTALLED )
1129 {
1130 painter->save();
1131 painter->setRenderHint( QPainter::Antialiasing );
1132 QMargins statusMargins( 5, 2, 5, 2 );
1133 QFont font( newopt.font );
1134 font.setBold( true );
1135 QFontMetrics metrics( font );
1136 painter->setFont( font );
1137 QRect statusRect = metrics.boundingRect( qtr("Installed") );
1138 statusRect.translate( newopt.rect.width() - statusRect.width(),
1139 newopt.rect.top() + statusRect.height() );
1140 statusRect.adjust( - statusMargins.left() - statusMargins.right(),
1141 0, 0,
1142 statusMargins.top() + statusMargins.bottom() );
1143 QPainterPath path;
1144 path.addRoundedRect( statusRect, 2.0, 2.0 );
1145 painter->fillPath( path, QColor( Qt::green ).darker( 125 ) );
1146 painter->setPen( Qt::white );
1147 painter->drawText(
1148 statusRect.adjusted( statusMargins.left(), statusMargins.top(),
1149 -statusMargins.right(), -statusMargins.bottom() ),
1150 qtr("Installed") );
1151 painter->restore();
1152 }
1153
1154 if ( newopt.state & QStyle::State_Selected )
1155 painter->setPen( newopt.palette.highlightedText().color() );
1156
1157 /* Start below text */
1158 QRect textrect( newopt.rect );
1159 textrect.adjust( 2 * margins.left() + margins.right() + newopt.decorationSize.width(),
1160 margins.top(),
1161 - margins.right(),
1162 - margins.bottom() - newopt.fontMetrics.height() );
1163 textrect.translate( 0, newopt.fontMetrics.height() * 2 );
1164
1165 /* Version */
1166 QString version = index.data( AddonsListModel::VersionRole ).toString();
1167 if ( !version.isEmpty() )
1168 painter->drawText( textrect, Qt::AlignLeft, qtr("Version %1").arg( version ) );
1169
1170 textrect.translate( 0, newopt.fontMetrics.height() );
1171
1172 /* Score */
1173 int i_score = index.data( AddonsListModel::ScoreRole ).toInt();
1174 QPixmap scoreicon;
1175 if ( i_score )
1176 {
1177 scoreicon = QPixmap( ":/addons/score" ).scaledToHeight(
1178 newopt.fontMetrics.height(), Qt::SmoothTransformation );
1179 int i_width = ( (float) i_score / ADDON_MAX_SCORE ) * scoreicon.width();
1180 /* Erase the end (value) of our pixmap with a shadow */
1181 QPainter erasepainter( &scoreicon );
1182 erasepainter.setCompositionMode( QPainter::CompositionMode_SourceIn );
1183 erasepainter.fillRect( QRect( i_width, 0,
1184 scoreicon.width() - i_width, scoreicon.height() ),
1185 newopt.palette.color( QPalette::Dark ) );
1186 erasepainter.end();
1187 painter->drawPixmap( textrect.topLeft(), scoreicon );
1188 }
1189
1190 /* Downloads # */
1191 int i_downloads = index.data( AddonsListModel::DownloadsCountRole ).toInt();
1192 if ( i_downloads )
1193 painter->drawText( textrect.translated( scoreicon.width() + margins.left(), 0 ),
1194 Qt::AlignLeft, qtr("%1 downloads").arg( i_downloads ) );
1195
1196 painter->restore();
1197
1198 if ( animator )
1199 {
1200 if ( animator->isRunning() && animator->getIndex() == index )
1201 {
1202 if ( i_state != ADDON_INSTALLING && i_state != ADDON_UNINSTALLING )
1203 animator->run( false );
1204 }
1205 /* Create our installation progress overlay */
1206
1207 if ( i_state == ADDON_INSTALLING || i_state == ADDON_UNINSTALLING )
1208 {
1209 painter->save();
1210 painter->setCompositionMode( QPainter::CompositionMode_SourceOver );
1211 painter->fillRect( newopt.rect, QColor( 255, 255, 255, 128 ) );
1212 if ( animator && index.isValid() )
1213 {
1214 animator->setIndex( index );
1215 animator->run( true );
1216 QSize adjustment = newopt.rect.size() / 4;
1217 progressbar->setGeometry(
1218 newopt.rect.adjusted( adjustment.width(), adjustment.height(),
1219 -adjustment.width(), -adjustment.height() ) );
1220 painter->drawPixmap( newopt.rect.left() + adjustment.width(),
1221 newopt.rect.top() + adjustment.height(),
1222 QPixmap::grabWidget( progressbar ) );
1223 }
1224 painter->restore();
1225 }
1226 }
1227}
