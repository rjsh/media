44OpenUrlDialog::OpenUrlDialog( intf_thread_t *_p_intf,
45 bool _bClipboard ) :
46 QVLCDialog( (QWidget*)_p_intf->p_sys->p_mi, _p_intf ), bClipboard( _bClipboard )
47{
48 setWindowTitle( qtr( "Open URL" ) );
49 setWindowRole( "vlc-open-url" );
50
51 /* Buttons */
52 QPushButton *but;
53
54 QDialogButtonBox *box = new QDialogButtonBox( this );
55 but = box->addButton( qtr( "&Play" ), QDialogButtonBox::AcceptRole );
56 CONNECT( but, clicked(), this, play() );
57
58 but = box->addButton( qtr( "&Enqueue" ), QDialogButtonBox::AcceptRole );
59 CONNECT( but, clicked(), this, enqueue() );
60
61 but = box->addButton( qtr( "&Cancel" ) , QDialogButtonBox::RejectRole );
62 CONNECT( box, rejected(), this, reject() );
63
64 /* Info label and line edit */
65 edit = new ClickLineEdit( qtr( "Enter URL here..." ), this );
66 edit->setValidator( new UrlValidator( edit ) );
67
68 QLabel *info = new QLabel( qtr( "Please enter the URL or path "
69 "to the media you want to play."),
70 this );
71
72 setToolTip( qtr( "If your clipboard contains a valid URL\n"
73 "or the path to a file on your computer,\n"
74 "it will be automatically selected." ) );
75
76 /* Layout */
77 QVBoxLayout *vlay = new QVBoxLayout( this );
78
79 vlay->addWidget( info );
80 vlay->addWidget( edit );
81 vlay->addWidget( box );
82}

84void OpenUrlDialog::enqueue()
85{
86 bShouldEnqueue = true;
87 lastUrl = edit->text();
88 accept();
89}
90
91void OpenUrlDialog::play()
92{
93 lastUrl = edit->text();
94 accept();
95}

97QString OpenUrlDialog::url() const
98{
99 return lastUrl;

102bool OpenUrlDialog::shouldEnqueue() const
103{
104 return bShouldEnqueue;

107/** Show Event:
108 * When the dialog is shown, try to extract a URL from the clipboard
109 * and paste it in the Edit box.
110 * showEvent can happen not only on exec() but I think it's cool to
111 * actualize the URL on showEvent (eg. change virtual desktop...)
112 **/
113void OpenUrlDialog::showEvent( QShowEvent *ev )
114{
115 (void) ev;
116 bShouldEnqueue = false;
117 edit->setFocus( Qt::OtherFocusReason );
118 if( !lastUrl.isEmpty() && edit->text().isEmpty() )
119 {
120 /* The text should not have been changed, excepted if the user
121 has clicked Cancel before */
122 edit->setText( lastUrl );
123 }
124 else
125 edit->clear();
126
127 if( bClipboard )
128 {
129 QClipboard *clipboard = QApplication::clipboard();
130 assert( clipboard != NULL );
131 QString txt = clipboard->text( QClipboard::Selection ).trimmed();
132
133 if( txt.isEmpty() || ( !txt.contains("://") && !QFile::exists(txt) ) )
134 txt = clipboard->text( QClipboard::Clipboard ).trimmed();
135
136 if( txt.contains( "://" ) || QFile::exists( txt ) )
137 edit->setText( txt );
138 }
139}
