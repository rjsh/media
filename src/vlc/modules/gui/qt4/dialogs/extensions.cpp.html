51ExtensionsDialogProvider::ExtensionsDialogProvider( intf_thread_t *_p_intf,
52 extensions_manager_t *p_mgr )
53 : QObject( NULL ), p_intf( _p_intf ), p_extensions_manager( p_mgr )
54{
55 // At this point, we consider that the Qt interface already called
56 // dialog_Register() in order to be the extension dialog provider
57 var_Create( p_intf, "dialog-extension", VLC_VAR_ADDRESS );
58 var_AddCallback( p_intf, "dialog-extension", DialogCallback, NULL );
59
60 CONNECT( this, SignalDialog( extension_dialog_t* ),
61 this, UpdateExtDialog( extension_dialog_t* ) );
62}
63
64ExtensionsDialogProvider::~ExtensionsDialogProvider()
65{
66 msg_Dbg( p_intf, "ExtensionsDialogProvider is quitting..." );
67 var_DelCallback( p_intf, "dialog-extension", DialogCallback, NULL );
68}

70/** Create a dialog
71 * Note: Lock on p_dialog->lock must be held. */
72ExtensionDialog* ExtensionsDialogProvider::CreateExtDialog(
73 extension_dialog_t *p_dialog )
74{
75 ExtensionDialog *dialog = new ExtensionDialog( p_intf,
76 p_extensions_manager,
77 p_dialog );
78 p_dialog->p_sys_intf = (void*) dialog;
79 CONNECT( dialog, destroyDialog( extension_dialog_t* ),
80 this, DestroyExtDialog( extension_dialog_t* ) );
81 return dialog;
82}
83
84/** Destroy a dialog
85 * Note: Lock on p_dialog->lock must be held. */
86int ExtensionsDialogProvider::DestroyExtDialog( extension_dialog_t *p_dialog )
87{
88 assert( p_dialog );
89 ExtensionDialog *dialog = ( ExtensionDialog* ) p_dialog->p_sys_intf;
90 if( !dialog )
91 return VLC_EGENERIC;
92 delete dialog;
93 p_dialog->p_sys_intf = NULL;
94 vlc_cond_signal( &p_dialog->cond );
95 return VLC_SUCCESS;
96}

98/**
99 * Update/Create/Destroy a dialog
100 **/
101ExtensionDialog* ExtensionsDialogProvider::UpdateExtDialog(
102 extension_dialog_t *p_dialog )
103{
104 assert( p_dialog );
105
106 ExtensionDialog *dialog = ( ExtensionDialog* ) p_dialog->p_sys_intf;
107 if( p_dialog->b_kill && !dialog )
108 {
109 /* This extension could not be activated properly but tried
110 to create a dialog. We must ignore it. */
111 return NULL;
112 }
113
114 vlc_mutex_lock( &p_dialog->lock );
115 if( !p_dialog->b_kill && !dialog )
116 {
117 dialog = CreateExtDialog( p_dialog );
118 dialog->setVisible( !p_dialog->b_hide );
119 dialog->has_lock = false;
120 }
121 else if( !p_dialog->b_kill && dialog )
122 {
123 dialog->has_lock = true;
124 dialog->UpdateWidgets();
125 if( strcmp( qtu( dialog->windowTitle() ),
126 p_dialog->psz_title ) != 0 )
127 dialog->setWindowTitle( qfu( p_dialog->psz_title ) );
128 dialog->has_lock = false;
129 dialog->setVisible( !p_dialog->b_hide );
130 }
131 else if( p_dialog->b_kill )
132 {
133 DestroyExtDialog( p_dialog );
134 }
135 vlc_cond_signal( &p_dialog->cond );
136 vlc_mutex_unlock( &p_dialog->lock );
137 return dialog;
138}

140/**
141 * Ask the dialog manager to create/update/kill the dialog. Thread-safe.
142 **/
143void ExtensionsDialogProvider::ManageDialog( extension_dialog_t *p_dialog )
144{
145 assert( p_dialog );
146 ExtensionsManager *extMgr = ExtensionsManager::getInstance( p_intf );
147 assert( extMgr != NULL );
148 if( !extMgr->isUnloading() )
149 emit SignalDialog( p_dialog ); // Safe because we signal Qt thread
150 else
151 UpdateExtDialog( p_dialog ); // This is safe, we're already in Qt thread
152}

154/**
155 * Ask the dialogs provider to create a new dialog
156 **/
157static int DialogCallback( vlc_object_t *p_this, const char *psz_variable,
158 vlc_value_t old_val, vlc_value_t new_val,
159 void *param )
160{
161 (void) p_this;
162 (void) psz_variable;
163 (void) old_val;
164 (void) param;
165
166 ExtensionsDialogProvider *p_edp = ExtensionsDialogProvider::getInstance();
167 if( !p_edp )
168 return VLC_EGENERIC;
169 if( !new_val.p_address )
170 return VLC_EGENERIC;
171
172 extension_dialog_t *p_dialog = ( extension_dialog_t* ) new_val.p_address;
173 p_edp->ManageDialog( p_dialog );
174 return VLC_SUCCESS;
175}

178ExtensionDialog::ExtensionDialog( intf_thread_t *_p_intf,
179 extensions_manager_t *p_mgr,
180 extension_dialog_t *_p_dialog )
181 : QDialog( NULL ), p_intf( _p_intf ), p_extensions_manager( p_mgr )
182 , p_dialog( _p_dialog ), has_lock(true)
183{
184 assert( p_dialog );
185 CONNECT( ExtensionsDialogProvider::getInstance(), destroyed(),
186 this, parentDestroyed() );
187
188 msg_Dbg( p_intf, "Creating a new dialog: '%s'", p_dialog->psz_title );
189 this->setWindowFlags( Qt::WindowMinMaxButtonsHint
190 | Qt::WindowCloseButtonHint );
191 this->setWindowTitle( qfu( p_dialog->psz_title ) );
192
193 layout = new QGridLayout( this );
194 clickMapper = new QSignalMapper( this );
195 CONNECT( clickMapper, mapped( QObject* ), this, TriggerClick( QObject* ) );
196 inputMapper = new QSignalMapper( this );
197 CONNECT( inputMapper, mapped( QObject* ), this, SyncInput( QObject* ) );
198 selectMapper = new QSignalMapper( this );
199 CONNECT( selectMapper, mapped( QObject* ), this, SyncSelection(QObject*) );
200
201 UpdateWidgets();
202}

204ExtensionDialog::~ExtensionDialog()
205{
206 msg_Dbg( p_intf, "Deleting extension dialog '%s'", qtu(windowTitle()) );
207}

209QWidget* ExtensionDialog::CreateWidget( extension_widget_t *p_widget )
210{
211 QLabel *label = NULL;
212 QPushButton *button = NULL;
213 QTextBrowser *textArea = NULL;
214 QLineEdit *textInput = NULL;
215 QCheckBox *checkBox = NULL;
216 QComboBox *comboBox = NULL;
217 QListWidget *list = NULL;
218 SpinningIcon *spinIcon = NULL;
219 struct extension_widget_t::extension_widget_value_t *p_value = NULL;
220
221 assert( p_widget->p_sys_intf == NULL );
222
223 switch( p_widget->type )
224 {
225 case EXTENSION_WIDGET_LABEL:
226 label = new QLabel( qfu( p_widget->psz_text ), this );
227 p_widget->p_sys_intf = label;
228 label->setTextFormat( Qt::RichText );
229 label->setOpenExternalLinks( true );
230 return label;
231
232 case EXTENSION_WIDGET_BUTTON:
233 button = new QPushButton( qfu( p_widget->psz_text ), this );
234 clickMapper->setMapping( button, new WidgetMapper( p_widget ) );
235 CONNECT( button, clicked(), clickMapper, map() );
236 p_widget->p_sys_intf = button;
237 return button;
238
239 case EXTENSION_WIDGET_IMAGE:
240 label = new QLabel( this );
241 label->setPixmap( QPixmap( qfu( p_widget->psz_text ) ) );
242 if( p_widget->i_width > 0 )
243 label->setMaximumWidth( p_widget->i_width );
244 if( p_widget->i_height > 0 )
245 label->setMaximumHeight( p_widget->i_height );
246 label->setScaledContents( true );
247 p_widget->p_sys_intf = label;
248 return label;
249
250 case EXTENSION_WIDGET_HTML:
251 textArea = new QTextBrowser( this );
252 textArea->setOpenExternalLinks( true );
253 textArea->setHtml( qfu( p_widget->psz_text ) );
254 p_widget->p_sys_intf = textArea;
255 return textArea;
256
257 case EXTENSION_WIDGET_TEXT_FIELD:
258 textInput = new QLineEdit( this );
259 textInput->setText( qfu( p_widget->psz_text ) );
260 textInput->setReadOnly( false );
261 textInput->setEchoMode( QLineEdit::Normal );
262 inputMapper->setMapping( textInput, new WidgetMapper( p_widget ) );
263 /// @note: maybe it would be wiser to use textEdited here?
264 CONNECT( textInput, textChanged(const QString &),
265 inputMapper, map() );
266 p_widget->p_sys_intf = textInput;
267 return textInput;
268
269 case EXTENSION_WIDGET_PASSWORD:
270 textInput = new QLineEdit( this );
271 textInput->setText( qfu( p_widget->psz_text ) );
272 textInput->setReadOnly( false );
273 textInput->setEchoMode( QLineEdit::Password );
274 inputMapper->setMapping( textInput, new WidgetMapper( p_widget ) );
275 /// @note: maybe it would be wiser to use textEdited here?
276 CONNECT( textInput, textChanged(const QString &),
277 inputMapper, map() );
278 p_widget->p_sys_intf = textInput;
279 return textInput;
280
281 case EXTENSION_WIDGET_CHECK_BOX:
282 checkBox = new QCheckBox( this );
283 checkBox->setText( qfu( p_widget->psz_text ) );
284 checkBox->setChecked( p_widget->b_checked );
285 clickMapper->setMapping( checkBox, new WidgetMapper( p_widget ) );
286 CONNECT( checkBox, stateChanged( int ), clickMapper, map() );
287 p_widget->p_sys_intf = checkBox;
288 return checkBox;
289
290 case EXTENSION_WIDGET_DROPDOWN:
291 comboBox = new QComboBox( this );
292 comboBox->setEditable( false );
293 for( p_value = p_widget->p_values;
294 p_value != NULL;
295 p_value = p_value->p_next )
296 {
297 comboBox->addItem( qfu( p_value->psz_text ), p_value->i_id );
298 }
299 /* Set current item */
300 if( p_widget->psz_text )
301 {
302 int idx = comboBox->findText( qfu( p_widget->psz_text ) );
303 if( idx >= 0 )
304 comboBox->setCurrentIndex( idx );
305 }
306 selectMapper->setMapping( comboBox, new WidgetMapper( p_widget ) );
307 CONNECT( comboBox, currentIndexChanged( const QString& ),
308 selectMapper, map() );
309 return comboBox;
310
311 case EXTENSION_WIDGET_LIST:
312 list = new QListWidget( this );
313 list->setSelectionMode( QAbstractItemView::ExtendedSelection );
314 for( p_value = p_widget->p_values;
315 p_value != NULL;
316 p_value = p_value->p_next )
317 {
318 QListWidgetItem *item =
319 new QListWidgetItem( qfu( p_value->psz_text ) );
320 item->setData( Qt::UserRole, p_value->i_id );
321 list->addItem( item );
322 }
323 selectMapper->setMapping( list, new WidgetMapper( p_widget ) );
324 CONNECT( list, itemSelectionChanged(),
325 selectMapper, map() );
326 return list;
327
328 case EXTENSION_WIDGET_SPIN_ICON:
329 spinIcon = new SpinningIcon( this );
330 spinIcon->play( p_widget->i_spin_loops );
331 p_widget->p_sys_intf = spinIcon;
332 return spinIcon;
333
334 default:
335 msg_Err( p_intf, "Widget type %d unknown", p_widget->type );
336 return NULL;
337 }
338}

340/**
341 * Forward click event to the extension
342 * @param object A WidgetMapper, whose data() is the p_widget
343 **/
344int ExtensionDialog::TriggerClick( QObject *object )
345{
346 assert( object != NULL );
347 WidgetMapper *mapping = static_cast< WidgetMapper* >( object );
348 extension_widget_t *p_widget = mapping->getWidget();
349
350 QCheckBox *checkBox = NULL;
351 int i_ret = VLC_EGENERIC;
352
353 bool lockedHere = false;
354 if( !has_lock )
355 {
356 vlc_mutex_lock( &p_dialog->lock );
357 has_lock = true;
358 lockedHere = true;
359 }
360
361 switch( p_widget->type )
362 {
363 case EXTENSION_WIDGET_BUTTON:
364 i_ret = extension_WidgetClicked( p_dialog, p_widget );
365 break;
366
367 case EXTENSION_WIDGET_CHECK_BOX:
368 checkBox = static_cast< QCheckBox* >( p_widget->p_sys_intf );
369 p_widget->b_checked = checkBox->isChecked();
370 i_ret = VLC_SUCCESS;
371 break;
372
373 default:
374 msg_Dbg( p_intf, "A click event was triggered by a wrong widget" );
375 break;
376 }
377
378 if( lockedHere )
379 {
380 vlc_mutex_unlock( &p_dialog->lock );
381 has_lock = false;
382 }
383
384 return i_ret;
385}

387/**
388 * Synchronize psz_text with the widget's text() value on update
389 * @param object A WidgetMapper
390 **/
391void ExtensionDialog::SyncInput( QObject *object )
392{
393 assert( object != NULL );
394
395 bool lockedHere = false;
396 if( !has_lock )
397 {
398 vlc_mutex_lock( &p_dialog->lock );
399 has_lock = true;
400 lockedHere = true;
401 }
402
403 WidgetMapper *mapping = static_cast< WidgetMapper* >( object );
404 extension_widget_t *p_widget = mapping->getWidget();
405 assert( p_widget->type == EXTENSION_WIDGET_TEXT_FIELD
406 || p_widget->type == EXTENSION_WIDGET_PASSWORD );
407 /* Synchronize psz_text with the new value */
408 QLineEdit *widget = static_cast< QLineEdit* >( p_widget->p_sys_intf );
409 char *psz_text = widget->text().isNull() ? NULL : strdup( qtu( widget->text() ) );
410 free( p_widget->psz_text );
411 p_widget->psz_text = psz_text;
412
413 if( lockedHere )
414 {
415 vlc_mutex_unlock( &p_dialog->lock );
416 has_lock = false;
417 }
418}

420/**
421 * Synchronize parameter b_selected in the values list
422 * @param object A WidgetMapper
423 **/
424void ExtensionDialog::SyncSelection( QObject *object )
425{
426 assert( object != NULL );
427 struct extension_widget_t::extension_widget_value_t *p_value;
428
429 bool lockedHere = false;
430 if( !has_lock )
431 {
432 vlc_mutex_lock( &p_dialog->lock );
433 has_lock = true;
434 lockedHere = true;
435 }
436
437 WidgetMapper *mapping = static_cast< WidgetMapper* >( object );
438 extension_widget_t *p_widget = mapping->getWidget();
439 assert( p_widget->type == EXTENSION_WIDGET_DROPDOWN
440 || p_widget->type == EXTENSION_WIDGET_LIST );
441
442 if( p_widget->type == EXTENSION_WIDGET_DROPDOWN )
443 {
444 QComboBox *combo = static_cast< QComboBox* >( p_widget->p_sys_intf );
445 for( p_value = p_widget->p_values;
446 p_value != NULL;
447 p_value = p_value->p_next )
448 {
449// if( !qstrcmp( p_value->psz_text, qtu( combo->currentText() ) ) )
450 if( combo->itemData( combo->currentIndex(), Qt::UserRole ).toInt()
451 == p_value->i_id )
452 {
453 p_value->b_selected = true;
454 }
455 else
456 {
457 p_value->b_selected = false;
458 }
459 }
460 free( p_widget->psz_text );
461 p_widget->psz_text = strdup( qtu( combo->currentText() ) );
462 }
463 else if( p_widget->type == EXTENSION_WIDGET_LIST )
464 {
465 QListWidget *list = static_cast<QListWidget*>( p_widget->p_sys_intf );
466 QList<QListWidgetItem *> selection = list->selectedItems();
467 for( p_value = p_widget->p_values;
468 p_value != NULL;
469 p_value = p_value->p_next )
470 {
471 bool b_selected = false;
472 foreach( const QListWidgetItem *item, selection )
473 {
474// if( !qstrcmp( qtu( item->text() ), p_value->psz_text ) )
475 if( item->data( Qt::UserRole ).toInt() == p_value->i_id )
476 {
477 b_selected = true;
478 break;
479 }
480 }
481 p_value->b_selected = b_selected;
482 }
483 }
484
485 if( lockedHere )
486 {
487 vlc_mutex_unlock( &p_dialog->lock );
488 has_lock = false;
489 }
490}

492void ExtensionDialog::UpdateWidgets()
493{
494 assert( p_dialog );
495 extension_widget_t *p_widget;
496 FOREACH_ARRAY( p_widget, p_dialog->widgets )
497 {
498 if( !p_widget ) continue; /* Some widgets may be NULL at this point */
499 QWidget *widget;
500 int row = p_widget->i_row - 1;
501 int col = p_widget->i_column - 1;
502 if( row < 0 )
503 {
504 row = layout->rowCount();
505 col = 0;
506 }
507 else if( col < 0 )
508 col = layout->columnCount();
509 int hsp = __MAX( 1, p_widget->i_horiz_span );
510 int vsp = __MAX( 1, p_widget->i_vert_span );
511 if( !p_widget->p_sys_intf && !p_widget->b_kill )
512 {
513 widget = CreateWidget( p_widget );
514 if( !widget )
515 {
516 msg_Warn( p_intf, "Could not create a widget for dialog %s",
517 p_dialog->psz_title );
518 continue;
519 }
520 widget->setVisible( !p_widget->b_hide );
521 layout->addWidget( widget, row, col, vsp, hsp );
522 if( ( p_widget->i_width > 0 ) && ( p_widget->i_height > 0 ) )
523 widget->resize( p_widget->i_width, p_widget->i_height );
524 p_widget->p_sys_intf = widget;
525 this->resize( sizeHint() );
526 /* If an update was required, cancel it as we just created the widget */
527 p_widget->b_update = false;
528 }
529 else if( p_widget->p_sys_intf && !p_widget->b_kill
530 && p_widget->b_update )
531 {
532 widget = UpdateWidget( p_widget );
533 if( !widget )
534 {
535 msg_Warn( p_intf, "Could not update a widget for dialog %s",
536 p_dialog->psz_title );
537 return;
538 }
539 widget->setVisible( !p_widget->b_hide );
540 layout->addWidget( widget, row, col, vsp, hsp );
541 if( ( p_widget->i_width > 0 ) && ( p_widget->i_height > 0 ) )
542 widget->resize( p_widget->i_width, p_widget->i_height );
543 p_widget->p_sys_intf = widget;
544 this->resize( sizeHint() );
545
546 /* Do not update again */
547 p_widget->b_update = false;
548 }
549 else if( p_widget->p_sys_intf && p_widget->b_kill )
550 {
551 DestroyWidget( p_widget );
552 p_widget->p_sys_intf = NULL;
553 this->resize( sizeHint() );
554 }
555 }
556 FOREACH_END()
557}

559QWidget* ExtensionDialog::UpdateWidget( extension_widget_t *p_widget )
560{
561 QLabel *label = NULL;
562 QPushButton *button = NULL;
563 QTextBrowser *textArea = NULL;
564 QLineEdit *textInput = NULL;
565 QCheckBox *checkBox = NULL;
566 QComboBox *comboBox = NULL;
567 QListWidget *list = NULL;
568 SpinningIcon *spinIcon = NULL;
569 struct extension_widget_t::extension_widget_value_t *p_value = NULL;
570
571 assert( p_widget->p_sys_intf != NULL );
572
573 switch( p_widget->type )
574 {
575 case EXTENSION_WIDGET_LABEL:
576 label = static_cast< QLabel* >( p_widget->p_sys_intf );
577 label->setText( qfu( p_widget->psz_text ) );
578 return label;
579
580 case EXTENSION_WIDGET_BUTTON:
581 // FIXME: looks like removeMappings does not work
582 button = static_cast< QPushButton* >( p_widget->p_sys_intf );
583 button->setText( qfu( p_widget->psz_text ) );
584 clickMapper->removeMappings( button );
585 clickMapper->setMapping( button, new WidgetMapper( p_widget ) );
586 CONNECT( button, clicked(), clickMapper, map() );
587 return button;
588
589 case EXTENSION_WIDGET_IMAGE:
590 label = static_cast< QLabel* >( p_widget->p_sys_intf );
591 label->setPixmap( QPixmap( qfu( p_widget->psz_text ) ) );
592 return label;
593
594 case EXTENSION_WIDGET_HTML:
595 textArea = static_cast< QTextBrowser* >( p_widget->p_sys_intf );
596 textArea->setHtml( qfu( p_widget->psz_text ) );
597 return textArea;
598
599 case EXTENSION_WIDGET_TEXT_FIELD:
600 textInput = static_cast< QLineEdit* >( p_widget->p_sys_intf );
601 textInput->setText( qfu( p_widget->psz_text ) );
602 return textInput;
603
604 case EXTENSION_WIDGET_PASSWORD:
605 textInput = static_cast< QLineEdit* >( p_widget->p_sys_intf );
606 textInput->setText( qfu( p_widget->psz_text ) );
607 return textInput;
608
609 case EXTENSION_WIDGET_CHECK_BOX:
610 checkBox = static_cast< QCheckBox* >( p_widget->p_sys_intf );
611 checkBox->setText( qfu( p_widget->psz_text ) );
612 checkBox->setChecked( p_widget->b_checked );
613 return checkBox;
614
615 case EXTENSION_WIDGET_DROPDOWN:
616 comboBox = static_cast< QComboBox* >( p_widget->p_sys_intf );
617 // method widget:clear()
618 if ( p_widget->p_values == NULL )
619 {
620 comboBox->clear();
621 return comboBox;
622 }
623 // method widget:addvalue()
624 for( p_value = p_widget->p_values;
625 p_value != NULL;
626 p_value = p_value->p_next )
627 {
628 if ( comboBox->findText( qfu( p_value->psz_text ) ) < 0 )
629 comboBox->addItem( qfu( p_value->psz_text ), p_value->i_id );
630 }
631 return comboBox;
632
633 case EXTENSION_WIDGET_LIST:
634 list = static_cast< QListWidget* >( p_widget->p_sys_intf );
635 list->clear();
636 for( p_value = p_widget->p_values;
637 p_value != NULL;
638 p_value = p_value->p_next )
639 {
640 QListWidgetItem *item =
641 new QListWidgetItem( qfu( p_value->psz_text ) );
642 item->setData( Qt::UserRole, p_value->i_id );
643 list->addItem( item );
644 }
645 return list;
646
647 case EXTENSION_WIDGET_SPIN_ICON:
648 spinIcon = static_cast< SpinningIcon* >( p_widget->p_sys_intf );
649 if( !spinIcon->isPlaying() && p_widget->i_spin_loops != 0 )
650 spinIcon->play( p_widget->i_spin_loops );
651 else if( spinIcon->isPlaying() && p_widget->i_spin_loops == 0 )
652 spinIcon->stop();
653 p_widget->i_height = p_widget->i_width = 16;
654 return spinIcon;
655
656 default:
657 msg_Err( p_intf, "Widget type %d unknown", p_widget->type );
658 return NULL;
659 }
660}

662void ExtensionDialog::DestroyWidget( extension_widget_t *p_widget,
663 bool b_cond )
664{
665 assert( p_widget && p_widget->b_kill );
666 QWidget *widget = static_cast< QWidget* >( p_widget->p_sys_intf );
667 delete widget;
668 p_widget->p_sys_intf = NULL;
669 if( b_cond )
670 vlc_cond_signal( &p_dialog->cond );
671}

682/** Grab some keyboard input (ESC, ...) and handle actions manually */
