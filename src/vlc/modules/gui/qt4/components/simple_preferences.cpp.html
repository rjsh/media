186static int getDefaultAudioVolume(vlc_object_t *obj, const char *aout)
187{
188 if (!strcmp(aout, "") || !strcmp(aout, "any"))
189 return -1;
190 else
191 /* Note: For hysterical raisins, this is sorted by decreasing priority
192 * order (then alphabetical order). */
193 if (!strcmp(aout, "pulse"))
194 return -1;
195 else
196#ifdef __linux__
197 if (!strcmp(aout, "alsa") && module_exists("alsa"))
198 return cbrtf(config_GetFloat(obj, "alsa-gain")) * 100.f + .5f;
199 else
200#endif
201#ifdef _WIN32
202 if (!strcmp(aout, "mmdevice"))
203 return -1;
204 else
205#endif
206 if (!strcmp(aout, "sndio"))
207 return -1;
208 else
209#ifdef __APPLE__
210 if (!strcmp("auhal") && module_exists("auhal"))
211 return (config_GetFloat(obj, "auhal-volume") * 100.f + .5f)
212 / AOUT_VOLUME_DEFAULT;
213 else
214#endif
215#ifdef _WIN32
216 if (!strcmp(aout, "directsound") && module_exists("directsound"))
217 return config_GetFloat(obj, "directx-volume") * 100.f + .5f;
218 else
219#endif
220 if (!strcmp(aout, "jack"))
221 return cbrtf(config_GetFloat(obj, "jack-gain")) * 100.f + 0.5f;
222 else
223#ifdef __OS2__
224 if (!strcmp(aout, "kai"))
225 return cbrtf(config_GetFloat(obj, "kai-gain")) * 100.f + .5f;
226 else
227#endif
228 if (!strcmp(aout, "oss"))
229 return -1;
230 else
231#ifdef _WIN32
232 if (!strcmp(aout, "waveout"))
233 return config_GetFloat(obj, "waveout-volume") * 100.f + .5f;
234 else
235#endif
236 return -1;
237}

242SPrefsCatList::SPrefsCatList( intf_thread_t *_p_intf, QWidget *_parent ) :
243 QWidget( _parent ), p_intf( _p_intf )
244{
245 QHBoxLayout *layout = new QHBoxLayout();
246
247 /* Use autoExclusive buttons and a mapper as QButtonGroup can't
248 set focus (keys) when it manages the buttons's exclusivity.
249 See QT bugs 131 & 816 and QAbstractButton's source code. */
250 QSignalMapper *mapper = new QSignalMapper( layout );
251 CONNECT( mapper, mapped(int), this, switchPanel(int) );
252
253 short icon_height = ICON_HEIGHT;
254
255#define ADD_CATEGORY( button, label, ltooltip, icon, numb ) \
256 QToolButton * button = new QToolButton( this ); \
257 button->setIcon( QIcon( ":/prefsmenu/" #icon ) ); \
258 button->setText( label ); \
259 button->setToolTip( ltooltip ); \
260 button->setToolButtonStyle( Qt::ToolButtonTextUnderIcon ); \
261 button->setIconSize( QSize( icon_height + 40 , icon_height ) ); \
262 button->setSizePolicy(QSizePolicy::MinimumExpanding, QSizePolicy::Preferred ); \
263 button->setAutoRaise( true ); \
264 button->setCheckable( true ); \
265 button->setAutoExclusive( true ); \
266 CONNECT( button, clicked(), mapper, map() ); \
267 mapper->setMapping( button, numb ); \
268 layout->addWidget( button );
269
270 ADD_CATEGORY( SPrefsInterface, qtr("Interface"), qtr("Interface Settings"),
271 cone_interface_64, 0 );
272 ADD_CATEGORY( SPrefsAudio, qtr("Audio"), qtr("Audio Settings"),
273 cone_audio_64, 1 );
274 ADD_CATEGORY( SPrefsVideo, qtr("Video"), qtr("Video Settings"),
275 cone_video_64, 2 );
276 ADD_CATEGORY( SPrefsSubtitles, qtr(SUBPIC_TITLE), qtr("Subtitle & On Screen Display Settings"),
277 cone_subtitles_64, 3 );
278 ADD_CATEGORY( SPrefsInputAndCodecs, qtr(INPUT_TITLE), qtr("Input & Codecs Settings"),
279 cone_input_64, 4 );
280 ADD_CATEGORY( SPrefsHotkeys, qtr("Hotkeys"), qtr("Configure Hotkeys"),
281 cone_hotkeys_64, 5 );
282
283#undef ADD_CATEGORY
284
285 SPrefsInterface->setChecked( true );
286 layout->setMargin( 0 );
287 layout->setSpacing( 1 );
288
289 setSizePolicy(QSizePolicy::MinimumExpanding,QSizePolicy::Preferred);
290 setMinimumWidth( ICON_HEIGHT * 6 + 10 );
291 setLayout( layout );
292}

1008/* Function called from the main Preferences dialog on each SPrefs Panel */
1009void SPrefsPanel::apply()
1010{
1011 /* Generic save for ever panel */
1012 QList<ConfigControl *>::const_iterator i;
1013 for( i = controls.begin() ; i != controls.end() ; ++i )
1014 {
1015 ConfigControl *c = qobject_cast<ConfigControl *>(*i);
1016 c->doApply();
1017 }
1018
1019 switch( number )
1020 {
1021 case SPrefsInputAndCodecs:
1022 {
1023 /* Device default selection */
1024 QByteArray devicepath =
1025 qobject_cast<QComboBox *>(optionWidgets["inputLE"])->currentText().toUtf8();
1026 if( devicepath.size() > 0 )
1027 {
1028 config_PutPsz( p_intf, "dvd", devicepath );
1029 config_PutPsz( p_intf, "vcd", devicepath );
1030 config_PutPsz( p_intf, "cd-audio", devicepath );
1031 }
1032
1033#define CaC( name, factor ) config_PutInt( p_intf, name, i_comboValue * factor )
1034 /* Caching */
1035 QComboBox *cachingCombo = qobject_cast<QComboBox *>(optionWidgets["cachingCoB"]);
1036 int i_comboValue = cachingCombo->itemData( cachingCombo->currentIndex() ).toInt();
1037 if( i_comboValue )
1038 {
1039 CaC( "file-caching", 1 );
1040 CaC( "network-caching", 10/3 );
1041 CaC( "disc-caching", 1 );
1042 CaC( "live-caching", 1 );
1043 }
1044 break;
1045#undef CaC
1046 }
1047
1048 /* Interfaces */
1049 case SPrefsInterface:
1050 {
1051 if( qobject_cast<QRadioButton *>(optionWidgets["skinRB"])->isChecked() )
1052 config_PutPsz( p_intf, "intf", "skins2,any" );
1053 else
1054 //if( qobject_cast<QRadioButton *>(optionWidgets[qtRB])->isChecked() )
1055 config_PutPsz( p_intf, "intf", "" );
1056 if( qobject_cast<QComboBox *>(optionWidgets["styleCB"]) )
1057 getSettings()->setValue( "MainWindow/QtStyle",
1058 qobject_cast<QComboBox *>(optionWidgets["styleCB"])->currentText() );
1059#ifdef _WIN32
1060 saveLang();
1061#endif
1062 break;
1063 }
1064
1065 case SPrefsAudio:
1066 {
1067 bool b_checked =
1068 qobject_cast<QCheckBox *>(optionWidgets["normalizerChB"])->isChecked();
1069 if( b_checked && !qs_filter.contains( "normvol" ) )
1070 qs_filter.append( "normvol" );
1071 if( !b_checked && qs_filter.contains( "normvol" ) )
1072 qs_filter.removeAll( "normvol" );
1073
1074 b_checked =
1075 qobject_cast<QCheckBox *>(optionWidgets["headphoneB"])->isChecked();
1076
1077 if( b_checked && !qs_filter.contains( "headphone" ) )
1078 qs_filter.append( "headphone" );
1079 if( !b_checked && qs_filter.contains( "headphone" ) )
1080 qs_filter.removeAll( "headphone" );
1081
1082 config_PutPsz( p_intf, "audio-filter", qtu( qs_filter.join( ":" ) ) );
1083
1084 /* Default volume */
1085 int i_volume =
1086 qobject_cast<QSlider *>(optionWidgets["defaultVolume"])->value();
1087 bool b_reset_volume =
1088 qobject_cast<QCheckBox *>(optionWidgets["resetVolumeCheckbox"])->isChecked();
1089 module_config_t *p_config = config_FindConfig( VLC_OBJECT(p_intf), "aout" );
1090 char *psz_aout = p_config->value.psz;
1091
1092
1093 float f_gain = powf( i_volume / 100.f, 3 );
1094
1095#define save_vol_aout( name ) \
1096 module_exists( name ) && ( !psz_aout || !strcmp( psz_aout, name ) || !strcmp( psz_aout, "any" ) )
1097
1098 //FIXME this is moot
1099#if defined( _WIN32 )
1100 VLC_UNUSED( f_gain );
1101 if( save_vol_aout( "directsound" ) )
1102 config_PutFloat( p_intf, "directx-volume", i_volume / 100.f );
1103 if( save_vol_aout( "waveout" ) )
1104 config_PutFloat( p_intf, "waveout-volume", i_volume / 100.f );
1105#elif defined( Q_OS_MAC )
1106 VLC_UNUSED( f_gain );
1107 if( save_vol_aout( "auhal" ) )
1108 config_PutFloat( p_intf, "auhal-volume", i_volume / 100.f
1109 * AOUT_VOLUME_DEFAULT );
1110#elif defined( __OS2__ )
1111 if( save_vol_aout( "kai" ) )
1112 config_PutFloat( p_intf, "kai-gain", f_gain );
1113#else
1114 if( save_vol_aout( "alsa" ) )
1115 config_PutFloat( p_intf, "alsa-gain", f_gain );
1116 if( save_vol_aout( "jack" ) )
1117 config_PutFloat( p_intf, "jack-gain", f_gain );
1118#endif
1119#undef save_vol_aout
1120
1121 config_PutInt( p_intf, "volume-save", !b_reset_volume );
1122
1123 break;
1124 }
1125 case SPrefsSubtitles:
1126 {
1127 bool b_checked = qobject_cast<QCheckBox *>(optionWidgets["shadowCB"])->isChecked();
1128 if( b_checked && config_GetInt( p_intf, "freetype-shadow-opacity" ) == 0 ) {
1129 config_PutInt( p_intf, "freetype-shadow-opacity", 128 );
1130 }
1131 else if (!b_checked ) {
1132 config_PutInt( p_intf, "freetype-shadow-opacity", 0 );

1137 config_PutInt( p_intf, "freetype-background-opacity", 128 );
1138 }
1139 else if (!b_checked ) {
1140 config_PutInt( p_intf, "freetype-background-opacity", 0 );

1142
1143 }
1144 }
