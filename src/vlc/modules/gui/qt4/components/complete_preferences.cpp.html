53PrefsTree::PrefsTree( intf_thread_t *_p_intf, QWidget *_parent ) :
54 QTreeWidget( _parent ), p_intf( _p_intf )
55{
56 b_show_only_loaded = false;
57 /* General Qt options */
58 setAlternatingRowColors( true );
59 setHeaderHidden( true );
60
61 setIconSize( QSize( ITEM_HEIGHT,ITEM_HEIGHT ) );
62 setTextElideMode( Qt::ElideNone );
63
64 setUniformRowHeights( true );
65 CONNECT( this, itemExpanded(QTreeWidgetItem*), this, resizeColumns() );
66
67 /* Nice icons */
68#define BI( a,b) QIcon a##_icon = QIcon( b )
69 BI( audio, ":/prefsmenu/advanced/audio" );
70 BI( video, ":/prefsmenu/advanced/video" );
71 BI( input, ":/prefsmenu/advanced/codec" );
72 BI( sout, ":/prefsmenu/advanced/sout" );
73 BI( advanced, ":/prefsmenu/advanced/extended" );
74 BI( playlist, ":/prefsmenu/advanced/playlist" );
75 BI( interface, ":/prefsmenu/advanced/intf" );
76#undef BI
77
78 /* Build the tree for the main module */
79 module_t *p_module = module_get_main();
80
81 /* Initialisation and get the confsize */
82 PrefsItemData *data = NULL;
83 PrefsItemData *data_sub = NULL;
84 QTreeWidgetItem *current_item = NULL;
85 unsigned confsize;
86 module_config_t *const p_config = module_config_get (p_module, &confsize);
87
88 /* Go through the list of conf */
89 for( size_t i = 0; i < confsize; i++ )
90 {
91 const char *psz_help;
92 QIcon icon;
93
94 /* Work on a new item */
95 module_config_t *p_item = p_config + i;
96
97 switch( p_item->i_type )
98 {
99 /* This is a category */
100 case CONFIG_CATEGORY:
101 if( p_item->value.i == -1 ) break;
102
103 /* PrefsItemData Init */
104 data = new PrefsItemData();
105 data->name = qtr( config_CategoryNameGet( p_item->value.i ) );
106 psz_help = config_CategoryHelpGet( p_item->value.i );
107 if( psz_help )
108 data->help = qtr( psz_help );
109 else
110 data->help.clear();
111 data->i_type = PrefsItemData::TYPE_CATEGORY;
112 data->i_object_id = p_item->value.i;
113
114 /* This is a category, put a nice icon */
115 switch( p_item->value.i )
116 {
117#define CI(a,b) case a: icon = b##_icon;break
118 CI( CAT_AUDIO, audio );
119 CI( CAT_VIDEO, video );
120 CI( CAT_INPUT, input );
121 CI( CAT_SOUT, sout );
122 CI( CAT_ADVANCED, advanced );
123 CI( CAT_PLAYLIST, playlist );
124 CI( CAT_INTERFACE, interface );
125 }
126#undef CI
127
128 /* Create a new QTreeItem to display it in the tree at top level */
129 current_item = new QTreeWidgetItem();
130 current_item->setText( 0, data->name );
131 current_item->setIcon( 0 , icon );
132 //current_item->setSizeHint( 0, QSize( -1, ITEM_HEIGHT ) );
133 current_item->setData( 0, Qt::UserRole,
134 qVariantFromValue( data ) );
135 addTopLevelItem( current_item );
136 expandItem( current_item );
137 break;
138
139 /* This is a subcategory */
140 case CONFIG_SUBCATEGORY:
141 if( p_item->value.i == -1 ) break;
142
143 /* Special cases: move the main subcategories to the parent cat*/
144 if( data &&
145 ( p_item->value.i == SUBCAT_VIDEO_GENERAL ||
146 p_item->value.i == SUBCAT_ADVANCED_MISC ||
147 p_item->value.i == SUBCAT_INPUT_GENERAL ||
148 p_item->value.i == SUBCAT_INTERFACE_GENERAL ||
149 p_item->value.i == SUBCAT_SOUT_GENERAL||
150 p_item->value.i == SUBCAT_PLAYLIST_GENERAL||
151 p_item->value.i == SUBCAT_AUDIO_GENERAL ) )
152 {
153 /* Data still contains the correct thing */
154 data->i_type = PrefsItemData::TYPE_CATSUBCAT;
155 data->i_subcat_id = p_item->value.i;
156 data->name = qtr( config_CategoryNameGet( p_item->value.i ) );
157 psz_help = config_CategoryHelpGet( p_item->value.i );
158 if( psz_help )
159 data->help = qtr( psz_help );
160 else
161 data->help.clear();
162 current_item->setData( 0, Qt::UserRole,
163 QVariant::fromValue( data ) );
164 continue;
165 }
166
167 /* Normal Subcategories */
168
169 /* Process the Data */
170 data_sub = new PrefsItemData();
171 data_sub->name = qtr( config_CategoryNameGet( p_item->value.i) );
172 psz_help = config_CategoryHelpGet( p_item->value.i );
173 if( psz_help )
174 data_sub->help = qtr( psz_help );
175 else
176 data_sub->help.clear();
177 data_sub->i_type = PrefsItemData::TYPE_SUBCATEGORY;
178 data_sub->i_object_id = p_item->value.i;
179
180 /* Create a new TreeWidget */
181 QTreeWidgetItem *subcat_item = new QTreeWidgetItem();
182 subcat_item->setText( 0, data_sub->name );
183 subcat_item->setData( 0, Qt::UserRole,
184 qVariantFromValue( data_sub ) );
185 //subcat_item->setSizeHint( 0, QSize( -1, ITEM_HEIGHT ) );
186
187 /* Add it to the parent */
188 assert( current_item );
189 current_item->addChild( subcat_item );
190 break;
191
192 /* Other items don't need yet a place on the tree */
193 }
194 }
195 module_config_free( p_config );
196
197 size_t count;
198 module_t **p_list = module_list_get( &count );
199 /* Build the tree of plugins */
200 for( size_t i = 0; i < count; i++ )
201 {
202 p_module = p_list[i];
203 // Main module excluded
204 if( module_is_main( p_module) ) continue;
205
206 unsigned confsize;
207 int i_subcategory = 0, i_category = 0;
208
209 bool b_options = false;
210 module_config_t *const p_config = module_config_get (p_module, &confsize);
211
212 /* Loop through the configurations items */
213 for (size_t i = 0; i < confsize; i++)
214 {
215 const module_config_t *p_item = p_config + i;
216
217 if( p_item->i_type == CONFIG_CATEGORY )
218 i_category = p_item->value.i;
219 else if( p_item->i_type == CONFIG_SUBCATEGORY )
220 i_subcategory = p_item->value.i;
221
222 if( CONFIG_ITEM(p_item->i_type) )
223 b_options = true;
224
225 if( b_options && i_category && i_subcategory )
226 break;
227 }
228 module_config_free (p_config);
229
230 /* Dummy item, please proceed */
231 if( !b_options || i_category == 0 || i_subcategory == 0 ) continue;
232
233
234 // Locate the category item;
235 QTreeWidgetItem *subcat_item = NULL;
236 bool b_found = false;
237
238 for( int i_cat_index = 0 ; i_cat_index < topLevelItemCount();
239 i_cat_index++ )
240 {
241 /* Get the treeWidgetItem that correspond to the category */
242 QTreeWidgetItem *cat_item = topLevelItem( i_cat_index );
243 PrefsItemData *data = cat_item->data( 0, Qt::UserRole ).
244 value<PrefsItemData *>();
245
246 /* If we match the good category */
247 if( data->i_object_id == i_category )
248 {
249 for( int i_sc_index = 0; i_sc_index < cat_item->childCount();
250 i_sc_index++ )
251 {
252 subcat_item = cat_item->child( i_sc_index );
253 PrefsItemData *sc_data = subcat_item->data(0, Qt::UserRole).
254 value<PrefsItemData *>();
255 if( sc_data && sc_data->i_object_id == i_subcategory )
256 {
257 b_found = true;
258 break;
259 }
260 }
261 if( !b_found )
262 {
263 subcat_item = cat_item;
264 b_found = true;
265 }
266 break;
267 }
268 }
269 if( !b_found ) continue;
270
271 PrefsItemData *module_data = new PrefsItemData();
272 module_data->i_type = PrefsItemData::TYPE_MODULE;
273 module_data->psz_shortcut = strdup( module_get_object( p_module ) );
274 module_data->name = qtr( module_get_name( p_module, false ) );
275 module_data->help.clear();
276 const char *psz_help = module_get_help( p_module );
277 if ( psz_help )
278 module_data->help = qtr( psz_help );
279
280 QTreeWidgetItem *module_item = new QTreeWidgetItem();
281 module_item->setText( 0, module_data->name );
282 module_item->setData( 0, Qt::UserRole,
283 QVariant::fromValue( module_data) );
284 //module_item->setSizeHint( 0, QSize( -1, ITEM_HEIGHT ) );
285 subcat_item->addChild( module_item );
286 }
287
288 /* We got everything, just sort a bit */
289 sortItems( 0, Qt::AscendingOrder );
290
291 module_list_free( p_list );
292 resizeColumnToContents( 0 );
293}

305void PrefsTree::doAll( bool doclean )
306{
307 for( int i_cat_index = 0 ; i_cat_index < topLevelItemCount();
308 i_cat_index++ )
309 {
310 QTreeWidgetItem *cat_item = topLevelItem( i_cat_index );
311 for( int i_sc_index = 0; i_sc_index < cat_item->childCount();
312 i_sc_index++ )
313 {
314 QTreeWidgetItem *sc_item = cat_item->child( i_sc_index );
315 for( int i_module = 0 ; i_module < sc_item->childCount();
316 i_module++ )
317 {
318 PrefsItemData *data = sc_item->child( i_module )->
319 data( 0, Qt::UserRole).value<PrefsItemData *>();
320 if( data->panel && doclean )
321 {
322 delete data->panel;
323 data->panel = NULL;
324 }
325 else if( data->panel )
326 data->panel->apply();
327 }
328 PrefsItemData *data = sc_item->data( 0, Qt::UserRole).
329 value<PrefsItemData *>();
330 if( data->panel && doclean )
331 {
332 delete data->panel;
333 data->panel = NULL;
334 }
335 else if( data->panel )
336 data->panel->apply();
337 }
338 PrefsItemData *data = cat_item->data( 0, Qt::UserRole).
339 value<PrefsItemData *>();
340 if( data->panel && doclean )
341 {
342 delete data->panel;
343 data->panel = NULL;
344 }
345 else if( data->panel )
346 data->panel->apply();
347 }

350/* apply filter on tree item and recursively on its sub items
351 * returns whether the item was filtered */
352bool PrefsTree::filterItems( QTreeWidgetItem *item, const QString &text,
353 Qt::CaseSensitivity cs )
354{
355 bool sub_filtered = true;
356
357 for( int i = 0; i < item->childCount(); i++ )
358 {
359 QTreeWidgetItem *sub_item = item->child( i );
360 if ( !filterItems( sub_item, text, cs ) )
361 {
362 /* not all the sub items were filtered */
363 sub_filtered = false;
364 }
365 }
366
367 PrefsItemData *data = item->data( 0, Qt::UserRole ).value<PrefsItemData *>();
368
369 bool filtered = sub_filtered && !data->contains( text, cs );
370 if ( b_show_only_loaded && sub_filtered && !data->b_loaded )
371 filtered = true;
372 item->setExpanded( !sub_filtered );
373 item->setHidden( filtered );
374
375 return filtered;
376}

379/* collapse item if it's not selected or one of its sub items
380 * returns whether the item was collapsed */
381bool PrefsTree::collapseUnselectedItems( QTreeWidgetItem *item )
382{
383 bool sub_collapsed = true;
384
385 for( int i = 0; i < item->childCount(); i++ )
386 {
387 QTreeWidgetItem *sub_item = item->child( i );
388 if ( !collapseUnselectedItems( sub_item ) )
389 {
390 /* not all the sub items were collapsed */
391 sub_collapsed = false;
392 }
393 }
394
395 bool collapsed = sub_collapsed && !item->isSelected();
396 item->setExpanded( !sub_collapsed );
397 item->setHidden( false );
398
399 return collapsed;
400}

402static void populateLoadedSet( QSet<QString> *loaded, vlc_object_t *p_node )
403{
404 Q_ASSERT( loaded );
405 char *name = vlc_object_get_name( p_node );
406 if ( !EMPTY_STR( name ) ) loaded->insert( QString( name ) );
407 free( name );
408
409 vlc_list_t *l = vlc_list_children( p_node );
410 for( int i=0; i < l->i_count; i++ )
411 populateLoadedSet( loaded, l->p_values[i].p_object );
412 vlc_list_release( l );
413}

415/* Updates the PrefsItemData loaded status to reflect currently
416 * running modules */
417void PrefsTree::updateLoadedStatus( QTreeWidgetItem *item = NULL,
418 QSet<QString> *loaded = NULL )
419{
420 bool b_release = false;
421
422 if( loaded == NULL )
423 {
424 vlc_object_t *p_root = VLC_OBJECT( p_intf->p_libvlc );
425 loaded = new QSet<QString>();
426 populateLoadedSet( loaded, p_root );
427 b_release = true;
428 }
429
430 if ( item == NULL )
431 {
432 for( int i = 0 ; i < topLevelItemCount(); i++ )
433 updateLoadedStatus( topLevelItem( i ), loaded );
434 }
435 else
436 {
437 PrefsItemData *data = item->data( 0, Qt::UserRole )
438 .value<PrefsItemData *>();
439 data->b_loaded = loaded->contains( QString( data->psz_shortcut ) );
440
441 for( int i = 0; i < item->childCount(); i++ )
442 updateLoadedStatus( item->child( i ), loaded );
443 }
444
445 if ( b_release )
446 delete loaded;
447}

449/* apply filter on tree */
450void PrefsTree::filter( const QString &text )
451{
452 bool clear_filter = text.isEmpty() && ! b_show_only_loaded;
453
454 updateLoadedStatus();
455
456 for( int i = 0 ; i < topLevelItemCount(); i++ )
457 {
458 QTreeWidgetItem *cat_item = topLevelItem( i );
459 if ( clear_filter )
460 {
461 collapseUnselectedItems( cat_item );
462 }
463 else
464 {
465 filterItems( cat_item, text, Qt::CaseInsensitive );
466 }
467 }
468}

490/* go over the module config items and search text in psz_text
491 * also search the module name and head */
492bool PrefsItemData::contains( const QString &text, Qt::CaseSensitivity cs )
493{
494 /* Find our module */
495 module_t *p_module = NULL;
496 if( this->i_type == TYPE_CATEGORY )
497 return false;
498 else if( this->i_type == TYPE_MODULE )
499 p_module = module_find( this->psz_shortcut );
500 else
501 {
502 p_module = module_get_main();
503 assert( p_module );
504 }
505
506 unsigned confsize;
507 module_config_t *const p_config = module_config_get (p_module, &confsize),
508 *p_item = p_config,
509 *p_end = p_config + confsize;
510
511 if( this->i_type == TYPE_SUBCATEGORY || this->i_type == TYPE_CATSUBCAT )
512 {
513 while ( p_item < p_end )
514 {
515 if( p_item->i_type == CONFIG_SUBCATEGORY &&
516 (
517 ( this->i_type == TYPE_SUBCATEGORY &&
518 p_item->value.i == this->i_object_id )
519 ||
520 ( this->i_type == TYPE_CATSUBCAT &&
521 p_item->value.i == this->i_subcat_id )
522 )
523 )
524 break;
525 p_item++;
526 }
527 }
528
529 QString head;
530
531 if( this->i_type == TYPE_SUBCATEGORY || this->i_type == TYPE_CATSUBCAT )
532 {
533 head.clear();
534 p_item++; // Why that ? +1
535 }
536 else
537 {
538 head = QString( qtr( module_GetLongName( p_module ) ) );
539 }
540
541 if (name.contains( text, cs ) || head.contains( text, cs ) || help.contains( text, cs ))
542 {
543 module_config_free( p_config );
544 return true;
545 }
546
547 if( p_item ) do
548 {
549 if (
550 (
551 ( this->i_type == TYPE_SUBCATEGORY && p_item->value.i != this->i_object_id )
552 ||
553 ( this->i_type == TYPE_CATSUBCAT && p_item->value.i != this->i_subcat_id )
554 ) &&
555 ( p_item->i_type == CONFIG_CATEGORY || p_item->i_type == CONFIG_SUBCATEGORY )
556 ) break;
557
558 if( p_item->b_internal ) continue;
559
560 if ( p_item->psz_text && qtr( p_item->psz_text ).contains( text, cs ) )
561 {
562 module_config_free( p_config );
563 return true;
564 }
565 }
566 while (
567 !(
568 ( this->i_type == TYPE_SUBCATEGORY || this->i_type == TYPE_CATSUBCAT )
569 &&
570 ( p_item->i_type == CONFIG_CATEGORY || p_item->i_type == CONFIG_SUBCATEGORY )
571 )
572 && ( ++p_item < p_end )
573 );
574
575 module_config_free( p_config );
576 return false;
577}

587AdvPrefsPanel::AdvPrefsPanel( intf_thread_t *_p_intf, QWidget *_parent,
588 PrefsItemData * data ) :
589 QWidget( _parent ), p_intf( _p_intf )
590{
591 /* Find our module */
592 module_t *p_module = NULL;
593 p_config = NULL;
594 if( data->i_type == PrefsItemData::TYPE_CATEGORY )
595 return;
596 else if( data->i_type == PrefsItemData::TYPE_MODULE )
597 p_module = module_find( data->psz_shortcut );
598 else
599 {
600 p_module = module_get_main();
601 assert( p_module );
602 }
603
604 unsigned confsize;
605 p_config = module_config_get( p_module, &confsize );
606 module_config_t *p_item = p_config,
607 *p_end = p_config + confsize;
608
609 if( data->i_type == PrefsItemData::TYPE_SUBCATEGORY ||
610 data->i_type == PrefsItemData::TYPE_CATSUBCAT )
611 {
612 while (p_item < p_end)
613 {
614 if( p_item->i_type == CONFIG_SUBCATEGORY &&
615 ( ( data->i_type == PrefsItemData::TYPE_SUBCATEGORY &&
616 p_item->value.i == data->i_object_id ) ||
617 ( data->i_type == PrefsItemData::TYPE_CATSUBCAT &&
618 p_item->value.i == data->i_subcat_id ) ) )
619 break;
620 p_item++;
621 }
622 }
623
624 /* Widgets now */
625 global_layout = new QVBoxLayout();
626 global_layout->setMargin( 2 );
627 QString head;
628 QString help;
629
630 help = QString( data->help );
631
632 if( data->i_type == PrefsItemData::TYPE_SUBCATEGORY ||
633 data->i_type == PrefsItemData::TYPE_CATSUBCAT )
634 {
635 head = QString( data->name );
636 p_item++; // Why that ?
637 }
638 else
639 {
640 head = QString( qtr( module_GetLongName( p_module ) ) );
641 }
642
643 QLabel *titleLabel = new QLabel( head );
644 QFont titleFont = QApplication::font();
645 titleFont.setPointSize( titleFont.pointSize() + 6 );
646 titleLabel->setFont( titleFont );
647
648 // Title <hr>
649 QFrame *title_line = new QFrame;
650 title_line->setFrameShape(QFrame::HLine);
651 title_line->setFrameShadow(QFrame::Sunken);
652
653 QLabel *helpLabel = new QLabel( help, this );
654 helpLabel->setWordWrap( true );
655
656 global_layout->addWidget( titleLabel );
657 global_layout->addWidget( title_line );
658 global_layout->addWidget( helpLabel );
659
660 QGroupBox *box = NULL;
661 QGridLayout *boxlayout = NULL;
662
663 QScrollArea *scroller= new QScrollArea;
664 scroller->setFrameStyle( QFrame::NoFrame );
665 QWidget *scrolled_area = new QWidget;
666
667 QGridLayout *layout = new QGridLayout();
668 int i_line = 0, i_boxline = 0;
669 bool has_hotkey = false;
670
671 if( p_item ) do
672 {
673 if( ( ( data->i_type == PrefsItemData::TYPE_SUBCATEGORY &&
674 p_item->value.i != data->i_object_id ) ||
675 ( data->i_type == PrefsItemData::TYPE_CATSUBCAT &&
676 p_item->value.i != data->i_subcat_id ) ) &&
677 ( p_item->i_type == CONFIG_CATEGORY ||
678 p_item->i_type == CONFIG_SUBCATEGORY ) )
679 break;
680 if( p_item->b_internal ) continue;
681
682 if( p_item->i_type == CONFIG_SECTION )
683 {
684 if( box )
685 {
686 box->setLayout( boxlayout );
687 box->show();
688 layout->addWidget( box, i_line, 0, 1, -1 );
689 i_line++;
690 }
691 box = new QGroupBox( qtr( p_item->psz_text ), this );
692 box->hide();
693 boxlayout = new QGridLayout();
694 }
695 /* Only one hotkey control */
696 if( p_item->i_type == CONFIG_ITEM_KEY )
697 {
698 if( has_hotkey )
699 continue;
700 has_hotkey = true;
701 }
702
703 ConfigControl *control;
704 if( ! box )
705 control = ConfigControl::createControl( VLC_OBJECT( p_intf ),
706 p_item, this, layout, i_line );
707 else
708 control = ConfigControl::createControl( VLC_OBJECT( p_intf ),
709 p_item, this, boxlayout, i_boxline );
710 if( !control )
711 continue;
712
713 if( box ) i_boxline++;
714 else i_line++;
715 controls.append( control );
716 }
717 while( !( ( data->i_type == PrefsItemData::TYPE_SUBCATEGORY ||
718 data->i_type == PrefsItemData::TYPE_CATSUBCAT ) &&
719 ( p_item->i_type == CONFIG_CATEGORY ||
720 p_item->i_type == CONFIG_SUBCATEGORY ) )
721 && ( ++p_item < p_end ) );
722
723 if( box && i_boxline > 0 )
724 {
725 box->setLayout( boxlayout );
726 box->show();
727 layout->addWidget( box, i_line, 0, 1, -1 );
728 }
729
730 scrolled_area->setSizePolicy( QSizePolicy::Preferred,QSizePolicy::Fixed );
731 scrolled_area->setLayout( layout );
732 scroller->setWidget( scrolled_area );
