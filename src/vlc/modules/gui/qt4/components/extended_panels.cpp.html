66QString OptionFromWidgetName( QObject *obj )
67{
68 /* Gruik ? ... nah */
69 QString option = obj->objectName().replace( "Slider", "" )
70 .replace( "Combo" , "" )
71 .replace( "Dial" , "" )
72 .replace( "Check" , "" )
73 .replace( "Spin" , "" )
74 .replace( "Text" , "" );
75 for( char a = 'A'; a <= 'Z'; a++ )
76 {
77 option = option.replace( QString( a ),
78 QString( '-' ) + QString( a + 'a' - 'A' ) );
79 }
80 return option;

278static char *ChangeFiltersString( struct intf_thread_t *p_intf, const char *psz_filter_type, const char *psz_name, bool b_add )
279{
280 char *psz_parser, *psz_string;
281
282 psz_string = config_GetPsz( p_intf, psz_filter_type );
283
284 if( !psz_string ) psz_string = strdup( "" );
285
286 psz_parser = strstr( psz_string, psz_name );
287
288 if( b_add )
289 {
290 if( !psz_parser )
291 {
292 psz_parser = psz_string;
293 if( asprintf( &psz_string, ( *psz_string ) ? "%s:%s" : "%s%s",
294 psz_string, psz_name ) == -1 )
295 {
296 free( psz_parser );
297 return NULL;
298 }
299 free( psz_parser );
300 }
301 else
302 {
303 free( psz_string );
304 return NULL;
305 }
306 }
307 else
308 {
309 if( psz_parser )
310 {
311 if( *( psz_parser + strlen( psz_name ) ) == ':' )
312 {
313 memmove( psz_parser, psz_parser + strlen( psz_name ) + 1,
314 strlen( psz_parser + strlen( psz_name ) + 1 ) + 1 );
315 }
316 else
317 {
318 *psz_parser = '\0';
319 }
320
321 /* Remove trailing : : */
322 size_t i_len = strlen( psz_string );
323 if( i_len > 0 && *( psz_string + i_len - 1 ) == ':' )
324 {
325 *( psz_string + i_len - 1 ) = '\0';
326 }
327 }
328 else
329 {
330 free( psz_string );
331 return NULL;
332 }
333 }
334 return psz_string;
335}

357static void ChangeVFiltersString( struct intf_thread_t *p_intf, const char *psz_name, bool b_add )
358{
359 char *psz_string;
360 const char *psz_filter_type;
361
362 module_t *p_obj = module_find( psz_name );
363 if( !p_obj )
364 {
365 msg_Err( p_intf, "Unable to find filter module \"%s\".", psz_name );
366 return;
367 }
368
369 if( module_provides( p_obj, "video splitter" ) )
370 {
371 psz_filter_type = "video-splitter";
372 }
373 else if( module_provides( p_obj, "video filter2" ) )
374 {
375 psz_filter_type = "video-filter";
376 }
377 else if( module_provides( p_obj, "sub source" ) )
378 {
379 psz_filter_type = "sub-source";
380 }
381 else if( module_provides( p_obj, "sub filter" ) )
382 {
383 psz_filter_type = "sub-filter";
384 }
385 else
386 {
387 msg_Err( p_intf, "Unknown video filter type." );
388 return;
389 }
390
391 psz_string = ChangeFiltersString( p_intf, psz_filter_type, psz_name, b_add );
392 if( !psz_string )
393 return;
394
395 /* Vout is not kept, so put that in the config */
396 config_PutPsz( p_intf, psz_filter_type, psz_string );
397
398 /* Try to set on the fly */
399 if( !strcmp( psz_filter_type, "video-splitter" ) )
400 {
401 playlist_t *p_playlist = THEPL;
402 var_SetString( p_playlist, psz_filter_type, psz_string );
403 }
404 else
405 {
406 vout_thread_t *p_vout = THEMIM->getVout();
407 if( p_vout )
408 {
409 var_SetString( p_vout, psz_filter_type, psz_string );
410 vlc_object_release( p_vout );
411 }
412 }
413
414 free( psz_string );
415}

453void ExtVideo::initComboBoxItems( QObject *widget )
454{
455 QComboBox *combobox = qobject_cast<QComboBox*>( widget );
456 if( !combobox ) return;
457
458 QString option = OptionFromWidgetName( widget );
459 module_config_t *p_item = config_FindConfig( VLC_OBJECT( p_intf ),
460 qtu( option ) );
461 if( p_item == NULL )
462 {
463 msg_Err( p_intf, "Couldn't find option \"%s\".", qtu( option ) );
464 return;
465 }
466
467 if( p_item->i_type == CONFIG_ITEM_INTEGER
468 || p_item->i_type == CONFIG_ITEM_BOOL )
469 {
470 int64_t *values;
471 char **texts;
472 ssize_t count = config_GetIntChoices( VLC_OBJECT( p_intf ),
473 qtu( option ), &values, &texts );
474 for( ssize_t i = 0; i < count; i++ )
475 {
476 combobox->addItem( qtr( texts[i] ), qlonglong(values[i]) );
477 free( texts[i] );
478 }
479 free( texts );
480 free( values );
481 }
482 else if( p_item->i_type == CONFIG_ITEM_STRING )
483 {
484 char **values;
485 char **texts;
486 ssize_t count = config_GetPszChoices( VLC_OBJECT( p_intf ),
487 qtu( option ), &values, &texts );
488 for( ssize_t i = 0; i < count; i++ )
489 {
490 combobox->addItem( qtr( texts[i] ), qfu(values[i]) );
491 free( texts[i] );
492 free( values[i] );
493 }
494 free( texts );
495 free( values );
496 }
497}

499void ExtVideo::setWidgetValue( QObject *widget )
500{
501 QString module = ModuleFromWidgetName( widget->parent() );
502 //std::cout << "Module name: " << module.toStdString() << std::endl;
503 QString option = OptionFromWidgetName( widget );
504 //std::cout << "Option name: " << option.toStdString() << std::endl;
505
506 vlc_object_t *p_obj = ( vlc_object_t * )
507 vlc_object_find_name( p_intf->p_libvlc, qtu( module ) );
508 int i_type;
509 vlc_value_t val;
510
511 if( !p_obj )
512 {
513#if 0
514 msg_Dbg( p_intf,
515 "Module instance %s not found, looking in config values.",
516 qtu( module ) );
517#endif
518 i_type = config_GetType( p_intf, qtu( option ) ) & VLC_VAR_CLASS;
519 switch( i_type )
520 {
521 case VLC_VAR_INTEGER:
522 case VLC_VAR_BOOL:
523 val.i_int = config_GetInt( p_intf, qtu( option ) );
524 break;
525 case VLC_VAR_FLOAT:
526 val.f_float = config_GetFloat( p_intf, qtu( option ) );
527 break;
528 case VLC_VAR_STRING:
529 val.psz_string = config_GetPsz( p_intf, qtu( option ) );
530 break;
531 }
532 }
533 else
534 {
535 i_type = var_Type( p_obj, qtu( option ) ) & VLC_VAR_CLASS;
536 var_Get( p_obj, qtu( option ), &val );
537 vlc_object_release( p_obj );
538 }
539
540 /* Try to cast to all the widgets we're likely to encounter. Only
541 * one of the casts is expected to work. */
542 QSlider *slider = qobject_cast<QSlider*> ( widget );
543 QCheckBox *checkbox = qobject_cast<QCheckBox*> ( widget );
544 QSpinBox *spinbox = qobject_cast<QSpinBox*> ( widget );
545 QDoubleSpinBox *doublespinbox = qobject_cast<QDoubleSpinBox*>( widget );
546 VLCQDial *dial = qobject_cast<VLCQDial*> ( widget );
547 QLineEdit *lineedit = qobject_cast<QLineEdit*> ( widget );
548 QComboBox *combobox = qobject_cast<QComboBox*> ( widget );
549
550 if( i_type == VLC_VAR_INTEGER || i_type == VLC_VAR_BOOL )
551 {
552 if( slider ) slider->setValue( val.i_int );
553 else if( checkbox ) checkbox->setCheckState( val.i_int? Qt::Checked
554 : Qt::Unchecked );
555 else if( spinbox ) spinbox->setValue( val.i_int );
556 else if( dial ) dial->setValue( ( 540-val.i_int )%360 );
557 else if( lineedit )
558 {
559 char str[30];
560 snprintf( str, sizeof(str), "%06"PRIX64, val.i_int );
561 lineedit->setText( str );
562 }
563 else if( combobox ) combobox->setCurrentIndex(
564 combobox->findData( qlonglong(val.i_int) ) );
565 else msg_Warn( p_intf, "Could not find the correct Integer widget" );
566 }
567 else if( i_type == VLC_VAR_FLOAT )
568 {
569 if( slider ) slider->setValue( ( int )( val.f_float*( double )slider->tickInterval() ) ); /* hack alert! */
570 else if( doublespinbox ) doublespinbox->setValue( val.f_float );
571 else if( dial ) dial->setValue( (540 - lroundf(val.f_float)) % 360 );
572 else msg_Warn( p_intf, "Could not find the correct Float widget" );
573 }
574 else if( i_type == VLC_VAR_STRING )
575 {
576 if( lineedit ) lineedit->setText( qfu( val.psz_string ) );
577 else if( combobox ) combobox->setCurrentIndex(
578 combobox->findData( qfu( val.psz_string ) ) );
579 else msg_Warn( p_intf, "Could not find the correct String widget" );
580 free( val.psz_string );
581 }
582 else
583 if( p_obj )
584 msg_Err( p_intf,
585 "Module %s's %s variable is of an unsupported type ( %d )",
586 qtu( module ),
587 qtu( option ),
588 i_type );
589}

591void ExtVideo::updateFilterOptions()
592{
593 QString module = ModuleFromWidgetName( sender()->parent() );
594 //msg_Dbg( p_intf, "Module name: %s", qtu( module ) );
595 QString option = OptionFromWidgetName( sender() );
596 //msg_Dbg( p_intf, "Option name: %s", qtu( option ) );
597
598 vlc_object_t *p_obj = ( vlc_object_t * )
599 vlc_object_find_name( p_intf->p_libvlc, qtu( module ) );
600 int i_type;
601 bool b_is_command;
602 if( !p_obj )
603 {
604 msg_Warn( p_intf, "Module %s not found. You'll need to restart the filter to take the change into account.", qtu( module ) );
605 i_type = config_GetType( p_intf, qtu( option ) );
606 b_is_command = false;
607 }
608 else
609 {
610 i_type = var_Type( p_obj, qtu( option ) );
611 if( i_type == 0 )
612 i_type = config_GetType( p_intf, qtu( option ) );
613 b_is_command = ( i_type & VLC_VAR_ISCOMMAND );
614 }
615
616 /* Try to cast to all the widgets we're likely to encounter. Only
617 * one of the casts is expected to work. */
618 QSlider *slider = qobject_cast<QSlider*> ( sender() );
619 QCheckBox *checkbox = qobject_cast<QCheckBox*> ( sender() );
620 QSpinBox *spinbox = qobject_cast<QSpinBox*> ( sender() );
621 QDoubleSpinBox *doublespinbox = qobject_cast<QDoubleSpinBox*>( sender() );
622 VLCQDial *dial = qobject_cast<VLCQDial*> ( sender() );
623 QLineEdit *lineedit = qobject_cast<QLineEdit*> ( sender() );
624 QComboBox *combobox = qobject_cast<QComboBox*> ( sender() );
625
626 i_type &= VLC_VAR_CLASS;
627 if( i_type == VLC_VAR_INTEGER || i_type == VLC_VAR_BOOL )
628 {
629 int i_int = 0;
630 if( slider ) i_int = slider->value();
631 else if( checkbox ) i_int = checkbox->checkState() == Qt::Checked;
632 else if( spinbox ) i_int = spinbox->value();
633 else if( dial ) i_int = ( 540-dial->value() )%360;
