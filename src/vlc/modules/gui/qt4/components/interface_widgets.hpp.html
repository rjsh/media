173class TimeLabel : public ClickableQLabel
174{
175 Q_OBJECT
176public:
177 enum Display
178 {
179 Elapsed,
180 Remaining,
181 Both
182 };
183
184 TimeLabel( intf_thread_t *_p_intf, TimeLabel::Display _displayType = TimeLabel::Both );
185protected:
186 virtual void mousePressEvent( QMouseEvent *event )
187 {
188 if( displayType == TimeLabel::Elapsed ) return;
189 toggleTimeDisplay();
190 event->accept();
191 }
192 virtual void mouseDoubleClickEvent( QMouseEvent *event )
193 {
194 if( displayType != TimeLabel::Both ) return;
195 event->accept();
196 toggleTimeDisplay();
197 ClickableQLabel::mouseDoubleClickEvent( event );
198 }
199private:
200 intf_thread_t *p_intf;
201 bool b_remainingTime;
202 int cachedLength;
203 TimeLabel::Display displayType;
204
205 char psz_length[MSTRTIME_MAX_SIZE];
206 char psz_time[MSTRTIME_MAX_SIZE];
207 void toggleTimeDisplay();
208private slots:
209 void setDisplayPosition( float pos, int64_t time, int length );
210 void setDisplayPosition( float pos );
211};
