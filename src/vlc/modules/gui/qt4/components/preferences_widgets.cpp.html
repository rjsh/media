60QString formatTooltip(const QString & tooltip)
61{
62 QString text = tooltip;
63 text.replace("\n", "<br/>");
64
65 QString formatted =
66 "<html><head><meta name=\"qrichtext\" content=\"1\" />"
67 "<style type=\"text/css\"> p, li { white-space: pre-wrap; } </style></head>"
68 "<body style=\" font-family:'Sans Serif'; "
69 "font-style:normal; text-decoration:none;\">"
70 "<p style=\" margin-top:0px; margin-bottom:0px; margin-left:0px; "
71 "margin-right:0px; -qt-block-indent:0; text-indent:0px;\">" +
72 text + "</p></body></html>";
73 return formatted;
74}

76ConfigControl *ConfigControl::createControl( vlc_object_t *p_this,
77 module_config_t *p_item,
78 QWidget *parent,
79 QGridLayout *l, int line )
80{
81 ConfigControl *p_control = NULL;
82
83 switch( p_item->i_type )
84 {
85 case CONFIG_ITEM_MODULE:
86 p_control = new StringListConfigControl( p_this, p_item, parent );
87 break;
88 case CONFIG_ITEM_MODULE_CAT:
89 p_control = new ModuleConfigControl( p_this, p_item, parent );
90 break;
91 case CONFIG_ITEM_MODULE_LIST:
92 p_control = new ModuleListConfigControl( p_this, p_item, parent, false );
93 break;
94 case CONFIG_ITEM_MODULE_LIST_CAT:
95 p_control = new ModuleListConfigControl( p_this, p_item, parent, true );
96 break;
97 case CONFIG_ITEM_STRING:
98 if( p_item->list.psz_cb )
99 p_control = new StringListConfigControl( p_this, p_item, parent );
100 else
101 p_control = new StringConfigControl( p_this, p_item, parent, false );
102 break;
103 case CONFIG_ITEM_PASSWORD:
104 p_control = new StringConfigControl( p_this, p_item, parent, true );
105 break;
106 case CONFIG_ITEM_RGB:
107 p_control = new ColorConfigControl( p_this, p_item, parent );
108 break;
109 case CONFIG_ITEM_INTEGER:
110 if( p_item->list.i_cb )
111 p_control = new IntegerListConfigControl( p_this, p_item, parent, false );
112 else if( p_item->min.i || p_item->max.i )
113 p_control = new IntegerRangeConfigControl( p_this, p_item, parent );
114 else
115 p_control = new IntegerConfigControl( p_this, p_item, parent );
116 break;
117 case CONFIG_ITEM_LOADFILE:
118 case CONFIG_ITEM_SAVEFILE:
119 p_control = new FileConfigControl( p_this, p_item, parent );
120 break;
121 case CONFIG_ITEM_DIRECTORY:
122 p_control = new DirectoryConfigControl( p_this, p_item, parent );
123 break;
124 case CONFIG_ITEM_FONT:
125 p_control = new FontConfigControl( p_this, p_item, parent );
126 break;
127 case CONFIG_ITEM_KEY:
128 p_control = new KeySelectorControl( p_this, p_item, parent );
129 break;
130 case CONFIG_ITEM_BOOL:
131 p_control = new BoolConfigControl( p_this, p_item, parent );
132 break;
133 case CONFIG_ITEM_FLOAT:
134 if( p_item->min.f || p_item->max.f )
135 p_control = new FloatRangeConfigControl( p_this, p_item, parent );
136 else
137 p_control = new FloatConfigControl( p_this, p_item, parent );
138 break;
139 default:
140 break;
141 }
142 if ( p_control ) p_control->insertIntoExistingGrid( l, line );
143 return p_control;
144}

429void StringListConfigControl::finish(module_config_t *p_module_config )
430{
431 combo->setEditable( false );
432 CONNECT( combo, currentIndexChanged ( int ), this, comboIndexChanged( int ) );
433
434 if(!p_module_config) return;
435
436 char **values, **texts;
437 ssize_t count = config_GetPszChoices( p_this, p_item->psz_name,
438 &values, &texts );
439 for( ssize_t i = 0; i < count && texts; i++ )
440 {
441 if( texts[i] == NULL || values[i] == NULL )
442 continue;
443
444 combo->addItem( qfu(texts[i]), QVariant( qfu(values[i])) );
445 if( !strcmp( p_item->value.psz ? p_item->value.psz : "", values[i] ) )
446 combo->setCurrentIndex( combo->count() - 1 );
447 free( texts[i] );
448 free( values[i] );
449 }
450 free( texts );
451 free( values );
452
453 if( p_module_config->psz_longtext )
454 {
455 QString tipText = qtr(p_module_config->psz_longtext);
456 combo->setToolTip( formatTooltip(tipText) );
457 if( label )
458 label->setToolTip( formatTooltip(tipText) );
459 }
460 if( label )
461 label->setBuddy( combo );
462}

469void setfillVLCConfigCombo( const char *configname, intf_thread_t *p_intf,
470 QComboBox *combo )
471{
472 module_config_t *p_config =
473 config_FindConfig( VLC_OBJECT(p_intf), configname );
474 if( p_config == NULL )
475 return;
476
477 if( (p_config->i_type & 0xF0) == CONFIG_ITEM_STRING )
478 {
479 char **values, **texts;
480 ssize_t count = config_GetPszChoices(VLC_OBJECT(p_intf),
481 configname, &values, &texts);
482 for( ssize_t i = 0; i < count; i++ )
483 {
484 combo->addItem( qtr(texts[i]), QVariant(qfu(values[i])) );
485 if( p_config->value.psz && !strcmp(p_config->value.psz, values[i]) )
486 combo->setCurrentIndex( i );
487 free( texts[i] );
488 free( values[i] );
489 }
490 free( texts );
491 free( values );
492 }
493 else
494 {
495 int64_t *values;
496 char **texts;
497 ssize_t count = config_GetIntChoices(VLC_OBJECT(p_intf), configname,
498 &values, &texts);
499 for( ssize_t i = 0; i < count; i++ )
500 {
501 combo->addItem( qtr(texts[i]), QVariant(qlonglong(values[i])) );
502 if( p_config->value.i == values[i] )
503 combo->setCurrentIndex( i );
504 free( texts[i] );
505 }
506 free( texts );
507 }
508
509 if( p_config->psz_longtext != NULL )
510 combo->setToolTip( qfu( p_config->psz_longtext ) );
511}

539void ModuleConfigControl::finish( )
540{
541 combo->setEditable( false );
542
543 /* build a list of available modules */
544 size_t count;
545 module_t **p_list = module_list_get( &count );
546 combo->addItem( qtr("Default") );
547 for( size_t i = 0; i < count; i++ )
548 {
549 module_t *p_parser = p_list[i];
550
551 if( !strcmp( module_get_object( p_parser ), "core" ) ) continue;
552
553 unsigned confsize;
554 module_config_t *p_config;
555
556 p_config = module_config_get (p_parser, &confsize);
557 for (size_t i = 0; i < confsize; i++)
558 {
559 /* Hack: required subcategory is stored in i_min */
560 const module_config_t *p_cfg = p_config + i;
561 if( p_cfg->i_type == CONFIG_SUBCATEGORY &&
562 p_cfg->value.i == p_item->min.i )
563 combo->addItem( qtr( module_GetLongName( p_parser )),
564 QVariant( module_get_object( p_parser ) ) );
565 if( p_item->value.psz && !strcmp( p_item->value.psz,
566 module_get_object( p_parser ) ) )
567 combo->setCurrentIndex( combo->count() - 1 );
568 }
569 module_config_free (p_config);
570 }
571 module_list_free( p_list );
572
573 if( p_item->psz_longtext )
574 {
575 QString tipText = qtr(p_item->psz_longtext);
576 combo->setToolTip( formatTooltip(tipText) );
577 if( label )
578 label->setToolTip( formatTooltip(tipText) );
579 }
580 if( label )
581 label->setBuddy( combo );
582}

655void ModuleListConfigControl::finish( bool bycat )
656{
657 /* build a list of available modules */
658 size_t count;
659 module_t **p_list = module_list_get( &count );
660 for( size_t i = 0; i < count; i++ )
661 {
662 module_t *p_parser = p_list[i];
663
664 if( bycat )
665 {
666 if( !strcmp( module_get_object( p_parser ), "core" ) ) continue;
667
668 unsigned confsize;
669 module_config_t *p_config = module_config_get (p_parser, &confsize);
670
671 for (size_t i = 0; i < confsize; i++)
672 {
673 module_config_t *p_cfg = p_config + i;
674 /* Hack: required subcategory is stored in i_min */
675 if( p_cfg->i_type == CONFIG_SUBCATEGORY &&
676 p_cfg->value.i == p_item->min.i )
677 {
678 checkbox_lists( p_parser );
679 }
680
681 /* Parental Advisory HACK:
682 * Selecting HTTP, RC and Telnet interfaces is difficult now
683 * since they are just the lua interface module */
684 if( p_cfg->i_type == CONFIG_SUBCATEGORY &&
685 !strcmp( module_get_object( p_parser ), "lua" ) &&
686 !strcmp( p_item->psz_name, "extraintf" ) &&
687 p_cfg->value.i == p_item->min.i )
688 {
689 checkbox_lists( "Web", "Lua HTTP", "http" );
690 checkbox_lists( "Telnet", "Lua Telnet", "telnet" );
691#ifndef _WIN32
692 checkbox_lists( "Console", "Lua CLI", "cli" );
693#endif
694 }
695 }
696 module_config_free (p_config);
697 }
698 else if( module_provides( p_parser, p_item->psz_type ) )
699 {
700 checkbox_lists(p_parser);
701 }
702 }
703 module_list_free( p_list );

1194
1195void KeySelectorControl::finish()
1196{
1197 if( label && p_item->psz_longtext )
1198 label->setToolTip( formatTooltip( qtr( p_item->psz_longtext ) ) );
1199
1200 /* Fill the table */
1201
1202 /* Get the main Module */
1203 module_t *p_main = module_get_main();
1204 assert( p_main );
1205
1206 /* Access to the module_config_t */
1207 unsigned confsize;
1208 module_config_t *p_config;
1209
1210 p_config = module_config_get (p_main, &confsize);
1211
1212 QMap<QString, QString> global_keys;
1213 for (size_t i = 0; i < confsize; i++)
1214 {
1215 module_config_t *p_config_item = p_config + i;
1216
1217 /* If we are a (non-global) key option not empty */
1218 if( CONFIG_ITEM(p_config_item->i_type) && p_config_item->psz_name != NULL
1219 && !strncmp( p_config_item->psz_name , "key-", 4 )
1220 && !EMPTY_STR( p_config_item->psz_text ) )
1221 {
1222 /*
1223 Each tree item has:
1224 - QString text in column 0
1225 - QString name in data of column 0
1226 - KeyValue in String in column 1
1227 */
1228 QTreeWidgetItem *treeItem = new QTreeWidgetItem();
1229 treeItem->setText( ACTION_COL, qtr( p_config_item->psz_text ) );
1230 treeItem->setData( ACTION_COL, Qt::UserRole,
1231 QVariant( qfu( p_config_item->psz_name ) ) );
1232
1233 QString keys = qfu(p_config_item->value.psz ? _(p_config_item->value.psz) : "");
1234 treeItem->setText( HOTKEY_COL, keys );
1235 treeItem->setToolTip( HOTKEY_COL, qtr("Double click to change.\nDelete key to remove.") );
1236 treeItem->setToolTip( GLOBAL_HOTKEY_COL, qtr("Double click to change.\nDelete key to remove.") );
1237 treeItem->setData( HOTKEY_COL, Qt::UserRole, QVariant( p_config_item->value.psz ) );
1238 table->addTopLevelItem( treeItem );
1239 continue;
1240 }
1241
1242 if( CONFIG_ITEM(p_config_item->i_type) && p_config_item->psz_name != NULL
1243 && !strncmp( p_config_item->psz_name , "global-key", 10 )
1244 && !EMPTY_STR( p_config_item->psz_text )
1245 && !EMPTY_STR( p_config_item->value.psz ) )
1246 {
1247 global_keys.insertMulti( qtr( p_config_item->psz_text ), qfu( p_config_item->value.psz ) );
1248 }
1249 }
1250
1251 QMap<QString, QString>::const_iterator i = global_keys.constBegin();
1252 while (i != global_keys.constEnd())
1253 {
1254 QList<QTreeWidgetItem *> list =
1255 table->findItems( i.key(), Qt::MatchExactly|Qt::MatchWrap, ACTION_COL );
1256 if( list.count() >= 1 )
1257 {
1258 QString keys = i.value();
1259 list[0]->setText( GLOBAL_HOTKEY_COL, keys );
1260 list[0]->setData( GLOBAL_HOTKEY_COL, Qt::UserRole, keys );
1261 }
1262 if( list.count() >= 2 )
1263 msg_Dbg( p_this, "This is probably wrong, %s", qtu(i.key()) );
1264
1265 i++;
1266 }
1267
1268 module_config_free (p_config);
1269
1270 table->resizeColumnToContents( 0 );
1271
1272 CONNECT( table, itemActivated( QTreeWidgetItem *, int ),
1273 this, selectKey( QTreeWidgetItem *, int ) );
1274}

1296void KeySelectorControl::selectKey( QTreeWidgetItem *keyItem, int column )
1297{
1298 /* This happens when triggered by ClickEater */
1299 if( keyItem == NULL ) keyItem = table->currentItem();
1300
1301 /* This can happen when nothing is selected on the treeView
1302 and the shortcutValue is clicked */
1303 if( !keyItem ) return;
1304
1305 /* If clicked on the first column, assuming user wants the normal hotkey */
1306 if( column == ACTION_COL ) column = HOTKEY_COL;
1307
1308 bool b_global = ( column == GLOBAL_HOTKEY_COL );
1309
1310 /* Launch a small dialog to ask for a new key */
1311 KeyInputDialog *d = new KeyInputDialog( table, keyItem->text( ACTION_COL ),
1312 table, b_global );
1313 d->setExistingkeysSet( &existingkeys );
1314 d->exec();
1315
1316 if( d->result() == QDialog::Accepted )
1317 {
1318 QString newKey = VLCKeyToString( d->keyValue, false );
1319
1320 /* In case of conflict, reset other keys*/
1321 if( d->conflicts )
1322 {
1323 QTreeWidgetItem *it;
1324 for( int i = 0; i < table->topLevelItemCount() ; i++ )
1325 {
1326 it = table->topLevelItem(i);
1327 if( ( keyItem != it ) &&
1328 ( it->data( column, Qt::UserRole ).toString() == newKey ) )
1329 {
1330 it->setText( column, NULL );
1331 it->setData( column, Qt::UserRole, QVariant() );
1332 }
1333 }
1334 }
1335
1336 keyItem->setText( column, VLCKeyToString( d->keyValue, true ) );
1337 keyItem->setData( column, Qt::UserRole, newKey );
1338 }
1339 else if( d->result() == 2 )
1340 {
1341 keyItem->setText( column, NULL );
1342 keyItem->setData( column, Qt::UserRole, QVariant() );
1343 }
1344
1345 delete d;
1346}

1348void KeySelectorControl::doApply()
1349{
1350 QTreeWidgetItem *it;
1351 for( int i = 0; i < table->topLevelItemCount() ; i++ )
1352 {
1353 it = table->topLevelItem(i);
1354 if( it->data( HOTKEY_COL, Qt::UserRole ).toInt() >= 0 )
1355 config_PutPsz( p_this,
1356 qtu( it->data( ACTION_COL, Qt::UserRole ).toString() ),
1357 qtu( it->data( HOTKEY_COL, Qt::UserRole ).toString() ) );
1358
1359 config_PutPsz( p_this,
1360 qtu( "global-" + it->data( ACTION_COL, Qt::UserRole ).toString() ),
1361 qtu( it->data( GLOBAL_HOTKEY_COL, Qt::UserRole ).toString() ) );

1365bool KeySelectorControl::eventFilter( QObject *obj, QEvent *e )
1366{
1367 if( obj != table || e->type() != QEvent::KeyPress )
1368 return ConfigControl::eventFilter(obj, e);
1369
1370 QKeyEvent *keyEv = static_cast<QKeyEvent*>(e);
1371 QTreeWidget *aTable = static_cast<QTreeWidget *>(obj);
1372 if( keyEv->key() == Qt::Key_Escape )
1373 {
1374 aTable->clearFocus();
1375 return true;
1376 }
1377 else if( keyEv->key() == Qt::Key_Return ||
1378 keyEv->key() == Qt::Key_Enter )
1379 {
1380 selectKey( aTable->currentItem(), aTable->currentColumn() );
1381 return true;
1382 }
1383 else if( keyEv->key() == Qt::Key_Delete )
1384 {
1385 if( aTable->currentColumn() != ACTION_COL )
1386 {
1387 aTable->currentItem()->setText( aTable->currentColumn(), NULL );
1388 aTable->currentItem()->setData( aTable->currentColumn(), Qt::UserRole, QVariant() );
1389 }

1450void KeyInputDialog::checkForConflicts( int i_vlckey, const QString &sequence )
1451{
1452 QList<QTreeWidgetItem *> conflictList =
1453 table->findItems( VLCKeyToString( i_vlckey, true ), Qt::MatchExactly,
1454 b_global ? 2 : 1 );
1455
1456 if( conflictList.count() &&
1457 !conflictList[0]->data( b_global ? 2 : 1, Qt::UserRole ).toString().isEmpty() &&
1458 conflictList[0]->data( b_global ? 2 : 1, Qt::UserRole ).toString() != "Unset" )
1459 {
1460 warning->setText( qtr("Warning: this key or combination is already assigned to ") +
1461 QString( "\"<b>%1</b>\"" ).arg( conflictList[0]->text( 0 ) ) );
1462 warning->show();
1463 ok->show();
1464 unset->hide();
1465
1466 conflicts = true;
1467 }
1468 else if( existingkeys && !sequence.isEmpty()
1469 && existingkeys->contains( sequence ) )
1470 {
1471 warning->setText(
1472 qtr( "Warning: <b>%1</b> is already an application menu shortcut" )
1473 .arg( sequence )
1474 );
1475 warning->show();
1476 ok->show();
1477 unset->hide();
1478
1479 conflicts = true;
1480 }
1481 else accept();
1482}

1484void KeyInputDialog::keyPressEvent( QKeyEvent *e )
1485{
1486 if( e->key() == Qt::Key_Tab ||
1487 e->key() == Qt::Key_Shift ||
1488 e->key() == Qt::Key_Control ||
1489 e->key() == Qt::Key_Meta ||
1490 e->key() == Qt::Key_Alt ||
1491 e->key() == Qt::Key_AltGr )
1492 return;
1493 int i_vlck = qtEventToVLCKey( e );
1494 QKeySequence sequence( e->key() | e->modifiers() );
1495 selected->setText( qtr( "Key or combination: " )
1496 + QString("<b>%1</b>").arg( VLCKeyToString( i_vlck, true ) ) );
1497 checkForConflicts( i_vlck, sequence.toString() );
1498 keyValue = i_vlck;
1499}
