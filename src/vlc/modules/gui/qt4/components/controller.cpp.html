59/******
60 * This is an abstract Toolbar/Controller
61 * This has helper to create any toolbar, any buttons and to manage the actions
62 *
63 *****/
64AbstractController::AbstractController( intf_thread_t * _p_i, QWidget *_parent )
65 : QFrame( _parent )
66{
67 p_intf = _p_i;
68 advControls = NULL;
69 buttonGroupLayout = NULL;
70
71 /* Main action provider */
72 toolbarActionsMapper = new QSignalMapper( this );
73 CONNECT( toolbarActionsMapper, mapped( int ),
74 ActionsManager::getInstance( p_intf ), doAction( int ) );
75 CONNECT( THEMIM->getIM(), playingStatusChanged( int ), this, setStatus( int ) );
76
77 setSizePolicy( QSizePolicy::MinimumExpanding, QSizePolicy::Fixed );
78}

80/* Reemit some signals on status Change to activate some buttons */
81void AbstractController::setStatus( int status )
82{
83 bool b_hasInput = THEMIM->getIM()->hasInput();
84 /* Activate the interface buttons according to the presence of the input */
85 emit inputExists( b_hasInput );
86
87 emit inputPlaying( status == PLAYING_S );
88
89 emit inputIsRecordable( b_hasInput &&
90 var_GetBool( THEMIM->getInput(), "can-record" ) );
91
92 emit inputIsTrickPlayable( b_hasInput &&
93 var_GetBool( THEMIM->getInput(), "can-rewind" ) );
94}

109/* Open the generic config line for the toolbar, parse it
110 * and create the widgets accordingly */
111void AbstractController::parseAndCreate( const QString& config,
112 QBoxLayout *newControlLayout )
113{
114 QStringList list = config.split( ";", QString::SkipEmptyParts ) ;
115 for( int i = 0; i < list.count(); i++ )
116 {
117 QStringList list2 = list.at( i ).split( "-" );
118 if( list2.count() < 1 )
119 {
120 msg_Warn( p_intf, "Parsing error 1. Please, report this." );
121 continue;
122 }
123
124 bool ok;
125 int i_option = WIDGET_NORMAL;
126 buttonType_e i_type = (buttonType_e)list2.at( 0 ).toInt( &ok );
127 if( !ok )
128 {
129 msg_Warn( p_intf, "Parsing error 2. Please, report this." );
130 continue;
131 }
132
133 if( list2.count() > 1 )
134 {
135 i_option = list2.at( 1 ).toInt( &ok );
136 if( !ok )
137 {
138 msg_Warn( p_intf, "Parsing error 3. Please, report this." );
139 continue;
140 }
141 }
142
143 createAndAddWidget( newControlLayout, -1, i_type, i_option );
144 }
145
146 if( buttonGroupLayout )
147 {
148 newControlLayout->addLayout( buttonGroupLayout );
149 buttonGroupLayout = NULL;
150 }
151}

153void AbstractController::createAndAddWidget( QBoxLayout *controlLayout_,
154 int i_index,
155 buttonType_e i_type,
156 int i_option )
157{
158 VLC_UNUSED( i_index ); // i_index should only be required for edition
159
160 /* Close the current buttonGroup if we have a special widget or a spacer */
161 if( buttonGroupLayout && i_type > BUTTON_MAX )
162 {
163 controlLayout_->addLayout( buttonGroupLayout );
164 buttonGroupLayout = NULL;
165 }
166
167 /* Special case for SPACERS, who aren't QWidgets */
168 if( i_type == WIDGET_SPACER )
169 {
170 controlLayout_->addSpacing( 12 );
171 }
172 else if( i_type == WIDGET_SPACER_EXTEND )
173 {
174 controlLayout_->addStretch( 12 );
175 }
176 else
177 {
178 /* Create the widget */
179 QWidget *widg = createWidget( i_type, i_option );
180 if( !widg ) return;
181
182 /* Buttons */
183 if( i_type < BUTTON_MAX )
184 {
185 if( !buttonGroupLayout )
186 {
187 buttonGroupLayout = new QHBoxLayout;
188
189 }
190 buttonGroupLayout->addWidget( widg );
191 }
192 else /* Special widgets */
193 {
194 controlLayout_->addWidget( widg );
195 }
196 }
197}

832void FullscreenControllerWidget::restoreFSC()
833{
834 if( !isWideFSC )
835 {
836 /* Restore half-bar and re-centre if needed */
837 setMinimumWidth( FSC_WIDTH );
838 adjustSize();
839
840 QRect currentRes = QApplication::desktop()->screenGeometry( targetScreen() );
841
842 if( currentRes == screenRes &&
843 QApplication::desktop()->screen()->geometry().contains( previousPosition, true ) )
844 {
845 /* Restore to the last known position */
846 move( previousPosition );
847 }
848 else
849 {
850 /* FSC is out of screen or screen resolution changed */
851 msg_Dbg( p_intf, "Recentering the Fullscreen Controller" );
852 centerFSC( targetScreen() );
853 screenRes = currentRes;
854 previousPosition = pos();
855 }
856 }
857 else
858 {
859 /* Dock at the bottom of the screen */
860 updateFullwidthGeometry( targetScreen() );
861 }

961/**
962 * event handling
963 * events: show, hide, start timer for hiding
964 */
965void FullscreenControllerWidget::customEvent( QEvent *event )
966{
967 bool b_fs;
968
969 switch( (int)event->type() )
970 {
971 /* This is used when the 'i' hotkey is used, to force quick toggle */
972 case IMEvent::FullscreenControlToggle:
973 vlc_mutex_lock( &lock );
974 b_fs = b_fullscreen;
975 vlc_mutex_unlock( &lock );
976
977 if( b_fs )
978 {
979 if( isHidden() )
980 {
981 p_hideTimer->stop();
982 showFSC();
983 }
984 else
985 hideFSC();
986 }
987 break;
988 /* Event called to Show the FSC on mouseChanged() */
989 case IMEvent::FullscreenControlShow:
990 vlc_mutex_lock( &lock );
991 b_fs = b_fullscreen;
992 vlc_mutex_unlock( &lock );
993
994 if( b_fs )
995 showFSC();
996
997 break;
998 /* Start the timer to hide later, called usually with above case */
999 case IMEvent::FullscreenControlPlanHide:
1000 if( !b_mouse_over ) // Only if the mouse is not over FSC
1001 planHideFSC();
1002 break;
1003 /* Hide */
1004 case IMEvent::FullscreenControlHide:
1005 hideFSC();
1006 break;
1007 default:
1008 break;
1009 }
1010}

1124 */
1125void FullscreenControllerWidget::setVoutList( vout_thread_t **pp_vout, int i_vout )
1126{
1127 QList<vout_thread_t*> del;
1128 QList<vout_thread_t*> add;
1129
1130 QList<vout_thread_t*> set;
1131
1132 /* */
1133 for( int i = 0; i < i_vout; i++ )
1134 set += pp_vout[i];
1135
1136 /* Vout to remove */
1137 vlc_mutex_lock( &lock );
1138 foreach( vout_thread_t *p_vout, vout )
1139 {
1140 if( !set.contains( p_vout ) )
1141 del += p_vout;
1142 }
1143 vlc_mutex_unlock( &lock );
1144
1145 foreach( vout_thread_t *p_vout, del )
1146 {
1147 var_DelCallback( p_vout, "fullscreen",
1148 FullscreenControllerWidgetFullscreenChanged, this );
1149 vlc_mutex_lock( &lock );
1150 fullscreenChanged( p_vout, false, 0 );
1151 vout.removeAll( p_vout );
1152 vlc_mutex_unlock( &lock );
1153
1154 vlc_object_release( VLC_OBJECT(p_vout) );
1155 }
1156
1157 /* Vout to track */
1158 vlc_mutex_lock( &lock );
1159 foreach( vout_thread_t *p_vout, set )
1160 {
1161 if( !vout.contains( p_vout ) )
1162 add += p_vout;
1163 }
1164 vlc_mutex_unlock( &lock );
1165
1166 foreach( vout_thread_t *p_vout, add )
1167 {
1168 vlc_object_hold( VLC_OBJECT(p_vout) );
1169
1170 vlc_mutex_lock( &lock );
1171 vout.append( p_vout );
1172 var_AddCallback( p_vout, "fullscreen",
1173 FullscreenControllerWidgetFullscreenChanged, this );
1174 /* I miss a add and fire */
1175 fullscreenChanged( p_vout, var_GetBool( p_vout, "fullscreen" ),
1176 var_GetInteger( p_vout, "mouse-hide-timeout" ) );
1177 vlc_mutex_unlock( &lock );
1178 }

1180/**
1181 * Register and unregister callback for mouse moving
1182 */
1183void FullscreenControllerWidget::fullscreenChanged( vout_thread_t *p_vout,
1184 bool b_fs, int i_timeout )
1185{
1186 /* FIXME - multiple vout (ie multiple mouse position ?) and thread safety if multiple vout ? */
1187
1188 vlc_mutex_lock( &lock );
1189 /* Entering fullscreen, register callback */
1190 if( b_fs && !b_fullscreen )
1191 {
1192 msg_Dbg( p_vout, "Qt: Entering Fullscreen" );
1193 b_fullscreen = true;
1194 i_hide_timeout = i_timeout;
1195 var_AddCallback( p_vout, "mouse-moved",
1196 FullscreenControllerWidgetMouseMoved, this );
1197 }
1198 /* Quitting fullscreen, unregistering callback */
1199 else if( !b_fs && b_fullscreen )
1200 {
1201 msg_Dbg( p_vout, "Qt: Quitting Fullscreen" );
1202 b_fullscreen = false;
1203 i_hide_timeout = i_timeout;
1204 var_DelCallback( p_vout, "mouse-moved",
1205 FullscreenControllerWidgetMouseMoved, this );
1206
1207 /* Force fs hiding */
1208 IMEvent *eHide = new IMEvent( IMEvent::FullscreenControlHide, 0 );
1209 QApplication::postEvent( this, eHide );
1210 }
1211 vlc_mutex_unlock( &lock );
1212}

1214/**
1215 * Mouse change callback (show/hide the controller on mouse movement)
1216 */
1217void FullscreenControllerWidget::mouseChanged( vout_thread_t *, int i_mousex, int i_mousey )
1218{
1219 bool b_toShow;
1220
1221 /* FIXME - multiple vout (ie multiple mouse position ?) and thread safety if multiple vout ? */
1222
1223 b_toShow = false;
1224 if( ( i_mouse_last_move_x == -1 || i_mouse_last_move_y == -1 ) ||
1225 ( abs( i_mouse_last_move_x - i_mousex ) > i_sensitivity ||
1226 abs( i_mouse_last_move_y - i_mousey ) > i_sensitivity ) )
1227 {
1228 i_mouse_last_move_x = i_mousex;
1229 i_mouse_last_move_y = i_mousey;
1230 b_toShow = true;
1231 }
1232
1233 if( b_toShow )
1234 {
1235 /* Show event */
1236 IMEvent *eShow = new IMEvent( IMEvent::FullscreenControlShow, 0 );
1237 QApplication::postEvent( this, eShow );
1238
1239 /* Plan hide event */
1240 IMEvent *eHide = new IMEvent( IMEvent::FullscreenControlPlanHide, 0 );
1241 QApplication::postEvent( this, eHide );
1242 }
