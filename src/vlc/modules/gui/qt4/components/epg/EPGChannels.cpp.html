30EPGChannels::EPGChannels( QWidget *parent, EPGView *m_epgView )
31 : QWidget( parent ), m_epgView( m_epgView ), m_offset( 0 )
32{
33 setContentsMargins( 0, 0, 0, 0 );
34}
35
36void EPGChannels::setOffset( int offset )
37{
38 m_offset = offset;
39 update();
40}
41
42void EPGChannels::addChannel( QString channelName )
43{
44 if ( !channelList.contains( channelName ) )
45 {
46 channelList << channelName;
47 channelList.sort();
48 update();
49 }
50}
51
52void EPGChannels::removeChannel( QString channelName )
53{
54 if ( channelList.removeOne( channelName ) ) update();
55}
56
57void EPGChannels::paintEvent( QPaintEvent *event )
58{
59 Q_UNUSED( event );
60
61 QPainter p( this );
62
63 /* Draw the top and the bottom lines. */
64 p.drawLine( 0, 0, width() - 1, 0 );
65
66 unsigned int i=0;
67 foreach( QString text, channelList )
68 {
69 /* try to remove the " [Program xxx]" end */
70 int i_idx_channel = text.lastIndexOf(" [");
71 if (i_idx_channel > 0)
72 text = text.left( i_idx_channel );
73
74 p.drawText( 0, - m_offset + ( i++ + 0.5 ) * TRACKS_HEIGHT - 4,
75 width(), 20, Qt::AlignLeft, text );
76
77 int i_width = fontMetrics().width( text );
78 if( width() < i_width )
79 setMinimumWidth( i_width );
80 }
81}
