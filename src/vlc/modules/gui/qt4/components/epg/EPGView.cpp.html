35void EPGGraphicsScene::drawBackground( QPainter *painter, const QRectF &rect)
36{
37 EPGView *epgView = qobject_cast<EPGView *>(parent());
38
39 /* day change */
40 QDateTime rectstarttime = epgView->startTime().addSecs( rect.left() );
41 QDateTime nextdaylimit = QDateTime( rectstarttime.date() );
42 QRectF area( rect );
43 while( area.left() < width() )
44 {
45 nextdaylimit = nextdaylimit.addDays( 1 );
46 area.setRight( epgView->startTime().secsTo( nextdaylimit ) );
47
48 if ( epgView->startTime().date().daysTo( nextdaylimit.date() ) % 2 != 0 )
49 painter->fillRect( area, palette().color( QPalette::Base ) );
50 else
51 painter->fillRect( area, palette().color( QPalette::AlternateBase ) );
52
53 area.setLeft( area.right() + 1 );
54 }
55
56 /* channels lines */
57 painter->setPen( QPen( QColor( 224, 224, 224 ) ) );
58 for( int y = rect.top() + TRACKS_HEIGHT ; y < rect.bottom() ; y += TRACKS_HEIGHT )
59 painter->drawLine( QLineF( rect.left(), y, rect.right(), y ) );
60
61 /* current hour line */
62 int x = epgView->startTime().secsTo( epgView->baseTime() );
63 painter->setPen( QPen( QColor( 255, 192, 192 ) ) );
64 painter->drawLine( QLineF( x, rect.top(), x, rect.bottom() ) );
65}

88void EPGView::updateStartTime()
89{
90 mutex.lock();
91 foreach( EPGEventByTimeQMap *epgItemByTime, epgitemsByChannel.values() )
92 {
93 foreach( EPGItem *epgItem, epgItemByTime->values() )
94 {
95 epgItem->updatePos();
96 }
97 }
98 mutex.unlock();
99}

101void EPGView::updateChannels()
102{
103 /* Make sure our items goes to the correct row */
104 unsigned int channelIndex = 0;
105 mutex.lock();
106 foreach( EPGEventByTimeQMap *epgItemByTime, epgitemsByChannel.values() )
107 {
108 foreach( EPGItem *epgItem, epgItemByTime->values() )
109 epgItem->setRow( channelIndex );
110 channelIndex++;
111 }
112 mutex.unlock();
113}

130static void cleanOverlapped( EPGEventByTimeQMap *epgItemByTime, EPGItem *epgItem, QGraphicsScene *scene )
131{
132 QDateTime epgItemTime = epgItem->start();
133 QDateTime epgItemTimeEnd = epgItem->end();
134 /* Clean overlapped programs */
135 foreach(const QDateTime existingTimes, epgItemByTime->keys())
136 {
137 if ( existingTimes > epgItemTimeEnd ) break; /* Can't overlap later items */
138 if ( existingTimes != epgItemTime )
139 {
140 EPGItem *otherEPGItem = epgItemByTime->value( existingTimes );
141 if ( otherEPGItem->playsAt( epgItemTime.addSecs( 1 ) )
142 || /* add/minus one sec because next one can start at prev end min */
143 otherEPGItem->playsAt( epgItemTimeEnd.addSecs( -1 ) ) )
144 {
145 epgItemByTime->remove( otherEPGItem->start() );
146 scene->removeItem( otherEPGItem );
147 delete otherEPGItem;
148 }
149 }
150 }
151}

153bool EPGView::addEPGEvent( vlc_epg_event_t *eventdata, QString channelName, bool b_current )
154{
155 /* Init our nested map if required */
156 EPGEventByTimeQMap *epgItemByTime;
157 EPGItem *epgItem;
158 bool b_refresh_channels = false;
159
160 QDateTime eventStart = QDateTime::fromTime_t( eventdata->i_start );
161 if ( eventStart.addSecs( eventdata->i_duration ) < m_baseTime )
162 return false; /* EPG feed sent expired item */
163 if ( eventStart < m_startTime )
164 {
165 m_startTime = eventStart;
166 emit startTimeChanged( m_startTime );
167 }
168
169 mutex.lock();
170 if ( !epgitemsByChannel.contains( channelName ) )
171 {
172 epgItemByTime = new EPGEventByTimeQMap();
173 epgitemsByChannel.insert( channelName, epgItemByTime );
174 emit channelAdded( channelName );
175 b_refresh_channels = true;
176 } else {
177 epgItemByTime = epgitemsByChannel.value( channelName );
178 }
179
180 if ( epgItemByTime->contains( eventStart ) )
181 {
182 /* Update our existing programs */
183 epgItem = epgItemByTime->value( eventStart );
184 epgItem->setCurrent( b_current );
185 if ( epgItem->setData( eventdata ) ) /* updates our entry */
186 cleanOverlapped( epgItemByTime, epgItem, scene() );
187 mutex.unlock();
188 return false;
189 } else {
190 /* Insert a new program entry */
191 epgItem = new EPGItem( eventdata, this );
192 cleanOverlapped( epgItemByTime, epgItem, scene() );
193 /* Effectively insert our new program */
194 epgItem->setCurrent( b_current );
195 epgItemByTime->insert( eventStart, epgItem );
196 scene()->addItem( epgItem );
197 /* update only our row (without calling the updatechannels()) */
198 epgItem->setRow( epgitemsByChannel.keys().indexOf( channelName ) );
199
200 /* First Insert, needs to focus by default then */
201 if ( epgitemsByChannel.keys().count() == 1 &&
202 epgItemByTime->count() == 1 )
203 focusItem( epgItem );
204 }
205 mutex.unlock();
206
207 /* Update rows on each item */
208 if ( b_refresh_channels ) updateChannels();
209
210 return true;
211}

213void EPGView::removeEPGEvent( vlc_epg_event_t *eventdata, QString channelName )
214{
215 EPGEventByTimeQMap *epgItemByTime;
216 QDateTime eventStart = QDateTime::fromTime_t( eventdata->i_start );
217 EPGItem *epgItem;
218 bool b_update_channels = false;
219
220 mutex.lock();
221 if ( epgitemsByChannel.contains( channelName ) )
222 {
223 epgItemByTime = epgitemsByChannel.value( channelName );
224
225 if ( epgItemByTime->contains( eventStart ) )
226 { /* delete our EPGItem */
227 epgItem = epgItemByTime->value( eventStart );
228 epgItemByTime->remove( eventStart );
229 scene()->removeItem( epgItem );
230 delete epgItem;
231 }
232
233 if ( epgItemByTime->keys().isEmpty() )
234 { /* Now unused channel */
235 epgitemsByChannel.remove( channelName );
236 delete epgItemByTime;
237 emit channelRemoved( channelName );
238 b_update_channels = true;
239 }
240 }
241 mutex.unlock();
242
243 if ( b_update_channels ) updateChannels();
244}

246void EPGView::reset()
247{
248 /* clean our items storage and remove them from the scene */
249 EPGEventByTimeQMap *epgItemByTime;
250 EPGItem *epgItem;
251 mutex.lock();
252 foreach( const QString &channelName, epgitemsByChannel.keys() )
253 {
254 epgItemByTime = epgitemsByChannel[ channelName ];
255 foreach( const QDateTime &key, epgItemByTime->keys() )
256 {
257 epgItem = epgItemByTime->value( key );
258 scene()->removeItem( epgItem );
259 epgItemByTime->remove( key );
260 delete epgItem;
261 }
262 epgitemsByChannel.remove( channelName );
263 delete epgItemByTime;
264 emit channelRemoved( channelName ); /* notify others */
265 }
266 mutex.unlock();
267}

269void EPGView::cleanup()
270{
271 /* remove expired items and clear their current flag */
272 EPGEventByTimeQMap *epgItemByTime;
273 EPGItem *epgItem;
274 m_baseTime = QDateTime::currentDateTime();
275 QDateTime lowestTime = m_baseTime;
276 bool b_timechanged = false;
277 bool b_update_channels = false;
278
279 mutex.lock();
280 foreach( const QString &channelName, epgitemsByChannel.keys() )
281 {
282 epgItemByTime = epgitemsByChannel[ channelName ];
283 foreach( const QDateTime &key, epgItemByTime->keys() )
284 {
285 epgItem = epgItemByTime->value( key );
286 if ( epgItem->endsBefore( baseTime() ) ) /* Expired item ? */
287 {
288 scene()->removeItem( epgItem );
289 epgItemByTime->remove( key );
290 delete epgItem;
291 } else {
292 epgItem->setCurrent( false ); /* if stream doesn't update */
293 if ( lowestTime > epgItem->start() )
294 {
295 lowestTime = epgItem->start(); /* update our reference */
296 b_timechanged = true;
297 }
298 }
299 }
300
301 if ( epgItemByTime->keys().isEmpty() )
302 { /* Now unused channel */
303 epgitemsByChannel.remove( channelName );
304 delete epgItemByTime;
305 emit channelRemoved( channelName );
306 b_update_channels = true;
307 }
308 }
309 mutex.unlock();
310
311 if ( b_timechanged )
312 {
313 m_startTime = lowestTime;
314 emit startTimeChanged( m_startTime );
315 }
316
317 if ( b_update_channels ) updateChannels();
318}

320EPGView::~EPGView()

322 reset();

325void EPGView::updateDuration()
326{
327 QDateTime maxItemTime;
328 mutex.lock();
329 foreach( EPGEventByTimeQMap *epgItemByTime, epgitemsByChannel.values() )
330 foreach( EPGItem *epgItem, epgItemByTime->values() )
331 if ( epgItem->end() > maxItemTime ) maxItemTime = epgItem->end();
332 mutex.unlock();
333 m_duration = m_startTime.secsTo( maxItemTime );
334 emit durationChanged( m_duration );
335}
