40EPGWidget::EPGWidget( QWidget *parent ) : QWidget( parent )
41{
42 b_input_type_known = false;
43 m_rulerWidget = new EPGRuler( this );
44 m_epgView = new EPGView( this );
45 m_channelsWidget = new EPGChannels( this, m_epgView );
46
47 m_channelsWidget->setMinimumWidth( 100 );
48
49 m_epgView->setSizePolicy( QSizePolicy::Expanding, QSizePolicy::Expanding );
50 setZoom( 1 );
51
52 rootWidget = new QStackedWidget( this );
53
54 QWidget *containerWidget = new QWidget( this );
55 QGridLayout* layout = new QGridLayout( this );
56 layout->addWidget( m_rulerWidget, 0, 1 );
57 layout->addWidget( m_channelsWidget, 1, 0 );
58 layout->addWidget( m_epgView, 1, 1 );
59 layout->setSpacing( 0 );
60 containerWidget->setLayout( layout );
61 rootWidget->insertWidget( EPGVIEW_WIDGET, containerWidget );
62
63 QLabel *noepgLabel = new QLabel( qtr("No EPG Data Available"), this );
64 noepgLabel->setAlignment( Qt::AlignCenter );
65 rootWidget->insertWidget( NOEPG_WIDGET, noepgLabel );
66
67 rootWidget->setCurrentIndex( 1 );
68 layout = new QGridLayout( this );
69 layout->addWidget( rootWidget );
70 setLayout( layout );
71
72 CONNECT( m_epgView, startTimeChanged(QDateTime),
73 m_rulerWidget, setStartTime(QDateTime) );
74 CONNECT( m_epgView, durationChanged(int),
75 m_rulerWidget, setDuration(int) );
76 CONNECT( m_epgView->horizontalScrollBar(), valueChanged(int),
77 m_rulerWidget, setOffset(int) );
78 CONNECT( m_epgView->verticalScrollBar(), valueChanged(int),
79 m_channelsWidget, setOffset(int) );
80 connect( m_epgView, SIGNAL( itemFocused(EPGItem*)),
81 this, SIGNAL(itemSelectionChanged(EPGItem*)) );
82 CONNECT( m_epgView, channelAdded(QString), m_channelsWidget, addChannel(QString) );
83 CONNECT( m_epgView, channelRemoved(QString), m_channelsWidget, removeChannel(QString) );
84}

86void EPGWidget::reset()
87{
88 m_epgView->reset();
89 m_epgView->updateDuration();
90 m_epgView->updateStartTime();
91}

93void EPGWidget::setZoom( int level )
94{
95 double scale = (double)level / 20;
96 m_epgView->setScale( scale );
97 m_rulerWidget->setScale( scale );
98}

100void EPGWidget::updateEPG( input_item_t *p_input_item )
101{
102 if( !p_input_item ) return;
103
104 /* flush our EPG data if input type has changed */
105 if ( b_input_type_known && p_input_item->i_type != i_event_source_type ) m_epgView->reset();
106 i_event_source_type = p_input_item->i_type;
107 b_input_type_known = true;
108
109 m_epgView->cleanup(); /* expire items and flags */
110 /* Fixme: input could have dissapeared */
111 vlc_mutex_lock( & p_input_item->lock );
112
113 for ( int i = 0; i < p_input_item->i_epg; ++i )
114 {
115 vlc_epg_t *p_epg = p_input_item->pp_epg[i];
116
117 /* Read current epg events from libvlc and try to insert them */
118 for ( int j = 0; j < p_epg->i_event; ++j )
119 {
120 vlc_epg_event_t *p_event = p_epg->pp_event[j];
121 m_epgView->addEPGEvent( p_event, qfu( p_epg->psz_name ),
122 ( p_epg->p_current == p_event ) );
123 }
124 }
125 vlc_mutex_unlock( & p_input_item->lock );
126
127 /* toggle our widget view */
128 rootWidget->setCurrentIndex(
129 m_epgView->hasValidData() ? EPGVIEW_WIDGET : NOEPG_WIDGET );
130
131 // Update the global duration and start time.
132 m_epgView->updateDuration();
133 m_epgView->updateStartTime();
134}
