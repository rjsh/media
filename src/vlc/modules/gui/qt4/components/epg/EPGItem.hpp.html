27#include "qt4.hpp"
28
29#include <vlc_epg.h>
30#include <QGraphicsItem>
31#include <QDateTime>
32
33class QPainter;
34class QString;
35
36class EPGView;
37
38class EPGItem : public QGraphicsItem
39{
40public:
41 EPGItem( vlc_epg_event_t *data, EPGView *view );
42
43 virtual QRectF boundingRect() const;
44 virtual void paint( QPainter *painter, const QStyleOptionGraphicsItem *option, QWidget *widget = 0 );
45
46 const QDateTime& start() const;
47 QDateTime end() const;
48
49 int duration() const;
50 const QString& name() const { return m_name; }
51 QString description() const;
52 int rating() const { return m_rating; }
53 bool setData( vlc_epg_event_t * );
54 void setRow( unsigned int );
55 void setCurrent( bool );
56 void setDuration( int duration );
57 void setRating( uint8_t i_rating );
58 void updatePos();
59 bool endsBefore( const QDateTime & ) const;
60 bool playsAt( const QDateTime & ) const;
61
62protected:
63 virtual void focusInEvent( QFocusEvent * event );
64 virtual void hoverEnterEvent ( QGraphicsSceneHoverEvent * );
65 virtual void hoverLeaveEvent ( QGraphicsSceneHoverEvent * );
66
67private:
68 EPGView *m_view;
69 QRectF m_boundingRect;
70 unsigned int i_row;
71
72 QDateTime m_start;
73 int m_duration;
74 QString m_name;
75 QString m_description;
76 QString m_shortDescription;
77 bool m_current;
78 uint8_t m_rating;
79};
