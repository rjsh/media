68void EPGRuler::paintEvent( QPaintEvent *event )
69{
70 Q_UNUSED( event );
71 const QSize margin( 0, contentsMargins().top() );
72 const QSize header( 0, maximumHeight() - contentsMargins().top() );
73 const int spacing = m_scale * 3600;
74 QPainter p( this );
75
76 QDateTime localStartTime;
77 localStartTime = m_startTime.addSecs( m_offset / m_scale );
78
79 QDateTime diff( localStartTime );
80 diff.setTime( QTime( localStartTime.time().hour(), 0, 0, 0 ) );
81
82 int secondsToHour = localStartTime.secsTo( diff );
83
84 /* draw hour blocks, with right bound being hourmark */
85 QPoint here( secondsToHour * m_scale, margin.height() );
86 QPoint previous( -1, 0 );
87 QDateTime current( localStartTime.addSecs( secondsToHour ) );
88 current = current.addSecs( -3600 );
89
90 QColor fillColor;
91 while ( here.rx() < width() + spacing )
92 {
93 QRect area( QPoint( previous.x() + 1, margin.height() ), here );
94 area.adjust( 0, 0, 0, header.height() );
95 QString timeString = current.toString( "hh'h'" );
96 /* Show Day */
97 if ( current.time().hour() == 0 )
98 timeString += current.date().toString( " ddd dd" );
99
100 if ( m_startTime.date().daysTo( current.date() ) % 2 == 0 )
101 fillColor = palette().color( QPalette::Window );
102 else
103 fillColor = palette().color( QPalette::Dark );
104 p.fillRect( area, fillColor );
105 p.drawLine( area.topRight(), area.bottomRight() );
106 p.drawText( area, Qt::AlignLeft, timeString );
107 previous = here;
108 here.rx() += spacing;
109 current = current.addSecs( 3600 );
110 }
111
112 /* draw current time line */
113 here.rx() = localStartTime.secsTo( QDateTime::currentDateTime() ) * m_scale;
114 if ( here.x() <= width() && here.x() >= 0 )
115 {
116 p.setPen( QPen( QColor( 255, 0 , 0, 128 ) ) );
117 p.drawLine( here, QPoint( here.x(), here.y() + header.height() ) );
118 }
119}
