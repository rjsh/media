41VLCStatsView::VLCStatsView( QWidget *parent ) : QGraphicsView( parent )
42{
43 QColor history(0, 0, 0, 255),
44 total(237, 109, 0, 160),
45 content(109, 237, 0, 160);
46
47 scale( 1.0, -1.0 ); /* invert our Y axis */
48 setOptimizationFlags( QGraphicsView::DontAdjustForAntialiasing );
49 setAlignment( Qt::AlignLeft );
50 setVerticalScrollBarPolicy( Qt::ScrollBarAlwaysOff );
51 setHorizontalScrollBarPolicy( Qt::ScrollBarAlwaysOff );
52 viewScene = new QGraphicsScene( this );
53 historyShape = viewScene->addPolygon( QPolygonF(), QPen(Qt::NoPen),
54 QBrush(history) );
55 totalbitrateShape = viewScene->addPolygon( QPolygonF(), QPen(Qt::NoPen),
56 QBrush(total) );
57 setScene( viewScene );
58 reset();
59
60 QPen linepen( Qt::DotLine );
61 linepen.setBrush( QBrush( QColor( 33, 33, 33 ) ) );
62 for ( int i=0; i<3; i++ )
63 rulers[i] = viewScene->addLine( QLineF(), linepen );
64}
65#undef ADD_LABEL

67void VLCStatsView::reset()
68{
69 historymergepointer = 0;
70 blocksize = 4;
71 valuesaccumulator = 0;
72 valuesaccumulatorcount = 0;
73 historyShape->setPolygon( QPolygonF() );
74 totalbitrateShape->setPolygon( QPolygonF() );
75}

77void VLCStatsView::addValue( float value )
78{
79 value /= 1000;
80
81 QPolygonF shape = totalbitrateShape->polygon();
82 if ( shape.count() > ( STATS_LENGTH + 2 ) ) /* keep only STATS_LENGTH samples */
83 {
84 shape.remove( 1 );
85 for(int i=1; i<( STATS_LENGTH + 2 ); i++)
86 ( (QPointF &) shape.at(i) ).setX( i - 1 ); /*move back values*/
87 }
88
89 int count = shape.count();
90 if ( count == 0 )
91 {
92 shape << QPointF( 0, 0 ); /* begin and close shape */
93 shape << QPointF( count, 0 );
94 }
95
96 shape.insert( shape.end() - 1, QPointF( count, value ) );
97 ( (QPointF &) shape.last() ).setX( count );
98 totalbitrateShape->setPolygon( shape );
99
100 addHistoryValue( value );
101
102 QRectF maxsizes = scene()->itemsBoundingRect();
103 maxsizes.setRight( STATS_LENGTH );
104 fitInView( maxsizes ); /* fix viewport */
105 drawRulers( maxsizes );
106}

108void VLCStatsView::drawRulers( const QRectF &maxsizes )
109{
110 float height = maxsizes.height() * 1000;
111 int texp = 0;
112 while( height > 1.0 ) { height /= 10; texp++; }
113 height = 1.0;
114 while( texp-- ) height *= 10;
115 for ( int i=0; i<3; i++ )
116 {
117 float y = ( height / 5 ) * ( i + 1 ) / 1000;
118 rulers[i]->setLine( QLineF( 0, y, STATS_LENGTH, y ) );
119 }
120}

122void VLCStatsView::addHistoryValue( float value )
123{
124 /* We keep a full history by creating virtual blocks for inserts, growing
125 by power of 2 when no more space is available. At this time, we also
126 free space by agregating the oldest values 2 by 2.
127 Each shown value finally being a mean of blocksize samples.
128 */
129 bool doinsert = false;
130 int next_blocksize = blocksize;
131 QPolygonF shape = historyShape->polygon();
132 int count = shape.count();
133 if ( count == 0 )
134 {
135 shape << QPointF( 0, 0 ); /* begin and close shape */
136 shape << QPointF( count, 0 );
137 }
138
139 valuesaccumulator += ( value / blocksize );
140 valuesaccumulatorcount++;
141
142 if ( valuesaccumulatorcount == blocksize )
143 {
144 valuesaccumulator = 0;
145 valuesaccumulatorcount = 0;
146 doinsert = true;
147 }
148
149 if ( doinsert )
150 {
151 if ( count > ( STATS_LENGTH + 2 ) )
152 {
153 float y = 0;
154 y += ((QPointF &) shape.at( historymergepointer + 1 )).y();
155 y += ((QPointF &) shape.at( historymergepointer + 2 )).y();
156 y /= 2;
157
158 /* merge */
159 shape.remove( historymergepointer + 2 );
160 ( (QPointF &) shape.at( historymergepointer + 1 ) ).setY( y );
161 for(int i=historymergepointer +1; i<( STATS_LENGTH + 2 ); i++)
162 ( (QPointF &) shape.at(i) ).setX( i - 1 ); /*move back values*/
163 historymergepointer++;
164 if ( historymergepointer > ( STATS_LENGTH - 1 ) )
165 {
166 historymergepointer = 0;
167 next_blocksize = ( blocksize << 1 );
168 }
169 }
170
171 shape.insert( shape.end() - 1, QPointF( count, value ) );
172 ( (QPointF &) shape.last() ).setX( count );
173 }
174 else
175 ( (QPointF &) shape.last() ).setX( count - 1 );
176
177 historyShape->setPolygon( shape );
178
179 blocksize = next_blocksize;
180}
