200/**
201 * Update all the MetaData and art on an "item-changed" event
202 **/
203void MetaPanel::update( input_item_t *p_item )
204{
205 if( !p_item )
206 {
207 clear();
208 return;
209 }
210
211 /* Don't update if you are in edit mode */
212 if( b_inEditMode ) return;
213 p_input = p_item;
214
215 char *psz_meta;
216#define UPDATE_META( meta, widget ) { \
217 psz_meta = input_item_Get##meta( p_item ); \
218 widget->setText( !EMPTY_STR( psz_meta ) ? qfu( psz_meta ) : "" ); \
219 free( psz_meta ); }
220
221#define UPDATE_META_INT( meta, widget ) { \
222 psz_meta = input_item_Get##meta( p_item ); \
223 if( !EMPTY_STR( psz_meta ) ) \
224 widget->setValue( atoi( psz_meta ) ); } \
225 free( psz_meta );
226
227 /* Name / Title */
228 psz_meta = input_item_GetTitleFbName( p_item );
229 if( psz_meta )
230 {
231 title_text->setText( qfu( psz_meta ) );
232 free( psz_meta );
233 }
234 else
235 title_text->setText( "" );
236
237 /* URL / URI */
238 psz_meta = input_item_GetURI( p_item );
239 if( !EMPTY_STR( psz_meta ) )
240 emit uriSet( qfu( psz_meta ) );
241 fingerprintButton->setVisible( Chromaprint::isSupported( QString( psz_meta ) ) );
242 free( psz_meta );
243
244 /* Other classic though */
245 UPDATE_META( Artist, artist_text );
246 UPDATE_META( Genre, genre_text );
247 UPDATE_META( Copyright, copyright_text );
248 UPDATE_META( Album, collection_text );
249 disconnect( description_text, SIGNAL(textChanged()), this,
250 SLOT(enterEditMode()) );
251 UPDATE_META( Description, description_text );
252 CONNECT( description_text, textChanged(), this, enterEditMode() );
253 UPDATE_META( Language, language_text );
254 UPDATE_META( NowPlaying, nowplaying_text );
255 UPDATE_META( Publisher, publisher_text );
256 UPDATE_META( EncodedBy, encodedby_text );
257
258 UPDATE_META( Date, date_text );
259 UPDATE_META( TrackNum, seqnum_text );
260 UPDATE_META( TrackTotal, seqtot_text );
261// UPDATE_META( Setting, setting_text );
262// UPDATE_META_INT( Rating, rating_text );
263
264 /* URL */
265 psz_meta = input_item_GetURL( p_item );
266 if( !EMPTY_STR( psz_meta ) )
267 {
268 QString newURL = qfu(psz_meta);
269 if( currentURL != newURL )
270 {
271 currentURL = newURL;
272 lblURL->setText( "<a href='" + currentURL + "'>" +
273 currentURL.remove( QRegExp( ".*://") ) + "</a>" );
274 }
275 }
276 free( psz_meta );
277#undef UPDATE_META_INT
278#undef UPDATE_META
279
280 // If a artURL is available as a local file, directly display it !
281
282 QString file;
283 char *psz_art = input_item_GetArtURL( p_item );
284 if( psz_art )
285 {
286 char *psz = make_path( psz_art );
287 free( psz_art );
288 file = qfu( psz );
289 free( psz );
290 }
291
292 art_cover->showArtUpdate( file );
293 art_cover->setItem( p_item );
294}

296/**
297 * Save the MetaData, triggered by parent->save Button
298 **/
299void MetaPanel::saveMeta()
300{
301 if( p_input == NULL )
302 return;
303
304 /* now we read the modified meta data */
305 input_item_SetTitle( p_input, qtu( title_text->text() ) );
306 input_item_SetArtist( p_input, qtu( artist_text->text() ) );
307 input_item_SetAlbum( p_input, qtu( collection_text->text() ) );
308 input_item_SetGenre( p_input, qtu( genre_text->text() ) );
309 input_item_SetTrackNum( p_input, qtu( seqnum_text->text() ) );
310 input_item_SetTrackTotal( p_input, qtu( seqtot_text->text() ) );
311 input_item_SetDate( p_input, qtu( date_text->text() ) );
312 input_item_SetLanguage( p_input, qtu( language_text->text() ) );
313
314 input_item_SetCopyright( p_input, qtu( copyright_text->text() ) );
315 input_item_SetPublisher( p_input, qtu( publisher_text->text() ) );
316 input_item_SetDescription( p_input, qtu( description_text->toPlainText() ) );
317
318 input_item_WriteMeta( VLC_OBJECT(THEPL), p_input );
319
320 /* Reset the status of the mode. No need to emit any signal because parent
321 is the only caller */
322 b_inEditMode = false;
323}

404/**
405 * Update the Extra Metadata from p_meta->i_extras
406 **/
407void ExtraMetaPanel::update( input_item_t *p_item )
408{
409 if( !p_item )
410 {
411 clear();
412 return;
413 }
414
415 QList<QTreeWidgetItem *> items;
416
417 extraMetaTree->clear();
418
419 vlc_mutex_lock( &p_item->lock );
420 vlc_meta_t *p_meta = p_item->p_meta;
421 if( !p_meta )
422 {
423 vlc_mutex_unlock( &p_item->lock );
424 return;
425 }
426
427 char ** ppsz_allkey = vlc_meta_CopyExtraNames( p_meta);
428
429 for( int i = 0; ppsz_allkey[i] ; i++ )
430 {
431 const char * psz_value = vlc_meta_GetExtra( p_meta, ppsz_allkey[i] );
432 QStringList tempItem;
433 tempItem.append( qfu( ppsz_allkey[i] ) + " : ");
434 tempItem.append( qfu( psz_value ) );
435 items.append( new QTreeWidgetItem ( extraMetaTree, tempItem ) );
436 free( ppsz_allkey[i] );
437 }
438 vlc_mutex_unlock( &p_item->lock );
439 free( ppsz_allkey );
440
441 extraMetaTree->addTopLevelItems( items );
442 extraMetaTree->resizeColumnToContents( 0 );
443}

480/**
481 * Update the Codecs information on parent->update()
482 **/
483void InfoPanel::update( input_item_t *p_item)
484{
485 if( !p_item )
486 {
487 clear();
488 return;
489 }
490
491 InfoTree->clear();
492 QTreeWidgetItem *current_item = NULL;
493 QTreeWidgetItem *child_item = NULL;
494
495 for( int i = 0; i< p_item->i_categories ; i++)
496 {
497 current_item = new QTreeWidgetItem();
498 current_item->setText( 0, qfu(p_item->pp_categories[i]->psz_name) );
499 InfoTree->addTopLevelItem( current_item );
500
501 for( int j = 0 ; j < p_item->pp_categories[i]->i_infos ; j++ )
502 {
503 child_item = new QTreeWidgetItem ();
504 child_item->setText( 0,
505 qfu(p_item->pp_categories[i]->pp_infos[j]->psz_name)
506 + ": "
507 + qfu(p_item->pp_categories[i]->pp_infos[j]->psz_value));
508
509 current_item->addChild(child_item);
510 }
511 InfoTree->setItemExpanded( current_item, true);
512 }

634/**
635 * Update the Statistics
636 **/
637void InputStatsPanel::update( input_item_t *p_item )
638{
639 if ( !isVisible() ) return;
640 assert( p_item );
641 vlc_mutex_lock( &p_item->p_stats->lock );
642
643#define UPDATE_INT( widget, calc... ) \
644 { widget->setText( 1, QString::number( (qulonglong)calc ) ); }
645
646#define UPDATE_FLOAT( widget, format, calc... ) \
647 { QString str; widget->setText( 1 , str.sprintf( format, ## calc ) ); }
648
649 UPDATE_INT( read_media_stat, (p_item->p_stats->i_read_bytes / 1024 ) );
650 UPDATE_FLOAT( input_bitrate_stat, "%6.0f", (float)(p_item->p_stats->f_input_bitrate * 8000 ));
651 UPDATE_INT( demuxed_stat, (p_item->p_stats->i_demux_read_bytes / 1024 ) );
652 UPDATE_FLOAT( stream_bitrate_stat, "%6.0f", (float)(p_item->p_stats->f_demux_bitrate * 8000 ));
653 UPDATE_INT( corrupted_stat, p_item->p_stats->i_demux_corrupted );
654 UPDATE_INT( discontinuity_stat, p_item->p_stats->i_demux_discontinuity );
655
656 statsView->addValue( p_item->p_stats->f_input_bitrate * 8000 );
657
658 /* Video */
659 UPDATE_INT( vdecoded_stat, p_item->p_stats->i_decoded_video );
660 UPDATE_INT( vdisplayed_stat, p_item->p_stats->i_displayed_pictures );
