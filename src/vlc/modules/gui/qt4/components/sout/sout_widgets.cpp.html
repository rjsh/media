68}

70void SoutInputBox::setMRL( const QString& mrl )
71{
72 QUrl uri = QUrl::fromEncoded( mrl.toLatin1() );
73 sourceLine->setText( uri.toString() );
74 QString type = uri.scheme();
75 if ( type.isEmpty() ) type = qtr( I_FILE_SLASH_DIR );
76 sourceValueLabel->setText( type );
77}

119
120QString FileDestBox::getMRL( const QString& mux )
121{
122 if( fileEdit->text().isEmpty() ) return "";
123
124 SoutMrl m;
125 m.begin( "file" );
126 QString outputfile = fileEdit->text();
127 if( !mux.isEmpty() )
128 {
129 if( outputfile.contains( QRegExp("\\..{2,4}$")) &&
130 !outputfile.endsWith(mux) )
131 {
132 /* Replace the extension according to muxer */
133 outputfile.replace(QRegExp("\\..{2,4}$"),"."+mux);
134 } else if (!outputfile.endsWith( mux ) )
135 {
136 m.option( "mux", mux );
137 }
138 }
139 m.option( "dst", outputfile );
140 m.option( "no-overwrite" );
141 m.end();
142
143 return m.getMrl();
144}

181QString HTTPDestBox::getMRL( const QString& mux )
182{
183 if( HTTPEdit->text().isEmpty() ) return "";
184
185 QString path = HTTPEdit->text();
186 if( path[0] != '/' )
187 path.prepend( qfu("/") );
188 QString port;
189 port.setNum( HTTPPort->value() );
190 QString dst = ":" + port + path;
191
192 SoutMrl m;
193 m.begin( "http" );
194 /* Path-extension is primary muxer to use if possible,
195 otherwise check for mux-choise and see that it isn't mp4
196 then fallback to flv*/
197 if ( !path.contains(QRegExp("\\..{2,3}$") ) )
198 {
199 if( !mux.isEmpty() && mux.compare("mp4") )
200 m.option( "mux", mux );
201 else
202 m.option( "mux", "ffmpeg{mux=flv}" );
203 }
204 m.option( "dst", dst );
205 m.end();
206
207 return m.getMrl();
208}

369QString RTPDestBox::getMRL( const QString& )
370{
371 QString addr = RTPEdit->text();
372 QString name = SAPName->text();
373
374 if( addr.isEmpty() ) return qfu("");
375
376 SoutMrl m;
377 m.begin( "rtp" );
378 m.option( "dst", RTPEdit->text() );
379 m.option( "port", RTPPort->value() );
380 /* mp4-mux ain't usable in rtp-output either */
381 if( !mux.isEmpty() )
382 m.option( "mux", mux );
383 if( !name.isEmpty() )
384 {
385 m.option( "sap" );
386 m.option( "name", name );
387 }
388 m.end();
389
390 return m.getMrl();
391}

434QString ICEDestBox::getMRL( const QString& )
435{
436 if( ICEEdit->text().isEmpty() ) return "";
437
438 SoutMrl m;
439 m.begin( "std" );
440 m.option( "access", "shout" );
441 m.option( "mux", "ogg" );
442
443 QString url = ICEPassEdit->text() + "@"
444 + ICEEdit->text()
445 + ":" + QString::number( ICEPort->value(), 10 )
446 + "/" + ICEMountEdit->text();
447
448 m.option( "dst", url );
449 m.end();
450 return m.getMrl();
