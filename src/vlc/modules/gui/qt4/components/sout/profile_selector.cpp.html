83inline void VLCProfileSelector::fillProfilesCombo()
84{
85 QSettings settings(
86#ifdef _WIN32
87 QSettings::IniFormat,
88#else
89 QSettings::NativeFormat,
90#endif
91 QSettings::UserScope, "vlc", "vlc-qt-interface" );
92
93 int i_size = settings.beginReadArray( "codecs-profiles" );
94
95 for( int i = 0; i < i_size; i++ )
96 {
97 settings.setArrayIndex( i );
98 if( settings.value( "Profile-Name" ).toString().isEmpty() ) continue;
99 profileBox->addItem( settings.value( "Profile-Name" ).toString(),
100 settings.value( "Profile-Value" ) );
101 }
102 if( i_size == 0 )
103 {
104 for( size_t i = 0; i < NB_PROFILE; i++ )
105 {
106 profileBox->addItem( video_profile_name_list[i],
107 video_profile_value_list[i] );
108 }
109 }
110 settings.endArray();

124void VLCProfileSelector::editProfile( const QString& qs, const QString& value )
125{
126 /* Create the Profile Editor */
127 VLCProfileEditor *editor = new VLCProfileEditor( qs, value, this );
128
129 /* Show it */
130 if( QDialog::Accepted == editor->exec() )
131 {
132 /* New Profile */
133 if( qs.isEmpty() )
134 profileBox->addItem( editor->name, QVariant( editor->transcodeValue() ) );
135 /* Update old profile */
136 else
137 {
138 /* Look for the profile */
139 int i_profile = profileBox->findText( qs );
140 assert( i_profile != -1 );
141 profileBox->setItemText( i_profile, editor->name );
142 profileBox->setItemData( i_profile, editor->transcodeValue() );
143 /* Force mrl recreation */
144 updateOptions( i_profile );
145 }
146 }
147 delete editor;
148
149 saveProfiles();
150 emit optionsChanged();
151}

159void VLCProfileSelector::saveProfiles()
160{
161 QSettings settings(
162#ifdef _WIN32
163 QSettings::IniFormat,
164#else
165 QSettings::NativeFormat,
166#endif
167 QSettings::UserScope, "vlc", "vlc-qt-interface" );
168
169 settings.remove( "codecs-profiles" ); /* Erase old profiles to be rewritten */
170 settings.beginWriteArray( "codecs-profiles" );
171 for( int i = 0; i < profileBox->count(); i++ )
172 {
173 settings.setArrayIndex( i );
174 settings.setValue( "Profile-Name", profileBox->itemText( i ) );
175 settings.setValue( "Profile-Value", profileBox->itemData( i ).toString() );
176 }
177 settings.endArray();
178}

180void VLCProfileSelector::updateOptions( int i )
181{
182 QString options = profileBox->itemData( i ).toString();
183 QRegExp rx(OLDFORMAT);
184 if ( !options.contains( ";" ) ) return;
185 if ( rx.indexIn( options ) != -1 )
186 return updateOptionsOldFormat( i );
187
188 transcode = "";
189
190 QStringList tuples = options.split( ";" );
191 typedef QHash<QString, QString> proptovalueHashType;
192 QHash<QString, proptovalueHashType *> categtopropHash;
193 proptovalueHashType *proptovalueHash;
194 QString value;
195
196 /* Build a double hash structure because we need to make ordered lookups */
197 foreach ( const QString &tuple, tuples )
198 {
199 QStringList keyvalue = tuple.split( "=" );
200 if ( keyvalue.count() != 2 ) continue;
201 QString key = keyvalue[0];
202 value = keyvalue[1];
203 keyvalue = key.split( "_" );
204 if ( keyvalue.count() != 2 ) continue;
205 QString categ = keyvalue[0];
206 QString prop = keyvalue[1];
207
208 if ( ! categtopropHash.contains( categ ) )
209 {
210 proptovalueHash = new proptovalueHashType();
211 categtopropHash.insert( categ, proptovalueHash );
212 } else {
213 proptovalueHash = categtopropHash.value( categ );
214 }
215 proptovalueHash->insert( prop, value );
216 }
217
218 /* Now we can build the/translate into MRL */
219#define HASHPICK( categ, prop ) \
220 if ( categtopropHash.contains( categ ) ) \
221 {\
222 proptovalueHash = categtopropHash.value( categ );\
223 value = proptovalueHash->take( prop );\
224 }\
225 else value = QString()
226
227 SoutMrl smrl;
228 smrl.begin( "transcode" );
229
230 /* First muxer options */
231 HASHPICK( "muxer", "mux" );
232 if ( value.isEmpty() ) goto cleanup;
233 mux = value;
234
235 HASHPICK( "video", "enable" );
236 if ( !value.isEmpty() )
237 {
238 HASHPICK( "video", "codec" );
239
240 if ( !value.isEmpty() )
241 {
242 smrl.option( "vcodec", value );
243
244 HASHPICK( "vcodec", "bitrate" );
245 if ( value.toInt() > 0 )
246 {
247 smrl.option( "vb", value.toInt() );
248 }
249
250 HASHPICK( "video", "filters" );
251 if ( !value.isEmpty() )
252 {
253 QStringList valuesList = QUrl::fromPercentEncoding( value.toLatin1() ).split( ";" );
254 smrl.option( "vfilter", valuesList.join( ":" ) );
255 }
256
257 /*if ( codec is h264 )*/
258 {
259 /* special handling */
260 QStringList codecoptions;
261
262 HASHPICK( "vcodec", "qp" );
263 if( value.toInt() > 0 )
264 codecoptions << QString( "qp=%1" ).arg( value );
265
266 HASHPICK( "vcodec", "custom" );
267 if( !value.isEmpty() )
268 codecoptions << QUrl::fromPercentEncoding( value.toLatin1() );
269
270 if ( codecoptions.count() )
271 smrl.option( "venc",
272 QString("x264{%1}").arg( codecoptions.join(",") ) );
273 }
274
275 HASHPICK( "vcodec", "framerate" );
276 if ( !value.isEmpty() && value.toInt() > 0 )
277 smrl.option( "fps", value );
278
279 HASHPICK( "vcodec", "scale" );
280 if ( !value.isEmpty() )
281 smrl.option( "scale", value );
282
283 HASHPICK( "vcodec", "width" );
284 if ( !value.isEmpty() && value.toInt() > 0 )
285 smrl.option( "width", value );
286
287 HASHPICK( "vcodec", "height" );
288 if ( !value.isEmpty() && value.toInt() > 0 )
289 smrl.option( "height", value );
290 }
291 } else {
292 smrl.option( "vcodec", "none" );
293 }
294
295 HASHPICK( "audio", "enable" );
296 if ( !value.isEmpty() )
297 {
298 HASHPICK( "audio", "codec" );
299 if ( !value.isEmpty() )
300 {
301 smrl.option( "acodec", value );
302
303 HASHPICK( "acodec", "bitrate" );
304 smrl.option( "ab", value.toInt() );
305
306 HASHPICK( "acodec", "channels" );
307 smrl.option( "channels", value.toInt() );
308
309 HASHPICK( "acodec", "samplerate" );
310 smrl.option( "samplerate", value.toInt() );
311
312 HASHPICK( "audio", "filters" );
313 if ( !value.isEmpty() )
314 {
315 QStringList valuesList = QUrl::fromPercentEncoding( value.toLatin1() ).split( ";" );
316 smrl.option( "afilter", valuesList.join( ":" ) );
317 }
318
319 }
320 } else {
321 smrl.option( "acodec", "none" );
322 }
323
324 HASHPICK( "subtitles", "enable" );
325 if( !value.isEmpty() )
326 {
327 HASHPICK( "subtitles", "codec" );
328 smrl.option( "scodec", value );
329
330 HASHPICK( "subtitles", "overlay" );
331 if ( !value.isEmpty() )
332 smrl.option( "soverlay" );
333 }
334 smrl.end();
335#undef HASHPICK
336
337 transcode = smrl.getMrl();
338
339 cleanup:
340 /* Temp hash tables cleanup */
341 foreach( proptovalueHashType *hash, categtopropHash )
342 delete hash;
343
344 emit optionsChanged();
345}

347void VLCProfileSelector::updateOptionsOldFormat( int i )
348{
349 QStringList options = profileBox->itemData( i ).toString().split( ";" );
350 if( options.count() < 16 )
351 return;
352
353 mux = options[0];
354
355 SoutMrl smrl;
356 if( options[1].toInt() || options[2].toInt() || options[3].toInt() )
357 {
358 smrl.begin( "transcode" );
359
360 if( options[1].toInt() )
361 {
362 smrl.option( "vcodec", options[4] );
363 if( options[4] != "none" )
364 {
365 smrl.option( "vb", options[5].toInt() );
366 if( !options[7].isEmpty() && options[7].toInt() > 0 )
367 smrl.option( "fps", options[7] );
368 if( !options[6].isEmpty() )
369 smrl.option( "scale", options[6] );
370 if( !options[8].isEmpty() && options[8].toInt() > 0 )
371 smrl.option( "width", options[8].toInt() );
372 if( !options[9].isEmpty() && options[9].toInt() > 0 )
373 smrl.option( "height", options[9].toInt() );
374 }
375 }
376
377 if( options[2].toInt() )
378 {
379 smrl.option( "acodec", options[10] );
380 if( options[10] != "none" )
381 {
382 smrl.option( "ab", options[11].toInt() );
383 smrl.option( "channels", options[12].toInt() );
384 smrl.option( "samplerate", options[13].toInt() );
385 }
386 }
387
388 if( options[3].toInt() )
389 {
390 smrl.option( "scodec", options[14] );
391 if( options[15].toInt() )
392 smrl.option( "soverlay" );
393 }
394
395 smrl.end();
396
397 transcode = smrl.getMrl();
398 }
399 else
400 transcode = "";
401 emit optionsChanged();
402}

448void VLCProfileEditor::loadCapabilities()
449{
450 size_t count;
451 module_t **p_all = module_list_get (&count);
452
453 /* Parse the module list for capabilities and probe each of them */
454 for (size_t i = 0; i < count; i++)
455 {
456 module_t *p_module = p_all[i];
457
458 if( module_provides( p_module, "sout mux" ) )
459 caps["muxers"].insert( module_get_object( p_module ) );
460// else if ( module_provides( p_module, "encoder" ) )
461// caps["encoders"].insert( module_get_object( p_module ) );
462 }
463 module_list_free (p_all);
464}

466inline void VLCProfileEditor::registerFilters()
467{
468 size_t count;
469 module_t **p_all = module_list_get (&count);
470
471 for (size_t i = 0; i < count; i++)
472 {
473 module_t *p_module = p_all[i];
474 if ( module_get_score( p_module ) > 0 ) continue;
475
476 QString capability = module_get_capability( p_module );
477 QListWidget *listWidget = NULL;
478 QListWidgetItem *item;
479
480 if ( capability == "video filter2" )
481 listWidget = ui.valueholder_video_filters;
482 else if ( capability == "audio filter" )
483 listWidget = ui.valueholder_audio_filters;
484
485 if ( !listWidget ) continue;
486
487 item = new QListWidgetItem( module_get_name( p_module, true ) );
488 item->setCheckState( Qt::Unchecked );
489 item->setToolTip( QString( module_get_help( p_module ) ) );
490 item->setData( Qt::UserRole, QString( module_get_object( p_module ) ) );
491 listWidget->addItem( item );
492 }
493 module_list_free (p_all);
494
495 ui.valueholder_video_filters->sortItems();
496 ui.valueholder_audio_filters->sortItems();
497}

621void VLCProfileEditor::fillProfile( const QString& qs )
622{
623 QRegExp rx(OLDFORMAT);
624 if ( rx.indexIn( qs ) != -1 ) return fillProfileOldFormat( qs );
625
626 QStringList tuples = qs.split( ";" );
627 foreach ( const QString &tuple, tuples )
628 {
629 QStringList keyvalue = tuple.split( "=" );
630 if ( keyvalue.count() != 2 ) continue;
631 QString key = keyvalue[0];
632 QString value = keyvalue[1];
633 QObject *object = findChild<QObject *>( CATPROP2NAME( key ) );
634 if ( object )
635 {
636 if( object->inherits( "QButtonGroup" ) )
637 { /* Buttongroup for Radios */
638 const QButtonGroup *group = qobject_cast<const QButtonGroup *>( object );
639 foreach( QAbstractButton *button, group->buttons() )
640 {
641 if ( button->property("sout").toString() == value )
642 {
643 button->setChecked( true );
644 break;/* radios are exclusive */
645 }
646 }
647 }
648 else if( object->inherits( "QCheckBox" ) )
649 {
650 QCheckBox *box = qobject_cast<QCheckBox *>( object );
651 box->setChecked( ! value.isEmpty() );
652 }
653 else if( object->inherits( "QGroupBox" ) )
654 {
655 QGroupBox *box = qobject_cast<QGroupBox *>( object );
656 box->setChecked( ! value.isEmpty() );
657 }
658 else if( object->inherits( "QSpinBox" ) )
659 {
660 QSpinBox *box = qobject_cast<QSpinBox *>( object );
661 box->setValue( value.toInt() );
662 }
663 else if( object->inherits( "QDoubleSpinBox" ) )
664 {
665 QDoubleSpinBox *box = qobject_cast<QDoubleSpinBox *>( object );
666 box->setValue( value.toDouble() );
667 }
668 else if( object->inherits( "QComboBox" ) )
669 {
670 QComboBox *box = qobject_cast<QComboBox *>( object );
671 box->setCurrentIndex( box->findData( value ) );
672 if ( box->lineEdit() && box->currentIndex() == -1 )
673 box->lineEdit()->setText( value );
674 }
675 else if( object->inherits( "QLineEdit" ) )
676 {
677 QLineEdit *box = qobject_cast<QLineEdit *>( object );
678 box->setText( QUrl::fromPercentEncoding( value.toLatin1() ) );
679 }
680 else if ( object->inherits( "QListWidget" ) )
681 {
682 QStringList valuesList = QUrl::fromPercentEncoding( value.toLatin1() ).split( ";" );
683 const QListWidget *list = qobject_cast<const QListWidget *>( object );
684 for( int i=0; i < list->count(); i++ )
685 {
686 QListWidgetItem *item = list->item( i );
687 if ( valuesList.contains( item->data( Qt::UserRole ).toString() ) )
688 item->setCheckState( Qt::Checked );
689 else
690 item->setCheckState( Qt::Unchecked );
691 }
692 }
693 }

697void VLCProfileEditor::fillProfileOldFormat( const QString& qs )
698{
699 QStringList options = qs.split( ";" );
700 if( options.count() < 16 )
701 return;
702
703 const QString mux = options[0];
704 for ( int i=0; i< ui.muxer->layout()->count(); i++ )
705 {
706 QRadioButton *current =
707 qobject_cast<QRadioButton *>(ui.muxer->layout()->itemAt(i)->widget());
708 if ( unlikely( !current ) ) continue;/* someone is messing up with ui */
709 if ( current->property("sout").toString() == mux )
710 {
711 current->setChecked( true );
712 break;/* radios are exclusive */
713 }
714 }
715
716 ui.valueholder_video_copy->setChecked( !options[1].toInt() );
717 ui.valueholder_video_enable->setChecked( ( options[4] != "none" ) );
718 ui.valueholder_audio_copy->setChecked( !options[2].toInt() );
719 ui.valueholder_audio_enable->setChecked( ( options[10] != "none" ) );
720 ui.valueholder_subtitles_enable->setChecked( options[3].toInt() );
721
722 ui.valueholder_video_codec->setCurrentIndex( ui.valueholder_video_codec->findData( options[4] ) );
723 ui.valueholder_vcodec_bitrate->setValue( options[5].toInt() );
724 if ( options[6].toInt() > 0 )
725 ui.valueholder_vcodec_scale->setEditText( options[6] );
726 else
727 ui.valueholder_vcodec_scale->setCurrentIndex( 0 );
728 ui.valueholder_vcodec_framerate->setValue( options[7].toDouble() );
729 ui.valueholder_vcodec_width->setValue( options[8].toInt() );
730 ui.valueholder_vcodec_height->setValue( options[9].toInt() );
731
732 ui.valueholder_audio_codec->setCurrentIndex( ui.valueholder_audio_codec->findData( options[10] ) );
733 ui.valueholder_acodec_bitrate->setValue( options[11].toInt() );
734 ui.valueholder_acodec_channels->setValue( options[12].toInt() );
735
736 int index = ui.valueholder_acodec_samplerate->findData( options[13] );
737 if ( index == -1 ) index = ui.valueholder_acodec_samplerate->findData( 44100 );
738 ui.valueholder_acodec_samplerate->setCurrentIndex( index );
739
740 ui.valueholder_subtitles_codec->setCurrentIndex( ui.valueholder_subtitles_codec->findData( options[14] ) );
741 ui.valueholder_subtitles_overlay->setChecked( options[15].toInt() );
742}

760QString VLCProfileEditor::transcodeValue()
761{
762 QList<QObject *> allwidgets = findChildren<QObject *>();
763 QStringList configuration;
764
765 foreach( const QObject *object, allwidgets )
766 {
767 if ( ! object->objectName().startsWith( CATPROP2NAME( "" ) ) )
768 continue;
769 if ( object->inherits( "QWidget" ) &&
770 ! qobject_cast<const QWidget *>(object)->isEnabled() ) continue;
771
772 QString name = object->objectName();
773 QStringList vals = object->objectName().split( "_" );
774 if ( vals.count() != 3 ) continue;
775 QString &categ = vals[1];
776 QString &prop = vals[2];
777 QString value;
778
779 if( object->inherits( "QButtonGroup" ) )
780 {
781 const QButtonGroup *group = qobject_cast<const QButtonGroup *>( object );
782 value = group->checkedButton()->property( "sout" ).toString();
783 }
784 else if( object->inherits( "QCheckBox" ) )
785 {
786 const QCheckBox *box = qobject_cast<const QCheckBox *>( object );
787 value = box->isChecked() ? "yes" : "";
788 }
789 else if( object->inherits( "QGroupBox" ) )
790 {
791 const QGroupBox *box = qobject_cast<const QGroupBox *>( object );
792 value = box->isChecked() ? "yes" : "";
793 }
794 else if( object->inherits( "QSpinBox" ) )
795 {
796 const QSpinBox *box = qobject_cast<const QSpinBox *>( object );
797 value = QString::number( box->value() );
798 }
799 else if( object->inherits( "QDoubleSpinBox" ) )
800 {
801 const QDoubleSpinBox *box = qobject_cast<const QDoubleSpinBox *>( object );
802 value = QString::number( box->value() );
803 }
804 else if( object->inherits( "QComboBox" ) )
805 {
806 const QComboBox *box = qobject_cast<const QComboBox *>( object );
807 value = currentData( box ).toString();
808 if ( value.isEmpty() && box->lineEdit() ) value = box->lineEdit()->text();
809 }
810 else if( object->inherits( "QLineEdit" ) )
811 {
812 const QLineEdit *box = qobject_cast<const QLineEdit *>( object );
813 value = QUrl::toPercentEncoding( box->text(), "", "_;" );
814 }
815 else if ( object->inherits( "QListWidget" ) )
816 {
817 const QListWidget *list = qobject_cast<const QListWidget *>( object );
818 QStringList valuesList;
819 for( int i=0; i < list->count(); i++ )
820 {
821 const QListWidgetItem *item = list->item( i );
822 if ( item->checkState() == Qt::Checked )
823 valuesList.append( item->data( Qt::UserRole ).toString() );
824 }
825 value = QUrl::toPercentEncoding( valuesList.join( ";" ), "", "_;" );
826 }
827
828 if ( !value.isEmpty() )
829 configuration << QString( "%1_%2=%3" )
830 .arg( categ ).arg( prop ).arg( value );
831 }
832
833 return configuration.join( ";" );
834}
