44/**
45 * SPECIAL Widgets that are a bit more than just a ToolButton
46 * and have an icon/behaviour that changes depending on the context:

48 * - A->B Button
49 * - Teletext group buttons
50 * - Sound Widget group
51 **/

55public slots:
56 void updateButtonIcons( bool );

73class AspectRatioComboBox : public QComboBox
74{
75 Q_OBJECT
76public:
77 AspectRatioComboBox( intf_thread_t* _p_intf ) : p_intf( _p_intf )
78 {
79 CONNECT( THEMIM->getIM(), voutChanged( bool ),
80 this, updateRatios() );
81 CONNECT( this, currentIndexChanged( int ),
82 this, updateAspectRatio( int ) );
83 updateRatios();
84 }
85
86public slots:
87 void updateRatios();
88 void updateAspectRatio( int );
89
90private:
91 intf_thread_t* p_intf;
92};
