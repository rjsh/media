42SoundWidget::SoundWidget( QWidget *_parent, intf_thread_t * _p_intf,
43 bool b_shiny, bool b_special )
44 : QWidget( _parent ), p_intf( _p_intf),
45 b_is_muted( false ), b_ignore_valuechanged( false )
46{
47 /* We need a layout for this widget */
48 QHBoxLayout *layout = new QHBoxLayout( this );
49 layout->setSpacing( 0 ); layout->setMargin( 0 );
50
51 /* We need a Label for the pix */
52 volMuteLabel = new QLabel;
53 volMuteLabel->setPixmap( QPixmap( ":/toolbar/volume-medium" ) );
54
55 /* We might need a subLayout too */
56 QVBoxLayout *subLayout;
57
58 volMuteLabel->installEventFilter( this );
59
60 /* Normal View, click on icon mutes */
61 if( !b_special )
62 {
63 volumeMenu = NULL; subLayout = NULL;
64 volumeControlWidget = NULL;
65
66 /* And add the label */
67 layout->addWidget( volMuteLabel, 0, b_shiny? Qt::AlignBottom : Qt::AlignCenter );
68 }
69 else
70 {
71 /* Special view, click on button shows the slider */
72 b_shiny = false;
73
74 volumeControlWidget = new QFrame( this );
75 subLayout = new QVBoxLayout( volumeControlWidget );
76 subLayout->setContentsMargins( 4, 4, 4, 4 );
77 volumeMenu = new QMenu( this );
78
79 QWidgetAction *widgetAction = new QWidgetAction( volumeControlWidget );
80 widgetAction->setDefaultWidget( volumeControlWidget );
81 volumeMenu->addAction( widgetAction );
82
83 /* And add the label */
84 layout->addWidget( volMuteLabel );
85 }
86
87 /* Slider creation: shiny or clean */
88 if( b_shiny )
89 {
90 volumeSlider = new SoundSlider( this,
91 config_GetFloat( p_intf, "volume-step" ),
92 var_InheritString( p_intf, "qt-slider-colours" ),
93 var_InheritInteger( p_intf, "qt-max-volume") );
94 }
95 else
96 {
97 volumeSlider = new QSlider( NULL );
98 volumeSlider->setAttribute( Qt::WA_MacSmallSize);
99 volumeSlider->setOrientation( b_special ? Qt::Vertical
100 : Qt::Horizontal );
101 volumeSlider->setMaximum( 200 );
102 }
103
104 volumeSlider->setFocusPolicy( Qt::NoFocus );
105 if( b_special )
106 subLayout->addWidget( volumeSlider );
107 else
108 layout->addWidget( volumeSlider, 0, b_shiny? Qt::AlignBottom : Qt::AlignCenter );
109
110 /* Set the volume from the config */
111 float volume = playlist_VolumeGet( THEPL );
112 libUpdateVolume( (volume >= 0.f) ? volume : 1.f );
113 /* Sync mute status */
114 if( playlist_MuteGet( THEPL ) > 0 )
115 updateMuteStatus( true );
116
117 /* Volume control connection */
118 volumeSlider->setTracking( true );
119 CONNECT( volumeSlider, valueChanged( int ), this, valueChangedFilter( int ) );
120 CONNECT( this, valueReallyChanged( int ), this, userUpdateVolume( int ) );
121 CONNECT( THEMIM, volumeChanged( float ), this, libUpdateVolume( float ) );
122 CONNECT( THEMIM, soundMuteChanged( bool ), this, updateMuteStatus( bool ) );
123}

125void SoundWidget::refreshLabels()
126{
127 int i_sliderVolume = volumeSlider->value();
128
129 if( b_is_muted )
130 {
131 volMuteLabel->setPixmap( QPixmap(":/toolbar/volume-muted" ) );
132 volMuteLabel->setToolTip(qfu(vlc_pgettext("Tooltip|Unmute", "Unmute")));
133 return;
134 }
135
136 if( i_sliderVolume < VOLUME_MAX / 3 )
137 volMuteLabel->setPixmap( QPixmap( ":/toolbar/volume-low" ) );
138 else if( i_sliderVolume > (VOLUME_MAX * 2 / 3 ) )
139 volMuteLabel->setPixmap( QPixmap( ":/toolbar/volume-high" ) );
140 else volMuteLabel->setPixmap( QPixmap( ":/toolbar/volume-medium" ) );
141 volMuteLabel->setToolTip( qfu(vlc_pgettext("Tooltip|Mute", "Mute")) );
142}

144/* volumeSlider changed value event slot */
145void SoundWidget::userUpdateVolume( int i_sliderVolume )
146{
147 /* Only if volume is set by user action on slider */
148 setMuted( false );
149 playlist_VolumeSet( THEPL, i_sliderVolume / 100.f );
150 refreshLabels();
151}
152
153/* libvlc changed value event slot */
154void SoundWidget::libUpdateVolume( float volume )
155{
156 long i_volume = lroundf(volume * 100.f);
157 if( i_volume != volumeSlider->value() )
158 {
159 b_ignore_valuechanged = true;
160 volumeSlider->setValue( i_volume );
161 b_ignore_valuechanged = false;
162 }
163 refreshLabels();
164}
165
166void SoundWidget::valueChangedFilter( int i_val )
167{
168 /* valueChanged is also emitted when the lib setValue() */
169 if ( !b_ignore_valuechanged ) emit valueReallyChanged( i_val );
170}

172/* libvlc mute/unmute event slot */
173void SoundWidget::updateMuteStatus( bool mute )
174{
175 b_is_muted = mute;
176
177 SoundSlider *soundSlider = qobject_cast<SoundSlider *>(volumeSlider);
178 if( soundSlider )
179 soundSlider->setMuted( mute );
180 refreshLabels();
181}

183void SoundWidget::showVolumeMenu( QPoint pos )
184{
185 volumeMenu->setFixedHeight( volumeMenu->sizeHint().height() );
186 volumeMenu->exec( QCursor::pos() - pos - QPoint( 0, volumeMenu->height()/2 )
187 + QPoint( width(), height() /2) );
188}
189
190void SoundWidget::setMuted( bool mute )
191{
192 b_is_muted = mute;
193 playlist_t *p_playlist = THEPL;
194 playlist_MuteSet( p_playlist, mute );
195}

197bool SoundWidget::eventFilter( QObject *obj, QEvent *e )
198{
199 VLC_UNUSED( obj );
200 if( e->type() == QEvent::MouseButtonPress )
201 {
202 QMouseEvent *event = static_cast<QMouseEvent*>(e);
203 if( event->button() == Qt::LeftButton )
204 {
205 if( volumeSlider->orientation() == Qt::Vertical )
206 {
207 showVolumeMenu( event->pos() );
208 }
209 else
210 {
211 setMuted( !b_is_muted );
212 }
213 e->accept();
214 return true;
215 }
216 }
217 e->ignore();
218 return false;
219}

258void AspectRatioComboBox::updateRatios()
259{
260 /* Clear the list before updating */
261 clear();
262 vlc_value_t val_list, text_list;
263 vout_thread_t* p_vout = THEMIM->getVout();
264
265 /* Disable if there is no vout */
266 if( p_vout == NULL )
267 {
268 addItem( qtr("Aspect Ratio") );
269 setDisabled( true );
270 return;
271 }
272
273 var_Change( p_vout, "aspect-ratio", VLC_VAR_GETLIST, &val_list, &text_list );
274 for( int i = 0; i < val_list.p_list->i_count; i++ )
275 addItem( qfu( text_list.p_list->p_values[i].psz_string ),
276 QString( val_list.p_list->p_values[i].psz_string ) );
277 setEnabled( true );
278 var_FreeList( &val_list, &text_list );
279 vlc_object_release( p_vout );
280}

282void AspectRatioComboBox::updateAspectRatio( int x )
283{
284 vout_thread_t* p_vout = THEMIM->getVout();
285 if( p_vout && x >= 0 )
286 {
287 var_SetString( p_vout, "aspect-ratio", qtu( itemData(x).toString() ) );
288 }
289 if( p_vout )
290 vlc_object_release( p_vout );
291}
