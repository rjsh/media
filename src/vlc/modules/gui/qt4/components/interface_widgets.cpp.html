96/**
97 * Request the video to avoid the conflicts
98 **/
99WId VideoWidget::request( int *pi_x, int *pi_y,
100 unsigned int *pi_width, unsigned int *pi_height,
101 bool b_keep_size )
102{
103 msg_Dbg( p_intf, "Video was requested %i, %i", *pi_x, *pi_y );
104
105 if( stable )
106 {
107 msg_Dbg( p_intf, "embedded video already in use" );
108 return 0;
109 }
110 if( b_keep_size )
111 {
112 *pi_width = size().width();
113 *pi_height = size().height();
114 }
115
116 /* The owner of the video window needs a stable handle (WinId). Reparenting
117 * in Qt4-X11 changes the WinId of the widget, so we need to create another
118 * dummy widget that stays within the reparentable widget. */
119 stable = new QWidget();
120 QPalette plt = palette();
121 plt.setColor( QPalette::Window, Qt::black );
122 stable->setPalette( plt );
123 stable->setAutoFillBackground(true);
124 /* Force the widget to be native so that it gets a winId() */
125 stable->setAttribute( Qt::WA_NativeWindow, true );
126 /* Indicates that the widget wants to draw directly onto the screen.
127 Widgets with this attribute set do not participate in composition
128 management */
129 /* This is currently disabled on X11 as it does not seem to improve
130 * performance, but causes the video widget to be transparent... */
131#if !defined (Q_WS_X11) && !defined (Q_WS_QPA)
132 stable->setAttribute( Qt::WA_PaintOnScreen, true );
133#endif
134
135 layout->addWidget( stable );
136
137#ifdef Q_WS_X11
138 /* HACK: Only one X11 client can subscribe to mouse button press events.
139 * VLC currently handles those in the video display.
140 * Force Qt4 to unsubscribe from mouse press and release events. */
141 Display *dpy = QX11Info::display();
142 Window w = stable->winId();
143 XWindowAttributes attr;
144
145 XGetWindowAttributes( dpy, w, &attr );
146 attr.your_event_mask &= ~(ButtonPressMask|ButtonReleaseMask);
147 XSelectInput( dpy, w, attr.your_event_mask );
148#endif
149 sync();
150 return stable->winId();
151}

153/* Set the Widget to the correct Size */
154/* Function has to be called by the parent
155 Parent has to care about resizing itself */
156void VideoWidget::SetSizing( unsigned int w, unsigned int h )
157{
158 resize( w, h );
159 emit sizeChanged( w, h );
160 /* Work-around a bug?misconception? that would happen when vout core resize
161 twice to the same size and would make the vout not centered.
162 This cause a small flicker.
163 See #3621
164 */
165 if( (unsigned)size().width() == w && (unsigned)size().height() == h )
166 updateGeometry();
167 sync();
168}

173
174 if( stable )
175 {
176 layout->removeWidget( stable );
177 stable->deleteLater();
178 stable = NULL;
179 }
180
181 updateGeometry();
182}

231
232void BackgroundWidget::paintEvent( QPaintEvent *e )
233{
234 if ( !b_withart )
235 {
236 /* we just want background autofill */
237 QWidget::paintEvent( e );
238 return;
239 }
240
241 int i_maxwidth, i_maxheight;
242 QPixmap pixmap = QPixmap( pixmapUrl );
243 QPainter painter(this);
244 QBitmap pMask;
245 float f_alpha = 1.0;
246
247 i_maxwidth = __MIN( maximumWidth(), width() ) - MARGIN * 2;
248 i_maxheight = __MIN( maximumHeight(), height() ) - MARGIN * 2;
249
250 painter.setOpacity( property( "opacity" ).toFloat() );
251
252 if ( height() > MARGIN * 2 )
253 {
254 /* Scale down the pixmap if the widget is too small */
255 if( pixmap.width() > i_maxwidth || pixmap.height() > i_maxheight )
256 {
257 pixmap = pixmap.scaled( i_maxwidth, i_maxheight,
258 Qt::KeepAspectRatio, Qt::SmoothTransformation );
259 }
260 else
261 if ( b_expandPixmap &&
262 pixmap.width() < width() && pixmap.height() < height() )
263 {
264 /* Scale up the pixmap to fill widget's size */
265 f_alpha = ( (float) pixmap.height() / (float) height() );
266 pixmap = pixmap.scaled(
267 width() - MARGIN * 2,
268 height() - MARGIN * 2,
269 Qt::KeepAspectRatio,
270 ( f_alpha < .2 )? /* Don't waste cpu when not visible */
271 Qt::SmoothTransformation:
272 Qt::FastTransformation
273 );
274 /* Non agressive alpha compositing when sizing up */
275 pMask = QBitmap( pixmap.width(), pixmap.height() );
276 pMask.fill( QColor::fromRgbF( 1.0, 1.0, 1.0, f_alpha ) );
277 pixmap.setMask( pMask );
278 }
279
280 painter.drawPixmap(
281 MARGIN + ( i_maxwidth - pixmap.width() ) /2,
282 MARGIN + ( i_maxheight - pixmap.height() ) /2,
283 pixmap);
284 }
285 QWidget::paintEvent( e );
286}

342void EasterEggBackgroundWidget::spawnFlakes()
343{
344 if ( ! isVisible() ) return;
345
346 double w = (double) width() / RAND_MAX;
347
348 int i_spawn = ( (double) qrand() / RAND_MAX ) * i_rate;
349
350 QLinkedList<flake *>::iterator it = flakes->begin();
351 while( it != flakes->end() )
352 {
353 flake *current = *it;
354 current->point.setY( current->point.y() + i_speed );
355 if ( current->point.y() + i_speed >= height() )
356 {
357 delete current;
358 it = flakes->erase( it );
359 }
360 else
361 it++;
362 }
363
364 if ( flakes->size() < MAX_FLAKES )
365 for ( int i=0; i<i_spawn; i++ )
366 {
367 flake *f = new flake;
368 f->point.setX( qrand() * w );
369 f->b_fat = ( qrand() < ( RAND_MAX * .33 ) );
370 flakes->append( f );
371 }
372 update();
373}
374
375void EasterEggBackgroundWidget::reset()
376{
377 while ( !flakes->isEmpty() )
378 delete flakes->takeFirst();
379}
380
381void EasterEggBackgroundWidget::paintEvent( QPaintEvent *e )
382{
383 QPainter painter(this);
384
385 painter.setBrush( QBrush( QColor(Qt::white) ) );
386 painter.setPen( QPen(Qt::white) );
387
388 QLinkedList<flake *>::const_iterator it = flakes->constBegin();
389 while( it != flakes->constEnd() )
390 {
391 const flake * const f = *(it++);
392 if ( f->b_fat )
393 {
394 /* Xsnow like :p */
395 painter.drawPoint( f->point.x(), f->point.y() -1 );
396 painter.drawPoint( f->point.x() + 1, f->point.y() );
397 painter.drawPoint( f->point.x(), f->point.y() +1 );
398 painter.drawPoint( f->point.x() - 1, f->point.y() );
399 }
400 else
401 {
402 painter.drawPoint( f->point );
403 }
404 }
405
406 BackgroundWidget::paintEvent( e );
407}

587void SpeedControlWidget::updateControls( float rate )
588{
589 if( speedSlider->isSliderDown() )
590 {
591 //We don't want to change anything if the user is using the slider
592 return;
593 }
594
595 double value = 17 * log( rate ) / log( 2. );
596 int sliderValue = (int) ( ( value > 0 ) ? value + .5 : value - .5 );
597
598 if( sliderValue < speedSlider->minimum() )
599 {
600 sliderValue = speedSlider->minimum();
601 }
602 else if( sliderValue > speedSlider->maximum() )
603 {
604 sliderValue = speedSlider->maximum();
605 }
606 lastValue = sliderValue;
607
608 speedSlider->setValue( sliderValue );
609 //spinBox->setValue( rate );
610}
611
612void SpeedControlWidget::updateRate( int sliderValue )
613{
614 if( sliderValue == lastValue )
615 return;
616
617 double speed = pow( 2, (double)sliderValue / 17 );
618 int rate = INPUT_RATE_DEFAULT / speed;
619
620 THEMIM->getIM()->setRate(rate);
621 //spinBox->setValue( var_InheritFloat( THEPL, "rate" ) );
622}

766void TimeLabel::setDisplayPosition( float pos, int64_t t, int length )
767{
768 if( pos == -1.f )
769 {
770 setMinimumSize( QSize( 0, 0 ) );
771 if( displayType == TimeLabel::Both )
772 setText( "--:--/--:--" );
773 else
774 setText( "--:--" );
775 return;
776 }
777
778 int time = t / 1000000;
779
780 secstotimestr( psz_length, length );
781 secstotimestr( psz_time, ( b_remainingTime && length ) ? length - time
782 : time );
783
784 // compute the minimum size that will be required for the psz_length
785 // and use it to enforce a minimal size to avoid "dancing" widgets
786 QSize minsize( 0, 0 );
787 if ( length > 0 )
788 {
789 QMargins margins = contentsMargins();
790 minsize += QSize(
791 fontMetrics().size( 0, QString( psz_length ), 0, 0 ).width(),
792 sizeHint().height()
793 );
794 minsize += QSize( margins.left() + margins.right() + 8, 0 ); /* +padding */
795
796 if ( b_remainingTime )
797 minsize += QSize( fontMetrics().size( 0, "-", 0, 0 ).width(), 0 );
798 }
799
800 switch( displayType )
801 {
802 case TimeLabel::Elapsed:
803 setMinimumSize( minsize );
804 setText( QString( psz_time ) );
805 break;
806 case TimeLabel::Remaining:
807 if( b_remainingTime )
808 {
809 setMinimumSize( minsize );
810 setText( QString("-") + QString( psz_time ) );
811 }
812 else
813 {
814 setMinimumSize( QSize( 0, 0 ) );
815 setText( QString( psz_length ) );
816 }
817 break;
818 case TimeLabel::Both:
819 default:
820 QString timestr = QString( "%1%2/%3" )
821 .arg( QString( (b_remainingTime && length) ? "-" : "" ) )
822 .arg( QString( psz_time ) )
823 .arg( QString( ( !length && time ) ? "--:--" : psz_length ) );
824
825 setText( timestr );
826 break;
827 }
828 cachedLength = length;
829}

831void TimeLabel::setDisplayPosition( float pos )
832{
833 if( pos == -1.f || cachedLength == 0 )
834 {
835 setText( " --:--/--:-- " );
836 return;
837 }
838
839 int time = pos * cachedLength;
840 secstotimestr( psz_time,
841 ( b_remainingTime && cachedLength ?
842 cachedLength - time : time ) );
843 QString timestr = QString( "%1%2/%3" )
844 .arg( QString( (b_remainingTime && cachedLength) ? "-" : "" ) )
845 .arg( QString( psz_time ) )
846 .arg( QString( ( !cachedLength && time ) ? "--:--" : psz_length ) );
847
848 setText( timestr );
849}
