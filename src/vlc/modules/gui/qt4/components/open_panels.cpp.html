63/* Populate a combobox with the devices matching a pattern.
64 Combobox will automatically do autocompletion on the edit zone */
65#define POPULATE_WITH_DEVS(ppsz_devlist, targetCombo) \
66 QStringList targetCombo ## StringList = QStringList(); \
67 for ( size_t i = 0; i< sizeof(ppsz_devlist) / sizeof(*ppsz_devlist); i++ ) \
68 targetCombo ## StringList << QString( ppsz_devlist[ i ] ); \
69 targetCombo->addItems( QDir( "/dev/" )\
70 .entryList( targetCombo ## StringList, QDir::System )\
71 .replaceInStrings( QRegExp("^"), "/dev/" ) \
72 );

126{

197void FileOpenPanel::dropEvent( QDropEvent *event )
198{
199 if( event->possibleActions() & Qt::CopyAction )
200 event->setDropAction( Qt::CopyAction );
201 else
202 return;
203
204 const QMimeData *mimeData = event->mimeData();
205 foreach( const QUrl &url, mimeData->urls() )
206 {
207 if( url.isValid() )
208 {
209 QListWidgetItem *item = new QListWidgetItem(
210 toNativeSeparators( url.toLocalFile() ),
211 ui.fileListWidg );
212 item->setFlags( Qt::ItemIsEditable | Qt::ItemIsEnabled );
213 ui.fileListWidg->addItem( item );
214 }
215 }
216 updateMRL();
217 updateButtons();
218 event->accept();
219}

261/* Update the current MRL */
262void FileOpenPanel::updateMRL()
263{
264 QStringList fileList;
265 QString mrl;
266
267 /* File Listing */
268 if( dialogBox == NULL )
269 for( int i = 0; i < ui.fileListWidg->count(); i++ )
270 {
271 if( !ui.fileListWidg->item( i )->text().isEmpty() )
272 fileList << toURI(ui.fileListWidg->item( i )->text());
273 }
274 else
275 {
276 fileList = dialogBox->selectedFiles();
277 for( int i = 0; i < fileList.count(); i++ )
278 fileList[i] = toURI( fileList[i] );
279 }
280
281 /* Options */
282 if( ui.subGroupBox->isChecked() && !ui.subInput->text().isEmpty() ) {
283 mrl.append( " :sub-file=" + colon_escape( ui.subInput->text() ) );
284 }
285
286 emit methodChanged( "file-caching" );
287 emit mrlUpdated( fileList, mrl );
288}

369#ifdef _WIN32 /* Disc drives probing for Windows */
370void DiscOpenPanel::onFocus()
371{
372 ui.deviceCombo->clear();
373 wchar_t szDrives[512];
374 szDrives[0] = '\0';
375 if( GetLogicalDriveStringsW( sizeof( szDrives ) - 1, szDrives ) )
376 {
377 wchar_t *drive = szDrives;
378 UINT oldMode = SetErrorMode( SEM_FAILCRITICALERRORS );
379 while( *drive )
380 {
381 if( GetDriveTypeW(drive) == DRIVE_CDROM )
382 {
383 wchar_t psz_name[512] = L"";
384 GetVolumeInformationW( drive, psz_name, 511, NULL, NULL, NULL, NULL, 0 );
385
386 QString displayName = FromWide( drive );
387 char *psz_title = FromWide( psz_name );
388 if( !EMPTY_STR(psz_title)) {
389 displayName = displayName + " - " + psz_title;
390 }
391
392 ui.deviceCombo->addItem( displayName, FromWide( drive ) );
393 free( psz_title );
394 }
395
396 /* go to next drive */
397 while( *(drive++) );
398 }
399 SetErrorMode(oldMode);
400 }
401
402 int temp = ui.deviceCombo->findData( config_GetPsz( p_intf, "dvd" ), Qt::UserRole, Qt::MatchStartsWith );
403 if( temp != -1 )
404 ui.deviceCombo->setCurrentIndex( temp );
405}
406#elif defined( __OS2__ ) /* Disc drives probing for OS/2 */
407void DiscOpenPanel::onFocus()
408{
409
410 ui.deviceCombo->clear();
411
412#define IOCTL_CDROMDISK2 0x82
413#define CDROMDISK2_DRIVELETTERS 0x60
414
415 HFILE hcd2;
416 ULONG ulAction;
417 ULONG ulParamLen;
418 ULONG ulData;
419 ULONG ulDataLen;
420
421 if( DosOpen(( PSZ )"CD-ROM2$", ( PHFILE )&hcd2, &ulAction, 0, FILE_NORMAL,
422 OPEN_ACTION_OPEN_IF_EXISTS | OPEN_ACTION_FAIL_IF_NEW,
423 OPEN_ACCESS_READONLY | OPEN_SHARE_DENYNONE,
424 NULL ))
425 return;
426
427 if( !DosDevIOCtl( hcd2, IOCTL_CDROMDISK2, CDROMDISK2_DRIVELETTERS,
428 NULL, 0, &ulParamLen, &ulData, sizeof( ulData ), &ulDataLen ))
429 {
430 char szDriveName[] = "X:\\";
431
432 int count = LOUSHORT( ulData );
433 int drive = HIUSHORT( ulData );
434
435 for( ; count; --count, ++drive )
436 {
437 szDriveName[ 0 ] = 'A' + drive;
438
439 QString name = qfu( szDriveName );
440
441 ui.deviceCombo->addItem( name, name );
442 }
443 }
444
445 DosClose( hcd2 );
446}
447#endif

536/* Update the current MRL */
537void DiscOpenPanel::updateMRL()
538{
539 QString discPath;
540 QStringList fileList;
541
542 discPath = ui.deviceCombo->currentText();
543
544 int tmp = ui.deviceCombo->findText( discPath );
545 if( tmp != -1 && ui.deviceCombo->itemData( tmp ) != QVariant::Invalid )
546 discPath = ui.deviceCombo->itemData( tmp ).toString();
547
548 /* MRL scheme */
549 const char *scheme;
550 /* DVD */
551 if( ui.dvdRadioButton->isChecked() ) {
552 if( !ui.dvdsimple->isChecked() )
553 scheme = "dvd";
554 else
555 scheme = "dvdsimple";
556 } else if ( ui.bdRadioButton->isChecked() )
557 scheme = "bluray";
558 /* VCD */
559 else if ( ui.vcdRadioButton->isChecked() )
560 scheme = "vcd";
561 /* CDDA */
562 else
563 scheme = "cdda";
564
565 char *mrl = vlc_path2uri( qtu(discPath), scheme );
566 if( unlikely(mrl == NULL) )
567 return;
568
569 /* Title/chapter encoded in MRL */
570 QString anchor = "";
571 if( ui.titleSpin->value() > 0 ) {
572 if( ui.dvdRadioButton->isChecked() || ui.bdRadioButton->isChecked() ) {
573 anchor = QString("#%1").arg( ui.titleSpin->value() );
574 if ( ui.chapterSpin->value() > 0 )
575 anchor += QString(":%1").arg( ui.chapterSpin->value() );
576 }
577 else if ( ui.vcdRadioButton->isChecked() )
578 anchor = QString("#%1").arg( ui.titleSpin->value() );
579 }
580
581 emit methodChanged( "disc-caching" );
582
583 fileList << (qfu(mrl) + anchor);
584 free(mrl);
585
586 QString opts = "";
587
588 /* Input item options */
589 if ( ui.dvdRadioButton->isChecked() || ui.vcdRadioButton->isChecked() )
590 {
591 if ( ui.audioSpin->value() >= 0 )
592 opts += " :audio-track=" +
593 QString("%1").arg( ui.audioSpin->value() );
594 if ( ui.subtitlesSpin->value() >= 0 )
595 opts += " :sub-track=" +
596 QString("%1").arg( ui.subtitlesSpin->value() );
597 }
598 else if( ui.audioCDRadioButton->isChecked() )
599 {
600 if( ui.titleSpin->value() > 0 )
601 opts += QString(" :cdda-track=%1").arg( ui.titleSpin->value() );
602 }
603 else if ( ui.bdRadioButton->isChecked() )
604 {
605 if ( ui.dvdsimple->isChecked() == false )
606 opts += " :bluray-menu";
607 }
608 emit mrlUpdated( fileList, opts );
609}

1116void CaptureOpenPanel::updateMRL()
1117{
1118 QString mrl = "";
1119 QStringList fileList;
1120 int i_devicetype = ui.deviceCombo->itemData(
1121 ui.deviceCombo->currentIndex() ).toInt();
1122 switch( i_devicetype )
1123 {
1124#ifdef _WIN32
1125 case DSHOW_DEVICE:
1126 fileList << "dshow://";
1127 mrl+= " :dshow-vdev=" +
1128 colon_escape( QString("%1").arg( vdevDshowW->getValue() ) );
1129 mrl+= " :dshow-adev=" +
1130 colon_escape( QString("%1").arg( adevDshowW->getValue() ) )+" ";
1131 if( dshowVSizeLine->isModified() )
1132 mrl += ":dshow-size=" + dshowVSizeLine->text();
1133 break;
1134#else
1135 case V4L2_DEVICE:
1136 fileList << "v4l2://" + v4l2VideoDevice->currentText();
1137 mrl += ":v4l2-standard="
1138 + v4l2StdBox->itemData( v4l2StdBox->currentIndex() ).toString();
1139 if( !v4l2AudioDevice->currentText().isEmpty() )
1140 mrl += " :input-slave=alsa://" + v4l2AudioDevice->currentText();
1141 break;
1142 case JACK_DEVICE:
1143 mrl = "jack://";
1144 mrl += "channels=" + QString::number( jackChannels->value() );
1145 mrl += ":ports=" + jackPortsSelected->text();
1146 fileList << mrl; mrl = "";
1147
1148 if ( jackPace->isChecked() )
1149 {
1150 mrl += " :jack-input-use-vlc-pace";
1151 }
1152 if ( jackConnect->isChecked() )
1153 {
1154 mrl += " :jack-input-auto-connect";
1155 }
1156 break;
1157 case PVR_DEVICE:
1158 fileList << "v4l2://" + pvrDevice->currentText();
1159 mrl += ":v4l2-standard="
1160 + pvrNormBox->itemData( pvrNormBox->currentIndex() ).toString();
1161 if( pvrFreq->value() )
1162 mrl += ":v4l2-tuner-frequency=" + QString::number( pvrFreq->value() );
1163 if( !pvrAudioDevice->currentText().isEmpty() )
1164 mrl += " :input-slave=" + pvrAudioDevice->currentText();
1165 break;
1166#endif
1167 case DTV_DEVICE:
1168 if( dvbc->isChecked() ) mrl = "dvb-c://";
1169 else
1170 if( dvbs->isChecked() ) mrl = "dvb-s://";
1171 else
1172 if( dvbs2->isChecked() ) mrl = "dvb-s2://";
1173 else
1174 if( dvbt->isChecked() ) mrl = "dvb-t://";
1175 else
1176 if( dvbt2->isChecked() ) mrl = "dvb-t2://";
1177 else
1178 if( atsc->isChecked() ) mrl = "atsc://";
1179 else
1180 if( cqam->isChecked() ) mrl = "cqam://";
1181
1182 mrl += "frequency=" + QString::number( dvbFreq->value() ) + "000";
1183
1184 if( dvbc->isChecked() || cqam->isChecked() )
1185 mrl += ":modulation="
1186 + dvbQamBox->itemData( dvbQamBox->currentIndex() ).toString();
1187 if( dvbs2->isChecked() )
1188 mrl += ":modulation="
1189 + dvbPskBox->itemData( dvbPskBox->currentIndex() ).toString();
1190 if( dvbc->isChecked() || dvbs->isChecked() || dvbs2->isChecked() )
1191 mrl += ":srate=" + QString::number( dvbSrate->value() );
1192 if( dvbt->isChecked() || dvbt2->isChecked() )
1193 mrl += ":bandwidth=" +
1194 QString::number( dvbBandBox->itemData(
1195 dvbBandBox->currentIndex() ).toInt() );
1196
1197 fileList << mrl; mrl= "";
1198 mrl += " :dvb-adapter=" + QString::number( dvbCard->value() );
1199 break;
1200 case SCREEN_DEVICE:
1201 fileList << "screen://";
1202 mrl = " :screen-fps=" + QString::number( screenFPS->value(), 'f' );
1203 updateButtons();
1204 break;
1205 }
1206 emit methodChanged( "live-caching" );
1207
1208 if( !advMRL.isEmpty() ) mrl += " " + advMRL;
1209
1210 emit mrlUpdated( fileList, mrl );
1211}

1310 QGridLayout *gLayout = new QGridLayout( advFrame );
1311
1312 scroll->setWidgetResizable( true );
1313 scroll->setWidget( advFrame );
1314
1315 /* Create the options inside the FrameLayout */
1316 for( int n = 0; n < (int)i_confsize; n++ )
1317 {
1318 module_config_t *p_item = p_config + n;
1319 QString name = p_item->psz_name;
1320
1321 if( name.isEmpty() || configList.contains( name ) )
1322 continue;
1323
1324 msg_Dbg( p_intf, "item %s", p_item->psz_name);
1325 ConfigControl *config = ConfigControl::createControl(
1326 VLC_OBJECT( p_intf ), p_item, advFrame, gLayout, n );
1327 if( config )
1328 controls.append( config );
1329 }
1330
1331 /* Button stuffs */
1332 QDialogButtonBox *advButtonBox = new QDialogButtonBox( adv );
1333 QPushButton *closeButton = new QPushButton( qtr( "OK" ) );
1334 QPushButton *cancelButton = new QPushButton( qtr( "Cancel" ) );
1335
1336 CONNECT( closeButton, clicked(), adv, accept() );
1337 CONNECT( cancelButton, clicked(), adv, reject() );
1338
1339 advButtonBox->addButton( closeButton, QDialogButtonBox::AcceptRole );
1340 advButtonBox->addButton( cancelButton, QDialogButtonBox::RejectRole );
1341
1342 mainLayout->addWidget( advButtonBox );
1343
1344 /* Creation of the MRL */
1345 if( adv->exec() )
1346 {
1347 QString tempMRL = "";
1348 for( int i = 0; i < controls.count(); i++ )
1349 {
1350 ConfigControl *control = controls[i];
1351
1352 tempMRL += (i ? " :" : ":");
1353
1354 if( control->getType() == CONFIG_ITEM_BOOL )
1355 if( !(qobject_cast<VIntConfigControl *>(control)->getValue() ) )
1356 tempMRL += "no-";
1357
1358 tempMRL += control->getName();
1359
1360 switch( control->getType() )
1361 {
1362 case CONFIG_ITEM_STRING:
1363 case CONFIG_ITEM_LOADFILE:
1364 case CONFIG_ITEM_SAVEFILE:
1365 case CONFIG_ITEM_DIRECTORY:
1366 case CONFIG_ITEM_MODULE:
1367 tempMRL += colon_escape( QString("=%1").arg( qobject_cast<VStringConfigControl *>(control)->getValue() ) );
1368 break;
1369 case CONFIG_ITEM_INTEGER:
1370 tempMRL += QString("=%1").arg( qobject_cast<VIntConfigControl *>(control)->getValue() );
1371 break;
1372 case CONFIG_ITEM_FLOAT:
1373 tempMRL += QString("=%1").arg( qobject_cast<VFloatConfigControl *>(control)->getValue() );
1374 break;
1375 }
1376 }
1377 advMRL = tempMRL;
1378 updateMRL();
1379 }
1380 qDeleteAll( controls );
1381 delete adv;
1382 module_config_free( p_config );
1383}
