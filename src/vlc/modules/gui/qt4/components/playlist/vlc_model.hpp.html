47/* Provides non Q_Object interface for Models.
48 This allows multiple inheritance on already QAbstractModel based
49 Qobjects like Q*ProxyModel.
50 Signals being a Q_Object property, they need to be redirected
51 using a QObject based class member.
52*/
53class VLCModelSubInterface
54{
55public:
56 VLCModelSubInterface();
57 virtual ~VLCModelSubInterface();
58 enum nodeRole
59 {
60 IsCurrentRole = Qt::UserRole,
61 IsLeafNodeRole,
62 IsCurrentsParentNodeRole
63 };
64 virtual void rebuild( playlist_item_t * p = NULL ) = 0;
65 virtual void doDelete( QModelIndexList ) = 0;
66 virtual void createNode( QModelIndex, QString ) = 0;
67 virtual void renameNode( QModelIndex, QString ) = 0;
68 virtual void removeAll() = 0;
69
70 virtual QModelIndex rootIndex() const = 0;
71 virtual void filter( const QString& search_text, const QModelIndex & root, bool b_recursive ) = 0;
72 virtual QModelIndex currentIndex() const = 0;
73 virtual QModelIndex indexByPLID( const int i_plid, const int c ) const = 0;
74 virtual QModelIndex indexByInputItemID( const int i_inputitem_id, const int c ) const = 0;
75 virtual int itemId( const QModelIndex &, int type ) const = 0;
76 virtual bool isTree() const = 0;
77 virtual bool canEdit() const = 0;
78 virtual QString getURI( const QModelIndex &index ) const = 0;
79 virtual input_item_t *getInputItem( const QModelIndex & ) const = 0;
80 virtual QString getTitle( const QModelIndex &index ) const = 0;
81 enum actions
82 {
83 ACTION_PLAY = 1,
84 ACTION_STREAM,
85 ACTION_SAVE,
86 ACTION_INFO,
87 ACTION_ADDTOPLAYLIST,
88 ACTION_REMOVE,
89 ACTION_SORT,
90 ACTION_EXPLORE,
91 ACTION_CREATENODE,
92 ACTION_RENAMENODE,
93 ACTION_CLEAR,
94 ACTION_ENQUEUEFILE,
95 ACTION_ENQUEUEDIR,
96 ACTION_ENQUEUEGENERIC,
97 ACTION_SAVETOPLAYLIST
98 };
99 struct actionsContainerType
100 {
101 actions action;
102 int column; /* for sorting */
103 QStringList uris; /* for enqueuing */
104 QString options;
105 };
106 virtual bool action( QAction *, const QModelIndexList & ) = 0;
107 virtual bool isSupportedAction( actions action, const QModelIndex & ) const = 0;
108 static int columnFromMeta( int meta_col );
109
110 virtual void activateItem( const QModelIndex &index ) = 0;
111 virtual void ensureArtRequested( const QModelIndex &index ) = 0;
112};
113
114/* Abstract VLC Model ; Base for custom models.
115 Only implements methods sharing the same code that would be
116 implemented in subclasses.
117 Any custom method here must be only used in implemented methods.
118*/
119class VLCModel : public QAbstractItemModel, public VLCModelSubInterface
120{
121 Q_OBJECT
122public:
123 VLCModel( intf_thread_t *_p_intf, QObject *parent = 0 );
124 virtual ~VLCModel();
125
126 /*** QAbstractItemModel subclassing ***/
127 virtual int columnCount( const QModelIndex &parent = QModelIndex() ) const;
128 QVariant headerData( int, Qt::Orientation, int ) const;
129
130 /*** VLCModelSubInterface subclassing ***/
131 virtual int itemId( const QModelIndex &, int type ) const;
132 virtual QString getURI( const QModelIndex &index ) const;
133 virtual input_item_t *getInputItem( const QModelIndex & ) const;
134 virtual QString getTitle( const QModelIndex &index ) const;
135
136 /* Custom */
137 static int columnToMeta( int _column );
138 static int metaToColumn( int meta );
139 static QString getMeta( const QModelIndex & index, int meta );
140 static QPixmap getArtPixmap( const QModelIndex & index, const QSize & size );
141
142public slots:
143 /* slots handlers */
144 virtual void ensureArtRequested( const QModelIndex &index );
145
146signals:
147 void currentIndexChanged( const QModelIndex& );
148 void rootIndexChanged();
149
150protected:
151 /* Custom methods / helpers */
152 virtual bool isCurrent( const QModelIndex &index ) const;
153 virtual bool isParent( const QModelIndex &index, const QModelIndex &current ) const = 0;
154 virtual bool isLeaf( const QModelIndex &index ) const = 0;
155 virtual AbstractPLItem *getItem( const QModelIndex & index ) const;
156
157 QIcon icons[ITEM_TYPE_NUMBER];
158
159 intf_thread_t *p_intf;
160};
