50void SelectorActionButton::paintEvent( QPaintEvent *event )
51{
52 QPainter p( this );
53 QColor color = palette().color( QPalette::HighlightedText );
54 color.setAlpha( 80 );
55 if( underMouse() )
56 p.fillRect( rect(), color );
57 p.setPen( color );
58 int frame = style()->pixelMetric( QStyle::PM_DefaultFrameWidth, 0, this );
59 p.drawLine( rect().topLeft() + QPoint( 0, frame ),
60 rect().bottomLeft() - QPoint( 0, frame ) );
61 QFramelessButton::paintEvent( event );
62}

78void PLSelItem::addAction( ItemAction act, const QString& tooltip )
79{
80 if( lblAction ) return; //might change later
81
82 QIcon icon;
83
84 switch( act )
85 {
86 case ADD_ACTION:
87 icon = QIcon( ":/buttons/playlist/playlist_add" ); break;
88 case RM_ACTION:
89 icon = QIcon( ":/buttons/playlist/playlist_remove" ); break;
90 default:
91 return;
92 }
93
94 lblAction = new SelectorActionButton();
95 lblAction->setIcon( icon );
96 lblAction->setMinimumWidth( lblAction->sizeHint().width() + 6 );
97
98 if( !tooltip.isEmpty() ) lblAction->setToolTip( tooltip );
99
100 layout->addWidget( lblAction, 0 );
101 lblAction->hide();
102
103 CONNECT( lblAction, clicked(), this, triggerAction() );
104}

194/*
195 * Reads and updates the playlist's duration as [xx:xx] after the label in the tree
196 * item - the treeview item to get the duration for
197 * prefix - the string to use before the time (should be the category name)
198 */
199void PLSelector::updateTotalDuration( PLSelItem* item, const char* prefix )
200{
201 /* Getting the playlist */
202 QVariant playlistVariant = item->treeItem()->data( 0, PL_ITEM_ROLE );
203 playlist_item_t* node = playlistVariant.value<playlist_item_t*>();
204
205 /* Get the duration of the playlist item */
206 playlist_Lock( THEPL );
207 mtime_t mt_duration = playlist_GetNodeDuration( node );
208 playlist_Unlock( THEPL );
209
210 /* Formatting time */
211 QString qs_timeLabel( prefix );
212
213 int i_seconds = mt_duration / 1000000;
214 int i_minutes = i_seconds / 60;
215 i_seconds = i_seconds % 60;
216 if( i_minutes >= 60 )
217 {
218 int i_hours = i_minutes / 60;
219 i_minutes = i_minutes % 60;
220 qs_timeLabel += QString(" [%1:%2:%3]").arg( i_hours ).arg( i_minutes, 2, 10, QChar('0') ).arg( i_seconds, 2, 10, QChar('0') );
221 }
222 else
223 qs_timeLabel += QString( " [%1:%2]").arg( i_minutes, 2, 10, QChar('0') ).arg( i_seconds, 2, 10, QChar('0') );
224
225 item->setText( qs_timeLabel );
226}

228void PLSelector::createItems()
229{
230 /* PL */
231 playlistItem = putPLData( addItem( PL_ITEM_TYPE, N_("Playlist"), true ),
232 THEPL->p_playing );
233 playlistItem->treeItem()->setData( 0, SPECIAL_ROLE, QVariant( IS_PL ) );
234 playlistItem->treeItem()->setData( 0, Qt::DecorationRole, QIcon( ":/sidebar/playlist" ) );
235 setCurrentItem( playlistItem->treeItem() );
236
237 /* ML */
238 PLSelItem *ml = putPLData( addItem( PL_ITEM_TYPE, N_("Media Library"), true ),
239 THEPL->p_media_library );
240 ml->treeItem()->setData( 0, SPECIAL_ROLE, QVariant( IS_ML ) );
241 ml->treeItem()->setData( 0, Qt::DecorationRole, QIcon( ":/sidebar/library" ) );
242
243 /* SD nodes */
244 QTreeWidgetItem *mycomp = addItem( CATEGORY_TYPE, N_("My Computer"), false, true )->treeItem();
245 QTreeWidgetItem *devices = addItem( CATEGORY_TYPE, N_("Devices"), false, true )->treeItem();
246 QTreeWidgetItem *lan = addItem( CATEGORY_TYPE, N_("Local Network"), false, true )->treeItem();
247 QTreeWidgetItem *internet = addItem( CATEGORY_TYPE, N_("Internet"), false, true )->treeItem();
248
249#define NOT_SELECTABLE(w) w->setFlags( w->flags() ^ Qt::ItemIsSelectable );
250 NOT_SELECTABLE( mycomp );
251 NOT_SELECTABLE( devices );
252 NOT_SELECTABLE( lan );
253 NOT_SELECTABLE( internet );
254#undef NOT_SELECTABLE
255
256 /* SD subnodes */
257 char **ppsz_longnames;
258 int *p_categories;
259 char **ppsz_names = vlc_sd_GetNames( THEPL, &ppsz_longnames, &p_categories );
260 if( !ppsz_names )
261 return;
262
263 char **ppsz_name = ppsz_names, **ppsz_longname = ppsz_longnames;
264 int *p_category = p_categories;
265 for( ; *ppsz_name; ppsz_name++, ppsz_longname++, p_category++ )
266 {
267 //msg_Dbg( p_intf, "Adding a SD item: %s", *ppsz_longname );
268
269 PLSelItem *selItem;
270 QIcon icon;
271 QString name( *ppsz_name );
272 switch( *p_category )
273 {
274 case SD_CAT_INTERNET:
275 {
276 selItem = addItem( SD_TYPE, *ppsz_longname, false, false, internet );
277 if( name.startsWith( "podcast" ) )
278 {
279 selItem->treeItem()->setData( 0, SPECIAL_ROLE, QVariant( IS_PODCAST ) );
280 selItem->addAction( ADD_ACTION, qtr( "Subscribe to a podcast" ) );
281 CONNECT( selItem, action( PLSelItem* ), this, podcastAdd( PLSelItem* ) );
282 podcastsParent = selItem->treeItem();
283 icon = QIcon( ":/sidebar/podcast" );
284 }
285 else if ( name.startsWith( "lua{" ) )
286 {
287 int i_head = name.indexOf( "sd='" ) + 4;
288 int i_tail = name.indexOf( '\'', i_head );
289 QString iconname = QString( ":/sidebar/sd/%1" ).arg( name.mid( i_head, i_tail - i_head ) );
290 QResource resource( iconname );
291 if ( !resource.isValid() )
292 icon = QIcon( ":/sidebar/network" );
293 else
294 icon = QIcon( iconname );
295 }
296 }
297 break;
298 case SD_CAT_DEVICES:
299 name = name.mid( 0, name.indexOf( '{' ) );
300 selItem = addItem( SD_TYPE, *ppsz_longname, false, false, devices );
301 if ( name == "xcb_apps" )
302 icon = QIcon( ":/sidebar/screen" );
303 else if ( name == "mtp" )
304 icon = QIcon( ":/sidebar/mtp" );
305 else if ( name == "disc" )
306 icon = QIcon( ":/sidebar/disc" );
307 else
308 icon = QIcon( ":/sidebar/capture" );
309 break;
310 case SD_CAT_LAN:
311 selItem = addItem( SD_TYPE, *ppsz_longname, false, false, lan );
312 icon = QIcon( ":/sidebar/lan" );
313 break;
314 case SD_CAT_MYCOMPUTER:
315 name = name.mid( 0, name.indexOf( '{' ) );
316 selItem = addItem( SD_TYPE, *ppsz_longname, false, false, mycomp );
317 if ( name == "video_dir" )
318 icon = QIcon( ":/sidebar/movie" );
319 else if ( name == "audio_dir" )
320 icon = QIcon( ":/sidebar/music" );
321 else if ( name == "picture_dir" )
322 icon = QIcon( ":/sidebar/pictures" );
323 else
324 icon = QIcon( ":/sidebar/movie" );
325 break;
326 default:
327 selItem = addItem( SD_TYPE, *ppsz_longname );
328 }
329
330 selItem->treeItem()->setData( 0, SD_CATEGORY_ROLE, *p_category );
331 putSDData( selItem, *ppsz_name, *ppsz_longname );
332 if ( ! icon.isNull() )
333 selItem->treeItem()->setData( 0, Qt::DecorationRole, icon );
334
335 free( *ppsz_name );
336 free( *ppsz_longname );
337 }
338 free( ppsz_names );
339 free( ppsz_longnames );
340 free( p_categories );
341
342 if( mycomp->childCount() == 0 ) delete mycomp;
343 if( devices->childCount() == 0 ) delete devices;
344 if( lan->childCount() == 0 ) delete lan;
345 if( internet->childCount() == 0 ) delete internet;
346}

348void PLSelector::setSource( QTreeWidgetItem *item )
349{
350 if( !item || item == curItem )
351 return;
352
353 bool b_ok;
354 int i_type = item->data( 0, TYPE_ROLE ).toInt( &b_ok );
355 if( !b_ok || i_type == CATEGORY_TYPE )
356 return;
357
358 bool sd_loaded;
359 if( i_type == SD_TYPE )
360 {
361 QString qs = item->data( 0, NAME_ROLE ).toString();
362 sd_loaded = playlist_IsServicesDiscoveryLoaded( THEPL, qtu( qs ) );
363 if( !sd_loaded )
364 {
365 if ( playlist_ServicesDiscoveryAdd( THEPL, qtu( qs ) ) != VLC_SUCCESS )
366 return ;
367
368 services_discovery_descriptor_t *p_test = new services_discovery_descriptor_t;
369 int i_ret = playlist_ServicesDiscoveryControl( THEPL, qtu( qs ), SD_CMD_DESCRIPTOR, p_test );
370 if( i_ret == VLC_SUCCESS && p_test->i_capabilities & SD_CAP_SEARCH )
371 item->setData( 0, CAP_SEARCH_ROLE, true );
372 }
373 }
374
375 curItem = item;
376
377 /* */
378 playlist_Lock( THEPL );
379 playlist_item_t *pl_item = NULL;
380
381 /* Special case for podcast */
382 // FIXME: simplify
383 if( i_type == SD_TYPE )
384 {
385 /* Find the right item for the SD */
386 pl_item = playlist_ChildSearchName( THEPL->p_root,
387 qtu( item->data(0, LONGNAME_ROLE ).toString() ) );
388
389 /* Podcasts */
390 if( item->data( 0, SPECIAL_ROLE ).toInt() == IS_PODCAST )
391 {
392 if( pl_item && !sd_loaded )
393 {
394 podcastsParentId = pl_item->i_id;
395 for( int i=0; i < pl_item->i_children; i++ )
396 addPodcastItem( pl_item->pp_children[i] );
397 }
398 pl_item = NULL; //to prevent activating it
399 }
400 }
401 else
402 pl_item = item->data( 0, PL_ITEM_ROLE ).value<playlist_item_t*>();
403
404 playlist_Unlock( THEPL );
405
406 /* */
407 if( pl_item )
408 {
409 emit categoryActivated( pl_item, false );
410 int i_cat = item->data( 0, SD_CATEGORY_ROLE ).toInt();
411 emit SDCategorySelected( i_cat == SD_CAT_INTERNET
412 || i_cat == SD_CAT_LAN );
413 }
414}

432PLSelItem *PLSelector::addPodcastItem( playlist_item_t *p_item )
433{
434 vlc_gc_incref( p_item->p_input );
435
436 char *psz_name = input_item_GetName( p_item->p_input );
437 PLSelItem *item = addItem( PL_ITEM_TYPE, psz_name, false, false, podcastsParent );
438 free( psz_name );
439
440 item->addAction( RM_ACTION, qtr( "Remove this podcast subscription" ) );
441 item->treeItem()->setData( 0, PL_ITEM_ROLE, QVariant::fromValue( p_item ) );
442 item->treeItem()->setData( 0, PL_ITEM_ID_ROLE, QVariant(p_item->i_id) );
443 item->treeItem()->setData( 0, IN_ITEM_ROLE, QVariant::fromValue( p_item->p_input ) );
444 CONNECT( item, action( PLSelItem* ), this, podcastRemove( PLSelItem* ) );
445 return item;
446}

455bool PLSelector::dropMimeData ( QTreeWidgetItem * parent, int,
456 const QMimeData * data, Qt::DropAction )
457{
458 if( !parent ) return false;
459
460 QVariant type = parent->data( 0, TYPE_ROLE );
461 if( type == QVariant() ) return false;
462
463 int i_truth = parent->data( 0, SPECIAL_ROLE ).toInt();
464 if( i_truth != IS_PL && i_truth != IS_ML ) return false;
465
466 bool to_pl = ( i_truth == IS_PL );
467
468 const PlMimeData *plMimeData = qobject_cast<const PlMimeData*>( data );
469 if( !plMimeData ) return false;
470
471 QList<input_item_t*> inputItems = plMimeData->inputItems();
472
473 playlist_Lock( THEPL );
474
475 foreach( input_item_t *p_input, inputItems )
476 {
477 playlist_item_t *p_item = playlist_ItemGetByInput( THEPL, p_input );
478 if( !p_item ) continue;
479
480 playlist_NodeAddCopy( THEPL, p_item,
481 to_pl ? THEPL->p_playing : THEPL->p_media_library,
482 PLAYLIST_END );
483 }
484
485 playlist_Unlock( THEPL );
486
487 return true;
488}

496void PLSelector::plItemAdded( int item, int parent )
497{
498 updateTotalDuration(playlistItem, "Playlist");
499 if( parent != podcastsParentId || podcastsParent == NULL ) return;
500
501 playlist_Lock( THEPL );
502
503 playlist_item_t *p_item = playlist_ItemGetById( THEPL, item );
504 if( !p_item ) {
505 playlist_Unlock( THEPL );
506 return;
507 }
508
509 int c = podcastsParent->childCount();
510 for( int i = 0; i < c; i++ )
511 {
512 QTreeWidgetItem *podItem = podcastsParent->child(i);
513 if( podItem->data( 0, PL_ITEM_ID_ROLE ).toInt() == item )
514 {
515 //msg_Dbg( p_intf, "Podcast already in: (%d) %s", item, p_item->p_input->psz_uri);
516 playlist_Unlock( THEPL );
517 return;
518 }
519 }
520
521 //msg_Dbg( p_intf, "Adding podcast: (%d) %s", item, p_item->p_input->psz_uri );
522 addPodcastItem( p_item );
523
524 playlist_Unlock( THEPL );
525
526 podcastsParent->setExpanded( true );
527}

529void PLSelector::plItemRemoved( int id )
530{
531 updateTotalDuration(playlistItem, "Playlist");
532 if( !podcastsParent ) return;
533
534 int c = podcastsParent->childCount();
535 for( int i = 0; i < c; i++ )
536 {
537 QTreeWidgetItem *item = podcastsParent->child(i);
538 if( item->data( 0, PL_ITEM_ID_ROLE ).toInt() == id )
539 {
540 input_item_t *p_input = item->data( 0, IN_ITEM_ROLE ).value<input_item_t*>();
541 //msg_Dbg( p_intf, "Removing podcast: (%d) %s", id, p_input->psz_uri );
542 vlc_gc_decref( p_input );
543 delete item;
544 return;
545 }
546 }

572void PLSelector::podcastAdd( PLSelItem * )
573{
574 assert( podcastsParent );
575
576 bool ok;
577 QString url = QInputDialog::getText( this, qtr( "Subscribe" ),
578 qtr( "Enter URL of the podcast to subscribe to:" ),
579 QLineEdit::Normal, QString(), &ok );
580 if( !ok || url.isEmpty() ) return;
581
582 setSource( podcastsParent ); //to load the SD in case it's not loaded
583
584 QString request("ADD:");
585 request += url.trimmed();
586 var_SetString( THEPL, "podcast-request", qtu( request ) );
587}

589void PLSelector::podcastRemove( PLSelItem* item )
590{
591 QString question ( qtr( "Do you really want to unsubscribe from %1?" ) );
592 question = question.arg( item->text() );
593 QMessageBox::StandardButton res =
594 QMessageBox::question( this, qtr( "Unsubscribe" ), question,
595 QMessageBox::Yes | QMessageBox::No,
596 QMessageBox::No );
597 if( res == QMessageBox::No ) return;
598
599 input_item_t *input = item->treeItem()->data( 0, IN_ITEM_ROLE ).value<input_item_t*>();
600 if( !input ) return;
601
602 QString request("RM:");
603 char *psz_uri = input_item_GetURI( input );
604 request += qfu( psz_uri );
605 var_SetString( THEPL, "podcast-request", qtu( request ) );
606 free( psz_uri );
607}
