50/*
51 Playlist item is just a wrapper, an abstraction of the playlist_item
52 in order to be managed by PLModel
53
54 PLItem have a parent, and id and a input Id
55*/
56
57void PLItem::init( playlist_item_t *_playlist_item, PLItem *parent )
58{
59 parentItem = parent; /* Can be NULL, but only for the rootItem */
60 i_playlist_id = _playlist_item->i_id; /* Playlist item specific id */
61 p_input = _playlist_item->p_input;
62 vlc_gc_incref( p_input );
63}
64
65/*
66 Constructors
67 Call the above function init
68 */
69PLItem::PLItem( playlist_item_t *p_item, PLItem *parent )
70{
71 init( p_item, parent );
72}
73
74PLItem::PLItem( playlist_item_t * p_item )
75{
76 init( p_item, NULL );
77}
78
79PLItem::~PLItem()
80{
81 vlc_gc_decref( p_input );
82 qDeleteAll( children );
83 children.clear();
84}

86int PLItem::id( int type )
87{
88 switch( type )
89 {
90 case INPUTITEM_ID:
91 return inputItem()->i_id;
92 case PLAYLIST_ID:
93 return i_playlist_id;
94 default:
95 case MLMEDIA_ID:
96 assert( 0 );
97 return -1;
98 }
99}

101void PLItem::takeChildAt( int index )
102{
103 AbstractPLItem *child = children[index];
104 child->parentItem = NULL;
105 children.removeAt( index );
106}

108/* This function is used to get one's parent's row number in the model */
109int PLItem::row()
110{
111 if( parentItem )
112 return parentItem->indexOf( this );
113 return 0;
114}
115
116bool PLItem::operator< ( AbstractPLItem& other )
117{
118 AbstractPLItem *item1 = this;
119 while( item1->parentItem )
120 {
121 AbstractPLItem *item2 = &other;
122 while( item2->parentItem )
123 {
124 if( item1 == item2->parentItem ) return true;
125 if( item2 == item1->parentItem ) return false;
126 if( item1->parentItem == item2->parentItem )
127 return item1->parentItem->indexOf( item1 ) <
128 item1->parentItem->indexOf( item2 );
129 item2 = item2->parentItem;
130 }
131 item1 = item1->parentItem;
132 }
133 return false;
134}

136QUrl PLItem::getURI() const
137{
138 QString uri;
139 vlc_mutex_lock( &p_input->lock );
140 uri = QString( p_input->psz_uri );
141 vlc_mutex_unlock( &p_input->lock );
142 return QUrl( uri );
143}

145QString PLItem::getTitle() const
146{
147 QString title;
148 char *fb_name = input_item_GetTitle( p_input );
149 if( EMPTY_STR( fb_name ) )
150 {
151 free( fb_name );
152 fb_name = input_item_GetName( p_input );
153 }
154 title = qfu(fb_name);
155 free(fb_name);
156 return title;
157}
