50class PLModel : public VLCModel
51{
52 Q_OBJECT
53
54public:
55 PLModel( playlist_t *, intf_thread_t *,
56 playlist_item_t *, QObject *parent = 0 );
57 virtual ~PLModel();
58
59 /* Qt main PLModel */
60 static PLModel* getPLModel( intf_thread_t *p_intf )
61 {
62 if(!p_intf->p_sys->pl_model )
63 {
64 playlist_Lock( THEPL );
65 playlist_item_t *p_root = THEPL->p_playing;
66 playlist_Unlock( THEPL );
67 p_intf->p_sys->pl_model = new PLModel( THEPL, p_intf, p_root, NULL );
68 }
69
70 return p_intf->p_sys->pl_model;
71 }
72
73 /*** QAbstractItemModel subclassing ***/
74
75 /* Data structure */
76 virtual QVariant data( const QModelIndex &index, const int role ) const;
77 virtual int rowCount( const QModelIndex &parent = QModelIndex() ) const;
78 virtual Qt::ItemFlags flags( const QModelIndex &index ) const;
79 virtual QModelIndex index( const int r, const int c, const QModelIndex &parent ) const;
80 virtual QModelIndex parent( const QModelIndex &index ) const;
81
82 /* Drag and Drop */
83 virtual Qt::DropActions supportedDropActions() const;
84 virtual QMimeData* mimeData( const QModelIndexList &indexes ) const;
85 virtual bool dropMimeData( const QMimeData *data, Qt::DropAction action,
86 int row, int column, const QModelIndex &target );
87 virtual QStringList mimeTypes() const;
88
89 /* Sort */
90 virtual void sort( const int column, Qt::SortOrder order = Qt::AscendingOrder );
91
92 /*** VLCModelSubInterface subclassing ***/
93 virtual void rebuild( playlist_item_t * p = NULL );
94 virtual void doDelete( QModelIndexList selected );
95 virtual void createNode( QModelIndex index, QString name );
96 virtual void renameNode( QModelIndex index, QString name );
97 virtual void removeAll();
98
99 /* Lookups */
100 virtual QModelIndex rootIndex() const;
101 virtual void filter( const QString& search_text, const QModelIndex & root, bool b_recursive );
102 virtual QModelIndex currentIndex() const;
103 virtual QModelIndex indexByPLID( const int i_plid, const int c ) const;
104 virtual QModelIndex indexByInputItemID( const int i_inputitem_id, const int c ) const;
105 virtual bool isTree() const;
106 virtual bool canEdit() const;
107 virtual bool action( QAction *action, const QModelIndexList &indexes );
108 virtual bool isSupportedAction( actions action, const QModelIndex & ) const;
109
110protected:
111 /* VLCModel subclassing */
112 bool isParent( const QModelIndex &index, const QModelIndex &current) const;
113 bool isLeaf( const QModelIndex &index ) const;
114 PLItem *getItem( const QModelIndex & index ) const;
115
116private:
117 /* General */
118 PLItem *rootItem;
119
120 playlist_t *p_playlist;
121
122 /* Custom model private methods */
123 /* Lookups */
124 QModelIndex index( PLItem *, const int c ) const;
125
126 /* Shallow actions (do not affect core playlist) */
127 void updateTreeItem( PLItem * );
128 void removeItem ( PLItem * );
129 void recurseDelete( QList<AbstractPLItem*> children, QModelIndexList *fullList );
130 void takeItem( PLItem * ); //will not delete item
131 void insertChildren( PLItem *node, QList<PLItem*>& items, int i_pos );
132 /* ...of which the following will not update the views */
133 void updateChildren( PLItem * );
134 void updateChildren( playlist_item_t *, PLItem * );
135
136 /* Deep actions (affect core playlist) */
137 void dropAppendCopy( const PlMimeData * data, PLItem *target, int pos );
138 void dropMove( const PlMimeData * data, PLItem *target, int new_pos );

140 /* */
141 void sort( QModelIndex caller, QModelIndex rootIndex, const int column, Qt::SortOrder order );
142
143 /* Lookups */
144 PLItem *findByPLId( PLItem *, int i_plitemid ) const;
145 PLItem *findByInputId( PLItem *, int i_input_itemid ) const;

162 void processItemAppend( int i_pl_itemid, int i_pl_itemidparent );
163 void activateItem( playlist_item_t *p_item );
164 void activateItem( const QModelIndex &index );
