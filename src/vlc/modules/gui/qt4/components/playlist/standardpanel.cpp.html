105 CONNECT( spinnerAnimation, pixmapReady( const QPixmap & ), this, updateViewport() );

148{

151 if ( !index.isValid() )

153 currentView->clearSelection();
154 }
155 else if ( ! currentView->selectionModel()->selectedIndexes().contains( index ) )

199 VLCModelSubInterface::ACTION_EXPLORE );
200
201 QIcon addIcon( ":/buttons/playlist/playlist_add" );
202
203 ADD_MENU_ENTRY( addIcon, qtr(I_POP_NEWFOLDER),
204 VLCModelSubInterface::ACTION_CREATENODE )

206 ADD_MENU_ENTRY( QIcon(), qtr(I_POP_RENAMEFOLDER),
207 VLCModelSubInterface::ACTION_RENAMENODE )

240 if ( model->isSupportedAction( VLCModelSubInterface::ACTION_SORT, index ) )

245 sortingColumns << COLUMN_TITLE << COLUMN_ARTIST << COLUMN_ALBUM << COLUMN_TRACK_NUMBER << COLUMN_URI;
246 container.action = VLCModelSubInterface::ACTION_SORT;

270 /* Display and forward the result */
271 if( !menu.isEmpty() )
272 {
273 menu.exec( point ); return true;
274 }

280void StandardPLPanel::popupAction( QAction *action )
281{
282 VLCModel *model = qobject_cast<VLCModel *>(currentView->model());
283 VLCModelSubInterface::actionsContainerType a =
284 action->data().value<VLCModelSubInterface::actionsContainerType>();
285 QModelIndexList list = currentView->selectionModel()->selectedRows();
286 QModelIndex index = popupIndex( currentView );
287 char *path = NULL;
288 OpenDialog *dialog;
289 QString temp;
290 QStringList uris;
291 bool ok;
292
293 /* first try to complete actions requiring missing parameters thru UI dialogs */
294 switch( a.action )
295 {
296 case VLCModelSubInterface::ACTION_INFO:
297 /* locally handled only */
298 if( index.isValid() )
299 {
300 input_item_t* p_input = model->getInputItem( index );
301 MediaInfoDialog *mid = new MediaInfoDialog( p_intf, p_input );
302 mid->setParent( PlaylistDialog::getInstance( p_intf ),
303 Qt::Dialog );
304 mid->show();
305 }
306 break;
307
308 case VLCModelSubInterface::ACTION_EXPLORE:
309 /* locally handled only */
310 temp = model->getURI( index );
311 if( ! temp.isEmpty() ) path = make_path( temp.toLatin1().constData() );
312 if( path == NULL ) return;
313 QDesktopServices::openUrl(
314 QUrl::fromLocalFile( QFileInfo( qfu( path ) ).absolutePath() ) );
315 free( path );
316 break;
317
318 case VLCModelSubInterface::ACTION_STREAM:
319 /* locally handled only */
320 temp = model->getURI( index );
321 if ( ! temp.isEmpty() )
322 THEDP->streamingDialog( NULL, temp, false );
323 break;
324
325 case VLCModelSubInterface::ACTION_SAVE:
326 /* locally handled only */
327 temp = model->getURI( index );
328 if ( ! temp.isEmpty() )
329 THEDP->streamingDialog( NULL, temp );
330 break;
331
332 case VLCModelSubInterface::ACTION_CREATENODE:
333 temp = QInputDialog::getText( PlaylistDialog::getInstance( p_intf ),
334 qtr( I_NEW_DIR ), qtr( I_NEW_DIR_NAME ),
335 QLineEdit::Normal, QString(), &ok);
336 if ( !ok ) return;
337 model->createNode( index, temp );
338 break;
339
340 case VLCModelSubInterface::ACTION_RENAMENODE:
341 temp = QInputDialog::getText( PlaylistDialog::getInstance( p_intf ),
342 qtr( I_RENAME_DIR ), qtr( I_RENAME_DIR_NAME ),
343 QLineEdit::Normal, model->getTitle( index ), &ok);
344 if ( !ok ) return;
345 model->renameNode( index, temp );
346 break;
347
348 case VLCModelSubInterface::ACTION_ENQUEUEFILE:
349 uris = THEDP->showSimpleOpen();
350 if ( uris.isEmpty() ) return;
351 uris.sort();
352 foreach( const QString &file, uris )
353 a.uris << qtu( toURI( toNativeSeparators( file ) ) );
354 action->setData( QVariant::fromValue( a ) );
355 if ( model->action( action, list ) )
356 foreach( const QString &file, a.uris )
357 RecentsMRL::getInstance( p_intf )->addRecent( file );
358 break;
359
360 case VLCModelSubInterface::ACTION_ENQUEUEDIR:
361 temp = DialogsProvider::getDirectoryDialog( p_intf );
362 if ( temp.isEmpty() ) return;
363 a.uris << temp;
364 action->setData( QVariant::fromValue( a ) );
365 model->action( action, list );
366 break;
367
368 case VLCModelSubInterface::ACTION_ENQUEUEGENERIC:
369 dialog = OpenDialog::getInstance( this, p_intf, false, SELECT, true, true );
370 dialog->showTab( OPEN_FILE_TAB );
371 dialog->exec(); /* make it modal */
372 a.uris = dialog->getMRLs( false );
373 a.options = dialog->getOptions();
374 if ( a.uris.isEmpty() ) return;
375 action->setData( QVariant::fromValue( a ) );
376 if ( model->action( action, list ) )
377 foreach( const QString &file, a.uris )
378 RecentsMRL::getInstance( p_intf )->addRecent( file );
379 break;
380
381 case VLCModelSubInterface::ACTION_SAVETOPLAYLIST:
382 THEDP->savePlayingToPlaylist();
383 break;
384 default:
385 model->action( action, list );
386 }
387}

389QMenu* StandardPLPanel::viewSelectionMenu( StandardPLPanel *panel )
390{
391 QMenu *viewMenu = new QMenu( qtr( "Playlist View Mode" ) );
392 QSignalMapper *viewSelectionMapper = new QSignalMapper( viewMenu );
393 CONNECT( viewSelectionMapper, mapped( int ), panel, showView( int ) );
394
395 QActionGroup *viewGroup = new QActionGroup( viewMenu );
396# define MAX_VIEW StandardPLPanel::VIEW_COUNT
397 for( int i = 0; i < MAX_VIEW; i++ )
398 {
399 QAction *action = viewMenu->addAction( viewNames[i] );
400 action->setCheckable( true );
401 viewGroup->addAction( action );
402 viewSelectionMapper->setMapping( action, i );
403 CONNECT( action, triggered(), viewSelectionMapper, map() );
404 if( panel->currentViewIndex() == i )
405 action->setChecked( true );
406 }
407 return viewMenu;
408}

419void StandardPLPanel::popupSelectColumn( QPoint )
420{
421 QMenu menu;
422 assert( treeView );
423
424 /* We do not offer the option to hide index 0 column, or
425 * QTreeView will behave weird */

427 {
428 QAction* option = menu.addAction( qfu( psz_column_title( i ) ) );
429 option->setCheckable( true );
430 option->setChecked( !treeView->isColumnHidden( j ) );
431 selectColumnsSigMapper->setMapping( option, j );
432 CONNECT( option, triggered(), selectColumnsSigMapper, map() );
433 }
434 menu.exec( QCursor::pos() );
435}

442/* Search in the playlist */
443void StandardPLPanel::search( const QString& searchText )
444{
445 int type;
446 QString name;
447 bool can_search;
448 p_selector->getCurrentItemInfos( &type, &can_search, &name );
449
450 if( type != SD_TYPE || !can_search )
451 {
452 bool flat = ( currentView == iconView ||
453 currentView == listView ||
454 currentView == picFlowView );
455 model->filter( searchText,
456 flat ? currentView->rootIndex() : QModelIndex(),
457 !flat );
458 }
459}

484void StandardPLPanel::browseInto( const QModelIndex &index )
485{
486 if( currentView == iconView || currentView == listView || currentView == picFlowView )
487 {
488
489 currentView->setRootIndex( index );
490
491 /* When going toward root in LocationBar, scroll to the item
492 that was previously as root */
493 QModelIndex newIndex = model->indexByPLID(currentRootIndexPLId,0);
494 while( newIndex.isValid() && (newIndex.parent() != index) )
495 newIndex = newIndex.parent();
496 if( newIndex.isValid() )
497 currentView->scrollTo( newIndex );
498
499 /* Store new rootindexid*/
500 currentRootIndexPLId = model->itemId( index, PLAYLIST_ID );
501
502 model->ensureArtRequested( index );
503 }
504
505 emit viewChanged( index );
506}

515void StandardPLPanel::wheelEvent( QWheelEvent *e )
516{
517 if( e->modifiers() & Qt::ControlModifier ) {
518 int numSteps = e->delta() / 8 / 15;
519 if( numSteps > 0)
520 increaseZoom();
521 else if( numSteps < 0)
522 decreaseZoom();
523 }
524 // Accept this event in order to prevent unwanted volume up/down changes
525 e->accept();
526}

528bool StandardPLPanel::eventFilter ( QObject *obj, QEvent * event )
529{
530 if (event->type() == QEvent::KeyPress)
531 {
532 QKeyEvent *keyEvent = static_cast<QKeyEvent*>(event);
533 if( keyEvent->key() == Qt::Key_Delete ||
534 keyEvent->key() == Qt::Key_Backspace )
535 {
536 deleteSelection();
537 return true;
538 }
539 }
540 else if ( event->type() == QEvent::Paint )
541 {/* Warn! Don't filter events from anything else than views ! */
542 if ( model->rowCount() == 0 && p_selector->getCurrentItemCategory() == PL_ITEM_TYPE )
543 {
544 QWidget *viewport = qobject_cast<QWidget *>( obj );
545 QStylePainter painter( viewport );
546 QPixmap dropzone(":/dropzone");
547 QRect rect = viewport->geometry();
548 QSize size = rect.size() / 2 - dropzone.size() / 2;
549 rect.adjust( 0, size.height(), 0 , 0 );
550 painter.drawItemPixmap( rect, Qt::AlignHCenter, dropzone );
551 /* now select the zone just below the drop zone and let Qt center
552 the text by itself */
553 rect.adjust( 0, dropzone.size().height() + 10, 0, 0 );
554 rect.setRight( viewport->geometry().width() );
555 rect.setLeft( 0 );
556 painter.drawItemText( rect,
557 Qt::AlignHCenter,
558 palette(),
559 true,
560 qtr("Playlist is currently empty.\n"
561 "Drop a file here or select a "
562 "media source from the left."),
563 QPalette::Text );
564 }
565 else if ( spinnerAnimation->state() == PixmapAnimator::Running )
566 {
567 if ( currentView->model()->rowCount() )
568 spinnerAnimation->stop(); /* Trick until SD emits events */
569 else
570 {
571 QWidget *viewport = qobject_cast<QWidget *>( obj );
572 QStylePainter painter( viewport );
573 QPixmap *spinner = spinnerAnimation->getPixmap();
574 QPoint point = viewport->geometry().center();
575 point -= QPoint( spinner->size().width() / 2, spinner->size().height() / 2 );
576 painter.drawPixmap( point, *spinner );
577 }
578 }
579 }
580 return false;
581}

666void StandardPLPanel::showView( int i_view )
667{
668 bool b_treeViewCreated = false;
669
670 switch( i_view )
671 {
672 case ICON_VIEW:
673 {
674 if( iconView == NULL )
675 createIconView();
676 currentView = iconView;
677 break;
678 }
679 case LIST_VIEW:
680 {
681 if( listView == NULL )
682 createListView();
683 currentView = listView;
684 break;
685 }
686 case PICTUREFLOW_VIEW:
687 {
688 if( picFlowView == NULL )
689 createCoverView();
690 currentView = picFlowView;
691 break;
692 }
693 default:
694 case TREE_VIEW:
695 {
696 if( treeView == NULL )
697 {
698 createTreeView();
699 b_treeViewCreated = true;
700 }
701 currentView = treeView;
702 break;
703 }
704 }
705
706 currentView->setModel( model );
707
708 /* Restoring the header Columns must come after changeModel */
709 if( b_treeViewCreated )
710 {
711 assert( treeView );
712 if( getSettings()->contains( "Playlist/headerStateV2" ) )
713 {
714 treeView->header()->restoreState(getSettings()
715 ->value( "Playlist/headerStateV2" ).toByteArray() );
716 /* if there is allready stuff in playlist, we don't sort it and we reset
717 sorting */
718 if( model->rowCount() )
719 {
720 treeView->header()->setSortIndicator( -1 , Qt::AscendingOrder );
721 }
722 }
723 else
724 {
725 for( int m = 1, c = 0; m != COLUMN_END; m <<= 1, c++ )
726 {
727 treeView->setColumnHidden( c, !( m & COLUMN_DEFAULT ) );
728 if( m == COLUMN_TITLE ) treeView->header()->resizeSection( c, 200 );
729 else if( m == COLUMN_DURATION ) treeView->header()->resizeSection( c, 80 );
730 }
731 }
732 }
733
734 updateZoom( i_zoom );
735 viewStack->setCurrentWidget( currentView );
736 browseInto();
737 gotoPlayingItem();
738}

785void StandardPLPanel::activate( const QModelIndex &index )
786{
787 if( currentView->model() == model )
788 {
789 /* If we are not a leaf node */
790 if( !index.data( VLCModelSubInterface::IsLeafNodeRole ).toBool() )
791 {
792 if( currentView != treeView )
793 browseInto( index );
794 }
795 else
796 {
797 playlist_Lock( THEPL );
798 playlist_item_t *p_item = playlist_ItemGetById( THEPL, model->itemId( index, PLAYLIST_ID ) );
799 if ( p_item )
800 {
801 p_item->i_flags |= PLAYLIST_SUBITEM_STOP_FLAG;
802 lastActivatedPLItemId = p_item->i_id;
803 }
804 playlist_Unlock( THEPL );
805 if ( p_item && index.isValid() )
806 model->activateItem( index );
807 }
808 }
809}

811void StandardPLPanel::browseInto( int i_pl_item_id )
812{
813 if( i_pl_item_id != lastActivatedPLItemId ) return;
814
815 QModelIndex index = model->indexByPLID( i_pl_item_id, 0 );
816
817 if( currentView == treeView )
818 treeView->setExpanded( index, true );
819 else
820 browseInto( index );
821
822 lastActivatedPLItemId = -1;
823}
