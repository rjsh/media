83{

112bool modelIndexLessThen( const QModelIndex &i1, const QModelIndex &i2 )
113{
114 if( !i1.isValid() || !i2.isValid() ) return false;
115 PLItem *item1 = static_cast<PLItem*>( i1.internalPointer() );
116 PLItem *item2 = static_cast<PLItem*>( i2.internalPointer() );
117 if( item1->hasSameParent( item2 ) ) return i1.row() < i2.row();
118 else return *item1 < *item2;
119}

121QMimeData *PLModel::mimeData( const QModelIndexList &indexes ) const
122{
123 PlMimeData *plMimeData = new PlMimeData();
124 QModelIndexList list;
125
126 foreach( const QModelIndex &index, indexes ) {
127 if( index.isValid() && index.column() == 0 )
128 list.append(index);
129 }
130
131 qSort(list.begin(), list.end(), modelIndexLessThen);
132
133 AbstractPLItem *item = NULL;
134 foreach( const QModelIndex &index, list ) {
135 if( item )
136 {
137 AbstractPLItem *testee = getItem( index );
138 while( testee->parent() )
139 {
140 if( testee->parent() == item ||
141 testee->parent() == item->parent() ) break;
142 testee = testee->parent();
143 }
144 if( testee->parent() == item ) continue;
145 item = getItem( index );
146 }
147 else
148 item = getItem( index );
149
150 plMimeData->appendItem( static_cast<PLItem*>(item)->inputItem() );
151 }
152
153 return plMimeData;
154}

156/* Drop operation */
157bool PLModel::dropMimeData( const QMimeData *data, Qt::DropAction action,
158 int row, int, const QModelIndex &parent )
159{
160 bool copy = action == Qt::CopyAction;
161 if( !copy && action != Qt::MoveAction )
162 return true;
163
164 const PlMimeData *plMimeData = qobject_cast<const PlMimeData*>( data );
165 if( plMimeData )
166 {
167 if( copy )
168 dropAppendCopy( plMimeData, getItem( parent ), row );
169 else
170 dropMove( plMimeData, getItem( parent ), row );
171 }
172 return true;
173}

175void PLModel::dropAppendCopy( const PlMimeData *plMimeData, PLItem *target, int pos )
176{
177 PL_LOCK;
178
179 playlist_item_t *p_parent =
180 playlist_ItemGetByInput( p_playlist, target->inputItem() );
181 if( !p_parent ) return;
182
183 if( pos == -1 ) pos = PLAYLIST_END;
184
185 QList<input_item_t*> inputItems = plMimeData->inputItems();
186
187 foreach( input_item_t* p_input, inputItems )
188 {
189 playlist_item_t *p_item = playlist_ItemGetByInput( p_playlist, p_input );
190 if( !p_item ) continue;
191 pos = playlist_NodeAddCopy( p_playlist, p_item, p_parent, pos );
192 }
193
194 PL_UNLOCK;
195}

197void PLModel::dropMove( const PlMimeData * plMimeData, PLItem *target, int row )
198{
199 QList<input_item_t*> inputItems = plMimeData->inputItems();
200 QList<PLItem*> model_items;
201 playlist_item_t **pp_items;
202 pp_items = (playlist_item_t **)
203 calloc( inputItems.count(), sizeof( playlist_item_t* ) );
204 if ( !pp_items ) return;
205
206 PL_LOCK;
207
208 playlist_item_t *p_parent =
209 playlist_ItemGetByInput( p_playlist, target->inputItem() );
210
211 if( !p_parent || row > p_parent->i_children )
212 {
213 PL_UNLOCK;
214 free( pp_items );
215 return;
216 }
217
218 int new_pos = row == -1 ? p_parent->i_children : row;
219 int model_pos = new_pos;
220 int i = 0;
221
222 foreach( input_item_t *p_input, inputItems )
223 {
224 playlist_item_t *p_item = playlist_ItemGetByInput( p_playlist, p_input );
225 if( !p_item ) continue;
226
227 PLItem *item = findByInputId( rootItem, p_input->i_id );
228 if( !item ) continue;
229
230 /* Better not try to move a node into itself.
231 Abort the whole operation in that case,
232 because it is ambiguous. */
233 AbstractPLItem *climber = target;
234 while( climber )
235 {
236 if( climber == item )
237 {
238 PL_UNLOCK;
239 free( pp_items );
240 return;
241 }
242 climber = climber->parent();
243 }
244
245 if( item->parent() == target &&
246 target->children.indexOf( item ) < new_pos )
247 model_pos--;
248
249 model_items.append( item );
250 pp_items[i] = p_item;
251 i++;
252 }
253
254 if( model_items.isEmpty() )
255 {
256 PL_UNLOCK;
257 free( pp_items );
258 return;
259 }
260
261 playlist_TreeMoveMany( p_playlist, i, pp_items, p_parent, new_pos );
262
263 PL_UNLOCK;
264
265 foreach( PLItem *item, model_items )
266 takeItem( item );
267
268 insertChildren( target, model_items, model_pos );
269 free( pp_items );
270}

299/****************** Base model mandatory implementations *****************/
300QVariant PLModel::data( const QModelIndex &index, const int role ) const
301{
302 if( !index.isValid() ) return QVariant();
303 PLItem *item = getItem( index );
304 if( role == Qt::DisplayRole )
305 {
306 int metadata = columnToMeta( index.column() );
307 if( metadata == COLUMN_END ) return QVariant();
308
309 QString returninfo;
310 if( metadata == COLUMN_NUMBER )
311 returninfo = QString::number( index.row() + 1 );
312 else if( metadata == COLUMN_COVER )
313 {
314 QString artUrl;
315 artUrl = InputManager::decodeArtURL( item->inputItem() );
316 if( artUrl.isEmpty() )
317 {
318 for( int i = 0; i < item->childCount(); i++ )
319 {
320 artUrl = InputManager::decodeArtURL( item->child( i )->inputItem() );
321 if( !artUrl.isEmpty() )
322 break;
323 }
324 }
325 return artUrl;
326 }
327 else
328 {
329 char *psz = psz_column_meta( item->inputItem(), metadata );
330 returninfo = qfu( psz );
331 free( psz );
332 }
333 return QVariant( returninfo );
334 }
335 else if( role == Qt::DecorationRole )
336 {
337 switch( columnToMeta(index.column()) )
338 {
339 case COLUMN_TITLE:
340 /* Used to segfault here because i_type wasn't always initialized */
341 return QVariant( icons[item->inputItem()->i_type] );
342 case COLUMN_COVER:
343 /* !warn: changes tree item line height. Otherwise, override
344 * delegate's sizehint */
345 return getArtPixmap( index, QSize(16,16) );
346 default:
347 return QVariant();
348 }
349 }
350 else if( role == Qt::FontRole )
351 {
352 return QVariant( QFont() );
353 }
354 else if( role == Qt::BackgroundRole && isCurrent( index ) )
355 {
356 return QVariant( QBrush( Qt::gray ) );
357 }
358 else if( role == IsCurrentRole )
359 {
360 return QVariant( isCurrent( index ) );
361 }
362 else if( role == IsLeafNodeRole )
363 {
364 return QVariant( isLeaf( index ) );
365 }
366 else if( role == IsCurrentsParentNodeRole )
367 {
368 return QVariant( isParent( index, currentIndex() ) );
369 }
370 return QVariant();
371}

373/* Seek from current index toward the top and see if index is one of parent nodes */
374bool PLModel::isParent( const QModelIndex &index, const QModelIndex &current ) const
375{
376 if( !index.isValid() )
377 return false;
378
379 if( index == current )
380 return true;
381
382 if( !current.isValid() || !current.parent().isValid() )
383 return false;
384
385 return isParent( index, current.parent() );
386}

460QModelIndex PLModel::parent( const QModelIndex &index ) const
461{
462 if( !index.isValid() ) return QModelIndex();
463
464 PLItem *childItem = getItem( index );
465 if( !childItem )
466 {
467 msg_Err( p_playlist, "Item not found" );
468 return QModelIndex();
469 }
470
471 PLItem *parentItem = static_cast<PLItem*>(childItem->parent());
472 if( !parentItem || parentItem == rootItem ) return QModelIndex();
473 if( !parentItem->parent() )
474 {
475 msg_Err( p_playlist, "No parent found, trying row 0. Please report this" );
476 return createIndex( 0, 0, parentItem );
477 }
478 return createIndex(parentItem->row(), 0, parentItem);
479}

499PLItem * PLModel::findInner( PLItem *root, int i_id, bool b_isinputid ) const
500{
501 if( !root ) return NULL;
502
503 if( !b_isinputid && root->id( PLAYLIST_ID ) == i_id )
504 return root;
505
506 else if( b_isinputid && root->id( INPUTITEM_ID ) == i_id )
507 return root;
508
509 QList<AbstractPLItem *>::iterator it = root->children.begin();
510 while ( it != root->children.end() )
511 {
512 PLItem *item = static_cast<PLItem *>(*it);
513 if( !b_isinputid && item->id( PLAYLIST_ID ) == i_id )
514 return item;
515
516 else if( b_isinputid && item->id( INPUTITEM_ID ) == i_id )
517 return item;
518
519 if( item->childCount() )
520 {
521 PLItem *childFound = findInner( item, i_id, b_isinputid );
522 if( childFound )
523 return childFound;
524 }
525 ++it;
526 }
527 return NULL;
528}

590void PLModel::processItemAppend( int i_pl_itemid, int i_pl_itemidparent )
591{
592 playlist_item_t *p_item = NULL;
593 PLItem *newItem = NULL;
594 int pos;
595
596 /* Find the Parent */
597 PLItem *nodeParentItem = findByPLId( rootItem, i_pl_itemidparent );
598 if( !nodeParentItem ) return;
599
600 /* Search for an already matching children */
601 foreach( AbstractPLItem *existing, nodeParentItem->children )
602 if( existing->id( PLAYLIST_ID ) == i_pl_itemid ) return;
603
604 /* Find the child */
605 PL_LOCK;
606 p_item = playlist_ItemGetById( p_playlist, i_pl_itemid );
607 if( !p_item || p_item->i_flags & PLAYLIST_DBL_FLAG )
608 {
609 PL_UNLOCK; return;
610 }
611
612 for( pos = p_item->p_parent->i_children - 1; pos >= 0; pos-- )
613 if( p_item->p_parent->pp_children[pos] == p_item ) break;
614
615 newItem = new PLItem( p_item, nodeParentItem );
616 PL_UNLOCK;
617
618 /* We insert the newItem (children) inside the parent */
619 beginInsertRows( index( nodeParentItem, 0 ), pos, pos );
620 nodeParentItem->insertChild( newItem, pos );
621 endInsertRows();
622 if ( newItem->inputItem() == THEMIM->currentInputItem() )
623 emit currentIndexChanged( index( newItem, 0 ) );
624
625 if( latestSearch.isEmpty() ) return;
626 filter( latestSearch, index( rootItem, 0), false /*FIXME*/ );
627}

629void PLModel::rebuild( playlist_item_t *p_root )
630{
631 beginResetModel();
632
633 PL_LOCK;
634 if( rootItem ) rootItem->clearChildren();
635 if( p_root ) // Can be NULL
636 {
637 if ( rootItem ) delete rootItem;
638 rootItem = new PLItem( p_root );
639 }
640 assert( rootItem );
641 /* Recreate from root */
642 updateChildren( rootItem );
643 PL_UNLOCK;
644
645 /* And signal the view */
646 endResetModel();
647 if( p_root ) emit rootIndexChanged();
648}

650void PLModel::takeItem( PLItem *item )
651{
652 assert( item );
653 PLItem *parent = static_cast<PLItem*>(item->parent());
654 assert( parent );
655 int i_index = parent->indexOf( item );
656
657 beginRemoveRows( index( parent, 0 ), i_index, i_index );
658 parent->takeChildAt( i_index );
659 endRemoveRows();
660}

662void PLModel::insertChildren( PLItem *node, QList<PLItem*>& items, int i_pos )
663{
664 assert( node );
665 int count = items.count();
666 if( !count ) return;
667 beginInsertRows( index( node, 0 ), i_pos, i_pos + count - 1 );
668 for( int i = 0; i < count; i++ )
669 {
670 node->children.insert( i_pos + i, items[i] );
671 items[i]->parentItem = node;
672 }
673 endInsertRows();
674}
675
676void PLModel::removeItem( PLItem *item )
677{
678 if( !item ) return;
679
680 if( item->parent() ) {
681 int i = item->parent()->indexOf( item );
682 beginRemoveRows( index( static_cast<PLItem*>(item->parent()), 0), i, i );
683 item->parent()->children.removeAt(i);
684 delete item;
685 endRemoveRows();
686 }
687 else delete item;
688
689 if(item == rootItem)
690 {
691 rootItem = NULL;
692 rebuild( p_playlist->p_playing );
693 }
694}

703/* This function must be entered WITH the playlist lock */
704void PLModel::updateChildren( playlist_item_t *p_node, PLItem *root )
705{

707 {
708 if( p_node->pp_children[i]->i_flags & PLAYLIST_DBL_FLAG ) continue;
709 PLItem *newItem = new PLItem( p_node->pp_children[i], root );
710 root->appendChild( newItem );
711 if( p_node->pp_children[i]->i_children != -1 )
712 updateChildren( p_node->pp_children[i], newItem );
713 }
714}

725/**
726 * Deletion, don't delete items childrens if item is going to be
727 * delete allready, so we remove childrens from selection-list.
728 */
729void PLModel::doDelete( QModelIndexList selected )
730{
731 if( !canEdit() ) return;
732
733 while( !selected.isEmpty() )
734 {
735 QModelIndex index = selected[0];
736 selected.removeAt( 0 );
737
738 if( index.column() != 0 ) continue;
739
740 PLItem *item = getItem( index );
741 if( item->childCount() )
742 recurseDelete( item->children, &selected );
743
744 PL_LOCK;
745 playlist_DeleteFromInput( p_playlist, item->inputItem(), pl_Locked );
746 PL_UNLOCK;
747
748 removeItem( item );
749 }
750}
751
752void PLModel::recurseDelete( QList<AbstractPLItem*> children, QModelIndexList *fullList )
753{
754 for( int i = children.count() - 1; i >= 0 ; i-- )
755 {
756 PLItem *item = static_cast<PLItem *>(children[i]);
757 if( item->childCount() )
758 recurseDelete( item->children, fullList );
759 fullList->removeAll( index( item, 0 ) );
760 }
761}

763/******* Volume III: Sorting and searching ********/
764void PLModel::sort( const int column, Qt::SortOrder order )
765{

769void PLModel::sort( QModelIndex caller, QModelIndex rootIndex, const int column, Qt::SortOrder order )
770{
771 msg_Dbg( p_intf, "Sorting by column %i, order %i", column, order );
772
773 int meta = columnToMeta( column );
774 if( meta == COLUMN_END ) return;
775
776 PLItem *item = ( rootIndex.isValid() ) ? getItem( rootIndex )
777 : rootItem;
778 if( !item ) return;
779
780 int i_root_id = item->id( PLAYLIST_ID );
781
782 QModelIndex qIndex = index( item, 0 );
783 int count = item->childCount();
784 if( count )
785 {
786 beginRemoveRows( qIndex, 0, count - 1 );
787 item->clearChildren();
788 endRemoveRows( );
789 }
790
791 PL_LOCK;
792 {
793 playlist_item_t *p_root = playlist_ItemGetById( p_playlist,
794 i_root_id );
795 if( p_root )
796 {
797 playlist_RecursiveNodeSort( p_playlist, p_root,
798 i_column_sorting( meta ),
799 order == Qt::AscendingOrder ?
800 ORDER_NORMAL : ORDER_REVERSE );
801 }
802 }
803
804 if( count )
805 {
806 beginInsertRows( qIndex, 0, count - 1 );
807 updateChildren( item );
808 endInsertRows( );
809 }
810 PL_UNLOCK;
811 /* if we have popup item, try to make sure that you keep that item visible */
812 if( caller.isValid() ) emit currentIndexChanged( caller );
813
814 else if( currentIndex().isValid() ) emit currentIndexChanged( currentIndex() );
815}

888bool PLModel::action( QAction *action, const QModelIndexList &indexes )
889{
890 QModelIndex index;
891 actionsContainerType a = action->data().value<actionsContainerType>();
892
893 switch ( a.action )
894 {
895
896 case ACTION_PLAY:
897 if ( !indexes.empty() && indexes.first().isValid() )
898 {
899 activateItem( indexes.first() );
900 return true;
901 }
902 break;
903
904 case ACTION_ADDTOPLAYLIST:
905 PL_LOCK;
906 foreach( const QModelIndex &currentIndex, indexes )
907 {
908 playlist_item_t *p_item = playlist_ItemGetById( THEPL, itemId( currentIndex, PLAYLIST_ID ) );
909 if( !p_item ) continue;
910
911 playlist_NodeAddCopy( THEPL, p_item,
912 THEPL->p_playing,
913 PLAYLIST_END );
914 }
915 PL_UNLOCK;
916 return true;
917
918 case ACTION_REMOVE:
919 doDelete( indexes );
920 return true;
921
922 case ACTION_SORT:
923 if ( indexes.empty() ) break;
924 index = indexes.first().parent();
925 if( !index.isValid() ) index = rootIndex();
926 sort( indexes.first(), index,
927 a.column > 0 ? a.column - 1 : -a.column - 1,
928 a.column > 0 ? Qt::AscendingOrder : Qt::DescendingOrder );
929 return true;
930
931 case ACTION_CLEAR:
932 removeAll();
933 return true;
934
935 case ACTION_ENQUEUEFILE:
936 foreach( const QString &uri, a.uris )
937 Open::openMRL( p_intf, uri.toLatin1().constData(),
938 false, getPLRootType() == ROOTTYPE_CURRENT_PLAYING );
939 return true;
940
941 case ACTION_ENQUEUEDIR:
942 if( a.uris.isEmpty() ) break;
943
944 Open::openMRL( p_intf, a.uris.first().toLatin1().constData(),
945 false, getPLRootType() == ROOTTYPE_CURRENT_PLAYING );
946
947 return true;
948
949 case ACTION_ENQUEUEGENERIC:
950 foreach( const QString &uri, a.uris )
951 {
952 QStringList options = a.options.split( " :" );
953 Open::openMRLwithOptions( p_intf, uri, &options, false );
954 }
955 return true;
956
957 default:
958 break;
959 }
960 return false;
961}

963bool PLModel::isSupportedAction( actions action, const QModelIndex &index ) const
964{
965 switch ( action )
966 {
967 case ACTION_ADDTOPLAYLIST:
968 /* Only if we are not already in Current Playing */
969 if ( getPLRootType() == ROOTTYPE_CURRENT_PLAYING ) return false;
970 if( index.isValid() && index != rootIndex() )
971 return ( itemId( index, PLAYLIST_ID ) != THEPL->p_playing->i_id );
972 case ACTION_SORT:
973 return rowCount();
974 case ACTION_PLAY:
975 case ACTION_STREAM:
976 case ACTION_SAVE:
977 case ACTION_INFO:
978 case ACTION_REMOVE:
979 return index.isValid() && index != rootIndex();
980 case ACTION_EXPLORE:
981 if( index.isValid() )
982 return getURI( index ).startsWith( "file://" );
983 case ACTION_CREATENODE:
984 return ( canEdit() && isTree() );
985 case ACTION_RENAMENODE:
986 return ( index != rootIndex() ) && !isLeaf( index );
987 break;
988 case ACTION_CLEAR:
989 return rowCount() && canEdit();
990 case ACTION_ENQUEUEFILE:
991 case ACTION_ENQUEUEDIR:
992 case ACTION_ENQUEUEGENERIC:
993 return canEdit();
994 case ACTION_SAVETOPLAYLIST:
995 return rowCount() > 0;
996 default:
997 return false;
998 }
999 return false;
1000}
