49PlaylistWidget::PlaylistWidget( intf_thread_t *_p_i, QWidget *_par )
50 : QWidget( _par ), p_intf ( _p_i )
51{
52
53 setContentsMargins( 0, 3, 0, 3 );
54
55 QGridLayout *layout = new QGridLayout( this );
56 layout->setMargin( 0 ); layout->setSpacing( 0 );
57
58 /*******************
59 * Left *
60 *******************/
61 /* We use a QSplitter for the left part */
62 leftSplitter = new QSplitter( Qt::Vertical, this );
63
64 /* Source Selector */
65 selector = new PLSelector( this, p_intf );
66 leftSplitter->addWidget( selector );
67
68 /* Create a Container for the Art Label
69 in order to have a beautiful resizing for the selector above it */
70 artContainer = new QStackedWidget;
71 artContainer->setMaximumHeight( 256 );
72
73 /* Art label */
74 CoverArtLabel *art = new CoverArtLabel( artContainer, p_intf );
75 art->setToolTip( qtr( "Double click to get media information" ) );
76 artContainer->addWidget( art );
77
78 CONNECT( THEMIM->getIM(), artChanged( QString ),
79 art, showArtUpdate( const QString& ) );
80 CONNECT( THEMIM->getIM(), artChanged( input_item_t * ),
81 art, showArtUpdate( input_item_t * ) );
82
83 leftSplitter->addWidget( artContainer );
84
85 /*******************
86 * Right *
87 *******************/
88 /* Initialisation of the playlist */
89 playlist_t * p_playlist = THEPL;
90 PL_LOCK;
91 playlist_item_t *p_root = p_playlist->p_playing;
92 PL_UNLOCK;
93
94 setMinimumWidth( 400 );
95
96 PLModel *model = PLModel::getPLModel( p_intf );
97
98 mainView = new StandardPLPanel( this, p_intf, p_root, selector, model );
99
100 /* Location Bar */
101 locationBar = new LocationBar( model );
102 locationBar->setSizePolicy( QSizePolicy::Ignored, QSizePolicy::Preferred );
103 layout->addWidget( locationBar, 0, 0, 1, 2 );
104 layout->setColumnStretch( 0, 5 );
105 CONNECT( locationBar, invoked( const QModelIndex & ),
106 mainView, browseInto( const QModelIndex & ) );
107
108 QHBoxLayout *topbarLayout = new QHBoxLayout();
109 layout->addLayout( topbarLayout, 0, 1 );
110 topbarLayout->setSpacing( 10 );
111
112 /* Button to switch views */
113 QToolButton *viewButton = new QToolButton( this );
114 viewButton->setIcon( style()->standardIcon( QStyle::SP_FileDialogDetailedView ) );
115 viewButton->setToolTip( qtr("Change playlistview") );
116 topbarLayout->addWidget( viewButton );
117
118 viewButton->setMenu( StandardPLPanel::viewSelectionMenu( mainView ));
119 CONNECT( viewButton, clicked(), mainView, cycleViews() );
120
121 /* Search */
122 searchEdit = new SearchLineEdit( this );
123 searchEdit->setMaximumWidth( 250 );
124 searchEdit->setMinimumWidth( 80 );
125 searchEdit->setToolTip( qtr("Search the playlist") );
126 topbarLayout->addWidget( searchEdit );
127 CONNECT( searchEdit, textChanged( const QString& ),
128 mainView, search( const QString& ) );
129 CONNECT( searchEdit, searchDelayedChanged( const QString& ),
130 mainView, searchDelayed( const QString & ) );
131
132 CONNECT( mainView, viewChanged( const QModelIndex& ),
133 this, changeView( const QModelIndex &) );
134
135 /* Connect the activation of the selector to a redefining of the PL */
136 DCONNECT( selector, categoryActivated( playlist_item_t *, bool ),
137 mainView, setRootItem( playlist_item_t *, bool ) );
138 mainView->setRootItem( p_root, false );
139 CONNECT( selector, SDCategorySelected(bool), mainView, setWaiting(bool) );
140
141 /* */
142 split = new PlaylistSplitter( this );
143
144 /* Add the two sides of the QSplitter */
145 split->addWidget( leftSplitter );
146 split->addWidget( mainView );
147
148 QList<int> sizeList;
149 sizeList << 180 << 420 ;
150 split->setSizes( sizeList );
151 split->setStretchFactor( 0, 0 );
152 split->setStretchFactor( 1, 3 );
153 split->setCollapsible( 1, false );
154 leftSplitter->setMaximumWidth( 250 );
155
156 /* In case we want to keep the splitter information */
157 // components shall never write there setting to a fixed location, may infer
158 // with other uses of the same component...
159 getSettings()->beginGroup("Playlist");
160 split->restoreState( getSettings()->value("splitterSizes").toByteArray());
161 leftSplitter->restoreState( getSettings()->value("leftSplitterGeometry").toByteArray() );
162 getSettings()->endGroup();
163
164 layout->addWidget( split, 1, 0, 1, -1 );
165
166 setAcceptDrops( true );
167 setWindowTitle( qtr( "Playlist" ) );
168 setWindowRole( "vlc-playlist" );
169 setWindowIcon( QApplication::windowIcon() );
170}
171
172PlaylistWidget::~PlaylistWidget()
173{
174 getSettings()->beginGroup("Playlist");
175 getSettings()->setValue( "splitterSizes", split->saveState() );
176 getSettings()->setValue( "leftSplitterGeometry", leftSplitter->saveState() );
177 getSettings()->endGroup();
178 msg_Dbg( p_intf, "Playlist Destroyed" );
179}

181void PlaylistWidget::dropEvent( QDropEvent *event )
182{
183 if( !( selector->getCurrentItemCategory() == IS_PL ||
184 selector->getCurrentItemCategory() == IS_ML ) ) return;
185
186 if( p_intf->p_sys->p_mi )
187 p_intf->p_sys->p_mi->dropEventPlay( event, false,
188 (selector->getCurrentItemCategory() == IS_PL) );
189}
190void PlaylistWidget::dragEnterEvent( QDragEnterEvent *event )
191{
192 event->acceptProposedAction();
193}

195void PlaylistWidget::closeEvent( QCloseEvent *event )
196{
197 if( THEDP->isDying() )
198 {
199 p_intf->p_sys->p_mi->playlistVisible = true;
200 event->accept();
201 }
202 else
203 {
204 p_intf->p_sys->p_mi->playlistVisible = false;
205 hide();
206 event->ignore();
207 }
208}

243void LocationBar::setIndex( const QModelIndex &index )
244{
245 qDeleteAll( buttons );
246 buttons.clear();
247 qDeleteAll( actions );
248 actions.clear();
249
250 QModelIndex i = index;
251 bool first = true;
252
253 while( true )
254 {
255 QString text = model->getTitle( i );
256
257 QAbstractButton *btn = new LocationButton( text, first, !first, this );
258 btn->setSizePolicy( QSizePolicy::Maximum, QSizePolicy::Fixed );
259 buttons.append( btn );
260
261 QAction *action = new QAction( text, this );
262 actions.append( action );
263 CONNECT( btn, clicked(), action, trigger() );
264
265 mapper->setMapping( action, model->itemId( i, PLAYLIST_ID ) );
266 CONNECT( action, triggered(), mapper, map() );
267
268 first = false;
269
270 if( i.isValid() ) i = i.parent();
271 else break;
272 }
273
274 QString prefix;
275 for( int a = actions.count() - 1; a >= 0 ; a-- )
276 {
277 actions[a]->setText( prefix + actions[a]->text() );
278 prefix += QString(" ");
279 }
280
281 if( isVisible() ) layOut( size() );
282}

295void LocationBar::layOut( const QSize& size )
296{
297 menuMore->clear();
298 widths.clear();
299
300 int count = buttons.count();
301 int totalWidth = 0;
302 for( int i = 0; i < count; i++ )
303 {
304 int w = buttons[i]->sizeHint().width();
305 widths.append( w );
306 totalWidth += w;
307 if( totalWidth > size.width() ) break;
308 }
309
310 int x = 0;
311 int shown = widths.count();
312
313 if( totalWidth > size.width() && count > 1 )
314 {
315 QSize sz = btnMore->sizeHint();
316 btnMore->setGeometry( 0, 0, sz.width(), size.height() );
317 btnMore->show();
318 x = sz.width();
319 totalWidth += x;
320 }
321 else
322 {
323 btnMore->hide();
324 }
325 for( int i = count - 1; i >= 0; i-- )
326 {
327 if( totalWidth <= size.width() || i == 0)
328 {
329 buttons[i]->setGeometry( x, 0, qMin( size.width() - x, widths[i] ), size.height() );
330 buttons[i]->show();
331 x += widths[i];
332 totalWidth -= widths[i];
333 }
334 else
335 {
336 menuMore->addAction( actions[i] );
337 buttons[i]->hide();
338 if( i < shown ) totalWidth -= widths[i];
339 }
340 }
341}

363#define PADDING 4
364
365void LocationButton::paintEvent ( QPaintEvent * )
366{
367 QStyleOptionButton option;
368 option.initFrom( this );
369 option.state |= QStyle::State_Enabled;
370 QPainter p( this );
371
372 if( underMouse() )
373 {
374 p.save();
375 p.setRenderHint( QPainter::Antialiasing, true );
376 QColor c = palette().color( QPalette::Highlight );
377 p.setPen( c );
378 p.setBrush( c.lighter( 150 ) );
379 p.setOpacity( 0.2 );
380 p.drawRoundedRect( option.rect.adjusted( 0, 2, 0, -2 ), 5, 5 );
381 p.restore();
382 }
383
384 QRect r = option.rect.adjusted( PADDING, 0, -PADDING - (b_arrow ? 10 : 0), 0 );
385
386 QString str( text() );
387 /* This check is absurd, but either it is not done properly inside elidedText(),
388 or boundingRect() is wrong */
389 if( r.width() < fontMetrics().boundingRect( text() ).width() )
390 str = fontMetrics().elidedText( text(), Qt::ElideRight, r.width() );
391 p.drawText( r, Qt::AlignVCenter | Qt::AlignLeft, str );
392
393 if( b_arrow )
394 {
395 option.rect.setWidth( 10 );
396 option.rect.moveRight( rect().right() );
397 style()->drawPrimitive( QStyle::PE_IndicatorArrowRight, &option, &p );
398 }
399}
400
401QSize LocationButton::sizeHint() const
402{
403 QSize s( fontMetrics().boundingRect( text() ).size() );
404 /* Add two pixels to width: font metrics are buggy, if you pass text through elidation
405 with exactly the width of its bounding rect, sometimes it still elides */
406 s.setWidth( s.width() + ( 2 * PADDING ) + ( b_arrow ? 10 : 0 ) + 2 );
407 s.setHeight( s.height() + 2 * PADDING );
408 return s;
409}
