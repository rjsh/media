43void ActionsManager::doAction( int id_action )
44{
45 switch( id_action )
46 {
47 case PLAY_ACTION:
48 play(); break;
49 case STOP_ACTION:
50 THEMIM->stop(); break;
51 case OPEN_ACTION:
52 THEDP->openDialog(); break;
53 case PREVIOUS_ACTION:
54 THEMIM->prev(); break;
55 case NEXT_ACTION:
56 THEMIM->next(); break;
57 case SLOWER_ACTION:
58 THEMIM->getIM()->slower(); break;
59 case FASTER_ACTION:
60 THEMIM->getIM()->faster(); break;
61 case FULLSCREEN_ACTION:
62 fullscreen(); break;
63 case EXTENDED_ACTION:
64 THEDP->extendedDialog(); break;
65 case PLAYLIST_ACTION:
66 playlist(); break;
67 case SNAPSHOT_ACTION:
68 snapshot(); break;
69 case RECORD_ACTION:
70 record(); break;
71 case FRAME_ACTION:
72 frame(); break;
73 case ATOB_ACTION:
74 THEMIM->getIM()->setAtoB(); break;
75 case REVERSE_ACTION:
76 THEMIM->getIM()->reverse(); break;
77 case SKIP_BACK_ACTION:
78 skipBackward();
79 break;
80 case SKIP_FW_ACTION:
81 skipForward();
82 break;
83 case QUIT_ACTION:
84 THEDP->quit(); break;
85 case RANDOM_ACTION:
86 THEMIM->toggleRandom(); break;
87 case INFO_ACTION:
88 THEDP->mediaInfoDialog(); break;
89 case OPEN_SUB_ACTION:
90 THEDP->loadSubtitlesFile(); break;
91 case FULLWIDTH_ACTION:
92 if( p_intf->p_sys->p_mi )
93 p_intf->p_sys->p_mi->getFullscreenControllerWidget()->toggleFullwidth();
94 break;
95 default:
96 msg_Warn( p_intf, "Action not supported: %i", id_action );
97 break;
98 }
99}

101void ActionsManager::play()
102{
103 if( THEPL->current.i_size == 0 && THEPL->items.i_size == 0 )
104 {
105 /* The playlist is empty, open a file requester */
106 THEDP->openFileDialog();
107 return;
108 }
109 THEMIM->togglePlayPause();
110}

112/**
113 * TODO
114 * This functions toggle the fullscreen mode
115 * If there is no video, it should first activate Visualisations...
116 * This has also to be fixed in enableVideo()
117 */
118void ActionsManager::fullscreen()
119{
120 bool fs = var_ToggleBool( THEPL, "fullscreen" );
121 vout_thread_t *p_vout = THEMIM->getVout();
122 if( p_vout)
123 {
124 var_SetBool( p_vout, "fullscreen", fs );
125 vlc_object_release( p_vout );
126 }
127}
128
129void ActionsManager::snapshot()
130{
131 vout_thread_t *p_vout = THEMIM->getVout();
132 if( p_vout )
133 {
134 var_TriggerCallback( p_vout, "video-snapshot" );
135 vlc_object_release( p_vout );
136 }
137}

139void ActionsManager::playlist()

145void ActionsManager::record()
146{
147 input_thread_t *p_input = THEMIM->getInput();
148 if( p_input )
149 {
150 /* This method won't work fine if the stream can't be cut anywhere */
151 var_ToggleBool( p_input, "record" );
152#if 0
153 else
154 {
155 /* 'record' access-filter is not loaded, we open Save dialog */
156 input_item_t *p_item = input_GetItem( p_input );
157 if( !p_item )
158 return;
159
160 char *psz = input_item_GetURI( p_item );
161 if( psz )
162 THEDP->streamingDialog( NULL, qfu(psz), true );
163 }
164#endif
165 }
166}

168void ActionsManager::frame()
169{
170 input_thread_t *p_input = THEMIM->getInput();
171 if( p_input )
172 var_TriggerCallback( p_input, "frame-next" );
173}
