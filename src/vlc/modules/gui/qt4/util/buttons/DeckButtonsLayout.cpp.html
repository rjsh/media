24DeckButtonsLayout::DeckButtonsLayout( QWidget* parent )
25 : QLayout( parent )
26{
27 backwardItem = 0;
28 goItem = 0;
29 forwardItem = 0;
30
31 setContentsMargins( 0, 0, 0, 0 );
32 setSpacing( 0 );
33
34 setBackwardButton( 0 );
35 setRoundButton( 0 );
36 setForwardButton( 0 );
37}

39DeckButtonsLayout::~DeckButtonsLayout()
40{
41 delete backwardItem;
42 delete goItem;
43 delete forwardItem;
44}

46QSize DeckButtonsLayout::sizeHint() const
47{
48 const int bbw = backwardButton ? backwardButton->sizeHint().width() : 0;
49 const int fbw = forwardButton ? forwardButton->sizeHint().width() : 0;
50 const int gbw = bbw +fbw == 0 ? ( RoundButton ? RoundButton->sizeHint().width() : 0 ) : bbw +fbw;
51 int left; int top; int right; int bottom;
52 QSize sh = QSize( gbw, 0 );
53
54 getContentsMargins( &left, &top, &right, &bottom );
55
56 sh.setHeight( qMax( sh.height(), backwardButton ? backwardButton->sizeHint().height() : 0 ) );
57 sh.setHeight( qMax( sh.height(), RoundButton ? RoundButton->sizeHint().height() : 0 ) );
58 sh.setHeight( qMax( sh.height(), forwardButton ? forwardButton->sizeHint().height() : 0 ) );
59
60 sh.rwidth() += left +right;
61 sh.rheight() += top +bottom;
62
63 return sh;
64}

71void DeckButtonsLayout::setGeometry( const QRect& _r )
72{
73 QLayout::setGeometry( _r );
74
75 int left; int top; int right; int bottom;
76 getContentsMargins( &left, &top, &right, &bottom );
77
78 const QRect r = _r.adjusted( left, top, right, bottom );
79 const QAbstractButton* button = backwardButton ? backwardButton : forwardButton;
80 qreal factor = 1;
81
82 if ( !button ) {
83 if ( RoundButton ) {
84 const int min = qMin( r.height(), r.width() );
85 QRect rect = QRect( QPoint(), QSize( min, min ) );
86
87 rect.moveCenter( r.center() );
88 RoundButton->setGeometry( rect );
89 }
90
91 return;
92 }
93 else if ( backwardButton && forwardButton ) {
94 factor = (qreal)r.width() /(qreal)( button->sizeHint().width() *2 );
95 }
96 else if ( RoundButton ) {
97 factor = (qreal)r.width() /(qreal)( button->sizeHint().width() +( RoundButton->sizeHint().width() /2 ) );
98 }
99 else {
100 factor = (qreal)r.width() /(qreal)( button->sizeHint().width() );
101 }
102
103 if ( RoundButton ) {
104 int height = (qreal)RoundButton->sizeHint().height() *factor;
105
106 while ( height > r.height() ) {
107 factor -= 0.1;
108 height = (qreal)RoundButton->sizeHint().height() *factor;
109 }
110
111 QRect rect( QPoint(), QSize( height, height ) );
112 rect.moveCenter( r.center() );
113
114 if ( backwardButton && forwardButton ) {
115 // nothing to do
116 }
117 else if ( backwardButton ) {
118 rect.moveRight( r.right() );
119 }
120 else if ( forwardButton ) {
121 rect.moveLeft( r.left() );
122 }
123
124 RoundButton->setGeometry( rect );
125 }
126 else {
127 int height = (qreal)button->sizeHint().height() *factor;
128
129 while ( height > r.height() ) {
130 factor -= 0.1;
131 height = (qreal)button->sizeHint().height() *factor;
132 }
133 }
134
135 const QSize bs = QSize( (qreal)button->sizeHint().width() *factor, (qreal)button->sizeHint().height() *factor );
136
137 if ( backwardButton ) {
138 QRect gr = RoundButton ? QRect( QPoint(), RoundButton->size() ) : r;
139 QRect rect = QRect( QPoint(), bs );
140
141 if ( RoundButton ) {
142 gr.moveTopLeft( RoundButton->pos() );
143 }
144
145 rect.moveCenter( gr.center() );
146 rect.moveRight( gr.center().x() +1 );
147
148 backwardButton->setGeometry( rect );
149 }
150
151 if ( forwardButton ) {
152 QRect gr = RoundButton ? QRect( QPoint(), RoundButton->size() ) : r;
153 QRect rect = QRect( QPoint(), bs );
154
155 if ( RoundButton ) {
156 gr.moveTopLeft( RoundButton->pos() );
157 }
158
159 rect.moveCenter( gr.center() );
160 rect.moveLeft( gr.center().x() );
161
162 forwardButton->setGeometry( rect );
163 }
164
165 if ( RoundButton ) {
166 RoundButton->raise();
167 }
168}

175QLayoutItem* DeckButtonsLayout::itemAt( int index ) const
176{
177 switch ( index ) {
178 case 0:
179 return backwardItem;
180 case 1:
181 return goItem;
182 case 2:
183 return forwardItem;
184 }
185
186 return 0;
187}
188
189QLayoutItem* DeckButtonsLayout::takeAt( int index )
190{
191 QLayoutItem* item = itemAt( index );
192
193 switch ( index ) {
194 case 0: {
195 backwardItem = 0;
196
197 if ( backwardButton ) {
198 backwardButton->setParent( 0 );
199 }
200
201 backwardButton = 0;
202 break;
203 }
204 case 1: {
205 goItem = 0;
206
207 if ( RoundButton ) {
208 RoundButton->setParent( 0 );
209 }
210
211 RoundButton = 0;
212 break;
213 }
214 case 2: {
215 forwardItem = 0;
216
217 if ( forwardButton ) {
218 forwardButton->setParent( 0 );
219 }
220
221 forwardButton = 0;
222 break;
223 }
224 }
225
226 update();
227
228 return item;
229}

231void DeckButtonsLayout::setBackwardButton( QAbstractButton* button )
232{
233 if ( backwardButton && button == backwardButton ) {
234 return;
235 }
236
237 if ( backwardItem ) {
238 delete takeAt( 0 );
239 }
240
241 if ( button ) {
242 addChildWidget( button );
243 }
244
245 backwardItem = new QWidgetItem( button );
246 backwardButton = button;
247
248 update();
249}
250
251void DeckButtonsLayout::setRoundButton( QAbstractButton* button )
252{
253 if ( RoundButton && button == RoundButton ) {
254 return;
255 }
256
257 if ( goItem ) {
258 delete takeAt( 1 );
259 }
260
261 if ( button ) {
262 addChildWidget( button );
263 }
264
265 goItem = new QWidgetItem( button );
266 RoundButton = button;
267
268 update();
269}
270
271void DeckButtonsLayout::setForwardButton( QAbstractButton* button )
272{
273 if ( forwardButton && button == forwardButton ) {
274 return;
275 }
276
277 if ( forwardItem ) {
278 delete takeAt( 2 );
279 }
280
281 if ( button ) {
282 addChildWidget( button );
283 }
284
285 forwardItem = new QWidgetItem( button );
286 forwardButton = button;
287
288 update();
289}
