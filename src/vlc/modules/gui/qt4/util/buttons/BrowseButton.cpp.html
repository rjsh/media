27BrowseButton::BrowseButton( QWidget* parent, BrowseButton::Type type )
28 : RoundButton( parent )
29{
30 setIconSize( QSize( 16, 16 ) );
31 setType( type );
32}
33
34BrowseButton::Type BrowseButton::type() const
35{
36 return mType;
37}
38
39void BrowseButton::setType( BrowseButton::Type type )
40{
41 //FIXME
42 switch ( type ) {
43 case BrowseButton::Backward:
44 setIcon( QIcon::fromTheme( "media-seek-backward" ) );
45 break;
46 case BrowseButton::Forward:
47 setIcon( QIcon::fromTheme( "media-seek-forward" ) );
48 break;
49 }
50
51 mType = type;
52}
53
54QSize BrowseButton::sizeHint() const
55{
56 return QSize( 50, 26 );
57}
58
59void BrowseButton::paintEvent( QPaintEvent* event )
60{
61 /*RoundButton::paintEvent( event );
62 return;*/
63
64 Q_UNUSED( event );
65
66 const int corner = 5;
67 const int margin = 5;
68 QPainter painter( this );
69 QStyleOptionToolButton option;
70
71 initStyleOption( &option );
72 painter.setRenderHint( QPainter::Antialiasing );
73
74 painter.setPen( QPen( pen( &option ), 1 ) );
75 painter.setBrush( brush( &option ) );
76 painter.drawRoundedRect( rect().adjusted( 1, 1, -1, -1 ), corner, corner );
77
78 switch ( mType ) {
79 case BrowseButton::Backward:
80 option.rect = option.rect.adjusted( 0, 0, -height() +margin, 0 );
81 break;
82 case BrowseButton::Forward:
83 option.rect = option.rect.adjusted( height() -margin, 0, 0, 0 );
84 break;
85 }
86
87 style()->drawControl( QStyle::CE_ToolButtonLabel, &option, &painter, this );
88}
