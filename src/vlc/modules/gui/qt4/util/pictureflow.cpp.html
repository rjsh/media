60// readjust the settings, call this when slide dimension is changed
61void PictureFlowState::reposition()
62{
63 angle = 70 * IANGLE_MAX / 360; // approx. 70 degrees tilted
64
65 offsetX = slideWidth / 2 * (PFREAL_ONE - fcos(angle));
66 offsetY = slideWidth / 2 * fsin(angle);
67 offsetX += slideWidth * PFREAL_ONE;
68 offsetY += slideWidth * PFREAL_ONE / 4;
69 spacing = 40;
70}

72// adjust slides so that they are in "steady state" position
73void PictureFlowState::reset()
74{
75 centerSlide.angle = 0;
76 centerSlide.cx = 0;
77 centerSlide.cy = 0;
78 centerSlide.slideIndex = centerIndex;
79 centerSlide.blend = 256;
80
81 leftSlides.resize(6);
82 for (int i = 0; i < (int)leftSlides.count(); i++) {
83 SlideInfo& si = leftSlides[i];
84 si.angle = angle;
85 si.cx = -(offsetX + spacing * i * PFREAL_ONE);
86 si.cy = offsetY;
87 si.slideIndex = centerIndex - 1 - i;
88 si.blend = 256;
89 if (i == (int)leftSlides.count() - 2)
90 si.blend = 128;
91 if (i == (int)leftSlides.count() - 1)
92 si.blend = 0;
93 }
94
95 rightSlides.resize(6);
96 for (int i = 0; i < (int)rightSlides.count(); i++) {
97 SlideInfo& si = rightSlides[i];
98 si.angle = -angle;
99 si.cx = offsetX + spacing * i * PFREAL_ONE;
100 si.cy = offsetY;
101 si.slideIndex = centerIndex + 1 + i;
102 si.blend = 256;
103 if (i == (int)rightSlides.count() - 2)
104 si.blend = 128;
105 if (i == (int)rightSlides.count() - 1)
106 si.blend = 0;
107 }
108}

117void PictureFlowAnimator::start(int slide)
118{
119 target = slide;
120 if (!animateTimer.isActive() && state) {
121 step = (target < state->centerSlide.slideIndex) ? -1 : 1;
122 animateTimer.start(30);
123 }
124}

126void PictureFlowAnimator::stop(int slide)
127{
128 step = 0;
129 target = slide;
130 frame = slide << 16;
131 animateTimer.stop();
132}
133
134void PictureFlowAnimator::update()
135{
136 if (!animateTimer.isActive())
137 return;
138 if (step == 0)
139 return;
140 if (!state)
141 return;
142
143 int speed = 16384/2;
144
145#if 1
146 // deaccelerate when approaching the target
147 const int max = 2 * 65536;
148
149 int fi = frame;
150 fi -= (target << 16);
151 if (fi < 0)
152 fi = -fi;
153 fi = qMin(fi, max);
154
155 int ia = IANGLE_MAX * (fi - max / 2) / (max * 2);
156 speed = 512 + 16384 * (PFREAL_ONE + fsin(ia)) / PFREAL_ONE;
157#endif
158
159 frame += speed * step;
160
161 int index = frame >> 16;
162 int pos = frame & 0xffff;
163 int neg = 65536 - pos;
164 int tick = (step < 0) ? neg : pos;
165 PFreal ftick = (tick * PFREAL_ONE) >> 16;
166
167 if (step < 0)
168 index++;
169
170 if (state->centerIndex != index) {
171 state->centerIndex = index;
172 frame = index << 16;
173 state->centerSlide.slideIndex = state->centerIndex;
174 for (int i = 0; i < (int)state->leftSlides.count(); i++)
175 state->leftSlides[i].slideIndex = state->centerIndex - 1 - i;
176 for (int i = 0; i < (int)state->rightSlides.count(); i++)
177 state->rightSlides[i].slideIndex = state->centerIndex + 1 + i;
178 }
179
180 state->centerSlide.angle = (step * tick * state->angle) >> 16;
181 state->centerSlide.cx = -step * fmul(state->offsetX, ftick);
182 state->centerSlide.cy = fmul(state->offsetY, ftick);
183
184 if (state->centerIndex == target) {
185 stop(target);
186 state->reset();
187 return;
188 }
189
190 for (int i = 0; i < (int)state->leftSlides.count(); i++) {
191 SlideInfo& si = state->leftSlides[i];
192 si.angle = state->angle;
193 si.cx = -(state->offsetX + state->spacing * i * PFREAL_ONE + step * state->spacing * ftick);
194 si.cy = state->offsetY;
195 }
196
197 for (int i = 0; i < (int)state->rightSlides.count(); i++) {
198 SlideInfo& si = state->rightSlides[i];
199 si.angle = -state->angle;
200 si.cx = state->offsetX + state->spacing * i * PFREAL_ONE - step * state->spacing * ftick;
201 si.cy = state->offsetY;
202 }
203
204 if (step > 0) {
205 PFreal ftick = (neg * PFREAL_ONE) >> 16;
206 state->rightSlides[0].angle = -(neg * state->angle) >> 16;
207 state->rightSlides[0].cx = fmul(state->offsetX, ftick);
208 state->rightSlides[0].cy = fmul(state->offsetY, ftick);
209 } else {
210 PFreal ftick = (pos * PFREAL_ONE) >> 16;
211 state->leftSlides[0].angle = (pos * state->angle) >> 16;
212 state->leftSlides[0].cx = -fmul(state->offsetX, ftick);
213 state->leftSlides[0].cy = fmul(state->offsetY, ftick);
214 }
215
216 // must change direction ?
217 if (target < index) if (step > 0)
218 step = -1;
219 if (target > index) if (step < 0)
220 step = 1;
221
222 // the first and last slide must fade in/fade out
223 int nleft = state->leftSlides.count();
224 int nright = state->rightSlides.count();
225 int fade = pos / 256;
226
227 for (int index = 0; index < nleft; index++) {
228 int blend = 256;
229 if (index == nleft - 1)
230 blend = (step > 0) ? 0 : 128 - fade / 2;
231 if (index == nleft - 2)
232 blend = (step > 0) ? 128 - fade / 2 : 256 - fade / 2;
233 if (index == nleft - 3)
234 blend = (step > 0) ? 256 - fade / 2 : 256;
235 state->leftSlides[index].blend = blend;
236 }
237 for (int index = 0; index < nright; index++) {
238 int blend = (index < nright - 2) ? 256 : 128;
239 if (index == nright - 1)
240 blend = (step > 0) ? fade / 2 : 0;
241 if (index == nright - 2)
242 blend = (step > 0) ? 128 + fade / 2 : fade / 2;
243 if (index == nright - 3)
244 blend = (step > 0) ? 256 : 128 + fade / 2;
245 state->rightSlides[index].blend = blend;
246 }
247
248}

264void PictureFlowSoftwareRenderer::paint()
265{
266 if (!widget)
267 return;
268
269 if (widget->size() != size)
270 init();
271
272 if (state->backgroundColor != bgcolor) {
273 bgcolor = state->backgroundColor;
274 }
275
276 if ((int)(state->reflectionEffect) != effect) {
277 effect = (int)state->reflectionEffect;
278 }
279
280 if (dirty)
281 render();
282
283 QPainter painter(widget);
284 painter.setCompositionMode(QPainter::CompositionMode_Source);
285 painter.drawImage(QPoint(0, 0), buffer);
286}

288void PictureFlowSoftwareRenderer::init()
289{
290 if (!widget)
291 return;
292
293 blankSurface = 0;
294
295 size = widget->size();
296 int ww = size.width();
297 int wh = size.height();
298 int w = (ww + 1) / 2;
299 int h = (wh + 1) / 2;
300
301 buffer = QImage(ww, wh, QImage::Format_ARGB32);
302 buffer.fill(bgcolor);
303
304 rays.resize(w*2);
305 for (int i = 0; i < w; i++) {
306 PFreal gg = ((PFREAL_ONE >> 1) + i * PFREAL_ONE) / (2 * h);
307 rays[w-i-1] = -gg;
308 rays[w+i] = gg;
309 }
310
311 dirty = true;
312}

314// TODO: optimize this with lookup tables
315static QRgb blendColor(QRgb c1, QRgb c2, int blend)
316{
317 unsigned int a,r,g,b,as,ad;
318 if(blend>255)
319 blend=255;
320 as=(qAlpha(c1)*blend)/256;
321 ad=qAlpha(c2);
322 a=as+((255-as)*ad)/256;
323 if(a>0)
324 {
325 r=(as*qRed(c1)+((255-as)*ad*qRed(c2))/256)/a;
326 g=(as*qGreen(c1)+((255-as)*ad*qGreen(c2))/256)/a;
327 b=(as*qBlue(c1)+((255-as)*ad*qBlue(c2))/256)/a;
328 }
329 else
330 {
331 r=g=b=0;
332 }
333 return qRgba(r, g, b, a);
334}

337static QImage* prepareSurface(const QImage* slideImage, int w, int h, QRgb bgcolor,
338 PictureFlow::ReflectionEffect reflectionEffect, QModelIndex index)
339{
340 Q_UNUSED(bgcolor);
341 Qt::TransformationMode mode = Qt::SmoothTransformation;
342 QImage img = slideImage->scaled(w, h, Qt::KeepAspectRatio, mode);
343
344 // slightly larger, to accomodate for the reflection
345 int hs = h * 2;
346 int hofs = h / 3;
347
348 // offscreen buffer: black is sweet
349 QImage* result = new QImage(hs, w, QImage::Format_ARGB32);
350 QFont font( index.data( Qt::FontRole ).value<QFont>() );
351 QPainter imagePainter( result );
352 QTransform rotation;
353 imagePainter.setFont( font );
354 rotation.rotate(90);
355 rotation.scale(1,-1);
356 rotation.translate( 0, hofs );
357 QRgb bg=qRgba(0, 0, 0, 0);
358 result->fill(bg);
359
360 // transpose the image, this is to speed-up the rendering
361 // because we process one column at a time
362 // (and much better and faster to work row-wise, i.e in one scanline)
363 /*
364 for (int x = 0; x < w; x++)
365 for (int y = 0; y < h; y++)
366 result->setPixel(hofs + y, x, img.pixel(x, y));
367 */
368 if (reflectionEffect != PictureFlow::NoReflection) {
369 // create the reflection
370 int ht = hs - h - hofs;
371 int hte = ht;
372 for (int x = 0; x < w; x++)
373 {
374 QRgb *line = (QRgb*)(result->scanLine( x ));
375 int xw=img.width(),yw=img.height();
376 QRgb color;
377 for (int y = 0; y < ht; y++) {
378 color=bg;
379 int x0=x-(w-xw)/2;
380 int y0=yw - y - 1+(h-yw)/2;
381 if(x0>=0 && x0<xw && y0>=0 && y0<yw)
382 color = img.pixel(x0, y0);
383 line[h+hofs+y] = blendColor( color, bg, 128*(hte-y)/hte );
384 //result->setPixel(h + hofs + y, x, blendColor(color, bgcolor, 128*(hte - y) / hte));
385 }
386 }
387
388 if (reflectionEffect == PictureFlow::BlurredReflection) {
389 // blur the reflection everything first
390 // Based on exponential blur algorithm by Jani Huhtanen
391 QRect rect(hs / 2, 0, hs / 2, w);
392 rect &= result->rect();
393
394 int r1 = rect.top();
395 int r2 = rect.bottom();
396 int c1 = rect.left();
397 int c2 = rect.right();
398
399 int bpl = result->bytesPerLine();
400 int rgba[4];
401 unsigned char* p;
402
403 // how many times blur is applied?
404 // for low-end system, limit this to only 1 loop
405 for (int loop = 0; loop < 2; loop++) {
406 for (int col = c1; col <= c2; col++) {
407 p = result->scanLine(r1) + col * 4;
408 for (int i = 0; i < 3; i++)
409 rgba[i] = p[i] << 4;
410
411 p += bpl;
412 for (int j = r1; j < r2; j++, p += bpl)
413 for (int i = 0; i < 3; i++)
414 p[i] = (rgba[i] += (((p[i] << 4) - rgba[i])) >> 1) >> 4;
415 }
416
417 for (int row = r1; row <= r2; row++) {
418 p = result->scanLine(row) + c1 * 4;
419 for (int i = 0; i < 3; i++)
420 rgba[i] = p[i] << 4;
421
422 p += 4;
423 for (int j = c1; j < c2; j++, p += 4)
424 for (int i = 0; i < 3; i++)
425 p[i] = (rgba[i] += (((p[i] << 4) - rgba[i])) >> 1) >> 4;
426 }
427
428 for (int col = c1; col <= c2; col++) {
429 p = result->scanLine(r2) + col * 4;
430 for (int i = 0; i < 3; i++)
431 rgba[i] = p[i] << 4;
432
433 p -= bpl;
434 for (int j = r1; j < r2; j++, p -= bpl)
435 for (int i = 0; i < 3; i++)
436 p[i] = (rgba[i] += (((p[i] << 4) - rgba[i])) >> 1) >> 4;
437 }
438
439 for (int row = r1; row <= r2; row++) {
440 p = result->scanLine(row) + c2 * 4;
441 for (int i = 0; i < 3; i++)
442 rgba[i] = p[i] << 4;
443
444 p -= 4;
445 for (int j = c1; j < c2; j++, p -= 4)
446 for (int i = 0; i < 3; i++)
447 p[i] = (rgba[i] += (((p[i] << 4) - rgba[i])) >> 1) >> 4;
448 }
449 }
450 }
451 }
452 // overdraw to leave only the reflection blurred (but not the actual image)
453 imagePainter.setTransform( rotation );
454 imagePainter.drawImage( (w-img.width())/2, (h-img.height())/2, img );
455 imagePainter.setBrush( QColor(bg));//QBrush( Qt::lightGray ) );
456 imagePainter.setPen( QColor( Qt::lightGray ) );
457 QFontMetrics fm = imagePainter.fontMetrics();
458 imagePainter.setPen( QColor( Qt::darkGray ) );
459 imagePainter.drawText( 0+1, 1+h-fm.height()*2, VLCModel::getMeta( index, COLUMN_TITLE ) );
460 imagePainter.setPen( QColor( Qt::lightGray ) );
461 imagePainter.drawText( 0, h-fm.height()*2, VLCModel::getMeta( index, COLUMN_TITLE ) );
462 imagePainter.setPen( QColor( Qt::darkGray ) );
463 imagePainter.drawText( 0+1, 1+h-fm.height()*1, VLCModel::getMeta( index, COLUMN_ARTIST ) );
464 imagePainter.setPen( QColor( Qt::lightGray ) );
465 imagePainter.drawText( 0, h-fm.height()*1, VLCModel::getMeta( index, COLUMN_ARTIST ) );
466
467 return result;
468}

470QImage* PictureFlowSoftwareRenderer::surface(QModelIndex index)
471{
472 if (!state || !index.isValid())
473 return 0;
474
475 QImage* img = new QImage(VLCModel::getArtPixmap( index,
476 QSize( state->slideWidth, state->slideHeight ) ).toImage());
477
478 QImage* sr = prepareSurface(img, state->slideWidth, state->slideHeight, bgcolor, state->reflectionEffect, index );
479
480 delete img;
481 return sr;
482}

484// Renders a slide to offscreen buffer. Returns a rect of the rendered area.
485// col1 and col2 limit the column for rendering.
486QRect PictureFlowSoftwareRenderer::renderSlide(const SlideInfo &slide, int col1, int col2)
487{
488 int blend = slide.blend;
489 if (!blend)
490 return QRect();
491
492 QModelIndex index;
493
494 QString artURL;
495
496 VLCModel *m = static_cast<VLCModel*>( state->model );
497
498 if( m )
499 {
500 index = m->index( slide.slideIndex, 0, m->currentIndex().parent() );
501 if( !index.isValid() )
502 return QRect();
503 artURL = m->data( index, COLUMN_COVER ).toString();
504 }
505
506 QString key = QString("%1%2%3%4").arg(VLCModel::getMeta( index, COLUMN_TITLE )).arg( VLCModel::getMeta( index, COLUMN_ARTIST ) ).arg(index.data( VLCModel::IsCurrentRole ).toBool() ).arg( artURL );
507
508 QImage* src;
509 if( cache.contains( key ) )
510 src = cache.value( key );
511 else
512 {
513 src = surface( index );
514 cache.insert( key, src );
515 }
516 if (!src)
517 return QRect();
518
519 QRect rect(0, 0, 0, 0);
520
521 int sw = src->height();
522 int sh = src->width();
523 int h = buffer.height();
524 int w = buffer.width();
525
526 if (col1 > col2) {
527 int c = col2;
528 col2 = col1;
529 col1 = c;
530 }
531
532 col1 = (col1 >= 0) ? col1 : 0;
533 col2 = (col2 >= 0) ? col2 : w - 1;
534 col1 = qMin(col1, w - 1);
535 col2 = qMin(col2, w - 1);
536
537 int zoom = 100;
538 int distance = h * 100 / zoom;
539 PFreal sdx = fcos(slide.angle);
540 PFreal sdy = fsin(slide.angle);
541 PFreal xs = slide.cx - state->slideWidth * sdx / 2;
542 PFreal ys = slide.cy - state->slideWidth * sdy / 2;
543 PFreal dist = distance * PFREAL_ONE;
544
545 int xi = qMax((PFreal)0, ((w * PFREAL_ONE / 2) + fdiv(xs * h, dist + ys)) >> PFREAL_SHIFT);
546 if (xi >= w)
547 {
548 return rect;
549 }
550
551 bool flag = false;
552 rect.setLeft(xi);
553 for (int x = qMax(xi, col1); x <= col2; x++) {
554 PFreal hity = 0;
555 PFreal fk = rays[x];
556 if (sdy) {
557 fk = fk - fdiv(sdx, sdy);
558 hity = -fdiv((rays[x] * distance - slide.cx + slide.cy * sdx / sdy), fk);
559 }
560
561 dist = distance * PFREAL_ONE + hity;
562 if (dist < 0)
563 continue;
564
565 PFreal hitx = fmul(dist, rays[x]);
566 PFreal hitdist = fdiv(hitx - slide.cx, sdx);
567
568 int column = sw / 2 + (hitdist >> PFREAL_SHIFT);
569 if (column >= sw)
570 break;
571 if (column < 0)
572 continue;
573
574 rect.setRight(x);
575 if (!flag)
576 rect.setLeft(x);
577 flag = true;
578
579 int y1 = h / 2;
580 int y2 = y1 + 1;
581 QRgb* pixel1 = (QRgb*)(buffer.scanLine(y1)) + x;
582 QRgb* pixel2 = (QRgb*)(buffer.scanLine(y2)) + x;
583 QRgb pixelstep = pixel2 - pixel1;
584
585 int center = (sh / 2);
586 int dy = dist / h;
587 int p1 = center * PFREAL_ONE - dy / 2;
588 int p2 = center * PFREAL_ONE + dy / 2;
589
590 const QRgb *ptr = (const QRgb*)(src->scanLine(column));
591 while ((y1 >= 0) && (y2 < h) && (p1 >= 0)) {
592 QRgb c1 = ptr[p1 >> PFREAL_SHIFT];
593 QRgb c2 = ptr[p2 >> PFREAL_SHIFT];
594 *pixel1 = blendColor(c1, *pixel1+0*bgcolor, blend);
595 *pixel2 = blendColor(c2, *pixel2+0*bgcolor, blend);
596 p1 -= dy;
597 p2 += dy;
598 y1--;
599 y2++;
600 pixel1 -= pixelstep;
601 pixel2 += pixelstep;
602 }
603 }
604
605 rect.setTop(0);
606 rect.setBottom(h - 1);
607 return rect;
608}
609
610void PictureFlowSoftwareRenderer::renderSlides()
611{
612 int nleft = state->leftSlides.count();
613 int nright = state->rightSlides.count();
614
615 for (int index = nleft-1; index >= 0; index--) {
616 renderSlide(state->leftSlides[index]);
617 }
618 for (int index = nright-1; index >= 0; index--) {
619 renderSlide(state->rightSlides[index]);
620 }
621 renderSlide(state->centerSlide);
622}
623
624// Render the slides. Updates only the offscreen buffer.
625void PictureFlowSoftwareRenderer::render()
626{
627 buffer.fill(state->backgroundColor);
628 renderSlides();
629 dirty = false;
630}

635PictureFlow::PictureFlow(QWidget* parent, QAbstractItemModel* _p_model): QWidget(parent)
636{
637 d = new PictureFlowPrivate;
638 d->picrole = Qt::DecorationRole;
639 d->textrole = Qt::DisplayRole;
640 d->piccolumn = 0;
641 d->textcolumn = 0;
642
643 d->state = new PictureFlowState;
644 d->state->model = 0;
645 d->state->reset();
646 d->state->reposition();
647
648 d->renderer = new PictureFlowSoftwareRenderer;
649 d->renderer->state = d->state;
650 d->renderer->widget = this;
651 d->renderer->init();
652
653 d->animator = new PictureFlowAnimator;
654 d->animator->state = d->state;
655 QObject::connect(&d->animator->animateTimer, SIGNAL(timeout()), this, SLOT(updateAnimation()));
656
657 QObject::connect(&d->triggerTimer, SIGNAL(timeout()), this, SLOT(render()));
658
659 setAttribute(Qt::WA_StaticContents, true);
660 setAttribute(Qt::WA_OpaquePaintEvent, true);
661 setAttribute(Qt::WA_NoSystemBackground, true);
662
663 d->setModel(_p_model);
664}
665
666PictureFlow::~PictureFlow()
667{
668 delete d->renderer;
669 delete d->animator;
670 delete d->state;
671 delete d;
672}

771void PictureFlow::showPrevious()
772{
773 int step = d->animator->step;
774 int center = d->state->centerIndex;
775
776 if (step > 0)
777 d->animator->start(center);
778
779 if (step == 0)
780 if (center > 0)
781 d->animator->start(center - 1);
782
783 if (step < 0)
784 d->animator->target = qMax(0, center - 2);
785}

787void PictureFlow::showNext()
788{
789 int step = d->animator->step;
790 int center = d->state->centerIndex;
791
792 if (step < 0)
793 d->animator->start(center);
794
795 if (step == 0)
796 if (center < slideCount() - 1)
797 d->animator->start(center + 1);
798
799 if (step > 0)
800 d->animator->target = qMin(center + 2, slideCount() - 1);
801}

813void PictureFlow::keyPressEvent(QKeyEvent* event)
814{

864void PictureFlow::wheelEvent(QWheelEvent * event)
865{
866 if (event->orientation() == Qt::Horizontal)

868 event->ignore();
869 }
870 else
871 {
872 int numSteps = -((event->delta() / 8) / 15);
873
874 if (numSteps > 0)
875 {
876 for (int i = 0;i < numSteps;i++)

880 }
881 else
882 {

884 {
885 showPrevious();
886 }
887 }
888 event->accept();
889 }
