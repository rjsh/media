51/* Input Slider derived from QSlider */
52class SeekSlider : public QSlider
53{
54 Q_OBJECT
55 Q_PROPERTY(qreal handleOpacity READ handleOpacity WRITE setHandleOpacity)
56public:
57 SeekSlider( Qt::Orientation q, QWidget *_parent = 0, bool _classic = false );
58 virtual ~SeekSlider();
59 void setChapters( SeekPoints * );
60
61protected:
62 virtual void mouseMoveEvent( QMouseEvent *event );
63 virtual void mousePressEvent( QMouseEvent* event );
64 virtual void mouseReleaseEvent( QMouseEvent *event );
65 virtual void wheelEvent( QWheelEvent *event );
66 virtual void enterEvent( QEvent * );
67 virtual void leaveEvent( QEvent * );
68 virtual void hideEvent( QHideEvent * );
69 virtual void paintEvent(QPaintEvent *ev);
70
71 virtual bool eventFilter( QObject *obj, QEvent *event );
72
73 virtual QSize sizeHint() const;
74
75 void processReleasedButton();
76 qreal handleOpacity() const;
77 void setHandleOpacity( qreal opacity );
78 int handleLength();
79
80private:
81 bool isSliding; /* Whether we are currently sliding by user action */
82 bool isJumping; /* if we requested a jump to another chapter */
83 int inputLength; /* InputLength that can change */
84 char psz_length[MSTRTIME_MAX_SIZE]; /* Used for the ToolTip */
85 QTimer *seekLimitTimer;
86 TimeTooltip *mTimeTooltip;
87 float f_buffering;
88 QTime bufferingStart;
89 SeekPoints* chapters;
90 bool b_classic;
91 bool b_seekable;
92 int mHandleLength;
93
94 /* Colors & gradients */
95 QSize gradientsTargetSize;
96 QLinearGradient backgroundGradient;
97 QLinearGradient foregroundGradient;
98 QLinearGradient handleGradient;
99 QColor tickpointForeground;
100 QColor shadowDark;
101 QColor shadowLight;
102 QCommonStyle *alternativeStyle;
103
104 /* Handle's animation */
105 qreal mHandleOpacity;
106 QPropertyAnimation *animHandle;
107 QTimer *hideHandleTimer;
108
109public slots:
110 void setPosition( float, int64_t, int );
111 void setSeekable( bool b ) { b_seekable = b ; }
112 void updateBuffering( float );
113 void hideHandle();
114
115private slots:
116 void startSeekTimer();
117 void updatePos();
118
119signals:
120 void sliderDragged( float );
121
122};

129class SoundSlider : public QAbstractSlider
130{
131 Q_OBJECT
132public:
133 SoundSlider(QWidget *_parent, float _i_step, char *psz_colors, int max = SOUNDMAX );
134 void setMuted( bool ); /* Set Mute status */
135
136protected:
137 const static int paddingL = 3;
138 const static int paddingR = 2;
139
140 virtual void paintEvent( QPaintEvent *);
141 virtual void wheelEvent( QWheelEvent *event );
142 virtual void mousePressEvent( QMouseEvent * );
143 virtual void mouseMoveEvent( QMouseEvent * );
144 virtual void mouseReleaseEvent( QMouseEvent * );
145
146 void processReleasedButton();
147
148private:
149 bool isSliding; /* Whether we are currently sliding by user action */
150 bool b_mouseOutside; /* Whether the mouse is outside or inside the Widget */
151 int i_oldvalue; /* Store the old Value before changing */
152 float f_step; /* How much do we increase each time we wheel */
153 bool b_isMuted;
154
155 QPixmap pixGradient; /* Gradient pix storage */
156 QPixmap pixGradient2; /* Muted Gradient pix storage */
157 QPixmap pixOutside; /* OutLine pix storage */
158 QPainter painter;
159 QColor background;
160 QColor foreground;
161 QFont textfont;
162 QRect textrect;
163
164 void changeValue( int x ); /* Function to modify the value from pixel x() */
165};
