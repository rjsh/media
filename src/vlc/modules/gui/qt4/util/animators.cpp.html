24#include <QWidget>
25#include <QPixmap>
26#include <QAbstractItemView>

28BasicAnimator::BasicAnimator( QObject *parent )
29 : QAbstractAnimation( parent ), current_frame( 0 )
30{
31 setFps( 15 );
32 setLoopCount( -1 );
33}
34
35void BasicAnimator::updateCurrentTime( int msecs )
36{
37 msecs += (interval / 2);
38 int i = ( msecs / interval );
39 if ( i != current_frame )
40 {
41 current_frame = i;
42 emit frameChanged();
43 }
44}
45
46PixmapAnimator::PixmapAnimator( QWidget *parent, QList<QString> frames )
47 : BasicAnimator( parent )
48{
49 foreach( QString name, frames )
50 pixmaps.append( new QPixmap( name ) );
51 currentPixmap = pixmaps.at( 0 );
52 setFps( frames.count() ); /* default to 1 sec loop */
53}
54
55PixmapAnimator::~PixmapAnimator()
56{
57 qDeleteAll( pixmaps );
58}
59
60void PixmapAnimator::updateCurrentTime( int msecs )
61{
62 int i = msecs / interval;
63 if ( i >= pixmaps.count() ) i = pixmaps.count() - 1; /* roundings */
64 if ( i != current_frame )
65 {
66 current_frame = i;
67 currentPixmap = pixmaps.at( current_frame );
68 emit pixmapReady( *currentPixmap );
69 }
70}

72DelegateAnimationHelper::DelegateAnimationHelper( QAbstractItemView *view_,
73 BasicAnimator *animator_ )
74 : QObject( view_ ), view( view_ ), animator( animator_ )
75{
76 if ( !animator )
77 {
78 animator = new BasicAnimator( this );
79 animator->setFps( 15 );
80 animator->setLoopCount( -1 );
81 }
82 setIndex( QModelIndex() );
83 CONNECT( animator, frameChanged(), this, updateDelegate() );
84}
85
86void DelegateAnimationHelper::setIndex( const QModelIndex &index_ )
87{
88 index = QPersistentModelIndex( index_ );
89}
90
91void DelegateAnimationHelper::run( bool b_run )
92{
93 if ( b_run )
94 {
95 if ( ! isRunning() ) animator->start();
96 }
97 else
98 animator->stop();
99}
100
101bool DelegateAnimationHelper::isRunning() const
102{
103 return ( animator->state() == QAbstractAnimation::Running );
104}
105
106const QPersistentModelIndex & DelegateAnimationHelper::getIndex() const
107{
108 return index;
109}
110
111void DelegateAnimationHelper::updateDelegate()
112{
113 /* Prevent running indefinitively if removed from model */
114 if ( !index.isValid() )
115 run( false );
116 else
117 {
118 if ( view->viewport() )
119 view->viewport()->update();
120 else
121 view->update( index );
122 }
123}
