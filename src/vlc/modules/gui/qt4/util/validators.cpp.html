29QValidator::State UrlValidator::validate( QString& str, int& ) const
30{
31 if( str.startsWith( ' ' ) )
32 return QValidator::Invalid;
33
34 if ( str.isEmpty() )
35 return QValidator::Intermediate;
36
37 QUrl url( str );
38 if ( url.scheme().isEmpty() )
39 return QValidator::Intermediate;
40
41 return ( url.isValid() ) ? QValidator::Acceptable : QValidator::Intermediate;
42}
43
44void UrlValidator::fixup( QString & input ) const
45{
46 while( input.startsWith( ' ' ) )
47 input.chop( 1 );
48 QUrl fixed( input, QUrl::TolerantMode );
49 input = fixed.toString();
50}
