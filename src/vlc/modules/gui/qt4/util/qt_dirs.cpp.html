29QString toURI( const QString& s )
30{
31 if( s.contains( qfu("://") ) )
32 return s;
33
34 char *psz = vlc_path2uri( qtu(s), NULL );
35 if( psz == NULL )
36 return qfu("");
37
38 QString uri = qfu( psz );
39 free( psz );
40 return uri;
41}
