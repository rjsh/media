48void QFramelessButton::paintEvent( QPaintEvent * )
49{
50 QPainter painter( this );
51 QPixmap pix = icon().pixmap( size() );
52 QPoint pos( (width() - pix.width()) / 2, (height() - pix.height()) / 2 );
53 painter.drawPixmap( QRect( pos.x(), pos.y(), pix.width(), pix.height() ), pix );
54}

109/***************************************************************************
110 * Hotkeys converters
111 ***************************************************************************/
112int qtKeyModifiersToVLC( QInputEvent* e )
113{
114 int i_keyModifiers = 0;
115 if( e->modifiers() & Qt::ShiftModifier ) i_keyModifiers |= KEY_MODIFIER_SHIFT;
116 if( e->modifiers() & Qt::AltModifier ) i_keyModifiers |= KEY_MODIFIER_ALT;
117 if( e->modifiers() & Qt::ControlModifier ) i_keyModifiers |= KEY_MODIFIER_CTRL;
118 if( e->modifiers() & Qt::MetaModifier ) i_keyModifiers |= KEY_MODIFIER_META;
119 return i_keyModifiers;
120}

260static int keycmp( const void *a, const void *b )
261{
262 const int *q = (const int *)a;
263 const vlc_qt_key_t *m = (const vlc_qt_key_t *)b;
264
265 return *q - m->qt;
266}

268int qtEventToVLCKey( QKeyEvent *e )
269{
270 int qtk = e->key();
271 uint32_t i_vlck = 0;
272
273 if( qtk <= 0xff )
274 {
275 /* VLC and X11 use lowercase whereas Qt uses uppercase, this
276 * method should be equal to towlower in case of latin1 */
277 if( qtk >= 'A' && qtk <= 'Z' ) i_vlck = qtk+32;
278 else if( qtk >= 0xC0 && qtk <= 0xDE && qtk != 0xD7) i_vlck = qtk+32;
279 else i_vlck = qtk;
280 }
281 else
282 {
283 const vlc_qt_key_t *map;
284
285 map = (const vlc_qt_key_t *)
286 bsearch( &qtk, (const void *)keys, sizeof(keys)/sizeof(keys[0]),
287 sizeof(*keys), keycmp );
288 if( map != NULL )
289 i_vlck = map->vlc;
290 }
291
292 /* Handle modifiers */
293 i_vlck |= qtKeyModifiersToVLC( e );
294 return i_vlck;
295}
296
297int qtWheelEventToVLCKey( QWheelEvent *e )
298{
299 int i_vlck = 0;
300 /* Handle modifiers */
301 i_vlck |= qtKeyModifiersToVLC( e );
302 if ( e->delta() > 0 )
303 i_vlck |= KEY_MOUSEWHEELUP;
304 else
305 i_vlck |= KEY_MOUSEWHEELDOWN;
306 return i_vlck;
307}

309QString VLCKeyToString( unsigned val, bool locale )
310{
311 char *base = vlc_keycode2str (val, locale);
312 if (base == NULL)
313 return qtr( "Unset" );
314
315 QString r = qfu( base );
316
317 free( base );
318 return r;
319}

337QToolButtonExt::QToolButtonExt(QWidget *parent, int ms )
338 : QToolButton( parent ),
339 shortClick( false ),
340 longClick( false )
341{
342 setAutoRepeat( true );
343 /* default to twice the doubleclick delay */
344 setAutoRepeatDelay( ( ms > 0 )? ms : 2 * QApplication::doubleClickInterval() );
345 setAutoRepeatInterval( 100 );
346 connect( this, SIGNAL(released()), this, SLOT(releasedSlot()) );
347 connect( this, SIGNAL(clicked()), this, SLOT(clickedSlot()) );
348}

350/* table illustrating the different scenarios and the events generated
351 * ====================
352 *
353 * event isDown()
354 *
355 * released false }
356 * clicked false }= short click
357 *
358 * released false = cancelled click (mouse released outside of button area,
359 * before long click delay kicks in)
360 *
361 * released true }
362 * clicked true }= long click (multiple of these generated)
363 * released false = stop long click (mouse released / moved outside of
364 * button area)
365 * (clicked false) = stop long click (additional event if mouse released
366 * inside of button area)
367 */
368
369void QToolButtonExt::releasedSlot()
370{
371 if( isDown() )
372 {
373 // we are beginning a long click
374 longClick = true;
375 shortClick = false;
376 }
377 else
378 {
379 if( longClick )
380 {
381 // we are stopping a long click
382 longClick = false;
383 shortClick = false;
384 }
385 else
386 {
387 // we are generating a short click
388 longClick = false;
389 shortClick = true;
390 }
391 }
392}
393
394void QToolButtonExt::clickedSlot()
395{
396 if( longClick )
397 emit longClicked();
398 else if( shortClick )
399 emit shortClicked();
400}
