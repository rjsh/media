22#include "timetooltip.hpp"

24#include <QApplication>
25#include <QPainter>
26#include <QPainterPath>
27#include <QBitmap>
28#include <QFontMetrics>
29#include <QDesktopWidget>
30
31#define TIP_HEIGHT 5
32
33TimeTooltip::TimeTooltip( QWidget *parent ) :
34 QWidget( parent )
35{
36 setWindowFlags( Qt::Window |
37 Qt::WindowStaysOnTopHint |
38 Qt::FramelessWindowHint |
39 Qt::X11BypassWindowManagerHint );
40
41 // Tell Qt that it doesn't need to erase the background before
42 // a paintEvent occurs. This should save some CPU cycles.
43 setAttribute( Qt::WA_OpaquePaintEvent );
44
45#if defined( Q_OS_WIN ) || defined( Q_OS_OS2 )
46 /*
47 - This attribute is required on Windows and OS/2 to avoid focus stealing of other windows.
48 - When set on Linux the TimeTooltip appears behind the FSController in fullscreen.
49 */
50 setAttribute( Qt::WA_ShowWithoutActivating );
51#endif
52
53 // Inherit from the system default font size -5
54 mFont = QFont( "Verdana", qMax( qApp->font().pointSize() - 5, 7 ) );
55 mTipX = -1;
56
57 // By default the widget is unintialized and should not be displayed
58 resize( 0, 0 );
59}
60
61void TimeTooltip::adjustPosition()
62{
63 if( mDisplayedText.isEmpty() )
64 {
65 resize( 0, 0 );
66 return;
67 }
68
69 // Get the bounding box required to print the text and add some padding
70 QFontMetrics metrics( mFont );
71 QRect textbox = metrics.boundingRect( mDisplayedText );
72 textbox.adjust( -2, -2, 2, 2 );
73 textbox.moveTo( 0, 0 );
74
75 // Resize the widget to fit our needs
76 QSize size( textbox.width() + 1, textbox.height() + TIP_HEIGHT + 1 );
77
78 // The desired label position is just above the target
79 QPoint position( mTarget.x() - size.width() / 2,
80 mTarget.y() - size.height() + TIP_HEIGHT / 2 );
81
82 // Keep the tooltip on the same screen if possible
83 QRect screen = QApplication::desktop()->screenGeometry( mTarget );
84 position.setX( qMax( screen.left(), qMin( position.x(),
85 screen.left() + screen.width() - size.width() ) ) );
86 position.setY( qMax( screen.top(), qMin( position.y(),
87 screen.top() + screen.height() - size.height() ) ) );
88
89 move( position );
90
91 int tipX = mTarget.x() - position.x();
92 if( mBox != textbox || mTipX != tipX )
93 {
94 mBox = textbox;
95 mTipX = tipX;
96
97 resize( size );
98 buildPath();
99 setMask( mMask );
100 }
101}
102
103void TimeTooltip::buildPath()
104{
105 // Prepare the painter path for future use so
106 // we only have to generate the text at runtime.
107
108 // Draw the text box
109 mPainterPath = QPainterPath();
110 mPainterPath.addRect( mBox );
111
112 // Draw the tip
113 QPolygon polygon;
114 polygon << QPoint( qMax( 0, mTipX - 3 ), mBox.height() )
115 << QPoint( mTipX, mBox.height() + TIP_HEIGHT )
116 << QPoint( qMin( mTipX + 3, mBox.width() ), mBox.height() );
117 mPainterPath.addPolygon( polygon );
118
119 // Store the simplified version of the path
120 mPainterPath = mPainterPath.simplified();
121
122 // Create the mask used to erase the background
123 // Note: this is a binary bitmap (black & white)
124 mMask = QBitmap( size() );
125 QPainter painter( &mMask );
126 painter.fillRect( mMask.rect(), Qt::white );
127 painter.setPen( Qt::black );
128 painter.setBrush( Qt::black );
129 painter.drawPath( mPainterPath );
130 painter.end();
131}

133void TimeTooltip::setTip( const QPoint& target, const QString& time, const QString& text )
134{
135 mDisplayedText = time;
136 if ( !text.isEmpty() )
137 mDisplayedText.append( " - " ).append( text );
138
139 if( mTarget != target || time.length() != mTime.length() || mText != text )
140 {
141 mTarget = target;
142 mTime = time;
143 mText = text;
144 adjustPosition();
145 }
146
147 update();
148 raise();
149}
150
151void TimeTooltip::show()
152{
153 setVisible( true );
154 raise();
155}
156
157void TimeTooltip::paintEvent( QPaintEvent * )
158{
159 QPainter p( this );
160 p.setRenderHints( QPainter::HighQualityAntialiasing | QPainter::TextAntialiasing );
161
162 p.setPen( Qt::black );
163 p.setBrush( qApp->palette().base() );
164 p.drawPath( mPainterPath );
165
166 p.setFont( mFont );
167 p.setPen( QPen( qApp->palette().text(), 1 ) );
168 p.drawText( mBox, Qt::AlignCenter, mDisplayedText );
169}
