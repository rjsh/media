21#ifndef ANIMATORS_HPP
22#define ANIMATORS_HPP
23
24#include <QObject>
25#include <QList>
26#include <QString>
27#include <QAbstractAnimation>
28#include <QPersistentModelIndex>
29
30class QWidget;
31class QPixmap;
32class QAbstractItemView;
33
34class BasicAnimator : public QAbstractAnimation
35{
36 Q_OBJECT
37
38public:
39 BasicAnimator( QObject *parent = 0 );
40 void setFps( int _fps ) { fps = _fps; interval = 1000.0 / fps; }
41 virtual int duration() const { return 1000; }
42
43signals:
44 void frameChanged();
45
46protected:
47 virtual void updateCurrentTime ( int msecs );
48 int fps;
49 int interval;
50 int current_frame;
51};
52
53/** An animated pixmap
54 * Use this widget to display an animated icon based on a series of
55 * pixmaps. The pixmaps will be stored in memory and should be kept small.
56 * First, create the widget, add frames and then start playing. Looping
57 * is supported.
58 **/
59class PixmapAnimator : public BasicAnimator
60{
61 Q_OBJECT
62
63public:
64 PixmapAnimator( QWidget *parent, QList<QString> _frames );
65 virtual int duration() const { return interval * pixmaps.count(); }
66 virtual ~PixmapAnimator();
67 QPixmap *getPixmap() { return currentPixmap; }
68protected:
69 virtual void updateCurrentTime ( int msecs );
70 QList<QPixmap *> pixmaps;
71 QPixmap *currentPixmap;
72signals:
73 void pixmapReady( const QPixmap & );
74};
75
76class DelegateAnimationHelper : public QObject
77{
78 Q_OBJECT
79
80public:
81 DelegateAnimationHelper( QAbstractItemView *view, BasicAnimator *animator = 0 );
82 void setIndex( const QModelIndex &index );
83 bool isRunning() const;
84 const QPersistentModelIndex & getIndex() const;
85
86public slots:
87 void run( bool );
88
89protected slots:
90 void updateDelegate();
91
92private:
93 QAbstractItemView *view;
94 BasicAnimator *animator;
95 QPersistentModelIndex index;
96};
