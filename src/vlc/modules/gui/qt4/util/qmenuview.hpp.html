27#include <QMenu>
28#include <QAbstractItemModel>
29
30class QMenuView : public QMenu
31{
32 Q_OBJECT
33
34public:
35 QMenuView( QWidget * parent = 0, int iMaxVisibleCount = 0 );
36 virtual ~QMenuView(){}
37
38 /* Model */
39 void setModel( QAbstractItemModel * model ) { m_model = model; }
40 QAbstractItemModel * model() const { return m_model; }
41
42 /* Size limit */
43 void setMaximumItemCount( int count ) { iMaxVisibleCount = count; }
44
45private:
46 QAbstractItemModel *m_model;
47
48 QAction *createActionFromIndex( QModelIndex index );
49
50 void build( const QModelIndex &parent );
51
52 int iMaxVisibleCount;
53
54private slots:
55 void rebuild();
56 void activate(QAction*);
57
58signals:
59 void activated( const QModelIndex & index );
60};
