54/*!
55 Class PictureFlow implements an image show widget with animation effect
56 like Apple's CoverFlow (in iTunes and iPod). Images are arranged in form
57 of slides, one main slide is shown at the center with few slides on
58 the left and right sides of the center slide. When the next or previous
59 slide is brought to the front, the whole slides flow to the right or
60 the right with smooth animation effect; until the new slide is finally
61 placed at the center.
62
63 */

192// for fixed-point arithmetic, we need minimum 32-bit long
193// long long (64-bit) might be useful for multiplication and division
194typedef long PFreal;
195#define PFREAL_SHIFT 10
196#define PFREAL_ONE (1 << PFREAL_SHIFT)
197
198#define IANGLE_MAX 1024
199#define IANGLE_MASK 1023
200
201inline PFreal fmul(PFreal a, PFreal b)
202{
203 return ((long long)(a))*((long long)(b)) >> PFREAL_SHIFT;
204}
205
206inline PFreal fdiv(PFreal num, PFreal den)
207{
208 long long p = (long long)(num) << (PFREAL_SHIFT * 2);
209 long long q = p / (long long)den;
210 long long r = q >> PFREAL_SHIFT;
211
212 return r;
213}
214
215inline PFreal fsin(int iangle)
216{
217 // warning: regenerate the table if IANGLE_MAX and PFREAL_SHIFT are changed!
218 static const PFreal tab[] = {
219 3, 103, 202, 300, 394, 485, 571, 652,
220 726, 793, 853, 904, 947, 980, 1004, 1019,
221 1023, 1018, 1003, 978, 944, 901, 849, 789,
222 721, 647, 566, 479, 388, 294, 196, 97,
223 -4, -104, -203, -301, -395, -486, -572, -653,
224 -727, -794, -854, -905, -948, -981, -1005, -1020,
225 -1024, -1019, -1004, -979, -945, -902, -850, -790,
226 -722, -648, -567, -480, -389, -295, -197, -98,
227 3
228 };
229
230 while (iangle < 0)
231 iangle += IANGLE_MAX;
232 iangle &= IANGLE_MASK;
233
234 int i = (iangle >> 4);
235 PFreal p = tab[i];
236 PFreal q = tab[(i+1)];
237 PFreal g = (q - p);
238 return p + g *(iangle - i*16) / 16;
239}
240
241inline PFreal fcos(int iangle)
242{
243 return fsin(iangle + (IANGLE_MAX >> 2));
244}
