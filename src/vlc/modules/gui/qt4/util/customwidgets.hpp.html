115/** This spinning icon, to the colors of the VLC cone, will show
116 * that there is some background activity running
117 **/
118class SpinningIcon : public QLabel
119{
120 Q_OBJECT
121
122public:
123 SpinningIcon( QWidget *parent );
124 void play( int loops = -1, int fps = 0 )
125 {
126 animator->setLoopCount( loops );
127 if ( fps ) animator->setFps( fps );
128 animator->start();
129 }
130 void stop() { animator->stop(); }
131 bool isPlaying() { return animator->state() == PixmapAnimator::Running; }
132private:
133 PixmapAnimator *animator;
134};
