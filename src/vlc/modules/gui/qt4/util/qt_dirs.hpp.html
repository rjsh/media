27#include <QString>
28#include <QDir>
29
30#define toNativeSeparators( s ) QDir::toNativeSeparators ( s )
31
32static inline QString removeTrailingSlash( QString s )
33{
34 if( ( s.length() > 1 ) && ( s[s.length()-1] == QLatin1Char( '/' ) ) )
35 s.remove( s.length() - 1, 1 );
36 return s;
37}
38
39#define savedirpathFromFile( a ) p_intf->p_sys->filepath = toNativeSeparators( QFileInfo( a ).path() )
40#define toNativeSepNoSlash( a ) toNativeSeparators( removeTrailingSlash( a ) )
41
42static inline QString colon_escape( QString s )
43{
44 return s.replace( ":", "\\:" );
45}
46static inline QString colon_unescape( QString s )
47{
48 return s.replace( "\\:", ":" ).trimmed();
49}
