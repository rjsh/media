102/* Define the Input used.
103 Add the callbacks on input
104 p_input is held once here */
105void InputManager::setInput( input_thread_t *_p_input )
106{
107 delInput();
108 p_input = _p_input;
109 if( p_input != NULL )
110 {
111 msg_Dbg( p_intf, "IM: Setting an input" );
112 vlc_object_hold( p_input );
113 addCallbacks();
114
115 UpdateStatus();
116 UpdateName();
117 UpdateArt();
118 UpdateTeletext();
119 UpdateNavigation();
120 UpdateVout();
121
122 p_item = input_GetItem( p_input );
123 emit rateChanged( var_GetFloat( p_input, "rate" ) );
124
125 /* Get Saved Time */
126 int i_time = RecentsMRL::getInstance( p_intf )->time( p_item->psz_uri );
127 if( i_time > 0 &&
128 !var_GetFloat( p_input, "run-time" ) &&
129 !var_GetFloat( p_input, "start-time" ) &&
130 !var_GetFloat( p_input, "stop-time" ) )
131 {
132 playlist_Pause( THEPL );
133
134 if( QMessageBox::question( NULL,
135 _("Continue playback?"),
136 _("Do you want to restart the playback where left off?"),
137 QMessageBox::Yes | QMessageBox::No, QMessageBox::Yes )
138 == QMessageBox::Yes )
139 var_SetTime( p_input, "time", (int64_t)i_time * 1000 );
140
141 playlist_Play( THEPL );
142 }
143 }
144 else
145 {
146 p_item = NULL;
147 assert( !p_input_vbi );
148 emit rateChanged( var_InheritFloat( p_intf, "rate" ) );
149 }
150}

152/* delete Input if it ever existed.
153 Delete the callbacls on input
154 p_input is released once here */
155void InputManager::delInput()
156{
157 if( !p_input ) return;
158 msg_Dbg( p_intf, "IM: Deleting the input" );
159
160 /* Save time / position */
161 float f_pos = var_GetFloat( p_input , "position" );
162 int64_t i_time = var_GetTime( p_input, "time");
163 int i_length = var_GetTime( p_input , "length" ) / CLOCK_FREQ;
164 if( f_pos < 0.05 || f_pos > 0.95 || i_length < 60) {
165 i_time = -1;
166 }
167 RecentsMRL::getInstance( p_intf )->setTime( p_item->psz_uri, i_time );
168
169 delCallbacks();
170 i_old_playing_status = END_S;
171 p_item = NULL;
172 oldName = "";
173 artUrl = "";
174 b_video = false;
175 timeA = 0;
176 timeB = 0;
177 f_rate = 0. ;
178
179 if( p_input_vbi )
180 {
181 vlc_object_release( p_input_vbi );
182 p_input_vbi = NULL;
183 }
184
185 vlc_object_release( p_input );
186 p_input = NULL;
187
188 emit positionUpdated( -1.0, 0 ,0 );
189 emit rateChanged( var_InheritFloat( p_intf, "rate" ) );
190 emit nameChanged( "" );
191 emit chapterChanged( 0 );
192 emit titleChanged( 0 );
193 emit playingStatusChanged( END_S );
194
195 emit teletextPossible( false );
196 emit AtoBchanged( false, false );
197 emit voutChanged( false );
198 emit voutListChanged( NULL, 0 );
199
200 /* Reset all InfoPanels but stats */
201 emit artChanged( NULL );
202 emit artChanged( "" );
203 emit infoChanged( NULL );
204 emit currentMetaChanged( (input_item_t *)NULL );
205
206 emit encryptionChanged( false );
207 emit recordingStateChanged( false );
208
209 emit cachingChanged( 1 );
210}

212/* Convert the event from the callbacks in actions */
213void InputManager::customEvent( QEvent *event )
214{
215 int i_type = event->type();
216 IMEvent *ple = static_cast<IMEvent *>(event);
217
218 if( i_type == IMEvent::ItemChanged )
219 UpdateMeta( ple->item() );
220
221 if( !hasInput() )
222 return;
223
224 /* Actions */
225 switch( i_type )
226 {
227 case IMEvent::PositionUpdate:
228 UpdatePosition();
229 break;
230 case IMEvent::StatisticsUpdate:
231 UpdateStats();
232 break;
233 case IMEvent::ItemChanged:
234 /* Ignore ItemChanged_Type event that does not apply to our input */
235 if( p_item == ple->item() )
236 {
237 UpdateStatus();
238 // UpdateName();
239 UpdateArt();
240 UpdateMeta();
241 /* Update duration of file */
242 }
243 break;
244 case IMEvent::ItemStateChanged:
245 UpdateStatus();
246 break;
247 case IMEvent::NameChanged:
248 UpdateName();
249 break;
250 case IMEvent::MetaChanged:
251 UpdateMeta();
252 UpdateName(); /* Needed for NowPlaying */
253 UpdateArt(); /* Art is part of meta in the core */
254 break;
255 case IMEvent::InfoChanged:
256 UpdateInfo();
257 break;
258 case IMEvent::ItemTitleChanged:
259 UpdateNavigation();
260 UpdateName(); /* Display the name of the Chapter, if exists */
261 break;
262 case IMEvent::ItemRateChanged:
263 UpdateRate();
264 break;
265 case IMEvent::ItemEsChanged:
266 UpdateTeletext();
267 // We don't do anything ES related. Why ?
268 break;
269 case IMEvent::ItemTeletextChanged:
270 UpdateTeletext();
271 break;
272 case IMEvent::InterfaceVoutUpdate:
273 UpdateVout();
274 break;
275 case IMEvent::SynchroChanged:
276 emit synchroChanged();
277 break;
278 case IMEvent::CachingEvent:
279 UpdateCaching();
280 break;
281 case IMEvent::BookmarksChanged:
282 emit bookmarksChanged();
283 break;
284 case IMEvent::InterfaceAoutUpdate:
285 UpdateAout();
286 break;
287 case IMEvent::RecordingEvent:
288 UpdateRecord();
289 break;
290 case IMEvent::ProgramChanged:
291 UpdateProgramEvent();
292 break;
293 case IMEvent::EPGEvent:
294 UpdateEPG();
295 break;
296 default:
297 msg_Warn( p_intf, "This shouldn't happen: %i", i_type );
298 assert(0);
299 }
300}

315static int ItemChanged( vlc_object_t *p_this, const char *psz_var,
316 vlc_value_t oldval, vlc_value_t newval, void *param )
317{
318 VLC_UNUSED( p_this ); VLC_UNUSED( psz_var ); VLC_UNUSED( oldval );
319
320 InputManager *im = (InputManager*)param;
321 input_item_t *p_item = static_cast<input_item_t *>(newval.p_address);
322
323 IMEvent *event = new IMEvent( IMEvent::ItemChanged, p_item );
324 QApplication::postEvent( im, event );
325 return VLC_SUCCESS;
326}

328static int InputEvent( vlc_object_t *p_this, const char *,
329 vlc_value_t, vlc_value_t newval, void *param )
330{
331 VLC_UNUSED( p_this );
332
333 InputManager *im = (InputManager*)param;
334 IMEvent *event;
335
336 switch( newval.i_int )
337 {
338 case INPUT_EVENT_STATE:
339 event = new IMEvent( IMEvent::ItemStateChanged );
340 break;
341 case INPUT_EVENT_RATE:
342 event = new IMEvent( IMEvent::ItemRateChanged );
343 break;
344 case INPUT_EVENT_POSITION:
345 //case INPUT_EVENT_LENGTH:
346 event = new IMEvent( IMEvent::PositionUpdate );
347 break;
348
349 case INPUT_EVENT_TITLE:
350 case INPUT_EVENT_CHAPTER:
351 event = new IMEvent( IMEvent::ItemTitleChanged );
352 break;
353
354 case INPUT_EVENT_ES:
355 event = new IMEvent( IMEvent::ItemEsChanged );
356 break;
357 case INPUT_EVENT_TELETEXT:
358 event = new IMEvent( IMEvent::ItemTeletextChanged );
359 break;
360
361 case INPUT_EVENT_STATISTICS:
362 event = new IMEvent( IMEvent::StatisticsUpdate );
363 break;
364
365 case INPUT_EVENT_VOUT:
366 event = new IMEvent( IMEvent::InterfaceVoutUpdate );
367 break;
368 case INPUT_EVENT_AOUT:
369 event = new IMEvent( IMEvent::InterfaceAoutUpdate );
370 break;
371
372 case INPUT_EVENT_ITEM_META: /* Codec MetaData + Art */
373 event = new IMEvent( IMEvent::MetaChanged );
374 break;
375 case INPUT_EVENT_ITEM_INFO: /* Codec Info */
376 event = new IMEvent( IMEvent::InfoChanged );
377 break;
378 case INPUT_EVENT_ITEM_NAME:
379 event = new IMEvent( IMEvent::NameChanged );
380 break;
381
382 case INPUT_EVENT_AUDIO_DELAY:
383 case INPUT_EVENT_SUBTITLE_DELAY:
384 event = new IMEvent( IMEvent::SynchroChanged );
385 break;
386
387 case INPUT_EVENT_CACHE:
388 event = new IMEvent( IMEvent::CachingEvent );
389 break;
390
391 case INPUT_EVENT_BOOKMARK:
392 event = new IMEvent( IMEvent::BookmarksChanged );
393 break;
394
395 case INPUT_EVENT_RECORD:
396 event = new IMEvent( IMEvent::RecordingEvent );
397 break;
398
399 case INPUT_EVENT_PROGRAM:
400 /* This is for PID changes */
401 event = new IMEvent( IMEvent::ProgramChanged );
402 break;
403
404 case INPUT_EVENT_ITEM_EPG:
405 /* EPG data changed */
406 event = new IMEvent( IMEvent::EPGEvent );
407 break;
408
409 case INPUT_EVENT_SIGNAL:
410 /* This is for capture-card signals */
411 /* event = new IMEvent( SignalChanged_Type );
412 break; */
413 default:
414 event = NULL;
415 break;
416 }
417
418 if( event )
419 QApplication::postEvent( im, event );
420 return VLC_SUCCESS;
421}

494void InputManager::UpdateName()
495{
496 assert( p_input );
497
498 /* Update text, name and nowplaying */
499 QString name;
500
501 /* Try to get the nowplaying */
502 char *format = var_InheritString( p_intf, "input-title-format" );
503 char *formated = str_format_meta( p_input, format );
504 free( format );
505 name = qfu(formated);
506 free( formated );
507
508 /* If we have Nothing */
509 if( name.simplified().isEmpty() )
510 {
511 char *uri = input_item_GetURI( input_GetItem( p_input ) );
512 char *file = uri ? strrchr( uri, '/' ) : NULL;
513 if( file != NULL )
514 {
515 decode_URI( ++file );
516 name = qfu(file);
517 }
518 else
519 name = qfu(uri);
520 free( uri );
521 }
522
523 name = name.trimmed();
524
525 if( oldName != name )
526 {
527 emit nameChanged( name );
528 oldName = name;
529 }
530}

567void InputManager::UpdateTeletext()
568{
569 if( hasInput() )
570 {
571 const bool b_enabled = var_CountChoices( p_input, "teletext-es" ) > 0;
572 const int i_teletext_es = var_GetInteger( p_input, "teletext-es" );
573
574 /* Teletext is possible. Show the buttons */
575 emit teletextPossible( b_enabled );
576
577 /* If Teletext is selected */
578 if( b_enabled && i_teletext_es >= 0 )
579 {
580 /* Then, find the current page */
581 int i_page = 100;
582 bool b_transparent = false;
583
584 if( p_input_vbi )
585 {
586 var_DelCallback( p_input_vbi, "vbi-page", VbiEvent, this );
587 vlc_object_release( p_input_vbi );
588 }
589
590 if( input_GetEsObjects( p_input, i_teletext_es, &p_input_vbi, NULL, NULL ) )
591 p_input_vbi = NULL;
592
593 if( p_input_vbi )
594 {
595 /* This callback is not remove explicitly, but interfaces
596 * are guaranted to outlive input */
597 var_AddCallback( p_input_vbi, "vbi-page", VbiEvent, this );
598
599 i_page = var_GetInteger( p_input_vbi, "vbi-page" );
600 b_transparent = !var_GetBool( p_input_vbi, "vbi-opaque" );
601 }
602 emit newTelexPageSet( i_page );
603 emit teletextTransparencyActivated( b_transparent );
604
605 }
606 emit teletextActivated( b_enabled && i_teletext_es >= 0 );
607 }
608 else
609 {
610 emit teletextActivated( false );
611 emit teletextPossible( false );
612 }
613}

671void InputManager::requestArtUpdate( input_item_t *p_item, bool b_forced )
672{
673 bool b_current_item = false;
674 if ( !p_item && hasInput() )
675 { /* default to current item */
676 p_item = input_GetItem( p_input );
677 b_current_item = true;
678 }
679
680 if ( p_item )
681 {
682 /* check if it has already been enqueued */
683 if ( p_item->p_meta && !b_forced )
684 {
685 int status = vlc_meta_GetStatus( p_item->p_meta );
686 if ( status & ( ITEM_ART_NOTFOUND|ITEM_ART_FETCHED ) )
687 return;
688 }
689 libvlc_ArtRequest( p_intf->p_libvlc, p_item,
690 (b_forced) ? META_REQUEST_OPTION_SCOPE_ANY
691 : META_REQUEST_OPTION_NONE );
692 /* No input will signal the cover art to update,
693 * let's do it ourself */
694 if ( b_current_item )
695 UpdateArt();
696 else
697 emit artChanged( p_item );
698 }
699}

998MainInputManager::MainInputManager( intf_thread_t *_p_intf )

1003{

1005 im = new InputManager( this, p_intf );
1006
1007 var_AddCallback( THEPL, "item-change", ItemChanged, im );
1008 var_AddCallback( THEPL, "activity", PLItemChanged, this );

1010 var_AddCallback( THEPL, "playlist-item-append", PLItemAppended, this );

1012 random.addCallback( this, SLOT(notifyRandom(bool)) );
1013 repeat.addCallback( this, SLOT(notifyRepeatLoop(bool)) );
1014 loop.addCallback( this, SLOT(notifyRepeatLoop(bool)) );
1015
1016 volume.addCallback( this, SLOT(notifyVolume(float)) );
1017 mute.addCallback( this, SLOT(notifyMute(bool)) );
1018
1019 /* Warn our embedded IM about input changes */
1020 DCONNECT( this, inputChanged( input_thread_t * ),
1021 im, setInput( input_thread_t * ) );
1022
1023 /* initialize p_input (an input can already be running) */
1024 p_input = playlist_CurrentInput( THEPL );
1025 if( p_input )
1026 emit inputChanged( p_input );
1027
1028 /* Audio Menu */
1029 menusAudioMapper = new QSignalMapper();
1030 CONNECT( menusAudioMapper, mapped(QString), this, menusUpdateAudio( QString ) );
1031}

1061void MainInputManager::customEvent( QEvent *event )
1062{
1063 int type = event->type();
1064
1065 PLEvent *plEv;
1066
1067 // msg_Dbg( p_intf, "New MainIM Event of type: %i", type );
1068 switch( type )
1069 {
1070 case PLEvent::PLItemAppended:
1071 plEv = static_cast<PLEvent*>( event );
1072 emit playlistItemAppended( plEv->getItemId(), plEv->getParentId() );
1073 return;
1074 case PLEvent::PLItemRemoved:
1075 plEv = static_cast<PLEvent*>( event );
1076 emit playlistItemRemoved( plEv->getItemId() );
1077 return;
1078 case PLEvent::PLEmpty:
1079 plEv = static_cast<PLEvent*>( event );
1080 emit playlistNotEmpty( plEv->getItemId() >= 0 );
1081 return;
1082 case PLEvent::LeafToParent:
1083 plEv = static_cast<PLEvent*>( event );
1084 emit leafBecameParent( plEv->getItemId() );
1085 return;
1086 default:
1087 if( type != IMEvent::ItemChanged ) return;
1088 }
1089
1090 if( p_input != NULL )
1091 vlc_object_release( p_input );
1092 p_input = playlist_CurrentInput( THEPL );
1093 emit inputChanged( p_input );
1094}
