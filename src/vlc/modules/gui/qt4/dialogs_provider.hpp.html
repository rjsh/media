64class DialogsProvider : public QObject
65{
66 Q_OBJECT
67 friend class VLCMenuBar;
68
69public:
70 static DialogsProvider *getInstance()
71 {
72 assert( instance );
73 return instance;
74 }
75 static DialogsProvider *getInstance( intf_thread_t *p_intf )
76 {
77 if( !instance )
78 instance = new DialogsProvider( p_intf );
79 return instance;
80 }
81 static void killInstance()
82 {
83 delete instance;
84 instance = NULL;
85 }

174class DialogEvent : public QEvent
175{
176public:
177 static const QEvent::Type DialogEvent_Type;
178 DialogEvent( int _i_dialog, int _i_arg, intf_dialog_args_t *_p_arg ) :
179 QEvent( DialogEvent_Type )
180 {
181 i_dialog = _i_dialog;
182 i_arg = _i_arg;
183 p_arg = _p_arg;
184 }
185
186 int i_arg, i_dialog;
