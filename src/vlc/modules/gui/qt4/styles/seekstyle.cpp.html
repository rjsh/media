47int SeekStyle::pixelMetric( PixelMetric metric, const QStyleOption *option, const QWidget *widget ) const
48{
49 const QStyleOptionSlider *slider;
50
51 if ( widget && ( slider = qstyleoption_cast<const QStyleOptionSlider *>( option ) ) )
52 {
53 switch( metric )
54 {
55 case QStyle::PM_SliderThickness:
56 case QStyle::PM_SliderLength:
57 return widget->minimumSize().height();
58 default:
59 break;
60 }
61 }
62
63 return QProxyStyle::pixelMetric( metric, option, widget );
64}

66void SeekStyle::drawComplexControl( ComplexControl cc, const QStyleOptionComplex *option, QPainter *painter, const QWidget *widget ) const
67{
68 if( cc == CC_Slider )
69 {
70 painter->setRenderHints( QPainter::Antialiasing );
71
72 if ( const SeekStyle::SeekStyleOption *slideroptions =
73 qstyleoption_cast<const SeekStyle::SeekStyleOption *>( option ) )
74 {
75 qreal sliderPos = -1;
76
77 /* Get the needed subcontrols to draw the slider */
78 QRect groove = subControlRect(CC_Slider, slideroptions, SC_SliderGroove, widget);
79 QRect handle = subControlRect(CC_Slider, slideroptions, SC_SliderHandle, widget);
80
81 /* Adjust the size of the groove so the handle stays centered */
82 groove.adjust( handle.width() / 2, 0, -handle.width() / 2, 0 );
83
84 /* Reduce the height of the groove */
85 // Note: in the original 2.0.0 code, the groove had a height of 9px and to
86 // comply with the original style (but still allow the widget to expand) I
87 // had to remove 1 from the rect bottom.
88 groove.adjust( 0, (qreal)groove.height() / 3.7, 0, (qreal)-groove.height() / 3.7 - 1 );
89
90 if ( ( slideroptions->subControls & SC_SliderGroove ) && groove.isValid() )
91 {
92 sliderPos = ( ( (qreal)groove.width() ) / (qreal)slideroptions->maximum )
93 * (qreal)slideroptions->sliderPosition;
94
95 /* set the background color and gradient */
96 QColor backgroundBase( slideroptions->palette.window().color() );
97 QLinearGradient backgroundGradient( 0, 0, 0, slideroptions->rect.height() );
98 backgroundGradient.setColorAt( 0.0, backgroundBase.darker( 140 ) );
99 backgroundGradient.setColorAt( 1.0, backgroundBase );
100
101 /* set the foreground color and gradient */
102 QColor foregroundBase( 50, 156, 255 );
103 QLinearGradient foregroundGradient( 0, 0, 0, groove.height() );
104 foregroundGradient.setColorAt( 0.0, foregroundBase );
105 foregroundGradient.setColorAt( 1.0, foregroundBase.darker( 125 ) );
106
107 /* draw a slight 3d effect on the bottom */
108 painter->setPen( QColor( 230, 230, 230 ) );
109 painter->setBrush( Qt::NoBrush );
110 painter->drawRoundedRect( groove.adjusted( 0, 2, 0, 0 ), RADIUS, RADIUS );
111
112 /* draw background */
113 painter->setPen( Qt::NoPen );
114 painter->setBrush( backgroundGradient );
115 painter->drawRoundedRect( groove, RADIUS, RADIUS );
116
117 /* adjusted foreground rectangle */
118 QRect valueRect = groove.adjusted( 1, 1, -1, 0 );
119
120 valueRect.setWidth( sliderPos );
121
122 /* draw foreground */
123 if ( slideroptions->sliderPosition > slideroptions->minimum && slideroptions->sliderPosition <= slideroptions->maximum )
124 {
125 painter->setPen( Qt::NoPen );
126 painter->setBrush( foregroundGradient );
127 painter->drawRoundedRect( valueRect, RADIUS, RADIUS );
128 }
129
130 /* draw buffering overlay */
131 if ( slideroptions->buffering < 1.0 )
132 {
133 QRect innerRect = groove.adjusted( 1, 1,
134 groove.width() * ( -1.0 + slideroptions->buffering ) - 1, 0 );
135 QColor overlayColor = QColor( "Orange" );
136 overlayColor.setAlpha( 128 );
137 painter->setBrush( overlayColor );
138 painter->drawRoundedRect( innerRect, RADIUS, RADIUS );
139 }
140 }
141
142 if ( slideroptions->subControls & SC_SliderTickmarks ) {
143 QStyleOptionSlider tmpSlider = *slideroptions;
144 tmpSlider.subControls = SC_SliderTickmarks;
145 QProxyStyle::drawComplexControl(cc, &tmpSlider, painter, widget);
146 }
147
148 if ( slideroptions->subControls & SC_SliderHandle && handle.isValid() )
149 {
150 /* Useful for debugging */
151 //painter->setBrush( QColor( 0, 0, 255, 150 ) );
152 //painter->drawRect( handle );
153
154 if ( option->state & QStyle::State_MouseOver || slideroptions->animate )
155 {
156 QPalette p = slideroptions->palette;
157
158 /* draw chapters tickpoints */
159 if ( slideroptions->points.size() && slideroptions->length && groove.width() )
160 {
161 QColor background = p.color( QPalette::Active, QPalette::Window );
162 QColor foreground = p.color( QPalette::Active, QPalette::WindowText );
163 foreground.setHsv( foreground.hue(),
164 ( background.saturation() + foreground.saturation() ) / 2,
165 ( background.value() + foreground.value() ) / 2 );
166 if ( slideroptions->orientation == Qt::Horizontal ) /* TODO: vertical */
167 {
168 foreach( int64_t time, slideroptions->points )
169 {
170 int x = groove.x() + time / 1000000.0 / slideroptions->length * groove.width();
171 painter->setPen( foreground );
172 painter->setBrush( Qt::NoBrush );
173 painter->drawLine( x, slideroptions->rect.height(), x, slideroptions->rect.height() - CHAPTERSSPOTSIZE );
174 }
175 }
176 }
177
178 /* draw handle */
179 if ( option->state & QStyle::State_Enabled && sliderPos != -1 )
180 {
181 QSize hSize = QSize( handle.height(), handle.height() ) - QSize( 6, 6 );;
182 QPoint pos = QPoint( handle.center().x() - ( hSize.width() / 2 ), handle.center().y() - ( hSize.height() / 2 ) );
183
184 QPoint shadowPos( pos - QPoint( 2, 2 ) );
185 QSize sSize( QSize( handle.height(), handle.height() ) - QSize( 2, 2 ) );
186
187 /* prepare the handle's gradient */
188 QLinearGradient handleGradient( 0, 0, 0, hSize.height() );
189 handleGradient.setColorAt( 0.0, p.window().color().lighter( 120 ) );
190 handleGradient.setColorAt( 0.9, p.window().color().darker( 120 ) );
191
192 /* prepare the handle's shadow gradient */
193 QColor shadowBase = p.shadow().color();
194 if( shadowBase.lightness() > 100 )
195 shadowBase = QColor( 60, 60, 60 ); // Palette's shadow is too bright
196 QColor shadowDark( shadowBase.darker( 150 ) );
197 QColor shadowLight( shadowBase.lighter( 180 ) );
198 shadowLight.setAlpha( 50 );
199
200 QRadialGradient shadowGradient( shadowPos.x() + ( sSize.width() / 2 ),
201 shadowPos.y() + ( sSize.height() / 2 ),
202 qMax( sSize.width(), sSize.height() ) / 2 );
203 shadowGradient.setColorAt( 0.4, shadowDark );
204 shadowGradient.setColorAt( 1.0, shadowLight );
205
206 painter->setPen( Qt::NoPen );
207 painter->setOpacity( slideroptions->animationopacity );
208
209 /* draw the handle's shadow */
210 painter->setBrush( shadowGradient );
211 painter->drawEllipse( shadowPos.x(), shadowPos.y() + 1, sSize.width(), sSize.height() );
212
213 /* finally draw the handle */
214 painter->setBrush( handleGradient );
215 painter->drawEllipse( pos.x(), pos.y(), hSize.width(), hSize.height() );
216 }
217 }
218 }
219 }
220 }
221 else
222 {
223 qWarning() << "SeekStyle: Drawing an unmanaged control";
224 QProxyStyle::drawComplexControl( cc, option, painter, widget );
225 }
