100#define CONNECT( a, b, c, d ) \
101 connect( a, SIGNAL(b), c, SLOT(d) )
102#define DCONNECT( a, b, c, d ) \
103 connect( a, SIGNAL(b), c, SLOT(d), Qt::DirectConnection )
104#define BUTTONACT( b, a ) connect( b, SIGNAL(clicked()), this, SLOT(a) )
105
106#define BUTTON_SET( button, text, tooltip ) \
107 button->setText( text ); \
108 button->setToolTip( tooltip );
109
110#define BUTTON_SET_ACT( button, text, tooltip, thisslot ) \
111 BUTTON_SET( button, text, tooltip ); \
112 BUTTONACT( button, thisslot );
113
114#define BUTTON_SET_IMG( button, text, image, tooltip ) \
115 BUTTON_SET( button, text, tooltip ); \
116 button->setIcon( QIcon( ":/"#image ) );
117
118#define BUTTON_SET_ACT_I( button, text, image, tooltip, thisslot ) \
119 BUTTON_SET_IMG( button, text, image, tooltip ); \
120 BUTTONACT( button, thisslot );

132static inline QString QVLCUserDir( vlc_userdir_t type )
133{
134 char *dir = config_GetUserDir( type );
135 if( !dir )
136 return "";
137 QString res = qfu( dir );
138 free( dir );
139 return res;
140}
