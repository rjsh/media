29Chromaprint::Chromaprint( intf_thread_t *_p_intf ) : p_intf( _p_intf )
30{
31 Q_ASSERT( p_intf );
32 p_fingerprinter = fingerprinter_Create( VLC_OBJECT( p_intf ) );
33 if ( p_fingerprinter )
34 var_AddCallback( p_fingerprinter, "results-available", results_available, this );
35}
36
37int Chromaprint::results_available( vlc_object_t *, const char *, vlc_value_t, vlc_value_t , void *param )
38{
39 Chromaprint *me = (Chromaprint *) param;
40 me->finish();
41 return 0;
42}
43
44fingerprint_request_t * Chromaprint::fetchResults()
45{
46 return p_fingerprinter->pf_getresults( p_fingerprinter );
47}
48
49void Chromaprint::apply( fingerprint_request_t *p_r, int i_id )
50{
51 p_fingerprinter->pf_apply( p_r, i_id );
52}
53
54bool Chromaprint::enqueue( input_item_t *p_item )
55{
56 if ( ! p_fingerprinter ) return false;
57 fingerprint_request_t *p_r = fingerprint_request_New( p_item );
58 if ( ! p_r ) return false;
59 mtime_t t = input_item_GetDuration( p_item );
60 if ( t ) p_r->i_duration = (unsigned int) ( t / 1000000 );
61 p_fingerprinter->pf_enqueue( p_fingerprinter, p_r );
62 return true;
63}
64
65bool Chromaprint::isSupported( QString uri )
66{
67 if ( !module_exists( "stream_out_chromaprint" ) )
68 return false;
69 else
70 return ( uri.startsWith( "file://" ) || uri.startsWith( "/" ) );
71}
72
73Chromaprint::~Chromaprint()
74{
75 if ( p_fingerprinter )
76 fingerprinter_Destroy( p_fingerprinter );
77}
