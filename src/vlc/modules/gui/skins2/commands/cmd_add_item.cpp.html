34void CmdAddItem::execute()
35{
36 playlist_t *pPlaylist = getIntf()->p_sys->p_playlist;
37 if( !pPlaylist )
38 return;
39
40 if( strstr( m_name.c_str(), "://" ) == NULL )
41 {
42 char *psz_uri = vlc_path2uri( m_name.c_str(), NULL );
43 if( !psz_uri )
44 return;
45 m_name = psz_uri;
46 free( psz_uri );
47 }
48 playlist_Add( pPlaylist, m_name.c_str(), NULL,
49 m_playNow ? PLAYLIST_APPEND | PLAYLIST_GO : PLAYLIST_APPEND,
50 PLAYLIST_END, true, false );
51}
