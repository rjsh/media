30void CmdPlay::execute()
31{
32 playlist_t *pPlaylist = getIntf()->p_sys->p_playlist;
33 if( pPlaylist == NULL )
34 return;
35
36 // if already playing an input, reset rate to normal speed
37 input_thread_t *pInput = playlist_CurrentInput( pPlaylist );
38 if( pInput )
39 {
40 var_SetFloat( pPlaylist, "rate", 1.0 );
41 vlc_object_release( pInput );
42 }
43
44 playlist_Lock( pPlaylist );
45 const bool b_empty = playlist_IsEmpty( pPlaylist );
46 playlist_Unlock( pPlaylist );
47
48 if( !b_empty )
49 {
50 playlist_Play( pPlaylist );
51 }
52 else
53 {
54 // If the playlist is empty, open a file requester instead
55 CmdDlgFile( getIntf() ).execute();
56 }
57}

76void CmdSlower::execute()
77{
78 playlist_t *pPlaylist = getIntf()->p_sys->p_playlist;
79 input_thread_t *pInput = playlist_CurrentInput( pPlaylist );
80
81 if( pInput )
82 {
83 var_TriggerCallback( pPlaylist, "rate-slower" );
84 vlc_object_release( pInput );
85 }
