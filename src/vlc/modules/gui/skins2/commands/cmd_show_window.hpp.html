35/// Command to show a window
36class CmdShowWindow: public CmdGeneric
37{
38public:
39 CmdShowWindow( intf_thread_t *pIntf, WindowManager &rWinManager,
40 TopWindow &rWin ):
41 CmdGeneric( pIntf ), m_rWinManager( rWinManager ), m_rWin( rWin ) { }
42 virtual ~CmdShowWindow() { }
43 virtual void execute() { m_rWinManager.show( m_rWin ); }
44 virtual string getType() const { return "show window"; }
45
46private:
47 /// Reference to the window manager
48 WindowManager &m_rWinManager;
49 /// Reference to the window
50 TopWindow &m_rWin;
51};

89/// Command to show a popup menu
90class CmdShowPopup: public CmdGeneric
91{
92public:
93 CmdShowPopup( intf_thread_t *pIntf, Popup &rPopup ):
94 CmdGeneric( pIntf ), m_rPopup( rPopup ) { }
95 virtual ~CmdShowPopup() { }
96
97 virtual void execute()
98 {
99 int x, y;
100 OSFactory::instance( getIntf() )->getMousePos( x, y );
101 m_rPopup.show( x, y );
102 }
103
104 virtual string getType() const { return "show popup"; }
105
106private:
107 /// Reference to the popup
108 Popup &m_rPopup;
109};
