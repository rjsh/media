32/// "Add item" command
33class CmdAddItem: public CmdGeneric
34{
35public:
36 CmdAddItem( intf_thread_t *pIntf, const string &rName, bool playNow )
37 : CmdGeneric( pIntf ), m_name( rName ), m_playNow( playNow ) { }
38 virtual ~CmdAddItem() { }
39 virtual void execute();
40 virtual string getType() const { return "add item"; }
41
42private:
43 /// Name of the item to enqueue
44 string m_name;
45 /// Should we play the item immediately?
46 bool m_playNow;
47};
