25#ifndef ASYNC_QUEUE_HPP
26#define ASYNC_QUEUE_HPP
27
28#include "cmd_generic.hpp"
29
30#include <list>
31#include <string>
32
33class OSTimer;
34
35
36/// Asynchronous queue for commands
37class AsyncQueue: public SkinObject
38{
39public:
40 /// Get the instance of AsyncQueue
41 /// Returns NULL if initialization failed.
42 static AsyncQueue *instance( intf_thread_t *pIntf );
43
44 /// Destroy the instance of AsyncQueue
45 static void destroy( intf_thread_t *pIntf );
46
47 /// Add a command in the queue, after having removed the commands
48 /// of the same type already in the queue if needed
49 void push( const CmdGenericPtr &rcCommand, bool removePrev = true );
50
51 /// Remove the commands of the given type
52 void remove( const string &rType , const CmdGenericPtr &rcCommand );
53
54 /// Flush the queue and execute the commands
55 void flush();
56
57private:
58 /// Command queue
59 typedef std::list<CmdGenericPtr> cmdList_t;
60 cmdList_t m_cmdList;
61 /// Timer
62 OSTimer *m_pTimer;
63 /// Mutex
64 vlc_mutex_t m_lock;
65
66 // Private because it is a singleton
67 AsyncQueue( intf_thread_t *pIntf );
68 virtual ~AsyncQueue();
69
70 // Callback to flush the queue
71 DEFINE_CALLBACK( AsyncQueue, Flush );
72};
73
74
75#endif
