38void CmdQuit::execute()
39{
40 if( getIntf()->p_sys->p_input )
41 {
42 vout_thread_t *pVout = input_GetVout( getIntf()->p_sys->p_input );
43 if( pVout )
44 {
45 vout_OSDMessage( pVout, SPU_DEFAULT_CHANNEL, "%s", _( "Quit" ) );
46 vlc_object_release( pVout );
47 }
48 }
49
50 // Kill libvlc
51 libvlc_Quit( getIntf()->p_libvlc );
52}
53
54
55void CmdExitLoop::execute()
56{
57 // Get the instance of OSFactory
58 OSFactory *pOsFactory = OSFactory::instance( getIntf() );
59
60 // Exit the main OS loop
61 pOsFactory->getOSLoop()->exit();
62}
