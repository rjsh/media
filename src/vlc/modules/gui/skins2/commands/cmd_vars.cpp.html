31void CmdItemUpdate::execute()
32{
33 if( !m_pItem )
34 return;
35
36 // update playtree
37 playlist_t* pPlaylist = getIntf()->p_sys->p_playlist;
38 playlist_Lock( pPlaylist );
39 playlist_item_t* p_plItem = playlist_ItemGetByInput( pPlaylist, m_pItem );
40 int id = p_plItem ? p_plItem->i_id : 0;
41 playlist_Unlock( pPlaylist );
42
43 if( id )
44 VlcProc::instance( getIntf() )->getPlaytreeVar().onUpdateItem( id );
45
46 // update current input if needed
47 input_item_t* p_current = NULL;
48 input_thread_t* pInput = getIntf()->p_sys->p_input;
49 if( pInput )
50 p_current = input_GetItem( pInput );
51
52 if( p_current == m_pItem )
53 VlcProc::instance( getIntf() )->update_current_input();
54}
55
56bool CmdItemUpdate::checkRemove( CmdGeneric *pQueuedCommand ) const
57{
58 // We don't use RTTI - Use C-style cast
59 CmdItemUpdate *pUpdateCommand = (CmdItemUpdate *)(pQueuedCommand);
60 return m_pItem == pUpdateCommand->m_pItem;
61}
