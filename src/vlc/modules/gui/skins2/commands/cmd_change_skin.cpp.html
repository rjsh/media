37void CmdChangeSkin::execute()
38{
39 // Save the old theme to restore it in case of problem
40 Theme *pOldTheme = getIntf()->p_sys->p_theme;
41
42 if( pOldTheme )
43 {
44 pOldTheme->getWindowManager().saveVisibility();
45 pOldTheme->getWindowManager().hideAll();
46 }
47
48 VoutManager::instance( getIntf() )->saveVoutConfig();
49
50 ThemeLoader loader( getIntf() );
51 if( loader.load( m_file ) )
52 {
53 // Everything went well
54 msg_Info( getIntf(), "new theme successfully loaded (%s)",
55 m_file.c_str() );
56 delete pOldTheme;
57
58 // restore vout config
59 VoutManager::instance( getIntf() )->restoreVoutConfig( true );
60 }
61 else if( pOldTheme )
62 {
63 msg_Warn( getIntf(), "a problem occurred when loading the new theme,"
64 " restoring the previous one" );
65 getIntf()->p_sys->p_theme = pOldTheme;
66 VoutManager::instance( getIntf() )->restoreVoutConfig( false );
67 pOldTheme->getWindowManager().restoreVisibility();
68 }
69 else
70 {
71 msg_Err( getIntf(), "cannot load the theme, aborting" );
72 // Quit
73 CmdQuit cmd( getIntf() );
74 cmd.execute();
75 }
76
77 // update the repository
78 ThemeRepository::instance( getIntf() )->updateRepository();
79}
