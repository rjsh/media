25#ifndef VAR_BOOL_HPP
26#define VAR_BOOL_HPP
27
28#include "variable.hpp"
29#include "observer.hpp"
30
31
32/// Interface for read-only boolean variable
33class VarBool: public Variable, public Subject<VarBool>
34{
35public:
36 /// Get the variable type
37 virtual const string &getType() const { return m_type; }
38
39 /// Get the boolean value
40 virtual bool get() const = 0;
41
42protected:
43 VarBool( intf_thread_t *pIntf ): Variable( pIntf ) { }
44 virtual ~VarBool() { }
45
46private:
47 /// Variable type
48 static const string m_type;
49};
50
51
52/// Constant true VarBool
53class VarBoolTrue: public VarBool
54{
55public:
56 VarBoolTrue( intf_thread_t *pIntf ): VarBool( pIntf ) { }
57 virtual ~VarBoolTrue() { }
58 virtual bool get() const { return true; }
59};
60
61
62/// Constant false VarBool
63class VarBoolFalse: public VarBool
64{
65public:
66 VarBoolFalse( intf_thread_t *pIntf ): VarBool( pIntf ) { }
67 virtual ~VarBoolFalse() { }
68 virtual bool get() const { return false; }
69};
70
71
72/// Boolean variable implementation (read/write)
73class VarBoolImpl: public VarBool
74{
75public:
76 VarBoolImpl( intf_thread_t *pIntf );
77 virtual ~VarBoolImpl() { }
78
79 // Get the boolean value
80 virtual bool get() const { return m_value; }
81
82 /// Set the internal value
83 virtual void set( bool value );
84
85private:
86 /// Boolean value
87 bool m_value;
88};
89
90
91/// Conjunction of two boolean variables (AND)
92class VarBoolAndBool: public VarBool, public Observer<VarBool>
93{
94public:
95 VarBoolAndBool( intf_thread_t *pIntf, VarBool &rVar1, VarBool &rVar2 );
96 virtual ~VarBoolAndBool();
97 virtual bool get() const { return m_value; }
98
99 // Called when one of the observed variables is changed
100 void onUpdate( Subject<VarBool> &rVariable, void* );
101
102private:
103 /// Boolean variables
104 VarBool &m_rVar1, &m_rVar2;
105 bool m_value;
106};
107
108
109/// Disjunction of two boolean variables (OR)
110class VarBoolOrBool: public VarBool, public Observer<VarBool>
111{
112public:
113 VarBoolOrBool( intf_thread_t *pIntf, VarBool &rVar1, VarBool &rVar2 );
114 virtual ~VarBoolOrBool();
115 virtual bool get() const { return m_value; }
116
117 // Called when one of the observed variables is changed
118 void onUpdate( Subject<VarBool> &rVariable, void* );
119
120private:
121 /// Boolean variables
122 VarBool &m_rVar1, &m_rVar2;
123 bool m_value;
124};
125
126
127/// Negation of a boolean variable (NOT)
128class VarNotBool: public VarBool, public Observer<VarBool>
129{
130public:
131 VarNotBool( intf_thread_t *pIntf, VarBool &rVar );
132 virtual ~VarNotBool();
133 virtual bool get() const { return !m_rVar.get(); }
134
135 // Called when the observed variable is changed
136 void onUpdate( Subject<VarBool> &rVariable, void* );
137
138private:
139 /// Boolean variable
140 VarBool &m_rVar;
141};
