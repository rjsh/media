25#include "var_percent.hpp"
26
27
28const string VarPercent::m_type = "percent";
29
30
31void VarPercent::set( float percentage )
32{
33 if( percentage < 0 )
34 {
35 percentage = 0;
36 }
37 if( percentage > 1 )
38 {
39 percentage = 1;
40 }
41
42 // If the value has changed, notify the observers
43 if( m_value != percentage )
44 {
45 m_value = percentage;
46 notify( NULL );
47 }
48}
