25#ifndef VAR_PERCENT_HPP
26#define VAR_PERCENT_HPP
27
28#include "variable.hpp"
29#include "observer.hpp"
30
31class VarPercent;
32
33
34/// Percentage variable
35class VarPercent: public Variable, public Subject<VarPercent>
36{
37public:
38 VarPercent( intf_thread_t *pIntf ) :
39 Variable( pIntf ), m_value( 0 ), m_step( .05f ) {}
40 virtual ~VarPercent() { }
41
42 /// Get the variable type
43 virtual const string &getType() const { return m_type; }
44
45 /// Set the internal value
46 virtual void set( float percentage );
47 virtual float get() const { return m_value; }
48
49 /// Get the variable preferred step
50 virtual float getStep() const { return m_step; }
51 virtual void setStep( float val ) { m_step = val; }
52
53 /// Increment or decrement variable
54 void increment( int num ) { return set( m_value + num * m_step ); }
55
56private:
57 /// Variable type
58 static const string m_type;
59 /// Percent value
60 float m_value;
61 /// preferred step (for scrolling)
62 float m_step;
63};
64
65#endif
