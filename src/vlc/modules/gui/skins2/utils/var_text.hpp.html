25#ifndef VAR_TEXT_HPP
26#define VAR_TEXT_HPP
27
28#include "variable.hpp"
29#include "var_percent.hpp"
30#include "observer.hpp"
31#include "ustring.hpp"
32
33
34/// String variable
35class VarText: public Variable, public Subject<VarText>,
36 public Observer<VarPercent>,
37 public Observer<VarText>
38{
39public:
40 // Set substVars to true to replace "$X" variables in the text
41 VarText( intf_thread_t *pIntf, bool substVars = true );
42 virtual ~VarText();
43
44 /// Get the variable type
45 virtual const string &getType() const { return m_type; }
46
47 /// Set the internal value
48 virtual void set( const UString &rText );
49 virtual const UString get() const;
50
51 /// Methods called when an observed variable is modified
52 virtual void onUpdate( Subject<VarPercent> &rVariable, void* );
53 virtual void onUpdate( Subject<VarText> &rVariable, void* );
54
55private:
56 /// Stop observing other variables
57 void delObservers();
58
59 /// Variable type
60 static const string m_type;
61 /// The text of the variable
62 UString m_text;
63 /// Actual text after having replaced the variables
64 UString m_lastText;
65 /// Flag to activate or not "$X" variables substitution
66 bool m_substVars;
67};
68
69#endif
