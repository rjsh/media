31VarList::VarList( intf_thread_t *pIntf ): Variable( pIntf )
32{
33 // Create the position variable
34 m_cPosition = VariablePtr( new VarPercent( pIntf ) );

44void VarList::add( const UStringPtr &rcString )
45{
46 m_list.push_back( Elem_t( rcString ) );
47 notify();
48}
49
50
51void VarList::delSelected()
52{
53 Iterator it = begin();
54 while( it != end() )
55 {
56 if( (*it).m_selected )
57 {
58 Iterator oldIt = it;
59 it++;
60 m_list.erase( oldIt );
61 }
62 else
63 {
64 it++;
65 }
66 }
67 notify();
68}
69
70
71void VarList::clear()
72{
73 m_list.clear();
74}

76
77VarList::Iterator VarList::operator[]( int n )
78{
79 Iterator it = begin();
80 for( int i = 0; i < n; i++ )
81 {
82 if( it != end() )
83 {
84 it++;
85 }
86 else
87 {
88 break;
89 }
90 }
91 return it;
92}
93
94
95VarList::ConstIterator VarList::operator[]( int n ) const
96{
97 ConstIterator it = begin();
98 for( int i = 0; i < n; i++ )
99 {
100 if( it != end() )
101 {
102 it++;
103 }
104 else
105 {
106 break;
107 }
108 }
109 return it;
110}
