80VarTree::Iterator VarTree::add( int id, const UStringPtr &rcString,
81 bool selected, bool playing, bool expanded, bool readonly,
82 int pos )
83{
84 Iterator it;
85 if( pos == -1 )
86 {
87 it = m_children.end();
88 }
89 else
90 {
91 it = m_children.begin();
92 for( int i = 0; i < pos && it != m_children.end(); ++it, i++ );
93 }
94
95 return m_children.insert( it,
96 VarTree( getIntf(), this, id, rcString,
97 selected, playing,
98 expanded, readonly ) );
99}

119VarTree::Iterator VarTree::getNextSiblingOrUncle()
120{
121 VarTree *p_parent = parent();
122 if( p_parent )
123 {
124 Iterator it = ++(getSelf());
125 if( it != p_parent->m_children.end() )
126 return it;
127 else
128 return next_uncle();
129 }
130 return root()->m_children.end();
131}
132
133VarTree::Iterator VarTree::getPrevSiblingOrUncle()
134{
135 VarTree *p_parent = parent();
136 if( p_parent )
137 {
138 Iterator it = getSelf();
139 if( it != p_parent->m_children.begin() )
140 return --it;
141 else
142 return prev_uncle();
143 }
144 return root()->m_children.end();
145}
146
147/* find iterator to next ancestor
148 * ... which means parent++ or grandparent++ or grandgrandparent++ ... */
149VarTree::Iterator VarTree::next_uncle()
150{
151 VarTree *p_parent = parent();
152 if( p_parent )
153 {
154 VarTree *p_grandparent = p_parent->parent();
155 while( p_grandparent )
156 {
157 Iterator it = ++(p_parent->getSelf());
158 if( it != p_grandparent->m_children.end() )
159 return it;
160 p_parent = p_grandparent;
161 p_grandparent = p_parent->parent();
162 }
163 }
164
165 /* if we didn't return before, it means that we've reached the end */
166 return root()->m_children.end();
167}
168
169VarTree::Iterator VarTree::prev_uncle()
170{
171 VarTree *p_parent = parent();
172 if( p_parent )
173 {
174 VarTree *p_grandparent = p_parent->parent();
175 while( p_grandparent )
176 {
177 Iterator it = p_parent->getSelf();
178 if( it != p_grandparent->m_children.begin() )
179 return --it;
180 p_parent = p_grandparent;
181 p_grandparent = p_parent->parent();
182 }
183 }
184
185 /* if we didn't return before, it means that we've reached the end */
186 return root()->m_children.end();
187}

189int VarTree::visibleItems()
190{
191 int i_count = size();
192 for( Iterator it = m_children.begin(); it != m_children.end(); ++it )
193 {
194 if( it->m_expanded )
195 {
196 i_count += it->visibleItems();
197 }
198 }
199 return i_count;
200}
201
202VarTree::Iterator VarTree::getVisibleItem( int n )
203{
204 Iterator it = m_children.begin();
205 while( it != m_children.end() )
206 {
207 n--;
208 if( n <= 0 )
209 return it;
210 if( it->m_expanded )
211 {
212 int i;
213 i = n - it->visibleItems();
214 if( i <= 0 ) return it->getVisibleItem( n );
215 n = i;
216 }
217 ++it;
218 }
219 return m_children.end();
220}
221
222VarTree::Iterator VarTree::getLeaf( int n )
223{
224 Iterator it = m_children.begin();
225 while( it != m_children.end() )
226 {
227 if( it->size() )
228 {
229 int i;
230 i = n - it->countLeafs();
231 if( i <= 0 ) return it->getLeaf( n );
232 n = i;
233 }
234 else
235 {
236 n--;
237 if( n <= 0 )
238 return it;
239 }
240 ++it;
241 }
242 return m_children.end();
243}
244
245VarTree::Iterator VarTree::getNextVisibleItem( Iterator it )
246{
247 if( it->m_expanded && it->size() )
248 {
249 it = it->m_children.begin();
250 }
251 else
252 {
253 Iterator it_old = it;
254 ++it;
255 // Was 'it' the last brother? If so, look for uncles
256 if( it_old->parent() && it_old->parent()->m_children.end() == it )
257 {
258 it = it_old->next_uncle();
259 }
260 }
261 return it;
262}
263
264VarTree::Iterator VarTree::getPrevVisibleItem( Iterator it )
265{
266 if( it == root()->m_children.begin() )
267 return it;
268
269 if( it == root()->m_children.end() )
270 {
271 --it;
272 while( it->size() && it->m_expanded )
273 it = --(it->m_children.end());
274 return it;
275 }
276
277 /* Was it the first child of its parent ? */
278 VarTree *p_parent = it->parent();
279 if( it == p_parent->m_children.begin() )
280 {
281 /* Yes, get its parent's it */
282 it = p_parent->getSelf();
283 }
284 else
285 {
286 --it;
287 /* We have found an older brother, take its last visible child */
288 while( it->size() && it->m_expanded )
289 it = --(it->m_children.end());
290 }
291 return it;
292}

294VarTree::Iterator VarTree::getNextItem( Iterator it )
295{
296 if( it->size() )
297 {
298 it = it->m_children.begin();
299 }
300 else
301 {
302 Iterator it_old = it;
303 ++it;
304 // Was 'it' the last brother? If so, look for uncles
305 if( it_old->parent() && it_old->parent()->m_children.end() == it )
306 {
307 it = it_old->next_uncle();
308 }
309 }
310 return it;
311}

313VarTree::Iterator VarTree::getPrevItem( Iterator it )
314{
315 if( it == root()->m_children.begin() )
316 return it;
317
318 if( it == root()->m_children.end() )
319 {
320 --it;
321 while( it->size() )
322 it = --(it->m_children.end());
323 return it;
324 }
325 /* Was it the first child of its parent ? */
326 VarTree *p_parent = it->parent();
327 if( it == p_parent->m_children.begin() )
328 {
329 /* Yes, get its parent's it */
330 it = p_parent->getSelf();
331 }
332 else
333 {
334 --it;
335 /* We have found an older brother, take its last child */
336 while( it->size() )
337 it = --(it->m_children.end());
338 }
339 return it;
340}

342VarTree::Iterator VarTree::getNextLeaf( Iterator it )
343{
344 do
345 {
346 it = getNextItem( it );
347 }
348 while( it != root()->m_children.end() && it->size() );
349 return it;
350}
351
352VarTree::Iterator VarTree::getPrevLeaf( Iterator it )
353{
354 Iterator it_new = it->getPrevSiblingOrUncle();
355 if( it_new == root()->end() )
356 return it_new;
357 while( it_new->size() )
358 it_new = --(it_new->m_children.end());
359 return it_new;
360}

371void VarTree::ensureExpanded( const Iterator& it )
372{
373 /// Don't expand ourselves, only our parents
374 VarTree *current = &(*it);
375 current = current->parent();
376 while( current->parent() )
377 {
378 current->m_expanded = true;
379 current = current->parent();
380 }
381}

383int VarTree::countLeafs()
384{
385 if( size() == 0 )
386 return 1;
387
388 int i_count = 0;
389 for( Iterator it = m_children.begin(); it != m_children.end(); ++it )
390 {
391 i_count += it->countLeafs();
392 }
393 return i_count;
394}

396VarTree::Iterator VarTree::firstLeaf()
397{
398 Iterator b = root()->m_children.begin();
399 if( b->size() ) return getNextLeaf( b );
400 return b;

403int VarTree::getIndex( const Iterator& item )
404{
405 int index = 0;
406 Iterator it;
407 for( it = m_flat ? firstLeaf() : m_children.begin();
408 it != m_children.end();
409 it = m_flat ? getNextLeaf( it ) : getNextVisibleItem( it ) )
410 {
411 if( it == item )
412 break;
413 index++;
414 }
415 return (it == item) ? index : -1;
416}

418VarTree::Iterator VarTree::getItemFromSlider()
419{
420 // a simple (int)(...) causes rounding errors !
421#ifdef _MSC_VER
422# define lrint (int)
423#endif
424 VarPercent &rVarPos = getPositionVar();
425 double percentage = rVarPos.get();
426
427 int indexMax = m_flat ? (countLeafs() - 1)
428 : (visibleItems() - 1);
429
430 int index = lrint( (1.0 - percentage)*(double)indexMax );
431
432 Iterator it_first = m_flat ? getLeaf( index + 1 )
433 : getVisibleItem( index + 1 );
434 return it_first;
435}

437void VarTree::setSliderFromItem( const Iterator& it )
438{
439 VarPercent &rVarPos = getPositionVar();
440
441 int indexMax = m_flat ? (countLeafs() - 1)
442 : (visibleItems() - 1);
443
444 int index = getIndex( it );
445 double percentage = (1.0 - (double)index/(double)indexMax);
446
447 m_dontMove = true;
448 rVarPos.set( (float)percentage );
449 m_dontMove = false;
450}
