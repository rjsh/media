25#ifndef EVT_INPUT_HPP
26#define EVT_INPUT_HPP
27
28#include "evt_generic.hpp"
29
30/// Base class for mouse and key events
31class EvtInput: public EvtGeneric
32{
33public:
34 virtual ~EvtInput() { }
35
36 /// Masks for modifier keys
37 static const int
38 kModNone, kModAlt, kModShift, kModCtrl, kModMeta, kModCmd;
39
40 /// Get the modifiers
41 int getMod() const { return m_mod; }
42
43protected:
44 EvtInput( intf_thread_t *pIntf, int mod = kModNone );
45
46 /// Add the modifier to the event string
47 void addModifier( string &rEvtString ) const;
48
49private:
50 /// Modifiers (special key(s) pressed during the mouse event)
51 int m_mod;
52};
53
54#endif
