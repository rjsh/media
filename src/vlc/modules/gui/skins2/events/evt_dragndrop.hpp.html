24#ifndef EVT_DRAGNDROP_HPP
25#define EVT_DRAGNDROP_HPP
26
27#include "evt_generic.hpp"
28#include <list>
29
30
31/// Drag'n'Drop generic event
32class EvtDrag: public EvtGeneric
33{
34public:
35 EvtDrag( intf_thread_t *pIntf ): EvtGeneric( pIntf ) { }
36 virtual ~EvtDrag() { }
37 virtual const string getAsString() const { return "drag"; }
38};
39
40
41class EvtDragEnter: public EvtDrag
42{
43public:
44 EvtDragEnter( intf_thread_t *pIntf ): EvtDrag( pIntf ) { }
45 virtual ~EvtDragEnter() { }
46 virtual const string getAsString() const { return "drag:enter"; }
47};
48
49
50class EvtDragLeave: public EvtDrag
51{
52public:
53 EvtDragLeave( intf_thread_t *pIntf ): EvtDrag( pIntf ) { }
54 virtual ~EvtDragLeave() { }
55 virtual const string getAsString() const { return "drag:leave"; }
56};
57
58
59class EvtDragOver: public EvtDrag
60{
61public:
62 EvtDragOver( intf_thread_t *pIntf, int x, int y )
63 : EvtDrag( pIntf ), m_xPos( x ), m_yPos( y ) { }
64 virtual ~EvtDragOver() { }
65 virtual const string getAsString() const { return "drag:over"; }
66 // Return the event coordinates
67 int getXPos() const { return m_xPos; }
68 int getYPos() const { return m_yPos; }
69private:
70 int m_xPos;
71 int m_yPos;
72};
73
74class EvtDragDrop: public EvtDrag
75{
76public:
77 EvtDragDrop( intf_thread_t *pIntf, int x, int y, const list<string>& files )
78 : EvtDrag( pIntf ), m_files( files ), m_xPos( x ), m_yPos( y ) { }
79 virtual ~EvtDragDrop() { }
80 virtual const string getAsString() const { return "drag:drop"; }
81 // Return the event coordinates
82 int getXPos() const { return m_xPos; }
83 int getYPos() const { return m_yPos; }
84 const list<string>& getFiles() const { return m_files; }
85private:
86 list<string> m_files;
87 int m_xPos;
88 int m_yPos;
89};
