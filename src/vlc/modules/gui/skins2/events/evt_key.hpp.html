25#ifndef EVT_KEY_HPP
26#define EVT_KEY_HPP
27
28#include "evt_input.hpp"
29
30
31/// Class for keyboard events
32class EvtKey: public EvtInput
33{
34public:
35 enum ActionType_t
36 {
37 kDown,
38 kUp
39 };
40
41 EvtKey( intf_thread_t *I, int key, ActionType_t actn, int mod = kModNone )
42 : EvtInput( I, mod ), m_key( key ), m_action( actn ) { }
43 virtual ~EvtKey() { }
44 virtual const string getAsString() const;
45
46 int getKey() const { return m_key; }
47 int getModKey() const { return m_key | getMod(); }
48
49 ActionType_t getKeyState() const { return m_action; }
50
51private:
52 /// The concerned key, stored according to the '#define's in vlc_keys.h
53 /// but without the modifiers (which are stored in EvtInput)
54 int m_key;
55 /// Type of action
56 ActionType_t m_action;
57};
58
59
60#endif
