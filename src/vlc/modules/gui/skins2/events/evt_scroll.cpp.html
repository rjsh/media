25#include "evt_scroll.hpp"
26
27
28const string EvtScroll::getAsString() const
29{
30 string event = "scroll";
31
32 // Add the direction
33 if( m_direction == kUp )
34 event += ":up";
35 else if( m_direction == kDown )
36 event += ":down";
37 else
38 msg_Warn( getIntf(), "unknown scrolling direction" );
39
40 // Add the modifier
41 addModifier( event );
42
43 return event;
44}
