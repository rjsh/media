25#ifndef EVT_REFRESH_HPP
26#define EVT_REFRESH_HPP
27
28#include "evt_generic.hpp"
29
30
31/// Refresh window event
32class EvtRefresh: public EvtGeneric
33{
34public:
35 /// Constructor with the coordinates of the area to refresh
36 EvtRefresh( intf_thread_t *pIntf, int xStart, int yStart,
37 int width, int height )
38 : EvtGeneric( pIntf ), m_xStart( xStart ), m_yStart( yStart ),
39 m_width( width ), m_height( height ) { }
40
41 virtual ~EvtRefresh() { }
42 virtual const string getAsString() const { return "refresh"; }
43
44 /// Getters
45 int getXStart() const { return m_xStart; }
46 int getYStart() const { return m_yStart; }
47 int getWidth() const { return m_width; }
48 int getHeight() const { return m_height; }
49
50private:
51 /// Coordinates and size of the area to refresh
52 int m_xStart, m_yStart, m_width, m_height;
53};
54
55
56#endif
