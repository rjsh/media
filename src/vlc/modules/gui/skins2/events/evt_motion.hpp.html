25#ifndef EVT_MOTION_HPP
26#define EVT_MOTION_HPP
27
28#include "evt_input.hpp"
29
30
31/// Mouse move event
32class EvtMotion: public EvtInput
33{
34public:
35 EvtMotion( intf_thread_t *pIntf, int xPos, int yPos )
36 : EvtInput( pIntf ), m_xPos( xPos ), m_yPos( yPos ) { }
37 virtual ~EvtMotion() { }
38 virtual const string getAsString() const { return "motion"; }
39
40 // Getters
41 int getXPos() const { return m_xPos; }
42 int getYPos() const { return m_yPos; }
43
44private:
45 /// Coordinates of the mouse (absolute or relative)
46 /**
47 * The coordinates are absolute when the event is sent to the
48 * GenericWindow, but are relative to the window when the event is
49 * forwarded to the controls
50 */
51 int m_xPos, m_yPos;
52};
