28const string EvtSpecial::getAsString() const
29{
30 string event = "special";
31
32 // Add the action
33 if( m_action == kShow )
34 event += ":show";
35 else if( m_action == kHide )
36 event += ":hide";
37 else if( m_action == kEnable )
38 event += ":enable";
39 else if( m_action == kDisable )
40 event += ":disable";
41 else
42 msg_Warn( getIntf(), "unknown action type" );
43
44 return event;
45}
