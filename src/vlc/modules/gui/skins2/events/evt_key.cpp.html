25#include "evt_key.hpp"
26#include <vlc_keys.h>
27
28
29const string EvtKey::getAsString() const
30{
31 string event = "key";
32
33 // Add the action
34 if( m_action == kDown )
35 event += ":down";
36 else if( m_action == kUp )
37 event += ":up";
38 else
39 msg_Warn( getIntf(), "Unknown action type" );
40
41 // Add the key
42 char *keyName = vlc_keycode2str( m_key & ~KEY_MODIFIER, true );
43 if( keyName )
44 {
45 event += (string)":" + keyName;
46 free( keyName );
47 }
48 else
49 msg_Warn( getIntf(), "Unknown key: %d", m_key );
50
51 // Add the modifier
52 addModifier( event );
53
54 return event;
55}
