25#ifndef EVT_SCROLL_HPP
26#define EVT_SCROLL_HPP
27
28#include "evt_input.hpp"
29
30
31/// Class for mouse scroll events
32class EvtScroll: public EvtInput
33{
34public:
35 enum Direction_t
36 {
37 kUp,
38 kDown
39 };
40
41 EvtScroll( intf_thread_t *pIntf, int xPos, int yPos,
42 Direction_t direction, int mod = kModNone )
43 : EvtInput( pIntf, mod ), m_xPos( xPos ), m_yPos( yPos ),
44 m_direction( direction ) { }
45 virtual ~EvtScroll() { }
46 virtual const string getAsString() const;
47
48 // Return the event coordinates
49 int getXPos() const { return m_xPos; }
50 int getYPos() const { return m_yPos; }
51
52 // Return the direction
53 Direction_t getDirection() const { return m_direction; }
54
55private:
56 /// Coordinates of the mouse relative to the window
57 int m_xPos, m_yPos;
58 /// Scroll direction
59 Direction_t m_direction;
60};
