25#ifndef EVT_FOCUS_HPP
26#define EVT_FOCUS_HPP
27
28#include "evt_generic.hpp"
29
30
31/// Focus change event
32class EvtFocus: public EvtGeneric
33{
34public:
35 EvtFocus( intf_thread_t *pIntf, bool focus )
36 : EvtGeneric( pIntf ), m_focus( focus ) { }
37 virtual ~EvtFocus() { }
38
39 virtual const string getAsString() const
40 {
41 return ( m_focus ? "focus:in" : "focus:out" );
42 }
43
44private:
45 /// true for a focus in, and false for a focus out
46 bool m_focus;
47};
