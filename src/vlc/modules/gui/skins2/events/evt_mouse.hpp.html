25#ifndef EVT_MOUSE_HPP
26#define EVT_MOUSE_HPP
27
28#include "evt_input.hpp"
29
30
31/// Class for mouse button events
32class EvtMouse: public EvtInput
33{
34public:
35 enum ButtonType_t
36 {
37 kLeft,
38 kMiddle,
39 kRight
40 };
41
42 enum ActionType_t
43 {
44 kDown,
45 kUp,
46 kDblClick
47 };
48
49 EvtMouse( intf_thread_t *pIntf, int xPos, int yPos, ButtonType_t button,
50 ActionType_t action, int mod = kModNone )
51 : EvtInput( pIntf, mod ), m_xPos( xPos ), m_yPos( yPos ),
52 m_button( button ), m_action( action ) { }
53 virtual ~EvtMouse() { }
54 virtual const string getAsString() const;
55
56 // Return the event coordinates
57 int getXPos() const { return m_xPos; }
58 int getYPos() const { return m_yPos; }
59
60 // Return the button and the action
61 ButtonType_t getButton() const { return m_button; }
62 ActionType_t getAction() const { return m_action; }
63
64private:
65 /// Coordinates of the mouse relative to the window
66 int m_xPos, m_yPos;
67 /// Mouse button involved in the event
68 ButtonType_t m_button;
69 /// Type of action
70 ActionType_t m_action;
71};
72
73
74#endif
