38static inline float scroll( bool up, float pct, float step )
39{
40 return pct + ( up ? step : -step );
41}

44CtrlSliderCursor::CtrlSliderCursor( intf_thread_t *pIntf,
45 const GenericBitmap &rBmpUp,
46 const GenericBitmap &rBmpOver,
47 const GenericBitmap &rBmpDown,
48 const Bezier &rCurve,
49 VarPercent &rVariable,
50 VarBool *pVisible,
51 const UString &rTooltip,
52 const UString &rHelp ):
53 CtrlGeneric( pIntf, rHelp, pVisible ), m_fsm( pIntf ),
54 m_rVariable( rVariable ), m_tooltip( rTooltip ),
55 m_width( rCurve.getWidth() ), m_height( rCurve.getHeight() ),
56 m_xPosition( 0 ), m_yPosition( 0 ),
57 m_cmdOverDown( this ), m_cmdDownOver( this ),
58 m_cmdOverUp( this ), m_cmdUpOver( this ),
59 m_cmdMove( this ), m_cmdScroll( this ),
60 m_lastCursorRect(), m_xOffset( 0 ), m_yOffset( 0 ),
61 m_pEvt( NULL ), m_rCurve( rCurve ),
62 m_pImgUp( rBmpUp.getGraphics() ),
63 m_pImgOver( rBmpOver.getGraphics() ),
64 m_pImgDown( rBmpDown.getGraphics() ),
65 m_pImg( m_pImgUp )
66{
67 // States
68 m_fsm.addState( "up" );
69 m_fsm.addState( "over" );
70 m_fsm.addState( "down" );
71
72 // Transitions
73 m_fsm.addTransition( "over", "mouse:left:down", "down",
74 &m_cmdOverDown );
75 m_fsm.addTransition( "down", "mouse:left:up", "over",
76 &m_cmdDownOver );
77 m_fsm.addTransition( "over", "leave", "up", &m_cmdOverUp );
78 m_fsm.addTransition( "up", "enter", "over", &m_cmdUpOver );
79 m_fsm.addTransition( "down", "motion", "down", &m_cmdMove );
80 m_fsm.addTransition( "over", "scroll", "over", &m_cmdScroll );
81
82 // Initial state
83 m_fsm.setState( "up" );
84
85 // Observe the position variable
86 m_rVariable.addObserver( this );
87}

105bool CtrlSliderCursor::mouseOver( int x, int y ) const
106{
107 if( m_pImg )
108 {
109 // Compute the position of the cursor
110 int xPos, yPos;
111 m_rCurve.getPoint( m_rVariable.get(), xPos, yPos );
112
113 // Compute the resize factors
114 float factorX, factorY;
115 getResizeFactors( factorX, factorY );
116 xPos = (int)(xPos * factorX);
117 yPos = (int)(yPos * factorY);
118
119 return m_pImg->hit( x - xPos + m_pImg->getWidth() / 2,
120 y - yPos + m_pImg->getHeight() / 2 );
121 }
122
123 return false;
124}

183void CtrlSliderCursor::CmdOverDown::execute()
184{
185 EvtMouse *pEvtMouse = static_cast<EvtMouse*>(m_pParent->m_pEvt);
186
187 // Compute the resize factors
188 float factorX, factorY;
189 m_pParent->getResizeFactors( factorX, factorY );
190
191 // Get the position of the control
192 const Position *pPos = m_pParent->getPosition();
193
194 // Compute the offset
195 int tempX, tempY;
196 m_pParent->m_rCurve.getPoint( m_pParent->m_rVariable.get(), tempX, tempY );
197 m_pParent->m_xOffset = pEvtMouse->getXPos() - pPos->getLeft()
198 - (int)(tempX * factorX);
199 m_pParent->m_yOffset = pEvtMouse->getYPos() - pPos->getTop()
200 - (int)(tempY * factorY);
201
202 m_pParent->captureMouse();
203
204 if( m_pParent->m_pImg == m_pParent->m_pImgDown )
205 return;
206 m_pParent->m_pImg = m_pParent->m_pImgDown;
207 m_pParent->refreshLayout();
208}

240void CtrlSliderCursor::CmdMove::execute()
241{
242 EvtMouse *pEvtMouse = static_cast<EvtMouse*>(m_pParent->m_pEvt);
243
244 // Get the position of the control
245 const Position *pPos = m_pParent->getPosition();
246
247 // Compute the resize factors
248 float factorX, factorY;
249 m_pParent->getResizeFactors( factorX, factorY );
250
251 // Compute the relative position of the centre of the cursor
252 float relX = pEvtMouse->getXPos() - pPos->getLeft() - m_pParent->m_xOffset;
253 float relY = pEvtMouse->getYPos() - pPos->getTop() - m_pParent->m_yOffset;
254 // Ponderate with the resize factors
255 int relXPond = (int)(relX / factorX);
256 int relYPond = (int)(relY / factorY);
257
258 float percentage =
259 m_pParent->m_rCurve.getNearestPercent( relXPond, relYPond );
260 m_pParent->m_rVariable.set( percentage );
261}

263void CtrlSliderCursor::CmdScroll::execute()
264{
265 int dir = static_cast<EvtScroll*>(m_pParent->m_pEvt)->getDirection();
266 m_pParent->m_rVariable.set( scroll( EvtScroll::kUp == dir,
267 m_pParent->m_rVariable.get(),
268 m_pParent->m_rVariable.getStep()) );

289void CtrlSliderCursor::refreshLayout( bool force )
290{
291 rect currRect = getCurrentCursorRect();
292 if( !force && currRect == m_lastCursorRect )
293 return;
294
295 rect join;
296 if( rect::join( m_lastCursorRect, currRect, &join ) )
297 {
298 m_lastCursorRect = currRect;
299
300 const Position *pPos = getPosition();
301 notifyLayout( join.width, join.height,
302 join.x - pPos->getLeft(),
303 join.y - pPos->getTop() );
304 }
305}

307
308rect CtrlSliderCursor::getCurrentCursorRect()
309{
310 const Position *pPos = getPosition();
311
312 // Compute the position of the cursor
313 int xPos, yPos;
314 m_rCurve.getPoint( m_rVariable.get(), xPos, yPos );
315
316 // Compute the resize factors
317 float factorX, factorY;
318 getResizeFactors( factorX, factorY );
319 xPos = (int)(xPos * factorX);
320 yPos = (int)(yPos * factorY);
321
322 int x = pPos->getLeft() + xPos - m_pImg->getWidth() / 2;
323 int y = pPos->getTop() + yPos - m_pImg->getHeight() / 2;
324
325 return rect( x, y, m_pImg->getWidth(), m_pImg->getHeight() );
326}

329CtrlSliderBg::CtrlSliderBg( intf_thread_t *pIntf,
330 const Bezier &rCurve, VarPercent &rVariable,
331 int thickness, GenericBitmap *pBackground,
332 int nbHoriz, int nbVert, int padHoriz, int padVert,
333 VarBool *pVisible, const UString &rHelp ):
334 CtrlGeneric( pIntf, rHelp, pVisible ), m_pCursor( NULL ),
335 m_rVariable( rVariable ), m_thickness( thickness ), m_rCurve( rCurve ),
336 m_width( rCurve.getWidth() ), m_height( rCurve.getHeight() ),
337 m_pImgSeq( pBackground ), m_pScaledBmp( NULL ),
338 m_nbHoriz( nbHoriz ), m_nbVert( nbVert ),
339 m_padHoriz( padHoriz ), m_padVert( padVert ),
340 m_bgWidth( 0 ), m_bgHeight( 0 ), m_position( 0 )
341{
342 if( m_pImgSeq )
343 {
344 // Observe the position variable
345 m_rVariable.addObserver( this );
346
347 // Initial position
348 m_position = (int)( m_rVariable.get() * (m_nbHoriz * m_nbVert - 1) );
349 }
350}

362bool CtrlSliderBg::mouseOver( int x, int y ) const
363{
364 // Compute the resize factors
365 float factorX, factorY;
366 getResizeFactors( factorX, factorY );
367
368 if( m_pScaledBmp )
369 {
370 // background size that is displayed
371 int width = m_bgWidth - (int)(m_padHoriz * factorX);
372 int height = m_bgHeight - (int)(m_padVert * factorY);
373
374 return x >= 0 && x < width &&
375 y >= 0 && y < height;
376 }
377 else
378 {
379 return m_rCurve.getMinDist( (int)(x / factorX), (int)(y / factorY),
380 factorX, factorY ) < m_thickness;
381 }
382}

385void CtrlSliderBg::draw( OSGraphics &rImage, int xDest, int yDest, int w, int h )
386{
387 if( !m_pScaledBmp || m_bgWidth <= 0 || m_bgHeight <= 0 )
388 return;
389
390 // Compute the resize factors
391 float factorX, factorY;
392 getResizeFactors( factorX, factorY );
393
394 // Locate the right image in the background bitmap
395 int x = m_bgWidth * ( m_position % m_nbHoriz );
396 int y = m_bgHeight * ( m_position / m_nbHoriz );
397
398 // Draw the background image
399 const Position *pPos = getPosition();
400 rect region( pPos->getLeft(), pPos->getTop(),
401 m_bgWidth - (int)(m_padHoriz * factorX),
402 m_bgHeight - (int)(m_padVert * factorY) );
403 rect clip( xDest, yDest, w, h );
404 rect inter;
405 if( rect::intersect( region, clip, &inter ) )
406 rImage.drawBitmap( *m_pScaledBmp,
407 x + inter.x - region.x,
408 y + inter.y - region.y,
409 inter.x, inter.y,
410 inter.width, inter.height );
411}

414void CtrlSliderBg::handleEvent( EvtGeneric &rEvent )
415{
416 if( rEvent.getAsString().find( "mouse:left:down" ) != string::npos )
417 {
418 // Compute the resize factors
419 float factorX, factorY;
420 getResizeFactors( factorX, factorY );
421
422 // Get the position of the control
423 const Position *pPos = getPosition();
424
425 // Get the value corresponding to the position of the mouse
426 EvtMouse &rEvtMouse = (EvtMouse&)rEvent;
427 int x = rEvtMouse.getXPos();
428 int y = rEvtMouse.getYPos();
429 m_rVariable.set( m_rCurve.getNearestPercent(
430 (int)((x - pPos->getLeft()) / factorX),
431 (int)((y - pPos->getTop()) / factorY) ) );
432
433 // Forward the clic to the cursor
434 EvtMouse evt( getIntf(), x, y, EvtMouse::kLeft, EvtMouse::kDown );
435 TopWindow *pWin = getWindow();
436 if( pWin && m_pCursor )
437 {
438 EvtEnter evtEnter( getIntf() );
439 // XXX It was not supposed to be implemented like that !!
440 pWin->forwardEvent( evtEnter, *m_pCursor );
441 pWin->forwardEvent( evt, *m_pCursor );
442 }
443 }
444 else if( rEvent.getAsString().find( "scroll" ) != string::npos )
445 {
446 int dir = static_cast<EvtScroll*>(&rEvent)->getDirection();
447 m_rVariable.set( scroll( EvtScroll::kUp == dir,
448 m_rVariable.get(), m_rVariable.getStep() ) );
449 }
450}

526
527void CtrlSliderBg::setCurrentImage()
528{
529 // Compute the resize factors
530 float factorX, factorY;
531 getResizeFactors( factorX, factorY );
532
533 // Build the background image sequence
534 // Note: we suppose that the last padding is not included in the
535 // given image
536 // TODO: we should probably change this assumption, as it would make
537 // the code a bit simpler and it would be more natural for the skins
538 // designers
539 // Size of one elementary background image (padding included)
540 m_bgWidth =
541 (int)((m_pImgSeq->getWidth() + m_padHoriz) * factorX / m_nbHoriz);
542 m_bgHeight =
543 (int)((m_pImgSeq->getHeight() + m_padVert) * factorY / m_nbVert);
544
545 // Rescale the image accordingly
546 int width = m_bgWidth * m_nbHoriz - (int)(m_padHoriz * factorX);
547 int height = m_bgHeight * m_nbVert - (int)(m_padVert * factorY);
548 if( !m_pScaledBmp ||
549 m_pScaledBmp->getWidth() != width ||
550 m_pScaledBmp->getHeight() != height )
551 {
552 // scaled bitmap
553 delete m_pScaledBmp;
554 m_pScaledBmp = new ScaledBitmap( getIntf(), *m_pImgSeq, width, height );
555 }
