34CtrlMove::CtrlMove( intf_thread_t *pIntf, WindowManager &rWindowManager,
35 CtrlFlat &rCtrl, TopWindow &rWindow,
36 const UString &rHelp, VarBool *pVisible ):
37 CtrlFlat( pIntf, rHelp, pVisible ), m_fsm( pIntf ),
38 m_rWindowManager( rWindowManager ),
39 m_rCtrl( rCtrl ), m_rWindow( rWindow ),
40 m_cmdMovingMoving( this ),
41 m_cmdStillMoving( this ),
42 m_cmdMovingStill( this )
43{
44 m_pEvt = NULL;
45 m_xPos = 0;
46 m_yPos = 0;
47
48 // States
49 m_fsm.addState( "moving" );
50 m_fsm.addState( "still" );
51
52 // Transitions
53 m_fsm.addTransition( "moving", "mouse:left:up:none", "still",
54 &m_cmdMovingStill );
55 m_fsm.addTransition( "still", "mouse:left:down:none", "moving",
56 &m_cmdStillMoving );
57 m_fsm.addTransition( "moving", "motion", "moving", &m_cmdMovingMoving );
58
59 m_fsm.setState( "still" );
60}

112void CtrlMove::CmdStillMoving::execute()
113{
114 EvtMouse *pEvtMouse = (EvtMouse*)m_pParent->m_pEvt;
115
116 m_pParent->m_xPos = pEvtMouse->getXPos();
117 m_pParent->m_yPos = pEvtMouse->getYPos();
118
119 m_pParent->captureMouse();
120
121 m_pParent->m_rWindowManager.startMove( m_pParent->m_rWindow );
122}
123
124
125void CtrlMove::CmdMovingMoving::execute()
126{
127 EvtMotion *pEvtMotion = (EvtMotion*)m_pParent->m_pEvt;
128
129 int xNewLeft = pEvtMotion->getXPos() - m_pParent->m_xPos +
130 m_pParent->m_rWindow.getLeft();
131 int yNewTop = pEvtMotion->getYPos() - m_pParent->m_yPos +
132 m_pParent->m_rWindow.getTop();
133
134 m_pParent->m_rWindowManager.move( m_pParent->m_rWindow, xNewLeft, yNewTop );
135}
136
137
138void CtrlMove::CmdMovingStill::execute()
139{
140 m_pParent->releaseMouse();
141
142 m_pParent->m_rWindowManager.stopMove();
143}
