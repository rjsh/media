35class GenericBitmap;
36class ScaledBitmap;
37class OSGraphics;
38class VarPercent;
39
40
41/// Cursor of a slider
42class CtrlSliderCursor: public CtrlGeneric, public Observer<VarPercent>
43{
44public:
45 /// Create a cursor with 3 images (which are NOT copied, be careful)
46 /// If pVisible is NULL, the control is always visible
47 CtrlSliderCursor( intf_thread_t *pIntf, const GenericBitmap &rBmpUp,
48 const GenericBitmap &rBmpOver,
49 const GenericBitmap &rBmpDown,
50 const Bezier &rCurve, VarPercent &rVariable,
51 VarBool *pVisible, const UString &rTooltip,
52 const UString &rHelp );
53
54 virtual ~CtrlSliderCursor();
55
56 /// Handle an event
57 virtual void handleEvent( EvtGeneric &rEvent );
58
59 /// Return true if the control can be scrollable
60 virtual bool isScrollable() const { return true; }
61
62 /// Check whether coordinates are inside the control
63 virtual bool mouseOver( int x, int y ) const;
64
65 /// Draw the control on the given graphics
66 virtual void draw( OSGraphics &rImage, int xDest, int yDest, int w, int h );
67
68 /// Called when the position is set
69 virtual void onPositionChange();
70
71 /// Method called when the control is resized
72 virtual void onResize();
73
74 /// Method called to notify are to be updated
75 virtual void notifyLayout( int witdh = -1, int height = -1,
76 int xOffSet = 0, int yOffSet = 0 );
77
78 /// Get the text of the tooltip
79 virtual UString getTooltipText() const { return m_tooltip; }
80
81 /// Get the type of control (custom RTTI)
82 virtual string getType() const { return "slider_cursor"; }
83
84private:
85 /// Finite state machine of the control
86 FSM m_fsm;
87 /// Variable associated to the cursor
88 VarPercent &m_rVariable;
89 /// Tooltip text
90 const UString m_tooltip;
91 /// Initial size of the control
92 int m_width, m_height;
93 /// Position of the cursor
94 int m_xPosition, m_yPosition;
95 /// Callback objects
96 DEFINE_CALLBACK( CtrlSliderCursor, OverDown )
97 DEFINE_CALLBACK( CtrlSliderCursor, DownOver )
98 DEFINE_CALLBACK( CtrlSliderCursor, OverUp )
99 DEFINE_CALLBACK( CtrlSliderCursor, UpOver )
100 DEFINE_CALLBACK( CtrlSliderCursor, Move )
101 DEFINE_CALLBACK( CtrlSliderCursor, Scroll )
102 /// Last saved cursor placement
103 rect m_lastCursorRect;
104 /// Offset between the mouse pointer and the center of the cursor
105 int m_xOffset, m_yOffset;
106 /// The last received event
107 EvtGeneric *m_pEvt;
108 /// Bezier curve of the slider
109 const Bezier &m_rCurve;
110 /// Images of the cursor in the different states
111 const OSGraphics * const m_pImgUp;
112 const OSGraphics * const m_pImgOver;
113 const OSGraphics * const m_pImgDown;
114 /// Current image
115 const OSGraphics *m_pImg;
116
117 /// Method called when the position variable is modified
118 virtual void onUpdate( Subject<VarPercent> &rVariable, void * );
119
120 /// Method to compute the resize factors
121 void getResizeFactors( float &rFactorX, float &rFactorY ) const;
122
123 /// Call notifyLayout
124 void refreshLayout( bool force = true );
125
126 /// getter for the current slider rectangle
127 rect getCurrentCursorRect();
128};

131/// Slider background
132class CtrlSliderBg: public CtrlGeneric, public Observer<VarPercent>
133{
134public:
135 CtrlSliderBg( intf_thread_t *pIntf,
136 const Bezier &rCurve, VarPercent &rVariable,
137 int thickness, GenericBitmap *pBackground, int nbHoriz,
138 int nbVert, int padHoriz, int padVert, VarBool *pVisible,
139 const UString &rHelp );
140 virtual ~CtrlSliderBg();
141
142 /// Return true if the control can be scrollable
143 virtual bool isScrollable() const { return true; }
144
145 /// Tell whether the mouse is over the control
146 virtual bool mouseOver( int x, int y ) const;
147
148 /// Draw the control on the given graphics
149 virtual void draw( OSGraphics &rImage, int xDest, int yDest, int w, int h );
150
151 /// Handle an event
152 virtual void handleEvent( EvtGeneric &rEvent );
153
154 /// Called when the position is set
155 virtual void onPositionChange();
156
157 /// Method called when the control is resized
158 virtual void onResize();
159
160 /// Method called to notify are to be updated
161 virtual void notifyLayout( int witdh = -1, int height = -1,
162 int xOffSet = 0, int yOffSet = 0 );
163
164 /// Get the type of control (custom RTTI)
165 virtual string getType() const { return "slider_bg"; }
166
167 /// Associate a cursor to this background
168 void associateCursor( CtrlSliderCursor &rCursor );
169
170private:
171 /// Cursor of the slider
172 CtrlSliderCursor *m_pCursor;
173 /// Variable associated to the slider
174 VarPercent &m_rVariable;
175 /// Thickness of the curve
176 int m_thickness;
177 /// Bezier curve of the slider
178 const Bezier &m_rCurve;
179 /// Initial size of the control
180 int m_width, m_height;
181 /// Background image sequence (optional)
182 GenericBitmap *m_pImgSeq;
183 /// Scaled bitmap if needed
184 ScaledBitmap *m_pScaledBmp;
185 /// Number of images in the background bitmap
186 int m_nbHoriz, m_nbVert;
187 /// Number of pixels between two images
188 int m_padHoriz, m_padVert;
189 /// Size of a background image
190 int m_bgWidth, m_bgHeight;
191 /// Index of the current background image
192 int m_position;
193
194 /// Method called when the observed variable is modified
195 virtual void onUpdate( Subject<VarPercent> &rVariable, void* );
196
197 /// Method to compute the resize factors
198 void getResizeFactors( float &rFactorX, float &rFactorY ) const;
199
200 /// Method to (re)set the current image
201 void setCurrentImage( );
202};
