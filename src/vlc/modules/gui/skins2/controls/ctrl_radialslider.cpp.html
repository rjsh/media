36CtrlRadialSlider::CtrlRadialSlider( intf_thread_t *pIntf,
37 const GenericBitmap &rBmpSeq, int numImg,
38 VarPercent &rVariable, float minAngle,
39 float maxAngle, const UString &rHelp,
40 VarBool *pVisible ):
41 CtrlGeneric( pIntf, rHelp, pVisible ), m_fsm( pIntf ), m_numImg( numImg ),
42 m_rVariable( rVariable ), m_minAngle( minAngle ), m_maxAngle( maxAngle ),
43 m_position( 0 ),
44 m_width( rBmpSeq.getWidth() ), m_height( rBmpSeq.getHeight() / numImg ),
45 m_pImgSeq( rBmpSeq.getGraphics() ),
46 m_cmdUpDown( this ), m_cmdDownUp( this ), m_cmdMove( this )
47{
48 // States
49 m_fsm.addState( "up" );
50 m_fsm.addState( "down" );
51
52 // Transitions
53 m_fsm.addTransition( "up", "mouse:left:down", "down", &m_cmdUpDown );
54 m_fsm.addTransition( "down", "mouse:left:up", "up", &m_cmdDownUp );
55 m_fsm.addTransition( "down", "motion", "down", &m_cmdMove );
56
57 // Initial state
58 m_fsm.setState( "up" );
59
60 // Observe the variable
61 m_rVariable.addObserver( this );
62}
63
64
65CtrlRadialSlider::~CtrlRadialSlider()
66{
67 m_rVariable.delObserver( this );
68}

71void CtrlRadialSlider::handleEvent( EvtGeneric &rEvent )
72{
73 // Save the event to use it in callbacks
74 m_pEvt = &rEvent;
75
76 m_fsm.handleTransition( rEvent.getAsString() );
77}

80bool CtrlRadialSlider::mouseOver( int x, int y ) const
81{
82 return m_pImgSeq->hit( x, y + m_position * m_height );
83}

86void CtrlRadialSlider::draw( OSGraphics &rImage, int xDest, int yDest, int w, int h )
87{
88 const Position *pPos = getPosition();
89 rect region( pPos->getLeft(), pPos->getTop(), m_width, m_height );
90 rect clip( xDest, yDest, w ,h );
91 rect inter;
92 if( rect::intersect( region, clip, &inter ) )
93 rImage.drawGraphics( *m_pImgSeq,
94 inter.x - region.x,
95 inter.y - region.y + m_position * m_height,
96 inter.x, inter.y,
97 inter.width, inter.height );
98}

101void CtrlRadialSlider::onUpdate( Subject<VarPercent> &rVariable, void *arg )
102{
103 (void)arg;
104 if( &rVariable == &m_rVariable )
105 {
106 int position = (int)( m_rVariable.get() * ( m_numImg - 1 ) );
107 if( position == m_position )
108 return;
109
110 m_position = position;
111 notifyLayout( m_width, m_height );
112 }
113}

116void CtrlRadialSlider::CmdUpDown::execute()
117{
118 EvtMouse *pEvtMouse = (EvtMouse*)m_pParent->m_pEvt;
119
120 // Change the position of the cursor, in non-blocking mode
121 m_pParent->setCursor( pEvtMouse->getXPos(), pEvtMouse->getYPos(), false );
122
123 m_pParent->captureMouse();
124}
125
126
127void CtrlRadialSlider::CmdDownUp::execute()
128{
129 m_pParent->releaseMouse();
130}

133void CtrlRadialSlider::CmdMove::execute()
134{
135 EvtMouse *pEvtMouse = static_cast<EvtMouse*>(m_pParent->m_pEvt);
136
137 // Change the position of the cursor, in blocking mode
138 m_pParent->setCursor( pEvtMouse->getXPos(), pEvtMouse->getYPos(), true );
139}

142void CtrlRadialSlider::setCursor( int posX, int posY, bool blocking )
143{
144 // Get the position of the control
145 const Position *pPos = getPosition();
146 if( !pPos )
147 {
148 return;
149 }
150
151 // Compute the position relative to the center
152 int x = posX - pPos->getLeft() - m_width / 2;
153 int y = posY - pPos->getTop() - m_height / 2;
154
155 // Compute the polar coordinates. angle is -(-j,OM)
156 float r = sqrt((float)(x*x + y*y));
157 if( r == 0 )
158 {
159 return;
160 }
161 float angle = acos(y/r);
162 if( x > 0 )
163 {
164 angle = 2*M_PI - angle;
165 }
166
167 if( angle >= m_minAngle && angle <= m_maxAngle )
168 {
169 float newVal = (angle - m_minAngle) / (m_maxAngle - m_minAngle);
170 // Avoid too fast moves of the cursor if blocking mode
171 if( !blocking || fabs( m_rVariable.get() - newVal ) < 0.5 )
172 {
173 m_rVariable.set( newVal );
174 }
175 }
176}
