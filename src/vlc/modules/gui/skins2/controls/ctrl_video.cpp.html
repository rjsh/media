35CtrlVideo::CtrlVideo( intf_thread_t *pIntf, GenericLayout &rLayout,
36 bool autoResize, const UString &rHelp,
37 VarBool *pVisible ):
38 CtrlGeneric( pIntf, rHelp, pVisible ), m_rLayout( rLayout ),
39 m_bAutoResize( autoResize ), m_xShift( 0 ), m_yShift( 0 ),
40 m_pVoutWindow( NULL )
41{
42 VarBool &rFullscreen = VlcProc::instance( getIntf() )->getFullscreenVar();
43 rFullscreen.addObserver( this );
44
45 // if global parameter set to no resize, override skins behavior
46 if( !var_InheritBool( pIntf, "qt-video-autoresize" ) )
47 m_bAutoResize = false;
48}

54 rFullscreen.delObserver( this );
55}

71void CtrlVideo::onResize()
72{
73 const Position *pPos = getPosition();
74 if( pPos && m_pVoutWindow )
75 {
76 m_pVoutWindow->move( pPos->getLeft(), pPos->getTop() );
77 m_pVoutWindow->resize( pPos->getWidth(), pPos->getHeight() );
78 }
79}

82void CtrlVideo::onPositionChange()
83{
84 // Compute the difference between layout size and video size
85 m_xShift = m_rLayout.getWidth() - getPosition()->getWidth();
86 m_yShift = m_rLayout.getHeight() - getPosition()->getHeight();
87}

90void CtrlVideo::draw( OSGraphics &rImage, int xDest, int yDest, int w, int h)
91{
92 const Position *pPos = getPosition();
93 rect region( pPos->getLeft(), pPos->getTop(),
94 pPos->getWidth(), pPos->getHeight() );
95 rect clip( xDest, yDest, w, h );
96 rect inter;
97
98 if( rect::intersect( region, clip, &inter ) )
99 {
100 // Draw a black rectangle under the video to avoid transparency
101 rImage.fillRect( inter.x, inter.y, inter.width, inter.height, 0 );
102 }
103
104 if( m_pVoutWindow )
105 {
106 m_pVoutWindow->show();
107 }
108}

132void CtrlVideo::resizeControl( int width, int height )
133{
134 if( !m_bAutoResize )
135 return;
136
137 WindowManager &rWindowManager =
138 getIntf()->p_sys->p_theme->getWindowManager();
139
140 const Position *pPos = getPosition();
141
142 if( width != pPos->getWidth() || height != pPos->getHeight() )
143 {
144 // new layout dimensions
145 int newWidth = width + m_xShift;
146 int newHeight = height + m_yShift;
147
148 // Resize the layout
149 rWindowManager.startResize( m_rLayout, WindowManager::kResizeSE );
150 rWindowManager.resize( m_rLayout, newWidth, newHeight );
151 rWindowManager.stopResize();
152
153 if( m_pVoutWindow )
154 {
155 m_pVoutWindow->resize( pPos->getWidth(), pPos->getHeight() );
156 m_pVoutWindow->move( pPos->getLeft(), pPos->getTop() );
157 }
158 }
159}

162void CtrlVideo::onUpdate( Subject<VarBool> &rVariable, void *arg )
163{
164 (void)arg;
165
166 // Visibility changed
167 if( &rVariable == m_pVisible )
168 {
169 msg_Dbg( getIntf(), "VideoCtrl : Visibility changed (visible=%d)",
170 isVisible() );
171 notifyLayout();
172 }
173
174 // Active Layout changed
175 if( &rVariable == &m_pLayout->getActiveVar() )
176 {
177 msg_Dbg( getIntf(), "VideoCtrl : Active Layout changed (isActive=%d)",
178 m_pLayout->getActiveVar().get() );
179 }
180
181 VarBool &rFullscreen = VlcProc::instance( getIntf() )->getFullscreenVar();
182 if( &rVariable == &rFullscreen )
183 {
184 msg_Dbg( getIntf(), "VideoCtrl : fullscreen toggled (fullscreen = %d)",
185 rFullscreen.get() );
186 }
187
188 if( isUseable() && !isUsed() )
189 {
190 VoutManager::instance( getIntf() )->requestVout( this );
191 }
192 else if( !isUseable() && isUsed() )
193 {
194 VoutManager::instance( getIntf() )->discardVout( this );
195 }
196}

198void CtrlVideo::attachVoutWindow( VoutWindow* pVoutWindow, int width, int height )
199{
200 width = ( width < 0 ) ? pVoutWindow->getOriginalWidth() : width;
201 height = ( height < 0 ) ? pVoutWindow->getOriginalHeight() : height;
202
203 WindowManager &rWindowManager =
204 getIntf()->p_sys->p_theme->getWindowManager();
205 TopWindow* pWin = getWindow();
206 rWindowManager.show( *pWin );
207
208 if( m_bAutoResize && width && height )
209 {
210 int newWidth = width + m_xShift;
211 int newHeight = height + m_yShift;
212
213 rWindowManager.startResize( m_rLayout, WindowManager::kResizeSE );
214 rWindowManager.resize( m_rLayout, newWidth, newHeight );
215 rWindowManager.stopResize();
216 }
217
218 pVoutWindow->setCtrlVideo( this );
219
220 m_pVoutWindow = pVoutWindow;
221}

224void CtrlVideo::detachVoutWindow( )

226 m_pVoutWindow->setCtrlVideo( NULL );
227 m_pVoutWindow = NULL;
228}

231bool CtrlVideo::isUseable( ) const
232{
233 VarBool &rFullscreen = VlcProc::instance( getIntf() )->getFullscreenVar();
234
235 return isVisible() && // video control is visible
236 m_pLayout->isVisible() && // layout is visible
237 !rFullscreen.get(); // fullscreen is off
238}

241bool CtrlVideo::isUsed( ) const
242{
243 return m_pVoutWindow ? true : false;
