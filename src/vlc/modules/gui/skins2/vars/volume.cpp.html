35Volume::Volume( intf_thread_t *pIntf ): VarPercent( pIntf )
36{
37 // compute preferred step in [0.,1.] range
38 m_step = config_GetFloat( pIntf, "volume-step" )
39 / (float)AOUT_VOLUME_MAX;
40
41 // set current volume from the playlist
42 playlist_t* pPlaylist = pIntf->p_sys->p_playlist;
43 setVolume( var_GetFloat( pPlaylist, "volume" ), false );
44}
45
46
47void Volume::set( float percentage, bool updateVLC )
48{
49 VarPercent::set( percentage );
50 if( updateVLC )
51 {
52 playlist_t* pPlaylist = getIntf()->p_sys->p_playlist;
53 playlist_VolumeSet( pPlaylist, getVolume() );
54 }
55}
56
57
58void Volume::setVolume( float volume, bool updateVLC )
59{
60 // translate from [0.,AOUT_VOLUME_MAX/AOUT_VOLUME_DEFAULT] into [0.,1.]
61 float percentage = (volume > 0.f ) ?
62 volume * AOUT_VOLUME_DEFAULT / AOUT_VOLUME_MAX :
63 0.f;
64 set( percentage, updateVLC );
65}
66
67
68float Volume::getVolume() const
69{
70 // translate from [0.,1.] into [0.,AOUT_VOLUME_MAX/AOUT_VOLUME_DEFAULT]
71 return get() * AOUT_VOLUME_MAX / AOUT_VOLUME_DEFAULT;
72}
73
74
75string Volume::getAsStringPercent() const
76{
77 int value = lround( getVolume() * 100. );
78 // 0 <= value <= 200, so we need 4 chars
79 char str[4];
80 snprintf( str, 4, "%i", value );
81 return string(str);
82}
