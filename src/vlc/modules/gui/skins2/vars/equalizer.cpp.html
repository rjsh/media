38EqualizerBands::EqualizerBands( intf_thread_t *pIntf ): SkinObject( pIntf ),
39 m_isUpdating( false )
40{
41 for( int i = 0; i < kNbBands; i++ )
42 {
43 // Create and observe the band variables
44 VarPercent *pVar = new VarPercent( pIntf );
45 m_cBands[i] = VariablePtr( pVar );
46 pVar->set( 0.5f );
47 pVar->addObserver( this );
48 }
49}
50
51
52EqualizerBands::~EqualizerBands()
53{
54 for( int i = 0; i < kNbBands; i++ )
55 {
56 ((VarPercent*)m_cBands[i].get())->delObserver( this );
57 }
58}

61void EqualizerBands::set( string bands )
62{
63 float val;
64 stringstream ss( bands );
65
66 m_isUpdating = true;
67 // Parse the string
68 for( int i = 0; i < kNbBands; i++ )
69 {
70 ss >> val;
71 // Set the band value in percent
72 ((VarPercent*)m_cBands[i].get())->set( (val + 20) / 40 );
73 }
74 m_isUpdating = false;
75}
76
77
78VariablePtr EqualizerBands::getBand( int band )
79{
80 return m_cBands[band];
81}

84void EqualizerBands::onUpdate( Subject<VarPercent> &rBand, void *arg )
85{
86 (void)rBand; (void)arg;
87 playlist_t *pPlaylist = getIntf()->p_sys->p_playlist;
88 audio_output_t *pAout = playlist_GetAout( pPlaylist );
89
90 // Make sure we are not called from set()
91 if (!m_isUpdating)
92 {
93 float val;
94 stringstream ss;
95 // Write one digit after the floating point
96 ss << setprecision( 1 ) << setiosflags( ios::fixed );
97
98 // Convert the band values to a string
99 val = 40 * ((VarPercent*)m_cBands[0].get())->get() - 20;
100 ss << val;
101 for( int i = 1; i < kNbBands; i++ )
102 {
103 val = 40 * ((VarPercent*)m_cBands[i].get())->get() - 20;
104 ss << " " << val;
105 }
106
107 string bands = ss.str();
108
109 config_PutPsz( getIntf(), "equalizer-bands", bands.c_str() );
110 if( pAout )
111 {
112 // Update the audio output
113 var_SetString( pAout, "equalizer-bands", (char*)bands.c_str() );
114 }
115 }
116
117 if( pAout )
118 vlc_object_release( pAout );
119}

122EqualizerPreamp::EqualizerPreamp( intf_thread_t *pIntf ): VarPercent( pIntf )
123{
124 // Initial value
125 VarPercent::set( 0.8 );
126}

128
129void EqualizerPreamp::set( float percentage, bool updateVLC )
130{
131 playlist_t *pPlaylist = getIntf()->p_sys->p_playlist;
132 audio_output_t *pAout = playlist_GetAout( pPlaylist );
133
134 VarPercent::set( percentage );
135
136 // Avoid infinite loop
137 if( updateVLC )
138 {
139 float val = 40 * percentage - 20;
140
141 config_PutFloat( getIntf(), "equalizer-preamp", val );
142 if( pAout )
143 {
144 // Update the audio output
145 var_SetFloat( pAout, "equalizer-preamp", val );
146 }
147 }
148
149 if( pAout )
150 vlc_object_release( pAout );
151}
