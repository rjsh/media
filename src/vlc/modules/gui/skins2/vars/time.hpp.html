25#ifndef TIME_HPP
26#define TIME_HPP
27
28#include "../utils/var_percent.hpp"
29#include <string>
30
31/// Variable for VLC stream time
32class StreamTime: public VarPercent
33{
34public:
35 StreamTime( intf_thread_t *pIntf ): VarPercent( pIntf ) { }
36 virtual ~StreamTime() { }
37
38 virtual void set( float percentage, bool updateVLC );
39
40 virtual void set( float percentage ) { set( percentage, true ); }
41
42 /// Return a string containing a value from 0 to 100
43 virtual string getAsStringPercent() const;
44 /// Return the current time formatted as a time display (h:mm:ss)
45 virtual string getAsStringCurrTime( bool bShortFormat = false ) const;
46 /// Return the time left formatted as a time display (h:mm:ss)
47 virtual string getAsStringTimeLeft( bool bShortFormat = false ) const;
48 /// Return the duration formatted as a time display (h:mm:ss)
49 virtual string getAsStringDuration( bool bShortFormat = false ) const;
50
51private:
52 /// Convert a number of seconds into "h:mm:ss" format
53 string formatTime( int seconds, bool bShortFormat ) const;
54 /// Return true when there is a non-null input and its position is not 0.0.
55 bool havePosition() const;
56};
57
58#endif
