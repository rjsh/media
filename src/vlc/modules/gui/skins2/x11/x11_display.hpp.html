25#ifndef X11_DISPLAY_HPP
26#define X11_DISPLAY_HPP
27
28#include <X11/Xlib.h>
29#include <X11/Xatom.h>
30#include "../src/skin_common.hpp"
31
32// Helper macros
33#define XDISPLAY m_rDisplay.getDisplay()
34#define XVISUAL m_rDisplay.getVisual()
35#define XPIXELSIZE m_rDisplay.getPixelSize()
36#define XGC m_rDisplay.getGC()
37
38#define NET_WM_SUPPORTED m_rDisplay.m_net_wm_supported
39#define NET_WM_STATE m_rDisplay.m_net_wm_state
40#define NET_WM_STATE_FULLSCREEN m_rDisplay.m_net_wm_state_fullscreen
41#define NET_WM_STATE_ABOVE m_rDisplay.m_net_wm_state_above
42
43#define NET_WM_STAYS_ON_TOP m_rDisplay.m_net_wm_stays_on_top
44#define NET_WM_WINDOW_OPACITY m_rDisplay.m_net_wm_window_opacity
45
46#define NET_WM_PID m_rDisplay.m_net_wm_pid
47
48/// Class for encapsulation of a X11 Display
49class X11Display: public SkinObject
50{
51public:
52 X11Display( intf_thread_t *pIntf );
53 virtual ~X11Display();
54
55 /// Get the display
56 Display *getDisplay() const { return m_pDisplay; }
57
58 /// Get the visual
59 Visual *getVisual() const { return m_pVisual; }
60
61 /// Get the pixel size
62 int getPixelSize() const { return m_pixelSize; }
63
64 /// Get the graphics context
65 GC getGC() const { return m_gc; }
66
67 /// Get the colormap
68 Colormap getColormap() const { return m_colormap; }
69
70 /// Type of function to put RGBA values into a pixel
71 typedef void (X11Display::*MakePixelFunc_t)( uint8_t *pPixel,
72 uint8_t r, uint8_t g, uint8_t b, uint8_t a ) const;
73
74 /// Get a pointer on the right blendPixel implementation
75 MakePixelFunc_t getBlendPixel() const { return blendPixelImpl; }
76
77 /// Get a pointer on the right putPixel implementation
78 MakePixelFunc_t getPutPixel() const { return putPixelImpl; }
79
80 /// Get the pixel value corresponding to the given colors
81 unsigned long getPixelValue( uint8_t r, uint8_t g, uint8_t b ) const;
82
83 /// Get the main window ID
84 Window getMainWindow() const { return m_mainWindow; }
85
86 /// EWMH spec
87 Atom m_net_wm_supported;
88 Atom m_net_wm_state;
89 Atom m_net_wm_state_above;
90 Atom m_net_wm_state_fullscreen;
91
92 Atom m_net_wm_stays_on_top;
93 Atom m_net_wm_window_opacity;
94
95 Atom m_net_wm_pid;
96
97 /// test EWMH capabilities
98 void testEWMH();
99
100private:
101 /// Dummy parent window for the task bar
102 Window m_mainWindow;
103 /// Display parameters
104 Display *m_pDisplay;
105 Visual *m_pVisual;
106 int m_pixelSize;
107 GC m_gc;
108 Colormap m_colormap;
109 int m_redLeftShift, m_redRightShift;
110 int m_greenLeftShift, m_greenRightShift;
111 int m_blueLeftShift, m_blueRightShift;
112 /// Pointer on the right implementation of blendPixel
113 MakePixelFunc_t blendPixelImpl;
114 /// Pointer on the right implementation of putPixel
115 MakePixelFunc_t putPixelImpl;
116
117 template<class type> type putPixel(type r, type g, type b) const;
118 template<class type>
119 type blendPixel(type v,type r, type g, type b,type a) const;
120
121 /// Calculate shifts from a color mask
122 static void getShifts( uint32_t mask, int &rLeftShift, int &rRightShift );
123
124 /// 8 bpp version of blendPixel
125 void blendPixel8( uint8_t *pPixel, uint8_t r, uint8_t g, uint8_t b,
126 uint8_t a ) const;
127
128 /// 16 bpp MSB first version of blendPixel
129 void blendPixel16MSB( uint8_t *pPixel, uint8_t r, uint8_t g, uint8_t b,
130 uint8_t a ) const;
131
132 /// 16 bpp LSB first version of blendPixel
133 void blendPixel16LSB( uint8_t *pPixel, uint8_t r, uint8_t g, uint8_t b,
134 uint8_t a ) const;
135
136 /// 24/32 bpp MSB first version of blendPixel
137 void blendPixel32MSB( uint8_t *pPixel, uint8_t r, uint8_t g, uint8_t b,
138 uint8_t a ) const;
139
140 /// 24/32 bpp LSB first version of blendPixel
141 void blendPixel32LSB( uint8_t *pPixel, uint8_t r, uint8_t g, uint8_t b,
142 uint8_t a ) const;
143
144 /// 8 bpp version of putPixel
145 void putPixel8( uint8_t *pPixel, uint8_t r, uint8_t g, uint8_t b,
146 uint8_t a ) const;
147
148 /// 16 bpp MSB first version of putPixel
149 void putPixel16MSB( uint8_t *pPixel, uint8_t r, uint8_t g, uint8_t b,
150 uint8_t a ) const;
151
152 /// 16 bpp LSB first version of putPixel
153 void putPixel16LSB( uint8_t *pPixel, uint8_t r, uint8_t g, uint8_t b,
154 uint8_t a ) const;
155
156 /// 24/32 bpp MSB first version of putPixel
157 void putPixel32MSB( uint8_t *pPixel, uint8_t r, uint8_t g, uint8_t b,
158 uint8_t a ) const;
159
160 /// 24/32 bpp LSB first version of putPixel
161 void putPixel32LSB( uint8_t *pPixel, uint8_t r, uint8_t g, uint8_t b,
162 uint8_t a ) const;
163
164};
165
166#endif
