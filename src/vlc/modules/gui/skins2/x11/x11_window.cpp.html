41X11Window::X11Window( intf_thread_t *pIntf, GenericWindow &rWindow,
42 X11Display &rDisplay, bool dragDrop, bool playOnDrop,
43 X11Window *pParentWindow, GenericWindow::WindowType_t type ):
44 OSWindow( pIntf ), m_rDisplay( rDisplay ), m_pParent( pParentWindow ),
45 m_dragDrop( dragDrop ), m_pDropTarget( NULL ), m_type ( type )
46{
47 XSetWindowAttributes attr;
48 unsigned long valuemask;
49 string name_type;
50
51 if( type == GenericWindow::FullscreenWindow )
52 {
53 m_wnd_parent = DefaultRootWindow( XDISPLAY );
54
55 int i_screen = DefaultScreen( XDISPLAY );
56
57 attr.event_mask = ExposureMask | StructureNotifyMask;
58 attr.background_pixel = BlackPixel( XDISPLAY, i_screen );
59 attr.backing_store = Always;
60 valuemask = CWBackingStore | CWBackPixel | CWEventMask;
61
62 if( NET_WM_STATE_FULLSCREEN == None )
63 {
64 attr.override_redirect = True;
65 valuemask = valuemask | CWOverrideRedirect;
66 }
67
68 name_type = "Fullscreen";
69 }
70 else if( type == GenericWindow::VoutWindow )
71 {
72 m_wnd_parent = pParentWindow->m_wnd;
73
74 int i_screen = DefaultScreen( XDISPLAY );
75
76 attr.event_mask = ExposureMask | StructureNotifyMask;
77 attr.backing_store = Always;
78 attr.background_pixel = BlackPixel( XDISPLAY, i_screen );
79 valuemask = CWBackingStore | CWBackPixel | CWEventMask;
80
81 name_type = "VoutWindow";
82 }
83 else if( type == GenericWindow::FscWindow )
84 {
85 m_wnd_parent = DefaultRootWindow( XDISPLAY );
86
87 attr.event_mask = ExposureMask | StructureNotifyMask;
88 valuemask = CWEventMask;
89
90 name_type = "FscWindow";
91 }
92 else
93 {
94 m_wnd_parent = DefaultRootWindow( XDISPLAY );
95
96 attr.event_mask = ExposureMask | StructureNotifyMask;
97 valuemask = CWEventMask;
98
99 name_type = "TopWindow";
100 }
101
102 // Create the window
103 m_wnd = XCreateWindow( XDISPLAY, m_wnd_parent, -10, 0, 10, 10, 0, 0,
104 InputOutput, CopyFromParent, valuemask, &attr );
105
106 // wait for X server to process the previous commands
107 XSync( XDISPLAY, false );
108
109 // Set the colormap for 8bpp mode
110 if( XPIXELSIZE == 1 )
111 {
112 XSetWindowColormap( XDISPLAY, m_wnd, m_rDisplay.getColormap() );
113 }
114
115 // Select events received by the window
116 long event_mask;
117 if( type == GenericWindow::VoutWindow )
118 {
119 event_mask = ExposureMask|KeyPressMask|
120 LeaveWindowMask|FocusChangeMask;
121 }
122 else
123 {
124 event_mask = ExposureMask|KeyPressMask|
125 PointerMotionMask|ButtonPressMask|ButtonReleaseMask|
126 LeaveWindowMask|FocusChangeMask;
127 }
128 XSelectInput( XDISPLAY, m_wnd, event_mask );
129
130 // Store a pointer on the generic window in a map
131 X11Factory *pFactory = (X11Factory*)X11Factory::instance( getIntf() );
132 pFactory->m_windowMap[m_wnd] = &rWindow;
133
134 // Changing decorations
135 struct {
136 unsigned long flags;
137 unsigned long functions;
138 unsigned long decorations;
139 signed long input_mode;
140 unsigned long status;
141 } motifWmHints;
142 Atom hints_atom = XInternAtom( XDISPLAY, "_MOTIF_WM_HINTS", False );
143 motifWmHints.flags = 2; // MWM_HINTS_DECORATIONS;
144 motifWmHints.decorations = 0;
145 XChangeProperty( XDISPLAY, m_wnd, hints_atom, hints_atom, 32,
146 PropModeReplace, (unsigned char *)&motifWmHints,
147 sizeof( motifWmHints ) / sizeof( uint32_t ) );
148
149 // Drag & drop
150 if( m_dragDrop )
151 {
152 // Create a Dnd object for this window
153 m_pDropTarget = new X11DragDrop( getIntf(), m_rDisplay, m_wnd,
154 playOnDrop, &rWindow );
155
156 // Register the window as a drop target
157 Atom xdndAtom = XInternAtom( XDISPLAY, "XdndAware", False );
158 char xdndVersion = 4;
159 XChangeProperty( XDISPLAY, m_wnd, xdndAtom, XA_ATOM, 32,
160 PropModeReplace, (unsigned char *)&xdndVersion, 1 );
161
162 // Store a pointer to be used in X11Loop
163 pFactory->m_dndMap[m_wnd] = m_pDropTarget;
164 }
165
166 // Change the window title
167 string name_window = "VLC (" + name_type + ")";
168 XStoreName( XDISPLAY, m_wnd, name_window.c_str() );
169
170 // Set the WM_TRANSIENT_FOR property
171 if( type == GenericWindow::FscWindow )
172 {
173 // Associate the fsc window to the fullscreen window
174 VoutManager* pVoutManager = VoutManager::instance( getIntf() );
175 GenericWindow* pWin = pVoutManager->getVoutMainWindow();
176 Window wnd = (Window) pWin->getOSHandle();
177 XSetTransientForHint( XDISPLAY, m_wnd, wnd );
178 }
179 else
180 {
181 // Associate the regular top-level window to the offscren main window
182 XSetTransientForHint( XDISPLAY, m_wnd, m_rDisplay.getMainWindow() );
183 }
184
185 // initialize Class Hint
186 XClassHint classhint;
187 classhint.res_name = (char*) "vlc";
188 classhint.res_class = (char*) "Vlc";
189 XSetClassHint( XDISPLAY, m_wnd, &classhint );
190
191 // copies WM_HINTS from the main window
192 XWMHints *wm = XGetWMHints( XDISPLAY, m_rDisplay.getMainWindow() );
193 if( wm )
194 {
195 XSetWMHints( XDISPLAY, m_wnd, wm );
196 XFree( wm );
197 }
198
199 // initialize WM_CLIENT_MACHINE
200 char* hostname = NULL;
201 long host_name_max = sysconf( _SC_HOST_NAME_MAX );
202 if( host_name_max <= 0 )
203 host_name_max = _POSIX_HOST_NAME_MAX;
204 hostname = new char[host_name_max];
205 if( hostname && gethostname( hostname, host_name_max ) == 0 )
206 {
207 hostname[host_name_max - 1] = '\0';
208
209 XTextProperty textprop;
210 textprop.value = (unsigned char *) hostname;
211 textprop.encoding = XA_STRING;
212 textprop.format = 8;
213 textprop.nitems = strlen( hostname );
214 XSetWMClientMachine( XDISPLAY, m_wnd, &textprop);
215 }
216 delete[] hostname;
217
218 // initialize EWMH pid
219 pid_t pid = getpid();
220 assert( NET_WM_PID != None );
221 XChangeProperty( XDISPLAY, m_wnd, NET_WM_PID, XA_CARDINAL, 32,
222 PropModeReplace, (unsigned char *)&pid, 1 );
223
224}

226
227X11Window::~X11Window()
228{
229 X11Factory *pFactory = (X11Factory*)X11Factory::instance( getIntf() );
230 pFactory->m_windowMap[m_wnd] = NULL;
231 pFactory->m_dndMap[m_wnd] = NULL;
232
233 delete m_pDropTarget;
234
235 XDestroyWindow( XDISPLAY, m_wnd );
236 XSync( XDISPLAY, False );
237}

239void X11Window::reparent( uint32_t OSHandle, int x, int y, int w, int h )
240{
241 // Reparent the window
242 Window new_parent =
243 OSHandle ? (Window) OSHandle : DefaultRootWindow( XDISPLAY );
244
245 XReparentWindow( XDISPLAY, m_wnd, new_parent, x, y);
246 if( w && h )
247 XResizeWindow( XDISPLAY, m_wnd, w, h );
248
249 m_wnd_parent = new_parent;
250}

253void X11Window::show() const
254{
255 // Map the window
256 if( m_type == GenericWindow::VoutWindow )
257 {
258 XLowerWindow( XDISPLAY, m_wnd );
259 XMapWindow( XDISPLAY, m_wnd );
260 }
261 else if( m_type == GenericWindow::FullscreenWindow )
262 {
263 XMapRaised( XDISPLAY, m_wnd );
264 setFullscreen();
265 toggleOnTop( true );
266 }
267 else if( m_type == GenericWindow::FscWindow )
268 {
269 XMapRaised( XDISPLAY, m_wnd );
270 toggleOnTop( true );
271 }
272 else
273 {
274 XMapRaised( XDISPLAY, m_wnd );
275 }
276}

279void X11Window::hide() const

281 // Unmap the window
282 XUnmapWindow( XDISPLAY, m_wnd );

285void X11Window::moveResize( int left, int top, int width, int height ) const
286{
287 if( width && height )
288 XMoveResizeWindow( XDISPLAY, m_wnd, left, top, width, height );

300void X11Window::setOpacity( uint8_t value ) const
301{
302 if( NET_WM_WINDOW_OPACITY == None )
303 return;
304
305 if( 255==value )
306 XDeleteProperty(XDISPLAY, m_wnd, NET_WM_WINDOW_OPACITY);
307 else
308 {
309 uint32_t opacity = value * ((uint32_t)-1/255);
310 XChangeProperty(XDISPLAY, m_wnd, NET_WM_WINDOW_OPACITY, XA_CARDINAL, 32,
311 PropModeReplace, (unsigned char *) &opacity, 1L);
312 }
313 XSync( XDISPLAY, False );
314}

317void X11Window::setFullscreen( ) const
318{
319 if( NET_WM_STATE_FULLSCREEN != None )
320 {
321 XClientMessageEvent event;
322 memset( &event, 0, sizeof( XClientMessageEvent ) );
323
324 event.type = ClientMessage;
325 event.message_type = NET_WM_STATE;
326 event.display = XDISPLAY;
327 event.window = m_wnd;
328 event.format = 32;
329 event.data.l[ 0 ] = 1;
330 event.data.l[ 1 ] = NET_WM_STATE_FULLSCREEN;
331
332 XSendEvent( XDISPLAY,
333 DefaultRootWindow( XDISPLAY ),
334 False, SubstructureNotifyMask|SubstructureRedirectMask,
335 (XEvent*)&event );
336 }
337}

340void X11Window::toggleOnTop( bool onTop ) const
341{
342 if( NET_WM_STAYS_ON_TOP != None )
343 {
344 /* Switch "on top" status */
345 XClientMessageEvent event;
346 memset( &event, 0, sizeof( XClientMessageEvent ) );
347
348 event.type = ClientMessage;
349 event.message_type = NET_WM_STATE;
350 event.display = XDISPLAY;
351 event.window = m_wnd;
352 event.format = 32;
353 event.data.l[ 0 ] = onTop; /* set property */
354 event.data.l[ 1 ] = NET_WM_STAYS_ON_TOP;
355
356 XSendEvent( XDISPLAY, DefaultRootWindow( XDISPLAY ),
357 False, SubstructureNotifyMask|SubstructureRedirectMask, (XEvent*)&event );
358 }
359 else if( NET_WM_STATE_ABOVE != None )
360 {
361 /* Switch "above" state */
362 XClientMessageEvent event;
363 memset( &event, 0, sizeof( XClientMessageEvent ) );
364
365 event.type = ClientMessage;
366 event.message_type = NET_WM_STATE;
367 event.display = XDISPLAY;
368 event.window = m_wnd;
369 event.format = 32;
370 event.data.l[ 0 ] = onTop; /* set property */
371 event.data.l[ 1 ] = NET_WM_STATE_ABOVE;
372
373 XSendEvent( XDISPLAY, DefaultRootWindow( XDISPLAY ),
374 False, SubstructureNotifyMask|SubstructureRedirectMask, (XEvent*)&event );
375 }
376}
