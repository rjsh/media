38X11Graphics::X11Graphics( intf_thread_t *pIntf, X11Display &rDisplay,
39 int width, int height ):
40 OSGraphics( pIntf ), m_rDisplay( rDisplay ), m_width( width ),
41 m_height( height )
42{
43 // Get the display paramaters
44 int screen = DefaultScreen( XDISPLAY );
45 int depth = DefaultDepth( XDISPLAY, screen );
46
47 // X11 doesn't accept that !
48 if( width == 0 || height == 0 )
49 {
50 // Avoid a X11 Bad Value error
51 width = height = 1;
52 msg_Err( getIntf(), "invalid image size (null width or height)" );
53 }
54
55 // Create a pixmap
56 m_pixmap = XCreatePixmap( XDISPLAY, DefaultRootWindow( XDISPLAY ),
57 width, height, depth);
58
59 // Create the transparency mask (everything is transparent initially)
60 m_mask = XCreateRegion();
61
62 // Create a Graphics Context that does not generate GraphicsExpose events
63 XGCValues xgcvalues;
64 xgcvalues.graphics_exposures = False;
65 m_gc = XCreateGC( XDISPLAY, m_pixmap, GCGraphicsExposures, &xgcvalues );
66}

68
69X11Graphics::~X11Graphics()
70{
71 XFreeGC( XDISPLAY, m_gc );
72 XDestroyRegion( m_mask );
73 XFreePixmap( XDISPLAY, m_pixmap );
74}

77void X11Graphics::clear( int xDest, int yDest, int width, int height )
78{
79 if( width <= 0 || height <= 0 )
80 {
81 // Clear the transparency mask completely
82 XDestroyRegion( m_mask );
83 m_mask = XCreateRegion();
84 }
85 else
86 {
87 // remove this area from the mask
88 XRectangle rect;
89 rect.x = xDest;
90 rect.y = yDest;
91 rect.width = width;
92 rect.height = height;
93 Region regMask = XCreateRegion();
94 XUnionRectWithRegion( &rect, regMask, regMask );
95 XSubtractRegion( m_mask, regMask, m_mask );
96 XDestroyRegion( regMask );
97 }
98}

101void X11Graphics::drawGraphics( const OSGraphics &rGraphics, int xSrc,
102 int ySrc, int xDest, int yDest, int width,
103 int height )
104{
105 const X11Graphics& rGraph = (X11Graphics&)rGraphics;
106
107 // check and adapt to source if needed
108 if( !checkBoundaries( 0, 0, rGraph.getWidth(), rGraph.getHeight(),
109 xSrc, ySrc, width, height ) )
110 {
111 msg_Err( getIntf(), "nothing to draw from graphics source" );
112 return;
113 }
114
115 // check destination
116 if( !checkBoundaries( 0, 0, m_width, m_height,
117 xDest, yDest, width, height ) )
118 {
119 msg_Err( getIntf(), "out of reach destination! pls, debug your skin" );
120 return;
121 }
122
123 // Source drawable
124 Drawable src = rGraph.getDrawable();
125
126 // Create the mask for transparency
127 Region voidMask = XCreateRegion();
128 XRectangle rect;
129 rect.x = xSrc;
130 rect.y = ySrc;
131 rect.width = width;
132 rect.height = height;
133 Region clipMask = XCreateRegion();
134 XUnionRectWithRegion( &rect, voidMask, clipMask );
135 Region mask = XCreateRegion();
136 XIntersectRegion( rGraph.getMask(), clipMask, mask );
137 XDestroyRegion( clipMask );
138 XDestroyRegion( voidMask );
139 XOffsetRegion( mask, xDest - xSrc, yDest - ySrc );
140
141 // Copy the pixmap
142 XSetRegion( XDISPLAY, m_gc, mask );
143 XCopyArea( XDISPLAY, src, m_pixmap, m_gc, xSrc, ySrc, width, height,
144 xDest, yDest );
145
146 // Add the source mask to the mask of the graphics
147 Region newMask = XCreateRegion();
148 XUnionRegion( m_mask, mask, newMask );
149 XDestroyRegion( mask );
150 XDestroyRegion( m_mask );
151 m_mask = newMask;
152}

155void X11Graphics::drawBitmap( const GenericBitmap &rBitmap, int xSrc,
156 int ySrc, int xDest, int yDest, int width,
157 int height, bool blend )
158{
159 // check and adapt to source if needed
160 if( !checkBoundaries( 0, 0, rBitmap.getWidth(), rBitmap.getHeight(),
161 xSrc, ySrc, width, height ) )
162 {
163 msg_Err( getIntf(), "empty source! pls, debug your skin" );
164 return;
165 }
166
167 // check destination
168 if( !checkBoundaries( 0, 0, m_width, m_height,
169 xDest, yDest, width, height ) )
170 {
171 msg_Err( getIntf(), "out of reach destination! pls, debug your skin" );
172 return;
173 }
174
175 // Get a buffer on the image data
176 uint8_t *pBmpData = rBitmap.getData();
177 if( pBmpData == NULL )
178 {
179 // Nothing to draw
180 return;
181 }
182
183 // Force pending XCopyArea to be sent to the X Server
184 // before issuing an XGetImage.
185 XSync( XDISPLAY, False );
186
187 // Get the image from the pixmap
188 XImage *pImage = XGetImage( XDISPLAY, m_pixmap, xDest, yDest, width,
189 height, AllPlanes, ZPixmap );
190 if( pImage == NULL )
191 {
192 msg_Dbg( getIntf(), "XGetImage returned NULL" );
193 return;
194 }
195 char *pData = pImage->data;
196
197 // Get the padding of this image
198 int pad = pImage->bitmap_pad >> 3;
199 int shift = ( pad - ( (width * XPIXELSIZE) % pad ) ) % pad;
200
201 // Mask for transparency
202 Region mask = XCreateRegion();
203
204 // Get a pointer on the right X11Display::makePixel method
205 X11Display::MakePixelFunc_t makePixelFunc = ( blend ?
206 m_rDisplay.getBlendPixel() : m_rDisplay.getPutPixel() );
207
208 // Skip the first lines of the image
209 pBmpData += 4 * ySrc * rBitmap.getWidth();
210
211 // Copy the bitmap on the image and compute the mask
212 for( int y = 0; y < height; y++ )
213 {
214 // Skip uninteresting bytes at the beginning of the line
215 pBmpData += 4 * xSrc;
216 // Flag to say whether the previous pixel on the line was visible
217 bool wasVisible = false;
218 // Beginning of the current visible segment on the line
219 int visibleSegmentStart = 0;
220 for( int x = 0; x < width; x++ )
221 {
222 uint8_t b = *(pBmpData++);
223 uint8_t g = *(pBmpData++);
224 uint8_t r = *(pBmpData++);
225 uint8_t a = *(pBmpData++);
226 // Draw the pixel
227 (m_rDisplay.*makePixelFunc)( (uint8_t*)pData, r, g, b, a );
228 pData += XPIXELSIZE;
229 if( a > 0 )
230 {
231 // Pixel is visible
232 if( ! wasVisible )
233 {
234 // Beginning of a visible segment
235 visibleSegmentStart = x;
236 }
237 wasVisible = true;
238 }
239 else
240 {
241 // Pixel is transparent
242 if( wasVisible )
243 {
244 // End of a visible segment: add it to the mask
245 addHSegmentInRegion( mask, visibleSegmentStart, x, y );
246 }
247 wasVisible = false;
248 }
249 }
250 if( wasVisible )
251 {
252 // End of a visible segment: add it to the mask
253 addHSegmentInRegion( mask, visibleSegmentStart, width, y );
254 }
255 pData += shift;
256 // Skip uninteresting bytes at the end of the line
257 pBmpData += 4 * (rBitmap.getWidth() - width - xSrc);
258 }
259
260 // Apply the mask to the graphics context
261 XOffsetRegion( mask, xDest, yDest );
262 XSetRegion( XDISPLAY, m_gc, mask );
263 // Copy the image on the pixmap
264 XPutImage( XDISPLAY, m_pixmap, m_gc, pImage, 0, 0, xDest, yDest, width,
265 height);
266 XDestroyImage( pImage );
267
268 // Add the bitmap mask to the global graphics mask
269 Region newMask = XCreateRegion();
270 XUnionRegion( mask, m_mask, newMask );
271 XDestroyRegion( m_mask );
272 m_mask = newMask;
273
274 XDestroyRegion( mask );
275}

278void X11Graphics::fillRect( int left, int top, int width, int height,
279 uint32_t color )
280{
281 // Update the mask with the rectangle area
282 Region newMask = XCreateRegion();
283 XRectangle rect;
284 rect.x = left;
285 rect.y = top;
286 rect.width = width;
287 rect.height = height;
288 XUnionRectWithRegion( &rect, m_mask, newMask );
289 XDestroyRegion( m_mask );
290 m_mask = newMask;
291
292 // Draw the rectangle
293 XGCValues gcVal;
294 gcVal.foreground = m_rDisplay.getPixelValue( color >> 16, color >> 8, color );
295 XChangeGC( XDISPLAY, m_gc, GCForeground, &gcVal );
296 XSetRegion( XDISPLAY, m_gc, m_mask );
297 XFillRectangle( XDISPLAY, m_pixmap, m_gc, left, top, width, height );
298}

301void X11Graphics::drawRect( int left, int top, int width, int height,
302 uint32_t color )
303{
304 // Update the mask with the rectangle
305 addHSegmentInRegion( m_mask, left, left + width, top );
306 addHSegmentInRegion( m_mask, left, left + width, top + height );
307 addVSegmentInRegion( m_mask, top, top + height, left );
308 addVSegmentInRegion( m_mask, top, top + height, left + width );
309
310 // Draw the rectangle
311 XGCValues gcVal;
312 gcVal.foreground = m_rDisplay.getPixelValue( color >> 16, color >> 8, color );
313 XChangeGC( XDISPLAY, m_gc, GCForeground, &gcVal );
314 XSetRegion( XDISPLAY, m_gc, m_mask );
315 XDrawRectangle( XDISPLAY, m_pixmap, m_gc, left, top, width - 1, height - 1 );
316}

319void X11Graphics::applyMaskToWindow( OSWindow &rWindow )
320{
321 // Get the target window
322 Window win = ((X11Window&)rWindow).getDrawable();
323
324 // Change the shape of the window
325 XShapeCombineRegion( XDISPLAY, win, ShapeBounding, 0, 0, m_mask,
326 ShapeSet );
327}

330void X11Graphics::copyToWindow( OSWindow &rWindow, int xSrc, int ySrc,
331 int width, int height, int xDest, int yDest )
332{
333 // Destination window
334 Drawable dest = ((X11Window&)rWindow).getDrawable();
335
336 XCopyArea( XDISPLAY, m_pixmap, dest, XGC, xSrc, ySrc, width, height,
337 xDest, yDest );

347inline void X11Graphics::addHSegmentInRegion( Region &rMask, int xStart,
348 int xEnd, int y )
349{
350 XRectangle rect;
351 rect.x = xStart;
352 rect.y = y;
353 rect.width = xEnd - xStart;
354 rect.height = 1;
355 Region newMask = XCreateRegion();
356 XUnionRectWithRegion( &rect, rMask, newMask );
357 XDestroyRegion( rMask );
358 rMask = newMask;
359}
360
361
362inline void X11Graphics::addVSegmentInRegion( Region &rMask, int yStart,
363 int yEnd, int x )
364{
365 XRectangle rect;
366 rect.x = x;
367 rect.y = yStart;
368 rect.width = 1;
369 rect.height = yEnd - yStart;
370 Region newMask = XCreateRegion();
371 XUnionRectWithRegion( &rect, rMask, newMask );
372 XDestroyRegion( rMask );
373 rMask = newMask;
374}

376bool X11Graphics::checkBoundaries( int x_src, int y_src,
377 int w_src, int h_src,
378 int& x_target, int& y_target,
379 int& w_target, int& h_target )
380{
381 // set valid width and height
382 w_target = (w_target > 0) ? w_target : w_src;
383 h_target = (h_target > 0) ? h_target : h_src;
384
385 // clip source if needed
386 rect srcRegion( x_src, y_src, w_src, h_src );
387 rect targetRegion( x_target, y_target, w_target, h_target );
388 rect inter;
389 if( rect::intersect( srcRegion, targetRegion, &inter ) )
390 {
391 x_target = inter.x;
392 y_target = inter.y;
393 w_target = inter.width;
394 h_target = inter.height;
395 return true;
396 }
397 return false;
398}
