25#ifndef X11_GRAPHICS_HPP
26#define X11_GRAPHICS_HPP
27
28#include <X11/Xlib.h>
29#include <X11/Xutil.h>
30
31#include "../src/os_graphics.hpp"
32
33class X11Display;
34class GenericWindow;
35class GenericBitmap;
36
37/// X11 implementation of OSGraphics
38class X11Graphics: public OSGraphics
39{
40public:
41 X11Graphics( intf_thread_t *pIntf, X11Display &rDisplay, int width,
42 int height);
43
44 virtual ~X11Graphics();
45
46 /// Clear the graphics
47 virtual void clear( int xDest = 0, int yDest = 0,
48 int width = -1, int height = -1 );
49
50 /// Draw another graphics on this one
51 virtual void drawGraphics( const OSGraphics &rGraphics, int xSrc = 0,
52 int ySrc = 0, int xDest = 0, int yDest = 0,
53 int width = -1, int height = -1 );
54
55 /// Render a bitmap on this graphics
56 virtual void drawBitmap( const GenericBitmap &rBitmap, int xSrc = 0,
57 int ySrc = 0, int xDest = 0, int yDest = 0,
58 int width = -1, int height = -1,
59 bool blend = false );
60
61 /// Draw a filled rectangle on the grahics (color is #RRGGBB)
62 virtual void fillRect( int left, int top, int width, int height,
63 uint32_t color );
64
65 /// Draw an empty rectangle on the grahics (color is #RRGGBB)
66 virtual void drawRect( int left, int top, int width, int height,
67 uint32_t color );
68
69 /// Set the shape of a window with the mask of this graphics.
70 virtual void applyMaskToWindow( OSWindow &rWindow );
71
72 /// Copy the graphics on a window
73 virtual void copyToWindow( OSWindow &rWindow, int xSrc,
74 int ySrc, int width, int height,
75 int xDest, int yDest );
76
77 /// Tell whether the pixel at the given position is visible
78 virtual bool hit( int x, int y ) const;
79
80 /// Getters
81 virtual int getWidth() const { return m_width; }
82 virtual int getHeight() const { return m_height; }
83
84 /// Get the pixmap ID
85 Pixmap getDrawable() const { return m_pixmap; }
86 /// Get the transparency mask
87 Region getMask() const { return m_mask; }
88
89private:
90 /// X11 display
91 X11Display &m_rDisplay;
92 /// Size of the image
93 int m_width, m_height;
94 /// Pixmap
95 Pixmap m_pixmap;
96 /// Transparency mask
97 Region m_mask;
98 /// Graphics context
99 GC m_gc;
100
101 /// Add an horizontal segment in a region
102 void addHSegmentInRegion( Region &rMask, int xStart, int xEnd, int y );
103 /// Add a vertical segment in a region
104 void addVSegmentInRegion( Region &rMask, int yStart, int yEnd, int x );
105 /// check boundaries for graphics and bitmaps
106 bool checkBoundaries( int x_src, int y_src, int w_src, int h_src,
107 int& x_target, int& y_target,
108 int& w_target, int& h_target );
109};
110
111
112#endif
