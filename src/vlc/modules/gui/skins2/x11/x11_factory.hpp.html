40/// Class used to instanciate X11 specific objects
41class X11Factory: public OSFactory
42{
43private:
44 /** ptrmap is an associative container (like std::map, in fact it builds
45 * on std::map) that provides only operator[] in non-const and const
46 * variants, and has the property that a (const) lookup of a non-existent
47 * key does not cause an empty node to be inserted. Instead it returns
48 * NULL if there was no entry; so it only stores pointers. */
49 template<class K, class V> class ptrmap {
50 private:
51 typedef V *value_type;
52 typedef std::map<K,value_type> map_t;
53 map_t m_map;
54 ptrmap &operator=(const ptrmap &);
55 ptrmap(const ptrmap &);
56 public:
57 ptrmap() { }
58 value_type &operator[](K k) { return m_map.operator[](k); }
59 value_type operator[](K k) const
60 {
61 typename map_t::const_iterator i=m_map.find(k);
62 return i!=m_map.end() ? i->second : NULL;
63 }
64 };
65public:
66 /// Map to find the GenericWindow* associated to a X11Window
67 ptrmap<Window, GenericWindow> m_windowMap;
68 /// Map to find the Dnd object (X11DragDrop*) associated to a X11Window
69 ptrmap<Window, X11DragDrop> m_dndMap;

71 X11Factory( intf_thread_t *pIntf );
