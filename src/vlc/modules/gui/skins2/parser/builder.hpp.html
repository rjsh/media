39class Theme;
40class Bezier;
41class CmdGeneric;
42class GenericFont;
43class Position;
44class GenericRect;

47/// Class for skin construction
48class Builder: public SkinObject
49{
50public:
51 Builder( intf_thread_t *pIntf, const BuilderData &rData,
52 const string &rPath );
53 virtual ~Builder();
54
55 /// Create a Theme object, ready to use.
56 /// Return NULL in case of problem
57 Theme *build();
58
59 /// Parse an action tag and returns a command
60 CmdGeneric *parseAction( const string &rAction );
61
62private:
63 /// Data from the XML
64 const BuilderData &m_rData;
65 /// Path of the theme
66 const string m_path;
67
68 /// Theme under construction
69 Theme *m_pTheme;
70
71 void addTheme( const BuilderData::Theme &rData );
72 void addIniFile( const BuilderData::IniFile &rData );
73 void addBitmap( const BuilderData::Bitmap &rData );
74 void addSubBitmap( const BuilderData::SubBitmap &rData );
75 void addBitmapFont( const BuilderData::BitmapFont &rData );
76 void addFont( const BuilderData::Font &rData );
77 void addPopupMenu( const BuilderData::PopupMenu &rData );
78 void addMenuItem( const BuilderData::MenuItem &rData );
79 void addMenuSeparator( const BuilderData::MenuSeparator &rData );
80 void addWindow( const BuilderData::Window &rData );
81 void addLayout( const BuilderData::Layout &rData );
82 void addAnchor( const BuilderData::Anchor &rData );
83 void addButton( const BuilderData::Button &rData );
84 void addCheckbox( const BuilderData::Checkbox &rData );
85 void addImage( const BuilderData::Image &rData );
86 void addPanel( const BuilderData::Panel &rData );
87 void addText( const BuilderData::Text &rData );
88 void addRadialSlider( const BuilderData::RadialSlider &rData );
89 void addSlider( const BuilderData::Slider &rData );
90 void addList( const BuilderData::List &rData );
91 void addTree( const BuilderData::Tree &rData );
92 void addVideo( const BuilderData::Video &rData );
93
94 /// Helper for build(); gluing BuilderData::list<T>s to addType(T&)
95 template<class T> void add_objects(const std::list<T> &list,
96 void (Builder::*addfn)(const T &));
97
98 /// Compute the position of a control
99 const Position makePosition( const string &rLeftTop,
100 const string &rRightBottom,
101 int xPos, int yPos, int width, int height,
102 const GenericRect &rRect,
103 bool xKeepRatio = false,
104 bool yKeepRatio = false ) const;
105
106 // Build the full path of a file
107 string getFilePath( const string &fileName ) const;
108
109 /// Get a font from its id
110 GenericFont *getFont( const string &fontId );
111
112 /// Function to parse "points" tags
113 Bezier *getPoints( const char *pTag ) const;
114
115 /// Compute a color value
116 uint32_t getColor( const string &rVal ) const;
117
118 /// Image handler (used to load image files)
119 image_handler_t *m_pImageHandler;
120};
