24#ifndef EXPR_EVALUATOR_HPP
25#define EXPR_EVALUATOR_HPP
26
27#include "../src/skin_common.hpp"
28#include <list>
29#include <string>
30
31/// Expression evaluator using Reverse Polish Notation
32class ExprEvaluator: public SkinObject
33{
34public:
35 ExprEvaluator( intf_thread_t *pIntf ): SkinObject( pIntf ) { }
36 ~ExprEvaluator() { }
37
38 /// Clear the RPN stack and parse an expression
39 void parse( const string &rExpr );
40
41 /// Pop the first token from the RPN stack.
42 /// Return NULL when the stack is empty.
43 string getToken();
44
45private:
46 /// RPN stack
47 list<string> m_stack;
48
49 /// Returns true if op1 has precedency over op2
50 bool hasPrecedency( const string &op1, const string &op2 ) const;
51};
52
53#endif
