30XMLParser::XMLParser( intf_thread_t *pIntf, const string &rFileName )
31 : SkinObject( pIntf ), m_pXML( NULL ), m_pReader( NULL ), m_pStream( NULL )
32{
33 m_pXML = xml_Create( pIntf );
34 if( !m_pXML )
35 {
36 msg_Err( getIntf(), "cannot initialize xml" );
37 return;
38 }
39
40 LoadCatalog();
41
42 char *psz_uri = vlc_path2uri( rFileName.c_str(), NULL );
43 m_pStream = stream_UrlNew( pIntf, psz_uri );
44 free( psz_uri );
45 if( !m_pStream )
46 {
47 msg_Err( getIntf(), "failed to open %s for reading",
48 rFileName.c_str() );
49 return;
50 }
51
52 m_pReader = xml_ReaderCreate( m_pXML, m_pStream );
53 if( !m_pReader )
54 {
55 msg_Err( getIntf(), "failed to open %s for parsing",
56 rFileName.c_str() );
57 return;
58 }
59
60 xml_ReaderUseDTD( m_pReader );
61}
62
63
64XMLParser::~XMLParser()
65{
66 if( m_pReader ) xml_ReaderDelete( m_pReader );
67 if( m_pXML ) xml_Delete( m_pXML );
68 if( m_pStream ) stream_Delete( m_pStream );
69}

72void XMLParser::LoadCatalog()
73{
74 // Get the resource path and look for the DTD
75 OSFactory *pOSFactory = OSFactory::instance( getIntf() );
76 const list<string> &resPath = pOSFactory->getResourcePath();
77 const string &sep = pOSFactory->getDirSeparator();
78 list<string>::const_iterator it;
79
80 struct stat statBuf;
81
82 // Try to load the catalog first (needed at least on win32 where
83 // we don't have a default catalog)
84 for( it = resPath.begin(); it != resPath.end(); ++it )
85 {
86 string catalog_path = (*it) + sep + "skin.catalog";
87 if( !stat( catalog_path.c_str(), &statBuf ) )
88 {
89 msg_Dbg( getIntf(), "Using catalog %s", catalog_path.c_str() );
90 xml_CatalogLoad( m_pXML, catalog_path.c_str() );
91 break;
92 }
93 }
94 if( it == resPath.end() )
95 {
96 // Ok, try the default one
97 xml_CatalogLoad( m_pXML, NULL );
98 }
99
100 for( it = resPath.begin(); it != resPath.end(); ++it )
101 {
102 string path = (*it) + sep + "skin.dtd";
103 if( !stat( path.c_str(), &statBuf ) )
104 {
105 // DTD found
106 msg_Dbg( getIntf(), "using DTD %s", path.c_str() );
107
108 // Add an entry in the default catalog
109 xml_CatalogAdd( m_pXML, "public",
110 "-//VideoLAN//DTD VLC Skins V"
111 SKINS_DTD_VERSION "//EN", path.c_str() );
112 break;
113 }
114 }
115 if( it == resPath.end() )
116 {
117 msg_Err( getIntf(), "cannot find the skins DTD");
118 }
119}
120
121bool XMLParser::parse()
122{
123 const char *node;
124 int type;
125
126 if( !m_pReader ) return false;
127
128 m_errors = false;
129
130 while( (type = xml_ReaderNextNode( m_pReader, &node )) > 0 )
131 {
132 if( m_errors ) return false;
133
134 switch( type )
135 {
136 case XML_READER_STARTELEM:
137 {
138 // Read the attributes
139 AttrList_t attributes;
140 const char *name, *value;
141 while( (name = xml_ReaderNextAttr( m_pReader, &value )) != NULL )
142 attributes[strdup(name)] = strdup(value);
143
144 handleBeginElement( node, attributes );
145
146 map<const char*, const char*, ltstr> ::iterator it =
147 attributes.begin();
148 while( it != attributes.end() )
149 {
150 free( (char *)it->first );
151 free( (char *)it->second );
152 ++it;
153 }
154 break;
155 }
156
157 // End element
158 case XML_READER_ENDELEM:
159 {
160 handleEndElement( node );
161 break;
162 }
163 }
164 }
165 return (type == 0 && !m_errors );
166}
