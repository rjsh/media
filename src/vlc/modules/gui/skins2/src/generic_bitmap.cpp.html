28GenericBitmap::GenericBitmap( intf_thread_t *pIntf,
29 int nbFrames, int fps, int nbLoops ):
30 SkinObject( pIntf ), m_nbFrames( nbFrames ),
31 m_frameRate( fps ), m_nbLoops( nbLoops ), m_pGraphics( NULL )
32{
33}
34
35const OSGraphics *GenericBitmap::getGraphics() const
36{
37 if( m_pGraphics )
38 return m_pGraphics;
39
40 OSFactory *pOsFactory = OSFactory::instance( getIntf() );
41
42 int width = getWidth();
43 int height = getHeight();
44 if( width > 0 && height > 0 )
45 {
46 m_pGraphics = pOsFactory->createOSGraphics( width, height );
47 m_pGraphics->drawBitmap( *this, 0, 0 );
48 return m_pGraphics;
49 }
50
51 msg_Err( getIntf(), "failed to create a graphics, please report" );
52 return NULL;
53}

56BitmapImpl::BitmapImpl( intf_thread_t *pIntf, int width, int height,
57 int nbFrames, int fps, int nbLoops ):
58 GenericBitmap( pIntf, nbFrames, fps, nbLoops ), m_width( width ),
59 m_height( height ), m_pData( NULL )
60{
61 m_pData = new uint8_t[width * height * 4];
62 memset( m_pData, 0, width * height * 4 );
63}
64
65
66BitmapImpl::~BitmapImpl()
67{
68 delete[] m_pData;
69}

72bool BitmapImpl::drawBitmap( const GenericBitmap &rSource, int xSrc, int ySrc,
73 int xDest, int yDest, int width, int height )
74{
75 int srcWidth = rSource.getWidth();
76 uint32_t *pSrc = (uint32_t*)rSource.getData() + ySrc * srcWidth + xSrc;
77 if( !pSrc )
78 {
79 return false;
80 }
81 if( xSrc < 0 || xSrc + width > srcWidth ||
82 ySrc < 0 || ySrc + height > rSource.getHeight() )
83 {
84 msg_Warn( getIntf(), "drawBitmap: source rect too small, ignoring" );
85 return false;
86 }
87 if( xDest < 0 || xDest + width > m_width ||
88 yDest < 0 || yDest + height > m_height )
89 {
90 msg_Warn( getIntf(), "drawBitmap: dest rect too small, ignoring" );
91 return false;
92 }
93
94 uint32_t *pDest = (uint32_t*)m_pData + yDest * m_width + xDest ;
95 for( int y = 0; y < height; y++ )
96 {
97 memcpy( pDest, pSrc, 4 * width );
98 pSrc += srcWidth;
99 pDest += m_width;
100 }
101 return true;
102}
