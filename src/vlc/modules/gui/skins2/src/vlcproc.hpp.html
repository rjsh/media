47/// Singleton object handling VLC internal state and playlist
48class VlcProc: public SkinObject
49{
50public:
51 /// Get the instance of VlcProc
52 /// Returns NULL if the initialization of the object failed
53 static VlcProc *instance( intf_thread_t *pIntf );
54
55 /// Delete the instance of VlcProc
56 static void destroy( intf_thread_t *pIntf );
57
58 /// Getter for the playtree variable
59 Playtree &getPlaytreeVar() { return *((Playtree*)m_cPlaytree.get()); }
60
61 /// Getter for the time variable
62 StreamTime &getTimeVar() { return *((StreamTime*)(m_cVarTime.get())); }
63
64 /// Getter for the volume variable
65 Volume &getVolumeVar() { return *((Volume*)(m_cVarVolume.get())); }
66
67 /// Getter for the current playback speed
68 VarText &getSpeedVar()
69 { return *((VarText*)(m_cVarSpeed.get())); }
70
71 /// Getter for the stream name variable
72 VarText &getStreamNameVar()
73 { return *((VarText*)(m_cVarStreamName.get())); }
74
75 /// Getter for the stream URI variable
76 VarText &getStreamURIVar()
77 { return *((VarText*)(m_cVarStreamURI.get())); }
78
79 /// Getter for the stream bitrate variable
80 VarText &getStreamBitRateVar()
81 { return *((VarText*)(m_cVarStreamBitRate.get())); }
82
83 /// Getter for the stream sample rate variable
84 VarText &getStreamSampleRateVar()
85 { return *((VarText*)(m_cVarStreamSampleRate.get())); }
86
87 /// Getter for the stream Art url variable
88 VarString &getStreamArtVar()
89 { return *((VarString*)(m_cVarStreamArt.get())); }
90
91 /// Getter/Setter for the fullscreen variable
92 VarBool &getFullscreenVar() { return *((VarBool*)(m_cVarFullscreen.get())); }
93 void setFullscreenVar( bool );
94
95 /// Indicate whether the embedded video output is currently used
96 bool isVoutUsed() const { return m_pVout != NULL; }
97
98 /// initialize equalizer
99 void init_equalizer( );
100
101 /// update global variables for the current input
102 void update_current_input( );
103
104 void on_intf_event_changed( vlc_object_t* p_obj, vlc_value_t newVal );
105 void on_bit_rate_changed( vlc_object_t* p_obj, vlc_value_t newVal );
106 void on_sample_rate_changed( vlc_object_t* p_obj, vlc_value_t newVal );
107 void on_can_record_changed( vlc_object_t* p_obj, vlc_value_t newVal );
108
109 void on_random_changed( vlc_object_t* p_obj, vlc_value_t newVal );
110 void on_loop_changed( vlc_object_t* p_obj, vlc_value_t newVal );
111 void on_repeat_changed( vlc_object_t* p_obj, vlc_value_t newVal );
112
113 void on_volume_changed( vlc_object_t* p_obj, vlc_value_t newVal );
114 void on_mute_changed( vlc_object_t* p_obj, vlc_value_t newVal );
115 void on_audio_filter_changed( vlc_object_t* p_obj, vlc_value_t newVal );
116
117 void on_intf_show_changed( vlc_object_t* p_obj, vlc_value_t newVal );
118
119 void on_mouse_moved_changed( vlc_object_t* p_obj, vlc_value_t newVal );
120
121protected:
122 // Protected because it is a singleton
123 VlcProc( intf_thread_t *pIntf );
124 virtual ~VlcProc();
125
126private:
127 /// Playtree variable
128 VariablePtr m_cPlaytree;
129 VariablePtr m_cVarRandom;
130 VariablePtr m_cVarLoop;
131 VariablePtr m_cVarRepeat;
132 /// Variable for current position of the stream
133 VariablePtr m_cVarTime;
134 /// Variable for audio volume
135 VariablePtr m_cVarVolume;
136 /// Variable for speed playback
137 VariablePtr m_cVarSpeed;
138 /// Variable for current stream properties
139 VariablePtr m_cVarStreamName;
140 VariablePtr m_cVarStreamURI;
141 VariablePtr m_cVarStreamBitRate;
142 VariablePtr m_cVarStreamSampleRate;
143 VariablePtr m_cVarStreamArt;
144 /// Variable for the "mute" state
145 VariablePtr m_cVarMute;
146 /// Variables related to the input
147 VariablePtr m_cVarPlaying;
148 VariablePtr m_cVarStopped;
149 VariablePtr m_cVarPaused;
150 VariablePtr m_cVarSeekable;
151 VariablePtr m_cVarRecordable;
152 VariablePtr m_cVarRecording;
153 /// Variables related to the vout
154 VariablePtr m_cVarFullscreen;
155 VariablePtr m_cVarHasVout;
156 /// Variables related to audio
157 VariablePtr m_cVarHasAudio;
158 /// Equalizer variables
159 EqualizerBands m_varEqBands;
160 VariablePtr m_cVarEqPreamp;
161 VariablePtr m_cVarEqualizer;
162 /// Variable for DVD detection
163 VariablePtr m_cVarDvdActive;
164
165 /// Vout thread
166 vout_thread_t *m_pVout;
167
168 // reset variables when input is over
169 void reset_input();
170
171 // init variables (libvlc and playlist levels)
172 void init_variables();
173
174 /// Callback for intf-show variable
175 static int onIntfShow( vlc_object_t *pObj, const char *pVariable,
176 vlc_value_t oldVal, vlc_value_t newVal,
177 void *pParam );
178
179 /// Callback for input-current variable
180 static int onInputNew( vlc_object_t *pObj, const char *pVariable,
181 vlc_value_t oldVal, vlc_value_t newVal,
182 void *pParam );
183
184 /// Callback for item-change variable
185 static int onItemChange( vlc_object_t *pObj, const char *pVariable,
186 vlc_value_t oldVal, vlc_value_t newVal,
187 void *pParam );
188
189 /// Callback for item-change variable
190 static int onItemAppend( vlc_object_t *pObj, const char *pVariable,
191 vlc_value_t oldVal, vlc_value_t newVal,
192 void *pParam );
193
194 /// Callback for item-change variable
195 static int onItemDelete( vlc_object_t *pObj, const char *pVariable,
196 vlc_value_t oldVal, vlc_value_t newVal,
197 void *pParam );
198
199 static int onInteraction( vlc_object_t *pObj, const char *pVariable,
200 vlc_value_t oldVal, vlc_value_t newVal,
201 void *pParam );
202
203 static int onEqBandsChange( vlc_object_t *pObj, const char *pVariable,
204 vlc_value_t oldVal, vlc_value_t newVal,
205 void *pParam );
206
207 static int onEqPreampChange( vlc_object_t *pObj, const char *pVariable,
208 vlc_value_t oldVal, vlc_value_t newVal,
209 void *pParam );
210
211 /// Generic Callback
212 static int onGenericCallback( vlc_object_t *pObj, const char *pVariable,
213 vlc_value_t oldVal, vlc_value_t newVal,
214 void *pParam );
215
216 /// Generic Callback for intf-event
217 static int onGenericCallback2( vlc_object_t *pObj, const char *pVariable,
218 vlc_value_t oldVal, vlc_value_t newVal,
219 void *pParam );
220};
221
222#endif
