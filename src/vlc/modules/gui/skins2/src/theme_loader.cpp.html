68bool ThemeLoader::load( const string &fileName )
69{
70 string path = getFilePath( fileName );
71
72 //Before all, let's see if the file is present
73 struct stat p_stat;
74 if( vlc_stat( fileName.c_str(), &p_stat ) )
75 return false;
76
77 // First, we try to un-targz the file, and if it fails we hope it's a XML
78 // file...
79
80#if defined( HAVE_ZLIB_H )
81 if( ! extract( sToLocale( fileName ) ) && ! parse( path, fileName ) )
82 return false;
83#else
84 if( ! parse( path, fileName ) )
85 return false;
86#endif
87
88 Theme *pNewTheme = getIntf()->p_sys->p_theme;
89 if( !pNewTheme )
90 return false;
91
92 // Restore the theme configuration
93 getIntf()->p_sys->p_theme->loadConfig();
94
95 // Retain new loaded skins in config
96 config_PutPsz( getIntf(), "skins2-last", fileName.c_str() );
97
98 return true;
99}

103bool ThemeLoader::extractTarGz( const string &tarFile, const string &rootDir )
104{
105 TAR *t;
106#if defined( HAVE_LIBTAR_H )
107 tartype_t gztype = { (openfunc_t) gzopen_frontend,
108 (closefunc_t) gzclose_frontend,
109 (readfunc_t) gzread_frontend,
110 (writefunc_t) gzwrite_frontend };
111
112 if( tar_open( &t, (char *)tarFile.c_str(), &gztype, O_RDONLY, 0,
113 TAR_GNU ) == -1 )
114#else
115 if( tar_open( &t, (char *)tarFile.c_str(), O_RDONLY ) == -1 )
116#endif
117 {
118 return false;
119 }
120
121 if( tar_extract_all( t, (char *)rootDir.c_str() ) != 0 )
122 {
123 tar_close( t );
124 return false;
125 }
126
127 if( tar_close( t ) != 0 )
128 {
129 return false;
130 }
131
132 return true;
133}

136bool ThemeLoader::extractZip( const string &zipFile, const string &rootDir )
137{
138 bool b_isWsz = strstr( zipFile.c_str(), ".wsz" );
139
140 // Try to open the ZIP file
141 unzFile file = unzOpen( zipFile.c_str() );
142 unz_global_info info;
143
144 if( unzGetGlobalInfo( file, &info ) != UNZ_OK )
145 {
146 return false;
147 }
148 // Extract all the files in the archive
149 for( unsigned long i = 0; i < info.number_entry; i++ )
150 {
151 if( !extractFileInZip( file, rootDir, b_isWsz ) )
152 {
153 msg_Warn( getIntf(), "error while unzipping %s",
154 zipFile.c_str() );
155 unzClose( file );
156 return false;
157 }
158
159 if( i < info.number_entry - 1 )
160 {
161 // Go the next file in the archive
162 if( unzGoToNextFile( file ) !=UNZ_OK )
163 {
164 msg_Warn( getIntf(), "error while unzipping %s",
165 zipFile.c_str() );
166 unzClose( file );
167 return false;
168 }
169 }
170 }
171 unzClose( file );
172 return true;
173}

176bool ThemeLoader::extractFileInZip( unzFile file, const string &rootDir,
177 bool isWsz )
178{
179 // Read info for the current file
180 char filenameInZip[256];
181 unz_file_info fileInfo;
182 if( unzGetCurrentFileInfo( file, &fileInfo, filenameInZip,
183 sizeof( filenameInZip), NULL, 0, NULL, 0 )
184 != UNZ_OK )
185 {
186 return false;
187 }
188
189 // Convert the file name to lower case, because some winamp skins
190 // use the wrong case...
191 if( isWsz )
192 for( size_t i = 0; i < strlen( filenameInZip ); i++ )
193 filenameInZip[i] = tolower( (unsigned char)filenameInZip[i] );
194
195 // Allocate the buffer
196 void *pBuffer = malloc( ZIP_BUFFER_SIZE );
197 if( !pBuffer )
198 return false;
199
200 // Get the path of the file
201 OSFactory *pOsFactory = OSFactory::instance( getIntf() );
202 string fullPath = rootDir
203 + pOsFactory->getDirSeparator()
204 + fixDirSeparators( filenameInZip );
205 string basePath = getFilePath( fullPath );
206
207 // Extract the file if is not a directory
208 if( basePath != fullPath )
209 {
210 if( unzOpenCurrentFile( file ) )
211 {
212 free( pBuffer );
213 return false;
214 }
215 makedir( basePath.c_str() );
216 FILE *fout = fopen( fullPath.c_str(), "wb" );
217 if( fout == NULL )
218 {
219 msg_Err( getIntf(), "error opening %s", fullPath.c_str() );
220 free( pBuffer );
221 return false;
222 }
223
224 // Extract the current file
225 int n;
226 do
227 {
228 n = unzReadCurrentFile( file, pBuffer, ZIP_BUFFER_SIZE );
229 if( n < 0 )
230 {
231 msg_Err( getIntf(), "error while reading zip file" );
232 fclose(fout);
233 free( pBuffer );
234 return false;
235 }
236 else if( n > 0 )
237 {
238 if( fwrite( pBuffer, n , 1, fout) != 1 )
239 {
240 msg_Err( getIntf(), "error while writing %s",
241 fullPath.c_str() );
242 fclose(fout);
243 free( pBuffer );
244 return false;
245 }
246 }
247 } while( n > 0 );
248
249 fclose(fout);
250
251 if( unzCloseCurrentFile( file ) != UNZ_OK )
252 {
253 free( pBuffer );
254 return false;
255 }
256 }
257
258 free( pBuffer );
259 return true;
260}

263bool ThemeLoader::extract( const string &fileName )
264{
265 bool result = true;
266 char *tmpdir = tempnam( NULL, "vlt" );
267 string tempPath = sFromLocale( tmpdir );
268 free( tmpdir );
269
270 // Extract the file in a temporary directory
271 if( ! extractTarGz( fileName, tempPath ) &&
272 ! extractZip( fileName, tempPath ) )
273 {
274 deleteTempFiles( tempPath );
275 return false;
276 }
277
278 string path;
279 string xmlFile;
280 OSFactory *pOsFactory = OSFactory::instance( getIntf() );
281 // Find the XML file in the theme
282 if( findFile( tempPath, DEFAULT_XML_FILE, xmlFile ) )
283 {
284 path = getFilePath( xmlFile );
285 }
286 else
287 {
288 // No XML file, check if it is a winamp2 skin
289 string mainBmp;
290 if( findFile( tempPath, "main.bmp", mainBmp ) )
291 {
292 msg_Dbg( getIntf(), "trying to load a winamp2 skin" );
293 path = getFilePath( mainBmp );
294
295 // Look for winamp2.xml in the resource path
296 list<string> resPath = pOsFactory->getResourcePath();
297 list<string>::const_iterator it;
298 for( it = resPath.begin(); it != resPath.end(); ++it )
299 {
300 if( findFile( *it, WINAMP2_XML_FILE, xmlFile ) )
301 break;
302 }
303 }
304 }
305
306 if( !xmlFile.empty() )
307 {
308 // Parse the XML file
309 if (! parse( path, xmlFile ) )
310 {
311 msg_Err( getIntf(), "error while parsing %s", xmlFile.c_str() );
312 result = false;
313 }
314 }
315 else
316 {
317 msg_Err( getIntf(), "no XML found in theme %s", fileName.c_str() );
318 result = false;
319 }
320
321 // Clean-up
322 deleteTempFiles( tempPath );
323 return result;
324}

334bool ThemeLoader::parse( const string &path, const string &xmlFile )
335{
336 // File loaded
337 msg_Dbg( getIntf(), "using skin file: %s", xmlFile.c_str() );
338
339 // Start the parser
340 SkinParser parser( getIntf(), xmlFile, path );
341 if( ! parser.parse() )
342 return false;
343
344 // Build and store the theme
345 Builder builder( getIntf(), parser.getData(), path );
346 getIntf()->p_sys->p_theme = builder.build();
347
348 return true;
349}

352string ThemeLoader::getFilePath( const string &rFullPath )
353{
354 OSFactory *pOsFactory = OSFactory::instance( getIntf() );
355 const string &sep = pOsFactory->getDirSeparator();
356 // Find the last separator ('/' or '\')
357 string::size_type p = rFullPath.rfind( sep, rFullPath.size() );
358 string basePath;
359 if( p != string::npos )
360 {
361 if( p < rFullPath.size() - 1)
362 {
363 basePath = rFullPath.substr( 0, p );
364 }
365 else
366 {
367 basePath = rFullPath;
368 }
369 }
370 return basePath;
371}

373
374string ThemeLoader::fixDirSeparators( const string &rPath )
375{
376 OSFactory *pOsFactory = OSFactory::instance( getIntf() );
377 const string &sep = pOsFactory->getDirSeparator();
378 string::size_type p = rPath.find( "/", 0 );
379 string newPath = rPath;
380 while( p != string::npos )
381 {
382 newPath = newPath.replace( p, 1, sep );
383 p = newPath.find( "/", p + 1 );
384 }
385 return newPath;
386}

389bool ThemeLoader::findFile( const string &rootDir, const string &rFileName,
390 string &themeFilePath )
391{
392 // Path separator
393 const string &sep = OSFactory::instance( getIntf() )->getDirSeparator();
394
395 const char *pszDirContent;
396
397 // Open the dir
398 DIR *pCurrDir = vlc_opendir( rootDir.c_str() );
399
400 if( pCurrDir == NULL )
401 {
402 // An error occurred
403 msg_Dbg( getIntf(), "cannot open directory %s", rootDir.c_str() );
404 return false;
405 }
406
407 // While we still have entries in the directory
408 while( ( pszDirContent = vlc_readdir( pCurrDir ) ) != NULL )
409 {
410 string newURI = rootDir + sep + pszDirContent;
411
412 // Skip . and ..
413 if( string( pszDirContent ) != "." &&
414 string( pszDirContent ) != ".." )
415 {
416#if defined( S_ISDIR )
417 struct stat stat_data;
418
419 if( ( vlc_stat( newURI.c_str(), &stat_data ) == 0 )
420 && S_ISDIR(stat_data.st_mode) )
421#elif defined( DT_DIR )
422 if( pDirContent->d_type & DT_DIR )
423#else
424 if( 0 )
425#endif
426 {
427 // Can we find the file in this subdirectory?
428 if( findFile( newURI, rFileName, themeFilePath ) )
429 {
430 closedir( pCurrDir );
431 return true;
432 }
433 }
434 else
435 {
436 // Found the theme file?
437 if( rFileName == string( pszDirContent ) )
438 {
439 themeFilePath = newURI;
440 closedir( pCurrDir );
441 return true;
442 }
443 }
444 }
445 }
446
447 closedir( pCurrDir );
448 return false;
449}

454/* Values used in typeflag field */
455#define REGTYPE '0' /* regular file */
456#define AREGTYPE '\0' /* regular file */
457#define DIRTYPE '5' /* directory */

459#define BLOCKSIZE 512

461struct tar_header
462{ /* byte offset */
463 char name[100]; /* 0 */
464 char mode[8]; /* 100 */
465 char uid[8]; /* 108 */
466 char gid[8]; /* 116 */
467 char size[12]; /* 124 */
468 char mtime[12]; /* 136 */
469 char chksum[8]; /* 148 */
470 char typeflag; /* 156 */
471 char linkname[100]; /* 157 */
472 char magic[6]; /* 257 */
473 char version[2]; /* 263 */
474 char uname[32]; /* 265 */
475 char gname[32]; /* 297 */
476 char devmajor[8]; /* 329 */
477 char devminor[8]; /* 337 */
478 char prefix[155]; /* 345 */
479 /* 500 */
480};

482
483union tar_buffer {
484 char buffer[BLOCKSIZE];
485 struct tar_header header;
486};

490int tar_open( TAR **t, char *pathname, int oflags )
491{
492 (void)oflags;
493
494 gzFile f = gzopen( pathname, "rb" );
495 if( f == NULL )
496 {
497 fprintf( stderr, "Couldn't gzopen %s\n", pathname );
498 return -1;
499 }
500
501 *t = (gzFile *)malloc( sizeof(gzFile) );
502 **t = f;
503 return 0;
504}

506
507int tar_extract_all( TAR *t, char *prefix )
508{
509 union tar_buffer buffer;
510 int len, err, getheader = 1, remaining = 0;
511 FILE *outfile = NULL;
512 char fname[BLOCKSIZE + PATH_MAX];
513
514 while( 1 )
515 {
516 len = gzread( *t, &buffer, BLOCKSIZE );
517 if( len < 0 )
518 {
519 fprintf( stderr, "%s\n", gzerror(*t, &err) );
520 }
521
522 /*
523 * Always expect complete blocks to process
524 * the tar information.
525 */
526 if( len != 0 && len != BLOCKSIZE )
527 {
528 fprintf( stderr, "gzread: incomplete block read\n" );
529 return -1;
530 }
531
532 /*
533 * If we have to get a tar header
534 */
535 if( getheader == 1 )
536 {
537 /*
538 * If we met the end of the tar
539 * or the end-of-tar block, we are done
540 */
541 if( (len == 0) || (buffer.header.name[0] == 0) )
542 {
543 break;
544 }
545
546 sprintf( fname, "%s/%s", prefix, buffer.header.name );
547
548 /* Check magic value in header */
549 if( strncmp( buffer.header.magic, "GNUtar", 6 ) &&
550 strncmp( buffer.header.magic, "ustar", 5 ) )
551 {
552 //fprintf(stderr, "not a tar file\n");
553 return -1;
554 }
555
556 switch( buffer.header.typeflag )
557 {
558 case DIRTYPE:
559 makedir( fname );
560 break;
561 case REGTYPE:
562 case AREGTYPE:
563 remaining = getoct( buffer.header.size, 12 );
564 if( !remaining ) outfile = NULL; else
565 {
566 outfile = fopen( fname, "wb" );
567 if( outfile == NULL )
568 {
569 /* try creating directory */
570 char *p = strrchr( fname, '/' );
571 if( p != NULL )
572 {
573 *p = '\0';
574 makedir( fname );
575 *p = '/';
576 outfile = fopen( fname, "wb" );
577 if( !outfile )
578 {
579 fprintf( stderr, "tar couldn't create %s\n",
580 fname );
581 }
582 }
583 }
584 }
585
586 /*
587 * could have no contents
588 */
589 getheader = (remaining) ? 0 : 1;
590 break;
591 default:
592 break;
593 }
594 }
595 else
596 {
597 unsigned int bytes = (remaining > BLOCKSIZE)?BLOCKSIZE:remaining;
598
599 if( outfile != NULL )
600 {
601 if( fwrite( &buffer, sizeof(char), bytes, outfile ) != bytes )
602 {
603 fprintf( stderr, "error writing %s skipping...\n", fname );
604 fclose( outfile );
605 outfile = NULL;
606 unlink( fname );
607 }
608 }
609 remaining -= bytes;
610 if( remaining == 0 )
611 {
612 getheader = 1;
613 if( outfile != NULL )
614 {
615 fclose(outfile);
616 outfile = NULL;
617 }
618 }
619 }
620 }
621
622 return 0;
623}

634/* helper functions */
635int getoct( char *p, int width )
636{
637 int result = 0;
638 char c;
639
640 while( width-- )
641 {
642 c = *p++;
643 if( c == ' ' )
644 continue;
645 if( c == 0 )
646 break;
647 result = result * 8 + (c - '0');
648 }
649 return result;
650}

654/* Recursive make directory
655 * Abort if you get an ENOENT errno somewhere in the middle
656 * e.g. ignore error "mkdir on existing directory"
657 *
658 * return 1 if OK, 0 on error
659 */
660int makedir( const char *newdir )
661{
662 char *p, *buffer = strdup( newdir );
663 int len = strlen( buffer );
664
665 if( len <= 0 )
666 {
667 free( buffer );
668 return 0;
669 }
670
671 if( buffer[len-1] == '/' )
672 {
673 buffer[len-1] = '\0';
674 }
675
676 if( vlc_mkdir( buffer, 0775 ) == 0 )
677 {
678 free( buffer );
679 return 1;
680 }
681
682 p = buffer + 1;
683 while( 1 )
684 {
685 char hold;
686
687 while( *p && *p != '\\' && *p != '/' ) p++;
688 hold = *p;
689 *p = 0;
690 if( ( vlc_mkdir( buffer, 0775 ) == -1 ) && ( errno == ENOENT ) )
691 {
692 fprintf( stderr, "couldn't create directory %s\n", buffer );
693 free( buffer );
694 return 0;
695 }
696 if( hold == 0 ) break;
697 *p++ = hold;
698 }
699 free( buffer );
700 return 1;
701}

705static int currentGzFd = -1;
706static void * currentGzVp = NULL;
707
708int gzopen_frontend( const char *pathname, int oflags, int mode )
709{
710 (void)mode;
711
712 const char *gzflags;
713 gzFile gzf;
714
715 switch( oflags )
716 {
717 case O_WRONLY:
718 gzflags = "wb";
719 break;
720 case O_RDONLY:
721 gzflags = "rb";
722 break;
723 case O_RDWR:
724 default:
725 errno = EINVAL;
726 return -1;
727 }
728
729 gzf = gzopen( pathname, gzflags );
730 if( !gzf )
731 {
732 errno = ENOMEM;
733 return -1;
734 }
735
736 /** Hum ... */
737 currentGzFd = 42;
738 currentGzVp = gzf;
739
740 return currentGzFd;
741}
742
743int gzclose_frontend( int fd )
744{
745 if( currentGzVp != NULL && fd != -1 )
746 {
747 void *toClose = currentGzVp;
748 currentGzVp = NULL; currentGzFd = -1;
749 return gzclose( (gzFile) toClose );
750 }
751 return -1;
752}
753
754int gzread_frontend( int fd, void *p_buffer, size_t i_length )
755{
756 if( currentGzVp != NULL && fd != -1 )
757 {
758 return gzread( (gzFile) currentGzVp, p_buffer, i_length );
759 }
760 return -1;
761}
762
763int gzwrite_frontend( int fd, const void * p_buffer, size_t i_length )
764{
765 if( currentGzVp != NULL && fd != -1 )
766 {
767 return gzwrite( (gzFile) currentGzVp, const_cast<void*>(p_buffer), i_length );
768 }
769 return -1;
770}
