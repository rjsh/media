36Tooltip::Tooltip( intf_thread_t *pIntf, const GenericFont &rFont, int delay ):
37 SkinObject( pIntf ), m_rFont( rFont ), m_delay( delay ), m_pImage( NULL ),
38 m_xPos( -1 ), m_yPos( -1 ), m_cmdShow( this )
39{
40 OSFactory *pOsFactory = OSFactory::instance( pIntf );
41 m_pTimer = pOsFactory->createOSTimer( m_cmdShow );
42 m_pOsTooltip = pOsFactory->createOSTooltip();
43
44 // Observe the tooltip text variable
45 VarManager::instance( pIntf )->getTooltipText().addObserver( this );
46}
47
48
49Tooltip::~Tooltip()
50{
51 VarManager::instance( getIntf() )->getTooltipText().delObserver( this );
52 delete m_pTimer;
53 delete m_pOsTooltip;
54 delete m_pImage;
55}

57
58void Tooltip::show()
59{
60 // (Re)start the timer
61 m_pTimer->start( m_delay, true );
62}
63
64
65void Tooltip::hide()
66{
67 m_pTimer->stop();
68 m_pOsTooltip->hide();
69 m_xPos = -1;
70 m_yPos = -1;
71}

74void Tooltip::onUpdate( Subject<VarText> &rVariable, void *arg )
75{
76 (void)arg;
77 // Redisplay the tooltip
78 displayText( ((VarText&)rVariable).get() );

82void Tooltip::displayText( const UString &rText )
83{
84 // Rebuild the image
85 makeImage( rText );
86
87 // Redraw the window if it is already visible
88 if( m_xPos != -1 )
89 {
90 m_pOsTooltip->show( m_xPos, m_yPos, *m_pImage );
91 }
92}

95void Tooltip::makeImage( const UString &rText )
96{
97 // Render the text on a bitmap
98 GenericBitmap *pBmpTip = m_rFont.drawString( rText, 0 );
99 if( !pBmpTip )
100 {
101 return;
102 }
103 int w = pBmpTip->getWidth() + 10;
104 int h = m_rFont.getSize() + 8;
105
106 // Create the image of the tooltip
107 delete m_pImage;
108 m_pImage = OSFactory::instance( getIntf() )->createOSGraphics( w, h );
109 m_pImage->fillRect( 0, 0, w, h, 0xffffd0 );
110 m_pImage->drawRect( 0, 0, w, h, 0x000000 );
111 m_pImage->drawBitmap( *pBmpTip, 0, 0, 5, 5, -1, -1, true );
112
113 delete pBmpTip;
114}

117void Tooltip::CmdShow::execute()
118{
119 if( m_pParent->m_pImage )
120 {
121 if( m_pParent->m_xPos == -1 )
122 {
123 // Get the mouse coordinates and the image size
124 OSFactory *pOsFactory = OSFactory::instance( m_pParent->getIntf() );
125 int x, y;
126 pOsFactory->getMousePos( x, y );
127 int scrWidth = pOsFactory->getScreenWidth();
128 int scrHeight = pOsFactory->getScreenHeight();
129 int w = m_pParent->m_pImage->getWidth();
130 int h = m_pParent->m_pImage->getHeight();
131
132 // Compute the position of the tooltip
133 x -= (w / 2 + 4);
134 y += (h + 5);
135 if( x + w > scrWidth )
136 x -= (x + w - scrWidth);
137 else if( x < 0 )
138 x = 0;
139 if( y + h > scrHeight )
140 y -= (2 * h + 20);
141
142 m_pParent->m_xPos = x;
143 m_pParent->m_yPos = y;
144 }
145
146 // Show the tooltip window
147 m_pParent->m_pOsTooltip->show( m_pParent->m_xPos, m_pParent->m_yPos,
148 *(m_pParent->m_pImage) );
149 }
