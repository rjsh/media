34/// Generic interface for bitmaps
35class GenericBitmap: public SkinObject, public Box
36{
37public:
38 virtual ~GenericBitmap() { delete m_pGraphics; }
39
40 /// Get a linear buffer containing the image data.
41 /// Each pixel is stored in 4 bytes in the order B,G,R,A
42 virtual uint8_t *getData() const = 0;
43
44 /// Get the bitmap as a graphics
45 virtual const OSGraphics *getGraphics() const;
46
47 /// Get the number of frames in the bitmap
48 int getNbFrames() const { return m_nbFrames; }
49
50 /// Get the number of frames per second (for animated bitmaps)
51 int getFrameRate() const { return m_frameRate; }
52
53 /// Get the number of Loops (for animated bitmaps)
54 int getNbLoops() const { return m_nbLoops; }
55
56protected:
57 GenericBitmap( intf_thread_t *pIntf, int nbFrames = 1, int fps = 0, int nbLoops = 0);
58
59private:
60 /// Number of frames
61 int m_nbFrames;
62 /// Frame rate
63 int m_frameRate;
64 /// Number of Loops
65 int m_nbLoops;
66
67 /// graphics copy of the bitmap
68 mutable OSGraphics* m_pGraphics;
69};
70
71
72/// Basic bitmap implementation
73class BitmapImpl: public GenericBitmap
74{
75public:
76 /// Create an empty bitmap of the given size
77 BitmapImpl( intf_thread_t *pIntf, int width, int height,
78 int nbFrames = 1, int fps = 0, int nbLoops = 0 );
79 ~BitmapImpl();
80
81 /// Get the width of the bitmap
82 virtual int getWidth() const { return m_width; }
83
84 /// Get the heighth of the bitmap
85 virtual int getHeight() const { return m_height; }
86
87 /// Get a linear buffer containing the image data.
88 /// Each pixel is stored in 4 bytes in the order B,G,R,A
89 virtual uint8_t *getData() const { return m_pData; }
90
91 // Copy a region of another bitmap on this bitmap
92 bool drawBitmap( const GenericBitmap &rSource, int xSrc, int ySrc,
93 int xDest, int yDest, int width, int height );
94
95private:
96 /// Size of the bitmap.
97 int m_width, m_height;
98 /// Buffer containing the image data.
99 uint8_t *m_pData;
100};

103typedef CountedPtr<GenericBitmap> GenericBitmapPtr;
