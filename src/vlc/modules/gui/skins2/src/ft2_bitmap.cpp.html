28FT2Bitmap::FT2Bitmap( intf_thread_t *pIntf, int width, int height ):
29 GenericBitmap( pIntf ), m_width( width ), m_height( height )
30{
31 // Allocate memory for the buffer
32 m_pData = new uint8_t[m_height * m_width * 4];
33 memset( m_pData, 0, m_height * m_width * 4 );
34}
35
36
37FT2Bitmap::~FT2Bitmap()
38{
39 delete[] m_pData;
40}
41
42
43void FT2Bitmap::draw( const FT_Bitmap &rBitmap, int left, int top,
44 uint32_t color )
45{
46 uint8_t *pBuf = rBitmap.buffer;
47
48 // Calculate colors
49 uint8_t blue = color & 0xff;
50 uint8_t green = (color >> 8) & 0xff;
51 uint8_t red = (color >> 16) & 0xff;
52
53 for( int y = top; y < top + rBitmap.rows && y < m_height; y++ )
54 {
55 uint8_t *pData = m_pData + 4 * (m_width * y + left);
56 for( int x = left; x < left + rBitmap.width && x < m_width; x++ )
57 {
58 // The buffer in FT_Bitmap contains alpha values
59 uint8_t val = *(pBuf++);
60 *(pData++) = (blue * val) >> 8;
61 *(pData++) = (green * val) >> 8;
62 *(pData++) = (red * val) >> 8;
63 *(pData++) = val;
64 }
65 }
66}
