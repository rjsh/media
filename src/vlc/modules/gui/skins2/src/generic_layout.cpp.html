37GenericLayout::GenericLayout( intf_thread_t *pIntf, int width, int height,
38 int minWidth, int maxWidth, int minHeight,
39 int maxHeight ):
40 SkinObject( pIntf ), m_pWindow( NULL ),
41 m_original_width( width ), m_original_height( height ),
42 m_rect( 0, 0, width, height ),
43 m_minWidth( minWidth ), m_maxWidth( maxWidth ),
44 m_minHeight( minHeight ), m_maxHeight( maxHeight ), m_pVideoCtrlSet(),
45 m_visible( false ), m_pVarActive( NULL )
46{
47 // Get the OSFactory
48 OSFactory *pOsFactory = OSFactory::instance( getIntf() );
49 // Create the graphics buffer
50 m_pImage = pOsFactory->createOSGraphics( width, height );
51
52 // Create the "active layout" variable and register it in the manager
53 m_pVarActive = new VarBoolImpl( pIntf );
54 VarManager::instance( pIntf )->registerVar( VariablePtr( m_pVarActive ) );
55}

58GenericLayout::~GenericLayout()
59{
60 delete m_pImage;
61
62 list<Anchor*>::const_iterator it;
63 for( it = m_anchorList.begin(); it != m_anchorList.end(); ++it )
64 {
65 delete *it;
66 }
67
68 list<LayeredControl>::const_iterator iter;
69 for( iter = m_controlList.begin(); iter != m_controlList.end(); ++iter )
70 {
71 CtrlGeneric *pCtrl = (*iter).m_pControl;
72 pCtrl->unsetLayout();
73 }
74
75}

106void GenericLayout::addControl( CtrlGeneric *pControl,
107 const Position &rPosition, int layer )
108{
109 if( pControl )
110 {
111 // Associate this layout to the control
112 pControl->setLayout( this, rPosition );
113
114 // Add the control in the list.
115 // This list must remain sorted by layer order
116 list<LayeredControl>::iterator it;
117 for( it = m_controlList.begin(); it != m_controlList.end(); ++it )
118 {
119 if( layer < (*it).m_layer )
120 {
121 m_controlList.insert( it, LayeredControl( pControl, layer ) );
122 break;
123 }
124 }
125 // If this control is in front of all the previous ones
126 if( it == m_controlList.end() )
127 {
128 m_controlList.push_back( LayeredControl( pControl, layer ) );
129 }
130
131 // Check if it is a video control
132 if( pControl->getType() == "video" )
133 {
134 m_pVideoCtrlSet.insert( (CtrlVideo*)pControl );
135 }
136 }
137 else
138 {
139 msg_Dbg( getIntf(), "adding NULL control in the layout" );
140 }
141}

150void GenericLayout::onControlUpdate( const CtrlGeneric &rCtrl,
151 int width, int height,
152 int xOffSet, int yOffSet )
153{
154 // Do nothing if the layout or control is hidden
155 if( !m_visible )
156 return;
157
158 const Position *pPos = rCtrl.getPosition();
159 if( width > 0 && height > 0 )
160 {
161 // make sure region is within the layout
162 rect region( pPos->getLeft() + xOffSet,
163 pPos->getTop() + yOffSet,
164 width, height );
165 rect layout( 0, 0, m_rect.getWidth(), m_rect.getHeight() );
166 rect inter;
167 if( rect::intersect( layout, region, &inter ) )
168 {
169 refreshRect( inter.x, inter.y, inter.width, inter.height );
170 }
171 }
172}

175void GenericLayout::resize( int width, int height )
176{
177 // check real resize
178 if( width == m_rect.getWidth() && height == m_rect.getHeight() )
179 return;
180
181 // Update the window size
182 m_rect = SkinsRect( 0, 0 , width, height );
183
184 // Recreate a new image
185 if( m_pImage )
186 {
187 delete m_pImage;
188 OSFactory *pOsFactory = OSFactory::instance( getIntf() );
189 m_pImage = pOsFactory->createOSGraphics( width, height );
190 }
191
192 // Notify all the controls that the size has changed and redraw them
193 list<LayeredControl>::const_iterator iter;
194 for( iter = m_controlList.begin(); iter != m_controlList.end(); ++iter )
195 {
196 iter->m_pControl->onResize();
197 }
198}

206
207void GenericLayout::refreshRect( int x, int y, int width, int height )
208{
209 // Do nothing if the layout is hidden
210 if( !m_visible )
211 return;
212
213 // update the transparency global mask
214 m_pImage->clear( x, y, width, height );
215
216 // Draw all the controls of the layout
217 list<LayeredControl>::const_iterator iter;
218 for( iter = m_controlList.begin(); iter != m_controlList.end(); ++iter )
219 {
220 CtrlGeneric *pCtrl = (*iter).m_pControl;
221 if( pCtrl->isVisible() )
222 {
223 pCtrl->draw( *m_pImage, x, y, width, height );
224 }
225 }
226
227 // Refresh the associated window
228 TopWindow *pWindow = getWindow();
229 if( pWindow )
230 {
231 // first apply new shape to the window
232 pWindow->updateShape();
233 pWindow->invalidateRect( x, y, width, height );
234 }
235}

264bool GenericLayout::isTightlyCoupledWith( const GenericLayout& otherLayout ) const
265{
266 return m_original_width == otherLayout.m_original_width &&
267 m_original_height == otherLayout.m_original_height &&
268 m_minWidth == otherLayout.m_minWidth &&
269 m_maxWidth == otherLayout.m_maxWidth &&
270 m_minHeight == otherLayout.m_minHeight &&
271 m_maxHeight == otherLayout.m_maxHeight;
272}
