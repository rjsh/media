23#include <ft2build.h>
24#include "ft2_err.h"
25
26/* Warning: This file includes the error definitions header twice.
27 * Further, freetype2 errors are not contiguous, so instead of a sparse
28 * array we first look up the actual entry by linear search and only
29 * then return the actual error string. It could be improved to binary
30 * search (assuming the freetype2 source stays sorted), but error
31 * reporting shouldn't need to be efficient.
32 */
33
34#define FT_NOERRORDEF_( sym, num, str ) num,
35#define FT_ERRORDEF_( sym, num, str ) num,
36
37static const unsigned short ft2_errorindex[] =
38{
39#include FT_ERROR_DEFINITIONS_H
40};
41
42#undef FT_NOERRORDEF_
43#undef FT_ERRORDEF_
44
45
46#define FT_NOERRORDEF_( sym, num, str ) str,
47#define FT_ERRORDEF_( sym, num, str ) str,
48
49static const char *ft2_errorstrings[] =
50{
51#include FT_ERROR_DEFINITIONS_H
52};
53
54#undef FT_NOERRORDEF_
55#undef FT_ERRORDEF_
56
57enum { ft2_num_errors = sizeof(ft2_errorindex)/sizeof(*ft2_errorindex) };
58
59const char *ft2_strerror(unsigned err)
60{
61 unsigned i;
62 for( i=0; i<ft2_num_errors; ++i )
63 if( err==ft2_errorindex[i] )
64 break;
65
66 return i<ft2_num_errors ? ft2_errorstrings[i] :
67 "An error freetype2 neglected to specify";
68}
