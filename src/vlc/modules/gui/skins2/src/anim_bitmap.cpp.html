31AnimBitmap::AnimBitmap( intf_thread_t *pIntf, const GenericBitmap &rBitmap ):
32 SkinObject( pIntf ),
33 m_rBitmap( rBitmap ), m_pImage( rBitmap.getGraphics() ),
34 m_nbFrames( rBitmap.getNbFrames() ), m_frameRate( rBitmap.getFrameRate() ),
35 m_nbLoops( rBitmap.getNbLoops() ), m_curFrame( 0 ), m_curLoop( 0 ),
36 m_pTimer( NULL ), m_cmdNextFrame( this )
37{
38 OSFactory *pOsFactory = OSFactory::instance( pIntf );
39
40 // Create the timer
41 m_pTimer = pOsFactory->createOSTimer( m_cmdNextFrame );
42}
43
44
45AnimBitmap::~AnimBitmap()
46{
47 delete m_pTimer;
48}
49
50
51void AnimBitmap::startAnim()
52{
53 if( m_nbFrames > 1 && m_frameRate > 0 )
54 m_pTimer->start( 1000 / m_frameRate, false );
55}
56
57
58void AnimBitmap::stopAnim()
59{
60 m_pTimer->stop();
61 m_curLoop = 0;
62 m_curFrame = 0;
63}
64
65
66void AnimBitmap::draw( OSGraphics &rImage, int xDest, int yDest, int w, int h,
67 int xOffset, int yOffset )
68{
69 // Draw the current frame
70 int height = m_pImage->getHeight() / m_nbFrames;
71 int ySrc = height * m_curFrame;
72
73 // The old way .... transparency was not taken care of
74 // rImage.drawGraphics( *m_pImage, 0, ySrc, xDest, yDest,
75 // m_pImage->getWidth(), height );
76
77 rImage.drawBitmap( m_rBitmap,
78 xOffset, ySrc + yOffset,
79 xDest, yDest, w, h, true );
80}

82
83bool AnimBitmap::hit( int x, int y ) const
84{
85 int height = m_pImage->getHeight() / m_nbFrames;
86 return y >= 0 && y < height &&
87 m_pImage->hit( x, m_curFrame * height + y );
88}
89
90
91int AnimBitmap::getWidth() const
92{
93 return m_pImage->getWidth();
94}
95
96
97int AnimBitmap::getHeight() const
98{
99 return m_pImage->getHeight() / m_nbFrames;
100}

102
103void AnimBitmap::CmdNextFrame::execute()
104{
105 // Go the next frame
106 m_pParent->m_curFrame = ( m_pParent->m_curFrame + 1 ) %
107 m_pParent->m_nbFrames;
108
109 if( m_pParent->m_nbLoops > 0 && m_pParent->m_curFrame == 0 )
110 {
111 m_pParent->m_curLoop += 1;
112
113 if( m_pParent->m_curLoop == m_pParent->m_nbLoops )
114 {
115 m_pParent->stopAnim();
116 m_pParent->m_curFrame = m_pParent->m_nbFrames - 1;
117 }
118 }
119
120 // Notify the observer so that it can display the next frame
121 m_pParent->notify();
122}
