70/// Wrapper around FromLocale, to avoid the need to call LocaleFree()
71static inline string sFromLocale( const string &rLocale )
72{
73 const char *s = FromLocale( rLocale.c_str() );
74 string res = s;
75 LocaleFree( s );
76 return res;
77}
78
79#ifdef _WIN32
80/// Wrapper around FromWide, to avoid the need to call free()
81static inline string sFromWide( const wstring &rWide )
82{
83 char *s = FromWide( rWide.c_str() );
84 string res = s;
85 free( s );
86 return res;
87}
88#endif
89
90/// Wrapper around ToLocale, to avoid the need to call LocaleFree()
91static inline string sToLocale( const string &rUTF8 )
92{
93 const char *s = ToLocale( rUTF8.c_str() );
94 string res = s;
95 LocaleFree( s );
96 return res;
97}

100//---------------------------------------------------------------------------
101// intf_sys_t: description and status of skin interface
102//---------------------------------------------------------------------------
103struct intf_sys_t
104{
105 /// The input thread
106 input_thread_t *p_input;
107
108 /// The playlist thread
109 playlist_t *p_playlist;
110
111 // "Singleton" objects: MUST be initialized to NULL !
112 /// Logger
113 Logger *p_logger;
114 /// Asynchronous command queue
115 AsyncQueue *p_queue;
116 /// Dialog provider
117 Dialogs *p_dialogs;
118 /// Script interpreter
119 Interpreter *p_interpreter;
120 /// Factory for OS specific classes
121 OSFactory *p_osFactory;
122 /// Main OS specific message loop
123 OSLoop *p_osLoop;
124 /// Variable manager
125 VarManager *p_varManager;
126 /// VLC state handler
127 VlcProc *p_vlcProc;
128 /// Vout manager
129 VoutManager *p_voutManager;
130 /// Art manager
131 ArtManager *p_artManager;
132 /// Theme repository
133 ThemeRepository *p_repository;
134
135 /// Current theme
136 Theme *p_theme;
137
138 /// synchronisation at start of interface
139 vlc_thread_t thread;
140 vlc_mutex_t init_lock;
141 vlc_cond_t init_wait;
142 bool b_error;
143 bool b_ready;
144};

147/// Base class for all skin classes
148class SkinObject
149{
150public:
151 SkinObject( intf_thread_t *pIntf ): m_pIntf( pIntf ) { }
152 virtual ~SkinObject() { }
153
154 /// Getter (public because it is used in C callbacks in the win32
155 /// interface)
156 intf_thread_t *getIntf() const { return m_pIntf; }
157
158private:
159 intf_thread_t *m_pIntf;
160};
