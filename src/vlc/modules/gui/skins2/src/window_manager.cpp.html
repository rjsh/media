34WindowManager::WindowManager( intf_thread_t *pIntf ):
35 SkinObject( pIntf ), m_magnet( 0 ), m_alpha( 255 ), m_moveAlpha( 255 ),
36 m_opacityEnabled( false ), m_opacity( 255 ), m_direction( kNone ),
37 m_maximizeRect(0, 0, 50, 50), m_pTooltip( NULL ), m_pPopup( NULL )
38{
39 // Create and register a variable for the "on top" status
40 VarManager *pVarManager = VarManager::instance( getIntf() );
41 m_cVarOnTop = VariablePtr( new VarBoolImpl( getIntf() ) );
42 pVarManager->registerVar( m_cVarOnTop, "vlc.isOnTop" );
43
44 // transparency switched on or off by user
45 m_opacityEnabled = var_InheritBool( getIntf(), "skins2-transparency" );
46
47 // opacity overridden by user
48 m_opacity = 255 * var_InheritFloat( getIntf(), "qt-opacity" );
49}

74void WindowManager::startMove( TopWindow &rWindow )
75{
76 // Rebuild the set of moving windows
77 m_movingWindows.clear();
78 buildDependSet( m_movingWindows, &rWindow );
79
80 if( isOpacityNeeded() )
81 {
82 // Change the opacity of the moving windows
83 WinSet_t::const_iterator it;
84 for( it = m_movingWindows.begin(); it != m_movingWindows.end(); ++it )
85 {
86 (*it)->setOpacity( m_moveAlpha );
87 }
88 }
89}

92void WindowManager::stopMove()
93{
94 WinSet_t::const_iterator itWin1, itWin2;
95 AncList_t::const_iterator itAnc1, itAnc2;
96
97 if( isOpacityNeeded() )
98 {
99 // Restore the opacity of the moving windows
100 WinSet_t::const_iterator it;
101 for( it = m_movingWindows.begin(); it != m_movingWindows.end(); ++it )
102 {
103 (*it)->setOpacity( m_alpha );
104 }
105 }
106
107 // Delete the dependencies
108 m_dependencies.clear();
109
110 // Now we rebuild the dependencies.
111 // Iterate through all the windows
112 for( itWin1 = m_allWindows.begin(); itWin1 != m_allWindows.end(); ++itWin1 )
113 {
114 // Get the anchors of the layout associated to the window
115 const AncList_t &ancList1 =
116 (*itWin1)->getActiveLayout().getAnchorList();
117
118 // Iterate through all the windows, starting with (*itWin1)
119 for( itWin2 = itWin1; itWin2 != m_allWindows.end(); ++itWin2 )
120 {
121 // A window can't anchor itself...
122 if( (*itWin2) == (*itWin1) )
123 continue;
124
125 // Now, check for anchoring between the 2 windows
126 const AncList_t &ancList2 =
127 (*itWin2)->getActiveLayout().getAnchorList();
128 for( itAnc1 = ancList1.begin(); itAnc1 != ancList1.end(); ++itAnc1 )
129 {
130 for( itAnc2 = ancList2.begin();
131 itAnc2 != ancList2.end(); ++itAnc2 )
132 {
133 if( (*itAnc1)->isHanging( **itAnc2 ) )
134 {
135 // (*itWin1) anchors (*itWin2)
136 m_dependencies[*itWin1].insert( *itWin2 );
137 }
138 else if( (*itAnc2)->isHanging( **itAnc1 ) )
139 {
140 // (*itWin2) anchors (*itWin1)
141 m_dependencies[*itWin2].insert( *itWin1 );
142 }
143 }
144 }
145 }
146 }
147}

150void WindowManager::move( TopWindow &rWindow, int left, int top ) const
151{
152 // Compute the real move offset
153 int xOffset = left - rWindow.getLeft();
154 int yOffset = top - rWindow.getTop();
155
156 // Check anchoring; this can change the values of xOffset and yOffset
157 checkAnchors( &rWindow, xOffset, yOffset );
158
159 // Move all the windows
160 WinSet_t::const_iterator it;
161 for( it = m_movingWindows.begin(); it != m_movingWindows.end(); ++it )
162 {
163 (*it)->move( (*it)->getLeft() + xOffset, (*it)->getTop() + yOffset );
164 }
165}

168void WindowManager::startResize( GenericLayout &rLayout, Direction_t direction )
169{
170 m_direction = direction;
171
172 // Rebuild the set of moving windows.
173 // From the resized window, we only take into account the anchors which
174 // are mobile with the current type of resizing, and that are hanging a
175 // window. The hanged windows will come will all their dependencies.
176
177 m_resizeMovingE.clear();
178 m_resizeMovingS.clear();
179 m_resizeMovingSE.clear();
180
181 WinSet_t::const_iterator itWin;
182 AncList_t::const_iterator itAnc1, itAnc2;
183 // Get the anchors of the layout
184 const AncList_t &ancList1 = rLayout.getAnchorList();
185
186 // Iterate through all the hanged windows
187 for( itWin = m_dependencies[rLayout.getWindow()].begin();
188 itWin != m_dependencies[rLayout.getWindow()].end(); ++itWin )
189 {
190 // Now, check for anchoring between the 2 windows
191 const AncList_t &ancList2 =
192 (*itWin)->getActiveLayout().getAnchorList();
193 for( itAnc1 = ancList1.begin(); itAnc1 != ancList1.end(); ++itAnc1 )
194 {
195 for( itAnc2 = ancList2.begin();
196 itAnc2 != ancList2.end(); ++itAnc2 )
197 {
198 if( (*itAnc1)->isHanging( **itAnc2 ) )
199 {
200 // Add the dependencies of the hanged window to one of the
201 // lists of moving windows
202 Position::Ref_t aRefPos =
203 (*itAnc1)->getPosition().getRefLeftTop();
204 if( aRefPos == Position::kRightTop )
205 buildDependSet( m_resizeMovingE, *itWin );
206 else if( aRefPos == Position::kLeftBottom )
207 buildDependSet( m_resizeMovingS, *itWin );
208 else if( aRefPos == Position::kRightBottom )
209 buildDependSet( m_resizeMovingSE, *itWin );
210 break;
211 }
212 }
213 }
214 }
215
216 // The checkAnchors() method will need to have m_movingWindows properly set
217 // so let's insert in it the contents of the other sets
218 m_movingWindows.clear();
219 m_movingWindows.insert( rLayout.getWindow() );
220 m_movingWindows.insert( m_resizeMovingE.begin(), m_resizeMovingE.end() );
221 m_movingWindows.insert( m_resizeMovingS.begin(), m_resizeMovingS.end() );
222 m_movingWindows.insert( m_resizeMovingSE.begin(), m_resizeMovingSE.end() );
223}

233void WindowManager::resize( GenericLayout &rLayout,
234 int width, int height ) const
235{
236 // TODO: handle anchored windows
237 // Compute the real resizing offset
238 int xOffset = width - rLayout.getWidth();
239 int yOffset = height - rLayout.getHeight();
240
241 // Check anchoring; this can change the values of xOffset and yOffset
242 checkAnchors( rLayout.getWindow(), xOffset, yOffset );
243 if( m_direction == kResizeS )
244 xOffset = 0;
245 if( m_direction == kResizeE )
246 yOffset = 0;
247
248 int newWidth = rLayout.getWidth() + xOffset;
249 int newHeight = rLayout.getHeight() + yOffset;
250
251 // Check boundaries
252 if( newWidth < rLayout.getMinWidth() )
253 {
254 newWidth = rLayout.getMinWidth();
255 }
256 if( newWidth > rLayout.getMaxWidth() )
257 {
258 newWidth = rLayout.getMaxWidth();
259 }
260 if( newHeight < rLayout.getMinHeight() )
261 {
262 newHeight = rLayout.getMinHeight();
263 }
264 if( newHeight > rLayout.getMaxHeight() )
265 {
266 newHeight = rLayout.getMaxHeight();
267 }
268
269 if( newWidth == rLayout.getWidth() && newHeight == rLayout.getHeight() )
270 {
271 return;
272 }
273
274 // New offset, after the last corrections
275 int xNewOffset = newWidth - rLayout.getWidth();
276 int yNewOffset = newHeight - rLayout.getHeight();
277
278 // Resize the window
279 TopWindow *pWindow = rLayout.getWindow();
280 pWindow->resize( newWidth, newHeight );
281
282 // Do the actual resizing
283 rLayout.resize( newWidth, newHeight );
284
285 // refresh content
286 rLayout.refreshAll();
287
288 // Move all the anchored windows
289 WinSet_t::const_iterator it;
290 if( m_direction == kResizeE ||
291 m_direction == kResizeSE )
292 {
293 for( it = m_resizeMovingE.begin(); it != m_resizeMovingE.end(); ++it )
294 {
295 (*it)->move( (*it)->getLeft() + xNewOffset,
296 (*it)->getTop() );
297 }
298 }
299 if( m_direction == kResizeS ||
300 m_direction == kResizeSE )
301 {
302 for( it = m_resizeMovingS.begin(); it != m_resizeMovingS.end(); ++it )
303 {
304 (*it)->move( (*it)->getLeft(),
305 (*it)->getTop( )+ yNewOffset );
306 }
307 }
308 if( m_direction == kResizeE ||
309 m_direction == kResizeS ||
310 m_direction == kResizeSE )
311 {
312 for( it = m_resizeMovingSE.begin(); it != m_resizeMovingSE.end(); ++it )
313 {
314 (*it)->move( (*it)->getLeft() + xNewOffset,
315 (*it)->getTop() + yNewOffset );
316 }
317 }
318}

321void WindowManager::maximize( TopWindow &rWindow )
322{
323 // Save the current position/size of the window, to be able to restore it
324 m_maximizeRect = SkinsRect( rWindow.getLeft(), rWindow.getTop(),
325 rWindow.getLeft() + rWindow.getWidth(),
326 rWindow.getTop() + rWindow.getHeight() );
327
328 SkinsRect workArea = OSFactory::instance( getIntf() )->getWorkArea();
329 // Move the window
330 startMove( rWindow );
331 move( rWindow, workArea.getLeft(), workArea.getTop() );
332 stopMove();
333 // Now resize it
334 // FIXME: Ugly const_cast
335 GenericLayout &rLayout = (GenericLayout&)rWindow.getActiveLayout();
336 startResize( rLayout, kResizeSE );
337 resize( rLayout, workArea.getWidth(), workArea.getHeight() );
338 stopResize();
339 rWindow.m_pVarMaximized->set( true );
340
341 // Make the window unmovable by unregistering it
342// unregisterWindow( rWindow );
343}

354 startResize( rLayout, kResizeSE );

357 // Now move it
358 startMove( rWindow );
359 move( rWindow, m_maximizeRect.getLeft(), m_maximizeRect.getTop() );
360 stopMove();
361 rWindow.m_pVarMaximized->set( false );
362}

479void WindowManager::buildDependSet( WinSet_t &rWinSet,
480 TopWindow *pWindow )
481{
482 // pWindow is in the set
483 rWinSet.insert( pWindow );
484
485 // Iterate through the anchored windows
486 const WinSet_t &anchored = m_dependencies[pWindow];
487 WinSet_t::const_iterator iter;
488 for( iter = anchored.begin(); iter != anchored.end(); ++iter )
489 {
490 // Check that the window isn't already in the set before adding it
491 if( rWinSet.find( *iter ) == rWinSet.end() )
492 {
493 buildDependSet( rWinSet, *iter );
494 }
495 }
496}

499void WindowManager::checkAnchors( TopWindow *pWindow,
500 int &xOffset, int &yOffset ) const
501{
502 (void)pWindow;
503 WinSet_t::const_iterator itMov, itSta;
504 AncList_t::const_iterator itAncMov, itAncSta;
505
506 // Check magnetism with screen edges first (actually it is the work area)
507 SkinsRect workArea = OSFactory::instance( getIntf() )->getWorkArea();
508 // Iterate through the moving windows
509 for( itMov = m_movingWindows.begin();
510 itMov != m_movingWindows.end(); ++itMov )
511 {
512 // Skip the invisible windows
513 if( ! (*itMov)->getVisibleVar().get() )
514 {
515 continue;
516 }
517
518 int newLeft = (*itMov)->getLeft() + xOffset;
519 int newTop = (*itMov)->getTop() + yOffset;
520 if( newLeft > workArea.getLeft() - m_magnet &&
521 newLeft < workArea.getLeft() + m_magnet )
522 {
523 xOffset = workArea.getLeft() - (*itMov)->getLeft();
524 }
525 if( newTop > workArea.getTop() - m_magnet &&
526 newTop < workArea.getTop() + m_magnet )
527 {
528 yOffset = workArea.getTop() - (*itMov)->getTop();
529 }
530 int right = workArea.getLeft() + workArea.getWidth();
531 if( newLeft + (*itMov)->getWidth() > right - m_magnet &&
532 newLeft + (*itMov)->getWidth() < right + m_magnet )
533 {
534 xOffset = right - (*itMov)->getLeft() - (*itMov)->getWidth();
535 }
536 int bottom = workArea.getTop() + workArea.getHeight();
537 if( newTop + (*itMov)->getHeight() > bottom - m_magnet &&
538 newTop + (*itMov)->getHeight() < bottom + m_magnet )
539 {
540 yOffset = bottom - (*itMov)->getTop() - (*itMov)->getHeight();
541 }
542 }
543
544 // Iterate through the moving windows
545 for( itMov = m_movingWindows.begin();
546 itMov != m_movingWindows.end(); ++itMov )
547 {
548 // Skip the invisible windows
549 if( ! (*itMov)->getVisibleVar().get() )
550 {
551 continue;
552 }
553
554 // Get the anchors in the main layout of this moving window
555 const AncList_t &movAnchors =
556 (*itMov)->getActiveLayout().getAnchorList();
557
558 // Iterate through the static windows
559 for( itSta = m_allWindows.begin();
560 itSta != m_allWindows.end(); ++itSta )
561 {
562 // Skip the moving windows and the invisible ones
563 if( m_movingWindows.find( (*itSta) ) != m_movingWindows.end() ||
564 ! (*itSta)->getVisibleVar().get() )
565 {
566 continue;
567 }
568
569 // Get the anchors in the main layout of this static window
570 const AncList_t &staAnchors =
571 (*itSta)->getActiveLayout().getAnchorList();
572
573 // Check if there is an anchoring between one of the movAnchors
574 // and one of the staAnchors
575 for( itAncMov = movAnchors.begin();
576 itAncMov != movAnchors.end(); ++itAncMov )
577 {
578 for( itAncSta = staAnchors.begin();
579 itAncSta != staAnchors.end(); ++itAncSta )
580 {
581 if( (*itAncSta)->canHang( **itAncMov, xOffset, yOffset ) )
582 {
583 // We have found an anchoring!
584 // There is nothing to do here, since xOffset and
585 // yOffset are automatically modified by canHang()
586
587 // Don't check the other anchors, one is enough...
588 return;
589 }
590 else
591 {
592 // Temporary variables
593 int xOffsetTemp = -xOffset;
594 int yOffsetTemp = -yOffset;
595 if( (*itAncMov)->canHang( **itAncSta, xOffsetTemp,
596 yOffsetTemp ) )
597 {
598 // We have found an anchoring!
599 // xOffsetTemp and yOffsetTemp have been updated,
600 // we just need to change xOffset and yOffset
601 xOffset = -xOffsetTemp;
602 yOffset = -yOffsetTemp;
603
604 // Don't check the other anchors, one is enough...
605 return;
606 }
607 }
608 }
609 }
610 }
