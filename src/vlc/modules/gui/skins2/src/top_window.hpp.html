25#ifndef TOP_WINDOW_HPP
26#define TOP_WINDOW_HPP
27
28#include "generic_window.hpp"
29#include "../utils/pointer.hpp"
30#include <list>
31
32class OSWindow;
33class OSGraphics;
34class GenericLayout;
35class CtrlGeneric;
36class WindowManager;
37
38
39/// Class to handle top-level windows
40class TopWindow: public GenericWindow
41{
42private:
43 friend class WindowManager;
44public:
45 TopWindow( intf_thread_t *pIntf, int xPos, int yPos,
46 WindowManager &rWindowManager,
47 bool dragDrop, bool playOnDrop, bool visible,
48 GenericWindow::WindowType_t type = GenericWindow::TopWindow );
49 virtual ~TopWindow();
50
51 /// Methods to process OS events.
52 virtual void processEvent( EvtFocus &rEvtFocus );
53 virtual void processEvent( EvtMenu &rEvtMenu );
54 virtual void processEvent( EvtMotion &rEvtMotion );
55 virtual void processEvent( EvtMouse &rEvtMouse );
56 virtual void processEvent( EvtLeave &rEvtLeave );
57 virtual void processEvent( EvtKey &rEvtKey );
58 virtual void processEvent( EvtScroll &rEvtScroll );
59 virtual void processEvent( EvtDragDrop &rEvtDragDrop );
60 virtual void processEvent( EvtDragOver &rEvtDragOver );
61 virtual void processEvent( EvtDragLeave &rEvtDragLeave );
62
63 /// Forward an event to a control
64 virtual void forwardEvent( EvtGeneric &rEvt, CtrlGeneric &rCtrl );
65
66 // Refresh an area of the window
67 virtual void refresh( int left, int top, int width, int height );
68
69 /// Get the active layout
70 virtual const GenericLayout& getActiveLayout() const;
71
72 /// Update the shape of the window from the active layout
73 virtual void updateShape();
74
75 /// Called by a control that wants to capture the mouse
76 virtual void onControlCapture( const CtrlGeneric &rCtrl );
77
78 /// Called by a control that wants to release the mouse
79 virtual void onControlRelease( const CtrlGeneric &rCtrl );
80
81 /// Called by a control when its tooltip changed
82 virtual void onTooltipChange( const CtrlGeneric &rCtrl );
83
84 /// Get the "maximized" variable
85 VarBool &getMaximizedVar() { return *m_pVarMaximized; }
86
87 /// Get the initial visibility status
88 bool getInitialVisibility() const { return m_initialVisibility; }
89
90protected:
91 /// Actually show the window
92 virtual void innerShow();
93
94 /// Actually hide the window
95 virtual void innerHide();
96
97private:
98 /**
99 * These methods are only used by the window manager
100 */
101 //@{
102 /// Change the active layout
103 virtual void setActiveLayout( GenericLayout *pLayout );
104 //@}
105
106 /// Initial visibility status
107 bool m_initialVisibility;
108 /// indicator if playback is requested on drag&drop
109 bool m_playOnDrop;
110 /// Window manager
111 WindowManager &m_rWindowManager;
112 /// Current active layout of the window
113 GenericLayout *m_pActiveLayout;
114 /// Last control on which the mouse was over
115 CtrlGeneric *m_pLastHitControl;
116 /// Control that has captured the mouse
117 CtrlGeneric *m_pCapturingControl;
118 /// Control that has the focus
119 CtrlGeneric *m_pFocusControl;
120 /// Control over which drag&drop is hovering
121 CtrlGeneric *m_pDragControl;
122 /// Current key modifier (also used for mouse)
123 int m_currModifier;
124
125 /// Variable for the visibility of the window
126 VarBoolImpl *m_pVarMaximized;
127
128 /**
129 * Find the uppest control in the layout hit by the mouse, and send
130 * it an enter event if needed
131 */
132 CtrlGeneric *findHitControl( int xPos, int yPos );
133
134 /**
135 * Update the lastHitControl pointer and send a leave event to the
136 * right control
137 */
138 void setLastHit( CtrlGeneric *pNewHitControl );
139};
140
141typedef CountedPtr<TopWindow> TopWindowPtr;
142
143
144#endif
