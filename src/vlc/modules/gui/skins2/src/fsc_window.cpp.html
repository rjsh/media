29/**
30 * Fading out is computed in the following way:
31 * - a timer is fired with a given period (m_delay)
32 * - a total of FSC_COUNT transitions are processed before
33 * hiding the controller
34 * - transparency is changed in the following way :
35 * - unaltered for the first FSC_COUNT2 transitions
36 * - then linear decrease from m_opacity to 0 till
37 * FSC_COUNT is reached.
38 **/
39#define FSC_DELAY 50 // period for fading out (in msec)
40#define FSC_COUNT 60 // total # of transitions for fading out
41#define FSC_COUNT2 20 // # of transitions with opacity unaltered
42
43FscWindow::FscWindow( intf_thread_t *pIntf, int left, int top,
44 WindowManager &rWindowManager,
45 bool dragDrop, bool playOnDrop, bool visible ) :
46 TopWindow( pIntf, left, top, rWindowManager, dragDrop,
47 playOnDrop, false, GenericWindow::FscWindow ),
48 m_pTimer( NULL ), m_count( 0 ),
49 m_cmdFscHide( this )
50{
51 (void)visible;
52 m_pTimer = OSFactory::instance( getIntf() )->createOSTimer( m_cmdFscHide );
53
54 VarBool &rFullscreen = VlcProc::instance( getIntf() )->getFullscreenVar();
55 rFullscreen.addObserver( this );
56
57 // opacity overridden by user
58 m_opacity = 255 * var_InheritFloat( getIntf(), "qt-fs-opacity" );
59
60 // fullscreen-controller timeout overridden by user
61 m_delay = var_InheritInteger( getIntf(), "mouse-hide-timeout" ) / FSC_COUNT;
62 if( m_delay <= 0 )
63 m_delay = FSC_DELAY;
64
65 /// activation overridden by user
66 m_enabled = var_InheritBool( getIntf(), "qt-fs-controller" );
67
68 // register Fsc
69 VoutManager::instance( getIntf())->registerFSC( this );
70}
71
72
73FscWindow::~FscWindow()
74{
75 // unregister Fsc
76 VoutManager::instance( getIntf())->registerFSC( NULL );
77
78 VarBool &rFullscreen = VlcProc::instance( getIntf() )->getFullscreenVar();
79 rFullscreen.delObserver( this );
80
81 delete m_pTimer;
82}

84
85void FscWindow::onMouseMoved( )
86{
87 VarBool &rFullscreen = VlcProc::instance( getIntf() )->getFullscreenVar();
88 if( rFullscreen.get() )
89 {
90 show();
91 if( m_count < FSC_COUNT - FSC_COUNT2 )
92 {
93 m_pTimer->stop();
94 m_count = FSC_COUNT;
95 setOpacity( m_opacity );
96 m_pTimer->start( m_delay, false );
97 }
98 }
99}
100
101
102void FscWindow::onTimerExpired()
103{
104 if( m_count )
105 {
106 if( m_count < FSC_COUNT - FSC_COUNT2 )
107 setOpacity( m_opacity * m_count / (FSC_COUNT - FSC_COUNT2 ) );
108 m_count--;
109 }
110
111 if( !m_count )
112 {
113 hide();
114 }
115}

117
118void FscWindow::processEvent( EvtMotion &rEvtMotion )
119{
120 if( m_count )
121 {
122 m_pTimer->stop();
123 m_count = 0;
124 setOpacity( m_opacity );
125 }
126
127 TopWindow::processEvent( rEvtMotion );
128}
129
130
131void FscWindow::processEvent( EvtLeave &rEvtLeave )
132{
133 if( m_count )
134 {
135 m_pTimer->stop();
136 m_count = 0;
137 }
138
139 m_count = FSC_COUNT;
140 setOpacity( m_opacity );
141 m_pTimer->start( m_delay, false );
142
143 TopWindow::processEvent( rEvtLeave );
144}

146
147void FscWindow::onUpdate( Subject<VarBool> &rVariable, void *arg )
148{
149 VarBool &rFullscreen = VlcProc::instance( getIntf() )->getFullscreenVar();
150 if( &rVariable == &rFullscreen )
151 {
152 if( !rFullscreen.get() )
153 {
154 hide();
155 }
156 }
157 TopWindow::onUpdate( rVariable, arg );
158}
159
160
161void FscWindow::innerShow()
162{
163 if( !m_enabled )
164 return;
165
166 TopWindow::innerShow();
167
168 m_count = FSC_COUNT;
169 setOpacity( m_opacity );
170 m_pTimer->start( m_delay, false );
171}
172
173
174void FscWindow::innerHide()
175{
176 m_pTimer->stop();
177 m_count = 0;
178
179 TopWindow::innerHide();
180}

183void FscWindow::moveTo( int x, int y, int width, int height )
184{
185 // relocate the fs controller
186 // (centered horizontally and lowered vertically with a 3% margin)
187 int x_fsc = x + ( width - getWidth() ) / 2;
188 int y_fsc = y + height - getHeight() - height * 3 / 100;
189 move( x_fsc, y_fsc );
190}
191
192
193void FscWindow::CmdFscHide::execute()
194{
195 m_pParent->onTimerExpired();
196}
