34VoutManager *VoutManager::instance( intf_thread_t *pIntf )
35{
36 if( pIntf->p_sys->p_voutManager == NULL )
37 {
38 pIntf->p_sys->p_voutManager = new VoutManager( pIntf );
39 }
40
41 return pIntf->p_sys->p_voutManager;
42}

45void VoutManager::destroy( intf_thread_t *pIntf )
46{
47 delete pIntf->p_sys->p_voutManager;
48 pIntf->p_sys->p_voutManager = NULL;
49}

52VoutManager::VoutManager( intf_thread_t *pIntf ): SkinObject( pIntf ),
53 m_pCtrlVideoVec(), m_pCtrlVideoVecBackup(), m_SavedWndVec(),
54 m_pVoutMainWindow( NULL ), m_pFscWindow( NULL )
55{
56 m_pVoutMainWindow = new VoutMainWindow( getIntf() );
57
58 OSFactory *pOsFactory = OSFactory::instance( getIntf() );
59 int width = pOsFactory->getScreenWidth();
60 int height = pOsFactory->getScreenHeight();
61
62 m_pVoutMainWindow->move( 0, 0 );
63 m_pVoutMainWindow->resize( width, height );
64
65 VarBool &rFullscreen = VlcProc::instance( getIntf() )->getFullscreenVar();
66 rFullscreen.addObserver( this );
67}

70VoutManager::~VoutManager( )
71{
72 VarBool &rFullscreen = VlcProc::instance( getIntf() )->getFullscreenVar();
73 rFullscreen.delObserver( this );
74
75 delete m_pVoutMainWindow;
76}

91void VoutManager::saveVoutConfig( )
92{
93 // Save width/height to be consistent across themes
94 // and detach Video Controls
95 vector<SavedWnd>::iterator it;
96 for( it = m_SavedWndVec.begin(); it != m_SavedWndVec.end(); ++it )
97 {
98 if( it->pCtrlVideo )
99 {
100 // detach vout thread from VideoControl
101 it->pCtrlVideo->detachVoutWindow( );
102
103 // memorize width/height before VideoControl is destroyed
104 it->width = it->pCtrlVideo->getPosition()->getWidth();
105 it->height = it->pCtrlVideo->getPosition()->getHeight();
106 it->pCtrlVideo = NULL;
107 }
108 }
109
110 // Create a backup copy and reset original for new theme
111 m_pCtrlVideoVecBackup = m_pCtrlVideoVec;
112 m_pCtrlVideoVec.clear();
113}
114
115
116void VoutManager::restoreVoutConfig( bool b_success )
117{
118 if( !b_success )
119 {
120 // loading new theme failed, restoring previous theme
121 m_pCtrlVideoVec = m_pCtrlVideoVecBackup;
122 }
123
124 // reattach vout(s) to Video Controls
125 vector<SavedWnd>::iterator it;
126 for( it = m_SavedWndVec.begin(); it != m_SavedWndVec.end(); ++it )
127 {
128 CtrlVideo* pCtrlVideo = getBestCtrlVideo();
129 if( pCtrlVideo )
130 {
131 pCtrlVideo->attachVoutWindow( it->pVoutWindow );
132 it->pCtrlVideo = pCtrlVideo;
133 }
134 }
135}

138void VoutManager::discardVout( CtrlVideo* pCtrlVideo )
139{
140 vector<SavedWnd>::iterator it;
141 for( it = m_SavedWndVec.begin(); it != m_SavedWndVec.end(); ++it )
142 {
143 if( it->pCtrlVideo == pCtrlVideo )
144 {
145 // detach vout thread from VideoControl
146 it->pCtrlVideo->detachVoutWindow( );
147 it->width = it->pCtrlVideo->getPosition()->getWidth();
148 it->height = it->pCtrlVideo->getPosition()->getHeight();
149 it->pCtrlVideo = NULL;
150 break;
151 }
152 }
153}

156void VoutManager::requestVout( CtrlVideo* pCtrlVideo )
157{
158 vector<SavedWnd>::iterator it;
159 for( it = m_SavedWndVec.begin(); it != m_SavedWndVec.end(); ++it )
160 {
161 if( it->pCtrlVideo == NULL )
162 {
163 pCtrlVideo->attachVoutWindow( it->pVoutWindow,
164 it->width, it->height );
165 it->pCtrlVideo = pCtrlVideo;
166 break;
167 }
168 }
169}

171
172CtrlVideo* VoutManager::getBestCtrlVideo( )
173{
174 vector<CtrlVideo*>::const_iterator it;
175
176 // first, look up a video control that is visible and unused
177 for( it = m_pCtrlVideoVec.begin(); it != m_pCtrlVideoVec.end(); ++it )
178 {
179 if( (*it)->isUseable() && !(*it)->isUsed() )
180 {
181 return (*it);
182 }
183 }
184
185 // as a fallback, look up any video control that is unused
186 for( it = m_pCtrlVideoVec.begin(); it != m_pCtrlVideoVec.end(); ++it )
187 {
188 if( !(*it)->isUsed() )
189 {
190 return (*it);
191 }
192 }
193
194 return NULL;
195}

202void VoutManager::acceptWnd( vout_window_t* pWnd, int width, int height )
203{
204 // Creation of a dedicated Window per vout thread
205 VoutWindow* pVoutWindow = new VoutWindow( getIntf(), pWnd, width, height,
206 (GenericWindow*) m_pVoutMainWindow );
207
208 // try to find a video Control within the theme
209 CtrlVideo* pCtrlVideo = getBestCtrlVideo();
210 if( pCtrlVideo )
211 {
212 // A Video Control is available
213 // directly attach vout thread to it
214 pCtrlVideo->attachVoutWindow( pVoutWindow );
215 }
216 else
217 {
218 pVoutWindow->setCtrlVideo( NULL );
219 }
220
221 // save vout characteristics
222 m_SavedWndVec.push_back( SavedWnd( pWnd, pVoutWindow, pCtrlVideo ) );
223
224 msg_Dbg( pWnd, "New vout : Ctrl = %p, w x h = %ix%i",
225 pCtrlVideo, width, height );
226}

228
229void VoutManager::releaseWnd( vout_window_t* pWnd )
230{
231 // remove vout thread from savedVec
232 vector<SavedWnd>::iterator it;
233 for( it = m_SavedWndVec.begin(); it != m_SavedWndVec.end(); ++it )
234 {
235 if( it->pWnd == pWnd )
236 {
237 msg_Dbg( getIntf(), "vout released vout=%p, VideoCtrl=%p",
238 pWnd, it->pCtrlVideo );
239
240 // if a video control was being used, detach from it
241 if( it->pCtrlVideo )
242 {
243 it->pCtrlVideo->detachVoutWindow( );
244 }
245
246 // remove resources
247 delete it->pVoutWindow;
248 m_SavedWndVec.erase( it );
249 break;
250 }
251 }
252
253 // force fullscreen to false so that user regains control
254 VlcProc::instance( getIntf() )->setFullscreenVar( false );
255}

258void VoutManager::setSizeWnd( vout_window_t *pWnd, int width, int height )
259{
260 msg_Dbg( pWnd, "setSize (%ix%i) received from vout thread",
261 width, height );
262
263 vector<SavedWnd>::iterator it;
264 for( it = m_SavedWndVec.begin(); it != m_SavedWndVec.end(); ++it )
265 {
266 if( it->pWnd == pWnd )
267 {
268 VoutWindow* pVoutWindow = it->pVoutWindow;
269
270 pVoutWindow->setOriginalWidth( width );
271 pVoutWindow->setOriginalHeight( height );
272
273 CtrlVideo* pCtrlVideo = pVoutWindow->getCtrlVideo();
274 if( pCtrlVideo )
275 {
276 pCtrlVideo->resizeControl( width, height );
277 }
278 break;
279 }
280 }
281}

284void VoutManager::setFullscreenWnd( vout_window_t *pWnd, bool b_fullscreen )
285{
286 msg_Dbg( pWnd, "setFullscreen (%i) received from vout thread",
287 b_fullscreen );
288
289 // reconfigure the fullscreen window (multiple screens)
290 if( b_fullscreen )
291 {
292 vector<SavedWnd>::iterator it;
293 for( it = m_SavedWndVec.begin(); it != m_SavedWndVec.end(); ++it )
294 {
295 if( it->pWnd == pWnd )
296 {
297 VoutWindow* pVoutWindow = it->pVoutWindow;
298 configureFullscreen( *pVoutWindow );
299 break;
300 }
301 }
302 }
303
304 // set fullscreen
305 VlcProc::instance( getIntf() )->setFullscreenVar( b_fullscreen );
306}

309void VoutManager::onUpdate( Subject<VarBool> &rVariable, void *arg )
310{
311 (void)arg;
312 VarBool &rFullscreen = VlcProc::instance( getIntf() )->getFullscreenVar();
313 if( &rVariable == &rFullscreen )
314 {
315 if( rFullscreen.get() )
316 m_pVoutMainWindow->show();
317 else
318 m_pVoutMainWindow->hide();
319 }
320}

322
323void VoutManager::configureFullscreen( VoutWindow& rWindow )
324{
325 int numScr = var_InheritInteger( getIntf(), "qt-fullscreen-screennumber" );
326 int x0 = m_pVoutMainWindow->getTop();
327 int y0 = m_pVoutMainWindow->getLeft();
328
329 int x, y, w, h;
330 if( numScr >= 0 )
331 {
332 // select screen requested by user
333 OSFactory *pOsFactory = OSFactory::instance( getIntf() );
334 pOsFactory->getMonitorInfo( numScr, &x, &y, &w, &h );
335 }
336 else
337 {
338 // select screen where display is already occurring
339 rWindow.getMonitorInfo( &x, &y, &w, &h );
340 }
341
342 if( x != x0 || y != y0 )
343 {
344 // move and resize fullscreen
345 m_pVoutMainWindow->move( x, y );
346 m_pVoutMainWindow->resize( w, h );
347
348 // ensure the fs controller is also moved
349 if( m_pFscWindow )
350 {
351 m_pFscWindow->moveTo( x, y, w, h );
352 }
353 }
