27#include "skin_common.hpp"
28#include <map>

31/// Singleton object handling the list of available themes
32class ThemeRepository: public SkinObject
33{
34public:
35 /// Get the instance of ThemeRepository
36 /// Returns NULL if the initialization of the object failed
37 static ThemeRepository *instance( intf_thread_t *pIntf );
38
39 /// Delete the instance of ThemeRepository
40 static void destroy( intf_thread_t *pIntf );
41
42 /// Update repository
43 void updateRepository();
44
45protected:
46 // Protected because it is a singleton
47 ThemeRepository( intf_thread_t *pIntf );
48 virtual ~ThemeRepository();
49
50private:
51 /// Look for themes in a directory
52 void parseDirectory( const string &rDir );
53
54 /// Callback for menu item selection
55 static int changeSkin( vlc_object_t *pThis, char const *pVariable,
56 vlc_value_t oldval, vlc_value_t newval,
57 void *pData );
58
59 /// list of skins available
60 map<string,string> m_skinsMap;
61};
