34FileBitmap::FileBitmap( intf_thread_t *pIntf, image_handler_t *pImageHandler,
35 string fileName, uint32_t aColor, int nbFrames,
36 int fps, int nbLoops ):
37 GenericBitmap( pIntf, nbFrames, fps, nbLoops ), m_width( 0 ), m_height( 0 ),
38 m_pData( NULL )
39{
40 video_format_t fmt_in, fmt_out;
41 picture_t *pPic;
42
43 video_format_Init( &fmt_in, 0 );
44 video_format_Init( &fmt_out, VLC_CODEC_RGBA );
45
46 if( strstr( fileName.c_str(), "://" ) == NULL )
47 {
48 char *psz_uri = vlc_path2uri( fileName.c_str(), NULL );
49 if( !psz_uri )
50 return;
51 fileName = psz_uri;
52 free( psz_uri );
53 }
54
55 pPic = image_ReadUrl( pImageHandler, fileName.c_str(), &fmt_in, &fmt_out );
56 if( !pPic )
57 return;
58
59 m_width = fmt_out.i_width;
60 m_height = fmt_out.i_height;
61
62 m_pData = new uint8_t[m_height * m_width * 4];
63
64 // Compute the alpha layer
65 uint8_t *pData = m_pData, *pSrc = pPic->p->p_pixels;
66 for( int y = 0; y < m_height; y++ )
67 {
68 for( int x = 0; x < m_width; x++ )
69 {
70 uint32_t r = *pSrc++;
71 uint32_t g = *pSrc++;
72 uint32_t b = *pSrc++;
73 uint8_t a = *pSrc++;
74
75 *(pData++) = b * a / 255;
76 *(pData++) = g * a / 255;
77 *(pData++) = r * a / 255;
78
79 // Transparent pixel ?
80 if( aColor == (r<<16 | g<<8 | b) )
81 {
82 *pData = 0;
83 }
84 else
85 {
86 *pData = a;
87 }
88 pData++;
89 }
90 pSrc += pPic->p->i_pitch - m_width * 4;
91 }
92
93 picture_Release( pPic );
94 return;
95}
96
97
98FileBitmap::~FileBitmap()
99{
100 delete[] m_pData;
101}
102
103
104uint8_t *FileBitmap::getData() const
105{
106 if( m_pData == NULL )
107 {
108 msg_Warn( getIntf(), "FileBitmap::getData() returns NULL" );
109 }
110 return m_pData;
111}
