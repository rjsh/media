25#ifndef OS_FACTORY_HPP
26#define OS_FACTORY_HPP
27
28#include "skin_common.hpp"
29#include "../utils/position.hpp"
30#include <string>
31#include <list>
32
33#include "../src/generic_window.hpp"
34class CmdGeneric;
35class OSBitmap;
36class OSGraphics;
37class OSLoop;
38class OSWindow;
39class OSTooltip;
40class OSPopup;
41class OSTimer;
42
43
44/// Abstract factory used to instantiate OS specific objects.
45class OSFactory: public SkinObject
46{
47public:
48 enum CursorType_t
49 {
50 kDefaultArrow,
51 kResizeNS,
52 kResizeWE,
53 kResizeNWSE,
54 kResizeNESW
55 };
56
57 /**
58 * Initialization method overloaded in derived classes.
59 * It must return false if the init failed.
60 */
61 virtual bool init() { return true; }
62
63 /**
64 * Get the right instance of OSFactory.
65 * Returns NULL if initialization of the concrete factory failed.
66 */
67 static OSFactory *instance( intf_thread_t *pIntf );
68
69 /// Delete the instance of OSFactory
70 static void destroy( intf_thread_t *pIntf );
71
72 /// Instantiate an object OSGraphics
73 virtual OSGraphics *createOSGraphics( int width, int height ) = 0;
74
75 /// Get the instance of the singleton OSLoop
76 virtual OSLoop *getOSLoop() = 0;
77
78 /// Destroy the instance of OSLoop
79 virtual void destroyOSLoop() = 0;
80
81 /// Minimize all the windows
82 virtual void minimize() = 0;
83
84 /// Restore the minimized windows
85 virtual void restore() = 0;
86
87 /// Add an icon in the system tray
88 virtual void addInTray() = 0;
89
90 /// Remove the icon from the system tray
91 virtual void removeFromTray() = 0;
92
93 /// Show the task in the task bar
94 virtual void addInTaskBar() = 0;
95
96 /// Remove the task from the task bar
97 virtual void removeFromTaskBar() = 0;
98
99 /// Instantiate an OSTimer with the given command
100 virtual OSTimer *createOSTimer( CmdGeneric &rCmd ) = 0;
101
102 /// Instantiate an object OSWindow
103 virtual OSWindow *createOSWindow( GenericWindow &rWindow,
104 bool dragDrop, bool playOnDrop,
105 OSWindow *pParent, GenericWindow::WindowType_t ) = 0;
106
107 /// Instantiate an object OSTooltip
108 virtual OSTooltip *createOSTooltip() = 0;
109
110 /// Instantiate an object OSPopup
111 virtual OSPopup *createOSPopup() = 0;
112
113 /// Get the directory separator
114 virtual const string &getDirSeparator() const = 0;
115
116 /// Get the resource path
117 virtual const list<string> &getResourcePath() const = 0;
118
119 /// Get the screen size
120 virtual int getScreenWidth() const = 0;
121 virtual int getScreenHeight() const = 0;
122
123 /// Get Monitor Information for a given Window
124 virtual void getMonitorInfo( const GenericWindow &rWindow,
125 int* x, int* y,
126 int* width, int* height ) const = 0;
127
128 /// Get Monitor Information (screens numbered from 0 upwards.)
129 virtual void getMonitorInfo( int num, int* x, int* y,
130 int* width, int* height ) const = 0;
131
132 /// Get the work area (screen area without taskbars)
133 virtual SkinsRect getWorkArea() const = 0;
134
135 /// Get the position of the mouse
136 virtual void getMousePos( int &rXPos, int &rYPos ) const = 0;
137
138 /// Change the cursor
139 virtual void changeCursor( CursorType_t type ) const = 0;
140
141 /// Delete a directory recursively
142 virtual void rmDir( const string &rPath ) = 0;
143
144protected:
145 // Protected because it's a singleton
146 OSFactory( intf_thread_t* pIntf ): SkinObject( pIntf ) { }
147 virtual ~OSFactory() { }
148};
149
150
151#endif
