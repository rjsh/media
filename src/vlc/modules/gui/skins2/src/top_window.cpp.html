56TopWindow::TopWindow( intf_thread_t *pIntf, int left, int top,
57 WindowManager &rWindowManager,
58 bool dragDrop, bool playOnDrop, bool visible,
59 GenericWindow::WindowType_t type ):
60 GenericWindow( pIntf, left, top, dragDrop, playOnDrop, NULL, type ),
61 m_initialVisibility( visible ), m_playOnDrop( playOnDrop ),
62 m_rWindowManager( rWindowManager ),
63 m_pActiveLayout( NULL ), m_pLastHitControl( NULL ),
64 m_pCapturingControl( NULL ), m_pFocusControl( NULL ),
65 m_pDragControl( NULL ), m_currModifier( 0 )
66{
67 // Register as a moving window
68 m_rWindowManager.registerWindow( *this );
69
70 // Create the "maximized" variable and register it in the manager
71 m_pVarMaximized = new VarBoolImpl( pIntf );
72 VarManager::instance( pIntf )->registerVar( VariablePtr( m_pVarMaximized ) );
73}

79 m_rWindowManager.unregisterWindow( *this );
80}

89void TopWindow::processEvent( EvtMenu &rEvtMenu )
90{

103void TopWindow::processEvent( EvtMotion &rEvtMotion )
104{
105 // New control hit by the mouse
106 CtrlGeneric *pNewHitControl =
107 findHitControl( rEvtMotion.getXPos() - getLeft(),
108 rEvtMotion.getYPos() - getTop() );
109
110 setLastHit( pNewHitControl );
111
112 /// Update the help text
113 VarManager *pVarManager = VarManager::instance( getIntf() );
114 if( pNewHitControl )
115 {
116 pVarManager->getHelpText().set( pNewHitControl->getHelpText() );
117 }
118
119 // Send a motion event to the hit control, or to the control
120 // that captured the mouse, if any
121 CtrlGeneric *pActiveControl = pNewHitControl;
122 if( m_pCapturingControl )
123 {
124 pActiveControl = m_pCapturingControl;
125 }
126 if( pActiveControl )
127 {
128 // Compute the coordinates relative to the window
129 int xPos = rEvtMotion.getXPos() - getLeft();
130 int yPos = rEvtMotion.getYPos() - getTop();
131 // Send a motion event
132 EvtMotion evt( getIntf(), xPos, yPos );
133 pActiveControl->handleEvent( evt );
134 }
135}

151void TopWindow::processEvent( EvtMouse &rEvtMouse )
152{
153 // Get the control hit by the mouse
154 CtrlGeneric *pNewHitControl = findHitControl( rEvtMouse.getXPos(),
155 rEvtMouse.getYPos() );
156 setLastHit( pNewHitControl );
157
158 // Change the focused control
159 if( rEvtMouse.getAction() == EvtMouse::kDown )
160 {
161 // Raise the window
162 m_rWindowManager.raise( *this );
163
164 if( m_pFocusControl != pNewHitControl )
165 {
166 if( m_pFocusControl )
167 {
168 // The previous control loses the focus
169 EvtFocus evt( getIntf(), false );
170 m_pFocusControl->handleEvent( evt );
171 m_pFocusControl = NULL;
172 }
173
174 if( pNewHitControl && pNewHitControl->isFocusable() )
175 {
176 // The hit control gains the focus
177 m_pFocusControl = pNewHitControl;
178 EvtFocus evt( getIntf(), true );
179 pNewHitControl->handleEvent( evt );
180 }
181 }
182 }
183
184 // Send a mouse event to the hit control, or to the control
185 // that captured the mouse, if any
186 CtrlGeneric *pActiveControl = pNewHitControl;
187 if( m_pCapturingControl )
188 {
189 pActiveControl = m_pCapturingControl;
190 }
191 if( pActiveControl )
192 {
193 pActiveControl->handleEvent( rEvtMouse );
194 }
195}

198void TopWindow::processEvent( EvtKey &rEvtKey )
199{
200 // Forward the event to the focused control, if any
201 if( m_pFocusControl )
202 {
203 m_pFocusControl->handleEvent( rEvtKey );
204 return;
205 }
206
207 // Only do the action when the key is down
208 if( rEvtKey.getKeyState() == EvtKey::kDown )
209 {
210 //XXX not to be hardcoded!
211 // Ctrl-S = Change skin
212 if( (rEvtKey.getMod() & EvtInput::kModCtrl) &&
213 rEvtKey.getKey() == 's' )
214 {
215 CmdDlgChangeSkin cmd( getIntf() );
216 cmd.execute();
217 return;
218 }
219
220 //XXX not to be hardcoded!
221 // Ctrl-T = Toggle on top
222 if( (rEvtKey.getMod() & EvtInput::kModCtrl) &&
223 rEvtKey.getKey() == 't' )
224 {
225 CmdOnTop cmd( getIntf() );
226 cmd.execute();
227 return;
228 }
229
230 var_SetInteger( getIntf()->p_libvlc, "key-pressed",
231 rEvtKey.getModKey() );
232 }
233
234 // Always store the modifier, which can be needed for scroll events.
235 m_currModifier = rEvtKey.getMod();
236}

238void TopWindow::processEvent( EvtScroll &rEvtScroll )
239{
240 // Raise the windows
241 raise();
242
243 // Get the control hit by the mouse
244 CtrlGeneric *pNewHitControl = findHitControl( rEvtScroll.getXPos(),
245 rEvtScroll.getYPos());
246 setLastHit( pNewHitControl );
247
248 // send a mouse event to the right control when scrollable
249 // if none, send it directly to the vlc core
250 CtrlGeneric *pHitControl = m_pCapturingControl ?
251 m_pCapturingControl : pNewHitControl;
252
253 if( pHitControl && pHitControl->isScrollable() )
254 {
255 pHitControl->handleEvent( rEvtScroll );
256 }
257 else
258 {
259 // Treat the scroll event as a hotkey plus current modifiers
260 int i = (rEvtScroll.getDirection() == EvtScroll::kUp ?
261 KEY_MOUSEWHEELUP : KEY_MOUSEWHEELDOWN) | m_currModifier;
262
263 var_SetInteger( getIntf()->p_libvlc, "key-pressed", i );
264 }
265}

267void TopWindow::processEvent( EvtDragDrop &rEvtDragDrop )
268{
269 // Get the control hit by the mouse
270 int xPos = rEvtDragDrop.getXPos() - getLeft();
271 int yPos = rEvtDragDrop.getYPos() - getTop();
272
273 CtrlGeneric *pHitControl = findHitControl( xPos, yPos );
274 if( pHitControl && pHitControl->getType() == "tree" )
275 {
276 // Send a dragDrop event
277 EvtDragDrop evt( getIntf(), xPos, yPos, rEvtDragDrop.getFiles() );
278 pHitControl->handleEvent( evt );
279 }
280 else
281 {
282 input_thread_t *pInput = getIntf()->p_sys->p_input;
283 bool is_subtitle = false;
284 list<string> files = rEvtDragDrop.getFiles();
285 if( files.size() == 1 && pInput != NULL )
286 {
287 list<string>::const_iterator it = files.begin();
288 char* psz_file = make_path( it->c_str() );
289 if( psz_file )
290 {
291 is_subtitle = !input_AddSubtitleOSD( pInput, psz_file, true, true );
292 free( psz_file );
293 }
294 }
295 if( !is_subtitle )
296 {
297 list<string>::const_iterator it = files.begin();
298 for( bool first = true; it != files.end(); ++it, first = false )
299 {
300 bool playOnDrop = m_playOnDrop && first;
301 CmdAddItem( getIntf(), it->c_str(), playOnDrop ).execute();
302 }
303 }
304 }
305 m_pDragControl = NULL;
306}

308void TopWindow::processEvent( EvtDragOver &rEvtDragOver )
309{
310 // Get the control hit by the mouse
311 int xPos = rEvtDragOver.getXPos() - getLeft();
312 int yPos = rEvtDragOver.getYPos() - getTop();
313
314 CtrlGeneric *pHitControl = findHitControl( xPos, yPos );
315
316 if( m_pDragControl && m_pDragControl != pHitControl )
317 {
318 EvtDragLeave evt( getIntf() );
319 m_pDragControl->handleEvent( evt );
320 }
321
322 m_pDragControl = pHitControl;
323
324 if( m_pDragControl )
325 {
326 // Send a dragOver event
327 EvtDragOver evt( getIntf(), xPos, yPos );
328 m_pDragControl->handleEvent( evt );
329 }
330}

360void TopWindow::setActiveLayout( GenericLayout *pLayout )
361{
362 bool isVisible = getVisibleVar().get();
363 if( m_pActiveLayout )
364 {
365 if( isVisible )
366 {
367 m_pActiveLayout->onHide();
368 }
369 // The current layout becomes inactive
370 m_pActiveLayout->getActiveVar().set( false );
371
372 // if both layouts have the same original size, infer a
373 // subsequent resize of the active layout has to be applied
374 // to the new layout about to become active
375 if( pLayout->isTightlyCoupledWith( *m_pActiveLayout ) )
376 pLayout->resize( m_pActiveLayout->getWidth(),
377 m_pActiveLayout->getHeight() );
378 }
379
380 pLayout->setWindow( this );
381 m_pActiveLayout = pLayout;
382 // Get the size of the layout and resize the window
383 resize( pLayout->getWidth(), pLayout->getHeight() );
384
385 if( isVisible )
386 {
387 pLayout->onShow();
388 }
389
390 // The new layout is active
391 pLayout->getActiveVar().set( true );
392}

447void TopWindow::onControlRelease( const CtrlGeneric &rCtrl )
448{
449 // Release the capturing control
450 if( m_pCapturingControl == &rCtrl )
451 {
452 m_pCapturingControl = NULL;
453 }
454 else
455 {
456 msg_Dbg( getIntf(), "control had not captured the mouse" );
457 }
458
459 // Send an enter event to the control under the mouse, if it doesn't
460 // have received it yet
461 if( m_pLastHitControl && m_pLastHitControl != &rCtrl )
462 {
463 EvtEnter evt( getIntf() );
464 m_pLastHitControl->handleEvent( evt );
465
466 // Show the tooltip
467 m_rWindowManager.hideTooltip();
468 UString tipText = m_pLastHitControl->getTooltipText();
469 if( tipText.length() > 0 )
470 {
471 // Set the tooltip text variable
472 VarManager *pVarManager = VarManager::instance( getIntf() );
473 pVarManager->getTooltipText().set( tipText );
474 m_rWindowManager.showTooltip();
475 }
476 }
477}

480void TopWindow::onTooltipChange( const CtrlGeneric &rCtrl )
481{
482 // Check that the control is the active one
483 if( m_pLastHitControl && m_pLastHitControl == &rCtrl )
484 {
485 if( rCtrl.getTooltipText().size() )
486 {
487 // Set the tooltip text variable
488 VarManager *pVarManager = VarManager::instance( getIntf() );
489 pVarManager->getTooltipText().set( rCtrl.getTooltipText() );
490 m_rWindowManager.showTooltip();
491 }
492 else
493 {
494 // Nothing to display, so hide the tooltip
495 m_rWindowManager.hideTooltip();
496 }
497 }

501CtrlGeneric *TopWindow::findHitControl( int xPos, int yPos )
502{
503 if( m_pActiveLayout == NULL )
504 {
505 return NULL;
506 }
507
508 // Get the controls in the active layout
509 const list<LayeredControl> &ctrlList = m_pActiveLayout->getControlList();
510 list<LayeredControl>::const_reverse_iterator iter;
511
512 // New control hit by the mouse
513 CtrlGeneric *pNewHitControl = NULL;
514
515 // Loop on the control list to find the uppest hit control
516 for( iter = ctrlList.rbegin(); iter != ctrlList.rend(); ++iter )
517 {
518 // Get the position of the control in the layout
519 const Position *pos = (*iter).m_pControl->getPosition();
520 if( pos != NULL )
521 {
522 // Compute the coordinates of the mouse relative to the control
523 int xRel = xPos - pos->getLeft();
524 int yRel = yPos - pos->getTop();
525
526 CtrlGeneric *pCtrl = (*iter).m_pControl;
527 // Control hit ?
528 if( pCtrl->isVisible() && pCtrl->mouseOver( xRel, yRel ) )
529 {
530 pNewHitControl = (*iter).m_pControl;
531 break;
532 }
533 }
534 else
535 {
536 msg_Dbg( getIntf(), "control at NULL position" );
537 }
538 }
539
540 // If the hit control has just been entered, send it an enter event
541 if( pNewHitControl && (pNewHitControl != m_pLastHitControl) )
542 {
543 // Don't send the event if another control captured the mouse
544 if( !m_pCapturingControl || (m_pCapturingControl == pNewHitControl ) )
545 {
546 EvtEnter evt( getIntf() );
547 pNewHitControl->handleEvent( evt );
548
549 if( !m_pCapturingControl )
550 {
551 // Show the tooltip
552 m_rWindowManager.hideTooltip();
553 UString tipText = pNewHitControl->getTooltipText();
554 if( tipText.length() > 0 )
555 {
556 // Set the tooltip text variable
557 VarManager *pVarManager = VarManager::instance( getIntf() );
558 pVarManager->getTooltipText().set( tipText );
559 m_rWindowManager.showTooltip();
560 }
561 }
562 }
563 }
564
565 return pNewHitControl;
566}

570void TopWindow::setLastHit( CtrlGeneric *pNewHitControl )
571{
572 // Send a leave event to the left control
573 if( m_pLastHitControl && (pNewHitControl != m_pLastHitControl) )
574 {
575 // Don't send the event if another control captured the mouse
576 if( !m_pCapturingControl || (m_pCapturingControl == m_pLastHitControl))
577 {
578 EvtLeave evt( getIntf() );
579 m_pLastHitControl->handleEvent( evt );
580 }
581 }
582
583 m_pLastHitControl = pNewHitControl;
584}
