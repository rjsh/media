25#ifndef TOOLTIP_HPP
26#define TOOLTIP_HPP
27
28#include "../utils/var_text.hpp"
29#include "../commands/cmd_generic.hpp"
30
31class GenericFont;
32class OSTooltip;
33class OSTimer;
34class OSGraphics;
35class UString;
36
37
38class Tooltip: public SkinObject, public Observer<VarText>
39{
40public:
41 /// Create a tooltip with the given font and delay (in milliseconds)
42 Tooltip( intf_thread_t *pIntf, const GenericFont &rFont, int delay );
43
44 virtual ~Tooltip();
45
46 /// Draw the tooltip and show it after the delay
47 void show();
48
49 /// Hide the tooltip and cancel the timer if a tooltip is waiting for
50 /// display
51 void hide();
52
53private:
54 /// Font
55 const GenericFont &m_rFont;
56 /// Delay before popping the tooltip
57 int m_delay;
58 /// Timer to wait before showing the tooltip
59 OSTimer *m_pTimer;
60 /// OS specific tooltip window
61 OSTooltip *m_pOsTooltip;
62 /// Image of the tooltip
63 OSGraphics *m_pImage;
64 /// Position of the tooltip
65 int m_xPos, m_yPos;
66
67 /// Method called when the observed variable is modified
68 virtual void onUpdate( Subject<VarText> &rVariable, void * );
69
70 /// Display text of the tooltip
71 void displayText( const UString &rText );
72
73 /// Build m_pImage with the given text
74 void makeImage( const UString &rText );
75
76 /// Callback to show the tooltip window
77 DEFINE_CALLBACK( Tooltip, Show );
78};
79
80
81#endif
