25#ifndef FILE_BITMAP_HPP
26#define FILE_BITMAP_HPP
27
28#include "generic_bitmap.hpp"
29#include <string>
30
31
32/// Class for file bitmaps
33class FileBitmap: public GenericBitmap
34{
35public:
36 /// Load a bitmap from a file. aColor is the transparency
37 /// color, in the format 0xRRGGBB
38 FileBitmap( intf_thread_t *pIntf, image_handler_t *pImageHandler,
39 string fileName, uint32_t aColor, int nbFrames = 1,
40 int fps = 0, int nbLoops = 0 );
41
42 virtual ~FileBitmap();
43
44 /// Get the width of the bitmap
45 virtual int getWidth() const { return m_width; }
46
47 /// Get the heighth of the bitmap
48 virtual int getHeight() const { return m_height; }
49
50 /// Get a linear buffer containing the image data.
51 /// Each pixel is stored in 4 bytes in the order B,G,R,A
52 virtual uint8_t *getData() const;
53
54private:
55 /// Size of the bitmap.
56 int m_width, m_height;
57 /// Buffer containing the image data.
58 uint8_t *m_pData;
59};
