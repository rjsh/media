25#ifndef OS_TIMER_HPP
26#define OS_TIMER_HPP
27
28#include "skin_common.hpp"
29
30
31// Base class for OS-specific timers
32class OSTimer: public SkinObject
33{
34public:
35 virtual ~OSTimer() { }
36
37 /// (Re)start the timer with the given delay (in ms). If oneShot is
38 /// true, stop it after the first execution of the callback.
39 virtual void start( int delay, bool oneShot ) = 0;
40
41 /// Stop the timer
42 virtual void stop() = 0;
43
44protected:
45 OSTimer( intf_thread_t *pIntf ): SkinObject( pIntf ) { }
46};
47
48
49#endif
