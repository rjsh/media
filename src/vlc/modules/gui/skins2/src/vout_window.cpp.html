36VoutWindow::VoutWindow( intf_thread_t *pIntf, vout_window_t* pWnd,
37 int width, int height, GenericWindow* pParent ) :
38 GenericWindow( pIntf, 0, 0, false, false, pParent,
39 GenericWindow::VoutWindow ),
40 m_pWnd( pWnd ), original_width( width ), original_height( height ),
41 m_pCtrlVideo( NULL ), m_pParentWindow( pParent )
42{
43 if( m_pWnd )
44 {
45 vlc_object_hold( m_pWnd );
46
47#ifdef X11_SKINS
48 m_pWnd->handle.xid = getOSHandle();
49 m_pWnd->display.x11 = NULL;
50#else
51 m_pWnd->handle.hwnd = getOSHandle();
52#endif
53 }
54}
55
56
57VoutWindow::~VoutWindow()
58{
59 if( m_pWnd )
60 {
61 vlc_object_release( m_pWnd );
62 }
63}

66void VoutWindow::setCtrlVideo( CtrlVideo* pCtrlVideo )
67{
68 if( pCtrlVideo )
69 {
70 hide();
71 const Position *pPos = pCtrlVideo->getPosition();
72 int x = pPos->getLeft();
73 int y = pPos->getTop();
74 int w = pPos->getWidth();
75 int h = pPos->getHeight();
76
77 setParent( pCtrlVideo->getWindow(), x, y, w, h );
78 m_pParentWindow = pCtrlVideo->getWindow();
79
80 show();
81 }
82 else
83 {
84 hide();
85 int w = VoutManager::instance( getIntf() )->getVoutMainWindow()->getWidth();
86 int h = VoutManager::instance( getIntf() )->getVoutMainWindow()->getHeight();
87
88 setParent( VoutManager::instance( getIntf() )->getVoutMainWindow(),
89 0, 0, w, h );
90 m_pParentWindow =
91 VoutManager::instance( getIntf() )->getVoutMainWindow();
92 show();
93 }
94
95 m_pCtrlVideo = pCtrlVideo;
96}

99void VoutWindow::processEvent( EvtKey &rEvtKey )
100{
101 // Only do the action when the key is down
102 if( rEvtKey.getKeyState() == EvtKey::kDown )
103 var_SetInteger( getIntf()->p_libvlc, "key-pressed",
104 rEvtKey.getModKey() );
105}
