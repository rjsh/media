25#ifndef FT2_BITMAP_HPP
26#define FT2_BITMAP_HPP
27
28#include <ft2build.h>
29#include FT_FREETYPE_H
30
31#include "generic_bitmap.hpp"
32
33
34/// Class for rendering freetype glyphs on a bitmap
35class FT2Bitmap: public GenericBitmap
36{
37public:
38 /// Create an empty bitmap
39 FT2Bitmap( intf_thread_t *pIntf, int width, int height );
40
41 virtual ~FT2Bitmap();
42
43 /// Get the width of the bitmap
44 virtual int getWidth() const { return m_width; }
45
46 /// Get the heighth of the bitmap
47 virtual int getHeight() const { return m_height; }
48
49 /// Get a linear buffer containing the image data.
50 /// Each pixel is stored in 4 bytes in the order B,G,R,A
51 virtual uint8_t *getData() const { return m_pData; }
52
53 /// Render a glyph on the bitmap at the given position
54 void draw( const FT_Bitmap &rBitmap, int left, int top,
55 uint32_t color );
56
57private:
58 /// Bitmap size
59 int m_width, m_height;
60 /// Image data buffer
61 uint8_t *m_pData;
62};
63
64
65#endif
