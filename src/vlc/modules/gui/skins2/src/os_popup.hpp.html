25#ifndef OS_POPUP_HPP
26#define OS_POPUP_HPP
27
28#include "skin_common.hpp"
29#include <string>
30
31class OSGraphics;
32
33
34/// Base class for OS specific Popup Windows
35class OSPopup: public SkinObject
36{
37public:
38 virtual ~OSPopup() { }
39
40 /// Show the popup menu at the given (absolute) corrdinates
41 virtual void show( int xPos, int yPos ) = 0;
42
43 /// Hide the popup menu
44 virtual void hide() = 0;
45
46 /// Append a new menu item with the given label to the popup menu
47 virtual void addItem( const string &rLabel, int pos ) = 0;
48
49 /// Create a dummy menu item to separate sections
50 virtual void addSeparator( int pos ) = 0;
51
52 /// Return the position of the item identified by the given id
53 virtual int getPosFromId( int id ) const = 0;
54
55protected:
56 OSPopup( intf_thread_t *pIntf ): SkinObject( pIntf ) { }
57};
58
59#endif
