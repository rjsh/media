33ThemeRepository *ThemeRepository::instance( intf_thread_t *pIntf )
34{
35 if( pIntf->p_sys->p_repository == NULL )
36 {
37 pIntf->p_sys->p_repository = new ThemeRepository( pIntf );

39
40 return pIntf->p_sys->p_repository;
41}

51ThemeRepository::ThemeRepository( intf_thread_t *pIntf ): SkinObject( pIntf )
52{
53 vlc_value_t val, text;
54
55 // Create a variable to add items in wxwindows popup menu
56 var_Create( pIntf, "intf-skins", VLC_VAR_STRING |
57 VLC_VAR_HASCHOICE | VLC_VAR_ISCOMMAND );
58 text.psz_string = _("Select skin");
59 var_Change( pIntf, "intf-skins", VLC_VAR_SETTEXT, &text, NULL );
60
61 // Scan vlt files in the resource path
62 OSFactory *pOsFactory = OSFactory::instance( pIntf );
63 list<string> resPath = pOsFactory->getResourcePath();
64 list<string>::const_iterator it;
65 for( it = resPath.begin(); it != resPath.end(); ++it )
66 {
67 parseDirectory( *it );
68 }
69
70 // retrieve skins from skins directories and locate default skins
71 map<string,string>::const_iterator itmap, itdefault;
72 bool b_default_found = false;
73 for( itmap = m_skinsMap.begin(); itmap != m_skinsMap.end(); ++itmap )
74 {
75 string name = itmap->first;
76 string path = itmap->second;
77 val.psz_string = (char*) path.c_str();
78 text.psz_string = (char*) name.c_str();
79 var_Change( getIntf(), "intf-skins", VLC_VAR_ADDCHOICE, &val,
80 &text );
81
82 if( name == "Default" )
83 {
84 itdefault = itmap;
85 b_default_found = true;
86 }
87 }
88
89 // retrieve last skins stored or skins requested by user
90 char* psz_current = var_InheritString( getIntf(), "skins2-last" );
91 string current = string( psz_current ? psz_current : "" );
92 free( psz_current );
93
94 // check if skins exists and is readable
95 bool b_readable = !ifstream( current.c_str() ).fail();
96
97 msg_Dbg( getIntf(), "requested skins %s is %s accessible",
98 current.c_str(), b_readable ? "" : "NOT" );
99
100 // set the default skins if given skins not accessible
101 if( !b_readable && b_default_found )
102 current = itdefault->second;
103
104 // save this valid skins for reuse
105 config_PutPsz( getIntf(), "skins2-last", current.c_str() );
106
107 // Update repository
108 updateRepository();
109
110 // Set the callback
111 var_AddCallback( pIntf, "intf-skins", changeSkin, this );
112
113 // variable for opening a dialog box to change skins
114 var_Create( pIntf, "intf-skins-interactive", VLC_VAR_VOID |
115 VLC_VAR_ISCOMMAND );
116 text.psz_string = _("Open skin ...");
117 var_Change( pIntf, "intf-skins-interactive", VLC_VAR_SETTEXT, &text, NULL );
118
119 // Set the callback
120 var_AddCallback( pIntf, "intf-skins-interactive", changeSkin, this );
121
122}

125ThemeRepository::~ThemeRepository()
126{
127 m_skinsMap.clear();
128
129 var_DelCallback( getIntf(), "intf-skins", changeSkin, this );

132 var_Destroy( getIntf(), "intf-skins" );
133 var_Destroy( getIntf(), "intf-skins-interactive" );
134}

137void ThemeRepository::parseDirectory( const string &rDir_locale )
138{
139 const char *pszDirContent;
140 // Path separator
141 const string &sep = OSFactory::instance( getIntf() )->getDirSeparator();
142
143 // Open the dir
144 // FIXME: parseDirectory should be invoked with UTF-8 input instead!!
145 string rDir = sFromLocale( rDir_locale );
146 DIR *pDir = vlc_opendir( rDir.c_str() );
147
148 if( pDir == NULL )
149 {
150 // An error occurred
151 msg_Dbg( getIntf(), "cannot open directory %s", rDir.c_str() );
152 return;
153 }
154
155 // While we still have entries in the directory
156 while( ( pszDirContent = vlc_readdir( pDir ) ) != NULL )
157 {
158 string name = pszDirContent;
159 string extension;
160 if( name.size() > 4 )
161 {
162 extension = name.substr( name.size() - 4, 4 );
163 }
164 if( extension == ".vlt" || extension == ".wsz" )
165 {
166 string path = rDir + sep + name;
167 string shortname = name.substr( 0, name.size() - 4 );
168 for( string::size_type i = 0; i < shortname.size(); i++ )
169 shortname[i] = ( i == 0 ) ?
170 toupper( (unsigned char)shortname[i] ) :
171 tolower( (unsigned char)shortname[i] );
172 m_skinsMap[shortname] = path;
173
174 msg_Dbg( getIntf(), "found skin %s", path.c_str() );
175 }
176 }
177
178 closedir( pDir );
179}

183int ThemeRepository::changeSkin( vlc_object_t *pIntf, char const *pVariable,
184 vlc_value_t oldval, vlc_value_t newval,
185 void *pData )
186{
187 (void)pIntf; (void)oldval;
188 ThemeRepository *pThis = (ThemeRepository*)(pData);
189
190 if( !strcmp( pVariable, "intf-skins-interactive" ) )
191 {
192 CmdDlgChangeSkin cmd( pThis->getIntf() );
193 cmd.execute();
194 }
195 else if( !strcmp( pVariable, "intf-skins" ) )
196 {
197 // Try to load the new skin
198 CmdChangeSkin *pCmd = new CmdChangeSkin( pThis->getIntf(),
199 newval.psz_string );
200 AsyncQueue *pQueue = AsyncQueue::instance( pThis->getIntf() );
201 pQueue->push( CmdGenericPtr( pCmd ) );
202 }
203
204 return VLC_SUCCESS;
205}

208void ThemeRepository::updateRepository()
209{
210 vlc_value_t val, text;
211
212 // retrieve the current skin
213 char* psz_current = config_GetPsz( getIntf(), "skins2-last" );
214 if( !psz_current )
215 return;
216
217 val.psz_string = psz_current;
218 text.psz_string = psz_current;
219
220 // add this new skins if not yet present in repository
221 string current( psz_current );
222 map<string,string>::const_iterator it;
223 for( it = m_skinsMap.begin(); it != m_skinsMap.end(); ++it )
224 {
225 if( it->second == current )
226 break;
227 }
228 if( it == m_skinsMap.end() )
229 {
230 var_Change( getIntf(), "intf-skins", VLC_VAR_ADDCHOICE, &val,
231 &text );
232 string name = psz_current;
233 m_skinsMap[name] = name;
234 }
235
236 // mark this current skins as 'checked' in list
237 var_Change( getIntf(), "intf-skins", VLC_VAR_SETVALUE, &val, NULL );
238
239 free( psz_current );
240}
