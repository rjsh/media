25#include "os_factory.hpp"
26
27#ifdef X11_SKINS
28#include "../x11/x11_factory.hpp"
29#elif defined WIN32_SKINS
30#include "../win32/win32_factory.hpp"
31#elif defined MACOSX_SKINS
32#include "../macosx/macosx_factory.hpp"
33#elif defined OS2_SKINS
34#include "../os2/os2_factory.hpp"
35#endif
36
37OSFactory *OSFactory::instance( intf_thread_t *pIntf )
38{
39 if( ! pIntf->p_sys->p_osFactory )
40 {
41 OSFactory *pOsFactory;
42#ifdef X11_SKINS
43 pOsFactory = new X11Factory( pIntf );
44#elif defined WIN32_SKINS
45 pOsFactory = new Win32Factory( pIntf );
46#elif defined MACOSX_SKINS
47 pOsFactory = new MacOSXFactory( pIntf );
48#elif defined OS2_SKINS
49 pOsFactory = new OS2Factory( pIntf );
50#else
51#error "No OSFactory implementation !"
52#endif
53
54 if( pOsFactory->init() )
55 {
56 // Initialization succeeded
57 pIntf->p_sys->p_osFactory = pOsFactory;
58 }
59 else
60 {
61 // Initialization failed
62 delete pOsFactory;
63 }
64 }
65 return pIntf->p_sys->p_osFactory;
66}
67
68
69void OSFactory::destroy( intf_thread_t *pIntf )
70{
71 delete pIntf->p_sys->p_osFactory;
72 pIntf->p_sys->p_osFactory = NULL;
73}
