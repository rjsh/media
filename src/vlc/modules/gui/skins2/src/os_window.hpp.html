25#ifndef OS_WINDOW_HPP
26#define OS_WINDOW_HPP
27
28#include "skin_common.hpp"
29
30class GenericWindow;
31class OSGraphics;
32
33
34/// OS specific delegate class for GenericWindow
35class OSWindow: public SkinObject
36{
37public:
38 virtual ~OSWindow() { }
39
40 // Show the window
41 virtual void show() const = 0;
42
43 // Hide the window
44 virtual void hide() const = 0;
45
46 /// Move and resize the window
47 virtual void moveResize( int left, int top,
48 int width, int height ) const = 0;
49
50 /// Bring the window on top
51 virtual void raise() const = 0;
52
53 /// Set the opacity of the window (0 = transparent, 255 = opaque)
54 virtual void setOpacity( uint8_t value ) const = 0;
55
56 /// Toggle the window on top
57 virtual void toggleOnTop( bool onTop ) const = 0;
58
59 /// getter for handler
60 virtual vlc_wnd_type getOSHandle( ) const = 0;
61
62 /// reparent the window
63 virtual void reparent( vlc_wnd_type OSHandle,
64 int x, int y, int w, int h ) = 0;
65
66 /// updateWindow (tell the OS we need to update the window)
67 virtual bool invalidateRect( int x, int y, int w, int h ) const = 0;
68
69protected:
70 OSWindow( intf_thread_t *pIntf ): SkinObject( pIntf ) { }
71};
72
73
74#endif
