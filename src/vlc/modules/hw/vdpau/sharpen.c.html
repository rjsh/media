33struct filter_sys_t
34{
35 atomic_uint_fast32_t sigma;
36};
37
38static float vlc_to_vdp_sigma(float sigma)
39{
40 sigma /= 2.f;
41 if (sigma > +1.f)
42 sigma = +1.f;
43 if (sigma < -1.f)
44 sigma = -1.f;
45 return sigma;
46}
47
48static int SharpenCallback(vlc_object_t *obj, const char *varname,
49 vlc_value_t prev, vlc_value_t cur, void *data)
50{
51 filter_sys_t *sys = data;
52 union { uint32_t u; float f; } u;
53
54 u.f = vlc_to_vdp_sigma(cur.f_float);
55 atomic_store(&sys->sigma, u.u);
56 (void) obj; (void) varname; (void) prev;
57 return VLC_SUCCESS;
58}
59
60static picture_t *Sharpen(filter_t *filter, picture_t *pic)
61{
62 filter_sys_t *sys = filter->p_sys;
63 vlc_vdp_video_field_t *f = pic->context;
64 union { uint32_t u; float f; } u;
65
66 if (unlikely(f == NULL))
67 return pic;
68
69 u.u = atomic_load(&sys->sigma);
70 f->sharpen += u.f;
71 if (f->sharpen > +1.f)
72 f->sharpen = +1.f;
73 if (f->sharpen < -1.f)
74 f->sharpen = -1.f;
75
76 return pic;
77}

79static const char *const options[] = { "sigma", NULL };
80
81static int Open(vlc_object_t *obj)
82{
83 filter_t *filter = (filter_t *)obj;
84
85 if (filter->fmt_in.video.i_chroma != VLC_CODEC_VDPAU_VIDEO_422
86 && filter->fmt_in.video.i_chroma != VLC_CODEC_VDPAU_VIDEO_420)
87 return VLC_EGENERIC;
88 if (!video_format_IsSimilar(&filter->fmt_in.video, &filter->fmt_out.video))
89 return VLC_EGENERIC;
90
91 /* Check for sharpen support */
92 vdp_t *vdp;
93 VdpDevice device;
94 VdpStatus err;
95 VdpBool ok;
96
97 err = vdp_get_x11(NULL, -1, &vdp, &device);
98 if (err != VDP_STATUS_OK)
99 return VLC_EGENERIC; /* Weird. The decoder should be active... */
100
101 err = vdp_video_mixer_query_feature_support(vdp, device,
102 VDP_VIDEO_MIXER_FEATURE_SHARPNESS, &ok);
103 if (err != VDP_STATUS_OK)
104 ok = VDP_FALSE;
105 vdp_release_x11(vdp);
106
107 if (ok != VDP_TRUE)
108 {
109 msg_Err(filter, "sharpening/blurring not supported by VDPAU device");
110 return VLC_EGENERIC;
111 }
112
113 /* Initialization */
114 filter_sys_t *sys = malloc(sizeof (*sys));
115 if (unlikely(sys == NULL))
116 return VLC_ENOMEM;
117
118 filter->pf_video_filter = Sharpen;
119 filter->p_sys = sys;
120
121 config_ChainParse(filter, "sharpen-", options, filter->p_cfg);
122 var_AddCallback(filter, "sharpen-sigma", SharpenCallback, sys);
123
124 union { uint32_t u; float f; } u;
125 u.f = vlc_to_vdp_sigma(var_CreateGetFloatCommand(filter, "sharpen-sigma"));
126 atomic_init(&sys->sigma, u.u);
127
128 return VLC_SUCCESS;
129}
130
131static void Close(vlc_object_t *obj)
132{
133 filter_t *filter = (filter_t *)obj;
134 filter_sys_t *sys = filter->p_sys;
135
136 var_DelCallback(filter, "sharpen-sigma", SharpenCallback, sys);
137 free(sys);
138}
