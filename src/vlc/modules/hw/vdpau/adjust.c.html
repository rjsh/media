34struct filter_sys_t
35{
36 vlc_atomic_float brightness;
37 vlc_atomic_float contrast;
38 vlc_atomic_float saturation;
39 vlc_atomic_float hue;
40};
41
42static float vlc_to_vdp_brightness(float brightness)
43{
44 brightness -= 1.f;
45 if (brightness > +1.f)
46 brightness = +1.f;
47 if (brightness < -1.f)
48 brightness = -1.f;
49 return brightness;
50}
51
52static int BrightnessCallback(vlc_object_t *obj, const char *varname,
53 vlc_value_t prev, vlc_value_t cur, void *data)
54{
55 vlc_atomic_store_float(data, vlc_to_vdp_brightness(cur.f_float));
56 (void) obj; (void) varname; (void) prev;
57 return VLC_SUCCESS;
58}
59
60static float vlc_to_vdp_contrast(float contrast)
61{
62 if (contrast > 10.f)
63 contrast = 10.f;
64 if (contrast < 0.f)
65 contrast = 0.f;
66 return contrast;
67}
68
69static int ContrastCallback(vlc_object_t *obj, const char *varname,
70 vlc_value_t prev, vlc_value_t cur, void *data)
71{
72 vlc_atomic_store_float(data, vlc_to_vdp_contrast(cur.f_float));
73 (void) obj; (void) varname; (void) prev;
74 return VLC_SUCCESS;
75}
76
77#define vlc_to_vdp_saturation vlc_to_vdp_contrast
78
79static int SaturationCallback(vlc_object_t *obj, const char *varname,
80 vlc_value_t prev, vlc_value_t cur, void *data)
81{
82 vlc_atomic_store_float(data, vlc_to_vdp_saturation(cur.f_float));
83 (void) obj; (void) varname; (void) prev;
84 return VLC_SUCCESS;
85}
86
87static float vlc_to_vdp_hue(int hue)
88{
89 hue %= 360;
90 if (hue > 180)
91 hue -= 360;
92 return (float)hue * (float)(M_PI / 180.);
93}
94
95static int HueCallback(vlc_object_t *obj, const char *varname,
96 vlc_value_t prev, vlc_value_t cur, void *data)
97{
98
99 vlc_atomic_store_float(data, vlc_to_vdp_hue(cur.i_int));
100 (void) obj; (void) varname; (void) prev;
101 return VLC_SUCCESS;
102}
103
104static picture_t *Adjust(filter_t *filter, picture_t *pic)
105{
106 filter_sys_t *sys = filter->p_sys;
107 vlc_vdp_video_field_t *f = pic->context;
108
109 if (unlikely(f == NULL))
110 return pic;
111
112 f->procamp.brightness = vlc_atomic_load_float(&sys->brightness);
113 f->procamp.contrast = vlc_atomic_load_float(&sys->contrast);
114 f->procamp.saturation = vlc_atomic_load_float(&sys->saturation);
115 f->procamp.hue = vlc_atomic_load_float(&sys->hue);
116
117 return pic;
118}
119
120static const char *const options[] = {
121 "brightness", "contrast", "saturation", "hue", NULL
122};
123
124static int Open(vlc_object_t *obj)
125{
126 filter_t *filter = (filter_t *)obj;
127
128 if (filter->fmt_in.video.i_chroma != VLC_CODEC_VDPAU_VIDEO_422
129 && filter->fmt_in.video.i_chroma != VLC_CODEC_VDPAU_VIDEO_420)
130 return VLC_EGENERIC;
131 if (!video_format_IsSimilar(&filter->fmt_in.video, &filter->fmt_out.video))
132 return VLC_EGENERIC;
133
134 filter_sys_t *sys = malloc(sizeof (*sys));
135 if (unlikely(sys == NULL))
136 return VLC_ENOMEM;
137
138 filter->pf_video_filter = Adjust;
139 filter->p_sys = sys;
140
141 config_ChainParse(filter, "", options, filter->p_cfg);
142
143 float f;
144 int i;
145
146 f = var_CreateGetFloatCommand(filter, "brightness");
147 var_AddCallback(filter, "brightness", BrightnessCallback,
148 &sys->brightness);
149 vlc_atomic_init_float(&sys->brightness, vlc_to_vdp_brightness(f));
150
151 f = var_CreateGetFloatCommand(filter, "contrast");
152 var_AddCallback(filter, "contrast", ContrastCallback, &sys->contrast);
153 vlc_atomic_init_float(&sys->contrast, vlc_to_vdp_contrast(f));
154
155 f = var_CreateGetFloatCommand(filter, "saturation");
156 var_AddCallback(filter, "saturation", SaturationCallback,
157 &sys->saturation);
158 vlc_atomic_init_float(&sys->saturation, vlc_to_vdp_saturation(f));
159
160 i = var_CreateGetIntegerCommand(filter, "hue");
161 var_AddCallback(filter, "hue", HueCallback, &sys->hue);
162 vlc_atomic_init_float(&sys->hue, vlc_to_vdp_hue(i));
163
164 return VLC_SUCCESS;
165}
166
167static void Close(vlc_object_t *obj)
168{
169 filter_t *filter = (filter_t *)obj;
170 filter_sys_t *sys = filter->p_sys;
171
172 var_DelCallback(filter, "hue", HueCallback, &sys->hue);
173 var_DelCallback(filter, "saturation", SaturationCallback,
174 &sys->saturation);
175 var_DelCallback(filter, "contrast", ContrastCallback, &sys->contrast);
176 var_DelCallback(filter, "brightness", BrightnessCallback,
177 &sys->brightness);
178 free(sys);
179}
