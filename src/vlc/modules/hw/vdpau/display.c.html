55struct vout_display_sys_t
56{
57 xcb_connection_t *conn; /**< XCB connection */
58 vout_window_t *embed; /**< parent window */
59 vdp_t *vdp; /**< VDPAU back-end */
60 picture_t *current; /**< Currently visible picture */
61
62 xcb_window_t window; /**< target window (owned by VDPAU back-end) */
63 xcb_cursor_t cursor; /**< blank cursor */
64 VdpDevice device; /**< VDPAU device handle */
65 VdpPresentationQueueTarget target; /**< VDPAU presentation queue target */

68
69 picture_pool_t *pool; /**< pictures pool */
70};

85static VdpStatus picture_NewVDPAU(vdp_t *vdp, VdpRGBAFormat rgb_fmt,
86 const video_format_t *restrict fmt,
87 picture_t **restrict picp)
88{
89 picture_sys_t *psys = malloc(sizeof (*psys));
90 if (unlikely(psys == NULL))
91 return VDP_STATUS_RESOURCES;
92
93 psys->vdp = vdp_hold_x11(vdp, &psys->device);
94
95 VdpStatus err = vdp_output_surface_create(psys->vdp, psys->device,
96 rgb_fmt, fmt->i_visible_width, fmt->i_visible_height,
97 &psys->surface);
98 if (err != VDP_STATUS_OK)
99 {
100 vdp_release_x11(psys->vdp);
101 free(psys);
102 return err;
103 }
104
105 picture_resource_t res = {
106 .p_sys = psys,
107 .pf_destroy = PictureDestroyVDPAU,
108 };
109
110 picture_t *pic = picture_NewFromResource(fmt, &res);
111 if (unlikely(pic == NULL))
112 {
113 pictureSys_DestroyVDPAU(psys);
114 return VDP_STATUS_RESOURCES;
115 }
116 *picp = pic;
117 return VDP_STATUS_OK;
118}

120static picture_pool_t *PoolAlloc(vout_display_t *vd, unsigned requested_count)
121{
122 vout_display_sys_t *sys = vd->sys;
123 picture_t *pics[requested_count];
124
125 unsigned count = 0;
126 while (count < requested_count)
127 {
128 VdpStatus err = picture_NewVDPAU(sys->vdp, sys->rgb_fmt, &vd->fmt,
129 pics + count);
130 if (err != VDP_STATUS_OK)
131 {
132 msg_Err(vd, "%s creation failure: %s", "output surface",
133 vdp_get_error_string(sys->vdp, err));
134 break;
135 }
136 count++;
137 }
138 sys->current = NULL;
139
140 if (count == 0)
141 return NULL;
142
143 picture_pool_t *pool = picture_pool_New(count, pics);
144 if (unlikely(pool == NULL))
145 while (count > 0)
146 picture_Release(pics[--count]);
147 return pool;
148}
149
150static void PoolFree(vout_display_t *vd, picture_pool_t *pool)
151{
152 vout_display_sys_t *sys = vd->sys;
153
154 if (sys->current != NULL)
155 picture_Release(sys->current);
156 picture_pool_Delete(pool);
157}
158
159static picture_pool_t *Pool(vout_display_t *vd, unsigned requested_count)
160{
161 vout_display_sys_t *sys = vd->sys;
162
163 if (sys->pool == NULL)
164 sys->pool = PoolAlloc(vd, requested_count);
165 return sys->pool;
166}

168static void RenderRegion(vout_display_t *vd, VdpOutputSurface target,
169 const subpicture_t *subpic,
170 const subpicture_region_t *reg)
171{
172 vout_display_sys_t *sys = vd->sys;
173 VdpBitmapSurface surface;
174#ifdef WORDS_BIGENDIAN
175 VdpRGBAFormat fmt = VDP_RGBA_FORMAT_B8G8R8A8;
176#else
177 VdpRGBAFormat fmt = VDP_RGBA_FORMAT_R8G8B8A8;
178#endif
179 VdpStatus err;
180
181 /* Create GPU surface for sub-picture */
182 err = vdp_bitmap_surface_create(sys->vdp, sys->device, fmt,
183 reg->fmt.i_visible_width, reg->fmt.i_visible_height, VDP_FALSE,
184 &surface);
185 if (err != VDP_STATUS_OK)
186 {
187 msg_Err(vd, "%s creation failure: %s", "bitmap surface",
188 vdp_get_error_string(sys->vdp, err));
189 return;
190 }
191
192 /* Upload sub-picture to GPU surface */
193 picture_t *pic = reg->p_picture;
194 const void *data = pic->p[0].p_pixels;
195 uint32_t pitch = pic->p[0].i_pitch;
196
197 err = vdp_bitmap_surface_put_bits_native(sys->vdp, surface, &data, &pitch,
198 NULL);
199 if (err != VDP_STATUS_OK)
200 {
201 msg_Err(vd, "subpicture upload failure: %s",
202 vdp_get_error_string(sys->vdp, err));
203 goto out;
204 }
205
206 /* Render onto main surface */
207 VdpRect area = {
208 reg->i_x * vd->fmt.i_visible_width
209 / subpic->i_original_picture_width,
210 reg->i_y * vd->fmt.i_visible_height
211 / subpic->i_original_picture_height,
212 (reg->i_x + reg->fmt.i_visible_width) * vd->fmt.i_visible_width
213 / subpic->i_original_picture_width,
214 (reg->i_y + reg->fmt.i_visible_height) * vd->fmt.i_visible_height
215 / subpic->i_original_picture_height,
216 };
217 VdpColor color = { 1.f, 1.f, 1.f,
218 reg->i_alpha * subpic->i_alpha / 65535.f };
219 VdpOutputSurfaceRenderBlendState state = {
220 .struct_version = VDP_OUTPUT_SURFACE_RENDER_BLEND_STATE_VERSION,
221 .blend_factor_source_color =
222 VDP_OUTPUT_SURFACE_RENDER_BLEND_FACTOR_SRC_ALPHA,
223 .blend_factor_destination_color =
224 VDP_OUTPUT_SURFACE_RENDER_BLEND_FACTOR_ONE_MINUS_SRC_ALPHA,
225 .blend_factor_source_alpha =
226 VDP_OUTPUT_SURFACE_RENDER_BLEND_FACTOR_ZERO,
227 .blend_factor_destination_alpha =
228 VDP_OUTPUT_SURFACE_RENDER_BLEND_FACTOR_ONE,
229 .blend_equation_color = VDP_OUTPUT_SURFACE_RENDER_BLEND_EQUATION_ADD,
230 .blend_equation_alpha = VDP_OUTPUT_SURFACE_RENDER_BLEND_EQUATION_ADD,
231 .blend_constant = { 0.f, 0.f, 0.f, 0.f },
232 };
233
234 err = vdp_output_surface_render_bitmap_surface(sys->vdp, target, &area,
235 surface, NULL, &color, &state, 0);
236 if (err != VDP_STATUS_OK)
237 msg_Err(vd, "blending failure: %s",
238 vdp_get_error_string(sys->vdp, err));
239
240out:/* Destroy GPU surface */
241 vdp_bitmap_surface_destroy(sys->vdp, surface);
242}

244static void Queue(vout_display_t *vd, picture_t *pic, subpicture_t *subpic)
245{
246 vout_display_sys_t *sys = vd->sys;
247 VdpOutputSurface surface = pic->p_sys->surface;
248 VdpStatus err;
249
250 VdpPresentationQueueStatus status;
251 VdpTime ts;
252 err = vdp_presentation_queue_query_surface_status(sys->vdp, sys->queue,
253 surface, &status, &ts);
254 if (err == VDP_STATUS_OK && status != VDP_PRESENTATION_QUEUE_STATUS_IDLE)
255 msg_Dbg(vd, "surface status: %u", status);
256
257 if (subpic != NULL)
258 for (subpicture_region_t *r = subpic->p_region; r != NULL;
259 r = r->p_next)
260 RenderRegion(vd, surface, subpic, r);
261
262 /* Compute picture presentation time */
263 mtime_t now = mdate();
264 VdpTime pts;
265
266 err = vdp_presentation_queue_get_time(sys->vdp, sys->queue, &pts);
267 if (err != VDP_STATUS_OK)
268 {
269 msg_Err(vd, "presentation queue time failure: %s",
270 vdp_get_error_string(sys->vdp, err));
271 if (err == VDP_STATUS_DISPLAY_PREEMPTED)
272 vout_display_SendEventPicturesInvalid(vd);
273 return;
274 }
275
276 mtime_t delay = pic->date - now;
277 if (delay < 0)
278 delay = 0; /* core bug: date is not updated during pause */
279 if (unlikely(delay > CLOCK_FREQ))
280 { /* We would get stuck if the delay was too long. */
281 msg_Dbg(vd, "picture date corrupt: delay of %"PRId64" us", delay);
282 delay = CLOCK_FREQ / 50;
283 }
284 pts += delay * 1000;
285
286 /* Queue picture */
287 err = vdp_presentation_queue_display(sys->vdp, sys->queue, surface, 0, 0,
288 pts);
289 if (err != VDP_STATUS_OK)
290 msg_Err(vd, "presentation queue display failure: %s",
291 vdp_get_error_string(sys->vdp, err));
292}

294static void Wait(vout_display_t *vd, picture_t *pic, subpicture_t *subpicture)
295{
296 vout_display_sys_t *sys = vd->sys;
297
298 picture_t *current = sys->current;
299 if (current != NULL)
300 {
301 picture_sys_t *psys = current->p_sys;
302 VdpTime pts;
303 VdpStatus err;
304
305 err = vdp_presentation_queue_block_until_surface_idle(sys->vdp,
306 sys->queue, psys->surface, &pts);
307 if (err != VDP_STATUS_OK)
308 {
309 msg_Err(vd, "presentation queue blocking error: %s",
310 vdp_get_error_string(sys->vdp, err));
311 picture_Release(pic);
312 return;
313 }
314 picture_Release(current);
315 }
316
317 sys->current = pic;
318 (void) subpicture;
319}

321static int Control(vout_display_t *vd, int query, va_list ap)
322{
323 vout_display_sys_t *sys = vd->sys;
324
325 switch (query)
326 {
327 case VOUT_DISPLAY_HIDE_MOUSE:
328 xcb_change_window_attributes(sys->conn, sys->embed->handle.xid,
329 XCB_CW_CURSOR, &(uint32_t){ sys->cursor });
330 break;
331 case VOUT_DISPLAY_RESET_PICTURES:
332 {
333 msg_Dbg(vd, "resetting pictures");
334 if (sys->pool != NULL)
335 {
336 PoolFree(vd, sys->pool);
337 sys->pool = NULL;
338 }
339
340 const video_format_t *src= &vd->source;
341 video_format_t *fmt = &vd->fmt;
342 vout_display_place_t place;
343
344 vout_display_PlacePicture(&place, src, vd->cfg, false);
345
346 fmt->i_width = src->i_width * place.width / src->i_visible_width;
347 fmt->i_height = src->i_height * place.height / src->i_visible_height;
348 fmt->i_visible_width = place.width;
349 fmt->i_visible_height = place.height;
350 fmt->i_x_offset = src->i_x_offset * place.width / src->i_visible_width;
351 fmt->i_y_offset = src->i_y_offset * place.height / src->i_visible_height;
352
353 const uint32_t values[] = { place.x, place.y,
354 place.width, place.height, };
355 xcb_configure_window(sys->conn, sys->window,
356 XCB_CONFIG_WINDOW_X|XCB_CONFIG_WINDOW_Y|
357 XCB_CONFIG_WINDOW_WIDTH|XCB_CONFIG_WINDOW_HEIGHT,
358 values);
359 break;
360 }
361 case VOUT_DISPLAY_CHANGE_FULLSCREEN:
362 {
363 const vout_display_cfg_t *c = va_arg(ap, const vout_display_cfg_t *);
364 return vout_window_SetFullScreen(sys->embed, c->is_fullscreen);
365 }
366 case VOUT_DISPLAY_CHANGE_WINDOW_STATE:
367 {
368 unsigned state = va_arg(ap, unsigned);
369 return vout_window_SetState(sys->embed, state);
370 }
371 case VOUT_DISPLAY_CHANGE_DISPLAY_SIZE:
372 {
373 const vout_display_cfg_t *cfg = va_arg(ap, const vout_display_cfg_t *);
374 bool forced = va_arg(ap, int);
375 if (forced)
376 {
377 vout_window_SetSize(sys->embed,
378 cfg->display.width, cfg->display.height);
379 return VLC_EGENERIC; /* Always fail. See x11.c for rationale. */
380 }
381
382 vout_display_place_t place;
383 vout_display_PlacePicture(&place, &vd->source, cfg, false);
384 if (place.width != vd->fmt.i_visible_width
385 || place.height != vd->fmt.i_visible_height)
386 {
387 vout_display_SendEventPicturesInvalid(vd);
388 return VLC_SUCCESS;
389 }
390
391 const uint32_t values[] = { place.x, place.y,
392 place.width, place.height, };
393 xcb_configure_window(sys->conn, sys->window,
394 XCB_CONFIG_WINDOW_X|XCB_CONFIG_WINDOW_Y|
395 XCB_CONFIG_WINDOW_WIDTH|XCB_CONFIG_WINDOW_HEIGHT,
396 values);
397 break;
398 }
399 case VOUT_DISPLAY_CHANGE_DISPLAY_FILLED:
400 case VOUT_DISPLAY_CHANGE_ZOOM:
401 case VOUT_DISPLAY_CHANGE_SOURCE_ASPECT:
402 case VOUT_DISPLAY_CHANGE_SOURCE_CROP:
403 vout_display_SendEventPicturesInvalid (vd);
404 return VLC_SUCCESS;
405 default:
406 msg_Err(vd, "unknown control request %d", query);
407 return VLC_EGENERIC;
408 }
409 xcb_flush (sys->conn);
410 return VLC_SUCCESS;
411}

421static int xcb_screen_num(xcb_connection_t *conn, const xcb_screen_t *screen)
422{
423 const xcb_setup_t *setup = xcb_get_setup(conn);
424 unsigned snum = 0;
425
426 for (xcb_screen_iterator_t i = xcb_setup_roots_iterator(setup);
427 i.rem > 0; xcb_screen_next(&i))
428 {
429 if (i.data->root == screen->root)
430 return snum;
431 snum++;
432 }
433 return -1;
434}

436static int Open(vlc_object_t *obj)
437{
438 if (!vlc_xlib_init(obj))
439 return VLC_EGENERIC;
440
441 vout_display_t *vd = (vout_display_t *)obj;
442 vout_display_sys_t *sys = malloc(sizeof (*sys));
443 if (unlikely(sys == NULL))
444 return VLC_ENOMEM;
445
446 const xcb_screen_t *screen;
447 uint16_t width, height;
448 sys->embed = XCB_parent_Create(vd, &sys->conn, &screen, &width, &height);
449 if (sys->embed == NULL)
450 {
451 free(sys);
452 return VLC_EGENERIC;
453 }
454
455 /* Load the VDPAU back-end and create a device instance */
456 VdpStatus err = vdp_get_x11(sys->embed->display.x11,
457 xcb_screen_num(sys->conn, screen),
458 &sys->vdp, &sys->device);
459 if (err != VDP_STATUS_OK)
460 {
461 msg_Dbg(obj, "device creation failure: error %d", (int)err);
462 xcb_disconnect(sys->conn);
463 vout_display_DeleteWindow(vd, sys->embed);
464 free(sys);
465 return VLC_EGENERIC;
466 }
467
468 const char *info;
469 if (vdp_get_information_string(sys->vdp, &info) == VDP_STATUS_OK)
470 msg_Dbg(vd, "using back-end %s", info);
471
472 /* Check source format */
473 video_format_t fmt;
474 VdpChromaType chroma;
475 VdpYCbCrFormat format;
476
477 video_format_ApplyRotation(&fmt, &vd->fmt);
478
479 if (fmt.i_chroma == VLC_CODEC_VDPAU_VIDEO_420
480 || fmt.i_chroma == VLC_CODEC_VDPAU_VIDEO_422)
481 ;
482 else
483 if (vlc_fourcc_to_vdp_ycc(fmt.i_chroma, &chroma, &format))
484 {
485 uint32_t w, h;
486 VdpBool ok;
487
488 err = vdp_video_surface_query_capabilities(sys->vdp, sys->device,
489 chroma, &ok, &w, &h);
490 if (err != VDP_STATUS_OK)
491 {
492 msg_Err(vd, "%s capabilities query failure: %s", "video surface",
493 vdp_get_error_string(sys->vdp, err));
494 goto error;
495 }
496 if (!ok || w < fmt.i_width || h < fmt.i_height)
497 {
498 msg_Err(vd, "source video %s not supported", "chroma type");
499 goto error;
500 }
501
502 err = vdp_video_surface_query_get_put_bits_y_cb_cr_capabilities(
503 sys->vdp, sys->device, chroma, format, &ok);
504 if (err != VDP_STATUS_OK)
505 {
506 msg_Err(vd, "%s capabilities query failure: %s", "video surface",
507 vdp_get_error_string(sys->vdp, err));
508 goto error;
509 }
510 if (!ok)
511 {
512 msg_Err(vd, "source video %s not supported", "YCbCr format");
513 goto error;
514 }
515 }
516 else
517 goto error;
518
519 /* Check video mixer capabilities */
520 {
521 uint32_t min, max;
522
523 err = vdp_video_mixer_query_parameter_value_range(sys->vdp,
524 sys->device, VDP_VIDEO_MIXER_PARAMETER_VIDEO_SURFACE_WIDTH,
525 &min, &max);
526 if (err != VDP_STATUS_OK)
527 {
528 msg_Err(vd, "%s capabilities query failure: %s",
529 "video mixer surface width",
530 vdp_get_error_string(sys->vdp, err));
531 goto error;
532 }
533 if (min > fmt.i_width || fmt.i_width > max)
534 {
535 msg_Err(vd, "source video %s not supported", "width");
536 goto error;
537 }
538
539 err = vdp_video_mixer_query_parameter_value_range(sys->vdp,
540 sys->device, VDP_VIDEO_MIXER_PARAMETER_VIDEO_SURFACE_HEIGHT,
541 &min, &max);
542 if (err != VDP_STATUS_OK)
543 {
544 msg_Err(vd, "%s capabilities query failure: %s",
545 "video mixer surface height",
546 vdp_get_error_string(sys->vdp, err));
547 goto error;
548 }
549 if (min > fmt.i_height || fmt.i_height > max)
550 {
551 msg_Err(vd, "source video %s not supported", "height");
552 goto error;
553 }
554 }
555 fmt.i_chroma = VLC_CODEC_VDPAU_OUTPUT;
556
557 /* Select surface format */
558 static const VdpRGBAFormat rgb_fmts[] = {
559 VDP_RGBA_FORMAT_R10G10B10A2, VDP_RGBA_FORMAT_B10G10R10A2,
560 VDP_RGBA_FORMAT_B8G8R8A8, VDP_RGBA_FORMAT_R8G8B8A8,
561 };
562 unsigned i;
563
564 for (i = 0; i < sizeof (rgb_fmts) / sizeof (rgb_fmts[0]); i++)
565 {
566 uint32_t w, h;
567 VdpBool ok;
568
569 err = vdp_output_surface_query_capabilities(sys->vdp, sys->device,
570 rgb_fmts[i], &ok, &w, &h);
571 if (err != VDP_STATUS_OK)
572 {
573 msg_Err(vd, "%s capabilities query failure: %s", "output surface",
574 vdp_get_error_string(sys->vdp, err));
575 continue;
576 }
577 /* NOTE: Wrong! No warranties that zoom <= 100%! */
578 if (!ok || w < fmt.i_width || h < fmt.i_height)
579 continue;
580
581 sys->rgb_fmt = rgb_fmts[i];
582 msg_Dbg(vd, "using RGBA format %u", sys->rgb_fmt);
583 break;
584 }
585 if (i == sizeof (rgb_fmts) / sizeof (rgb_fmts[0]))
586 {
587 msg_Err(vd, "no supported output surface format");
588 goto error;
589 }
590
591 /* VDPAU-X11 requires a window dedicated to the back-end */
592 {
593 xcb_pixmap_t pix = xcb_generate_id(sys->conn);
594 xcb_create_pixmap(sys->conn, screen->root_depth, pix,
595 screen->root, 1, 1);
596
597 uint32_t mask =
598 XCB_CW_BACK_PIXMAP | XCB_CW_BACK_PIXEL |
599 XCB_CW_BORDER_PIXMAP | XCB_CW_BORDER_PIXEL |
600 XCB_CW_EVENT_MASK | XCB_CW_COLORMAP;
601 const uint32_t values[] = {
602 pix, screen->black_pixel, pix, screen->black_pixel,
603 XCB_EVENT_MASK_VISIBILITY_CHANGE, screen->default_colormap
604 };
605 vout_display_place_t place;
606
607 vout_display_PlacePicture (&place, &vd->source, vd->cfg, false);
608 sys->window = xcb_generate_id(sys->conn);
609
610 xcb_void_cookie_t c =
611 xcb_create_window_checked(sys->conn, screen->root_depth,
612 sys->window, sys->embed->handle.xid, place.x, place.y,
613 place.width, place.height, 0, XCB_WINDOW_CLASS_INPUT_OUTPUT,
614 screen->root_visual, mask, values);
615 if (XCB_error_Check(vd, sys->conn, "window creation failure", c))
616 goto error;
617 msg_Dbg(vd, "using X11 window 0x%08"PRIx32, sys->window);
618 xcb_map_window(sys->conn, sys->window);
619 }
620
621 /* Check bitmap capabilities (for SPU) */
622 const vlc_fourcc_t *spu_chromas = NULL;
623 {
624#ifdef WORDS_BIGENDIAN
625 static const vlc_fourcc_t subpicture_chromas[] = { VLC_CODEC_ARGB, 0 };
626#else
627 static const vlc_fourcc_t subpicture_chromas[] = { VLC_CODEC_RGBA, 0 };
628#endif
629 uint32_t w, h;
630 VdpBool ok;
631
632 err = vdp_bitmap_surface_query_capabilities(sys->vdp, sys->device,
633 VDP_RGBA_FORMAT_R8G8B8A8, &ok, &w, &h);
634 if (err != VDP_STATUS_OK)
635 {
636 msg_Err(vd, "%s capabilities query failure: %s", "output surface",
637 vdp_get_error_string(sys->vdp, err));
638 ok = VDP_FALSE;
639 }
640 if (ok)
641 spu_chromas = subpicture_chromas;
642 }
643
644 /* Initialize VDPAU queue */
645 err = vdp_presentation_queue_target_create_x11(sys->vdp, sys->device,
646 sys->window, &sys->target);
647 if (err != VDP_STATUS_OK)
648 {
649 msg_Err(vd, "%s creation failure: %s", "presentation queue target",
650 vdp_get_error_string(sys->vdp, err));
651 goto error;
652 }
653
654 err = vdp_presentation_queue_create(sys->vdp, sys->device, sys->target,
655 &sys->queue);
656 if (err != VDP_STATUS_OK)
657 {
658 msg_Err(vd, "%s creation failure: %s", "presentation queue",
659 vdp_get_error_string(sys->vdp, err));
660 vdp_presentation_queue_target_destroy(sys->vdp, sys->target);
661 goto error;
662 }
663
664 VdpColor black = { 0.f, 0.f, 0.f, 1.f };
665 vdp_presentation_queue_set_background_color(sys->vdp, sys->queue, &black);
666
667 sys->cursor = XCB_cursor_Create(sys->conn, screen);
668 sys->pool = NULL;
669
670 /* */
671 vd->sys = sys;
672 vd->info.has_pictures_invalid = true;
673 vd->info.has_event_thread = true;
674 vd->info.subpicture_chromas = spu_chromas;
675 vd->fmt = fmt;
676
677 vd->pool = Pool;
678 vd->prepare = Queue;
679 vd->display = Wait;
680 vd->control = Control;
681 vd->manage = Manage;
682
683 /* */
684 bool is_fullscreen = vd->cfg->is_fullscreen;
685 if (is_fullscreen && vout_window_SetFullScreen(sys->embed, true))
686 is_fullscreen = false;
687 vout_display_SendEventFullscreen(vd, is_fullscreen);
688 vout_display_SendEventDisplaySize(vd, width, height, is_fullscreen);
689
690 return VLC_SUCCESS;
691
692error:
693 vdp_release_x11(sys->vdp);
694 xcb_disconnect(sys->conn);
695 vout_display_DeleteWindow(vd, sys->embed);
696 free(sys);
697 return VLC_EGENERIC;
698}

700static void Close(vlc_object_t *obj)
701{
702 vout_display_t *vd = (vout_display_t *)obj;
703 vout_display_sys_t *sys = vd->sys;
704
705 /* Restore cursor explicitly (parent window connection will survive) */
706 xcb_change_window_attributes(sys->conn, sys->embed->handle.xid,
707 XCB_CW_CURSOR, &(uint32_t) { XCB_CURSOR_NONE });
708 xcb_flush(sys->conn);
709
710 vdp_presentation_queue_destroy(sys->vdp, sys->queue);
711 vdp_presentation_queue_target_destroy(sys->vdp, sys->target);
712
713 if (sys->pool != NULL)
714 PoolFree(vd, sys->pool);
715
716 vdp_release_x11(sys->vdp);
717 xcb_disconnect(sys->conn);
718 vout_display_DeleteWindow(vd, sys->embed);
719 free(sys);
720}
