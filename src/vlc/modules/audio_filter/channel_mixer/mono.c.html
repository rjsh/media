119/*****************************************************************************
120 * Init: initialize internal data structures
121 * and computes the needed atomic operations
122 *****************************************************************************/
123/* x and z represent the coordinates of the virtual speaker
124 * relatively to the center of the listener's head, measured in meters :
125 *
126 * left right
127 *Z
128 *-
129 *a head
130 *x
131 *i
132 *s
133 * rear left rear right
134 *
135 * x-axis
136 * */
137static void ComputeChannelOperations( struct filter_sys_t * p_data,
138 unsigned int i_rate, unsigned int i_next_atomic_operation,
139 int i_source_channel_offset, double d_x, double d_z,
140 double d_compensation_length, double d_channel_amplitude_factor )
141{
142 double d_c = 340; /*sound celerity (unit: m/s)*/
143 double d_compensation_delay = (d_compensation_length-0.1) / d_c * i_rate;
144
145 /* Left ear */
146 p_data->p_atomic_operations[i_next_atomic_operation]
147 .i_source_channel_offset = i_source_channel_offset;
148 p_data->p_atomic_operations[i_next_atomic_operation]
149 .i_dest_channel_offset = 0;/* left */
150 p_data->p_atomic_operations[i_next_atomic_operation]
151 .i_delay = (int)( sqrt( (-0.1-d_x)*(-0.1-d_x) + (0-d_z)*(0-d_z) )
152 / d_c * i_rate - d_compensation_delay );
153 if( d_x < 0 )
154 {
155 p_data->p_atomic_operations[i_next_atomic_operation]
156 .d_amplitude_factor = d_channel_amplitude_factor * 1.1 / 2;
157 }
158 else if( d_x > 0 )
159 {
160 p_data->p_atomic_operations[i_next_atomic_operation]
161 .d_amplitude_factor = d_channel_amplitude_factor * 0.9 / 2;
162 }
163 else
164 {
165 p_data->p_atomic_operations[i_next_atomic_operation]
166 .d_amplitude_factor = d_channel_amplitude_factor / 2;
167 }
168
169 /* Right ear */
170 p_data->p_atomic_operations[i_next_atomic_operation + 1]
171 .i_source_channel_offset = i_source_channel_offset;
172 p_data->p_atomic_operations[i_next_atomic_operation + 1]
173 .i_dest_channel_offset = 1;/* right */
174 p_data->p_atomic_operations[i_next_atomic_operation + 1]
175 .i_delay = (int)( sqrt( (0.1-d_x)*(0.1-d_x) + (0-d_z)*(0-d_z) )
176 / d_c * i_rate - d_compensation_delay );
177 if( d_x < 0 )
178 {
179 p_data->p_atomic_operations[i_next_atomic_operation + 1]
180 .d_amplitude_factor = d_channel_amplitude_factor * 0.9 / 2;
181 }
182 else if( d_x > 0 )
183 {
184 p_data->p_atomic_operations[i_next_atomic_operation + 1]
185 .d_amplitude_factor = d_channel_amplitude_factor * 1.1 / 2;
186 }
187 else
188 {
189 p_data->p_atomic_operations[i_next_atomic_operation + 1]
190 .d_amplitude_factor = d_channel_amplitude_factor / 2;
191 }
192}

194static int Init( vlc_object_t *p_this, struct filter_sys_t * p_data,
195 unsigned int i_nb_channels, uint32_t i_physical_channels,
196 unsigned int i_rate )
197{
198 double d_x = var_InheritInteger( p_this, "headphone-dim" );
199 double d_z = d_x;
200 double d_z_rear = -d_x/3;
201 double d_min = 0;
202 unsigned int i_next_atomic_operation;
203 int i_source_channel_offset;
204 unsigned int i;
205
206 if( var_InheritBool( p_this, "headphone-compensate" ) )
207 {
208 /* minimal distance to any speaker */
209 if( i_physical_channels & AOUT_CHAN_REARCENTER )
210 {
211 d_min = d_z_rear;
212 }
213 else
214 {
215 d_min = d_z;
216 }
217 }
218
219 /* Number of elementary operations */
220 p_data->i_nb_atomic_operations = i_nb_channels * 2;
221 if( i_physical_channels & AOUT_CHAN_CENTER )
222 {
223 p_data->i_nb_atomic_operations += 2;
224 }
225 p_data->p_atomic_operations = malloc( sizeof(struct atomic_operation_t)
226 * p_data->i_nb_atomic_operations );
227 if( p_data->p_atomic_operations == NULL )
228 return -1;
229
230 /* For each virtual speaker, computes elementary wave propagation time
231 * to each ear */
232 i_next_atomic_operation = 0;
233 i_source_channel_offset = 0;
234 if( i_physical_channels & AOUT_CHAN_LEFT )
235 {
236 ComputeChannelOperations( p_data , i_rate
237 , i_next_atomic_operation , i_source_channel_offset
238 , -d_x , d_z , d_min , 2.0 / i_nb_channels );
239 i_next_atomic_operation += 2;
240 i_source_channel_offset++;
241 }
242 if( i_physical_channels & AOUT_CHAN_RIGHT )
243 {
244 ComputeChannelOperations( p_data , i_rate
245 , i_next_atomic_operation , i_source_channel_offset
246 , d_x , d_z , d_min , 2.0 / i_nb_channels );
247 i_next_atomic_operation += 2;
248 i_source_channel_offset++;
249 }
250 if( i_physical_channels & AOUT_CHAN_MIDDLELEFT )
251 {
252 ComputeChannelOperations( p_data , i_rate
253 , i_next_atomic_operation , i_source_channel_offset
254 , -d_x , 0 , d_min , 1.5 / i_nb_channels );
255 i_next_atomic_operation += 2;
256 i_source_channel_offset++;
257 }
258 if( i_physical_channels & AOUT_CHAN_MIDDLERIGHT )
259 {
260 ComputeChannelOperations( p_data , i_rate
261 , i_next_atomic_operation , i_source_channel_offset
262 , d_x , 0 , d_min , 1.5 / i_nb_channels );
263 i_next_atomic_operation += 2;
264 i_source_channel_offset++;
265 }
266 if( i_physical_channels & AOUT_CHAN_REARLEFT )
267 {
268 ComputeChannelOperations( p_data , i_rate
269 , i_next_atomic_operation , i_source_channel_offset
270 , -d_x , d_z_rear , d_min , 1.5 / i_nb_channels );
271 i_next_atomic_operation += 2;
272 i_source_channel_offset++;
273 }
274 if( i_physical_channels & AOUT_CHAN_REARRIGHT )
275 {
276 ComputeChannelOperations( p_data , i_rate
277 , i_next_atomic_operation , i_source_channel_offset
278 , d_x , d_z_rear , d_min , 1.5 / i_nb_channels );
279 i_next_atomic_operation += 2;
280 i_source_channel_offset++;
281 }
282 if( i_physical_channels & AOUT_CHAN_REARCENTER )
283 {
284 ComputeChannelOperations( p_data , i_rate
285 , i_next_atomic_operation , i_source_channel_offset
286 , 0 , -d_z , d_min , 1.5 / i_nb_channels );
287 i_next_atomic_operation += 2;
288 i_source_channel_offset++;
289 }
290 if( i_physical_channels & AOUT_CHAN_CENTER )
291 {
292 /* having two center channels increases the spatialization effect */
293 ComputeChannelOperations( p_data , i_rate
294 , i_next_atomic_operation , i_source_channel_offset
295 , d_x / 5.0 , d_z , d_min , 0.75 / i_nb_channels );
296 i_next_atomic_operation += 2;
297 ComputeChannelOperations( p_data , i_rate
298 , i_next_atomic_operation , i_source_channel_offset
299 , -d_x / 5.0 , d_z , d_min , 0.75 / i_nb_channels );
300 i_next_atomic_operation += 2;
301 i_source_channel_offset++;
302 }
303 if( i_physical_channels & AOUT_CHAN_LFE )
304 {
305 ComputeChannelOperations( p_data , i_rate
306 , i_next_atomic_operation , i_source_channel_offset
307 , 0 , d_z_rear , d_min , 5.0 / i_nb_channels );
308 i_next_atomic_operation += 2;
309 i_source_channel_offset++;
310 }
311
312 /* Initialize the overflow buffer
313 * we need it because the process induce a delay in the samples */
314 p_data->i_overflow_buffer_size = 0;
315 for( i = 0 ; i < p_data->i_nb_atomic_operations ; i++ )
316 {
317 if( p_data->i_overflow_buffer_size
318 < p_data->p_atomic_operations[i].i_delay * 2 * sizeof (int16_t) )
319 {
320 p_data->i_overflow_buffer_size
321 = p_data->p_atomic_operations[i].i_delay * 2 * sizeof (int16_t);
322 }
323 }
324 p_data->p_overflow_buffer = malloc( p_data->i_overflow_buffer_size );
325 if( p_data->p_overflow_buffer == NULL )
326 {
327 free( p_data->p_atomic_operations );
328 return -1;
329 }
330 memset( p_data->p_overflow_buffer, 0, p_data->i_overflow_buffer_size );
331
332 /* end */
333 return 0;
334}

336/*****************************************************************************
337 * OpenFilter
338 *****************************************************************************/
339static int OpenFilter( vlc_object_t *p_this )
340{
341 filter_t * p_filter = (filter_t *)p_this;
342 filter_sys_t *p_sys = NULL;
343
344 if( aout_FormatNbChannels( &(p_filter->fmt_in.audio) ) == 1 )
345 {
346 /*msg_Dbg( p_filter, "filter discarded (incompatible format)" );*/
347 return VLC_EGENERIC;
348 }
349
350 if( (p_filter->fmt_in.audio.i_format != p_filter->fmt_out.audio.i_format) ||
351 (p_filter->fmt_in.audio.i_format != VLC_CODEC_S16N) ||
352 (p_filter->fmt_out.audio.i_format != VLC_CODEC_S16N) )
353 {
354 /*msg_Err( p_this, "couldn't load mono filter" );*/
355 return VLC_EGENERIC;
356 }
357
358 /* Allocate the memory needed to store the module's structure */
359 p_sys = p_filter->p_sys = malloc( sizeof(filter_sys_t) );
360 if( p_sys == NULL )
361 return VLC_EGENERIC;
362
363 p_sys->b_downmix = var_InheritBool( p_this, MONO_CFG "downmix" );
364 p_sys->i_channel_selected = var_InheritInteger( p_this, MONO_CFG "channel" );
365
366 p_sys->i_nb_channels = aout_FormatNbChannels( &(p_filter->fmt_in.audio) );
367 p_sys->i_bitspersample = p_filter->fmt_out.audio.i_bitspersample;
368
369 p_sys->i_overflow_buffer_size = 0;
370 p_sys->p_overflow_buffer = NULL;
371 p_sys->i_nb_atomic_operations = 0;
372 p_sys->p_atomic_operations = NULL;
373
374 if( Init( VLC_OBJECT(p_filter), p_filter->p_sys,
375 aout_FormatNbChannels( &p_filter->fmt_in.audio ),
376 p_filter->fmt_in.audio.i_physical_channels,
377 p_filter->fmt_in.audio.i_rate ) < 0 )
378 {
379 free( p_sys );
380 return VLC_EGENERIC;
381 }
382
383 if( p_sys->b_downmix )
384 {
385 msg_Dbg( p_this, "using stereo to mono downmix" );
386 p_filter->fmt_out.audio.i_physical_channels = AOUT_CHAN_CENTER;
387 p_filter->fmt_out.audio.i_channels = 1;
388 }
389 else
390 {
391 msg_Dbg( p_this, "using pseudo mono" );
392 p_filter->fmt_out.audio.i_physical_channels = AOUT_CHANS_STEREO;
393 p_filter->fmt_out.audio.i_channels = 2;
394 }
395 p_filter->fmt_out.audio.i_rate = p_filter->fmt_in.audio.i_rate;
396 p_filter->pf_audio_filter = Convert;
397
398 msg_Dbg( p_this, "%4.4s->%4.4s, channels %d->%d, bits per sample: %i->%i",
399 (char *)&p_filter->fmt_in.i_codec,
400 (char *)&p_filter->fmt_out.i_codec,
401 p_filter->fmt_in.audio.i_physical_channels,
402 p_filter->fmt_out.audio.i_physical_channels,
403 p_filter->fmt_in.audio.i_bitspersample,
404 p_filter->fmt_out.audio.i_bitspersample );
405
406 return VLC_SUCCESS;
407}

422/*****************************************************************************
423 * Convert
424 *****************************************************************************/
425static block_t *Convert( filter_t *p_filter, block_t *p_block )
426{
427 block_t *p_out;
428 int i_out_size;
429
430 if( !p_block || !p_block->i_nb_samples )
431 {
432 if( p_block )
433 block_Release( p_block );
434 return NULL;
435 }
436
437 i_out_size = p_block->i_nb_samples * p_filter->p_sys->i_bitspersample/8 *
438 aout_FormatNbChannels( &(p_filter->fmt_out.audio) );
439
440 p_out = block_Alloc( i_out_size );
441 if( !p_out )
442 {
443 msg_Warn( p_filter, "can't get output buffer" );
444 block_Release( p_block );
445 return NULL;
446 }
447 p_out->i_nb_samples =
448 (p_block->i_nb_samples / p_filter->p_sys->i_nb_channels) *
449 aout_FormatNbChannels( &(p_filter->fmt_out.audio) );
450
451#if 0
452 unsigned int i_in_size = in_buf.i_nb_samples * (p_filter->p_sys->i_bitspersample/8) *
453 aout_FormatNbChannels( &(p_filter->fmt_in.audio) );
454 if( (in_buf.i_buffer != i_in_size) && ((i_in_size % 32) != 0) ) /* is it word aligned?? */
455 {
456 msg_Err( p_filter, "input buffer is not word aligned" );
457 /* Fix output buffer to be word aligned */
458 }
459#endif
460
461 memset( p_out->p_buffer, 0, i_out_size );
462 if( p_filter->p_sys->b_downmix )
463 {
464 stereo2mono_downmix( p_filter, p_block, p_out );
465 mono( p_filter, p_out, p_block );
466 }
467 else
468 {
469 stereo_to_mono( p_filter, p_out, p_block );
470 }
471
472 block_Release( p_block );
473 return p_out;
474}

476/* stereo2mono_downmix - stereo channels into one mono channel.
477 * Code taken from modules/audio_filter/channel_mixer/headphone.c
478 * converted from float into int16_t based downmix
479 * Written by Boris Dor√®s <babal@via.ecp.fr>
480 */
481static void stereo2mono_downmix( filter_t * p_filter,
482 block_t * p_in_buf, block_t * p_out_buf )
483{
484 filter_sys_t *p_sys = (filter_sys_t *)p_filter->p_sys;
485
486 int i_input_nb = aout_FormatNbChannels( &p_filter->fmt_in.audio );
487 int i_output_nb = aout_FormatNbChannels( &p_filter->fmt_out.audio );
488
489 int16_t * p_in = (int16_t*) p_in_buf->p_buffer;
490 uint8_t * p_out;
491 uint8_t * p_overflow;
492 uint8_t * p_slide;
493
494 size_t i_overflow_size; /* in bytes */
495 size_t i_out_size; /* in bytes */
496
497 unsigned int i, j;
498
499 int i_source_channel_offset;
500 int i_dest_channel_offset;
501 unsigned int i_delay;
502 double d_amplitude_factor;
503
504 /* out buffer characterisitcs */
505 p_out_buf->i_nb_samples = p_in_buf->i_nb_samples;
506 p_out_buf->i_buffer = p_in_buf->i_buffer * i_output_nb / i_input_nb;
507 p_out = p_out_buf->p_buffer;
508 i_out_size = p_out_buf->i_buffer;
509
510 /* Slide the overflow buffer */
511 p_overflow = p_sys->p_overflow_buffer;
512 i_overflow_size = p_sys->i_overflow_buffer_size;
513
514 if ( i_out_size > i_overflow_size )
515 memcpy( p_out, p_overflow, i_overflow_size );
516 else
517 memcpy( p_out, p_overflow, i_out_size );
518
519 p_slide = p_sys->p_overflow_buffer;
520 while( p_slide < p_overflow + i_overflow_size )
521 {
522 if( p_slide + i_out_size < p_overflow + i_overflow_size )
523 {
524 memset( p_slide, 0, i_out_size );
525 if( p_slide + 2 * i_out_size < p_overflow + i_overflow_size )
526 memcpy( p_slide, p_slide + i_out_size, i_out_size );
527 else
528 memcpy( p_slide, p_slide + i_out_size,
529 p_overflow + i_overflow_size - ( p_slide + i_out_size ) );
530 }
531 else
532 {
533 memset( p_slide, 0, p_overflow + i_overflow_size - p_slide );
534 }
535 p_slide += i_out_size;
536 }
537
538 /* apply the atomic operations */
539 for( i = 0; i < p_sys->i_nb_atomic_operations; i++ )
540 {
541 /* shorter variable names */
542 i_source_channel_offset
543 = p_sys->p_atomic_operations[i].i_source_channel_offset;
544 i_dest_channel_offset
545 = p_sys->p_atomic_operations[i].i_dest_channel_offset;
546 i_delay = p_sys->p_atomic_operations[i].i_delay;
547 d_amplitude_factor
548 = p_sys->p_atomic_operations[i].d_amplitude_factor;
549
550 if( p_out_buf->i_nb_samples > i_delay )
551 {
552 /* current buffer coefficients */
553 for( j = 0; j < p_out_buf->i_nb_samples - i_delay; j++ )
554 {
555 ((int16_t*)p_out)[ (i_delay+j)*i_output_nb + i_dest_channel_offset ]
556 += p_in[ j * i_input_nb + i_source_channel_offset ]
557 * d_amplitude_factor;
558 }
559
560 /* overflow buffer coefficients */
561 for( j = 0; j < i_delay; j++ )
562 {
563 ((int16_t*)p_overflow)[ j*i_output_nb + i_dest_channel_offset ]
564 += p_in[ (p_out_buf->i_nb_samples - i_delay + j)
565 * i_input_nb + i_source_channel_offset ]
566 * d_amplitude_factor;
567 }
568 }
569 else
570 {
571 /* overflow buffer coefficients only */
572 for( j = 0; j < p_out_buf->i_nb_samples; j++ )
573 {
574 ((int16_t*)p_overflow)[ (i_delay - p_out_buf->i_nb_samples + j)
575 * i_output_nb + i_dest_channel_offset ]
576 += p_in[ j * i_input_nb + i_source_channel_offset ]
577 * d_amplitude_factor;
578 }
579 }
580 }

583/* Simple stereo to mono mixing. */
584static unsigned int mono( filter_t *p_filter,
585 block_t *p_output, block_t *p_input )
586{
587 filter_sys_t *p_sys = (filter_sys_t *)p_filter->p_sys;
588 int16_t *p_in, *p_out;
589 unsigned int n = 0, r = 0;
590
591 p_in = (int16_t *) p_input->p_buffer;
592 p_out = (int16_t *) p_output->p_buffer;
593
594 while( n < (p_input->i_nb_samples * p_sys->i_nb_channels) )
595 {
596 p_out[r] = (p_in[n] + p_in[n+1]) >> 1;
597 r++;
598 n += 2;
599 }
600 return r;
601}

603/* Simple stereo to mono mixing. */
604static unsigned int stereo_to_mono( filter_t *p_filter,
605 block_t *p_output, block_t *p_input )
606{
607 filter_sys_t *p_sys = (filter_sys_t *)p_filter->p_sys;
608 int16_t *p_in, *p_out;
609 unsigned int n;
610
611 p_in = (int16_t *) p_input->p_buffer;
612 p_out = (int16_t *) p_output->p_buffer;
613
614 for( n = 0; n < (p_input->i_nb_samples * p_sys->i_nb_channels); n++ )
615 {
616 /* Fake real mono. */
617 if( p_sys->i_channel_selected == -1)
618 {
619 p_out[n] = p_out[n+1] = (p_in[n] + p_in[n+1]) >> 1;
620 n++;
