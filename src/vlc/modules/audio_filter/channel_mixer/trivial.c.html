55/*****************************************************************************
56 * Create: allocate trivial channel mixer
57 *****************************************************************************/
58static int Create( vlc_object_t *p_this )
59{
60 filter_t * p_filter = (filter_t *)p_this;
61
62 if ( (p_filter->fmt_in.audio.i_physical_channels
63 == p_filter->fmt_out.audio.i_physical_channels
64 && p_filter->fmt_in.audio.i_original_channels
65 == p_filter->fmt_out.audio.i_original_channels)
66 || p_filter->fmt_in.audio.i_format != p_filter->fmt_out.audio.i_format
67 || p_filter->fmt_in.audio.i_rate != p_filter->fmt_out.audio.i_rate
68 || p_filter->fmt_in.audio.i_format != VLC_CODEC_FL32 )
69 {
70 return VLC_EGENERIC;
71 }
72
73 p_filter->pf_audio_filter = DoWork;
74 return VLC_SUCCESS;
75}

77/*****************************************************************************
78 * SparseCopy: trivially downmix or upmix a buffer
79 *****************************************************************************/
80static void SparseCopy( float * p_dest, const float * p_src, size_t i_len,
81 int i_output_stride, int i_input_stride )
82{
83 int i;
84 for ( i = i_len; i--; )
85 {
86 int j;
87 for ( j = 0; j < i_output_stride; j++ )
88 {
89 p_dest[j] = p_src[j % i_input_stride];
90 }
91 p_src += i_input_stride;
92 p_dest += i_output_stride;
93 }
94}
95
96/*****************************************************************************
97 * DoWork: convert a buffer
98 *****************************************************************************/
99static block_t *DoWork( filter_t * p_filter, block_t * p_in_buf )
100{
101 int i_input_nb = aout_FormatNbChannels( &p_filter->fmt_in.audio );
102 int i_output_nb = aout_FormatNbChannels( &p_filter->fmt_out.audio );
103
104 block_t *p_out_buf;
105 if( i_input_nb >= i_output_nb )
106 {
107 p_out_buf = p_in_buf; /* mix in place */
108 p_out_buf->i_buffer = p_in_buf->i_buffer * i_output_nb / i_input_nb;
109 }
110 else
111 {
112 p_out_buf = block_Alloc(
113 p_in_buf->i_buffer * i_output_nb / i_input_nb );
114 if( !p_out_buf )
115 goto out;
116 /* on upmixing case, zero out buffer */
117 memset( p_out_buf->p_buffer, 0, p_out_buf->i_buffer );
118 p_out_buf->i_nb_samples = p_in_buf->i_nb_samples;
119 p_out_buf->i_dts = p_in_buf->i_dts;
120 p_out_buf->i_pts = p_in_buf->i_pts;
121 p_out_buf->i_length = p_in_buf->i_length;
122 }
123
124 float * p_dest = (float *)p_out_buf->p_buffer;
125 const float * p_src = (float *)p_in_buf->p_buffer;
126 const bool b_reverse_stereo = p_filter->fmt_out.audio.i_original_channels & AOUT_CHAN_REVERSESTEREO;
127 bool b_dualmono2stereo = (p_filter->fmt_in.audio.i_original_channels & AOUT_CHAN_DUALMONO );
128 b_dualmono2stereo &= (p_filter->fmt_out.audio.i_physical_channels & ( AOUT_CHAN_LEFT | AOUT_CHAN_RIGHT ));
129 b_dualmono2stereo &= ((p_filter->fmt_out.audio.i_physical_channels & AOUT_CHAN_PHYSMASK) != (p_filter->fmt_in.audio.i_physical_channels & AOUT_CHAN_PHYSMASK));
130
131 if( likely( !b_reverse_stereo && ! b_dualmono2stereo ) )
132 {
133 SparseCopy( p_dest, p_src, p_in_buf->i_nb_samples, i_output_nb,
134 i_input_nb );
135 }
136 /* Special case from dual mono to stereo */
137 else if ( b_dualmono2stereo )
138 {
139 int i;
140 /* This is a bit special. */
141 if ( !(p_filter->fmt_out.audio.i_original_channels & AOUT_CHAN_LEFT) )
142 {
143 p_src++;
144 }
145 if ( p_filter->fmt_out.audio.i_physical_channels == AOUT_CHAN_CENTER )
146 {
147 /* Mono mode */
148 for ( i = p_in_buf->i_nb_samples; i--; )
149 {
150 *p_dest = *p_src;
151 p_dest++;
152 p_src += 2;
153 }
154 }
155 else
156 {
157 /* Fake-stereo mode */
158 for ( i = p_in_buf->i_nb_samples; i--; )
159 {
160 *p_dest = *p_src;
161 p_dest++;
162 *p_dest = *p_src;
163 p_dest++;
164 p_src += 2;
165 }
166 }
167 }
168 else if ( b_reverse_stereo )
169 {
170 /* Reverse-stereo mode */
171 int i;
172 for ( i = p_in_buf->i_nb_samples; i--; )
173 {
174 float i_tmp = p_src[0];
175 p_dest[0] = p_src[1];
176 p_dest[1] = i_tmp;
177
178 p_dest += 2;
179 p_src += 2;
180 }
181 }
182out:
183 if( p_in_buf != p_out_buf )
184 block_Release( p_in_buf );
185 return p_out_buf;
186}
