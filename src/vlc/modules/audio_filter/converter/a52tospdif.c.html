56/*****************************************************************************
57 * Create:
58 *****************************************************************************/
59static int Create( vlc_object_t *p_this )
60{
61 filter_t * p_filter = (filter_t *)p_this;
62
63 if ( p_filter->fmt_in.audio.i_format != VLC_CODEC_A52 ||
64 ( p_filter->fmt_out.audio.i_format != VLC_CODEC_SPDIFB &&
65 p_filter->fmt_out.audio.i_format != VLC_CODEC_SPDIFL ) )
66 {
67 return VLC_EGENERIC;
68 }
69
70 p_filter->pf_audio_filter = DoWork;
71
72 return VLC_SUCCESS;
73}
74
75/*****************************************************************************
76 * DoWork: convert a buffer
77 *****************************************************************************/
78static block_t *DoWork( filter_t * p_filter, block_t *p_in_buf )
79{
80 /* AC3 is natively big endian. Most SPDIF devices have the native
81 * endianness of the computer system.
82 * On Mac OS X however, little endian devices are also common.
83 */
84 static const uint8_t p_sync_le[6] = { 0x72, 0xF8, 0x1F, 0x4E, 0x01, 0x00 };
85 static const uint8_t p_sync_be[6] = { 0xF8, 0x72, 0x4E, 0x1F, 0x00, 0x01 };
86 uint16_t i_frame_size = p_in_buf->i_buffer / 2;
87 uint8_t * p_in = p_in_buf->p_buffer;
88
89 block_t *p_out_buf = block_Alloc( AOUT_SPDIF_SIZE );
90 if( !p_out_buf )
91 goto out;
92 uint8_t * p_out = p_out_buf->p_buffer;
93
94 /* Copy the S/PDIF headers. */
95 if( p_filter->fmt_out.audio.i_format == VLC_CODEC_SPDIFB )
96 {
97 memcpy( p_out, p_sync_be, 6 );
98 p_out[4] = p_in[5] & 0x7; /* bsmod */
99 SetWBE( p_out + 6, i_frame_size << 4 );
100 memcpy( &p_out[8], p_in, i_frame_size * 2 );
101 }
102 else
103 {
104 memcpy( p_out, p_sync_le, 6 );
105 p_out[5] = p_in[5] & 0x7; /* bsmod */
106 SetWLE( p_out + 6, i_frame_size << 4 );
107 swab( p_in, &p_out[8], i_frame_size * 2 );
108 }
109 memset( p_out + 8 + i_frame_size * 2, 0,
110 AOUT_SPDIF_SIZE - i_frame_size * 2 - 8 );
111
112 p_out_buf->i_dts = p_in_buf->i_dts;
113 p_out_buf->i_pts = p_in_buf->i_pts;
114 p_out_buf->i_nb_samples = p_in_buf->i_nb_samples;
115 p_out_buf->i_buffer = AOUT_SPDIF_SIZE;
116out:
117 block_Release( p_in_buf );
118 return p_out_buf;
119}
