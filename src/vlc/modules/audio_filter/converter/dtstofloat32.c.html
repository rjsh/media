58/* libdca channel order
59 * libdca currently only decodes 5.1, even if you have a DTS-ES source. */
60static const uint32_t pi_channels_in[] =
61{ AOUT_CHAN_CENTER, AOUT_CHAN_LEFT, AOUT_CHAN_RIGHT,
62 AOUT_CHAN_REARCENTER, AOUT_CHAN_REARLEFT, AOUT_CHAN_REARRIGHT,
63 AOUT_CHAN_LFE,
64 0 };

66/*****************************************************************************
67 * Local structures
68 *****************************************************************************/
69struct filter_sys_t
70{
71 dca_state_t * p_libdca; /* libdca internal structure */
72 bool b_dynrng; /* see below */
73 int i_flags; /* libdca flags, see dtsdec/doc/libdts.txt */
74 bool b_dontwarn;
75 int i_nb_channels; /* number of float32 per sample */
76
77 uint8_t pi_chan_table[AOUT_CHAN_MAX]; /* channel reordering */
78};

101/*****************************************************************************
102 * Open:
103 *****************************************************************************/
104static int Open( vlc_object_t *p_this, filter_sys_t *p_sys,
105 const audio_format_t *restrict input,
106 const audio_format_t *restrict output )
107{
108 p_sys->b_dynrng = var_InheritBool( p_this, "dts-dynrng" );
109 p_sys->b_dontwarn = 0;
110
111 /* We'll do our own downmixing, thanks. */
112 p_sys->i_nb_channels = aout_FormatNbChannels( output );
113 switch ( output->i_physical_channels & ~AOUT_CHAN_LFE )
114 {
115 case AOUT_CHAN_CENTER:
116 if ( (output->i_original_channels & AOUT_CHAN_CENTER)
117 || (output->i_original_channels
118 & (AOUT_CHAN_LEFT | AOUT_CHAN_RIGHT)) )
119 {
120 p_sys->i_flags = DCA_MONO;
121 }
122 break;
123
124 case AOUT_CHAN_LEFT | AOUT_CHAN_RIGHT:
125 if ( output->i_original_channels & AOUT_CHAN_DOLBYSTEREO )
126 {
127 p_sys->i_flags = DCA_DOLBY;
128 }
129 else if ( input->i_original_channels == AOUT_CHAN_CENTER )
130 {
131 p_sys->i_flags = DCA_MONO;
132 }
133 else if ( input->i_original_channels & AOUT_CHAN_DUALMONO )
134 {
135 p_sys->i_flags = DCA_CHANNEL;
136 }
137 else
138 {
139 p_sys->i_flags = DCA_STEREO;
140 }
141 break;
142
143 case AOUT_CHAN_LEFT | AOUT_CHAN_RIGHT | AOUT_CHAN_CENTER:
144 p_sys->i_flags = DCA_3F;
145 break;
146
147 case AOUT_CHAN_LEFT | AOUT_CHAN_RIGHT | AOUT_CHAN_REARCENTER:
148 p_sys->i_flags = DCA_2F1R;
149 break;
150
151 case AOUT_CHAN_LEFT | AOUT_CHAN_RIGHT | AOUT_CHAN_CENTER
152 | AOUT_CHAN_REARCENTER:
153 p_sys->i_flags = DCA_3F1R;
154 break;
155
156 case AOUT_CHAN_LEFT | AOUT_CHAN_RIGHT
157 | AOUT_CHAN_REARLEFT | AOUT_CHAN_REARRIGHT:
158 p_sys->i_flags = DCA_2F2R;
159 break;
160
161 case AOUT_CHAN_LEFT | AOUT_CHAN_RIGHT | AOUT_CHAN_CENTER
162 | AOUT_CHAN_REARLEFT | AOUT_CHAN_REARRIGHT:
163 p_sys->i_flags = DCA_3F2R;
164 break;
165
166 default:
167 msg_Warn( p_this, "unknown sample format!" );
168 free( p_sys );
169 return -1;
170 }
171 if ( output->i_physical_channels & AOUT_CHAN_LFE )
172 {
173 p_sys->i_flags |= DCA_LFE;
174 }
175 //p_sys->i_flags |= DCA_ADJUST_LEVEL;
176
177 /* Initialize libdca */
178 p_sys->p_libdca = dca_init( 0 );
179 if( p_sys->p_libdca == NULL )
180 {
181 msg_Err( p_this, "unable to initialize libdca" );
182 return VLC_EGENERIC;
183 }
184
185 aout_CheckChannelReorder( pi_channels_in, NULL,
186 output->i_physical_channels,
187 p_sys->pi_chan_table );
188
189 return VLC_SUCCESS;
190}

192/*****************************************************************************
193 * Interleave: helper function to interleave channels
194 *****************************************************************************/
195static void Interleave( float * p_out, const float * p_in, int i_nb_channels,
196 uint8_t *pi_chan_table )
197{
198 /* We do not only have to interleave, but also reorder the channels. */
199
200 int i, j;
201 for ( j = 0; j < i_nb_channels; j++ )
202 {
203 for ( i = 0; i < 256; i++ )
204 {
205 p_out[i * i_nb_channels + pi_chan_table[j]] = p_in[j * 256 + i];
206 }
207 }
208}
209
210/*****************************************************************************
211 * Duplicate: helper function to duplicate a unique channel
212 *****************************************************************************/
213static void Duplicate( float * p_out, const float * p_in )
214{
215 int i;
216
217 for ( i = 256; i--; )
218 {
219 *p_out++ = *p_in;
220 *p_out++ = *p_in;
221 p_in++;
222 }
223}
224
225/*****************************************************************************
226 * Exchange: helper function to exchange left & right channels
227 *****************************************************************************/
228static void Exchange( float * p_out, const float * p_in )
229{
230 int i;
231 const float * p_first = p_in + 256;
232 const float * p_second = p_in;
233
234 for ( i = 0; i < 256; i++ )
235 {
236 *p_out++ = *p_first++;
237 *p_out++ = *p_second++;
238 }
239}

241/*****************************************************************************
242 * Convert: decode a DTS frame.
243 *****************************************************************************/
244static block_t *Convert( filter_t *p_filter, block_t *p_in_buf )
245{
246 filter_sys_t *p_sys = p_filter->p_sys;
247 sample_t i_sample_level = 1;
248 int i_flags = p_sys->i_flags;
249 size_t i_bytes_per_block = 256 * p_sys->i_nb_channels
250 * sizeof(float);
251
252 block_t *p_out_buf = block_Alloc( 6 * i_bytes_per_block );
253 if( unlikely(p_out_buf == NULL) )
254 goto out;
255
256 /*
257 * Do the actual decoding now.
258 */
259
260 /* Needs to be called so the decoder knows which type of bitstream it is
261 * dealing with. */
262 int i_sample_rate, i_bit_rate, i_frame_length;
263 if( !dca_syncinfo( p_sys->p_libdca, p_in_buf->p_buffer, &i_flags,
264 &i_sample_rate, &i_bit_rate, &i_frame_length ) )
265 {
266 msg_Warn( p_filter, "libdca couldn't sync on frame" );
267 p_out_buf->i_nb_samples = p_out_buf->i_buffer = 0;
268 goto out;
269 }
270
271 i_flags = p_sys->i_flags;
272 dca_frame( p_sys->p_libdca, p_in_buf->p_buffer,
273 &i_flags, &i_sample_level, 0 );
274
275 if ( (i_flags & DCA_CHANNEL_MASK) != (p_sys->i_flags & DCA_CHANNEL_MASK)
276 && !p_sys->b_dontwarn )
277 {
278 msg_Warn( p_filter,
279 "libdca couldn't do the requested downmix 0x%x->0x%x",
280 p_sys->i_flags & DCA_CHANNEL_MASK,
281 i_flags & DCA_CHANNEL_MASK );
282
283 p_sys->b_dontwarn = 1;
284 }
285
286 if( 0)//!p_sys->b_dynrng )
287 {
288 dca_dynrng( p_sys->p_libdca, NULL, NULL );
289 }
290
291 for( int i = 0; i < dca_blocks_num(p_sys->p_libdca); i++ )
292 {
293 sample_t * p_samples;
294
295 if( dca_block( p_sys->p_libdca ) )
296 {
297 msg_Warn( p_filter, "dca_block failed for block %d", i );
298 break;
299 }
300
301 p_samples = dca_samples( p_sys->p_libdca );
302
303 if ( (p_sys->i_flags & DCA_CHANNEL_MASK) == DCA_MONO
304 && (p_filter->fmt_out.audio.i_physical_channels
305 & (AOUT_CHAN_LEFT | AOUT_CHAN_RIGHT)) )
306 {
307 Duplicate( (float *)(p_out_buf->p_buffer + i * i_bytes_per_block),
308 p_samples );
309 }
310 else if ( p_filter->fmt_out.audio.i_original_channels
311 & AOUT_CHAN_REVERSESTEREO )
312 {
313 Exchange( (float *)(p_out_buf->p_buffer + i * i_bytes_per_block),
314 p_samples );
315 }
316 else
317 {
318 /* Interleave the *$% samples. */
319 Interleave( (float *)(p_out_buf->p_buffer + i * i_bytes_per_block),
320 p_samples, p_sys->i_nb_channels, p_sys->pi_chan_table);
321 }
322 }
323
324 p_out_buf->i_buffer = p_in_buf->i_nb_samples * 4 * p_sys->i_nb_channels;
325 p_out_buf->i_nb_samples = p_in_buf->i_nb_samples;
326 p_out_buf->i_dts = p_in_buf->i_dts;
327 p_out_buf->i_pts = p_in_buf->i_pts;
328 p_out_buf->i_length = p_in_buf->i_length;
329out:
330 block_Release( p_in_buf );
331 return p_out_buf;
332}

334/*****************************************************************************
335 * OpenFilter:
336 *****************************************************************************/
337static int OpenFilter( vlc_object_t *p_this )
338{
339 filter_t *p_filter = (filter_t *)p_this;
340 filter_sys_t *p_sys;
341 int i_ret;
342
343 if( p_filter->fmt_in.i_codec != VLC_CODEC_DTS
344 || p_filter->fmt_out.audio.i_format != VLC_CODEC_FL32 )
345 return VLC_EGENERIC;
346
347 /* Allocate the memory needed to store the module's structure */
348 p_sys = p_filter->p_sys = malloc( sizeof(filter_sys_t) );
349 if( p_sys == NULL )
350 return VLC_ENOMEM;
351
352 i_ret = Open( VLC_OBJECT(p_filter), p_sys,
353 &p_filter->fmt_in.audio, &p_filter->fmt_out.audio );
354
355 p_filter->pf_audio_filter = Convert;
356 p_filter->fmt_out.audio.i_rate = p_filter->fmt_in.audio.i_rate;
357
358 return i_ret;
359}

361/*****************************************************************************
362 * CloseFilter : deallocate data structures
363 *****************************************************************************/
364static void CloseFilter( vlc_object_t *p_this )
365{
366 filter_t *p_filter = (filter_t *)p_this;
367 filter_sys_t *p_sys = p_filter->p_sys;
368
369 dca_free( p_sys->p_libdca );
370 free( p_sys );
371}
