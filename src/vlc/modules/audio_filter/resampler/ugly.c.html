50 set_capability( "audio converter", 2 )
51 set_category( CAT_AUDIO )
52 set_subcategory( SUBCAT_AUDIO_MISC )

57 set_callbacks( CreateResampler, NULL )

60/*****************************************************************************
61 * Create: allocate ugly resampler
62 *****************************************************************************/
63static int Create( vlc_object_t *p_this )
64{
65 filter_t * p_filter = (filter_t *)p_this;
66
67 if( p_filter->fmt_in.audio.i_rate == p_filter->fmt_out.audio.i_rate )
68 return VLC_EGENERIC;
69 return CreateResampler( p_this );
70}
71
72static int CreateResampler( vlc_object_t *p_this )
73{
74 filter_t * p_filter = (filter_t *)p_this;
75
76 if( p_filter->fmt_in.audio.i_format != p_filter->fmt_out.audio.i_format
77 || p_filter->fmt_in.audio.i_physical_channels
78 != p_filter->fmt_out.audio.i_physical_channels
79 || p_filter->fmt_in.audio.i_original_channels
80 != p_filter->fmt_out.audio.i_original_channels
81 || !AOUT_FMT_LINEAR( &p_filter->fmt_in.audio ) )
82 return VLC_EGENERIC;
83
84 p_filter->pf_audio_filter = DoWork;
85 return VLC_SUCCESS;
86}
87
88/*****************************************************************************
89 * DoWork: convert a buffer
90 *****************************************************************************/
91static block_t *DoWork( filter_t * p_filter, block_t * p_in_buf )
92{
93 /* Check if we really need to run the resampler */
94 if( p_filter->fmt_out.audio.i_rate == p_filter->fmt_in.audio.i_rate )
95 return p_in_buf;
96
97 block_t *p_out_buf = p_in_buf;
98 unsigned int i_out_nb = p_in_buf->i_nb_samples
99 * p_filter->fmt_out.audio.i_rate / p_filter->fmt_in.audio.i_rate;
100 const unsigned framesize = (p_filter->fmt_in.audio.i_bitspersample / 8)
101 * aout_FormatNbChannels( &p_filter->fmt_in.audio );
102
103 if( p_filter->fmt_out.audio.i_rate > p_filter->fmt_in.audio.i_rate )
104 {
105 p_out_buf = block_Alloc( i_out_nb * framesize );
106 if( !p_out_buf )
107 goto out;
108 }
109
110 unsigned char *p_out = p_out_buf->p_buffer;
111 unsigned char *p_in = p_in_buf->p_buffer;
112 unsigned int i_remainder = 0;
113
114 p_out_buf->i_nb_samples = i_out_nb;
115 p_out_buf->i_buffer = i_out_nb * framesize;
116 p_out_buf->i_pts = p_in_buf->i_pts;
117 p_out_buf->i_length = p_out_buf->i_nb_samples *
118 1000000 / p_filter->fmt_out.audio.i_rate;
119
120 while( i_out_nb )
121 {
122 if( p_out != p_in )
123 memcpy( p_out, p_in, framesize );
124 p_out += framesize;
125 i_out_nb--;
126
127 i_remainder += p_filter->fmt_in.audio.i_rate;
128 while( i_remainder >= p_filter->fmt_out.audio.i_rate )
129 {
130 p_in += framesize;
131 i_remainder -= p_filter->fmt_out.audio.i_rate;
132 }
133 }
134
135 if( p_in_buf != p_out_buf )
136out:
137 block_Release( p_in_buf );
138 return p_out_buf;
139}
