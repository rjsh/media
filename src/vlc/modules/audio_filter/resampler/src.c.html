41#define SRC_CONV_TYPE_LONGTEXT N_( \
42 "Different resampling algorithms are supported. " \
43 "The best one is slower, while the fast one exhibits low quality.")
44static const int conv_type_values[] = {
45 SRC_SINC_BEST_QUALITY, SRC_SINC_MEDIUM_QUALITY, SRC_SINC_FASTEST,
46 SRC_ZERO_ORDER_HOLD, SRC_LINEAR,
47};
48static const char *const conv_type_texts[] = {
49 N_("Sinc function (best quality)"), N_("Sinc function (medium quality)"),
50 N_("Sinc function (fast)"), N_("Zero Order Hold (fastest)"), N_("Linear (fastest)"),
51};

73static block_t *Resample (filter_t *, block_t *);
74
75static int Open (vlc_object_t *obj)
76{
77 filter_t *filter = (filter_t *)obj;
78
79 /* Will change rate */
80 if (filter->fmt_in.audio.i_rate == filter->fmt_out.audio.i_rate)
81 return VLC_EGENERIC;
82 return OpenResampler (obj);
83}
84
85static int OpenResampler (vlc_object_t *obj)
86{
87 filter_t *filter = (filter_t *)obj;
88
89 /* Only float->float */
90 if (filter->fmt_in.audio.i_format != VLC_CODEC_FL32
91 || filter->fmt_out.audio.i_format != VLC_CODEC_FL32
92 /* No channels remapping */
93 || filter->fmt_in.audio.i_physical_channels
94 != filter->fmt_out.audio.i_physical_channels
95 || filter->fmt_in.audio.i_original_channels
96 != filter->fmt_out.audio.i_original_channels)
97 return VLC_EGENERIC;
98
99 int type = var_InheritInteger (obj, "src-converter-type");
100 int channels = aout_FormatNbChannels (&filter->fmt_in.audio);
101 int err;
102
103 SRC_STATE *s = src_new (type, channels, &err);
104 if (s == NULL)
105 {
106 msg_Err (obj, "cannot initialize resampler: %s", src_strerror (err));
107 return VLC_EGENERIC;
108 }
109
110 filter->p_sys = (filter_sys_t *)s;
111 filter->pf_audio_filter = Resample;
112 return VLC_SUCCESS;
113}
114
115static void Close (vlc_object_t *obj)
116{
117 filter_t *filter = (filter_t *)obj;
118 SRC_STATE *s = (SRC_STATE *)filter->p_sys;
119
120 src_delete (s);
121}
122
123static block_t *Resample (filter_t *filter, block_t *in)
124{
125 block_t *out = NULL;
126 const size_t framesize = filter->fmt_out.audio.i_bytes_per_frame;
127
128 SRC_STATE *s = (SRC_STATE *)filter->p_sys;
129 SRC_DATA src;
130
131 src.src_ratio = (double)filter->fmt_out.audio.i_rate
132 / (double)filter->fmt_in.audio.i_rate;
133
134 int err = src_set_ratio (s, src.src_ratio);
135 if (err != 0)
136 {
137 msg_Err (filter, "cannot update resampling ratio: %s",
138 src_strerror (err));
139 goto error;
140 }
141
142 src.input_frames = in->i_nb_samples;
143 src.output_frames = ceil (src.src_ratio * src.input_frames);
144 src.end_of_input = 0;
145
146 out = block_Alloc (src.output_frames * framesize);
147 if (unlikely(out == NULL))
148 goto error;
149
150 src.data_in = (float *)in->p_buffer;
151 src.data_out = (float *)out->p_buffer;
152
153 err = src_process (s, &src);
154 if (err != 0)
155 {
156 msg_Err (filter, "cannot resample: %s", src_strerror (err));
157 block_Release (out);
158 out = NULL;
159 goto error;
160 }
161
162 if (src.input_frames_used < src.input_frames)
163 msg_Err (filter, "lost %ld of %ld input frames",
164 src.input_frames - src.input_frames_used, src.input_frames);
165
166 out->i_buffer = src.output_frames_gen * framesize;
167 out->i_nb_samples = src.output_frames_gen;
168 out->i_pts = in->i_pts;
169 out->i_length = src.output_frames_gen * CLOCK_FREQ
170 / filter->fmt_out.audio.i_rate;
171error:
172 block_Release (in);
173 return out;
174}
