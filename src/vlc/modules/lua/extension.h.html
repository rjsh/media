31/* List of available commands */
32typedef enum
33{
34 CMD_ACTIVATE = 1,
35 CMD_DEACTIVATE,
36 CMD_TRIGGERMENU, /* Arg1 = int*, pointing to id to trigger. free */
37 CMD_CLICK, /* Arg1 = extension_widget_t* */
38 CMD_CLOSE,
39 CMD_SET_INPUT, /* No arg. Just signal current input changed */
40 CMD_UPDATE_META, /* No arg. Just signal current input item meta changed */
41 CMD_PLAYING_CHANGED /* Arg1 = int*, New playing status */
42} command_type_e;
43
44//Data types
45typedef enum
46{
47 LUA_END = 0,
48 LUA_NUM,
49 LUA_TEXT
50} lua_datatype_e;
51
52struct extensions_manager_sys_t
53{
54 /* List of activated extensions */
55 DECL_ARRAY( extension_t* ) activated_extensions;
56
57 /* Lock for this list */
58 vlc_mutex_t lock;
59
60 /* Flag indicating that the module is about to be unloaded */
61 bool b_killed;
62};
63
64struct extension_sys_t
65{
66 /* Extension general */
67 int i_capabilities;
68
69 /* Lua specific */
70 lua_State *L;
71
72 vlclua_dtable_t dtable;
73
74 /* Thread data */
75 vlc_thread_t thread;
76 vlc_mutex_t command_lock;
77 vlc_mutex_t running_lock;
78 vlc_cond_t wait;
79
80 /* The input this extension should use for vlc.input
81 * or NULL if it should use playlist's current input */
82 struct input_thread_t *p_input;
83
84 extensions_manager_t *p_mgr; ///< Parent
85 /* Queue of commands to execute */
86 struct command_t
87 {
88 command_type_e i_command;
89 void *data[10]; ///< Optional void* arguments
90 struct command_t *next; ///< Next command
91 } *command;
92
93 // The two following booleans are protected by command_lock
94 dialog_progress_bar_t *progress;
95 vlc_timer_t timer; ///< This timer makes sure Lua never gets stuck >5s
96
97 bool b_exiting;
98};

100/* Extensions: manager functions */
101int Activate( extensions_manager_t *p_mgr, extension_t * );
102bool IsActivated( extensions_manager_t *p_mgr, extension_t * );
103int Deactivate( extensions_manager_t *p_mgr, extension_t * );
104void KillExtension( extensions_manager_t *p_mgr, extension_t *p_ext );
105int __PushCommand( extension_t *ext, bool unique, command_type_e cmd, va_list options );
106static inline int PushCommand( extension_t *ext, int cmd, ... )
107{
108 va_list args;
109 va_start( args, cmd );
110 int i_ret = __PushCommand( ext, false, cmd, args );
111 va_end( args );
112 return i_ret;
113}
114static inline int PushCommandUnique( extension_t *ext, int cmd, ... )
115{
116 va_list args;
117 va_start( args, cmd );
118 int i_ret = __PushCommand( ext, true, cmd, args );
119 va_end( args );
120 return i_ret;
121}
122bool LockExtension( extension_t *p_ext );
123void UnlockExtension( extension_t *p_ext );

130int lua_ExecuteFunctionVa( extensions_manager_t *p_mgr, extension_t *p_ext,

132int lua_ExecuteFunction( extensions_manager_t *p_mgr, extension_t *p_ext,
