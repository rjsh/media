64 input_item_t **pp_node = (input_item_t **)luaL_checkudata( L, 1, "input_item_t" );\

66 {\
67 if( lua_isstring( L, -1 ) )\
68 {\
69 input_item_Set ## normal ( *pp_node, lua_tostring( L, -1 ) );\
70 } else\
71 msg_Err( p_sd, "Error parsing set_ " # lowercase " arguments" );\
72 }\
73 return 1;\
74}

127static int vlclua_sd_get_services_names( lua_State *L )
128{
129 playlist_t *p_playlist = vlclua_get_playlist_internal( L );
130 char **ppsz_longnames;
131 char **ppsz_names = vlc_sd_GetNames( p_playlist, &ppsz_longnames, NULL );
132 if( !ppsz_names )
133 return 0;
134
135 char **ppsz_longname = ppsz_longnames;
136 char **ppsz_name = ppsz_names;
137 lua_settop( L, 0 );
138 lua_newtable( L );
139 for( ; *ppsz_name; ppsz_name++,ppsz_longname++ )
140 {
141 lua_pushstring( L, *ppsz_longname );
142 lua_setfield( L, -2, *ppsz_name );
143 free( *ppsz_name );
144 free( *ppsz_longname );
145 }
146 free( ppsz_names );
147 free( ppsz_longnames );
148 return 1;
149}

175static int vlclua_sd_add_node( lua_State *L )
176{
177 services_discovery_t *p_sd = (services_discovery_t *)vlclua_get_this( L );
178 if( lua_istable( L, -1 ) )
179 {
180 lua_getfield( L, -1, "title" );
181 if( lua_isstring( L, -1 ) )
182 {
183 const char *psz_name = lua_tostring( L, -1 );
184 input_item_t *p_input = input_item_NewWithType( "vlc://nop",
185 psz_name, 0, NULL, 0,
186 -1, ITEM_TYPE_NODE );
187 lua_pop( L, 1 );
188
189 if( p_input )
190 {
191 lua_getfield( L, -1, "arturl" );
192 if( lua_isstring( L, -1 ) && strcmp( lua_tostring( L, -1 ), "" ) )
193 {
194 char *psz_value = strdup( lua_tostring( L, -1 ) );
195 EnsureUTF8( psz_value );
196 msg_Dbg( p_sd, "ArtURL: %s", psz_value );
197 /** @todo Ask for art download if not local file */
198 input_item_SetArtURL( p_input, psz_value );
199 free( psz_value );
200 }
201 lua_pop( L, 1 );
202 lua_getfield( L, -1, "category" );
203 if( lua_isstring( L, -1 ) )
204 services_discovery_AddItem( p_sd, p_input, luaL_checkstring( L, -1 ) );
205 else
206 services_discovery_AddItem( p_sd, p_input, NULL );
207 input_item_t **udata = (input_item_t **)
208 lua_newuserdata( L, sizeof( input_item_t * ) );
209 *udata = p_input;
210 if( luaL_newmetatable( L, "node" ) )
211 {
212 lua_newtable( L );
213 luaL_register( L, NULL, vlclua_node_reg );
214 lua_setfield( L, -2, "__index" );
215 }
216 lua_setmetatable( L, -2 );
217 }
218 }
219 else
220 msg_Err( p_sd, "vlc.sd.add_node: the \"title\" parameter can't be empty" );
221 }
222 else
223 msg_Err( p_sd, "Error parsing add_node arguments" );
224 return 1;
225}

227static int vlclua_sd_add_item( lua_State *L )
228{
229 services_discovery_t *p_sd = (services_discovery_t *)vlclua_get_this( L );
230 if( lua_istable( L, -1 ) )
231 {
232 lua_getfield( L, -1, "path" );
233 if( lua_isstring( L, -1 ) )
234 {
235 const char *psz_path = lua_tostring( L, -1 );
236
237 lua_getfield( L, -2, "title" );
238 const char *psz_title = luaL_checkstring( L, -1 ) ? luaL_checkstring( L, -1 ) : psz_path;
239
240 /* The table must be at the top of the stack when calling
241 * vlclua_read_options() */
242 char **ppsz_options = NULL;
243 int i_options = 0;
244 lua_pushvalue( L, -3 );
245 vlclua_read_options( p_sd, L, &i_options, &ppsz_options );
246
247 input_item_t *p_input = input_item_NewExt( psz_path, psz_title,
248 i_options,
249 (const char **)ppsz_options,
250 VLC_INPUT_OPTION_TRUSTED, -1 );
251 lua_pop( L, 3 );
252
253 if( p_input )
254 {
255 vlclua_read_meta_data( p_sd, L, p_input );
256 /* This one is to be tested... */
257 vlclua_read_custom_meta_data( p_sd, L, p_input );
258 /* The duration is given in seconds, convert to microseconds */
259 lua_getfield( L, -1, "duration" );
260 if( lua_isnumber( L, -1 ) )
261 input_item_SetDuration( p_input, (lua_tonumber( L, -1 )*1e6) );
262 else if( !lua_isnil( L, -1 ) )
263 msg_Warn( p_sd, "Item duration should be a number (in seconds)." );
264 lua_pop( L, 1 );
265 lua_getfield( L, -1, "category" );
266 if( lua_isstring( L, -1 ) )
267 services_discovery_AddItem( p_sd, p_input, luaL_checkstring( L, -1 ) );
268 else
269 services_discovery_AddItem( p_sd, p_input, NULL );
270 lua_pop( L, 1 );
271
272 /* string to build the input item uid */
273 lua_getfield( L, -1, "uiddata" );
274 if( lua_isstring( L, -1 ) )
275 {
276 char *s = strdup( luaL_checkstring( L, -1 ) );
277 if ( s )
278 {
279 struct md5_s md5;
280 InitMD5( &md5 );
281 AddMD5( &md5, s, strlen( s ) );
282 EndMD5( &md5 );
283 free( s );
284 s = psz_md5_hash( &md5 );
285 if ( s )
286 input_item_AddInfo( p_input, "uid", "md5", "%s", s );
287 free( s );
288 }
289 }
290 lua_pop( L, 1 );
291
292 input_item_t **udata = (input_item_t **)
293 lua_newuserdata( L, sizeof( input_item_t * ) );
294 *udata = p_input;
295 if( luaL_newmetatable( L, "input_item_t" ) )
296 {
297 lua_newtable( L );
298 luaL_register( L, NULL, vlclua_item_reg );
299 lua_setfield( L, -2, "__index" );
300 lua_pushliteral( L, "none of your business" );
301 lua_setfield( L, -2, "__metatable" );
302 }
303 lua_setmetatable( L, -2 );
304 vlc_gc_decref( p_input );
305 }
306 while( i_options > 0 )
307 free( ppsz_options[--i_options] );
308 free( ppsz_options );
309 }
310 else
311 msg_Err( p_sd, "vlc.sd.add_item: the \"path\" parameter can't be empty" );
312 }
313 else
314 msg_Err( p_sd, "Error parsing add_item arguments" );
315 return 1;
316}

318static int vlclua_sd_remove_item( lua_State *L )
319{
320 services_discovery_t *p_sd = (services_discovery_t *)vlclua_get_this( L );
321 if( !lua_isnil( L, 1 ) )
322 {
323 input_item_t **pp_input = luaL_checkudata( L, 1, "input_item_t" );
324 if( *pp_input )
325 services_discovery_RemoveItem( p_sd, *pp_input );
326 /* Make sure we won't try to remove it again */
327 *pp_input = NULL;
328 }
329 return 1;
330}

354static int vlclua_node_add_subitem( lua_State *L )
355{
356 services_discovery_t *p_sd = (services_discovery_t *)vlclua_get_this( L );
357 input_item_t **pp_node = (input_item_t **)luaL_checkudata( L, 1, "node" );
358 if( *pp_node )
359 {
360 if( lua_istable( L, -1 ) )
361 {
362 lua_getfield( L, -1, "path" );
363 if( lua_isstring( L, -1 ) )
364 {
365 const char *psz_path = lua_tostring( L, -1 );
366
367 /* The table must be at the top of the stack when calling
368 * vlclua_read_options() */
369 char **ppsz_options = NULL;
370 int i_options = 0;
371 lua_pushvalue( L, -2 );
372 vlclua_read_options( p_sd, L, &i_options, &ppsz_options );
373
374 input_item_t *p_input = input_item_NewExt( psz_path,
375 psz_path, i_options,
376 (const char **)ppsz_options,
377 VLC_INPUT_OPTION_TRUSTED, -1 );
378 lua_pop( L, 2 );
379
380 if( p_input )
381 {
382 input_item_node_t *p_input_node = input_item_node_Create( *pp_node );
383
384 vlclua_read_meta_data( p_sd, L, p_input );
385 /* This one is to be tested... */
386 vlclua_read_custom_meta_data( p_sd, L, p_input );
387 lua_getfield( L, -1, "duration" );
388 if( lua_isnumber( L, -1 ) )
389 input_item_SetDuration( p_input, (lua_tonumber( L, -1 )*1e6) );
390 else if( !lua_isnil( L, -1 ) )
391 msg_Warn( p_sd, "Item duration should be a number (in seconds)." );
392 lua_pop( L, 1 );
393 input_item_node_AppendItem( p_input_node, p_input );
394 input_item_node_PostAndDelete( p_input_node );
395 input_item_t **udata = (input_item_t **)
396 lua_newuserdata( L, sizeof( input_item_t * ) );
397 *udata = p_input;
398 if( luaL_newmetatable( L, "input_item_t" ) )
399 {
400 lua_newtable( L );
401 luaL_register( L, NULL, vlclua_item_reg );
402 lua_setfield( L, -2, "__index" );
403 lua_pushliteral( L, "none of your business" );
404 lua_setfield( L, -2, "__metatable" );
405 }
406 lua_setmetatable( L, -2 );
407 vlc_gc_decref( p_input );
408 }
409 while( i_options > 0 )
410 free( ppsz_options[--i_options] );
411 free( ppsz_options );
412 }
413 else
414 msg_Err( p_sd, "node:add_subitem: the \"path\" parameter can't be empty" );
415 }
416 else
417 msg_Err( p_sd, "Error parsing add_subitem arguments" );
418 }
419 return 1;
420}
