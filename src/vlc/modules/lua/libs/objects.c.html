43/*****************************************************************************
44 * Generic vlc_object_t wrapper creation

46static int vlclua_object_release( lua_State *L )
47{
48 vlc_object_t **p_obj = (vlc_object_t **)luaL_checkudata( L, 1, "vlc_object" );
49 lua_pop( L, 1 );
50 vlc_object_release( *p_obj );
51 return 0;
52}
53
54static int vlclua_object_find( lua_State *L )
55{
56 lua_pushnil( L );
57 return 1;
58}
59
60static int vlclua_get_libvlc( lua_State *L )
61{
62 libvlc_int_t *p_libvlc = vlclua_get_this( L )->p_libvlc;
63 vlc_object_hold( p_libvlc );
64 vlclua_push_vlc_object( L, p_libvlc );
65 return 1;
66}
67
68static int vlclua_get_playlist( lua_State *L )
69{
70 playlist_t *p_playlist = vlclua_get_playlist_internal( L );
71 if( p_playlist )
72 {
73 vlc_object_hold( p_playlist );
74 vlclua_push_vlc_object( L, p_playlist );
75 }
76 else lua_pushnil( L );
77 return 1;
78}
79
80static int vlclua_get_input( lua_State *L )
81{
82 input_thread_t *p_input = vlclua_get_input_internal( L );
83 if( p_input )
84 {
85 /* NOTE: p_input is already held by vlclua_get_input_internal() */
86 vlclua_push_vlc_object( L, p_input );
87 }
88 else lua_pushnil( L );
89 return 1;
90}
91
92#undef vlclua_push_vlc_object
93int vlclua_push_vlc_object( lua_State *L, vlc_object_t *p_obj )
94{
95 vlc_object_t **udata = (vlc_object_t **)
96 lua_newuserdata( L, sizeof( vlc_object_t * ) );
97 *udata = p_obj;
98
99 if( luaL_newmetatable( L, "vlc_object" ) )
100 {
101 /* Hide the metatable */
102 lua_pushliteral( L, "none of your business" );
103 lua_setfield( L, -2, "__metatable" );
104 /* Set the garbage collector if needed */
105 lua_pushcfunction( L, vlclua_object_release );
106 lua_setfield( L, -2, "__gc" );
107 }
108 lua_setmetatable( L, -2 );
109 return 1;
110}
111static int vlclua_get_vout( lua_State *L )
112{
113 input_thread_t *p_input = vlclua_get_input_internal( L );
114 if( p_input )
115 {
116 vout_thread_t *p_vout = input_GetVout( p_input );
117 vlc_object_release( p_input );
118 if(p_vout)
119 {
120 vlclua_push_vlc_object( L, (vlc_object_t *) p_vout );
121 return 1;
122 }
123 }
124 lua_pushnil( L );
125 return 1;
126}
127static int vlclua_get_aout( lua_State *L )
128{
129 playlist_t *p_playlist = vlclua_get_playlist_internal( L );
130 if( p_playlist != NULL )
131 {
132 audio_output_t *p_aout = playlist_GetAout( p_playlist );
133 if( p_aout != NULL )
134 {
135 vlclua_push_vlc_object( L, (vlc_object_t *)p_aout );
136 return 1;
137 }
138 }
139 lua_pushnil( L );
140 return 1;
141}

145static const luaL_Reg vlclua_object_reg[] = {
146 { "input", vlclua_get_input },
147 { "playlist", vlclua_get_playlist },
148 { "libvlc", vlclua_get_libvlc },
149 { "find", vlclua_object_find },
150 { "vout", vlclua_get_vout},
151 { "aout", vlclua_get_aout},
152 { NULL, NULL }
153};
154
155void luaopen_object( lua_State *L )
156{
157 lua_newtable( L );
158 luaL_register( L, NULL, vlclua_object_reg );
159 lua_setfield( L, -2, "object" );
160}
