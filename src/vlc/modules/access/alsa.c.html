74static void Dump (vlc_object_t *obj, const char *msg,
75 int (*cb)(void *, snd_output_t *), void *p)
76{
77 snd_output_t *output;
78 char *str;
79
80 if (unlikely(snd_output_buffer_open (&output)))
81 return;
82
83 int val = cb (p, output);
84 if (val)
85 {
86 msg_Warn (obj, "cannot get info: %s", snd_strerror (val));
87 return;
88 }
89
90 size_t len = snd_output_buffer_string (output, &str);
91 if (len > 0 && str[len - 1])
92 len--; /* strip trailing newline */
93 msg_Dbg (obj, "%s%.*s", msg, (int)len, str);
94 snd_output_close (output);
95}
96#define Dump(o, m, cb, p) \
97 Dump(VLC_OBJECT(o), m, (int (*)(void *, snd_output_t *))(cb), p)

99static void DumpDevice (vlc_object_t *obj, snd_pcm_t *pcm)
100{
101 snd_pcm_info_t *info;
102
103 Dump (obj, " ", snd_pcm_dump, pcm);
104 snd_pcm_info_alloca (&info);
105 if (snd_pcm_info (pcm, info) == 0)
106 {
107 msg_Dbg (obj, " device name : %s", snd_pcm_info_get_name (info));
108 msg_Dbg (obj, " device ID : %s", snd_pcm_info_get_id (info));
109 msg_Dbg (obj, " subdevice name: %s",
110 snd_pcm_info_get_subdevice_name (info));
111 }
112}

125struct demux_sys_t
126{
127 snd_pcm_t *pcm;
128 es_out_id_t *es;
129 vlc_thread_t thread;
130
131 mtime_t start;
132 mtime_t caching;
133 snd_pcm_uframes_t period_size;
134 unsigned rate;
135};
136
137static void Poll (snd_pcm_t *pcm, int canc)
138{
139 int n = snd_pcm_poll_descriptors_count (pcm);
140 struct pollfd ufd[n];
141 unsigned short revents;
142
143 snd_pcm_poll_descriptors (pcm, ufd, n);
144 do
145 {
146 vlc_restorecancel (canc);
147 while (poll (ufd, n, -1) == -1);
148 canc = vlc_savecancel ();
149 snd_pcm_poll_descriptors_revents (pcm, ufd, n, &revents);
150 }
151 while (!revents);
152}
153
154static void *Thread (void *data)
155{
156 demux_t *demux = data;
157 demux_sys_t *sys = demux->p_sys;
158 snd_pcm_t *pcm = sys->pcm;
159 size_t bytes;
160 int canc, val;
161
162 canc = vlc_savecancel ();
163 bytes = snd_pcm_frames_to_bytes (pcm, sys->period_size);
164 val = snd_pcm_start (pcm);
165 if (val)
166 {
167 msg_Err (demux, "cannot prepare device: %s", snd_strerror (val));
168 return NULL;
169 }
170
171 for (;;)
172 {
173 block_t *block = block_Alloc (bytes);
174 if (unlikely(block == NULL))
175 break;
176
177 /* Wait for data */
178 Poll (pcm, canc);
179
180 /* Read data */
181 snd_pcm_sframes_t frames, delay;
182 mtime_t pts;
183
184 frames = snd_pcm_readi (pcm, block->p_buffer, sys->period_size);
185 pts = mdate ();
186 if (frames < 0)
187 {
188 if (frames == -EAGAIN)
189 continue;
190
191 val = snd_pcm_recover (pcm, frames, 1);
192 if (val == 0)
193 {
194 msg_Warn (demux, "cannot read samples: %s",
195 snd_strerror (frames));
196 continue;
197 }
198 msg_Err (demux, "cannot recover record stream: %s",
199 snd_strerror (val));
200 DumpDeviceStatus (demux, pcm);
201 break;
202 }
203
204 /* Compute time stamp */
205 if (snd_pcm_delay (pcm, &delay))
206 delay = 0;
207 delay += frames;
208 pts -= (CLOCK_FREQ * delay) / sys->rate;
209
210 block->i_buffer = snd_pcm_frames_to_bytes (pcm, frames);
211 block->i_nb_samples = frames;
212 block->i_pts = pts;
213 block->i_length = (CLOCK_FREQ * frames) / sys->rate;
214
215 es_out_Control (demux->out, ES_OUT_SET_PCR, block->i_pts);
216 es_out_Send (demux->out, sys->es, block);
217 }
218 return NULL;
219}

221static int Control (demux_t *demux, int query, va_list ap)
222{
223 demux_sys_t *sys = demux->p_sys;
224
225 switch (query)
226 {
227 case DEMUX_GET_TIME:
228 *va_arg (ap, int64_t *) = mdate () - sys->start;
229 break;
230
231 case DEMUX_GET_PTS_DELAY:
232 *va_arg (ap, int64_t *) = sys->caching;
233 break;
234
235 //case DEMUX_SET_NEXT_DEMUX_TIME: still needed?
236
237 case DEMUX_HAS_UNSUPPORTED_META:
238 case DEMUX_CAN_RECORD:
239 case DEMUX_CAN_PAUSE:
240 case DEMUX_CAN_CONTROL_PACE:
241 case DEMUX_CAN_CONTROL_RATE:
242 case DEMUX_CAN_SEEK:
243 *va_arg (ap, bool *) = false;
244 break;;
245
246 default:
247 return VLC_EGENERIC;
248 }
249
250 return VLC_SUCCESS;
251}

324static int Open (vlc_object_t *obj)
325{
326 demux_t *demux = (demux_t *)obj;
327 demux_sys_t *sys = malloc (sizeof (*sys));
328
329 if (unlikely(sys == NULL))
330 return VLC_ENOMEM;
331
332 /* Open the device */
333 const char *device = demux->psz_location;
334 if (device == NULL || !device[0])
335 device = "default";
336
337 const int mode = SND_PCM_NONBLOCK
338 /*| SND_PCM_NO_AUTO_RESAMPLE*/
339 | SND_PCM_NO_AUTO_CHANNELS
340 /*| SND_PCM_NO_AUTO_FORMAT*/;
341 snd_pcm_t *pcm;
342 int val = snd_pcm_open (&pcm, device, SND_PCM_STREAM_CAPTURE, mode);
343 if (val != 0)
344 {
345 msg_Err (demux, "cannot open ALSA device \"%s\": %s", device,
346 snd_strerror (val));
347 free (sys);
348 return VLC_EGENERIC;
349 }
350 sys->pcm = pcm;
351 msg_Dbg (demux, "using ALSA device: %s", device);
352 DumpDevice (VLC_OBJECT(demux), pcm);
353
354 /* Negotiate capture parameters */
355 snd_pcm_hw_params_t *hw;
356 es_format_t fmt;
357 unsigned param;
358 int dir;
359
360 snd_pcm_hw_params_alloca (&hw);
361 snd_pcm_hw_params_any (pcm, hw);
362 Dump (demux, "initial hardware setup:\n", snd_pcm_hw_params_dump, hw);
363
364 val = snd_pcm_hw_params_set_rate_resample (pcm, hw, 0);
365 if (val)
366 {
367 msg_Err (demux, "cannot disable resampling: %s", snd_strerror (val));
368 goto error;
369 }
370
371 val = snd_pcm_hw_params_set_access (pcm, hw,
372 SND_PCM_ACCESS_RW_INTERLEAVED);
373 if (val)
374 {
375 msg_Err (demux, "cannot set access mode: %s", snd_strerror (val));
376 goto error;
377 }
378
379 snd_pcm_format_t format = SND_PCM_FORMAT_UNKNOWN;
380 for (size_t i = 0; i < sizeof (choices) / sizeof (choices[0]); i++)
381 if (snd_pcm_hw_params_test_format (pcm, hw, choices[i]) == 0)
382 {
383 val = snd_pcm_hw_params_set_format (pcm, hw, choices[i]);
384 if (val)
385 {
386 msg_Err (demux, "cannot set sample format: %s",
387 snd_strerror (val));
388 goto error;
389 }
390 format = choices[i];
391 break;
392 }
393
394 if (format == SND_PCM_FORMAT_UNKNOWN)
395 {
396 msg_Err (demux, "no supported sample format");
397 goto error;
398 }
399
400 assert ((size_t)format < (sizeof (formats) / sizeof (formats[0])));
401 es_format_Init (&fmt, AUDIO_ES, formats[format]);
402 fmt.audio.i_format = fmt.i_codec;
403
404 param = 1 + var_InheritBool (demux, "alsa-stereo");
405 val = snd_pcm_hw_params_set_channels_max (pcm, hw, &param);
406 if (val)
407 {
408 msg_Err (demux, "cannot restrict channels count: %s",
409 snd_strerror (val));
410 goto error;
411 }
412 val = snd_pcm_hw_params_set_channels_last (pcm, hw, &param);
413 if (val)
414 {
415 msg_Err (demux, "cannot set channels count: %s", snd_strerror (val));
416 goto error;
417 }
418 assert (param > 0);
419 assert (param < (sizeof (channel_maps) / sizeof (channel_maps[0])));
420 fmt.audio.i_channels = param;
421 fmt.audio.i_original_channels =
422 fmt.audio.i_physical_channels = channel_maps[param - 1];
423
424 param = var_InheritInteger (demux, "alsa-samplerate");
425 val = snd_pcm_hw_params_set_rate_max (pcm, hw, &param, NULL);
426 if (val)
427 {
428 msg_Err (demux, "cannot restrict rate to %u Hz or less: %s", 192000,
429 snd_strerror (val));
430 goto error;
431 }
432 val = snd_pcm_hw_params_set_rate_last (pcm, hw, &param, &dir);
433 if (val)
434 {
435 msg_Err (demux, "cannot set sample rate: %s", snd_strerror (val));
436 goto error;
437 }
438 if (dir)
439 msg_Warn (demux, "sample rate is not integral");
440 fmt.audio.i_rate = param;
441 sys->rate = param;
442
443 sys->start = mdate ();
444 sys->caching = INT64_C(1000) * var_InheritInteger (demux, "live-caching");
445 param = sys->caching;
446 val = snd_pcm_hw_params_set_buffer_time_near (pcm, hw, &param, NULL);
447 if (val)
448 {
449 msg_Err (demux, "cannot set buffer duration: %s", snd_strerror (val));
450 goto error;
451 }
452
453 param /= 4;
454 val = snd_pcm_hw_params_set_period_time_near (pcm, hw, &param, NULL);
455 if (val)
456 {
457 msg_Err (demux, "cannot set period: %s", snd_strerror (val));
458 goto error;
459 }
460
461 val = snd_pcm_hw_params_get_period_size (hw, &sys->period_size, &dir);
462 if (val)
463 {
464 msg_Err (demux, "cannot get period size: %s", snd_strerror (val));
465 goto error;
466 }
467 if (dir > 0)
468 sys->period_size++;
469
470 /* Commit hardware parameters */
471 val = snd_pcm_hw_params (pcm, hw);
472 if (val)
473 {
474 msg_Err (demux, "cannot commit hardware parameters: %s",
475 snd_strerror (val));
476 goto error;
477 }
478 Dump (demux, "final HW setup:\n", snd_pcm_hw_params_dump, hw);
479
480 /* Kick recording */
481 aout_FormatPrepare (&fmt.audio);
482 sys->es = es_out_Add (demux->out, &fmt);
483 demux->p_sys = sys;
484
485 if (vlc_clone (&sys->thread, Thread, demux, VLC_THREAD_PRIORITY_INPUT))
486 {
487 es_out_Del (demux->out, sys->es);
488 goto error;
489 }
490
491 demux->pf_demux = NULL;
492 demux->pf_control = Control;
493 return VLC_SUCCESS;
494error:
495 snd_pcm_close (pcm);
496 free (sys);
497 return VLC_EGENERIC;
498}

500static void Close (vlc_object_t *obj)
501{
502 demux_t *demux = (demux_t *)obj;
503 demux_sys_t *sys = demux->p_sys;
504
505 vlc_cancel (sys->thread);
506 vlc_join (sys->thread, NULL);
507
508 snd_pcm_close (sys->pcm);
509 free (sys);
510}
