105/*****************************************************************************
106 * Open: open the frontend device
107 *****************************************************************************/
108static int Open( vlc_object_t *p_this )
109{
110 access_t *p_access = (access_t*)p_this;
111 access_sys_t *p_sys;
112
113 /* Only if selected */
114 if( *p_access->psz_access == '\0' )
115 return VLC_EGENERIC;
116
117 p_access->p_sys = p_sys = calloc( 1, sizeof( access_sys_t ) );
118 if( !p_sys )
119 return VLC_ENOMEM;
120
121 /* Create all variables */
122 VarInit( p_access );
123
124 /* Parse the command line */
125 if( ParseMRL( p_access ) )
126 {
127 free( p_sys );
128 return VLC_EGENERIC;
129 }
130
131 bool b_scan_mode = var_GetInteger( p_access, "dvb-frequency" ) == 0;
132 if( b_scan_mode )
133 {
134 msg_Dbg( p_access, "DVB scan mode selected" );
135 p_access->pf_block = BlockScan;
136 }
137 else
138 return VLC_EGENERIC; /* let the DTV plugin do the work */
139
140 /* Getting frontend info */
141 if( FrontendOpen( p_access) )
142 {
143 free( p_sys );
144 return VLC_EGENERIC;
145 }
146
147 /* Opening DVR device */
148 if( DVROpen( p_access ) < 0 )
149 {
150 FrontendClose( p_access );
151 free( p_sys );
152 return VLC_EGENERIC;
153 }
154
155 {
156 scan_parameter_t parameter;
157 scan_t *p_scan;
158
159 msg_Dbg( p_access, "setting filter on PAT/NIT/SDT (DVB only)" );
160 FilterSet( p_access, 0x00, OTHER_TYPE ); // PAT
161 FilterSet( p_access, 0x10, OTHER_TYPE ); // NIT
162 FilterSet( p_access, 0x11, OTHER_TYPE ); // SDT
163
164 if( FrontendGetScanParameter( p_access, &parameter ) ||
165 (p_scan = scan_New( VLC_OBJECT(p_access), &parameter )) == NULL )
166 {
167 Close( VLC_OBJECT(p_access) );
168 return VLC_EGENERIC;
169 }
170 p_sys->scan = p_scan;
171 p_sys->i_read_once = DVB_READ_ONCE_SCAN;
172 }
173
174 /* Set up access */
175 p_access->pf_read = NULL;
176 p_access->pf_control = Control;
177 p_access->pf_seek = NULL;
178 access_InitFields( p_access );
179
180 return VLC_SUCCESS;
181}

183/*****************************************************************************
184 * Close : Close the device
185 *****************************************************************************/
186static void Close( vlc_object_t *p_this )
187{
188 access_t *p_access = (access_t*)p_this;
189 access_sys_t *p_sys = p_access->p_sys;
190
191 FilterUnset( p_access, MAX_DEMUX );
192
193 DVRClose( p_access );
194 FrontendClose( p_access );
195 scan_Destroy( p_sys->scan );
196
197 free( p_sys );
198}

200/*****************************************************************************
201 * BlockScan:
202 *****************************************************************************/
203static block_t *BlockScan( access_t *p_access )
204{
205 access_sys_t *p_sys = p_access->p_sys;
206 scan_t *p_scan = p_sys->scan;
207 scan_configuration_t cfg;
208
209 /* */
210 if( scan_Next( p_scan, &cfg ) )
211 {
212 const bool b_first_eof = !p_access->info.b_eof;
213
214 if( b_first_eof )
215 msg_Warn( p_access, "Scanning finished" );
216
217 /* */
218 p_access->info.b_eof = true;
219 return b_first_eof ? scan_GetM3U( p_scan ) : NULL;
220 }
221
222 /* */
223 scan_session_t *session = scan_session_New( VLC_OBJECT(p_access), &cfg );
224 if( session == NULL )
225 return NULL;
226
227 /* */
228 msg_Dbg( p_access, "Scanning frequency %d", cfg.i_frequency );
229 var_SetInteger( p_access, "dvb-frequency", cfg.i_frequency );
230 msg_Dbg( p_access, " bandwidth %d", cfg.i_bandwidth );
231 var_SetInteger( p_access, "dvb-bandwidth", cfg.i_bandwidth );
232 if ( cfg.c_polarization )
233 var_SetInteger( p_access, "dvb-voltage", cfg.c_polarization == 'H' ? 18 : 13 );
234
235 if ( cfg.i_symbolrate )
236 var_SetInteger( p_access, "dvb-srate", cfg.i_symbolrate );
237
238 /* Setting frontend parameters for tuning the hardware */
239 if( FrontendSet( p_access ) < 0 )
240 {
241 msg_Err( p_access, "Failed to tune the frontend" );
242 p_access->info.b_eof = true;
243 scan_session_Destroy( p_scan, session );
244 return NULL;
245 }
246
247 /* */
248 int64_t i_scan_start = mdate();
249
250 bool b_has_dvb_signal = false;
251 bool b_has_lock = false;
252 int i_best_snr = -1;
253
254 for ( ; ; )
255 {
256 struct pollfd ufds[2];
257 int i_ret;
258
259 /* Initialize file descriptor sets */
260 memset (ufds, 0, sizeof (ufds));
261 ufds[0].fd = p_sys->i_handle;
262 ufds[0].events = POLLIN;
263 ufds[1].fd = p_sys->i_frontend_handle;
264 ufds[1].events = POLLPRI;
265
266 /* We'll wait 0.1 second if nothing happens */
267 /* Find if some data is available */
268 i_ret = poll( ufds, 2, 100 );
269
270 if( !vlc_object_alive (p_access) || scan_IsCancelled( p_scan ) )
271 break;
272
273 if( i_ret <= 0 )
274 {
275 const mtime_t i_scan_time = mdate() - i_scan_start;
276 frontend_status_t status;
277
278 FrontendGetStatus( p_access, &status );
279
280 b_has_dvb_signal |= status.b_has_carrier;
281 b_has_lock |= status.b_has_lock;
282
283 if( ( !b_has_dvb_signal && i_scan_time > DVB_SCAN_MAX_SIGNAL_TIME ) ||
284 ( !b_has_lock && i_scan_time > DVB_SCAN_MAX_LOCK_TIME ) ||
285 ( i_scan_time > DVB_SCAN_MAX_PROBE_TIME ) )
286 {
287 msg_Dbg( p_access, "timed out scanning current frequency (s=%d l=%d)", b_has_dvb_signal, b_has_lock );
288 break;
289 }
290 }
291
292 if( i_ret < 0 )
293 {
294 if( errno == EINTR )
295 continue;
296
297 msg_Err( p_access, "poll error: %s", vlc_strerror_c(errno) );
298 scan_session_Destroy( p_scan, session );
299
300 p_access->info.b_eof = true;
301 return NULL;
302 }
303
304 if( ufds[1].revents )
305 {
306 frontend_statistic_t stat;
307
308 FrontendPoll( p_access );
309
310 if( !FrontendGetStatistic( p_access, &stat ) )
311 {
312 if( stat.i_snr > i_best_snr )
313 i_best_snr = stat.i_snr;
314 }
315 }
316
317 if ( p_sys->i_frontend_timeout && mdate() > p_sys->i_frontend_timeout )
318 {
319 msg_Warn( p_access, "no lock, tuning again" );
320 FrontendSet( p_access );
321 }
322
323 if ( ufds[0].revents )
324 {
325 const int i_read_once = 1;
326 block_t *p_block = block_Alloc( i_read_once * TS_PACKET_SIZE );
327
328 if( ( i_ret = read( p_sys->i_handle, p_block->p_buffer,
329 i_read_once * TS_PACKET_SIZE ) ) <= 0 )
330 {
331 msg_Warn( p_access, "read failed: %s", vlc_strerror_c(errno) );
332 block_Release( p_block );
333 continue;
334 }
335 p_block->i_buffer = i_ret;
336
337 /* */
338 if( scan_session_Push( session, p_block ) )
339 {
340 msg_Dbg( p_access, "finished scanning current frequency" );
341 break;
342 }
343 }
344 }
345
346 /* */
347 if( i_best_snr > 0 )
348 scan_service_SetSNR( session, i_best_snr );
349
350 scan_session_Destroy( p_scan, session );
351 return NULL;
352}

354/*****************************************************************************
355 * Control:
356 *****************************************************************************/
357static int Control( access_t *p_access, int i_query, va_list args )
358{
359 bool *pb_bool;
360 int64_t *pi_64;
361 double *pf1, *pf2;
362 frontend_statistic_t stat;
363
364 switch( i_query )
365 {
366 case ACCESS_CAN_SEEK:
367 case ACCESS_CAN_FASTSEEK:
368 case ACCESS_CAN_PAUSE:
369 case ACCESS_CAN_CONTROL_PACE:
370 pb_bool = (bool*)va_arg( args, bool* );
371 *pb_bool = false;
372 break;
373
374 case ACCESS_GET_CONTENT_TYPE:
375 *va_arg( args, char** ) = strdup("application/vnd.apple.mpegurl"); // m3u8
376 return VLC_SUCCESS;
377
378 case ACCESS_GET_PTS_DELAY:
379 pi_64 = (int64_t*)va_arg( args, int64_t * );
380 *pi_64 = DEFAULT_PTS_DELAY;
381 break;
382
383 case ACCESS_GET_SIGNAL:
384 pf1 = (double*)va_arg( args, double * );
385 pf2 = (double*)va_arg( args, double * );
386
387 *pf1 = *pf2 = 0;
388 if( !FrontendGetStatistic( p_access, &stat ) )
389 {
390 *pf1 = (double)stat.i_snr / 65535.0;
391 *pf2 = (double)stat.i_signal_strenth / 65535.0;
392 }
393 break;
394
395 default:
396 return VLC_EGENERIC;
397
398 }
399 return VLC_SUCCESS;
400}

402/*****************************************************************************
403 * FilterSet/FilterUnset:
404 *****************************************************************************/
405static void FilterSet( access_t *p_access, int i_pid, int i_type )
406{
407 access_sys_t *p_sys = p_access->p_sys;
408 int i;
409
410 /* Find first free slot */
411 for( i = 0; i < MAX_DEMUX; i++ )
412 {
413 if( !p_sys->p_demux_handles[i].i_type )
414 break;
415
416 if( p_sys->p_demux_handles[i].i_pid == i_pid )
417 return; /* Already set */
418 }
419
420 if( i >= MAX_DEMUX )
421 {
422 msg_Err( p_access, "no free p_demux_handles !" );
423 return;
424 }
425
426 if( DMXSetFilter( p_access, i_pid,
427 &p_sys->p_demux_handles[i].i_handle, i_type ) )
428 {
429 msg_Err( p_access, "DMXSetFilter failed" );
430 return;
431 }
432 p_sys->p_demux_handles[i].i_type = i_type;
433 p_sys->p_demux_handles[i].i_pid = i_pid;
434
435 if( p_sys->i_read_once < DVB_READ_ONCE )
436 p_sys->i_read_once++;
437}
438
439static void FilterUnset( access_t *p_access, int i_max )
440{
441 access_sys_t *p_sys = p_access->p_sys;
442 int i;
443
444 for( i = 0; i < i_max; i++ )
445 {
446 if( p_sys->p_demux_handles[i].i_type )
447 {
448 DMXUnsetFilter( p_access, p_sys->p_demux_handles[i].i_handle );
449 p_sys->p_demux_handles[i].i_type = 0;
450 }
451 }
452}

482/* */
483static int ParseMRL( access_t *p_access )
484{
485 char *psz_dup = strdup( p_access->psz_location );
486 char *psz_parser = psz_dup;
487 vlc_value_t val;
488
489#define GET_OPTION_INT( option ) \
490 if ( !strncmp( psz_parser, option "=", strlen(option "=") ) ) \
491 { \
492 val.i_int = strtol( psz_parser + strlen(option "="), &psz_parser, \
493 0 ); \
494 var_Set( p_access, "dvb-" option, val ); \
495 }
496
497#define GET_OPTION_BOOL( option ) \
498 if ( !strncmp( psz_parser, option "=", strlen(option "=") ) ) \
499 { \
500 val.b_bool = strtol( psz_parser + strlen(option "="), &psz_parser, \
501 0 ); \
502 var_Set( p_access, "dvb-" option, val ); \
503 }
504
505#define GET_OPTION_STRING( option ) \
506 if ( !strncmp( psz_parser, option "=", strlen( option "=" ) ) ) \
507 { \
508 psz_parser += strlen( option "=" ); \
509 val.psz_string = psz_parser; \
510 char *p_save; \
511 char *tok = strtok_r(val.psz_string, ":", &p_save); \
512 val.psz_string[tok - val.psz_string - 1] = 0; \
513 var_Set( p_access, "dvb-" option, val ); \
514 psz_parser += strlen( val.psz_string ); \
515 }
516
517 while( *psz_parser )
518 {
519 GET_OPTION_INT("adapter")
520 else GET_OPTION_INT("device")
521 else GET_OPTION_INT("frequency")
522 else GET_OPTION_INT("inversion")
523 else GET_OPTION_BOOL("probe")
524 else GET_OPTION_BOOL("budget-mode")
525
526 else GET_OPTION_STRING("satellite")
527 else GET_OPTION_INT("voltage")
528 else GET_OPTION_BOOL("high-voltage")
529 else GET_OPTION_INT("tone")
530 else GET_OPTION_INT("satno")
531 else GET_OPTION_INT("srate")
532 else GET_OPTION_INT("lnb-lof1")
533 else GET_OPTION_INT("lnb-lof2")
534 else GET_OPTION_INT("lnb-slof")
535
536 else GET_OPTION_INT("bandwidth")
537 else GET_OPTION_INT("transmission")
538 else GET_OPTION_INT("hierarchy")
539
540 /* Redundant with voltage but much easier to use */
541 else if( !strncmp( psz_parser, "polarization=",
542 strlen( "polarization=" ) ) )
543 {
544 psz_parser += strlen( "polarization=" );
545 if ( *psz_parser == 'V' || *psz_parser == 'v' )
546 val.i_int = 13;
547 else if ( *psz_parser == 'H' || *psz_parser == 'h' )
548 val.i_int = 18;
549 else
550 {
551 msg_Err( p_access, "illegal polarization %c", *psz_parser );
552 free( psz_dup );
553 return VLC_EGENERIC;
554 }
555 var_Set( p_access, "dvb-voltage", val );
556 }
557 else
558 {
559 msg_Err( p_access, "unknown option (%s)", psz_parser );
560 free( psz_dup );
561 return VLC_EGENERIC;
562 }
563
564 if ( *psz_parser )
565 psz_parser++;
566 }
567#undef GET_OPTION_INT
568#undef GET_OPTION_BOOL
569#undef GET_OPTION_STRING
570
571 free( psz_dup );
572 return VLC_SUCCESS;
573}
