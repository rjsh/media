24typedef enum
25{
26 SCAN_NONE,
27 SCAN_DVB_T,
28 SCAN_DVB_S,
29 SCAN_DVB_C,
30} scan_type_t;

32typedef struct
33{
34 int i_frequency;
35 int i_symbol_rate;
36 int i_fec;
37 char c_polarization;

40typedef struct scan_parameter_t
41{
42 scan_type_t type;
43 bool b_exhaustive;
44 bool b_use_nit;
45 bool b_free_only;
46 bool b_modulation_set;
47 bool b_symbolrate_set;
48
49 int i_modulation;
50 int i_symbolrate;

52 {
53 int i_min;
54 int i_max;
55 int i_step;

59
60 struct
61 {
62 /* Bandwidth should be 6, 7 or 8 */
63 int i_min;
64 int i_max;
65 int i_step;
66
67 int i_count;
68 } bandwidth;
69
70 struct
71 {
72 char *psz_name; /* satellite name */
73 char *psz_path; /* config file path */
74
75 scan_dvbs_transponder_t *p_transponders;
76 int i_count;
77 } sat_info;
78} scan_parameter_t;

80typedef struct
81{
82 int i_frequency;

84 {
85 int i_bandwidth;
