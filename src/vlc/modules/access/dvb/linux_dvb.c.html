72/*****************************************************************************
73 * FrontendOpen : Determine frontend device information and capabilities
74 *****************************************************************************/
75int FrontendOpen( access_t *p_access )
76{
77 access_sys_t *p_sys = p_access->p_sys;
78 frontend_t * p_frontend;
79 unsigned int i_adapter, i_device;
80 bool b_probe;
81 char frontend[128];
82
83 i_adapter = var_GetInteger( p_access, "dvb-adapter" );
84 i_device = var_GetInteger( p_access, "dvb-device" );
85 b_probe = var_GetBool( p_access, "dvb-probe" );
86
87 if( snprintf( frontend, sizeof(frontend), FRONTEND, i_adapter, i_device ) >= (int)sizeof(frontend) )
88 {
89 msg_Err( p_access, "snprintf() truncated string for FRONTEND" );
90 frontend[sizeof(frontend) - 1] = '\0';
91 }
92
93 p_sys->p_frontend = p_frontend = malloc( sizeof(frontend_t) );
94 if( !p_frontend )
95 return VLC_ENOMEM;
96
97 msg_Dbg( p_access, "Opening device %s", frontend );
98 if( (p_sys->i_frontend_handle = vlc_open(frontend, O_RDWR | O_NONBLOCK)) < 0 )
99 {
100 msg_Err( p_access, "FrontEndOpen: opening device failed: %s",
101 vlc_strerror_c(errno) );
102 free( p_frontend );
103 return VLC_EGENERIC;
104 }
105
106 if( b_probe )
107 {
108 const char * psz_expected = NULL;
109 const char * psz_real;
110
111 if( FrontendInfo( p_access ) < 0 )
112 {
113 close( p_sys->i_frontend_handle );
114 free( p_frontend );
115 return VLC_EGENERIC;
116 }
117
118 switch( p_frontend->info.type )
119 {
120 case FE_OFDM:
121 psz_real = "DVB-T";
122 break;
123 case FE_QAM:
124 psz_real = "DVB-C";
125 break;
126 case FE_QPSK:
127 psz_real = "DVB-S";
128 break;
129 case FE_ATSC:
130 psz_real = "ATSC";
131 break;
132 default:
133 psz_real = "unknown";
134 }
135
136 /* Sanity checks */
137 if( (!strncmp( p_access->psz_access, "qpsk", 4 ) ||
138 !strncmp( p_access->psz_access, "dvb-s", 5 ) ||
139 !strncmp( p_access->psz_access, "satellite", 9 ) ) &&
140 (p_frontend->info.type != FE_QPSK) )
141 {
142 psz_expected = "DVB-S";
143 }
144 if( (!strncmp( p_access->psz_access, "cable", 5 ) ||
145 !strncmp( p_access->psz_access, "dvb-c", 5 ) ) &&
146 (p_frontend->info.type != FE_QAM) )
147 {
148 psz_expected = "DVB-C";
149 }
150 if( (!strncmp( p_access->psz_access, "terrestrial", 11 ) ||
151 !strncmp( p_access->psz_access, "dvb-t", 5 ) ) &&
152 (p_frontend->info.type != FE_OFDM) )
153 {
154 psz_expected = "DVB-T";
155 }
156
157 if( (!strncmp( p_access->psz_access, "usdigital", 9 ) ||
158 !strncmp( p_access->psz_access, "atsc", 4 ) ) &&
159 (p_frontend->info.type != FE_ATSC) )
160 {
161 psz_expected = "ATSC";
162 }
163
164 if( psz_expected != NULL )
165 {
166 msg_Err( p_access, "requested type %s not supported by %s tuner",
167 psz_expected, psz_real );
168 close( p_sys->i_frontend_handle );
169 free( p_frontend );
170 return VLC_EGENERIC;
171 }
172 }
173 else /* no frontend probing is done so use default border values. */
174 {
175 msg_Dbg( p_access, "using default values for frontend info" );
176
177 msg_Dbg( p_access, "method of access is %s", p_access->psz_access );
178 p_frontend->info.type = FE_QPSK;
179 if( !strncmp( p_access->psz_access, "qpsk", 4 ) ||
180 !strncmp( p_access->psz_access, "dvb-s", 5 ) )
181 p_frontend->info.type = FE_QPSK;
182 else if( !strncmp( p_access->psz_access, "cable", 5 ) ||
183 !strncmp( p_access->psz_access, "dvb-c", 5 ) )
184 p_frontend->info.type = FE_QAM;
185 else if( !strncmp( p_access->psz_access, "terrestrial", 11 ) ||
186 !strncmp( p_access->psz_access, "dvb-t", 5 ) )
187 p_frontend->info.type = FE_OFDM;
188 else if( !strncmp( p_access->psz_access, "usdigital", 9 ) ||
189 !strncmp( p_access->psz_access, "atsc", 4 ) )
190 p_frontend->info.type = FE_ATSC;
191 }
192
193 return VLC_SUCCESS;
194}

212/*****************************************************************************
213 * FrontendSet : Tune !
214 *****************************************************************************/
215int FrontendSet( access_t *p_access )
216{
217 access_sys_t *p_sys = p_access->p_sys;
218
219 switch( p_sys->p_frontend->info.type )
220 {
221 /* DVB-S */
222 case FE_QPSK:
223 if( FrontendSetQPSK( p_access ) )
224 {
225 msg_Err( p_access, "DVB-S tuning error" );
226 return VLC_EGENERIC;
227 }
228 break;
229
230 /* DVB-C */
231 case FE_QAM:
232 if( FrontendSetQAM( p_access ) )
233 {
234 msg_Err( p_access, "DVB-C tuning error" );
235 return VLC_EGENERIC;
236 }
237 break;
238
239 /* DVB-T */
240 case FE_OFDM:
241 if( FrontendSetOFDM( p_access ) )
242 {
243 msg_Err( p_access, "DVB-T tuning error" );
244 return VLC_EGENERIC;
245 }
246 break;
247
248 /* ATSC */
249 case FE_ATSC:
250 if( FrontendSetATSC( p_access ) )
251 {
252 msg_Err( p_access, "ATSC tuning error" );
253 return VLC_EGENERIC;
254 }
255 break;
256
257 default:
258 msg_Err( p_access, "tuner type %s not supported",
259 p_sys->p_frontend->info.name );
260 return VLC_EGENERIC;
261 }
262 p_sys->p_frontend->i_last_status = 0;
263 p_sys->i_frontend_timeout = mdate() + FRONTEND_LOCK_TIMEOUT;
264 return VLC_SUCCESS;
265}

267/*****************************************************************************
268 * FrontendPoll : Poll for frontend events
269 *****************************************************************************/
270void FrontendPoll( access_t *p_access )
271{
272 access_sys_t *p_sys = p_access->p_sys;
273 frontend_t * p_frontend = p_sys->p_frontend;
274 struct dvb_frontend_event event;
275 fe_status_t i_status, i_diff;
276
277 for( ;; )
278 {
279 if( ioctl( p_sys->i_frontend_handle, FE_GET_EVENT, &event ) < 0 )
280 {
281 if( errno != EWOULDBLOCK )
282 msg_Err( p_access, "frontend event error: %s",
283 vlc_strerror_c(errno) );
284 return;
285 }
286
287 i_status = event.status;
288 i_diff = i_status ^ p_frontend->i_last_status;
289 p_frontend->i_last_status = i_status;
290
291 {
292#define IF_UP( x ) \
293 } \
294 if ( i_diff & (x) ) \
295 { \
296 if ( i_status & (x) )
297
298 IF_UP( FE_HAS_SIGNAL )
299 msg_Dbg( p_access, "frontend has acquired signal" );
300 else
301 msg_Dbg( p_access, "frontend has lost signal" );
302
303 IF_UP( FE_HAS_CARRIER )
304 msg_Dbg( p_access, "frontend has acquired carrier" );
305 else
306 msg_Dbg( p_access, "frontend has lost carrier" );
307
308 IF_UP( FE_HAS_VITERBI )
309 msg_Dbg( p_access, "frontend has acquired stable FEC" );
310 else
311 msg_Dbg( p_access, "frontend has lost FEC" );
312
313 IF_UP( FE_HAS_SYNC )
314 msg_Dbg( p_access, "frontend has acquired sync" );
315 else
316 msg_Dbg( p_access, "frontend has lost sync" );
317
318 IF_UP( FE_HAS_LOCK )
319 {
320 frontend_statistic_t stat;
321
322 msg_Dbg( p_access, "frontend has acquired lock" );
323 p_sys->i_frontend_timeout = 0;
324
325 /* Read some statistics */
326 if( !FrontendGetStatistic( p_access, &stat ) )
327 {
328 if( stat.i_ber >= 0 )
329 msg_Dbg( p_access, "- Bit error rate: %d", stat.i_ber );
330 if( stat.i_signal_strenth >= 0 )
331 msg_Dbg( p_access, "- Signal strength: %d", stat.i_signal_strenth );
332 if( stat.i_snr >= 0 )
333 msg_Dbg( p_access, "- SNR: %d", stat.i_snr );
334 }
335 }
336 else
337 {
338 msg_Dbg( p_access, "frontend has lost lock" );
339 p_sys->i_frontend_timeout = mdate() + FRONTEND_LOCK_TIMEOUT;
340 }
341
342 IF_UP( FE_REINIT )
343 {
344 /* The frontend was reinited. */
345 msg_Warn( p_access, "reiniting frontend");
346 FrontendSet( p_access );
347 }
348 }
349#undef IF_UP
350 }
351}
352
353int FrontendGetStatistic( access_t *p_access, frontend_statistic_t *p_stat )
354{
355 access_sys_t *p_sys = p_access->p_sys;
356 frontend_t * p_frontend = p_sys->p_frontend;
357
358 if( (p_frontend->i_last_status & FE_HAS_LOCK) == 0 )
359 return VLC_EGENERIC;
360
361 memset( p_stat, 0, sizeof(*p_stat) );
362 if( ioctl( p_sys->i_frontend_handle, FE_READ_BER, &p_stat->i_ber ) < 0 )
363 p_stat->i_ber = -1;
364 if( ioctl( p_sys->i_frontend_handle, FE_READ_SIGNAL_STRENGTH, &p_stat->i_signal_strenth ) < 0 )
365 p_stat->i_signal_strenth = -1;
366 if( ioctl( p_sys->i_frontend_handle, FE_READ_SNR, &p_stat->i_snr ) < 0 )
367 p_stat->i_snr = -1;
368
369 return VLC_SUCCESS;
370}

372void FrontendGetStatus( access_t *p_access, frontend_status_t *p_status )
373{
374 access_sys_t *p_sys = p_access->p_sys;
375 frontend_t * p_frontend = p_sys->p_frontend;
376
377 p_status->b_has_signal = (p_frontend->i_last_status & FE_HAS_SIGNAL) != 0;
378 p_status->b_has_carrier = (p_frontend->i_last_status & FE_HAS_CARRIER) != 0;
379 p_status->b_has_lock = (p_frontend->i_last_status & FE_HAS_LOCK) != 0;
380}

382static int ScanParametersDvbS( access_t *p_access, scan_parameter_t *p_scan )
383{
384 const frontend_t *p_frontend = p_access->p_sys->p_frontend;
385
386 memset( p_scan, 0, sizeof(*p_scan) );
387 p_scan->type = SCAN_DVB_S;
388
389 p_scan->frequency.i_min = p_frontend->info.frequency_min;
390 p_scan->frequency.i_max = p_frontend->info.frequency_max;
391 /* set satellite config file path */
392 p_scan->sat_info.psz_name = var_InheritString( p_access, "dvb-satellite" );
393
394 return VLC_SUCCESS;
395}

397static int ScanParametersDvbC( access_t *p_access, scan_parameter_t *p_scan )
398{
399 const frontend_t *p_frontend = p_access->p_sys->p_frontend;
400
401 memset( p_scan, 0, sizeof(*p_scan) );
402 p_scan->type = SCAN_DVB_C;
403 p_scan->b_exhaustive = false;
404
405 /* */
406 p_scan->frequency.i_min = p_frontend->info.frequency_min;
407 p_scan->frequency.i_max = p_frontend->info.frequency_max;
408 p_scan->frequency.i_step = p_frontend->info.frequency_stepsize
409 ? p_frontend->info.frequency_stepsize : 166667;
410 p_scan->frequency.i_count = (p_scan->frequency.i_max-p_scan->frequency.i_min)/p_scan->frequency.i_step;
411
412 /* if frontend can do auto, don't scan them */
413 if( p_frontend->info.caps & FE_CAN_QAM_AUTO )
414 {
415 p_scan->b_modulation_set = true;
416 } else {
417 p_scan->b_modulation_set = false;
418 /* our scanning code flips modulation from 16..256 automaticly*/
419 p_scan->i_modulation = 0;
420 }
421
422 /* if user supplies symbolrate, don't scan those */
423 if( var_GetInteger( p_access, "dvb-srate" ) )
424 p_scan->b_symbolrate_set = true;
425 else
426 p_scan->b_symbolrate_set = false;
427
428 /* */
429 p_scan->bandwidth.i_min = 6;
430 p_scan->bandwidth.i_max = 8;
431 p_scan->bandwidth.i_step = 1;
432 p_scan->bandwidth.i_count = 3;
433 return VLC_SUCCESS;
434}

459int FrontendGetScanParameter( access_t *p_access, scan_parameter_t *p_scan )
460{
461 access_sys_t *p_sys = p_access->p_sys;
462 const frontend_t *p_frontend = p_sys->p_frontend;
463
464 if( p_frontend->info.type == FE_OFDM ) /* DVB-T */
465 return ScanParametersDvbT( p_access, p_scan );
466 else if( p_frontend->info.type == FE_QAM ) /* DVB-C */
467 return ScanParametersDvbC( p_access, p_scan );
468 else if( p_frontend->info.type == FE_QPSK )
469 return ScanParametersDvbS( p_access, p_scan ); /* DVB-S */
470
471 msg_Err( p_access, "frontend scanning not supported" );
472 return VLC_EGENERIC;
473}

475/*****************************************************************************
476 * FrontendInfo : Return information about given frontend
477 *****************************************************************************/
478static int FrontendInfo( access_t *p_access )
479{
480 access_sys_t *p_sys = p_access->p_sys;
481 frontend_t *p_frontend = p_sys->p_frontend;
482
483 /* Determine type of frontend */
484 if( ioctl( p_sys->i_frontend_handle, FE_GET_INFO, &p_frontend->info ) < 0 )
485 {
486 msg_Err( p_access, "frontend info request error: %s",
487 vlc_strerror_c(errno) );
488 return VLC_EGENERIC;
489 }
490
491 /* Print out frontend capabilities. */
492 msg_Dbg(p_access, "Frontend Info:" );
493 msg_Dbg(p_access, " name = %s", p_frontend->info.name );
494 switch( p_frontend->info.type )
495 {
496 case FE_QPSK:
497 msg_Dbg( p_access, " type = QPSK (DVB-S)" );
498 break;
499 case FE_QAM:
500 msg_Dbg( p_access, " type = QAM (DVB-C)" );
501 break;
502 case FE_OFDM:
503 msg_Dbg( p_access, " type = OFDM (DVB-T)" );
504 break;
505 case FE_ATSC:
506 msg_Dbg( p_access, " type = ATSC (USA)" );
507 break;
508#if 0 /* DVB_API_VERSION == 3 */
509 case FE_MEMORY:
510 msg_Dbg(p_access, " type = MEMORY" );
511 break;
512 case FE_NET:
513 msg_Dbg(p_access, " type = NETWORK" );
514 break;
515#endif
516 default:
517 msg_Err( p_access, " unknown frontend type (%d)",
518 p_frontend->info.type );
519 return VLC_EGENERIC;
520 }
521 msg_Dbg(p_access, " frequency_min = %u (kHz)",
522 p_frontend->info.frequency_min);
523 msg_Dbg(p_access, " frequency_max = %u (kHz)",
524 p_frontend->info.frequency_max);
525 msg_Dbg(p_access, " frequency_stepsize = %u",
526 p_frontend->info.frequency_stepsize);
527 msg_Dbg(p_access, " frequency_tolerance = %u",
528 p_frontend->info.frequency_tolerance);
529 msg_Dbg(p_access, " symbol_rate_min = %u (kHz)",
530 p_frontend->info.symbol_rate_min);
531 msg_Dbg(p_access, " symbol_rate_max = %u (kHz)",
532 p_frontend->info.symbol_rate_max);
533 msg_Dbg(p_access, " symbol_rate_tolerance (ppm) = %u",
534 p_frontend->info.symbol_rate_tolerance);
535 msg_Dbg(p_access, " notifier_delay (ms) = %u",
536 p_frontend->info.notifier_delay );
537
538 msg_Dbg(p_access, "Frontend Info capability list:");
539 if( p_frontend->info.caps == FE_IS_STUPID)
540 msg_Dbg(p_access, " no capabilities - frontend is stupid!");
541 if( p_frontend->info.caps & FE_CAN_INVERSION_AUTO)
542 msg_Dbg(p_access, " inversion auto");
543 if( p_frontend->info.caps & FE_CAN_FEC_1_2)
544 msg_Dbg(p_access, " forward error correction 1/2");
545 if( p_frontend->info.caps & FE_CAN_FEC_2_3)
546 msg_Dbg(p_access, " forward error correction 2/3");
547 if( p_frontend->info.caps & FE_CAN_FEC_3_4)
548 msg_Dbg(p_access, " forward error correction 3/4");
549 if( p_frontend->info.caps & FE_CAN_FEC_4_5)
550 msg_Dbg(p_access, " forward error correction 4/5");
551 if( p_frontend->info.caps & FE_CAN_FEC_5_6)
552 msg_Dbg(p_access, " forward error correction 5/6");
553 if( p_frontend->info.caps & FE_CAN_FEC_6_7)
554 msg_Dbg(p_access, " forward error correction 6/7");
555 if( p_frontend->info.caps & FE_CAN_FEC_7_8)
556 msg_Dbg(p_access, " forward error correction 7/8");
557 if( p_frontend->info.caps & FE_CAN_FEC_8_9)
558 msg_Dbg(p_access, " forward error correction 8/9");
559 if( p_frontend->info.caps & FE_CAN_FEC_AUTO)
560 msg_Dbg(p_access, " forward error correction auto");
561 if( p_frontend->info.caps & FE_CAN_QPSK)
562 msg_Dbg(p_access, " QPSK modulation");
563 if( p_frontend->info.caps & FE_CAN_QAM_16)
564 msg_Dbg(p_access, " QAM 16 modulation");
565 if( p_frontend->info.caps & FE_CAN_QAM_32)
566 msg_Dbg(p_access, " QAM 32 modulation");
567 if( p_frontend->info.caps & FE_CAN_QAM_64)
568 msg_Dbg(p_access, " QAM 64 modulation");
569 if( p_frontend->info.caps & FE_CAN_QAM_128)
570 msg_Dbg(p_access, " QAM 128 modulation");
571 if( p_frontend->info.caps & FE_CAN_QAM_256)
572 msg_Dbg(p_access, " QAM 256 modulation");
573 if( p_frontend->info.caps & FE_CAN_QAM_AUTO)
574 msg_Dbg(p_access, " QAM auto modulation");
575 if( p_frontend->info.caps & FE_CAN_TRANSMISSION_MODE_AUTO)
576 msg_Dbg(p_access, " transmission mode auto");
577 if( p_frontend->info.caps & FE_CAN_BANDWIDTH_AUTO)
578 msg_Dbg(p_access, " bandwidth mode auto");
579 if( p_frontend->info.caps & FE_CAN_GUARD_INTERVAL_AUTO)
580 msg_Dbg(p_access, " guard interval mode auto");
581 if( p_frontend->info.caps & FE_CAN_HIERARCHY_AUTO)
582 msg_Dbg(p_access, " hierarchy mode auto");
583 if( p_frontend->info.caps & FE_CAN_8VSB)
584 msg_Dbg(p_access, " 8-level VSB modulation");
585 if( p_frontend->info.caps & FE_CAN_16VSB)
586 msg_Dbg(p_access, " 16-level VSB modulation");
587 if( p_frontend->info.caps & FE_HAS_EXTENDED_CAPS)
588 msg_Dbg(p_access, " extended capabilities");
589 /* 3 capabilities that don't exist yet HERE */
590#if (DVB_API_VERSION > 5) \
591 || ((DVB_API_VERSION == 5 && DVB_API_VERSION_MINOR >= 2))
592 if( p_frontend->info.caps & FE_CAN_TURBO_FEC)
593 msg_Dbg(p_access, " Turbo FEC modulation");
594#else
595# warning Please update your Linux kernel headers!
596#endif
597 if( p_frontend->info.caps & FE_CAN_2G_MODULATION)
598 msg_Dbg(p_access, " 2nd generation modulation (DVB-S2)");
599 /* FE_NEEDS_BENDING is deprecated */

663static int DoDiseqc( access_t *p_access )
664{
665 access_sys_t *p_sys = p_access->p_sys;
666 int i_val;
667 bool b_val;
668 int i_frequency, i_lnb_slof;
669 fe_sec_voltage_t fe_voltage;
670 fe_sec_tone_mode_t fe_tone;
671
672 i_frequency = var_GetInteger( p_access, "dvb-frequency" );
673 i_lnb_slof = var_GetInteger( p_access, "dvb-lnb-slof" );
674
675 i_val = var_GetInteger( p_access, "dvb-tone" );
676 if( i_val == -1 /* auto */ )
677 {
678 if( i_frequency >= i_lnb_slof )
679 i_val = 1;
680 else
681 i_val = 0;
682 var_SetInteger( p_access, "dvb-tone", i_val );
683 }
684
685 fe_voltage = DecodeVoltage( p_access );
686 fe_tone = DecodeTone( p_access );
687
688 /* Switch off continuous tone. */
689 if( ioctl( p_sys->i_frontend_handle, FE_SET_TONE, SEC_TONE_OFF ) < 0 )
690 {
691 msg_Err( p_access, "switching tone %s error: %s", "off",
692 vlc_strerror_c(errno) );
693 return VLC_EGENERIC;
694 }
695
696 /* Configure LNB voltage. */
697 if( ioctl( p_sys->i_frontend_handle, FE_SET_VOLTAGE, fe_voltage ) < 0 )
698 {
699 msg_Err( p_access, "voltage error: %s", vlc_strerror_c(errno) );
700 return VLC_EGENERIC;
701 }
702
703 b_val = var_GetBool( p_access, "dvb-high-voltage" );
704 if( ioctl( p_sys->i_frontend_handle,
705 FE_ENABLE_HIGH_LNB_VOLTAGE, b_val ) < 0 && b_val )
706 {
707 msg_Err( p_access, "high LNB voltage error: %s",
708 vlc_strerror_c(errno) );
709 }
710
711 /* Wait for at least 15 ms. */
712 msleep(15000);
713
714 i_val = var_GetInteger( p_access, "dvb-satno" );
715 if( i_val > 0 && i_val < 5 )
716 {
717 /* digital satellite equipment control,
718 * specification is available from http://www.eutelsat.com/
719 */
720
721 /* 1.x compatible equipment */
722 struct diseqc_cmd_t cmd = { {{0xe0, 0x10, 0x38, 0xf0, 0x00, 0x00}, 4}, 0 };
723
724 /* param: high nibble: reset bits, low nibble set bits,
725 * bits are: option, position, polarization, band
726 */
727 cmd.cmd.msg[3] = 0xf0 /* reset bits */
728 | (((i_val - 1) * 4) & 0xc)
729 | (fe_voltage == SEC_VOLTAGE_13 ? 0 : 2)
730 | (fe_tone == SEC_TONE_ON ? 1 : 0);
731
732 if( ioctl( p_sys->i_frontend_handle, FE_DISEQC_SEND_MASTER_CMD,
733 &cmd.cmd ) )
734 {
735 msg_Err( p_access, "master command sending error: %s",
736 vlc_strerror_c(errno) );
737 return VLC_EGENERIC;
738 }
739
740 msleep(15000 + cmd.wait * 1000);
741
742 /* A or B simple diseqc ("diseqc-compatible") */
743 if( ioctl( p_sys->i_frontend_handle, FE_DISEQC_SEND_BURST,
744 ((i_val - 1) % 2) ? SEC_MINI_B : SEC_MINI_A ) )
745 {
746 msg_Err( p_access, "burst sending error: %s",
747 vlc_strerror_c(errno) );
748 return VLC_EGENERIC;
749 }
750
751 msleep(15000);
752 }
753
754 if( ioctl( p_sys->i_frontend_handle, FE_SET_TONE, fe_tone ) )
755 {
756 msg_Err( p_access, "switching tone %s error: %s",
757 (fe_tone == SEC_TONE_ON) ? "on" : "off",
758 vlc_strerror_c(errno) );
759 return VLC_EGENERIC;
760 }
761
762 msleep(50000);
763 return 0;
764}

766static int FrontendSetQPSK( access_t *p_access )
767{
768 access_sys_t *p_sys = p_access->p_sys;
769 struct dvb_frontend_parameters fep;
770 int i_val;
771 int i_frequency, i_lnb_slof = 0, i_lnb_lof1, i_lnb_lof2 = 0;
772
773 /* Prepare the fep structure */
774 i_frequency = var_GetInteger( p_access, "dvb-frequency" );
775
776 i_val = var_GetInteger( p_access, "dvb-lnb-lof1" );
777 if( i_val == 0 )
778 {
779 /* Automatic mode. */
780 if ( i_frequency >= 950000 && i_frequency <= 2150000 )
781 {
782 msg_Dbg( p_access, "frequency %d is in IF-band", i_frequency );
783 i_lnb_lof1 = 0;
784 }
785 else if ( i_frequency >= 2500000 && i_frequency <= 2700000 )
786 {
787 msg_Dbg( p_access, "frequency %d is in S-band", i_frequency );
788 i_lnb_lof1 = 3650000;
789 }
790 else if ( i_frequency >= 3400000 && i_frequency <= 4200000 )
791 {
792 msg_Dbg( p_access, "frequency %d is in C-band (lower)",
793 i_frequency );
794 i_lnb_lof1 = 5150000;
795 }
796 else if ( i_frequency >= 4500000 && i_frequency <= 4800000 )
797 {
798 msg_Dbg( p_access, "frequency %d is in C-band (higher)",
799 i_frequency );
800 i_lnb_lof1 = 5950000;
801 }
802 else if ( i_frequency >= 10700000 && i_frequency <= 13250000 )
803 {
804 msg_Dbg( p_access, "frequency %d is in Ku-band",
805 i_frequency );
806 i_lnb_lof1 = 9750000;
807 i_lnb_lof2 = 10600000;
808 i_lnb_slof = 11700000;
809 }
810 else
811 {
812 msg_Err( p_access, "frequency %d is out of any known band",
813 i_frequency );
814 msg_Err( p_access, "specify dvb-lnb-lof1 manually for the local "
815 "oscillator frequency" );
816 return VLC_EGENERIC;
817 }
818 var_SetInteger( p_access, "dvb-lnb-lof1", i_lnb_lof1 );
819 var_SetInteger( p_access, "dvb-lnb-lof2", i_lnb_lof2 );
820 var_SetInteger( p_access, "dvb-lnb-slof", i_lnb_slof );
821 }
822 else
823 {
824 i_lnb_lof1 = i_val;
825 i_lnb_lof2 = var_GetInteger( p_access, "dvb-lnb-lof2" );
826 i_lnb_slof = var_GetInteger( p_access, "dvb-lnb-slof" );
827 }
828
829 if( i_lnb_slof && i_frequency >= i_lnb_slof )
830 {
831 i_frequency -= i_lnb_lof2;
832 }
833 else
834 {
835 i_frequency -= i_lnb_lof1;
836 }
837 fep.frequency = i_frequency >= 0 ? i_frequency : -i_frequency;
838
839 fep.inversion = DecodeInversion( p_access );
840
841 fep.u.qpsk.symbol_rate = var_GetInteger( p_access, "dvb-srate" );
842
843 fep.u.qpsk.fec_inner = FEC_NONE;
844
845 if( DoDiseqc( p_access ) < 0 )
846 {
847 return VLC_EGENERIC;
848 }
849
850 /* Empty the event queue */
851 for( ; ; )
852 {
853 struct dvb_frontend_event event;
854 if ( ioctl( p_sys->i_frontend_handle, FE_GET_EVENT, &event ) < 0
855 && errno == EWOULDBLOCK )
856 break;
857 }
858
859 /* Now send it all to the frontend device */
860 if( ioctl( p_sys->i_frontend_handle, FE_SET_FRONTEND, &fep ) < 0 )
861 {
862 msg_Err( p_access, "frontend error: %s", vlc_strerror_c(errno) );
863 return VLC_EGENERIC;
864 }
865
866 return VLC_SUCCESS;
867}

1232 dvr[sizeof(dvr) - 1] = '\0';
1233 }
1234
1235 msg_Dbg( p_access, "Opening device %s", dvr );

1237 {

1239 vlc_strerror_c(errno) );

1241 }

1246 vlc_strerror_c(errno) );
