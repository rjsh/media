205/*****************************************************************************
206 * DemuxOpen:
207 *****************************************************************************/
208static int DemuxOpen( vlc_object_t *p_this )
209{
210 demux_t *p_demux = (demux_t *)p_this;
211 demux_sys_t *p_sys;
212 char *psz_parser;
213
214 /* Fill p_demux field */
215 p_demux->pf_demux = DemuxDemux;
216 p_demux->pf_control = DemuxControl;
217 p_demux->p_sys = p_sys = calloc( 1, sizeof( demux_sys_t ) );
218 if( unlikely(!p_sys ) )
219 return VLC_ENOMEM;
220
221 p_sys->i_state = STATE_NOSYNC;
222 p_sys->i_last_state_change = mdate();
223
224 /* SDI AR */
225 char *psz_ar = var_InheritString( p_demux, "linsys-sdi-aspect-ratio" );
226 if ( psz_ar != NULL )
227 {
228 psz_parser = strchr( psz_ar, ':' );
229 if ( psz_parser )
230 {
231 *psz_parser++ = '\0';
232 p_sys->i_forced_aspect = p_sys->i_aspect =
233 strtol( psz_ar, NULL, 0 ) * VOUT_ASPECT_FACTOR
234 / strtol( psz_parser, NULL, 0 );
235 }
236 else
237 p_sys->i_forced_aspect = 0;
238 free( psz_ar );
239 }
240
241 /* */
242 p_sys->i_id_video = var_InheritInteger( p_demux, "linsys-sdi-id-video" );
243
244 /* Audio ES */
245 char *psz_string = psz_parser = var_InheritString( p_demux,
246 "linsys-sdi-audio" );
247 int i = 0;
248
249 while ( psz_parser != NULL && *psz_parser )
250 {
251 int i_id, i_group, i_pair;
252 char *psz_next = strchr( psz_parser, '=' );
253 if ( psz_next != NULL )
254 {
255 *psz_next = '\0';
256 i_id = strtol( psz_parser, NULL, 0 );
257 psz_parser = psz_next + 1;
258 }
259 else
260 i_id = 0;
261
262 psz_next = strchr( psz_parser, ':' );
263 if ( psz_next != NULL )
264 {
265 *psz_next = '\0';
266 psz_next++;
267 }
268
269 if ( sscanf( psz_parser, "%d,%d", &i_group, &i_pair ) == 2 )
270 {
271 p_sys->p_audios[i].i_group = i_group;
272 p_sys->p_audios[i].i_pair = i_pair;
273 p_sys->p_audios[i].i_id = i_id;
274 i++;
275 }
276 else
277 msg_Warn( p_demux, "malformed audio configuration (%s)",
278 psz_parser );
279
280 psz_parser = psz_next;
281 }
282 free( psz_string );
283
284 /* Teletext ES */
285 p_sys->psz_telx = var_InheritString( p_demux, "linsys-sdi-telx" );
286
287 p_sys->psz_telx_lang = var_InheritString( p_demux, "linsys-sdi-telx-lang" );
288
289 return VLC_SUCCESS;
290}

292static int Open( vlc_object_t *p_this )
293{
294 demux_t *p_demux = (demux_t *)p_this;
295 demux_sys_t *p_sys;
296 int i_ret;
297
298 if ( (i_ret = DemuxOpen( p_this )) != VLC_SUCCESS )
299 return i_ret;
300
301 /* Fill p_demux field */
302 p_demux->pf_demux = Demux;
303 p_demux->pf_control = Control;
304 p_sys = p_demux->p_sys;
305
306 p_sys->i_link = var_InheritInteger( p_demux, "linsys-sdi-link" );
307
308 if( InitCapture( p_demux ) != VLC_SUCCESS )
309 {
310 free( p_sys );
311 return VLC_EGENERIC;
312 }
313
314 return VLC_SUCCESS;
315}

409static int NewFrame( demux_t *p_demux )
410{
411 demux_sys_t *p_sys = p_demux->p_sys;
412
413 p_sys->p_current_picture = block_Alloc( p_sys->i_block_size );
414 if( unlikely( !p_sys->p_current_picture ) )
415 return VLC_ENOMEM;
416 p_sys->p_y = p_sys->p_current_picture->p_buffer;
417 p_sys->p_u = p_sys->p_y + p_sys->i_width * p_sys->i_height;
418 p_sys->p_v = p_sys->p_u + p_sys->i_width * p_sys->i_height / 4;
419
420 for ( int i = 0; i < MAX_AUDIOS; i++ )
421 {
422 sdi_audio_t *p_audio = &p_sys->p_audios[i];
423 p_audio->i_left_samples = p_audio->i_right_samples = 0;
424 }
425 return VLC_SUCCESS;

428static int StartDecode( demux_t *p_demux )
429{
430 demux_sys_t *p_sys = p_demux->p_sys;
431 es_format_t fmt;
432 char *psz_parser;
433
434 p_sys->i_next_date = START_DATE;
435 p_sys->i_incr = 1000000 * p_sys->i_frame_rate_base / p_sys->i_frame_rate;
436 p_sys->i_block_size = p_sys->i_width * p_sys->i_height * 3 / 2
437 + sizeof(struct block_extension_t);
438 if( NewFrame( p_demux ) != VLC_SUCCESS )
439 return VLC_ENOMEM;
440
441 /* Video ES */
442 es_format_Init( &fmt, VIDEO_ES, VLC_CODEC_I420 );
443 fmt.i_id = p_sys->i_id_video;
444 fmt.video.i_frame_rate = p_sys->i_frame_rate;
445 fmt.video.i_frame_rate_base = p_sys->i_frame_rate_base;
446 fmt.video.i_width = p_sys->i_width;
447 fmt.video.i_height = p_sys->i_height;
448 int i_aspect = p_sys->i_forced_aspect ? p_sys->i_forced_aspect
449 : p_sys->i_aspect;
450 fmt.video.i_sar_num = i_aspect * fmt.video.i_height
451 / fmt.video.i_width;
452 fmt.video.i_sar_den = VOUT_ASPECT_FACTOR;
453 p_sys->p_es_video = es_out_Add( p_demux->out, &fmt );
454
455 if ( p_sys->b_vbi && InitWSS( p_demux ) != VLC_SUCCESS )
456 p_sys->b_vbi = 0;
457
458 /* Teletext ES */
459 psz_parser = p_sys->psz_telx;
460 if ( psz_parser != NULL && *psz_parser )
461 {
462 if ( !p_sys->b_vbi )
463 {
464 msg_Warn( p_demux, "VBI is unsupported on this input stream" );
465 }
466 else
467 {
468 int i_id;
469 char *psz_next = strchr( psz_parser, '=' );
470 if ( psz_next != NULL )
471 {
472 *psz_next = '\0';
473 i_id = strtol( psz_parser, NULL, 0 );
474 psz_parser = psz_next + 1;
475 }
476 else
477 i_id = 0;
478
479 psz_next = strchr( psz_parser, '-' );
480 if ( psz_next != NULL )
481 *psz_next++ = '\0';
482
483 p_sys->i_telx_line = strtol( psz_parser, NULL, 0 ) - 1;
484 if ( psz_next != NULL )
485 p_sys->i_telx_count = strtol( psz_next, NULL, 0 )
486 - p_sys->i_telx_line - 1 + 1;
487 else
488 p_sys->i_telx_count = 1;
489
490 if ( InitTelx( p_demux ) == VLC_SUCCESS )
491 {
492 int i_dr_size = 0;
493 uint8_t *p_dr = NULL;
494
495 msg_Dbg( p_demux, "capturing VBI lines %d-%d and %d-%d",
496 p_sys->i_telx_line + 1,
497 p_sys->i_telx_line + 1 + p_sys->i_telx_count - 1,
498 p_sys->i_telx_line + 1 + 313,
499 p_sys->i_telx_line + 1 + 313
500 + p_sys->i_telx_count - 1 );
501
502 es_format_Init( &fmt, SPU_ES, VLC_CODEC_TELETEXT );
503 fmt.i_id = i_id;
504
505 /* Teletext language & type */
506 psz_parser = p_sys->psz_telx_lang;
507
508 while ( (psz_next = strchr( psz_parser, '=' )) != NULL )
509 {
510 int i_page;
511 *psz_next++ = '\0';
512 if ( !psz_next[0] || !psz_next[1] || !psz_next[2] )
513 break;
514 i_page = strtol( psz_parser, NULL, 0 );
515 i_dr_size += 5;
516 p_dr = realloc( p_dr, i_dr_size );
517 p_dr[i_dr_size - 5] = *psz_next++;
518 p_dr[i_dr_size - 4] = *psz_next++;
519 p_dr[i_dr_size - 3] = *psz_next++;
520 if ( *psz_next == '/' )
521 {
522 psz_next++;
523 p_dr[i_dr_size - 2] = strtol( psz_next, &psz_next, 0 )
524 << 3;
525 }
526 else /* subtitle for hearing impaired */
527 p_dr[i_dr_size - 2] = 0x5 << 3;
528 p_dr[i_dr_size - 2] |= (i_page / 100) & 0x7;
529 p_dr[i_dr_size - 1] = i_page % 100;
530
531 if ( *psz_next == ',' )
532 psz_next++;
533 psz_parser = psz_next;
534 }
535
536 fmt.i_extra = i_dr_size;
537 fmt.p_extra = p_dr;
538 p_sys->p_es_telx = es_out_Add( p_demux->out, &fmt );
539 }
540 else
541 p_sys->i_telx_count = 0;
542 }
543 }
544 return VLC_SUCCESS;
545}

547static void StopDecode( demux_t *p_demux )
548{
549 demux_sys_t *p_sys = p_demux->p_sys;
550
551 if ( p_sys->i_state != STATE_SYNC )
552 return;
553
554 free( p_sys->p_line_buffer );
555
556 block_Release( p_sys->p_current_picture );
557 p_sys->p_current_picture = NULL;
558 es_out_Del( p_demux->out, p_sys->p_es_video );
559
560 if ( p_sys->b_vbi )
561 {
562 free( p_sys->p_wss_buffer );
563 p_sys->p_wss_buffer = NULL;
564 vbi_raw_decoder_destroy( &p_sys->rd_wss );
565
566 if ( p_sys->p_es_telx )
567 {
568 es_out_Del( p_demux->out, p_sys->p_es_telx );
569 free( p_sys->p_telx_buffer );
570 p_sys->p_telx_buffer = NULL;
571 vbi_raw_decoder_destroy( &p_sys->rd_telx );
572 }
573 }
574
575 for ( int i = 0; i < MAX_AUDIOS; i++ )
576 {
577 sdi_audio_t *p_audio = &p_sys->p_audios[i];
578 if ( p_audio->i_group && p_audio->p_es != NULL )
579 {
580 es_out_Del( p_demux->out, p_audio->p_es );
581 p_audio->p_es = NULL;
582 free( p_audio->p_buffer );
583 p_audio->p_buffer = NULL;
584 }
585 }

588static void InitVideo( demux_t *p_demux )
589{
590 demux_sys_t *p_sys = p_demux->p_sys;
591 int i_total_width = (p_sys->i_anc_size + p_sys->i_active_size) * 4 / 5;
592 p_sys->i_width = (p_sys->i_active_size - 5) * 4 / 10;
593 if ( p_sys->i_nb_lines == 625 )
594 {
595 /* PAL */
596 p_sys->i_frame_rate = 25;
597 p_sys->i_frame_rate_base = 1;
598 p_sys->i_height = 576;
599 p_sys->i_aspect = 4 * VOUT_ASPECT_FACTOR / 3;
600 p_sys->b_hd = false;
601 }
602 else if ( p_sys->i_nb_lines == 525 )
603 {
604 /* NTSC */
605 p_sys->i_frame_rate = 30000;
606 p_sys->i_frame_rate_base = 1001;
607 p_sys->i_height = 480;
608 p_sys->i_aspect = 4 * VOUT_ASPECT_FACTOR / 3;
609 p_sys->b_hd = false;
610 }
611 else if ( p_sys->i_nb_lines == 1125 && i_total_width == 2640 )
612 {
613 /* 1080i50 or 1080p25 */
614 p_sys->i_frame_rate = 25;
615 p_sys->i_frame_rate_base = 1;
616 p_sys->i_height = 1080;
617 p_sys->i_aspect = 16 * VOUT_ASPECT_FACTOR / 9;
618 p_sys->b_hd = true;
619 }
620 else if ( p_sys->i_nb_lines == 1125 && i_total_width == 2200 )
621 {
622 /* 1080i60 or 1080p30 */
623 p_sys->i_frame_rate = 30000;
624 p_sys->i_frame_rate_base = 1001;
625 p_sys->i_height = 1080;
626 p_sys->i_aspect = 16 * VOUT_ASPECT_FACTOR / 9;
627 p_sys->b_hd = true;
628 }
629 else if ( p_sys->i_nb_lines == 750 && i_total_width == 1980 )
630 {
631 /* 720p50 */
632 p_sys->i_frame_rate = 50;
633 p_sys->i_frame_rate_base = 1;
634 p_sys->i_height = 720;
635 p_sys->i_aspect = 16 * VOUT_ASPECT_FACTOR / 9;
636 p_sys->b_hd = true;
637 }
638 else if ( p_sys->i_nb_lines == 750 && i_total_width == 1650 )
639 {
640 /* 720p60 */
641 p_sys->i_frame_rate = 60000;
642 p_sys->i_frame_rate_base = 1001;
643 p_sys->i_height = 720;
644 p_sys->i_aspect = 16 * VOUT_ASPECT_FACTOR / 9;
645 p_sys->b_hd = true;
646 }
647 else

718static void DecodeWSS( demux_t *p_demux )
719{
720 demux_sys_t *p_sys = p_demux->p_sys;
721 vbi_sliced p_sliced[1];
722
723 if ( vbi_raw_decode( &p_sys->rd_wss, p_sys->p_wss_buffer, p_sliced ) == 0 )
724 {
725 p_sys->i_aspect = 4 * VOUT_ASPECT_FACTOR / 3;
726 }
727 else
728 {
729 unsigned int i_old_aspect = p_sys->i_aspect;
730 uint8_t *p = p_sliced[0].data;
731 int i_aspect, i_parity;
732
733 i_aspect = p[0] & 15;
734 i_parity = i_aspect;
735 i_parity ^= i_parity >> 2;
736 i_parity ^= i_parity >> 1;
737 i_aspect &= 7;
738
739 if ( !(i_parity & 1) )
740 msg_Warn( p_demux, "WSS parity error" );
741 else if ( i_aspect == 7 )
742 p_sys->i_aspect = 16 * VOUT_ASPECT_FACTOR / 9;
743 else
744 p_sys->i_aspect = 4 * VOUT_ASPECT_FACTOR / 3;
745
746 if ( p_sys->i_aspect != i_old_aspect )
747 msg_Dbg( p_demux, "new WSS information (ra=%x md=%x cod=%x hlp=%x rvd=%x sub=%x pos=%x srd=%x c=%x cp=%x)",
748 i_aspect, (p[0] & 0x10) >> 4, (p[0] & 0x20) >> 5,
749 (p[0] & 0x40) >> 6, (p[0] & 0x80) >> 7, p[1] & 0x01,
750 (p[1] >> 1) & 3, (p[1] & 0x08) >> 3, (p[1] & 0x10) >> 4,
751 (p[1] & 0x20) >> 5 );
752 }

791static int DecodeTelx( demux_t *p_demux )
792{
793 demux_sys_t *p_sys = p_demux->p_sys;
794 vbi_sliced p_sliced[p_sys->i_telx_count * 2];
795 int i_nb_slices = vbi_raw_decode( &p_sys->rd_telx, p_sys->p_telx_buffer,
796 p_sliced );
797
798 if ( i_nb_slices )
799 {
800 /* 3, 7, 11, 15, etc. */
801 int i_nb_slices_rounded = 3 + (i_nb_slices / 4) * 4;
802 int i;
803 uint8_t *p;
804 block_t *p_block = block_Alloc( 1 + i_nb_slices_rounded * 46 );
805 if( unlikely( !p_block ) )
806 return VLC_ENOMEM;
807 p_block->p_buffer[0] = 0x10; /* FIXME ? data_identifier */
808 p = p_block->p_buffer + 1;
809
810 for ( i = 0; i < i_nb_slices; i++ )
811 {
812 int i_line = p_sliced[i].line;
813 p[0] = 0x3; /* FIXME data_unit_id == subtitles */
814 p[1] = 0x2c; /* data_unit_length */
815 /* reserved | field_parity (kind of inverted) | line */
816 p[2] = 0xc0 | (i_line > 313 ? 0 : 0x20) | (i_line % 313);
817 p[3] = 0xe4; /* framing code */
818 for ( int j = 0; j < 42; j++ )
819 p[4 + j] = vbi_rev8( p_sliced[i].data[j] );
820 p += 46;
821 }
822
823 /* Let's stuff */
824 for ( ; i < i_nb_slices_rounded; i++ )
825 {
826 p[0] = 0xff;
827 p[1] = 0x2c;
828 memset( p + 2, 0xff, 44 );
829 p += 46;
830 }
831
832 p_block->i_dts = p_block->i_pts = p_sys->i_next_date;
833 es_out_Send( p_demux->out, p_sys->p_es_telx, p_block );
834 }
835 return VLC_SUCCESS;
836}

873/* Fast and efficient linear resampling routine */
874static void ResampleAudio( int16_t *p_out, int16_t *p_in,
875 unsigned int i_out, unsigned int i_in )
876{
877 unsigned int i_remainder = 0;
878 float f_last_sample = (float)*p_in / 32768.0;
879
880 *p_out = *p_in;
881 p_out += 2;
882 p_in += 2;
883
884 for ( unsigned int i = 1; i < i_in; i++ )
885 {
886 float f_in = (float)*p_in / 32768.0;
887 while ( i_remainder < i_out )
888 {
889 float f_out = f_last_sample;
890 f_out += (f_in - f_last_sample) * i_remainder / i_out;
891 if ( f_out >= 1.0 ) *p_out = 32767;
892 else if ( f_out < -1.0 ) *p_out = -32768;
893 else *p_out = f_out * 32768.0;
894 p_out += 2;
895 i_remainder += i_in;
896 }
897
898 f_last_sample = f_in;
899 p_in += 2;
900 i_remainder -= i_out;
901 }
902}

904static int DecodeAudio( demux_t *p_demux, sdi_audio_t *p_audio )
905{
906 demux_sys_t *p_sys = p_demux->p_sys;
907 block_t *p_block;
908 int16_t *p_output;
909
910 if ( p_audio->p_buffer == NULL )
911 return VLC_EGENERIC;
912 if ( !p_audio->i_left_samples && !p_audio->i_right_samples )
913 {
914 msg_Warn( p_demux, "no audio %u/%u", p_audio->i_group,
915 p_audio->i_pair );
916 return VLC_EGENERIC;
917 }
918 if ( p_audio->i_left_samples <
919 (float)p_audio->i_nb_samples * (1. - SAMPLERATE_TOLERANCE) ||
920 p_audio->i_left_samples >
921 (float)p_audio->i_nb_samples * (1. + SAMPLERATE_TOLERANCE) )
922 {
923 msg_Warn( p_demux,
924 "left samplerate out of tolerance for audio %u/%u (%u vs. %u)",
925 p_audio->i_group, p_audio->i_pair,
926 p_audio->i_left_samples, p_audio->i_nb_samples );
927 return VLC_EGENERIC;
928 }
929 if ( p_audio->i_right_samples <
930 (float)p_audio->i_nb_samples * (1. - SAMPLERATE_TOLERANCE) ||
931 p_audio->i_right_samples >
932 (float)p_audio->i_nb_samples * (1. + SAMPLERATE_TOLERANCE) )
933 {
934 msg_Warn( p_demux,
935 "right samplerate out of tolerance for audio %u/%u (%u vs. %u)",
936 p_audio->i_group, p_audio->i_pair,
937 p_audio->i_right_samples, p_audio->i_nb_samples );
938 return VLC_EGENERIC;
939 }
940
941 p_block = block_Alloc( p_audio->i_nb_samples * sizeof(int16_t) * 2 );
942 if( unlikely( !p_block ) )
943 return VLC_ENOMEM;
944 p_block->i_dts = p_block->i_pts = p_sys->i_next_date
945 + (mtime_t)p_audio->i_delay * INT64_C(1000000) / p_audio->i_rate;
946 p_output = (int16_t *)p_block->p_buffer;
947
948 if ( p_audio->i_left_samples == p_audio->i_nb_samples &&
949 p_audio->i_right_samples == p_audio->i_nb_samples )
950 memcpy( p_output, p_audio->p_buffer,
951 p_audio->i_nb_samples * sizeof(int16_t) * 2 );
952 else
953 {
954 ResampleAudio( p_output, p_audio->p_buffer,
955 p_audio->i_nb_samples, p_audio->i_left_samples );
956
957 ResampleAudio( p_output + 1, p_audio->p_buffer + 1,
958 p_audio->i_nb_samples, p_audio->i_right_samples );
959 }
960
961 es_out_Send( p_demux->out, p_audio->p_es, p_block );
962 return VLC_SUCCESS;
963}

1207 {
1208 const uint16_t *x = p_anc;
1209
1210 /* SMPTE 272M says that when parsing a frame, if an audio config
1211 * structure is present we will encounter it first. Otherwise
1212 * it is assumed to be 48 kHz. */
1213 if ( p_audio->p_es == NULL )
1214 {
1215 p_audio->i_rate = 48000;
1216 p_audio->i_delay = 0;
1217 if( InitAudio( p_demux, p_audio ) != VLC_SUCCESS )
1218 return;
1219 }
1220
1221 if ( i_block_number )
1222 {
1223 if ( p_audio->i_block_number + 1 != i_block_number )
1224 msg_Warn( p_demux,
1225 "audio data block discontinuity (%"PRIu8"->%"PRIu8") for group %"PRIu8,
1226 p_audio->i_block_number, i_block_number,
1227 i_group );
1228 if ( i_block_number == 0xff )
1229 p_audio->i_block_number = 0;
1230 else
1231 p_audio->i_block_number = i_block_number;
1232 }
1233
1234 while ( x < p_anc + i_data_count )
1235 {
1236 if ( ((*x & 0x4) && p_audio->i_pair == 2)
1237 || (!(*x & 0x4) && p_audio->i_pair == 1) )
1238 {
1239 uint32_t i_tmp = (uint32_t)((x[0] & 0x1f1) >> 3)
1240 | ((x[1] & 0x1ff) << 6)
1241 | ((x[2] & 0x1f) << 15);
1242 int32_t i_sample;
1243 if ( x[2] & 0x10 )
1244 i_sample = i_tmp | 0xfff00000;
1245 else
1246 i_sample = i_tmp;
1247
1248 if ( x[0] & 0x2 )
1249 {
1250 if ( p_audio->i_right_samples < p_audio->i_max_samples )
1251 p_audio->p_buffer[2 * p_audio->i_right_samples
1252 + 1] = (i_sample + 8) / 16;
1253 p_audio->i_right_samples++;
1254 }
1255 else
1256 {
1257 if ( p_audio->i_left_samples < p_audio->i_max_samples )
1258 p_audio->p_buffer[2 * p_audio->i_left_samples]
1259 = (i_sample + 8) / 16;
1260 p_audio->i_left_samples++;
1261 }
1262 }
1263 x += 3;
1264 }
1265 }
1266 }
1267}

1269static void HandleAudioConfig( demux_t *p_demux, const uint16_t *p_anc,
1270 uint8_t i_data_count, uint8_t i_group )
1271{
1272 demux_sys_t *p_sys = p_demux->p_sys;
1273
1274 if ( i_data_count != 18 )
1275 {
1276 msg_Warn( p_demux, "malformed audio config for group %u", i_group );
1277 return;
1278 }
1279
1280 for ( int i = 0; i < MAX_AUDIOS; i++ )
1281 {
1282 sdi_audio_t *p_audio = &p_sys->p_audios[i];
1283 if ( p_audio->i_group == i_group && p_audio->p_es == NULL )
1284 {
1285 unsigned int i_rate;
1286
1287 if ( p_audio->i_pair == 2 )
1288 {
1289 i_rate = (p_anc[2] & 0xe0) >> 5;
1290 if ( p_anc[7] & 0x1 )
1291 {
1292 uint32_t i_tmp = ((p_anc[7] & 0x1fe) >> 1)
1293 | ((p_anc[8] & 0x1ff) << 8)
1294 | ((p_anc[9] & 0x1ff) << 17);
1295 if ( p_anc[9] & 0x80 )
1296 p_audio->i_delay = i_tmp | 0xfc000000;
1297 else
1298 p_audio->i_delay = i_tmp;
1299 }
1300 if ( p_anc[13] & 0x1 )
1301 msg_Warn( p_demux, "asymetric audio is not supported" );
1302 }
1303 else
1304 {
1305 i_rate = (p_anc[2] & 0xe) >> 1;
1306 if ( p_anc[4] & 0x1 )
1307 {
1308 uint32_t i_tmp = ((p_anc[4] & 0x1fe) >> 1)
1309 | ((p_anc[5] & 0x1ff) << 8)
1310 | ((p_anc[6] & 0x1ff) << 17);
1311 if ( p_anc[6] & 0x80 )
1312 p_audio->i_delay = i_tmp | 0xfc000000;
1313 else
1314 p_audio->i_delay = i_tmp;
1315 }
1316 if ( p_anc[10] & 0x1 )
1317 msg_Warn( p_demux, "asymetric audio is not supported" );
1318 }
1319
1320 switch ( i_rate )
1321 {
1322 case 0: p_audio->i_rate = 48000; break;
1323 case 1: p_audio->i_rate = 44100; break;
1324 case 2: p_audio->i_rate = 32000; break;
1325 default:
1326 msg_Warn( p_demux, "unknown rate for audio %u/%u (%u)",
1327 i_group, p_sys->p_audios[i].i_pair, i_rate );
1328 continue;
1329 }
1330
1331 if( InitAudio( p_demux, p_audio ) != VLC_SUCCESS )
1332 return;
1333 }
1334 }
1335}

1337/*
1338 * Ancillary packet structure:
1339 * byte 0: Ancillary Data Flag (0)
1340 * byte 1: Ancillary Data Flag (0x3ff)
1341 * byte 2: Ancillary Data Flag (0x3ff)
1342 * byte 3: Data ID (2 high order bits = parity)
1343 * byte 4: Data Block Number 1-255 or 0=unknown (if DID < 0x80)
1344 * or Secondary Data ID (if DID >= 0x80)
1345 * byte 5: Data Count (10 bits)
1346 * byte 6+DC: Checksum
1347 */
1348static void HandleAncillary( demux_t *p_demux, const uint16_t *p_anc,
1349 unsigned int i_size )
1350{
1351 uint8_t i_data_count;
1352
1353 if ( i_size < 7
1354 || p_anc[0] != 0x0 || p_anc[1] != 0x3ff || p_anc[2] != 0x3ff )
1355 return;
1356
1357 i_data_count = p_anc[5] & 0xff;
1358 if ( i_size - 6 < i_data_count )
1359 {
1360 msg_Warn( p_demux, "malformed ancillary packet (size %u > %u)",
1361 i_data_count, i_size - 6 );
1362 return;
1363 }
1364
1365 switch ( p_anc[3] ) /* Data ID */
1366 {
1367 case 0x2ff:
1368 HandleAudioData( p_demux, p_anc + 6, i_data_count, 1, p_anc[4] & 0xff );
1369 break;
1370 case 0x1fd:
1371 HandleAudioData( p_demux, p_anc + 6, i_data_count, 2, p_anc[4] & 0xff );
1372 break;
1373 case 0x1fb:
1374 HandleAudioData( p_demux, p_anc + 6, i_data_count, 3, p_anc[4] & 0xff );
1375 break;
1376 case 0x2f9:
1377 HandleAudioData( p_demux, p_anc + 6, i_data_count, 4, p_anc[4] & 0xff );
1378 break;
1379
1380 case 0x1ef:
1381 HandleAudioConfig( p_demux, p_anc + 6, i_data_count, 1 );
1382 break;
1383 case 0x2ee:
1384 HandleAudioConfig( p_demux, p_anc + 6, i_data_count, 2 );
1385 break;
1386 case 0x2ed:
1387 HandleAudioConfig( p_demux, p_anc + 6, i_data_count, 3 );
1388 break;
1389 case 0x1ec:
1390 HandleAudioConfig( p_demux, p_anc + 6, i_data_count, 4 );
1391 break;
1392
1393 /* Extended data packets, same order */
1394 case 0x1fe:
1395 case 0x2fc:
1396 case 0x2fa:
1397 case 0x1f8:
1398
1399 default:
1400 break;
1401
1402 case 0x88: /* non-conforming ANC packet */
1403 p_anc += 7;
1404 i_size -= 7;
1405 while ( i_size >= 7 && (p_anc[0] != 0x0 || p_anc[1] != 0x3ff
1406 || p_anc[2] != 0x3ff) )
1407 {
1408 p_anc++;
1409 i_size--;
1410 }
1411 if ( i_size >= 7 )
1412 HandleAncillary( p_demux, p_anc, i_size );
1413 return;
1414 }
1415
1416 return HandleAncillary( p_demux, p_anc + i_data_count + 7,
1417 i_size - i_data_count - 7 );
1418
1419}

1421static int HandleSDBuffer( demux_t *p_demux, uint8_t *p_buffer,
1422 unsigned int i_buffer_size )
1423{
1424 demux_sys_t *p_sys = p_demux->p_sys;
1425 const uint8_t *p_parser = p_buffer;
1426 const uint8_t *p_end = p_parser + i_buffer_size;
1427 const uint8_t *p_line;
1428
1429 if ( p_sys->i_state != STATE_SYNC
1430 && p_sys->i_last_state_change < mdate() - RESYNC_TIMEOUT )
1431 {
1432 p_sys->i_state = STATE_NOSYNC;
1433 p_sys->i_last_state_change = mdate();
1434 return VLC_EGENERIC;
1435 }
1436
1437 switch ( p_sys->i_state )
1438 {
1439 case STATE_NOSYNC:
1440 default:
1441 p_parser = FindReferenceCode( FIELD_2_VBLANK_SAV, p_parser, p_end );
1442 if ( p_parser == NULL )
1443 break;
1444 p_sys->i_state = STATE_STARTSYNC;
1445 p_sys->i_last_state_change = mdate();
1446
1447 case STATE_STARTSYNC:
1448 p_parser = FindReferenceCode( FIELD_1_VBLANK_EAV, p_parser, p_end );
1449 if ( p_parser == NULL )
1450 break;
1451 p_sys->i_anc_size = 0;
1452 p_sys->i_state = STATE_ANCSYNC;
1453 p_sys->i_last_state_change = mdate();
1454
1455 case STATE_ANCSYNC:
1456 p_parser = CountReference( &p_sys->i_anc_size,
1457 FIELD_1_VBLANK_SAV, p_parser, p_end );
1458 if ( p_parser == NULL )
1459 break;
1460 p_sys->i_active_size = 0;
1461 p_sys->i_state = STATE_LINESYNC;
1462 p_sys->i_last_state_change = mdate();
1463
1464 case STATE_LINESYNC:
1465 p_parser = CountReference( &p_sys->i_active_size,
1466 FIELD_1_VBLANK_EAV, p_parser, p_end );
1467 if ( p_parser == NULL )
1468 break;
1469 p_sys->i_picture_size = p_sys->i_anc_size + p_sys->i_active_size;
1470 p_sys->i_state = STATE_ACTIVESYNC;
1471 p_sys->i_last_state_change = mdate();
1472
1473 case STATE_ACTIVESYNC:
1474 p_parser = CountReference( &p_sys->i_picture_size,
1475 FIELD_1_ACTIVE_EAV, p_parser, p_end );
1476 if ( p_parser == NULL )
1477 break;
1478 p_sys->i_line_offset = p_sys->i_picture_size
1479 / (p_sys->i_anc_size + p_sys->i_active_size);
1480 p_sys->i_state = STATE_VBLANKSYNC;
1481 p_sys->i_last_state_change = mdate();
1482
1483 case STATE_VBLANKSYNC:
1484 p_parser = CountReference( &p_sys->i_picture_size,
1485 FIELD_2_ACTIVE_EAV, p_parser, p_end );
1486 if ( p_parser == NULL )
1487 break;
1488 p_sys->i_state = STATE_PICSYNC;
1489 p_sys->i_last_state_change = mdate();
1490
1491 case STATE_PICSYNC:
1492 p_parser = CountReference( &p_sys->i_picture_size,
1493 FIELD_1_VBLANK_EAV, p_parser, p_end );
1494 if ( p_parser == NULL )
1495 break;
1496
1497 if ( p_sys->i_picture_size
1498 % (p_sys->i_anc_size + p_sys->i_active_size) )
1499 {
1500 msg_Warn( p_demux, "wrong picture size (anc=%d active=%d total=%d offset=%d), syncing",
1501 p_sys->i_anc_size, p_sys->i_active_size,
1502 p_sys->i_picture_size, p_sys->i_line_offset + 1 );
1503 p_sys->i_state = STATE_NOSYNC;
1504 p_sys->i_last_state_change = mdate();
1505 break;
1506 }
1507
1508 p_sys->i_nb_lines = p_sys->i_picture_size
1509 / (p_sys->i_anc_size + p_sys->i_active_size);
1510 InitVideo( p_demux );
1511 msg_Dbg( p_demux,
1512 "acquired sync, anc=%d active=%d lines=%d offset=%d",
1513 p_sys->i_anc_size, p_sys->i_active_size,
1514 p_sys->i_nb_lines, p_sys->i_line_offset + 1 );
1515 p_sys->i_state = STATE_SYNC;
1516 if( StartDecode( p_demux ) != VLC_SUCCESS )
1517 {
1518 StopDecode( p_demux );
1519 return VLC_ENOMEM;
1520 }
1521 p_sys->i_current_line = 0;
1522 p_sys->p_line_buffer = malloc( p_sys->i_anc_size
1523 + p_sys->i_active_size );
1524 if( !p_sys->p_line_buffer )
1525 {
1526 StopDecode( p_demux );
1527 return VLC_ENOMEM;
1528 }
1529 p_sys->i_line_buffer = 0;
1530
1531 case STATE_SYNC:
1532 while ( (p_line = GetLine( p_demux, &p_parser, p_end )) != NULL )
1533 {
1534 bool b_field = p_sys->b_hd ? false :
1535 (p_sys->i_current_line >= p_sys->i_nb_lines / 2);
1536 unsigned int i_field_height = p_sys->b_hd ? p_sys->i_height :
1537 p_sys->i_height / 2;
1538 unsigned int i_field_line = b_field ?
1539 p_sys->i_current_line - (p_sys->i_nb_lines + 1) / 2 :
1540 p_sys->i_current_line;
1541 bool b_vbi = i_field_line < p_sys->i_line_offset ||
1542 i_field_line >= p_sys->i_line_offset + i_field_height;
1543 unsigned int anc = p_sys->i_anc_size;
1544
1545 if ( p_line[0] != 0xff || p_line[1] != 0x3
1546 || p_line[2] != 0x0 || p_line[3] != 0x0
1547 || p_line[anc+0] != 0xff || p_line[anc+1] != 0x3
1548 || p_line[anc+2] != 0x0 || p_line[anc+3] != 0x0
1549 || (!b_field && b_vbi &&
1550 (p_line[4] != FIELD_1_VBLANK_EAV ||
1551 p_line[anc+4] != FIELD_1_VBLANK_SAV))
1552 || (!b_field && !b_vbi &&
1553 (p_line[4] != FIELD_1_ACTIVE_EAV ||
1554 p_line[anc+4] != FIELD_1_ACTIVE_SAV))
1555 || (b_field && b_vbi &&
1556 (p_line[4] != FIELD_2_VBLANK_EAV ||
1557 p_line[anc+4] != FIELD_2_VBLANK_SAV))
1558 || (b_field && !b_vbi &&
1559 (p_line[4] != FIELD_2_ACTIVE_EAV ||
1560 p_line[anc+4] != FIELD_2_ACTIVE_SAV)) )
1561 {
1562 msg_Warn( p_demux, "lost sync line:%u SAV:%x EAV:%x",
1563 p_sys->i_current_line + 1, p_line[4], p_line[anc+4] );
1564 StopDecode( p_demux );
1565 p_sys->i_state = STATE_NOSYNC;
1566 p_sys->i_last_state_change = mdate();
1567 break;
1568 }
1569
1570 if ( HasAncillary( p_line + 5 ) )
1571 {
1572 /* HANC */
1573 unsigned int i_anc_words = (p_sys->i_anc_size - 5) * 4 / 5;
1574 uint16_t p_anc[i_anc_words];
1575 UnpackAnc( p_line + 5, p_sys->i_anc_size - 5, p_anc );
1576 HandleAncillary( p_demux, p_anc, i_anc_words );
1577 }
1578
1579 if ( !b_vbi )
1580 {
1581 unsigned int i_active_field_line = i_field_line
1582 - p_sys->i_line_offset;
1583 unsigned int i_active_line = b_field
1584 + i_active_field_line * 2;
1585 if ( !(i_active_field_line % 2) && !b_field )
1586 Unpack01( p_line + anc + 5, p_sys->i_active_size - 5,
1587 p_sys->p_y + p_sys->i_width * i_active_line,
1588 p_sys->p_u + (p_sys->i_width / 2)
1589 * (i_active_line / 2),
1590 p_sys->p_v + (p_sys->i_width / 2)
1591 * (i_active_line / 2) );
1592 else if ( !(i_active_field_line % 2) )
1593 Unpack01( p_line + anc + 5, p_sys->i_active_size - 5,
1594 p_sys->p_y + p_sys->i_width * i_active_line,
1595 p_sys->p_u + (p_sys->i_width / 2)
1596 * (i_active_line / 2 + 1),
1597 p_sys->p_v + (p_sys->i_width / 2)
1598 * (i_active_line / 2 + 1) );
1599 else if ( !b_field )
1600 Unpack2( p_line + anc + 5, p_sys->i_active_size - 5,
1601 p_sys->p_y + p_sys->i_width * i_active_line,
1602 p_sys->p_u + (p_sys->i_width / 2)
1603 * (i_active_line / 2 - 1),
1604 p_sys->p_v + (p_sys->i_width / 2)
1605 * (i_active_line / 2 - 1) );
1606 else
1607 Unpack3( p_line + anc + 5, p_sys->i_active_size - 5,
1608 p_sys->p_y + p_sys->i_width * i_active_line,
1609 p_sys->p_u + (p_sys->i_width / 2)
1610 * (i_active_line / 2),
1611 p_sys->p_v + (p_sys->i_width / 2)
1612 * (i_active_line / 2) );
1613
1614 if ( p_sys->b_vbi && p_sys->i_height == 576
1615 && p_sys->i_current_line == p_sys->i_line_offset )
1616 {
1617 /* Line 23 is half VBI, half active */
1618 UnpackVBI( p_line + anc + 5, p_sys->i_active_size - 5,
1619 p_sys->p_wss_buffer );
1620 }
1621 }
1622 else if ( p_sys->b_vbi && p_sys->i_telx_count &&
1623 i_field_line >= p_sys->i_telx_line &&
1624 i_field_line < p_sys->i_telx_line
1625 + p_sys->i_telx_count )
1626 {
1627 UnpackVBI( p_line + anc + 5, p_sys->i_active_size - 5,
1628 &p_sys->p_telx_buffer[(i_field_line
1629 - p_sys->i_telx_line + b_field * p_sys->i_telx_count)
1630 * p_sys->i_width * 2] );
1631 }
1632 else if ( b_vbi && HasAncillary( p_line + anc + 5 ) )
1633 {
1634 /* VANC */
1635 unsigned int i_anc_words = (p_sys->i_active_size - 5) * 4 / 5;
1636 uint16_t p_anc[i_anc_words];
1637 UnpackAnc( p_line + 5, p_sys->i_active_size - 5,
1638 p_anc );
1639 HandleAncillary( p_demux, p_anc, i_anc_words );
1640 }
1641
1642 p_sys->i_current_line++;
1643 if ( p_sys->i_current_line == p_sys->i_nb_lines )
1644 {
1645 p_sys->i_current_line %= p_sys->i_nb_lines;
1646 if( DecodeFrame( p_demux ) != VLC_SUCCESS )
1647 return VLC_EGENERIC;
1648 }
1649 }
1650 break;
1651 }
1652
1653 return VLC_SUCCESS;
1654}

1706static int InitCapture( demux_t *p_demux )
1707{

1709 const int i_page_size = getpagesize();
1710 unsigned int i_bufmemsize;
1711 int i_ret;
1712 char psz_dev[MAXLEN];
1713
1714 /* 10-bit mode or nothing */
1715 if ( WriteULSysfs( SDI_MODE_FILE, p_sys->i_link, SDI_CTL_MODE_10BIT ) < 0 )
1716 {
1717 msg_Err( p_demux, "couldn't write file " SDI_MODE_FILE, p_sys->i_link );
1718 return VLC_EGENERIC;
1719 }
1720
1721 if ( (i_ret = ReadULSysfs( SDI_BUFFERS_FILE, p_sys->i_link )) < 0 )
1722 {
1723 msg_Err( p_demux, "couldn't read file " SDI_BUFFERS_FILE,
1724 p_sys->i_link );
1725 return VLC_EGENERIC;
1726 }
1727 p_sys->i_buffers = i_ret;
1728 p_sys->i_current_buffer = 0;
1729
1730 if ( (i_ret = ReadULSysfs( SDI_BUFSIZE_FILE, p_sys->i_link )) < 0 )
1731 {
1732 msg_Err( p_demux, "couldn't read file " SDI_BUFSIZE_FILE,
1733 p_sys->i_link );
1734 return VLC_EGENERIC;
1735 }
1736 p_sys->i_buffer_size = i_ret;
1737 if ( p_sys->i_buffer_size % 20 )
1738 {
1739 msg_Err( p_demux, "buffer size must be a multiple of 20" );
1740 return VLC_EGENERIC;
1741 }
1742
1743 snprintf( psz_dev, sizeof(psz_dev) - 1, SDI_DEVICE, p_sys->i_link );
1744 if ( (p_sys->i_fd = vlc_open( psz_dev, O_RDONLY ) ) < 0 )
1745 {
1746 msg_Err( p_demux, "couldn't open device %s", psz_dev );
1747 return VLC_EGENERIC;
1748 }
1749
1750 i_bufmemsize = ((p_sys->i_buffer_size + i_page_size - 1) / i_page_size)
1751 * i_page_size;
1752 p_sys->pp_buffers = malloc( p_sys->i_buffers * sizeof(uint8_t *) );
1753 if( !p_sys->pp_buffers )
1754 return VLC_ENOMEM;
1755
1756 for ( unsigned int i = 0; i < p_sys->i_buffers; i++ )
1757 {
1758 if ( (p_sys->pp_buffers[i] = mmap( NULL, p_sys->i_buffer_size,
1759 PROT_READ, MAP_SHARED, p_sys->i_fd,
1760 i * i_bufmemsize )) == MAP_FAILED )
1761 {
1762 msg_Err( p_demux, "couldn't mmap(%d): %s", i,
1763 vlc_strerror_c(errno) );

1765 return VLC_EGENERIC;
1766 }

1783static int Capture( demux_t *p_demux )
1784{
1785 demux_sys_t *p_sys = p_demux->p_sys;
1786 struct pollfd pfd;
1787
1788 pfd.fd = p_sys->i_fd;
1789 pfd.events = POLLIN | POLLPRI;
1790
1791 if ( poll( &pfd, 1, READ_TIMEOUT ) < 0 )
1792 {
1793 msg_Warn( p_demux, "couldn't poll(): %s", vlc_strerror_c(errno) );
1794 return VLC_EGENERIC;
1795 }
1796
1797 if ( pfd.revents & POLLPRI )
1798 {
1799 unsigned int i_val;
1800
1801 if ( ioctl( p_sys->i_fd, SDI_IOC_RXGETEVENTS, &i_val ) < 0 )
1802 msg_Warn( p_demux, "couldn't SDI_IOC_RXGETEVENTS: %s",
1803 vlc_strerror_c(errno) );
1804 else
1805 {
1806 if ( i_val & SDI_EVENT_RX_BUFFER )
1807 msg_Warn( p_demux, "driver receive buffer queue overrun" );
1808 if ( i_val & SDI_EVENT_RX_FIFO )
1809 msg_Warn( p_demux, "onboard receive FIFO overrun");
1810 if ( i_val & SDI_EVENT_RX_CARRIER )
1811 msg_Warn( p_demux, "carrier status change");
1812 }
1813
1814 p_sys->i_next_date += CLOCK_GAP;
1815 }
1816
1817 if ( pfd.revents & POLLIN )
1818 {
1819 int i_ret;
1820
1821 if ( ioctl( p_sys->i_fd, SDI_IOC_DQBUF, p_sys->i_current_buffer ) < 0 )
1822 {
1823 msg_Warn( p_demux, "couldn't SDI_IOC_DQBUF: %s",
1824 vlc_strerror_c(errno) );
1825 return VLC_EGENERIC;
1826 }
1827
1828 i_ret = HandleSDBuffer( p_demux,
1829 p_sys->pp_buffers[p_sys->i_current_buffer],
1830 p_sys->i_buffer_size );
1831
1832 if ( ioctl( p_sys->i_fd, SDI_IOC_QBUF, p_sys->i_current_buffer ) < 0 )
1833 {
1834 msg_Warn( p_demux, "couldn't SDI_IOC_QBUF: %s",
1835 vlc_strerror_c(errno) );
1836 return VLC_EGENERIC;
1837 }
1838
1839 if ( i_ret == VLC_SUCCESS )
1840 {
1841 p_sys->i_current_buffer++;
1842 p_sys->i_current_buffer %= p_sys->i_buffers;
1843 }
1844 else
1845 {
1846 /* Reference codes do not start on a multiple of 5. This sometimes
1847 * happen. We really don't want to allow this. */
1848 msg_Warn( p_demux, "resetting board" );
1849 CloseCapture( p_demux );
1850 InitCapture( p_demux );
1851 }
1852 }
1853
1854 return VLC_SUCCESS;
1855}
