166/*****************************************************************************
167 * DemuxOpen:
168 *****************************************************************************/
169static int Open( vlc_object_t *p_this )
170{
171 demux_t *p_demux = (demux_t *)p_this;
172 demux_sys_t *p_sys;
173 char *psz_parser;
174
175 /* Fill p_demux field */
176 p_demux->pf_demux = Demux;
177 p_demux->pf_control = Control;
178 p_demux->p_sys = p_sys = calloc( 1, sizeof( demux_sys_t ) );
179 if( unlikely(!p_sys) )
180 return VLC_ENOMEM;
181
182 /* HDSDI AR */
183 char *psz_ar = var_InheritString( p_demux, "linsys-hdsdi-aspect-ratio" );
184 if ( psz_ar != NULL )
185 {
186 psz_parser = strchr( psz_ar, ':' );
187 if ( psz_parser )
188 {
189 *psz_parser++ = '\0';
190 p_sys->i_forced_aspect = p_sys->i_aspect =
191 strtol( psz_ar, NULL, 0 ) * VOUT_ASPECT_FACTOR
192 / strtol( psz_parser, NULL, 0 );
193 }
194 else
195 p_sys->i_forced_aspect = 0;
196 free( psz_ar );
197 }
198
199 /* */
200 p_sys->i_id_video = var_InheritInteger( p_demux, "linsys-hdsdi-id-video" );
201
202 /* Audio ES */
203 char *psz_string = psz_parser = var_InheritString( p_demux,
204 "linsys-hdsdi-audio" );
205 int i = 0;
206 p_sys->i_max_channel = -1;
207
208 while ( psz_parser != NULL && *psz_parser )
209 {
210 int i_id, i_group, i_pair;
211 char *psz_next = strchr( psz_parser, '=' );
212 if ( psz_next != NULL )
213 {
214 *psz_next = '\0';
215 i_id = strtol( psz_parser, NULL, 0 );
216 psz_parser = psz_next + 1;
217 }
218 else
219 i_id = 0;
220
221 psz_next = strchr( psz_parser, ':' );
222 if ( psz_next != NULL )
223 {
224 *psz_next = '\0';
225 psz_next++;
226 }
227
228 if ( sscanf( psz_parser, "%d,%d", &i_group, &i_pair ) == 2 )
229 {
230 p_sys->p_audios[i].i_channel = (i_group - 1) * 2 + (i_pair - 1);
231 if ( p_sys->p_audios[i].i_channel > p_sys->i_max_channel )
232 p_sys->i_max_channel = p_sys->p_audios[i].i_channel;
233 p_sys->p_audios[i].i_id = i_id;
234 i++;
235 }
236 else
237 msg_Warn( p_demux, "malformed audio configuration (%s)",
238 psz_parser );
239
240 psz_parser = psz_next;
241 }
242 free( psz_string );
243 for ( ; i < MAX_AUDIOS; i++ )
244 p_sys->p_audios[i].i_channel = -1;
245
246
247 p_sys->i_link = var_InheritInteger( p_demux, "linsys-hdsdi-link" );
248
249 if( InitCapture( p_demux ) != VLC_SUCCESS )
250 {
251 free( p_sys );
252 return VLC_EGENERIC;
253 }
254
255 return VLC_SUCCESS;
256}

308/*****************************************************************************
309 * HDSDI syntax parsing stuff
310 *****************************************************************************/
311#define U (uint16_t)(p_line[0])
312#define Y1 (uint16_t)(p_line[1])
313#define V (uint16_t)(p_line[2])
314#define Y2 (uint16_t)(p_line[3])
315
316/* For lines 0 [4] or 1 [4] */
317static void Unpack01( const uint8_t *p_line, unsigned int i_size,
318 uint8_t *p_y, uint8_t *p_u, uint8_t *p_v )
319{
320 const uint8_t *p_end = p_line + i_size;
321
322 while ( p_line < p_end )
323 {
324 *p_u++ = U;
325 *p_y++ = Y1;
326 *p_v++ = V;
327 *p_y++ = Y2;
328 p_line += 4;
329 }
330}
331
332/* For lines 2 [4] */
333static void Unpack2( const uint8_t *p_line, unsigned int i_size,
334 uint8_t *p_y, uint8_t *p_u, uint8_t *p_v )
335{
336 const uint8_t *p_end = p_line + i_size;
337
338 while ( p_line < p_end )
339 {
340 uint16_t tmp;
341 tmp = 3 * *p_u;
342 tmp += U;
343 *p_u++ = tmp / 4;
344 *p_y++ = Y1;
345 tmp = 3 * *p_v;
346 tmp += V;
347 *p_v++ = tmp / 4;
348 *p_y++ = Y2;
349 p_line += 4;
350 }
351}
352
353/* For lines 3 [4] */
354static void Unpack3( const uint8_t *p_line, unsigned int i_size,
355 uint8_t *p_y, uint8_t *p_u, uint8_t *p_v )
356{
357 const uint8_t *p_end = p_line + i_size;
358
359 while ( p_line < p_end )
360 {
361 uint16_t tmp;
362 tmp = *p_u;
363 tmp += 3 * U;
364 *p_u++ = tmp / 4;
365 *p_y++ = Y1;
366 tmp = *p_v;
367 tmp += 3 * V;
368 *p_v++ = tmp / 4;
369 *p_y++ = Y2;
370 p_line += 4;
371 }
372}
373
374#undef U
375#undef Y1
376#undef V
377#undef Y2

379static void SparseCopy( int16_t *p_dest, const int16_t *p_src,
380 size_t i_nb_samples, size_t i_offset, size_t i_stride )
381{
382 for ( size_t i = 0; i < i_nb_samples; i++ )
383 {
384 p_dest[2 * i] = p_src[i_offset];
385 p_dest[2 * i + 1] = p_src[i_offset + 1];
386 i_offset += 2 * i_stride;
387 }
388}

401static void StopDecode( demux_t *p_demux )
402{
403 demux_sys_t *p_sys = p_demux->p_sys;
404
405 es_out_Del( p_demux->out, p_sys->p_es_video );
406
407 for ( int i = 0; i < MAX_AUDIOS; i++ )
408 {
409 hdsdi_audio_t *p_audio = &p_sys->p_audios[i];
410 if ( p_audio->i_channel != -1 && p_audio->p_es != NULL )
411 {
412 es_out_Del( p_demux->out, p_audio->p_es );
413 p_audio->p_es = NULL;
414 }
415 }
416}

418static int InitVideo( demux_t *p_demux )
419{
420 demux_sys_t *p_sys = p_demux->p_sys;
421 es_format_t fmt;
422
423 msg_Dbg( p_demux, "found standard %d", p_sys->i_standard );
424 switch ( p_sys->i_standard )
425 {
426 case SDIVIDEO_CTL_BT_601_576I_50HZ:
427 /* PAL */
428 p_sys->i_frame_rate = 25;
429 p_sys->i_frame_rate_base = 1;
430 p_sys->i_width = 720;
431 p_sys->i_height = 576;
432 p_sys->i_aspect = 4 * VOUT_ASPECT_FACTOR / 3;
433 break;
434
435 case SDIVIDEO_CTL_SMPTE_296M_720P_50HZ:
436 p_sys->i_frame_rate = 50;
437 p_sys->i_frame_rate_base = 1;
438 p_sys->i_width = 1280;
439 p_sys->i_height = 720;
440 p_sys->i_aspect = 16 * VOUT_ASPECT_FACTOR / 9;
441 break;
442
443 case SDIVIDEO_CTL_SMPTE_296M_720P_60HZ:
444 p_sys->i_frame_rate = 60;
445 p_sys->i_frame_rate_base = 1;
446 p_sys->i_width = 1280;
447 p_sys->i_height = 720;
448 p_sys->i_aspect = 16 * VOUT_ASPECT_FACTOR / 9;
449 break;
450
451 case SDIVIDEO_CTL_SMPTE_295M_1080I_50HZ:
452 case SDIVIDEO_CTL_SMPTE_274M_1080I_50HZ:
453 case SDIVIDEO_CTL_SMPTE_274M_1080PSF_25HZ:
454 /* 1080i50 or 1080p25 */
455 p_sys->i_frame_rate = 25;
456 p_sys->i_frame_rate_base = 1;
457 p_sys->i_width = 1920;
458 p_sys->i_height = 1080;
459 p_sys->i_aspect = 16 * VOUT_ASPECT_FACTOR / 9;
460 break;
461
462 case SDIVIDEO_CTL_SMPTE_274M_1080I_59_94HZ:
463 p_sys->i_frame_rate = 30000;
464 p_sys->i_frame_rate_base = 1001;
465 p_sys->i_width = 1920;
466 p_sys->i_height = 1080;
467 p_sys->i_aspect = 16 * VOUT_ASPECT_FACTOR / 9;
468 break;
469
470 case SDIVIDEO_CTL_SMPTE_274M_1080I_60HZ:
471 p_sys->i_frame_rate = 30;
472 p_sys->i_frame_rate_base = 1;
473 p_sys->i_width = 1920;
474 p_sys->i_height = 1080;
475 p_sys->i_aspect = 16 * VOUT_ASPECT_FACTOR / 9;
476 break;
477
478 default:
479 msg_Err( p_demux, "unsupported standard %d", p_sys->i_standard );
480 return VLC_EGENERIC;
481 }
482
483 p_sys->i_next_vdate = START_DATE;
484 p_sys->i_incr = 1000000 * p_sys->i_frame_rate_base / p_sys->i_frame_rate;
485 p_sys->i_vblock_size = p_sys->i_width * p_sys->i_height * 3 / 2
486 + sizeof(struct block_extension_t);
487
488 /* Video ES */
489 es_format_Init( &fmt, VIDEO_ES, VLC_FOURCC('I','4','2','0') );
490 fmt.i_id = p_sys->i_id_video;
491 fmt.video.i_frame_rate = p_sys->i_frame_rate;
492 fmt.video.i_frame_rate_base = p_sys->i_frame_rate_base;
493 fmt.video.i_width = fmt.video.i_visible_width = p_sys->i_width;
494 fmt.video.i_height = fmt.video.i_visible_height = p_sys->i_height;
495 fmt.video.i_sar_num = p_sys->i_aspect * fmt.video.i_height
496 / fmt.video.i_width;
497 fmt.video.i_sar_den = VOUT_ASPECT_FACTOR;
498 p_sys->p_es_video = es_out_Add( p_demux->out, &fmt );
499
500 return VLC_SUCCESS;
501}

503static int InitAudio( demux_t *p_demux )
504{
505 demux_sys_t *p_sys = p_demux->p_sys;
506 es_format_t fmt;
507
508 for ( int i = 0; i < MAX_AUDIOS; i++ )
509 {
510 hdsdi_audio_t *p_audio = &p_sys->p_audios[i];
511
512 if ( p_audio->i_channel == -1 ) continue;
513
514 msg_Dbg( p_demux, "starting audio %u/%u rate:%u delay:%d",
515 1 + p_audio->i_channel / 2, 1 + (p_audio->i_channel % 2),
516 p_sys->i_sample_rate, p_audio->i_delay );
517
518 es_format_Init( &fmt, AUDIO_ES, VLC_FOURCC('a','r','a','w') );
519 fmt.i_id = p_audio->i_id;
520 fmt.audio.i_channels = 2;
521 fmt.audio.i_original_channels =
522 fmt.audio.i_physical_channels = AOUT_CHANS_STEREO;
523 fmt.audio.i_rate = p_sys->i_sample_rate;
524 fmt.audio.i_bitspersample = 16;
525 fmt.audio.i_blockalign = fmt.audio.i_channels *
526 fmt.audio.i_bitspersample / 8;
527 fmt.i_bitrate = fmt.audio.i_channels * fmt.audio.i_rate *
528 fmt.audio.i_bitspersample;
529 p_audio->p_es = es_out_Add( p_demux->out, &fmt );
530 }
531
532 p_sys->i_next_adate = START_DATE;
533 p_sys->i_ablock_size = p_sys->i_sample_rate * 4 * p_sys->i_frame_rate_base / p_sys->i_frame_rate;
534 p_sys->i_aincr = 1000000. * p_sys->i_ablock_size / p_sys->i_sample_rate / 4;
535
536 return VLC_SUCCESS;
537}

539static int HandleVideo( demux_t *p_demux, const uint8_t *p_buffer )
540{
541 demux_sys_t *p_sys = p_demux->p_sys;
542 block_t *p_current_picture = block_Alloc( p_sys->i_vblock_size );
543 if( unlikely( !p_current_picture ) )
544 return VLC_ENOMEM;
545 uint8_t *p_y = p_current_picture->p_buffer;
546 uint8_t *p_u = p_y + p_sys->i_width * p_sys->i_height;
547 uint8_t *p_v = p_u + p_sys->i_width * p_sys->i_height / 4;
548 unsigned int i_total_size = p_sys->i_width * 2;
549 unsigned int i_current_line;
550 struct block_extension_t ext;
551
552 for ( i_current_line = 0; i_current_line < p_sys->i_height;
553 i_current_line++ )
554 {
555 bool b_field = (i_current_line >= p_sys->i_height / 2);
556 unsigned int i_field_line = b_field ?
557 i_current_line - (p_sys->i_height + 1) / 2 :
558 i_current_line;
559 unsigned int i_real_line = b_field + i_field_line * 2;
560 const uint8_t *p_line = p_buffer + i_current_line * p_sys->i_width * 2;
561
562 if ( !(i_field_line % 2) && !b_field )
563 Unpack01( p_line, i_total_size,
564 p_y + p_sys->i_width * i_real_line,
565 p_u + (p_sys->i_width / 2) * (i_real_line / 2),
566 p_v + (p_sys->i_width / 2) * (i_real_line / 2) );
567 else if ( !(i_field_line % 2) )
568 Unpack01( p_line, i_total_size,
569 p_y + p_sys->i_width * i_real_line,
570 p_u + (p_sys->i_width / 2) * (i_real_line / 2 + 1),
571 p_v + (p_sys->i_width / 2) * (i_real_line / 2 + 1) );
572 else if ( !b_field )
573 Unpack2( p_line, i_total_size,
574 p_y + p_sys->i_width * i_real_line,
575 p_u + (p_sys->i_width / 2) * (i_real_line / 2 - 1),
576 p_v + (p_sys->i_width / 2) * (i_real_line / 2 - 1) );
577 else
578 Unpack3( p_line, i_total_size,
579 p_y + p_sys->i_width * i_real_line,
580 p_u + (p_sys->i_width / 2) * (i_real_line / 2),
581 p_v + (p_sys->i_width / 2) * (i_real_line / 2) );
582 }
583
584 /* FIXME: progressive formats ? */
585 ext.b_progressive = false;
586 ext.i_nb_fields = 2;
587 ext.b_top_field_first = true;
588 ext.i_aspect = p_sys->i_forced_aspect ? p_sys->i_forced_aspect :
589 p_sys->i_aspect;
590
591 memcpy( &p_current_picture->p_buffer[p_sys->i_vblock_size
592 - sizeof(struct block_extension_t)],
593 &ext, sizeof(struct block_extension_t) );
594
595 p_current_picture->i_dts = p_current_picture->i_pts = p_sys->i_next_vdate;
596 es_out_Send( p_demux->out, p_sys->p_es_video, p_current_picture );
597
598 es_out_Control( p_demux->out, ES_OUT_SET_PCR, p_sys->i_next_vdate );
599 p_sys->i_next_vdate += p_sys->i_incr;
600 return VLC_SUCCESS;
601}

603static int HandleAudio( demux_t *p_demux, const uint8_t *p_buffer )
604{
605 demux_sys_t *p_sys = p_demux->p_sys;
606
607 for ( int i = 0; i < MAX_AUDIOS; i++ )
608 {
609 hdsdi_audio_t *p_audio = &p_sys->p_audios[i];
610 if ( p_audio->i_channel != -1 && p_audio->p_es != NULL )
611 {
612 block_t *p_block = block_Alloc( p_sys->i_ablock_size );
613 if( unlikely( !p_block ) )
614 return VLC_ENOMEM;
615 SparseCopy( (int16_t *)p_block->p_buffer, (const int16_t *)p_buffer,
616 p_sys->i_ablock_size / 4,
617 p_audio->i_channel * 2, p_sys->i_max_channel + 1 );
618
619 p_block->i_dts = p_block->i_pts
620 = p_sys->i_next_adate + (mtime_t)p_audio->i_delay
621 * INT64_C(1000000) / p_sys->i_sample_rate;
622 p_block->i_length = p_sys->i_aincr;
623 es_out_Send( p_demux->out, p_audio->p_es, p_block );
624 }
625 }
626 p_sys->i_next_adate += p_sys->i_aincr;
627 return VLC_SUCCESS;
628}

635static ssize_t WriteULSysfs( const char *psz_fmt, unsigned int i_link,
636 unsigned int i_buf )
637{
638 char psz_file[MAXLEN], psz_data[MAXLEN];
639 int i_fd;
640 ssize_t i_ret;
641
642 snprintf( psz_file, sizeof(psz_file) -1, psz_fmt, i_link );
643
644 snprintf( psz_data, sizeof(psz_data) -1, "%u\n", i_buf );
645
646 if ( (i_fd = vlc_open( psz_file, O_WRONLY )) < 0 )
647 return i_fd;
648
649 i_ret = write( i_fd, psz_data, strlen(psz_data) + 1 );
650 close( i_fd );
651 return i_ret;
652}

654static int InitCapture( demux_t *p_demux )
655{
656 demux_sys_t *p_sys = p_demux->p_sys;
657#ifdef HAVE_MMAP_SDIVIDEO
658 const int i_page_size = getpagesize();
659 unsigned int i_bufmemsize;
660#endif
661 char psz_vdev[MAXLEN];
662
663 snprintf( psz_vdev, sizeof(psz_vdev), SDIVIDEO_DEVICE, p_sys->i_link );
664 psz_vdev[sizeof(psz_vdev) - 1] = '\0';
665 if ( (p_sys->i_vfd = vlc_open( psz_vdev, O_RDONLY ) ) < 0 )
666 {
667 msg_Err( p_demux, "couldn't open device %s", psz_vdev );
668 return VLC_EGENERIC;
669 }
670
671 /* Wait for standard to settle down */
672 while ( vlc_object_alive(p_demux) )
673 {
674 struct pollfd pfd[1];
675
676 pfd[0].fd = p_sys->i_vfd;
677 pfd[0].events = POLLIN | POLLPRI;
678
679 if ( poll( pfd, 1, READ_TIMEOUT ) < 0 )
680 {
681 msg_Warn( p_demux, "couldn't poll(): %s", vlc_strerror_c(errno) );
682 close( p_sys->i_vfd );
683 return VLC_EGENERIC;
684 }
685
686 if ( pfd[0].revents & POLLPRI )
687 {
688 unsigned int i_val;
689
690 if ( ioctl( p_sys->i_vfd, SDIVIDEO_IOC_RXGETEVENTS, &i_val ) < 0 )
691 msg_Warn( p_demux, "couldn't SDIVIDEO_IOC_RXGETEVENTS: %s",
692 vlc_strerror_c(errno) );
693 else
694 {
695 if ( i_val & SDIVIDEO_EVENT_RX_BUFFER )
696 msg_Warn( p_demux, "driver receive buffer queue overrun" );
697 if ( i_val & SDIVIDEO_EVENT_RX_FIFO )
698 msg_Warn( p_demux, "onboard receive FIFO overrun");
699 if ( i_val & SDIVIDEO_EVENT_RX_CARRIER )
700 msg_Warn( p_demux, "carrier status change");
701 if ( i_val & SDIVIDEO_EVENT_RX_DATA )
702 msg_Warn( p_demux, "data status change");
703 if ( i_val & SDIVIDEO_EVENT_RX_STD )
704 {
705 msg_Warn( p_demux, "standard status change");
706 break;
707 }
708 }
709 }
710 }
711 if ( !vlc_object_alive(p_demux) )
712 {
713 close( p_sys->i_vfd );
714 return VLC_EGENERIC;
715 }
716
717 if ( ioctl( p_sys->i_vfd, SDIVIDEO_IOC_RXGETVIDSTATUS, &p_sys->i_standard )
718 < 0 )
719 {
720 msg_Warn( p_demux, "couldn't SDIVIDEO_IOC_RXGETVIDSTATUS: %s",
721 vlc_strerror_c(errno) );
722 close( p_sys->i_vfd );
723 return VLC_EGENERIC;
724 }
725 close( p_sys->i_vfd );
726
727 if ( InitVideo( p_demux ) != VLC_SUCCESS )
728 return VLC_EGENERIC;
729 p_sys->i_vbuffer_size = p_sys->i_height * p_sys->i_width * 2;
730
731 /* First open the audio for synchronization reasons */
732 if ( p_sys->i_max_channel != -1 )
733 {
734 unsigned int i_rate;
735 char psz_adev[MAXLEN];
736
737 snprintf( psz_adev, sizeof(psz_adev), SDIAUDIO_DEVICE, p_sys->i_link );
738 psz_adev[sizeof(psz_adev) - 1] = '\0';
739 if ( (p_sys->i_afd = vlc_open( psz_adev, O_RDONLY ) ) < 0 )
740 {
741 msg_Err( p_demux, "couldn't open device %s", psz_adev );
742 return VLC_EGENERIC;
743 }
744
745 if ( ioctl( p_sys->i_afd, SDIAUDIO_IOC_RXGETAUDRATE, &i_rate ) < 0 )
746 {
747 msg_Warn( p_demux, "couldn't SDIAUDIO_IOC_RXGETAUDRATE: %s",
748 vlc_strerror_c(errno) );
749 return VLC_EGENERIC;
750 }
751 switch ( i_rate )
752 {
753 case SDIAUDIO_CTL_ASYNC_48_KHZ:
754 case SDIAUDIO_CTL_SYNC_48_KHZ:
755 p_sys->i_sample_rate = 48000;
756 break;
757 case SDIAUDIO_CTL_ASYNC_44_1_KHZ:
758 case SDIAUDIO_CTL_SYNC_44_1_KHZ:
759 p_sys->i_sample_rate = 44100;
760 break;
761 case SDIAUDIO_CTL_ASYNC_32_KHZ:
762 case SDIAUDIO_CTL_SYNC_32_KHZ:
763 p_sys->i_sample_rate = 32000;
764 break;
765 case SDIAUDIO_CTL_ASYNC_96_KHZ:
766 case SDIAUDIO_CTL_SYNC_96_KHZ:
767 p_sys->i_sample_rate = 96000;
768 break;
769 case SDIAUDIO_CTL_ASYNC_FREE_RUNNING:
770 case SDIAUDIO_CTL_SYNC_FREE_RUNNING:
771 default:
772 msg_Err( p_demux, "unknown sample rate %u", i_rate );
773 return VLC_EGENERIC;
774 }
775 close( p_sys->i_afd );
776
777 if ( InitAudio( p_demux ) != VLC_SUCCESS )
778 return VLC_EGENERIC;
779 p_sys->i_abuffer_size = p_sys->i_ablock_size
780 * (1 + p_sys->i_max_channel);
781
782 /* Use 16-bit audio */
783 if ( WriteULSysfs( SDIAUDIO_SAMPLESIZE_FILE, p_sys->i_link,
784 SDIAUDIO_CTL_AUDSAMP_SZ_16 ) < 0 )
785 {
786 msg_Err( p_demux, "couldn't write file " SDIAUDIO_SAMPLESIZE_FILE,
787 p_sys->i_link );
788 return VLC_EGENERIC;
789 }
790
791 if ( WriteULSysfs( SDIAUDIO_CHANNELS_FILE, p_sys->i_link,
792 (p_sys->i_max_channel + 1) * 2 ) < 0 )
793 {
794 msg_Err( p_demux, "couldn't write file " SDIAUDIO_CHANNELS_FILE,
795 p_sys->i_link );
796 return VLC_EGENERIC;
797 }
798
799#ifdef HAVE_MMAP_SDIAUDIO
800 if ( (p_sys->i_abuffers = ReadULSysfs( SDIAUDIO_BUFFERS_FILE,
801 p_sys->i_link )) < 0 )
802 {
803 msg_Err( p_demux, "couldn't read file " SDIAUDIO_BUFFERS_FILE,
804 p_sys->i_link );
805 return VLC_EGENERIC;
806 }
807 p_sys->i_current_abuffer = 0;
808#endif
809
810 if ( WriteULSysfs( SDIAUDIO_BUFSIZE_FILE, p_sys->i_link,
811 p_sys->i_abuffer_size ) < 0 )
812 {
813 msg_Err( p_demux, "couldn't write file " SDIAUDIO_BUFSIZE_FILE,
814 p_sys->i_link );
815 return VLC_EGENERIC;
816 }
817
818 if ( (p_sys->i_afd = open( psz_adev, O_RDONLY ) ) < 0 )
819 {
820 msg_Err( p_demux, "couldn't open device %s", psz_adev );
821 return VLC_EGENERIC;
822 }
823
824#ifdef HAVE_MMAP_SDIAUDIO
825 i_bufmemsize = ((p_sys->i_abuffer_size + i_page_size - 1) / i_page_size)
826 * i_page_size;
827 p_sys->pp_abuffers = malloc( p_sys->i_abuffers * sizeof(uint8_t *) );
828 if( unlikely( !p_sys->pp_abuffers ) )
829 return VLC_ENOMEM;
830 for ( unsigned int i = 0; i < p_sys->i_abuffers; i++ )
831 {
832 if ( (p_sys->pp_abuffers[i] = mmap( NULL, p_sys->i_abuffer_size,
833 PROT_READ, MAP_SHARED, p_sys->i_afd,
834 i * i_bufmemsize )) == MAP_FAILED )
835 {
836 msg_Err( p_demux, "couldn't mmap(%d): %s", i,
837 vlc_strerror_c(errno) );
838 return VLC_EGENERIC;
839 }
840 }
841#endif
842 }
843
844 /* Use 8-bit video */
845 if ( WriteULSysfs( SDIVIDEO_MODE_FILE, p_sys->i_link,
846 SDIVIDEO_CTL_MODE_UYVY ) < 0 )
847 {
848 msg_Err( p_demux, "couldn't write file " SDIVIDEO_MODE_FILE,
849 p_sys->i_link );
850 return VLC_EGENERIC;
851 }
852
853 if ( WriteULSysfs( SDIVIDEO_BUFFERS_FILE, p_sys->i_link,
854 NB_VBUFFERS ) < 0 )
855 {
856 msg_Err( p_demux, "couldn't write file " SDIVIDEO_BUFFERS_FILE,
857 p_sys->i_link );
858 return VLC_EGENERIC;
859 }
860#ifdef HAVE_MMAP_SDIVIDEO
861 p_sys->i_vbuffers = NB_VBUFFERS;
862#endif
863
864 if ( WriteULSysfs( SDIVIDEO_BUFSIZE_FILE, p_sys->i_link,
865 p_sys->i_vbuffer_size ) < 0 )
866 {
867 msg_Err( p_demux, "couldn't write file " SDIVIDEO_BUFSIZE_FILE,
868 p_sys->i_link );
869 return VLC_EGENERIC;
870 }
871
872 if ( (p_sys->i_vfd = open( psz_vdev, O_RDONLY ) ) < 0 )
873 {
874 msg_Err( p_demux, "couldn't open device %s", psz_vdev );
875 return VLC_EGENERIC;
876 }
877
878#ifdef HAVE_MMAP_SDIVIDEO
879 p_sys->i_current_vbuffer = 0;
880 i_bufmemsize = ((p_sys->i_vbuffer_size + i_page_size - 1) / i_page_size)
881 * i_page_size;
882 p_sys->pp_vbuffers = malloc( p_sys->i_vbuffers * sizeof(uint8_t *) );
883 if( unlikely( !p_sys->pp_vbuffers ) )
884 return VLC_ENOMEM;
885 for ( unsigned int i = 0; i < p_sys->i_vbuffers; i++ )
886 {
887 if ( (p_sys->pp_vbuffers[i] = mmap( NULL, p_sys->i_vbuffer_size,
888 PROT_READ, MAP_SHARED, p_sys->i_vfd,
889 i * i_bufmemsize )) == MAP_FAILED )
890 {
891 msg_Err( p_demux, "couldn't mmap(%d): %s", i,
892 vlc_strerror_c(errno) );
893 return VLC_EGENERIC;
894 }
895 }
896#endif
897
898 return VLC_SUCCESS;
899}

902{
903 demux_sys_t *p_sys = p_demux->p_sys;
904
905 StopDecode( p_demux );
906#ifdef HAVE_MMAP_SDIVIDEO

917 free( p_sys->pp_abuffers );
918#endif
919 close( p_sys->i_afd );
920 }

923static int Capture( demux_t *p_demux )
924{
925 demux_sys_t *p_sys = p_demux->p_sys;
926 struct pollfd pfd[2];
927
928 pfd[0].fd = p_sys->i_vfd;
929 pfd[0].events = POLLIN | POLLPRI;
930 if ( p_sys->i_max_channel != -1 )
931 {
932 pfd[1].fd = p_sys->i_afd;
933 pfd[1].events = POLLIN | POLLPRI;
934 }
935
936 if ( poll( pfd, 1 + (p_sys->i_max_channel != -1), READ_TIMEOUT ) < 0 )
937 {
938 msg_Warn( p_demux, "couldn't poll(): %s", vlc_strerror_c(errno) );
939 return VLC_EGENERIC;
940 }
941
942 if ( pfd[0].revents & POLLPRI )
943 {
944 unsigned int i_val;
945
946 if ( ioctl( p_sys->i_vfd, SDIVIDEO_IOC_RXGETEVENTS, &i_val ) < 0 )
947 msg_Warn( p_demux, "couldn't SDIVIDEO_IOC_RXGETEVENTS: %s",
948 vlc_strerror_c(errno) );
949 else
950 {
951 if ( i_val & SDIVIDEO_EVENT_RX_BUFFER )
952 msg_Warn( p_demux, "driver receive buffer queue overrun" );
953 if ( i_val & SDIVIDEO_EVENT_RX_FIFO )
954 msg_Warn( p_demux, "onboard receive FIFO overrun");
955 if ( i_val & SDIVIDEO_EVENT_RX_CARRIER )
956 msg_Warn( p_demux, "carrier status change");
957 if ( i_val & SDIVIDEO_EVENT_RX_DATA )
958 msg_Warn( p_demux, "data status change");
959 if ( i_val & SDIVIDEO_EVENT_RX_STD )
960 msg_Warn( p_demux, "standard status change");
961 }
962
963 p_sys->i_next_adate += CLOCK_GAP;
964 p_sys->i_next_vdate += CLOCK_GAP;
965 }
966
967 if ( p_sys->i_max_channel != -1 && pfd[1].revents & POLLPRI )
968 {
969 unsigned int i_val;
970
971 if ( ioctl( p_sys->i_afd, SDIAUDIO_IOC_RXGETEVENTS, &i_val ) < 0 )
972 msg_Warn( p_demux, "couldn't SDIAUDIO_IOC_RXGETEVENTS: %s",
973 vlc_strerror_c(errno) );
974 else
975 {
976 if ( i_val & SDIAUDIO_EVENT_RX_BUFFER )
977 msg_Warn( p_demux, "driver receive buffer queue overrun" );
978 if ( i_val & SDIAUDIO_EVENT_RX_FIFO )
979 msg_Warn( p_demux, "onboard receive FIFO overrun");
980 if ( i_val & SDIAUDIO_EVENT_RX_CARRIER )
981 msg_Warn( p_demux, "carrier status change");
982 if ( i_val & SDIAUDIO_EVENT_RX_DATA )
983 msg_Warn( p_demux, "data status change");
984 }
985
986 p_sys->i_next_adate += CLOCK_GAP;
987 p_sys->i_next_vdate += CLOCK_GAP;
988 }
989
990 if ( pfd[0].revents & POLLIN )
991 {
992#ifdef HAVE_MMAP_SDIVIDEO
993 if ( ioctl( p_sys->i_vfd, SDIVIDEO_IOC_DQBUF, p_sys->i_current_vbuffer )
994 < 0 )
995 {
996 msg_Warn( p_demux, "couldn't SDIVIDEO_IOC_DQBUF: %s",
997 vlc_strerror_c(errno) );
998 return VLC_EGENERIC;
999 }
1000
1001 if( HandleVideo( p_demux, p_sys->pp_vbuffers[p_sys->i_current_vbuffer] ) != VLC_SUCCESS )
1002 return VLC_ENOMEM;
1003
1004 if ( ioctl( p_sys->i_vfd, SDIVIDEO_IOC_QBUF, p_sys->i_current_vbuffer )
1005 < 0 )
1006 {
1007 msg_Warn( p_demux, "couldn't SDIVIDEO_IOC_QBUF: %s",
1008 vlc_strerror_c(errno) );
1009 return VLC_EGENERIC;
1010 }
1011
1012 p_sys->i_current_vbuffer++;
1013 p_sys->i_current_vbuffer %= p_sys->i_vbuffers;
1014#else
1015 uint8_t *p_buffer = malloc( p_sys->i_vbuffer_size );
1016 if( unlikely( !p_buffer ) )
1017 return VLC_ENOMEM;
1018
1019 if ( read( p_sys->i_vfd, p_buffer, p_sys->i_vbuffer_size ) < 0 )
1020 {
1021 msg_Warn( p_demux, "couldn't read: %s", vlc_strerror_c(errno) );
1022 free( p_buffer );
1023 return VLC_EGENERIC;
1024 }
1025
1026 if( HandleVideo( p_demux, p_buffer ) != VLC_SUCCESS )
1027 {
1028 free( p_buffer );
1029 return VLC_ENOMEM;
1030 }
1031 free( p_buffer );
1032#endif
1033 }
1034
1035 if ( p_sys->i_max_channel != -1 && pfd[1].revents & POLLIN )
1036 {
1037#ifdef HAVE_MMAP_SDIAUDIO
1038 if ( ioctl( p_sys->i_afd, SDIAUDIO_IOC_DQBUF, p_sys->i_current_abuffer )
1039 < 0 )
1040 {
1041 msg_Warn( p_demux, "couldn't SDIAUDIO_IOC_DQBUF: %s",
1042 vlc_strerror_c(errno) );
1043 return VLC_EGENERIC;
1044 }
1045
1046 if( HandleAudio( p_demux, p_sys->pp_abuffers[p_sys->i_current_abuffer] ) != VLC_SUCCESS )
1047 return VLC_ENOMEM;
1048
1049 if ( ioctl( p_sys->i_afd, SDIAUDIO_IOC_QBUF, p_sys->i_current_abuffer )
1050 < 0 )
1051 {
1052 msg_Warn( p_demux, "couldn't SDIAUDIO_IOC_QBUF: %s",
1053 vlc_strerror_c(errno) );
1054 return VLC_EGENERIC;
1055 }
1056
1057 p_sys->i_current_abuffer++;
1058 p_sys->i_current_abuffer %= p_sys->i_abuffers;
1059#else
1060 uint8_t *p_buffer = malloc( p_sys->i_abuffer_size );
1061 if( unlikely( !p_buffer ) )
1062 return VLC_ENOMEM;
1063
1064 if ( read( p_sys->i_afd, p_buffer, p_sys->i_abuffer_size ) < 0 )
1065 {
1066 msg_Warn( p_demux, "couldn't read: %s", vlc_strerror_c(errno) );
1067 free( p_buffer );
1068 return VLC_EGENERIC;
1069 }
1070
1071 if( HandleAudio( p_demux, p_buffer ) != VLC_SUCCESS )
1072 {
1073 free( p_buffer );
1074 return VLC_ENOMEM;
1075 }
1076 free( p_buffer );
1077#endif
1078 }
1079
1080 return VLC_SUCCESS;
1081}
