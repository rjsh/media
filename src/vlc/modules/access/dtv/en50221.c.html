149#define SIZE_INDICATOR 0x80
150
151static uint8_t *GetLength( uint8_t *p_data, int *pi_length )
152{
153 *pi_length = *p_data++;
154
155 if ( (*pi_length & SIZE_INDICATOR) != 0 )
156 {
157 int l = *pi_length & ~SIZE_INDICATOR;
158 int i;
159
160 *pi_length = 0;
161 for ( i = 0; i < l; i++ )
162 *pi_length = (*pi_length << 8) | *p_data++;
163 }
164
165 return p_data;
166}
167
168static uint8_t *SetLength( uint8_t *p_data, int i_length )
169{
170 uint8_t *p = p_data;
171
172 if ( i_length < 128 )
173 {
174 *p++ = i_length;
175 }
176 else if ( i_length < 256 )
177 {
178 *p++ = SIZE_INDICATOR | 0x1;
179 *p++ = i_length;
180 }
181 else if ( i_length < 65536 )
182 {
183 *p++ = SIZE_INDICATOR | 0x2;
184 *p++ = i_length >> 8;
185 *p++ = i_length & 0xff;
186 }
187 else if ( i_length < 16777216 )
188 {
189 *p++ = SIZE_INDICATOR | 0x3;
190 *p++ = i_length >> 16;
191 *p++ = (i_length >> 8) & 0xff;
192 *p++ = i_length & 0xff;
193 }
194 else
195 {
196 *p++ = SIZE_INDICATOR | 0x4;
197 *p++ = i_length >> 24;
198 *p++ = (i_length >> 16) & 0xff;
199 *p++ = (i_length >> 8) & 0xff;
200 *p++ = i_length & 0xff;
201 }
202
203 return p;
204}

242/*****************************************************************************
243 * TPDUSend
244 *****************************************************************************/
245static int TPDUSend( cam_t * p_cam, uint8_t i_slot, uint8_t i_tag,
246 const uint8_t *p_content, int i_length )
247{
248 uint8_t i_tcid = i_slot + 1;
249 uint8_t p_data[MAX_TPDU_SIZE];
250 int i_size;
251
252 i_size = 0;
253 p_data[0] = i_slot;
254 p_data[1] = i_tcid;
255 p_data[2] = i_tag;
256
257 switch ( i_tag )
258 {
259 case T_RCV:
260 case T_CREATE_TC:
261 case T_CTC_REPLY:
262 case T_DELETE_TC:
263 case T_DTC_REPLY:
264 case T_REQUEST_TC:
265 p_data[3] = 1; /* length */
266 p_data[4] = i_tcid;
267 i_size = 5;
268 break;
269
270 case T_NEW_TC:
271 case T_TC_ERROR:
272 p_data[3] = 2; /* length */
273 p_data[4] = i_tcid;
274 p_data[5] = p_content[0];
275 i_size = 6;
276 break;
277
278 case T_DATA_LAST:
279 case T_DATA_MORE:
280 {
281 /* i_length <= MAX_TPDU_DATA */
282 uint8_t *p = p_data + 3;
283 p = SetLength( p, i_length + 1 );
284 *p++ = i_tcid;
285
286 if ( i_length )
287 memcpy( p, p_content, i_length );
288 i_size = i_length + (p - p_data);
289 break;
290 }
291
292 default:
293 break;
294 }
295 Dump( true, p_data, i_size );
296
297 if ( write( p_cam->fd, p_data, i_size ) != i_size )
298 {
299 msg_Err( p_cam->obj, "cannot write to CAM device: %s",
300 vlc_strerror_c(errno) );
301 return VLC_EGENERIC;
302 }
303
304 return VLC_SUCCESS;
305}

308/*****************************************************************************
309 * TPDURecv
310 *****************************************************************************/
311#define CAM_READ_TIMEOUT 3500 // ms
312
313static int TPDURecv( cam_t *p_cam, uint8_t i_slot, uint8_t *pi_tag,
314 uint8_t *p_data, int *pi_size )
315{
316 uint8_t i_tcid = i_slot + 1;
317 int i_size;
318 struct pollfd pfd[1];
319
320 pfd[0].fd = p_cam->fd;
321 pfd[0].events = POLLIN;
322
323 while( poll(pfd, 1, CAM_READ_TIMEOUT ) == -1 )
324 if( errno != EINTR )
325 {
326 msg_Err( p_cam->obj, "poll error: %s", vlc_strerror_c(errno) );
327 return VLC_EGENERIC;
328 }
329
330 if ( !(pfd[0].revents & POLLIN) )
331 {
332 msg_Err( p_cam->obj, "CAM device poll time-out" );
333 return VLC_EGENERIC;
334 }
335
336 if ( pi_size == NULL )
337 {
338 p_data = xmalloc( MAX_TPDU_SIZE );
339 }
340
341 for ( ; ; )
342 {
343 i_size = read( p_cam->fd, p_data, MAX_TPDU_SIZE );
344
345 if ( i_size >= 0 || errno != EINTR )
346 break;
347 }
348
349 if ( i_size < 5 )
350 {
351 msg_Err( p_cam->obj, "cannot read from CAM device (%d): %s", i_size,
352 vlc_strerror_c(errno) );
353 if( pi_size == NULL )
354 free( p_data );
355 return VLC_EGENERIC;
356 }
357
358 if ( p_data[1] != i_tcid )
359 {
360 msg_Err( p_cam->obj, "invalid read from CAM device (%d instead of %d)",
361 p_data[1], i_tcid );
362 if( pi_size == NULL )
363 free( p_data );
364 return VLC_EGENERIC;
365 }
366
367 *pi_tag = p_data[2];
368 p_cam->pb_tc_has_data[i_slot] = (i_size >= 4
369 && p_data[i_size - 4] == T_SB
370 && p_data[i_size - 3] == 2
371 && (p_data[i_size - 1] & DATA_INDICATOR))
372 ? true : false;
373
374 Dump( false, p_data, i_size );
375
376 if ( pi_size == NULL )
377 free( p_data );
378 else
379 *pi_size = i_size;
380
381 return VLC_SUCCESS;
382}

400#define RI_RESOURCE_MANAGER 0x00010041
401#define RI_APPLICATION_INFORMATION 0x00020041
402#define RI_CONDITIONAL_ACCESS_SUPPORT 0x00030041
403#define RI_HOST_CONTROL 0x00200041
404#define RI_DATE_TIME 0x00240041
405#define RI_MMI 0x00400041
406
407static int ResourceIdToInt( uint8_t *p_data )
408{
409 return ((int)p_data[0] << 24) | ((int)p_data[1] << 16)
410 | ((int)p_data[2] << 8) | p_data[3];
411}

413/*****************************************************************************
414 * SPDUSend
415 *****************************************************************************/
416static int SPDUSend( cam_t * p_cam, int i_session_id,
417 uint8_t *p_data, int i_size )
418{
419 uint8_t *p_spdu = xmalloc( i_size + 4 );
420 uint8_t *p = p_spdu;
421 uint8_t i_tag;
422 uint8_t i_slot = p_cam->p_sessions[i_session_id - 1].i_slot;
423
424 *p++ = ST_SESSION_NUMBER;
425 *p++ = 0x02;
426 *p++ = (i_session_id >> 8);
427 *p++ = i_session_id & 0xff;
428
429 memcpy( p, p_data, i_size );
430
431 i_size += 4;
432 p = p_spdu;
433
434 while ( i_size > 0 )
435 {
436 if ( i_size > MAX_TPDU_DATA )
437 {
438 if ( TPDUSend( p_cam, i_slot, T_DATA_MORE, p,
439 MAX_TPDU_DATA ) != VLC_SUCCESS )
440 {
441 msg_Err( p_cam->obj, "couldn't send TPDU on session %d",
442 i_session_id );
443 free( p_spdu );
444 return VLC_EGENERIC;
445 }
446 p += MAX_TPDU_DATA;
447 i_size -= MAX_TPDU_DATA;
448 }
449 else
450 {
451 if ( TPDUSend( p_cam, i_slot, T_DATA_LAST, p, i_size )
452 != VLC_SUCCESS )
453 {
454 msg_Err( p_cam->obj, "couldn't send TPDU on session %d",
455 i_session_id );
456 free( p_spdu );
457 return VLC_EGENERIC;
458 }
459 i_size = 0;
460 }
461
462 if ( TPDURecv( p_cam, i_slot, &i_tag, NULL, NULL ) != VLC_SUCCESS
463 || i_tag != T_SB )
464 {
465 msg_Err( p_cam->obj, "couldn't recv TPDU on session %d",
466 i_session_id );
467 free( p_spdu );
468 return VLC_EGENERIC;
469 }
470 }
471
472 free( p_spdu );
473 return VLC_SUCCESS;
474}

476/*****************************************************************************
477 * SessionOpen
478 *****************************************************************************/
479static void SessionOpen( cam_t * p_cam, uint8_t i_slot,
480 uint8_t *p_spdu, int i_size )
481{
482 VLC_UNUSED( i_size );
483
484 int i_session_id;
485 int i_resource_id = ResourceIdToInt( &p_spdu[2] );
486 uint8_t p_response[16];
487 int i_status = SS_NOT_ALLOCATED;
488 uint8_t i_tag;
489
490 for ( i_session_id = 1; i_session_id <= MAX_SESSIONS; i_session_id++ )
491 {
492 if ( !p_cam->p_sessions[i_session_id - 1].i_resource_id )
493 break;
494 }
495 if ( i_session_id > MAX_SESSIONS )
496 {
497 msg_Err( p_cam->obj, "too many sessions !" );
498 return;
499 }
500 p_cam->p_sessions[i_session_id - 1].i_slot = i_slot;
501 p_cam->p_sessions[i_session_id - 1].i_resource_id = i_resource_id;
502 p_cam->p_sessions[i_session_id - 1].pf_close = NULL;
503 p_cam->p_sessions[i_session_id - 1].pf_manage = NULL;
504
505 if ( i_resource_id == RI_RESOURCE_MANAGER
506 || i_resource_id == RI_APPLICATION_INFORMATION
507 || i_resource_id == RI_CONDITIONAL_ACCESS_SUPPORT
508 || i_resource_id == RI_DATE_TIME
509 || i_resource_id == RI_MMI )
510 {
511 i_status = SS_OK;
512 }
513
514 p_response[0] = ST_OPEN_SESSION_RESPONSE;
515 p_response[1] = 0x7;
516 p_response[2] = i_status;
517 p_response[3] = p_spdu[2];
518 p_response[4] = p_spdu[3];
519 p_response[5] = p_spdu[4];
520 p_response[6] = p_spdu[5];
521 p_response[7] = i_session_id >> 8;
522 p_response[8] = i_session_id & 0xff;
523
524 if ( TPDUSend( p_cam, i_slot, T_DATA_LAST, p_response, 9 ) !=
525 VLC_SUCCESS )
526 {
527 msg_Err( p_cam->obj,
528 "SessionOpen: couldn't send TPDU on slot %d", i_slot );
529 return;
530 }
531 if ( TPDURecv( p_cam, i_slot, &i_tag, NULL, NULL ) != VLC_SUCCESS )
532 {
533 msg_Err( p_cam->obj,
534 "SessionOpen: couldn't recv TPDU on slot %d", i_slot );
535 return;
536 }
537
538 switch ( i_resource_id )
539 {
540 case RI_RESOURCE_MANAGER:
541 ResourceManagerOpen( p_cam, i_session_id ); break;
542 case RI_APPLICATION_INFORMATION:
543 ApplicationInformationOpen( p_cam, i_session_id ); break;
544 case RI_CONDITIONAL_ACCESS_SUPPORT:
545 ConditionalAccessOpen( p_cam, i_session_id ); break;
546 case RI_DATE_TIME:
547 DateTimeOpen( p_cam, i_session_id ); break;
548 case RI_MMI:
549 MMIOpen( p_cam, i_session_id ); break;
550
551 case RI_HOST_CONTROL:
552 default:
553 msg_Err( p_cam->obj, "unknown resource id (0x%x)", i_resource_id );
554 p_cam->p_sessions[i_session_id - 1].i_resource_id = 0;
555 }
556}

628 switch ( i_resource_id )

630 case RI_RESOURCE_MANAGER:
631 ResourceManagerOpen( p_cam, i_session_id ); break;
632 case RI_APPLICATION_INFORMATION:
633 ApplicationInformationOpen( p_cam, i_session_id ); break;
634 case RI_CONDITIONAL_ACCESS_SUPPORT:
635 ConditionalAccessOpen( p_cam, i_session_id ); break;
636 case RI_DATE_TIME:
637 DateTimeOpen( p_cam, i_session_id ); break;

711 * SPDUHandle
712 *****************************************************************************/
713static void SPDUHandle( cam_t * p_cam, uint8_t i_slot,
714 uint8_t *p_spdu, int i_size )
715{
716 int i_session_id;
717
718 switch ( p_spdu[0] )
719 {
720 case ST_SESSION_NUMBER:
721 if ( i_size <= 4 )
722 return;
723 i_session_id = ((int)p_spdu[2] << 8) | p_spdu[3];
724 p_cam->p_sessions[i_session_id - 1].pf_handle( p_cam, i_session_id,
725 p_spdu + 4, i_size - 4 );
726 break;
727
728 case ST_OPEN_SESSION_REQUEST:
729 if ( i_size != 6 || p_spdu[1] != 0x4 )
730 return;
731 SessionOpen( p_cam, i_slot, p_spdu, i_size );
732 break;
733
734 case ST_CREATE_SESSION_RESPONSE:
735 if ( i_size != 9 || p_spdu[1] != 0x7 )
736 return;
737 SessionCreateResponse( p_cam, i_slot, p_spdu, i_size );
738 break;
739
740 case ST_CLOSE_SESSION_REQUEST:
741 if ( i_size != 4 || p_spdu[1] != 0x2 )
742 return;
743 i_session_id = ((int)p_spdu[2] << 8) | p_spdu[3];
744 SessionClose( p_cam, i_session_id );
745 break;
746
747 case ST_CLOSE_SESSION_RESPONSE:
748 if ( i_size != 5 || p_spdu[1] != 0x3 )
749 return;
750 i_session_id = ((int)p_spdu[3] << 8) | p_spdu[4];
751 if ( p_spdu[2] )
752 {
753 msg_Err( p_cam->obj,
754 "closing a session which is not allocated (%d)",
755 i_session_id );
756 }
757 else
758 {
759 if ( p_cam->p_sessions[i_session_id - 1].pf_close != NULL )
760 p_cam->p_sessions[i_session_id - 1].pf_close( p_cam,
761 i_session_id );
762 p_cam->p_sessions[i_session_id - 1].i_resource_id = 0;
763 }
764 break;
765
766 default:
767 msg_Err( p_cam->obj, "unexpected tag in SPDUHandle (%x)", p_spdu[0] );
768 break;
769 }
770}

852 *****************************************************************************/
853static int APDUSend( cam_t * p_cam, int i_session_id, int i_tag,
854 uint8_t *p_data, int i_size )
855{
856 uint8_t *p_apdu = xmalloc( i_size + 12 );
857 uint8_t *p = p_apdu;
858 ca_msg_t ca_msg;
859 int i_ret;
860
861 *p++ = (i_tag >> 16);
862 *p++ = (i_tag >> 8) & 0xff;
863 *p++ = i_tag & 0xff;
864 p = SetLength( p, i_size );
865 if ( i_size )
866 memcpy( p, p_data, i_size );
867 if ( p_cam->i_ca_type == CA_CI_LINK )
868 {
869 i_ret = SPDUSend( p_cam, i_session_id, p_apdu, i_size + p - p_apdu );
870 }
871 else
872 {
873 if ( i_size + p - p_apdu > 256 )
874 {
875 msg_Err( p_cam->obj, "CAM: apdu overflow" );
876 i_ret = VLC_EGENERIC;
877 }
878 else
879 {
880 ca_msg.length = i_size + p - p_apdu;
881 if ( i_size == 0 ) ca_msg.length=3;
882 memcpy( ca_msg.msg, p_apdu, i_size + p - p_apdu );
883 i_ret = ioctl( p_cam->fd, CA_SEND_MSG, &ca_msg );
884 if ( i_ret < 0 )
885 {
886 msg_Err( p_cam->obj, "Error sending to CAM: %s",
887 vlc_strerror_c(errno) );
888 i_ret = VLC_EGENERIC;
889 }
890 }
891 }
892 free( p_apdu );
893 return i_ret;
894}

1434 int i_interval;
1435 mtime_t i_last;
1436} date_time_t;

1438/*****************************************************************************
1439 * DateTimeSend
1440 *****************************************************************************/
1441static void DateTimeSend( cam_t * p_cam, int i_session_id )
1442{
1443 date_time_t *p_date =
1444 (date_time_t *)p_cam->p_sessions[i_session_id - 1].p_sys;
1445
1446 time_t t = time(NULL);
1447 struct tm tm_gmt;
1448 struct tm tm_loc;
1449
1450 if ( gmtime_r(&t, &tm_gmt) && localtime_r(&t, &tm_loc) )
1451 {
1452 int Y = tm_gmt.tm_year;
1453 int M = tm_gmt.tm_mon + 1;
1454 int D = tm_gmt.tm_mday;
1455 int L = (M == 1 || M == 2) ? 1 : 0;
1456 int MJD = 14956 + D + (int)((Y - L) * 365.25)
1457 + (int)((M + 1 + L * 12) * 30.6001);
1458 uint8_t p_response[7];
1459
1460#define DEC2BCD(d) (((d / 10) << 4) + (d % 10))
1461
1462 SetWBE( &p_response[0], MJD );
1463 p_response[2] = DEC2BCD(tm_gmt.tm_hour);
1464 p_response[3] = DEC2BCD(tm_gmt.tm_min);
1465 p_response[4] = DEC2BCD(tm_gmt.tm_sec);
1466 SetWBE( &p_response[5], tm_loc.tm_gmtoff / 60 );
1467
1468 APDUSend( p_cam, i_session_id, AOT_DATE_TIME, p_response, 7 );
1469
1470 p_date->i_last = mdate();
1471 }
1472}
