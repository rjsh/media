444static int Open (vlc_object_t *obj)
445{
446 access_t *access = (access_t *)obj;
447 access_sys_t *sys = malloc (sizeof (*sys));
448 if (unlikely(sys == NULL))
449 return VLC_ENOMEM;
450
451 var_LocationParse (obj, access->psz_location, "dvb-");
452
453 dvb_device_t *dev = dvb_open (obj);
454 if (dev == NULL)
455 {
456 free (sys);
457 return VLC_EGENERIC;
458 }
459
460 sys->dev = dev;
461 sys->signal_poll = 0;
462 access->p_sys = sys;
463
464 uint64_t freq = var_InheritFrequency (obj);
465 if (freq != 0)
466 {
467 const delsys_t *delsys = GuessSystem (access->psz_access, dev);
468 if (delsys == NULL || Tune (obj, dev, delsys, freq))
469 {
470 msg_Err (obj, "tuning to %"PRIu64" Hz failed", freq);
471 dialog_Fatal (obj, N_("Digital broadcasting"),
472 N_("The selected digital tuner does not support "
473 "the specified parameters.\n"
474 "Please check the preferences."));
475 goto error;
476 }
477 }
478 dvb_add_pid (dev, 0);
479
480 access->pf_block = Read;
481 access->pf_control = Control;
482 free (access->psz_demux);
483 access->psz_demux = strdup ("ts");
484 return VLC_SUCCESS;
485
486error:
487 Close (obj);
488 access->p_sys = NULL;
489 return VLC_EGENERIC;
490}
491
492static void Close (vlc_object_t *obj)
493{
494 access_t *access = (access_t *)obj;
495 access_sys_t *sys = access->p_sys;
496
497 dvb_close (sys->dev);
498 free (sys);
499}
500
501static block_t *Read (access_t *access)
502{
503#define BUFSIZE (20*188)
504 block_t *block = block_Alloc (BUFSIZE);
505 if (unlikely(block == NULL))
506 return NULL;
507
508 access_sys_t *sys = access->p_sys;
509 ssize_t val = dvb_read (sys->dev, block->p_buffer, BUFSIZE);
510
511 if (val <= 0)
512 {
513 if (val == 0)
514 access->info.b_eof = true;
515 block_Release (block);
516 return NULL;
517 }
518
519 block->i_buffer = val;
520
521 return block;
522}

524static int Control (access_t *access, int query, va_list args)
525{
526 access_sys_t *sys = access->p_sys;
527 dvb_device_t *dev = sys->dev;
528
529 switch (query)
530 {
531 case ACCESS_CAN_SEEK:
532 case ACCESS_CAN_FASTSEEK:
533 case ACCESS_CAN_PAUSE:
534 case ACCESS_CAN_CONTROL_PACE:
535 *va_arg (args, bool *) = false;
536 break;
537
538 case ACCESS_GET_PTS_DELAY:
539 {
540 int64_t *v = va_arg (args, int64_t *);
541 *v = var_InheritInteger (access, "live-caching") * INT64_C(1000);
542 break;
543 }
544
545 case ACCESS_GET_CONTENT_TYPE:
546 *va_arg (args, char **) = strdup ("video/MP2T");
547 break;
548
549 case ACCESS_GET_SIGNAL:
550 /* Fetch the signal levels only every so often to avoid stressing
551 * the device bus. */
552 if ((sys->signal_poll++))
553 return VLC_EGENERIC;
554
555 *va_arg (args, double *) = dvb_get_snr (dev);
556 *va_arg (args, double *) = dvb_get_signal_strength (dev);
557 return VLC_SUCCESS;
558
559 case ACCESS_SET_PRIVATE_ID_STATE:
560 {
561 unsigned pid = va_arg (args, unsigned);
562 bool add = va_arg (args, unsigned);
563
564 if (unlikely(pid > 0x1FFF))
565 return VLC_EGENERIC;
566 if (add)
567 {
568 if (dvb_add_pid (dev, pid))
569 return VLC_EGENERIC;
570 }
571 else
572 dvb_remove_pid (dev, pid);
573 break;
574 }
575
576#ifdef HAVE_DVBPSI
577 case ACCESS_SET_PRIVATE_ID_CA:
578 {
579 struct dvbpsi_pmt_s *pmt = va_arg (args, struct dvbpsi_pmt_s *);
580
581 dvb_set_ca_pmt (dev, pmt);
582 break;
583 }
584#endif
585 /*case ACCESS_GET_PRIVATE_ID_STATE: TODO? */
586
587 default:
588 return VLC_EGENERIC;
589 }
590
591 return VLC_SUCCESS;
592}

597/** Determines which delivery system to use. */
598static const delsys_t *GuessSystem (const char *scheme, dvb_device_t *dev)
599{
600 /* Specific delivery system is specified */
601 if (!strcasecmp (scheme, "atsc"))
602 return &atsc;
603 if (!strcasecmp (scheme, "cqam"))
604 return &cqam;
605 if (!strcasecmp (scheme, "dvb-c"))
606 return &dvbc;
607 if (!strcasecmp (scheme, "dvb-s"))
608 return &dvbs;
609 if (!strcasecmp (scheme, "dvb-s2"))
610 return &dvbs2;
611 if (!strcasecmp (scheme, "dvb-t"))
612 return &dvbt;
613 if (!strcasecmp (scheme, "dvb-t2"))
614 return &dvbt2;
615 if (!strcasecmp (scheme, "isdb-c"))
616 return &isdbc;
617 if (!strcasecmp (scheme, "isdb-s"))
618 return &isdbs;
619 if (!strcasecmp (scheme, "isdb-t"))
620 return &isdbt;
621
622 /* If the demodulator supports 2G, we cannot guess whether
623 * 1G or 2G is intended. For backward compatibility, 1G is assumed
624 * (this is not a limitation of Linux DVB). We will probably need something
625 * smarter when 2G (semi automatic) scanning is implemented. */
626 unsigned systems = dvb_enum_systems (dev);
627
628 /* Only wave carrier is specified */
629 if (!strcasecmp (scheme, "cable"))
630 {
631 if (systems & DVB_C)
632 return &dvbc;
633 if (systems & CQAM)
634 return &cqam;
635 if (systems & ISDB_C)
636 return &isdbc;
637 }
638 if (!strcasecmp (scheme, "satellite"))
639 {
640 if (systems & DVB_S)
641 return &dvbs;
642 if (systems & ISDB_S)
643 return &isdbs;
644 }
645 if (!strcasecmp (scheme, "terrestrial"))
646 {
647 if (systems & DVB_T)
648 return &dvbc;
649 if (systems & ATSC)
650 return &cqam;
651 if (systems & ISDB_T)
652 return &isdbt;
653 }
654
655 /* Only standards family or nothing is specified */
656 if (systems & DVB_C)
657 return &dvbc;
658 if (systems & DVB_S)
659 return &dvbs;
660 if (systems & DVB_T)
661 return &dvbt;
662 if (systems & ATSC)
663 return &atsc;
664 return NULL;
665}

690static uint32_t var_InheritCodeRate (vlc_object_t *obj, const char *varname)
691{
692 char *code_rate = var_InheritString (obj, varname);
693 if (code_rate == NULL)
694 return VLC_FEC_AUTO;
695
696 uint16_t a, b;
697 int v = sscanf (code_rate, "%"SCNu16"/%"SCNu16, &a, &b);
698 free (code_rate);
699 switch (v)
700 {
701 case 2:
702 return VLC_FEC(a, b);
703 case 1:
704 if (a == 0)
705 return 0;
706 /* Backward compatibility with VLC < 1.2 (= Linux DVBv3 enum) */
707 if (a < 9)
708 {
709 msg_Warn (obj, "\"%s=%"PRIu16"\" option is obsolete. "
710 "Use \"%s=%"PRIu16"/%"PRIu16"\" instead.",
711 varname + 4, a, varname + 4, a, a + 1);
712 return VLC_FEC(a, a + 1);
713 }
714 else
715 msg_Warn (obj, "\"fec=9\" option is obsolete.");
716 }
717 return VLC_FEC_AUTO;
718}

725static const char *var_InheritModulation (vlc_object_t *obj, const char *var)
726{
727 char *mod = var_InheritString (obj, var);
728 if (mod == NULL)
729 return "";
730
731 size_t n = sizeof (modulation_vlc) / sizeof (modulation_vlc[0]);
732 const char *const *p = lfind (mod, modulation_vlc, &n, sizeof (mod), modcmp);
733 if (p != NULL)
734 {
735 free (mod);
736 return *p;
737 }
738
739 /* Backward compatibility with VLC < 1.2 */
740 const char *str;
741 switch (atoi (mod))
742 {
743 case -1: str = "QPSK"; break;
744 case 0: str = "QAM"; break;
745 case 8: str = "8VSB"; break;
746 case 16: str = "16QAM"; break;
747 case 32: str = "32QAM"; break;
748 case 64: str = "64QAM"; break;
749 case 128: str = "128QAM"; break;
750 case 256: str = "256QAM"; break;
751 default: return "";
752 }
753
754 msg_Warn (obj, "\"modulation=%s\" option is obsolete. "
755 "Use \"modulation=%s\" instead.", mod, str);
756 free (mod);
757 return str;
758}

760static unsigned var_InheritGuardInterval (vlc_object_t *obj)
761{
762 char *guard = var_InheritString (obj, "dvb-guard");
763 if (guard == NULL)
764 return VLC_GUARD_AUTO;
765
766 uint16_t a, b;
767 int v = sscanf (guard, "%"SCNu16"/%"SCNu16, &a, &b);
768 free (guard);
769 switch (v)
770 {
771 case 1:
772 /* Backward compatibility with VLC < 1.2 */
773 if (a == 0)
774 break;
775 msg_Warn (obj, "\"guard=%"PRIu16"\" option is obsolete. "
776 "Use \"guard=1/%"PRIu16" instead.", a, a);
777 b = a;
778 a = 1;
779 case 2:
780 return VLC_GUARD(a, b);
781 }
782 return VLC_GUARD_AUTO;
783}
