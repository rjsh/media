89struct vlc_v4l2_ctrl
90{
91 int fd;
92 uint32_t id;
93 uint8_t type;
94 char name[32];
95 int32_t default_value;

99static int ControlSet (const vlc_v4l2_ctrl_t *c, int_fast32_t value)
100{
101 struct v4l2_control ctrl = {
102 .id = c->id,
103 .value = value,
104 };
105 if (v4l2_ioctl (c->fd, VIDIOC_S_CTRL, &ctrl) < 0)
106 return -1;
107 return 0;
108}

148static int ControlSetCallback (vlc_object_t *obj, const char *var,
149 vlc_value_t old, vlc_value_t cur, void *data)
150{
151 const vlc_v4l2_ctrl_t *ctrl = data;
152 int ret;
153
154 switch (ctrl->type)
155 {
156 case V4L2_CTRL_TYPE_INTEGER:
157 case V4L2_CTRL_TYPE_MENU:
158 case V4L2_CTRL_TYPE_BITMASK:
159 case V4L2_CTRL_TYPE_INTEGER_MENU:
160 ret = ControlSet (ctrl, cur.i_int);
161 break;
162 case V4L2_CTRL_TYPE_BOOLEAN:
163 ret = ControlSet (ctrl, cur.b_bool);
164 break;
165 case V4L2_CTRL_TYPE_BUTTON:
166 ret = ControlSet (ctrl, 0);
167 break;
168 case V4L2_CTRL_TYPE_INTEGER64:
169 ret = ControlSet64 (ctrl, cur.i_int);
170 break;
171 case V4L2_CTRL_TYPE_STRING:
172 ret = ControlSetStr (ctrl, cur.psz_string);
173 break;
174 default:
175 assert (0);
176 }
177
178 if (ret)
179 {
180 msg_Err (obj, "cannot set control %s: %s", var, vlc_strerror_c(errno));
181 return VLC_EGENERIC;
182 }
183 (void) old;
184 return VLC_SUCCESS;
185}

187static void ControlsReset (vlc_object_t *obj, vlc_v4l2_ctrl_t *list)
188{
189 while (list != NULL)
190 {
191 switch (list->type)
192 {
193 case V4L2_CTRL_TYPE_INTEGER:
194 case V4L2_CTRL_TYPE_MENU:
195 case V4L2_CTRL_TYPE_INTEGER_MENU:
196 var_SetInteger (obj, list->name, list->default_value);
197 break;
198 case V4L2_CTRL_TYPE_BOOLEAN:
199 var_SetBool (obj, list->name, list->default_value);
200 break;
201 default:;
202 }
203 list = list->next;
204 }
205}

208 vlc_value_t old, vlc_value_t cur, void *data)
209{
210 ControlsReset (obj, data);
211 (void) var; (void) old; (void) cur;
212 return VLC_SUCCESS;

215static void ControlsSetFromString (vlc_object_t *obj,
216 const vlc_v4l2_ctrl_t *list)
217{
218 char *buf = var_InheritString (obj, CFG_PREFIX"set-ctrls");
219 if (buf == NULL)
220 return;
221
222 char *p = buf;
223 if (*p == '{')
224 p++;
225
226 char *end = strchr (p, '}');
227 if (end != NULL)
228 *end = '\0';
229next:
230 while (p != NULL && *p)
231 {
232 const char *name, *value;
233
234 p += strspn (p, ", ");
235 name = p;
236 end = strchr (p, ',');
237 if (end != NULL)
238 *(end++) = '\0';
239 p = end; /* next name/value pair */
240
241 end = strchr (name, '=');
242 if (end == NULL)
243 {
244 /* TODO? support button controls that way? */
245 msg_Err (obj, "syntax error in \"%s\": missing '='", name);
246 continue;
247 }
248 *(end++) = '\0';
249 value = end;
250
251 for (const vlc_v4l2_ctrl_t *c = list; c != NULL; c = c->next)
252 if (!strcasecmp (name, c->name))
253 switch (c->type)
254 {
255 case V4L2_CTRL_TYPE_INTEGER:
256 case V4L2_CTRL_TYPE_BOOLEAN:
257 case V4L2_CTRL_TYPE_MENU:
258 case V4L2_CTRL_TYPE_INTEGER_MENU:
259 {
260 long val = strtol (value, &end, 0);
261 if (*end)
262 {
263 msg_Err (obj, "syntax error in \"%s\": "
264 " not an integer", value);
265 goto next;
266 }
267 ControlSet (c, val);
268 break;
269 }
270
271 case V4L2_CTRL_TYPE_INTEGER64:
272 {
273 long long val = strtoll (value, &end, 0);
274 if (*end)
275 {
276 msg_Err (obj, "syntax error in \"%s\": "
277 " not an integer", value);
278 goto next;
279 }
280 ControlSet64 (c, val);
281 break;
282 }
283
284 case V4L2_CTRL_TYPE_STRING:
285 ControlSetStr (c, value);
286 break;
287
288 case V4L2_CTRL_TYPE_BITMASK:
289 {
290 unsigned long val = strtoul (value, &end, 0);
291 if (*end)
292 {
293 msg_Err (obj, "syntax error in \"%s\": "
294 " not an integer", value);
295 goto next;
296 }
297 ControlSet (c, val);
298 break;
299 }
300
301 default:
302 msg_Err (obj, "setting \"%s\" not supported", name);
303 goto next;
304 }
305
306 msg_Err (obj, "control \"%s\" not available", name);
307 }
308 free (buf);
309}

311static int cidcmp (const void *a, const void *b)
312{
313 const uint32_t *id = a;
314 const vlc_v4l2_ctrl_name_t *name = b;
315
316 return (int32_t)(*id - name->cid);
317}

319/**
320 * Creates a VLC-V4L2 control structure:
321 * In particular, determines a name suitable for a VLC object variable.
322 * \param query V4L2 control query structure [IN]
323 * \return NULL on error
324 */
325static vlc_v4l2_ctrl_t *ControlCreate (int fd,
326 const struct v4l2_queryctrl *query)
327{
328 vlc_v4l2_ctrl_t *ctrl = malloc (sizeof (*ctrl));
329 if (unlikely(ctrl == NULL))
330 return NULL;
331
332 ctrl->fd = fd;
333 ctrl->id = query->id;
334 ctrl->type = query->type;
335
336 /* Search for a well-known control */
337 const vlc_v4l2_ctrl_name_t *known;
338 known = bsearch (&query->id, controls, sizeof (controls) / sizeof (*known),
339 sizeof (*known), cidcmp);
340 if (known != NULL)
341 strcpy (ctrl->name, known->name);
342 else
343 /* Fallback to automatically-generated control name */
344 {
345 size_t i;
346 for (i = 0; query->name[i]; i++)
347 {
348 unsigned char c = query->name[i];
349 if (c == ' ' || c == ',')
350 c = '_';
351 if (c < 128)
352 c = tolower (c);
353 ctrl->name[i] = c;
354 }
355 ctrl->name[i] = '\0';
356 }
357
358 ctrl->default_value = query->default_value;
359 return ctrl;
360}

363#define CTRL_FLAGS_IGNORE \
364 (V4L2_CTRL_FLAG_DISABLED /* not implemented at all */ \
365 |V4L2_CTRL_FLAG_READ_ONLY /* value is constant */ \

368static vlc_v4l2_ctrl_t *ControlAddInteger (vlc_object_t *obj, int fd,
369 const struct v4l2_queryctrl *query)
370{
371 msg_Dbg (obj, " integer %s (%08"PRIX32")", query->name, query->id);
372 if (query->flags & CTRL_FLAGS_IGNORE)
373 return NULL;
374
375 vlc_v4l2_ctrl_t *c = ControlCreate (fd, query);
376 if (unlikely(c == NULL))
377 return NULL;
378
379 if (var_Create (obj, c->name, VLC_VAR_INTEGER | VLC_VAR_ISCOMMAND))
380 {
381 free (c);
382 return NULL;
383 }
384
385 vlc_value_t val;
386 struct v4l2_control ctrl = { .id = query->id };
387
388 if (v4l2_ioctl (fd, VIDIOC_G_CTRL, &ctrl) >= 0)
389 {
390 msg_Dbg (obj, " current: %3"PRId32", default: %3"PRId32,
391 ctrl.value, query->default_value);
392 val.i_int = ctrl.value;
393 var_Change (obj, c->name, VLC_VAR_SETVALUE, &val, NULL);
394 }
395 val.i_int = query->minimum;
396 var_Change (obj, c->name, VLC_VAR_SETMIN, &val, NULL);
397 val.i_int = query->maximum;
398 var_Change (obj, c->name, VLC_VAR_SETMAX, &val, NULL);
399 if (query->step != 1)
400 {
401 val.i_int = query->step;
402 var_Change (obj, c->name, VLC_VAR_SETSTEP, &val, NULL);
403 }
404 val.i_int = query->default_value;
405 var_Change (obj, c->name, VLC_VAR_SETDEFAULT, &val, NULL);
406 return c;
407}

442static vlc_v4l2_ctrl_t *ControlAddMenu (vlc_object_t *obj, int fd,
443 const struct v4l2_queryctrl *query)
444{
445 msg_Dbg (obj, " menu %s (%08"PRIX32")", query->name, query->id);
446 if (query->flags & CTRL_FLAGS_IGNORE)
447 return NULL;
448
449 vlc_v4l2_ctrl_t *c = ControlCreate (fd, query);
450 if (unlikely(c == NULL))
451 return NULL;
452
453 if (var_Create (obj, c->name, VLC_VAR_INTEGER | VLC_VAR_HASCHOICE
454 | VLC_VAR_ISCOMMAND))
455 {
456 free (c);
457 return NULL;
458 }
459
460 vlc_value_t val;
461 struct v4l2_control ctrl = { .id = query->id };
462
463 if (v4l2_ioctl (fd, VIDIOC_G_CTRL, &ctrl) >= 0)
464 {
465 msg_Dbg (obj, " current: %"PRId32", default: %"PRId32,
466 ctrl.value, query->default_value);
467 val.i_int = ctrl.value;
468 var_Change (obj, c->name, VLC_VAR_SETVALUE, &val, NULL);
469 }
470 val.i_int = query->minimum;
471 var_Change (obj, c->name, VLC_VAR_SETMIN, &val, NULL);
472 val.i_int = query->maximum;
473 var_Change (obj, c->name, VLC_VAR_SETMAX, &val, NULL);
474 val.i_int = query->default_value;
475 var_Change (obj, c->name, VLC_VAR_SETDEFAULT, &val, NULL);
476
477 /* Import menu choices */
478 for (uint_fast32_t idx = query->minimum;
479 idx <= (uint_fast32_t)query->maximum;
480 idx++)
481 {
482 struct v4l2_querymenu menu = { .id = query->id, .index = idx };
483
484 if (v4l2_ioctl (fd, VIDIOC_QUERYMENU, &menu) < 0)
485 continue;
486 msg_Dbg (obj, " choice %"PRIu32") %s", menu.index, menu.name);
487
488 vlc_value_t text;
489 val.i_int = menu.index;
490 text.psz_string = (char *)menu.name;
491 var_Change (obj, c->name, VLC_VAR_ADDCHOICE, &val, &text);
492 }
493 return c;
494}

559static vlc_v4l2_ctrl_t *ControlAddString (vlc_object_t *obj, int fd,
560 const struct v4l2_queryctrl *query)
561{
562 msg_Dbg (obj, " string %s (%08"PRIX32")", query->name, query->id);
563 if ((query->flags & CTRL_FLAGS_IGNORE) || query->maximum > 65535)
564 return NULL;
565
566 vlc_v4l2_ctrl_t *c = ControlCreate (fd, query);
567 if (unlikely(c == NULL))
568 return NULL;
569
570 if (var_Create (obj, c->name, VLC_VAR_STRING | VLC_VAR_ISCOMMAND))
571 {
572 free (c);
573 return NULL;
574 }
575
576 /* Get current value */
577 char *buf = malloc (query->maximum + 1);
578 if (likely(buf != NULL))
579 {
580 struct v4l2_ext_control ext_ctrl = {
581 .id = c->id,
582 .size = query->maximum + 1,
583 };
584 ext_ctrl.string = buf;
585 struct v4l2_ext_controls ext_ctrls = {
586 .ctrl_class = V4L2_CTRL_ID2CLASS(c->id),
587 .count = 1,
588 .error_idx = 0,
589 .controls = &ext_ctrl,
590 };
591
592 if (v4l2_ioctl (c->fd, VIDIOC_G_EXT_CTRLS, &ext_ctrls) >= 0)
593 {
594 vlc_value_t val = { .psz_string = buf };
595
596 msg_Dbg (obj, " current: \"%s\"", buf);
597 var_Change (obj, c->name, VLC_VAR_SETVALUE, &val, NULL);
598 }
599 free (buf);
600 }
601
602 return c;
603}

662 if (v4l2_ioctl (fd, VIDIOC_G_CTRL, &ctrl) >= 0)
663 {
664 msg_Dbg (obj, " current: %"PRId32", default: %"PRId32,
665 ctrl.value, query->default_value);
666 val.i_int = ctrl.value;
667 var_Change (obj, c->name, VLC_VAR_SETVALUE, &val, NULL);
668 }
669 val.i_int = query->minimum;
670 var_Change (obj, c->name, VLC_VAR_SETMIN, &val, NULL);
671 val.i_int = query->maximum;
672 var_Change (obj, c->name, VLC_VAR_SETMAX, &val, NULL);
673 val.i_int = query->default_value;
674 var_Change (obj, c->name, VLC_VAR_SETDEFAULT, &val, NULL);
675
676 /* Import menu choices */
677 for (uint_fast32_t idx = query->minimum;
678 idx <= (uint_fast32_t)query->maximum;
679 idx++)
680 {
681 struct v4l2_querymenu menu = { .id = query->id, .index = idx };
682 char name[sizeof ("-9223372036854775808")];
683
684 if (v4l2_ioctl (fd, VIDIOC_QUERYMENU, &menu) < 0)
685 continue;
686 msg_Dbg (obj, " choice %"PRIu32") %"PRId64, menu.index,
687 (uint64_t)menu.value);
688
689 vlc_value_t text;
690 val.i_int = menu.index;
691 sprintf (name, "%"PRId64, (int64_t)menu.value);
692 text.psz_string = name;
693 var_Change (obj, c->name, VLC_VAR_ADDCHOICE, &val, &text);
694 }
695 return c;
696}

698static vlc_v4l2_ctrl_t *ControlAddUnknown (vlc_object_t *obj, int fd,
699 const struct v4l2_queryctrl *query)
700{
701 msg_Dbg (obj, " unknown %s (%08"PRIX32")", query->name, query->id);
702 msg_Warn (obj, " unknown control type %u", (unsigned)query->type);
703 (void) fd;
704 return NULL;
705}

707typedef vlc_v4l2_ctrl_t *(*ctrl_type_cb) (vlc_object_t *, int,
708 const struct v4l2_queryctrl *);
709
710/**
711 * Lists all user-class v4l2 controls, sets them to the user specified
712 * value and create the relevant variables to enable run-time changes.
713 */
714vlc_v4l2_ctrl_t *ControlsInit (vlc_object_t *obj, int fd)
715{
716 /* A list of controls that can be modified at run-time is stored in the
717 * "controls" variable. The V4L2 controls dialog can be built from this. */
718 var_Create (obj, "controls", VLC_VAR_INTEGER | VLC_VAR_HASCHOICE);
719
720 static const ctrl_type_cb handlers[] =
721 {
722 [V4L2_CTRL_TYPE_INTEGER] = ControlAddInteger,
723 [V4L2_CTRL_TYPE_BOOLEAN] = ControlAddBoolean,
724 [V4L2_CTRL_TYPE_MENU] = ControlAddMenu,
725 [V4L2_CTRL_TYPE_BUTTON] = ControlAddButton,
726 [V4L2_CTRL_TYPE_INTEGER64] = ControlAddInteger64,
727 [V4L2_CTRL_TYPE_CTRL_CLASS] = ControlAddClass,
728 [V4L2_CTRL_TYPE_STRING] = ControlAddString,
729 [V4L2_CTRL_TYPE_BITMASK] = ControlAddBitMask,
730 [V4L2_CTRL_TYPE_INTEGER_MENU] = ControlAddIntMenu,
731 };
732
733 vlc_v4l2_ctrl_t *list = NULL;
734 struct v4l2_queryctrl query;
735
736 query.id = V4L2_CTRL_FLAG_NEXT_CTRL;
737 while (v4l2_ioctl (fd, VIDIOC_QUERYCTRL, &query) >= 0)
738 {
739 ctrl_type_cb handler = NULL;
740 if (query.type < (sizeof (handlers) / sizeof (handlers[0])))
741 handler = handlers[query.type];
742 if (handler == NULL)
743 handler = ControlAddUnknown;
744
745 vlc_v4l2_ctrl_t *c = handler (obj, fd, &query);
746 if (c != NULL)
747 {
748 vlc_value_t val, text;
749
750 var_AddCallback (obj, c->name, ControlSetCallback, c);
751 text.psz_string = (char *)query.name;
752 var_Change (obj, c->name, VLC_VAR_SETTEXT, &text, NULL);
753 val.i_int = query.id;
754 text.psz_string = (char *)c->name;
755 var_Change (obj, "controls", VLC_VAR_ADDCHOICE, &val, &text);
756
757 c->next = list;
758 list = c;
759 }
760 query.id |= V4L2_CTRL_FLAG_NEXT_CTRL;
761 }
762
763 /* Set well-known controls from VLC configuration */
764 for (vlc_v4l2_ctrl_t *ctrl = list; ctrl != NULL; ctrl = ctrl->next)
765 {
766 if (!CTRL_CID_KNOWN (ctrl->id))
767 continue;
768
769 char varname[sizeof (CFG_PREFIX) + sizeof (ctrl->name) - 1];
770 sprintf (varname, CFG_PREFIX"%s", ctrl->name);
771
772 int64_t val = var_InheritInteger (obj, varname);
773 if (val == -1)
774 continue; /* the VLC default value: "do not modify" */
775 ControlSet (ctrl, val); /* NOTE: all known are integers or booleans */
776 }
777
778 /* Set any control from the VLC configuration control string */
779 ControlsSetFromString (obj, list);
780
781 /* Add a control to reset all controls to their default values */
782 {
783 vlc_value_t val, text;
784
785 var_Create (obj, "reset", VLC_VAR_VOID | VLC_VAR_ISCOMMAND);
786 val.psz_string = _("Reset defaults");
787 var_Change (obj, "reset", VLC_VAR_SETTEXT, &val, NULL);
788 val.i_int = -1;
789
790 text.psz_string = (char *)"reset";
791 var_Change (obj, "controls", VLC_VAR_ADDCHOICE, &val, &text);
792 var_AddCallback (obj, "reset", ControlsResetCallback, list);
793 }
794 if (var_InheritBool (obj, CFG_PREFIX"controls-reset"))
795 ControlsReset (obj, list);
796
797 return list;
798}
799
800void ControlsDeinit (vlc_object_t *obj, vlc_v4l2_ctrl_t *list)
801{
802 var_DelCallback (obj, "reset", ControlsResetCallback, list);
803 var_Destroy (obj, "reset");
804
805 while (list != NULL)
806 {
807 vlc_v4l2_ctrl_t *next = list->next;
808
809 var_DelCallback (obj, list->name, ControlSetCallback, list);
810 var_Destroy (obj, list->name);
811 free (list);
812 list = next;
813 }
814
815 var_Destroy (obj, "controls");
816}
