73int DemuxOpen( vlc_object_t *obj )
74{
75 demux_t *demux = (demux_t *)obj;
76
77 demux_sys_t *sys = malloc (sizeof (*sys));
78 if (unlikely(sys == NULL))
79 return VLC_ENOMEM;
80 demux->p_sys = sys;
81#ifdef ZVBI_COMPILED
82 sys->vbi = NULL;
83#endif
84
85 ParseMRL( obj, demux->psz_location );
86
87 char *path = var_InheritString (obj, CFG_PREFIX"dev");
88 if (unlikely(path == NULL))
89 goto error; /* probably OOM */
90
91 uint32_t caps;
92 int fd = OpenDevice (obj, path, &caps);
93 free (path);
94 if (fd == -1)
95 goto error;
96 sys->fd = fd;
97
98 if (InitVideo (demux, fd, caps))
99 {
100 v4l2_close (fd);
101 goto error;
102 }
103
104 sys->controls = ControlsInit (VLC_OBJECT(demux), fd);
105 sys->start = mdate ();
106 demux->pf_demux = NULL;
107 demux->pf_control = DemuxControl;
108 demux->info.i_update = 0;
109 demux->info.i_title = 0;
110 demux->info.i_seekpoint = 0;
111 return VLC_SUCCESS;
112error:
113 free (sys);
114 return VLC_EGENERIC;
115}

225static const vlc_v4l2_fmt_t *vlc_from_v4l2_fourcc (uint32_t fourcc)
226{
227 for (size_t i = 0; i < sizeof (v4l2_fmts) / sizeof (v4l2_fmts[0]); i++)
228 if (v4l2_fmts[i].v4l2 == fourcc)
229 return v4l2_fmts + i;
230 return NULL;
231}
232
233static size_t vlc_v4l2_fmt_rank (const vlc_v4l2_fmt_t *fmt)
234{
235 if (fmt == NULL)
236 return SIZE_MAX;
237
238 ptrdiff_t d = fmt - v4l2_fmts;
239 assert (d >= 0);
240 assert (d < (ptrdiff_t)(sizeof (v4l2_fmts) / sizeof (v4l2_fmts[0])));
241 return d;
242}

244static vlc_fourcc_t var_InheritFourCC (vlc_object_t *obj, const char *varname)
245{
246 char *str = var_InheritString (obj, varname);
247 if (str == NULL)
248 return 0;
249
250 vlc_fourcc_t fourcc = vlc_fourcc_GetCodecFromString (VIDEO_ES, str);
251 if (fourcc == 0)
252 msg_Err (obj, "invalid codec %s", str);
253 free (str);
254 return fourcc;
255}
256#define var_InheritFourCC(o, v) var_InheritFourCC(VLC_OBJECT(o), v)

258static void GetAR (int fd, unsigned *restrict num, unsigned *restrict den)
259{
260 struct v4l2_cropcap cropcap = { .type = V4L2_BUF_TYPE_VIDEO_CAPTURE };
261
262 /* TODO: get CROPCAP only once (see ResetCrop()). */
263 if (v4l2_ioctl (fd, VIDIOC_CROPCAP, &cropcap) < 0)
264 {
265 *num = *den = 1;
266 return;
267 }
268 *num = cropcap.pixelaspect.numerator;
269 *den = cropcap.pixelaspect.denominator;
270}

272static int InitVideo (demux_t *demux, int fd, uint32_t caps)
273{
274 demux_sys_t *sys = demux->p_sys;
275 v4l2_std_id std;
276
277 if (!(caps & V4L2_CAP_VIDEO_CAPTURE))
278 {
279 msg_Err (demux, "not a video capture device");
280 return -1;
281 }
282
283 if (SetupInput (VLC_OBJECT(demux), fd, &std))
284 return -1;
285
286 /* Picture format negotiation */
287 const vlc_v4l2_fmt_t *selected = NULL;
288 vlc_fourcc_t reqfourcc = var_InheritFourCC (demux, CFG_PREFIX"chroma");
289 bool native = false;
290
291 for (struct v4l2_fmtdesc codec = { .type = V4L2_BUF_TYPE_VIDEO_CAPTURE };
292 v4l2_ioctl (fd, VIDIOC_ENUM_FMT, &codec) >= 0;
293 codec.index++)
294 { /* Enumerate available chromas */
295 const vlc_v4l2_fmt_t *dsc = vlc_from_v4l2_fourcc (codec.pixelformat);
296
297 msg_Dbg (demux, " %s %s format %4.4s (%4.4s): %s",
298 (codec.flags & V4L2_FMT_FLAG_EMULATED) ? "emulates" : "supports",
299 (codec.flags & V4L2_FMT_FLAG_COMPRESSED) ? "compressed" : "raw",
300 (char *)&codec.pixelformat,
301 (dsc != NULL) ? (const char *)&dsc->vlc : "N.A.",
302 codec.description);
303
304 if (dsc == NULL)
305 continue; /* ignore VLC-unsupported codec */
306
307 if (dsc->vlc == reqfourcc)
308 {
309 msg_Dbg (demux, " matches the requested format");
310 selected = dsc;
311 break; /* always select the requested format if found */
312 }
313
314 if (codec.flags & V4L2_FMT_FLAG_EMULATED)
315 {
316 if (native)
317 continue; /* ignore emulated format if possible */
318 }
319 else
320 native = true;
321
322 if (vlc_v4l2_fmt_rank (dsc) > vlc_v4l2_fmt_rank (selected))
323 continue; /* ignore if rank is worse */
324
325 selected = dsc;
326 }
327
328 if (selected == NULL)
329 {
330 msg_Err (demux, "cannot negotiate supported video format");
331 return -1;
332 }
333 msg_Dbg (demux, "selected format %4.4s (%4.4s)",
334 (const char *)&selected->v4l2, (const char *)&selected->vlc);
335
336 /* Find best resolution and frame rate available */
337 struct v4l2_format fmt;
338 struct v4l2_streamparm parm;
339 if (SetupFormat (demux, fd, selected->v4l2, &fmt, &parm))
340 return -1;
341
342 /* Print extra info */
343 msg_Dbg (demux, "%d bytes maximum for complete image",
344 fmt.fmt.pix.sizeimage);
345 /* Check interlacing */
346 sys->block_flags = 0;
347 switch (fmt.fmt.pix.field)
348 {
349 case V4L2_FIELD_NONE:
350 msg_Dbg (demux, "Interlacing setting: progressive");
351 break;
352 case V4L2_FIELD_TOP:
353 msg_Dbg (demux, "Interlacing setting: top field only");
354 break;
355 case V4L2_FIELD_BOTTOM:
356 msg_Dbg (demux, "Interlacing setting: bottom field only");
357 break;
358 case V4L2_FIELD_INTERLACED:
359 msg_Dbg (demux, "Interlacing setting: interleaved");
360 /*if (NTSC)
361 sys->block_flags = BLOCK_FLAG_BOTTOM_FIELD_FIRST;
362 else*/
363 sys->block_flags = BLOCK_FLAG_TOP_FIELD_FIRST;
364 break;
365 case V4L2_FIELD_SEQ_TB:
366 msg_Dbg (demux, "Interlacing setting: sequential top bottom (TODO)");
367 break;
368 case V4L2_FIELD_SEQ_BT:
369 msg_Dbg (demux, "Interlacing setting: sequential bottom top (TODO)");
370 break;
371 case V4L2_FIELD_ALTERNATE:
372 msg_Dbg (demux, "Interlacing setting: alternate fields (TODO)");
373 fmt.fmt.pix.height *= 2;
374 break;
375 case V4L2_FIELD_INTERLACED_TB:
376 msg_Dbg (demux, "Interlacing setting: interleaved top bottom");
377 sys->block_flags = BLOCK_FLAG_TOP_FIELD_FIRST;
378 break;
379 case V4L2_FIELD_INTERLACED_BT:
380 msg_Dbg (demux, "Interlacing setting: interleaved bottom top");
381 sys->block_flags = BLOCK_FLAG_BOTTOM_FIELD_FIRST;
382 break;
383 default:
384 msg_Warn (demux, "Interlacing setting: unknown type (%d)",
385 fmt.fmt.pix.field);
386 break;
387 }
388
389 /* Declare our unique elementary (video) stream */
390 es_format_t es_fmt;
391
392 es_format_Init (&es_fmt, VIDEO_ES, selected->vlc);
393 es_fmt.video.i_rmask = selected->red;
394 es_fmt.video.i_gmask = selected->green;
395 es_fmt.video.i_bmask = selected->blue;
396 es_fmt.video.i_visible_width = fmt.fmt.pix.width;
397 if (fmt.fmt.pix.bytesperline != 0 && selected->bpp != 0)
398 es_fmt.video.i_width = fmt.fmt.pix.bytesperline / selected->bpp;
399 else
400 es_fmt.video.i_width = fmt.fmt.pix.width;
401 es_fmt.video.i_visible_height =
402 es_fmt.video.i_height = fmt.fmt.pix.height;
403 es_fmt.video.i_frame_rate = parm.parm.capture.timeperframe.denominator;
404 es_fmt.video.i_frame_rate_base = parm.parm.capture.timeperframe.numerator;
405 GetAR (fd, &es_fmt.video.i_sar_num, &es_fmt.video.i_sar_den);
406
407 msg_Dbg (demux, "added new video ES %4.4s %ux%u", (char *)&es_fmt.i_codec,
408 es_fmt.video.i_width, es_fmt.video.i_height);
409 msg_Dbg (demux, " frame rate: %u/%u", es_fmt.video.i_frame_rate,
410 es_fmt.video.i_frame_rate_base);
411 msg_Dbg (demux, " aspect ratio: %u/%u", es_fmt.video.i_sar_num,
412 es_fmt.video.i_sar_den);
413 sys->es = es_out_Add (demux->out, &es_fmt);
414
415 /* Init I/O method */
416 void *(*entry) (void *);
417 if (caps & V4L2_CAP_STREAMING)
418 {
419 if (0 /* BROKEN */ && StartUserPtr (VLC_OBJECT(demux), fd) == 0)
420 {
421 /* In principles, mmap() will pad the length to a multiple of the
422 * page size, so there is no need to care. Nevertheless with the
423 * page size, block->i_size can be set optimally. */
424 const long pagemask = sysconf (_SC_PAGE_SIZE) - 1;
425
426 sys->blocksize = (fmt.fmt.pix.sizeimage + pagemask) & ~pagemask;
427 sys->bufv = NULL;
428 entry = UserPtrThread;
429 msg_Dbg (demux, "streaming with %"PRIu32"-bytes user buffers",
430 sys->blocksize);
431 }
432 else /* fall back to memory map */
433 {
434 sys->bufc = 4;
435 sys->bufv = StartMmap (VLC_OBJECT(demux), fd, &sys->bufc);
436 if (sys->bufv == NULL)
437 return -1;
438 entry = MmapThread;
439 msg_Dbg (demux, "streaming with %"PRIu32" memory-mapped buffers",
440 sys->bufc);
441 }
442 }
443 else if (caps & V4L2_CAP_READWRITE)
444 {
445 sys->blocksize = fmt.fmt.pix.sizeimage;
446 sys->bufv = NULL;
447 entry = ReadThread;
448 msg_Dbg (demux, "reading %"PRIu32" bytes at a time", sys->blocksize);
449 }
450 else
451 {
452 msg_Err (demux, "no supported capture method");
453 return -1;
454 }
455
456#ifdef ZVBI_COMPILED
457 if (std & V4L2_STD_NTSC_M)
458 {
459 char *vbi_path = var_InheritString (demux, CFG_PREFIX"vbidev");
460 if (vbi_path != NULL)
461 sys->vbi = OpenVBI (demux, vbi_path);
462 free(vbi_path);
463 }
464#endif
465
466 if (vlc_clone (&sys->thread, entry, demux, VLC_THREAD_PRIORITY_INPUT))
467 {
468#ifdef ZVBI_COMPILED
469 if (sys->vbi != NULL)
470 CloseVBI (sys->vbi);
471#endif
472 if (sys->bufv != NULL)
473 StopMmap (sys->fd, sys->bufv, sys->bufc);
474 return -1;
475 }
476 return 0;
477}

479void DemuxClose( vlc_object_t *obj )
480{
481 demux_t *demux = (demux_t *)obj;
482 demux_sys_t *sys = demux->p_sys;
483
484 vlc_cancel (sys->thread);
485 vlc_join (sys->thread, NULL);
486 if (sys->bufv != NULL)
487 StopMmap (sys->fd, sys->bufv, sys->bufc);
488 ControlsDeinit( obj, sys->controls );
489 v4l2_close (sys->fd);
490
491#ifdef ZVBI_COMPILED
492 if (sys->vbi != NULL)
493 CloseVBI (sys->vbi);
494#endif
495
496 free( sys );
497}

499/** Allocates and queue a user buffer using mmap(). */
500static block_t *UserPtrQueue (vlc_object_t *obj, int fd, size_t length)
501{
502 void *ptr = mmap (NULL, length, PROT_READ | PROT_WRITE,
503 MAP_PRIVATE | MAP_ANONYMOUS, -1, 0);
504 if (ptr == MAP_FAILED)
505 {
506 msg_Err (obj, "cannot allocate %zu-bytes buffer: %s", length,
507 vlc_strerror_c(errno));
508 return NULL;
509 }
510
511 block_t *block = block_mmap_Alloc (ptr, length);
512 if (unlikely(block == NULL))
513 {
514 munmap (ptr, length);
515 return NULL;
516 }
517
518 struct v4l2_buffer buf = {
519 .type = V4L2_BUF_TYPE_VIDEO_CAPTURE,
520 .memory = V4L2_MEMORY_USERPTR,
521 .m = {
522 .userptr = (uintptr_t)ptr,
523 },
524 .length = length,
525 };
526
527 if (v4l2_ioctl (fd, VIDIOC_QBUF, &buf) < 0)
528 {
529 msg_Err (obj, "cannot queue buffer: %s", vlc_strerror_c(errno));
530 block_Release (block);
531 return NULL;
532 }
533 return block;
534}

536static void *UserPtrThread (void *data)
537{
538 demux_t *demux = data;
539 demux_sys_t *sys = demux->p_sys;
540 int fd = sys->fd;
541 struct pollfd ufd[2];
542 nfds_t numfds = 1;
543
544 ufd[0].fd = fd;
545 ufd[0].events = POLLIN;
546
547 int canc = vlc_savecancel ();
548 for (;;)
549 {
550 struct v4l2_buffer buf = {
551 .type = V4L2_BUF_TYPE_VIDEO_CAPTURE,
552 .memory = V4L2_MEMORY_USERPTR,
553 };
554 block_t *block = UserPtrQueue (VLC_OBJECT(demux), fd, sys->blocksize);
555 if (block == NULL)
556 break;
557
558 /* Wait for data */
559 vlc_restorecancel (canc);
560 block_cleanup_push (block);
561 while (poll (ufd, numfds, -1) == -1)
562 if (errno != EINTR)
563 msg_Err (demux, "poll error: %s", vlc_strerror_c(errno));
564 vlc_cleanup_pop ();
565 canc = vlc_savecancel ();
566
567 if (v4l2_ioctl (fd, VIDIOC_DQBUF, &buf) < 0)
568 {
569 msg_Err (demux, "cannot dequeue buffer: %s",
570 vlc_strerror_c(errno));
571 block_Release (block);
572 continue;
573 }
574
575 assert (block->p_buffer == (void *)buf.m.userptr);
576 block->i_buffer = buf.length;
577 block->i_pts = block->i_dts = GetBufferPTS (&buf);
578 block->i_flags |= sys->block_flags;
579 es_out_Control (demux->out, ES_OUT_SET_PCR, block->i_pts);
580 es_out_Send (demux->out, sys->es, block);
581 }
582 vlc_restorecancel (canc); /* <- hmm, this is purely cosmetic */
583 return NULL;
584}

586static void *MmapThread (void *data)
587{
588 demux_t *demux = data;
589 demux_sys_t *sys = demux->p_sys;
590 int fd = sys->fd;
591 struct pollfd ufd[2];
592 nfds_t numfds = 1;
593
594 ufd[0].fd = fd;
595 ufd[0].events = POLLIN;
596
597#ifdef ZVBI_COMPILED
598 if (sys->vbi != NULL)
599 {
600 ufd[1].fd = GetFdVBI (sys->vbi);
601 ufd[1].events = POLLIN;
602 numfds++;
603 }
604#endif
605
606 for (;;)
607 {
608 /* Wait for data */
609 if (poll (ufd, numfds, -1) == -1)
610 {
611 if (errno != EINTR)
612 msg_Err (demux, "poll error: %s", vlc_strerror_c(errno));
613 continue;
614 }
615
616 if( ufd[0].revents )
617 {
618 int canc = vlc_savecancel ();
619 block_t *block = GrabVideo (VLC_OBJECT(demux), fd, sys->bufv);
620 if (block != NULL)
621 {
622 block->i_flags |= sys->block_flags;
623 es_out_Control (demux->out, ES_OUT_SET_PCR, block->i_pts);
624 es_out_Send (demux->out, sys->es, block);
625 }
626 vlc_restorecancel (canc);
627 }
628#ifdef ZVBI_COMPILED
629 if (sys->vbi != NULL && ufd[1].revents)
630 GrabVBI (demux, sys->vbi);
631#endif
632 }
633
634 assert (0);
635}

637static void *ReadThread (void *data)
638{
639 demux_t *demux = data;
640 demux_sys_t *sys = demux->p_sys;
641 int fd = sys->fd;
642 struct pollfd ufd[2];
643 nfds_t numfds = 1;
644
645 ufd[0].fd = fd;
646 ufd[0].events = POLLIN;
647
648#ifdef ZVBI_COMPILED
649 if (sys->vbi != NULL)
650 {
651 ufd[1].fd = GetFdVBI (sys->vbi);
652 ufd[1].events = POLLIN;
653 numfds++;
654 }
655#endif
656
657 for (;;)
658 {
659 /* Wait for data */
660 if (poll (ufd, numfds, -1) == -1)
661 {
662 if (errno != EINTR)
663 msg_Err (demux, "poll error: %s", vlc_strerror_c(errno));
664 continue;
665 }
666
667 if( ufd[0].revents )
668 {
669 block_t *block = block_Alloc (sys->blocksize);
670 if (unlikely(block == NULL))
671 {
672 msg_Err (demux, "read error: %s", vlc_strerror_c(errno));
673 v4l2_read (fd, NULL, 0); /* discard frame */
674 continue;
675 }
676 block->i_pts = block->i_dts = mdate ();
677 block->i_flags |= sys->block_flags;
678
679 int canc = vlc_savecancel ();
680 ssize_t val = v4l2_read (fd, block->p_buffer, block->i_buffer);
681 if (val != -1)
682 {
683 block->i_buffer = val;
684 es_out_Control (demux->out, ES_OUT_SET_PCR, block->i_pts);
685 es_out_Send (demux->out, sys->es, block);
686 }
687 else
688 block_Release (block);
689 vlc_restorecancel (canc);
690 }
691#ifdef ZVBI_COMPILED
692 if (sys->vbi != NULL && ufd[1].revents)
693 GrabVBI (demux, sys->vbi);
694#endif
695 }
696 assert (0);
697}

699static int DemuxControl( demux_t *demux, int query, va_list args )
700{
701 demux_sys_t *sys = demux->p_sys;
702
703 switch( query )
704 {
705 /* Special for access_demux */
706 case DEMUX_CAN_PAUSE:
707 case DEMUX_CAN_SEEK:
708 case DEMUX_CAN_CONTROL_PACE:
709 *va_arg( args, bool * ) = false;
710 return VLC_SUCCESS;
711
712 case DEMUX_GET_PTS_DELAY:
713 *va_arg(args,int64_t *) = INT64_C(1000)
714 * var_InheritInteger( demux, "live-caching" );
715 return VLC_SUCCESS;
716
717 case DEMUX_GET_TIME:
718 *va_arg (args, int64_t *) = mdate() - sys->start;
719 return VLC_SUCCESS;
720
721 /* TODO implement others */
722 default:
723 return VLC_EGENERIC;
724 }
725
726 return VLC_EGENERIC;
727}
