39struct access_sys_t
40{
41 int fd;
42 uint32_t block_flags;
43 union
44 {
45 uint32_t bufc;
46 uint32_t blocksize;
47 };
48 struct buffer_t *bufv;
49 vlc_v4l2_ctrl_t *controls;
50};
51
52static block_t *MMapBlock (access_t *);
53static block_t *ReadBlock (access_t *);
54static int AccessControl( access_t *, int, va_list );
55static int InitVideo(access_t *, int, uint32_t);

57int AccessOpen( vlc_object_t *obj )
58{
59 access_t *access = (access_t *)obj;
60
61 access_InitFields( access );
62
63 access_sys_t *sys = calloc (1, sizeof (*sys));
64 if( unlikely(sys == NULL) )
65 return VLC_ENOMEM;
66 access->p_sys = sys;
67
68 ParseMRL( obj, access->psz_location );
69
70 char *path = var_InheritString (obj, CFG_PREFIX"dev");
71 if (unlikely(path == NULL))
72 goto error; /* probably OOM */
73
74 uint32_t caps;
75 int fd = OpenDevice (obj, path, &caps);
76 free (path);
77 if (fd == -1)
78 goto error;
79 sys->fd = fd;
80
81 if (InitVideo (access, fd, caps))
82 {
83 v4l2_close (fd);
84 goto error;
85 }
86
87 sys->controls = ControlsInit (VLC_OBJECT(access), fd);
88 access->pf_seek = NULL;
89 access->pf_control = AccessControl;
90 return VLC_SUCCESS;
91error:
92 free (sys);
93 return VLC_EGENERIC;
94}

96int InitVideo (access_t *access, int fd, uint32_t caps)
97{
98 access_sys_t *sys = access->p_sys;
99
100 if (!(caps & V4L2_CAP_VIDEO_CAPTURE))
101 {
102 msg_Err (access, "not a video capture device");
103 return -1;
104 }
105
106 v4l2_std_id std;
107 if (SetupInput (VLC_OBJECT(access), fd, &std))
108 return -1;
109
110 /* NOTE: The V4L access_demux expects a VLC FOURCC as "chroma". It is used to set the
111 * es_format_t structure correctly. However, the V4L access (*here*) has no use for a
112 * VLC FOURCC and expects a V4L2 format directly instead. That is confusing :-( */
113 uint32_t pixfmt = 0;
114 char *fmtstr = var_InheritString (access, CFG_PREFIX"chroma");
115 if (fmtstr != NULL && strlen (fmtstr) <= 4)
116 {
117 memcpy (&pixfmt, fmtstr, strlen (fmtstr));
118 free (fmtstr);
119 }
120 else
121 /* Use the default^Wprevious format if none specified */
122 {
123 struct v4l2_format fmt = { .type = V4L2_BUF_TYPE_VIDEO_CAPTURE };
124 if (v4l2_ioctl (fd, VIDIOC_G_FMT, &fmt) < 0)
125 {
126 msg_Err (access, "cannot get default format: %s",
127 vlc_strerror_c(errno));
128 return -1;
129 }
130 pixfmt = fmt.fmt.pix.pixelformat;
131 }
132 msg_Dbg (access, "selected format %4.4s", (const char *)&pixfmt);
133
134 struct v4l2_format fmt;
135 struct v4l2_streamparm parm;
136 if (SetupFormat (access, fd, pixfmt, &fmt, &parm))
137 return -1;
138
139 msg_Dbg (access, "%"PRIu32" bytes for complete image", fmt.fmt.pix.sizeimage);
140 /* Check interlacing */
141 switch (fmt.fmt.pix.field)
142 {
143 case V4L2_FIELD_INTERLACED:
144 msg_Dbg (access, "Interlacing setting: interleaved");
145 /*if (NTSC)
146 sys->block_flags = BLOCK_FLAG_BOTTOM_FIELD_FIRST;
147 else*/
148 sys->block_flags = BLOCK_FLAG_TOP_FIELD_FIRST;
149 break;
150 case V4L2_FIELD_INTERLACED_TB:
151 msg_Dbg (access, "Interlacing setting: interleaved top bottom" );
152 sys->block_flags = BLOCK_FLAG_TOP_FIELD_FIRST;
153 break;
154 case V4L2_FIELD_INTERLACED_BT:
155 msg_Dbg (access, "Interlacing setting: interleaved bottom top" );
156 sys->block_flags = BLOCK_FLAG_BOTTOM_FIELD_FIRST;
157 break;
158 default:
159 break;
160 }
161
162 /* Init I/O method */
163 if (caps & V4L2_CAP_STREAMING)
164 {
165 sys->bufc = 4;
166 sys->bufv = StartMmap (VLC_OBJECT(access), fd, &sys->bufc);
167 if (sys->bufv == NULL)
168 return -1;
169 access->pf_block = MMapBlock;
170 }
171 else if (caps & V4L2_CAP_READWRITE)
172 {
173 sys->blocksize = fmt.fmt.pix.sizeimage;
174 sys->bufv = NULL;
175 access->pf_block = ReadBlock;
176 }
177 else
178 {
179 msg_Err (access, "no supported capture method");
180 return -1;
181 }
182
183 return 0;
184}

186void AccessClose( vlc_object_t *obj )
187{
188 access_t *access = (access_t *)obj;
189 access_sys_t *sys = access->p_sys;
190
191 if (sys->bufv != NULL)
192 StopMmap (sys->fd, sys->bufv, sys->bufc);
193 ControlsDeinit( obj, sys->controls );
194 v4l2_close (sys->fd);
195 free( sys );
196}

198/* Wait for data */
199static int AccessPoll (access_t *access)
200{
201 access_sys_t *sys = access->p_sys;
202 struct pollfd ufd;
203
204 ufd.fd = sys->fd;
205 ufd.events = POLLIN;
206
207 switch (poll (&ufd, 1, 500))
208 {
209 case -1:
210 if (errno == EINTR)
211 case 0:
212 /* FIXME: kill this case (arbitrary timeout) */
213 return -1;
214 msg_Err (access, "poll error: %s", vlc_strerror_c(errno));
215 access->info.b_eof = true;
216 return -1;
217 }
218 return 0;
219}

222static block_t *MMapBlock (access_t *access)
223{
224 access_sys_t *sys = access->p_sys;
225
226 if (AccessPoll (access))
227 return NULL;
228
229 block_t *block = GrabVideo (VLC_OBJECT(access), sys->fd, sys->bufv);
230 if( block != NULL )
231 {
232 block->i_pts = block->i_dts = mdate();
233 block->i_flags |= sys->block_flags;
234 }
235 return block;
236}
237
238static block_t *ReadBlock (access_t *access)
239{
240 access_sys_t *sys = access->p_sys;
241
242 if (AccessPoll (access))
243 return NULL;
244
245 block_t *block = block_Alloc (sys->blocksize);
246 if (unlikely(block == NULL))
247 return NULL;
248
249 ssize_t val = v4l2_read (sys->fd, block->p_buffer, block->i_buffer);
250 if (val < 0)
251 {
252 block_Release (block);
253 msg_Err (access, "cannot read buffer: %s", vlc_strerror_c(errno));
254 access->info.b_eof = true;
255 return NULL;
256 }
257
258 block->i_buffer = val;
259 access->info.i_pos += val;
260 return block;
261}
262
263static int AccessControl( access_t *access, int query, va_list args )
264{
265 switch( query )
266 {
267 case ACCESS_CAN_SEEK:
268 case ACCESS_CAN_FASTSEEK:
269 case ACCESS_CAN_PAUSE:
270 case ACCESS_CAN_CONTROL_PACE:
271 *va_arg( args, bool* ) = false;
272 break;
273
274 case ACCESS_GET_PTS_DELAY:
275 *va_arg(args,int64_t *) = INT64_C(1000)
276 * var_InheritInteger( access, "live-caching" );
277 break;
278
279 case ACCESS_SET_PAUSE_STATE:
280 /* Nothing to do */
281 break;
282
283 default:
284 return VLC_EGENERIC;
285
286 }
287 return VLC_SUCCESS;
288}
