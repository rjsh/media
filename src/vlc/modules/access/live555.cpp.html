279/*****************************************************************************
280 * DemuxOpen:
281 *****************************************************************************/
282static int Open ( vlc_object_t *p_this )
283{
284 demux_t *p_demux = (demux_t*)p_this;
285 demux_sys_t *p_sys = NULL;
286
287 int i_return;
288 int i_error = VLC_EGENERIC;
289
290 if( p_demux->s )
291 {
292 /* See if it looks like a SDP
293 v, o, s fields are mandatory and in this order */
294 const uint8_t *p_peek;
295 if( stream_Peek( p_demux->s, &p_peek, 7 ) < 7 ) return VLC_EGENERIC;
296
297 if( memcmp( p_peek, "v=0\r\n", 5 ) &&
298 memcmp( p_peek, "v=0\n", 4 ) &&
299 ( p_peek[0] < 'a' || p_peek[0] > 'z' || p_peek[1] != '=' ) )
300 {
301 return VLC_EGENERIC;
302 }
303 }
304
305 p_demux->pf_demux = Demux;
306 p_demux->pf_control= Control;
307 p_demux->p_sys = p_sys = (demux_sys_t*)calloc( 1, sizeof( demux_sys_t ) );
308 if( !p_sys ) return VLC_ENOMEM;
309
310 msg_Dbg( p_demux, "version "LIVEMEDIA_LIBRARY_VERSION_STRING );
311
312 TAB_INIT( p_sys->i_track, p_sys->track );
313 p_sys->f_npt = 0.;
314 p_sys->f_npt_start = 0.;
315 p_sys->f_npt_length = 0.;
316 p_sys->b_no_data = true;
317 p_sys->psz_path = strdup( p_demux->psz_location );
318 p_sys->b_force_mcast = var_InheritBool( p_demux, "rtsp-mcast" );
319 p_sys->f_seek_request = -1;
320
321 /* parse URL for rtsp://[user:[passwd]@]serverip:port/options */
322 vlc_UrlParse( &p_sys->url, p_sys->psz_path, 0 );
323
324 if( ( p_sys->scheduler = BasicTaskScheduler::createNew() ) == NULL )
325 {
326 msg_Err( p_demux, "BasicTaskScheduler::createNew failed" );
327 goto error;
328 }
329 if( !( p_sys->env = BasicUsageEnvironment::createNew(*p_sys->scheduler) ) )
330 {
331 msg_Err( p_demux, "BasicUsageEnvironment::createNew failed" );
332 goto error;
333 }
334
335 if( strcasecmp( p_demux->psz_access, "sdp" ) )
336 {
337 char *p = p_sys->psz_path;
338 while( (p = strchr( p, ' ' )) != NULL ) *p = '+';
339 }
340
341 if( p_demux->s != NULL )
342 {
343 /* Gather the complete sdp file */
344 int i_sdp = 0;
345 int i_sdp_max = 1000;
346 uint8_t *p_sdp = (uint8_t*) malloc( i_sdp_max );
347
348 if( !p_sdp )
349 {
350 i_error = VLC_ENOMEM;
351 goto error;
352 }
353
354 for( ;; )
355 {
356 int i_read = stream_Read( p_demux->s, &p_sdp[i_sdp],
357 i_sdp_max - i_sdp - 1 );
358
359 if( !vlc_object_alive (p_demux) )
360 {
361 free( p_sdp );
362 goto error;
363 }
364
365 if( i_read < 0 )
366 {
367 msg_Err( p_demux, "failed to read SDP" );
368 free( p_sdp );
369 goto error;
370 }
371
372 i_sdp += i_read;
373
374 if( i_read < i_sdp_max - i_sdp - 1 )
375 {
376 p_sdp[i_sdp] = '\0';
377 break;
378 }
379
380 i_sdp_max += 1000;
381 p_sdp = (uint8_t*)xrealloc( p_sdp, i_sdp_max );
382 }
383 p_sys->p_sdp = (char*)p_sdp;
384 }
385 else if( ( i_return = Connect( p_demux ) ) != VLC_SUCCESS )
386 {
387 msg_Err( p_demux, "Failed to connect with rtsp://%s", p_sys->psz_path );
388 goto error;
389 }
390
391 if( p_sys->p_sdp == NULL )
392 {
393 msg_Err( p_demux, "Failed to retrieve the RTSP Session Description" );
394 i_error = VLC_ENOMEM;
395 goto error;
396 }
397
398 if( ( i_return = SessionsSetup( p_demux ) ) != VLC_SUCCESS )
399 {
400 msg_Err( p_demux, "Nothing to play for rtsp://%s", p_sys->psz_path );
401 goto error;
402 }
403
404 if( p_sys->b_real ) goto error;
405
406 if( ( i_return = Play( p_demux ) ) != VLC_SUCCESS )
407 goto error;
408
409 if( p_sys->p_out_asf && ParseASF( p_demux ) )
410 {
411 msg_Err( p_demux, "cannot find a usable asf header" );
412 /* TODO Clean tracks */
413 goto error;
414 }
415
416 if( p_sys->i_track <= 0 )
417 goto error;
418
419 return VLC_SUCCESS;
420
421error:
422 Close( p_this );
423 return i_error;
424}

426/*****************************************************************************
427 * DemuxClose:
428 *****************************************************************************/
429static void Close( vlc_object_t *p_this )
430{
431 demux_t *p_demux = (demux_t*)p_this;
432 demux_sys_t *p_sys = p_demux->p_sys;
433
434 if( p_sys->p_timeout )
435 {
436 vlc_cancel( p_sys->p_timeout->handle );
437 vlc_join( p_sys->p_timeout->handle, NULL );
438 free( p_sys->p_timeout );
439 }
440
441 if( p_sys->rtsp && p_sys->ms ) p_sys->rtsp->sendTeardownCommand( *p_sys->ms, NULL );
442 if( p_sys->ms ) Medium::close( p_sys->ms );
443 if( p_sys->rtsp ) RTSPClient::close( p_sys->rtsp );
444 if( p_sys->env ) p_sys->env->reclaim();
445
446 for( int i = 0; i < p_sys->i_track; i++ )
447 {
448 live_track_t *tk = p_sys->track[i];
449
450 if( tk->b_muxed ) stream_Delete( tk->p_out_muxed );
451 es_format_Clean( &tk->fmt );
452 free( tk->p_buffer );
453 free( tk );
454 }
455 TAB_CLEAN( p_sys->i_track, p_sys->track );
456 if( p_sys->p_out_asf ) stream_Delete( p_sys->p_out_asf );
457 delete p_sys->scheduler;
458 free( p_sys->p_sdp );
459 free( p_sys->psz_path );
460
461 vlc_UrlClean( &p_sys->url );
462
463 free( p_sys );
464}

479/* return true if the RTSP command succeeded */
480static bool wait_Live555_response( demux_t *p_demux, int i_timeout = 0 /* ms */ )
481{
482 TaskToken task;
483 demux_sys_t * p_sys = p_demux->p_sys;
484 p_sys->event_rtsp = 0;
485 if( i_timeout > 0 )
486 {
487 /* Create a task that will be called if we wait more than timeout ms */
488 task = p_sys->scheduler->scheduleDelayedTask( i_timeout*1000,
489 TaskInterruptRTSP,
490 p_demux );
491 }
492 p_sys->event_rtsp = 0;
493 p_sys->b_error = true;
494 p_sys->i_live555_ret = 0;
495 p_sys->scheduler->doEventLoop( &p_sys->event_rtsp );
496 //here, if b_error is true and i_live555_ret = 0 we didn't receive a response
497 if( i_timeout > 0 )
498 {
499 /* remove the task */
500 p_sys->scheduler->unscheduleDelayedTask( task );
501 }
502 return !p_sys->b_error;
503}

505static void continueAfterDESCRIBE( RTSPClient* client, int result_code,
506 char* result_string )
507{
508 RTSPClientVlc *client_vlc = static_cast<RTSPClientVlc *> ( client );
509 demux_sys_t *p_sys = client_vlc->p_sys;
510 p_sys->i_live555_ret = result_code;
511 if ( result_code == 0 )
512 {
513 char* sdpDescription = result_string;
514 free( p_sys->p_sdp );
515 p_sys->p_sdp = NULL;
516 if( sdpDescription )
517 {
518 p_sys->p_sdp = strdup( sdpDescription );
519 p_sys->b_error = false;
520 }
521 }
522 else
523 p_sys->b_error = true;
524 delete[] result_string;
525 p_sys->event_rtsp = 1;
526}

528static void continueAfterOPTIONS( RTSPClient* client, int result_code,
529 char* result_string )
530{
531 RTSPClientVlc *client_vlc = static_cast<RTSPClientVlc *> (client);
532 demux_sys_t *p_sys = client_vlc->p_sys;
533 p_sys->b_get_param =
534 // If OPTIONS fails, assume GET_PARAMETER is not supported but
535 // still continue on with the stream. Some servers (foscam)
536 // return 501/not implemented for OPTIONS.
537 result_code == 0
538 && result_string != NULL
539 && strstr( result_string, "GET_PARAMETER" ) != NULL;
540 client->sendDescribeCommand( continueAfterDESCRIBE );
541 delete[] result_string;
542}

544/*****************************************************************************
545 * Connect: connects to the RTSP server to setup the session DESCRIBE
546 *****************************************************************************/
547static int Connect( demux_t *p_demux )
548{
549 demux_sys_t *p_sys = p_demux->p_sys;
550 Authenticator authenticator;
551 char *psz_user = NULL;
552 char *psz_pwd = NULL;
553 char *psz_url = NULL;
554 int i_http_port = 0;
555 int i_ret = VLC_SUCCESS;
556 const int i_timeout = var_InheritInteger( p_demux, "ipv4-timeout" );
557
558 /* Get the user name and password */
559 if( p_sys->url.psz_username || p_sys->url.psz_password )
560 {
561 /* Create the URL by stripping away the username/password part */
562 if( p_sys->url.i_port == 0 )
563 p_sys->url.i_port = 554;
564 if( asprintf( &psz_url, "rtsp://%s:%d%s",
565 strempty( p_sys->url.psz_host ),
566 p_sys->url.i_port,
567 strempty( p_sys->url.psz_path ) ) == -1 )
568 return VLC_ENOMEM;
569
570 psz_user = strdup( strempty( p_sys->url.psz_username ) );
571 psz_pwd = strdup( strempty( p_sys->url.psz_password ) );
572 }
573 else
574 {
575 if( asprintf( &psz_url, "rtsp://%s", p_sys->psz_path ) == -1 )
576 return VLC_ENOMEM;
577
578 psz_user = var_InheritString( p_demux, "rtsp-user" );
579 psz_pwd = var_InheritString( p_demux, "rtsp-pwd" );
580 }
581
582createnew:
583 if( !vlc_object_alive (p_demux) )
584 {
585 i_ret = VLC_EGENERIC;
586 goto bailout;
587 }
588
589 if( var_CreateGetBool( p_demux, "rtsp-http" ) )
590 i_http_port = var_InheritInteger( p_demux, "rtsp-http-port" );
591
592 p_sys->rtsp = new RTSPClientVlc( *p_sys->env, psz_url,
593 var_InheritInteger( p_demux, "verbose" ) > 1 ? 1 : 0,
594 "LibVLC/"VERSION, i_http_port, p_sys );
595 if( !p_sys->rtsp )
596 {
597 msg_Err( p_demux, "RTSPClient::createNew failed (%s)",
598 p_sys->env->getResultMsg() );
599 i_ret = VLC_EGENERIC;
600 goto bailout;
601 }
602
603 /* Kasenna enables KeepAlive by analysing the User-Agent string.
604 * Appending _KA to the string should be enough to enable this feature,
605 * however, there is a bug where the _KA doesn't get parsed from the
606 * default User-Agent as created by VLC/Live555 code. This is probably due
607 * to spaces in the string or the string being too long. Here we override
608 * the default string with a more compact version.
609 */
610 if( var_InheritBool( p_demux, "rtsp-kasenna" ))
611 {
612 p_sys->rtsp->setUserAgentString( "VLC_MEDIA_PLAYER_KA" );
613 }
614
615describe:
616 authenticator.setUsernameAndPassword( psz_user, psz_pwd );
617
618 p_sys->rtsp->sendOptionsCommand( &continueAfterOPTIONS, &authenticator );
619
620 if( !wait_Live555_response( p_demux, i_timeout ) )
621 {
622 int i_code = p_sys->i_live555_ret;
623 if( i_code == 401 )
624 {
625 msg_Dbg( p_demux, "authentication failed" );
626
627 free( psz_user );
628 free( psz_pwd );
629 dialog_Login( p_demux, &psz_user, &psz_pwd,
630 _("RTSP authentication"), "%s",
631 _("Please enter a valid login name and a password.") );
632 if( psz_user != NULL && psz_pwd != NULL )
633 {
634 msg_Dbg( p_demux, "retrying with user=%s", psz_user );
635 goto describe;
636 }
637 }
638 else if( i_code > 0 && i_code != 404 && !var_GetBool( p_demux, "rtsp-http" ) )
639 {
640 /* Perhaps a firewall is being annoying. Try HTTP tunneling mode */
641 msg_Dbg( p_demux, "we will now try HTTP tunneling mode" );
642 var_SetBool( p_demux, "rtsp-http", true );
643 if( p_sys->rtsp ) RTSPClient::close( p_sys->rtsp );
644 p_sys->rtsp = NULL;
645 goto createnew;
646 }
647 else
648 {
649 if( i_code == 0 )
650 msg_Dbg( p_demux, "connection timeout" );
651 else
652 {
653 msg_Dbg( p_demux, "connection error %d", i_code );
654 if( i_code == 403 )
655 dialog_Fatal( p_demux, _("RTSP connection failed"),
656 _("Access to the stream is denied by the server configuration.") );
657 }
658 if( p_sys->rtsp ) RTSPClient::close( p_sys->rtsp );
659 p_sys->rtsp = NULL;
660 }
661 i_ret = VLC_EGENERIC;
662 }
663
664bailout:
665 /* malloc-ated copy */
666 free( psz_url );
667 free( psz_user );
668 free( psz_pwd );
669
670 return i_ret;
671}

1209/*****************************************************************************
1210 * Play: starts the actual playback of the stream
1211 *****************************************************************************/
1212static int Play( demux_t *p_demux )
1213{
1214 demux_sys_t *p_sys = p_demux->p_sys;
1215
1216 if( p_sys->rtsp )
1217 {
1218 /* The PLAY */
1219 p_sys->rtsp->sendPlayCommand( *p_sys->ms, default_live555_callback, p_sys->f_npt_start, -1, 1 );
1220
1221 if( !wait_Live555_response(p_demux) )
1222 {
1223 msg_Err( p_demux, "RTSP PLAY failed %s", p_sys->env->getResultMsg() );
1224 return VLC_EGENERIC;
1225 }
1226
1227 /* Retrieve the timeout value and set up a timeout prevention thread */
1228 p_sys->i_timeout = p_sys->rtsp->sessionTimeoutParameter();
1229 if( p_sys->i_timeout <= 0 )
1230 p_sys->i_timeout = 60; /* default value from RFC2326 */
1231
1232 /* start timeout-thread only if GET_PARAMETER is supported by the server */
1233 /* or start it if wmserver dialect, since they don't report that GET_PARAMETER is supported correctly */
1234 if( !p_sys->p_timeout && ( p_sys->b_get_param || var_InheritBool( p_demux, "rtsp-wmserver" ) ) )
1235 {
1236 msg_Dbg( p_demux, "We have a timeout of %d seconds", p_sys->i_timeout );
1237 p_sys->p_timeout = (timeout_thread_t *)malloc( sizeof(timeout_thread_t) );
1238 if( p_sys->p_timeout )
1239 {
1240 memset( p_sys->p_timeout, 0, sizeof(timeout_thread_t) );
1241 p_sys->p_timeout->p_sys = p_demux->p_sys; /* lol, object recursion :D */
1242 if( vlc_clone( &p_sys->p_timeout->handle, TimeoutPrevention,
1243 p_sys->p_timeout, VLC_THREAD_PRIORITY_LOW ) )
1244 {
1245 msg_Err( p_demux, "cannot spawn liveMedia timeout thread" );
1246 free( p_sys->p_timeout );
1247 p_sys->p_timeout = NULL;
1248 }
1249 else
1250 msg_Dbg( p_demux, "spawned timeout thread" );
1251 }
1252 else
1253 msg_Err( p_demux, "cannot spawn liveMedia timeout thread" );
1254 }
1255 }
1256 p_sys->i_pcr = 0;
1257
1258 /* Retrieve the starttime if possible */
1259 p_sys->f_npt_start = p_sys->ms->playStartTime();
1260 if( p_sys->ms->playEndTime() > 0 )
1261 p_sys->f_npt_length = p_sys->ms->playEndTime();
1262
1263 msg_Dbg( p_demux, "play start: %f stop:%f", p_sys->f_npt_start, p_sys->f_npt_length );
1264 return VLC_SUCCESS;
1265}

1271static int Demux( demux_t *p_demux )

1273 demux_sys_t *p_sys = p_demux->p_sys;
1274 TaskToken task;
1275
1276 bool b_send_pcr = true;
1277 int i;
1278
1279 /* Check if we need to send the server a Keep-A-Live signal */
1280 if( p_sys->b_timeout_call && p_sys->rtsp && p_sys->ms )
1281 {
1282 char *psz_bye = NULL;
1283 p_sys->rtsp->sendGetParameterCommand( *p_sys->ms, NULL, psz_bye );
1284 p_sys->b_timeout_call = false;
1285 }
1286
1287 for( i = 0; i < p_sys->i_track; i++ )
1288 {
1289 live_track_t *tk = p_sys->track[i];
1290
1291 if( tk->p_es )
1292 {
1293 bool b;
1294 es_out_Control( p_demux->out, ES_OUT_GET_ES_STATE, tk->p_es, &b );
1295 if( !b && tk->b_selected )
1296 {
1297 tk->b_selected = false;
1298 p_sys->rtsp->sendTeardownCommand( *tk->sub, NULL );
1299 }
1300 else if( b && !tk->b_selected)
1301 {
1302 bool b_rtsp_tcp = var_GetBool( p_demux, "rtsp-tcp" ) ||
1303 var_GetBool( p_demux, "rtsp-http" );
1304 p_sys->rtsp->sendSetupCommand( *tk->sub, default_live555_callback, False,
1305 toBool( b_rtsp_tcp ),
1306 toBool( p_sys->b_force_mcast && !b_rtsp_tcp ) );
1307 if( !wait_Live555_response( p_demux ) )
1308 {
1309 msg_Err( p_demux, "SETUP of'%s/%s' failed %s",
1310 tk->sub->mediumName(), tk->sub->codecName(),
1311 p_sys->env->getResultMsg() );
1312 }
1313 else
1314 {
1315 p_sys->rtsp->sendPlayCommand( *tk->sub, default_live555_callback, -1, -1, p_sys->ms->scale() );
1316 if( !wait_Live555_response(p_demux) )
1317 {
1318 msg_Err( p_demux, "RTSP PLAY failed %s", p_sys->env->getResultMsg() );
1319 p_sys->rtsp->sendTeardownCommand( *tk->sub, NULL );
1320 }
1321 else
1322 tk->b_selected = true;
1323 }
1324 if( !tk->b_selected )
1325 es_out_Control( p_demux->out, ES_OUT_SET_ES_STATE, tk->p_es, false );
1326 }
1327 }
1328
1329 if( tk->b_asf || tk->b_muxed )
1330 b_send_pcr = false;
1331 }
1332 if( p_sys->i_pcr > 0 )
1333 {
1334 if( b_send_pcr )
1335 es_out_Control( p_demux->out, ES_OUT_SET_PCR, 1 + p_sys->i_pcr );
1336 }
1337
1338 /* First warn we want to read data */
1339 p_sys->event_data = 0;
1340 for( i = 0; i < p_sys->i_track; i++ )
1341 {
1342 live_track_t *tk = p_sys->track[i];
1343
1344 if( tk->waiting == 0 )
1345 {
1346 tk->waiting = 1;
1347 tk->sub->readSource()->getNextFrame( tk->p_buffer, tk->i_buffer,
1348 StreamRead, tk, StreamClose, tk );
1349 }
1350 }
1351 /* Create a task that will be called if we wait more than 300ms */
1352 task = p_sys->scheduler->scheduleDelayedTask( 300000, TaskInterruptData, p_demux );
1353
1354 /* Do the read */
1355 p_sys->scheduler->doEventLoop( &p_sys->event_data );
1356
1357 /* remove the task */
1358 p_sys->scheduler->unscheduleDelayedTask( task );
1359
1360 /* Check for gap in pts value */
1361 for( i = 0; i < p_sys->i_track; i++ )
1362 {
1363 live_track_t *tk = p_sys->track[i];
1364
1365 if( !tk->b_muxed && !tk->b_rtcp_sync &&
1366 tk->sub->rtpSource() && tk->sub->rtpSource()->hasBeenSynchronizedUsingRTCP() )
1367 {
1368 msg_Dbg( p_demux, "tk->rtpSource->hasBeenSynchronizedUsingRTCP()" );
1369
1370 es_out_Control( p_demux->out, ES_OUT_RESET_PCR );
1371 tk->b_rtcp_sync = true;
1372 /* reset PCR */
1373 tk->i_pts = VLC_TS_INVALID;
1374 tk->f_npt = 0.;
1375 p_sys->i_pcr = 0;
1376 p_sys->f_npt = 0.;
1377 }
1378 }
1379
1380 if( p_sys->b_multicast && p_sys->b_no_data &&
1381 ( p_sys->i_no_data_ti > 120 ) )
1382 {
1383 /* FIXME Make this configurable
1384 msg_Err( p_demux, "no multicast data received in 36s, aborting" );
1385 return 0;
1386 */
1387 }
1388 else if( !p_sys->b_multicast && !p_sys->b_paused &&
1389 p_sys->b_no_data && ( p_sys->i_no_data_ti > 34 ) )
1390 {
1391 bool b_rtsp_tcp = var_GetBool( p_demux, "rtsp-tcp" ) ||
1392 var_GetBool( p_demux, "rtsp-http" );
1393
1394 if( !b_rtsp_tcp && p_sys->rtsp && p_sys->ms )
1395 {
1396 msg_Warn( p_demux, "no data received in 10s. Switching to TCP" );
1397 if( RollOverTcp( p_demux ) )
1398 {
1399 msg_Err( p_demux, "TCP rollover failed, aborting" );
1400 return 0;
1401 }
1402 return 1;
1403 }
1404 msg_Err( p_demux, "no data received in 10s, aborting" );
1405 return 0;
1406 }
1407 else if( !p_sys->b_multicast && !p_sys->b_paused &&
1408 ( p_sys->i_no_data_ti > 34 ) )
1409 {
1410 /* EOF ? */
1411 msg_Warn( p_demux, "no data received in 10s, eof ?" );
1412 return 0;
1413 }
1414 return p_sys->b_error ? 0 : 1;
1415}

1417/*****************************************************************************
1418 * Control:
1419 *****************************************************************************/
1420static int Control( demux_t *p_demux, int i_query, va_list args )
1421{
1422 demux_sys_t *p_sys = p_demux->p_sys;
1423 int64_t *pi64, i64;
1424 double *pf, f;
1425 bool *pb, *pb2;
1426 int *pi_int;
1427
1428 switch( i_query )
1429 {
1430 case DEMUX_GET_TIME:
1431 pi64 = (int64_t*)va_arg( args, int64_t * );
1432 if( p_sys->f_npt > 0 )
1433 {
1434 *pi64 = (int64_t)(p_sys->f_npt * 1000000.);
1435 return VLC_SUCCESS;
1436 }
1437 return VLC_EGENERIC;
1438
1439 case DEMUX_GET_LENGTH:
1440 pi64 = (int64_t*)va_arg( args, int64_t * );
1441 if( p_sys->f_npt_length > 0 )
1442 {
1443 double d_length = p_sys->f_npt_length * 1000000.0;
1444 if( d_length >= INT64_MAX )
1445 *pi64 = INT64_MAX;
1446 else
1447 *pi64 = (int64_t)d_length;
1448 return VLC_SUCCESS;
1449 }
1450 return VLC_EGENERIC;
1451
1452 case DEMUX_GET_POSITION:
1453 pf = (double*)va_arg( args, double* );
1454 if( (p_sys->f_npt_length > 0) && (p_sys->f_npt > 0) )
1455 {
1456 *pf = p_sys->f_npt / p_sys->f_npt_length;
1457 return VLC_SUCCESS;
1458 }
1459 return VLC_EGENERIC;
1460
1461 case DEMUX_SET_POSITION:
1462 case DEMUX_SET_TIME:
1463 if( p_sys->rtsp && (p_sys->f_npt_length > 0) )
1464 {
1465 int i;
1466 float time;
1467
1468 if( (i_query == DEMUX_SET_TIME) && (p_sys->f_npt > 0) )
1469 {
1470 i64 = (int64_t)va_arg( args, int64_t );
1471 time = (float)(i64 / 1000000.0); /* in second */
1472 }
1473 else if( i_query == DEMUX_SET_TIME )
1474 return VLC_EGENERIC;
1475 else
1476 {
1477 f = (double)va_arg( args, double );
1478 time = f * p_sys->f_npt_length; /* in second */
1479 }
1480
1481 if( p_sys->b_paused )
1482 {
1483 p_sys->f_seek_request = time;
1484 return VLC_SUCCESS;
1485 }
1486
1487 p_sys->rtsp->sendPauseCommand( *p_sys->ms, default_live555_callback );
1488
1489 if( !wait_Live555_response( p_demux ) )
1490 {
1491 msg_Err( p_demux, "PAUSE before seek failed %s",
1492 p_sys->env->getResultMsg() );
1493 return VLC_EGENERIC;
1494 }
1495
1496 p_sys->rtsp->sendPlayCommand( *p_sys->ms, default_live555_callback, time, -1, 1 );
1497
1498 if( !wait_Live555_response( p_demux ) )
1499 {
1500 msg_Err( p_demux, "seek PLAY failed %s",
1501 p_sys->env->getResultMsg() );
1502 return VLC_EGENERIC;
1503 }
1504 p_sys->i_pcr = 0;
1505
1506 for( i = 0; i < p_sys->i_track; i++ )
1507 {
1508 p_sys->track[i]->b_rtcp_sync = false;
1509 p_sys->track[i]->i_pts = VLC_TS_INVALID;
1510 }
1511
1512 /* Retrieve the starttime if possible */
1513 p_sys->f_npt = p_sys->f_npt_start = p_sys->ms->playStartTime();
1514
1515 /* Retrieve the duration if possible */
1516 if( p_sys->ms->playEndTime() > 0 )
1517 p_sys->f_npt_length = p_sys->ms->playEndTime();
1518
1519 msg_Dbg( p_demux, "seek start: %f stop:%f", p_sys->f_npt_start, p_sys->f_npt_length );
1520 return VLC_SUCCESS;
1521 }
1522 return VLC_EGENERIC;
1523
1524 /* Special for access_demux */
1525 case DEMUX_CAN_PAUSE:
1526 case DEMUX_CAN_SEEK:
1527 pb = (bool*)va_arg( args, bool * );
1528 if( p_sys->rtsp && p_sys->f_npt_length > 0 )
1529 /* Not always true, but will be handled in SET_PAUSE_STATE */
1530 *pb = true;
1531 else
1532 *pb = false;
1533 return VLC_SUCCESS;
1534
1535 case DEMUX_CAN_CONTROL_PACE:
1536 pb = (bool*)va_arg( args, bool * );
1537
1538#if 1 /* Disable for now until we have a clock synchro algo
1539 * which works with something else than MPEG over UDP */
1540 *pb = false;
1541#else
1542 *pb = true;
1543#endif
1544 return VLC_SUCCESS;
1545
1546 case DEMUX_CAN_CONTROL_RATE:
1547 pb = (bool*)va_arg( args, bool * );
1548 pb2 = (bool*)va_arg( args, bool * );
1549
1550 *pb = (p_sys->rtsp != NULL) &&
1551 (p_sys->f_npt_length > 0) &&
1552 ( !var_GetBool( p_demux, "rtsp-kasenna" ) ||
1553 !var_GetBool( p_demux, "rtsp-wmserver" ) );
1554 *pb2 = false;
1555 return VLC_SUCCESS;
1556
1557 case DEMUX_SET_RATE:
1558 {
1559 double f_scale, f_old_scale;
1560
1561 if( !p_sys->rtsp || (p_sys->f_npt_length <= 0) ||
1562 var_GetBool( p_demux, "rtsp-kasenna" ) ||
1563 var_GetBool( p_demux, "rtsp-wmserver" ) )
1564 return VLC_EGENERIC;
1565
1566 /* According to RFC 2326 p56 chapter 12.35 a RTSP server that
1567 * supports Scale:
1568 *
1569 * "[...] should try to approximate the viewing rate, but
1570 * may restrict the range of scale values that it supports.
1571 * The response MUST contain the actual scale value chosen
1572 * by the server."
1573 *
1574 * Scale = 1 indicates normal play
1575 * Scale > 1 indicates fast forward
1576 * Scale < 1 && Scale > 0 indicates slow motion
1577 * Scale < 0 value indicates rewind
1578 */
1579
1580 pi_int = (int*)va_arg( args, int * );
1581 f_scale = (double)INPUT_RATE_DEFAULT / (*pi_int);
1582 f_old_scale = p_sys->ms->scale();
1583
1584 /* Passing -1 for the start and end time will mean liveMedia won't
1585 * create a Range: section for the RTSP message. The server should
1586 * pick up from the current position */
1587 p_sys->rtsp->sendPlayCommand( *p_sys->ms, default_live555_callback, -1, -1, f_scale );
1588
1589 if( !wait_Live555_response( p_demux ) )
1590 {
1591 msg_Err( p_demux, "PLAY with Scale %0.2f failed %s", f_scale,
1592 p_sys->env->getResultMsg() );
1593 return VLC_EGENERIC;
1594 }
1595
1596 if( p_sys->ms->scale() == f_old_scale )
1597 {
1598 msg_Err( p_demux, "no scale change using old Scale %0.2f",
1599 p_sys->ms->scale() );
1600 return VLC_EGENERIC;
1601 }
1602
1603 /* ReSync the stream */
1604 p_sys->f_npt_start = 0;
1605 p_sys->i_pcr = 0;
1606 p_sys->f_npt = 0.0;
1607
1608 *pi_int = (int)( INPUT_RATE_DEFAULT / p_sys->ms->scale() );
1609 msg_Dbg( p_demux, "PLAY with new Scale %0.2f (%d)", p_sys->ms->scale(), (*pi_int) );
1610 return VLC_SUCCESS;
1611 }
1612
1613 case DEMUX_SET_PAUSE_STATE:
1614 {
1615 bool b_pause = (bool)va_arg( args, int );
1616 if( p_sys->rtsp == NULL )
1617 return VLC_EGENERIC;
1618
1619 if( b_pause == p_sys->b_paused )
1620 return VLC_SUCCESS;
1621 if( b_pause )
1622 p_sys->rtsp->sendPauseCommand( *p_sys->ms, default_live555_callback );
1623 else
1624 p_sys->rtsp->sendPlayCommand( *p_sys->ms, default_live555_callback, p_sys->f_seek_request,
1625 -1.0f, p_sys->ms->scale() );
1626
1627 if( !wait_Live555_response( p_demux ) )
1628 {
1629 msg_Err( p_demux, "PLAY or PAUSE failed %s", p_sys->env->getResultMsg() );
1630 return VLC_EGENERIC;
1631 }
1632 p_sys->f_seek_request = -1;
1633 p_sys->b_paused = b_pause;
1634
1635 /* When we Pause, we'll need the TimeoutPrevention thread to
1636 * handle sending the "Keep Alive" message to the server.
1637 * Unfortunately Live555 isn't thread safe and so can't
1638 * do this normally while the main Demux thread is handling
1639 * a live stream. We end up with the Timeout thread blocking
1640 * waiting for a response from the server. So when we PAUSE
1641 * we set a flag that the TimeoutPrevention function will check
1642 * and if it's set, it will trigger the GET_PARAMETER message */
1643 if( p_sys->b_paused && p_sys->p_timeout != NULL )
1644 p_sys->p_timeout->b_handle_keep_alive = true;
1645 else if( !p_sys->b_paused && p_sys->p_timeout != NULL )
1646 p_sys->p_timeout->b_handle_keep_alive = false;
1647
1648 if( !p_sys->b_paused )
1649 {
1650 for( int i = 0; i < p_sys->i_track; i++ )
1651 {
1652 live_track_t *tk = p_sys->track[i];
1653 tk->b_rtcp_sync = false;
1654 tk->i_pts = VLC_TS_INVALID;
1655 p_sys->i_pcr = 0;
1656 es_out_Control( p_demux->out, ES_OUT_RESET_PCR );
1657 }
1658 }
1659
1660 /* Reset data received counter */
1661 p_sys->i_no_data_ti = 0;
1662
1663 /* Retrieve the starttime if possible */
1664 p_sys->f_npt_start = p_sys->ms->playStartTime();
1665
1666 /* Retrieve the duration if possible */
1667 if( p_sys->ms->playEndTime() )
1668 p_sys->f_npt_length = p_sys->ms->playEndTime();
1669
1670 msg_Dbg( p_demux, "pause start: %f stop:%f", p_sys->f_npt_start, p_sys->f_npt_length );
1671 return VLC_SUCCESS;
1672 }
1673 case DEMUX_GET_TITLE_INFO:
1674 case DEMUX_SET_TITLE:
1675 case DEMUX_SET_SEEKPOINT:
1676 return VLC_EGENERIC;
1677
1678 case DEMUX_GET_PTS_DELAY:
1679 pi64 = (int64_t*)va_arg( args, int64_t * );
1680 *pi64 = INT64_C(1000)
1681 * var_InheritInteger( p_demux, "network-caching" );
1682 return VLC_SUCCESS;
1683
1684 default:
1685 return VLC_EGENERIC;
1686 }
1687}

1756/*****************************************************************************
1757 *
1758 *****************************************************************************/
1759static block_t *StreamParseAsf( demux_t *p_demux, live_track_t *tk,
1760 bool b_marker,
1761 const uint8_t *p_data, unsigned i_size )
1762{
1763 const unsigned i_packet_size = p_demux->p_sys->asfh.i_min_data_packet_size;
1764 block_t *p_list = NULL;
1765
1766 while( i_size >= 4 )
1767 {
1768 unsigned i_flags = p_data[0];
1769 unsigned i_length_offset = (p_data[1] << 16) |
1770 (p_data[2] << 8) |
1771 (p_data[3] );
1772 bool b_length = i_flags & 0x40;
1773 bool b_relative_ts = i_flags & 0x20;
1774 bool b_duration = i_flags & 0x10;
1775 bool b_location_id = i_flags & 0x08;
1776
1777 //msg_Dbg( p_demux, "ASF: marker=%d size=%d : %c=%d id=%d",
1778 // b_marker, i_size, b_length ? 'L' : 'O', i_length_offset );
1779 unsigned i_header_size = 4;
1780 if( b_relative_ts )
1781 i_header_size += 4;
1782 if( b_duration )
1783 i_header_size += 4;
1784 if( b_location_id )
1785 i_header_size += 4;
1786
1787 if( i_header_size > i_size )
1788 {
1789 msg_Warn( p_demux, "Invalid header size" );
1790 break;
1791 }
1792
1793 /* XXX
1794 * When b_length is true, the streams I found do not seems to respect
1795 * the documentation.
1796 * From them, I have failed to find which choice between '__MIN()' or
1797 * 'i_length_offset - i_header_size' is the right one.
1798 */
1799 unsigned i_payload;
1800 if( b_length )
1801 i_payload = __MIN( i_length_offset, i_size - i_header_size);
1802 else
1803 i_payload = i_size - i_header_size;
1804
1805 if( !tk->p_asf_block )
1806 {
1807 tk->p_asf_block = block_Alloc( i_packet_size );
1808 if( !tk->p_asf_block )
1809 break;
1810 tk->p_asf_block->i_buffer = 0;
1811 }
1812 unsigned i_offset = b_length ? 0 : i_length_offset;
1813 if( i_offset == tk->p_asf_block->i_buffer && i_offset + i_payload <= i_packet_size )
1814 {
1815 memcpy( &tk->p_asf_block->p_buffer[i_offset], &p_data[i_header_size], i_payload );
1816 tk->p_asf_block->i_buffer += i_payload;
1817 if( b_marker )
1818 {
1819 /* We have a complete packet */
1820 tk->p_asf_block->i_buffer = i_packet_size;
1821 block_ChainAppend( &p_list, tk->p_asf_block );
1822 tk->p_asf_block = NULL;
1823 }
1824 }
1825 else
1826 {
1827 /* Reset on broken stream */
1828 msg_Err( p_demux, "Broken packet detected (%d vs %zu or %d + %d vs %d)",
1829 i_offset, tk->p_asf_block->i_buffer, i_offset, i_payload, i_packet_size);
1830 tk->p_asf_block->i_buffer = 0;
1831 }
1832
1833 /* */
1834 p_data += i_header_size + i_payload;
1835 i_size -= i_header_size + i_payload;
1836 }
1837 return p_list;
1838}

1840/*****************************************************************************
1841 *
1842 *****************************************************************************/
1843static void StreamRead( void *p_private, unsigned int i_size,
1844 unsigned int i_truncated_bytes, struct timeval pts,
1845 unsigned int duration )
1846{
1847 VLC_UNUSED( duration );
1848
1849 live_track_t *tk = (live_track_t*)p_private;
1850 demux_t *p_demux = tk->p_demux;
1851 demux_sys_t *p_sys = p_demux->p_sys;
1852 block_t *p_block;
1853
1854 //msg_Dbg( p_demux, "pts: %d", pts.tv_sec );
1855
1856 int64_t i_pts = (int64_t)pts.tv_sec * INT64_C(1000000) +
1857 (int64_t)pts.tv_usec;
1858
1859 /* XXX Beurk beurk beurk Avoid having negative value XXX */
1860 i_pts &= INT64_C(0x00ffffffffffffff);
1861
1862 /* Retrieve NPT for this pts */
1863 tk->f_npt = tk->sub->getNormalPlayTime(pts);
1864
1865 if( tk->b_quicktime && tk->p_es == NULL )
1866 {
1867 QuickTimeGenericRTPSource *qtRTPSource =
1868 (QuickTimeGenericRTPSource*)tk->sub->rtpSource();
1869 QuickTimeGenericRTPSource::QTState &qtState = qtRTPSource->qtState;
1870 uint8_t *sdAtom = (uint8_t*)&qtState.sdAtom[4];
1871
1872 /* Get codec information from the quicktime atoms :
1873 * http://developer.apple.com/quicktime/icefloe/dispatch026.html */
1874 if( tk->fmt.i_cat == VIDEO_ES ) {
1875 if( qtState.sdAtomSize < 16 + 32 )
1876 {
1877 /* invalid */
1878 p_sys->event_data = 0xff;
1879 tk->waiting = 0;
1880 return;
1881 }
1882 tk->fmt.i_codec = VLC_FOURCC(sdAtom[0],sdAtom[1],sdAtom[2],sdAtom[3]);
1883 tk->fmt.video.i_width = (sdAtom[28] << 8) | sdAtom[29];
1884 tk->fmt.video.i_height = (sdAtom[30] << 8) | sdAtom[31];
1885
1886 if( tk->fmt.i_codec == VLC_FOURCC('a', 'v', 'c', '1') )
1887 {
1888 uint8_t *pos = (uint8_t*)qtRTPSource->qtState.sdAtom + 86;
1889 uint8_t *endpos = (uint8_t*)qtRTPSource->qtState.sdAtom
1890 + qtRTPSource->qtState.sdAtomSize;
1891 while (pos+8 < endpos) {
1892 unsigned int atomLength = pos[0]<<24 | pos[1]<<16 | pos[2]<<8 | pos[3];
1893 if( atomLength == 0 || atomLength > (unsigned int)(endpos-pos)) break;
1894 if( memcmp(pos+4, "avcC", 4) == 0 &&
1895 atomLength > 8 &&
1896 atomLength <= INT_MAX )
1897 {
1898 tk->fmt.i_extra = atomLength-8;
1899 tk->fmt.p_extra = xmalloc( tk->fmt.i_extra );
1900 memcpy(tk->fmt.p_extra, pos+8, atomLength-8);
1901 break;
1902 }
1903 pos += atomLength;
1904 }
1905 }
1906 else
1907 {
1908 tk->fmt.i_extra = qtState.sdAtomSize - 16;
1909 tk->fmt.p_extra = xmalloc( tk->fmt.i_extra );
1910 memcpy( tk->fmt.p_extra, &sdAtom[12], tk->fmt.i_extra );
1911 }
1912 }
1913 else {
1914 if( qtState.sdAtomSize < 24 )
1915 {
1916 /* invalid */
1917 p_sys->event_data = 0xff;
1918 tk->waiting = 0;
1919 return;
1920 }
1921 tk->fmt.i_codec = VLC_FOURCC(sdAtom[0],sdAtom[1],sdAtom[2],sdAtom[3]);
1922 tk->fmt.audio.i_bitspersample = (sdAtom[22] << 8) | sdAtom[23];
1923 }
1924 tk->p_es = es_out_Add( p_demux->out, &tk->fmt );
1925 }
1926
1927#if 0
1928 fprintf( stderr, "StreamRead size=%d pts=%lld\n",
1929 i_size,
1930 pts.tv_sec * 1000000LL + pts.tv_usec );
1931#endif
1932
1933 /* grow buffer if it looks like buffer is too small, but don't eat
1934 * up all the memory on strange streams */
1935 if( i_truncated_bytes > 0 )
1936 {
1937 if( tk->i_buffer < 2000000 )
1938 {
1939 void *p_tmp;
1940 msg_Dbg( p_demux, "lost %d bytes", i_truncated_bytes );
1941 msg_Dbg( p_demux, "increasing buffer size to %d", tk->i_buffer * 2 );
1942 p_tmp = realloc( tk->p_buffer, tk->i_buffer * 2 );
1943 if( p_tmp == NULL )
1944 {
1945 msg_Warn( p_demux, "realloc failed" );
1946 }
1947 else
1948 {
1949 tk->p_buffer = (uint8_t*)p_tmp;
1950 tk->i_buffer *= 2;
1951 }
1952 }
1953
1954 if( tk->b_discard_trunc )
1955 {
1956 p_sys->event_data = 0xff;
1957 tk->waiting = 0;
1958 return;
1959 }
1960 }
1961
1962 assert( i_size <= tk->i_buffer );
1963
1964 if( tk->fmt.i_codec == VLC_CODEC_AMR_NB ||
1965 tk->fmt.i_codec == VLC_CODEC_AMR_WB )
1966 {
1967 AMRAudioSource *amrSource = (AMRAudioSource*)tk->sub->readSource();
1968
1969 p_block = block_Alloc( i_size + 1 );
1970 p_block->p_buffer[0] = amrSource->lastFrameHeader();
1971 memcpy( p_block->p_buffer + 1, tk->p_buffer, i_size );
1972 }
1973 else if( tk->fmt.i_codec == VLC_CODEC_H261 )
1974 {
1975 H261VideoRTPSource *h261Source = (H261VideoRTPSource*)tk->sub->rtpSource();
1976 uint32_t header = h261Source->lastSpecialHeader();
1977 p_block = block_Alloc( i_size + 4 );
1978 memcpy( p_block->p_buffer, &header, 4 );
1979 memcpy( p_block->p_buffer + 4, tk->p_buffer, i_size );
1980
1981 if( tk->sub->rtpSource()->curPacketMarkerBit() )
1982 p_block->i_flags |= BLOCK_FLAG_END_OF_FRAME;
1983 }
1984 else if( tk->fmt.i_codec == VLC_CODEC_H264 || tk->fmt.i_codec == VLC_CODEC_HEVC )
1985 {
1986 if( tk->fmt.i_codec == VLC_CODEC_H264 && (tk->p_buffer[0] & 0x1f) >= 24 )
1987 msg_Warn( p_demux, "unsupported NAL type for H264" );
1988 else if( tk->fmt.i_codec == VLC_CODEC_HEVC && ((tk->p_buffer[0] & 0x7e)>>1) >= 48 )
1989 msg_Warn( p_demux, "unsupported NAL type for H265" );
1990
1991 /* Normal NAL type */
1992 p_block = block_Alloc( i_size + 4 );
1993 p_block->p_buffer[0] = 0x00;
1994 p_block->p_buffer[1] = 0x00;
1995 p_block->p_buffer[2] = 0x00;
1996 p_block->p_buffer[3] = 0x01;
1997 memcpy( &p_block->p_buffer[4], tk->p_buffer, i_size );
1998 }
1999 else if( tk->b_asf )
2000 {
2001 p_block = StreamParseAsf( p_demux, tk,
2002 tk->sub->rtpSource()->curPacketMarkerBit(),
2003 tk->p_buffer, i_size );
2004 }
2005 else
2006 {
2007 p_block = block_Alloc( i_size );
2008 memcpy( p_block->p_buffer, tk->p_buffer, i_size );
2009 }
2010
2011 if( p_sys->i_pcr < i_pts )
2012 {
2013 p_sys->i_pcr = i_pts;
2014 }
2015
2016 /* Update our global npt value */
2017 if( tk->f_npt > 0 &&
2018 ( tk->f_npt < p_sys->f_npt_length || p_sys->f_npt_length <= 0 ) )
2019 p_sys->f_npt = tk->f_npt;
2020
2021 if( p_block )
2022 {
2023 if( !tk->b_muxed && !tk->b_asf )
2024 {
2025 if( i_pts != tk->i_pts )
2026 p_block->i_pts = VLC_TS_0 + i_pts;
2027 /*FIXME: for h264 you should check that packetization-mode=1 in sdp-file */
2028 p_block->i_dts = ( tk->fmt.i_codec == VLC_CODEC_MPGV ) ? VLC_TS_INVALID : (VLC_TS_0 + i_pts);
2029 }
2030
2031 if( tk->b_muxed )
2032 stream_DemuxSend( tk->p_out_muxed, p_block );
2033 else if( tk->b_asf )
2034 stream_DemuxSend( p_sys->p_out_asf, p_block );
2035 else
2036 es_out_Send( p_demux->out, tk->p_es, p_block );
2037 }
2038
2039 /* warn that's ok */
2040 p_sys->event_data = 0xff;
2041
2042 /* we have read data */
2043 tk->waiting = 0;
2044 p_demux->p_sys->b_no_data = false;
2045 p_demux->p_sys->i_no_data_ti = 0;
2046
2047 if( i_pts > 0 && !tk->b_muxed )
2048 {
2049 tk->i_pts = i_pts;
2050 }
2051}
