38struct stream_sys_t {
39 stream_t *payload;
40};
41
42static int Read(stream_t *s, void *data, unsigned size)
43{
44 return stream_Read(s->p_sys->payload, data, size);
45}
46
47static int Peek( stream_t *s, const uint8_t **data, unsigned size)
48{
49 return stream_Peek(s->p_sys->payload, data, size);
50}
51
52static int Control(stream_t *s, int query, va_list args)
53{
54 switch (query) {
55 case STREAM_GET_CONTENT_TYPE: {
56 char **mime = va_arg(args, char **);
57 *mime = strdup("audio/x-mpegurl");
58 return VLC_SUCCESS;
59 }
60 default:
61 return stream_vaControl(s->p_sys->payload, query, args);
62 }
63}

65int RarStreamOpen(vlc_object_t *object)
66{
67 stream_t *s = (stream_t*)object;
68
69 if (RarProbe(s->p_source))
70 return VLC_EGENERIC;
71
72 int count;
73 rar_file_t **files;
74 const int64_t position = stream_Tell(s->p_source);
75 if (RarParse(s->p_source, &count, &files, false) &&
76 RarParse(s->p_source, &count, &files, true ) )
77 {
78 stream_Seek(s->p_source, position);
79 msg_Err(s, "Invalid or unsupported RAR archive");
80 free(files);
81 return VLC_EGENERIC;
82 }
83
84 /* TODO use xspf to have node for directories
85 * Reusing WriteXSPF from the zip access is probably a good idea
86 * (becareful about '\' and '/'.
87 */
88 char *mrl;
89 if (asprintf(&mrl, "%s://%s", s->psz_access, s->psz_path)< 0)
90 mrl = NULL;
91 char *base;
92 char *encoded = mrl ? encode_URI_component(mrl) : NULL;
93 free(mrl);
94
95 if (!encoded || asprintf(&base, "rar://%s", encoded) < 0)
96 base = NULL;
97 free(encoded);
98
99 char *data = strdup("#EXTM3U\n");
100 for (int i = 0; i < count; i++) {
101 rar_file_t *f = files[i];
102 char *next;
103 if (base && data &&
104 asprintf(&next, "%s"
105 "#EXTINF:,,%s\n"
106 "%s|%s\n",
107 data, f->name, base, f->name) >= 0) {
108 free(data);
109 data = next;
110 }
111 RarFileDelete(f);
112 }
113 free(base);
114 free(files);
115 if (!data)
116 return VLC_EGENERIC;
117 stream_t *payload = stream_MemoryNew(s, (uint8_t*)data, strlen(data), false);
118 if (!payload) {
119 free(data);
120 return VLC_EGENERIC;
121 }
122
123 s->pf_read = Read;
124 s->pf_peek = Peek;
125 s->pf_control = Control;
126
127 stream_sys_t *sys = s->p_sys = malloc(sizeof(*sys));
128 if (!sys) {
129 stream_Delete(payload);
130 return VLC_ENOMEM;
131 }
132 sys->payload = payload;
133
134 char *tmp;
135 if (asprintf(&tmp, "%s.m3u", s->psz_path) < 0) {
136 RarStreamClose(object);
137 return VLC_ENOMEM;
138 }
139 free(s->psz_path);
140 s->psz_path = tmp;
141
142 return VLC_SUCCESS;
143}
144
145void RarStreamClose(vlc_object_t *object)
146{
147 stream_t *s = (stream_t*)object;
148 stream_sys_t *sys = s->p_sys;
149
150 stream_Delete(sys->payload);
151 free(sys);
152}
