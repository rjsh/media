84/*****************************************************************************
85 * vlc_DosDevIOCtl: high memory safe wrapper for DosDevIOCtl
86 *****************************************************************************
87 * Unfortunately, DosDevIOCtl() is not high memory safe API, and is not
88 * covered by os2safe.h. So define a wrapper function for it here.
89 *****************************************************************************/
90
91static APIRET vlc_DosDevIOCtl( HFILE hdevice, ULONG category, ULONG function,
92 PVOID pParams, ULONG cbParamLenMax,
93 PULONG pcbParamLen, PVOID pData,
94 ULONG cbDataLenMax, PULONG pcbDataLen )
95{
96 PVOID pParamsLow = NULL;
97 PVOID pDataLow = NULL;
98 ULONG cbParamLenLow;
99 ULONG cbDataLenLow;
100
101 APIRET rc;
102
103 rc = DosAllocMem( &pParamsLow, cbParamLenMax, fALLOC );
104 if( rc )
105 goto exit_free;
106
107 rc = DosAllocMem( &pDataLow, cbDataLenMax, fALLOC );
108 if( rc )
109 goto exit_free;
110
111 memcpy( pParamsLow, pParams, cbParamLenMax );
112 memcpy( pDataLow, pData, cbDataLenMax );
113
114 cbParamLenLow = *pcbParamLen;
115 cbDataLenLow = *pcbDataLen;
116
117 rc = DosDevIOCtl( hdevice, category, function, pParamsLow,
118 cbParamLenMax, &cbParamLenLow, pDataLow, cbDataLenMax,
119 &cbDataLenLow );
120
121 if( !rc )
122 {
123 memcpy( pParams, pParamsLow, cbParamLenMax );
124 memcpy( pData, pDataLow, cbDataLenMax );
125
126 *pcbParamLen = cbParamLenLow;
127 *pcbDataLen = cbDataLenLow;
128 }
129
130exit_free:
131 DosFreeMem( pParamsLow);
132 DosFreeMem( pDataLow);
133
134 return rc;
135}

146/*****************************************************************************
147 * ioctl_Open: Opens a VCD device or file and returns an opaque handle
148 *****************************************************************************/
149vcddev_t *ioctl_Open( vlc_object_t *p_this, const char *psz_dev )
150{
151 int i_ret;
152 int b_is_file;
153 vcddev_t *p_vcddev;
154#if !defined( _WIN32 ) && !defined( __OS2__ )
155 struct stat fileinfo;
156#endif
157
158 if( !psz_dev ) return NULL;
159
160 /*
161 * Initialize structure with default values
162 */
163 p_vcddev = malloc( sizeof(*p_vcddev) );
164 if( p_vcddev == NULL )
165 return NULL;
166 p_vcddev->i_vcdimage_handle = -1;
167 p_vcddev->psz_dev = NULL;
168 b_is_file = 1;
169
170 /*
171 * Check if we are dealing with a device or a file (vcd image)
172 */
173#if defined( _WIN32 ) || defined( __OS2__ )
174 if( (strlen( psz_dev ) == 2 && psz_dev[1] == ':') )
175 {
176 b_is_file = 0;
177 }
178
179#else
180 if( vlc_stat( psz_dev, &fileinfo ) < 0 )
181 {
182 free( p_vcddev );
183 return NULL;
184 }
185
186 /* Check if this is a block/char device */
187 if( S_ISBLK( fileinfo.st_mode ) || S_ISCHR( fileinfo.st_mode ) )
188 b_is_file = 0;
189#endif
190
191 if( b_is_file )
192 {
193 i_ret = OpenVCDImage( p_this, psz_dev, p_vcddev );
194 }
195 else
196 {
197 /*
198 * open the vcd device
199 */
200
201#ifdef _WIN32
202 i_ret = win32_vcd_open( p_this, psz_dev, p_vcddev );
203#elif defined( __OS2__ )
204 i_ret = os2_vcd_open( p_this, psz_dev, p_vcddev );
205#else
206 p_vcddev->i_device_handle = -1;
207 p_vcddev->i_device_handle = vlc_open( psz_dev, O_RDONLY | O_NONBLOCK );
208 i_ret = (p_vcddev->i_device_handle == -1) ? -1 : 0;
209#endif
210 }
211
212 if( i_ret == 0 )
213 {
214 p_vcddev->psz_dev = (char *)strdup( psz_dev );
215 }
216 else
217 {
218 free( p_vcddev );
219 p_vcddev = NULL;
220 }
221
222 return p_vcddev;
223}

225/*****************************************************************************
226 * ioctl_Close: Closes an already opened VCD device or file.
227 *****************************************************************************/
228void ioctl_Close( vlc_object_t * p_this, vcddev_t *p_vcddev )
229{
230 free( p_vcddev->psz_dev );
231
232 if( p_vcddev->i_vcdimage_handle != -1 )
233 {
234 /*
235 * vcd image mode
236 */
237
238 CloseVCDImage( p_this, p_vcddev );
239 return;
240 }
241
242 /*
243 * vcd device mode
244 */
245
246#ifdef _WIN32
247 if( p_vcddev->h_device_handle )
248 CloseHandle( p_vcddev->h_device_handle );
249#elif defined( __OS2__ )
250 if( p_vcddev->hcd )
251 DosClose( p_vcddev->hcd );
252#else
253 if( p_vcddev->i_device_handle != -1 )
254 close( p_vcddev->i_device_handle );
255#endif
256 free( p_vcddev );
257}

259/*****************************************************************************
260 * ioctl_GetTracksMap: Read the Table of Content, fill in the pp_sectors map
261 * if pp_sectors is not null and return the number of
262 * tracks available.
263 *****************************************************************************/
264int ioctl_GetTracksMap( vlc_object_t *p_this, const vcddev_t *p_vcddev,
265 int **pp_sectors )
266{
267 int i_tracks = 0;
268
269 if( p_vcddev->i_vcdimage_handle != -1 )
270 {
271 /*
272 * vcd image mode
273 */
274
275 i_tracks = p_vcddev->i_tracks;
276
277 if( pp_sectors )
278 {
279 *pp_sectors = calloc( i_tracks + 1, sizeof(**pp_sectors) );
280 if( *pp_sectors == NULL )
281 return 0;
282 memcpy( *pp_sectors, p_vcddev->p_sectors,
283 (i_tracks + 1) * sizeof(**pp_sectors) );
284 }
285
286 return i_tracks;
287 }
288 else
289 {
290
291 /*
292 * vcd device mode
293 */
294
295#if defined( __APPLE__ )
296
297 CDTOC *pTOC;
298 int i_descriptors;
299
300 if( ( pTOC = darwin_getTOC( p_this, p_vcddev ) ) == NULL )
301 {
302 msg_Err( p_this, "failed to get the TOC" );
303 return 0;
304 }
305
306 i_descriptors = CDTOCGetDescriptorCount( pTOC );
307 i_tracks = darwin_getNumberOfTracks( pTOC, i_descriptors );
308
309 if( pp_sectors )
310 {
311 int i, i_leadout = -1;
312 CDTOCDescriptor *pTrackDescriptors;
313 u_char track;
314
315 *pp_sectors = calloc( i_tracks + 1, sizeof(**pp_sectors) );
316 if( *pp_sectors == NULL )
317 {
318 darwin_freeTOC( pTOC );
319 return 0;
320 }
321
322 pTrackDescriptors = pTOC->descriptors;
323
324 for( i_tracks = 0, i = 0; i < i_descriptors; i++ )
325 {
326 track = pTrackDescriptors[i].point;
327
328 if( track == 0xA2 )
329 i_leadout = i;
330
331 if( track > CD_MAX_TRACK_NO || track < CD_MIN_TRACK_NO )
332 continue;
333
334 (*pp_sectors)[i_tracks++] =
335 CDConvertMSFToLBA( pTrackDescriptors[i].p );
336 }
337
338 if( i_leadout == -1 )
339 {
340 msg_Err( p_this, "leadout not found" );
341 free( *pp_sectors );
342 darwin_freeTOC( pTOC );
343 return 0;
344 }
345
346 /* set leadout sector */
347 (*pp_sectors)[i_tracks] =
348 CDConvertMSFToLBA( pTrackDescriptors[i_leadout].p );
349 }
350
351 darwin_freeTOC( pTOC );
352
353#elif defined( _WIN32 )
354 DWORD dwBytesReturned;
355 CDROM_TOC cdrom_toc;
356
357 if( DeviceIoControl( p_vcddev->h_device_handle, IOCTL_CDROM_READ_TOC,
358 NULL, 0, &cdrom_toc, sizeof(CDROM_TOC),
359 &dwBytesReturned, NULL ) == 0 )
360 {
361 msg_Err( p_this, "could not read TOCHDR" );
362 return 0;
363 }
364
365 i_tracks = cdrom_toc.LastTrack - cdrom_toc.FirstTrack + 1;
366
367 if( pp_sectors )
368 {
369 *pp_sectors = calloc( i_tracks + 1, sizeof(**pp_sectors) );
370 if( *pp_sectors == NULL )
371 return 0;
372
373 for( int i = 0 ; i <= i_tracks ; i++ )
374 {
375 (*pp_sectors)[ i ] = MSF_TO_LBA2(
376 cdrom_toc.TrackData[i].Address[1],
377 cdrom_toc.TrackData[i].Address[2],
378 cdrom_toc.TrackData[i].Address[3] );
379 msg_Dbg( p_this, "p_sectors: %i, %i", i, (*pp_sectors)[i]);
380 }
381 }
382
383#elif defined( __OS2__ )
384 cdrom_get_tochdr_t get_tochdr = {{'C', 'D', '0', '1'}};
385 cdrom_tochdr_t tochdr;
386
387 ULONG param_len;
388 ULONG data_len;
389 ULONG rc;
390
391 rc = DosDevIOCtl( p_vcddev->hcd, IOCTL_CDROMAUDIO,
392 CDROMAUDIO_GETAUDIODISK,
393 &get_tochdr, sizeof( get_tochdr ), &param_len,
394 &tochdr, sizeof( tochdr ), &data_len );
395 if( rc )
396 {
397 msg_Err( p_this, "could not read TOCHDR" );
398 return 0;
399 }
400
401 i_tracks = tochdr.last_track - tochdr.first_track + 1;
402
403 if( pp_sectors )
404 {
405 cdrom_get_track_t get_track = {{'C', 'D', '0', '1'}, };
406 cdrom_track_t track;
407 int i;
408
409 *pp_sectors = calloc( i_tracks + 1, sizeof(**pp_sectors) );
410 if( *pp_sectors == NULL )
411 return 0;
412
413 for( i = 0 ; i < i_tracks ; i++ )
414 {
415 get_track.track = tochdr.first_track + i;
416 rc = DosDevIOCtl( p_vcddev->hcd, IOCTL_CDROMAUDIO,
417 CDROMAUDIO_GETAUDIOTRACK,
418 &get_track, sizeof(get_track), &param_len,
419 &track, sizeof(track), &data_len );
420 if (rc)
421 {
422 msg_Err( p_this, "could not read %d track",
423 get_track.track );
424 return 0;
425 }
426
427 (*pp_sectors)[ i ] = MSF_TO_LBA2(
428 track.start.minute,
429 track.start.second,
430 track.start.frame );
431 msg_Dbg( p_this, "p_sectors: %i, %i", i, (*pp_sectors)[i]);
432 }
433
434 /* for lead-out track */
435 (*pp_sectors)[ i ] = MSF_TO_LBA2(
436 tochdr.lead_out.minute,
437 tochdr.lead_out.second,
438 tochdr.lead_out.frame );
439 msg_Dbg( p_this, "p_sectors: %i, %i", i, (*pp_sectors)[i]);
440 }
441
442#elif defined( HAVE_IOC_TOC_HEADER_IN_SYS_CDIO_H ) \
443 || defined( HAVE_SCSIREQ_IN_SYS_SCSIIO_H )
444 struct ioc_toc_header tochdr;
445 struct ioc_read_toc_entry toc_entries;
446
447 if( ioctl( p_vcddev->i_device_handle, CDIOREADTOCHEADER, &tochdr )
448 == -1 )
449 {
450 msg_Err( p_this, "could not read TOCHDR" );
451 return 0;
452 }
453
454 i_tracks = tochdr.ending_track - tochdr.starting_track + 1;
455
456 if( pp_sectors )
457 {
458 int i;
459
460 *pp_sectors = calloc( i_tracks + 1, sizeof(**pp_sectors) );
461 if( *pp_sectors == NULL )
462 return 0;
463
464 toc_entries.address_format = CD_LBA_FORMAT;
465 toc_entries.starting_track = 0;
466 toc_entries.data_len = ( i_tracks + 1 ) *
467 sizeof( struct cd_toc_entry );
468 toc_entries.data = (struct cd_toc_entry *)
469 malloc( toc_entries.data_len );
470 if( toc_entries.data == NULL )
471 {
472 free( *pp_sectors );
473 return 0;
474 }
475
476 /* Read the TOC */
477 if( ioctl( p_vcddev->i_device_handle, CDIOREADTOCENTRYS,
478 &toc_entries ) == -1 )
479 {
480 msg_Err( p_this, "could not read the TOC" );
481 free( *pp_sectors );
482 free( toc_entries.data );
483 return 0;
484 }
485
486 /* Fill the p_sectors structure with the track/sector matches */
487 for( i = 0 ; i <= i_tracks ; i++ )
488 {
489#if defined( HAVE_SCSIREQ_IN_SYS_SCSIIO_H )
490 /* FIXME: is this ok? */
491 (*pp_sectors)[ i ] = toc_entries.data[i].addr.lba;
492#else
493 (*pp_sectors)[ i ] = ntohl( toc_entries.data[i].addr.lba );
494#endif
495 }
496 }
497#else
498 struct cdrom_tochdr tochdr;
499 struct cdrom_tocentry tocent;
500
501 /* First we read the TOC header */
502 if( ioctl( p_vcddev->i_device_handle, CDROMREADTOCHDR, &tochdr )
503 == -1 )
504 {
505 msg_Err( p_this, "could not read TOCHDR" );
506 return 0;
507 }
508
509 i_tracks = tochdr.cdth_trk1 - tochdr.cdth_trk0 + 1;
510
511 if( pp_sectors )
512 {
513 int i;
514
515 *pp_sectors = calloc( i_tracks + 1, sizeof(**pp_sectors) );
516 if( *pp_sectors == NULL )
517 return 0;
518
519 /* Fill the p_sectors structure with the track/sector matches */
520 for( i = 0 ; i <= i_tracks ; i++ )
521 {
522 tocent.cdte_format = CDROM_LBA;
523 tocent.cdte_track =
524 ( i == i_tracks ) ? CDROM_LEADOUT : tochdr.cdth_trk0 + i;
525
526 if( ioctl( p_vcddev->i_device_handle, CDROMREADTOCENTRY,
527 &tocent ) == -1 )
528 {
529 msg_Err( p_this, "could not read TOCENTRY" );
530 free( *pp_sectors );
531 return 0;
532 }
533
534 (*pp_sectors)[ i ] = tocent.cdte_addr.lba;
535 }
536 }
537#endif
538
539 return i_tracks;
540 }
541}

543/****************************************************************************
544 * ioctl_ReadSector: Read VCD or CDDA sectors
545 ****************************************************************************/
546int ioctl_ReadSectors( vlc_object_t *p_this, const vcddev_t *p_vcddev,
547 int i_sector, uint8_t *p_buffer, int i_nb, int i_type )
548{
549 uint8_t *p_block;
550 int i;
551
552 if( i_type == VCD_TYPE )
553 p_block = malloc( VCD_SECTOR_SIZE * i_nb );
554 else
555 p_block = p_buffer;
556
557 if( p_vcddev->i_vcdimage_handle != -1 )
558 {
559 /*
560 * vcd image mode
561 */
562 if( lseek( p_vcddev->i_vcdimage_handle, i_sector * VCD_SECTOR_SIZE,
563 SEEK_SET ) == -1 )
564 {
565 msg_Err( p_this, "Could not lseek to sector %d", i_sector );
566 goto error;
567 }
568
569 if( read( p_vcddev->i_vcdimage_handle, p_block, VCD_SECTOR_SIZE * i_nb)
570 == -1 )
571 {
572 msg_Err( p_this, "Could not read sector %d", i_sector );
573 goto error;
574 }
575
576 }
577 else
578 {
579
580 /*
581 * vcd device mode
582 */
583
584#if defined( __APPLE__ )
585 dk_cd_read_t cd_read;
586
587 memset( &cd_read, 0, sizeof(cd_read) );
588
589 cd_read.offset = i_sector * VCD_SECTOR_SIZE;
590 cd_read.sectorArea = kCDSectorAreaSync | kCDSectorAreaHeader |
591 kCDSectorAreaSubHeader | kCDSectorAreaUser |
592 kCDSectorAreaAuxiliary;
593 cd_read.sectorType = kCDSectorTypeUnknown;
594
595 cd_read.buffer = p_block;
596 cd_read.bufferLength = VCD_SECTOR_SIZE * i_nb;
597
598 if( ioctl( p_vcddev->i_device_handle, DKIOCCDREAD, &cd_read ) == -1 )
599 {
600 msg_Err( p_this, "could not read block %d", i_sector );
601 goto error;
602 }
603
604#elif defined( _WIN32 )
605 DWORD dwBytesReturned;
606 RAW_READ_INFO cdrom_raw;
607
608 /* Initialize CDROM_RAW_READ structure */
609 cdrom_raw.DiskOffset.QuadPart = CD_SECTOR_SIZE * i_sector;
610 cdrom_raw.SectorCount = i_nb;
611 cdrom_raw.TrackMode = i_type == VCD_TYPE ? XAForm2 : CDDA;
612
613 if( DeviceIoControl( p_vcddev->h_device_handle, IOCTL_CDROM_RAW_READ,
614 &cdrom_raw, sizeof(RAW_READ_INFO), p_block,
615 VCD_SECTOR_SIZE * i_nb, &dwBytesReturned,
616 NULL ) == 0 )
617 {
618 if( i_type == VCD_TYPE )
619 {
620 /* Retry in YellowMode2 */
621 cdrom_raw.TrackMode = YellowMode2;
622 if( DeviceIoControl( p_vcddev->h_device_handle,
623 IOCTL_CDROM_RAW_READ, &cdrom_raw,
624 sizeof(RAW_READ_INFO), p_block,
625 VCD_SECTOR_SIZE * i_nb, &dwBytesReturned,
626 NULL ) == 0 )
627 goto error;
628 }
629 else return -1;
630 }
631
632#elif defined( __OS2__ )
633 cdrom_readlong_t readlong = {{'C', 'D', '0', '1'}, };
634
635 ULONG param_len;
636 ULONG data_len;
637 ULONG rc;
638
639 readlong.addr_mode = 0; /* LBA mode */
640 readlong.sectors = i_nb;
641 readlong.start = i_sector;
642
643 rc = DosDevIOCtl( p_vcddev->hcd, IOCTL_CDROMDISK, CDROMDISK_READLONG,
644 &readlong, sizeof( readlong ), &param_len,
645 p_block, VCD_SECTOR_SIZE * i_nb, &data_len );
646 if( rc )
647 {
648 msg_Err( p_this, "could not read block %d", i_sector );
649 goto error;
650 }
651
652#elif defined( HAVE_SCSIREQ_IN_SYS_SCSIIO_H )
653 struct scsireq sc;
654 int i_ret;
655
656 memset( &sc, 0, sizeof(sc) );
657 sc.cmd[0] = 0xBE;
658 sc.cmd[1] = i_type == VCD_TYPE ? SECTOR_TYPE_MODE2_FORM2:
659 SECTOR_TYPE_CDDA;
660 sc.cmd[2] = (i_sector >> 24) & 0xff;
661 sc.cmd[3] = (i_sector >> 16) & 0xff;
662 sc.cmd[4] = (i_sector >> 8) & 0xff;
663 sc.cmd[5] = (i_sector >> 0) & 0xff;
664 sc.cmd[6] = (i_nb >> 16) & 0xff;
665 sc.cmd[7] = (i_nb >> 8) & 0xff;
666 sc.cmd[8] = (i_nb ) & 0xff;
667 sc.cmd[9] = i_type == VCD_TYPE ? READ_CD_RAW_MODE2 : READ_CD_USERDATA;
668 sc.cmd[10] = 0; /* sub channel */
669 sc.cmdlen = 12;
670 sc.databuf = (caddr_t)p_block;
671 sc.datalen = VCD_SECTOR_SIZE * i_nb;
672 sc.senselen = sizeof( sc.sense );
673 sc.flags = SCCMD_READ;
674 sc.timeout = 10000;
675
676 i_ret = ioctl( p_vcddev->i_device_handle, SCIOCCOMMAND, &sc );
677 if( i_ret == -1 )
678 {
679 msg_Err( p_this, "SCIOCCOMMAND failed" );
680 goto error;
681 }
682 if( sc.retsts || sc.error )
683 {
684 msg_Err( p_this, "SCSI command failed: status %d error %d",
685 sc.retsts, sc.error );
686 goto error;
687 }
688
689#elif defined( HAVE_IOC_TOC_HEADER_IN_SYS_CDIO_H )
690 int i_size = VCD_SECTOR_SIZE;
691
692 if( ioctl( p_vcddev->i_device_handle, CDRIOCSETBLOCKSIZE, &i_size )
693 == -1 )
694 {
695 msg_Err( p_this, "Could not set block size" );
696 goto error;
697 }
698
699 if( lseek( p_vcddev->i_device_handle,
700 i_sector * VCD_SECTOR_SIZE, SEEK_SET ) == -1 )
701 {
702 msg_Err( p_this, "Could not lseek to sector %d", i_sector );
703 goto error;
704 }
705
706 if( read( p_vcddev->i_device_handle,
707 p_block, VCD_SECTOR_SIZE * i_nb ) == -1 )
708 {
709 msg_Err( p_this, "Could not read sector %d", i_sector );
710 goto error;
711 }
712
713#else
714 for( i = 0; i < i_nb; i++ )
715 {
716 int i_dummy = i_sector + i + 2 * CD_FRAMES;
717
718#define p_msf ((struct cdrom_msf0 *)(p_block + i * VCD_SECTOR_SIZE))
719 p_msf->minute = i_dummy / (CD_FRAMES * CD_SECS);
720 p_msf->second = ( i_dummy % (CD_FRAMES * CD_SECS) ) / CD_FRAMES;
721 p_msf->frame = ( i_dummy % (CD_FRAMES * CD_SECS) ) % CD_FRAMES;
722#undef p_msf
723
724 if( ioctl( p_vcddev->i_device_handle, CDROMREADRAW,
725 p_block + i * VCD_SECTOR_SIZE ) == -1 )
726 {
727 msg_Err( p_this, "could not read block %i from disc",
728 i_sector );
729
730 if( i == 0 )
731 goto error;
732 else
733 break;
734 }
735 }
736#endif
737 }
738
739 /* For VCDs, we don't want to keep the header and footer of the
740 * sectors read */
741 if( i_type == VCD_TYPE )
742 {
743 for( i = 0; i < i_nb; i++ )
744 {
745 memcpy( p_buffer + i * VCD_DATA_SIZE,
746 p_block + i * VCD_SECTOR_SIZE + VCD_DATA_START,
747 VCD_DATA_SIZE );
748 }
749 free( p_block );
750 }
751
752 return( 0 );
753
754error:
755 if( i_type == VCD_TYPE )
756 free( p_block );
757 return( -1 );
758}

764/****************************************************************************
765 * OpenVCDImage: try to open a vcd image from a .cue file
766 ****************************************************************************/
767static int OpenVCDImage( vlc_object_t * p_this, const char *psz_dev,
768 vcddev_t *p_vcddev )
769{
770 int i_ret = -1;
771 char *p_pos;
772 char *psz_vcdfile = NULL;
773 char *psz_cuefile = NULL;
774 FILE *cuefile = NULL;
775 int *p_sectors = NULL;
776 char line[1024];
777 bool b_found = false;
778
779 /* Check if we are dealing with a .cue file */
780 p_pos = strrchr( psz_dev, '.' );
781 if( p_pos && !strcmp( p_pos, ".cue" ) )
782 {
783 /* psz_dev must be the cue file. Let's assume there's a .bin
784 * file with the same filename */
785 psz_vcdfile = malloc( p_pos - psz_dev + 5 /* ".bin" */ );
786 strncpy( psz_vcdfile, psz_dev, p_pos - psz_dev );
787 strcpy( psz_vcdfile + (p_pos - psz_dev), ".bin");
788 psz_cuefile = strdup( psz_dev );
789 }
790 else
791 {
792 /* psz_dev must be the actual vcd file. Let's assume there's a .cue
793 * file with the same filename */
794 if( p_pos )
795 {
796 psz_cuefile = malloc( p_pos - psz_dev + 5 /* ".cue" */ );
797 strncpy( psz_cuefile, psz_dev, p_pos - psz_dev );
798 strcpy( psz_cuefile + (p_pos - psz_dev), ".cue");
799 }
800 else
801 {
802 if( asprintf( &psz_cuefile, "%s.cue", psz_dev ) == -1 )
803 psz_cuefile = NULL;
804 }
805 /* If we need to look up the .cue file, then we don't have to look for the vcd */
806 psz_vcdfile = strdup( psz_dev );
807 }
808
809 /* Open the cue file and try to parse it */
810 msg_Dbg( p_this,"trying .cue file: %s", psz_cuefile );
811 cuefile = vlc_fopen( psz_cuefile, "rt" );
812 if( cuefile == NULL )
813 {
814 msg_Dbg( p_this, "could not find .cue file" );
815 goto error;
816 }
817
818 msg_Dbg( p_this,"guessing vcd image file: %s", psz_vcdfile );
819 p_vcddev->i_vcdimage_handle = vlc_open( psz_vcdfile,
820 O_RDONLY | O_NONBLOCK | O_BINARY );
821
822 while( fgets( line, 1024, cuefile ) && !b_found )
823 {
824 /* We have a cue file, but no valid vcd file yet */
825 char filename[1024];
826 char type[16];
827 int i_temp = sscanf( line, "FILE \"%1023[^\"]\" %15s", filename, type );
828 switch( i_temp )
829 {
830 case 2:
831 msg_Dbg( p_this, "the cue file says the data file is %s", type );
832 if( strcasecmp( type, "BINARY" ) )
833 goto error; /* Error if not binary, otherwise treat as case 1 */
834 case 1:
835 if( p_vcddev->i_vcdimage_handle == -1 )
836 {
837 msg_Dbg( p_this, "we could not find the data file, but we found a new path" );
838 free( psz_vcdfile);
839 if( *filename != '/' && ((p_pos = strrchr( psz_cuefile, '/' ))
840 || (p_pos = strrchr( psz_cuefile, '\\' ) )) )
841 {
842 psz_vcdfile = malloc( strlen(filename) +
843 (p_pos - psz_cuefile + 1) + 1 );
844 strncpy( psz_vcdfile, psz_cuefile, (p_pos - psz_cuefile + 1) );
845 strcpy( psz_vcdfile + (p_pos - psz_cuefile + 1), filename );
846 } else psz_vcdfile = strdup( filename );
847 msg_Dbg( p_this,"using vcd image file: %s", psz_vcdfile );
848 p_vcddev->i_vcdimage_handle = vlc_open( psz_vcdfile,
849 O_RDONLY | O_NONBLOCK | O_BINARY );
850 }
851 b_found = true;
852 default:
853 break;
854 }
855 }
856
857 if( p_vcddev->i_vcdimage_handle == -1)
858 goto error;
859
860 /* Try to parse the i_tracks and p_sectors info so we can just forget
861 * about the cuefile */
862 size_t i_tracks = 0;
863
864 while( fgets( line, 1024, cuefile ) && i_tracks < INT_MAX-1 )
865 {
866 /* look for a TRACK line */
867 char psz_dummy[10];
868 if( !sscanf( line, "%9s", psz_dummy ) || strcmp(psz_dummy, "TRACK") )
869 continue;
870
871 /* look for an INDEX line */
872 while( fgets( line, 1024, cuefile ) )
873 {
874 int i_num, i_min, i_sec, i_frame;
875
876 if( (sscanf( line, "%*9s %2u %2u:%2u:%2u", &i_num,
877 &i_min, &i_sec, &i_frame ) != 4) || (i_num != 1) )
878 continue;
879
880 int *buf = realloc (p_sectors, (i_tracks + 1) * sizeof (*buf));
881 if (buf == NULL)
882 goto error;
883 p_sectors = buf;
884 p_sectors[i_tracks] = MSF_TO_LBA(i_min, i_sec, i_frame);
885 msg_Dbg( p_this, "vcd track %i begins at sector:%i",
886 (int)i_tracks, (int)p_sectors[i_tracks] );
887 i_tracks++;
888 break;
889 }
890 }
891
892 /* fill in the last entry */
893 int *buf = realloc (p_sectors, (i_tracks + 1) * sizeof (*buf));
894 if (buf == NULL)
895 goto error;
896 p_sectors = buf;
897 p_sectors[i_tracks] = lseek(p_vcddev->i_vcdimage_handle, 0, SEEK_END)
898 / VCD_SECTOR_SIZE;
899 msg_Dbg( p_this, "vcd track %i, begins at sector:%i",
900 (int)i_tracks, (int)p_sectors[i_tracks] );
901 p_vcddev->i_tracks = ++i_tracks;
902 p_vcddev->p_sectors = p_sectors;
903 i_ret = 0;
904
905error:
906 if( cuefile ) fclose( cuefile );
907 free( p_sectors );
908 free( psz_cuefile );
909 free( psz_vcdfile );
910
911 return i_ret;
912}

929/****************************************************************************
930 * darwin_getTOC: get the TOC
931 ****************************************************************************/
932static CDTOC *darwin_getTOC( vlc_object_t * p_this, const vcddev_t *p_vcddev )
933{
934 mach_port_t port;
935 char *psz_devname;
936 kern_return_t ret;
937 CDTOC *pTOC = NULL;
938 io_iterator_t iterator;
939 io_registry_entry_t service;
940 CFMutableDictionaryRef properties;
941 CFDataRef data;
942
943 /* get the device name */
944 if( ( psz_devname = strrchr( p_vcddev->psz_dev, '/') ) != NULL )
945 ++psz_devname;
946 else
947 psz_devname = p_vcddev->psz_dev;
948
949 /* unraw the device name */
950 if( *psz_devname == 'r' )
951 ++psz_devname;
952
953 /* get port for IOKit communication */
954 if( ( ret = IOMasterPort( MACH_PORT_NULL, &port ) ) != KERN_SUCCESS )
955 {
956 msg_Err( p_this, "IOMasterPort: 0x%08x", ret );
957 return( NULL );
958 }
959
960 /* get service iterator for the device */
961 if( ( ret = IOServiceGetMatchingServices(
962 port, IOBSDNameMatching( port, 0, psz_devname ),
963 &iterator ) ) != KERN_SUCCESS )
964 {
965 msg_Err( p_this, "IOServiceGetMatchingServices: 0x%08x", ret );
966 return( NULL );
967 }
968
969 /* first service */
970 service = IOIteratorNext( iterator );
971 IOObjectRelease( iterator );
972
973 /* search for kIOCDMediaClass */
974 while( service && !IOObjectConformsTo( service, kIOCDMediaClass ) )
975 {
976 if( ( ret = IORegistryEntryGetParentIterator( service,
977 kIOServicePlane, &iterator ) ) != KERN_SUCCESS )
978 {
979 msg_Err( p_this, "IORegistryEntryGetParentIterator: 0x%08x", ret );
980 IOObjectRelease( service );
981 return( NULL );
982 }
983
984 IOObjectRelease( service );
985 service = IOIteratorNext( iterator );
986 IOObjectRelease( iterator );
987 }
988
989 if( !service )
990 {
991 msg_Err( p_this, "search for kIOCDMediaClass came up empty" );
992 return( NULL );
993 }
994
995 /* create a CF dictionary containing the TOC */
996 if( ( ret = IORegistryEntryCreateCFProperties( service, &properties,
997 kCFAllocatorDefault, kNilOptions ) ) != KERN_SUCCESS )
998 {
999 msg_Err( p_this, "IORegistryEntryCreateCFProperties: 0x%08x", ret );
1000 IOObjectRelease( service );
1001 return( NULL );
1002 }
1003
1004 /* get the TOC from the dictionary */
1005 if( ( data = (CFDataRef) CFDictionaryGetValue( properties,
1006 CFSTR(kIOCDMediaTOCKey) ) ) != NULL )
1007 {
1008 CFRange range;
1009 CFIndex buf_len;
1010
1011 buf_len = CFDataGetLength( data ) + 1;
1012 range = CFRangeMake( 0, buf_len );
1013
1014 if( ( pTOC = malloc( buf_len ) ) != NULL )
1015 {
1016 CFDataGetBytes( data, range, (u_char *)pTOC );
1017 }
1018 }
1019 else
1020 {
1021 msg_Err( p_this, "CFDictionaryGetValue failed" );
1022 }
1023
1024 CFRelease( properties );
1025 IOObjectRelease( service );
1026
1027 return( pTOC );
1028}

1145 {

1147 char psz_text[12+1];
1148
1149 const int i_pack_type = p_block[0];
1150 if( i_pack_type < 0x80 || i_pack_type > 0x8f )
1151 continue;
1152
1153 const int i_track_number = (p_block[1] >> 0)&0x7f;
1154 const int i_extension_flag = ( p_block[1] >> 7)& 0x01;
1155 if( i_extension_flag )
1156 continue;
1157
1158 //const int i_sequence_number = p_block[2];
1159 //const int i_charater_position = (p_block[3] >> 0) &0x0f;
1160 //const int i_block_number = (p_block[3] >> 4) &0x07;

1163 //const int i_unicode = ( p_block[3] >> 7)&0x01;
1164 //const int i_crc = (p_block[4+12] << 8) | (p_block[4+13] << 0);
1165
1166 /* */
1167 memcpy( psz_text, &p_block[4], 12 );
1168 psz_text[12] = '\0';
1169
1170 /* */
1171 int i_track = i_track_number;
1172 char *psz_track = &psz_text[0];
1173 while( i_track <= 127 && psz_track < &psz_text[12] )
1174 {
1175 //fprintf( stderr, "t=%d psz_track=%p end=%p", i_track, psz_track, &psz_text[12] );
1176 if( *psz_track )
1177 {
1178 astrcat( &pppsz_info[i_track][i_pack_type-0x80], psz_track );
1179 i_track_last = __MAX( i_track_last, i_track );
1180 }
1181
1182 i_track++;
1183 psz_track += 1 + strlen(psz_track);
1184 }
1185 }
1186
1187 if( i_track_last < 0 )
1188 return -1;
1189
1190 vlc_meta_t **pp_tracks = calloc( i_track_last+1, sizeof(*pp_tracks) );
1191 if( !pp_tracks )
1192 goto exit;
1193
1194 for( int j = 0; j < 0x10; j++ )
1195 {
1196 for( int i = 0; i <= i_track_last; i++ )
1197 {
1198 /* */
1199 if( pppsz_info[i][j] )
1200 EnsureUTF8( pppsz_info[i][j] );
1201
1202 /* */
1203 const char *psz_default = pppsz_info[0][j];
1204 const char *psz_value = pppsz_info[i][j];
1205
1206 if( !psz_value && !psz_default )
1207 continue;
1208 vlc_meta_t *p_track = pp_tracks[i];
1209 if( !p_track )
1210 {
1211 p_track = pp_tracks[i] = vlc_meta_New();
1212 if( !p_track )
1213 continue;
1214 }
1215 switch( j )
1216 {
1217 case 0x00: /* Album/Title */
1218 if( i == 0 )
1219 {
1220 vlc_meta_SetAlbum( p_track, psz_value );
1221 }
1222 else
1223 {
1224 if( psz_value )
1225 vlc_meta_SetTitle( p_track, psz_value );
1226 if( psz_default )
1227 vlc_meta_SetAlbum( p_track, psz_default );
1228 }
1229 break;
1230 case 0x01: /* Performer */
1231 vlc_meta_SetArtist( p_track,
1232 psz_value ? psz_value : psz_default );
1233 break;
1234 case 0x05: /* Messages */
1235 vlc_meta_SetDescription( p_track,
1236 psz_value ? psz_value : psz_default );
1237 break;
1238 case 0x07: /* Genre */
1239 vlc_meta_SetGenre( p_track,
1240 psz_value ? psz_value : psz_default );
1241 break;
1242 /* FIXME unsupported:
1243 * 0x02: songwriter
1244 * 0x03: composer
1245 * 0x04: arrenger
1246 * 0x06: disc id */
1247 }
1248 }
1249 }
1250 /* */
1251exit:
1252 for( int j = 0; j < 0x10; j++ )
1253 for( int i = 0; i <= i_track_last; i++ )
1254 free( pppsz_info[i][j] );
1255
1256 *ppp_tracks = pp_tracks;
1257 *pi_tracks = i_track_last+1;
1258 return pp_tracks ? 0 : -1;
1259}

1313static int CdTextRead( vlc_object_t *p_object, const vcddev_t *p_vcddev,

1315{

1321 struct cdrom_generic_command gc;
1322 uint8_t header[4];

1324 /* Read CD-TEXT size */
1325 memset( header, 0, sizeof(header) );
1326 memset( &gc, 0, sizeof(gc) );
1327 gc.cmd[0] = 0x43; /* Read TOC */
1328 gc.cmd[1] = 0x02; /* MSF */
1329 gc.cmd[2] = 5; /* CD-Text */
1330 gc.cmd[7] = ( sizeof(header) >> 8 ) & 0xff;
1331 gc.cmd[8] = ( sizeof(header) >> 0 ) & 0xff;
1332
1333 gc.buflen = sizeof(header);
1334 gc.buffer = header;
1335 gc.data_direction = CGC_DATA_READ;
1336 gc.timeout = 1000;
1337
1338 if( ioctl( p_vcddev->i_device_handle, CDROM_SEND_PACKET, &gc ) == -1 )
1339 return VLC_EGENERIC;
1340
1341 /* If the size is less than 4 it is an error, if it 4 then
1342 * it means no text data */
1343 const int i_text = 2 + (header[0] << 8) + header[1];
1344 if( i_text <= 4 )
1345 return VLC_EGENERIC;
1346
1347 /* Read complete CD-TEXT */
1348 uint8_t *p_text = calloc( 1, i_text );
1349 if( !p_text )
1350 return VLC_EGENERIC;
1351
1352 memset( &gc, 0, sizeof(gc) );
1353 gc.cmd[0] = 0x43; /* Read TOC */
1354 gc.cmd[1] = 0x02; /* MSF */
1355 gc.cmd[2] = 5; /* CD-Text */
1356 gc.cmd[7] = ( i_text >> 8 ) & 0xff;
1357 gc.cmd[8] = ( i_text >> 0 ) & 0xff;
