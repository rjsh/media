25/*****************************************************************************
26 * The vcddev structure
27 *****************************************************************************/
28struct vcddev_s
29{
30 char *psz_dev; /* vcd device name */
31
32 /* Section used in vcd image mode */
33 int i_vcdimage_handle; /* vcd image file descriptor */
34 int i_tracks; /* number of tracks of the vcd */
35 int *p_sectors; /* tracks layout on the vcd */
36
37 /* Section used in vcd device mode */
38
39#ifdef _WIN32
40 HANDLE h_device_handle; /* vcd device descriptor */
41#elif defined( __OS2__ )
42 HFILE hcd; /* vcd device descriptor */
43#else
44 int i_device_handle; /* vcd device descriptor */
45#endif
46
47};

53/* LBA = msf.frame + 75 * ( msf.second + 60 * msf.minute ) */
54#define MSF_TO_LBA(min, sec, frame) ((int)frame + 75 * (sec + 60 * min))
55/* LBA = msf.frame + 75 * ( msf.second - 2 + 60 * msf.minute ) */
56#define MSF_TO_LBA2(min, sec, frame) ((int)frame + 75 * (sec -2 + 60 * min))

75/* Win32 DeviceIoControl specifics */
76#ifndef MAXIMUM_NUMBER_TRACKS
77# define MAXIMUM_NUMBER_TRACKS 100
78#endif
79typedef struct _TRACK_DATA {
80 UCHAR Reserved;
81 UCHAR Control : 4;
82 UCHAR Adr : 4;
83 UCHAR TrackNumber;
84 UCHAR Reserved1;
85 UCHAR Address[4];
86} TRACK_DATA, *PTRACK_DATA;
87typedef struct _CDROM_TOC {
88 UCHAR Length[2];
89 UCHAR FirstTrack;
90 UCHAR LastTrack;
91 TRACK_DATA TrackData[MAXIMUM_NUMBER_TRACKS];
92} CDROM_TOC, *PCDROM_TOC;
93typedef enum _TRACK_MODE_TYPE {
94 YellowMode2,
95 XAForm2,
96 CDDA
97} TRACK_MODE_TYPE, *PTRACK_MODE_TYPE;
98typedef struct __RAW_READ_INFO {
99 LARGE_INTEGER DiskOffset;
100 ULONG SectorCount;
101 TRACK_MODE_TYPE TrackMode;
102} RAW_READ_INFO, *PRAW_READ_INFO;
103typedef struct _CDROM_READ_TOC_EX {
104 UCHAR Format : 4;
105 UCHAR Reserved1 : 3;
106 UCHAR Msf : 1;
107 UCHAR SessionTrack;
108 UCHAR Reserved2;
109 UCHAR Reserved3;
110} CDROM_READ_TOC_EX, *PCDROM_READ_TOC_EX;
111
112#ifndef IOCTL_CDROM_BASE
113# define IOCTL_CDROM_BASE FILE_DEVICE_CD_ROM
114#endif
115#ifndef IOCTL_CDROM_READ_TOC
116# define IOCTL_CDROM_READ_TOC CTL_CODE(IOCTL_CDROM_BASE, 0x0000, \
117 METHOD_BUFFERED, FILE_READ_ACCESS)
118#endif
119#ifndef IOCTL_CDROM_RAW_READ
120#define IOCTL_CDROM_RAW_READ CTL_CODE(IOCTL_CDROM_BASE, 0x000F, \
121 METHOD_OUT_DIRECT, FILE_READ_ACCESS)
122#endif
123#define IOCTL_CDROM_READ_TOC_EX CTL_CODE(IOCTL_CDROM_BASE, 0x0015, \
124 METHOD_BUFFERED, FILE_READ_ACCESS)
