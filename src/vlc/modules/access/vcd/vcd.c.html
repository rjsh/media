62/* how many blocks VCDRead will read in each loop */
63#define VCD_BLOCKS_ONCE 20
64#define VCD_DATA_ONCE (VCD_BLOCKS_ONCE * VCD_DATA_SIZE)
65
66struct access_sys_t
67{
68 vcddev_t *vcddev; /* vcd device descriptor */
69
70 /* Title infos */
71 int i_titles;
72 input_title_t *title[99]; /* No more that 99 track in a vcd ? */
73 int i_current_title;
74 int i_current_seekpoint;
75
76 int i_sector; /* Current Sector */
77 int *p_sectors; /* Track sectors */
78};
79
80static block_t *Block( access_t * );
81static int Seek( access_t *, uint64_t );
82static int Control( access_t *, int, va_list );
83static int EntryPoints( access_t * );

85/*****************************************************************************
86 * VCDOpen: open vcd
87 *****************************************************************************/
88static int Open( vlc_object_t *p_this )
89{
90 access_t *p_access = (access_t *)p_this;
91 access_sys_t *p_sys;
92 if( p_access->psz_filepath == NULL )
93 return VLC_EGENERIC;
94
95 char *psz_dup = ToLocaleDup( p_access->psz_filepath );
96 char *psz;
97 int i_title = 0;
98 int i_chapter = 0;
99 vcddev_t *vcddev;
100
101 /* Command line: vcd://[dev_path][#title[,chapter]] */
102 if( ( psz = strchr( psz_dup, '#' ) ) )
103 {
104 *psz++ = '\0';
105
106 i_title = strtol( psz, &psz, 0 );
107 if( *psz )
108 i_chapter = strtol( psz+1, &psz, 0 );
109 }
110
111 if( *psz_dup == '\0' )
112 {
113 free( psz_dup );
114
115 /* Only when selected */
116 if( strcmp( p_access->psz_access, "vcd" ) &&
117 strcmp( p_access->psz_access, "svcd" ) )
118 return VLC_EGENERIC;
119
120 psz_dup = var_CreateGetString( p_access, "vcd" );
121 if( *psz_dup == '\0' )
122 {
123 free( psz_dup );
124 return VLC_EGENERIC;
125 }
126 }
127
128#if defined( _WIN32 ) || defined( __OS2__ )
129 if( psz_dup[0] && psz_dup[1] == ':' &&
130 psz_dup[2] == '\\' && psz_dup[3] == '\0' ) psz_dup[2] = '\0';
131#endif
132
133 /* Open VCD */
134 vcddev = ioctl_Open( p_this, psz_dup );
135 free( psz_dup );
136 if( !vcddev )
137 return VLC_EGENERIC;
138
139 /* Set up p_access */
140 p_access->p_sys = p_sys = calloc( 1, sizeof( access_sys_t ) );
141 if( unlikely(!p_sys ))
142 goto error;
143 p_sys->vcddev = vcddev;
144
145 /* We read the Table Of Content information */
146 p_sys->i_titles = ioctl_GetTracksMap( VLC_OBJECT(p_access),
147 p_sys->vcddev, &p_sys->p_sectors );
148 if( p_sys->i_titles < 0 )
149 {
150 msg_Err( p_access, "unable to count tracks" );
151 goto error;
152 }
153 else if( p_sys->i_titles <= 1 )
154 {
155 msg_Err( p_access, "no movie tracks found" );
156 goto error;
157 }
158
159 /* The first title isn't usable */
160 p_sys->i_titles--;
161
162 /* Build title table */
163 for( int i = 0; i < p_sys->i_titles; i++ )
164 {
165 input_title_t *t = p_sys->title[i] = vlc_input_title_New();
166
167 msg_Dbg( p_access, "title[%d] start=%d", i, p_sys->p_sectors[1+i] );
168 msg_Dbg( p_access, "title[%d] end=%d", i, p_sys->p_sectors[i+2] );
169
170 t->i_size = ( p_sys->p_sectors[i+2] - p_sys->p_sectors[i+1] ) *
171 (int64_t)VCD_DATA_SIZE;
172 }
173
174 /* Map entry points into chapters */
175 if( EntryPoints( p_access ) )
176 {
177 msg_Warn( p_access, "could not read entry points, will not use them" );
178 }
179
180 /* Starting title/chapter and sector */
181 if( i_title >= p_sys->i_titles )
182 i_title = 0;
183 if( i_chapter >= p_sys->title[i_title]->i_seekpoint )
184 i_chapter = 0;
185
186 p_sys->i_sector = p_sys->p_sectors[1+i_title];
187 if( i_chapter > 0 )
188 {
189 p_sys->i_sector += ( p_sys->title[i_title]->seekpoint[i_chapter]->i_byte_offset /
190 VCD_DATA_SIZE );
191 }
192
193 /* p_access */
194 p_access->pf_read = NULL;
195 p_access->pf_block = Block;
196 p_access->pf_control = Control;
197 p_access->pf_seek = Seek;
198
199 p_access->info.b_eof = false;
200
201 p_sys->i_current_title = i_title;
202 p_sys->i_current_seekpoint = i_chapter;
203 p_access->info.i_pos = (uint64_t)( p_sys->i_sector - p_sys->p_sectors[1+i_title] ) *
204 VCD_DATA_SIZE;
205
206 free( p_access->psz_demux );
207 p_access->psz_demux = strdup( "ps" );
208
209 return VLC_SUCCESS;
210
211error:
212 ioctl_Close( VLC_OBJECT(p_access), vcddev );
213 free( p_sys );
214 return VLC_EGENERIC;
215}

217/*****************************************************************************
218 * Close: closes vcd
219 *****************************************************************************/
220static void Close( vlc_object_t *p_this )
221{
222 access_t *p_access = (access_t *)p_this;
223 access_sys_t *p_sys = p_access->p_sys;
224
225 ioctl_Close( p_this, p_sys->vcddev );
226 free( p_sys );
227}

229/*****************************************************************************
230 * Control:
231 *****************************************************************************/
232static int Control( access_t *p_access, int i_query, va_list args )
233{
234 access_sys_t *p_sys = p_access->p_sys;
235 input_title_t ***ppp_title;
236
237 switch( i_query )
238 {
239 /* */
240 case ACCESS_CAN_SEEK:
241 case ACCESS_CAN_FASTSEEK:
242 case ACCESS_CAN_PAUSE:
243 case ACCESS_CAN_CONTROL_PACE:
244 *va_arg( args, bool* ) = true;
245 break;
246
247 case ACCESS_GET_SIZE:
248 *va_arg( args, uint64_t * ) =
249 p_sys->title[p_sys->i_current_title]->i_size;
250 break;
251
252 /* */
253 case ACCESS_GET_PTS_DELAY:
254 *va_arg( args, int64_t * ) = INT64_C(1000)
255 * var_InheritInteger(p_access, "disc-caching");
256 break;
257
258 /* */
259 case ACCESS_SET_PAUSE_STATE:
260 break;
261
262 case ACCESS_GET_TITLE_INFO:
263 ppp_title = va_arg( args, input_title_t*** );
264 *va_arg( args, int* ) = p_sys->i_titles;
265
266 /* Duplicate title infos */
267 *ppp_title = malloc( sizeof(input_title_t *) * p_sys->i_titles );
268 for( int i = 0; i < p_sys->i_titles; i++ )
269 (*ppp_title)[i] = vlc_input_title_Duplicate( p_sys->title[i] );
270 break;
271
272 case ACCESS_GET_TITLE:
273 *va_arg( args, unsigned * ) = p_sys->i_current_title;
274 break;
275
276 case ACCESS_GET_SEEKPOINT:
277 *va_arg( args, unsigned * ) = p_sys->i_current_seekpoint;
278 break;
279
280 case ACCESS_SET_TITLE:
281 {
282 int i = va_arg( args, int );
283 if( i != p_sys->i_current_title )
284 {
285 /* Update info */
286 p_sys->i_current_title = i;
287 p_sys->i_current_seekpoint = 0;
288 p_access->info.i_pos = 0;
289
290 /* Next sector to read */
291 p_sys->i_sector = p_sys->p_sectors[1+i];
292 }
293 break;
294 }
295
296 case ACCESS_SET_SEEKPOINT:
297 {
298 int i = va_arg( args, int );
299 unsigned i_title = p_sys->i_current_title;
300 input_title_t *t = p_sys->title[i_title];
301
302 if( t->i_seekpoint > 0 )
303 {
304 p_sys->i_current_seekpoint = i;
305
306 p_sys->i_sector = p_sys->p_sectors[1 + i_title] +
307 t->seekpoint[i]->i_byte_offset / VCD_DATA_SIZE;
308
309 p_access->info.i_pos = (uint64_t)(p_sys->i_sector -
310 p_sys->p_sectors[1 + i_title]) *VCD_DATA_SIZE;
311 }
312 break;
313 }
314
315 default:
316 return VLC_EGENERIC;
317 }
318 return VLC_SUCCESS;
319}

321/*****************************************************************************
322 * Block:
323 *****************************************************************************/
324static block_t *Block( access_t *p_access )
325{
326 access_sys_t *p_sys = p_access->p_sys;
327 int i_blocks = VCD_BLOCKS_ONCE;
328 block_t *p_block;
329
330 /* Check end of file */
331 if( p_access->info.b_eof ) return NULL;
332
333 /* Check end of title */
334 while( p_sys->i_sector >= p_sys->p_sectors[p_sys->i_current_title + 2] )
335 {
336 if( p_sys->i_current_title + 2 >= p_sys->i_titles )
337 {
338 p_access->info.b_eof = true;
339 return NULL;
340 }
341
342 p_sys->i_current_title++;
343 p_sys->i_current_seekpoint = 0;
344 p_access->info.i_pos = 0;
345 }
346
347 /* Don't read after the end of a title */
348 if( p_sys->i_sector + i_blocks >=
349 p_sys->p_sectors[p_sys->i_current_title + 2] )
350 {
351 i_blocks = p_sys->p_sectors[p_sys->i_current_title + 2 ] - p_sys->i_sector;
352 }
353
354 /* Do the actual reading */
355 if( i_blocks < 0 || !( p_block = block_Alloc( i_blocks * VCD_DATA_SIZE ) ) )
356 {
357 msg_Err( p_access, "cannot get a new block of size: %i",
358 i_blocks * VCD_DATA_SIZE );
359 return NULL;
360 }
361
362 if( ioctl_ReadSectors( VLC_OBJECT(p_access), p_sys->vcddev,
363 p_sys->i_sector, p_block->p_buffer, i_blocks, VCD_TYPE ) < 0 )
364 {
365 msg_Err( p_access, "cannot read sector %i", p_sys->i_sector );
366 block_Release( p_block );
367
368 /* Try to skip one sector (in case of bad sectors) */
369 p_sys->i_sector++;
370 p_access->info.i_pos += VCD_DATA_SIZE;
371 return NULL;
372 }
373
374 /* Update seekpoints */
375 for( int i_read = 0; i_read < i_blocks; i_read++ )
376 {
377 input_title_t *t = p_sys->title[p_sys->i_current_title];
378
379 if( t->i_seekpoint > 0 &&
380 p_sys->i_current_seekpoint + 1 < t->i_seekpoint &&
381 (int64_t) /* Unlikely to go over 8192 PetaB */
382 (p_access->info.i_pos + i_read * VCD_DATA_SIZE) >=
383 t->seekpoint[p_sys->i_current_seekpoint + 1]->i_byte_offset )
384 {
385 msg_Dbg( p_access, "seekpoint change" );
386 p_sys->i_current_seekpoint++;
387 }
388 }
389
390 /* Update a few values */
391 p_sys->i_sector += i_blocks;
392 p_access->info.i_pos += p_block->i_buffer;
393
394 return p_block;
395}

397/*****************************************************************************
398 * Seek:
399 *****************************************************************************/
400static int Seek( access_t *p_access, uint64_t i_pos )
401{
402 access_sys_t *p_sys = p_access->p_sys;
403 input_title_t *t = p_sys->title[p_sys->i_current_title];
404 int i_seekpoint;
405
406 /* Next sector to read */
407 p_access->info.i_pos = i_pos;
408 p_sys->i_sector = i_pos / VCD_DATA_SIZE +
409 p_sys->p_sectors[p_sys->i_current_title + 1];
410
411 /* Update current seekpoint */
412 for( i_seekpoint = 0; i_seekpoint < t->i_seekpoint; i_seekpoint++ )
413 {
414 if( i_seekpoint + 1 >= t->i_seekpoint ) break;
415 if( 0 < t->seekpoint[i_seekpoint + 1]->i_byte_offset &&
416 i_pos < (uint64_t)t->seekpoint[i_seekpoint + 1]->i_byte_offset ) break;
417 }
418
419 if( i_seekpoint != p_sys->i_current_seekpoint )
420 {
421 msg_Dbg( p_access, "seekpoint change" );
422 p_sys->i_current_seekpoint = i_seekpoint;
423 }
424
425 /* Reset eof */
426 p_access->info.b_eof = false;
427
428 return VLC_SUCCESS;
429}

431/*****************************************************************************
432 * EntryPoints: Reads the information about the entry points on the disc.
433 *****************************************************************************/
434static int EntryPoints( access_t *p_access )
435{
436 access_sys_t *p_sys = p_access->p_sys;
437 uint8_t sector[VCD_DATA_SIZE];
438
439 entries_sect_t entries;
440 int i_nb;
441
442 /* Read the entry point sector */
443 if( ioctl_ReadSectors( VLC_OBJECT(p_access), p_sys->vcddev,
444 VCD_ENTRIES_SECTOR, sector, 1, VCD_TYPE ) < 0 )
445 {
446 msg_Err( p_access, "could not read entry points sector" );
447 return VLC_EGENERIC;
448 }
449 memcpy( &entries, sector, CD_SECTOR_SIZE );
450
451 i_nb = GetWBE( &entries.i_entries_nb );
452 if( i_nb > 500 )
453 {
454 msg_Err( p_access, "invalid entry points number" );
455 return VLC_EGENERIC;
456 }
457
458 if( strncmp( entries.psz_id, "ENTRYVCD", sizeof( entries.psz_id ) ) &&
459 strncmp( entries.psz_id, "ENTRYSVD", sizeof( entries.psz_id ) ) )
460 {
461 msg_Err( p_access, "unrecognized entry points format" );
462 return VLC_EGENERIC;
463 }
464
465 for( int i = 0; i < i_nb; i++ )
466 {
467 const int i_title = BCD_TO_BIN(entries.entry[i].i_track) - 2;
468 const int i_sector =
469 (MSF_TO_LBA2( BCD_TO_BIN( entries.entry[i].msf.minute ),
470 BCD_TO_BIN( entries.entry[i].msf.second ),
471 BCD_TO_BIN( entries.entry[i].msf.frame ) ));
472 seekpoint_t *s;
473
474 if( i_title < 0 ) continue; /* Should not occur */
475 if( i_title >= p_sys->i_titles ) continue;
476
477 msg_Dbg( p_access, "Entry[%d] title=%d sector=%d",
478 i, i_title, i_sector );
479
480 s = vlc_seekpoint_New();
481 s->i_byte_offset = (i_sector - p_sys->p_sectors[i_title+1]) *
482 VCD_DATA_SIZE;
483
484 TAB_APPEND( p_sys->title[i_title]->i_seekpoint,
485 p_sys->title[i_title]->seekpoint, s );
486 }
487
488 return VLC_SUCCESS;
489}
