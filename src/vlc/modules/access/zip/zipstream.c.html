124inline static void free_all_node( node *root )
125{
126 while( root )
127 {
128 free_all_node( root->child );

130 node *tmp = root->next;
131 free( root );
132 root = tmp;
133 }
134}

136/* Allocate strcat and format */
137static int astrcatf( char **ppsz_dest, const char *psz_fmt_src, ... )
138{
139 char *psz_tmp;
140 va_list args;
141
142 va_start( args, psz_fmt_src );
143 int i_ret = vasprintf( &psz_tmp, psz_fmt_src, args );
144 va_end( args );
145
146 if( i_ret == -1 ) return -1;
147
148 int i_len = strlen( *ppsz_dest ) + strlen( psz_tmp ) + 1;
149 char *psz_out = realloc( *ppsz_dest, i_len );
150 if( !psz_out ) return -1;
151
152 strcat( psz_out, psz_tmp );
153 free( psz_tmp );
154
155 *ppsz_dest = psz_out;
156 return i_len;
157}

166/** **************************************************************************
167 * Open
168 *****************************************************************************/
169int StreamOpen( vlc_object_t *p_this )
170{
171 stream_t *s = (stream_t*) p_this;
172 stream_sys_t *p_sys;
173
174 /* Verify file format */
175 const uint8_t *p_peek;
176 if( stream_Peek( s->p_source, &p_peek, i_zip_marker ) < i_zip_marker )
177 return VLC_EGENERIC;
178 if( memcmp( p_peek, p_zip_marker, i_zip_marker ) )
179 return VLC_EGENERIC;
180
181 s->p_sys = p_sys = calloc( 1, sizeof( *p_sys ) );
182 if( !p_sys )
183 return VLC_ENOMEM;
184
185 s->pf_read = Read;
186 s->pf_peek = Peek;
187 s->pf_control = Control;
188
189 p_sys->fileFunctions = ( zlib_filefunc_def * )
190 calloc( 1, sizeof( zlib_filefunc_def ) );
191 if( !p_sys->fileFunctions )
192 {
193 free( p_sys );
194 return VLC_ENOMEM;
195 }
196 p_sys->fileFunctions->zopen_file = ZipIO_Open;
197 p_sys->fileFunctions->zread_file = ZipIO_Read;
198 p_sys->fileFunctions->zwrite_file = ZipIO_Write;
199 p_sys->fileFunctions->ztell_file = ZipIO_Tell;
200 p_sys->fileFunctions->zseek_file = ZipIO_Seek;
201 p_sys->fileFunctions->zclose_file = ZipIO_Close;
202 p_sys->fileFunctions->zerror_file = ZipIO_Error;
203 p_sys->fileFunctions->opaque = ( void * ) s;
204 p_sys->zipFile = unzOpen2( NULL /* path */, p_sys->fileFunctions );
205 if( !p_sys->zipFile )
206 {
207 msg_Warn( s, "unable to open file" );
208 free( p_sys->fileFunctions );
209 free( p_sys );
210 return VLC_EGENERIC;
211 }
212
213 /* Find the stream uri */
214 char *psz_tmp;
215 if( asprintf( &psz_tmp, "%s.xspf", s->psz_path ) == -1 )
216 {
217 free( p_sys->fileFunctions );
218 free( p_sys );
219 return VLC_ENOMEM;
220 }
221 p_sys->psz_path = s->psz_path;
222 s->psz_path = psz_tmp;
223
224 return VLC_SUCCESS;
225}

248/** *************************************************************************
249 * Read
250 ****************************************************************************/
251static int Read( stream_t *s, void *p_read, unsigned int i_read )
252{
253 stream_sys_t *p_sys = s->p_sys;
254
255 /* Fill the buffer */
256 if( Fill( s ) )
257 return -1;
258
259 /* Read the buffer */
260 unsigned i_len = __MIN( i_read, p_sys->i_len - p_sys->i_pos );
261 if( p_read )
262 memcpy( p_read, p_sys->psz_xspf + p_sys->i_pos, i_len );
263 p_sys->i_pos += i_len;
264
265 return i_len;
266}

268/** *************************************************************************
269 * Peek
270 ****************************************************************************/
271static int Peek( stream_t *s, const uint8_t **pp_peek, unsigned int i_peek )
272{
273 stream_sys_t *p_sys = s->p_sys;
274
275 /* Fill the buffer */
276 if( Fill( s ) )
277 return -1;
278
279 /* Point to the buffer */
280 int i_len = __MIN( i_peek, p_sys->i_len - p_sys->i_pos );
281 *pp_peek = (uint8_t*) p_sys->psz_xspf + p_sys->i_pos;
282
283 return i_len;
284}

286/** *************************************************************************
287 * Control
288 ****************************************************************************/
289static int Control( stream_t *s, int i_query, va_list args )
290{
291 stream_sys_t *p_sys = s->p_sys;
292
293 switch( i_query )
294 {
295 case STREAM_SET_POSITION:
296 {
297 uint64_t i_position = va_arg( args, uint64_t );
298 if( i_position >= p_sys->i_len )
299 return VLC_EGENERIC;
300 else
301 {
302 p_sys->i_pos = (size_t) i_position;
303 return VLC_SUCCESS;
304 }
305 }
306
307 case STREAM_GET_POSITION:
308 {
309 uint64_t *pi_position = va_arg( args, uint64_t* );
310 *pi_position = p_sys->i_pos;
311 return VLC_SUCCESS;
312 }
313
314 case STREAM_GET_SIZE:
315 {
316 uint64_t *pi_size = va_arg( args, uint64_t* );
317 *pi_size = p_sys->i_len;
318 return VLC_SUCCESS;
319 }
320
321 case STREAM_GET_CONTENT_TYPE:
322 return VLC_EGENERIC;
323
324 case STREAM_UPDATE_SIZE:
325 case STREAM_CAN_SEEK:
326 case STREAM_CAN_FASTSEEK:
327 case STREAM_SET_RECORD_STATE:
328 return stream_vaControl( s->p_source, i_query, args );
329
330 default:
331 return VLC_EGENERIC;
332 }
333}

335static int Fill( stream_t *s )
336{
337 stream_sys_t *p_sys = s->p_sys;
338
339 if( p_sys->psz_xspf )
340 return VLC_SUCCESS;
341
342 if( CreatePlaylist( s, &p_sys->psz_xspf ) < 0 )
343 return VLC_EGENERIC;
344
345 p_sys->i_len = strlen( p_sys->psz_xspf );
346 p_sys->i_pos = 0;
347 return VLC_SUCCESS;
348}
349
350static int CreatePlaylist( stream_t *s, char **pp_buffer )
351{
352 stream_sys_t *p_sys = s->p_sys;
353
354 unzFile file = p_sys->zipFile;
355 if( !file )
356 return -1;
357
358 /* Get some infos about zip archive */
359 int i_ret = 0;
360 vlc_array_t *p_filenames = vlc_array_new(); /* Will contain char* */
361
362 /* List all file names in Zip archive */
363 i_ret = GetFilesInZip( s, file, p_filenames, NULL );
364 if( i_ret < 0 )
365 {
366 i_ret = -1;
367 goto exit;
368 }
369
370 /* Construct the xspf playlist */
371 i_ret = WriteXSPF( pp_buffer, p_filenames, p_sys->psz_path );
372 if( i_ret > 0 )
373 i_ret = 1;
374 else if( i_ret < 0 )
375 i_ret = -1;
376
377exit:
378 /* Close archive */
379 unzClose( file );
380 p_sys->zipFile = NULL;
381
382 for( int i = 0; i < vlc_array_count( p_filenames ); i++ )
383 {
384 free( vlc_array_item_at_index( p_filenames, i ) );
385 }
386 vlc_array_destroy( p_filenames );
387 return i_ret;
388}

395/** **************************************************************************
396 * \brief List files in zip and append their names to p_array
397 * \param p_this
398 * \param file Opened zip file
399 * \param p_array vlc_array_t which will receive all filenames
400 *
401 * In case of error, returns VLC_EGENERIC.
402 * In case of success, returns number of files found, and goes back to first file.
403 *****************************************************************************/
404static int GetFilesInZip( stream_t *p_this, unzFile file,
405 vlc_array_t *p_filenames, vlc_array_t *p_fileinfos )
406{
407 if( !p_filenames || !p_this )
408 return VLC_EGENERIC;
409
410 int i_ret = 0;
411
412 /* Get global info */
413 unz_global_info info;
414
415 if( unzGetGlobalInfo( file, &info ) != UNZ_OK )
416 {
417 msg_Warn( p_this, "this is not a valid zip archive" );
418 return VLC_EGENERIC;
419 }
420
421 /* Go to first file in archive */
422 unzGoToFirstFile( file );
423
424 /* Get info about each file */
425 for( unsigned long i = 0; i < info.number_entry; i++ )
426 {
427 char *psz_fileName = calloc( ZIP_FILENAME_LEN, 1 );
428 unz_file_info *p_fileInfo = calloc( 1, sizeof( unz_file_info ) );
429
430 if( !p_fileInfo || !psz_fileName )
431 {
432 free( psz_fileName );
433 free( p_fileInfo );
434 return VLC_ENOMEM;
435 }
436
437 if( unzGetCurrentFileInfo( file, p_fileInfo, psz_fileName,
438 ZIP_FILENAME_LEN, NULL, 0, NULL, 0 )
439 != UNZ_OK )
440 {
441 msg_Warn( p_this, "can't get info about file in zip" );
442 free( psz_fileName );
443 free( p_fileInfo );
444 return VLC_EGENERIC;
445 }
446
447 if( p_filenames )
448 vlc_array_append( p_filenames, strdup( psz_fileName ) );
449 free( psz_fileName );
450
451 if( p_fileinfos )
452 vlc_array_append( p_fileinfos, p_fileInfo );
453 else
454 free( p_fileInfo );
455
456 if( i < ( info.number_entry - 1 ) )
457 {
458 /* Go the next file in the archive */
459 if( unzGoToNextFile( file ) != UNZ_OK )
460 {
461 msg_Warn( p_this, "can't go to next file in zip" );
462 return VLC_EGENERIC;
463 }
464 }
465
466 i_ret++;
467 }
468
469 /* i_ret should be equal to info.number_entry */
470 unzGoToFirstFile( file );
471 return i_ret;
472}

476 * XSPF generation functions

479/** **************************************************************************
480 * \brief Check a character for allowance in the Xml.
481 * Allowed chars are: a-z, A-Z, 0-9, \, /, ., ' ', _ and :
482 *****************************************************************************/
483bool isAllowedChar( char c )
484{
485 return ( c >= 'a' && c <= 'z' )
486 || ( c >= 'A' && c <= 'Z' )
487 || ( c >= '0' && c <= '9' )
488 || ( c == ':' ) || ( c == '/' )
489 || ( c == '\\' ) || ( c == '.' )
490 || ( c == ' ' ) || ( c == '_' );
491}
492
493/** **************************************************************************
494 * \brief Escape string to be XML valid
495 * Allowed chars are defined by the above function isAllowedChar()
496 * Invalid chars are escaped using non standard '?XX' notation.
497 * NOTE: We cannot trust VLC internal Web encoding functions
498 * because they are not able to encode and decode some rare utf-8
499 * characters properly. Also, we don't control exactly when they are
500 * called (from this module).
501 *****************************************************************************/
502static int escapeToXml( char **ppsz_encoded, const char *psz_url )
503{
504 char *psz_iter, *psz_tmp;
505
506 /* Count number of unallowed characters in psz_url */
507 size_t i_num = 0, i_len = 0;
508 for( psz_iter = (char*) psz_url; *psz_iter; ++psz_iter )
509 {
510 if( isAllowedChar( *psz_iter ) )
511 {
512 i_len++;
513 }
514 else
515 {
516 i_len++;
517 i_num++;
518 }
519 }
520
521 /* Special case */
522 if( i_num == 0 )
523 {
524 *ppsz_encoded = malloc( i_len + 1 );
525 memcpy( *ppsz_encoded, psz_url, i_len + 1 );
526 return VLC_SUCCESS;
527 }
528
529 /* Copy string, replacing invalid characters */
530 char *psz_ret = malloc( i_len + 3*i_num + 2 );
531 if( !psz_ret ) return VLC_ENOMEM;
532
533 for( psz_iter = (char*) psz_url, psz_tmp = psz_ret;
534 *psz_iter; ++psz_iter, ++psz_tmp )
535 {
536 if( isAllowedChar( *psz_iter ) )
537 {
538 *psz_tmp = *psz_iter;
539 }
540 else
541 {
542 *(psz_tmp++) = '?';
543 snprintf( psz_tmp, 3, "%02x", ( *psz_iter & 0x000000FF ) );
544 psz_tmp++;
545 }
546 }
547 *psz_tmp = '\0';
548
549 /* Return success */
550 *ppsz_encoded = psz_ret;
551 return VLC_SUCCESS;
552}

554/** **************************************************************************
555 * \brief Write the XSPF playlist given the list of files
556 *****************************************************************************/
557static int WriteXSPF( char **pp_buffer, vlc_array_t *p_filenames,
558 const char *psz_zippath )
559{
560 char *psz_zip = strrchr( psz_zippath, DIR_SEP_CHAR );
561 psz_zip = convert_xml_special_chars( psz_zip ? (psz_zip+1) : psz_zippath );
562
563 if( asprintf( pp_buffer, "<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n"
564 "<playlist version=\"1\" xmlns=\"http://xspf.org/ns/0/\" "
565 "xmlns:vlc=\"http://www.videolan.org/vlc/playlist/ns/0/\">\n"
566 " <title>%s</title>\n"
567 " <trackList>\n", psz_zip ) == -1)
568 return -1;
569
570 /* Root node */
571 node *playlist = new_node( psz_zip );
572
573 /* Encode the URI and append ZIP_SEP */
574 char *psz_pathtozip;
575 escapeToXml( &psz_pathtozip, psz_zippath );
576 if( astrcatf( &psz_pathtozip, "%s", ZIP_SEP ) < 0 ) return -1;
577
578 int i_track = 0;
579 for( int i = 0; i < vlc_array_count( p_filenames ); ++i )
580 {
581 char *psz_name = (char*) vlc_array_item_at_index( p_filenames, i );
582 int i_len = strlen( psz_name );
583
584 if( !i_len ) continue;
585
586 /* Is it a folder ? */
587 if( psz_name[i_len-1] == '/' )
588 {
589 /* Do nothing */
590 }
591 else /* File */
592 {
593 /* Extract file name */
594 char *psz_file = strrchr( psz_name, '/' );
595 psz_file = convert_xml_special_chars( psz_file ?
596 (psz_file+1) : psz_name );
597
598 /* Build full MRL */
599 char *psz_path = strdup( psz_pathtozip );
600 char *psz_escapedName;
601 escapeToXml( &psz_escapedName, psz_name );
602 if( astrcatf( &psz_path, "%s", psz_escapedName ) < 0 )
603 {
604 free( psz_escapedName );
605 return -1;
606 }
607 free( psz_escapedName );
608
609 /* Track information */
610 if( astrcatf( pp_buffer,
611 " <track>\n"
612 " <location>zip://%s</location>\n"
613 " <title>%s</title>\n"
614 " <extension application=\"http://www.videolan.org/vlc/playlist/0\">\n"
615 " <vlc:id>%d</vlc:id>\n"
616 " </extension>\n"
617 " </track>\n",
618 psz_path, psz_file, i_track ) < 0 ) return -1;
619
620 free( psz_file );
621 free( psz_path );
622
623 /* Find the parent node */
624 node *parent = findOrCreateParentNode( playlist, psz_name );
625 assert( parent );
626
627 /* Add the item to this node */
628 item *tmp = parent->media;
629 if( !tmp )
630 {
631 parent->media = new_item( i_track );
632 }
633 else
634 {
635 while( tmp->next )
636 {
637 tmp = tmp->next;
638 }
639 tmp->next = new_item( i_track );
640 }
641
642 ++i_track;
643 }
644 }
645
646 free( psz_pathtozip );
647
648 /* Close tracklist, open the extension */
649 if( astrcatf( pp_buffer,
650 " </trackList>\n"
651 " <extension application=\"http://www.videolan.org/vlc/playlist/0\">\n"
652 ) < 0 ) return -1;
653
654 /* Write the tree */
655 if( nodeToXSPF( pp_buffer, playlist, true ) < 0 ) return -1;
656
657 /* Close extension and playlist */
658 if( astrcatf( pp_buffer, " </extension>\n</playlist>\n" ) < 0 ) return -1;
659
660 /* printf( "%s", *pp_buffer ); */
661
662 free_all_node( playlist );
663
664 return VLC_SUCCESS;
665}

667/** **************************************************************************
668 * \brief Recursively convert a node to its XSPF representation
669 *****************************************************************************/
670static int nodeToXSPF( char **pp_buffer, node *n, bool b_root )
671{
672 if( !b_root )
673 {
674 if( astrcatf( pp_buffer, " <vlc:node title=\"%s\">\n", n->name ) < 0 )
675 return -1;
676 }
677 if( n->child )
678 nodeToXSPF( pp_buffer, n->child, false );
679 item *i = n->media;
680 while( i )
681 {
682 if( astrcatf( pp_buffer, " <vlc:item tid=\"%d\" />\n", i->id ) < 0 )
683 return -1;
684 i = i->next;
685 }
686 if( !b_root )
687 {
688 if( astrcatf( pp_buffer, " </vlc:node>\n" ) < 0 )
689 return -1;
690 }
691 return VLC_SUCCESS;
692}

694/** **************************************************************************
695 * \brief Either create or find the parent node of the item
696 *****************************************************************************/
697static node* findOrCreateParentNode( node *root, const char *fullpath )
698{
699 char *folder;
700 char *path = strdup( fullpath );
701 folder = path;
702
703 assert( root );
704
705 char *sep = strchr( folder, '/' );
706 if( !sep )
707 {
708 free( path );
709 return root;
710 }
711
712 *sep = '\0';
713 ++sep;
714
715 node *current = root->child;
716
717 while( current )
718 {
719 if( !strcmp( current->name, folder ) )
720 {
721 /* We found the folder, go recursively deeper */
722 node *parentNode = findOrCreateParentNode( current, sep );
723 free( path );
724 return parentNode;
725 }
726 current = current->next;
727 }
728
729 /* If we are here, then it means that we did not find the parent */
730 node *ret = new_node( folder );
731 if( !root->child )
732 root->child = ret;
733 else
734 {
735 current = root->child;
736 while( current->next )
737 {
738 current = current->next;
739 }
740 current->next = ret;
741 }
742
743 /* And now, create the subfolders */
744 ret = findOrCreateParentNode( ret, sep );
745
746 free( path );
747 return ret;
748}

755/** **************************************************************************
756 * \brief interface for unzip module to open a file using a vlc_stream
757 * \param opaque
758 * \param filename
759 * \param mode how to open the file (read/write ?). We support only read
760 * \return opaque
761 *****************************************************************************/
762static void ZCALLBACK *ZipIO_Open( void *opaque, const char *file, int mode )
763{
764 (void) file;
765 stream_t *s = (stream_t*) opaque;

772}
