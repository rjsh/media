24#include <libavformat/avformat.h>
25#include <libavformat/avio.h>
26
27int OpenAvio (vlc_object_t *);
28void CloseAvio(vlc_object_t *);
29int OutOpenAvio (vlc_object_t *);
30void OutCloseAvio(vlc_object_t *);
31
32#define AVIO_MODULE \
33 set_shortname(N_("AVIO")) \
34 set_description(N_("libavformat AVIO access") ) \
35 set_category(CAT_INPUT) \
36 set_subcategory(SUBCAT_INPUT_ACCESS) \
37 set_capability("access", -1) \
38 add_shortcut("avio", "rtmp", "rtmpe", "rtmps", "rtmpt", "rtmpte", "rtmpts") \
39 set_callbacks(OpenAvio, CloseAvio) \
40 set_section(N_("Input"), NULL ) \
41 add_string("avio-options", NULL, AV_OPTIONS_TEXT, AV_OPTIONS_LONGTEXT, true) \
42 add_submodule () \
43 set_shortname( "AVIO" ) \
44 set_description( N_("libavformat AVIO access output") ) \
45 set_capability( "sout access", -1 ) \
46 set_category( CAT_SOUT ) \
47 set_subcategory( SUBCAT_SOUT_ACO ) \
48 add_shortcut( "avio", "rtmp" ) \
49 set_callbacks( OutOpenAvio, OutCloseAvio) \
50 set_section(N_("Stream output"), NULL ) \
51 add_string("sout-avio-options", NULL, AV_OPTIONS_TEXT, AV_OPTIONS_LONGTEXT, true)
