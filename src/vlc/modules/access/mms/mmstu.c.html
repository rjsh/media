86int MMSTUOpen( access_t *p_access )
87{
88 access_sys_t *p_sys;
89 int i_proto;
90 int i_status;
91
92 /* Set up p_access */
93 access_InitFields( p_access );
94 p_access->pf_read = NULL;
95 p_access->pf_block = Block;
96 p_access->pf_control = Control;
97 p_access->pf_seek = Seek;
98
99 p_access->p_sys = p_sys = calloc( 1, sizeof( access_sys_t ) );
100 if( !p_sys ) return VLC_ENOMEM;
101
102 p_sys->i_timeout = var_CreateGetInteger( p_access, "mms-timeout" );
103
104 vlc_mutex_init( &p_sys->lock_netwrite );
105
106 /* *** Parse URL and get server addr/port and path *** */
107 vlc_UrlParse( &p_sys->url, p_access->psz_location, 0 );
108 if( p_sys->url.psz_host == NULL || *p_sys->url.psz_host == '\0' )
109 {
110 msg_Err( p_access, "invalid server name" );
111 vlc_UrlClean( &p_sys->url );
112 vlc_mutex_destroy( &p_sys->lock_netwrite );
113 free( p_sys );
114 return VLC_EGENERIC;
115 }
116 if( p_sys->url.i_port <= 0 )
117 {
118 p_sys->url.i_port = 1755;
119 }
120
121 /* *** connect to this server *** */
122 /* look at requested protocol (udp/tcp) */
123 i_proto = MMS_PROTO_AUTO;
124 if( *p_access->psz_access )
125 {
126 if( !strncmp( p_access->psz_access, "mmsu", 4 ) )
127 {
128 i_proto = MMS_PROTO_UDP;
129 }
130 else if( !strncmp( p_access->psz_access, "mmst", 4 ) )
131 {
132 i_proto = MMS_PROTO_TCP;
133 }
134 }
135
136 /* connect */
137 if( i_proto == MMS_PROTO_AUTO )
138 { /* first try with TCP and then UDP*/
139 if( ( i_status = MMSOpen( p_access, &p_sys->url, MMS_PROTO_TCP ) ) )
140 {
141 if( vlc_object_alive(p_access) )
142 i_status = MMSOpen( p_access, &p_sys->url, MMS_PROTO_UDP );
143 }
144 }
145 else
146 {
147 i_status = MMSOpen( p_access, &p_sys->url, i_proto );
148 }
149
150 if( i_status )
151 {
152 msg_Err( p_access, "cannot connect to server" );
153 vlc_UrlClean( &p_sys->url );
154 vlc_mutex_destroy( &p_sys->lock_netwrite );
155 free( p_sys );
156 return VLC_EGENERIC;
157 }
158
159 msg_Dbg( p_access, "connected to %s:%d", p_sys->url.psz_host, p_sys->url.i_port );
160 /*
161 * i_flags_broadcast
162 * yy xx ?? ??
163 * broadcast yy=0x02, xx= 0x00
164 * pre-recorded yy=0x01, xx= 0x80 if video, 0x00 no video
165 */
166 if( p_sys->i_packet_count <= 0 && p_sys->asfh.i_data_packets_count > 0 )
167 {
168 p_sys->i_packet_count = p_sys->asfh.i_data_packets_count;
169 }
170 if( p_sys->i_packet_count <= 0 || ( p_sys->i_flags_broadcast >> 24 ) == 0x02 )
171 {
172 p_sys->b_seekable = false;
173 }
174 else
175 {
176 p_sys->b_seekable = true;
177 p_sys->i_size =
178 (uint64_t)p_sys->i_header +
179 (uint64_t)p_sys->i_packet_count * (uint64_t)p_sys->i_packet_length;
180 }
181 p_sys->b_keep_alive = false;
182
183 /* *** Start stream *** */
184 if( MMSStart( p_access, 0xffffffff ) < 0 )
185 {
186 msg_Err( p_access, "cannot start stream" );
187 MMSTUClose ( p_access );
188 return VLC_EGENERIC;
189 }
190
191 return VLC_SUCCESS;
192}

195 * Close: free unused data structures

197void MMSTUClose( access_t *p_access )
198{
199 access_sys_t *p_sys = p_access->p_sys;
200
201 KeepAliveStop( p_access );
202
203 /* close connection with server */
204 MMSClose( p_access );
205

208
209 free( p_sys );
210}

212/*****************************************************************************
213 * Control:
214 *****************************************************************************/
215static int Control( access_t *p_access, int i_query, va_list args )
216{
217 access_sys_t *p_sys = p_access->p_sys;
218 bool *pb_bool;
219 bool b_bool;
220 int64_t *pi_64;
221 int i_int;
222
223 switch( i_query )
224 {
225 case ACCESS_CAN_SEEK:
226 pb_bool = (bool*)va_arg( args, bool* );
227 *pb_bool = p_sys->b_seekable;
228 break;
229
230 case ACCESS_CAN_FASTSEEK:
231 pb_bool = (bool*)va_arg( args, bool* );
232 *pb_bool = false;
233 break;
234
235 case ACCESS_CAN_PAUSE:
236 pb_bool = (bool*)va_arg( args, bool* );
237 *pb_bool = true;
238 break;
239
240 case ACCESS_CAN_CONTROL_PACE:
241 pb_bool = (bool*)va_arg( args, bool* );
242
243#if 0 /* Disable for now until we have a clock synchro algo
244 * which works with something else than MPEG over UDP */
245 *pb_bool = false;
246#endif
247 *pb_bool = true;
248 break;
249
250 case ACCESS_GET_SIZE:
251 *va_arg( args, uint64_t * ) = p_sys->i_size;
252 break;
253
254 case ACCESS_GET_PTS_DELAY:
255 pi_64 = (int64_t*)va_arg( args, int64_t * );
256 *pi_64 = INT64_C(1000)
257 * var_InheritInteger( p_access, "network-caching" );
258 break;
259
260 case ACCESS_GET_PRIVATE_ID_STATE:
261 i_int = (int)va_arg( args, int );
262 pb_bool = (bool *)va_arg( args, bool * );
263
264 if( i_int < 0 || i_int > 127 )
265 return VLC_EGENERIC;
266 *pb_bool = p_sys->asfh.stream[i_int].i_selected ? true : false;
267 break;
268
269 case ACCESS_SET_PAUSE_STATE:
270 b_bool = (bool)va_arg( args, int );
271 if( b_bool )
272 {
273 MMSStop( p_access );
274 KeepAliveStart( p_access );
275 }
276 else
277 {
278 KeepAliveStop( p_access );
279 Seek( p_access, p_access->info.i_pos );
280 }
281 break;
282
283 default:
284 return VLC_EGENERIC;
285 }
286 return VLC_SUCCESS;
287}

289/*****************************************************************************
290 * Seek: try to go at the right place
291 *****************************************************************************/
292static int Seek( access_t * p_access, uint64_t i_pos )
293{
294 access_sys_t *p_sys = p_access->p_sys;
295 uint32_t i_packet;
296 uint32_t i_offset;
297 var_buffer_t buffer;
298
299 if( i_pos < p_sys->i_header)
300 {
301 if( p_access->info.i_pos < p_sys->i_header )
302 {
303 /* no need to restart stream, it was already one
304 * or no stream was yet read */
305 p_access->info.i_pos = i_pos;
306 return VLC_SUCCESS;
307 }
308 else
309 {
310 i_packet = 0xffffffff;
311 i_offset = 0;
312 }
313 }
314 else
315 {
316 i_packet = ( i_pos - p_sys->i_header ) / p_sys->i_packet_length;
317 i_offset = ( i_pos - p_sys->i_header ) % p_sys->i_packet_length;
318 }
319 if( p_sys->b_seekable && i_packet >= p_sys->i_packet_count )
320 return VLC_EGENERIC;
321
322 msg_Dbg( p_access, "seeking to %"PRIu64 " (packet:%u)", i_pos, i_packet );
323
324 MMSStop( p_access );
325 msg_Dbg( p_access, "stream stopped (seek)" );
326
327 /* *** restart stream *** */
328 var_buffer_initwrite( &buffer, 0 );
329 var_buffer_add64( &buffer, 0 ); /* seek point in second */
330 var_buffer_add32( &buffer, 0xffffffff );
331 var_buffer_add32( &buffer, i_packet ); // begin from start
332 var_buffer_add8( &buffer, 0xff ); // stream time limit
333 var_buffer_add8( &buffer, 0xff ); // on 3bytes ...
334 var_buffer_add8( &buffer, 0xff ); //
335 var_buffer_add8( &buffer, 0x00 ); // don't use limit
336 var_buffer_add32( &buffer, p_sys->i_media_packet_id_type );
337
338 mms_CommandSend( p_access, 0x07, p_sys->i_command_level, 0x0001ffff,
339 buffer.p_data, buffer.i_data );
340
341 var_buffer_free( &buffer );
342
343
344 while( vlc_object_alive (p_access) )
345 {
346 if( mms_HeaderMediaRead( p_access, MMS_PACKET_CMD ) < 0 )
347 {
348 p_access->info.b_eof = true;
349 return VLC_EGENERIC;
350 }
351
352 if( p_sys->i_command == 0x1e )
353 {
354 msg_Dbg( p_access, "received 0x1e (seek)" );
355 break;
356 }
357 }
358
359 while( vlc_object_alive (p_access) )
360 {
361 if( mms_HeaderMediaRead( p_access, MMS_PACKET_CMD ) < 0 )
362 {
363 p_access->info.b_eof = true;
364 return VLC_EGENERIC;
365 }
366 if( p_sys->i_command == 0x05 )
367 {
368 msg_Dbg( p_access, "received 0x05 (seek)" );
369 break;
370 }
371 }
372
373 /* get a packet */
374 if( mms_HeaderMediaRead( p_access, MMS_PACKET_MEDIA ) < 0 )
375 {
376 p_access->info.b_eof = true;
377 return VLC_EGENERIC;
378 }
379
380 msg_Dbg( p_access, "Streaming restarted" );
381
382 p_sys->i_media_used += i_offset;
383 p_access->info.i_pos = i_pos;
384 p_access->info.b_eof = false;
385
386 return VLC_SUCCESS;
387}

389/*****************************************************************************
390 * Block:
391 *****************************************************************************/
392static block_t *Block( access_t *p_access )
393{
394 access_sys_t *p_sys = p_access->p_sys;
395
396 if( p_access->info.b_eof )
397 return NULL;
398
399 if( p_access->info.i_pos < p_sys->i_header )
400 {
401 const size_t i_copy = p_sys->i_header - p_access->info.i_pos;
402
403 block_t *p_block = block_Alloc( i_copy );
404 if( !p_block )
405 return NULL;
406
407 memcpy( p_block->p_buffer, &p_sys->p_header[p_access->info.i_pos], i_copy );
408 p_access->info.i_pos += i_copy;
409 return p_block;
410 }
411 else if( p_sys->p_media && p_sys->i_media_used < __MAX( p_sys->i_media, p_sys->i_packet_length ) )
412 {
413 size_t i_copy = 0;
414 size_t i_padding = 0;
415
416 if( p_sys->i_media_used < p_sys->i_media )
417 i_copy = p_sys->i_media - p_sys->i_media_used;
418 if( __MAX( p_sys->i_media, p_sys->i_media_used ) < p_sys->i_packet_length )
419 i_padding = p_sys->i_packet_length - __MAX( p_sys->i_media, p_sys->i_media_used );
420
421 block_t *p_block = block_Alloc( i_copy + i_padding );
422 if( !p_block )
423 return NULL;
424
425 if( i_copy > 0 )
426 memcpy( &p_block->p_buffer[0], &p_sys->p_media[p_sys->i_media_used], i_copy );
427 if( i_padding > 0 )
428 memset( &p_block->p_buffer[i_copy], 0, i_padding );
429
430 p_sys->i_media_used += i_copy + i_padding;
431 p_access->info.i_pos += i_copy + i_padding;
432 return p_block;
433 }
434
435 mms_HeaderMediaRead( p_access, MMS_PACKET_MEDIA );
436 return NULL;
437}

439/****************************************************************************
440 * MMSOpen : Open a connection with the server over mmst or mmsu
441 ****************************************************************************/
442static int MMSOpen( access_t *p_access, vlc_url_t *p_url, int i_proto )
443{
444 access_sys_t *p_sys = p_access->p_sys;
445 int b_udp = ( i_proto == MMS_PROTO_UDP ) ? 1 : 0;
446
447 var_buffer_t buffer;
448 char *tmp;
449 uint16_t *p;
450 int i_server_version;
451 int i_tool_version;
452 int i_update_player_url;
453 int i_encryption_type;
454 int i;
455 int i_streams;
456 int i_first;
457 char *mediapath;
458
459
460 /* *** Open a TCP connection with server *** */
461 msg_Dbg( p_access, "waiting for connection..." );
462 p_sys->i_handle_tcp = net_ConnectTCP( p_access, p_url->psz_host, p_url->i_port );
463 if( p_sys->i_handle_tcp < 0 )
464 {
465 msg_Err( p_access, "failed to open a connection (tcp)" );
466 return VLC_EGENERIC;
467 }
468 msg_Dbg( p_access,
469 "connection(tcp) with \"%s:%d\" successful",
470 p_url->psz_host,
471 p_url->i_port );
472
473 /* *** Bind port if UDP protocol is selected *** */
474 if( b_udp )
475 {
476 if( net_GetSockAddress( p_sys->i_handle_tcp, p_sys->sz_bind_addr,
477 NULL ) )
478 {
479 net_Close( p_sys->i_handle_tcp );
480 return VLC_EGENERIC;
481 }
482
483 p_sys->i_handle_udp = net_ListenUDP1( (vlc_object_t *)p_access, p_sys->sz_bind_addr,
484 7000 );
485 if( p_sys->i_handle_udp < 0 )
486 {
487 msg_Err( p_access, "failed to open a connection (udp)" );
488 net_Close( p_sys->i_handle_tcp );
489 return VLC_EGENERIC;
490 }
491 msg_Dbg( p_access,
492 "connection(udp) at \"%s:%d\" successful",
493 p_sys->sz_bind_addr, 7000 );
494 }
495
496 /* *** Init context for mms prototcol *** */
497 GenerateGuid ( &p_sys->guid ); /* used to identify client by server */
498 msg_Dbg( p_access,
499 "generated guid: "GUID_FMT,
500 GUID_PRINT( p_sys->guid ) );
501 p_sys->i_command_level = 1; /* updated after 0x1A command */
502 p_sys->i_seq_num = 0;
503 p_sys->i_media_packet_id_type = 0x04;
504 p_sys->i_header_packet_id_type = 0x02;
505 p_sys->i_proto = i_proto;
506 p_sys->i_packet_seq_num = 0;
507 p_sys->p_header = NULL;
508 p_sys->i_header = 0;
509 p_sys->p_media = NULL;
510 p_sys->i_media = 0;
511 p_sys->i_media_used = 0;
512
513 p_access->info.i_pos = 0;
514 p_sys->i_buffer_tcp = 0;
515 p_sys->i_buffer_udp = 0;
516 p_sys->p_cmd = NULL;
517 p_sys->i_cmd = 0;
518 p_access->info.b_eof = false;
519
520 /* *** send command 1 : connection request *** */
521 var_buffer_initwrite( &buffer, 0 );
522 var_buffer_add16( &buffer, 0x001c );
523 var_buffer_add16( &buffer, 0x0003 );
524 if( asprintf( &tmp,
525 "NSPlayer/7.0.0.1956; {"GUID_FMT"}; Host: %s",
526 GUID_PRINT( p_sys->guid ),
527 p_url->psz_host ) < 0 )
528 {
529 var_buffer_free( &buffer );
530 net_Close( p_sys->i_handle_tcp );
531 return VLC_ENOMEM;
532 }
533
534 var_buffer_addUTF16( p_access, &buffer, tmp );
535 free( tmp );
536
537 mms_CommandSend( p_access,
538 0x01, /* connexion request */
539 0x00000000, /* flags, FIXME */
540 0x0004000b, /* ???? */
541 buffer.p_data,
542 buffer.i_data );
543
544 if( mms_CommandRead( p_access, 0x01, 0 ) < 0 )
545 {
546 var_buffer_free( &buffer );
547 MMSClose( p_access );
548 return VLC_EGENERIC;
549 }
550
551 i_server_version = GetDWLE( p_sys->p_cmd + MMS_CMD_HEADERSIZE + 32 );
552 i_tool_version = GetDWLE( p_sys->p_cmd + MMS_CMD_HEADERSIZE + 36 );
553 i_update_player_url = GetDWLE( p_sys->p_cmd + MMS_CMD_HEADERSIZE + 40 );
554 i_encryption_type = GetDWLE( p_sys->p_cmd + MMS_CMD_HEADERSIZE + 44 );
555 p = (uint16_t*)( p_sys->p_cmd + MMS_CMD_HEADERSIZE + 48 );
556#define GETUTF16( psz, size ) \
557 { \
558 int i; \
559 psz = xmalloc( size + 1); \
560 for( i = 0; i < size; i++ ) \
561 { \
562 psz[i] = p[i]; \
563 } \
564 psz[size] = '\0'; \
565 p += ( size ); \
566 }
567 GETUTF16( p_sys->psz_server_version, i_server_version );
568 GETUTF16( p_sys->psz_tool_version, i_tool_version );
569 GETUTF16( p_sys->psz_update_player_url, i_update_player_url );
570 GETUTF16( p_sys->psz_encryption_type, i_encryption_type );
571#undef GETUTF16
572 msg_Dbg( p_access,
573 "0x01 --> server_version:\"%s\" tool_version:\"%s\" update_player_url:\"%s\" encryption_type:\"%s\"",
574 p_sys->psz_server_version,
575 p_sys->psz_tool_version,
576 p_sys->psz_update_player_url,
577 p_sys->psz_encryption_type );
578
579 /* *** should make an 18 command to make data timing *** */
580
581 /* *** send command 2 : transport protocol selection *** */
582 var_buffer_reinitwrite( &buffer, 0 );
583 var_buffer_add32( &buffer, 0x00000000 );
584 var_buffer_add32( &buffer, 0x000a0000 );
585 var_buffer_add32( &buffer, 0x00000002 );
586 if( b_udp )
587 {
588 if( asprintf( &tmp,
589 "\\\\%s\\UDP\\%d",
590 p_sys->sz_bind_addr,
591 7000 ) < 0) // FIXME
592 {
593 var_buffer_free( &buffer );
594 MMSClose( p_access );
595 return VLC_EGENERIC;
596 }
597 }
598 else
599 {
600 if( asprintf( &tmp, "\\\\192.168.0.1\\TCP\\1242" ) < 0 )
601 {
602 var_buffer_free( &buffer );
603 MMSClose( p_access );
604 return VLC_EGENERIC;
605 }
606 }
607 var_buffer_addUTF16( p_access, &buffer, tmp );
608 var_buffer_add16( &buffer, '0' );
609 free( tmp );
610
611 mms_CommandSend( p_access,
612 0x02, /* connexion request */
613 0x00000000, /* flags, FIXME */
614 0xffffffff, /* ???? */
615 buffer.p_data,
616 buffer.i_data );
617
618 /* *** response from server, should be 0x02 or 0x03 *** */
619 mms_CommandRead( p_access, 0x02, 0x03 );
620 if( p_sys->i_command == 0x03 )
621 {
622 msg_Err( p_access,
623 "%s protocol selection failed", b_udp ? "UDP" : "TCP" );
624 var_buffer_free( &buffer );
625 MMSClose( p_access );
626 return VLC_EGENERIC;
627 }
628 else if( p_sys->i_command != 0x02 )
629 {
630 msg_Warn( p_access, "received command isn't 0x02 in reponse to 0x02" );
631 }
632
633 /* *** send command 5 : media file name/path requested *** */
634 var_buffer_reinitwrite( &buffer, 0 );
635 var_buffer_add64( &buffer, 0 );
636
637 /* media file path shouldn't start with / character */
638 mediapath = p_url->psz_path;
639 if ( mediapath && *mediapath == '/' )
640 {
641 mediapath++;
642 }
643 var_buffer_addUTF16( p_access, &buffer, mediapath );
644
645 mms_CommandSend( p_access,
646 0x05,
647 p_sys->i_command_level,
648 0xffffffff,
649 buffer.p_data,
650 buffer.i_data );
651
652 /* *** wait for reponse *** */
653 mms_CommandRead( p_access, 0x1a, 0x06 );
654
655 /* test if server send 0x1A answer */
656 if( p_sys->i_command == 0x1A )
657 {
658 msg_Err( p_access, "id/password requested (not yet supported)" );
659 /* FIXME */
660 var_buffer_free( &buffer );
661 MMSClose( p_access );
662 return VLC_EGENERIC;
663 }
664 if( p_sys->i_command != 0x06 )
665 {
666 msg_Err( p_access,
667 "unknown answer (0x%x instead of 0x06)",
668 p_sys->i_command );
669 var_buffer_free( &buffer );
670 MMSClose( p_access );
671 return( -1 );
672 }
673
674 /* 1 for file ok, 2 for authen ok */
675 switch( GetDWLE( p_sys->p_cmd + MMS_CMD_HEADERSIZE ) )
676 {
677 case 0x0001:
678 msg_Dbg( p_access, "media file name/path accepted" );
679 break;
680 case 0x0002:
681 msg_Dbg( p_access, "authentication accepted" );
682 break;
683 case -1:
684 default:
685 msg_Err( p_access, "error while asking for file %d",
686 GetDWLE( p_sys->p_cmd + MMS_CMD_HEADERSIZE ) );
687 var_buffer_free( &buffer );
688 MMSClose( p_access );
689 return VLC_EGENERIC;
690 }
691
692 p_sys->i_flags_broadcast =
693 GetDWLE( p_sys->p_cmd + MMS_CMD_HEADERSIZE + 12 );
694 p_sys->i_media_length =
695 GetDWLE( p_sys->p_cmd + MMS_CMD_HEADERSIZE + 24 );
696 p_sys->i_packet_length =
697 GetDWLE( p_sys->p_cmd + MMS_CMD_HEADERSIZE + 44 );
698 p_sys->i_packet_count =
699 GetDWLE( p_sys->p_cmd + MMS_CMD_HEADERSIZE + 48 );
700 p_sys->i_max_bit_rate =
701 GetDWLE( p_sys->p_cmd + MMS_CMD_HEADERSIZE + 56 );
702 p_sys->i_header_size =
703 GetDWLE( p_sys->p_cmd + MMS_CMD_HEADERSIZE + 60 );
704
705 msg_Dbg( p_access,
706 "answer 0x06 flags:0x%8.8"PRIx32" media_length:%"PRIu32"s "
707 "packet_length:%zu packet_count:%"PRIu32" max_bit_rate:%d "
708 "header_size:%zu",
709 p_sys->i_flags_broadcast,
710 p_sys->i_media_length,
711 p_sys->i_packet_length,
712 p_sys->i_packet_count,
713 p_sys->i_max_bit_rate,
714 p_sys->i_header_size );
715
716 /* *** send command 15 *** */
717
718 var_buffer_reinitwrite( &buffer, 0 );
719 var_buffer_add32( &buffer, 0 );
720 var_buffer_add32( &buffer, 0x8000 );
721 var_buffer_add32( &buffer, 0xffffffff );
722 var_buffer_add32( &buffer, 0x00 );
723 var_buffer_add32( &buffer, 0x00 );
724 var_buffer_add32( &buffer, 0x00 );
725 var_buffer_add64( &buffer, (((uint64_t)0x40ac2000)<<32) );
726 var_buffer_add32( &buffer, p_sys->i_header_packet_id_type );
727 var_buffer_add32( &buffer, 0x00 );
728 mms_CommandSend( p_access, 0x15, p_sys->i_command_level, 0x00,
729 buffer.p_data, buffer.i_data );
730
731 /* *** wait for reponse *** */
732 /* Commented out because it fails on some stream (no 0x11 answer) */
733#if 0
734 mms_CommandRead( p_access, 0x11, 0 );
735
736 if( p_sys->i_command != 0x11 )
737 {
738 msg_Err( p_access,
739 "unknown answer (0x%x instead of 0x11)",
740 p_sys->i_command );
741 var_buffer_free( &buffer );
742 MMSClose( p_access );
743 return( -1 );
744 }
745#endif
746
747 /* *** now read header packet *** */
748 /* XXX could be split over multiples packets */
749 msg_Dbg( p_access, "reading header" );
750 for( ;; )
751 {
752 if( mms_HeaderMediaRead( p_access, MMS_PACKET_HEADER ) < 0 )
753 {
754 msg_Err( p_access, "cannot receive header" );
755 var_buffer_free( &buffer );
756 MMSClose( p_access );
757 return VLC_EGENERIC;
758 }
759 if( p_sys->i_header >= p_sys->i_header_size )
760 {
761 msg_Dbg( p_access,
762 "header complete(%zu)",
763 p_sys->i_header );
764 break;
765 }
766 msg_Dbg( p_access,
767 "header incomplete (%zu/%zu), reading more",
768 p_sys->i_header,
769 p_sys->i_header_size );
770 }
771
772 /* *** parse header and get stream and their id *** */
773 /* get all streams properties,
774 *
775 * TODO : stream bitrates properties(optional)
776 * and bitrate mutual exclusion(optional) */
777 asf_HeaderParse ( &p_sys->asfh,
778 p_sys->p_header, p_sys->i_header );
779 asf_StreamSelect( &p_sys->asfh,
780 var_InheritInteger( p_access, "mms-maxbitrate" ),
781 var_InheritBool( p_access, "mms-all" ),
782 var_InheritBool( p_access, "audio" ),
783 var_InheritBool( p_access, "video" ) );
784
785 /* *** now select stream we want to receive *** */
786 /* TODO take care of stream bitrate TODO */
787 i_streams = 0;
788 i_first = -1;
789 var_buffer_reinitwrite( &buffer, 0 );
790 /* for now, select first audio and video stream */
791 for( i = 1; i < 128; i++ )
792 {
793
794 if( p_sys->asfh.stream[i].i_cat != ASF_CODEC_TYPE_UNKNOWN )
795 {
796 i_streams++;
797 if( i_first != -1 )
798 {
799 var_buffer_add16( &buffer, 0xffff );
800 var_buffer_add16( &buffer, i );
801 }
802 else
803 {
804 i_first = i;
805 }
806 if( p_sys->asfh.stream[i].i_selected )
807 {
808 var_buffer_add16( &buffer, 0x0000 );
809 msg_Info( p_access,
810 "selecting stream[0x%x] %s (%d Kib/s)",
811 i,
812 ( p_sys->asfh.stream[i].i_cat == ASF_CODEC_TYPE_AUDIO ) ?
813 "audio" : "video" ,
814 p_sys->asfh.stream[i].i_bitrate / 1024);
815 }
816 else
817 {
818 var_buffer_add16( &buffer, 0x0002 );
819 msg_Info( p_access,
820 "ignoring stream[0x%x] %s (%d Kib/s)",
821 i,
822 ( p_sys->asfh.stream[i].i_cat == ASF_CODEC_TYPE_AUDIO ) ?
823 "audio" : "video" ,
824 p_sys->asfh.stream[i].i_bitrate / 1024);
825
826 }
827 }
828 }
829
830 if( i_streams == 0 )
831 {
832 msg_Err( p_access, "cannot find any stream" );
833 var_buffer_free( &buffer );
834 MMSClose( p_access );
835 return VLC_EGENERIC;
836 }
837 mms_CommandSend( p_access, 0x33,
838 i_streams,
839 0xffff | ( i_first << 16 ),
840 buffer.p_data, buffer.i_data );
841
842 mms_CommandRead( p_access, 0x21, 0 );
843 if( p_sys->i_command != 0x21 )
844 {
845 msg_Err( p_access,
846 "unknown answer (0x%x instead of 0x21)",
847 p_sys->i_command );
848 var_buffer_free( &buffer );
849 MMSClose( p_access );
850 return VLC_EGENERIC;
851 }
852
853
854 var_buffer_free( &buffer );
855
856 msg_Info( p_access, "connection successful" );
857
858 return VLC_SUCCESS;
859}

890 "unknown answer (0x%x instead of 0x05)",

893 }

895 {
896 /* get a packet */

901 }

953/****************************************************************************
954 *
955 * MMS specific functions
956 *
957 ****************************************************************************/
958static int mms_CommandSend( access_t *p_access, int i_command,
959 uint32_t i_prefix1, uint32_t i_prefix2,
960 uint8_t *p_data, int i_data_old )
961{
962 var_buffer_t buffer;
963 access_sys_t *p_sys = p_access->p_sys;
964 int i_data_by8, i_ret;
965 int i_data = i_data_old;
966
967 while( i_data & 0x7 ) i_data++;
968 i_data_by8 = i_data >> 3;
969
970 /* first init buffer */
971 var_buffer_initwrite( &buffer, 0 );
972
973 var_buffer_add32( &buffer, 0x00000001 ); /* start sequence */
974 var_buffer_add32( &buffer, 0xB00BFACE );
975 /* size after protocol type */
976 var_buffer_add32( &buffer, i_data + MMS_CMD_HEADERSIZE - 16 );
977 var_buffer_add32( &buffer, 0x20534d4d ); /* protocol "MMS " */
978 var_buffer_add32( &buffer, i_data_by8 + 4 );
979 var_buffer_add32( &buffer, p_sys->i_seq_num ); p_sys->i_seq_num++;
980 var_buffer_add64( &buffer, 0 );
981 var_buffer_add32( &buffer, i_data_by8 + 2 );
982 var_buffer_add32( &buffer, 0x00030000 | i_command ); /* dir | command */
983 var_buffer_add32( &buffer, i_prefix1 ); /* command specific */
984 var_buffer_add32( &buffer, i_prefix2 ); /* command specific */
985
986 /* specific command data */
987 if( p_data && i_data > 0 )
988 {
989 var_buffer_addmemory( &buffer, p_data, i_data_old );
990 }
991
992 /* Append padding to the command data */
993 var_buffer_add64( &buffer, 0 );
994
995 /* send it */
996 vlc_mutex_lock( &p_sys->lock_netwrite );
997 i_ret = net_Write( p_access, p_sys->i_handle_tcp, NULL, buffer.p_data,
998 buffer.i_data - ( 8 - ( i_data - i_data_old ) ) );
999 vlc_mutex_unlock( &p_sys->lock_netwrite );
1000
1001 if( i_ret != buffer.i_data - ( 8 - ( i_data - i_data_old ) ) )
1002 {
1003 var_buffer_free( &buffer );
1004 msg_Err( p_access, "failed to send command" );
1005 return VLC_EGENERIC;
1006 }
1007
1008 var_buffer_free( &buffer );
1009 return VLC_SUCCESS;
1010}

1012static int NetFillBuffer( access_t *p_access )
1013{
1014 access_sys_t *p_sys = p_access->p_sys;
1015 int i_ret;
1016 struct pollfd ufd[2];
1017 unsigned timeout, nfd;
1018
1019 /* FIXME when using udp */
1020 ssize_t i_tcp, i_udp;
1021 ssize_t i_tcp_read, i_udp_read;
1022 int i_try = 0;
1023
1024 i_tcp = MMS_BUFFER_SIZE/2 - p_sys->i_buffer_tcp;
1025
1026 if( p_sys->i_proto == MMS_PROTO_UDP )
1027 {
1028 i_udp = MMS_BUFFER_SIZE/2 - p_sys->i_buffer_udp;
1029 }
1030 else
1031 {
1032 i_udp = 0; /* there isn't udp socket */
1033 }
1034
1035 if( ( i_udp <= 0 ) && ( i_tcp <= 0 ) )
1036 {
1037 msg_Warn( p_access, "nothing to read %d:%d", (int)i_tcp, (int)i_udp );
1038 return 0;
1039 }
1040 else
1041 {
1042 /* msg_Warn( p_access, "ask for tcp:%d udp:%d", i_tcp, i_udp ); */
1043 }
1044
1045 /* Find if some data is available */
1046 do
1047 {
1048 i_try++;
1049
1050 /* Initialize file descriptor set */
1051 memset (ufd, 0, sizeof (ufd));
1052 nfd = 0;
1053
1054 if( i_tcp > 0 )
1055 {
1056 ufd[nfd].fd = p_sys->i_handle_tcp;
1057 ufd[nfd].events = POLLIN;
1058 nfd++;
1059 }
1060 if( i_udp > 0 )
1061 {
1062 ufd[nfd].fd = p_sys->i_handle_udp;
1063 ufd[nfd].events = POLLIN;
1064 nfd++;
1065 }
1066
1067 /* We'll wait 0.5 second if nothing happens */
1068 timeout = __MIN( 500, p_sys->i_timeout );
1069
1070 if( i_try * timeout > p_sys->i_timeout )
1071 {
1072 msg_Err(p_access, "no data received");
1073 return -1;
1074 }
1075
1076 if( i_try > 3 && (p_sys->i_buffer_tcp > 0 || p_sys->i_buffer_udp > 0) )
1077 {
1078 return -1;
1079 }
1080
1081 if( !vlc_object_alive (p_access) )
1082 return -1;
1083
1084 //msg_Dbg( p_access, "NetFillBuffer: trying again (select)" );
1085
1086 } while( !(i_ret = poll( ufd, nfd, timeout)) ||
1087 (i_ret < 0 && errno == EINTR) );
1088
1089 if( i_ret < 0 )
1090 {
1091 msg_Err( p_access, "network poll error: %s", vlc_strerror_c(errno) );
1092 return -1;
1093 }
1094
1095 i_tcp_read = i_udp_read = 0;
1096
1097 if( ( i_tcp > 0 ) && ufd[0].revents )
1098 {
1099 i_tcp_read =
1100 recv( p_sys->i_handle_tcp,
1101 p_sys->buffer_tcp + p_sys->i_buffer_tcp,
1102 i_tcp + MMS_BUFFER_SIZE/2, 0 );
1103 }
1104
1105 if( i_udp > 0 && ufd[i_tcp > 0].revents )
1106 {
1107 i_udp_read = recv( p_sys->i_handle_udp,
1108 p_sys->buffer_udp + p_sys->i_buffer_udp,
1109 i_udp + MMS_BUFFER_SIZE/2, 0 );
1110 }
1111
1112#ifdef MMS_DEBUG
1113 if( p_sys->i_proto == MMS_PROTO_UDP )
1114 {
1115 msg_Dbg( p_access, "filling buffer TCP:%d+%d UDP:%d+%d",
1116 p_sys->i_buffer_tcp, i_tcp_read,
1117 p_sys->i_buffer_udp, i_udp_read );
1118 }
1119 else
1120 {
1121 msg_Dbg( p_access, "filling buffer TCP:%d+%d",
1122 p_sys->i_buffer_tcp, i_tcp_read );
1123 }
1124#endif
1125
1126 if( i_tcp_read > 0 ) p_sys->i_buffer_tcp += i_tcp_read;
1127 if( i_udp_read > 0 ) p_sys->i_buffer_udp += i_udp_read;
1128
1129 return i_tcp_read + i_udp_read;
1130}

1132static int mms_ParseCommand( access_t *p_access,
1133 uint8_t *p_data,
1134 size_t i_data,
1135 int *pi_used )
1136{
1137 #define GET32( i_pos ) \
1138 ( p_sys->p_cmd[i_pos] + ( p_sys->p_cmd[i_pos +1] << 8 ) + \
1139 ( p_sys->p_cmd[i_pos + 2] << 16 ) + \
1140 ( p_sys->p_cmd[i_pos + 3] << 24 ) )
1141
1142 access_sys_t *p_sys = p_access->p_sys;
1143 uint32_t i_length;
1144 uint32_t i_id;
1145
1146 free( p_sys->p_cmd );
1147 p_sys->i_cmd = i_data;
1148 p_sys->p_cmd = xmalloc( i_data );
1149 memcpy( p_sys->p_cmd, p_data, i_data );
1150
1151 *pi_used = i_data; /* by default */
1152
1153 if( i_data < MMS_CMD_HEADERSIZE )
1154 {
1155 msg_Warn( p_access, "truncated command (header incomplete)" );
1156 p_sys->i_command = 0;
1157 return -1;
1158 }
1159 i_id = GetDWLE( p_data + 4 );
1160 i_length = GetDWLE( p_data + 8 ) + 16;
1161
1162 if( i_id != 0xb00bface || i_length < 16 )
1163 {
1164 msg_Err( p_access,
1165 "incorrect command header (0x%"PRIx32")", i_id );
1166 p_sys->i_command = 0;
1167 return -1;
1168 }
1169
1170 if( i_length > p_sys->i_cmd )
1171 {
1172 msg_Warn( p_access,
1173 "truncated command (missing %zu bytes)",
1174 (size_t)i_length - i_data );
1175 p_sys->i_command = 0;
1176 return -1;
1177 }
1178 else if( i_length < p_sys->i_cmd )
1179 {
1180 p_sys->i_cmd = i_length;
1181 *pi_used = i_length;
1182 }
1183
1184 msg_Dbg( p_access,
1185 "recv command start_sequence:0x%8.8x command_id:0x%8.8x length:%d len8:%d sequence 0x%8.8x len8_II:%d dir_comm:0x%8.8x",
1186 GET32( 0 ),
1187 GET32( 4 ),
1188 GET32( 8 ),
1189 /* 12: protocol type "MMS " */
1190 GET32( 16 ),
1191 GET32( 20 ),
1192 /* 24: unknown (0) */
1193 /* 28: unknown (0) */
1194 GET32( 32 ),
1195 GET32( 36 )
1196 /* 40: switches */
1197 /* 44: extra */ );
1198
1199 p_sys->i_command = GET32( 36 ) & 0xffff;
1200#undef GET32
1201
1202 return MMS_PACKET_CMD;
1203}

1205static int mms_ParsePacket( access_t *p_access,
1206 uint8_t *p_data, size_t i_data,
1207 int *pi_used )
1208{
1209 access_sys_t *p_sys = p_access->p_sys;
1210 int i_packet_seq_num;
1211 size_t i_packet_length;
1212 uint32_t i_packet_id;
1213
1214 *pi_used = i_data; /* default */
1215 if( i_data <= 8 )
1216 {
1217 msg_Warn( p_access, "truncated packet (header incomplete)" );
1218 return -1;
1219 }
1220
1221 i_packet_id = p_data[4];
1222 i_packet_seq_num = GetDWLE( p_data );
1223 i_packet_length = GetWLE( p_data + 6 );
1224
1225 //msg_Warn( p_access, "------->i_packet_length=%d, i_data=%d", i_packet_length, i_data );
1226
1227 if( i_packet_length > i_data || i_packet_length <= 8)
1228 {
1229 /* msg_Dbg( p_access,
1230 "truncated packet (Declared %d bytes, Actual %d bytes)",
1231 i_packet_length, i_data ); */
1232 *pi_used = 0;
1233 return -1;
1234 }
1235 else if( i_packet_length < i_data )
1236 {
1237 *pi_used = i_packet_length;
1238 }
1239
1240 if( i_packet_id == 0xff )
1241 {
1242 msg_Warn( p_access,
1243 "receive MMS UDP pair timing" );
1244 return( MMS_PACKET_UDP_TIMING );
1245 }
1246
1247 if( i_packet_id != p_sys->i_header_packet_id_type &&
1248 i_packet_id != p_sys->i_media_packet_id_type )
1249 {
1250 msg_Warn( p_access, "incorrect Packet Id Type (0x%x)", i_packet_id );
1251 return -1;
1252 }
1253
1254 /* we now have a media or a header packet */
1255 if( i_packet_seq_num != p_sys->i_packet_seq_num )
1256 {
1257#if 0
1258 /* FIXME for udp could be just wrong order ? */
1259 msg_Warn( p_access,
1260 "detected packet lost (%d != %d)",
1261 i_packet_seq_num,
1262 p_sys->i_packet_seq_num );
1263#endif
1264 }
1265 p_sys->i_packet_seq_num = i_packet_seq_num + 1;
1266
1267 if( i_packet_id == p_sys->i_header_packet_id_type )
1268 {
1269 if( p_sys->p_header )
1270 {
1271 p_sys->p_header = xrealloc( p_sys->p_header,
1272 p_sys->i_header + i_packet_length - 8 );
1273 memcpy( &p_sys->p_header[p_sys->i_header],
1274 p_data + 8, i_packet_length - 8 );
1275 p_sys->i_header += i_packet_length - 8;
1276
1277 }
1278 else
1279 {
1280 uint8_t* p_packet = xmalloc( i_packet_length - 8 ); // don't bother with preheader
1281 memcpy( p_packet, p_data + 8, i_packet_length - 8 );
1282 p_sys->p_header = p_packet;
1283 p_sys->i_header = i_packet_length - 8;
1284 }
1285/* msg_Dbg( p_access,
1286 "receive header packet (%d bytes)",
1287 i_packet_length - 8 ); */
1288
1289 return MMS_PACKET_HEADER;
1290 }
1291 else
1292 {
1293 uint8_t* p_packet = xmalloc( i_packet_length - 8 ); // don't bother with preheader
1294 memcpy( p_packet, p_data + 8, i_packet_length - 8 );
1295 FREENULL( p_sys->p_media );
1296 p_sys->p_media = p_packet;
1297 p_sys->i_media = i_packet_length - 8;
1298 p_sys->i_media_used = 0;
1299/* msg_Dbg( p_access,
1300 "receive media packet (%d bytes)",
1301 i_packet_length - 8 ); */
1302
1303 return MMS_PACKET_MEDIA;
1304 }

1307static int mms_ReceivePacket( access_t *p_access )
1308{
1309 access_sys_t *p_sys = p_access->p_sys;
1310 int i_packet_tcp_type;
1311 int i_packet_udp_type;
1312
1313 for( ;; )
1314 {
1315 bool b_refill = true;
1316
1317 /* first if we need to refill buffer */
1318 if( p_sys->i_buffer_tcp >= MMS_CMD_HEADERSIZE )
1319 {
1320 if( GetDWLE( p_sys->buffer_tcp + 4 ) == 0xb00bface )
1321 {
1322 if( GetDWLE( p_sys->buffer_tcp + 8 ) + 16 <=
1323 (uint32_t)p_sys->i_buffer_tcp )
1324 {
1325 b_refill = false;
1326 }
1327 }
1328 else if( GetWLE( p_sys->buffer_tcp + 6 ) <= p_sys->i_buffer_tcp )
1329 {
1330 b_refill = false;
1331 }
1332 }
1333 if( p_sys->i_proto == MMS_PROTO_UDP && p_sys->i_buffer_udp >= 8 &&
1334 GetWLE( p_sys->buffer_udp + 6 ) <= p_sys->i_buffer_udp )
1335 {
1336 b_refill = false;
1337 }
1338
1339 if( b_refill && NetFillBuffer( p_access ) < 0 )
1340 {
1341 msg_Warn( p_access, "cannot fill buffer" );
1342 return -1;
1343 }
1344
1345 i_packet_tcp_type = -1;
1346 i_packet_udp_type = -1;
1347
1348 if( p_sys->i_buffer_tcp > 0 )
1349 {
1350 int i_used;
1351
1352 if( GetDWLE( p_sys->buffer_tcp + 4 ) == 0xb00bface )
1353 {
1354 i_packet_tcp_type =
1355 mms_ParseCommand( p_access, p_sys->buffer_tcp,
1356 p_sys->i_buffer_tcp, &i_used );
1357
1358 }
1359 else
1360 {
1361 i_packet_tcp_type =
1362 mms_ParsePacket( p_access, p_sys->buffer_tcp,
1363 p_sys->i_buffer_tcp, &i_used );
1364 }
1365 if( i_used > 0 && i_used < MMS_BUFFER_SIZE )
1366 {
1367 memmove( p_sys->buffer_tcp, p_sys->buffer_tcp + i_used,
1368 MMS_BUFFER_SIZE - i_used );
1369 }
1370 p_sys->i_buffer_tcp -= i_used;
1371 }
1372 else if( p_sys->i_buffer_udp > 0 )
1373 {
1374 int i_used;
1375
1376 i_packet_udp_type =
1377 mms_ParsePacket( p_access, p_sys->buffer_udp,
1378 p_sys->i_buffer_udp, &i_used );
1379
1380 if( i_used > 0 && i_used < MMS_BUFFER_SIZE )
1381 {
1382 memmove( p_sys->buffer_udp, p_sys->buffer_udp + i_used,
1383 MMS_BUFFER_SIZE - i_used );
1384 }
1385 p_sys->i_buffer_udp -= i_used;
1386 }
1387
1388 if( i_packet_tcp_type == MMS_PACKET_CMD && p_sys->i_command == 0x1b )
1389 {
1390 mms_CommandSend( p_access, 0x1b, 0, 0, NULL, 0 );
1391 i_packet_tcp_type = -1;
1392 }
1393
1394 if( i_packet_tcp_type != -1 )
1395 {
1396 return i_packet_tcp_type;
1397 }
1398 else if( i_packet_udp_type != -1 )
1399 {
1400 return i_packet_udp_type;
1401 }
1402 }
1403}

1405static int mms_ReceiveCommand( access_t *p_access )
1406{
1407 access_sys_t *p_sys = p_access->p_sys;
1408
1409 for( ;; )
1410 {
1411 int i_used;
1412 int i_status;
1413
1414 if( NetFillBuffer( p_access ) < 0 )
1415 {
1416 msg_Warn( p_access, "cannot fill buffer" );
1417 return VLC_EGENERIC;
1418 }
1419 if( p_sys->i_buffer_tcp > 0 )
1420 {
1421 i_status = mms_ParseCommand( p_access, p_sys->buffer_tcp,
1422 p_sys->i_buffer_tcp, &i_used );
1423 if( i_used < MMS_BUFFER_SIZE )
1424 {
1425 memmove( p_sys->buffer_tcp, p_sys->buffer_tcp + i_used,
1426 MMS_BUFFER_SIZE - i_used );
1427 }
1428 p_sys->i_buffer_tcp -= i_used;
1429
1430 if( i_status < 0 )
1431 {
1432 return VLC_EGENERIC;
1433 }
1434
1435 if( p_sys->i_command == 0x1b )
1436 {
1437 mms_CommandSend( p_access, 0x1b, 0, 0, NULL, 0 );
1438 }
1439 else
1440 {
1441 break;
1442 }
1443 }
1444 else
1445 {
1446 return VLC_EGENERIC;
1447 }
1448 }
1449
1450 return VLC_SUCCESS;
1451}

1453#define MMS_RETRY_MAX 10
1454#define MMS_RETRY_SLEEP 50000
1455
1456static int mms_CommandRead( access_t *p_access, int i_command1,
1457 int i_command2 )
1458{
1459 access_sys_t *p_sys = p_access->p_sys;
1460 int i_count;
1461 int i_status;
1462
1463 for( i_count = 0; i_count < MMS_RETRY_MAX; )
1464 {
1465 i_status = mms_ReceiveCommand( p_access );
1466 if( i_status < 0 || p_sys->i_command == 0 )
1467 {
1468 i_count++;
1469 msleep( MMS_RETRY_SLEEP );
1470 }
1471 else if( i_command1 == 0 && i_command2 == 0)
1472 {
1473 return VLC_SUCCESS;
1474 }
1475 else if( p_sys->i_command == i_command1 ||
1476 p_sys->i_command == i_command2 )
1477 {
1478 return VLC_SUCCESS;
1479 }
1480 else
1481 {
1482 switch( p_sys->i_command )
1483 {
1484 case 0x03:
1485 msg_Warn( p_access, "socket closed by server" );
1486 p_access->info.b_eof = true;
1487 return VLC_EGENERIC;
1488 case 0x1e:
1489 msg_Warn( p_access, "end of media stream" );
1490 p_access->info.b_eof = true;
1491 return VLC_EGENERIC;
1492 default:
1493 break;
1494 }
1495 }
1496 }
1497 p_access->info.b_eof = true;
1498 msg_Warn( p_access, "failed to receive command (aborting)" );
1499
1500 return VLC_EGENERIC;
1501}

1504static int mms_HeaderMediaRead( access_t *p_access, int i_type )
1505{
1506 access_sys_t *p_sys = p_access->p_sys;
1507 int i_count;
1508
1509 for( i_count = 0; i_count < MMS_RETRY_MAX; )
1510 {
1511 int i_status;
1512
1513 if( !vlc_object_alive (p_access) )
1514 return -1;
1515
1516 i_status = mms_ReceivePacket( p_access );
1517 if( i_status < 0 )
1518 {
1519 i_count++;
1520 msg_Warn( p_access, "cannot receive header (%d/%d)",
1521 i_count, MMS_RETRY_MAX );
1522 msleep( MMS_RETRY_SLEEP );
1523 }
1524 else if( i_status == i_type || i_type == MMS_PACKET_ANY )
1525 {
1526 return i_type;
1527 }
1528 else if( i_status == MMS_PACKET_CMD )
1529 {
1530 switch( p_sys->i_command )
1531 {
1532 case 0x03:
1533 msg_Warn( p_access, "socket closed by server" );
1534 p_access->info.b_eof = true;
1535 return -1;
1536 case 0x1e:
1537 msg_Warn( p_access, "end of media stream" );
1538 p_access->info.b_eof = true;
1539 return -1;
1540 case 0x20:
1541 /* XXX not too dificult to be done EXCEPT that we
1542 * need to restart demuxer... and I don't see how we
1543 * could do that :p */
1544 msg_Err( p_access,
1545 "reinitialization needed --> unsupported" );
1546 p_access->info.b_eof = true;
1547 return -1;
1548 default:
1549 break;
1550 }
1551 }
1552 }
1553
1554 msg_Err( p_access, "cannot receive %s (aborting)",
1555 ( i_type == MMS_PACKET_HEADER ) ? "header" : "media data" );
1556 p_access->info.b_eof = true;
1557 return -1;
1558}

1561static void *KeepAliveThread( void *p_data )
1562{
1563 access_t *p_access = p_data;
1564
1565 for( ;; )

1568 int canc = vlc_savecancel();
