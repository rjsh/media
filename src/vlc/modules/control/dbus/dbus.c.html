159static int Open( vlc_object_t *p_this )
160{
161 intf_thread_t *p_intf = (intf_thread_t*)p_this;
162
163 /* initialisation of the connection */
164 if( !dbus_threads_init_default() )
165 return VLC_EGENERIC;
166
167 intf_sys_t *p_sys = calloc( 1, sizeof( intf_sys_t ) );
168 if( unlikely(!p_sys) )
169 return VLC_ENOMEM;
170
171 playlist_t *p_playlist;
172 DBusConnection *p_conn;
173 p_sys->i_player_caps = PLAYER_CAPS_NONE;
174 p_sys->i_playing_state = PLAYBACK_STATE_INVALID;
175
176 if( vlc_pipe( p_sys->p_pipe_fds ) )
177 {
178 free( p_sys );
179 msg_Err( p_intf, "Could not create pipe" );
180 return VLC_EGENERIC;
181 }
182
183 DBusError error;
184 dbus_error_init( &error );
185
186 /* connect privately to the session bus
187 * the connection will not be shared with other vlc modules which use dbus,
188 * thus avoiding a whole class of concurrency issues */
189 p_conn = dbus_bus_get_private( DBUS_BUS_SESSION, &error );
190 if( !p_conn )
191 {
192 msg_Err( p_this, "Failed to connect to the D-Bus session daemon: %s",
193 error.message );
194 dbus_error_free( &error );
195 free( p_sys );
196 return VLC_EGENERIC;
197 }
198
199 dbus_connection_set_exit_on_disconnect( p_conn, FALSE );
200
201 /* register an instance-specific well known name of the form
202 * org.mpris.MediaPlayer2.vlc.instanceXXXX where XXXX is the
203 * current process's pid */
204 size_t i_length = sizeof( DBUS_MPRIS_BUS_NAME ) +
205 sizeof( DBUS_INSTANCE_ID_PREFIX ) + 10;
206
207 char unique_service[i_length];
208
209 snprintf( unique_service, sizeof (unique_service),
210 DBUS_MPRIS_BUS_NAME"."DBUS_INSTANCE_ID_PREFIX"%"PRIu32,
211 (uint32_t)getpid() );
212
213 dbus_bus_request_name( p_conn, unique_service, 0, &error );
214
215 if( dbus_error_is_set( &error ) )
216 {
217 msg_Err( p_this, "Error requesting service name %s: %s",
218 unique_service, error.message );
219 dbus_error_free( &error );
220 free( p_sys );
221 return VLC_EGENERIC;
222 }
223 msg_Dbg( p_intf, "listening on dbus as: %s", unique_service );
224
225 /* Try to register org.mpris.MediaPlayer2.vlc as well in case we are
226 * the only VLC instance currently connected to the bus */
227 dbus_bus_request_name( p_conn, DBUS_MPRIS_BUS_NAME, 0, NULL );
228
229 /* Register the entry point object path */
230 dbus_connection_register_object_path( p_conn, DBUS_MPRIS_OBJECT_PATH,
231 &dbus_mpris_vtable, p_this );
232
233 dbus_connection_flush( p_conn );
234
235 p_intf->p_sys = p_sys;
236 p_sys->p_conn = p_conn;
237 p_sys->p_events = vlc_array_new();
238 p_sys->p_timeouts = vlc_array_new();
239 p_sys->p_watches = vlc_array_new();
240 vlc_mutex_init( &p_sys->lock );
241
242 p_playlist = pl_Get( p_intf );
243 p_sys->p_playlist = p_playlist;
244
245 var_AddCallback( p_playlist, "activity", AllCallback, p_intf );
246 var_AddCallback( p_playlist, "intf-change", AllCallback, p_intf );
247 var_AddCallback( p_playlist, "volume", AllCallback, p_intf );
248 var_AddCallback( p_playlist, "mute", AllCallback, p_intf );
249 var_AddCallback( p_playlist, "playlist-item-append", AllCallback, p_intf );
250 var_AddCallback( p_playlist, "playlist-item-deleted", AllCallback, p_intf );
251 var_AddCallback( p_playlist, "random", AllCallback, p_intf );
252 var_AddCallback( p_playlist, "repeat", AllCallback, p_intf );
253 var_AddCallback( p_playlist, "loop", AllCallback, p_intf );
254 var_AddCallback( p_playlist, "fullscreen", AllCallback, p_intf );
255
256 if( !dbus_connection_set_timeout_functions( p_conn,
257 add_timeout,
258 remove_timeout,
259 timeout_toggled,
260 p_intf, NULL ) )
261 goto error;
262
263 if( !dbus_connection_set_watch_functions( p_conn,
264 add_watch,
265 remove_watch,
266 watch_toggled,
267 p_intf, NULL ) )
268 goto error;
269
270 if( vlc_clone( &p_sys->thread, Run, p_intf, VLC_THREAD_PRIORITY_LOW ) )
271 goto error;
272
273 return VLC_SUCCESS;
274
275error:
276 /* The dbus connection is private,
277 * so we are responsible for closing it
278 * XXX: Does this make sense when OOM ? */
279 dbus_connection_close( p_sys->p_conn );
280 dbus_connection_unref( p_conn );
281
282 vlc_array_destroy( p_sys->p_events );
283 vlc_array_destroy( p_sys->p_timeouts );
284 vlc_array_destroy( p_sys->p_watches );
285
286 vlc_mutex_destroy( &p_sys->lock );
287
288 free( p_sys );
289 return VLC_ENOMEM;
290}

292/*****************************************************************************
293 * Close: destroy interface

296static void Close ( vlc_object_t *p_this )
297{
298 intf_thread_t *p_intf = (intf_thread_t*) p_this;
299 intf_sys_t *p_sys = p_intf->p_sys;
300 playlist_t *p_playlist = p_sys->p_playlist;
301
302 vlc_cancel( p_sys->thread );
303 vlc_join( p_sys->thread, NULL );
304
305 var_DelCallback( p_playlist, "activity", AllCallback, p_intf );
306 var_DelCallback( p_playlist, "intf-change", AllCallback, p_intf );
307 var_DelCallback( p_playlist, "volume", AllCallback, p_intf );
308 var_DelCallback( p_playlist, "mute", AllCallback, p_intf );
309 var_DelCallback( p_playlist, "playlist-item-append", AllCallback, p_intf );
310 var_DelCallback( p_playlist, "playlist-item-deleted", AllCallback, p_intf );
311 var_DelCallback( p_playlist, "random", AllCallback, p_intf );
312 var_DelCallback( p_playlist, "repeat", AllCallback, p_intf );
313 var_DelCallback( p_playlist, "loop", AllCallback, p_intf );
314 var_DelCallback( p_playlist, "fullscreen", AllCallback, p_intf );
315
316 if( p_sys->p_input )
317 {
318 var_DelCallback( p_sys->p_input, "intf-event", InputCallback, p_intf );
319 var_DelCallback( p_sys->p_input, "can-pause", AllCallback, p_intf );
320 var_DelCallback( p_sys->p_input, "can-seek", AllCallback, p_intf );
321 vlc_object_release( p_sys->p_input );
322 }
323
324 /* The dbus connection is private, so we are responsible
325 * for closing it */
326 dbus_connection_close( p_sys->p_conn );
327 dbus_connection_unref( p_sys->p_conn );
328
329 // Free the events array
330 for( int i = 0; i < vlc_array_count( p_sys->p_events ); i++ )
331 {
332 callback_info_t* info = vlc_array_item_at_index( p_sys->p_events, i );
333 free( info );
334 }
335 vlc_mutex_destroy( &p_sys->lock );
336 vlc_array_destroy( p_sys->p_events );
337 vlc_array_destroy( p_sys->p_timeouts );
338 vlc_array_destroy( p_sys->p_watches );
339 free( p_sys );
340}

342static dbus_bool_t add_timeout( DBusTimeout *p_timeout, void *p_data )
343{
344 intf_thread_t *p_intf = (intf_thread_t*) p_data;
345 intf_sys_t *p_sys = (intf_sys_t*) p_intf->p_sys;
346
347 timeout_info_t *p_info = calloc( 1, sizeof( timeout_info_t ) );
348 p_info->i_remaining = dbus_timeout_get_interval( p_timeout ) * 1000;/* µs */
349 p_info->p_timeout = p_timeout;
350
351 dbus_timeout_set_data( p_timeout, p_info, free );
352
353 vlc_mutex_lock( &p_sys->lock );
354 vlc_array_append( p_sys->p_timeouts, p_timeout );
355 vlc_mutex_unlock( &p_sys->lock );
356
357 return TRUE;
358}

414/**
415 * GetPollFds() fills an array of pollfd data structures with :
416 * - the set of enabled dbus watches
417 * - the unix pipe which we use to manually wake up the main loop
418 *
419 * This function must be called with p_sys->lock locked
420 *
421 * @return The number of file descriptors
422 *
423 * @param intf_thread_t *p_intf this interface thread's state
424 * @param struct pollfd *p_fds a pointer to a pollfd array large enough to
425 * contain all the returned data (number of enabled dbus watches + 1)
426 */
427static int GetPollFds( intf_thread_t *p_intf, struct pollfd *p_fds )
428{
429 intf_sys_t *p_sys = p_intf->p_sys;
430 int i_fds = 1, i_watches = vlc_array_count( p_sys->p_watches );
431
432 p_fds[0].fd = p_sys->p_pipe_fds[PIPE_OUT];
433 p_fds[0].events = POLLIN | POLLPRI;
434
435 for( int i = 0; i < i_watches; i++ )
436 {
437 DBusWatch *p_watch = NULL;
438 p_watch = vlc_array_item_at_index( p_sys->p_watches, i );
439 if( !dbus_watch_get_enabled( p_watch ) )
440 continue;
441
442 p_fds[i_fds].fd = dbus_watch_get_unix_fd( p_watch );
443 int i_flags = dbus_watch_get_flags( p_watch );
444
445 if( i_flags & DBUS_WATCH_READABLE )
446 p_fds[i_fds].events |= POLLIN | POLLPRI;
447
448 if( i_flags & DBUS_WATCH_WRITABLE )
449 p_fds[i_fds].events |= POLLOUT;
450
451 i_fds++;
452 }
453
454 return i_fds;
455}

457/**
458 * UpdateTimeouts() updates the remaining time for each timeout and
459 * returns how much time is left until the next timeout.
460 *
461 * This function must be called with p_sys->lock locked
462 *
463 * @return int The time remaining until the next timeout, in milliseconds
464 * or -1 if there are no timeouts
465 *
466 * @param intf_thread_t *p_intf This interface thread's state
467 * @param mtime_t i_loop_interval The time which has elapsed since the last
468 * call to this function
469 */
470static int UpdateTimeouts( intf_thread_t *p_intf, mtime_t i_loop_interval )
471{
472 intf_sys_t *p_sys = p_intf->p_sys;
473 mtime_t i_next_timeout = LAST_MDATE;
474 unsigned int i_timeouts = vlc_array_count( p_sys->p_timeouts );
475
476 if( 0 == i_timeouts )
477 return -1;
478
479 for( unsigned int i = 0; i < i_timeouts; i++ )
480 {
481 timeout_info_t *p_info = NULL;
482 DBusTimeout *p_timeout = NULL;
483 mtime_t i_interval = 0;
484
485 p_timeout = vlc_array_item_at_index( p_sys->p_timeouts, i );
486 i_interval = dbus_timeout_get_interval( p_timeout ) * 1000; /* µs */
487 p_info = (timeout_info_t*) dbus_timeout_get_data( p_timeout );
488
489 p_info->i_remaining -= __MAX( 0, i_loop_interval ) % i_interval;
490
491 if( !dbus_timeout_get_enabled( p_timeout ) )
492 continue;
493
494 /* The correct poll timeout value is the shortest one
495 * in the dbus timeouts list */
496 i_next_timeout = __MIN( i_next_timeout,
497 __MAX( 0, p_info->i_remaining ) );
498 }
499
500 /* next timeout in milliseconds */
501 return i_next_timeout / 1000;
502}

504/**
505 * ProcessEvents() reacts to a list of events originating from other VLC threads
506 *
507 * This function must be called with p_sys->lock unlocked
508 *
509 * @param intf_thread_t *p_intf This interface thread state
510 * @param callback_info_t *p_events the list of events to process
511 */
512static void ProcessEvents( intf_thread_t *p_intf,
513 callback_info_t **p_events, int i_events )
514{
515 bool b_can_play = p_intf->p_sys->b_can_play;
516
517 vlc_dictionary_t player_properties, tracklist_properties, root_properties;
518 vlc_dictionary_init( &player_properties, 0 );
519 vlc_dictionary_init( &tracklist_properties, 0 );
520 vlc_dictionary_init( &root_properties, 0 );
521
522 for( int i = 0; i < i_events; i++ )
523 {
524 switch( p_events[i]->signal )
525 {
526 case SIGNAL_ITEM_CURRENT:
527 TrackChange( p_intf );
528
529 // rate depends on current item
530 if( !vlc_dictionary_has_key( &player_properties, "Rate" ) )
531 vlc_dictionary_insert( &player_properties, "Rate", NULL );
532
533 vlc_dictionary_insert( &player_properties, "Metadata", NULL );
534 break;
535 case SIGNAL_INTF_CHANGE:
536 case SIGNAL_PLAYLIST_ITEM_APPEND:
537 case SIGNAL_PLAYLIST_ITEM_DELETED:
538 {
539 playlist_t *p_playlist = p_intf->p_sys->p_playlist;
540 PL_LOCK;
541 b_can_play = playlist_CurrentSize( p_playlist ) > 0;
542 PL_UNLOCK;
543
544 if( b_can_play != p_intf->p_sys->b_can_play )
545 {
546 p_intf->p_sys->b_can_play = b_can_play;
547 vlc_dictionary_insert( &player_properties, "CanPlay", NULL );
548 }
549
550 if( !vlc_dictionary_has_key( &tracklist_properties, "Tracks" ) )
551 vlc_dictionary_insert( &tracklist_properties, "Tracks", NULL );
552 break;
553 }
554 case SIGNAL_VOLUME_MUTED:
555 case SIGNAL_VOLUME_CHANGE:
556 vlc_dictionary_insert( &player_properties, "Volume", NULL );
557 break;
558 case SIGNAL_RANDOM:
559 vlc_dictionary_insert( &player_properties, "Shuffle", NULL );
560 break;
561 case SIGNAL_FULLSCREEN:
562 vlc_dictionary_insert( &root_properties, "Fullscreen", NULL );
563 break;
564 case SIGNAL_REPEAT:
565 case SIGNAL_LOOP:
566 vlc_dictionary_insert( &player_properties, "LoopStatus", NULL );
567 break;
568 case SIGNAL_STATE:
569 vlc_dictionary_insert( &player_properties, "PlaybackStatus", NULL );
570 break;
571 case SIGNAL_RATE:
572 vlc_dictionary_insert( &player_properties, "Rate", NULL );
573 break;
574 case SIGNAL_INPUT_METADATA:
575 {
576 input_thread_t *p_input = pl_CurrentInput( p_intf );
577 input_item_t *p_item;
578 if( p_input )
579 {
580 p_item = input_GetItem( p_input );
581 vlc_object_release( p_input );
582
583 if( p_item )
584 vlc_dictionary_insert( &player_properties,
585 "Metadata", NULL );
586 }
587 break;
588 }
589 case SIGNAL_CAN_SEEK:
590 vlc_dictionary_insert( &player_properties, "CanSeek", NULL );
591 break;
592 case SIGNAL_CAN_PAUSE:
593 vlc_dictionary_insert( &player_properties, "CanPause", NULL );
594 break;
595 case SIGNAL_SEEK:
596 {
597 input_thread_t *p_input;
598 input_item_t *p_item;
599 p_input = pl_CurrentInput( p_intf );
600 if( p_input )
601 {
602 p_item = input_GetItem( p_input );
603 vlc_object_release( p_input );
604
605 if( p_item && ( p_item->i_id == p_events[i]->i_item ) )
606 SeekedEmit( p_intf );
607 }
608 break;
609 }
610 default:
611 assert(0);
612 }
613 free( p_events[i] );
614 }
615
616 if( vlc_dictionary_keys_count( &player_properties ) )
617 PlayerPropertiesChangedEmit( p_intf, &player_properties );
618
619 if( vlc_dictionary_keys_count( &tracklist_properties ) )
620 TrackListPropertiesChangedEmit( p_intf, &tracklist_properties );
621
622 if( vlc_dictionary_keys_count( &root_properties ) )
623 RootPropertiesChangedEmit( p_intf, &root_properties );
624
625 vlc_dictionary_clear( &player_properties, NULL, NULL );
626 vlc_dictionary_clear( &tracklist_properties, NULL, NULL );
627 vlc_dictionary_clear( &root_properties, NULL, NULL );
628}

630/**
631 * ProcessWatches() handles a list of dbus watches after poll() has returned
632 *
633 * This function must be called with p_sys->lock unlocked
634 *
635 * @param intf_thread_t *p_intf This interface thread state
636 * @param DBusWatch **p_watches The list of dbus watches to process
637 * @param int i_watches The size of the p_watches array
638 * @param struct pollfd *p_fds The result of a poll() call
639 * @param int i_fds The number of file descriptors processed by poll()
640 */
641static void ProcessWatches( intf_thread_t *p_intf,
642 DBusWatch **p_watches, int i_watches,
643 struct pollfd *p_fds, int i_fds )
644{
645 VLC_UNUSED(p_intf);
646
647 /* Process watches */
648 for( int i = 0; i < i_watches; i++ )
649 {
650 DBusWatch *p_watch = p_watches[i];
651 if( !dbus_watch_get_enabled( p_watch ) )
652 continue;
653
654 for( int j = 0; j < i_fds; j++ )
655 {
656 if( p_fds[j].fd != dbus_watch_get_unix_fd( p_watch ) )
657 continue;
658
659 int i_flags = 0;
660 int i_revents = p_fds[j].revents;
661
662 if( i_revents & POLLIN )
663 i_flags |= DBUS_WATCH_READABLE;
664
665 if( i_revents & POLLOUT )
666 i_flags |= DBUS_WATCH_WRITABLE;
667
668 if( i_revents & POLLERR )
669 i_flags |= DBUS_WATCH_ERROR;
670
671 if( i_revents & POLLHUP )
672 i_flags |= DBUS_WATCH_HANGUP;
673
674 if( i_flags )
675 dbus_watch_handle( p_watch, i_flags );
676 }
677 }
678}

680/**
681 * ProcessTimeouts() handles DBus timeouts
682 *
683 * This function must be called with p_sys->lock locked
684 *
685 * @param intf_thread_t *p_intf This interface thread state
686 * @param DBusTimeout **p_timeouts List of timeouts to process
687 * @param int i_timeouts Size of p_timeouts
688 */
689static void ProcessTimeouts( intf_thread_t *p_intf,
690 DBusTimeout **p_timeouts, int i_timeouts )
691{
692 VLC_UNUSED( p_intf );
693
694 for( int i = 0; i < i_timeouts; i++ )
695 {
696 timeout_info_t *p_info = NULL;
697
698 p_info = (timeout_info_t*) dbus_timeout_get_data( p_timeouts[i] );
699
700 if( !dbus_timeout_get_enabled( p_info->p_timeout ) )
701 continue;
702
703 if( p_info->i_remaining > 0 )
704 continue;
705
706 dbus_timeout_handle( p_info->p_timeout );
707 p_info->i_remaining = dbus_timeout_get_interval( p_info->p_timeout );
708 }
709}

711/**
712 * DispatchDBusMessages() dispatches incoming dbus messages
713 * (indirectly invoking the callbacks), then it sends outgoing
714 * messages which needs to be sent on the bus (method replies and signals)
715 *
716 * This function must be called with p_sys->lock unlocked
717 *
718 * @param intf_thread_t *p_intf This interface thread state
719 */
720static void DispatchDBusMessages( intf_thread_t *p_intf )
721{
722 DBusDispatchStatus status;
723 intf_sys_t *p_sys = p_intf->p_sys;
724
725 /* Dispatch incoming messages */
726 status = dbus_connection_get_dispatch_status( p_sys->p_conn );
727 while( status != DBUS_DISPATCH_COMPLETE )
728 {
729 dbus_connection_dispatch( p_sys->p_conn );
730 status = dbus_connection_get_dispatch_status( p_sys->p_conn );
731 }
732
733 /* Send outgoing data */
734 if( dbus_connection_has_messages_to_send( p_sys->p_conn ) )
735 dbus_connection_flush( p_sys->p_conn );
736}

738/**
739 * MPRISEntryPoint() routes incoming messages to their respective interface
740 * implementation.
741 *
742 * This function is called during dbus_connection_dispatch()
743 */
744static DBusHandlerResult
745MPRISEntryPoint ( DBusConnection *p_conn, DBusMessage *p_from, void *p_this )
746{
747 const char *psz_target_interface;
748 const char *psz_interface = dbus_message_get_interface( p_from );
749 const char *psz_method = dbus_message_get_member( p_from );
750
751 DBusError error;
752
753 if( psz_interface && strcmp( psz_interface, DBUS_INTERFACE_PROPERTIES ) )
754 psz_target_interface = psz_interface;
755
756 else
757 {
758 dbus_error_init( &error );
759 dbus_message_get_args( p_from, &error,
760 DBUS_TYPE_STRING, &psz_target_interface,
761 DBUS_TYPE_INVALID );
762
763 if( dbus_error_is_set( &error ) )
764 {
765 msg_Err( (vlc_object_t*) p_this, "D-Bus error on %s.%s: %s",
766 psz_interface, psz_method,
767 error.message );
768 dbus_error_free( &error );
769 return DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
770 }
771 }
772
773 if( !strcmp( psz_target_interface, DBUS_INTERFACE_INTROSPECTABLE ) )
774 return handle_introspect( p_conn, p_from, p_this );
775
776 if( !strcmp( psz_target_interface, DBUS_MPRIS_ROOT_INTERFACE ) )
777 return handle_root( p_conn, p_from, p_this );
778
779 if( !strcmp( psz_target_interface, DBUS_MPRIS_PLAYER_INTERFACE ) )
780 return handle_player( p_conn, p_from, p_this );
781
782 if( !strcmp( psz_target_interface, DBUS_MPRIS_TRACKLIST_INTERFACE ) )
783 return handle_tracklist( p_conn, p_from, p_this );
784
785 return DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
786}

788/*****************************************************************************
789 * Run: main loop
790 *****************************************************************************/
791
792static void *Run( void *data )
793{
794 intf_thread_t *p_intf = data;
795 intf_sys_t *p_sys = p_intf->p_sys;
796 mtime_t i_last_run = mdate();
797
798 for( ;; )
799 {
800 int canc = vlc_savecancel();
801 vlc_mutex_lock( &p_sys->lock );
802
803 int i_watches = vlc_array_count( p_sys->p_watches );
804 struct pollfd fds[i_watches];
805 memset(fds, 0, sizeof fds);
806
807 int i_fds = GetPollFds( p_intf, fds );
808
809 mtime_t i_now = mdate(), i_loop_interval = i_now - i_last_run;
810
811 int i_next_timeout = UpdateTimeouts( p_intf, i_loop_interval );
812 i_last_run = i_now;
813
814 vlc_mutex_unlock( &p_sys->lock );
815
816 /* thread cancellation is allowed while the main loop sleeps */
817 vlc_restorecancel( canc );
818
819 int i_pollres = poll( fds, i_fds, i_next_timeout );
820
821 canc = vlc_savecancel();
822
823 if( -1 == i_pollres )
824 { /* XXX: What should we do when poll() fails ? */
825 msg_Err( p_intf, "poll() failed: %s", vlc_strerror_c(errno) );
826 vlc_restorecancel( canc );
827 continue;
828 }
829
830 /* Was the main loop woken up manually ? */
831 if( 0 < i_pollres && ( fds[0].revents & POLLIN ) )
832 {
833 char buf;
834 (void)read( fds[0].fd, &buf, 1 );
835 }
836
837 /* We need to lock the mutex while building lists of events,
838 * timeouts and watches to process but we can't keep the lock while
839 * processing them, or else we risk a deadlock:
840 *
841 * The signal functions could lock mutex X while p_events is locked;
842 * While some other function in vlc (playlist) might lock mutex X
843 * and then set a variable which would call AllCallback(), which itself
844 * needs to lock p_events to add a new event.
845 */
846 vlc_mutex_lock( &p_intf->p_sys->lock );
847
848 /* Get the list of timeouts to process */
849 unsigned int i_timeouts = vlc_array_count( p_sys->p_timeouts );
850 DBusTimeout *p_timeouts[i_timeouts ? i_timeouts : 1];
851 for( unsigned int i = 0; i < i_timeouts; i++ )
852 {
853 p_timeouts[i] = vlc_array_item_at_index( p_sys->p_timeouts, i );
854 }
855
856 /* Get the list of watches to process */
857 i_watches = vlc_array_count( p_sys->p_watches );
858 DBusWatch *p_watches[i_watches];
859 for( int i = 0; i < i_watches; i++ )
860 {
861 p_watches[i] = vlc_array_item_at_index( p_sys->p_watches, i );
862 }
863
864 /* Get the list of events to process */
865 int i_events = vlc_array_count( p_intf->p_sys->p_events );
866 callback_info_t* p_info[i_events ? i_events : 1];
867 for( int i = i_events - 1; i >= 0; i-- )
868 {
869 p_info[i] = vlc_array_item_at_index( p_intf->p_sys->p_events, i );
870 vlc_array_remove( p_intf->p_sys->p_events, i );
871 }
872
873 /* now we can release the lock and process what's pending */
874 vlc_mutex_unlock( &p_intf->p_sys->lock );
875
876 ProcessEvents( p_intf, p_info, i_events );
877 ProcessWatches( p_intf, p_watches, i_watches, fds, i_fds );
878
879 ProcessTimeouts( p_intf, p_timeouts, i_timeouts );
880 DispatchDBusMessages( p_intf );
881
882 vlc_restorecancel( canc );
883 }
884 assert(0);
885}

896/* Flls a callback_info_t data structure in response
897 * to an "intf-event" input event.
898 *
899 * @warning This function executes in the input thread.
900 *
901 * @return VLC_SUCCESS on success, VLC_E* on error.
902 */
903static int InputCallback( vlc_object_t *p_this, const char *psz_var,
904 vlc_value_t oldval, vlc_value_t newval, void *data )
905{
906 input_thread_t *p_input = (input_thread_t *)p_this;
907 intf_thread_t *p_intf = data;
908 intf_sys_t *p_sys = p_intf->p_sys;
909
910 dbus_int32_t i_state = PLAYBACK_STATE_INVALID;
911
912 callback_info_t *p_info = calloc( 1, sizeof( callback_info_t ) );
913 if( unlikely(p_info == NULL) )
914 return VLC_ENOMEM;
915
916 switch( newval.i_int )
917 {
918 case INPUT_EVENT_DEAD:
919 case INPUT_EVENT_ABORT:
920 i_state = PLAYBACK_STATE_STOPPED;
921 break;
922 case INPUT_EVENT_STATE:
923 switch( var_GetInteger( p_input, "state" ) )
924 {
925 case OPENING_S:
926 case PLAYING_S:
927 i_state = PLAYBACK_STATE_PLAYING;
928 break;
929 case PAUSE_S:
930 i_state = PLAYBACK_STATE_PAUSED;
931 break;
932 default:
933 i_state = PLAYBACK_STATE_STOPPED;
934 }
935 break;
936 case INPUT_EVENT_ITEM_META:
937 p_info->signal = SIGNAL_INPUT_METADATA;
938 break;
939 case INPUT_EVENT_RATE:
940 p_info->signal = SIGNAL_RATE;
941 break;
942 case INPUT_EVENT_POSITION:
943 {
944 mtime_t i_now = mdate(), i_pos, i_projected_pos, i_interval;
945 float f_current_rate;
946
947 /* Detect seeks
948 * XXX: This is way more convoluted than it should be... */
949 i_pos = var_GetTime( p_input, "time" );
950
951 if( !p_intf->p_sys->i_last_input_pos_event ||
952 !( var_GetInteger( p_input, "state" ) == PLAYING_S ) )
953 {
954 p_intf->p_sys->i_last_input_pos_event = i_now;
955 p_intf->p_sys->i_last_input_pos = i_pos;
956 break;
957 }
958
959 f_current_rate = var_GetFloat( p_input, "rate" );
960 i_interval = ( i_now - p_intf->p_sys->i_last_input_pos_event );
961
962 i_projected_pos = p_intf->p_sys->i_last_input_pos +
963 ( i_interval * f_current_rate );
964
965 p_intf->p_sys->i_last_input_pos_event = i_now;
966 p_intf->p_sys->i_last_input_pos = i_pos;
967
968 if( llabs( i_pos - i_projected_pos ) < SEEK_THRESHOLD )
969 break;
970
971 p_info->signal = SIGNAL_SEEK;
972 p_info->i_item = input_GetItem( p_input )->i_id;
973 break;
974 }
975 default:
976 free( p_info );
977 return VLC_SUCCESS; /* don't care */
978 }
979
980 vlc_mutex_lock( &p_sys->lock );
981 if( i_state != PLAYBACK_STATE_INVALID &&
982 i_state != p_sys->i_playing_state )
983 {
984 p_sys->i_playing_state = i_state;
985 p_info->signal = SIGNAL_STATE;
986 }
987 if( p_info->signal )
988 vlc_array_append( p_intf->p_sys->p_events, p_info );
989 else
990 free( p_info );
991 vlc_mutex_unlock( &p_intf->p_sys->lock );
992
993 wakeup_main_loop( p_intf );
994
995 (void)psz_var;
996 (void)oldval;
997 return VLC_SUCCESS;
998}

1000// Get all the callbacks
1001static int AllCallback( vlc_object_t *p_this, const char *psz_var,
1002 vlc_value_t oldval, vlc_value_t newval, void *p_data )
1003{
1004 intf_thread_t *p_intf = p_data;
1005 callback_info_t info = { .signal = SIGNAL_NONE };
1006
1007 // Wich event is it ?
1008 if( !strcmp( "activity", psz_var ) )
1009 info.signal = SIGNAL_ITEM_CURRENT;
1010 else if( !strcmp( "volume", psz_var ) )
1011 {
1012 if( oldval.f_float != newval.f_float )
1013 info.signal = SIGNAL_VOLUME_CHANGE;
1014 }
1015 else if( !strcmp( "mute", psz_var ) )
1016 {
1017 if( oldval.b_bool != newval.b_bool )
1018 info.signal = SIGNAL_VOLUME_MUTED;
1019 }
1020 else if( !strcmp( "intf-change", psz_var ) )
1021 info.signal = SIGNAL_INTF_CHANGE;
1022 else if( !strcmp( "playlist-item-append", psz_var ) )
1023 {
1024 info.signal = SIGNAL_PLAYLIST_ITEM_APPEND;
1025 info.i_node = ((playlist_add_t*)newval.p_address)->i_node;
1026 }
1027 else if( !strcmp( "playlist-item-deleted", psz_var ) )
1028 info.signal = SIGNAL_PLAYLIST_ITEM_DELETED;
1029 else if( !strcmp( "random", psz_var ) )
1030 info.signal = SIGNAL_RANDOM;
1031 else if( !strcmp( "fullscreen", psz_var ) )
1032 info.signal = SIGNAL_FULLSCREEN;
1033 else if( !strcmp( "repeat", psz_var ) )
1034 info.signal = SIGNAL_REPEAT;
1035 else if( !strcmp( "loop", psz_var ) )
1036 info.signal = SIGNAL_LOOP;
1037 else if( !strcmp( "can-seek", psz_var ) )
1038 info.signal = SIGNAL_CAN_SEEK;
1039 else if( !strcmp( "can-pause", psz_var ) )
1040 info.signal = SIGNAL_CAN_PAUSE;
1041 else
1042 assert(0);
1043
1044 if( info.signal == SIGNAL_NONE )
1045 return VLC_SUCCESS;
1046
1047 callback_info_t *p_info = malloc( sizeof( *p_info ) );
1048 if( unlikely(p_info == NULL) )
1049 return VLC_ENOMEM;
1050
1051 // Append the event
1052 *p_info = info;
1053 vlc_mutex_lock( &p_intf->p_sys->lock );
1054 vlc_array_append( p_intf->p_sys->p_events, p_info );
1055 vlc_mutex_unlock( &p_intf->p_sys->lock );
1056
1057 wakeup_main_loop( p_intf );
1058 (void) p_this;
1059 return VLC_SUCCESS;
1060}

1062/*****************************************************************************
1063 * TrackChange: callback on playlist "activity"
1064 *****************************************************************************/
1065static int TrackChange( intf_thread_t *p_intf )
1066{
1067 intf_sys_t *p_sys = p_intf->p_sys;
1068 input_thread_t *p_input = NULL;
1069 input_item_t *p_item = NULL;
1070
1071 if( p_intf->p_sys->b_dead )
1072 return VLC_SUCCESS;
1073
1074 if( p_sys->p_input )
1075 {
1076 var_DelCallback( p_sys->p_input, "intf-event", InputCallback, p_intf );
1077 var_DelCallback( p_sys->p_input, "can-pause", AllCallback, p_intf );
1078 var_DelCallback( p_sys->p_input, "can-seek", AllCallback, p_intf );
1079 vlc_object_release( p_sys->p_input );
1080 p_sys->p_input = NULL;
1081 }
1082
1083 p_sys->b_meta_read = false;
1084
1085 p_input = pl_CurrentInput( p_intf );
1086 if( !p_input )
1087 {
1088 return VLC_SUCCESS;
1089 }
1090
1091 p_item = input_GetItem( p_input );
1092 if( !p_item )
1093 {
1094 vlc_object_release( p_input );
1095 return VLC_EGENERIC;
1096 }
1097
1098 if( input_item_IsPreparsed( p_item ) )
1099 p_sys->b_meta_read = true;
1100
1101 p_sys->p_input = p_input;
1102 var_AddCallback( p_input, "intf-event", InputCallback, p_intf );
1103 var_AddCallback( p_input, "can-pause", AllCallback, p_intf );
1104 var_AddCallback( p_input, "can-seek", AllCallback, p_intf );
1105
1106 return VLC_SUCCESS;
1107}
