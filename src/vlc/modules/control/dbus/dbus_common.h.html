43#define DBUS_METHOD( method_function ) \
44 static DBusHandlerResult method_function \
45 ( DBusConnection *p_conn, DBusMessage *p_from, void *p_this )
46
47#define DBUS_SIGNAL( signal_function ) \
48 static DBusHandlerResult signal_function \
49 ( DBusConnection *p_conn, void *p_data )
50
51#define REPLY_INIT \
52 DBusMessage* p_msg = dbus_message_new_method_return( p_from ); \
53 if( !p_msg ) return DBUS_HANDLER_RESULT_NEED_MEMORY; \
54
55#define REPLY_SEND \
56 if( !dbus_connection_send( p_conn, p_msg, NULL ) ) \
57 return DBUS_HANDLER_RESULT_NEED_MEMORY; \
58 dbus_connection_flush( p_conn ); \
59 dbus_message_unref( p_msg ); \
60 return DBUS_HANDLER_RESULT_HANDLED
61
62#define SIGNAL_INIT( interface, path, signal ) \
63 DBusMessage *p_msg = dbus_message_new_signal( path, \
64 interface, signal ); \
65 if( !p_msg ) return DBUS_HANDLER_RESULT_NEED_MEMORY; \
66
67#define SIGNAL_SEND \
68 if( !dbus_connection_send( p_conn, p_msg, NULL ) ) \
69 return DBUS_HANDLER_RESULT_NEED_MEMORY; \
70 dbus_message_unref( p_msg ); \
71 dbus_connection_flush( p_conn ); \
72 return DBUS_HANDLER_RESULT_HANDLED
73
74#define OUT_ARGUMENTS \
75 DBusMessageIter args; \
76 dbus_message_iter_init_append( p_msg, &args )
77
78#define DBUS_ADD( dbus_type, value ) \
79 if( !dbus_message_iter_append_basic( &args, dbus_type, value ) ) \
80 return DBUS_HANDLER_RESULT_NEED_MEMORY
81
82#define ADD_STRING( s ) DBUS_ADD( DBUS_TYPE_STRING, s )
83#define ADD_DOUBLE( d ) DBUS_ADD( DBUS_TYPE_DOUBLE, d )
84#define ADD_BOOL( b ) DBUS_ADD( DBUS_TYPE_BOOLEAN, b )
85#define ADD_INT32( i ) DBUS_ADD( DBUS_TYPE_INT32, i )
86#define ADD_INT64( i ) DBUS_ADD( DBUS_TYPE_INT64, i )
87#define ADD_BYTE( b ) DBUS_ADD( DBUS_TYPE_BYTE, b )
