42static int
43MarshalPosition( intf_thread_t *p_intf, DBusMessageIter *container )
44{
45 /* returns position in microseconds */
46 dbus_int64_t i_pos;
47 input_thread_t *p_input = pl_CurrentInput( p_intf );
48
49 if( !p_input )
50 i_pos = 0;
51
52 else
53 {
54 i_pos = var_GetTime( p_input, "time" );
55 vlc_object_release( p_input );
56 }
57
58 if( !dbus_message_iter_append_basic( container, DBUS_TYPE_INT64, &i_pos ) )
59 return VLC_ENOMEM;
60
61 return VLC_SUCCESS;
62}

64DBUS_METHOD( SetPosition )
65{ /* set position in microseconds */
66
67 REPLY_INIT;
68 dbus_int64_t i_pos;
69 vlc_value_t position;
70 char *psz_trackid, *psz_dbus_trackid;
71 input_item_t *p_item;
72
73 DBusError error;
74 dbus_error_init( &error );
75
76 dbus_message_get_args( p_from, &error,
77 DBUS_TYPE_OBJECT_PATH, &psz_dbus_trackid,
78 DBUS_TYPE_INT64, &i_pos,
79 DBUS_TYPE_INVALID );
80
81 if( dbus_error_is_set( &error ) )
82 {
83 msg_Err( (vlc_object_t*) p_this, "D-Bus message reading : %s",
84 error.message );
85 dbus_error_free( &error );
86 return DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
87 }
88
89 input_thread_t *p_input = pl_CurrentInput( p_this );
90
91 if( p_input )
92 {
93 if( ( p_item = input_GetItem( p_input ) ) )
94 {
95 if( -1 == asprintf( &psz_trackid,
96 MPRIS_TRACKID_FORMAT,
97 p_item->i_id ) )
98 {
99 vlc_object_release( p_input );
100 return DBUS_HANDLER_RESULT_NEED_MEMORY;
101 }
102
103 if( !strcmp( psz_trackid, psz_dbus_trackid ) )
104 {
105 position.i_time = (mtime_t) i_pos;
106 var_Set( p_input, "time", position );
107 }
108 free( psz_trackid );
109 }
110
111 vlc_object_release( p_input );
112 }
113
114
115 REPLY_SEND;
116}

118DBUS_METHOD( Seek )
119{
120 REPLY_INIT;
121 dbus_int64_t i_step;
122 vlc_value_t newpos;
123 mtime_t i_pos;
124
125 DBusError error;
126 dbus_error_init( &error );
127
128 dbus_message_get_args( p_from, &error,
129 DBUS_TYPE_INT64, &i_step,
130 DBUS_TYPE_INVALID );
131
132 if( dbus_error_is_set( &error ) )
133 {
134 msg_Err( (vlc_object_t*) p_this, "D-Bus message reading : %s",
135 error.message );
136 dbus_error_free( &error );
137 return DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
138 }
139
140 input_thread_t *p_input = pl_CurrentInput( p_this );
141 if( p_input && var_GetBool( p_input, "can-seek" ) )
142 {
143 i_pos = var_GetTime( p_input, "time" );
144 newpos.i_time = (mtime_t) i_step + i_pos;
145
146 if( newpos.i_time < 0 )
147 newpos.i_time = 0;
148
149 var_Set( p_input, "time", newpos );
150 }
151
152 if( p_input )
153 vlc_object_release( p_input );
154
155 REPLY_SEND;
156}

398static int
399MarshalPlaybackStatus( intf_thread_t *p_intf, DBusMessageIter *container )
400{
401 input_thread_t *p_input = pl_CurrentInput( p_intf );
402 const char *psz_playback_status;
403
404 if( p_input != NULL )
405 {
406 switch( var_GetInteger( p_input, "state" ) )
407 {
408 case OPENING_S:
409 case PLAYING_S:
410 psz_playback_status = PLAYBACK_STATUS_PLAYING;
411 break;
412 case PAUSE_S:
413 psz_playback_status = PLAYBACK_STATUS_PAUSED;
414 break;
415 default:
416 psz_playback_status = PLAYBACK_STATUS_STOPPED;
417 }
418
419 vlc_object_release( (vlc_object_t*) p_input );
420 }
421 else
422 psz_playback_status = PLAYBACK_STATUS_STOPPED;
423
424 if( !dbus_message_iter_append_basic( container, DBUS_TYPE_STRING,
425 &psz_playback_status ) )
426 return VLC_ENOMEM;
427
428 return VLC_SUCCESS;
429}

497static int
498MarshalLoopStatus( intf_thread_t *p_intf, DBusMessageIter *container )
499{
500 const char *psz_loop_status;
501
502 if( var_GetBool( p_intf->p_sys->p_playlist, "repeat" ) )
503 psz_loop_status = LOOP_STATUS_TRACK;
504
505 else if( var_GetBool( p_intf->p_sys->p_playlist, "loop" ) )
506 psz_loop_status = LOOP_STATUS_PLAYLIST;
507
508 else
509 psz_loop_status = LOOP_STATUS_NONE;
510
511 if( !dbus_message_iter_append_basic( container, DBUS_TYPE_STRING,
512 &psz_loop_status ) )
513 return VLC_ENOMEM;
514
515 return VLC_SUCCESS;
516}

518DBUS_METHOD( LoopStatusSet )
519{
520 REPLY_INIT;
521 char *psz_loop_status;
522
523 if( VLC_SUCCESS != DemarshalSetPropertyValue( p_from, &psz_loop_status ) )
524 return DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
525
526 if( !strcmp( psz_loop_status, LOOP_STATUS_NONE ) )
527 {
528 var_SetBool( PL, "loop", FALSE );
529 var_SetBool( PL, "repeat", FALSE );
530 }
531 else if( !strcmp( psz_loop_status, LOOP_STATUS_TRACK ) )
532 {
533 var_SetBool( PL, "loop", FALSE );
534 var_SetBool( PL, "repeat", TRUE );
535 }
536 else if( !strcmp( psz_loop_status, LOOP_STATUS_PLAYLIST ) )
537 {
538 var_SetBool( PL, "loop", TRUE );
539 var_SetBool( PL, "repeat", FALSE );
540 }
541 else
542 return DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
543
544 REPLY_SEND;
545}

547static int
548MarshalMetadata( intf_thread_t *p_intf, DBusMessageIter *container )
549{
550 DBusMessageIter a;
551 input_thread_t *p_input = pl_CurrentInput( p_intf );
552 input_item_t *p_item = NULL;
553
554 if( p_input != NULL )
555 {
556 p_item = input_GetItem( p_input );
557
558 if( p_item )
559 {
560 int result = GetInputMeta( p_item, container );
561
562 if (result != VLC_SUCCESS)
563 {
564 vlc_object_release( (vlc_object_t*) p_input );
565 return result;
566 }
567 }
568
569 vlc_object_release( (vlc_object_t*) p_input );
570 }
571
572 if (!p_item)
573 {
574 // avoid breaking the type marshalling
575 if( !dbus_message_iter_open_container( container, DBUS_TYPE_ARRAY, "{sv}", &a ) ||
576 !dbus_message_iter_close_container( container, &a ) )
577 return VLC_ENOMEM;
578 }
579
580 return VLC_SUCCESS;
581}

606 SIGNAL_SEND;
607}

721DBUS_METHOD( GetAllProperties )
722{
723 REPLY_INIT;
724 OUT_ARGUMENTS;
725
726 DBusError error;
727 DBusMessageIter dict;
728
729 char *const psz_interface_name = NULL;
730
731 dbus_error_init( &error );
732 dbus_message_get_args( p_from, &error,
733 DBUS_TYPE_STRING, &psz_interface_name,
734 DBUS_TYPE_INVALID );
735
736 if( dbus_error_is_set( &error ) )
737 {
738 msg_Err( (vlc_object_t*) p_this, "D-Bus message reading : %s",
739 error.message );
740 dbus_error_free( &error );
741 return DBUS_HANDLER_RESULT_NOT_YET_HANDLED;
742 }
743
744 msg_Dbg( (vlc_object_t*) p_this, "Getting All properties" );
745

824/**
825 * PropertiesChangedSignal() synthetizes and sends the
826 * org.freedesktop.DBus.Properties.PropertiesChanged signal
827 */
828static DBusHandlerResult
829PropertiesChangedSignal( intf_thread_t *p_intf,
830 vlc_dictionary_t *p_changed_properties )
831{
832 DBusConnection *p_conn = p_intf->p_sys->p_conn;
833 DBusMessageIter changed_properties, invalidated_properties;
834 const char *psz_interface_name = DBUS_MPRIS_PLAYER_INTERFACE;
835 char **ppsz_properties = NULL;
836 int i_properties = 0;
837
838 SIGNAL_INIT( DBUS_INTERFACE_PROPERTIES,
839 DBUS_MPRIS_OBJECT_PATH,
840 "PropertiesChanged" );
841
842 OUT_ARGUMENTS;
843 ADD_STRING( &psz_interface_name );
844
845 if( !dbus_message_iter_open_container( &args, DBUS_TYPE_ARRAY, "{sv}",
846 &changed_properties ) )
847 return DBUS_HANDLER_RESULT_NEED_MEMORY;
848
849 i_properties = vlc_dictionary_keys_count( p_changed_properties );
850 ppsz_properties = vlc_dictionary_all_keys( p_changed_properties );
851
852 if( unlikely(!ppsz_properties) )
853 {
854 dbus_message_iter_abandon_container( &args, &changed_properties );
855 return DBUS_HANDLER_RESULT_NEED_MEMORY;
856 }
857
858 for( int i = 0; i < i_properties; i++ )
859 {
860 PROPERTY_MAPPING_BEGIN
861 PROPERTY_ENTRY( Metadata, "a{sv}" )
862 PROPERTY_ENTRY( PlaybackStatus, "s" )
863 PROPERTY_ENTRY( LoopStatus, "s" )
864 PROPERTY_ENTRY( Rate, "d" )
865 PROPERTY_ENTRY( Shuffle, "b" )
866 PROPERTY_ENTRY( Volume, "d" )
867 PROPERTY_ENTRY( CanSeek, "b" )
868 PROPERTY_ENTRY( CanPlay, "b" )
869 PROPERTY_ENTRY( CanPause, "b" )
870 PROPERTY_MAPPING_END
871
872 free( ppsz_properties[i] );
873 }
874
875 free( ppsz_properties );
876
877 if( !dbus_message_iter_close_container( &args, &changed_properties ) )
878 return DBUS_HANDLER_RESULT_NEED_MEMORY;
879
880 if( !dbus_message_iter_open_container( &args, DBUS_TYPE_ARRAY, "s",
881 &invalidated_properties ) )
882 return DBUS_HANDLER_RESULT_NEED_MEMORY;
883
884 if( !dbus_message_iter_close_container( &args, &invalidated_properties ) )
885 return DBUS_HANDLER_RESULT_NEED_MEMORY;
886
887 SIGNAL_SEND;
888}

891#undef PROPERTY_ADD
892#undef PROPERTY_MAPPING_END
