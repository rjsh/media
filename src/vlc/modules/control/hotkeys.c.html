44/*****************************************************************************
45 * intf_sys_t: description and status of FB interface
46 *****************************************************************************/
47struct intf_sys_t
48{
49 vout_thread_t *p_last_vout;
50 int slider_chan;
51
52 /*subtitle_delaybookmarks: placeholder for storing subtitle sync timestamps*/
53 struct
54 {
55 int64_t i_time_subtitle;
56 int64_t i_time_audio;
57 } subtitle_delaybookmarks;
58};

97/*****************************************************************************
98 * Open: initialize interface
99 *****************************************************************************/
100static int Open( vlc_object_t *p_this )
101{
102 intf_thread_t *p_intf = (intf_thread_t *)p_this;
103 intf_sys_t *p_sys;
104 p_sys = malloc( sizeof( intf_sys_t ) );
105 if( !p_sys )
106 return VLC_ENOMEM;
107
108 p_intf->p_sys = p_sys;
109
110 p_sys->p_last_vout = NULL;
111 p_sys->subtitle_delaybookmarks.i_time_audio = 0;
112 p_sys->subtitle_delaybookmarks.i_time_subtitle = 0;
113
114 var_AddCallback( p_intf->p_libvlc, "key-action", ActionEvent, p_intf );
115 return VLC_SUCCESS;
116}
117
118/*****************************************************************************
119 * Close: destroy interface
120 *****************************************************************************/
121static void Close( vlc_object_t *p_this )
122{
123 intf_thread_t *p_intf = (intf_thread_t *)p_this;
124 intf_sys_t *p_sys = p_intf->p_sys;
125
126 var_DelCallback( p_intf->p_libvlc, "key-action", ActionEvent, p_intf );
127
128 /* Destroy structure */
129 free( p_sys );
130}
131
132static int PutAction( intf_thread_t *p_intf, int i_action )
133{
134 intf_sys_t *p_sys = p_intf->p_sys;
135 playlist_t *p_playlist = pl_Get( p_intf );
136
137 /* Update the input */
138 input_thread_t *p_input = playlist_CurrentInput( p_playlist );
139
140 /* Update the vout */
141 vout_thread_t *p_vout = p_input ? input_GetVout( p_input ) : NULL;
142
143 /* Register OSD channels */
144 /* FIXME: this check can fail if the new vout is reallocated at the same
145 * address as the old one... We should rather listen to vout events.
146 * Alternatively, we should keep a reference to the vout thread. */
147 if( p_vout && p_vout != p_sys->p_last_vout )
148 p_sys->slider_chan = vout_RegisterSubpictureChannel( p_vout );
149 p_sys->p_last_vout = p_vout;
150
151 /* Quit */
152 switch( i_action )
153 {
154 /* Libvlc / interface actions */
155 case ACTIONID_QUIT:
156 libvlc_Quit( p_intf->p_libvlc );
157
158 ClearChannels( p_intf, p_vout );
159 DisplayMessage( p_vout, _( "Quit" ) );
160 break;
161
162 case ACTIONID_INTF_TOGGLE_FSC:
163 case ACTIONID_INTF_HIDE:
164 var_TriggerCallback( p_intf->p_libvlc, "intf-toggle-fscontrol" );
165 break;
166 case ACTIONID_INTF_BOSS:
167 var_TriggerCallback( p_intf->p_libvlc, "intf-boss" );
168 break;
169 case ACTIONID_INTF_POPUP_MENU:
170 var_TriggerCallback( p_intf->p_libvlc, "intf-popupmenu" );
171 break;
172
173 /* Playlist actions (including audio) */
174 case ACTIONID_LOOP:
175 {
176 /* Toggle Normal -> Loop -> Repeat -> Normal ... */
177 const char *mode;
178 if( var_GetBool( p_playlist, "repeat" ) )
179 {
180 var_SetBool( p_playlist, "repeat", false );
181 mode = N_("Off");
182 }
183 else
184 if( var_GetBool( p_playlist, "loop" ) )
185 { /* FIXME: this is not atomic, we should use a real tristate */
186 var_SetBool( p_playlist, "loop", false );
187 var_SetBool( p_playlist, "repeat", true );
188 mode = N_("One");
189 }
190 else
191 {
192 var_SetBool( p_playlist, "loop", true );
193 mode = N_("All");
194 }
195 DisplayMessage( p_vout, _("Loop: %s"), vlc_gettext(mode) );
196 break;
197 }
198
199 case ACTIONID_RANDOM:
200 {
201 const bool state = var_ToggleBool( p_playlist, "random" );
202 DisplayMessage( p_vout, _("Random: %s"),
203 vlc_gettext( state ? N_("On") : N_("Off") ) );
204 break;
205 }
206
207 case ACTIONID_NEXT:
208 DisplayMessage( p_vout, _("Next") );
209 playlist_Next( p_playlist );
210 break;
211 case ACTIONID_PREV:
212 DisplayMessage( p_vout, _("Previous") );
213 playlist_Prev( p_playlist );
214 break;
215
216 case ACTIONID_STOP:
217 playlist_Stop( p_playlist );
218 break;
219
220 case ACTIONID_RATE_NORMAL:
221 var_SetFloat( p_playlist, "rate", 1.f );
222 DisplayRate( p_vout, 1.f );
223 break;
224 case ACTIONID_FASTER:
225 var_TriggerCallback( p_playlist, "rate-faster" );
226 DisplayRate( p_vout, var_GetFloat( p_playlist, "rate" ) );
227 break;
228 case ACTIONID_SLOWER:
229 var_TriggerCallback( p_playlist, "rate-slower" );
230 DisplayRate( p_vout, var_GetFloat( p_playlist, "rate" ) );
231 break;
232 case ACTIONID_RATE_FASTER_FINE:
233 case ACTIONID_RATE_SLOWER_FINE:
234 {
235 const int i_dir = i_action == ACTIONID_RATE_FASTER_FINE ? 1 : -1;
236 float rate = AdjustRateFine( VLC_OBJECT(p_playlist), i_dir );
237
238 var_SetFloat( p_playlist, "rate", rate );
239 DisplayRate( p_vout, rate );
240 break;
241 }
242
243 case ACTIONID_PLAY_BOOKMARK1:
244 case ACTIONID_PLAY_BOOKMARK2:
245 case ACTIONID_PLAY_BOOKMARK3:
246 case ACTIONID_PLAY_BOOKMARK4:
247 case ACTIONID_PLAY_BOOKMARK5:
248 case ACTIONID_PLAY_BOOKMARK6:
249 case ACTIONID_PLAY_BOOKMARK7:
250 case ACTIONID_PLAY_BOOKMARK8:
251 case ACTIONID_PLAY_BOOKMARK9:
252 case ACTIONID_PLAY_BOOKMARK10:
253 PlayBookmark( p_intf, i_action - ACTIONID_PLAY_BOOKMARK1 + 1 );
254 break;
255
256 case ACTIONID_SET_BOOKMARK1:
257 case ACTIONID_SET_BOOKMARK2:
258 case ACTIONID_SET_BOOKMARK3:
259 case ACTIONID_SET_BOOKMARK4:
260 case ACTIONID_SET_BOOKMARK5:
261 case ACTIONID_SET_BOOKMARK6:
262 case ACTIONID_SET_BOOKMARK7:
263 case ACTIONID_SET_BOOKMARK8:
264 case ACTIONID_SET_BOOKMARK9:
265 case ACTIONID_SET_BOOKMARK10:
266 SetBookmark( p_intf, i_action - ACTIONID_SET_BOOKMARK1 + 1 );
267 break;
268 case ACTIONID_PLAY_CLEAR:
269 {
270 playlist_t *p_playlist = pl_Get( p_intf );
271 playlist_Clear( p_playlist, pl_Unlocked );
272 break;
273 }
274 case ACTIONID_VOL_UP:
275 {
276 float vol;
277 if( playlist_VolumeUp( p_playlist, 1, &vol ) == 0 )
278 DisplayVolume( p_intf, p_vout, vol );
279 break;
280 }
281 case ACTIONID_VOL_DOWN:
282 {
283 float vol;
284 if( playlist_VolumeDown( p_playlist, 1, &vol ) == 0 )
285 DisplayVolume( p_intf, p_vout, vol );
286 break;
287 }
288 case ACTIONID_VOL_MUTE:
289 {
290 int mute = playlist_MuteGet( p_playlist );
291 if( mute < 0 )
292 break;
293 mute = !mute;
294 if( playlist_MuteSet( p_playlist, mute ) )
295 break;
296
297 float vol = playlist_VolumeGet( p_playlist );
298 if( mute || vol == 0.f )
299 {
300 ClearChannels( p_intf, p_vout );
301 DisplayIcon( p_vout, OSD_MUTE_ICON );
302 }
303 else
304 DisplayVolume( p_intf, p_vout, vol );
305 break;
306 }
307
308 case ACTIONID_AUDIODEVICE_CYCLE:
309 {
310 audio_output_t *p_aout = playlist_GetAout( p_playlist );
311 if( p_aout == NULL )
312 break;
313
314 char **ids, **names;
315 int n = aout_DevicesList( p_aout, &ids, &names );
316 if( n == -1 )
317 break;
318
319 char *dev = aout_DeviceGet( p_aout );
320 const char *devstr = (dev != NULL) ? dev : "";
321
322 int idx = 0;
323 for( int i = 0; i < n; i++ )
324 {
325 if( !strcmp(devstr, ids[i]) )
326 idx = (i + 1) % n;
327 }
328 free( dev );
329
330 if( !aout_DeviceSet( p_aout, ids[idx] ) )
331 DisplayMessage( p_vout, _("Audio Device: %s"), names[idx] );
332 vlc_object_release( p_aout );
333
334 for( int i = 0; i < n; i++ )
335 {
336 free( ids[i] );
337 free( names[i] );
338 }
339 free( ids );
340 free( names );
341 break;
342 }
343
344 /* Playlist + input actions */
345 case ACTIONID_PLAY_PAUSE:
346 if( p_input )
347 {
348 ClearChannels( p_intf, p_vout );
349
350 int state = var_GetInteger( p_input, "state" );
351 DisplayIcon( p_vout, state != PAUSE_S ? OSD_PAUSE_ICON : OSD_PLAY_ICON );
352 playlist_Pause( p_playlist );
353 }
354 else
355 playlist_Play( p_playlist );
356 break;
357
358 case ACTIONID_PLAY:
359 if( p_input && var_GetFloat( p_input, "rate" ) != 1. )
360 /* Return to normal speed */
361 var_SetFloat( p_input, "rate", 1. );
362 else
363 {
364 ClearChannels( p_intf, p_vout );
365 DisplayIcon( p_vout, OSD_PLAY_ICON );
366 playlist_Play( p_playlist );
367 }
368 break;
369
370 /* Playlist + video output actions */
371 case ACTIONID_WALLPAPER:
372 {
373 bool wp = var_ToggleBool( p_playlist, "video-wallpaper" );
374 if( p_vout )
375 var_SetBool( p_vout, "video-wallpaper", wp );
376 break;
377 }
378
379 /* Input actions */
380 case ACTIONID_PAUSE:
381 if( p_input && var_GetInteger( p_input, "state" ) != PAUSE_S )
382 {
383 ClearChannels( p_intf, p_vout );
384 DisplayIcon( p_vout, OSD_PAUSE_ICON );
385 var_SetInteger( p_input, "state", PAUSE_S );
386 }
387 break;
388
389 case ACTIONID_RECORD:
390 if( p_input && var_GetBool( p_input, "can-record" ) )
391 {
392 const bool on = var_ToggleBool( p_input, "record" );
393 DisplayMessage( p_vout, vlc_gettext(on
394 ? N_("Recording") : N_("Recording done")) );
395 }
396 break;
397
398 case ACTIONID_FRAME_NEXT:
399 if( p_input )
400 {
401 var_TriggerCallback( p_input, "frame-next" );
402 DisplayMessage( p_vout, _("Next frame") );
403 }
404 break;
405
406 case ACTIONID_SUBSYNC_MARKAUDIO:
407 {
408 p_sys->subtitle_delaybookmarks.i_time_audio = mdate();
409 DisplayMessage( p_vout, _("Sub sync: bookmarked audio time"));
410 break;
411 }
412 case ACTIONID_SUBSYNC_MARKSUB:
413 if( p_input )
414 {
415 vlc_value_t val, list, list2;
416 int i_count;
417 var_Get( p_input, "spu-es", &val );
418
419 var_Change( p_input, "spu-es", VLC_VAR_GETCHOICES,
420 &list, &list2 );
421 i_count = list.p_list->i_count;
422 if( i_count < 1 || val.i_int < 0 )
423 {
424 DisplayMessage( p_vout, _("No active subtitle") );
425 var_FreeList( &list, &list2 );
426 break;
427 }
428 p_sys->subtitle_delaybookmarks.i_time_subtitle = mdate();
429 DisplayMessage( p_vout,
430 _("Sub sync: bookmarked subtitle time"));
431 var_FreeList( &list, &list2 );
432 }
433 break;
434 case ACTIONID_SUBSYNC_APPLY:
435 {
436 /* Warning! Can yield a pause in the playback.
437 * For example, the following succession of actions will yield a 5 second delay :
438 * - Pressing Shift-H (ACTIONID_SUBSYNC_MARKAUDIO)
439 * - wait 5 second
440 * - Press Shift-J (ACTIONID_SUBSYNC_MARKSUB)
441 * - Press Shift-K (ACTIONID_SUBSYNC_APPLY)
442 * --> 5 seconds pause
443 * This is due to var_SetTime() (and ultimately UpdatePtsDelay())
444 * which causes the video to pause for an equivalent duration
445 * (This problem is also present in the "Track synchronization" window) */
446 if ( p_input )
447 {
448 if ( (p_sys->subtitle_delaybookmarks.i_time_audio == 0) || (p_sys->subtitle_delaybookmarks.i_time_subtitle == 0) )
449 {
450 DisplayMessage( p_vout, _( "Sub sync: set bookmarks first!" ) );
451 }
452 else
453 {
454 int64_t i_current_subdelay = var_GetTime( p_input, "spu-delay" );
455 int64_t i_additional_subdelay = p_sys->subtitle_delaybookmarks.i_time_audio - p_sys->subtitle_delaybookmarks.i_time_subtitle;
456 int64_t i_total_subdelay = i_current_subdelay + i_additional_subdelay;
457 var_SetTime( p_input, "spu-delay", i_total_subdelay);
458 ClearChannels( p_intf, p_vout );
459 DisplayMessage( p_vout, _( "Sub sync: corrected %i ms (total delay = %i ms)" ),
460 (int)(i_additional_subdelay / 1000),
461 (int)(i_total_subdelay / 1000) );
462 p_sys->subtitle_delaybookmarks.i_time_audio = 0;
463 p_sys->subtitle_delaybookmarks.i_time_subtitle = 0;
464 }
465 }
466 break;
467 }
468 case ACTIONID_SUBSYNC_RESET:
469 {
470 var_SetTime( p_input, "spu-delay", 0);
471 ClearChannels( p_intf, p_vout );
472 DisplayMessage( p_vout, _( "Sub sync: delay reset" ) );
473 p_sys->subtitle_delaybookmarks.i_time_audio = 0;
474 p_sys->subtitle_delaybookmarks.i_time_subtitle = 0;
475 break;
476 }
477
478 case ACTIONID_SUBDELAY_DOWN:
479 case ACTIONID_SUBDELAY_UP:
480 {
481 int diff = (i_action == ACTIONID_SUBDELAY_UP) ? 50000 : -50000;
482 if( p_input )
483 {
484 vlc_value_t val, list, list2;
485 int i_count;
486 var_Get( p_input, "spu-es", &val );
487
488 var_Change( p_input, "spu-es", VLC_VAR_GETCHOICES,
489 &list, &list2 );
490 i_count = list.p_list->i_count;
491 if( i_count < 1 || val.i_int < 0 )
492 {
493 DisplayMessage( p_vout, _("No active subtitle") );
494 var_FreeList( &list, &list2 );
495 break;
496 }
497 int64_t i_delay = var_GetTime( p_input, "spu-delay" ) + diff;
498
499 var_SetTime( p_input, "spu-delay", i_delay );
500 ClearChannels( p_intf, p_vout );
501 DisplayMessage( p_vout, _( "Subtitle delay %i ms" ),
502 (int)(i_delay/1000) );
503 var_FreeList( &list, &list2 );
504 }
505 break;
506 }
507 case ACTIONID_AUDIODELAY_DOWN:
508 case ACTIONID_AUDIODELAY_UP:
509 {
510 int diff = (i_action == ACTIONID_AUDIODELAY_UP) ? 50000 : -50000;
511 if( p_input )
512 {
513 int64_t i_delay = var_GetTime( p_input, "audio-delay" ) + diff;
514
515 var_SetTime( p_input, "audio-delay", i_delay );
516 ClearChannels( p_intf, p_vout );
517 DisplayMessage( p_vout, _( "Audio delay %i ms" ),
518 (int)(i_delay/1000) );
519 }
520 break;
521 }
522
523 case ACTIONID_AUDIO_TRACK:
524 if( p_input )
525 {
526 vlc_value_t val, list, list2;
527 int i_count, i;
528 var_Get( p_input, "audio-es", &val );
529 var_Change( p_input, "audio-es", VLC_VAR_GETCHOICES,
530 &list, &list2 );
531 i_count = list.p_list->i_count;
532 if( i_count > 1 )
533 {
534 for( i = 0; i < i_count; i++ )
535 {
536 if( val.i_int == list.p_list->p_values[i].i_int )
537 {
538 break;
539 }
540 }
541 /* value of audio-es was not in choices list */
542 if( i == i_count )
543 {
544 msg_Warn( p_input,
545 "invalid current audio track, selecting 0" );
546 i = 0;
547 }
548 else if( i == i_count - 1 )
549 i = 1;
550 else
551 i++;
552 var_Set( p_input, "audio-es", list.p_list->p_values[i] );
553 DisplayMessage( p_vout, _("Audio track: %s"),
554 list2.p_list->p_values[i].psz_string );
555 }
556 var_FreeList( &list, &list2 );
557 }
558 break;
559 case ACTIONID_SUBTITLE_TRACK:
560 if( p_input )
561 {
562 vlc_value_t val, list, list2;
563 int i_count, i;
564 var_Get( p_input, "spu-es", &val );
565
566 var_Change( p_input, "spu-es", VLC_VAR_GETCHOICES,
567 &list, &list2 );
568 i_count = list.p_list->i_count;
569 if( i_count <= 1 )
570 {
571 DisplayMessage( p_vout, _("Subtitle track: %s"),
572 _("N/A") );
573 var_FreeList( &list, &list2 );
574 break;
575 }
576 for( i = 0; i < i_count; i++ )
577 {
578 if( val.i_int == list.p_list->p_values[i].i_int )
579 {
580 break;
581 }
582 }
583 /* value of spu-es was not in choices list */
584 if( i == i_count )
585 {
586 msg_Warn( p_input,
587 "invalid current subtitle track, selecting 0" );
588 i = 0;
589 }
590 else if( i == i_count - 1 )
591 i = 0;
592 else
593 i++;
594 var_SetInteger( p_input, "spu-es", list.p_list->p_values[i].i_int );
595 var_SetInteger( p_input, "spu-choice", list.p_list->p_values[i].i_int );
596 DisplayMessage( p_vout, _("Subtitle track: %s"),
597 list2.p_list->p_values[i].psz_string );
598 var_FreeList( &list, &list2 );
599 }
600 break;
601 case ACTIONID_SUBTITLE_TOGGLE:
602 if( p_input )
603 {
604 vlc_value_t list, list2;
605 int i_count, i_sel_index, i_sel_id, i_old_id, i_new_index;
606 i_old_id = var_GetInteger( p_input, "spu-es" );
607 i_sel_id = var_GetInteger( p_input, "spu-choice" );
608
609 var_Change( p_input, "spu-es", VLC_VAR_GETCHOICES,
610 &list, &list2 );
611 i_count = list.p_list->i_count;
612 if( i_count <= 1 )
613 {
614 DisplayMessage( p_vout, _("Subtitle track: %s"),
615 _("N/A") );
616 var_FreeList( &list, &list2 );
617 break;
618 }
619 for( i_sel_index = 0; i_sel_index < i_count; i_sel_index++ )
620 {
621 if( i_sel_id == list.p_list->p_values[i_sel_index].i_int )
622 {
623 break;
624 }
625 }
626 /* if there is nothing to toggle choose the first track */
627 if( !i_sel_index ) {
628 i_sel_index = 1;
629 i_sel_id = list.p_list->p_values[1].i_int;
630 var_SetInteger( p_input, "spu-choice", i_sel_id );
631 }
632
633 i_new_index = 0;
634 if( i_old_id != i_sel_id )
635 {
636 if( i_sel_index >= i_count )
637 {
638 var_SetInteger( p_input, "spu-choice", list.p_list->p_values[0].i_int );
639 }
640 else
641 {
642 i_new_index = i_sel_index;
643 }
644 }
645 var_SetInteger( p_input, "spu-es", list.p_list->p_values[i_new_index].i_int );
646 DisplayMessage( p_vout, _("Subtitle track: %s"),
647 list2.p_list->p_values[i_new_index].psz_string );
648 var_FreeList( &list, &list2 );
649 }
650 break;
651 case ACTIONID_PROGRAM_SID_NEXT:
652 case ACTIONID_PROGRAM_SID_PREV:
653 if( p_input )
654 {
655 vlc_value_t val, list, list2;
656 int i_count, i;
657 var_Get( p_input, "program", &val );
658
659 var_Change( p_input, "program", VLC_VAR_GETCHOICES,
660 &list, &list2 );
661 i_count = list.p_list->i_count;
662 if( i_count <= 1 )
663 {
664 DisplayMessage( p_vout, _("Program Service ID: %s"),
665 _("N/A") );
666 var_FreeList( &list, &list2 );
667 break;
668 }
669 for( i = 0; i < i_count; i++ )
670 {
671 if( val.i_int == list.p_list->p_values[i].i_int )
672 {
673 break;
674 }
675 }
676 /* value of program sid was not in choices list */
677 if( i == i_count )
678 {
679 msg_Warn( p_input,
680 "invalid current program SID, selecting 0" );

1231static float AdjustRateFine( vlc_object_t *p_obj, const int i_dir )
1232{
1233 const float f_rate_min = (float)INPUT_RATE_DEFAULT / INPUT_RATE_MAX;
1234 const float f_rate_max = (float)INPUT_RATE_DEFAULT / INPUT_RATE_MIN;
1235 float f_rate = var_GetFloat( p_obj, "rate" );
1236
1237 int i_sign = f_rate < 0 ? -1 : 1;
1238
1239 f_rate = floor( fabs(f_rate) / 0.1 + i_dir + 0.05 ) * 0.1;
1240
1241 if( f_rate < f_rate_min )
1242 f_rate = f_rate_min;
1243 else if( f_rate > f_rate_max )
1244 f_rate = f_rate_max;
1245 f_rate *= i_sign;
1246
1247 return f_rate;
1248}
