32#include <vlc_common.h>
33#include <vlc_plugin.h>
34#include <vlc_interface.h>
35
36#ifdef _WIN32
37#define QUIET_TEXT N_("Do not open a DOS command box interface")
38#define QUIET_LONGTEXT N_( \
39 "By default the dummy interface plugin will start a DOS command box. " \
40 "Enabling the quiet mode will not bring this command box but can also " \
41 "be pretty annoying when you want to stop VLC and no video window is " \
42 "open." )
43#endif
44
45static int Open( vlc_object_t * );
46
47vlc_module_begin ()
48 set_shortname( N_("Dummy") )
49 set_description( N_("Dummy interface") )
50 set_capability( "interface", 0 )
51 set_callbacks( Open, NULL )
52#ifdef _WIN32
53 add_bool( "dummy-quiet", false, QUIET_TEXT, QUIET_LONGTEXT, false )
54#endif
55vlc_module_end ()
56
57/*****************************************************************************
58 * Open: initialize dummy interface
59 *****************************************************************************/
60static int Open( vlc_object_t *p_this )
61{
62 intf_thread_t *p_intf = (intf_thread_t*) p_this;
63
64#ifdef _WIN32
65 bool b_quiet;
66 b_quiet = var_InheritBool( p_intf, "dummy-quiet" );
67 if( !b_quiet )
68 CONSOLE_INTRO_MSG;
69#endif
70
71 msg_Info( p_intf, "using the dummy interface module..." );
72
73 return VLC_SUCCESS;
74}
