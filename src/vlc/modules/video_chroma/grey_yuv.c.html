59/*****************************************************************************
60 * Activate: allocate a chroma function
61 *****************************************************************************
62 * This function allocates and initializes a chroma function
63 *****************************************************************************/
64static int Activate( vlc_object_t *p_this )
65{
66 filter_t *p_filter = (filter_t *)p_this;
67
68 if( p_filter->fmt_out.video.i_width & 1
69 || p_filter->fmt_out.video.i_height & 1 )
70 {
71 return -1;
72 }
73
74 if( p_filter->fmt_in.video.i_width != p_filter->fmt_out.video.i_width
75 || p_filter->fmt_in.video.i_height != p_filter->fmt_out.video.i_height
76 || p_filter->fmt_in.video.orientation != p_filter->fmt_out.video.orientation )
77 return -1;
78
79 switch( p_filter->fmt_in.video.i_chroma )
80 {
81 case VLC_CODEC_GREY:
82 switch( p_filter->fmt_out.video.i_chroma )
83 {
84 case VLC_CODEC_I420:
85 p_filter->pf_video_filter = GREY_I420_Filter;
86 break;
87 case VLC_CODEC_YUYV:
88 p_filter->pf_video_filter = GREY_YUY2_Filter;
89 break;
90 default:
91 return -1;
92 }
93 break;
94
95 default:
96 return -1;
97 }
98
99 return 0;
100}

107/*****************************************************************************
108 * GREY_I420: 8-bit grayscale to planar YUV 4:2:0
109 *****************************************************************************/
110static void GREY_I420( filter_t *p_filter, picture_t *p_source,
111 picture_t *p_dest )
112{
113 uint8_t *p_line = p_source->p->p_pixels;
114 uint8_t *p_y = p_dest->Y_PIXELS;
115 uint8_t *p_u = p_dest->U_PIXELS;
116 uint8_t *p_v = p_dest->V_PIXELS;
117
118 int i_x, i_y;
119
120 const int i_source_margin = p_source->p->i_pitch
121 - p_source->p->i_visible_pitch;
122 const int i_dest_margin = p_dest->p[0].i_pitch
123 - p_dest->p[0].i_visible_pitch;
124 const int i_dest_margin_c = p_dest->p[1].i_pitch
125 - p_dest->p[1].i_visible_pitch;
126
127 for( i_y = p_filter->fmt_in.video.i_height / 2; i_y-- ; )
128 {
129 memset(p_u, 0x80, p_dest->p[1].i_visible_pitch);
130 p_u += i_dest_margin_c;
131
132 memset(p_v, 0x80, p_dest->p[1].i_visible_pitch);
133 p_v += i_dest_margin_c;
134 }
135
136 for( i_y = p_filter->fmt_in.video.i_height; i_y-- ; )
137 {
138 for( i_x = p_filter->fmt_in.video.i_width / 8; i_x-- ; )
139 {
140 *p_y++ = *p_line++; *p_y++ = *p_line++;
141 *p_y++ = *p_line++; *p_y++ = *p_line++;
142 *p_y++ = *p_line++; *p_y++ = *p_line++;
143 *p_y++ = *p_line++; *p_y++ = *p_line++;
144 }
145
146 for( i_x = p_filter->fmt_in.video.i_width % 8; i_x-- ; )
147 {
148 *p_y++ = *p_line++;
149 }
150
151 p_line += i_source_margin;
152 p_y += i_dest_margin;
153 }
154}

156/*****************************************************************************
157 * GREY_YUY2: 8-bit grayscale to packed YUY2
158 *****************************************************************************/
159static void GREY_YUY2( filter_t *p_filter, picture_t *p_source,
160 picture_t *p_dest )
161{
162 uint8_t *p_in = p_source->p->p_pixels;
163 uint8_t *p_out = p_dest->p->p_pixels;
164
165 int i_x, i_y;
166
167 const int i_source_margin = p_source->p->i_pitch
168 - p_source->p->i_visible_pitch;
169 const int i_dest_margin = p_dest->p->i_pitch
170 - p_dest->p->i_visible_pitch;
171
172 for( i_y = p_filter->fmt_out.video.i_height; i_y-- ; )
173 {
174 for( i_x = p_filter->fmt_out.video.i_width / 8; i_x-- ; )
175 {
176 *p_out++ = *p_in++; *p_out++ = 0x80;
177 *p_out++ = *p_in++; *p_out++ = 0x80;
178 *p_out++ = *p_in++; *p_out++ = 0x80;
179 *p_out++ = *p_in++; *p_out++ = 0x80;
180 *p_out++ = *p_in++; *p_out++ = 0x80;
181 *p_out++ = *p_in++; *p_out++ = 0x80;
182 *p_out++ = *p_in++; *p_out++ = 0x80;
183 *p_out++ = *p_in++; *p_out++ = 0x80;
184 }
185
186 for( i_x = (p_filter->fmt_out.video.i_width % 8) / 2; i_x-- ; )
187 {
188 *p_out++ = *p_in++; *p_out++ = 0x80;
189 *p_out++ = *p_in++; *p_out++ = 0x80;
190 }
191
192 p_in += i_source_margin;
193 p_out += i_dest_margin;
194 }
