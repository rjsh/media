64static const vlc_fourcc_t pi_allowed_chromas[] = {
65 VLC_CODEC_I420,
66 VLC_CODEC_I422,
67 VLC_CODEC_RGB32,
68 VLC_CODEC_RGB24,
69 0
70};

79/*****************************************************************************
80 * Activate: allocate a chroma function
81 *****************************************************************************
82 * This function allocates and initializes a chroma function
83 *****************************************************************************/
84static int Activate( vlc_object_t *p_this )
85{
86 filter_t *p_filter = (filter_t *)p_this;
87 filter_sys_t *p_sys;
88 int i_ret;
89
90 const bool b_chroma = p_filter->fmt_in.video.i_chroma != p_filter->fmt_out.video.i_chroma;
91 const bool b_resize = p_filter->fmt_in.video.i_width != p_filter->fmt_out.video.i_width ||
92 p_filter->fmt_in.video.i_height != p_filter->fmt_out.video.i_height;
93 const bool b_transform = p_filter->fmt_in.video.orientation != p_filter->fmt_out.video.orientation;
94
95 if( !b_chroma && !b_resize && !b_transform)
96 return VLC_EGENERIC;
97
98 p_sys = p_filter->p_sys = calloc( 1, sizeof( *p_sys ) );
99 if( !p_sys )
100 return VLC_ENOMEM;
101
102 p_sys->p_chain = filter_chain_New( p_filter, "video filter2", false, BufferAllocationInit, NULL, p_filter );
103 if( !p_sys->p_chain )
104 {
105 free( p_sys );
106 return VLC_EGENERIC;
107 }
108
109 if( b_transform )
110 i_ret = BuildTransformChain( p_filter );
111 else if( b_chroma && b_resize )
112 i_ret = BuildChromaResize( p_filter );
113 else if( b_chroma )
114 i_ret = BuildChromaChain( p_filter );
115 else
116 i_ret = VLC_EGENERIC;
117
118 if( i_ret )
119 {
120 /* Hum ... looks like this really isn't going to work. Too bad. */
121 filter_chain_Delete( p_sys->p_chain );
122 free( p_sys );
123 return VLC_EGENERIC;
124 }
125 /* */
126 p_filter->pf_video_filter = Chain;
127 return VLC_SUCCESS;
128}
129
130static void Destroy( vlc_object_t *p_this )
131{
132 filter_t *p_filter = (filter_t *)p_this;
133 filter_chain_Delete( p_filter->p_sys->p_chain );
134 free( p_filter->p_sys );
135}

149static int BuildTransformChain( filter_t *p_filter )
150{
151
152 es_format_t fmt_mid;
153 int i_ret;
154
155 /* Lets try transform first, then (potentially) resize+chroma */
156 msg_Dbg( p_filter, "Trying to build transform, then chroma+resize" );
157 es_format_Copy( &fmt_mid, &p_filter->fmt_in );
158 video_format_TransformTo(&fmt_mid.video, p_filter->fmt_out.video.orientation);
159 i_ret = CreateChain( p_filter, &fmt_mid, NULL );
160 es_format_Clean( &fmt_mid );
161 if( i_ret == VLC_SUCCESS )
162 return VLC_SUCCESS;
163
164 /* Lets try resize+chroma first, then transform */
165 msg_Dbg( p_filter, "Trying to build chroma+resize" );
166 EsFormatMergeSize( &fmt_mid, &p_filter->fmt_out, &p_filter->fmt_in );
167 i_ret = CreateChain( p_filter, &fmt_mid, NULL );
168 es_format_Clean( &fmt_mid );
169 if( i_ret == VLC_SUCCESS )
170 return VLC_SUCCESS;
171
172 return VLC_EGENERIC;
173}
174
175static int BuildChromaResize( filter_t *p_filter )
176{
177 es_format_t fmt_mid;
178 int i_ret;
179
180 /* Lets try resizing and then doing the chroma conversion */
181 msg_Dbg( p_filter, "Trying to build resize+chroma" );
182 EsFormatMergeSize( &fmt_mid, &p_filter->fmt_in, &p_filter->fmt_out );
183 i_ret = CreateChain( p_filter, &fmt_mid, NULL );
184 es_format_Clean( &fmt_mid );
185 if( i_ret == VLC_SUCCESS )
186 return VLC_SUCCESS;
187
188 /* Lets try it the other way arround (chroma and then resize) */
189 msg_Dbg( p_filter, "Trying to build chroma+resize" );
190 EsFormatMergeSize( &fmt_mid, &p_filter->fmt_out, &p_filter->fmt_in );
191 i_ret = CreateChain( p_filter, &fmt_mid, NULL );
192 es_format_Clean( &fmt_mid );
193 if( i_ret == VLC_SUCCESS )
194 return VLC_SUCCESS;
195
196 return VLC_EGENERIC;
197}

199static int BuildChromaChain( filter_t *p_filter )
200{
201 es_format_t fmt_mid;
202
203 /* We have to protect ourself against a too high recursion */
204 const char *psz_option = MODULE_STRING"-level";
205 int i_level = 0;
206 for( const config_chain_t *c = p_filter->p_cfg; c != NULL; c = c->p_next)
207 {
208 if( c->psz_name && c->psz_value && !strcmp(c->psz_name, psz_option) )
209 {
210 i_level = atoi(c->psz_value);
211 if( i_level < 0 || i_level > CHAIN_LEVEL_MAX )
212 {
213 msg_Err( p_filter, "Too high level of recursion (%d)", i_level );
214 return VLC_EGENERIC;
215 }
216 break;
217 }
218 }
219
220 /* */
221 int i_ret = VLC_EGENERIC;
222
223 /* */
224 config_chain_t cfg_level;
225 memset(&cfg_level, 0, sizeof(cfg_level));
226 cfg_level.psz_name = strdup(psz_option);
227 if( asprintf( &cfg_level.psz_value, "%d", i_level + 1) < 0 )
228 cfg_level.psz_value = NULL;
229 if( !cfg_level.psz_name || !cfg_level.psz_value )
230 goto exit;
231
232 /* Now try chroma format list */
233 for( int i = 0; pi_allowed_chromas[i]; i++ )
234 {
235 const vlc_fourcc_t i_chroma = pi_allowed_chromas[i];
236 if( i_chroma == p_filter->fmt_in.i_codec ||
237 i_chroma == p_filter->fmt_out.i_codec )
238 continue;
239
240 msg_Dbg( p_filter, "Trying to use chroma %4.4s as middle man",
241 (char*)&i_chroma );
242
243 es_format_Copy( &fmt_mid, &p_filter->fmt_in );
244 fmt_mid.i_codec =
245 fmt_mid.video.i_chroma = i_chroma;
246 fmt_mid.video.i_rmask = 0;
247 fmt_mid.video.i_gmask = 0;
248 fmt_mid.video.i_bmask = 0;
249 video_format_FixRgb(&fmt_mid.video);
250
251 i_ret = CreateChain( p_filter, &fmt_mid, &cfg_level );
252 es_format_Clean( &fmt_mid );
253
254 if( i_ret == VLC_SUCCESS )
255 break;
256 }
257
258exit:
259 free( cfg_level.psz_name );
260 free( cfg_level.psz_value );
261 return i_ret;
262}

287/*****************************************************************************
288 *
289 *****************************************************************************/
290static int CreateChain( filter_t *p_parent, es_format_t *p_fmt_mid, config_chain_t *p_cfg )
291{
292 filter_chain_Reset( p_parent->p_sys->p_chain, &p_parent->fmt_in, &p_parent->fmt_out );
293
294 filter_t *p_filter;
295
296 if( p_parent->fmt_in.video.orientation != p_fmt_mid->video.orientation)
297 {
298 p_filter = AppendTransform( p_parent->p_sys->p_chain, &p_parent->fmt_in, p_fmt_mid );
299 }
300 else
301 {
302 p_filter = filter_chain_AppendFilter( p_parent->p_sys->p_chain, NULL, p_cfg, NULL, p_fmt_mid );
303 }
304
305 if( !p_filter )
306 return VLC_EGENERIC;
307
308 //Check if first filter was enough (transform filter most likely):
309 if( es_format_IsSimilar(&p_filter->fmt_out, &p_parent->fmt_out ))
310 return VLC_SUCCESS;
311
312 if( p_fmt_mid->video.orientation != p_parent->fmt_out.video.orientation)
313 {
314 p_filter = AppendTransform( p_parent->p_sys->p_chain, p_fmt_mid, &p_parent->fmt_out );
315 }
316 else
317 {
318 p_filter = filter_chain_AppendFilter( p_parent->p_sys->p_chain, NULL, p_cfg, p_fmt_mid, NULL );
319 }
320
321 if( !p_filter )
322 {
323 //Clean up.
324 filter_chain_Reset( p_parent->p_sys->p_chain, NULL, NULL );
325 return VLC_EGENERIC;
326 }
327
328 return VLC_SUCCESS;
329}

331static filter_t * AppendTransform( filter_chain_t *p_chain, es_format_t *p_fmt1, es_format_t *p_fmt2 )
332{
333 video_transform_t transform = video_format_GetTransform(p_fmt1->video.orientation, p_fmt2->video.orientation);
334
335 const char *type;
336
337 switch ( transform ) {
338
339 case TRANSFORM_R90:
340 type = "90";
341 break;
342 case TRANSFORM_R180:
343 type = "180";
344 break;
345 case TRANSFORM_R270:
346 type = "270";
347 break;
348 case TRANSFORM_HFLIP:
349 type = "hflip";
350 break;
351 case TRANSFORM_VFLIP:
352 type = "vflip";
353 break;
354 case TRANSFORM_TRANSPOSE:
355 type = "transpose";
356 break;
357 case TRANSFORM_ANTI_TRANSPOSE:
358 type = "antitranspose";
359 break;
360 default:
361 type = NULL;
362 break;
363 }
364
365 if( !type )
366 return NULL;
367
368 config_chain_t *cfg;
369 char *name;
370 char config[100];
371 snprintf( config, 100, "transform{type=%s}", type );
372 char *next = config_ChainCreate( &name, &cfg, config );
373
374 filter_t *p_filter = filter_chain_AppendFilter( p_chain, name, cfg, p_fmt1, p_fmt2 );
375
376 config_ChainDestroy(cfg);
377 free(name);
378 free(next);
379
380 return p_filter;
381}

383static void EsFormatMergeSize( es_format_t *p_dst,
384 const es_format_t *p_base,
385 const es_format_t *p_size )
386{
387 es_format_Copy( p_dst, p_base );
388
389 p_dst->video.i_width = p_size->video.i_width;
390 p_dst->video.i_height = p_size->video.i_height;
391
392 p_dst->video.i_visible_width = p_size->video.i_visible_width;
393 p_dst->video.i_visible_height = p_size->video.i_visible_height;
394
395 p_dst->video.i_x_offset = p_size->video.i_x_offset;
396 p_dst->video.i_y_offset = p_size->video.i_y_offset;
397
398 p_dst->video.orientation = p_size->video.orientation;
399}
