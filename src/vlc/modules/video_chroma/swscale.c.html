140/*****************************************************************************
141 * OpenScaler: probe the filter and return score
142 *****************************************************************************/
143static int OpenScaler( vlc_object_t *p_this )
144{
145 filter_t *p_filter = (filter_t*)p_this;
146 filter_sys_t *p_sys;
147
148 int i_sws_mode;
149
150 if( GetParameters( NULL,
151 &p_filter->fmt_in.video,
152 &p_filter->fmt_out.video, 0 ) )
153 return VLC_EGENERIC;
154
155 /* */
156 p_filter->pf_video_filter = Filter;
157 /* Allocate the memory needed to store the decoder's structure */
158 if( ( p_filter->p_sys = p_sys = malloc(sizeof(filter_sys_t)) ) == NULL )
159 return VLC_ENOMEM;
160
161 /* Set CPU capabilities */
162 p_sys->i_cpu_mask = GetSwsCpuMask();
163
164 /* */
165 i_sws_mode = var_CreateGetInteger( p_filter, "swscale-mode" );
166 switch( i_sws_mode )
167 {
168 case 0: p_sys->i_sws_flags = SWS_FAST_BILINEAR; break;
169 case 1: p_sys->i_sws_flags = SWS_BILINEAR; break;
170 case 2: p_sys->i_sws_flags = SWS_BICUBIC; break;
171 case 3: p_sys->i_sws_flags = SWS_X; break;
172 case 4: p_sys->i_sws_flags = SWS_POINT; break;
173 case 5: p_sys->i_sws_flags = SWS_AREA; break;
174 case 6: p_sys->i_sws_flags = SWS_BICUBLIN; break;
175 case 7: p_sys->i_sws_flags = SWS_GAUSS; break;
176 case 8: p_sys->i_sws_flags = SWS_SINC; break;
177 case 9: p_sys->i_sws_flags = SWS_LANCZOS; break;
178 case 10: p_sys->i_sws_flags = SWS_SPLINE; break;
179 default: p_sys->i_sws_flags = SWS_BICUBIC; i_sws_mode = 2; break;
180 }
181
182 p_sys->p_src_filter = NULL;
183 p_sys->p_dst_filter = NULL;
184
185 /* Misc init */
186 p_sys->ctx = NULL;
187 p_sys->ctxA = NULL;
188 p_sys->p_src_a = NULL;
189 p_sys->p_dst_a = NULL;
190 p_sys->p_src_e = NULL;
191 p_sys->p_dst_e = NULL;
192 memset( &p_sys->fmt_in, 0, sizeof(p_sys->fmt_in) );
193 memset( &p_sys->fmt_out, 0, sizeof(p_sys->fmt_out) );
194
195 if( Init( p_filter ) )
196 {
197 if( p_sys->p_src_filter )
198 sws_freeFilter( p_sys->p_src_filter );
199 free( p_sys );
200 return VLC_EGENERIC;
201 }
202
203 msg_Dbg( p_filter, "%ix%i (%ix%i) chroma: %4.4s -> %ix%i (%ix%i) chroma: %4.4s with scaling using %s",
204 p_filter->fmt_in.video.i_visible_width, p_filter->fmt_in.video.i_visible_height,
205 p_filter->fmt_in.video.i_width, p_filter->fmt_in.video.i_height,
206 (char *)&p_filter->fmt_in.video.i_chroma,
207 p_filter->fmt_out.video.i_visible_width, p_filter->fmt_out.video.i_visible_height,
208 p_filter->fmt_out.video.i_width, p_filter->fmt_out.video.i_height,
209 (char *)&p_filter->fmt_out.video.i_chroma,
210 ppsz_mode_descriptions[i_sws_mode] );
211
212 return VLC_SUCCESS;
213}

232static int GetSwsCpuMask(void)
233{
234 int i_sws_cpu = 0;
235
236#if defined(__i386__) || defined(__x86_64__)
237 if( vlc_CPU_MMX() )
238 i_sws_cpu |= SWS_CPU_CAPS_MMX;
239#if (LIBSWSCALE_VERSION_INT >= ((0<<16)+(5<<8)+0))
240 if( vlc_CPU_MMXEXT() )
241 i_sws_cpu |= SWS_CPU_CAPS_MMX2;
242#endif
243 if( vlc_CPU_3dNOW() )
244 i_sws_cpu |= SWS_CPU_CAPS_3DNOW;
245#elif defined(__ppc__) || defined(__ppc64__) || defined(__powerpc__)
246 if( vlc_CPU_ALTIVEC() )
247 i_sws_cpu |= SWS_CPU_CAPS_ALTIVEC;
248#endif
249
250 return i_sws_cpu;
251}

253static void FixParameters( int *pi_fmt, bool *pb_has_a, bool *pb_swap_uv, vlc_fourcc_t fmt )
254{
255 switch( fmt )
256 {
257 case VLC_CODEC_YUV422A:
258 *pi_fmt = PIX_FMT_YUV422P;
259 *pb_has_a = true;
260 break;
261 case VLC_CODEC_YUV420A:
262 *pi_fmt = PIX_FMT_YUV420P;
263 *pb_has_a = true;
264 break;
265 case VLC_CODEC_YUVA:
266 *pi_fmt = PIX_FMT_YUV444P;
267 *pb_has_a = true;
268 break;
269 case VLC_CODEC_RGBA:
270 *pi_fmt = PIX_FMT_BGR32;
271 *pb_has_a = true;
272 break;
273 case VLC_CODEC_ARGB:
274 *pi_fmt = PIX_FMT_BGR32_1;
275 *pb_has_a = true;
276 break;
277 case VLC_CODEC_BGRA:
278 *pi_fmt = PIX_FMT_RGB32;
279 *pb_has_a = true;
280 break;
281 case VLC_CODEC_YV12:
282 *pi_fmt = PIX_FMT_YUV420P;
283 *pb_swap_uv = true;
284 break;
285 case VLC_CODEC_YV9:
286 *pi_fmt = PIX_FMT_YUV410P;
287 *pb_swap_uv = true;
288 break;
289 default:
290 break;
291 }

294static int GetParameters( ScalerConfiguration *p_cfg,
295 const video_format_t *p_fmti,
296 const video_format_t *p_fmto,
297 int i_sws_flags_default )
298{
299 int i_fmti = -1;
300 int i_fmto = -1;
301
302 bool b_has_ai = false;
303 bool b_has_ao = false;
304 int i_sws_flags = i_sws_flags_default;
305 bool b_swap_uvi = false;
306 bool b_swap_uvo = false;
307
308 GetFfmpegChroma( &i_fmti, p_fmti );
309 GetFfmpegChroma( &i_fmto, p_fmto );
310
311 if( p_fmti->i_chroma == p_fmto->i_chroma )
312 {
313 if( p_fmti->i_chroma == VLC_CODEC_YUVP && ALLOW_YUVP )
314 {
315 i_fmti = i_fmto = PIX_FMT_GRAY8;
316 i_sws_flags = SWS_POINT;
317 }
318 }
319
320 FixParameters( &i_fmti, &b_has_ai, &b_swap_uvi, p_fmti->i_chroma );
321 FixParameters( &i_fmto, &b_has_ao, &b_swap_uvo, p_fmto->i_chroma );
322
323#if !defined (__ANDROID__) && !defined(TARGET_OS_IPHONE)
324 /* FIXME TODO removed when ffmpeg is fixed
325 * Without SWS_ACCURATE_RND the quality is really bad for some conversions */
326 switch( i_fmto )
327 {
328 case PIX_FMT_ARGB:
329 case PIX_FMT_RGBA:
330 case PIX_FMT_ABGR:
331 i_sws_flags |= SWS_ACCURATE_RND;
332 break;
333 }
334#endif
335
336 if( p_cfg )
337 {
338 p_cfg->i_fmti = i_fmti;
339 p_cfg->i_fmto = i_fmto;
340 p_cfg->b_has_a = b_has_ai && b_has_ao;
341 p_cfg->b_add_a = (!b_has_ai) && b_has_ao;
342 p_cfg->b_copy = i_fmti == i_fmto &&
343 p_fmti->i_visible_width == p_fmto->i_visible_width &&
344 p_fmti->i_visible_height == p_fmto->i_visible_height;
345 p_cfg->b_swap_uvi = b_swap_uvi;
346 p_cfg->b_swap_uvo = b_swap_uvo;
347 p_cfg->i_sws_flags = i_sws_flags;
348 }
349
350 if( i_fmti < 0 || i_fmto < 0 )
351 return VLC_EGENERIC;
352
353 return VLC_SUCCESS;
354}

356static int Init( filter_t *p_filter )
357{
358 filter_sys_t *p_sys = p_filter->p_sys;
359 const video_format_t *p_fmti = &p_filter->fmt_in.video;
360 video_format_t *p_fmto = &p_filter->fmt_out.video;
361
362 if( p_fmti->orientation != p_fmto->orientation )
363 return VLC_EGENERIC;
364
365 if( video_format_IsSimilar( p_fmti, &p_sys->fmt_in ) &&
366 video_format_IsSimilar( p_fmto, &p_sys->fmt_out ) &&
367 p_sys->ctx )
368 {
369 return VLC_SUCCESS;
370 }
371
372 Clean( p_filter );
373
374 /* Init with new parameters */
375 ScalerConfiguration cfg;
376 if( GetParameters( &cfg, p_fmti, p_fmto, p_sys->i_sws_flags ) )
377 {
378 msg_Err( p_filter, "format not supported" );
379 return VLC_EGENERIC;
380 }
381 if( p_fmti->i_visible_width <= 0 || p_fmti->i_visible_height <= 0 ||
382 p_fmto->i_visible_width <= 0 || p_fmto->i_visible_height <= 0 )
383 {
384 msg_Err( p_filter, "invalid scaling: %ix%i -> %ix%i",
385 p_fmti->i_visible_width, p_fmti->i_visible_height,
386 p_fmto->i_visible_width, p_fmto->i_visible_height);
387 return VLC_EGENERIC;
388 }
389
390 /* swscale does not like too small width */
391 p_sys->i_extend_factor = 1;
392 while( __MIN( p_fmti->i_visible_width, p_fmto->i_visible_width ) * p_sys->i_extend_factor < MINIMUM_WIDTH)
393 p_sys->i_extend_factor++;
394
395 const unsigned i_fmti_visible_width = p_fmti->i_visible_width * p_sys->i_extend_factor;
396 const unsigned i_fmto_visible_width = p_fmto->i_visible_width * p_sys->i_extend_factor;
397 for( int n = 0; n < (cfg.b_has_a ? 2 : 1); n++ )
398 {
399 const int i_fmti = n == 0 ? cfg.i_fmti : PIX_FMT_GRAY8;
400 const int i_fmto = n == 0 ? cfg.i_fmto : PIX_FMT_GRAY8;
401 struct SwsContext *ctx;
402
403 ctx = sws_getContext( i_fmti_visible_width, p_fmti->i_visible_height, i_fmti,
404 i_fmto_visible_width, p_fmto->i_visible_height, i_fmto,
405 cfg.i_sws_flags | p_sys->i_cpu_mask,
406 p_sys->p_src_filter, p_sys->p_dst_filter, 0 );
407 if( n == 0 )
408 p_sys->ctx = ctx;
409 else
410 p_sys->ctxA = ctx;
411 }
412 if( p_sys->ctxA )
413 {
414 p_sys->p_src_a = picture_New( VLC_CODEC_GREY, i_fmti_visible_width, p_fmti->i_visible_height, 0, 1 );
415 p_sys->p_dst_a = picture_New( VLC_CODEC_GREY, i_fmto_visible_width, p_fmto->i_visible_height, 0, 1 );
416 }
417 if( p_sys->i_extend_factor != 1 )
418 {
419 p_sys->p_src_e = picture_New( p_fmti->i_chroma, i_fmti_visible_width, p_fmti->i_visible_height, 0, 1 );
420 p_sys->p_dst_e = picture_New( p_fmto->i_chroma, i_fmto_visible_width, p_fmto->i_visible_height, 0, 1 );
421
422 if( p_sys->p_src_e )
423 memset( p_sys->p_src_e->p[0].p_pixels, 0, p_sys->p_src_e->p[0].i_pitch * p_sys->p_src_e->p[0].i_lines );
424 if( p_sys->p_dst_e )
425 memset( p_sys->p_dst_e->p[0].p_pixels, 0, p_sys->p_dst_e->p[0].i_pitch * p_sys->p_dst_e->p[0].i_lines );
426 }
427
428 if( !p_sys->ctx ||
429 ( cfg.b_has_a && ( !p_sys->ctxA || !p_sys->p_src_a || !p_sys->p_dst_a ) ) ||
430 ( p_sys->i_extend_factor != 1 && ( !p_sys->p_src_e || !p_sys->p_dst_e ) ) )
431 {
432 msg_Err( p_filter, "could not init SwScaler and/or allocate memory" );
433 Clean( p_filter );
434 return VLC_EGENERIC;
435 }
436
437 if (p_filter->b_allow_fmt_out_change)
438 {
439 /*
440 * If the transformation is not homothetic we must modify the
441 * aspect ratio of the output format in order to have the
442 * output picture displayed correctly and not stretched
443 * horizontally or vertically.
444 * WARNING: this is a hack, ideally this should not be needed
445 * and the vout should update its video format instead.
446 */
447 unsigned i_sar_num = p_fmti->i_sar_num * p_fmti->i_visible_width;
448 unsigned i_sar_den = p_fmti->i_sar_den * p_fmto->i_visible_width;
449 vlc_ureduce(&i_sar_num, &i_sar_den, i_sar_num, i_sar_den, 65536);
450 i_sar_num *= p_fmto->i_visible_height;
451 i_sar_den *= p_fmti->i_visible_height;
452 vlc_ureduce(&i_sar_num, &i_sar_den, i_sar_num, i_sar_den, 65536);
453 p_fmto->i_sar_num = i_sar_num;
454 p_fmto->i_sar_den = i_sar_den;
455 }
456
457 p_sys->b_add_a = cfg.b_add_a;
458 p_sys->b_copy = cfg.b_copy;
459 p_sys->fmt_in = *p_fmti;
460 p_sys->fmt_out = *p_fmto;
461 p_sys->b_swap_uvi = cfg.b_swap_uvi;
462 p_sys->b_swap_uvo = cfg.b_swap_uvo;
463
464#if 0
465 msg_Dbg( p_filter, "%ix%i (%ix%i) chroma: %4.4s -> %ix%i (%ix%i) chroma: %4.4s extend by %d",
466 p_fmti->i_visible_width, p_fmti->i_visible_height, p_fmti->i_width, p_fmti->i_height, (char *)&p_fmti->i_chroma,
467 p_fmto->i_visible_width, p_fmto->i_visible_height, p_fmto->i_width, p_fmto->i_height, (char *)&p_fmto->i_chroma,
468 p_sys->i_extend_factor );
469#endif
470 return VLC_SUCCESS;
471}
472static void Clean( filter_t *p_filter )
473{
474 filter_sys_t *p_sys = p_filter->p_sys;
475
476 if( p_sys->p_src_e )
477 picture_Release( p_sys->p_src_e );
478 if( p_sys->p_dst_e )
479 picture_Release( p_sys->p_dst_e );
480
481 if( p_sys->p_src_a )
482 picture_Release( p_sys->p_src_a );
483 if( p_sys->p_dst_a )
484 picture_Release( p_sys->p_dst_a );
485
486 if( p_sys->ctxA )
487 sws_freeContext( p_sys->ctxA );
488
489 if( p_sys->ctx )
490 sws_freeContext( p_sys->ctx );
491
492 /* We have to set it to null has we call be called again :( */
493 p_sys->ctx = NULL;
494 p_sys->ctxA = NULL;
495 p_sys->p_src_a = NULL;
496 p_sys->p_dst_a = NULL;
497 p_sys->p_src_e = NULL;
498 p_sys->p_dst_e = NULL;
499}

501static void GetPixels( uint8_t *pp_pixel[4], int pi_pitch[4],
502 const picture_t *p_picture,
503 int i_plane_start, int i_plane_count,
504 bool b_swap_uv )
505{
506 assert( !b_swap_uv || i_plane_count >= 3 );
507 int n;
508 for( n = 0; n < __MIN(i_plane_count, p_picture->i_planes-i_plane_start ); n++ )
509 {
510 const int nd = ( b_swap_uv && n >= 1 && n <= 2 ) ? (3 - n) : n;
511 pp_pixel[nd] = p_picture->p[i_plane_start+n].p_pixels;
512 pi_pitch[nd] = p_picture->p[i_plane_start+n].i_pitch;
513 }
514 for( ; n < 4; n++ )
515 {
516 pp_pixel[n] = NULL;
517 pi_pitch[n] = 0;
518 }
519}

521static void ExtractA( picture_t *p_dst, const picture_t *restrict p_src,
522 unsigned i_width, unsigned i_height, unsigned offset )
523{
524 plane_t *d = &p_dst->p[0];
525 const plane_t *s = &p_src->p[0];
526
527 for( unsigned y = 0; y < i_height; y++ )
528 for( unsigned x = 0; x < i_width; x++ )
529 d->p_pixels[y*d->i_pitch+x] = s->p_pixels[y*s->i_pitch+4*x+offset];
530}

532static void InjectA( picture_t *p_dst, const picture_t *restrict p_src,
533 unsigned i_width, unsigned i_height, unsigned offset )
534{
535 plane_t *d = &p_dst->p[0];
536 const plane_t *s = &p_src->p[0];
537
538 for( unsigned y = 0; y < i_height; y++ )
539 for( unsigned x = 0; x < i_width; x++ )
540 d->p_pixels[y*d->i_pitch+4*x+offset] = s->p_pixels[y*s->i_pitch+x];
541}

543static void FillA( plane_t *d, unsigned i_offset )
544{
545 for( int y = 0; y < d->i_visible_lines; y++ )
546 for( int x = 0; x < d->i_visible_pitch; x += d->i_pixel_pitch )
547 d->p_pixels[y*d->i_pitch+x+i_offset] = 0xff;
548}

550static void CopyPad( picture_t *p_dst, const picture_t *p_src )
551{
552 picture_Copy( p_dst, p_src );
553 for( int n = 0; n < p_dst->i_planes; n++ )
554 {
555 const plane_t *s = &p_src->p[n];
556 plane_t *d = &p_dst->p[n];
557
558 for( int y = 0; y < s->i_lines; y++ )
559 {
560 for( int x = s->i_visible_pitch; x < d->i_visible_pitch; x += s->i_pixel_pitch )
561 memcpy( &d->p_pixels[y*d->i_pitch + x], &d->p_pixels[y*d->i_pitch + s->i_visible_pitch - s->i_pixel_pitch], s->i_pixel_pitch );
562 }
563 }

566static void SwapUV( picture_t *p_dst, const picture_t *p_src )
567{
568 picture_t tmp = *p_src;
569 tmp.p[1] = p_src->p[2];
570 tmp.p[2] = p_src->p[1];
571
572 picture_CopyPixels( p_dst, &tmp );
573}
574static void Convert( filter_t *p_filter, struct SwsContext *ctx,
575 picture_t *p_dst, picture_t *p_src, int i_height, int i_plane_start, int i_plane_count,
576 bool b_swap_uvi, bool b_swap_uvo )
577{
578 uint8_t palette[AVPALETTE_SIZE];
579
580 uint8_t *src[4]; int src_stride[4];
581 uint8_t *dst[4]; int dst_stride[4];
582
583 GetPixels( src, src_stride, p_src, i_plane_start, i_plane_count, b_swap_uvi );
584 if( p_filter->fmt_in.video.i_chroma == VLC_CODEC_RGBP )
585 {
586 memset( palette, 0, sizeof(palette) );
587 if( p_filter->fmt_in.video.p_palette )
588 memcpy( palette, p_filter->fmt_in.video.p_palette->palette,
589 __MIN( sizeof(video_palette_t), AVPALETTE_SIZE ) );
590 src[1] = palette;
591 src_stride[1] = 4;
592 }
593
594 GetPixels( dst, dst_stride, p_dst, i_plane_start, i_plane_count, b_swap_uvo );
595
596#if LIBSWSCALE_VERSION_INT >= ((0<<16)+(5<<8)+0)
597 sws_scale( ctx, src, src_stride, 0, i_height,
598 dst, dst_stride );
599#else
600 sws_scale_ordered( ctx, src, src_stride, 0, i_height,
601 dst, dst_stride );
602#endif
603}

605/****************************************************************************
606 * Filter: the whole thing
607 ****************************************************************************
608 * This function is called just after the thread is launched.
609 ****************************************************************************/
610static picture_t *Filter( filter_t *p_filter, picture_t *p_pic )
611{
612 filter_sys_t *p_sys = p_filter->p_sys;
613 const video_format_t *p_fmti = &p_filter->fmt_in.video;
614 const video_format_t *p_fmto = &p_filter->fmt_out.video;
615 picture_t *p_pic_dst;
616
617 /* Check if format properties changed */
618 if( Init( p_filter ) )
619 {
620 picture_Release( p_pic );
621 return NULL;
622 }
623
624 /* Request output picture */
625 p_pic_dst = filter_NewPicture( p_filter );
626 if( !p_pic_dst )
627 {
628 picture_Release( p_pic );
629 return NULL;
630 }
631
632 /* */
633 picture_t *p_src = p_pic;
634 picture_t *p_dst = p_pic_dst;
635 if( p_sys->i_extend_factor != 1 )
636 {
637 p_src = p_sys->p_src_e;
638 p_dst = p_sys->p_dst_e;
639
640 CopyPad( p_src, p_pic );
641 }
642
643 if( p_sys->b_copy && p_sys->b_swap_uvi == p_sys->b_swap_uvo )
644 picture_CopyPixels( p_dst, p_src );
645 else if( p_sys->b_copy )
646 SwapUV( p_dst, p_src );
647 else
648 Convert( p_filter, p_sys->ctx, p_dst, p_src, p_fmti->i_visible_height, 0, 3,
649 p_sys->b_swap_uvi, p_sys->b_swap_uvo );
650 if( p_sys->ctxA )
651 {
652 /* We extract the A plane to rescale it, and then we reinject it. */
653 if( p_fmti->i_chroma == VLC_CODEC_RGBA || p_fmti->i_chroma == VLC_CODEC_BGRA )
654 ExtractA( p_sys->p_src_a, p_src, p_fmti->i_visible_width * p_sys->i_extend_factor, p_fmti->i_visible_height, OFFSET_A );
655 else if( p_fmti->i_chroma == VLC_CODEC_ARGB )
656 ExtractA( p_sys->p_src_a, p_src, p_fmti->i_visible_width * p_sys->i_extend_factor, p_fmti->i_visible_height, 0 );
657 else
658 plane_CopyPixels( p_sys->p_src_a->p, p_src->p+A_PLANE );
659
660 Convert( p_filter, p_sys->ctxA, p_sys->p_dst_a, p_sys->p_src_a, p_fmti->i_visible_height, 0, 1, false, false );
661 if( p_fmto->i_chroma == VLC_CODEC_RGBA || p_fmto->i_chroma == VLC_CODEC_BGRA )
662 InjectA( p_dst, p_sys->p_dst_a, p_fmto->i_visible_width * p_sys->i_extend_factor, p_fmto->i_visible_height, OFFSET_A );
663 else if( p_fmto->i_chroma == VLC_CODEC_ARGB )
664 InjectA( p_dst, p_sys->p_dst_a, p_fmto->i_visible_width * p_sys->i_extend_factor, p_fmto->i_visible_height, 0 );
665 else
666 plane_CopyPixels( p_dst->p+A_PLANE, p_sys->p_dst_a->p );
667 }
668 else if( p_sys->b_add_a )
669 {
670 /* We inject a complete opaque alpha plane */
671 if( p_fmto->i_chroma == VLC_CODEC_RGBA || p_fmto->i_chroma == VLC_CODEC_BGRA )
672 FillA( &p_dst->p[0], OFFSET_A );
673 else if( p_fmto->i_chroma == VLC_CODEC_ARGB )
674 FillA( &p_dst->p[0], 0 );
675 else
676 FillA( &p_dst->p[A_PLANE], 0 );
677 }
678
679 if( p_sys->i_extend_factor != 1 )
680 {
681 picture_CopyPixels( p_pic_dst, p_dst );
682 }
683
684 picture_CopyProperties( p_pic_dst, p_pic );
685 picture_Release( p_pic );
686 return p_pic_dst;
687}
