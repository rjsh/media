125/*
126 * This is a global list of existing decoders.
127 * The reason for this list is that:
128 * - I want to be able to reconfigure Tiger when user prefs change
129 * - User prefs are variables which are not specific to a decoder (eg, if
130 * there are several decoders, there will still be one set of variables)
131 * - A callback set on those will not be passed a pointer to the decoder
132 * (as the decoder isn't known, and there could be multiple ones)
133 * - Creating variables in the decoder will create different ones, with
134 * values copied from the relevant user pref, so a callback on those
135 * won't be called when the user pref is changed
136 * Therefore, each decoder will register/unregister itself with this list,
137 * callbacks will be set for the user prefs, and these will in turn walk
138 * through this list and tell each decoder to update the relevant variable.
139 * HOWEVER.
140 * VLC's variable system is still in my way as I can't get the value of
141 * the user pref variables at decoder start *unless* I create my own vars
142 * which inherit from the user ones, but those are utterly useless after
143 * that first use, since they'll be detached from the ones the user can
144 * change. So, I create them, read their values, and then destroy them.
145 */
146static vlc_mutex_t kate_decoder_list_mutex = VLC_STATIC_MUTEX;
147static size_t kate_decoder_list_size = 0;
148static decoder_t **kate_decoder_list = NULL;

334/*****************************************************************************
335 * OpenDecoder: probe the decoder and return score
336 *****************************************************************************
337 * Tries to launch a decoder and return score so that the interface is able
338 * to chose.
339 *****************************************************************************/
340static int OpenDecoder( vlc_object_t *p_this )
341{
342 decoder_t *p_dec = (decoder_t*)p_this;
343 decoder_sys_t *p_sys;
344
345 if( p_dec->fmt_in.i_codec != VLC_CODEC_KATE )
346 {
347 return VLC_EGENERIC;
348 }
349
350 msg_Dbg( p_dec, "kate: OpenDecoder");
351
352 /* Set callbacks */
353 p_dec->pf_decode_sub = (subpicture_t *(*)(decoder_t *, block_t **))
354 DecodeBlock;
355 p_dec->pf_packetize = (block_t *(*)(decoder_t *, block_t **))
356 DecodeBlock;
357
358 /* Allocate the memory needed to store the decoder's structure */
359 if( ( p_dec->p_sys = p_sys = malloc(sizeof(*p_sys)) ) == NULL )
360 return VLC_ENOMEM;
361
362 vlc_mutex_init( &p_sys->lock );
363 p_sys->i_refcount = 0;
364 DecSysHold( p_sys );
365
366 /* init of p_sys */
367#ifdef ENABLE_PACKETIZER
368 p_sys->b_packetizer = false;
369#endif
370 p_sys->b_ready = false;
371 p_sys->i_pts =
372 p_sys->i_max_stop = VLC_TS_INVALID;
373
374 kate_comment_init( &p_sys->kc );
375 kate_info_init( &p_sys->ki );
376
377 p_sys->b_has_headers = false;
378
379 /* retrieve options */
380 p_sys->b_formatted = var_CreateGetBool( p_dec, "kate-formatted" );
381
382 vlc_mutex_lock( &kate_decoder_list_mutex );
383
384#ifdef HAVE_TIGER
385
386 p_sys->b_use_tiger = var_CreateGetBool( p_dec, "kate-use-tiger" );
387
388 p_sys->p_tr = NULL;
389
390 /* get initial value of configuration */
391 p_sys->i_tiger_default_font_color = GetTigerColor( p_dec, "kate-tiger-default-font" );
392 p_sys->i_tiger_default_background_color = GetTigerColor( p_dec, "kate-tiger-default-background" );
393 p_sys->e_tiger_default_font_effect = GetTigerInteger( p_dec, "kate-tiger-default-font-effect" );
394 p_sys->f_tiger_default_font_effect_strength = GetTigerFloat( p_dec, "kate-tiger-default-font-effect-strength" );
395 p_sys->psz_tiger_default_font_desc = GetTigerString( p_dec, "kate-tiger-default-font-desc" );
396 p_sys->f_tiger_quality = GetTigerFloat( p_dec, "kate-tiger-quality" );
397
398 if( p_sys->b_use_tiger )
399 {
400 int i_ret = tiger_renderer_create( &p_sys->p_tr );
401 if( i_ret < 0 )
402 {
403 msg_Warn ( p_dec, "Failed to create Tiger renderer, falling back to basic rendering" );
404 p_sys->p_tr = NULL;
405 p_sys->b_use_tiger = false;
406 }
407 else {
408 CHECK_TIGER_RET( tiger_renderer_set_surface_clear_color( p_sys->p_tr, 1, 0, 0, 0, 0 ) );
409
410 UpdateTigerFontEffect( p_dec );
411 UpdateTigerFontColor( p_dec );
412 UpdateTigerBackgroundColor( p_dec );
413 UpdateTigerQuality( p_dec );
414 UpdateTigerFontDesc( p_dec );
415 }
416 }
417
418#else
419
420 p_sys->b_use_tiger = false;
421
422#endif
423
424 es_format_Init( &p_dec->fmt_out, SPU_ES, 0 );
425
426 /* add the decoder to the global list */
427 decoder_t **list = realloc( kate_decoder_list, (kate_decoder_list_size+1) * sizeof( *list ));
428 if( list )
429 {
430 list[ kate_decoder_list_size++ ] = p_dec;
431 kate_decoder_list = list;
432 }
433
434 vlc_mutex_unlock( &kate_decoder_list_mutex );
435
436 return VLC_SUCCESS;
437}

456/****************************************************************************
457 * DecodeBlock: the whole thing
458 ****************************************************************************
459 * This function must be fed with kate packets.
460 ****************************************************************************/
461static subpicture_t *DecodeBlock( decoder_t *p_dec, block_t **pp_block )
462{
463 decoder_sys_t *p_sys = p_dec->p_sys;
464 block_t *p_block;
465 kate_packet kp;
466
467 if( !pp_block || !*pp_block )
468 return NULL;
469
470 p_block = *pp_block;
471
472 if( p_block->i_flags & (BLOCK_FLAG_DISCONTINUITY|BLOCK_FLAG_CORRUPTED) )
473 {
474#ifdef HAVE_TIGER
475 if( p_block->i_flags & BLOCK_FLAG_DISCONTINUITY)
476 {
477 /* Hmm, should we wait before flushing the renderer ? I think not, but not certain... */
478 vlc_mutex_lock( &p_sys->lock );
479 tiger_renderer_seek( p_sys->p_tr, 0 );
480 vlc_mutex_unlock( &p_sys->lock );
481 }
482#endif
483 p_sys->i_max_stop = VLC_TS_INVALID;
484 block_Release( p_block );
485 return NULL;
486 }
487
488 /* Block to Kate packet */
489 kate_packet_wrap(&kp, p_block->i_buffer, p_block->p_buffer);
490
491 if( !p_sys->b_has_headers )
492 {
493 if( ProcessHeaders( p_dec ) )
494 {
495 block_Release( *pp_block );
496 return NULL;
497 }
498 p_sys->b_has_headers = true;
499 }
500
501 return ProcessPacket( p_dec, &kp, pp_block );
502}

504/*****************************************************************************
505 * ProcessHeaders: process Kate headers.
506 *****************************************************************************/
507static int ProcessHeaders( decoder_t *p_dec )
508{
509 decoder_sys_t *p_sys = p_dec->p_sys;
510 kate_packet kp;
511
512 unsigned pi_size[XIPH_MAX_HEADER_COUNT];
513 void *pp_data[XIPH_MAX_HEADER_COUNT];
514 unsigned i_count;
515 if( xiph_SplitHeaders( pi_size, pp_data, &i_count,
516 p_dec->fmt_in.i_extra, p_dec->fmt_in.p_extra) )
517 return VLC_EGENERIC;
518
519 if( i_count < 1 )
520 return VLC_EGENERIC;
521
522 /* Take care of the initial Kate header */
523 kp.nbytes = pi_size[0];
524 kp.data = pp_data[0];
525 int i_ret = kate_decode_headerin( &p_sys->ki, &p_sys->kc, &kp );
526 if( i_ret < 0 )
527 {
528 msg_Err( p_dec, "this bitstream does not contain Kate data (%d)", i_ret );
529 return VLC_EGENERIC;
530 }
531
532 msg_Dbg( p_dec, "%s %s text, granule rate %f, granule shift %d",
533 p_sys->ki.language, p_sys->ki.category,
534 (double)p_sys->ki.gps_numerator/p_sys->ki.gps_denominator,
535 p_sys->ki.granule_shift);
536
537 /* parse all remaining header packets */
538 for( unsigned i_headeridx = 1; i_headeridx < i_count; i_headeridx++ )
539 {
540 kp.nbytes = pi_size[i_headeridx];
541 kp.data = pp_data[i_headeridx];
542 i_ret = kate_decode_headerin( &p_sys->ki, &p_sys->kc, &kp );
543 if( i_ret < 0 )
544 {
545 msg_Err( p_dec, "Kate header %d is corrupted: %d", i_headeridx, i_ret );
546 return VLC_EGENERIC;
547 }
548
549 /* header 1 is comments */
550 if( i_headeridx == 1 )
551 {
552 ParseKateComments( p_dec );
553 }
554 }
555
556#ifdef ENABLE_PACKETIZER
557 if( !p_sys->b_packetizer )
558#endif
559 {
560 /* We have all the headers, initialize decoder */
561 msg_Dbg( p_dec, "we have all headers, initialize libkate for decoding" );
562 i_ret = kate_decode_init( &p_sys->k, &p_sys->ki );
563 if (i_ret < 0)
564 {
565 msg_Err( p_dec, "Kate failed to initialize for decoding: %d", i_ret );
566 return VLC_EGENERIC;
567 }
568 p_sys->b_ready = true;
569 }
570#ifdef ENABLE_PACKETIZER
571 else
572 {
573 p_dec->fmt_out.i_extra = p_dec->fmt_in.i_extra;
574 p_dec->fmt_out.p_extra = xrealloc( p_dec->fmt_out.p_extra,
575 p_dec->fmt_out.i_extra );
576 memcpy( p_dec->fmt_out.p_extra,
577 p_dec->fmt_in.p_extra, p_dec->fmt_out.i_extra );
578 }
579#endif
580
581 return VLC_SUCCESS;
582}

584/*****************************************************************************
585 * ProcessPacket: processes a kate packet.
586 *****************************************************************************/
587static subpicture_t *ProcessPacket( decoder_t *p_dec, kate_packet *p_kp,
588 block_t **pp_block )
589{
590 decoder_sys_t *p_sys = p_dec->p_sys;
591 block_t *p_block = *pp_block;
592 subpicture_t *p_buf = NULL;
593
594 /* Date management */
595 if( p_block->i_pts > VLC_TS_INVALID && p_block->i_pts != p_sys->i_pts )
596 {
597 p_sys->i_pts = p_block->i_pts;
598 }
599
600 *pp_block = NULL; /* To avoid being fed the same packet again */
601
602#ifdef ENABLE_PACKETIZER
603 if( p_sys->b_packetizer )
604 {
605 /* Date management */
606 p_block->i_dts = p_block->i_pts = p_sys->i_pts;
607
608 if( p_sys->i_headers >= p_sys->i_num_headers )
609 p_block->i_length = p_sys->i_pts - p_block->i_pts;
610 else
611 p_block->i_length = 0;
612
613 p_buf = p_block;
614 }
615 else
616#endif
617 {
618 p_buf = DecodePacket( p_dec, p_kp, p_block );
619
620 if( p_block )
621 block_Release( p_block );
622 }
623
624 return p_buf;
625}

627/* nicked off blend.c */
628static inline void rgb_to_yuv( uint8_t *y, uint8_t *u, uint8_t *v,
629 int r, int g, int b )
630{
631 *y = ( ( ( 66 * r + 129 * g + 25 * b + 128 ) >> 8 ) + 16 );
632 *u = ( ( -38 * r - 74 * g + 112 * b + 128 ) >> 8 ) + 128 ;
633 *v = ( ( 112 * r - 94 * g - 18 * b + 128 ) >> 8 ) + 128 ;
634}

636/*
637 This retrieves the size of the video.
638 The best case is when the original video size is known, as we can then
639 scale images to match. In this case, since VLC autoscales, we want to
640 return the original size and let VLC scale everything.
641 if the original size is not known, then VLC can't resize, so we return
642 the size of the incoming video. If sizes in the Kate stream are in
643 relative units, it works fine. If they are absolute, you get what you
644 ask for. Images aren't rescaled.
645*/
646static void GetVideoSize( decoder_t *p_dec, int *w, int *h )
647{
648 /* searching for vout to get its size is frowned upon, so we don't and
649 use a default size if the original canvas size is not specified. */
650 decoder_sys_t *p_sys = p_dec->p_sys;
651 if( p_sys->ki.original_canvas_width > 0 && p_sys->ki.original_canvas_height > 0 )
652 {
653 *w = p_sys->ki.original_canvas_width;
654 *h = p_sys->ki.original_canvas_height;
655 msg_Dbg( p_dec, "original canvas %zu %zu",
656 p_sys->ki.original_canvas_width, p_sys->ki.original_canvas_height );
657 }
658 else
659 {
660 /* nothing, leave defaults */
661 msg_Dbg( p_dec, "original canvas size unknown");
662 }
663}

692static void SetupText( decoder_t *p_dec, subpicture_t *p_spu, const kate_event *ev )
693{
694 decoder_sys_t *p_sys = p_dec->p_sys;
695
696 if( ev->text_encoding != kate_utf8 )
697 {
698 msg_Warn( p_dec, "Text isn't UTF-8, unsupported, ignored" );
699 return;
700 }
701
702 switch( ev->text_markup_type )
703 {
704 case kate_markup_none:
705 p_spu->p_region->psz_text = strdup( ev->text ); /* no leak, this actually gets killed by the core */
706 break;
707 case kate_markup_simple:
708 if( p_sys->b_formatted )
709 {
710 /* the HTML renderer expects a top level text tag pair */
711 char *buffer = NULL;
712 if( asprintf( &buffer, "<text>%s</text>", ev->text ) >= 0 )
713 {
714 p_spu->p_region->psz_html = buffer;
715 }
716 break;
717 }
718 /* if not formatted, we fall through */
719 default:
720 /* we don't know about this one, so remove markup and display as text */
721 {
722 char *copy = strdup( ev->text );
723 size_t len0 = strlen( copy ) + 1;
724 kate_text_remove_markup( ev->text_encoding, copy, &len0 );
725 p_spu->p_region->psz_text = copy;
726 }
727 break;
728 }
729}

733static void TigerDestroySubpicture( subpicture_t *p_subpic )
734{
735 DecSysRelease( p_subpic->updater.p_sys->p_dec_sys );
736 free( p_subpic->updater.p_sys );

738/*
739 * We get premultiplied alpha, but VLC doesn't expect this, so we demultiply
740 * alpha to avoid double multiply (and thus thinner text than we should)).
741 * Best would be to have VLC be able to handle premultiplied alpha, as it
742 * would be faster to blend as well.
743 *
744 * Also, we flip color components around for big endian machines, as Tiger
745 * outputs ARGB or ABGR (the one we selected here) in host endianness.
746 */
747static void PostprocessTigerImage( plane_t *p_plane, unsigned int i_width )
748{
749 PROFILE_START( tiger_renderer_postprocess );
750 int y;
751 for( y=0; y<p_plane->i_lines; ++y )
752 {
753 uint8_t *p_line = (uint8_t*)(p_plane->p_pixels + y*p_plane->i_pitch);
754 unsigned int x;
755 for( x=0; x<i_width; ++x )
756 {
757 uint8_t *p_pixel = p_line+x*4;
758#ifdef WORDS_BIGENDIAN
759 uint8_t a = p_pixel[0];
760#else
761 uint8_t a = p_pixel[3];
762#endif
763 if( a )
764 {
765#ifdef WORDS_BIGENDIAN
766 uint8_t tmp = p_pixel[2];
767 p_pixel[0] = clip_uint8_vlc((p_pixel[3] * 255 + a / 2) / a);
768 p_pixel[3] = a;
769 p_pixel[2] = clip_uint8_vlc((p_pixel[1] * 255 + a / 2) / a);
770 p_pixel[1] = clip_uint8_vlc((tmp * 255 + a / 2) / a);
771#else
772 p_pixel[0] = clip_uint8_vlc((p_pixel[0] * 255 + a / 2) / a);
773 p_pixel[1] = clip_uint8_vlc((p_pixel[1] * 255 + a / 2) / a);
774 p_pixel[2] = clip_uint8_vlc((p_pixel[2] * 255 + a / 2) / a);
775#endif
776 }
777 else
778 {
779 p_pixel[0] = 0;
780 p_pixel[1] = 0;
781 p_pixel[2] = 0;
782 p_pixel[3] = 0;
783 }
784 }
785 }
786 PROFILE_STOP( tiger_renderer_postprocess );
787}

789static int TigerValidateSubpicture( subpicture_t *p_subpic,
790 bool b_fmt_src, const video_format_t *p_fmt_src,
791 bool b_fmt_dst, const video_format_t *p_fmt_dst,
792 mtime_t ts )
793{
794 VLC_UNUSED(p_fmt_src); VLC_UNUSED(p_fmt_dst);
795
796 decoder_sys_t *p_sys = p_subpic->updater.p_sys->p_dec_sys;
797
798 if( b_fmt_src || b_fmt_dst )
799 return VLC_EGENERIC;
800
801 PROFILE_START( TigerValidateSubpicture );
802
803 /* time in seconds from the start of the stream */
804 kate_float t = (p_subpic->updater.p_sys->i_start + ts - p_subpic->i_start ) / 1000000.0f;
805
806 /* it is likely that the current region (if any) can be kept as is; test for this */
807 vlc_mutex_lock( &p_sys->lock );
808 int i_ret;
809 if( p_sys->b_dirty || tiger_renderer_is_dirty( p_sys->p_tr ) )
810 {
811 i_ret = VLC_EGENERIC;
812 goto exit;
813 }
814 if( tiger_renderer_update( p_sys->p_tr, t, 1 ) >= 0 &&
815 tiger_renderer_is_dirty( p_sys->p_tr ) )
816 {
817 i_ret = VLC_EGENERIC;
818 goto exit;
819 }
820
821 i_ret = VLC_SUCCESS;
822exit:
823 vlc_mutex_unlock( &p_sys->lock );
824 PROFILE_STOP( TigerValidateSubpicture );
825 return i_ret;
826}

828/* Tiger renders can end up looking a bit crap since they get overlaid on top of
829 a subsampled YUV image, so there can be a fair amount of chroma bleeding.
830 Looks good with white though since it's all luma. Hopefully that will be the
831 common case. */
832static void TigerUpdateSubpicture( subpicture_t *p_subpic,
833 const video_format_t *p_fmt_src,
834 const video_format_t *p_fmt_dst,
835 mtime_t ts )
836{
837 decoder_sys_t *p_sys = p_subpic->updater.p_sys->p_dec_sys;
838 plane_t *p_plane;
839 kate_float t;
840 int i_ret;
841
842
843 /* time in seconds from the start of the stream */
844 t = (p_subpic->updater.p_sys->i_start + ts - p_subpic->i_start ) / 1000000.0f;
845
846 PROFILE_START( TigerUpdateSubpicture );
847
848 /* create a full frame region - this will also tell Tiger the size of the frame */
849 video_format_t fmt = *p_fmt_dst;
850 fmt.i_chroma = VLC_CODEC_RGBA;
851 fmt.i_bits_per_pixel = 0;
852 fmt.i_width =
853 fmt.i_visible_width = p_fmt_src->i_width;
854 fmt.i_height =
855 fmt.i_visible_height = p_fmt_src->i_height;
856 fmt.i_x_offset = fmt.i_y_offset = 0;
857
858 subpicture_region_t *p_r = subpicture_region_New( &fmt );
859 if( !p_r )
860 return;
861
862 p_r->i_x = 0;
863 p_r->i_y = 0;
864 p_r->i_align = SUBPICTURE_ALIGN_TOP | SUBPICTURE_ALIGN_LEFT;
865
866 vlc_mutex_lock( &p_sys->lock );
867
868 p_plane = &p_r->p_picture->p[0];
869 i_ret = tiger_renderer_set_buffer( p_sys->p_tr, p_plane->p_pixels, fmt.i_width, p_plane->i_lines, p_plane->i_pitch, 1 );
870 if( i_ret < 0 )
871 {
872 goto failure;
873 }
874
875 PROFILE_START( tiger_renderer_update );
876 i_ret = tiger_renderer_update( p_sys->p_tr, t, 1 );
877 if( i_ret < 0 )
878 {
879 goto failure;
880 }
881 PROFILE_STOP( tiger_renderer_update );
882
883 PROFILE_START( tiger_renderer_render );
884 i_ret = tiger_renderer_render( p_sys->p_tr );
885 if( i_ret < 0 )
886 {
887 goto failure;
888 }
889 PROFILE_STOP( tiger_renderer_render );
890
891 PostprocessTigerImage( p_plane, fmt.i_width );
892 p_subpic->p_region = p_r;
893 p_sys->b_dirty = false;
894
895 PROFILE_STOP( TigerUpdateSubpicture );
896
897 vlc_mutex_unlock( &p_sys->lock );
898
899 return;
900
901failure:
902 vlc_mutex_unlock( &p_sys->lock );
903 subpicture_region_ChainDelete( p_r );
904}

1110/*****************************************************************************
1111 * DecodePacket: decodes a Kate packet.
1112 *****************************************************************************/
1113static subpicture_t *DecodePacket( decoder_t *p_dec, kate_packet *p_kp, block_t *p_block )
1114{
1115 decoder_sys_t *p_sys = p_dec->p_sys;
1116 const kate_event *ev = NULL;
1117 subpicture_t *p_spu = NULL;
1118 int i_ret;
1119
1120 if( !p_sys->b_ready )
1121 {
1122 msg_Err( p_dec, "Cannot decode Kate packet, decoder not initialized" );
1123 return NULL;
1124 }
1125
1126 i_ret = kate_decode_packetin( &p_sys->k, p_kp );
1127 if( i_ret < 0 )
1128 {
1129 msg_Err( p_dec, "Kate failed to decode packet: %d", i_ret );
1130 return NULL;
1131 }
1132
1133 i_ret = kate_decode_eventout( &p_sys->k, &ev );
1134 if( i_ret < 0 )
1135 {
1136 msg_Err( p_dec, "Kate failed to retrieve event: %d", i_ret );
1137 return NULL;
1138 }
1139 if( i_ret > 0 )
1140 {
1141 /* no event to go with this packet, this is normal */
1142 return NULL;
1143 }
1144
1145 /* we have an event */
1146
1147 /* Get a new spu */
1148 subpicture_updater_sys_t *p_spu_sys = NULL;
1149 if( p_sys->b_use_tiger)
1150 {
1151 p_spu_sys = malloc( sizeof(*p_spu_sys) );
1152 if( !p_spu_sys )
1153 return NULL;
1154 }
1155 subpicture_updater_t updater = {
1156#ifdef HAVE_TIGER
1157 .pf_validate = TigerValidateSubpicture,
1158 .pf_update = TigerUpdateSubpicture,
1159 .pf_destroy = TigerDestroySubpicture,
1160#endif
1161 .p_sys = p_spu_sys,
1162 };
1163 p_spu = decoder_NewSubpicture( p_dec, p_sys->b_use_tiger ? &updater : NULL );
1164 if( !p_spu )
1165 {
1166 free( p_spu_sys );
1167 /* this will happen for lyrics as there is no vout - so no error */
1168 /* msg_Err( p_dec, "Failed to allocate spu buffer" ); */
1169 return NULL;
1170 }
1171
1172 p_spu->i_start = p_block->i_pts;
1173 p_spu->i_stop = p_block->i_pts + INT64_C(1000000)*ev->duration*p_sys->ki.gps_denominator/p_sys->ki.gps_numerator;
1174 p_spu->b_ephemer = false;
1175 p_spu->b_absolute = false;
1176
1177#ifdef HAVE_TIGER
1178 if( p_sys->b_use_tiger)
1179 {
1180 p_spu_sys->p_dec_sys = p_sys;
1181 p_spu_sys->i_start = p_block->i_pts;
1182 DecSysHold( p_sys );
1183
1184 p_spu->i_stop = __MAX( p_sys->i_max_stop, p_spu->i_stop );
1185 p_spu->b_ephemer = true;
1186 p_spu->b_absolute = true;
1187
1188 /* add the event to tiger */
1189 vlc_mutex_lock( &p_sys->lock );
1190 CHECK_TIGER_RET( tiger_renderer_add_event( p_sys->p_tr, ev->ki, ev ) );
1191 vlc_mutex_unlock( &p_sys->lock );
1192 }
1193 else
1194#endif
1195 {
1196 p_spu = SetupSimpleKateSPU( p_dec, p_spu, ev );
1197 }
1198
1199 return p_spu;
1200}

1202/*****************************************************************************
1203 * SetupSimpleKateSPU: creates text/bitmap regions where appropriate
1204 *****************************************************************************/
1205static subpicture_t *SetupSimpleKateSPU( decoder_t *p_dec, subpicture_t *p_spu,
1206 const kate_event *ev )
1207{
1208 decoder_sys_t *p_sys = p_dec->p_sys;
1209 video_format_t fmt;
1210 subpicture_region_t *p_bitmap_region = NULL;
1211 video_palette_t palette;
1212 kate_tracker kin;
1213 bool b_tracker_valid = false;
1214 int i_ret;
1215
1216 /* these may be 0 for "not specified" */
1217 p_spu->i_original_picture_width = p_sys->ki.original_canvas_width;
1218 p_spu->i_original_picture_height = p_sys->ki.original_canvas_height;
1219
1220 /* Create a new subpicture region */
1221 memset( &fmt, 0, sizeof(video_format_t) );
1222
1223 if (p_sys->b_formatted)
1224 {
1225 i_ret = kate_tracker_init( &kin, &p_sys->ki, ev );
1226 if( i_ret < 0)
1227 {
1228 msg_Err( p_dec, "failed to initialize kate tracker, event will be unformatted: %d", i_ret );
1229 }
1230 else
1231 {
1232 int w = 720, h = 576; /* give sensible defaults just in case we fail to get the actual size */
1233 GetVideoSize(p_dec, &w, &h);
1234 i_ret = kate_tracker_update(&kin, 0, w, h, 0, 0, w, h);
1235 if( i_ret < 0)
1236 {
1237 kate_tracker_clear(&kin);
1238 msg_Err( p_dec, "failed to update kate tracker, event will be unformatted: %d", i_ret );
1239 }
1240 else
1241 {
1242 // TODO: parse tracker and set style, init fmt
1243 b_tracker_valid = true;
1244 }
1245 }
1246 }
1247
1248 if (ev->bitmap && ev->bitmap->type==kate_bitmap_type_paletted && ev->palette) {
1249
1250 /* create a separate region for the bitmap */
1251 memset( &fmt, 0, sizeof(video_format_t) );
1252 fmt.i_chroma = VLC_CODEC_YUVP;
1253 fmt.i_width = fmt.i_visible_width = ev->bitmap->width;
1254 fmt.i_height = fmt.i_visible_height = ev->bitmap->height;
1255 fmt.i_x_offset = fmt.i_y_offset = 0;
1256 fmt.p_palette = &palette;
1257 CreateKatePalette( fmt.p_palette, ev->palette );
1258
1259 p_bitmap_region = subpicture_region_New( &fmt );
1260 if( !p_bitmap_region )
1261 {
1262 msg_Err( p_dec, "cannot allocate SPU region" );
1263 decoder_DeleteSubpicture( p_dec, p_spu );
1264 return NULL;
1265 }
1266
1267 /* create the bitmap */
1268 CreateKateBitmap( p_bitmap_region->p_picture, ev->bitmap );
1269
1270 msg_Dbg(p_dec, "Created bitmap, %zux%zu, %zu colors", ev->bitmap->width, ev->bitmap->height, ev->palette->ncolors);
1271 }
1272
1273 /* text region */
1274 fmt.i_chroma = VLC_CODEC_TEXT;
1275 fmt.i_sar_num = 0;
1276 fmt.i_sar_den = 1;
1277 fmt.i_width = fmt.i_height = 0;
1278 fmt.i_x_offset = fmt.i_y_offset = 0;
1279 p_spu->p_region = subpicture_region_New( &fmt );
1280 if( !p_spu->p_region )
1281 {
1282 msg_Err( p_dec, "cannot allocate SPU region" );
1283 if( p_bitmap_region )
1284 subpicture_region_Delete( p_bitmap_region );
1285 decoder_DeleteSubpicture( p_dec, p_spu );
1286 return NULL;
1287 }
1288
1289 SetupText( p_dec, p_spu, ev );
1290
1291 /* default positioning */
1292 p_spu->p_region->i_align = SUBPICTURE_ALIGN_BOTTOM;
1293 if (p_bitmap_region)
1294 {
1295 p_bitmap_region->i_align = SUBPICTURE_ALIGN_BOTTOM;
1296 }
1297 p_spu->p_region->i_x = 0;
1298 p_spu->p_region->i_y = 10;
1299
1300 /* override if tracker info present */
1301 if (b_tracker_valid)
1302 {
1303 if (kin.has.region)
1304 {
1305 p_spu->p_region->i_x = kin.region_x;
1306 p_spu->p_region->i_y = kin.region_y;
1307 if (p_bitmap_region)
1308 {
1309 p_bitmap_region->i_x = kin.region_x;
1310 p_bitmap_region->i_y = kin.region_y;
1311 }
1312 p_spu->b_absolute = true;
1313 }
1314
1315 kate_tracker_clear(&kin);
1316 }
1317
1318 /* if we have a bitmap, chain it before the text */
1319 if (p_bitmap_region)
1320 {
1321 p_bitmap_region->p_next = p_spu->p_region;
1322 p_spu->p_region = p_bitmap_region;
1323 }
1324
1325 return p_spu;
1326}

1328/*****************************************************************************
1329 * ParseKateComments:
1330 *****************************************************************************/
1331static void ParseKateComments( decoder_t *p_dec )
1332{
1333 char *psz_name, *psz_value, *psz_comment;
1334 int i = 0;
1335
1336 while ( i < p_dec->p_sys->kc.comments )
1337 {
1338 psz_comment = strdup( p_dec->p_sys->kc.user_comments[i] );
1339 if( !psz_comment )
1340 break;
1341 psz_name = psz_comment;
1342 psz_value = strchr( psz_comment, '=' );
1343 if( psz_value )
1344 {
1345 *psz_value = '\0';
1346 psz_value++;
1347
1348 if( !p_dec->p_description )
1349 p_dec->p_description = vlc_meta_New();
1350 if( p_dec->p_description )
1351 vlc_meta_AddExtra( p_dec->p_description, psz_name, psz_value );
1352 }
1353 free( psz_comment );
1354 i++;
1355 }
1356}

1359 * CloseDecoder: clean up the decoder
1360 *****************************************************************************/
1361static void CloseDecoder( vlc_object_t *p_this )
1362{
1363 decoder_t *p_dec = (decoder_t *)p_this;
1364 size_t i_index;
1365
1366 /* remove the decoder from the global list */
1367 vlc_mutex_lock( &kate_decoder_list_mutex );
1368 for( i_index = 0; i_index < kate_decoder_list_size; i_index++ )
1369 {
1370 if( kate_decoder_list[ i_index ] == p_dec )
1371 {
1372 kate_decoder_list[ i_index ] = kate_decoder_list[ --kate_decoder_list_size ];
1373 break;
1374 }
1375 }
1376 vlc_mutex_unlock( &kate_decoder_list_mutex );
1377
1378 msg_Dbg( p_dec, "Closing Kate decoder" );
1379 DecSysRelease( p_dec->p_sys );
1380}

1389static void DecSysRelease( decoder_sys_t *p_sys )
1390{

1393 if( p_sys->i_refcount > 0)
1394 {
1395 vlc_mutex_unlock( &p_sys->lock );
1396 return;
1397 }
1398
1399 vlc_mutex_unlock( &p_sys->lock );
1400 vlc_mutex_destroy( &p_sys->lock );
1401
1402#ifdef HAVE_TIGER
1403 if( p_sys->p_tr )
1404 tiger_renderer_destroy( p_sys->p_tr );
1405 free( p_sys->psz_tiger_default_font_desc );
1406#endif
1407
1408 if (p_sys->b_ready)
1409 kate_clear( &p_sys->k );
1410 kate_info_clear( &p_sys->ki );
1411 kate_comment_clear( &p_sys->kc );
1412
1413 free( p_sys );
1414}
