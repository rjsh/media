164/*****************************************************************************
165 * Open: probe the decoder and return score
166 *****************************************************************************
167 * Tries to launch a decoder and return score so that the interface is able
168 * to chose.
169 *****************************************************************************/
170static int Open( vlc_object_t *p_this )
171{
172 decoder_t *p_dec = (decoder_t *) p_this;
173 decoder_sys_t *p_sys = NULL;
174 int i_val;
175
176
177 if( p_dec->fmt_in.i_codec != VLC_CODEC_TELETEXT)
178 {
179 return VLC_EGENERIC;
180 }
181
182 p_dec->pf_decode_sub = Decode;
183 p_sys = p_dec->p_sys = calloc( 1, sizeof(*p_sys) );
184 if( p_sys == NULL )
185 return VLC_ENOMEM;
186 p_dec->fmt_out.i_cat = SPU_ES;
187 p_dec->fmt_out.i_codec = 0;
188
189 p_sys->i_align = 0;
190 for ( int i = 0; i < 9; i++ )
191 p_sys->pi_active_national_set[i] = ppi_national_subsets[1];
192
193 i_val = var_CreateGetInteger( p_dec, "telx-override-page" );
194 if( i_val == -1 && p_dec->fmt_in.subs.teletext.i_magazine != -1 &&
195 ( p_dec->fmt_in.subs.teletext.i_magazine != 1 ||
196 p_dec->fmt_in.subs.teletext.i_page != 0 ) ) /* ignore if TS demux wants page 100 (unlikely to be sub) */
197 {
198 bool b_val;
199 p_sys->i_wanted_magazine = p_dec->fmt_in.subs.teletext.i_magazine;
200 p_sys->i_wanted_page = p_dec->fmt_in.subs.teletext.i_page;
201
202 b_val = var_CreateGetBool( p_dec, "telx-french-workaround" );
203 if( p_sys->i_wanted_page < 100 &&
204 (b_val || (p_sys->i_wanted_page % 16) >= 10))
205 {
206 /* See http://www.nada.kth.se/~ragge/vdr/ttxtsubs/TROUBLESHOOTING.txt
207 * paragraph about French channels - they mix up decimal and
208 * hexadecimal */
209 p_sys->i_wanted_page = (p_sys->i_wanted_page / 10) * 16 +
210 (p_sys->i_wanted_page % 10);
211 }
212 }
213 else if( i_val <= 0 )
214 {
215 p_sys->i_wanted_magazine = -1;
216 p_sys->i_wanted_page = -1;
217 }
218 else
219 {
220 p_sys->i_wanted_magazine = i_val / 100;
221 p_sys->i_wanted_page = (((i_val % 100) / 10) << 4)
222 |((i_val % 100) % 10);
223 }
224 p_sys->b_ignore_sub_flag = var_CreateGetBool( p_dec,
225 "telx-ignore-subtitle-flag" );
226
227 msg_Dbg( p_dec, "starting telx on magazine %d page %02x flag %d",
228 p_sys->i_wanted_magazine, p_sys->i_wanted_page,
229 p_sys->b_ignore_sub_flag );
230
231 return VLC_SUCCESS;
232
233/* error: */
234/* if (p_sys) { */
235/* free(p_sys); */
236/* p_sys = NULL; */
237/* } */
238/* return VLC_EGENERIC; */
239}

252/**************************
253 * change bits endianness *
254 **************************/
255static uint8_t bytereverse( int n )
256{
257 n = (((n >> 1) & 0x55) | ((n << 1) & 0xaa));
258 n = (((n >> 2) & 0x33) | ((n << 2) & 0xcc));
259 n = (((n >> 4) & 0x0f) | ((n << 4) & 0xf0));
260 return n;
261}
262
263static int hamming_8_4( int a )
264{
265 switch (a) {
266 case 0xA8:
267 return 0;
268 case 0x0B:
269 return 1;
270 case 0x26:
271 return 2;
272 case 0x85:
273 return 3;
274 case 0x92:
275 return 4;
276 case 0x31:
277 return 5;
278 case 0x1C:
279 return 6;
280 case 0xBF:
281 return 7;
282 case 0x40:
283 return 8;
284 case 0xE3:
285 return 9;
286 case 0xCE:
287 return 10;
288 case 0x6D:
289 return 11;
290 case 0x7A:
291 return 12;
292 case 0xD9:
293 return 13;
294 case 0xF4:
295 return 14;
296 case 0x57:
297 return 15;
298 default:
299 return -1; // decoding error , not yet corrected
300 }
301}

303// utc-2 --> utf-8
304// this is not a general function, but it's enough for what we do here
305// the result buffer need to be at least 4 bytes long
306static void to_utf8( char * res, uint16_t ch )
307{
308 if( ch >= 0x80 )
309 {
310 if( ch >= 0x800 )
311 {
312 res[0] = (ch >> 12) | 0xE0;
313 res[1] = ((ch >> 6) & 0x3F) | 0x80;
314 res[2] = (ch & 0x3F) | 0x80;
315 res[3] = 0;
316 }
317 else
318 {
319 res[0] = (ch >> 6) | 0xC0;
320 res[1] = (ch & 0x3F) | 0x80;
321 res[2] = 0;
322 }
323 }
324 else
325 {
326 res[0] = ch;
327 res[1] = 0;
328 }
329}

331static void decode_string( char * res, int res_len,
332 decoder_sys_t *p_sys, int magazine,
333 uint8_t * packet, int len )
334{
335 char utf8[7];
336 char * pt = res;
337 int i;
338
339 for ( i = 0; i < len; i++ )
340 {
341 int in = bytereverse( packet[i] ) & 0x7f;
342 uint16_t out = 32;
343 size_t l;
344
345 switch ( in )
346 {
347 /* special national characters */
348 case 0x23:
349 out = p_sys->pi_active_national_set[magazine][0];
350 break;
351 case 0x24:
352 out = p_sys->pi_active_national_set[magazine][1];
353 break;
354 case 0x40:
355 out = p_sys->pi_active_national_set[magazine][2];
356 break;
357 case 0x5b:
358 out = p_sys->pi_active_national_set[magazine][3];
359 break;
360 case 0x5c:
361 out = p_sys->pi_active_national_set[magazine][4];
362 break;
363 case 0x5d:
364 out = p_sys->pi_active_national_set[magazine][5];
365 break;
366 case 0x5e:
367 out = p_sys->pi_active_national_set[magazine][6];
368 break;
369 case 0x5f:
370 out = p_sys->pi_active_national_set[magazine][7];
371 break;
372 case 0x60:
373 out = p_sys->pi_active_national_set[magazine][8];
374 break;
375 case 0x7b:
376 out = p_sys->pi_active_national_set[magazine][9];
377 break;
378 case 0x7c:
379 out = p_sys->pi_active_national_set[magazine][10];
380 break;
381 case 0x7d:
382 out = p_sys->pi_active_national_set[magazine][11];
383 break;
384 case 0x7e:
385 out = p_sys->pi_active_national_set[magazine][12];
386 break;
387
388 /* some special control characters (empirical) */
389 case 0x0d:
390 /* apparently this starts a sequence that ends with 0xb 0xb */
391 while ( i + 1 < len && (bytereverse( packet[i+1] ) & 0x7f) != 0x0b )
392 i++;
393 i += 2;
394 break;
395 /* goto skip; */
396
397 default:
398 /* non documented national range 0x08 - 0x0f */
399 if ( in >= 0x08 && in <= 0x0f )
400 {
401 out = p_sys->pi_active_national_set[magazine][13 + in - 8];
402 break;
403 }
404
405 /* normal ascii */
406 if ( in > 32 && in < 0x7f )
407 out = in;
408 }
409
410 /* handle undefined national characters */
411 if ( out == 0 )
412 out = 32;
413
414 /* convert to utf-8 */
415 to_utf8( utf8, out );
416 l = strlen( utf8 );
417 if ( pt + l < res + res_len - 1 )
418 {
419 strcpy(pt, utf8);
420 pt += l;
421 }
422
423 /* skip: ; */
424 }
425 /* end: */
426 *pt++ = 0;
427}

429/*****************************************************************************
430 * Decode:
431 *****************************************************************************/
432static subpicture_t *Decode( decoder_t *p_dec, block_t **pp_block )
433{
434 decoder_sys_t *p_sys = p_dec->p_sys;
435 block_t *p_block;
436 subpicture_t *p_spu = NULL;
437 video_format_t fmt;
438 /* int erase = 0; */
439 int len, offset;
440#if 0
441 int i_wanted_magazine = i_conf_wanted_page / 100;
442 int i_wanted_page = 0x10 * ((i_conf_wanted_page % 100) / 10)
443 | (i_conf_wanted_page % 10);
444#endif
445 bool b_update = false;
446 char psz_text[512], *pt = psz_text;
447 char psz_line[256];
448 int i, total;
449
450 if( pp_block == NULL || *pp_block == NULL )
451 return NULL;
452 p_block = *pp_block;
453 *pp_block = NULL;
454
455 dbg((p_dec, "start of telx packet with header %2x\n",
456 * (uint8_t *) p_block->p_buffer));
457 len = p_block->i_buffer;
458 for ( offset = 1; offset + 46 <= len; offset += 46 )
459 {
460 uint8_t * packet = (uint8_t *) p_block->p_buffer+offset;
461// int vbi = ((0x20 & packet[2]) != 0 ? 0 : 313) + (0x1F & packet[2]);
462
463// dbg((p_dec, "vbi %d header %02x %02x %02x\n", vbi, packet[0], packet[1], packet[2]));
464 if ( packet[0] == 0xFF ) continue;
465
466/* if (packet[1] != 0x2C) { */
467/* printf("wrong header\n"); */
468/* //goto error; */
469/* continue; */
470/* } */
471
472 int mpag = (hamming_8_4( packet[4] ) << 4) | hamming_8_4( packet[5] );
473 int row, magazine;
474 if ( mpag < 0 )
475 {
476 /* decode error */
477 dbg((p_dec, "mpag hamming error\n"));
478 continue;
479 }
480
481 row = 0xFF & bytereverse(mpag);
482 magazine = (7 & row) == 0 ? 8 : (7 & row);
483 row >>= 3;
484
485 if ( p_sys->i_wanted_page != -1
486 && magazine != p_sys->i_wanted_magazine )
487 continue;
488
489 if ( row == 0 )
490 {
491 /* row 0 : flags and header line */
492 int flag = 0;
493 int a;
494
495 for ( a = 0; a < 6; a++ )
496 {
497 flag |= (0xF & (bytereverse( hamming_8_4(packet[8 + a]) ) >> 4))
498 << (a * 4);
499 }
500
501 /* if (!p_sys->b_ignore_sub_flag && !(1 & flag>>15)) */
502 /* continue; */
503
504 p_sys->i_page[magazine] = (0xF0 & bytereverse( hamming_8_4(packet[7]) )) |
505 (0xF & (bytereverse( hamming_8_4(packet[6]) ) >> 4) );
506
507 decode_string( psz_line, sizeof(psz_line), p_sys, magazine,
508 packet + 14, 40 - 14 );
509
510 dbg((p_dec, "mag %d flags %x page %x character set %d subtitles %d", magazine, flag,
511 p_sys->i_page[magazine],
512 7 & flag>>21, 1 & flag>>15, psz_line));
513
514 p_sys->pi_active_national_set[magazine] =
515 ppi_national_subsets[7 & (flag >> 21)];
516
517 p_sys->b_is_subtitle[magazine] = p_sys->b_ignore_sub_flag
518 || ( (1 & (flag >> 15))
519 && (1 & (flag>>16)) );
520
521 dbg(( p_dec, "FLAGS%s%s%s%s%s%s%s mag_ser %d",
522 (1 & (flag>>14))? " news" : "",
523 (1 & (flag>>15))? " subtitle" : "",
524 (1 & (flag>>7))? " erase" : "",
525 (1 & (flag>>16))? " suppressed_head" : "",
526 (1 & (flag>>17))? " update" : "",
527 (1 & (flag>>18))? " interrupt" : "",
528 (1 & (flag>>19))? " inhibit" : "",
529 (1 & (flag>>20)) ));
530
531 if ( (p_sys->i_wanted_page != -1
532 && p_sys->i_page[magazine] != p_sys->i_wanted_page)
533 || !p_sys->b_is_subtitle[magazine] )
534 continue;
535
536 p_sys->b_erase[magazine] = (1 & (flag >> 7));
537
538 dbg((p_dec, "%ld --> %ld\n", (long int) p_block->i_pts, (long int)(p_sys->prev_pts+1500000)));
539 /* kludge here :
540 * we ignore the erase flag if it happens less than 1.5 seconds
541 * before last caption
542 * TODO make this time configurable
543 * UPDATE the kludge seems to be no more necessary
544 * so it's commented out*/
545 if ( /*p_block->i_pts > p_sys->prev_pts + 1500000 && */
546 p_sys->b_erase[magazine] )
547 {
548 int i;
549
550 dbg((p_dec, "ERASE !\n"));
551
552 p_sys->b_erase[magazine] = 0;
553 for ( i = 1; i < 32; i++ )
554 {
555 if ( !p_sys->ppsz_lines[i][0] ) continue;
556 /* b_update = true; */
557 p_sys->ppsz_lines[i][0] = 0;
558 }
559 }
560
561 /* replace the row if it's different */
562 if ( strcmp(psz_line, p_sys->ppsz_lines[row]) )
563 {
564 strncpy( p_sys->ppsz_lines[row], psz_line,
565 sizeof(p_sys->ppsz_lines[row]) - 1);
566 }
567 b_update = true;
568
569 }
570 else if ( row < 24 )
571 {
572 char * t;
573 int i;
574 /* row 1-23 : normal lines */
575
576 if ( (p_sys->i_wanted_page != -1
577 && p_sys->i_page[magazine] != p_sys->i_wanted_page)
578 || !p_sys->b_is_subtitle[magazine]
579 || (p_sys->i_wanted_page == -1
580 && p_sys->i_page[magazine] > 0x99) )
581 continue;
582
583 decode_string( psz_line, sizeof(psz_line), p_sys, magazine,
584 packet + 6, 40 );
585 t = psz_line;
586
587 /* remove starting spaces */
588 while ( *t == 32 ) t++;
589
590 /* remove trailing spaces */
591 for ( i = strlen(t) - 1; i >= 0 && t[i] == 32; i-- );
592 t[i + 1] = 0;
593
594 /* replace the row if it's different */
595 if ( strcmp( t, p_sys->ppsz_lines[row] ) )
596 {
597 strncpy( p_sys->ppsz_lines[row], t,
598 sizeof(p_sys->ppsz_lines[row]) - 1 );
599 b_update = true;
600 }
601
602 if (t[0])
603 p_sys->prev_pts = p_block->i_pts;
604
605 dbg((p_dec, "%d %d : ", magazine, row));
606 dbg((p_dec, "%s\n", t));
607
608#ifdef TELX_DEBUG
609 {
610 char dbg[256];
611 dbg[0] = 0;
612 for ( i = 0; i < 40; i++ )
613 {
614 int in = bytereverse(packet[6 + i]) & 0x7f;
615 sprintf(dbg + strlen(dbg), "%02x ", in);
616 }
617 dbg((p_dec, "%s\n", dbg));
618 dbg[0] = 0;
619 for ( i = 0; i < 40; i++ )
620 {
621 decode_string( psz_line, sizeof(psz_line), p_sys, magazine,
622 packet + 6 + i, 1 );
623 sprintf( dbg + strlen(dbg), "%s ", psz_line );
624 }
625 dbg((p_dec, "%s\n", dbg));
626 }
627#endif
628 }
629 else if ( row == 25 )
630 {
631 /* row 25 : alternate header line */
632 if ( (p_sys->i_wanted_page != -1
633 && p_sys->i_page[magazine] != p_sys->i_wanted_page)
634 || !p_sys->b_is_subtitle[magazine] )
635 continue;
636
637 decode_string( psz_line, sizeof(psz_line), p_sys, magazine,
638 packet + 6, 40 );
639
640 /* replace the row if it's different */
641 if ( strcmp( psz_line, p_sys->ppsz_lines[0] ) )
642 {
643 strncpy( p_sys->ppsz_lines[0], psz_line,
644 sizeof(p_sys->ppsz_lines[0]) - 1 );
645 /* b_update = true; */
646 }
647 }
648/* else if (row == 26) { */
649/* // row 26 : TV listings */
650/* } else */
651/* dbg((p_dec, "%d %d : %s\n", magazine, row, decode_string(p_sys, magazine, packet+6, 40))); */
652 }
653
654 if ( !b_update )
655 goto error;
656
657 total = 0;
658 for ( i = 1; i < 24; i++ )
659 {
660 size_t l = strlen( p_sys->ppsz_lines[i] );
661
662 if ( l > sizeof(psz_text) - total - 1 )
663 l = sizeof(psz_text) - total - 1;
664
665 if ( l > 0 )
666 {
667 memcpy( pt, p_sys->ppsz_lines[i], l );
668 total += l;
669 pt += l;
670 if ( sizeof(psz_text) - total - 1 > 0 )
671 {
672 *pt++ = '\n';
673 total++;
674 }
675 }
676 }
677 *pt = 0;
678
679 if ( !strcmp(psz_text, p_sys->psz_prev_text) )
680 goto error;
681
682 dbg((p_dec, "UPDATE TELETEXT PICTURE\n"));
683
684 assert( sizeof(p_sys->psz_prev_text) >= sizeof(psz_text) );
685 strcpy( p_sys->psz_prev_text, psz_text );
686
687 /* Create the subpicture unit */
688 p_spu = decoder_NewSubpicture( p_dec, NULL );
689 if( !p_spu )
690 {
691 msg_Warn( p_dec, "can't get spu buffer" );
692 goto error;
693 }
694
695 /* Create a new subpicture region */
696 memset( &fmt, 0, sizeof(video_format_t) );
697 fmt.i_chroma = VLC_CODEC_TEXT;
698 fmt.i_width = fmt.i_height = 0;
699 fmt.i_x_offset = fmt.i_y_offset = 0;
700 p_spu->p_region = subpicture_region_New( &fmt );
701 if( p_spu->p_region == NULL )
702 {
703 msg_Err( p_dec, "cannot allocate SPU region" );
704 goto error;
705 }
706
707 /* Normal text subs, easy markup */
708 p_spu->p_region->i_align = SUBPICTURE_ALIGN_BOTTOM | p_sys->i_align;
709 p_spu->p_region->i_x = p_sys->i_align ? 20 : 0;
710 p_spu->p_region->i_y = 10;
711 p_spu->p_region->psz_text = strdup(psz_text);
712
713 p_spu->i_start = p_block->i_pts;
714 p_spu->i_stop = p_block->i_pts + p_block->i_length;
715 p_spu->b_ephemer = (p_block->i_length == 0);
716 p_spu->b_absolute = false;
717 dbg((p_dec, "%ld --> %ld\n", (long int) p_block->i_pts/100000, (long int)p_block->i_length/100000));
718
719 block_Release( p_block );
720 return p_spu;
721
722error:
723 if ( p_spu != NULL )
724 {
725 decoder_DeleteSubpicture( p_dec, p_spu );
726 p_spu = NULL;
727 }
728
729 block_Release( p_block );
730 return NULL;
731}
