48/**
49 * Initialize subtitle decoder
50 */
51int InitSubtitleDec(decoder_t *dec, AVCodecContext *context,
52 AVCodec *codec, int codec_id, const char *namecodec)
53{
54 decoder_sys_t *sys;
55
56 /* */
57 switch (codec_id) {
58 case AV_CODEC_ID_HDMV_PGS_SUBTITLE:
59 case AV_CODEC_ID_XSUB:
60 case AV_CODEC_ID_DVB_SUBTITLE:
61 break;
62 default:
63 msg_Warn(dec, "refusing to decode non validated subtitle codec");
64 return VLC_EGENERIC;
65 }
66
67 /* */
68 dec->p_sys = sys = malloc(sizeof(*sys));
69 if (!sys)
70 return VLC_ENOMEM;
71
72 codec->type = AVMEDIA_TYPE_SUBTITLE;
73 context->codec_type = AVMEDIA_TYPE_SUBTITLE;
74 context->codec_id = codec_id;
75 sys->p_context = context;
76 sys->p_codec = codec;
77 sys->i_codec_id = codec_id;
78 sys->psz_namecodec = namecodec;
79 sys->b_delayed_open = false;
80
81 /* */
82 context->extradata_size = 0;
83 context->extradata = NULL;
84
85 /* */
86 int ret;
87 char *psz_opts = var_InheritString(dec, "avcodec-options");
88 AVDictionary *options = NULL;
89 if (psz_opts && *psz_opts)
90 options = vlc_av_get_options(psz_opts);
91 free(psz_opts);
92
93 vlc_avcodec_lock();
94 ret = avcodec_open2(context, codec, options ? &options : NULL);
95 vlc_avcodec_unlock();
96
97 AVDictionaryEntry *t = NULL;
98 while ((t = av_dict_get(options, "", t, AV_DICT_IGNORE_SUFFIX))) {
99 msg_Err(dec, "Unknown option \"%s\"", t->key);
100 }
101 av_dict_free(&options);
102
103 if (ret < 0) {
104 msg_Err(dec, "cannot open codec (%s)", namecodec);
105 free(context->extradata);
106 free(sys);
107 return VLC_EGENERIC;
108 }
109
110 /* */
111 msg_Dbg(dec, "libavcodec codec (%s) started", namecodec);
112 dec->fmt_out.i_cat = SPU_ES;
113
114 return VLC_SUCCESS;
115}

117/**
118 * Decode one subtitle
119 */
120subpicture_t *DecodeSubtitle(decoder_t *dec, block_t **block_ptr)
121{
122 decoder_sys_t *sys = dec->p_sys;
123
124 if (!block_ptr || !*block_ptr)
125 return NULL;
126
127 block_t *block = *block_ptr;
128
129 if (block->i_flags & (BLOCK_FLAG_DISCONTINUITY|BLOCK_FLAG_CORRUPTED)) {
130 block_Release(block);
131 avcodec_flush_buffers(sys->p_context);
132 return NULL;
133 }
134
135 if (block->i_buffer <= 0) {
136 block_Release(block);
137 return NULL;
138 }
139
140 *block_ptr =
141 block = block_Realloc(block,
142 0,
143 block->i_buffer + FF_INPUT_BUFFER_PADDING_SIZE);
144 if (!block)
145 return NULL;
146 block->i_buffer -= FF_INPUT_BUFFER_PADDING_SIZE;
147 memset(&block->p_buffer[block->i_buffer], 0, FF_INPUT_BUFFER_PADDING_SIZE);
148
149 /* */
150 AVSubtitle subtitle;
151 memset(&subtitle, 0, sizeof(subtitle));
152
153 AVPacket pkt;
154 av_init_packet(&pkt);
155 pkt.data = block->p_buffer;
156 pkt.size = block->i_buffer;
157
158 int has_subtitle = 0;
159 int used = avcodec_decode_subtitle2(sys->p_context,
160 &subtitle, &has_subtitle, &pkt);
161
162 if (used < 0) {
163 msg_Warn(dec, "cannot decode one subtitle (%zu bytes)",
164 block->i_buffer);
165
166 block_Release(block);
167 return NULL;
168 } else if ((size_t)used > block->i_buffer) {
169 used = block->i_buffer;
170 }
171
172 block->i_buffer -= used;
173 block->p_buffer += used;
174
175 /* */
176 subpicture_t *spu = NULL;
177 if (has_subtitle)
178 spu = ConvertSubtitle(dec, &subtitle,
179 block->i_pts > 0 ? block->i_pts : block->i_dts,
180 sys->p_context);
181
182 /* */
183 if (!spu)
184 block_Release(block);
185 return spu;
186}

188/**
189 * Convert a RGBA libavcodec region to our format.
190 */
191static subpicture_region_t *ConvertRegionRGBA(AVSubtitleRect *ffregion)
192{
193 if (ffregion->w <= 0 || ffregion->h <= 0)
194 return NULL;
195
196 video_format_t fmt;
197 memset(&fmt, 0, sizeof(fmt));
198 fmt.i_chroma = VLC_FOURCC('R','G','B','A');
199 fmt.i_width =
200 fmt.i_visible_width = ffregion->w;
201 fmt.i_height =
202 fmt.i_visible_height = ffregion->h;
203 fmt.i_x_offset = 0;
204 fmt.i_y_offset = 0;
205
206 subpicture_region_t *region = subpicture_region_New(&fmt);
207 if (!region)
208 return NULL;
209
210 region->i_x = ffregion->x;
211 region->i_y = ffregion->y;
212 region->i_align = SUBPICTURE_ALIGN_TOP | SUBPICTURE_ALIGN_LEFT;
213
214 const plane_t *p = &region->p_picture->p[0];
215 for (int y = 0; y < ffregion->h; y++) {
216 for (int x = 0; x < ffregion->w; x++) {
217 /* I don't think don't have paletized RGB_A_ */
218 const uint8_t index = ffregion->pict.data[0][y * ffregion->w+x];
219 assert(index < ffregion->nb_colors);
220
221 uint32_t color;
222 memcpy(&color, &ffregion->pict.data[1][4*index], 4);
223
224 uint8_t *p_rgba = &p->p_pixels[y * p->i_pitch + x * p->i_pixel_pitch];
225 p_rgba[0] = (color >> 16) & 0xff;
226 p_rgba[1] = (color >> 8) & 0xff;
227 p_rgba[2] = (color >> 0) & 0xff;
228 p_rgba[3] = (color >> 24) & 0xff;
229 }
230 }
231
232 return region;
233}

235/**
236 * Convert a libavcodec subtitle to our format.
237 */
238static subpicture_t *ConvertSubtitle(decoder_t *dec, AVSubtitle *ffsub, mtime_t pts,
239 AVCodecContext *avctx)
240{
241 subpicture_t *spu = decoder_NewSubpicture(dec, NULL);
242 if (!spu)
243 return NULL;
244
245 //msg_Err(dec, "%lld %d %d",
246 // pts, ffsub->start_display_time, ffsub->end_display_time);
247 spu->i_start = pts + ffsub->start_display_time * INT64_C(1000);
248 spu->i_stop = pts + ffsub->end_display_time * INT64_C(1000);
249 spu->b_absolute = true; /* FIXME How to set it right ? */
250 spu->b_ephemer = true; /* FIXME How to set it right ? */
251
252 if (avctx->coded_width != 0 && avctx->coded_height != 0) {
253 spu->i_original_picture_width = avctx->coded_width;
254 spu->i_original_picture_height = avctx->coded_height;
255 } else {
256 spu->i_original_picture_width =
257 dec->fmt_in.subs.spu.i_original_frame_width;
258 spu->i_original_picture_height =
259 dec->fmt_in.subs.spu.i_original_frame_height;
260 }
261
262 subpicture_region_t **region_next = &spu->p_region;
263
264 for (unsigned i = 0; i < ffsub->num_rects; i++) {
265 AVSubtitleRect *rec = ffsub->rects[i];
266
267 //msg_Err(dec, "SUBS RECT[%d]: %dx%d @%dx%d",
268 // i, rec->w, rec->h, rec->x, rec->y);
269
270 subpicture_region_t *region = NULL;
271 switch (ffsub->format) {
272 case 0:
273 region = ConvertRegionRGBA(rec);
274 break;
275 default:
276 msg_Warn(dec, "unsupported subtitle type");
277 region = NULL;
278 break;
279 }
280 if (region) {
281 *region_next = region;
282 region_next = &region->p_next;
283 }
284 /* Free AVSubtitleRect */
285 avpicture_free(&rec->pict);
286 av_free(rec);
287 }
288 av_free(ffsub->rects);
289
290 return spu;
291}
