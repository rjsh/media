24#ifndef VLC_AVCODEC_VA_H
25#define VLC_AVCODEC_VA_H 1
26
27typedef struct vlc_va_t vlc_va_t;
28typedef struct vlc_va_sys_t vlc_va_sys_t;
29
30struct vlc_va_t {
31 VLC_COMMON_MEMBERS
32
33 vlc_va_sys_t *sys;
34 module_t *module;
35 const char *description;
36 int pix_fmt;
37
38 int (*setup)(vlc_va_t *, void **hw, vlc_fourcc_t *output,
39 int width, int height);
40 int (*get)(vlc_va_t *, void **opaque, uint8_t **data);
41 void (*release)(void *opaque, uint8_t *surface);
42 int (*extract)(vlc_va_t *, picture_t *dst, void *opaque, uint8_t *data);
43};
44
45/**
46 * Creates an accelerated video decoding back-end for libavcodec.
47 * @param obj parent VLC object
48 * @param fmt VLC format of the content to decode
49 * @return a new VLC object on success, NULL on error.
50 */
51vlc_va_t *vlc_va_New(vlc_object_t *obj, AVCodecContext *, const es_format_t *fmt);
52
53/**
54 * Initializes the acceleration video decoding back-end for libavcodec.
55 * @param hw pointer to libavcodec hardware context pointer [OUT]
56 * @param output pointer to video chroma output by the back-end [OUT]
57 * @param width coded video width in pixels
58 * @param height coded video height in pixels
59 * @return VLC_SUCCESS on success, otherwise an error code.
60 */
61static inline int vlc_va_Setup(vlc_va_t *va, void **hw, vlc_fourcc_t *output,
62 int width, int height)
63{
64 return va->setup(va, hw, output, width, height);
65}
66
67/**
68 * Allocates a hardware video surface for a libavcodec frame.
69 * The surface will be used as output for the hardware decoder, and possibly
70 * also as a reference frame to decode other surfaces.
71 *
72 * @param opaque pointer to storage space for surface internal data [OUT]
73 * @param data pointer to the AVFrame data[0] and data[3] pointers [OUT]
74 *
75 * @note This function needs not be reentrant. However it may be called
76 * concurrently with vlc_va_Extract() and/or vlc_va_Release() from other
77 * threads and other frames.
78 *
79 * @param frame libavcodec frame [IN/OUT]
80 * @return VLC_SUCCESS on success, otherwise an error code.
81 */
82static inline int vlc_va_Get(vlc_va_t *va, void **opaque, uint8_t **data)
83{
84 return va->get(va, opaque, data);
85}
86
87/**
88 * Releases a hardware surface from a libavcodec frame.
89 * The surface has been previously allocated with vlc_va_Get().
90 *
91 * @param opaque opaque data pointer of the AVFrame set by vlc_va_Get()
92 * @param data data[0] pointer of the AVFrame set by vlc_va_Get()
93 *
94 * @note This function needs not be reentrant. However it may be called
95 * concurrently with vlc_va_Get() and/or vlc_va_Extract() from other threads
96 * and other frames.
97 *
98 * @param frame libavcodec frame previously allocated by vlc_va_Get()
99 */
100static inline void vlc_va_Release(vlc_va_t *va, void *opaque, uint8_t *data)
101{
102 va->release(opaque, data);
103}
104
105/**
106 * Extracts a hardware surface from a libavcodec frame into a VLC picture.
107 * The surface has been previously allocated with vlc_va_Get() and decoded
108 * by the libavcodec hardware acceleration.
109 * The surface may still be used by libavcodec as a reference frame until it is
110 * freed with vlc_va_Release().
111 *
112 * @note This function needs not be reentrant, but it may run concurrently with
113 * vlc_va_Get() or vlc_va_Release() in other threads (with distinct frames).
114 *
115 * @param frame libavcodec frame previously allocated by vlc_va_Get()
116 */
117static inline int vlc_va_Extract(vlc_va_t *va, picture_t *dst, void *opaque,
118 uint8_t *data)
119{
120 return va->extract(va, dst, opaque, data);
121}
122
123/**
124 * Destroys a libavcodec hardware acceleration back-end.
125 * All allocated surfaces shall have been released beforehand.
126 */
127void vlc_va_Delete(vlc_va_t *);
128
129#endif
