113static int Open( vlc_va_t *va, int i_codec_id, int i_thread_count )
114{
115 vlc_va_sys_t *sys = calloc( 1, sizeof(*sys) );
116 if ( unlikely(sys == NULL) )
117 return VLC_ENOMEM;
118
119 VAProfile i_profile, *p_profiles_list;
120 bool b_supported_profile = false;
121 int i_profiles_nb = 0;
122 int i_surface_count;
123
124 /* */
125 switch( i_codec_id )
126 {
127 case AV_CODEC_ID_MPEG1VIDEO:
128 case AV_CODEC_ID_MPEG2VIDEO:
129 i_profile = VAProfileMPEG2Main;
130 i_surface_count = 2 + 2;
131 break;
132 case AV_CODEC_ID_MPEG4:
133 i_profile = VAProfileMPEG4AdvancedSimple;
134 i_surface_count = 2+1;
135 break;
136 case AV_CODEC_ID_WMV3:
137 i_profile = VAProfileVC1Main;
138 i_surface_count = 2+1;
139 break;
140 case AV_CODEC_ID_VC1:
141 i_profile = VAProfileVC1Advanced;
142 i_surface_count = 2+1;
143 break;
144 case AV_CODEC_ID_H264:
145 i_profile = VAProfileH264High;
146 i_surface_count = 16 + i_thread_count + 2;
147 break;;
148 default:
149 free( sys );
150 return VLC_EGENERIC;
151 }
152
153 /* */
154 sys->i_config_id = VA_INVALID_ID;
155 sys->i_context_id = VA_INVALID_ID;
156 sys->image.image_id = VA_INVALID_ID;
157
158 /* Create a VA display */
159#ifdef VLC_VA_BACKEND_XLIB
160 sys->p_display_x11 = XOpenDisplay(NULL);
161 if( !sys->p_display_x11 )
162 {
163 msg_Err( va, "Could not connect to X server" );
164 goto error;
165 }
166
167 sys->p_display = vaGetDisplay( sys->p_display_x11 );
168#endif
169#ifdef VLC_VA_BACKEND_DRM
170 sys->drm_fd = vlc_open("/dev/dri/card0", O_RDWR);
171 if( sys->drm_fd == -1 )
172 {
173 msg_Err( va, "Could not access rendering device: %m" );
174 goto error;
175 }
176
177 sys->p_display = vaGetDisplayDRM( sys->drm_fd );
178#endif
179 if( !sys->p_display )
180 {
181 msg_Err( va, "Could not get a VAAPI device" );
182 goto error;
183 }
184
185 int major, minor;
186
187 if( vaInitialize( sys->p_display, &major, &minor ) )
188 {
189 msg_Err( va, "Failed to initialize the VAAPI device" );
190 goto error;
191 }
192
193 /* Check if the selected profile is supported */
194 i_profiles_nb = vaMaxNumProfiles( sys->p_display );
195 p_profiles_list = calloc( i_profiles_nb, sizeof( VAProfile ) );
196 if( !p_profiles_list )
197 goto error;
198
199 VAStatus i_status = vaQueryConfigProfiles( sys->p_display, p_profiles_list, &i_profiles_nb );
200 if ( i_status == VA_STATUS_SUCCESS )
201 {
202 for( int i = 0; i < i_profiles_nb; i++ )
203 {
204 if ( p_profiles_list[i] == i_profile )
205 {
206 b_supported_profile = true;
207 break;
208 }
209 }
210 }
211 free( p_profiles_list );
212 if ( !b_supported_profile )
213 {
214 msg_Dbg( va, "Codec and profile not supported by the hardware" );
215 goto error;
216 }
217
218 /* Create a VA configuration */
219 VAConfigAttrib attrib;
220 memset( &attrib, 0, sizeof(attrib) );
221 attrib.type = VAConfigAttribRTFormat;
222 if( vaGetConfigAttributes( sys->p_display,
223 i_profile, VAEntrypointVLD, &attrib, 1 ) )
224 goto error;
225
226 /* Not sure what to do if not, I don't have a way to test */
227 if( (attrib.value & VA_RT_FORMAT_YUV420) == 0 )
228 goto error;
229 if( vaCreateConfig( sys->p_display,
230 i_profile, VAEntrypointVLD, &attrib, 1, &sys->i_config_id ) )
231 {
232 sys->i_config_id = VA_INVALID_ID;
233 goto error;
234 }
235
236 sys->i_surface_count = i_surface_count;
237
238 sys->b_supports_derive = false;
239
240 vlc_mutex_init(&sys->lock);
241
242 va->sys = sys;
243 va->description = vaQueryVendorString( sys->p_display );
244 return VLC_SUCCESS;
245
246error:
247 if( sys->p_display != NULL )
248 vaTerminate( sys->p_display );
249#ifdef VLC_VA_BACKEND_XLIB
250 if( sys->p_display_x11 != NULL )
251 XCloseDisplay( sys->p_display_x11 );
252#endif
253#ifdef VLC_VA_BACKEND_DRM
254 if( sys->drm_fd != -1 )
255 close( sys->drm_fd );
256#endif
257 free( sys );
258 return VLC_EGENERIC;
259}

261static void DestroySurfaces( vlc_va_sys_t *sys )
262{
263 if( sys->image.image_id != VA_INVALID_ID )
264 {
265 CopyCleanCache( &sys->image_cache );
266 vaDestroyImage( sys->p_display, sys->image.image_id );
267 }
268 else if(sys->b_supports_derive)
269 {
270 CopyCleanCache( &sys->image_cache );
271 }
272
273 if( sys->i_context_id != VA_INVALID_ID )
274 vaDestroyContext( sys->p_display, sys->i_context_id );
275
276 for( int i = 0; i < sys->i_surface_count && sys->p_surface; i++ )
277 {
278 vlc_va_surface_t *p_surface = &sys->p_surface[i];
279
280 if( p_surface->i_id != VA_INVALID_SURFACE )
281 vaDestroySurfaces( sys->p_display, &p_surface->i_id, 1 );
282 }
283 free( sys->p_surface );
284
285 /* */
286 sys->image.image_id = VA_INVALID_ID;
287 sys->i_context_id = VA_INVALID_ID;
288 sys->p_surface = NULL;
289 sys->i_surface_width = 0;
290 sys->i_surface_height = 0;
291 vlc_mutex_destroy(&sys->lock);
292}

294static int CreateSurfaces( vlc_va_sys_t *sys, void **pp_hw_ctx, vlc_fourcc_t *pi_chroma,
295 int i_width, int i_height )
296{
297 assert( i_width > 0 && i_height > 0 );
298
299 /* */
300 sys->p_surface = calloc( sys->i_surface_count, sizeof(*sys->p_surface) );
301 if( !sys->p_surface )
302 return VLC_EGENERIC;
303 sys->image.image_id = VA_INVALID_ID;
304 sys->i_context_id = VA_INVALID_ID;
305
306 /* Create surfaces */
307 VASurfaceID pi_surface_id[sys->i_surface_count];
308 if( vaCreateSurfaces( sys->p_display, VA_RT_FORMAT_YUV420, i_width, i_height,
309 pi_surface_id, sys->i_surface_count, NULL, 0 ) )
310 {
311 for( int i = 0; i < sys->i_surface_count; i++ )
312 sys->p_surface[i].i_id = VA_INVALID_SURFACE;
313 goto error;
314 }
315
316 for( int i = 0; i < sys->i_surface_count; i++ )
317 {
318 vlc_va_surface_t *p_surface = &sys->p_surface[i];
319
320 p_surface->i_id = pi_surface_id[i];
321 p_surface->i_refcount = 0;
322 p_surface->i_order = 0;
323 p_surface->p_lock = &sys->lock;
324 }
325
326 /* Create a context */
327 if( vaCreateContext( sys->p_display, sys->i_config_id,
328 i_width, i_height, VA_PROGRESSIVE,
329 pi_surface_id, sys->i_surface_count, &sys->i_context_id ) )
330 {
331 sys->i_context_id = VA_INVALID_ID;
332 goto error;
333 }
334
335 /* Find and create a supported image chroma */
336 int i_fmt_count = vaMaxNumImageFormats( sys->p_display );
337 VAImageFormat *p_fmt = calloc( i_fmt_count, sizeof(*p_fmt) );
338 if( !p_fmt )
339 goto error;
340
341 if( vaQueryImageFormats( sys->p_display, p_fmt, &i_fmt_count ) )
342 {
343 free( p_fmt );
344 goto error;
345 }
346
347 VAImage test_image;
348 if(vaDeriveImage(sys->p_display, pi_surface_id[0], &test_image) == VA_STATUS_SUCCESS)
349 {
350 sys->b_supports_derive = true;
351 vaDestroyImage(sys->p_display, test_image.image_id);
352 }
353
354 vlc_fourcc_t i_chroma = 0;
355 for( int i = 0; i < i_fmt_count; i++ )
356 {
357 if( p_fmt[i].fourcc == VA_FOURCC_YV12 ||
358 p_fmt[i].fourcc == VA_FOURCC_IYUV ||
359 p_fmt[i].fourcc == VA_FOURCC_NV12 )
360 {
361 if( vaCreateImage( sys->p_display, &p_fmt[i], i_width, i_height, &sys->image ) )
362 {
363 sys->image.image_id = VA_INVALID_ID;
364 continue;
365 }
366 /* Validate that vaGetImage works with this format */
367 if( vaGetImage( sys->p_display, pi_surface_id[0],
368 0, 0, i_width, i_height,
369 sys->image.image_id) )
370 {
371 vaDestroyImage( sys->p_display, sys->image.image_id );
372 sys->image.image_id = VA_INVALID_ID;
373 continue;
374 }
375
376 i_chroma = VLC_CODEC_YV12;
377 break;
378 }
379 }
380 free( p_fmt );
381 if( !i_chroma )
382 goto error;
383 *pi_chroma = i_chroma;
384
385 if(sys->b_supports_derive)
386 {
387 vaDestroyImage( sys->p_display, sys->image.image_id );
388 sys->image.image_id = VA_INVALID_ID;
389 }
390
391 if( unlikely(CopyInitCache( &sys->image_cache, i_width )) )
392 goto error;
393
394 /* Setup the ffmpeg hardware context */
395 *pp_hw_ctx = &sys->hw_ctx;
396
397 memset( &sys->hw_ctx, 0, sizeof(sys->hw_ctx) );
398 sys->hw_ctx.display = sys->p_display;
399 sys->hw_ctx.config_id = sys->i_config_id;
400 sys->hw_ctx.context_id = sys->i_context_id;
401
402 /* */
403 sys->i_surface_chroma = i_chroma;
404 sys->i_surface_width = i_width;
405 sys->i_surface_height = i_height;
406 return VLC_SUCCESS;
407
408error:
409 DestroySurfaces( sys );
410 return VLC_EGENERIC;
411}

413static int Setup( vlc_va_t *va, void **pp_hw_ctx, vlc_fourcc_t *pi_chroma,
414 int i_width, int i_height )
415{
416 vlc_va_sys_t *sys = va->sys;
417
418 if( sys->i_surface_width == i_width &&
419 sys->i_surface_height == i_height )
420 {
421 *pp_hw_ctx = &sys->hw_ctx;
422 *pi_chroma = sys->i_surface_chroma;
423 return VLC_SUCCESS;
424 }
425
426 *pp_hw_ctx = NULL;
427 *pi_chroma = 0;
428 if( sys->i_surface_width || sys->i_surface_height )
429 DestroySurfaces( sys );
430
431 if( i_width > 0 && i_height > 0 )
432 return CreateSurfaces( sys, pp_hw_ctx, pi_chroma, i_width, i_height );
433
434 return VLC_EGENERIC;
435}

437static int Extract( vlc_va_t *va, picture_t *p_picture, void *opaque,
438 uint8_t *data )
439{
440 vlc_va_sys_t *sys = va->sys;
441 VASurfaceID i_surface_id = (VASurfaceID)(uintptr_t)data;
442
443#if VA_CHECK_VERSION(0,31,0)
444 if( vaSyncSurface( sys->p_display, i_surface_id ) )
445#else
446 if( vaSyncSurface( sys->p_display, sys->i_context_id, i_surface_id ) )
447#endif
448 return VLC_EGENERIC;
449
450 if(sys->b_supports_derive)
451 {
452 if(vaDeriveImage(sys->p_display, i_surface_id, &(sys->image)) != VA_STATUS_SUCCESS)
453 return VLC_EGENERIC;
454 }
455 else
456 {
457 if( vaGetImage( sys->p_display, i_surface_id,
458 0, 0, sys->i_surface_width, sys->i_surface_height,
459 sys->image.image_id) )
460 return VLC_EGENERIC;
461 }
462
463 void *p_base;
464 if( vaMapBuffer( sys->p_display, sys->image.buf, &p_base ) )
465 return VLC_EGENERIC;
466
467 const uint32_t i_fourcc = sys->image.format.fourcc;
468 if( i_fourcc == VA_FOURCC_YV12 ||
469 i_fourcc == VA_FOURCC_IYUV )
470 {
471 bool b_swap_uv = i_fourcc == VA_FOURCC_IYUV;
472 uint8_t *pp_plane[3];
473 size_t pi_pitch[3];
474
475 for( int i = 0; i < 3; i++ )
476 {
477 const int i_src_plane = (b_swap_uv && i != 0) ? (3 - i) : i;
478 pp_plane[i] = (uint8_t*)p_base + sys->image.offsets[i_src_plane];
479 pi_pitch[i] = sys->image.pitches[i_src_plane];
480 }
481 CopyFromYv12( p_picture, pp_plane, pi_pitch,
482 sys->i_surface_width,
483 sys->i_surface_height,
484 &sys->image_cache );
485 }
486 else
487 {
488 assert( i_fourcc == VA_FOURCC_NV12 );
489 uint8_t *pp_plane[2];
490 size_t pi_pitch[2];
491
492 for( int i = 0; i < 2; i++ )
493 {
494 pp_plane[i] = (uint8_t*)p_base + sys->image.offsets[i];
495 pi_pitch[i] = sys->image.pitches[i];
496 }
497 CopyFromNv12( p_picture, pp_plane, pi_pitch,
498 sys->i_surface_width,
499 sys->i_surface_height,
500 &sys->image_cache );
501 }
502
503 if( vaUnmapBuffer( sys->p_display, sys->image.buf ) )
504 return VLC_EGENERIC;
505
506 if(sys->b_supports_derive)
507 {
508 vaDestroyImage( sys->p_display, sys->image.image_id );
509 sys->image.image_id = VA_INVALID_ID;
510 }
511 (void) opaque;
512 return VLC_SUCCESS;
513}

515static int Get( vlc_va_t *va, void **opaque, uint8_t **data )
516{
517 vlc_va_sys_t *sys = va->sys;
518 int i_old;
519 int i;
520
521 vlc_mutex_lock( &sys->lock );
522 /* Grab an unused surface, in case none are, try the oldest
523 * XXX using the oldest is a workaround in case a problem happens with ffmpeg */
524 for( i = 0, i_old = 0; i < sys->i_surface_count; i++ )
525 {
526 vlc_va_surface_t *p_surface = &sys->p_surface[i];
527
528 if( !p_surface->i_refcount )
529 break;
530
531 if( p_surface->i_order < sys->p_surface[i_old].i_order )
532 i_old = i;
533 }
534 if( i >= sys->i_surface_count )
535 i = i_old;
536 vlc_mutex_unlock( &sys->lock );
537
538 vlc_va_surface_t *p_surface = &sys->p_surface[i];
539
540 p_surface->i_refcount = 1;
541 p_surface->i_order = sys->i_surface_order++;
542 *data = (void *)(uintptr_t)p_surface->i_id;
543 *opaque = p_surface;
544 return VLC_SUCCESS;
545}
546
547static void Release( void *opaque, uint8_t *data )
548{
549 vlc_va_surface_t *p_surface = opaque;
550
551 vlc_mutex_lock( p_surface->p_lock );
552 p_surface->i_refcount--;
553 vlc_mutex_unlock( p_surface->p_lock );
554 (void) data;
555}

557static void Close( vlc_va_sys_t *sys )
558{
559 if( sys->i_surface_width || sys->i_surface_height )
560 DestroySurfaces( sys );
561
562 if( sys->i_config_id != VA_INVALID_ID )
563 vaDestroyConfig( sys->p_display, sys->i_config_id );
564 vaTerminate( sys->p_display );
565#ifdef VLC_VA_BACKEND_XLIB
566 XCloseDisplay( sys->p_display_x11 );
567#endif
568#ifdef VLC_VA_BACKEND_DRM
569 close( sys->drm_fd );
570#endif
571}
572
573static void Delete( vlc_va_t *va )
574{
575 vlc_va_sys_t *sys = va->sys;
576 Close( sys );
577 free( sys );
578}

580static int Create( vlc_va_t *p_va, AVCodecContext *ctx,
581 const es_format_t *fmt )
582{
583#ifdef VLC_VA_BACKEND_XLIB
584 if( !vlc_xlib_init( VLC_OBJECT(p_va) ) )
585 {
586 msg_Warn( p_va, "Ignoring VA API" );
587 return VLC_EGENERIC;
588 }
589#endif
590
591 (void) fmt;
592
593 int err = Open( p_va, ctx->codec_id, ctx->thread_count );
594 if( err )
595 return err;
596
597 /* Only VLD supported */
598 p_va->pix_fmt = PIX_FMT_VAAPI_VLD;
599 p_va->setup = Setup;
600 p_va->get = Get;
601 p_va->release = Release;
602 p_va->extract = Extract;
603 return VLC_SUCCESS;
604}
