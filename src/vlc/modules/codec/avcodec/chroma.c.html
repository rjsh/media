148/* FIXME special case the RGB formats */
149int GetFfmpegChroma( int *restrict i_ffmpeg_chroma, const video_format_t *fmt )
150{
151 for( int i = 0; chroma_table[i].i_chroma != 0; i++ )
152 {
153 if( chroma_table[i].i_chroma == fmt->i_chroma )
154 {
155 if( ( chroma_table[i].i_rmask == 0 &&
156 chroma_table[i].i_gmask == 0 &&
157 chroma_table[i].i_bmask == 0 ) ||
158 ( chroma_table[i].i_rmask == fmt->i_rmask &&
159 chroma_table[i].i_gmask == fmt->i_gmask &&
160 chroma_table[i].i_bmask == fmt->i_bmask ) )
161 {
162 *i_ffmpeg_chroma = chroma_table[i].i_chroma_id;
163 return VLC_SUCCESS;
164 }
165 }
166 }
167 return VLC_EGENERIC;
168}

170vlc_fourcc_t FindVlcChroma( int i_ffmpeg_id )
171{
172 for( int i = 0; chroma_table[i].i_chroma != 0; i++ )
173 if( chroma_table[i].i_chroma_id == i_ffmpeg_id )
174 return chroma_table[i].i_chroma;
175 return 0;
176}
177
178int GetVlcChroma( video_format_t *fmt, int i_ffmpeg_chroma )
179{
180 /* TODO FIXME for rgb format we HAVE to set rgb mask/shift */
181 for( int i = 0; chroma_table[i].i_chroma != 0; i++ )
182 {
183 if( chroma_table[i].i_chroma_id == i_ffmpeg_chroma )
184 {
185 fmt->i_rmask = chroma_table[i].i_rmask;
186 fmt->i_gmask = chroma_table[i].i_gmask;
187 fmt->i_bmask = chroma_table[i].i_bmask;
188 fmt->i_chroma = chroma_table[i].i_chroma;
189 return VLC_SUCCESS;
190 }
191 }
192 return VLC_EGENERIC;
193}
194
195int FindFfmpegChroma( vlc_fourcc_t fourcc )
196{
197 for( int i = 0; chroma_table[i].i_chroma != 0; i++ )
198 if( chroma_table[i].i_chroma == fourcc )
199 return chroma_table[i].i_chroma_id;
200 return PIX_FMT_NONE;
201}
