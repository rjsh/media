45/*****************************************************************************
46 * decoder_sys_t : decoder descriptor
47 *****************************************************************************/
48struct decoder_sys_t
49{
50 AVCODEC_COMMON_MEMBERS
51
52 /* Video decoder specific part */
53 mtime_t i_pts;
54
55 AVFrame *p_ff_pic;
56
57 /* for frame skipping algo */
58 bool b_hurry_up;
59 enum AVDiscard i_skip_frame;
60 enum AVDiscard i_skip_idct;
61
62 /* how many decoded frames are late */
63 int i_late_frames;
64 mtime_t i_late_frames_start;
65
66 /* for direct rendering */
67 bool b_direct_rendering;
68 int i_direct_rendering_used;
69
70 bool b_has_b_frames;
71
72 /* Hack to force display of still pictures */
73 bool b_first_frame;
74
75
76 /* */
77#if LIBAVCODEC_VERSION_MAJOR < 54
78 AVPaletteControl palette;
79#else
80 bool palette_sent;
81#endif
82
83 /* */
84 bool b_flush;
85
86 /* VA API */
87 vlc_va_t *p_va;
88
89 vlc_sem_t sem_mt;
90};
91

114static uint32_t ffmpeg_CodecTag( vlc_fourcc_t fcc )
115{
116 uint8_t *p = (uint8_t*)&fcc;
117 return p[0] | (p[1] << 8) | (p[2] << 16) | (p[3] << 24);
118}

124/* Returns a new picture buffer */
125static inline picture_t *ffmpeg_NewPictBuf( decoder_t *p_dec,
126 AVCodecContext *p_context )
127{
128 decoder_sys_t *p_sys = p_dec->p_sys;
129 int width = p_context->coded_width;
130 int height = p_context->coded_height;
131
132 if( p_sys->p_va == NULL )
133 {
134 int aligns[AV_NUM_DATA_POINTERS];
135
136 avcodec_align_dimensions2(p_context, &width, &height, aligns);
137 }
138
139
140 if( width == 0 || height == 0 || width > 8192 || height > 8192 )
141 {
142 msg_Err( p_dec, "Invalid frame size %dx%d.", width, height );
143 return NULL; /* invalid display size */
144 }
145 p_dec->fmt_out.video.i_width = width;
146 p_dec->fmt_out.video.i_height = height;
147
148 if( width != p_context->width || height != p_context->height )
149 {
150 p_dec->fmt_out.video.i_visible_width = p_context->width;
151 p_dec->fmt_out.video.i_visible_height = p_context->height;
152 }
153 else
154 {
155 p_dec->fmt_out.video.i_visible_width = width;
156 p_dec->fmt_out.video.i_visible_height = height;
157 }
158
159 if( !p_sys->p_va && GetVlcChroma( &p_dec->fmt_out.video, p_context->pix_fmt ) )
160 {
161 /* we are doomed, but not really, because most codecs set their pix_fmt
162 * much later
163 * FIXME does it make sense here ? */
164 p_dec->fmt_out.video.i_chroma = VLC_CODEC_I420;
165 }
166 p_dec->fmt_out.i_codec = p_dec->fmt_out.video.i_chroma;
167
168 /* If an aspect-ratio was specified in the input format then force it */
169 if( p_dec->fmt_in.video.i_sar_num > 0 && p_dec->fmt_in.video.i_sar_den > 0 )
170 {
171 p_dec->fmt_out.video.i_sar_num = p_dec->fmt_in.video.i_sar_num;
172 p_dec->fmt_out.video.i_sar_den = p_dec->fmt_in.video.i_sar_den;
173 }
174 else
175 {
176 p_dec->fmt_out.video.i_sar_num = p_context->sample_aspect_ratio.num;
177 p_dec->fmt_out.video.i_sar_den = p_context->sample_aspect_ratio.den;
178
179 if( !p_dec->fmt_out.video.i_sar_num || !p_dec->fmt_out.video.i_sar_den )
180 {
181 p_dec->fmt_out.video.i_sar_num = 1;
182 p_dec->fmt_out.video.i_sar_den = 1;
183 }
184 }
185
186 if( p_dec->fmt_in.video.i_frame_rate > 0 &&
187 p_dec->fmt_in.video.i_frame_rate_base > 0 )
188 {
189 p_dec->fmt_out.video.i_frame_rate =
190 p_dec->fmt_in.video.i_frame_rate;
191 p_dec->fmt_out.video.i_frame_rate_base =
192 p_dec->fmt_in.video.i_frame_rate_base;
193 }
194 else if( p_context->time_base.num > 0 && p_context->time_base.den > 0 )
195 {
196 p_dec->fmt_out.video.i_frame_rate = p_context->time_base.den;
197 p_dec->fmt_out.video.i_frame_rate_base = p_context->time_base.num * __MAX( p_context->ticks_per_frame, 1 );
198 }
199
200 return decoder_NewPicture( p_dec );
201}

203/*****************************************************************************
204 * InitVideo: initialize the video decoder
205 *****************************************************************************
206 * the ffmpeg codec will be opened, some memory allocated. The vout is not yet
207 * opened (done after the first decoded frame).
208 *****************************************************************************/
209int InitVideoDec( decoder_t *p_dec, AVCodecContext *p_context,
210 AVCodec *p_codec, int i_codec_id, const char *psz_namecodec )
211{
212 decoder_sys_t *p_sys;
213 int i_val;
214
215 /* Allocate the memory needed to store the decoder's structure */
216 if( ( p_dec->p_sys = p_sys = calloc( 1, sizeof(decoder_sys_t) ) ) == NULL )
217 return VLC_ENOMEM;
218
219 p_codec->type = AVMEDIA_TYPE_VIDEO;
220 p_context->codec_type = AVMEDIA_TYPE_VIDEO;
221 p_context->codec_id = i_codec_id;
222 p_sys->p_context = p_context;
223 p_sys->p_codec = p_codec;
224 p_sys->i_codec_id = i_codec_id;
225 p_sys->psz_namecodec = psz_namecodec;
226 p_sys->p_ff_pic = avcodec_alloc_frame();
227 p_sys->b_delayed_open = true;
228 p_sys->p_va = NULL;
229 vlc_sem_init( &p_sys->sem_mt, 0 );
230
231 /* ***** Fill p_context with init values ***** */
232 p_sys->p_context->codec_tag = ffmpeg_CodecTag( p_dec->fmt_in.i_original_fourcc ?: p_dec->fmt_in.i_codec );
233
234 /* ***** Get configuration of ffmpeg plugin ***** */
235 p_sys->p_context->workaround_bugs =
236 var_InheritInteger( p_dec, "avcodec-workaround-bugs" );
237 p_sys->p_context->err_recognition =
238 var_InheritInteger( p_dec, "avcodec-error-resilience" );
239
240 if( var_CreateGetBool( p_dec, "grayscale" ) )
241 p_sys->p_context->flags |= CODEC_FLAG_GRAY;
242
243 /* ***** Output always the frames ***** */
244#if LIBAVCODEC_VERSION_CHECK(55, 23, 1, 40, 101)
245 p_sys->p_context->flags |= CODEC_FLAG_OUTPUT_CORRUPT;
246#endif
247
248 i_val = var_CreateGetInteger( p_dec, "avcodec-vismv" );
249 if( i_val ) p_sys->p_context->debug_mv = i_val;
250
251 i_val = var_CreateGetInteger( p_dec, "avcodec-skiploopfilter" );
252 if( i_val >= 4 ) p_sys->p_context->skip_loop_filter = AVDISCARD_ALL;
253 else if( i_val == 3 ) p_sys->p_context->skip_loop_filter = AVDISCARD_NONKEY;
254 else if( i_val == 2 ) p_sys->p_context->skip_loop_filter = AVDISCARD_BIDIR;
255 else if( i_val == 1 ) p_sys->p_context->skip_loop_filter = AVDISCARD_NONREF;
256
257 if( var_CreateGetBool( p_dec, "avcodec-fast" ) )
258 p_sys->p_context->flags2 |= CODEC_FLAG2_FAST;
259
260 /* ***** libavcodec frame skipping ***** */
261 p_sys->b_hurry_up = var_CreateGetBool( p_dec, "avcodec-hurry-up" );
262
263 i_val = var_CreateGetInteger( p_dec, "avcodec-skip-frame" );
264 if( i_val >= 4 ) p_sys->p_context->skip_frame = AVDISCARD_ALL;
265 else if( i_val == 3 ) p_sys->p_context->skip_frame = AVDISCARD_NONKEY;
266 else if( i_val == 2 ) p_sys->p_context->skip_frame = AVDISCARD_BIDIR;
267 else if( i_val == 1 ) p_sys->p_context->skip_frame = AVDISCARD_NONREF;
268 else if( i_val == -1 ) p_sys->p_context->skip_frame = AVDISCARD_NONE;
269 else p_sys->p_context->skip_frame = AVDISCARD_DEFAULT;
270 p_sys->i_skip_frame = p_sys->p_context->skip_frame;
271
272 i_val = var_CreateGetInteger( p_dec, "avcodec-skip-idct" );
273 if( i_val >= 4 ) p_sys->p_context->skip_idct = AVDISCARD_ALL;
274 else if( i_val == 3 ) p_sys->p_context->skip_idct = AVDISCARD_NONKEY;
275 else if( i_val == 2 ) p_sys->p_context->skip_idct = AVDISCARD_BIDIR;
276 else if( i_val == 1 ) p_sys->p_context->skip_idct = AVDISCARD_NONREF;
277 else if( i_val == -1 ) p_sys->p_context->skip_idct = AVDISCARD_NONE;
278 else p_sys->p_context->skip_idct = AVDISCARD_DEFAULT;
279 p_sys->i_skip_idct = p_sys->p_context->skip_idct;
280
281 /* ***** libavcodec direct rendering ***** */
282 p_sys->b_direct_rendering = false;
283 p_sys->i_direct_rendering_used = -1;
284 if( var_CreateGetBool( p_dec, "avcodec-dr" ) &&
285 (p_sys->p_codec->capabilities & CODEC_CAP_DR1) &&
286 /* No idea why ... but this fixes flickering on some TSCC streams */
287 p_sys->i_codec_id != AV_CODEC_ID_TSCC && p_sys->i_codec_id != AV_CODEC_ID_CSCD &&
288 p_sys->i_codec_id != AV_CODEC_ID_CINEPAK &&
289 !p_sys->p_context->debug_mv )
290 {
291 /* Some codecs set pix_fmt only after the 1st frame has been decoded,
292 * so we need to do another check in ffmpeg_GetFrameBuf() */
293 p_sys->b_direct_rendering = true;
294 }
295
296 /* libavcodec doesn't properly release old pictures when frames are skipped */
297 //if( p_sys->b_hurry_up ) p_sys->b_direct_rendering = false;
298 if( p_sys->b_direct_rendering )
299 {
300 msg_Dbg( p_dec, "trying to use direct rendering" );
301 p_sys->p_context->flags |= CODEC_FLAG_EMU_EDGE;
302 }
303 else
304 {
305 msg_Dbg( p_dec, "direct rendering is disabled" );
306 }
307
308 p_sys->p_context->get_format = ffmpeg_GetFormat;
309 /* Always use our get_buffer wrapper so we can calculate the
310 * PTS correctly */
311#if LIBAVCODEC_VERSION_MAJOR >= 55
312 p_sys->p_context->get_buffer2 = lavc_GetFrame;
313#else
314 p_sys->p_context->get_buffer = ffmpeg_GetFrameBuf;
315 p_sys->p_context->reget_buffer = avcodec_default_reget_buffer;
316 p_sys->p_context->release_buffer = ffmpeg_ReleaseFrameBuf;
317#endif
318 p_sys->p_context->opaque = p_dec;
319
320#ifdef HAVE_AVCODEC_MT
321 int i_thread_count = var_InheritInteger( p_dec, "avcodec-threads" );
322 if( i_thread_count <= 0 )
323 {
324 i_thread_count = vlc_GetCPUCount();
325 if( i_thread_count > 1 )
326 i_thread_count++;
327
328 //FIXME: take in count the decoding time
329 i_thread_count = __MIN( i_thread_count, 4 );
330 }
331 i_thread_count = __MIN( i_thread_count, 16 );
332 msg_Dbg( p_dec, "allowing %d thread(s) for decoding", i_thread_count );
333 p_sys->p_context->thread_count = i_thread_count;
334 p_sys->p_context->thread_safe_callbacks = true;
335
336 switch( i_codec_id )
337 {
338 case AV_CODEC_ID_MPEG4:
339 case AV_CODEC_ID_H263:
340 p_sys->p_context->thread_type = 0;
341 break;
342 case AV_CODEC_ID_MPEG1VIDEO:
343 case AV_CODEC_ID_MPEG2VIDEO:
344 p_sys->p_context->thread_type &= ~FF_THREAD_SLICE;
345 /* fall through */
346# if (LIBAVCODEC_VERSION_INT < AV_VERSION_INT(55, 1, 0))
347 case AV_CODEC_ID_H264:
348 case AV_CODEC_ID_VC1:
349 case AV_CODEC_ID_WMV3:
350 p_sys->p_context->thread_type &= ~FF_THREAD_FRAME;
351# endif
352 }
353
354 if( p_sys->p_context->thread_type & FF_THREAD_FRAME )
355 p_dec->i_extra_picture_buffers = 2 * p_sys->p_context->thread_count;
356#endif
357
358 /* ***** misc init ***** */
359 p_sys->i_pts = VLC_TS_INVALID;
360 p_sys->b_has_b_frames = false;
361 p_sys->b_first_frame = true;
362 p_sys->b_flush = false;
363 p_sys->i_late_frames = 0;
364
365 /* Set output properties */
366 p_dec->fmt_out.i_cat = VIDEO_ES;
367 if( GetVlcChroma( &p_dec->fmt_out.video, p_context->pix_fmt ) != VLC_SUCCESS )
368 {
369 /* we are doomed. but not really, because most codecs set their pix_fmt later on */
370 p_dec->fmt_out.i_codec = VLC_CODEC_I420;
371 }
372 p_dec->fmt_out.i_codec = p_dec->fmt_out.video.i_chroma;
373
374 p_dec->fmt_out.video.orientation = p_dec->fmt_in.video.orientation;
375
376#if LIBAVCODEC_VERSION_MAJOR < 54
377 /* Setup palette */
378 memset( &p_sys->palette, 0, sizeof(p_sys->palette) );
379 if( p_dec->fmt_in.video.p_palette )
380 {
381 p_sys->palette.palette_changed = 1;
382
383 for( int i = 0; i < __MIN( AVPALETTE_COUNT, p_dec->fmt_in.video.p_palette->i_entries ); i++ )
384 {
385 union {
386 uint32_t u;
387 uint8_t a[4];
388 } c;
389 c.a[0] = p_dec->fmt_in.video.p_palette->palette[i][0];
390 c.a[1] = p_dec->fmt_in.video.p_palette->palette[i][1];
391 c.a[2] = p_dec->fmt_in.video.p_palette->palette[i][2];
392 c.a[3] = p_dec->fmt_in.video.p_palette->palette[i][3];
393
394 p_sys->palette.palette[i] = c.u;
395 }
396 p_sys->p_context->palctrl = &p_sys->palette;
397
398 p_dec->fmt_out.video.p_palette = malloc( sizeof(video_palette_t) );
399 if( p_dec->fmt_out.video.p_palette )
400 *p_dec->fmt_out.video.p_palette = *p_dec->fmt_in.video.p_palette;
401 }
402 else if( p_sys->i_codec_id != CODEC_ID_MSVIDEO1 && p_sys->i_codec_id != CODEC_ID_CINEPAK )
403 {
404 p_sys->p_context->palctrl = &p_sys->palette;
405 }
406#else
407 if( p_dec->fmt_in.video.p_palette ) {
408 p_sys->palette_sent = false;
409 p_dec->fmt_out.video.p_palette = malloc( sizeof(video_palette_t) );
410 if( p_dec->fmt_out.video.p_palette )
411 *p_dec->fmt_out.video.p_palette = *p_dec->fmt_in.video.p_palette;
412 } else
413 p_sys->palette_sent = true;
414#endif
415
416 /* ***** init this codec with special data ***** */
417 ffmpeg_InitCodec( p_dec );
418
419 /* ***** Open the codec ***** */
420 if( ffmpeg_OpenCodec( p_dec ) < 0 )
421 {
422 msg_Err( p_dec, "cannot open codec (%s)", p_sys->psz_namecodec );
423 av_free( p_sys->p_ff_pic );
424 vlc_sem_destroy( &p_sys->sem_mt );
425 free( p_sys );
426 return VLC_EGENERIC;
427 }
428
429 return VLC_SUCCESS;
430}

432/*****************************************************************************
433 * DecodeVideo: Called to decode one or more frames
434 *****************************************************************************/
435picture_t *DecodeVideo( decoder_t *p_dec, block_t **pp_block )
436{
437 decoder_sys_t *p_sys = p_dec->p_sys;
438 AVCodecContext *p_context = p_sys->p_context;
439 int b_drawpicture;
440 block_t *p_block;
441
442 if( !pp_block )
443 return NULL;
444
445 if( !p_context->extradata_size && p_dec->fmt_in.i_extra )
446 {
447 ffmpeg_InitCodec( p_dec );
448 if( p_sys->b_delayed_open )
449 {
450 if( ffmpeg_OpenCodec( p_dec ) )
451 msg_Err( p_dec, "cannot open codec (%s)", p_sys->psz_namecodec );
452 }
453 }
454
455 p_block = *pp_block;
456 if(!p_block && !(p_sys->p_codec->capabilities & CODEC_CAP_DELAY) )
457 return NULL;
458
459 if( p_sys->b_delayed_open )
460 {
461 if( p_block )
462 block_Release( p_block );
463 return NULL;
464 }
465
466 if( p_block)
467 {
468 if( p_block->i_flags & (BLOCK_FLAG_DISCONTINUITY|BLOCK_FLAG_CORRUPTED) )
469 {
470 p_sys->i_pts = VLC_TS_INVALID; /* To make sure we recover properly */
471
472 p_sys->i_late_frames = 0;
473
474 post_mt( p_sys );
475 if( p_block->i_flags & BLOCK_FLAG_DISCONTINUITY )
476 avcodec_flush_buffers( p_context );
477 wait_mt( p_sys );
478
479 block_Release( p_block );
480 return NULL;
481 }
482
483 if( p_block->i_flags & BLOCK_FLAG_PREROLL )
484 {
485 /* Do not care about late frames when prerolling
486 * TODO avoid decoding of non reference frame
487 * (ie all B except for H264 where it depends only on nal_ref_idc) */
488 p_sys->i_late_frames = 0;
489 }
490 }
491
492 if( !p_dec->b_pace_control && (p_sys->i_late_frames > 0) &&
493 (mdate() - p_sys->i_late_frames_start > INT64_C(5000000)) )
494 {
495 if( p_sys->i_pts > VLC_TS_INVALID )
496 {
497 p_sys->i_pts = VLC_TS_INVALID; /* To make sure we recover properly */
498 }
499 if( p_block )
500 block_Release( p_block );
501 p_sys->i_late_frames--;
502 msg_Err( p_dec, "more than 5 seconds of late video -> "
503 "dropping frame (computer too slow ?)" );
504 return NULL;
505 }
506
507 /* A good idea could be to decode all I pictures and see for the other */
508 if( !p_dec->b_pace_control &&
509 p_sys->b_hurry_up &&
510 (p_sys->i_late_frames > 4) )
511 {
512 b_drawpicture = 0;
513 if( p_sys->i_late_frames < 12 )
514 {
515 p_context->skip_frame =
516 (p_sys->i_skip_frame <= AVDISCARD_NONREF) ?
517 AVDISCARD_NONREF : p_sys->i_skip_frame;
518 }
519 else
520 {
521 /* picture too late, won't decode
522 * but break picture until a new I, and for mpeg4 ...*/
523 p_sys->i_late_frames--; /* needed else it will never be decrease */
524 if( p_block )
525 block_Release( p_block );
526 msg_Warn( p_dec, "More than 4 late frames, dropping frame" );
527 return NULL;
528 }
529 }
530 else
531 {
532 if( p_sys->b_hurry_up )
533 p_context->skip_frame = p_sys->i_skip_frame;
534 if( !p_block || !(p_block->i_flags & BLOCK_FLAG_PREROLL) )
535 b_drawpicture = 1;
536 else
537 b_drawpicture = 0;
538 }
539
540 if( p_context->width <= 0 || p_context->height <= 0 )
541 {
542 if( p_sys->b_hurry_up )
543 p_context->skip_frame = p_sys->i_skip_frame;
544 }
545 else if( !b_drawpicture )
546 {
547 /* It creates broken picture
548 * FIXME either our parser or ffmpeg is broken */
549#if 0
550 if( p_sys->b_hurry_up )
551 p_context->skip_frame = __MAX( p_context->skip_frame,
552 AVDISCARD_NONREF );
553#endif
554 }
555
556 /*
557 * Do the actual decoding now */
558
559 /* Don't forget that libavcodec requires a little more bytes
560 * that the real frame size */
561 if( p_block && p_block->i_buffer > 0 )
562 {
563 p_sys->b_flush = ( p_block->i_flags & BLOCK_FLAG_END_OF_SEQUENCE ) != 0;
564
565 p_block = block_Realloc( p_block, 0,
566 p_block->i_buffer + FF_INPUT_BUFFER_PADDING_SIZE );
567 if( !p_block )
568 return NULL;
569 p_block->i_buffer -= FF_INPUT_BUFFER_PADDING_SIZE;
570 *pp_block = p_block;
571 memset( p_block->p_buffer + p_block->i_buffer, 0,
572 FF_INPUT_BUFFER_PADDING_SIZE );
573 }
574
575 while( !p_block || p_block->i_buffer > 0 || p_sys->b_flush )
576 {
577 int i_used, b_gotpicture;
578 picture_t *p_pic;
579 AVPacket pkt;
580
581 post_mt( p_sys );
582
583 av_init_packet( &pkt );
584 if( p_block )
585 {
586 pkt.data = p_block->p_buffer;
587 pkt.size = p_block->i_buffer;
588 pkt.pts = p_block->i_pts;
589 pkt.dts = p_block->i_dts;
590 }
591 else
592 {
593 /* Return delayed frames if codec has CODEC_CAP_DELAY */
594 pkt.data = NULL;
595 pkt.size = 0;
596 }
597
598#if LIBAVCODEC_VERSION_MAJOR >= 54
599 if( !p_sys->palette_sent )
600 {
601 uint8_t *pal = av_packet_new_side_data(&pkt, AV_PKT_DATA_PALETTE, AVPALETTE_SIZE);
602 if (pal) {
603 memcpy(pal, p_dec->fmt_in.video.p_palette->palette, AVPALETTE_SIZE);
604 p_sys->palette_sent = true;
605 }
606 }
607#endif
608
609 /* Make sure we don't reuse the same timestamps twice */
610 if( p_block )
611 {
612 p_block->i_pts =
613 p_block->i_dts = VLC_TS_INVALID;
614 }
615
616 i_used = avcodec_decode_video2( p_context, p_sys->p_ff_pic,
617 &b_gotpicture, &pkt );
618
619 wait_mt( p_sys );
620
621 if( p_sys->b_flush )
622 p_sys->b_first_frame = true;
623
624 if( p_block )
625 {
626 if( p_block->i_buffer <= 0 )
627 p_sys->b_flush = false;
628
629 if( i_used < 0 )
630 {
631 if( b_drawpicture )
632 msg_Warn( p_dec, "cannot decode one frame (%zu bytes)",
633 p_block->i_buffer );
634 block_Release( p_block );
635 return NULL;
636 }
637 else if( (unsigned)i_used > p_block->i_buffer ||
638 p_context->thread_count > 1 )
639 {
640 i_used = p_block->i_buffer;
641 }
642
643 /* Consumed bytes */
644 p_block->i_buffer -= i_used;
645 p_block->p_buffer += i_used;
646 }
647
648 /* Nothing to display */
649 if( !b_gotpicture )
650 {
651 if( i_used == 0 ) break;
652 continue;
653 }
654
655 /* Sanity check (seems to be needed for some streams) */
656 if( p_sys->p_ff_pic->pict_type == AV_PICTURE_TYPE_B)
657 {
658 p_sys->b_has_b_frames = true;
659 }
660
661 /* Compute the PTS */
662 mtime_t i_pts =
663 p_sys->p_ff_pic->pkt_pts;
664 if (i_pts <= VLC_TS_INVALID)
665 i_pts = p_sys->p_ff_pic->pkt_dts;
666
667 if( i_pts <= VLC_TS_INVALID )
668 i_pts = p_sys->i_pts;
669
670 /* Interpolate the next PTS */
671 if( i_pts > VLC_TS_INVALID )
672 p_sys->i_pts = i_pts;
673 if( p_sys->i_pts > VLC_TS_INVALID )
674 {
675 /* interpolate the next PTS */
676 if( p_dec->fmt_in.video.i_frame_rate > 0 &&
677 p_dec->fmt_in.video.i_frame_rate_base > 0 )
678 {
679 p_sys->i_pts += INT64_C(1000000) *
680 (2 + p_sys->p_ff_pic->repeat_pict) *
681 p_dec->fmt_in.video.i_frame_rate_base /
682 (2 * p_dec->fmt_in.video.i_frame_rate);
683 }
684 else if( p_context->time_base.den > 0 )
685 {
686 int i_tick = p_context->ticks_per_frame;
687 if( i_tick <= 0 )
688 i_tick = 1;
689
690 p_sys->i_pts += INT64_C(1000000) *
691 (2 + p_sys->p_ff_pic->repeat_pict) *
692 i_tick * p_context->time_base.num /
693 (2 * p_context->time_base.den);
694 }
695 }
696
697 /* Update frame late count (except when doing preroll) */
698 mtime_t i_display_date = 0;
699 if( !p_block || !(p_block->i_flags & BLOCK_FLAG_PREROLL) )
700 i_display_date = decoder_GetDisplayDate( p_dec, i_pts );
701
702 if( i_display_date > 0 && i_display_date <= mdate() )
703 {
704 p_sys->i_late_frames++;
705 if( p_sys->i_late_frames == 1 )
706 p_sys->i_late_frames_start = mdate();
707 }
708 else
709 {
710 p_sys->i_late_frames = 0;
711 }
712
713 if( !b_drawpicture || ( !p_sys->p_va && !p_sys->p_ff_pic->linesize[0] ) )
714 continue;
715
716 if( p_sys->p_va != NULL || p_sys->p_ff_pic->opaque == NULL )
717 {
718 /* Get a new picture */
719 p_pic = ffmpeg_NewPictBuf( p_dec, p_context );
720 if( !p_pic )
721 {
722 if( p_block )
723 block_Release( p_block );
724 return NULL;
725 }
726
727 /* Fill p_picture_t from AVVideoFrame and do chroma conversion
728 * if needed */
729 ffmpeg_CopyPicture( p_dec, p_pic, p_sys->p_ff_pic );
730 }
731 else
732 {
733 p_pic = (picture_t *)p_sys->p_ff_pic->opaque;
734 decoder_LinkPicture( p_dec, p_pic );
735 }
736
737 if( !p_dec->fmt_in.video.i_sar_num || !p_dec->fmt_in.video.i_sar_den )
738 {
739 /* Fetch again the aspect ratio in case it changed */
740 p_dec->fmt_out.video.i_sar_num
741 = p_context->sample_aspect_ratio.num;
742 p_dec->fmt_out.video.i_sar_den
743 = p_context->sample_aspect_ratio.den;
744
745 if( !p_dec->fmt_out.video.i_sar_num || !p_dec->fmt_out.video.i_sar_den )
746 {
747 p_dec->fmt_out.video.i_sar_num = 1;
748 p_dec->fmt_out.video.i_sar_den = 1;
749 }
750 }
751
752 /* Send decoded frame to vout */
753 if( i_pts > VLC_TS_INVALID)
754 {
755 p_pic->date = i_pts;
756
757 if( p_sys->b_first_frame )
758 {
759 /* Hack to force display of still pictures */
760 p_sys->b_first_frame = false;
761 p_pic->b_force = true;
762 }
763
764 p_pic->i_nb_fields = 2 + p_sys->p_ff_pic->repeat_pict;
765 p_pic->b_progressive = !p_sys->p_ff_pic->interlaced_frame;
766 p_pic->b_top_field_first = p_sys->p_ff_pic->top_field_first;
767
768 return p_pic;
769 }
770 else
771 {
772 decoder_DeletePicture( p_dec, p_pic );
773 }
774 }
775
776 if( p_block )
777 block_Release( p_block );
778 return NULL;
779}

808/*****************************************************************************
809 * ffmpeg_InitCodec: setup codec extra initialization data for ffmpeg
810 *****************************************************************************/
811static void ffmpeg_InitCodec( decoder_t *p_dec )
812{
813 decoder_sys_t *p_sys = p_dec->p_sys;
814 int i_size = p_dec->fmt_in.i_extra;
815
816 if( !i_size ) return;
817
818 if( p_sys->i_codec_id == AV_CODEC_ID_SVQ3 )
819 {
820 uint8_t *p;
821
822 p_sys->p_context->extradata_size = i_size + 12;
823 p = p_sys->p_context->extradata =
824 av_malloc( p_sys->p_context->extradata_size +
825 FF_INPUT_BUFFER_PADDING_SIZE );
826 if( !p )
827 return;
828
829 memcpy( &p[0], "SVQ3", 4 );
830 memset( &p[4], 0, 8 );
831 memcpy( &p[12], p_dec->fmt_in.p_extra, i_size );
832
833 /* Now remove all atoms before the SMI one */
834 if( p_sys->p_context->extradata_size > 0x5a &&
835 strncmp( (char*)&p[0x56], "SMI ", 4 ) )
836 {
837 uint8_t *psz = &p[0x52];
838
839 while( psz < &p[p_sys->p_context->extradata_size - 8] )
840 {
841 int i_size = GetDWBE( psz );
842 if( i_size <= 1 )
843 {
844 /* FIXME handle 1 as long size */
845 break;
846 }
847 if( !strncmp( (char*)&psz[4], "SMI ", 4 ) )
848 {
849 memmove( &p[0x52], psz,
850 &p[p_sys->p_context->extradata_size] - psz );
851 break;
852 }
853
854 psz += i_size;
855 }
856 }
857 }
858 else
859 {
860 p_sys->p_context->extradata_size = i_size;
861 p_sys->p_context->extradata =
862 av_malloc( i_size + FF_INPUT_BUFFER_PADDING_SIZE );
863 if( p_sys->p_context->extradata )
864 {
865 memcpy( p_sys->p_context->extradata,
866 p_dec->fmt_in.p_extra, i_size );
867 memset( p_sys->p_context->extradata + i_size,
868 0, FF_INPUT_BUFFER_PADDING_SIZE );
869 }
870 }
871}

873/*****************************************************************************
874 * ffmpeg_CopyPicture: copy a picture from ffmpeg internal buffers to a
875 * picture_t structure (when not in direct rendering mode).
876 *****************************************************************************/
877static void ffmpeg_CopyPicture( decoder_t *p_dec,
878 picture_t *p_pic, AVFrame *p_ff_pic )
879{
880 decoder_sys_t *p_sys = p_dec->p_sys;
881
882 if( p_sys->p_va )
883 {
884 vlc_va_Extract( p_sys->p_va, p_pic, p_ff_pic->opaque,
885 p_ff_pic->data[3] );
886 }
887 else if( FindVlcChroma( p_sys->p_context->pix_fmt ) )
888 {
889 int i_plane, i_size, i_line;
890 uint8_t *p_dst, *p_src;
891 int i_src_stride, i_dst_stride;
892
893 for( i_plane = 0; i_plane < p_pic->i_planes; i_plane++ )
894 {
895 p_src = p_ff_pic->data[i_plane];
896 p_dst = p_pic->p[i_plane].p_pixels;
897 i_src_stride = p_ff_pic->linesize[i_plane];
898 i_dst_stride = p_pic->p[i_plane].i_pitch;
899
900 i_size = __MIN( i_src_stride, i_dst_stride );
901 for( i_line = 0; i_line < p_pic->p[i_plane].i_visible_lines;
902 i_line++ )
903 {
904 memcpy( p_dst, p_src, i_size );
905 p_src += i_src_stride;
906 p_dst += i_dst_stride;
907 }
908 }
909 }
910 else
911 {
912 const char *name = av_get_pix_fmt_name( p_sys->p_context->pix_fmt );
913 msg_Err( p_dec, "Unsupported decoded output format %d (%s)",
914 p_sys->p_context->pix_fmt, name ? name : "unknown" );
915 p_dec->b_error = 1;
916 }
917}

969static picture_t *lavc_dr_GetFrame(struct AVCodecContext *ctx,
970 AVFrame *frame, int flags)
971{
972 decoder_t *dec = (decoder_t *)ctx->opaque;
973 decoder_sys_t *sys = dec->p_sys;
974
975 if (GetVlcChroma(&dec->fmt_out.video, ctx->pix_fmt) != VLC_SUCCESS)
976 return NULL;
977 dec->fmt_out.i_codec = dec->fmt_out.video.i_chroma;
978 if (ctx->pix_fmt == PIX_FMT_PAL8)
979 return NULL;
980
981 int width = frame->width;
982 int height = frame->height;
983 int aligns[AV_NUM_DATA_POINTERS];
984
985 avcodec_align_dimensions2(ctx, &width, &height, aligns);
986
987 picture_t *pic = ffmpeg_NewPictBuf(dec, ctx);
988 if (pic == NULL)
989 return NULL;
990
991 /* Check that the picture is suitable for libavcodec */
992 if (pic->p[0].i_pitch < width * pic->p[0].i_pixel_pitch)
993 {
994 if (sys->i_direct_rendering_used != 0)
995 msg_Dbg(dec, "plane 0: pitch too small (%d/%d*%d)",
996 pic->p[0].i_pitch, width, pic->p[0].i_pixel_pitch);
997 goto no_dr;
998 }
999
1000 if (pic->p[0].i_lines < height)
1001 {
1002 if (sys->i_direct_rendering_used != 0)
1003 msg_Dbg(dec, "plane 0: lines too few (%d/%d)",
1004 pic->p[0].i_lines, height);
1005 goto no_dr;
1006 }
1007
1008 for (int i = 0; i < pic->i_planes; i++)
1009 {
1010 if (pic->p[i].i_pitch % aligns[i])
1011 {
1012 if (sys->i_direct_rendering_used != 0)
1013 msg_Dbg(dec, "plane %d: pitch not aligned (%d%%%d)",
1014 i, pic->p[i].i_pitch, aligns[i]);
1015 goto no_dr;
1016 }
1017 if (((uintptr_t)pic->p[i].p_pixels) % aligns[i])
1018 {
1019 if (sys->i_direct_rendering_used != 0)
1020 msg_Warn(dec, "plane %d not aligned", i);
1021 goto no_dr;
1022 }
1023 }
1024
1025 /* Allocate buffer references */
1026 for (int i = 0; i < pic->i_planes; i++)
1027 {
1028 lavc_pic_ref_t *ref = malloc(sizeof (*ref));
1029 if (ref == NULL)
1030 goto error;
1031 ref->decoder = dec;
1032 ref->picture = pic;
1033 decoder_LinkPicture(dec, pic);
1034
1035 uint8_t *data = pic->p[i].p_pixels;
1036 int size = pic->p[i].i_pitch * pic->p[i].i_lines;
1037
1038 frame->buf[i] = av_buffer_create(data, size, lavc_dr_ReleaseFrame,
1039 ref, 0);
1040 if (unlikely(frame->buf[i] == NULL))
1041 {
1042 lavc_dr_ReleaseFrame(ref, data);
1043 goto error;
1044 }
1045 }
1046 decoder_UnlinkPicture(dec, pic);
1047 (void) flags;
1048 return pic;
1049error:
1050 for (unsigned i = 0; frame->buf[i] != NULL; i++)
1051 av_buffer_unref(&frame->buf[i]);
1052no_dr:
1053 decoder_DeletePicture(dec, pic);
1054 return NULL;
1055}

1057/**
1058 * Callback used by libavcodec to get a frame buffer.
1059 *
1060 * It is used for direct rendering as well as to get the right PTS for each
1061 * decoded picture (even in indirect rendering mode).
1062 */
1063static int lavc_GetFrame(struct AVCodecContext *ctx, AVFrame *frame, int flags)
1064{
1065 decoder_t *dec = ctx->opaque;
1066 decoder_sys_t *sys = dec->p_sys;
1067 picture_t *pic;
1068
1069 for (unsigned i = 0; i < AV_NUM_DATA_POINTERS; i++)
1070 {
1071 frame->data[i] = NULL;
1072 frame->linesize[i] = 0;
1073 frame->buf[i] = NULL;
1074 }
1075
1076 if (sys->p_va != NULL)
1077 return lavc_va_GetFrame(ctx, frame, flags);
1078
1079 frame->opaque = NULL;
1080 if (!sys->b_direct_rendering)
1081 return avcodec_default_get_buffer2(ctx, frame, flags);
1082
1083 /* Some codecs set pix_fmt only after the 1st frame has been decoded,
1084 * so we need to check for direct rendering again. */
1085 wait_mt(sys);
1086 pic = lavc_dr_GetFrame(ctx, frame, flags);
1087 if (pic == NULL)
1088 {
1089 if (sys->i_direct_rendering_used != 0)
1090 {
1091 msg_Warn(dec, "disabling direct rendering");
1092 sys->i_direct_rendering_used = 0;
1093 }
1094 post_mt(sys);
1095 return avcodec_default_get_buffer2(ctx, frame, flags);
1096 }
1097
1098 if (sys->i_direct_rendering_used != 1)
1099 {
1100 msg_Dbg(dec, "enabling direct rendering");
1101 sys->i_direct_rendering_used = 1;
1102 }
1103 post_mt(sys);
1104
1105 frame->opaque = pic;
1106 static_assert(PICTURE_PLANE_MAX <= AV_NUM_DATA_POINTERS, "Oops!");
1107 for (unsigned i = 0; i < PICTURE_PLANE_MAX; i++)
1108 {
1109 frame->data[i] = pic->p[i].p_pixels;
1110 frame->linesize[i] = pic->p[i].i_pitch;
1111 }
1112 return 0;
1113}

1205static int ffmpeg_GetFrameBuf( struct AVCodecContext *p_context,
1206 AVFrame *p_ff_pic )
1207{
1208 decoder_t *p_dec = (decoder_t *)p_context->opaque;
1209 decoder_sys_t *p_sys = p_dec->p_sys;
1210
1211 /* */
1212 p_ff_pic->opaque = NULL;
1213#if ! LIBAVCODEC_VERSION_CHECK(54, 34, 0, 79, 101)
1214 p_ff_pic->pkt_pts = p_context->pkt ? p_context->pkt->pts : AV_NOPTS_VALUE;
1215#endif
1216#if LIBAVCODEC_VERSION_MAJOR < 54
1217 p_ff_pic->age = 256*256*256*64;
1218#endif
1219
1220 if( p_sys->p_va )
1221 return ffmpeg_va_GetFrameBuf(p_context, p_ff_pic);
1222
1223 if( !p_sys->b_direct_rendering )
1224 return avcodec_default_get_buffer( p_context, p_ff_pic );
1225
1226 wait_mt( p_sys );
1227 /* Some codecs set pix_fmt only after the 1st frame has been decoded,
1228 * so we need to check for direct rendering again. */
1229
1230 picture_t *p_pic = ffmpeg_dr_GetFrameBuf(p_context);
1231 if (!p_pic) {
1232 if( p_sys->i_direct_rendering_used != 0 )
1233 {
1234 msg_Warn( p_dec, "disabling direct rendering" );
1235 p_sys->i_direct_rendering_used = 0;
1236 }
1237
1238 post_mt( p_sys );
1239 return avcodec_default_get_buffer( p_context, p_ff_pic );
1240 }
1241
1242 if( p_sys->i_direct_rendering_used != 1 ) {
1243 msg_Dbg( p_dec, "using direct rendering" );
1244 p_sys->i_direct_rendering_used = 1;
1245 }
1246
1247 p_context->draw_horiz_band = NULL;
1248 post_mt( p_sys );
1249
1250 p_ff_pic->opaque = (void*)p_pic;
1251 p_ff_pic->type = FF_BUFFER_TYPE_USER;
1252 p_ff_pic->data[0] = p_pic->p[0].p_pixels;
1253 p_ff_pic->data[1] = p_pic->p[1].p_pixels;
1254 p_ff_pic->data[2] = p_pic->p[2].p_pixels;
1255 p_ff_pic->data[3] = NULL; /* alpha channel but I'm not sure */
1256
1257 p_ff_pic->linesize[0] = p_pic->p[0].i_pitch;
1258 p_ff_pic->linesize[1] = p_pic->p[1].i_pitch;
1259 p_ff_pic->linesize[2] = p_pic->p[2].i_pitch;
1260 p_ff_pic->linesize[3] = 0;
1261
1262 return 0;
1263}

1287static enum PixelFormat ffmpeg_GetFormat( AVCodecContext *p_context,
1288 const enum PixelFormat *pi_fmt )
1289{
1290 decoder_t *p_dec = p_context->opaque;
1291 decoder_sys_t *p_sys = p_dec->p_sys;
1292 vlc_va_t *p_va = p_sys->p_va;
1293
1294 if( p_va != NULL )
1295 vlc_va_Delete( p_va );
1296
1297 /* Enumerate available formats */
1298 bool can_hwaccel = false;
1299 for( size_t i = 0; pi_fmt[i] != PIX_FMT_NONE; i++ )
1300 {
1301 const AVPixFmtDescriptor *dsc = av_pix_fmt_desc_get(pi_fmt[i]);
1302 if (dsc == NULL)
1303 continue;
1304 bool hwaccel = (dsc->flags & AV_PIX_FMT_FLAG_HWACCEL) != 0;
1305
1306 msg_Dbg( p_dec, "available %sware decoder output format %d (%s)",
1307 hwaccel ? "hard" : "soft", pi_fmt[i], dsc->name );
1308 if (hwaccel)
1309 can_hwaccel = true;
1310 }
1311
1312 if (!can_hwaccel)
1313 goto end;
1314
1315 /* Profile and level information is needed now.
1316 * TODO: avoid code duplication with avcodec.c */
1317 if( p_context->profile != FF_PROFILE_UNKNOWN)
1318 p_dec->fmt_in.i_profile = p_context->profile;
1319 if( p_context->level != FF_LEVEL_UNKNOWN)
1320 p_dec->fmt_in.i_level = p_context->level;
1321
1322 p_va = vlc_va_New( VLC_OBJECT(p_dec), p_context, &p_dec->fmt_in );
1323 if( p_va == NULL )
1324 goto end;
1325
1326 for( size_t i = 0; pi_fmt[i] != PIX_FMT_NONE; i++ )
1327 {
1328 if( p_va->pix_fmt != pi_fmt[i] )
1329 continue;
1330
1331 /* We try to call vlc_va_Setup when possible to detect errors when
1332 * possible (later is too late) */
1333 if( p_context->width > 0 && p_context->height > 0
1334 && vlc_va_Setup( p_va, &p_context->hwaccel_context,
1335 &p_dec->fmt_out.video.i_chroma,
1336 p_context->width, p_context->height ) )
1337 {
1338 msg_Err( p_dec, "acceleration setup failure" );
1339 break;
1340 }
1341
1342 if( p_va->description )
1343 msg_Info( p_dec, "Using %s for hardware decoding.",
1344 p_va->description );
1345
1346 /* FIXME this will disable direct rendering
1347 * even if a new pixel format is renegotiated
1348 */
1349 p_sys->b_direct_rendering = false;
1350 p_sys->p_va = p_va;
1351 p_context->draw_horiz_band = NULL;
1352 return pi_fmt[i];
1353 }
1354
1355 vlc_va_Delete( p_va );
1356
1357end:
1358 /* Fallback to default behaviour */
1359 p_sys->p_va = NULL;
1360 return avcodec_default_get_format( p_context, pi_fmt );
1361}
