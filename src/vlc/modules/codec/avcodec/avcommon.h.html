44#define AV_OPTIONS_TEXT "Advanced options"
45#define AV_OPTIONS_LONGTEXT "Advanced options, in the form {opt=val,opt2=val2}."
46
47static inline AVDictionary *vlc_av_get_options(const char *psz_opts)
48{
49 AVDictionary *options = NULL;
50 config_chain_t *cfg = NULL;
51 config_ChainParseOptions(&cfg, psz_opts);
52 while (cfg) {
53 config_chain_t *next = cfg->p_next;
54 av_dict_set(&options, cfg->psz_name, cfg->psz_value,
55 AV_DICT_DONT_STRDUP_KEY | AV_DICT_DONT_STRDUP_VAL);
56 free(cfg);
57 cfg = next;
58 }
59 return options;
60}

62static inline void vlc_init_avutil(vlc_object_t *obj)
63{
64 int level = AV_LOG_QUIET;
65
66 if (!var_InheritBool(obj, "quiet")) {
67 int64_t verbose = var_InheritInteger(obj, "verbose");
68 if (verbose >= 0) switch(verbose + VLC_MSG_ERR) {
69 case VLC_MSG_ERR:
70 level = AV_LOG_ERROR;
71 break;
72 case VLC_MSG_WARN:
73 level = AV_LOG_WARNING;
74 break;
75 case VLC_MSG_DBG:
76 level = AV_LOG_DEBUG;
77 default:
78 break;
79 }
80 }
81
82 av_log_set_level(level);
83
84 msg_Dbg(obj, "CPU flags: 0x%08x", av_get_cpu_flags());
85}

90static inline void vlc_init_avformat(vlc_object_t *obj)
91{

94 vlc_init_avutil(obj);

96 avformat_network_init();
