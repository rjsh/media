251/*****************************************************************************
252 * OpenDecoder: probe the decoder and return score
253 *****************************************************************************/
254static int OpenDecoder( vlc_object_t *p_this )
255{
256 decoder_t *p_dec = (decoder_t*) p_this;
257 unsigned i_codec_id;
258 int i_cat, i_result;
259 const char *psz_namecodec;
260
261 AVCodecContext *p_context = NULL;
262 AVCodec *p_codec = NULL;
263
264 /* *** determine codec type *** */
265 if( !GetFfmpegCodec( p_dec->fmt_in.i_codec, &i_cat, &i_codec_id,
266 &psz_namecodec ) )
267 {
268 return VLC_EGENERIC;
269 }
270
271 /* Initialization must be done before avcodec_find_decoder() */
272 vlc_init_avcodec(p_this);
273
274 /* *** ask ffmpeg for a decoder *** */
275 char *psz_decoder = var_CreateGetString( p_this, "avcodec-codec" );
276 if( psz_decoder && *psz_decoder )
277 {
278 p_codec = avcodec_find_decoder_by_name( psz_decoder );
279 if( !p_codec )
280 msg_Err( p_this, "Decoder `%s' not found", psz_decoder );
281 else if( p_codec->id != i_codec_id )
282 {
283 msg_Err( p_this, "Decoder `%s' can't handle %4.4s",
284 psz_decoder, (char*)&p_dec->fmt_in.i_codec );
285 p_codec = NULL;
286 }
287 }
288 free( psz_decoder );
289 if( !p_codec )
290 p_codec = avcodec_find_decoder( i_codec_id );
291 if( !p_codec )
292 {
293 msg_Dbg( p_dec, "codec not found (%s)", psz_namecodec );
294 return VLC_EGENERIC;
295 }
296
297 /* *** get a p_context *** */
298 p_context = avcodec_alloc_context3(p_codec);
299 if( !p_context )
300 return VLC_ENOMEM;
301 p_context->debug = var_InheritInteger( p_dec, "avcodec-debug" );
302 p_context->opaque = (void *)p_this;
303
304 p_dec->b_need_packetized = true;
305 switch( i_cat )
306 {
307 case VIDEO_ES:
308 p_dec->pf_decode_video = DecodeVideo;
309 i_result = InitVideoDec ( p_dec, p_context, p_codec,
310 i_codec_id, psz_namecodec );
311 break;
312 case AUDIO_ES:
313 p_dec->pf_decode_audio = DecodeAudio;
314 i_result = InitAudioDec ( p_dec, p_context, p_codec,
315 i_codec_id, psz_namecodec );
316 break;
317 case SPU_ES:
318 p_dec->pf_decode_sub = DecodeSubtitle;
319 i_result = InitSubtitleDec( p_dec, p_context, p_codec,
320 i_codec_id, psz_namecodec );
321 break;
322 default:
323 i_result = VLC_EGENERIC;
324 }
325
326 if( i_result == VLC_SUCCESS )
327 {
328 p_dec->p_sys->i_cat = i_cat;
329 if( p_context->profile != FF_PROFILE_UNKNOWN)
330 p_dec->fmt_in.i_profile = p_context->profile;
331 if( p_context->level != FF_LEVEL_UNKNOWN)
332 p_dec->fmt_in.i_level = p_context->level;
333 }
334
335 return i_result;
336}

338/*****************************************************************************
339 * CloseDecoder: decoder destruction
340 *****************************************************************************/
341static void CloseDecoder( vlc_object_t *p_this )
342{
343 decoder_t *p_dec = (decoder_t *)p_this;
344 decoder_sys_t *p_sys = p_dec->p_sys;
345
346 switch( p_sys->i_cat )
347 {
348 case VIDEO_ES:
349 EndVideoDec ( p_dec );
350 break;
351 }
352
353 if( p_sys->p_context )
354 {
355 av_free( p_sys->p_context->extradata );
356 p_sys->p_context->extradata = NULL;
357
358 if( !p_sys->b_delayed_open )
359 {
360 vlc_avcodec_lock();
361 avcodec_close( p_sys->p_context );
362 vlc_avcodec_unlock();
363 }
364 msg_Dbg( p_dec, "ffmpeg codec (%s) stopped", p_sys->psz_namecodec );
365 av_free( p_sys->p_context );
366 }
367
368 free( p_sys );
369}

371/*****************************************************************************
372 * ffmpeg_OpenCodec:
373 *****************************************************************************/
374int ffmpeg_OpenCodec( decoder_t *p_dec )
375{
376 decoder_sys_t *p_sys = p_dec->p_sys;
377
378 if( p_sys->p_context->extradata_size <= 0 )
379 {
380 if( p_sys->i_codec_id == AV_CODEC_ID_VC1 ||
381 p_sys->i_codec_id == AV_CODEC_ID_VORBIS ||
382 p_sys->i_codec_id == AV_CODEC_ID_THEORA ||
383 ( p_sys->i_codec_id == AV_CODEC_ID_AAC &&
384 !p_dec->fmt_in.b_packetized ) )
385 {
386 msg_Warn( p_dec, "waiting for extra data for codec %s",
387 p_sys->psz_namecodec );
388 return 1;
389 }
390 }
391 if( p_dec->fmt_in.i_cat == VIDEO_ES )
392 {
393 p_sys->p_context->width = p_dec->fmt_in.video.i_visible_width;
394 p_sys->p_context->height = p_dec->fmt_in.video.i_visible_height;
395 if (p_sys->p_context->width == 0)
396 p_sys->p_context->width = p_dec->fmt_in.video.i_width;
397 else if (p_sys->p_context->width != p_dec->fmt_in.video.i_width)
398 p_sys->p_context->coded_width = p_dec->fmt_in.video.i_width;
399 if (p_sys->p_context->height == 0)
400 p_sys->p_context->height = p_dec->fmt_in.video.i_height;
401 else if (p_sys->p_context->height != p_dec->fmt_in.video.i_height)
402 p_sys->p_context->coded_height = p_dec->fmt_in.video.i_height;
403 p_sys->p_context->bits_per_coded_sample = p_dec->fmt_in.video.i_bits_per_pixel;
404 }
405 else if( p_dec->fmt_in.i_cat == AUDIO_ES )
406 {
407 p_sys->p_context->sample_rate = p_dec->fmt_in.audio.i_rate;
408 p_sys->p_context->channels = p_dec->fmt_in.audio.i_channels;
409
410 p_sys->p_context->block_align = p_dec->fmt_in.audio.i_blockalign;
411 p_sys->p_context->bit_rate = p_dec->fmt_in.i_bitrate;
412 p_sys->p_context->bits_per_coded_sample = p_dec->fmt_in.audio.i_bitspersample;
413 if( p_sys->i_codec_id == AV_CODEC_ID_ADPCM_G726 &&
414 p_sys->p_context->bit_rate > 0 &&
415 p_sys->p_context->sample_rate > 0)
416 p_sys->p_context->bits_per_coded_sample = p_sys->p_context->bit_rate /
417 p_sys->p_context->sample_rate;
418 }
419 int ret;
420 char *psz_opts = var_InheritString( p_dec, "avcodec-options" );
421 AVDictionary *options = NULL;
422 if (psz_opts && *psz_opts)
423 options = vlc_av_get_options(psz_opts);
424 free(psz_opts);
425
426 vlc_avcodec_lock();
427 ret = avcodec_open2( p_sys->p_context, p_sys->p_codec, options ? &options : NULL );
428 vlc_avcodec_unlock();
429
430 AVDictionaryEntry *t = NULL;
431 while ((t = av_dict_get(options, "", t, AV_DICT_IGNORE_SUFFIX))) {
432 msg_Err( p_dec, "Unknown option \"%s\"", t->key );
433 }
434 av_dict_free(&options);
435
436 if( ret < 0 )
437 return VLC_EGENERIC;
438 msg_Dbg( p_dec, "avcodec codec (%s) started", p_sys->psz_namecodec );
439
440#ifdef HAVE_AVCODEC_MT
441 if( p_dec->fmt_in.i_cat == VIDEO_ES )
442 {
443 switch( p_sys->p_context->active_thread_type )
444 {
445 case FF_THREAD_FRAME:
446 msg_Dbg( p_dec, "using frame thread mode with %d threads",
447 p_sys->p_context->thread_count );
448 break;
449 case FF_THREAD_SLICE:
450 msg_Dbg( p_dec, "using slice thread mode with %d threads",
451 p_sys->p_context->thread_count );
452 break;
453 case 0:
454 if( p_sys->p_context->thread_count > 1 )
455 msg_Warn( p_dec, "failed to enable threaded decoding" );
456 break;
457 default:
458 msg_Warn( p_dec, "using unknown thread mode with %d threads",
459 p_sys->p_context->thread_count );
460 break;
461 }
462 }
463#endif
464
465 p_sys->b_delayed_open = false;
466
467 return VLC_SUCCESS;
468}
