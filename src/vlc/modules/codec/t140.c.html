44static int Open( vlc_object_t *p_this )
45{
46 encoder_t *p_enc = (encoder_t *)p_this;
47
48 switch( p_enc->fmt_out.i_codec )
49 {
50 case VLC_CODEC_SUBT:
51 if( ( p_enc->fmt_out.subs.psz_encoding != NULL )
52 && strcasecmp( p_enc->fmt_out.subs.psz_encoding, "utf8" )
53 && strcasecmp( p_enc->fmt_out.subs.psz_encoding, "UTF-8" ) )
54 {
55 msg_Err( p_this, "Only UTF-8 encoding supported" );
56 return VLC_EGENERIC;
57 }
58 case VLC_CODEC_ITU_T140:
59 break;
60
61 default:
62 if( !p_enc->b_force )
63 return VLC_EGENERIC;
64
65 p_enc->fmt_out.i_codec = VLC_CODEC_ITU_T140;
66 }
67
68 p_enc->p_sys = NULL;
69
70 p_enc->pf_encode_sub = Encode;
71 p_enc->fmt_out.i_cat = SPU_ES;
72 return VLC_SUCCESS;
73}

82static block_t *Encode( encoder_t *p_enc, subpicture_t *p_spu )
83{
84 VLC_UNUSED( p_enc );
85
86 subpicture_region_t *p_region;
87 block_t *p_block;
88 size_t len;
89
90 if( p_spu == NULL )
91 return NULL;
92
93 p_region = p_spu->p_region;
94 if( ( p_region == NULL )
95 || ( p_region->fmt.i_chroma != VLC_CODEC_TEXT )
96 || ( p_region->psz_text == NULL ) )
97 return NULL;
98
99 /* This should already be UTF-8 encoded, so not much effort... */
100 len = strlen( p_region->psz_text );
101 p_block = block_Alloc( len );
102 memcpy( p_block->p_buffer, p_region->psz_text, len );
103
104 p_block->i_pts = p_block->i_dts = p_spu->i_start;
105 if( !p_spu->b_ephemer && ( p_spu->i_stop > p_spu->i_start ) )
106 p_block->i_length = p_spu->i_stop - p_spu->i_start;
107
108 return p_block;
109}
