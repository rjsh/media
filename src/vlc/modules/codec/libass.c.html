123/*****************************************************************************
124 * Create: Open libass decoder.
125 *****************************************************************************/
126static int Create( vlc_object_t *p_this )
127{
128 decoder_t *p_dec = (decoder_t *)p_this;
129 decoder_sys_t *p_sys;
130
131 if( p_dec->fmt_in.i_codec != VLC_CODEC_SSA )
132 return VLC_EGENERIC;
133
134 p_dec->pf_decode_sub = DecodeBlock;
135
136 p_dec->p_sys = p_sys = malloc( sizeof( decoder_sys_t ) );
137 if( !p_sys )
138 return VLC_ENOMEM;
139
140 /* */
141 vlc_mutex_init( &p_sys->lock );
142 p_sys->i_refcount = 1;
143 memset( &p_sys->fmt, 0, sizeof(p_sys->fmt) );
144 p_sys->i_max_stop = VLC_TS_INVALID;
145 p_sys->p_library = NULL;
146 p_sys->p_renderer = NULL;
147 p_sys->p_track = NULL;
148
149 /* Create libass library */
150 ASS_Library *p_library = p_sys->p_library = ass_library_init();
151 if( !p_library )
152 {
153 msg_Warn( p_dec, "Libass library creation failed" );
154 DecSysRelease( p_sys );
155 return VLC_EGENERIC;
156 }
157
158 /* load attachments */
159 input_attachment_t **pp_attachments;
160 int i_attachments;
161 if( decoder_GetInputAttachments( p_dec, &pp_attachments, &i_attachments ))
162 {
163 i_attachments = 0;
164 pp_attachments = NULL;
165 }
166 for( int k = 0; k < i_attachments; k++ )
167 {
168 input_attachment_t *p_attach = pp_attachments[k];
169
170 bool found = false;
171
172 /* Check mimetype*/
173 if( !strcasecmp( p_attach->psz_mime, "application/x-truetype-font" ) )
174 found = true;
175 /* Then extension */
176 else if( !found && strlen( p_attach->psz_name ) > 4 )
177 {
178 char *ext = p_attach->psz_name + strlen( p_attach->psz_name ) - 4;
179
180 if( !strcasecmp( ext, ".ttf" ) || !strcasecmp( ext, ".otf" ) || !strcasecmp( ext, ".ttc" ) )
181 found = true;
182 }
183
184 if( found )
185 {
186 msg_Dbg( p_dec, "adding embedded font %s", p_attach->psz_name );
187
188 ass_add_font( p_sys->p_library, p_attach->psz_name, p_attach->p_data, p_attach->i_data );
189 }
190 vlc_input_attachment_Delete( p_attach );
191 }
192 free( pp_attachments );
193
194 ass_set_extract_fonts( p_library, true );
195 ass_set_style_overrides( p_library, NULL );
196
197 /* Create the renderer */
198 ASS_Renderer *p_renderer = p_sys->p_renderer = ass_renderer_init( p_library );
199 if( !p_renderer )
200 {
201 msg_Warn( p_dec, "Libass renderer creation failed" );
202 DecSysRelease( p_sys );
203 return VLC_EGENERIC;
204 }
205
206 ass_set_use_margins( p_renderer, false);
207 //if( false )
208 // ass_set_margins( p_renderer, int t, int b, int l, int r);
209 ass_set_hinting( p_renderer, ASS_HINTING_LIGHT );
210 ass_set_font_scale( p_renderer, 1.0 );
211 ass_set_line_spacing( p_renderer, 0.0 );
212
213#if defined( __ANDROID__ )
214 const char *psz_font = "/system/fonts/DroidSans-Bold.ttf";
215 const char *psz_family = "Droid Sans Bold";
216#else
217 const char *psz_font = NULL; /* We don't ship a default font with VLC */
218 const char *psz_family = "Arial"; /* Use Arial if we can't find anything more suitable */
219#endif
220
221#ifdef HAVE_FONTCONFIG
222#if defined(_WIN32)
223 dialog_progress_bar_t *p_dialog =
224 dialog_ProgressCreate( p_dec,
225 _("Building font cache"),
226 _( "Please wait while your font cache is rebuilt.\n"
227 "This should take less than a minute." ), NULL );
228#endif
229 ass_set_fonts( p_renderer, psz_font, psz_family, true, NULL, 1 ); // setup default font/family
230#ifdef _WIN32
231 if( p_dialog )
232 {
233 dialog_ProgressSet( p_dialog, NULL, 1.0 );
234 dialog_ProgressDestroy( p_dialog );
235 }
236#endif
237#else
238 /* FIXME you HAVE to give him a font if no fontconfig */
239 ass_set_fonts( p_renderer, psz_font, psz_family, false, NULL, 1 );
240#endif
241
242 /* Add a track */
243 ASS_Track *p_track = p_sys->p_track = ass_new_track( p_sys->p_library );
244 if( !p_track )
245 {
246 DecSysRelease( p_sys );
247 return VLC_EGENERIC;
248 }
249 ass_process_codec_private( p_track, p_dec->fmt_in.p_extra, p_dec->fmt_in.i_extra );
250
251 p_dec->fmt_out.i_cat = SPU_ES;
252 p_dec->fmt_out.i_codec = VLC_CODEC_RGBA;
253
254 return VLC_SUCCESS;
255}

267static void DecSysHold( decoder_sys_t *p_sys )
268{
269 vlc_mutex_lock( &p_sys->lock );
270 p_sys->i_refcount++;
271 vlc_mutex_unlock( &p_sys->lock );
272}
273static void DecSysRelease( decoder_sys_t *p_sys )
274{
275 /* */
276 vlc_mutex_lock( &p_sys->lock );
277 p_sys->i_refcount--;
278 if( p_sys->i_refcount > 0 )
279 {
280 vlc_mutex_unlock( &p_sys->lock );
281 return;
282 }
283 vlc_mutex_unlock( &p_sys->lock );
284 vlc_mutex_destroy( &p_sys->lock );
285
286 if( p_sys->p_track )
287 ass_free_track( p_sys->p_track );
288 if( p_sys->p_renderer )
289 ass_renderer_done( p_sys->p_renderer );
290 if( p_sys->p_library )
291 ass_library_done( p_sys->p_library );
292
293 free( p_sys );
294}

296/****************************************************************************
297 * DecodeBlock:
298 ****************************************************************************/
299static subpicture_t *DecodeBlock( decoder_t *p_dec, block_t **pp_block )
300{
301 decoder_sys_t *p_sys = p_dec->p_sys;
302
303 subpicture_t *p_spu = NULL;
304 block_t *p_block;
305
306 if( !pp_block || *pp_block == NULL )
307 return NULL;
308
309 p_block = *pp_block;
310 if( p_block->i_flags & (BLOCK_FLAG_DISCONTINUITY|BLOCK_FLAG_CORRUPTED) )
311 {
312 p_sys->i_max_stop = VLC_TS_INVALID;
313 block_Release( p_block );
314 return NULL;
315 }
316 *pp_block = NULL;
317
318 if( p_block->i_buffer == 0 || p_block->p_buffer[0] == '\0' )
319 {
320 block_Release( p_block );
321 return NULL;
322 }
323
324 subpicture_updater_sys_t *p_spu_sys = malloc( sizeof(*p_spu_sys) );
325 if( !p_spu_sys )
326 {
327 block_Release( p_block );
328 return NULL;
329 }
330
331 subpicture_updater_t updater = {
332 .pf_validate = SubpictureValidate,
333 .pf_update = SubpictureUpdate,
334 .pf_destroy = SubpictureDestroy,
335 .p_sys = p_spu_sys,
336 };
337 p_spu = decoder_NewSubpicture( p_dec, &updater );
338 if( !p_spu )
339 {
340 msg_Warn( p_dec, "can't get spu buffer" );
341 free( p_spu_sys );
342 block_Release( p_block );
343 return NULL;
344 }
345
346 p_spu_sys->p_img = NULL;
347 p_spu_sys->p_dec_sys = p_sys;
348 p_spu_sys->i_subs_len = p_block->i_buffer;
349 p_spu_sys->p_subs_data = malloc( p_block->i_buffer );
350 p_spu_sys->i_pts = p_block->i_pts;
351 if( !p_spu_sys->p_subs_data )
352 {
353 decoder_DeleteSubpicture( p_dec, p_spu );
354 block_Release( p_block );
355 return NULL;
356 }
357 memcpy( p_spu_sys->p_subs_data, p_block->p_buffer,
358 p_block->i_buffer );
359
360 p_spu->i_start = p_block->i_pts;
361 p_spu->i_stop = __MAX( p_sys->i_max_stop, p_block->i_pts + p_block->i_length );
362 p_spu->b_ephemer = true;
363 p_spu->b_absolute = true;
364
365 p_sys->i_max_stop = p_spu->i_stop;
366
367 vlc_mutex_lock( &p_sys->lock );
368 if( p_sys->p_track )
369 {
370 ass_process_chunk( p_sys->p_track, p_spu_sys->p_subs_data, p_spu_sys->i_subs_len,
371 p_block->i_pts / 1000, p_block->i_length / 1000 );
372 }
373 vlc_mutex_unlock( &p_sys->lock );
374
375 DecSysHold( p_sys ); /* Keep a reference for the returned subpicture */
376
377 block_Release( p_block );
378
379 return p_spu;
380}

382/****************************************************************************
383 *
384 ****************************************************************************/
385static int SubpictureValidate( subpicture_t *p_subpic,
386 bool b_fmt_src, const video_format_t *p_fmt_src,
387 bool b_fmt_dst, const video_format_t *p_fmt_dst,
388 mtime_t i_ts )
389{
390 decoder_sys_t *p_sys = p_subpic->updater.p_sys->p_dec_sys;
391
392 vlc_mutex_lock( &p_sys->lock );
393
394 video_format_t fmt = *p_fmt_dst;
395 fmt.i_chroma = VLC_CODEC_RGBA;
396 fmt.i_bits_per_pixel = 0;
397 fmt.i_visible_width = fmt.i_width;
398 fmt.i_visible_height = fmt.i_height;
399 fmt.i_x_offset = 0;
400 fmt.i_y_offset = 0;
401 if( b_fmt_src || b_fmt_dst )
402 {
403 ass_set_frame_size( p_sys->p_renderer, fmt.i_width, fmt.i_height );
404 const double src_ratio = (double)p_fmt_src->i_width / p_fmt_src->i_height;
405 const double dst_ratio = (double)p_fmt_dst->i_width / p_fmt_dst->i_height;
406 ass_set_aspect_ratio( p_sys->p_renderer, dst_ratio / src_ratio, 1 );
407 p_sys->fmt = fmt;
408 }
409
410 /* */
411 const mtime_t i_stream_date = p_subpic->updater.p_sys->i_pts + (i_ts - p_subpic->i_start);
412 int i_changed;
413 ASS_Image *p_img = ass_render_frame( p_sys->p_renderer, p_sys->p_track,
414 i_stream_date/1000, &i_changed );
415
416 if( !i_changed && !b_fmt_src && !b_fmt_dst &&
417 (p_img != NULL) == (p_subpic->p_region != NULL) )
418 {
419 vlc_mutex_unlock( &p_sys->lock );
420 return VLC_SUCCESS;
421 }
422 p_subpic->updater.p_sys->p_img = p_img;
423
424 /* The lock is released by SubpictureUpdate */
425 return VLC_EGENERIC;
426}

428static void SubpictureUpdate( subpicture_t *p_subpic,
429 const video_format_t *p_fmt_src,
430 const video_format_t *p_fmt_dst,
431 mtime_t i_ts )
432{
433 VLC_UNUSED( p_fmt_src ); VLC_UNUSED( p_fmt_dst ); VLC_UNUSED( i_ts );
434
435 decoder_sys_t *p_sys = p_subpic->updater.p_sys->p_dec_sys;
436
437 video_format_t fmt = p_sys->fmt;
438 ASS_Image *p_img = p_subpic->updater.p_sys->p_img;
439
440 /* */
441 p_subpic->i_original_picture_height = fmt.i_height;
442 p_subpic->i_original_picture_width = fmt.i_width;
443
444 /* XXX to improve efficiency we merge regions that are close minimizing
445 * the lost surface.
446 * libass tends to create a lot of small regions and thus spu engine
447 * reinstanciate a lot the scaler, and as we do not support subpel blending
448 * it looks ugly (text unaligned).
449 */
450 const int i_max_region = 4;
451 rectangle_t region[i_max_region];
452 const int i_region = BuildRegions( region, i_max_region, p_img, fmt.i_width, fmt.i_height );
453
454 if( i_region <= 0 )
455 {
456 vlc_mutex_unlock( &p_sys->lock );
457 return;
458 }
459
460 /* Allocate the regions and draw them */
461 subpicture_region_t **pp_region_last = &p_subpic->p_region;
462
463 for( int i = 0; i < i_region; i++ )
464 {
465 subpicture_region_t *r;
466 video_format_t fmt_region;
467
468 /* */
469 fmt_region = fmt;
470 fmt_region.i_width =
471 fmt_region.i_visible_width = region[i].x1 - region[i].x0;
472 fmt_region.i_height =
473 fmt_region.i_visible_height = region[i].y1 - region[i].y0;
474
475 r = subpicture_region_New( &fmt_region );
476 if( !r )
477 break;
478 r->i_x = region[i].x0;
479 r->i_y = region[i].y0;
480 r->i_align = SUBPICTURE_ALIGN_TOP | SUBPICTURE_ALIGN_LEFT;
481
482 /* */
483 RegionDraw( r, p_img );
484
485 /* */
486 *pp_region_last = r;
487 pp_region_last = &r->p_next;
488 }
489 vlc_mutex_unlock( &p_sys->lock );
490
491}
492static void SubpictureDestroy( subpicture_t *p_subpic )
493{
494 subpicture_updater_sys_t *p_sys = p_subpic->updater.p_sys;
495
496 DecSysRelease( p_sys->p_dec_sys );
497 free( p_sys->p_subs_data );
498 free( p_sys );
499}

501static rectangle_t r_create( int x0, int y0, int x1, int y1 )
502{
503 rectangle_t r = { x0, y0, x1, y1 };
504 return r;
505}
506static rectangle_t r_img( const ASS_Image *p_img )
507{
508 return r_create( p_img->dst_x, p_img->dst_y, p_img->dst_x+p_img->w, p_img->dst_y+p_img->h );
509}
510static void r_add( rectangle_t *r, const rectangle_t *n )
511{
512 r->x0 = __MIN( r->x0, n->x0 );
513 r->y0 = __MIN( r->y0, n->y0 );
514 r->x1 = __MAX( r->x1, n->x1 );
515 r->y1 = __MAX( r->y1, n->y1 );
516}
517static int r_surface( const rectangle_t *r )
518{
519 return (r->x1-r->x0) * (r->y1-r->y0);
520}
521static bool r_overlap( const rectangle_t *a, const rectangle_t *b, int i_dx, int i_dy )
522{
523 return __MAX(a->x0-i_dx, b->x0) < __MIN( a->x1+i_dx, b->x1 ) &&
524 __MAX(a->y0-i_dy, b->y0) < __MIN( a->y1+i_dy, b->y1 );
525}

527static int BuildRegions( rectangle_t *p_region, int i_max_region, ASS_Image *p_img_list, int i_width, int i_height )
528{
529 ASS_Image *p_tmp;
530 int i_count;
531
532#ifdef DEBUG_REGION
533 int64_t i_ck_start = mdate();
534#endif
535
536 for( p_tmp = p_img_list, i_count = 0; p_tmp != NULL; p_tmp = p_tmp->next )
537 if( p_tmp->w > 0 && p_tmp->h > 0 )
538 i_count++;
539 if( i_count <= 0 )
540 return 0;
541
542 ASS_Image **pp_img = calloc( i_count, sizeof(*pp_img) );
543 if( !pp_img )
544 return 0;
545
546 for( p_tmp = p_img_list, i_count = 0; p_tmp != NULL; p_tmp = p_tmp->next )
547 if( p_tmp->w > 0 && p_tmp->h > 0 )
548 pp_img[i_count++] = p_tmp;
549
550 /* */
551 const int i_w_inc = __MAX( ( i_width + 49 ) / 50, 32 );
552 const int i_h_inc = __MAX( ( i_height + 99 ) / 100, 32 );
553 int i_maxh = i_w_inc;
554 int i_maxw = i_h_inc;
555 int i_region;
556 rectangle_t region[i_max_region+1];
557
558 i_region = 0;
559 for( int i_used = 0; i_used < i_count; )
560 {
561 int n;
562 for( n = 0; n < i_count; n++ )
563 {
564 if( pp_img[n] )
565 break;
566 }
567 assert( i_region < i_max_region + 1 );
568 region[i_region++] = r_img( pp_img[n] );
569 pp_img[n] = NULL; i_used++;
570
571 bool b_ok;
572 do {
573 b_ok = false;
574 for( n = 0; n < i_count; n++ )
575 {
576 ASS_Image *p_img = pp_img[n];
577 if( !p_img )
578 continue;
579 rectangle_t r = r_img( p_img );
580
581 int k;
582 int i_best = -1;
583 int i_best_s = INT_MAX;
584 for( k = 0; k < i_region; k++ )
585 {
586 if( !r_overlap( &region[k], &r, i_maxw, i_maxh ) )
587 continue;
588 int s = r_surface( &r );
589 if( s < i_best_s )
590 {
591 i_best_s = s;
592 i_best = k;
593 }
594 }
595 if( i_best >= 0 )
596 {
597 r_add( &region[i_best], &r );
598 pp_img[n] = NULL; i_used++;
599 b_ok = true;
600 }
601 }
602 } while( b_ok );
603
604 if( i_region > i_max_region )
605 {
606 int i_best_i = -1;
607 int i_best_j = -1;
608 int i_best_ds = INT_MAX;
609
610 /* merge best */
611 for( int i = 0; i < i_region; i++ )
612 {
613 for( int j = i+1; j < i_region; j++ )
614 {
615 rectangle_t n = region[i];
616 r_add( &n, &region[j] );
617 int ds = r_surface( &n ) - r_surface( &region[i] ) - r_surface( &region[j] );
618
619 if( ds < i_best_ds )
620 {
621 i_best_i = i;
622 i_best_j = j;
623 i_best_ds = ds;
624 }
625 }
626 }
627#ifdef DEBUG_REGION
628 msg_Err( p_spu, "Merging %d and %d", i_best_i, i_best_j );
629#endif
630 r_add( &region[i_best_i], &region[i_best_j] );
631
632 if( i_best_j+1 < i_region )
633 memmove( &region[i_best_j], &region[i_best_j+1], sizeof(*region) * ( i_region - (i_best_j+1) ) );
634 i_region--;
635 }
636 }
637
638 /* */
639 for( int n = 0; n < i_region; n++ )
640 p_region[n] = region[n];
641
642#ifdef DEBUG_REGION
643 int64_t i_ck_time = mdate() - i_ck_start;
644 msg_Err( p_spu, "ASS: %d objects merged into %d region in %d micros", i_count, i_region, (int)(i_ck_time) );
645#endif
646
647 free( pp_img );
648
649 return i_region;
650}

652static void RegionDraw( subpicture_region_t *p_region, ASS_Image *p_img )
653{
654 const plane_t *p = &p_region->p_picture->p[0];
655 const int i_x = p_region->i_x;
656 const int i_y = p_region->i_y;
657 const int i_width = p_region->fmt.i_width;
658 const int i_height = p_region->fmt.i_height;
659
660 memset( p->p_pixels, 0x00, p->i_pitch * p->i_lines );
661 for( ; p_img != NULL; p_img = p_img->next )
662 {
663 if( p_img->dst_x < i_x || p_img->dst_x + p_img->w > i_x + i_width ||
664 p_img->dst_y < i_y || p_img->dst_y + p_img->h > i_y + i_height )
665 continue;
666
667 const unsigned r = (p_img->color >> 24)&0xff;
668 const unsigned g = (p_img->color >> 16)&0xff;
669 const unsigned b = (p_img->color >> 8)&0xff;
670 const unsigned a = (p_img->color )&0xff;
671 int x, y;
672
673 for( y = 0; y < p_img->h; y++ )
674 {
675 for( x = 0; x < p_img->w; x++ )
676 {
677 const unsigned alpha = p_img->bitmap[y*p_img->stride+x];
678 const unsigned an = (255 - a) * alpha / 255;
679
680 uint8_t *p_rgba = &p->p_pixels[(y+p_img->dst_y-i_y) * p->i_pitch + 4 * (x+p_img->dst_x-i_x)];
681 const unsigned ao = p_rgba[3];
682
683 /* Native endianness, but RGBA ordering */
684 if( ao == 0 )
685 {
686 /* Optimized but the else{} will produce the same result */
687 p_rgba[0] = r;
688 p_rgba[1] = g;
689 p_rgba[2] = b;
690 p_rgba[3] = an;
691 }
692 else
693 {
694 p_rgba[3] = 255 - ( 255 - p_rgba[3] ) * ( 255 - an ) / 255;
695 if( p_rgba[3] != 0 )
696 {
697 p_rgba[0] = ( p_rgba[0] * ao * (255-an) / 255 + r * an ) / p_rgba[3];
698 p_rgba[1] = ( p_rgba[1] * ao * (255-an) / 255 + g * an ) / p_rgba[3];
699 p_rgba[2] = ( p_rgba[2] * ao * (255-an) / 255 + b * an ) / p_rgba[3];
700 }
701 }
702 }
703 }
704 }
705
706#ifdef DEBUG_REGION
707 /* XXX Draw a box for debug */
708#define P(x,y) ((uint32_t*)&p->p_pixels[(y)*p->i_pitch + 4*(x)])
709 for( int y = 0; y < p->i_lines; y++ )
710 *P(0,y) = *P(p->i_visible_pitch/4-1,y) = 0xff000000;
711 for( int x = 0; x < p->i_visible_pitch; x++ )
712 *P(x/4,0) = *P(x/4,p->i_visible_lines-1) = 0xff000000;
713#undef P
714#endif
715}
