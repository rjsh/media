62/*****************************************************************************
63 * decoder_sys_t : vorbis decoder descriptor
64 *****************************************************************************/
65struct decoder_sys_t
66{
67 /* Module mode */
68 bool b_packetizer;
69
70 bool b_has_headers;
71
72 /*
73 * Vorbis properties
74 */
75 vorbis_info vi; /* struct that stores all the static vorbis bitstream
76 settings */
77 vorbis_comment vc; /* struct that stores all the bitstream user
78 * comments */
79 vorbis_dsp_state vd; /* central working state for the packet->PCM
80 * decoder */
81 vorbis_block vb; /* local working space for packet->PCM decode */
82
83 /*
84 * Common properties
85 */
86 date_t end_date;
87 int i_last_block_size;
88
89 /*
90 ** Channel reordering
91 */
92 uint8_t pi_chan_table[AOUT_CHAN_MAX];
93};

228/*****************************************************************************
229 * OpenDecoder: probe the decoder and return score
230 *****************************************************************************/
231static int OpenDecoder( vlc_object_t *p_this )
232{
233 decoder_t *p_dec = (decoder_t*)p_this;
234 decoder_sys_t *p_sys;
235
236 if( p_dec->fmt_in.i_codec != VLC_CODEC_VORBIS )
237 return VLC_EGENERIC;
238
239 /* Allocate the memory needed to store the decoder's structure */
240 p_dec->p_sys = p_sys = malloc( sizeof(*p_sys) );
241 if( unlikely( !p_sys ) )
242 return VLC_ENOMEM;
243
244 /* Misc init */
245 date_Set( &p_sys->end_date, 0 );
246 p_sys->i_last_block_size = 0;
247 p_sys->b_packetizer = false;
248 p_sys->b_has_headers = false;
249
250 /* Take care of vorbis init */
251 vorbis_info_init( &p_sys->vi );
252 vorbis_comment_init( &p_sys->vc );
253
254 /* Set output properties */
255 p_dec->fmt_out.i_cat = AUDIO_ES;
256#ifdef MODULE_NAME_IS_tremor
257 p_dec->fmt_out.i_codec = VLC_CODEC_S32N;
258#else
259 p_dec->fmt_out.i_codec = VLC_CODEC_FL32;
260#endif
261
262 /* Set callbacks */
263 p_dec->pf_decode_audio = DecodeBlock;
264 p_dec->pf_packetize = DecodeBlock;
265
266 return VLC_SUCCESS;
267}

269static int OpenPacketizer( vlc_object_t *p_this )
270{
271 decoder_t *p_dec = (decoder_t*)p_this;
272
273 int i_ret = OpenDecoder( p_this );
274
275 if( i_ret == VLC_SUCCESS )
276 {
277 p_dec->p_sys->b_packetizer = true;
278 p_dec->fmt_out.i_codec = VLC_CODEC_VORBIS;
279 }
280
281 return i_ret;
282}

284/****************************************************************************
285 * DecodeBlock: the whole thing
286 ****************************************************************************
287 * This function must be fed with ogg packets.
288 ****************************************************************************/
289static block_t *DecodeBlock( decoder_t *p_dec, block_t **pp_block )
290{
291 decoder_sys_t *p_sys = p_dec->p_sys;
292 ogg_packet oggpacket;
293
294 if( !pp_block ) return NULL;
295
296 if( *pp_block )
297 {
298 /* Block to Ogg packet */
299 oggpacket.packet = (*pp_block)->p_buffer;
300 oggpacket.bytes = (*pp_block)->i_buffer;
301 }
302 else
303 {
304 if( p_sys->b_packetizer ) return NULL;
305
306 /* Block to Ogg packet */
307 oggpacket.packet = NULL;
308 oggpacket.bytes = 0;
309 }
310
311 oggpacket.granulepos = -1;
312 oggpacket.b_o_s = 0;
313 oggpacket.e_o_s = 0;
314 oggpacket.packetno = 0;
315
316 /* Check for headers */
317 if( !p_sys->b_has_headers )
318 {
319 if( ProcessHeaders( p_dec ) )
320 {
321 block_Release( *pp_block );
322 return NULL;
323 }
324 p_sys->b_has_headers = true;
325 }
326
327 return ProcessPacket( p_dec, &oggpacket, pp_block );
328}

330/*****************************************************************************
331 * ProcessHeaders: process Vorbis headers.
332 *****************************************************************************/
333static int ProcessHeaders( decoder_t *p_dec )
334{
335 decoder_sys_t *p_sys = p_dec->p_sys;
336 ogg_packet oggpacket;
337
338 unsigned pi_size[XIPH_MAX_HEADER_COUNT];
339 void *pp_data[XIPH_MAX_HEADER_COUNT];
340 unsigned i_count;
341 if( xiph_SplitHeaders( pi_size, pp_data, &i_count,
342 p_dec->fmt_in.i_extra, p_dec->fmt_in.p_extra) )
343 return VLC_EGENERIC;
344 if( i_count < 3 )
345 return VLC_EGENERIC;
346
347 oggpacket.granulepos = -1;
348 oggpacket.e_o_s = 0;
349 oggpacket.packetno = 0;
350
351 /* Take care of the initial Vorbis header */
352 oggpacket.b_o_s = 1; /* yes this actually is a b_o_s packet :) */
353 oggpacket.bytes = pi_size[0];
354 oggpacket.packet = pp_data[0];
355 if( vorbis_synthesis_headerin( &p_sys->vi, &p_sys->vc, &oggpacket ) < 0 )
356 {
357 msg_Err( p_dec, "this bitstream does not contain Vorbis audio data");
358 return VLC_EGENERIC;
359 }
360
361 /* Setup the format */
362 p_dec->fmt_out.audio.i_rate = p_sys->vi.rate;
363 p_dec->fmt_out.audio.i_channels = p_sys->vi.channels;
364
365 if( p_dec->fmt_out.audio.i_channels >= ARRAY_SIZE(pi_channels_maps) )
366 {
367 msg_Err( p_dec, "invalid number of channels (1-%zu): %i",
368 ARRAY_SIZE(pi_channels_maps),
369 p_dec->fmt_out.audio.i_channels );
370 return VLC_EGENERIC;
371 }
372
373 p_dec->fmt_out.audio.i_physical_channels =
374 p_dec->fmt_out.audio.i_original_channels =
375 pi_channels_maps[p_sys->vi.channels];
376 p_dec->fmt_out.i_bitrate = __MAX( 0, (int32_t) p_sys->vi.bitrate_nominal );
377
378 date_Init( &p_sys->end_date, p_sys->vi.rate, 1 );
379
380 msg_Dbg( p_dec, "channels:%d samplerate:%ld bitrate:%ud",
381 p_sys->vi.channels, p_sys->vi.rate, p_dec->fmt_out.i_bitrate );
382
383 /* The next packet in order is the comments header */
384 oggpacket.b_o_s = 0;
385 oggpacket.bytes = pi_size[1];
386 oggpacket.packet = pp_data[1];
387 if( vorbis_synthesis_headerin( &p_sys->vi, &p_sys->vc, &oggpacket ) < 0 )
388 {
389 msg_Err( p_dec, "2nd Vorbis header is corrupted" );
390 return VLC_EGENERIC;
391 }
392 ParseVorbisComments( p_dec );
393
394 /* The next packet in order is the codebooks header
395 * We need to watch out that this packet is not missing as a
396 * missing or corrupted header is fatal. */
397 oggpacket.b_o_s = 0;
398 oggpacket.bytes = pi_size[2];
399 oggpacket.packet = pp_data[2];
400 if( vorbis_synthesis_headerin( &p_sys->vi, &p_sys->vc, &oggpacket ) < 0 )
401 {
402 msg_Err( p_dec, "3rd Vorbis header is corrupted" );
403 return VLC_EGENERIC;
404 }
405
406 if( !p_sys->b_packetizer )
407 {
408 /* Initialize the Vorbis packet->PCM decoder */
409 vorbis_synthesis_init( &p_sys->vd, &p_sys->vi );
410 vorbis_block_init( &p_sys->vd, &p_sys->vb );
411 }
412 else
413 {
414 p_dec->fmt_out.i_extra = p_dec->fmt_in.i_extra;
415 p_dec->fmt_out.p_extra = xrealloc( p_dec->fmt_out.p_extra,
416 p_dec->fmt_out.i_extra );
417 memcpy( p_dec->fmt_out.p_extra,
418 p_dec->fmt_in.p_extra, p_dec->fmt_out.i_extra );
419 }
420
421 ConfigureChannelOrder(p_sys->pi_chan_table, p_sys->vi.channels,
422 p_dec->fmt_out.audio.i_physical_channels, true);
423
424 return VLC_SUCCESS;
425}

427/*****************************************************************************
428 * ProcessPacket: processes a Vorbis packet.
429 *****************************************************************************/
430static void *ProcessPacket( decoder_t *p_dec, ogg_packet *p_oggpacket,
431 block_t **pp_block )
432{
433 decoder_sys_t *p_sys = p_dec->p_sys;
434 block_t *p_block = *pp_block;
435
436 /* Date management */
437 if( p_block && p_block->i_pts > VLC_TS_INVALID &&
438 p_block->i_pts != date_Get( &p_sys->end_date ) )
439 {
440 date_Set( &p_sys->end_date, p_block->i_pts );
441 }
442
443 if( !date_Get( &p_sys->end_date ) )
444 {
445 /* We've just started the stream, wait for the first PTS. */
446 if( p_block ) block_Release( p_block );
447 return NULL;
448 }
449
450 *pp_block = NULL; /* To avoid being fed the same packet again */
451
452 if( p_sys->b_packetizer )
453 {
454 return SendPacket( p_dec, p_oggpacket, p_block );
455 }
456 else
457 {
458 block_t *p_aout_buffer = DecodePacket( p_dec, p_oggpacket );
459 if( p_block )
460 block_Release( p_block );
461 return p_aout_buffer;
462 }
463}

465/*****************************************************************************
466 * Interleave: helper function to interleave channels
467 *****************************************************************************/
468static void Interleave( INTERLEAVE_TYPE *p_out, const INTERLEAVE_TYPE **pp_in,
469 int i_nb_channels, int i_samples, uint8_t *pi_chan_table)
470{
471 for( int j = 0; j < i_samples; j++ )
472 for( int i = 0; i < i_nb_channels; i++ )
473 {
474#ifdef MODULE_NAME_IS_tremor
475 union { int32_t i; uint32_t u;} spl;
476
477 spl.u = ((uint32_t)pp_in[i][j]) << 8;
478 p_out[j * i_nb_channels + pi_chan_table[i]] = spl.i;
479#else
480 p_out[j * i_nb_channels + pi_chan_table[i]] = pp_in[i][j];
481#endif
482 }
483}

485/*****************************************************************************
486 * DecodePacket: decodes a Vorbis packet.
487 *****************************************************************************/
488static block_t *DecodePacket( decoder_t *p_dec, ogg_packet *p_oggpacket )
489{
490 decoder_sys_t *p_sys = p_dec->p_sys;
491 int i_samples;
492
493 INTERLEAVE_TYPE **pp_pcm;
494
495 if( p_oggpacket->bytes &&
496 vorbis_synthesis( &p_sys->vb, p_oggpacket ) == 0 )
497 vorbis_synthesis_blockin( &p_sys->vd, &p_sys->vb );
498
499 /* **pp_pcm is a multichannel float vector. In stereo, for
500 * example, pp_pcm[0] is left, and pp_pcm[1] is right. i_samples is
501 * the size of each channel. Convert the float values
502 * (-1.<=range<=1.) to whatever PCM format and write it out */
503
504 if( ( i_samples = vorbis_synthesis_pcmout( &p_sys->vd, &pp_pcm ) ) > 0 )
505 {
506
507 block_t *p_aout_buffer;
508
509 p_aout_buffer =
510 decoder_NewAudioBuffer( p_dec, i_samples );
511
512 if( p_aout_buffer == NULL ) return NULL;
513
514 /* Interleave the samples */
515 Interleave( (INTERLEAVE_TYPE*)p_aout_buffer->p_buffer,
516 (const INTERLEAVE_TYPE**)pp_pcm, p_sys->vi.channels, i_samples,
517 p_sys->pi_chan_table);
518
519 /* Tell libvorbis how many samples we actually consumed */
520 vorbis_synthesis_read( &p_sys->vd, i_samples );
521
522 /* Date management */
523 p_aout_buffer->i_pts = date_Get( &p_sys->end_date );
524 p_aout_buffer->i_length = date_Increment( &p_sys->end_date,
525 i_samples ) - p_aout_buffer->i_pts;
526 return p_aout_buffer;
527 }
528 else
529 {
530 return NULL;
531 }
532}

534/*****************************************************************************
535 * SendPacket: send an ogg dated packet to the stream output.
536 *****************************************************************************/
537static block_t *SendPacket( decoder_t *p_dec, ogg_packet *p_oggpacket,
538 block_t *p_block )
539{
540 decoder_sys_t *p_sys = p_dec->p_sys;
541 int i_block_size, i_samples;
542
543 i_block_size = vorbis_packet_blocksize( &p_sys->vi, p_oggpacket );
544 if( i_block_size < 0 ) i_block_size = 0; /* non audio packet */
545 i_samples = ( p_sys->i_last_block_size + i_block_size ) >> 2;
546 p_sys->i_last_block_size = i_block_size;
547
548 /* Date management */
549 p_block->i_dts = p_block->i_pts = date_Get( &p_sys->end_date );
550
551 p_block->i_length = date_Increment( &p_sys->end_date, i_samples ) - p_block->i_pts;
552
553 return p_block;
554}

556/*****************************************************************************
557 * ParseVorbisComments
558 *****************************************************************************/
559static void ParseVorbisComments( decoder_t *p_dec )
560{
561 char *psz_name, *psz_value, *psz_comment;
562 int i = 0;
563
564 while( i < p_dec->p_sys->vc.comments )
565 {
566 psz_comment = strdup( p_dec->p_sys->vc.user_comments[i] );
567 if( !psz_comment )
568 break;
569 psz_name = psz_comment;
570 psz_value = strchr( psz_comment, '=' );
571 /* Don't add empty values */
572 if( psz_value && psz_value[1] != '\0')
573 {
574 *psz_value = '\0';
575 psz_value++;
576
577 if( !strcasecmp( psz_name, "REPLAYGAIN_TRACK_GAIN" ) ||
578 !strcasecmp( psz_name, "RG_RADIO" ) )
579 {
580 audio_replay_gain_t *r = &p_dec->fmt_out.audio_replay_gain;
581
582 r->pb_gain[AUDIO_REPLAY_GAIN_TRACK] = true;
583 r->pf_gain[AUDIO_REPLAY_GAIN_TRACK] = us_atof( psz_value );
584 }
585 else if( !strcasecmp( psz_name, "REPLAYGAIN_TRACK_PEAK" ) ||
586 !strcasecmp( psz_name, "RG_PEAK" ) )
587 {
588 audio_replay_gain_t *r = &p_dec->fmt_out.audio_replay_gain;
589
590 r->pb_peak[AUDIO_REPLAY_GAIN_TRACK] = true;
591 r->pf_peak[AUDIO_REPLAY_GAIN_TRACK] = us_atof( psz_value );
592 }
593 else if( !strcasecmp( psz_name, "REPLAYGAIN_ALBUM_GAIN" ) ||
594 !strcasecmp( psz_name, "RG_AUDIOPHILE" ) )
595 {
596 audio_replay_gain_t *r = &p_dec->fmt_out.audio_replay_gain;
597
598 r->pb_gain[AUDIO_REPLAY_GAIN_ALBUM] = true;
599 r->pf_gain[AUDIO_REPLAY_GAIN_ALBUM] = us_atof( psz_value );
600 }
601 else if( !strcasecmp( psz_name, "REPLAYGAIN_ALBUM_PEAK" ) )
602 {
603 audio_replay_gain_t *r = &p_dec->fmt_out.audio_replay_gain;
604
605 r->pb_peak[AUDIO_REPLAY_GAIN_ALBUM] = true;
606 r->pf_peak[AUDIO_REPLAY_GAIN_ALBUM] = us_atof( psz_value );
607 }
608 else if( !strcasecmp( psz_name, "METADATA_BLOCK_PICTURE" ) )
609 { /* Do nothing, for now */ }
610 else
611 {
612 if( !p_dec->p_description )
613 p_dec->p_description = vlc_meta_New();
614 if( p_dec->p_description )
615 vlc_meta_AddExtra( p_dec->p_description, psz_name, psz_value );
616 }
617
618 }
619 free( psz_comment );
620 i++;
621 }
622}

624/*****************************************************************************
625 *
626 *****************************************************************************/
627static void ConfigureChannelOrder(uint8_t *pi_chan_table, int i_channels,
628 uint32_t i_channel_mask, bool b_decode)
629{
630 const uint32_t *pi_channels_in;
631 switch( i_channels )
632 {
633 case 8:
634 pi_channels_in = pi_8channels_in;
635 break;
636 case 7:
637 pi_channels_in = pi_7channels_in;
638 break;
639 case 6:
640 case 5:
641 pi_channels_in = pi_6channels_in;
642 break;
643 case 4:
644 pi_channels_in = pi_4channels_in;
645 break;
646 case 3:
647 pi_channels_in = pi_3channels_in;
648 break;
649 default:
650 {
651 int i;
652 for( i = 0; i< i_channels; ++i )
653 {
654 pi_chan_table[i] = i;
655 }
656 return;
657 }
658 }
659
660 if( b_decode )
661 aout_CheckChannelReorder( pi_channels_in, NULL,
662 i_channel_mask, pi_chan_table );
663 else
664 aout_CheckChannelReorder( NULL, pi_channels_in,
665 i_channel_mask, pi_chan_table );
666}

716/*****************************************************************************
717 * OpenEncoder: probe the encoder and return score
718 *****************************************************************************/
719static int OpenEncoder( vlc_object_t *p_this )
720{
721 encoder_t *p_enc = (encoder_t *)p_this;
722 encoder_sys_t *p_sys;
723 int i_quality, i_min_bitrate, i_max_bitrate;
724 ogg_packet header[3];
725
726 if( p_enc->fmt_out.i_codec != VLC_CODEC_VORBIS &&
727 !p_enc->b_force )
728 {
729 return VLC_EGENERIC;
730 }
731
732 /* Allocate the memory needed to store the decoder's structure */
733 if( ( p_sys = (encoder_sys_t *)malloc(sizeof(encoder_sys_t)) ) == NULL )
734 return VLC_ENOMEM;
735 p_enc->p_sys = p_sys;
736
737 p_enc->pf_encode_audio = Encode;
738 p_enc->fmt_in.i_codec = VLC_CODEC_FL32;
739 p_enc->fmt_out.i_codec = VLC_CODEC_VORBIS;
740
741 config_ChainParse( p_enc, ENC_CFG_PREFIX, ppsz_enc_options, p_enc->p_cfg );
742
743 i_quality = var_GetInteger( p_enc, ENC_CFG_PREFIX "quality" );
744 if( i_quality > 10 ) i_quality = 10;
745 if( i_quality < 0 ) i_quality = 0;
746
747 if( var_GetBool( p_enc, ENC_CFG_PREFIX "cbr" ) ) i_quality = 0;
748 i_max_bitrate = var_GetInteger( p_enc, ENC_CFG_PREFIX "max-bitrate" );
749 i_min_bitrate = var_GetInteger( p_enc, ENC_CFG_PREFIX "min-bitrate" );
750
751 /* Initialize vorbis encoder */
752 vorbis_info_init( &p_sys->vi );
753
754 if( i_quality > 0 )
755 {
756 /* VBR mode */
757 if( vorbis_encode_setup_vbr( &p_sys->vi,
758 p_enc->fmt_in.audio.i_channels, p_enc->fmt_in.audio.i_rate,
759 i_quality * 0.1 ) )
760 {
761 vorbis_info_clear( &p_sys->vi );
762 free( p_enc->p_sys );
763 msg_Err( p_enc, "VBR mode initialisation failed" );
764 return VLC_EGENERIC;
765 }
766
767 /* Do we have optional hard quality restrictions? */
768 if( i_max_bitrate > 0 || i_min_bitrate > 0 )
769 {
770 struct ovectl_ratemanage_arg ai;
771 vorbis_encode_ctl( &p_sys->vi, OV_ECTL_RATEMANAGE_GET, &ai );
772
773 ai.bitrate_hard_min = i_min_bitrate;
774 ai.bitrate_hard_max = i_max_bitrate;
775 ai.management_active = 1;
776
777 vorbis_encode_ctl( &p_sys->vi, OV_ECTL_RATEMANAGE_SET, &ai );
778
779 }
780 else
781 {
782 /* Turn off management entirely */
783 vorbis_encode_ctl( &p_sys->vi, OV_ECTL_RATEMANAGE_SET, NULL );
784 }
785 }
786 else
787 {
788 if( vorbis_encode_setup_managed( &p_sys->vi,
789 p_enc->fmt_in.audio.i_channels, p_enc->fmt_in.audio.i_rate,
790 i_min_bitrate > 0 ? i_min_bitrate * 1000: -1,
791 p_enc->fmt_out.i_bitrate,
792 i_max_bitrate > 0 ? i_max_bitrate * 1000: -1 ) )
793 {
794 vorbis_info_clear( &p_sys->vi );
795 msg_Err( p_enc, "CBR mode initialisation failed" );
796 free( p_enc->p_sys );
797 return VLC_EGENERIC;
798 }
799 }
800
801 vorbis_encode_setup_init( &p_sys->vi );
802
803 /* Add a comment */
804 vorbis_comment_init( &p_sys->vc);
805 vorbis_comment_add_tag( &p_sys->vc, "ENCODER", "VLC media player");
806
807 /* Set up the analysis state and auxiliary encoding storage */
808 vorbis_analysis_init( &p_sys->vd, &p_sys->vi );
809 vorbis_block_init( &p_sys->vd, &p_sys->vb );
810
811 /* Create and store headers */
812 vorbis_analysis_headerout( &p_sys->vd, &p_sys->vc,
813 &header[0], &header[1], &header[2]);
814 for( int i = 0; i < 3; i++ )
815 {
816 if( xiph_AppendHeaders( &p_enc->fmt_out.i_extra, &p_enc->fmt_out.p_extra,
817 header[i].bytes, header[i].packet ) )
818 {
819 p_enc->fmt_out.i_extra = 0;
820 p_enc->fmt_out.p_extra = NULL;
821 }
822 }
823
824 p_sys->i_channels = p_enc->fmt_in.audio.i_channels;
825 p_sys->i_last_block_size = 0;
826 p_sys->i_samples_delay = 0;
827
828 ConfigureChannelOrder(p_sys->pi_chan_table, p_sys->vi.channels,
829 p_enc->fmt_in.audio.i_physical_channels, true);
830
831 return VLC_SUCCESS;
832}

834/****************************************************************************
835 * Encode: the whole thing
836 ****************************************************************************
837 * This function spits out ogg packets.
838 ****************************************************************************/
839static block_t *Encode( encoder_t *p_enc, block_t *p_aout_buf )
840{
841 encoder_sys_t *p_sys = p_enc->p_sys;
842 ogg_packet oggpacket;
843 block_t *p_block, *p_chain = NULL;
844 float **buffer;
845
846 /* FIXME: flush buffers in here */
847 if( unlikely( !p_aout_buf ) ) return NULL;
848
849 mtime_t i_pts = p_aout_buf->i_pts -
850 (mtime_t)1000000 * (mtime_t)p_sys->i_samples_delay /
851 (mtime_t)p_enc->fmt_in.audio.i_rate;
852
853 p_sys->i_samples_delay += p_aout_buf->i_nb_samples;
854
855 buffer = vorbis_analysis_buffer( &p_sys->vd, p_aout_buf->i_nb_samples );
856
857 /* convert samples to float and uninterleave */
858 for( unsigned int i = 0; i < p_sys->i_channels; i++ )
859 {
860 for( unsigned int j = 0 ; j < p_aout_buf->i_nb_samples ; j++ )
861 {
862 buffer[i][j]= ((float *)p_aout_buf->p_buffer)
863 [j * p_sys->i_channels + p_sys->pi_chan_table[i]];
864 }
865 }
866
867 vorbis_analysis_wrote( &p_sys->vd, p_aout_buf->i_nb_samples );
868
869 while( vorbis_analysis_blockout( &p_sys->vd, &p_sys->vb ) == 1 )
870 {
871 int i_samples;
872
873 vorbis_analysis( &p_sys->vb, NULL );
874 vorbis_bitrate_addblock( &p_sys->vb );
875
876 while( vorbis_bitrate_flushpacket( &p_sys->vd, &oggpacket ) )
877 {
878 int i_block_size;
879 p_block = block_Alloc( oggpacket.bytes );
880 memcpy( p_block->p_buffer, oggpacket.packet, oggpacket.bytes );
881
882 i_block_size = vorbis_packet_blocksize( &p_sys->vi, &oggpacket );
883
884 if( i_block_size < 0 ) i_block_size = 0;
885 i_samples = ( p_sys->i_last_block_size + i_block_size ) >> 2;
886 p_sys->i_last_block_size = i_block_size;
887
888 p_block->i_length = (mtime_t)1000000 *
889 (mtime_t)i_samples / (mtime_t)p_enc->fmt_in.audio.i_rate;
890
891 p_block->i_dts = p_block->i_pts = i_pts;
892
893 p_sys->i_samples_delay -= i_samples;
894
895 /* Update pts */
896 i_pts += p_block->i_length;
897 block_ChainAppend( &p_chain, p_block );
898 }
899 }
900
901 return p_chain;
902}
