145static int DecoderOpen( vlc_object_t *p_this )
146{
147 decoder_t *p_dec = (decoder_t*)p_this;
148 const int16_t *table;
149
150 switch( p_dec->fmt_in.i_codec )
151 {
152 case VLC_CODEC_ALAW:
153 table = alawtos16;
154 break;
155 case VLC_CODEC_MULAW:
156 table = ulawtos16;
157 break;
158 default:
159 return VLC_EGENERIC;
160 }
161
162 if( p_dec->fmt_in.audio.i_channels <= 0 ||
163 p_dec->fmt_in.audio.i_channels > AOUT_CHAN_MAX )
164 {
165 msg_Err( p_dec, "bad channels count (1-9): %i",
166 p_dec->fmt_in.audio.i_channels );
167 return VLC_EGENERIC;
168 }
169
170 if( p_dec->fmt_in.audio.i_rate <= 0 )
171 {
172 msg_Err( p_dec, "bad samplerate: %d Hz", p_dec->fmt_in.audio.i_rate );
173 return VLC_EGENERIC;
174 }
175
176 msg_Dbg( p_dec, "samplerate:%dHz channels:%d",
177 p_dec->fmt_in.audio.i_rate, p_dec->fmt_in.audio.i_channels );
178
179
180 /* Allocate the memory needed to store the decoder's structure */
181 decoder_sys_t *p_sys = malloc(sizeof(*p_sys));
182 if( unlikely(p_sys == NULL) )
183 return VLC_ENOMEM;
184
185 /* Set output properties */
186 p_dec->pf_decode_audio = DecodeBlock;
187 p_dec->p_sys = p_sys;
188
189 p_dec->fmt_out.i_cat = AUDIO_ES;
190 p_dec->fmt_out.i_codec = VLC_CODEC_S16N;
191 p_dec->fmt_out.audio.i_format = VLC_CODEC_S16N;
192 p_dec->fmt_out.audio.i_rate = p_dec->fmt_in.audio.i_rate;
193 if( p_dec->fmt_in.audio.i_physical_channels )
194 p_dec->fmt_out.audio.i_physical_channels =
195 p_dec->fmt_in.audio.i_physical_channels;
196 else
197 p_dec->fmt_out.audio.i_physical_channels =
198 pi_channels_maps[p_dec->fmt_in.audio.i_channels];
199 if( p_dec->fmt_in.audio.i_original_channels )
200 p_dec->fmt_out.audio.i_original_channels =
201 p_dec->fmt_in.audio.i_original_channels;
202 else
203 p_dec->fmt_out.audio.i_original_channels =
204 p_dec->fmt_out.audio.i_physical_channels;
205 aout_FormatPrepare( &p_dec->fmt_out.audio );
206
207 p_sys->table = table;
208
209 date_Init( &p_sys->end_date, p_dec->fmt_out.audio.i_rate, 1 );
210 date_Set( &p_sys->end_date, 0 );
211
212 return VLC_SUCCESS;
213}

215static block_t *DecodeBlock( decoder_t *p_dec, block_t **pp_block )
216{
217 decoder_sys_t *p_sys = p_dec->p_sys;
218
219 if( pp_block == NULL )
220 return NULL;
221 block_t *p_block = *pp_block;
222 if( p_block == NULL )
223 return NULL;
224
225 if( p_block->i_pts > VLC_TS_INVALID &&
226 p_block->i_pts != date_Get( &p_sys->end_date ) )
227 {
228 date_Set( &p_sys->end_date, p_block->i_pts );
229 }
230 else if( !date_Get( &p_sys->end_date ) )
231 {
232 /* We've just started the stream, wait for the first PTS. */
233 block_Release( p_block );
234 return NULL;
235 }
236
237 /* Don't re-use the same pts twice */
238 p_block->i_pts = VLC_TS_INVALID;
239
240 unsigned samples = p_block->i_buffer / p_dec->fmt_in.audio.i_channels;
241 if( samples == 0 )
242 {
243 block_Release( p_block );
244 return NULL;
245 }
246
247 /* Create chunks of max 1024 samples */
248 if( samples > 1024 ) samples = 1024;
249
250 block_t *p_out = decoder_NewAudioBuffer( p_dec, samples );
251 if( p_out == NULL )
252 {
253 block_Release( p_block );
254 return NULL;
255 }
256
257 assert( p_out->i_nb_samples == samples );
258 assert( p_out->i_buffer == samples * 2 * p_dec->fmt_in.audio.i_channels );
259
260 p_out->i_pts = date_Get( &p_sys->end_date );
261 p_out->i_length = date_Increment( &p_sys->end_date, samples )
262 - p_out->i_pts;
263
264 const uint8_t *src = p_block->p_buffer;
265 int16_t *dst = (int16_t *)p_out->p_buffer;
266
267 samples *= p_dec->fmt_in.audio.i_channels;
268
269 for( unsigned i = 0; i < samples; i++ )
270 *(dst++) = p_sys->table[*(src++)];
271
272 p_block->p_buffer += samples;
273 p_block->i_buffer -= samples;
274
275 return p_out;
276}

282 free( p_dec->p_sys );
