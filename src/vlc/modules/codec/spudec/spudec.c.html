45#define DVDSUBTRANS_DISABLE_TEXT N_("Disable DVD subtitle transparency")
46#define DVDSUBTRANS_DISABLE_LONGTEXT N_("Removes all transparency effects " \
47 "used in DVD subtitles.")
48
49vlc_module_begin ()
50 set_description( N_("DVD subtitles decoder") )
51 set_shortname( N_("DVD subtitles") )
52 set_capability( "decoder", 75 )
53 set_category( CAT_INPUT )
54 set_subcategory( SUBCAT_INPUT_SCODEC )
55 set_callbacks( DecoderOpen, Close )
56
57 add_bool( "dvdsub-transparency", false,
58 DVDSUBTRANS_DISABLE_TEXT, DVDSUBTRANS_DISABLE_LONGTEXT, true )
59 add_submodule ()
60 set_description( N_("DVD subtitles packetizer") )
61 set_capability( "packetizer", 50 )
62 set_callbacks( PacketizerOpen, Close )
63vlc_module_end ()

68static block_t * Reassemble( decoder_t *, block_t ** );

72/*****************************************************************************
73 * DecoderOpen
74 *****************************************************************************
75 * Tries to launch a decoder and return score so that the interface is able
76 * to chose.
77 *****************************************************************************/
78static int DecoderOpen( vlc_object_t *p_this )
79{
80 decoder_t *p_dec = (decoder_t*)p_this;
81 decoder_sys_t *p_sys;
82
83 if( p_dec->fmt_in.i_codec != VLC_CODEC_SPU )
84 return VLC_EGENERIC;
85
86 p_dec->p_sys = p_sys = malloc( sizeof( decoder_sys_t ) );
87
88 p_sys->b_packetizer = false;
89 p_sys->b_disabletrans = var_InheritBool( p_dec, "dvdsub-transparency" );
90 p_sys->i_spu_size = 0;
91 p_sys->i_spu = 0;
92 p_sys->p_block = NULL;
93
94 es_format_Init( &p_dec->fmt_out, SPU_ES, VLC_CODEC_SPU );
95
96 p_dec->pf_decode_sub = Decode;
97 p_dec->pf_packetize = NULL;
98
99 return VLC_SUCCESS;
100}
101
102/*****************************************************************************
103 * PacketizerOpen
104 *****************************************************************************
105 * Tries to launch a decoder and return score so that the interface is able
106 * to chose.
107 *****************************************************************************/
108static int PacketizerOpen( vlc_object_t *p_this )
109{
110 decoder_t *p_dec = (decoder_t*)p_this;
111
112 if( DecoderOpen( p_this ) )
113 {
114 return VLC_EGENERIC;
115 }
116 p_dec->pf_packetize = Packetize;
117 p_dec->p_sys->b_packetizer = true;
118 es_format_Copy( &p_dec->fmt_out, &p_dec->fmt_in );
119 p_dec->fmt_out.i_codec = VLC_CODEC_SPU;
120
121 return VLC_SUCCESS;
122}
123
124/*****************************************************************************
125 * Close:
126 *****************************************************************************/
127static void Close( vlc_object_t *p_this )
128{
129 decoder_t *p_dec = (decoder_t*)p_this;
130 decoder_sys_t *p_sys = p_dec->p_sys;
131
132 if( p_sys->p_block )
133 {
134 block_ChainRelease( p_sys->p_block );
135 }
136
137 free( p_sys );
138}

140/*****************************************************************************
141 * Decode:
142 *****************************************************************************/
143static subpicture_t *Decode( decoder_t *p_dec, block_t **pp_block )
144{
145 decoder_sys_t *p_sys = p_dec->p_sys;
146 block_t *p_spu_block;
147 subpicture_t *p_spu;
148
149 p_spu_block = Reassemble( p_dec, pp_block );
150
151 if( ! p_spu_block )
152 {
153 return NULL;
154 }
155
156 /* FIXME: what the, we shouldnâ€™t need to allocate 64k of buffer --sam. */
157 p_sys->i_spu = block_ChainExtract( p_spu_block, p_sys->buffer, 65536 );
158 p_sys->i_pts = p_spu_block->i_pts;
159 block_ChainRelease( p_spu_block );
160
161 /* Parse and decode */
162 p_spu = ParsePacket( p_dec );
163
164 /* reinit context */
165 p_sys->i_spu_size = 0;
166 p_sys->i_rle_size = 0;
167 p_sys->i_spu = 0;
168 p_sys->p_block = NULL;
169
170 return p_spu;
171}

173/*****************************************************************************
174 * Packetize:
175 *****************************************************************************/
176static block_t *Packetize( decoder_t *p_dec, block_t **pp_block )
177{
178 decoder_sys_t *p_sys = p_dec->p_sys;
179 block_t *p_spu = Reassemble( p_dec, pp_block );
180
181 if( ! p_spu )
182 {
183 return NULL;
184 }
185
186 p_spu->i_dts = p_spu->i_pts;
187 p_spu->i_length = 0;
188
189 /* reinit context */
190 p_sys->i_spu_size = 0;
191 p_sys->i_rle_size = 0;
192 p_sys->i_spu = 0;
193 p_sys->p_block = NULL;
194
195 return block_ChainGather( p_spu );
196}
197
198/*****************************************************************************
199 * Reassemble:
200 *****************************************************************************/
201static block_t *Reassemble( decoder_t *p_dec, block_t **pp_block )
202{
203 decoder_sys_t *p_sys = p_dec->p_sys;
204 block_t *p_block;
205
206 if( pp_block == NULL || *pp_block == NULL ) return NULL;
207 p_block = *pp_block;
208 *pp_block = NULL;
209
210 if( p_sys->i_spu_size <= 0 &&
211 ( p_block->i_pts <= VLC_TS_INVALID || p_block->i_buffer < 4 ) )
212 {
213 msg_Dbg( p_dec, "invalid starting packet (size < 4 or pts <=0)" );
214 msg_Dbg( p_dec, "spu size: %d, i_pts: %"PRId64" i_buffer: %zu",
215 p_sys->i_spu_size, p_block->i_pts, p_block->i_buffer );
216 block_Release( p_block );
217 return NULL;
218 }
219
220 block_ChainAppend( &p_sys->p_block, p_block );
221 p_sys->i_spu += p_block->i_buffer;
222
223 if( p_sys->i_spu_size <= 0 )
224 {
225 p_sys->i_spu_size = ( p_block->p_buffer[0] << 8 )|
226 p_block->p_buffer[1];
227 p_sys->i_rle_size = ( ( p_block->p_buffer[2] << 8 )|
228 p_block->p_buffer[3] ) - 4;
229
230 /* msg_Dbg( p_dec, "i_spu_size=%d i_rle=%d",
231 p_sys->i_spu_size, p_sys->i_rle_size ); */
232
233 if( p_sys->i_spu_size <= 0 || p_sys->i_rle_size >= p_sys->i_spu_size )
234 {
235 p_sys->i_spu_size = 0;
236 p_sys->i_rle_size = 0;
237 p_sys->i_spu = 0;
238 p_sys->p_block = NULL;
239
240 block_Release( p_block );
241 return NULL;
242 }
243 }
244
245 if( p_sys->i_spu >= p_sys->i_spu_size )
246 {
247 /* We have a complete sub */
248 if( p_sys->i_spu > p_sys->i_spu_size )
249 msg_Dbg( p_dec, "SPU packets size=%d should be %d",
250 p_sys->i_spu, p_sys->i_spu_size );
251
252 return p_sys->p_block;
253 }
254 return NULL;
255}
