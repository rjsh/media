72static bool isSmoothStreaming( stream_t *s )
73{
74 const uint8_t *peek;
75 int i_size = stream_Peek( s->p_source, &peek, 512 );
76 if( i_size < 512 )
77 return false;
78
79 char *peeked = malloc( 512 );
80 if( unlikely( peeked == NULL ) )
81 return false;
82
83 memcpy( peeked, peek, 512 );
84 peeked[511] = peeked[510] = '\0';
85
86 char *str;
87
88 if( !memcmp( peeked, "\xFF\xFE", 2 ) )
89 {
90 str = FromCharset( "UTF-16LE", peeked, 512 );
91 free( peeked );
92 }
93 else if( !memcmp( peeked, "\xFE\xFF", 2 ) )
94 {
95 str = FromCharset( "UTF-16BE", peeked, 512 );
96 free( peeked );
97 }
98 else
99 str = peeked;
100
101 if( str == NULL )
102 return false;
103
104 bool ret = strstr( str, "<SmoothStreamingMedia" ) != NULL;
105 free( str );
106 return ret;
107}

119static int parse_Manifest( stream_t *s )
120{
121 stream_sys_t *p_sys = s->p_sys;
122 xml_t *vlc_xml = NULL;
123 xml_reader_t *vlc_reader = NULL;
124 int type = UNKNOWN_ES;
125 const char *name, *value;
126 stream_t *st = s->p_source;
127 msg_Dbg( s, "Manifest parsing\n" );
128
129 vlc_xml = xml_Create( st );
130 if( !vlc_xml )
131 {
132 msg_Err( s, "Failed to open XML parser" );
133 return VLC_EGENERIC;
134 }
135
136 vlc_reader = xml_ReaderCreate( vlc_xml, st );
137 if( !vlc_reader )
138 {
139 msg_Err( s, "Failed to open source for parsing" );
140 xml_Delete( vlc_xml );
141 return VLC_EGENERIC;
142 }
143
144 const char *node;
145 uint8_t *WaveFormatEx;
146 sms_stream_t *sms = NULL;
147 quality_level_t *ql = NULL;
148 int64_t start_time = 0, duration = 0;
149 int64_t computed_start_time = 0, computed_duration = 0;
150 unsigned next_track_id = 1;
151 unsigned next_qid = 1;
152 int loop_count = 0;
153 bool b_weird = false;
154
155#define TIMESCALE 10000000
156 while( (type = xml_ReaderNextNode( vlc_reader, &node )) > 0 )
157 {
158 switch( type )
159 {
160 case XML_READER_STARTELEM:
161
162 if( !strcmp( node, "SmoothStreamingMedia" ) )
163 {
164 while( (name = xml_ReaderNextAttr( vlc_reader, &value )) )
165 {
166 if( !strcmp( name, "Duration" ) )
167 p_sys->vod_duration = strtoull( value, NULL, 10 );
168 else if( !strcmp( name, "TimeScale" ) )
169 p_sys->timescale = strtoull( value, NULL, 10 );
170 }
171 if( !p_sys->timescale )
172 p_sys->timescale = TIMESCALE;
173 }
174 else if( !strcmp( node, "StreamIndex" ) )
175 {
176 sms = sms_New();
177 if( unlikely( !sms ) )
178 {
179 xml_ReaderDelete( vlc_reader );
180 xml_Delete( vlc_xml );
181 return VLC_ENOMEM;
182 }
183 sms->id = next_track_id;
184 next_track_id++;
185
186 while( (name = xml_ReaderNextAttr( vlc_reader, &value )) )
187 {
188 if( !strcmp( name, "Type" ) )
189 {
190 if( !strcmp( value, "video" ) )
191 sms->type = VIDEO_ES;
192 else if( !strcmp( value, "audio" ) )
193 sms->type = AUDIO_ES;
194 else if( !strcmp( value, "text" ) )
195 sms->type = SPU_ES;
196 }
197
198 else if( !strcmp( name, "Name" ) )
199 sms->name = strdup( value );
200 else if( !strcmp( name, "TimeScale" ) )
201 sms->timescale = strtoull( value, NULL, 10 );
202 else if( !strcmp( name, "FourCC" ) )
203 sms->default_FourCC =
204 VLC_FOURCC( value[0], value[1], value[2], value[3] );
205
206 else if( !strcmp( name, "Chunks" ) )
207 {
208 sms->vod_chunks_nb = strtol( value, NULL, 10 );
209 if( sms->vod_chunks_nb == 0 ) /* live */
210 sms->vod_chunks_nb = UINT32_MAX;
211 }
212
213 else if( !strcmp( name, "QualityLevels" ) )
214 sms->qlevel_nb = strtoul( value, NULL, 10 );
215 else if( !strcmp( name, "Url" ) )
216 sms->url_template = strdup(value);
217 }
218
219 if( !sms->timescale )
220 sms->timescale = TIMESCALE;
221
222 if( !sms->name )
223 {
224 if( sms->type == VIDEO_ES )
225 sms->name = strdup( "video" );
226 else if( sms->type == AUDIO_ES )
227 sms->name = strdup( "audio" );
228 else if( sms->type == SPU_ES )
229 sms->name = strdup( "text" );
230 }
231 }
232 else if( !strcmp( node, "QualityLevel" ) )
233 {
234 if ( !sms )
235 break;
236
237 ql = ql_New();
238 if( !ql )
239 {
240 sms_Free( sms );
241 xml_ReaderDelete( vlc_reader );
242 xml_Delete( vlc_xml );
243 return VLC_ENOMEM;
244 }
245 ql->id = next_qid;
246 next_qid++;
247 while( (name = xml_ReaderNextAttr( vlc_reader, &value )) )
248 {
249 if( !strcmp( name, "Index" ) )
250 ql->Index = strtol( value, NULL, 10 );
251 else if( !strcmp( name, "Bitrate" ) )
252 ql->Bitrate = strtoull( value, NULL, 10 );
253 else if( !strcmp( name, "PacketSize" ) )
254 ql->nBlockAlign = strtoull( value, NULL, 10 );
255 else if( !strcmp( name, "FourCC" ) )
256 ql->FourCC = VLC_FOURCC( value[0], value[1],
257 value[2], value[3] );
258 else if( !strcmp( name, "CodecPrivateData" ) )
259 ql->CodecPrivateData = strdup( value );
260 else if( !strcmp( name, "WaveFormatEx" ) )
261 {
262 WaveFormatEx = decode_string_hex_to_binary( value );
263 uint16_t data_len = ((uint16_t *)WaveFormatEx)[8];
264 ql->CodecPrivateData = strndup( value + 36, data_len * 2 );
265
266 uint16_t wf_tag = ((uint16_t *)WaveFormatEx)[0];
267 wf_tag_to_fourcc( wf_tag, &ql->FourCC, NULL );
268
269 ql->Channels = ((uint16_t *)WaveFormatEx)[1];
270 ql->SamplingRate = ((uint32_t *)WaveFormatEx)[1];
271 ql->nBlockAlign = ((uint16_t *)WaveFormatEx)[6];
272 ql->BitsPerSample = ((uint16_t *)WaveFormatEx)[7];
273 free( WaveFormatEx );
274 }
275 else if( !strcmp( name, "MaxWidth" ) || !strcmp( name, "Width" ) )
276 ql->MaxWidth = strtoul( value, NULL, 10 );
277 else if( !strcmp( name, "MaxHeight" ) || !strcmp( name, "Height" ) )
278 ql->MaxHeight = strtoul( value, NULL, 10 );
279 else if( !strcmp( name, "Channels" ) )
280 ql->Channels = strtoul( value, NULL, 10 );
281 else if( !strcmp( name, "SamplingRate" ) )
282 ql->SamplingRate = strtoul( value, NULL, 10 );
283 else if( !strcmp( name, "BitsPerSample" ) )
284 ql->BitsPerSample = strtoul( value, NULL, 10 );
285 }
286
287 vlc_array_append( sms->qlevels, ql );
288 }
289 else if ( !strcmp( node, "Content" ) && sms && !sms->url_template )
290 {
291 /* empty(@Url) && ./Content == manifest embedded content */
292 sms_Free( sms );
293 sms = NULL;
294 }
295 else if( !strcmp( node, "c" ) )
296 {
297 if ( !sms )
298 break;
299 loop_count++;
300 start_time = duration = -1;
301 while( (name = xml_ReaderNextAttr( vlc_reader, &value )) )
302 {
303 if( !strcmp( name, "t" ) )
304 start_time = strtoull( value, NULL, 10 );
305 if( !strcmp( name, "d" ) )
306 duration = strtoull( value, NULL, 10 );
307 }
308 if( start_time == -1 )
309 {
310 assert( duration != -1 );
311 computed_start_time += computed_duration;
312 computed_duration = duration;
313 }
314 else if( duration == -1 )
315 {
316 assert( start_time != -1 );
317 /* Handle weird Manifests which give only the start time
318 * of the first segment. In those cases, we have to look
319 * at the start time of the second segment to compute
320 * the duration of the first one. */
321 if( loop_count == 1 )
322 {
323 b_weird = true;
324 computed_start_time = start_time;
325 continue;
326 }
327
328 computed_duration = start_time - computed_start_time;
329 if( !b_weird )
330 computed_start_time = start_time;
331 }
332 else
333 {
334 if( b_weird )
335 computed_duration = start_time - computed_start_time;
336 else
337 {
338 computed_start_time = start_time;
339 computed_duration = duration;
340 }
341 }
342
343 if( unlikely( chunk_New( sms, computed_duration,
344 computed_start_time ) == NULL ) )
345 {
346 sms_Free( sms );
347 xml_ReaderDelete( vlc_reader );
348 xml_Delete( vlc_xml );
349 return VLC_ENOMEM;
350 }
351 if( b_weird && start_time != -1 )
352 computed_start_time = start_time;
353 }
354 break;
355
356 case XML_READER_ENDELEM:
357 if( strcmp( node, "StreamIndex" ) )
358 break;
359
360 if ( sms )
361 {
362 vlc_array_append( p_sys->sms_streams, sms );
363
364 computed_start_time = 0;
365 computed_duration = 0;
366 loop_count = 0;
367 if( b_weird && !chunk_New( sms, computed_duration, computed_start_time ) )
368 {
369 sms_Free( sms );
370 xml_ReaderDelete( vlc_reader );
371 xml_Delete( vlc_xml );
372 return VLC_ENOMEM;
373 }
374
375 b_weird = false;
376 next_qid = 1;
377
378 if( sms->qlevel_nb == 0 )
379 sms->qlevel_nb = vlc_array_count( sms->qlevels );
380
381 sms = NULL;
382 }
383 break;
384
385 case XML_READER_TEXT:
386 break;
387 default:
388 sms_Free( sms );
389 xml_ReaderDelete( vlc_reader );
390 xml_Delete( vlc_xml );
391 return VLC_EGENERIC;
392 }
393 }
394#undef TIMESCALE
395
396 sms_Free( sms );
397 xml_ReaderDelete( vlc_reader );
398 xml_Delete( vlc_xml );
399
400 return VLC_SUCCESS;
401}

403static void SysCleanup( stream_sys_t *p_sys )
404{
405 if ( p_sys->sms_streams )
406 {
407 for ( int i=0; i< p_sys->sms_streams->i_count ; i++ )
408 sms_Free( p_sys->sms_streams->pp_elems[i] );
409 vlc_array_destroy( p_sys->sms_streams );
410 }
411 vlc_array_destroy( p_sys->selected_st );
412 vlc_array_destroy( p_sys->download.chunks );
413 if ( p_sys->init_chunks )
414 {
415 for ( int i=0; i< p_sys->init_chunks->i_count ; i++ )
416 chunk_Free( p_sys->init_chunks->pp_elems[i] );
417 vlc_array_destroy( p_sys->init_chunks );
418 }
419 sms_queue_free( p_sys->bws );
420 free( p_sys->base_url );
421}

423static int Open( vlc_object_t *p_this )
424{
425 stream_t *s = (stream_t*)p_this;
426 stream_sys_t *p_sys;
427
428 if( !isSmoothStreaming( s ) )
429 return VLC_EGENERIC;
430
431 msg_Info( p_this, "Smooth Streaming (%s)", s->psz_path );
432
433 s->p_sys = p_sys = calloc( 1, sizeof(*p_sys ) );
434 if( unlikely( p_sys == NULL ) )
435 return VLC_ENOMEM;
436
437 char *uri = NULL;
438 if( unlikely( asprintf( &uri, "%s://%s", s->psz_access, s->psz_path ) < 0 ) )
439 {
440 free( p_sys );
441 return VLC_ENOMEM;
442 }
443
444 /* remove the last part of the url */
445 char *pos = strrchr( uri, '/');
446 *pos = '\0';
447 p_sys->base_url = uri;
448
449 /* XXX I don't know wether or not we should allow caching */
450 p_sys->b_cache = false;
451
452 p_sys->sms_streams = vlc_array_new();
453 p_sys->selected_st = vlc_array_new();
454 p_sys->download.chunks = vlc_array_new();
455 p_sys->init_chunks = vlc_array_new();
456 if( unlikely( !p_sys->sms_streams || !p_sys->download.chunks ||
457 !p_sys->selected_st || !p_sys->init_chunks ) )
458 {
459 SysCleanup( p_sys );
460 free( p_sys );
461 return VLC_ENOMEM;
462 }
463
464 /* Parse SMS ismc content. */
465 if( parse_Manifest( s ) != VLC_SUCCESS )
466 {
467 SysCleanup( p_sys );
468 free( p_sys );
469 return VLC_EGENERIC;
470 }
471
472 if( !p_sys->vod_duration )
473 p_sys->b_live = true;
474
475 p_sys->i_tracks = vlc_array_count( p_sys->sms_streams );
476
477 /* Choose first video / audio / subtitle stream available */
478 sms_stream_t *tmp = NULL, *selected = NULL;
479 for( unsigned i = 0; i < p_sys->i_tracks; i++ )
480 {
481 tmp = vlc_array_item_at_index( p_sys->sms_streams, i );
482 selected = SMS_GET_SELECTED_ST( tmp->type );
483 if( !selected )
484 vlc_array_append( p_sys->selected_st, tmp );
485 }
486
487 /* Choose lowest quality for the first chunks */
488 quality_level_t *wanted, *qlvl;
489 sms_stream_t *sms = NULL;
490
491 for( unsigned i = 0; i < p_sys->i_tracks; i++ )
492 {
493 wanted = qlvl = NULL;
494 sms = vlc_array_item_at_index( p_sys->sms_streams, i );
495 if ( vlc_array_count( sms->qlevels ) )
496 {
497 wanted = vlc_array_item_at_index( sms->qlevels, 0 );
498 for( unsigned i=1; i < sms->qlevel_nb; i++ )
499 {
500 qlvl = vlc_array_item_at_index( sms->qlevels, i );
501 if( qlvl->Bitrate < wanted->Bitrate )
502 wanted = qlvl;
503 }
504 sms->download_qlvl = wanted->id;
505 }
506 }
507
508 vlc_mutex_init( &p_sys->download.lock_wait );
509 vlc_cond_init( &p_sys->download.wait );
510
511 /* */
512 s->pf_read = Read;
513 s->pf_peek = Peek;
514 s->pf_control = Control;
515
516 if( vlc_clone( &p_sys->thread, sms_Thread, s, VLC_THREAD_PRIORITY_INPUT ) )
517 {
518 SysCleanup( p_sys );
519 vlc_mutex_destroy( &p_sys->download.lock_wait );
520 vlc_cond_destroy( &p_sys->download.wait );
521 free( p_sys );
522 return VLC_EGENERIC;
523 }
524
525 return VLC_SUCCESS;
526}

528static void Close( vlc_object_t *p_this )
529{
530 stream_t *s = (stream_t*)p_this;
531 stream_sys_t *p_sys = s->p_sys;
532
533 vlc_mutex_lock( &p_sys->download.lock_wait );
534 p_sys->b_close = true;
535 /* Negate the condition variable's predicate */
536 for( int i = 0; i < 3; i++ )
537 p_sys->download.lead[i] = 0;
538 p_sys->playback.toffset = 0;
539 vlc_cond_signal(&p_sys->download.wait);
540 vlc_mutex_unlock( &p_sys->download.lock_wait );
541
542 vlc_join( p_sys->thread, NULL );
543 vlc_mutex_destroy( &p_sys->download.lock_wait );
544 vlc_cond_destroy( &p_sys->download.wait );
545
546 SysCleanup( p_sys );
547 free( p_sys );
548}

550static chunk_t *get_chunk( stream_t *s, const bool wait )
551{
552 stream_sys_t *p_sys = s->p_sys;
553 unsigned count;
554 chunk_t *chunk = NULL;
555
556 vlc_mutex_lock( &p_sys->download.lock_wait );
557 count = vlc_array_count( p_sys->download.chunks );
558 while( p_sys->playback.index >= count || p_sys->b_tseek )
559 {
560 /* Yes I know, checking for p_sys->b_die is not reliable,
561 * that's why vlc_object_alive() has been deprecated. But if I
562 * understood well, there is no good solution with a stream_filter
563 * module anyaway. */
564 if( !wait || p_sys->b_error )
565 {
566 vlc_mutex_unlock( &p_sys->download.lock_wait );
567 msg_Warn( s, "get_chunk failed! (playback index %u)",
568 p_sys->playback.index );
569 return NULL;
570 }
571 if( NO_MORE_CHUNKS )
572 {
573 vlc_mutex_unlock( &p_sys->download.lock_wait );
574 msg_Info( s, "No more chunks, end of the VOD" );
575 return NULL;
576 }
577
578 msg_Dbg( s, "get_chunk is waiting !!!" );
579 vlc_cond_timedwait( &p_sys->download.wait,
580 &p_sys->download.lock_wait, mdate() + 500000 );
581 count = vlc_array_count( p_sys->download.chunks );
582 msg_Dbg( s, "count is %u, and index is %u", count, p_sys->playback.index );
583 }
584 chunk = vlc_array_item_at_index( p_sys->download.chunks, p_sys->playback.index );
585
586 vlc_mutex_unlock( &p_sys->download.lock_wait );
587
588 return chunk;
589}

591static int sms_Read( stream_t *s, uint8_t *p_read, int i_read )
592{
593 stream_sys_t *p_sys = s->p_sys;
594 int copied = 0;
595 chunk_t *chunk = NULL;
596
597 do
598 {
599 chunk = get_chunk( s, true );
600 if( !chunk )
601 return copied;
602
603 if( chunk->read_pos >= (int)chunk->size )
604 {
605 if( chunk->type == VIDEO_ES ||
606 ( !SMS_GET_SELECTED_ST( VIDEO_ES ) && chunk->type == AUDIO_ES ) )
607 {
608 vlc_mutex_lock( &p_sys->download.lock_wait );
609 p_sys->playback.toffset += chunk->duration;
610 vlc_mutex_unlock( &p_sys->download.lock_wait );
611 vlc_cond_signal( &p_sys->download.wait);
612 }
613 if( !p_sys->b_cache || p_sys->b_live )
614 {
615 FREENULL( chunk->data );
616 chunk->read_pos = 0;
617 }
618
619 chunk->read_pos = 0;
620
621 p_sys->playback.index += 1;
622 msg_Dbg( s, "Incrementing playback index" );
623
624 continue;
625 }
626
627 if( chunk->read_pos == 0 )
628 {
629 const char *verb = p_read == NULL ? "skipping" : "reading";
630 msg_Dbg( s, "%s chunk %u (%u bytes), type %i",
631 verb, chunk->sequence, i_read, chunk->type );
632 /* check integrity */
633 uint32_t type;
634 uint8_t *slice = chunk->data;
635 SMS_GET4BYTES( type );
636 SMS_GETFOURCC( type );
637 assert( type == ATOM_moof || type == ATOM_uuid );
638 }
639
640 int len = -1;
641 uint8_t *src = chunk->data + chunk->read_pos;
642 if( i_read <= chunk->size - chunk->read_pos )
643 len = i_read;
644 else
645 len = chunk->size - chunk->read_pos;
646
647 if( len > 0 )
648 {
649 if( p_read ) /* otherwise caller skips data */
650 memcpy( p_read + copied, src, len );
651 chunk->read_pos += len;
652 copied += len;
653 i_read -= len;
654 }
655
656
657 } while ( i_read > 0 );
658
659 return copied;
660}

662static int Read( stream_t *s, void *buffer, unsigned i_read )
663{
664 stream_sys_t *p_sys = s->p_sys;
665 int length = 0;
666
667 if( p_sys->b_error )
668 return 0;
669
670 length = sms_Read( s, (uint8_t*) buffer, i_read );
671 if( length < 0 )
672 return 0;
673
674 /* This call to sms_Read will increment p_sys->playback.index
675 * in case the last chunk we read into is entirely read */
676 sms_Read( s, NULL, 0 );
677
678 p_sys->playback.boffset += length;
679 if( (unsigned)length < i_read )
680 msg_Warn( s, "could not read %i bytes, only %i!", i_read, length );
681
682 return length;
683}

685/* The MP4 demux should never have to to peek outside the current chunk */
686static int Peek( stream_t *s, const uint8_t **pp_peek, unsigned i_peek )
687{
688 chunk_t *chunk = get_chunk( s, true );
689 if( !chunk || !chunk->data )
690 return 0;
691
692 int bytes = chunk->size - chunk->read_pos;
693 assert( bytes > 0 );
694
695 if( (unsigned)bytes < i_peek )
696 {
697 msg_Err( s, "could not peek %u bytes, only %i!", i_peek, bytes );
698 }
699 msg_Dbg( s, "peeking at chunk %u!", chunk->sequence );
700 *pp_peek = chunk->data + chunk->read_pos;
701
702 return bytes;
703}
704
705/* Normaly a stream_filter is not able to provide *time* seeking, since a
706 * stream_filter operates on a byte stream. Thus, in order to circumvent this
707 * limitation, I treat a STREAM_SET_POSITION request which value "pos" is less
708 * than FAKE_STREAM_SIZE as a *time* seek request, and more precisely a request
709 * to jump at time position: pos / FAKE_STREAM_SIZE * total_video_duration.
710 * For exemple, it pos == 500, it would be interpreted as a request to jump at
711 * the middle of the video.
712 * If pos > 1000, it would be treated as a normal byte seek request. That means
713 * the demux is not able to request a byte seek with 0 <= pos <= 1000
714 * (unless it is in the current chunk), but that doesn't matter in practice.
715 * Of course this a bit hack-ish, but if Smooth Streaming doesn't die, its
716 * implementation will be moved to a access_demux module, and this hack won't
717 * be needed anymore (among others). */
718static int chunk_Seek( stream_t *s, const uint64_t pos )
719{
720 stream_sys_t *p_sys = s->p_sys;
721
722 if( pos == p_sys->playback.boffset )
723 return VLC_SUCCESS;
724
725 chunk_t *chunk = get_chunk( s, false );
726 if( chunk == NULL )
727 return VLC_EGENERIC;
728
729 bool inside_chunk = pos >= chunk->offset &&
730 pos < (chunk->offset + chunk->size) ? true : false;
731
732 if( inside_chunk )
733 {
734 chunk->read_pos = pos - chunk->offset;
735 p_sys->playback.boffset = pos;
736 return VLC_SUCCESS;
737 }
738 else
739 {
740 if( p_sys->b_live )
741 {
742 msg_Err( s, "Cannot seek outside the current chunk for a live stream" );
743 return VLC_EGENERIC;
744 }
745
746 msg_Info( s, "Seeking outside the current chunk" );
747 assert( pos <= FAKE_STREAM_SIZE );
748
749 vlc_mutex_lock( &p_sys->download.lock_wait );
750
751 p_sys->b_tseek = true;
752 p_sys->time_pos = p_sys->vod_duration * pos / FAKE_STREAM_SIZE;
753 for( int i = 0; i < 3; i++ )
754 p_sys->download.lead[i] = 0;
755 p_sys->playback.toffset = 0;
756
757 vlc_cond_signal( &p_sys->download.wait);
758 vlc_mutex_unlock( &p_sys->download.lock_wait );
759
760 return VLC_SUCCESS;
761 }
762}
763
764static int Control( stream_t *s, int i_query, va_list args )
765{
766 stream_sys_t *p_sys = s->p_sys;
767
768 switch( i_query )
769 {
770 case STREAM_CAN_SEEK:
771 *(va_arg( args, bool * )) = true;
772 break;
773 case STREAM_CAN_FASTSEEK:
774 case STREAM_CAN_PAUSE: /* TODO */
775 case STREAM_CAN_CONTROL_PACE:
776 *(va_arg( args, bool * )) = false;
777 break;
778 case STREAM_GET_POSITION:
779 *(va_arg( args, uint64_t * )) = p_sys->playback.boffset;
780 break;
781 case STREAM_SET_POSITION:
782 {
783 uint64_t pos = (uint64_t)va_arg(args, uint64_t);
784 int ret = chunk_Seek(s, pos);
785 if( ret == VLC_SUCCESS )
786 break;
787 else
788 return VLC_EGENERIC;
789 }
790 case STREAM_GET_SIZE:
791 *(va_arg( args, uint64_t * )) = FAKE_STREAM_SIZE;
792 break;
793 case STREAM_GET_PTS_DELAY:
794 *va_arg (args, int64_t *) = INT64_C(1000) *
795 var_InheritInteger(s, "network-caching");
796 break;
797 default:
798 return VLC_EGENERIC;
799 }
800 return VLC_SUCCESS;
801}
