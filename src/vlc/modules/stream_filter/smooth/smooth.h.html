29typedef struct item_s
30{
31 uint64_t value;
32 struct item_s *next;
33
34} item_t;
35
36typedef struct sms_queue_s
37{
38 int length;
39 item_t *first;
40} sms_queue_t;
41
42typedef struct chunk_s
43{
44 int64_t duration; /* chunk duration (seconds / TimeScale) */
45 int64_t start_time; /* PTS (seconds / TimeScale) */
46 int size; /* chunk size in bytes */
47 unsigned sequence; /* unique sequence number */
48 uint64_t offset; /* offset in the media */
49 int read_pos; /* position in the chunk */
50 int type; /* video, audio, or subtitles */
51
52 uint8_t *data;
53} chunk_t;
54
55typedef struct quality_level_s
56{
57 int Index;
58 uint32_t FourCC;
59 unsigned Bitrate;
60 unsigned MaxWidth;
61 unsigned MaxHeight;
62 unsigned SamplingRate;
63 unsigned Channels;
64 unsigned BitsPerSample;
65 unsigned AudioTag;
66 unsigned nBlockAlign;
67 unsigned id;
68 char *CodecPrivateData; /* hex encoded string */
69
70} quality_level_t;
71
72typedef struct sms_stream_s
73{
74 vlc_array_t *qlevels; /* list of available Quality Levels */
75 vlc_array_t *chunks; /* list of chunks */
76 uint32_t default_FourCC;
77 unsigned vod_chunks_nb; /* total num of chunks of the VOD stream */
78 unsigned timescale;
79 unsigned qlevel_nb; /* number of quality levels */
80 unsigned id; /* track id, will be set arbitrarily */
81 char *name;
82 char *url_template;
83 int type;
84 unsigned download_qlvl; /* current quality level ID for Download() */
85
86} sms_stream_t;
87
88struct stream_sys_t
89{
90 char *base_url; /* URL common part for chunks */
91 vlc_thread_t thread; /* SMS chunk download thread */
92
93 vlc_array_t *sms_streams; /* available streams */
94 vlc_array_t *selected_st; /* selected streams */
95 vlc_array_t *init_chunks;
96 unsigned i_tracks; /* Total number of tracks in the Manifest */
97 sms_queue_t *bws; /* Measured bandwidths of the N last chunks */
98 uint64_t vod_duration; /* total duration of the VOD media */
99 int64_t time_pos;
100 unsigned timescale;
101
102 /* Download */
103 struct sms_download_s
104 {
105 uint64_t lead[3]; /* how much audio/video/text data is available
106 (downloaded), in seconds / TimeScale */
107
108 unsigned ck_index[3]; /* current chunk for download */
109
110 uint64_t next_chunk_offset;
111 vlc_array_t *chunks; /* chunks that have been downloaded */
112 vlc_mutex_t lock_wait; /* protect chunk download counter. */
113 vlc_cond_t wait; /* some condition to wait on */
114 } download;
115
116 /* Playback */
117 struct sms_playback_s
118 {
119 uint64_t boffset; /* current byte offset in media */
120 uint64_t toffset; /* current time offset in media */
121 unsigned index; /* current chunk for playback */
122 } playback;
123
124 /* state */
125 bool b_cache; /* can cache files */
126 bool b_live; /* live stream? or vod? */
127 bool b_error; /* parsing error */
128 bool b_close; /* set by Close() */
129 bool b_tseek; /* time seeking */
130};

132#define SMS_GET4BYTES( dst ) do { \
133 dst = U32_AT( slice ); \
134 slice += 4; \
135 } while(0)
136
137#define SMS_GET1BYTE( dst ) do { \
138 dst = *slice; \
139 slice += 1; \
140 } while(0)
141
142#define SMS_GET3BYTES( dst ) do { \
143 dst = Get24bBE( slice ); \
144 slice += 3; \
145 } while(0)
146
147#define SMS_GET8BYTES( dst ) do { \
148 dst = U64_AT( slice ); \
149 slice += 8; \
150 } while(0)
151
152#define SMS_GET4or8BYTES( dst ) \
153 if( (version) == 0 ) \
154 SMS_GET4BYTES( dst ); \
155 else \
156 SMS_GET8BYTES( dst ); \
157
158#define SMS_GETFOURCC( dst ) do { \
159 memcpy( &dst, slice, 4 ); \
160 slice += 4; \
161 } while(0)
