35static char *ConstructUrl( const char *template, const char *base_url,
36 const uint64_t bandwidth, const uint64_t start_time )
37{
38 char *frag, *end, *qual;
39 char *url_template = strdup( template );
40 char *saveptr = NULL;
41
42 qual = strtok_r( url_template, "{", &saveptr );
43 strtok_r( NULL, "}", &saveptr );
44 frag = strtok_r( NULL, "{", &saveptr );
45 strtok_r( NULL, "}", &saveptr );
46 end = strtok_r( NULL, "", &saveptr );
47 char *url = NULL;
48
49 if( asprintf( &url, "%s/%s%"PRIu64"%s%"PRIu64"%s", base_url, qual,
50 bandwidth, frag, start_time, end) < 0 )
51 {
52 return NULL;
53 }
54
55 free( url_template );
56 return url;
57}

59static chunk_t * chunk_Get( sms_stream_t *sms, const int64_t start_time )
60{
61 int len = vlc_array_count( sms->chunks );
62 for( int i = 0; i < len; i++ )
63 {
64 chunk_t * chunk = vlc_array_item_at_index( sms->chunks, i );
65 if( !chunk ) return NULL;
66
67 if( chunk->start_time <= start_time &&
68 chunk->start_time + chunk->duration > start_time )
69 {
70 return chunk;
71 }
72 }
73 return NULL;
74}

76static unsigned set_track_id( chunk_t *chunk, const unsigned tid )
77{
78 uint32_t size, type;
79 if( !chunk->data )
80 return 0;
81 uint8_t *slice = chunk->data;
82 if( !slice )
83 return 0;
84
85 SMS_GET4BYTES( size );
86 SMS_GETFOURCC( type );
87 assert( type == ATOM_moof );
88
89 SMS_GET4BYTES( size );
90 SMS_GETFOURCC( type );
91 assert( type == ATOM_mfhd );
92 slice += size - 8;
93
94 SMS_GET4BYTES( size );
95 SMS_GETFOURCC( type );
96 assert( type == ATOM_traf );
97
98 SMS_GET4BYTES( size );
99 SMS_GETFOURCC( type );
100 if( type != ATOM_tfhd )
101 return 0;
102
103 unsigned ret = bswap32( ((uint32_t *)slice)[1] );
104 ((uint32_t *)slice)[1] = bswap32( tid );
105
106 return ret;
107}

109static int sms_Download( stream_t *s, chunk_t *chunk, char *url )
110{
111 stream_sys_t *p_sys = s->p_sys;
112
113 stream_t *p_ts = stream_UrlNew( s, url );
114 free( url );
115 if( p_ts == NULL )
116 return VLC_EGENERIC;
117
118 int64_t size = stream_Size( p_ts );
119
120 chunk->size = size;
121 chunk->offset = p_sys->download.next_chunk_offset;
122 p_sys->download.next_chunk_offset += chunk->size;
123
124 chunk->data = malloc( size );
125
126 if( chunk->data == NULL )
127 {
128 stream_Delete( p_ts );
129 return VLC_ENOMEM;
130 }
131
132 int read = stream_Read( p_ts, chunk->data, size );
133 if( read < size )
134 {
135 msg_Warn( s, "sms_Download: I requested %"PRIi64" bytes, "\
136 "but I got only %i", size, read );
137 chunk->data = realloc( chunk->data, read );
138 }
139
140 stream_Delete( p_ts );
141
142 vlc_mutex_lock( &p_sys->download.lock_wait );
143 int index = es_cat_to_index( chunk->type );
144 p_sys->download.lead[index] += chunk->duration;
145 vlc_mutex_unlock( &p_sys->download.lock_wait );
146
147 return VLC_SUCCESS;
148}

158static unsigned BandwidthAdaptation( stream_t *s,
159 sms_stream_t *sms, uint64_t bandwidth )
160{
161 if( sms->type != VIDEO_ES )
162 return sms->download_qlvl;
163
164 uint64_t bw_candidate = 0;
165 quality_level_t *qlevel;
166 unsigned ret = sms->download_qlvl;
167
168 for( unsigned i = 0; i < sms->qlevel_nb; i++ )
169 {
170 qlevel = vlc_array_item_at_index( sms->qlevels, i );
171 if( unlikely( !qlevel ) )
172 {
173 msg_Err( s, "Could no get %uth quality level", i );
174 return 0;
175 }
176
177 if( qlevel->Bitrate < (bandwidth - bandwidth / 3) &&
178 qlevel->Bitrate > bw_candidate )
179 {
180 bw_candidate = qlevel->Bitrate;
181 ret = qlevel->id;
182 }
183 }
184
185 return ret;
186}

189static int get_new_chunks( stream_t *s, chunk_t *ck )
190{
191 stream_sys_t *p_sys = s->p_sys;
192
193 uint8_t *slice = ck->data;
194 if( !slice )
195 return VLC_EGENERIC;
196 uint8_t version, fragment_count;
197 uint32_t size, type, flags;
198 sms_stream_t *sms;
199 UUID_t uuid;
200 TfrfBoxDataFields_t *tfrf_df;
201
202 sms = SMS_GET_SELECTED_ST( ck->type );
203
204 SMS_GET4BYTES( size );
205 SMS_GETFOURCC( type );
206 assert( type == ATOM_moof );
207
208 SMS_GET4BYTES( size );
209 SMS_GETFOURCC( type );
210 assert( type == ATOM_mfhd );
211 slice += size - 8;
212
213 SMS_GET4BYTES( size );
214 SMS_GETFOURCC( type );
215 assert( type == ATOM_traf );
216
217 for(;;)
218 {
219 SMS_GET4BYTES( size );
220 assert( size > 1 );
221 SMS_GETFOURCC( type );
222 if( type == ATOM_mdat )
223 {
224 msg_Err( s, "No uuid box found :-(" );
225 return VLC_EGENERIC;
226 }
227 else if( type == ATOM_uuid )
228 {
229 GetUUID( &uuid, slice);
230 if( !CmpUUID( &uuid, &TfrfBoxUUID ) )
231 break;
232 }
233 slice += size - 8;
234 }
235
236 slice += 16;
237 SMS_GET1BYTE( version );
238 SMS_GET3BYTES( flags );
239 SMS_GET1BYTE( fragment_count );
240
241 tfrf_df = calloc( fragment_count, sizeof( TfrfBoxDataFields_t ) );
242 if( unlikely( tfrf_df == NULL ) )
243 return VLC_EGENERIC;
244
245 for( uint8_t i = 0; i < fragment_count; i++ )
246 {
247 SMS_GET4or8BYTES( tfrf_df[i].i_fragment_abs_time );
248 SMS_GET4or8BYTES( tfrf_df[i].i_fragment_duration );
249 }
250
251 msg_Dbg( s, "read box: \"tfrf\" version %d, flags 0x%x, "\
252 "fragment count %"PRIu8, version, flags, fragment_count );
253
254 for( uint8_t i = 0; i < fragment_count; i++ )
255 {
256 int64_t dur = tfrf_df[i].i_fragment_duration;
257 int64_t stime = tfrf_df[i].i_fragment_abs_time;
258 msg_Dbg( s, "\"tfrf\" fragment duration %"PRIu64", "\
259 "fragment abs time %"PRIu64, dur, stime);
260
261 if( !chunk_Get( sms, stime + dur ) )
262 chunk_New( sms, dur, stime );
263 }
264 free( tfrf_df );
265
266 return VLC_SUCCESS;
267}

272/* SmooBox is a very simple MP4 box, used only to pass information
273 * to the demux layer. As this box is not aimed to travel accross networks,
274 * simplicity of the design is better than compactness */
275static int build_smoo_box( stream_t *s, uint8_t *smoo_box )
276{
277 stream_sys_t *p_sys = s->p_sys;
278 sms_stream_t *sms = NULL;
279 uint32_t FourCC;
280
281 /* smoo */
282 memset( smoo_box, 0, SMOO_SIZE );
283 smoo_box[2] = (SMOO_SIZE & 0xff00)>>8;
284 smoo_box[3] = SMOO_SIZE & 0xff;
285 smoo_box[4] = 'u';
286 smoo_box[5] = 'u';
287 smoo_box[6] = 'i';
288 smoo_box[7] = 'd';
289
290 /* UUID is e1da72ba-24d7-43c3-a6a5-1b5759a1a92c */
291 ((uint32_t *)smoo_box)[2] = bswap32( 0xe1da72ba );
292 ((uint32_t *)smoo_box)[3] = bswap32( 0x24d743c3 );
293 ((uint32_t *)smoo_box)[4] = bswap32( 0xa6a51b57 );
294 ((uint32_t *)smoo_box)[5] = bswap32( 0x59a1a92c );
295
296 uint8_t *stra_box;
297 for( int i = 0; i < 3; i++ )
298 {
299 sms = NULL;
300 int cat = UNKNOWN_ES;
301 stra_box = smoo_box + i * STRA_SIZE;
302
303 stra_box[26] = (STRA_SIZE & 0xff00)>>8;
304 stra_box[27] = STRA_SIZE & 0xff;
305 stra_box[28] = 'u';
306 stra_box[29] = 'u';
307 stra_box[30] = 'i';
308 stra_box[31] = 'd';
309
310 /* UUID is b03ef770-33bd-4bac-96c7-bf25f97e2447 */
311 ((uint32_t *)stra_box)[8] = bswap32( 0xb03ef770 );
312 ((uint32_t *)stra_box)[9] = bswap32( 0x33bd4bac );
313 ((uint32_t *)stra_box)[10] = bswap32( 0x96c7bf25 );
314 ((uint32_t *)stra_box)[11] = bswap32( 0xf97e2447 );
315
316 cat = index_to_es_cat( i );
317 stra_box[48] = cat;
318 sms = SMS_GET_SELECTED_ST( cat );
319
320 stra_box[49] = 0; /* reserved */
321 if( sms == NULL )
322 continue;
323 stra_box[50] = (sms->id & 0xff00)>>8;
324 stra_box[51] = sms->id & 0xff;
325
326 ((uint32_t *)stra_box)[13] = bswap32( sms->timescale );
327 ((uint64_t *)stra_box)[7] = bswap64( p_sys->vod_duration );
328
329 quality_level_t * qlvl = get_qlevel( sms, sms->download_qlvl );
330
331 if ( qlvl )
332 {
333 FourCC = qlvl->FourCC ? qlvl->FourCC : sms->default_FourCC;
334 ((uint32_t *)stra_box)[16] = bswap32( FourCC );
335 ((uint32_t *)stra_box)[17] = bswap32( qlvl->Bitrate );
336 ((uint32_t *)stra_box)[18] = bswap32( qlvl->MaxWidth );
337 ((uint32_t *)stra_box)[19] = bswap32( qlvl->MaxHeight );
338 ((uint32_t *)stra_box)[20] = bswap32( qlvl->SamplingRate );
339 ((uint32_t *)stra_box)[21] = bswap32( qlvl->Channels );
340 ((uint32_t *)stra_box)[22] = bswap32( qlvl->BitsPerSample );
341 ((uint32_t *)stra_box)[23] = bswap32( qlvl->AudioTag );
342 ((uint16_t *)stra_box)[48] = bswap16( qlvl->nBlockAlign );
343
344 if( !qlvl->CodecPrivateData )
345 continue;
346 stra_box[98] = stra_box[99] = stra_box[100] = 0; /* reserved */
347 assert( strlen( qlvl->CodecPrivateData ) < 512 );
348 stra_box[101] = strlen( qlvl->CodecPrivateData ) / 2;
349 uint8_t *binary_cpd = decode_string_hex_to_binary( qlvl->CodecPrivateData );
350 memcpy( stra_box + 102, binary_cpd, stra_box[101] );
351 free( binary_cpd );
352 }
353 }
354
355 return VLC_SUCCESS;
356}

358static chunk_t *build_init_chunk( stream_t *s )
359{
360 chunk_t *ret = calloc( 1, sizeof( chunk_t ) );
361 if( unlikely( ret == NULL ) )
362 goto build_init_chunk_error;
363
364 ret->size = SMOO_SIZE;
365 ret->data = malloc( SMOO_SIZE );
366 if( !ret->data )
367 goto build_init_chunk_error;
368
369 if( build_smoo_box( s, ret->data ) == VLC_SUCCESS)
370 return ret;
371
372 free( ret->data );
373build_init_chunk_error:
374 free( ret );
375 msg_Err( s, "build_init_chunk failed" );
376 return NULL;
377}

379static int Download( stream_t *s, sms_stream_t *sms )
380{
381 stream_sys_t *p_sys = s->p_sys;
382
383 int index = es_cat_to_index( sms->type );
384 if ( unlikely( index == -1 ) )
385 {
386 msg_Err( s, "invalid stream type" );
387 return VLC_EGENERIC;
388 }
389 int64_t start_time = p_sys->download.lead[index];
390
391 quality_level_t *qlevel = get_qlevel( sms, sms->download_qlvl );
392 if( unlikely( !qlevel ) )
393 {
394 msg_Err( s, "Could not get quality level id %u", sms->download_qlvl );
395 return VLC_EGENERIC;
396 }
397
398
399 chunk_t *chunk = chunk_Get( sms, start_time );
400 if( !chunk )
401 {
402 msg_Warn( s, "Could not find a chunk for stream %s, "\
403 "start time = %"PRIu64"", sms->name, start_time );
404 return VLC_EGENERIC;
405 }
406 if( chunk->data != NULL )
407 {
408 /* Segment already downloaded */
409 msg_Warn( s, "Segment already downloaded" );
410 return VLC_SUCCESS;
411 }
412
413 chunk->type = sms->type;
414
415 char *url = ConstructUrl( sms->url_template, p_sys->base_url,
416 qlevel->Bitrate, chunk->start_time );
417 if( !url )
418 {
419 msg_Err( s, "ConstructUrl returned NULL" );
420 return VLC_EGENERIC;
421 }
422
423 /* sanity check - can we download this chunk on time? */
424 uint64_t avg_bw = sms_queue_avg( p_sys->bws );
425 if( (avg_bw > 0) && (qlevel->Bitrate > 0) )
426 {
427 /* duration in ms */
428 unsigned chunk_duration = chunk->duration * 1000 / sms->timescale;
429 uint64_t size = chunk_duration * qlevel->Bitrate / 1000; /* bits */
430 unsigned estimated = size * 1000 / avg_bw;
431 if( estimated > chunk_duration )
432 {
433 msg_Warn( s,"downloading of chunk %d would take %d ms, "\
434 "which is longer than its playback (%d ms)",
435 chunk->sequence, estimated, chunk_duration );
436 }
437 }
438
439 mtime_t start = mdate();
440 if( sms_Download( s, chunk, url ) != VLC_SUCCESS )
441 {
442 msg_Err( s, "downloaded chunk %u from stream %s at quality\
443 %u failed", chunk->sequence, sms->name, qlevel->Bitrate );
444 return VLC_EGENERIC;
445 }
446 mtime_t duration = mdate() - start;
447
448 unsigned real_id = set_track_id( chunk, sms->id );
449 if( real_id == 0)
450 {
451 msg_Err( s, "tfhd box not found or invalid chunk" );
452 return VLC_EGENERIC;
453 }
454
455 //msg_Dbg( s, "chunk ID was %i and is now %i", real_id, sms->id );
456
457 if( p_sys->b_live )
458 get_new_chunks( s, chunk );
459
460 vlc_mutex_lock( &p_sys->download.lock_wait );
461 vlc_array_append( p_sys->download.chunks, chunk );
462 vlc_cond_signal( &p_sys->download.wait );
463 vlc_mutex_unlock( &p_sys->download.lock_wait );
464
465 msg_Info( s, "downloaded chunk %d from stream %s at quality %u",
466 chunk->sequence, sms->name, qlevel->Bitrate );
467
468 uint64_t actual_lead = chunk->start_time + chunk->duration;
469 int ind = es_cat_to_index( sms->type );
470 p_sys->download.ck_index[ind] = chunk->sequence;
471 p_sys->download.lead[ind] = __MIN( p_sys->download.lead[ind], actual_lead );
472
473 if( sms->type == VIDEO_ES ||
474 ( !SMS_GET_SELECTED_ST( VIDEO_ES ) && sms->type == AUDIO_ES ) )
475 {
476 p_sys->playback.toffset = __MIN( p_sys->playback.toffset,
477 (uint64_t)chunk->start_time );
478 }
479
480 unsigned dur_ms = __MAX( 1, duration / 1000 );
481 uint64_t bw = chunk->size * 8 * 1000 / dur_ms; /* bits / s */
482 if( sms_queue_put( p_sys->bws, bw ) != VLC_SUCCESS )
483 return VLC_EGENERIC;
484 avg_bw = sms_queue_avg( p_sys->bws );
485
486 if( sms->type != VIDEO_ES )
487 return VLC_SUCCESS;
488
489 /* Track could get disabled in mp4 demux if we trigger adaption too soon. */
490 if( chunk->sequence <= 1 )
491 return VLC_SUCCESS;
492
493 unsigned new_qlevel_id = BandwidthAdaptation( s, sms, avg_bw );
494 quality_level_t *new_qlevel = get_qlevel( sms, new_qlevel_id );
495 if( unlikely( !new_qlevel ) )
496 {
497 msg_Err( s, "Could not get quality level id %u", new_qlevel_id );
498 return VLC_EGENERIC;
499 }
500
501 if( new_qlevel->Bitrate != qlevel->Bitrate )
502 {
503 msg_Warn( s, "detected %s bandwidth (%u) stream",
504 (new_qlevel->Bitrate >= qlevel->Bitrate) ? "faster" : "lower",
505 new_qlevel->Bitrate );
506
507 sms->download_qlvl = new_qlevel_id;
508 }
509
510 if( new_qlevel->MaxWidth != qlevel->MaxWidth ||
511 new_qlevel->MaxHeight != qlevel->MaxHeight )
512 {
513 chunk_t *new_init_ck = build_init_chunk( s );
514 if( !new_init_ck )
515 {
516 return VLC_EGENERIC;
517 }
518
519 new_init_ck->offset = p_sys->download.next_chunk_offset;
520 p_sys->download.next_chunk_offset += new_init_ck->size;
521
522 vlc_mutex_lock( &p_sys->download.lock_wait );
523 vlc_array_append( p_sys->download.chunks, new_init_ck );
524 vlc_array_append( p_sys->init_chunks, new_init_ck );
525 vlc_mutex_unlock( &p_sys->download.lock_wait );
526 }
527 return VLC_SUCCESS;
528}

530static inline int64_t get_lead( stream_t *s )
531{
532 stream_sys_t *p_sys = s->p_sys;
533 int64_t lead = 0;
534 int64_t alead = p_sys->download.lead[es_cat_to_index( AUDIO_ES )];
535 int64_t vlead = p_sys->download.lead[es_cat_to_index( VIDEO_ES )];
536 bool video = SMS_GET_SELECTED_ST( VIDEO_ES ) ? true : false;
537 bool audio = SMS_GET_SELECTED_ST( AUDIO_ES ) ? true : false;
538
539 if( video && audio )
540 lead = __MIN( vlead, alead );
541 else if( video )
542 lead = vlead;
543 else
544 lead = alead;
545
546 lead -= p_sys->playback.toffset;
547 return lead;
548}

550static int next_track( stream_t *s )
551{
552 stream_sys_t *p_sys = s->p_sys;
553 uint64_t tmp, min = 0;
554 int cat, ret = UNKNOWN_ES;
555 for( int i = 0; i < 3; i++ )
556 {
557 tmp = p_sys->download.lead[i];
558 cat = index_to_es_cat( i );
559 if( (!min || tmp < min) && SMS_GET_SELECTED_ST( cat ) )
560 {
561 min = tmp;
562 ret = cat;
563 }
564 }
565 return ret;
566}

568void* sms_Thread( void *p_this )
569{
570 stream_t *s = (stream_t *)p_this;
571 stream_sys_t *p_sys = s->p_sys;
572 sms_stream_t *sms = NULL;
573 chunk_t *chunk;
574
575 int canc = vlc_savecancel();
576
577 /* We compute the average bandwidth of the 4 last downloaded
578 * chunks, but feel free to replace '4' by whatever you wish */
579 p_sys->bws = sms_queue_init( 4 );
580 if( !p_sys->bws )
581 goto cancel;
582
583 chunk_t *init_ck = build_init_chunk( s );
584 if( !init_ck )
585 goto cancel;
586
587 vlc_mutex_lock( &p_sys->download.lock_wait );
588 vlc_array_append( p_sys->download.chunks, init_ck );
589 vlc_array_append( p_sys->init_chunks, init_ck );
590 vlc_mutex_unlock( &p_sys->download.lock_wait );
591
592 p_sys->download.next_chunk_offset = init_ck->size;
593
594 /* XXX Sometimes, the video stream is cut into pieces of one exact length,
595 * while the audio stream fragments can't be made to match exactly,
596 * and for some reason the n^th advertised video fragment is related to
597 * the n+1^th advertised audio chunk or vice versa */
598
599 int64_t start_time = 0, lead = 0;
600
601 for( int i = 0; i < 3; i++ )
602 {
603 sms = SMS_GET_SELECTED_ST( index_to_es_cat( i ) );
604 if( sms && vlc_array_count( sms->chunks ) )
605 {
606 chunk = vlc_array_item_at_index( sms->chunks, 0 );
607 p_sys->download.lead[i] = chunk->start_time + p_sys->timescale / 1000;
608 if( !start_time )
609 start_time = chunk->start_time;
610
611 if( Download( s, sms ) != VLC_SUCCESS )
612 goto cancel;
613 }
614 }
615
616 while( 1 )
617 {
618 /* XXX replace magic number 10 by a value depending on
619 * LookAheadFragmentCount and DVRWindowLength */
620 vlc_mutex_lock( &p_sys->download.lock_wait );
621
622 if( p_sys->b_close )
623 {
624 vlc_mutex_unlock( &p_sys->download.lock_wait );
625 break;
626 }
627
628 lead = get_lead( s );
629
630 while( lead > 10 * p_sys->timescale + start_time || NO_MORE_CHUNKS )
631 {
632 vlc_cond_wait( &p_sys->download.wait, &p_sys->download.lock_wait );
633 lead = get_lead( s );
634
635 if( p_sys->b_close )
636 break;
637 }
638
639 if( p_sys->b_tseek )
640 {
641 int count = vlc_array_count( p_sys->download.chunks );
642 chunk_t *ck = NULL;
643 for( int i = 0; i < count; i++ )
644 {
645 ck = vlc_array_item_at_index( p_sys->download.chunks, i );
646 if( unlikely( !ck ) )
647 goto cancel;
648 ck->read_pos = 0;
649 if( ck->data == NULL )
650 continue;
651 FREENULL( ck->data );
652 }
653
654 vlc_array_destroy( p_sys->download.chunks );
655 p_sys->download.chunks = vlc_array_new();
656
657 p_sys->playback.toffset = p_sys->time_pos;
658 for( int i = 0; i < 3; i++ )
659 {
660 p_sys->download.lead[i] = p_sys->time_pos;
661 p_sys->download.ck_index[i] = 0;
662 }
663 p_sys->download.next_chunk_offset = 0;
664
665 p_sys->playback.boffset = 0;
666 p_sys->playback.index = 0;
667
668 chunk_t *new_init_ck = build_init_chunk( s );
669 if( !new_init_ck )
670 goto cancel;
671
672 new_init_ck->offset = p_sys->download.next_chunk_offset;
673 p_sys->download.next_chunk_offset += new_init_ck->size;
674
675 vlc_array_append( p_sys->download.chunks, new_init_ck );
676 vlc_array_append( p_sys->init_chunks, new_init_ck );
677 p_sys->b_tseek = false;
678 }
679 vlc_mutex_unlock( &p_sys->download.lock_wait );
680
681 sms = SMS_GET_SELECTED_ST( next_track( s ) );
682 if ( vlc_array_count( sms->chunks ) )
683 {
684 if( Download( s, sms ) != VLC_SUCCESS )
685 goto cancel;
686 }
687 }
688
689cancel:
690 p_sys->b_error = true;
691 msg_Warn(s, "Canceling download thread!");
692 vlc_restorecancel( canc );
693 return NULL;
694}
