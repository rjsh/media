34static int hex_digit( const char c )
35{
36 if (c >= 'A' && c <= 'F')
37 return c - 'A' + 10;
38 else if (c >= 'a' && c <= 'f')
39 return c - 'a' + 10;
40 else if (c >= '0' && c<= '9')
41 return c - '0';
42 else
43 return -1;
44}

46uint8_t *decode_string_hex_to_binary( const char *psz_src )
47{
48 int i = 0, j = 0, first_digit, second_digit;
49 int i_len = strlen( psz_src );
50 assert( i_len % 2 == 0 );
51 uint8_t *p_data = malloc( i_len / 2 );
52
53 if( !p_data )
54 return NULL;
55
56 while( i < i_len )
57 {
58 first_digit = hex_digit( psz_src[i++] );
59 second_digit = hex_digit( psz_src[i++] );
60 assert( first_digit >= 0 && second_digit >= 0 );
61 p_data[j++] = ( first_digit << 4 ) | second_digit;
62 }
63
64 return p_data;
65}

83chunk_t *chunk_New( sms_stream_t* sms, uint64_t duration,\
84 uint64_t start_time )
85{
86 chunk_t *chunk = calloc( 1, sizeof( chunk_t ) );
87 if( unlikely( chunk == NULL ) )
88 return NULL;
89
90 chunk->duration = duration;
91 chunk->start_time = start_time;
92 chunk->type = UNKNOWN_ES;
93 chunk->sequence = vlc_array_count( sms->chunks );
94 vlc_array_append( sms->chunks, chunk );
95 return chunk;
96}
97
98void chunk_Free( chunk_t *chunk )
99{
100 free( chunk->data );
101 free( chunk );
102}

104sms_stream_t * sms_New( void )
105{
106 sms_stream_t *sms = calloc( 1, sizeof( sms_stream_t ) );
107 if( unlikely( !sms ) ) return NULL;
108
109 sms->qlevels = vlc_array_new();
110 sms->chunks = vlc_array_new();
111 if ( unlikely(!sms->qlevels || !sms->chunks) )
112 {
113 sms_Free( sms );
114 return NULL;
115 }
116 sms->type = UNKNOWN_ES;
117 return sms;
118}
119
120void sms_Free( sms_stream_t *sms )
121{
122 if ( !sms )
123 return;
124 if( sms->qlevels )
125 {
126 for( int n = 0; n < vlc_array_count( sms->qlevels ); n++ )
127 {
128 quality_level_t *qlevel = vlc_array_item_at_index( sms->qlevels, n );
129 if( qlevel ) ql_Free( qlevel );
130 }
131 vlc_array_destroy( sms->qlevels );
132 }
133
134 if( sms->chunks )
135 {
136 for( int n = 0; n < vlc_array_count( sms->chunks ); n++ )
137 {
138 chunk_t *chunk = vlc_array_item_at_index( sms->chunks, n );
139 if( chunk) chunk_Free( chunk );
140 }
141 vlc_array_destroy( sms->chunks );
142 }
143
144 free( sms->name );
145 free( sms->url_template );
146 free( sms );
147}

171void sms_queue_free( sms_queue_t* queue )
172{
173 item_t *item = queue->first, *next = NULL;
174 while( item )
175 {
176 next = item->next;
177 free( item );
178 item = next;
179 }
180 free( queue );
181}

183int sms_queue_put( sms_queue_t *queue, const uint64_t value )
184{
185 /* Remove the last (and oldest) item */
186 item_t *item, *prev = NULL;
187 int count = 0;
188 for( item = queue->first; item != NULL; item = item->next )
189 {
190 count++;
191 if( count == queue->length )
192 {
193 FREENULL( item );
194 if( prev ) prev->next = NULL;
195 break;
196 }
197 else
198 prev = item;
199 }
200
201 /* Now insert the new item */
202 item_t *new = malloc( sizeof( item_t ) );
203 if( unlikely( !new ) )
204 return VLC_ENOMEM;
205
206 new->value = value;
207 new->next = queue->first;
208 queue->first = new;
209
210 return VLC_SUCCESS;
211}

213uint64_t sms_queue_avg( sms_queue_t *queue )
214{
215 item_t *last = queue->first;
216 if( last == NULL )
217 return 0;
218 uint64_t sum = queue->first->value;
219 for( int i = 0; i < queue->length - 1; i++ )
220 {
221 if( last )
222 {
223 last = last->next;
224 if( last )
225 sum += last->value;
226 }
227 }
228 return sum / queue->length;
229}

231sms_stream_t * sms_get_stream_by_cat( vlc_array_t *streams, int i_cat )
232{
233 sms_stream_t *ret = NULL;
234 int count = vlc_array_count( streams );
235 assert( count >= 0 && count <= 3 );
236
237 for( int i = 0; i < count; i++ )
238 {
239 ret = vlc_array_item_at_index( streams, i );
240 if( ret->type == i_cat )
241 return ret;
242 }
243 return NULL;
244}

276bool no_more_chunks( unsigned *indexes, vlc_array_t *streams )
277{
278 sms_stream_t *sms = NULL;
279 int count = vlc_array_count( streams );
280 unsigned ck_index;
281 for( int i = 0; i < count; i++ )
282 {
283 sms = vlc_array_item_at_index( streams, i );
284 ck_index = indexes[es_cat_to_index( sms->type )];
285 if( ck_index < sms->vod_chunks_nb - 1 )
286 return false;
287 }
288 return true;
289}
