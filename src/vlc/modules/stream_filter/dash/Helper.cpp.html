32std::string Helper::combinePaths (const std::string &path1, const std::string &path2)
33{
34 char path1Last = path1.at(path1.size() - 1);
35 char path2First = path2.at(0);
36
37 if(path1Last == '/' && path2First == '/')
38 return path1 + path2.substr(1, path2.size());
39
40 if(path1Last != '/' && path2First != '/')
41 return path1 + "/" + path2;
42
43 return path1 + path2;
44}
45std::string Helper::getDirectoryPath (const std::string &path)
46{
47 int pos = path.find_last_of('/');
48
49 return path.substr(0, pos);
50}
