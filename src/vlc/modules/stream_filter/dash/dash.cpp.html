79struct stream_sys_t
80{
81 dash::DASHManager *p_dashManager;
82 dash::mpd::MPD *p_mpd;
83 uint64_t position;
84 bool isLive;
85};

91/*****************************************************************************
92 * Open:
93 *****************************************************************************/
94static int Open(vlc_object_t *p_obj)
95{
96 stream_t *p_stream = (stream_t*) p_obj;
97
98 if(!dash::xml::DOMParser::isDash(p_stream->p_source))
99 return VLC_EGENERIC;
100
101 //Build a XML tree
102 dash::xml::DOMParser parser(p_stream->p_source);
103 if( !parser.parse() )
104 {
105 msg_Dbg( p_stream, "Could not parse mpd file." );
106 return VLC_EGENERIC;
107 }
108
109 //Begin the actual MPD parsing:
110 dash::mpd::MPD *mpd = dash::mpd::MPDFactory::create(parser.getRootNode(), p_stream->p_source, parser.getProfile());
111
112 if(mpd == NULL)
113 return VLC_EGENERIC;
114
115 stream_sys_t *p_sys = (stream_sys_t *) malloc(sizeof(stream_sys_t));
116 if (unlikely(p_sys == NULL))
117 return VLC_ENOMEM;
118
119 p_sys->p_mpd = mpd;
120 dash::DASHManager*p_dashManager = new dash::DASHManager(p_sys->p_mpd,
121 dash::logic::IAdaptationLogic::RateBased,
122 p_stream);
123
124 if(!p_dashManager->start())
125 {
126 delete p_dashManager;
127 free( p_sys );
128 return VLC_EGENERIC;
129 }
130 p_sys->p_dashManager = p_dashManager;
131 p_sys->position = 0;
132 p_sys->isLive = p_dashManager->getMpdManager()->getMPD()->isLive();
133 p_stream->p_sys = p_sys;
134 p_stream->pf_read = Read;
135 p_stream->pf_peek = Peek;
136 p_stream->pf_control = Control;
137
138 msg_Dbg(p_obj,"opening mpd file (%s)", p_stream->psz_path);
139
140 return VLC_SUCCESS;
141}
142/*****************************************************************************
143 * Close:
144 *****************************************************************************/
145static void Close(vlc_object_t *p_obj)
146{
147 stream_t *p_stream = (stream_t*) p_obj;
148 stream_sys_t *p_sys = (stream_sys_t *) p_stream->p_sys;
149 dash::DASHManager *p_dashManager = p_sys->p_dashManager;
150
151 delete(p_dashManager);
152 free(p_sys);
153}

157static int Seek ( stream_t *p_stream, uint64_t pos )
158{
159 stream_sys_t *p_sys = (stream_sys_t *) p_stream->p_sys;
160 dash::DASHManager *p_dashManager = p_sys->p_dashManager;
161 int i_ret = 0;
162 unsigned i_len = 0;
163 long i_read = 0;
164
165 if( pos < p_sys->position )
166 {
167 if( p_sys->position - pos > UINT_MAX )
168 {
169 msg_Err( p_stream, "Cannot seek backward that far!" );
170 return VLC_EGENERIC;
171 }
172 i_len = p_sys->position - pos;
173 i_ret = p_dashManager->seekBackwards( i_len );
174 if( i_ret == VLC_EGENERIC )
175 {
176 msg_Err( p_stream, "Cannot seek backward outside the current block :-/" );
177 return VLC_EGENERIC;
178 }
179 else
180 return VLC_SUCCESS;
181 }
182
183 /* Seek forward */
184 if( pos - p_sys->position > UINT_MAX )
185 {
186 msg_Err( p_stream, "Cannot seek forward that far!" );
187 return VLC_EGENERIC;
188 }
189 i_len = pos - p_sys->position;
190 i_read = Read( p_stream, (void *)NULL, i_len );
191 if( (unsigned)i_read == i_len )
192 return VLC_SUCCESS;
193 else
194 return VLC_EGENERIC;
195}
196
197static int Read (stream_t *p_stream, void *p_ptr, unsigned int i_len)
198{
199 stream_sys_t *p_sys = (stream_sys_t *) p_stream->p_sys;
200 dash::DASHManager *p_dashManager = p_sys->p_dashManager;
201 uint8_t *p_buffer = (uint8_t*)p_ptr;
202 int i_ret = 0;
203 int i_read = 0;
204
205 while( i_len > 0 )
206 {
207 i_read = p_dashManager->read( p_buffer, i_len );
208 if( i_read < 0 )
209 break;
210 p_buffer += i_read;
211 i_ret += i_read;
212 i_len -= i_read;
213 }
214 p_buffer -= i_ret;
215
216 if (i_read < 0)
217 {
218 switch (errno)
219 {
220 case EINTR:
221 case EAGAIN:
222 break;
223 default:
224 msg_Dbg(p_stream, "DASH Read: failed to read (%s)",
225 vlc_strerror_c(errno));
226 return 0;
227 }
228 return 0;
229 }
230
231 p_sys->position += i_ret;
232
233 return i_ret;
234}

236static int Peek (stream_t *p_stream, const uint8_t **pp_peek, unsigned int i_peek)
237{
238 stream_sys_t *p_sys = (stream_sys_t *) p_stream->p_sys;

240
241 return p_dashManager->peek( pp_peek, i_peek );
242}
243
244static int Control (stream_t *p_stream, int i_query, va_list args)
245{
246 stream_sys_t *p_sys = p_stream->p_sys;
247
248 switch (i_query)
249 {
250 case STREAM_CAN_SEEK:
251 case STREAM_CAN_FASTSEEK:
252 /*TODO Support Seek */
253 *(va_arg (args, bool *)) = SEEK;
254 break;
255 case STREAM_CAN_PAUSE:
256 case STREAM_CAN_CONTROL_PACE:
257 *(va_arg (args, bool *)) = false; /* TODO */
258 break;
259
260 case STREAM_GET_POSITION:
261 *(va_arg (args, uint64_t *)) = p_sys->position;
262 break;
263 case STREAM_SET_POSITION:
264 {
265 uint64_t pos = (uint64_t)va_arg(args, uint64_t);
266 if(Seek(p_stream, pos) == VLC_SUCCESS)
267 {
268 p_sys->position = pos;
269 break;
270 }
271 else
272 return VLC_EGENERIC;
273 }
274 case STREAM_GET_SIZE:
275 {
276 uint64_t* res = (va_arg (args, uint64_t *));
277 if(p_sys->isLive)
278 *res = 0;
279 else
280 {
281 const dash::mpd::Representation *rep = p_sys->p_dashManager->getAdaptionLogic()->getCurrentRepresentation();
282 if ( rep == NULL )
283 *res = 0;
284 else
285 *res = p_sys->p_mpd->getDuration() * rep->getBandwidth() / 8;
286 }
287 break;
288 }
289 case STREAM_GET_PTS_DELAY:
290 *va_arg (args, int64_t *) = INT64_C(1000) *
291 var_InheritInteger(p_stream, "network-caching");
292 break;
293
294 default:
295 return VLC_EGENERIC;
296 }
297 return VLC_SUCCESS;
298}
