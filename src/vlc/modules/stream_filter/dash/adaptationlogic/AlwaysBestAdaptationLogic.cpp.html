35AlwaysBestAdaptationLogic::AlwaysBestAdaptationLogic (IMPDManager *mpdManager, stream_t *stream) :
36 AbstractAdaptationLogic (mpdManager, stream)
37{
38 this->mpdManager = mpdManager;
39 this->count = 0;
40 this->initSchedule();
41}
42AlwaysBestAdaptationLogic::~AlwaysBestAdaptationLogic ()
43{
44}
45
46Chunk* AlwaysBestAdaptationLogic::getNextChunk()
47{
48 if(this->schedule.size() == 0)
49 return NULL;
50
51 if(this->count == this->schedule.size())
52 return NULL;
53
54 if ( this->count < this->schedule.size() )
55 {
56 Chunk *chunk = new Chunk();
57 chunk->setUrl(this->schedule.at( this->count )->getSourceUrl());
58 this->count++;
59 return chunk;
60 }
61 return NULL;
62}
63
64const Representation *AlwaysBestAdaptationLogic::getCurrentRepresentation() const
65{
66 if ( this->count < this->schedule.size() )
67 return this->schedule.at( this->count )->getParentRepresentation();
68 return NULL;
69}
70
71void AlwaysBestAdaptationLogic::initSchedule ()
72{
73 if(this->mpdManager != NULL)
74 {
75 std::vector<Period *> periods = this->mpdManager->getPeriods();
76
77 for(size_t i = 0; i < periods.size(); i++)
78 {
79 Representation *best = this->mpdManager->getBestRepresentation(periods.at(i));
80
81 if(best != NULL)
82 {
83 std::vector<Segment *> segments = this->mpdManager->getSegments(best);
84 for(size_t j = 0; j < segments.size(); j++)
85 {
86 this->schedule.push_back(segments.at(j));
87 }
88 }
89 }
90 }
