35RateBasedAdaptationLogic::RateBasedAdaptationLogic (IMPDManager *mpdManager, stream_t *stream) :
36 AbstractAdaptationLogic (mpdManager, stream),
37 mpdManager (mpdManager),
38 count (0),
39 currentPeriod (mpdManager->getFirstPeriod()),
40 width (0),
41 height (0)
42{
43 this->width = var_InheritInteger(stream, "dash-prefwidth");
44 this->height = var_InheritInteger(stream, "dash-prefheight");
45}
46
47Chunk* RateBasedAdaptationLogic::getNextChunk()
48{
49 if(this->mpdManager == NULL)
50 return NULL;
51
52 if(this->currentPeriod == NULL)
53 return NULL;
54
55 uint64_t bitrate = this->getBpsAvg();
56
57 if(this->getBufferPercent() < MINBUFFER)
58 bitrate = 0;
59
60 Representation *rep = this->mpdManager->getRepresentation(this->currentPeriod, bitrate, this->width, this->height);
61
62 if ( rep == NULL )
63 return NULL;
64
65 std::vector<Segment *> segments = this->mpdManager->getSegments(rep);
66
67 if ( this->count == segments.size() )
68 {
69 this->currentPeriod = this->mpdManager->getNextPeriod(this->currentPeriod);
70 this->count = 0;
71 return this->getNextChunk();
72 }
73
74 if ( segments.size() > this->count )
75 {
76 Segment *seg = segments.at( this->count );
77 Chunk *chunk = seg->toChunk();
78 //In case of UrlTemplate, we must stay on the same segment.
79 if ( seg->isSingleShot() == true )
80 this->count++;
81 seg->done();
82 return chunk;
83 }
84 return NULL;
85}
86
87const Representation *RateBasedAdaptationLogic::getCurrentRepresentation() const
88{
89 return this->mpdManager->getRepresentation( this->currentPeriod, this->getBpsAvg() );
90}
