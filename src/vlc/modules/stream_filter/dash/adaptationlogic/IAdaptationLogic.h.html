31#include "buffer/IBufferObserver.h"

33namespace dash
34{
35 namespace logic
36 {

38 {
39 public:
40
41 enum LogicType
42 {
43 Default,
44 AlwaysBest,
45 AlwaysLowest,
46 RateBased
47 };
48
49 virtual dash::http::Chunk* getNextChunk () = 0;

51 /**
52 * \return The average bitrate in bits per second.
53 */
54 virtual uint64_t getBpsAvg () const = 0;
55 virtual uint64_t getBpsLastChunk () const = 0;
56 };
57 }
58}
