32HTTPConnection::HTTPConnection (stream_t *stream) :
33 stream (stream),
34 peekBufferLen (0),
35 contentLength (0)
36{
37 this->peekBuffer = new uint8_t[PEEKBUFFER];
38}
39HTTPConnection::~HTTPConnection ()
40{
41 delete[] this->peekBuffer;
42 this->closeSocket();
43}
44
45int HTTPConnection::read (void *p_buffer, size_t len)
46{
47 if(this->peekBufferLen == 0)
48 {
49 int size = net_Read(this->stream, this->httpSocket, NULL, p_buffer, len, false);
50
51 if(size <= 0)
52 return 0;
53
54 return size;
55 }
56
57 memcpy(p_buffer, this->peekBuffer, this->peekBufferLen);
58 int ret = this->peekBufferLen;
59 this->peekBufferLen = 0;
60 return ret;
61}
62int HTTPConnection::peek (const uint8_t **pp_peek, size_t i_peek)
63{
64 if(this->peekBufferLen == 0)
65 this->peekBufferLen = this->read(this->peekBuffer, PEEKBUFFER);
66
67 int size = i_peek > this->peekBufferLen ? this->peekBufferLen : i_peek;
68
69 uint8_t *peek = new uint8_t [size];
70 memcpy(peek, this->peekBuffer, size);
71 *pp_peek = peek;
72 return size;
73}
74std::string HTTPConnection::prepareRequest (Chunk *chunk)
75{
76 std::string request;
77
78 if(!chunk->useByteRange())
79 {
80 request = "GET " + chunk->getPath() + " HTTP/1.1" + "\r\n" +
81 "Host: " + chunk->getHostname() + "\r\n" +
82 "Connection: close\r\n\r\n";
83 }
84 else
85 {
86 std::stringstream req;
87 req << "GET " << chunk->getPath() << " HTTP/1.1\r\n" <<
88 "Host: " << chunk->getHostname() << "\r\n" <<
89 "Range: bytes=" << chunk->getStartByte() << "-" << chunk->getEndByte() << "\r\n" <<
90 "Connection: close\r\n\r\n";
91
92 request = req.str();
93 }
94
95 return request;
96}
97bool HTTPConnection::init (Chunk *chunk)
98{
99 if(!chunk->hasHostname())
100 if(!this->setUrlRelative(chunk))
101 return false;
102
103 this->httpSocket = net_ConnectTCP(this->stream, chunk->getHostname().c_str(), chunk->getPort());
104
105 if(this->httpSocket == -1)
106 return false;
107
108 if(this->sendData(this->prepareRequest(chunk)))
109 return this->parseHeader();
110
111 return false;
112}
113bool HTTPConnection::parseHeader ()
114{
115 std::string line = this->readLine();
116
117 if(line.size() == 0)
118 return false;
119
120 while(line.compare("\r\n"))
121 {
122 if(!strncasecmp(line.c_str(), "Content-Length", 14))
123 this->contentLength = atoi(line.substr(15,line.size()).c_str());
124
125 line = this->readLine();
126
127 if(line.size() == 0)
128 return false;
129 }
130
131 return true;
132}
133std::string HTTPConnection::readLine ()
134{
135 std::stringstream ss;
136 char c[1];
137 size_t size = net_Read(this->stream, this->httpSocket, NULL, c, 1, false);
138
139 while(size)
140 {
141 ss << c[0];
142 if(c[0] == '\n')
143 break;
144
145 size = net_Read(this->stream, this->httpSocket, NULL, c, 1, false);
146 }
147
148 if(size > 0)
149 return ss.str();
150
151 return "";
152}
153bool HTTPConnection::sendData (const std::string& data)
154{
155 ssize_t size = net_Write(this->stream, this->httpSocket, NULL, data.c_str(), data.size());
156 if (size == -1)
157 {
158 return false;
159 }
160 if ((size_t)size != data.length())
161 {
162 this->sendData(data.substr(size, data.size()));
163 }
164
165 return true;
166}
167void HTTPConnection::closeSocket ()
168{
169 net_Close(this->httpSocket);
170}
171bool HTTPConnection::setUrlRelative (Chunk *chunk)
172{
173 std::stringstream ss;
174 ss << stream->psz_access << "://" << Helper::combinePaths(Helper::getDirectoryPath(stream->psz_path), chunk->getUrl());
175 chunk->setUrl(ss.str());
176
177 return chunk->hasHostname();
178}
