64void Chunk::setUrl (const std::string& url )
65{
66 this->url = url;
67
68 if(this->url.compare(0, 4, "http"))
69 {
70 this->isHostname = false;
71 return;
72 }
73
74 vlc_url_t url_components;
75 vlc_UrlParse(&url_components, url.c_str(), 0);
76
77 this->path = url_components.psz_path;
78 this->port = url_components.i_port ? url_components.i_port : 80;
79 this->hostname = url_components.psz_host;
80 this->isHostname = true;
81
82 vlc_UrlClean(&url_components);
83}
