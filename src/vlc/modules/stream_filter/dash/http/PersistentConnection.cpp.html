43int PersistentConnection::read (void *p_buffer, size_t len)
44{
45 if(this->chunkQueue.size() == 0)
46 return -1;
47
48 Chunk *readChunk = this->chunkQueue.front();
49
50 if(readChunk->getBytesRead() == 0)
51 {
52 if(!this->initChunk(readChunk))
53 {
54 this->chunkQueue.pop_front();
55 return -1;
56 }
57 }
58
59 if(readChunk->getBytesToRead() == 0)
60 {
61 this->chunkQueue.pop_front();
62 return 0;
63 }
64
65 int ret = 0;
66 if(len > readChunk->getBytesToRead())
67 ret = HTTPConnection::read(p_buffer, readChunk->getBytesToRead());
68 else
69 ret = HTTPConnection::read(p_buffer, len);
70
71 if(ret <= 0)
72 {
73 readChunk->setStartByte(readChunk->getStartByte() + readChunk->getBytesRead());
74 readChunk->setBytesRead(0);
75 if(!this->reconnect(readChunk))
76 {
77 this->chunkQueue.pop_front();
78 return -1;
79 }
80
81 return this->read(p_buffer, len);
82 }
83
84 readChunk->setBytesRead(readChunk->getBytesRead() + ret);
85
86 return ret;
87}
88std::string PersistentConnection::prepareRequest (Chunk *chunk)
89{
90 std::string request;
91 if(!chunk->useByteRange())
92 {
93 request = "GET " + chunk->getPath() + " HTTP/1.1" + "\r\n" +
94 "Host: " + chunk->getHostname() + "\r\n\r\n";
95 }
96 else
97 {
98 std::stringstream req;
99 req << "GET " << chunk->getPath() << " HTTP/1.1\r\n" <<
100 "Host: " << chunk->getHostname() << "\r\n" <<
101 "Range: bytes=" << chunk->getStartByte() << "-" << chunk->getEndByte() << "\r\n\r\n";
102
103 request = req.str();
104 }
105 return request;
106}
107bool PersistentConnection::init (Chunk *chunk)
108{
109 if(this->isInit)
110 return true;
111
112 if(chunk == NULL)
113 return false;
114
115 if(!chunk->hasHostname())
116 if(!this->setUrlRelative(chunk))
117 return false;
118
119 this->httpSocket = net_ConnectTCP(this->stream, chunk->getHostname().c_str(), chunk->getPort());
120
121 if(this->httpSocket == -1)
122 return false;
123
124 if(this->sendData(this->prepareRequest(chunk)))
125 this->isInit = true;
126
127 this->chunkQueue.push_back(chunk);
128 this->hostname = chunk->getHostname();
129
130 return this->isInit;
131}
132bool PersistentConnection::addChunk (Chunk *chunk)
133{
134 if(chunk == NULL)
135 return false;
136
137 if(!this->isInit)
138 return this->init(chunk);
139
140 if(!chunk->hasHostname())
141 if(!this->setUrlRelative(chunk))
142 return false;
143
144 if(chunk->getHostname().compare(this->hostname))
145 return false;
146
147 if(this->sendData(this->prepareRequest(chunk)))
148 {
149 this->chunkQueue.push_back(chunk);
150 return true;
151 }
152
153 return false;
154}
155bool PersistentConnection::initChunk (Chunk *chunk)
156{
157 if(this->parseHeader())
158 {
159 chunk->setLength(this->contentLength);
160 return true;
161 }
162
163 if(!this->reconnect(chunk))
164 return false;
165
166 if(this->parseHeader())
167 {
168 chunk->setLength(this->contentLength);
169 return true;
170 }
171
172 return false;
173}
174bool PersistentConnection::reconnect (Chunk *chunk)
175{
176 int count = 0;
177 std::string request = this->prepareRequest(chunk);
178
179 while(count < this->RETRY)
180 {
181 this->httpSocket = net_ConnectTCP(this->stream, chunk->getHostname().c_str(), chunk->getPort());
182 if(this->httpSocket != -1)
183 if(this->resendAllRequests())
184 return true;
185
186 count++;
187 }
188
189 return false;
190}
191const std::string& PersistentConnection::getHostname () const
192{
193 return this->hostname;
194}
195bool PersistentConnection::isConnected () const
196{
197 return this->isInit;
198}
199bool PersistentConnection::resendAllRequests ()
200{
201 for(size_t i = 0; i < this->chunkQueue.size(); i++)
202 if(!this->sendData((this->prepareRequest(this->chunkQueue.at(i)))))
203 return false;
204
205 return true;
206}
