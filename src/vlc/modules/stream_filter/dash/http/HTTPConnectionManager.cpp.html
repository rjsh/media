34const size_t HTTPConnectionManager::PIPELINE = 80;
35const size_t HTTPConnectionManager::PIPELINELENGTH = 2;
36const uint64_t HTTPConnectionManager::CHUNKDEFAULTBITRATE = 1;
37
38HTTPConnectionManager::HTTPConnectionManager (logic::IAdaptationLogic *adaptationLogic, stream_t *stream) :
39 adaptationLogic (adaptationLogic),
40 stream (stream),
41 chunkCount (0),
42 bpsAvg (0),
43 bpsLastChunk (0),
44 bpsCurrentChunk (0),
45 bytesReadSession (0),
46 bytesReadChunk (0),
47 timeSession (0),
48 timeChunk (0)
49{
50}
51HTTPConnectionManager::~HTTPConnectionManager ()
52{
53 this->closeAllConnections();
54}
55
56void HTTPConnectionManager::closeAllConnections ()
57{
58 vlc_delete_all(this->connectionPool);
59 vlc_delete_all(this->downloadQueue);
60}
61int HTTPConnectionManager::read (block_t *block)
62{
63 if(this->downloadQueue.size() == 0)
64 if(!this->addChunk(this->adaptationLogic->getNextChunk()))
65 return 0;
66
67 if(this->downloadQueue.front()->getPercentDownloaded() > HTTPConnectionManager::PIPELINE &&
68 this->downloadQueue.size() < HTTPConnectionManager::PIPELINELENGTH)
69 this->addChunk(this->adaptationLogic->getNextChunk());
70
71 int ret = 0;
72
73 mtime_t start = mdate();
74 ret = this->downloadQueue.front()->getConnection()->read(block->p_buffer, block->i_buffer);
75 mtime_t end = mdate();
76
77 block->i_length = (mtime_t)((ret * 8) / ((float)this->downloadQueue.front()->getBitrate() / 1000000));
78
79 double time = ((double)(end - start)) / 1000000;
80
81 if(ret <= 0)
82 {
83 this->bpsLastChunk = this->bpsCurrentChunk;
84 this->bytesReadChunk = 0;
85 this->timeChunk = 0;
86
87 delete(this->downloadQueue.front());
88 this->downloadQueue.pop_front();
89
90 return this->read(block);
91 }
92 else
93 {
94 this->updateStatistics(ret, time);
95 }
96
97 return ret;
98}
99void HTTPConnectionManager::attach (IDownloadRateObserver *observer)
100{
101 this->rateObservers.push_back(observer);
102}
103void HTTPConnectionManager::notify ()
104{
105 if ( this->bpsAvg == 0 )
106 return ;
107 for(size_t i = 0; i < this->rateObservers.size(); i++)
108 this->rateObservers.at(i)->downloadRateChanged(this->bpsAvg, this->bpsLastChunk);
109}
110std::vector<PersistentConnection *> HTTPConnectionManager::getConnectionsForHost (const std::string &hostname)
111{
112 std::vector<PersistentConnection *> cons;
113
114 for(size_t i = 0; i < this->connectionPool.size(); i++)
115 if(!this->connectionPool.at(i)->getHostname().compare(hostname) || !this->connectionPool.at(i)->isConnected())
116 cons.push_back(this->connectionPool.at(i));
117
118 return cons;
119}
120void HTTPConnectionManager::updateStatistics (int bytes, double time)
121{
122 this->bytesReadSession += bytes;
123 this->bytesReadChunk += bytes;
124 this->timeSession += time;
125 this->timeChunk += time;
126
127 this->bpsAvg = (int64_t) ((this->bytesReadSession * 8) / this->timeSession);
128 this->bpsCurrentChunk = (int64_t) ((this->bytesReadChunk * 8) / this->timeChunk);
129
130 if(this->bpsAvg < 0)
131 this->bpsAvg = 0;
132
133 if(this->bpsCurrentChunk < 0)
134 this->bpsCurrentChunk = 0;
135
136 this->notify();
137}
138bool HTTPConnectionManager::addChunk (Chunk *chunk)
139{
140 if(chunk == NULL)
141 return false;
142
143 this->downloadQueue.push_back(chunk);
144
145 std::vector<PersistentConnection *> cons = this->getConnectionsForHost(chunk->getHostname());
146
147 if(cons.size() == 0)
148 {
149 PersistentConnection *con = new PersistentConnection(this->stream);
150 this->connectionPool.push_back(con);
151 cons.push_back(con);
152 }
153
154 size_t pos = this->chunkCount % cons.size();
155
156 cons.at(pos)->addChunk(chunk);
157
158 chunk->setConnection(cons.at(pos));
159
160 this->chunkCount++;
161
162 if(chunk->getBitrate() <= 0)
163 chunk->setBitrate(HTTPConnectionManager::CHUNKDEFAULTBITRATE);
164
165 return true;
166}
