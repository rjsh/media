36DASHDownloader::DASHDownloader (HTTPConnectionManager *conManager, BlockBuffer *buffer)
37{
38 this->t_sys = (thread_sys_t *) malloc(sizeof(thread_sys_t));
39 this->t_sys->conManager = conManager;
40 this->t_sys->buffer = buffer;
41}
42DASHDownloader::~DASHDownloader ()
43{
44 this->t_sys->buffer->setEOF(true);
45 vlc_join(this->dashDLThread, NULL);
46 free(this->t_sys);
47}
48
49bool DASHDownloader::start ()
50{
51 if(vlc_clone(&(this->dashDLThread), download, (void*)this->t_sys, VLC_THREAD_PRIORITY_LOW))
52 return false;
53
54 return true;
55}
56void* DASHDownloader::download (void *thread_sys)
57{
58 thread_sys_t *t_sys = (thread_sys_t *) thread_sys;
59 HTTPConnectionManager *conManager = t_sys->conManager;
60 BlockBuffer *buffer = t_sys->buffer;
61 block_t *block = block_Alloc(BLOCKSIZE);
62 int ret = 0;
63
64 do
65 {
66 ret = conManager->read(block);
67 if(ret > 0)
68 {
69 block_t *bufBlock = block_Alloc(ret);
70 memcpy(bufBlock->p_buffer, block->p_buffer, ret);
71
72 bufBlock->i_length = block->i_length;
73 buffer->put(bufBlock);
74 }
75 }while(ret && !buffer->getEOF());
76
77 buffer->setEOF(true);
78 block_Release(block);
79
80 return NULL;
81}
