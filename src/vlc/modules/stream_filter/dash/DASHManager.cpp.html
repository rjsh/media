37DASHManager::DASHManager ( MPD *mpd,
38 IAdaptationLogic::LogicType type, stream_t *stream) :
39 conManager ( NULL ),
40 currentChunk ( NULL ),
41 adaptationLogic( NULL ),
42 logicType ( type ),
43 mpdManager ( NULL ),
44 mpd ( mpd ),
45 stream ( stream ),
46 downloader ( NULL ),
47 buffer ( NULL )
48{
49}
50DASHManager::~DASHManager ()
51{
52 delete this->downloader;
53 delete this->buffer;
54 delete this->conManager;
55 delete this->adaptationLogic;
56 delete this->mpdManager;
57}
58
59bool DASHManager::start()
60{
61 this->mpdManager = mpd::MPDManagerFactory::create( mpd );
62
63 if ( this->mpdManager == NULL )
64 return false;
65
66 this->adaptationLogic = AdaptationLogicFactory::create( this->logicType, this->mpdManager, this->stream);
67
68 if ( this->adaptationLogic == NULL )
69 return false;
70
71 this->conManager = new dash::http::HTTPConnectionManager(this->adaptationLogic, this->stream);
72 this->buffer = new BlockBuffer(this->stream);
73 this->downloader = new DASHDownloader(this->conManager, this->buffer);
74
75 this->conManager->attach(this->adaptationLogic);
76 this->buffer->attach(this->adaptationLogic);
77
78 return this->downloader->start();
79}
80int DASHManager::read( void *p_buffer, size_t len )
81{
82 return this->buffer->get(p_buffer, len);
83}
84
85int DASHManager::seekBackwards( unsigned i_len )

90int DASHManager::peek( const uint8_t **pp_peek, size_t i_peek )

92 return this->buffer->peek(pp_peek, i_peek);
93}
