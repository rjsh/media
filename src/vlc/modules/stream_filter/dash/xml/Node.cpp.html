36const std::string Node::EmptyString = "";
37
38Node::Node() :
39 type( -1 )
40{
41}
42Node::~Node ()
43{
44 for(size_t i = 0; i < this->subNodes.size(); i++)
45 delete(this->subNodes.at(i));
46}
47
48const std::vector<Node*>& Node::getSubNodes () const
49{
50 return this->subNodes;
51}
52void Node::addSubNode (Node *node)
53{
54 this->subNodes.push_back(node);
55}
56const std::string& Node::getName () const
57{
58 return this->name;
59}
60void Node::setName (const std::string& name)
61{
62 this->name = name;
63}
64
65bool Node::hasAttribute (const std::string& name)
66{
67 if(this->attributes.find(name) != this->attributes.end())
68 return true;
69
70 return false;
71}
72const std::string& Node::getAttributeValue (const std::string& key) const
73{
74 std::map<std::string, std::string>::const_iterator it = this->attributes.find( key );
75
76 if ( it != this->attributes.end() )
77 return it->second;
78 return EmptyString;
79}
80
81void Node::addAttribute ( const std::string& key, const std::string& value)
82{
83 this->attributes[key] = value;
84}
85std::vector<std::string> Node::getAttributeKeys () const
86{
87 std::vector<std::string> keys;
88 std::map<std::string, std::string>::const_iterator it;
89
90 for(it = this->attributes.begin(); it != this->attributes.end(); ++it)
91 {
92 keys.push_back(it->first);
93 }
94 return keys;
95}
96
97bool Node::hasText () const
98{
99 return false;
100}
101
102const std::string& Node::getText () const
103{
104 if ( this->type == XML_READER_TEXT )
105 return this->text;
106 else
107 {
108 assert( this->subNodes.size() == 1 );
109 return this->subNodes[0]->getText();
110 }
111}
112
113void Node::setText(const std::string &text)
114{
115 this->text = text;
116}
117
118const std::map<std::string,std::string>& Node::getAttributes () const
119{
120 return this->attributes;
121}
122
123int Node::getType() const
124{
125 return this->type;
126}
127
128void Node::setType(int type)
129{
130 this->type = type;
131}
