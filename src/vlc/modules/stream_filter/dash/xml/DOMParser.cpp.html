35DOMParser::DOMParser (stream_t *stream) :

43DOMParser::~DOMParser ()
44{
45 delete this->root;
46 if(this->vlc_reader)
47 xml_ReaderDelete(this->vlc_reader);
48 if ( this->vlc_xml )
49 xml_Delete( this->vlc_xml );
50}

52Node* DOMParser::getRootNode ()
53{

56bool DOMParser::parse ()
57{
58 this->vlc_xml = xml_Create(this->stream);
59
60 if(!this->vlc_xml)
61 return false;
62
63 this->vlc_reader = xml_ReaderCreate(this->vlc_xml, this->stream);
64
65 if(!this->vlc_reader)
66 return false;
67
68 this->root = this->processNode();
69 if ( this->root == NULL )
70 return false;
71
72 return true;
73}
74Node* DOMParser::processNode ()
75{
76 const char *data;
77 int type = xml_ReaderNextNode(this->vlc_reader, &data);
78 if(type != -1 && type != XML_READER_NONE && type != XML_READER_ENDELEM)
79 {
80 Node *node = new Node();
81 node->setType( type );
82
83 if ( type != XML_READER_TEXT )
84 {
85 std::string name = data;
86 bool isEmpty = xml_ReaderIsEmptyElement(this->vlc_reader);
87 node->setName(name);
88
89 this->addAttributesToNode(node);
90
91 if(isEmpty)
92 return node;
93
94 Node *subnode = NULL;
95
96 while((subnode = this->processNode()) != NULL)
97 node->addSubNode(subnode);
98 }
99 else
100 node->setText( data );
101 return node;
102 }
103 return NULL;
104}
105void DOMParser::addAttributesToNode (Node *node)
106{
107 const char *attrValue;
108 const char *attrName;
109
110 while((attrName = xml_ReaderNextAttr(this->vlc_reader, &attrValue)) != NULL)
111 {
112 std::string key = attrName;
113 std::string value = attrValue;
114 node->addAttribute(key, value);
115 }
116}
117void DOMParser::print (Node *node, int offset)
118{
119 for(int i = 0; i < offset; i++)
120 msg_Dbg(this->stream, " ");
121
122 msg_Dbg(this->stream, "%s", node->getName().c_str());
123
124 std::vector<std::string> keys = node->getAttributeKeys();
125
126 for(size_t i = 0; i < keys.size(); i++)
127 msg_Dbg(this->stream, " %s=%s", keys.at(i).c_str(), node->getAttributeValue(keys.at(i)).c_str());
128
129 msg_Dbg(this->stream, "\n");
130
131 offset++;
132
133 for(size_t i = 0; i < node->getSubNodes().size(); i++)
134 {
135 this->print(node->getSubNodes().at(i), offset);
136 }
137}
138void DOMParser::print ()
139{
140 this->print(this->root, 0);
141}
142bool DOMParser::isDash (stream_t *stream)
143{
144 const char* psz_namespaceDIS = "urn:mpeg:mpegB:schema:DASH:MPD:DIS2011";
145 const char* psz_namespaceIS = "urn:mpeg:DASH:schema:MPD:2011";
146
147 const uint8_t *peek;
148 int peek_size = stream_Peek(stream, &peek, 1024);
149 if (peek_size < (int)strlen(psz_namespaceDIS))
150 return false;
151
152 std::string header((const char*)peek, peek_size);
153 return (header.find(psz_namespaceDIS) != std::string::npos) || (header.find(psz_namespaceIS) != std::string::npos);
154}
155Profile DOMParser::getProfile ()
156{
157 if(this->root == NULL)
158 return dash::mpd::UnknownProfile;
159
160 std::string profile = this->root->getAttributeValue("profiles");
161 if ( profile.length() == 0 )
162 profile = this->root->getAttributeValue("profile"); //The standard spells it the both ways...
163
164 if(profile.find("urn:mpeg:mpegB:profile:dash:isoff-basic-on-demand:cm") != std::string::npos ||
165 profile.find("urn:mpeg:dash:profile:isoff-ondemand:2011") != std::string::npos ||
166 profile.find("urn:mpeg:dash:profile:isoff-on-demand:2011") != std::string::npos)
167 return dash::mpd::BasicCM;
168
169 if(profile.find("urn:mpeg:dash:profile:isoff-main:2011") != std::string::npos)
170 return dash::mpd::IsoffMain;
171
172 return dash::mpd::UnknownProfile;
173}
