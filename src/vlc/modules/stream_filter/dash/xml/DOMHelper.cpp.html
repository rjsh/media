32std::vector<Node *> DOMHelper::getElementByTagName (Node *root, const std::string& name, bool selfContain)
33{
34 std::vector<Node *> elements;
35
36 for(size_t i = 0; i < root->getSubNodes().size(); i++)
37 {
38 getElementsByTagName(root->getSubNodes().at(i), name, &elements, selfContain);
39 }
40
41 return elements;
42}
43
44std::vector<Node *> DOMHelper::getChildElementByTagName (Node *root, const std::string& name)
45{
46 std::vector<Node *> elements;
47
48 for(size_t i = 0; i < root->getSubNodes().size(); i++)
49 {
50 if( root->getSubNodes().at(i)->getName() == name )
51 elements.push_back(root->getSubNodes().at(i));
52 }
53
54 return elements;
55}
56
57void DOMHelper::getElementsByTagName (Node *root, const std::string& name, std::vector<Node*> *elements, bool selfContain)
58{
59 if(!selfContain && !root->getName().compare(name))
60 {
61 elements->push_back(root);
62 return;
63 }
64
65 if(!root->getName().compare(name))
66 elements->push_back(root);
67
68 for(size_t i = 0; i < root->getSubNodes().size(); i++)
69 {
70 getElementsByTagName(root->getSubNodes().at(i), name, elements, selfContain);
71 }
72}
73
74Node* DOMHelper::getFirstChildElementByName( Node *root, const std::string &name )
75{
76 for(size_t i = 0; i < root->getSubNodes().size(); i++)
77 {
78 if( root->getSubNodes().at( i )->getName() == name )
79 return root->getSubNodes().at( i );
80 }
81 return NULL;
82}
