40SegmentTemplate::SegmentTemplate( bool containRuntimeIdentifier,
41 Representation* representation ) :
42 Segment( representation ),
43 containRuntimeIdentifier( containRuntimeIdentifier ),
44 beginTime( std::string::npos ),
45 beginIndex( std::string::npos ),
46 currentSegmentIndex( 0 )
47{
48}
49
50std::string SegmentTemplate::getSourceUrl() const
51{
52 std::string res = this->sourceUrl;
53
54 if ( this->containRuntimeIdentifier == false )
55 return Segment::getSourceUrl();
56
57 if ( this->beginIndex != std::string::npos )
58 std::cerr << "Unhandled identifier \"$Index$\"" << std::endl;
59 if ( this->beginTime != std::string::npos )
60 {
61 //FIXME: This should use the current representation SegmentInfo
62 //which "inherits" the SegmentInfoDefault values.
63 if ( this->parentRepresentation->getParentGroup()->getSegmentInfoDefault() != NULL &&
64 this->parentRepresentation->getParentGroup()->getSegmentInfoDefault()->getSegmentTimeline() != NULL )
65 {
66 const SegmentTimeline::Element *el = this->parentRepresentation->getParentGroup()->
67 getSegmentInfoDefault()->getSegmentTimeline()->getElement( this->currentSegmentIndex );
68 if ( el != NULL )
69 {
70 std::ostringstream oss;
71 oss << el->t;
72 res.replace( this->beginTime, strlen("$Time$"), oss.str() );
73 }
74 }
75 }
76 return res;
77}
78
79void SegmentTemplate::setSourceUrl( const std::string &url )
80{
81 if ( this->containRuntimeIdentifier == true )
82 {
83 this->beginTime = url.find( "$Time$" );
84 this->beginIndex = url.find( "$Index$" );
85 }
86 Segment::setSourceUrl( url );
87}
88
89bool SegmentTemplate::isSingleShot() const
90{
91 return false;
92}
93
94void SegmentTemplate::done()
95{
96 this->currentSegmentIndex++;
97}
