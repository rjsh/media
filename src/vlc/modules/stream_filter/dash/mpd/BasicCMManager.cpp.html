43std::vector<Segment*> BasicCMManager::getSegments( const Representation *rep )
44{
45 std::vector<Segment *> retSegments;
46 SegmentInfo* info = rep->getSegmentInfo();
47 Segment* initSegment = info->getInitialisationSegment();
48
49 if ( initSegment )
50 retSegments.push_back( initSegment );
51 retSegments.insert( retSegments.end(), info->getSegments().begin(),
52 info->getSegments().end() );
53 return retSegments;
54}

60Representation* BasicCMManager::getBestRepresentation (Period *period)
61{
62 std::vector<AdaptationSet *> adaptSet = period->getAdaptationSets();
63
64 uint64_t bitrate = 0;
65 Representation *best = NULL;
66
67 for(size_t i = 0; i < adaptSet.size(); i++)
68 {
69 std::vector<Representation *> reps = adaptSet.at(i)->getRepresentations();
70 for(size_t j = 0; j < reps.size(); j++)
71 {
72 uint64_t currentBitrate = reps.at(j)->getBandwidth();
73
74 if( currentBitrate > bitrate)
75 {
76 bitrate = currentBitrate;
77 best = reps.at(j);
78 }
79 }
80 }
81 return best;
82}
83Period* BasicCMManager::getFirstPeriod ()
84{
85 std::vector<Period *> periods = this->mpd->getPeriods();
86
87 if(periods.size() == 0)
88 return NULL;
89
90 return periods.at(0);
91}

93Representation* BasicCMManager::getRepresentation(Period *period, uint64_t bitrate ) const
94{
95 std::vector<AdaptationSet *> adaptSet = period->getAdaptationSets();
96
97 Representation *best = NULL;
98
99 for(size_t i = 0; i < adaptSet.size(); i++)
100 {
101 std::vector<Representation *> reps = adaptSet.at(i)->getRepresentations();
102 for( size_t j = 0; j < reps.size(); j++ )
103 {
104 uint64_t currentBitrate = reps.at(j)->getBandwidth();
105
106 if ( best == NULL ||
107 ( currentBitrate > best->getBandwidth() &&
108 currentBitrate < bitrate ) )
109 {
110 best = reps.at( j );
111 }
112 }
113 }
114 return best;
115}
116Period* BasicCMManager::getNextPeriod (Period *period)
117{
118 std::vector<Period *> periods = this->mpd->getPeriods();
119
120 for(size_t i = 0; i < periods.size(); i++)
121 {
122 if(periods.at(i) == period && (i + 1) < periods.size())
123 return periods.at(i + 1);
124 }
125
126 return NULL;
127}
