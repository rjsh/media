31using namespace dash::xml;
32using namespace dash::mpd;
33
34MPD* MPDFactory::create (dash::xml::Node *root, stream_t *p_stream, Profile profile)
35{
36 switch(profile)
37 {
38 case dash::mpd::Full2011:
39 case dash::mpd::Basic:
40 case dash::mpd::BasicCM: return MPDFactory::createBasicCMMPD(root, p_stream);
41 case dash::mpd::IsoffMain: return MPDFactory::createIsoffMainMPD(root, p_stream);
42
43 default: return NULL;
44 }
45}
46MPD* MPDFactory::createBasicCMMPD (dash::xml::Node *root, stream_t *p_stream)
47{
48 dash::mpd::BasicCMParser mpdParser(root, p_stream);
49
50 if(mpdParser.parse() == false || mpdParser.getMPD() == NULL)
51 return NULL;
52 mpdParser.getMPD()->setProfile( dash::mpd::BasicCM );
53 return mpdParser.getMPD();
54}
55MPD* MPDFactory::createIsoffMainMPD (dash::xml::Node *root, stream_t *p_stream)
56{
57 dash::mpd::IsoffMainParser mpdParser(root, p_stream);
58
59 if(mpdParser.parse() == false || mpdParser.getMPD() == NULL)
60 return NULL;
61 mpdParser.getMPD()->setProfile( dash::mpd::IsoffMain );
62 return mpdParser.getMPD();
63}
