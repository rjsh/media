36Segment::Segment(const Representation *parent) :
37 startByte (-1),
38 endByte (-1),
39 parentRepresentation( parent )
40{
41 assert( parent != NULL );
42 if ( parent->getSegmentInfo() != NULL && parent->getSegmentInfo()->getDuration() >= 0 )
43 this->size = parent->getBandwidth() * parent->getSegmentInfo()->getDuration();
44 else
45 this->size = -1;
46}

87dash::http::Chunk* Segment::toChunk ()
88{
89 Chunk *chunk = new Chunk();
90
91 if(this->startByte != -1 && this->endByte != -1)
92 {
93 chunk->setUseByteRange(true);
94 chunk->setStartByte(this->startByte);
95 chunk->setEndByte(this->endByte);
96 }
97
98 if(this->baseUrls.size() > 0)
99 {
100 std::stringstream ss;
101 ss << this->baseUrls.at(0)->getUrl() << this->sourceUrl;
102 chunk->setUrl(ss.str());
103 ss.clear();
104
105 for(size_t i = 1; i < this->baseUrls.size(); i++)
106 {
107 ss << this->baseUrls.at(i)->getUrl() << this->sourceUrl;
108 chunk->addOptionalUrl(ss.str());
109 ss.clear();
110 }
111 }
112 else
113 {
114 chunk->setUrl(this->sourceUrl);
115 }
116
117 chunk->setBitrate(this->parentRepresentation->getBandwidth());
118
119 return chunk;
120}
