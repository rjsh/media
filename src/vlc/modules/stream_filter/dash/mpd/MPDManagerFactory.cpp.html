32IMPDManager* MPDManagerFactory::create( MPD *mpd )
33{
34 switch( mpd->getProfile() )
35 {
36 case mpd::BasicCM:
37 case mpd::Full2011: return new BasicCMManager (mpd);
38 case mpd::IsoffMain: return new IsoffMainManager (mpd);
39 case mpd::Basic:
40 case mpd::UnknownProfile:
41 default:
42 return NULL;
43 }
44}
