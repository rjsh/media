38AdaptationSet::AdaptationSet() :
39 subsegmentAlignmentFlag( false ),
40 segmentInfoDefault( NULL ),
41 isBitstreamSwitching( false )
42{
43}
44
45AdaptationSet::~AdaptationSet ()
46{
47 delete this->segmentInfoDefault;
48 vlc_delete_all( this->representations );
49}
50
51bool AdaptationSet::getSubsegmentAlignmentFlag() const
52{
53 return this->subsegmentAlignmentFlag;
54}
55
56void AdaptationSet::setSubsegmentAlignmentFlag(bool alignment)
57{
58 this->subsegmentAlignmentFlag = alignment;
59}
60
61std::vector<Representation*> AdaptationSet::getRepresentations ()
62{
63 return this->representations;
64}
65
66const Representation *AdaptationSet::getRepresentationById(const std::string &id) const
67{
68 std::vector<Representation*>::const_iterator it = this->representations.begin();
69 std::vector<Representation*>::const_iterator end = this->representations.end();
70
71 while ( it != end )
72 {
73 if ( (*it)->getId() == id )
74 return *it;
75 ++it;
76 }
77 return NULL;
78}

80const SegmentInfoDefault *AdaptationSet::getSegmentInfoDefault() const
