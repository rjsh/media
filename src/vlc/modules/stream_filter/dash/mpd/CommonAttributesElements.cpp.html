37CommonAttributesElements::CommonAttributesElements() :
38 width( -1 ),
39 height( -1 ),
40 parX( 1 ),
41 parY( 1 ),
42 frameRate( -1 )
43{
44}
45
46CommonAttributesElements::~CommonAttributesElements()
47{
48 vlc_delete_all( this->contentProtections );
49 vlc_delete_all( this->accessibilities );
50 vlc_delete_all( this->ratings );
51 vlc_delete_all( this->viewpoints );
52}
