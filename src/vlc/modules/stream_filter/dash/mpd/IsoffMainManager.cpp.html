42std::vector<Segment*> IsoffMainManager::getSegments (const Representation *rep)
43{
44 std::vector<Segment *> retSegments;
45 SegmentList* list= rep->getSegmentList();
46
47 if(rep->getSegmentBase())
48 {
49 Segment* initSegment = rep->getSegmentBase()->getInitSegment();
50
51 if(initSegment)
52 retSegments.push_back(initSegment);
53 }
54
55 retSegments.insert(retSegments.end(), list->getSegments().begin(), list->getSegments().end());
56 return retSegments;
57}

62Representation* IsoffMainManager::getBestRepresentation (Period *period)
63{
64 std::vector<AdaptationSet *> adaptationSets = period->getAdaptationSets();
65
66 int bitrate = 0;
67 Representation *best = NULL;
68
69 for(size_t i = 0; i < adaptationSets.size(); i++)
70 {
71 std::vector<Representation *> reps = adaptationSets.at(i)->getRepresentations();
72 for(size_t j = 0; j < reps.size(); j++)
73 {
74 int currentBitrate = reps.at(j)->getBandwidth();
75
76 if(currentBitrate > bitrate)
77 {
78 bitrate = currentBitrate;
79 best = reps.at(j);
80 }
81 }
82 }
83 return best;
84}
85Period* IsoffMainManager::getFirstPeriod ()

87 std::vector<Period *> periods = this->mpd->getPeriods();
88
89 if(periods.size() == 0)
90 return NULL;

94Representation* IsoffMainManager::getRepresentation (Period *period, uint64_t bitrate) const
95{
96 if(period == NULL)
97 return NULL;
98
99 std::vector<AdaptationSet *> adaptationSets = period->getAdaptationSets();
100
101 Representation *best = NULL;
102
103 for(size_t i = 0; i < adaptationSets.size(); i++)
104 {
105 std::vector<Representation *> reps = adaptationSets.at(i)->getRepresentations();
106 for( size_t j = 0; j < reps.size(); j++ )
107 {
108 uint64_t currentBitrate = reps.at(j)->getBandwidth();
109
110 if(best == NULL || (currentBitrate > best->getBandwidth() && currentBitrate < bitrate))
111 {
112 best = reps.at( j );
113 }
114 }
115 }
116 return best;
117}
118Period* IsoffMainManager::getNextPeriod (Period *period)
119{
120 std::vector<Period *> periods = this->mpd->getPeriods();
121
122 for(size_t i = 0; i < periods.size(); i++)
123 {
124 if(periods.at(i) == period && (i + 1) < periods.size())
125 return periods.at(i + 1);
126 }
127
128 return NULL;
129}

134Representation* IsoffMainManager::getRepresentation (Period *period, uint64_t bitrate, int width, int height) const
135{
136 if(period == NULL)
137 return NULL;
138
139 std::vector<AdaptationSet *> adaptationSets = period->getAdaptationSets();
140 std::vector<Representation *> resMatchReps;
141
142 for(size_t i = 0; i < adaptationSets.size(); i++)
143 {
144 std::vector<Representation *> reps = adaptationSets.at(i)->getRepresentations();
145 for( size_t j = 0; j < reps.size(); j++ )
146 {
147 if(reps.at(j)->getWidth() == width && reps.at(j)->getHeight() == height)
148 resMatchReps.push_back(reps.at(j));
149 }
150 }
151
152 if(resMatchReps.size() == 0)
153 return this->getRepresentation(period, bitrate);
154
155 Representation *best = NULL;
156 for( size_t j = 0; j < resMatchReps.size(); j++ )
157 {
158 uint64_t currentBitrate = resMatchReps.at(j)->getBandwidth();
159
160 if(best == NULL || (currentBitrate > best->getBandwidth() && currentBitrate < bitrate))
161 {
162 best = resMatchReps.at(j);
163 }
164 }
165
166 return best;
167}
