25#ifndef SEGMENT_H_
26#define SEGMENT_H_
27
28#include <string>
29#include <sstream>
30#include <vector>
31#include "mpd/BaseUrl.h"
32#include "http/Chunk.h"
33
34namespace dash
35{
36 namespace mpd
37 {
38 class Representation;
39 class Segment
40 {
41 public:
42 Segment( const Representation *parent );
43 virtual ~Segment(){}
44 virtual std::string getSourceUrl() const;
45 virtual void setSourceUrl( const std::string &url );
46 /**
47 * @return true if the segment should be dropped after being read.
48 * That is basically true when using an Url, and false
49 * when using an UrlTemplate
50 */
51 virtual bool isSingleShot () const;
52 virtual void done ();
53 virtual void addBaseUrl (BaseUrl *url);
54 virtual const std::vector<BaseUrl *>& getBaseUrls () const;
55 virtual void setByteRange (int start, int end);
56 virtual int getStartByte () const;
57 virtual int getEndByte () const;
58 virtual dash::http::Chunk* toChunk ();
59 const Representation* getParentRepresentation() const;
60 virtual int getSize() const;
61
62 protected:
63 std::string sourceUrl;
64 std::vector<BaseUrl *> baseUrls;
65 int startByte;
66 int endByte;
67 const Representation* parentRepresentation;
68 int size;
69 };
70 }
71}
72
73#endif /* SEGMENT_H_ */
