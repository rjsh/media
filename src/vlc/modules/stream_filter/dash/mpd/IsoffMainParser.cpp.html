34IsoffMainParser::IsoffMainParser (Node *root, stream_t *p_stream) :
35 root (root),
36 p_stream (p_stream),
37 mpd (NULL),
38 currentRepresentation( NULL )
39{
40}
41IsoffMainParser::~IsoffMainParser ()
42{
43}

45bool IsoffMainParser::parse ()
46{
47 this->mpd = new MPD();
48
49 this->setMPDAttributes();
50 this->setMPDBaseUrl();
51 this->setPeriods();
52
53 return true;
54}
55MPD* IsoffMainParser::getMPD ()
56{
57 return this->mpd;
58}
59void IsoffMainParser::setMPDAttributes ()
60{
61 const std::map<std::string, std::string> attr = this->root->getAttributes();
62
63 std::map<std::string, std::string>::const_iterator it;
64
65 it = attr.find("mediaPresentationDuration");
66 if(it != attr.end())
67 this->mpd->setDuration(str_duration(it->second.c_str()));
68
69 it = attr.find("minBufferTime");
70 if(it != attr.end())
71 this->mpd->setMinBufferTime(str_duration( it->second.c_str()));
72
73}
74void IsoffMainParser::setMPDBaseUrl ()
75{
76 std::vector<Node *> baseUrls = DOMHelper::getChildElementByTagName(this->root, "BaseURL");
77
78 for(size_t i = 0; i < baseUrls.size(); i++)
79 {
80 BaseUrl *url = new BaseUrl(baseUrls.at(i)->getText());
81 this->mpd->addBaseUrl(url);
82 }
83}
84void IsoffMainParser::setPeriods ()
85{
86 std::vector<Node *> periods = DOMHelper::getElementByTagName(this->root, "Period", false);
87
88 for(size_t i = 0; i < periods.size(); i++)
89 {
90 Period *period = new Period();
91 this->setAdaptationSets(periods.at(i), period);
92 this->mpd->addPeriod(period);
93 }
94}
95void IsoffMainParser::setAdaptationSets (Node *periodNode, Period *period)
96{
97 std::vector<Node *> adaptationSets = DOMHelper::getElementByTagName(periodNode, "AdaptationSet", false);
98
99 for(size_t i = 0; i < adaptationSets.size(); i++)
100 {
101 AdaptationSet *adaptationSet = new AdaptationSet();
102 this->setRepresentations(adaptationSets.at(i), adaptationSet);
103 period->addAdaptationSet(adaptationSet);
104 }
105}
106void IsoffMainParser::setRepresentations (Node *adaptationSetNode, AdaptationSet *adaptationSet)
107{
108 std::vector<Node *> representations = DOMHelper::getElementByTagName(adaptationSetNode, "Representation", false);
109
110 for(size_t i = 0; i < representations.size(); i++)
111 {
112 this->currentRepresentation = new Representation;
113 Node *repNode = representations.at(i);
114
115 if(repNode->hasAttribute("width"))
116 this->currentRepresentation->setWidth(atoi(repNode->getAttributeValue("width").c_str()));
117
118 if(repNode->hasAttribute("height"))
119 this->currentRepresentation->setHeight(atoi(repNode->getAttributeValue("height").c_str()));
120
121 if(repNode->hasAttribute("bandwidth"))
122 this->currentRepresentation->setBandwidth(atoi(repNode->getAttributeValue("bandwidth").c_str()));
123
124 this->setSegmentBase(repNode, this->currentRepresentation);
125 this->setSegmentList(repNode, this->currentRepresentation);
126 adaptationSet->addRepresentation(this->currentRepresentation);
127 }
128}
129void IsoffMainParser::setSegmentBase (dash::xml::Node *repNode, Representation *rep)
130{
131 std::vector<Node *> segmentBase = DOMHelper::getElementByTagName(repNode, "SegmentBase", false);
132
133 if(segmentBase.size() > 0)
134 {
135 SegmentBase *base = new SegmentBase();
136 this->setInitSegment(segmentBase.at(0), base);
137 rep->setSegmentBase(base);
138 }
139}
140void IsoffMainParser::setSegmentList (dash::xml::Node *repNode, Representation *rep)
141{
142 std::vector<Node *> segmentList = DOMHelper::getElementByTagName(repNode, "SegmentList", false);
143
144 if(segmentList.size() > 0)
145 {
146 SegmentList *list = new SegmentList();
147 this->setSegments(segmentList.at(0), list);
148 rep->setSegmentList(list);
149 }
150
151}
152void IsoffMainParser::setInitSegment (dash::xml::Node *segBaseNode, SegmentBase *base)
153{
154 std::vector<Node *> initSeg = DOMHelper::getElementByTagName(segBaseNode, "Initialisation", false);
155
156 if(initSeg.size() == 0)
157 initSeg = DOMHelper::getElementByTagName(segBaseNode, "Initialization", false);
158
159 if(initSeg.size() > 0)
160 {
161 Segment *seg = new Segment( this->currentRepresentation );
162 seg->setSourceUrl(initSeg.at(0)->getAttributeValue("sourceURL"));
163
164 if(initSeg.at(0)->hasAttribute("range"))
165 {
166 std::string range = initSeg.at(0)->getAttributeValue("range");
167 size_t pos = range.find("-");
168 seg->setByteRange(atoi(range.substr(0, pos).c_str()), atoi(range.substr(pos + 1, range.size()).c_str()));
169 }
170
171 for(size_t i = 0; i < this->mpd->getBaseUrls().size(); i++)
172 seg->addBaseUrl(this->mpd->getBaseUrls().at(i));
173
174 base->addInitSegment(seg);
175 }
176}
177void IsoffMainParser::setSegments (dash::xml::Node *segListNode, SegmentList *list)
178{
179 std::vector<Node *> segments = DOMHelper::getElementByTagName(segListNode, "SegmentURL", false);
180
181 for(size_t i = 0; i < segments.size(); i++)
182 {
183 Segment *seg = new Segment( this->currentRepresentation );
184 seg->setSourceUrl(segments.at(i)->getAttributeValue("media"));
185
186 if(segments.at(i)->hasAttribute("mediaRange"))
187 {
188 std::string range = segments.at(i)->getAttributeValue("mediaRange");
189 size_t pos = range.find("-");
190 seg->setByteRange(atoi(range.substr(0, pos).c_str()), atoi(range.substr(pos + 1, range.size()).c_str()));
191 }
192
193 for(size_t j = 0; j < this->mpd->getBaseUrls().size(); j++)
194 seg->addBaseUrl(this->mpd->getBaseUrls().at(j));
195
196 list->addSegment(seg);
197 }
198}
199void IsoffMainParser::print ()
200{
201 if(this->mpd)
202 {
203 msg_Dbg(this->p_stream, "MPD profile=%d mediaPresentationDuration=%ld minBufferTime=%ld", this->mpd->getProfile(),
204 this->mpd->getDuration(),
205 this->mpd->getMinBufferTime());
206 const std::vector<BaseUrl *> baseurls = this->mpd->getBaseUrls();
207
208 for(size_t i = 0; i < baseurls.size(); i++)
209 msg_Dbg(this->p_stream, "BaseUrl=%s", baseurls.at(i)->getUrl().c_str());
210
211 const std::vector<Period *> periods = this->mpd->getPeriods();
212
213 for(size_t i = 0; i < periods.size(); i++)
214 {
215 Period *period = periods.at(i);
216 msg_Dbg(this->p_stream, " Period");
217 for(size_t j = 0; j < period->getAdaptationSets().size(); j++)
218 {
219 AdaptationSet *aset = period->getAdaptationSets().at(j);
220 msg_Dbg(this->p_stream, " AdaptationSet");
221 for(size_t k = 0; k < aset->getRepresentations().size(); k++)
222 {
223 Representation *rep = aset->getRepresentations().at(k);
224 msg_Dbg(this->p_stream, " Representation");
225 Segment *initSeg = rep->getSegmentBase()->getInitSegment();
226 msg_Dbg(this->p_stream, " InitSeg url=%s", initSeg->getSourceUrl().c_str());
227 for(size_t l = 0; l < rep->getSegmentList()->getSegments().size(); l++)
228 {
229 Segment *seg = rep->getSegmentList()->getSegments().at(l);
230 msg_Dbg(this->p_stream, " Segment url=%s", seg->getSourceUrl().c_str());
231 }
232 }
233 }
