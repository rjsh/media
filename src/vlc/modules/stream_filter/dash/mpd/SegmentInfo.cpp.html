30using namespace dash::mpd;
31
32SegmentInfo::SegmentInfo() :
33 initSeg( NULL )
34{
35}
36
37SegmentInfo::~SegmentInfo ()
38{
39 for(size_t i = 0; i < this->segments.size(); i++)
40 delete(this->segments.at(i));
41
42 delete(this->initSeg);
43}
44
45const std::vector<Segment*>& SegmentInfo::getSegments () const
46{
47 return this->segments;
48}
49
50void SegmentInfo::addSegment (Segment *seg)
51{
52 this->segments.push_back(seg);
53}
