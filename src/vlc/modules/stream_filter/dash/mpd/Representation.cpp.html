34Representation::Representation () :

36 qualityRanking ( -1 ),
37 segmentInfo ( NULL ),
38 trickModeType ( NULL ),
39 parentGroup ( NULL ),
40 segmentBase ( NULL ),
41 segmentList ( NULL ),
42 width (0),
43 height (0)
44
45{
46}

48Representation::~Representation ()
49{

52}
