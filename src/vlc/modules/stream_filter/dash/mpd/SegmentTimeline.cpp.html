57void dash::mpd::SegmentTimeline::addElement(dash::mpd::SegmentTimeline::Element *e)
58{
59 int64_t offset = 0;
60
61 for ( int i = 0; i < e->r; ++i )
62 {
63 this->elements.push_back( e );
64 e = new SegmentTimeline::Element( *e );
65 offset += e->d;
66 e->t += offset;
67 }
68 this->elements.push_back( e );
69}
70
71const SegmentTimeline::Element* SegmentTimeline::getElement( unsigned int index ) const
72{
73 if ( this->elements.size() <= index )
74 return NULL;
75 std::list<Element*>::const_iterator it = this->elements.begin();
76 std::list<Element*>::const_iterator end = this->elements.end();
77 unsigned int i = 0;
78 while ( it != end )
79 {
80 if ( i == index )
81 return *it;
82 ++it;
83 ++i;
84 }
85 return NULL;
86}
