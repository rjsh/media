50/*****************************************************************************
51 * mux_sys_t: mux descriptor
52 *****************************************************************************/
53struct sout_mux_sys_t
54{
55 AVIOContext *io;
56 int io_buffer_size;
57 uint8_t *io_buffer;

60
61 bool b_write_header;
62 bool b_write_keyframe;
63 bool b_error;
64};

77/*****************************************************************************
78 * Open
79 *****************************************************************************/
80int OpenMux( vlc_object_t *p_this )
81{
82 AVOutputFormat *file_oformat;
83 sout_mux_t *p_mux = (sout_mux_t*)p_this;
84 bool dummy = !strcmp( p_mux->p_access->psz_access, "dummy");
85
86 if( dummy && strlen(p_mux->p_access->psz_path)
87 >= sizeof (((AVFormatContext *)NULL)->filename) )
88 return VLC_EGENERIC;
89
90 vlc_init_avformat(p_this);
91
92 config_ChainParse( p_mux, "sout-avformat-", ppsz_mux_options, p_mux->p_cfg );
93
94 /* Find the requested muxer */
95 char *psz_mux = var_InheritString( p_mux, "sout-avformat-mux" );
96 if( psz_mux )
97 {
98 file_oformat = av_guess_format( psz_mux, NULL, NULL );
99 free( psz_mux );
100 }
101 else
102 {
103 file_oformat =
104 av_guess_format( NULL, p_mux->p_access->psz_path, NULL);
105 }
106 if (!file_oformat)
107 {
108 msg_Err( p_mux, "unable for find a suitable output format" );
109 return VLC_EGENERIC;
110 }
111
112 sout_mux_sys_t *p_sys = malloc( sizeof( sout_mux_sys_t ) );
113 if( unlikely(p_sys == NULL) )
114 return VLC_ENOMEM;
115
116 p_mux->p_sys = p_sys;
117 p_sys->oc = avformat_alloc_context();
118 p_sys->oc->oformat = file_oformat;
119 /* If we use dummy access, let avformat write output */
120 if( dummy )
121 strcpy( p_sys->oc->filename, p_mux->p_access->psz_path );
122
123 /* Create I/O wrapper */
124 p_sys->io_buffer_size = 10 * 1024 * 1024; /* FIXME */
125 p_sys->io_buffer = malloc( p_sys->io_buffer_size );
126
127 bool b_can_seek;
128 if( sout_AccessOutControl( p_mux->p_access, ACCESS_OUT_CAN_SEEK, &b_can_seek ) )
129 b_can_seek = false;
130 p_sys->io = avio_alloc_context(
131 p_sys->io_buffer, p_sys->io_buffer_size,
132 1, p_mux, NULL, IOWrite, b_can_seek ? IOSeek : NULL );
133
134 p_sys->oc->pb = p_sys->io;
135 p_sys->oc->nb_streams = 0;
136
137 p_sys->b_write_header = true;
138 p_sys->b_write_keyframe = false;
139 p_sys->b_error = false;
140
141 /* Fill p_mux fields */
142 p_mux->pf_control = Control;
143 p_mux->pf_addstream = AddStream;
144 p_mux->pf_delstream = DelStream;
145 p_mux->pf_mux = Mux;
146
147 return VLC_SUCCESS;
148}

150/*****************************************************************************
151 * Close
152 *****************************************************************************/
153void CloseMux( vlc_object_t *p_this )
154{
155 sout_mux_t *p_mux = (sout_mux_t*)p_this;
156 sout_mux_sys_t *p_sys = p_mux->p_sys;
157
158 if( !p_sys->b_write_header && !p_sys->b_error && av_write_trailer( p_sys->oc ) < 0 )
159 {
160 msg_Err( p_mux, "could not write trailer" );
161 }
162
163 avformat_free_context(p_sys->oc);
164
165 free( p_sys->io_buffer );
166 free( p_sys );
167}

169/*****************************************************************************
170 * AddStream
171 *****************************************************************************/
172static int AddStream( sout_mux_t *p_mux, sout_input_t *p_input )
173{
174 sout_mux_sys_t *p_sys = p_mux->p_sys;
175 es_format_t *fmt = p_input->p_fmt;
176 AVCodecContext *codec;
177 AVStream *stream;
178 unsigned i_codec_id;
179
180 msg_Dbg( p_mux, "adding input" );
181
182 if( !GetFfmpegCodec( fmt->i_codec, 0, &i_codec_id, 0 ) )
183 {
184 msg_Dbg( p_mux, "couldn't find codec for fourcc '%4.4s'",
185 (char *)&fmt->i_codec );
186 return VLC_EGENERIC;
187 }
188
189 unsigned opus_size[XIPH_MAX_HEADER_COUNT];
190 void *opus_packet[XIPH_MAX_HEADER_COUNT];
191 if( fmt->i_codec == VLC_CODEC_OPUS )
192 {
193 unsigned count;
194 /* Only transmits the first packet (OpusHead) */
195 if( xiph_SplitHeaders(opus_size, opus_packet, &count, fmt->i_extra, fmt->p_extra ) ) {
196 count = 0;
197 }
198 if (count != 2 || opus_size[0] < 19) {
199 msg_Err(p_mux, "Invalid Opus header");
200 return VLC_EGENERIC;
201 }
202 }
203
204 p_input->p_sys = malloc( sizeof( int ) );
205 *((int *)p_input->p_sys) = p_sys->oc->nb_streams;
206
207 if( fmt->i_cat != VIDEO_ES && fmt->i_cat != AUDIO_ES)
208 {
209 msg_Warn( p_mux, "Unhandled ES category" );
210 return VLC_EGENERIC;
211 }
212
213 stream = avformat_new_stream( p_sys->oc, NULL);
214 if( !stream )
215 {
216 free( p_input->p_sys );
217 return VLC_EGENERIC;
218 }
219 codec = stream->codec;
220
221 codec->opaque = p_mux;
222
223 switch( fmt->i_cat )
224 {
225 case AUDIO_ES:
226 codec->codec_type = AVMEDIA_TYPE_AUDIO;
227 codec->channels = fmt->audio.i_channels;
228 codec->sample_rate = fmt->audio.i_rate;
229 codec->time_base = (AVRational){1, codec->sample_rate};
230 codec->frame_size = fmt->audio.i_frame_length;
231 if (fmt->i_bitrate == 0) {
232 msg_Warn( p_mux, "Missing audio bitrate, assuming 64k" );
233 fmt->i_bitrate = 64000;
234 }
235 break;
236
237 case VIDEO_ES:
238 if( !fmt->video.i_frame_rate || !fmt->video.i_frame_rate_base ) {
239 msg_Warn( p_mux, "Missing frame rate, assuming 25fps" );
240 fmt->video.i_frame_rate = 25;
241 fmt->video.i_frame_rate_base = 1;
242 } else
243 msg_Dbg( p_mux, "Muxing framerate will be %d/%d = %.2f fps",
244 fmt->video.i_frame_rate,
245 fmt->video.i_frame_rate_base,
246 (double)fmt->video.i_frame_rate/(double)fmt->video.i_frame_rate_base );
247
248 codec->codec_type = AVMEDIA_TYPE_VIDEO;
249 codec->width = fmt->video.i_width;
250 codec->height = fmt->video.i_height;
251 av_reduce( &codec->sample_aspect_ratio.num,
252 &codec->sample_aspect_ratio.den,
253 fmt->video.i_sar_num,
254 fmt->video.i_sar_den, 1 << 30 /* something big */ );
255 msg_Dbg(p_mux, "Muxing aspect ratio will be %d/%d",
256 fmt->video.i_sar_num, fmt->video.i_sar_den);
257 stream->sample_aspect_ratio.den = codec->sample_aspect_ratio.den;
258 stream->sample_aspect_ratio.num = codec->sample_aspect_ratio.num;
259 codec->time_base.den = fmt->video.i_frame_rate;
260 codec->time_base.num = fmt->video.i_frame_rate_base;
261 if (fmt->i_bitrate == 0) {
262 msg_Warn( p_mux, "Missing video bitrate, assuming 512k" );
263 fmt->i_bitrate = 512000;
264 } else
265 msg_Dbg( p_mux, "Muxing video bitrate will be %d", fmt->i_bitrate );
266 break;
267
268 }
269
270 codec->bit_rate = fmt->i_bitrate;
271 codec->codec_tag = av_codec_get_tag( p_sys->oc->oformat->codec_tag, i_codec_id );
272 if( !codec->codec_tag && i_codec_id == AV_CODEC_ID_MP2 )
273 {
274 i_codec_id = AV_CODEC_ID_MP3;
275 codec->codec_tag = av_codec_get_tag( p_sys->oc->oformat->codec_tag, i_codec_id );
276 }
277 codec->codec_id = i_codec_id;
278
279 if( fmt->i_extra )
280 {
281 if( fmt->i_codec == VLC_CODEC_OPUS )
282 {
283 codec->extradata_size = opus_size[0];
284 codec->extradata = av_malloc( opus_size[0] );
285 memcpy( codec->extradata, opus_packet[0], opus_size[0] );
286 }
287 else
288 {
289 codec->extradata_size = fmt->i_extra;
290 codec->extradata = av_malloc( fmt->i_extra );
291 memcpy( codec->extradata, fmt->p_extra, fmt->i_extra );
292 }
293 }
294
295 return VLC_SUCCESS;
296}

308static int MuxBlock( sout_mux_t *p_mux, sout_input_t *p_input )
309{
310 sout_mux_sys_t *p_sys = p_mux->p_sys;
311 block_t *p_data = block_FifoGet( p_input->p_fifo );
312 int i_stream = *((int *)p_input->p_sys);
313 AVStream *p_stream = p_sys->oc->streams[i_stream];
314 AVPacket pkt;
315
316 memset( &pkt, 0, sizeof(AVPacket) );
317
318 av_init_packet(&pkt);
319 pkt.data = p_data->p_buffer;
320 pkt.size = p_data->i_buffer;
321 pkt.stream_index = i_stream;
322
323 if( p_data->i_flags & BLOCK_FLAG_TYPE_I )
324 {
325#ifdef AVFMT_ALLOW_FLUSH
326 /* Make sure we don't inadvertedly mark buffered data as keyframes. */
327 if( p_sys->oc->oformat->flags & AVFMT_ALLOW_FLUSH )
328 av_write_frame( p_sys->oc, NULL );
329#endif
330
331 p_sys->b_write_keyframe = true;
332 pkt.flags |= AV_PKT_FLAG_KEY;
333 }
334
335 if( p_data->i_pts > 0 )
336 pkt.pts = p_data->i_pts * p_stream->time_base.den /
337 CLOCK_FREQ / p_stream->time_base.num;
338 if( p_data->i_dts > 0 )
339 pkt.dts = p_data->i_dts * p_stream->time_base.den /
340 CLOCK_FREQ / p_stream->time_base.num;
341
342 /* this is another hack to prevent libavformat from triggering the "non monotone timestamps" check in avformat/utils.c */
343 p_stream->cur_dts = ( p_data->i_dts * p_stream->time_base.den /
344 CLOCK_FREQ / p_stream->time_base.num ) - 1;
345
346 if( av_write_frame( p_sys->oc, &pkt ) < 0 )
347 {
348 msg_Err( p_mux, "could not write frame (pts: %"PRId64", dts: %"PRId64") "
349 "(pkt pts: %"PRId64", dts: %"PRId64")",
350 p_data->i_pts, p_data->i_dts, pkt.pts, pkt.dts );
351 block_Release( p_data );
352 return VLC_EGENERIC;
353 }
354
355 block_Release( p_data );
356 return VLC_SUCCESS;
357}

359/*****************************************************************************
360 * Mux: multiplex available data in input fifos
361 *****************************************************************************/
362static int Mux( sout_mux_t *p_mux )
363{
364 sout_mux_sys_t *p_sys = p_mux->p_sys;
365
366 if( p_sys->b_error ) return VLC_EGENERIC;
367
368 if( p_sys->b_write_header )
369 {
370 int error;
371 msg_Dbg( p_mux, "writing header" );
372
373 char *psz_opts = var_GetNonEmptyString( p_mux, "sout-avformat-options" );
374 AVDictionary *options = NULL;
375 if (psz_opts && *psz_opts)
376 options = vlc_av_get_options(psz_opts);
377 free(psz_opts);
378 error = avformat_write_header( p_sys->oc, options ? &options : NULL);
379 AVDictionaryEntry *t = NULL;
380 while ((t = av_dict_get(options, "", t, AV_DICT_IGNORE_SUFFIX))) {
381 msg_Err( p_mux, "Unknown option \"%s\"", t->key );
382 }
383 av_dict_free(&options);
384 if( error < 0 )
385 {
386 msg_Err( p_mux, "could not write header: %s",
387 vlc_strerror_c(AVUNERROR(error)) );
388 p_sys->b_write_header = false;
389 p_sys->b_error = true;
390 return VLC_EGENERIC;
391 }
392
393 avio_flush( p_sys->oc->pb );
394 p_sys->b_write_header = false;
395 }
396
397 for( ;; )
398 {
399 mtime_t i_dts;
400
401 int i_stream = sout_MuxGetStream( p_mux, 1, &i_dts );
402 if( i_stream < 0 )
403 return VLC_SUCCESS;
404
405 MuxBlock( p_mux, p_mux->pp_inputs[i_stream] );
406 }
407
408 return VLC_SUCCESS;
409}

411/*****************************************************************************
412 * Control:
413 *****************************************************************************/
414static int Control( sout_mux_t *p_mux, int i_query, va_list args )
415{
416 bool *pb_bool;
417
418 switch( i_query )
419 {
420 case MUX_CAN_ADD_STREAM_WHILE_MUXING:
421 pb_bool = (bool*)va_arg( args, bool * );
422 *pb_bool = false;
423 return VLC_SUCCESS;
424
425 case MUX_GET_ADD_STREAM_WAIT:
426 pb_bool = (bool*)va_arg( args, bool * );
427 *pb_bool = true;
428 return VLC_SUCCESS;
429
430 case MUX_GET_MIME:
431 {
432 char **ppsz = (char**)va_arg( args, char ** );
433 *ppsz = strdup( p_mux->p_sys->oc->oformat->mime_type );
434 return VLC_SUCCESS;
435 }
436
437 default:
438 return VLC_EGENERIC;
439 }
440}

442/*****************************************************************************
443 * I/O wrappers for libavformat
444 *****************************************************************************/
445static int IOWrite( void *opaque, uint8_t *buf, int buf_size )
446{
447 sout_mux_t *p_mux = opaque;
448 sout_mux_sys_t *p_sys = p_mux->p_sys;
449 int i_ret;
450
451#ifdef AVFORMAT_DEBUG
452 msg_Dbg( p_mux, "IOWrite %i bytes", buf_size );
453#endif
454
455 block_t *p_buf = block_Alloc( buf_size );
456 if( buf_size > 0 ) memcpy( p_buf->p_buffer, buf, buf_size );
457
458 if( p_sys->b_write_header )
459 p_buf->i_flags |= BLOCK_FLAG_HEADER;
460
461 if( p_sys->b_write_keyframe )
462 {
463 p_buf->i_flags |= BLOCK_FLAG_TYPE_I;
464 p_sys->b_write_keyframe = false;
465 }
466
467 i_ret = sout_AccessOutWrite( p_mux->p_access, p_buf );
468 return i_ret ? i_ret : -1;
469}

471static int64_t IOSeek( void *opaque, int64_t offset, int whence )
472{
473 sout_mux_t *p_mux = opaque;
474
475#ifdef AVFORMAT_DEBUG
476 msg_Dbg( p_mux, "IOSeek offset: %"PRId64", whence: %i", offset, whence );
477#endif
478
479 switch( whence )
480 {
481 case SEEK_SET:
482 return sout_AccessOutSeek( p_mux->p_access, offset );
483 case SEEK_CUR:
484 case SEEK_END:
485 default:
486 return -1;
487 }
