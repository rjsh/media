25#ifndef MERGE_FFMPEG
26#ifdef HAVE_CONFIG_H
27# include "config.h"
28#endif
29
30#include <vlc_common.h>
31#include <vlc_plugin.h>
32
33#include "avformat.h"
34#include "../../codec/avcodec/avcommon.h"
35
36vlc_module_begin ()
37#endif /* MERGE_FFMPEG */
38 add_shortcut( "ffmpeg", "avformat" )
39 set_category( CAT_INPUT )
40 set_subcategory( SUBCAT_INPUT_DEMUX )
41 set_description( N_("Avformat demuxer" ) )
42 set_shortname( N_("Avformat") )
43 set_capability( "demux", 2 )
44 set_callbacks( OpenDemux, CloseDemux )
45 set_section( N_("Demuxer"), NULL )
46 add_string( "avformat-format", NULL, FORMAT_TEXT, FORMAT_LONGTEXT, true )
47 add_obsolete_string("ffmpeg-format") /* removed since 2.1.0 */
48#if LIBAVFORMAT_VERSION_INT >= ((53<<16)+(26<<8)+0)
49 add_string( "avformat-options", NULL, AV_OPTIONS_TEXT, AV_OPTIONS_LONGTEXT, true )
50#endif
51
52#ifdef ENABLE_SOUT
53 /* mux submodule */
54 add_submodule ()
55 add_shortcut( "ffmpeg", "avformat" )
56 set_description( N_("Avformat muxer" ) )
57 set_capability( "sout mux", 2 )
58 set_section( N_("Muxer"), NULL )
59 add_string( "sout-avformat-mux", NULL, MUX_TEXT, MUX_LONGTEXT, true )
60 add_obsolete_string("ffmpeg-mux") /* removed since 2.1.0 */
61 add_string( "sout-avformat-options", NULL, AV_OPTIONS_TEXT, AV_OPTIONS_LONGTEXT, true )
62 set_callbacks( OpenMux, CloseMux )
63#endif
64#ifndef MERGE_FFMPEG
65vlc_module_end ()
66#endif
