40static vlc_fourcc_t GetFOURCC( const uint8_t *p_buff )
41{
42 return VLC_FOURCC( p_buff[0], p_buff[1], p_buff[2], p_buff[3] );
43}

45/****************************************************************************
46 *
47 * Basics functions to manipulates chunks
48 *
49 ****************************************************************************/
50static int AVI_ChunkReadCommon( stream_t *s, avi_chunk_t *p_chk )
51{
52 const uint8_t *p_peek;
53
54 memset( p_chk, 0, sizeof( avi_chunk_t ) );
55
56 if( stream_Peek( s, &p_peek, 8 ) < 8 )
57 return VLC_EGENERIC;
58
59 p_chk->common.i_chunk_fourcc = GetFOURCC( p_peek );
60 p_chk->common.i_chunk_size = GetDWLE( p_peek + 4 );
61 p_chk->common.i_chunk_pos = stream_Tell( s );
62
63 p_chk->common.p_father = NULL;
64 p_chk->common.p_next = NULL;
65 p_chk->common.p_first = NULL;
66 p_chk->common.p_next = NULL;
67
68#ifdef AVI_DEBUG
69 msg_Dbg( (vlc_object_t*)s,
70 "found chunk, fourcc: %4.4s size:%"PRId64" pos:%"PRId64,
71 (char*)&p_chk->common.i_chunk_fourcc,
72 p_chk->common.i_chunk_size,
73 p_chk->common.i_chunk_pos );
74#endif
75 return VLC_SUCCESS;
76}

78static int AVI_NextChunk( stream_t *s, avi_chunk_t *p_chk )
79{
80 avi_chunk_t chk;
81
82 if( !p_chk )
83 {
84 if( AVI_ChunkReadCommon( s, &chk ) )
85 {
86 return VLC_EGENERIC;
87 }
88 p_chk = &chk;
89 }
90
91 if( p_chk->common.p_father )
92 {
93 if( p_chk->common.p_father->common.i_chunk_pos +
94 __EVEN( p_chk->common.p_father->common.i_chunk_size ) + 8 <
95 p_chk->common.i_chunk_pos +
96 __EVEN( p_chk->common.i_chunk_size ) + 8 )
97 {
98 return VLC_EGENERIC;
99 }
100 }
101 return stream_Seek( s, p_chk->common.i_chunk_pos +
102 __EVEN( p_chk->common.i_chunk_size ) + 8 );
103}

110static int AVI_ChunkRead_list( stream_t *s, avi_chunk_t *p_container )
111{
112 avi_chunk_t *p_chk;
113 const uint8_t *p_peek;
114 bool b_seekable;
115 int i_ret = VLC_SUCCESS;
116
117 if( p_container->common.i_chunk_size > 0 && p_container->common.i_chunk_size < 4 )
118 {
119 /* empty box */
120 msg_Warn( (vlc_object_t*)s, "empty list chunk" );
121 return VLC_EGENERIC;
122 }
123 if( stream_Peek( s, &p_peek, 12 ) < 12 )
124 {
125 msg_Warn( (vlc_object_t*)s, "cannot peek while reading list chunk" );
126 return VLC_EGENERIC;
127 }
128
129 stream_Control( s, STREAM_CAN_FASTSEEK, &b_seekable );
130
131 p_container->list.i_type = GetFOURCC( p_peek + 8 );
132
133 /* XXX fixed for on2 hack */
134 if( p_container->common.i_chunk_fourcc == AVIFOURCC_ON2 && p_container->list.i_type == AVIFOURCC_ON2f )
135 {
136 p_container->common.i_chunk_fourcc = AVIFOURCC_RIFF;
137 p_container->list.i_type = AVIFOURCC_AVI;
138 }
139
140 if( p_container->common.i_chunk_fourcc == AVIFOURCC_LIST &&
141 p_container->list.i_type == AVIFOURCC_movi )
142 {
143 msg_Dbg( (vlc_object_t*)s, "skipping movi chunk" );
144 if( b_seekable )
145 {
146 return AVI_NextChunk( s, p_container );
147 }
148 return VLC_SUCCESS; /* point at begining of LIST-movi */
149 }
150
151 if( stream_Read( s, NULL, 12 ) != 12 )
152 {
153 msg_Warn( (vlc_object_t*)s, "cannot enter chunk" );
154 return VLC_EGENERIC;
155 }
156
157#ifdef AVI_DEBUG
158 msg_Dbg( (vlc_object_t*)s,
159 "found LIST chunk: \'%4.4s\'",
160 (char*)&p_container->list.i_type );
161#endif
162 msg_Dbg( (vlc_object_t*)s, "<list \'%4.4s\'>", (char*)&p_container->list.i_type );
163 for( ; ; )
164 {
165 p_chk = xmalloc( sizeof( avi_chunk_t ) );
166 memset( p_chk, 0, sizeof( avi_chunk_t ) );
167 if( !p_container->common.p_first )
168 {
169 p_container->common.p_first = p_chk;
170 }
171 else
172 {
173 p_container->common.p_last->common.p_next = p_chk;
174 }
175 p_container->common.p_last = p_chk;
176
177 i_ret = AVI_ChunkRead( s, p_chk, p_container );
178 if( i_ret )
179 {
180 break;
181 }
182 if( p_chk->common.p_father->common.i_chunk_size > 0 &&
183 ( stream_Tell( s ) >
184 (off_t)p_chk->common.p_father->common.i_chunk_pos +
185 (off_t)__EVEN( p_chk->common.p_father->common.i_chunk_size ) ) )
186 {
187 break;
188 }
189
190 /* If we can't seek then stop when we 've found LIST-movi */
191 if( p_chk->common.i_chunk_fourcc == AVIFOURCC_LIST &&
192 p_chk->list.i_type == AVIFOURCC_movi &&
193 ( !b_seekable || p_chk->common.i_chunk_size == 0 ) )
194 {
195 break;
196 }
197
198 }
199 msg_Dbg( (vlc_object_t*)s, "</list \'%4.4s\'>", (char*)&p_container->list.i_type );
200
201 if ( i_ret == AVI_ZERO_FOURCC ) return i_ret;
202 return VLC_SUCCESS;
203}

205/* Allow to append indexes after starting playback */
206int AVI_ChunkFetchIndexes( stream_t *s, avi_chunk_t *p_riff )
207{
208 avi_chunk_t *p_movi = AVI_ChunkFind( p_riff, AVIFOURCC_movi, 0 );
209 if ( !p_movi )
210 return VLC_EGENERIC;
211
212 avi_chunk_t *p_chk;
213 uint64_t i_indexpos = 8 + p_movi->common.i_chunk_pos + p_movi->common.i_chunk_size;
214 bool b_seekable = false;
215 int i_ret = VLC_SUCCESS;
216
217 stream_Control( s, STREAM_CAN_SEEK, &b_seekable );
218 if ( !b_seekable || stream_Seek( s, i_indexpos ) )
219 return VLC_EGENERIC;
220
221 for( ; ; )
222 {
223 p_chk = xmalloc( sizeof( avi_chunk_t ) );
224 memset( p_chk, 0, sizeof( avi_chunk_t ) );
225 if (unlikely( !p_riff->common.p_first ))
226 p_riff->common.p_first = p_chk;
227 else
228 p_riff->common.p_last->common.p_next = p_chk;
229 p_riff->common.p_last = p_chk;
230
231 i_ret = AVI_ChunkRead( s, p_chk, p_riff );
232 if( i_ret )
233 break;
234
235 if( p_chk->common.p_father->common.i_chunk_size > 0 &&
236 ( stream_Tell( s ) >
237 (off_t)p_chk->common.p_father->common.i_chunk_pos +
238 (off_t)__EVEN( p_chk->common.p_father->common.i_chunk_size ) ) )
239 {
240 break;
241 }
242
243 /* If we can't seek then stop when we 've found any index */
244 if( p_chk->common.i_chunk_fourcc == AVIFOURCC_indx ||
245 p_chk->common.i_chunk_fourcc == AVIFOURCC_idx1 )
246 {
247 break;
248 }
249
250 }
251
252 return i_ret;
253}

255#define AVI_READCHUNK_ENTER \
256 int64_t i_read = __EVEN(p_chk->common.i_chunk_size ) + 8; \
257 if( i_read > 100000000 ) \
258 { \
259 msg_Err( s, "Big chunk ignored" ); \
260 return VLC_EGENERIC; \
261 } \
262 uint8_t *p_read, *p_buff; \
263 if( !( p_read = p_buff = malloc(i_read ) ) ) \
264 { \
265 return VLC_EGENERIC; \
266 } \
267 i_read = stream_Read( s, p_read, i_read ); \
268 if( i_read < (int64_t)__EVEN(p_chk->common.i_chunk_size ) + 8 ) \
269 { \
270 free( p_buff ); \
271 return VLC_EGENERIC; \
272 }\
273 p_read += 8; \
274 i_read -= 8
275
276#define AVI_READ( res, func, size ) \
277 if( i_read < size ) { \
278 free( p_buff); \
279 return VLC_EGENERIC; \
280 } \
281 i_read -= size; \
282 res = func( p_read ); \
283 p_read += size \
284
285#define AVI_READCHUNK_EXIT( code ) \
286 free( p_buff ); \
287 return code

309static int AVI_ChunkRead_avih( stream_t *s, avi_chunk_t *p_chk )
310{
311 AVI_READCHUNK_ENTER;
312
313 p_chk->common.i_chunk_fourcc = AVIFOURCC_avih;
314 AVI_READ4BYTES( p_chk->avih.i_microsecperframe);
315 AVI_READ4BYTES( p_chk->avih.i_maxbytespersec );
316 AVI_READ4BYTES( p_chk->avih.i_reserved1 );
317 AVI_READ4BYTES( p_chk->avih.i_flags );
318 AVI_READ4BYTES( p_chk->avih.i_totalframes );
319 AVI_READ4BYTES( p_chk->avih.i_initialframes );
320 AVI_READ4BYTES( p_chk->avih.i_streams );
321 AVI_READ4BYTES( p_chk->avih.i_suggestedbuffersize );
322 AVI_READ4BYTES( p_chk->avih.i_width );
323 AVI_READ4BYTES( p_chk->avih.i_height );
324 AVI_READ4BYTES( p_chk->avih.i_scale );
325 AVI_READ4BYTES( p_chk->avih.i_rate );
326 AVI_READ4BYTES( p_chk->avih.i_start );
327 AVI_READ4BYTES( p_chk->avih.i_length );
328#ifdef AVI_DEBUG
329 msg_Dbg( (vlc_object_t*)s,
330 "avih: streams:%d flags:%s%s%s%s %dx%d",
331 p_chk->avih.i_streams,
332 p_chk->avih.i_flags&AVIF_HASINDEX?" HAS_INDEX":"",
333 p_chk->avih.i_flags&AVIF_MUSTUSEINDEX?" MUST_USE_INDEX":"",
334 p_chk->avih.i_flags&AVIF_ISINTERLEAVED?" IS_INTERLEAVED":"",
335 p_chk->avih.i_flags&AVIF_TRUSTCKTYPE?" TRUST_CKTYPE":"",
336 p_chk->avih.i_width, p_chk->avih.i_height );
337#endif
338 AVI_READCHUNK_EXIT( VLC_SUCCESS );
339}

341static int AVI_ChunkRead_strh( stream_t *s, avi_chunk_t *p_chk )
342{
343 AVI_READCHUNK_ENTER;
344
345 AVI_READFOURCC( p_chk->strh.i_type );
346 AVI_READFOURCC( p_chk->strh.i_handler );
347 AVI_READ4BYTES( p_chk->strh.i_flags );
348 AVI_READ4BYTES( p_chk->strh.i_reserved1 );
349 AVI_READ4BYTES( p_chk->strh.i_initialframes );
350 AVI_READ4BYTES( p_chk->strh.i_scale );
351 AVI_READ4BYTES( p_chk->strh.i_rate );
352 AVI_READ4BYTES( p_chk->strh.i_start );
353 AVI_READ4BYTES( p_chk->strh.i_length );
354 AVI_READ4BYTES( p_chk->strh.i_suggestedbuffersize );
355 AVI_READ4BYTES( p_chk->strh.i_quality );
356 AVI_READ4BYTES( p_chk->strh.i_samplesize );
357#ifdef AVI_DEBUG
358 msg_Dbg( (vlc_object_t*)s,
359 "strh: type:%4.4s handler:0x%8.8x samplesize:%d %.2ffps",
360 (char*)&p_chk->strh.i_type,
361 p_chk->strh.i_handler,
362 p_chk->strh.i_samplesize,
363 ( p_chk->strh.i_scale ?
364 (float)p_chk->strh.i_rate / (float)p_chk->strh.i_scale : -1) );
365#endif
366
367 AVI_READCHUNK_EXIT( VLC_SUCCESS );
368}

370static int AVI_ChunkRead_strf( stream_t *s, avi_chunk_t *p_chk )
371{
372 avi_chunk_t *p_strh;
373
374 AVI_READCHUNK_ENTER;
375 if( p_chk->common.p_father == NULL )
376 {
377 msg_Err( (vlc_object_t*)s, "malformed avi file" );
378 AVI_READCHUNK_EXIT( VLC_EGENERIC );
379 }
380 if( !( p_strh = AVI_ChunkFind( p_chk->common.p_father, AVIFOURCC_strh, 0 ) ) )
381 {
382 msg_Err( (vlc_object_t*)s, "malformed avi file" );
383 AVI_READCHUNK_EXIT( VLC_EGENERIC );
384 }
385
386 switch( p_strh->strh.i_type )
387 {
388 case( AVIFOURCC_auds ):
389 p_chk->strf.auds.i_cat = AUDIO_ES;
390 p_chk->strf.auds.p_wf = xmalloc( __MAX( p_chk->common.i_chunk_size, sizeof( WAVEFORMATEX ) ) );
391 if ( !p_chk->strf.auds.p_wf )
392 {
393 AVI_READCHUNK_EXIT( VLC_ENOMEM );
394 }
395 AVI_READ2BYTES( p_chk->strf.auds.p_wf->wFormatTag );
396 AVI_READ2BYTES( p_chk->strf.auds.p_wf->nChannels );
397 AVI_READ4BYTES( p_chk->strf.auds.p_wf->nSamplesPerSec );
398 AVI_READ4BYTES( p_chk->strf.auds.p_wf->nAvgBytesPerSec );
399 AVI_READ2BYTES( p_chk->strf.auds.p_wf->nBlockAlign );
400 AVI_READ2BYTES( p_chk->strf.auds.p_wf->wBitsPerSample );
401
402 if( p_chk->strf.auds.p_wf->wFormatTag != WAVE_FORMAT_PCM
403 && p_chk->common.i_chunk_size > sizeof( WAVEFORMATEX ) )
404 {
405 AVI_READ2BYTES( p_chk->strf.auds.p_wf->cbSize );
406
407 /* prevent segfault */
408 if( p_chk->strf.auds.p_wf->cbSize >
409 p_chk->common.i_chunk_size - sizeof( WAVEFORMATEX ) )
410 {
411 p_chk->strf.auds.p_wf->cbSize =
412 p_chk->common.i_chunk_size - sizeof( WAVEFORMATEX );
413 }
414
415 if( p_chk->strf.auds.p_wf->wFormatTag == WAVE_FORMAT_EXTENSIBLE )
416 {
417 msg_Dbg( s, "Extended header found" );
418 }
419 }
420 else
421 {
422 p_chk->strf.auds.p_wf->cbSize = 0;
423 }
424 if( p_chk->strf.auds.p_wf->cbSize > 0 )
425 {
426 memcpy( &p_chk->strf.auds.p_wf[1] ,
427 p_buff + 8 + sizeof( WAVEFORMATEX ), /* 8=fourcc+size */
428 p_chk->strf.auds.p_wf->cbSize );
429 }
430#ifdef AVI_DEBUG
431 msg_Dbg( (vlc_object_t*)s,
432 "strf: audio:0x%4.4x channels:%d %dHz %dbits/sample %dkb/s",
433 p_chk->strf.auds.p_wf->wFormatTag,
434 p_chk->strf.auds.p_wf->nChannels,
435 p_chk->strf.auds.p_wf->nSamplesPerSec,
436 p_chk->strf.auds.p_wf->wBitsPerSample,
437 p_chk->strf.auds.p_wf->nAvgBytesPerSec * 8 / 1024 );
438#endif
439 break;
440 case( AVIFOURCC_vids ):
441 p_strh->strh.i_samplesize = 0; /* XXX for ffmpeg avi file */
442 p_chk->strf.vids.i_cat = VIDEO_ES;
443 p_chk->strf.vids.p_bih = xmalloc( __MAX( p_chk->common.i_chunk_size,
444 sizeof( *p_chk->strf.vids.p_bih ) ) );
445 if ( !p_chk->strf.vids.p_bih )
446 {
447 AVI_READCHUNK_EXIT( VLC_ENOMEM );
448 }
449 AVI_READ4BYTES( p_chk->strf.vids.p_bih->biSize );
450 AVI_READ4BYTES( p_chk->strf.vids.p_bih->biWidth );
451 AVI_READ4BYTES( p_chk->strf.vids.p_bih->biHeight );
452 AVI_READ2BYTES( p_chk->strf.vids.p_bih->biPlanes );
453 AVI_READ2BYTES( p_chk->strf.vids.p_bih->biBitCount );
454 AVI_READFOURCC( p_chk->strf.vids.p_bih->biCompression );
455 AVI_READ4BYTES( p_chk->strf.vids.p_bih->biSizeImage );
456 AVI_READ4BYTES( p_chk->strf.vids.p_bih->biXPelsPerMeter );
457 AVI_READ4BYTES( p_chk->strf.vids.p_bih->biYPelsPerMeter );
458 AVI_READ4BYTES( p_chk->strf.vids.p_bih->biClrUsed );
459 AVI_READ4BYTES( p_chk->strf.vids.p_bih->biClrImportant );
460 if( p_chk->strf.vids.p_bih->biSize > p_chk->common.i_chunk_size )
461 {
462 p_chk->strf.vids.p_bih->biSize = p_chk->common.i_chunk_size;
463 }
464 uint64_t i_extrasize = p_chk->common.i_chunk_size - sizeof(VLC_BITMAPINFOHEADER);
465 if( i_extrasize > 0 )
466 {
467 /* There's a color palette appended, set up VLC_BITMAPINFO */
468 memcpy( &p_chk->strf.vids.p_bih[1],
469 p_buff + 8 + sizeof(VLC_BITMAPINFOHEADER), /* 8=fourrc+size */
470 i_extrasize );
471
472 if ( !p_chk->strf.vids.p_bih->biClrUsed )
473 p_chk->strf.vids.p_bih->biClrUsed = (1 << p_chk->strf.vids.p_bih->biBitCount);
474
475 if( i_extrasize > (UINT32_MAX * sizeof(uint32_t)) )
476 p_chk->strf.vids.p_bih->biClrUsed = UINT32_MAX;
477 else
478 {
479 p_chk->strf.vids.p_bih->biClrUsed =
480 __MAX( i_extrasize / sizeof(uint32_t),
481 p_chk->strf.vids.p_bih->biClrUsed );
482 }
483
484 /* stay within VLC's limits */
485 p_chk->strf.vids.p_bih->biClrUsed =
486 __MIN( VIDEO_PALETTE_COLORS_MAX, p_chk->strf.vids.p_bih->biClrUsed );
487 }
488 else p_chk->strf.vids.p_bih->biClrUsed = 0;
489#ifdef AVI_DEBUG
490 msg_Dbg( (vlc_object_t*)s,
491 "strf: video:%4.4s %"PRIu32"x%"PRIu32" planes:%d %dbpp",
492 (char*)&p_chk->strf.vids.p_bih->biCompression,
493 (uint32_t)p_chk->strf.vids.p_bih->biWidth,
494 (uint32_t)p_chk->strf.vids.p_bih->biHeight,
495 p_chk->strf.vids.p_bih->biPlanes,
496 p_chk->strf.vids.p_bih->biBitCount );
497#endif
498 break;
499 case AVIFOURCC_iavs:
500 case AVIFOURCC_ivas:
501 p_chk->strf.common.i_cat = UNKNOWN_ES;
502 break;
503 case( AVIFOURCC_txts ):
504 p_chk->strf.common.i_cat = SPU_ES;
505 break;
506 default:
507 msg_Warn( (vlc_object_t*)s, "unknown stream type: %4.4s",
508 (char*)&p_strh->strh.i_type );
509 p_chk->strf.common.i_cat = UNKNOWN_ES;
510 break;
511 }
512 AVI_READCHUNK_EXIT( VLC_SUCCESS );
513}
514static void AVI_ChunkFree_strf( avi_chunk_t *p_chk )
515{
516 avi_chunk_strf_t *p_strf = (avi_chunk_strf_t*)p_chk;
517 if( p_strf->common.i_cat == AUDIO_ES )
518 {
519 FREENULL( p_strf->auds.p_wf );
520 }
521 else if( p_strf->common.i_cat == VIDEO_ES )
522 {
523 FREENULL( p_strf->vids.p_bih );
524 }
525}

538 AVI_READCHUNK_EXIT( VLC_SUCCESS );
539}

546static int AVI_ChunkRead_idx1( stream_t *s, avi_chunk_t *p_chk )
547{
548 unsigned int i_count, i_index;
549
550 AVI_READCHUNK_ENTER;
551

564 AVI_READ4BYTES( p_chk->idx1.entry[i_index].i_pos );
565 AVI_READ4BYTES( p_chk->idx1.entry[i_index].i_length );
566 }
567 }
568 else
569 {
570 p_chk->idx1.entry = NULL;
571 }
572#ifdef AVI_DEBUG
573 msg_Dbg( (vlc_object_t*)s, "idx1: index entry:%d", i_count );
574#endif
575 AVI_READCHUNK_EXIT( VLC_SUCCESS );

587static int AVI_ChunkRead_indx( stream_t *s, avi_chunk_t *p_chk )
588{
589 unsigned int i_count, i;
590 int32_t i_dummy;
591 avi_chunk_indx_t *p_indx = (avi_chunk_indx_t*)p_chk;
592
593 AVI_READCHUNK_ENTER;
594
595 AVI_READ2BYTES( p_indx->i_longsperentry );
596 AVI_READ1BYTE ( p_indx->i_indexsubtype );
597 AVI_READ1BYTE ( p_indx->i_indextype );
598 AVI_READ4BYTES( p_indx->i_entriesinuse );
599
600 AVI_READ4BYTES( p_indx->i_id );
601 p_indx->idx.std = NULL;
602 p_indx->idx.field = NULL;
603 p_indx->idx.super = NULL;
604
605 if( p_indx->i_indextype == AVI_INDEX_OF_CHUNKS && p_indx->i_indexsubtype == 0 )
606 {
607 AVI_READ8BYTES( p_indx->i_baseoffset );
608 AVI_READ4BYTES( i_dummy );
609
610 i_count = __MIN( p_indx->i_entriesinuse, i_read / 8 );
611 p_indx->i_entriesinuse = i_count;
612 p_indx->idx.std = xcalloc( i_count, sizeof( indx_std_entry_t ) );
613
614 for( i = 0; i < i_count; i++ )
615 {
616 AVI_READ4BYTES( p_indx->idx.std[i].i_offset );
617 AVI_READ4BYTES( p_indx->idx.std[i].i_size );
618 }
619 }
620 else if( p_indx->i_indextype == AVI_INDEX_OF_CHUNKS && p_indx->i_indexsubtype == AVI_INDEX_2FIELD )
621 {
622 AVI_READ8BYTES( p_indx->i_baseoffset );
623 AVI_READ4BYTES( i_dummy );
624
625 i_count = __MIN( p_indx->i_entriesinuse, i_read / 12 );
626 p_indx->i_entriesinuse = i_count;
627 p_indx->idx.field = xcalloc( i_count, sizeof( indx_field_entry_t ) );
628 for( i = 0; i < i_count; i++ )
629 {
630 AVI_READ4BYTES( p_indx->idx.field[i].i_offset );
631 AVI_READ4BYTES( p_indx->idx.field[i].i_size );
632 AVI_READ4BYTES( p_indx->idx.field[i].i_offsetfield2 );
633 }
634 }
635 else if( p_indx->i_indextype == AVI_INDEX_OF_INDEXES )
636 {
637 p_indx->i_baseoffset = 0;
638 AVI_READ4BYTES( i_dummy );
639 AVI_READ4BYTES( i_dummy );
640 AVI_READ4BYTES( i_dummy );
641
642 i_count = __MIN( p_indx->i_entriesinuse, i_read / 16 );
643 p_indx->i_entriesinuse = i_count;
644 p_indx->idx.super = xcalloc( i_count, sizeof( indx_super_entry_t ) );
645
646 for( i = 0; i < i_count; i++ )
647 {
648 AVI_READ8BYTES( p_indx->idx.super[i].i_offset );
649 AVI_READ4BYTES( p_indx->idx.super[i].i_size );
650 AVI_READ4BYTES( p_indx->idx.super[i].i_duration );
651 }
652 }
653 else
654 {
655 msg_Warn( (vlc_object_t*)s, "unknow type/subtype index" );
656 }
657
658#ifdef AVI_DEBUG
659 msg_Dbg( (vlc_object_t*)s, "indx: type=%d subtype=%d entry=%d", p_indx->i_indextype, p_indx->i_indexsubtype, p_indx->i_entriesinuse );
660#endif
661 AVI_READCHUNK_EXIT( VLC_SUCCESS );
662}

894
895int AVI_ChunkRead( stream_t *s, avi_chunk_t *p_chk, avi_chunk_t *p_father )
896{
897 int i_index;
898
899 if( !p_chk )
900 {
901 msg_Warn( (vlc_object_t*)s, "cannot read null chunk" );
902 return VLC_EGENERIC;
903 }
904
905 if( AVI_ChunkReadCommon( s, p_chk ) )
906 {
907 msg_Warn( (vlc_object_t*)s, "cannot read one chunk" );
908 return VLC_EGENERIC;
909 }
910
911 if( p_chk->common.i_chunk_fourcc == VLC_FOURCC( 0, 0, 0, 0 ) )
912 {
913 msg_Warn( (vlc_object_t*)s, "found null fourcc chunk (corrupted file?)" );
914 return AVI_ZERO_FOURCC;
915 }
916 p_chk->common.p_father = p_father;
917
918 i_index = AVI_ChunkFunctionFind( p_chk->common.i_chunk_fourcc );
919 if( AVI_Chunk_Function[i_index].AVI_ChunkRead_function )
920 {
921 int i_return = AVI_Chunk_Function[i_index].AVI_ChunkRead_function( s, p_chk );
922 if ( i_return == AVI_STRD_ZERO_CHUNK || i_return == AVI_ZERO_FOURCC )
923 {
924 if ( !p_father ) return VLC_EGENERIC;
925 return AVI_NextChunk( s, p_father );
926 }
927 return i_return;
928 }
929 else if( ( ((char*)&p_chk->common.i_chunk_fourcc)[0] == 'i' &&
930 ((char*)&p_chk->common.i_chunk_fourcc)[1] == 'x' ) ||
931 ( ((char*)&p_chk->common.i_chunk_fourcc)[2] == 'i' &&
932 ((char*)&p_chk->common.i_chunk_fourcc)[3] == 'x' ) )
933 {
934 p_chk->common.i_chunk_fourcc = AVIFOURCC_indx;
935 return AVI_ChunkRead_indx( s, p_chk );
936 }
937
938 msg_Warn( (vlc_object_t*)s, "unknown chunk: %4.4s (not loaded)",
939 (char*)&p_chk->common.i_chunk_fourcc );
940 return AVI_NextChunk( s, p_chk );
941}
942
943void AVI_ChunkFree( stream_t *s,
944 avi_chunk_t *p_chk )
945{
946 int i_index;
947 avi_chunk_t *p_child, *p_next;
948
949 if( !p_chk )
950 {
951 return;
952 }
953
954 /* Free all child chunk */
955 p_child = p_chk->common.p_first;
956 while( p_child )
957 {
958 p_next = p_child->common.p_next;
959 AVI_ChunkFree( s, p_child );
960 free( p_child );
961 p_child = p_next;
962 }
963
964 i_index = AVI_ChunkFunctionFind( p_chk->common.i_chunk_fourcc );
965 if( AVI_Chunk_Function[i_index].AVI_ChunkFree_function )
966 {
967#ifdef AVI_DEBUG
968 msg_Dbg( (vlc_object_t*)s, "free chunk %4.4s",
969 (char*)&p_chk->common.i_chunk_fourcc );
970#endif
971 AVI_Chunk_Function[i_index].AVI_ChunkFree_function( p_chk);
972 }
973 else
974 {
975 msg_Warn( (vlc_object_t*)s, "unknown chunk: %4.4s (not unloaded)",
976 (char*)&p_chk->common.i_chunk_fourcc );
977 }
978 p_chk->common.p_first = NULL;
979 p_chk->common.p_last = NULL;
980
981 return;
982}

984static void AVI_ChunkDumpDebug_level( vlc_object_t *p_obj,
985 avi_chunk_t *p_chk, unsigned i_level )
986{
987 avi_chunk_t *p_child;
988
989 char str[512];
990 if( i_level >= (sizeof(str) - 1)/4 )
991 return;
992
993 memset( str, ' ', sizeof( str ) );
994 for( unsigned i = 1; i < i_level; i++ )
995 {
996 str[i * 4] = '|';
997 }
998 if( p_chk->common.i_chunk_fourcc == AVIFOURCC_RIFF ||
999 p_chk->common.i_chunk_fourcc == AVIFOURCC_ON2 ||
1000 p_chk->common.i_chunk_fourcc == AVIFOURCC_LIST )
1001 {
1002 snprintf( &str[i_level * 4], sizeof(str) - 4*i_level,
1003 "%c %4.4s-%4.4s size:%"PRIu64" pos:%"PRIu64,
1004 i_level ? '+' : '*',
1005 (char*)&p_chk->common.i_chunk_fourcc,
1006 (char*)&p_chk->list.i_type,
1007 p_chk->common.i_chunk_size,
1008 p_chk->common.i_chunk_pos );
1009 }
1010 else
1011 {
1012 snprintf( &str[i_level * 4], sizeof(str) - 4*i_level,
1013 "+ %4.4s size:%"PRIu64" pos:%"PRIu64,
1014 (char*)&p_chk->common.i_chunk_fourcc,
1015 p_chk->common.i_chunk_size,
1016 p_chk->common.i_chunk_pos );
1017 }
1018 msg_Dbg( p_obj, "%s", str );
1019
1020 p_child = p_chk->common.p_first;
1021 while( p_child )
1022 {
1023 AVI_ChunkDumpDebug_level( p_obj, p_child, i_level + 1 );
1024 p_child = p_child->common.p_next;
1025 }

1028int AVI_ChunkReadRoot( stream_t *s, avi_chunk_t *p_root )
1029{
1030 avi_chunk_list_t *p_list = (avi_chunk_list_t*)p_root;
1031 avi_chunk_t *p_chk;
1032 bool b_seekable;
1033
1034 stream_Control( s, STREAM_CAN_FASTSEEK, &b_seekable );
1035
1036 p_list->i_chunk_pos = 0;
1037 p_list->i_chunk_size = stream_Size( s );
1038 p_list->i_chunk_fourcc = AVIFOURCC_LIST;
1039 p_list->p_father = NULL;
1040 p_list->p_next = NULL;
1041 p_list->p_first = NULL;
1042 p_list->p_last = NULL;
1043
1044 p_list->i_type = VLC_FOURCC( 'r', 'o', 'o', 't' );
1045
1046 for( ; ; )
1047 {
1048 p_chk = xmalloc( sizeof( avi_chunk_t ) );
1049 memset( p_chk, 0, sizeof( avi_chunk_t ) );
1050 if( !p_root->common.p_first )
1051 {
1052 p_root->common.p_first = p_chk;
1053 }
1054 else
1055 {
1056 p_root->common.p_last->common.p_next = p_chk;
1057 }
1058 p_root->common.p_last = p_chk;
1059
1060 if( AVI_ChunkRead( s, p_chk, p_root ) ||
1061 ( stream_Tell( s ) >=
1062 (off_t)p_chk->common.p_father->common.i_chunk_pos +
1063 (off_t)__EVEN( p_chk->common.p_father->common.i_chunk_size ) ) )
1064 {
1065 break;
1066 }
1067 /* If we can't seek then stop when we 've found first RIFF-AVI */
1068 if( p_chk->common.i_chunk_fourcc == AVIFOURCC_RIFF &&
1069 p_chk->list.i_type == AVIFOURCC_AVI && !b_seekable )
1070 {
1071 break;
1072 }
1073 }
1074
1075 AVI_ChunkDumpDebug_level( (vlc_object_t*)s, p_root, 0 );
1076 return VLC_SUCCESS;
1077}
