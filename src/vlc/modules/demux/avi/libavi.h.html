31/* Flags for index */
32#define AVIIF_LIST 0x00000001L /* chunk is a 'LIST' */
33#define AVIIF_KEYFRAME 0x00000010L /* this frame is a key frame.*/
34#define AVIIF_NOTIME 0x00000100L /* this frame doesn't take any time */
