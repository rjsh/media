51/*****************************************************************************
52 * Local prototypes

54static int Demux ( demux_t * );

56
57struct demux_sys_t
58{
59 es_format_t fmt;
60 es_out_id_t *p_es;
61
62 date_t pts;
63};
64
65#define CDG_FRAME_SIZE (96)
66#define CDG_FRAME_RATE (75)
67
68/*****************************************************************************
69 * Open: check file and initializes structures
70 *****************************************************************************/
71static int Open( vlc_object_t * p_this )
72{
73 demux_t *p_demux = (demux_t*)p_this;
74 demux_sys_t *p_sys;
75
76 /* Identify cdg file by extension, as there is no simple way to
77 * detect it */
78 if( !demux_IsPathExtension( p_demux, ".cdg" ) && !demux_IsForced( p_demux, "cdg" ) )
79 return VLC_EGENERIC;
80
81 /* CDG file size has to be multiple of CDG_FRAME_SIZE (it works even
82 * if size is unknown ie 0) */
83// if( (stream_Size( p_demux->s ) % CDG_FRAME_SIZE) != 0 )
84// {
85// msg_Err( p_demux, "Reject CDG file based on its size" );
86// return VLC_EGENERIC;
87// }
88
89 p_sys = malloc( sizeof( demux_sys_t ) );
90 if( unlikely(p_sys == NULL) )
91 return VLC_ENOMEM;
92
93 p_demux->pf_demux = Demux;
94 p_demux->pf_control = Control;
95 p_demux->p_sys = p_sys;
96
97 /* */
98 es_format_Init( &p_sys->fmt, VIDEO_ES, VLC_CODEC_CDG );
99 p_sys->fmt.video.i_width = 300-2*6;
100 p_sys->fmt.video.i_height = 216-2*12 ;
101
102 p_sys->p_es = es_out_Add( p_demux->out, &p_sys->fmt );
103
104 /* There is CDG_FRAME_RATE frames per second */
105 date_Init( &p_sys->pts, CDG_FRAME_RATE, 1 );
106 date_Set( &p_sys->pts, 0 );
107
108 return VLC_SUCCESS;
109}
110
111/*****************************************************************************
112 * Demux: read packet and send them to decoders
113 *****************************************************************************
114 * Returns -1 in case of error, 0 in case of EOF, 1 otherwise
115 *****************************************************************************/
116static int Demux( demux_t *p_demux )
117{
118 demux_sys_t *p_sys = p_demux->p_sys;
119 block_t *p_block;
120
121 p_block = stream_Block( p_demux->s, CDG_FRAME_SIZE );
122 if( p_block == NULL )
123 {
124 msg_Dbg( p_demux, "cannot read data, eof" );
125 return 0;
126 }
127
128 p_block->i_dts =
129 p_block->i_pts = VLC_TS_0 + date_Get( &p_sys->pts );
130
131 es_out_Control( p_demux->out, ES_OUT_SET_PCR, p_block->i_pts );
132
133 es_out_Send( p_demux->out, p_sys->p_es, p_block );
134
135 date_Increment( &p_sys->pts, 1 );
136
137 return 1;
138}

140/*****************************************************************************
141 * Close: frees unused data
142 *****************************************************************************/
143static void Close ( vlc_object_t * p_this )
144{
145 demux_t *p_demux = (demux_t *)p_this;
146 demux_sys_t *p_sys = p_demux->p_sys;
147
148 free( p_sys );
149}
150
151/*****************************************************************************
152 * Control:
153 *****************************************************************************/
154static int Control( demux_t *p_demux, int i_query, va_list args )
155{
156 int i_ret = demux_vaControlHelper( p_demux->s, 0, -1,
157 8*CDG_FRAME_SIZE*CDG_FRAME_RATE, CDG_FRAME_SIZE,
158 i_query, args );
159 if( !i_ret && ( i_query == DEMUX_SET_POSITION || i_query == DEMUX_SET_TIME ) )
160 date_Set( &p_demux->p_sys->pts,
161 stream_Tell( p_demux->s ) / CDG_FRAME_SIZE *
162 INT64_C(1000000) / CDG_FRAME_RATE );
163
164 return i_ret;
165}
