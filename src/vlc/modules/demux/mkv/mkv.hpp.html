133class attachment_c
134{
135public:
136 attachment_c( const std::string& _psz_file_name, const std::string& _psz_mime_type, int _i_size )
137 :i_size(_i_size)
138 ,psz_file_name( _psz_file_name)
139 ,psz_mime_type( _psz_mime_type)
140 {
141 p_data = NULL;
142 }
143 ~attachment_c() { free( p_data ); }
144
145 /* Allocs the data space. Returns true if allocation went ok */
146 bool init()
147 {
148 p_data = malloc( i_size );
149 return (p_data != NULL);
150 }
151
152 const char* fileName() const { return psz_file_name.c_str(); }
153 const char* mimeType() const { return psz_mime_type.c_str(); }
154 int size() const { return i_size; }
155
156 void *p_data;
157private:
158 int i_size;
159 std::string psz_file_name;
160 std::string psz_mime_type;
161};
