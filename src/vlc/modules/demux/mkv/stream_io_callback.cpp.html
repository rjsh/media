33vlc_stream_io_callback::vlc_stream_io_callback( stream_t *s_, bool b_owner_ )
34 : s( s_), b_owner( b_owner_ )
35{
36 mb_eof = false;
37}
38
39uint32 vlc_stream_io_callback::read( void *p_buffer, size_t i_size )
40{
41 if( i_size <= 0 || mb_eof )
42 return 0;
43
44 return stream_Read( s, p_buffer, i_size );
45}
46
47void vlc_stream_io_callback::setFilePointer(int64_t i_offset, seek_mode mode )
48{
49 int64_t i_pos, i_size;
50
51 switch( mode )
52 {
53 case seek_beginning:
54 i_pos = i_offset;
55 break;
56 case seek_end:
57 i_pos = stream_Size( s ) - i_offset;
58 break;
59 default:
60 i_pos= stream_Tell( s ) + i_offset;
61 break;
62 }
63
64 if( i_pos < 0 || ( ( i_size = stream_Size( s ) ) != 0 && i_pos >= i_size ) )
65 {
66 mb_eof = true;
67 return;
68 }
69
70 mb_eof = false;
71 if( stream_Seek( s, i_pos ) )
72 {
73 mb_eof = true;
74 }
75 return;
76}
77
78uint64 vlc_stream_io_callback::getFilePointer( void )
79{
80 if ( s == NULL )
81 return 0;
82 return stream_Tell( s );
83}
84
85size_t vlc_stream_io_callback::write(const void *, size_t )
86{
87 return 0;
88}
89
90uint64 vlc_stream_io_callback::toRead( void )
91{
92 uint64_t i_size;
93
94 if( s == NULL)
95 return 0;
96
97 stream_Control( s, STREAM_GET_SIZE, &i_size );
98
99 if( i_size == 0 )
100 return UINT64_MAX;
101
102 return (uint64) i_size - stream_Tell( s );
103}
