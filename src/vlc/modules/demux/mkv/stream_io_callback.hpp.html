26/*****************************************************************************
27 * Stream managment
28 *****************************************************************************/
29class vlc_stream_io_callback: public IOCallback
30{
31 private:
32 stream_t *s;
33 bool mb_eof;
34 bool b_owner;
35
36 public:
37 vlc_stream_io_callback( stream_t *, bool );
38
39 virtual ~vlc_stream_io_callback()
40 {
41 if( b_owner )
42 stream_Delete( s );
43 }
44
45 virtual uint32 read ( void *p_buffer, size_t i_size);
46 virtual void setFilePointer ( int64_t i_offset, seek_mode mode = seek_beginning );
47 virtual size_t write ( const void *p_buffer, size_t i_size);
48 virtual uint64 getFilePointer ( void );
49 virtual void close ( void ) { return; }
50 uint64 toRead ( void );
51};
