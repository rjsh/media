70{
71 for( size_t i_track = 0; i_track < tracks.size(); i_track++ )
72 {

107void matroska_segment_c::LoadCues( KaxCues *cues )
108{
109 bool b_invalid_cue;
110 EbmlParser *ep;
111 EbmlElement *el;
112
113 if( b_cues )
114 {
115 msg_Err( &sys.demuxer, "There can be only 1 Cues per section." );
116 return;
117 }
118
119 ep = new EbmlParser( &es, cues, &sys.demuxer );
120 while( ( el = ep->Get() ) != NULL )
121 {
122 if( MKV_IS_ID( el, KaxCuePoint ) )
123 {
124 b_invalid_cue = false;
125#define idx p_indexes[i_index]
126
127 idx.i_track = -1;
128 idx.i_block_number= -1;
129 idx.i_position = -1;
130 idx.i_time = 0;
131 idx.b_key = true;
132
133 ep->Down();
134 while( ( el = ep->Get() ) != NULL )
135 {
136 if( MKV_IS_ID( el, KaxCueTime ) )
137 {
138 KaxCueTime &ctime = *(KaxCueTime*)el;
139 try
140 {
141 if( unlikely( ctime.GetSize() >= SIZE_MAX ) )
142 {
143 msg_Err( &sys.demuxer, "CueTime size too big");
144 b_invalid_cue = true;
145 break;
146 }
147 ctime.ReadData( es.I_O() );
148 }
149 catch(...)
150 {
151 msg_Err( &sys.demuxer, "Error while reading CueTime" );
152 b_invalid_cue = true;
153 break;
154 }
155 idx.i_time = uint64( ctime ) * i_timescale / (mtime_t)1000;
156 }
157 else if( MKV_IS_ID( el, KaxCueTrackPositions ) )
158 {
159 ep->Down();
160 try
161 {
162 while( ( el = ep->Get() ) != NULL )
163 {
164 if( unlikely( el->GetSize() >= SIZE_MAX ) )
165 {
166 ep->Up();
167 msg_Err( &sys.demuxer, "Error %s too big, aborting", typeid(*el).name() );
168 b_invalid_cue = true;
169 break;
170 }
171
172 if( MKV_IS_ID( el, KaxCueTrack ) )
173 {
174 KaxCueTrack &ctrack = *(KaxCueTrack*)el;
175 ctrack.ReadData( es.I_O() );
176 idx.i_track = uint16( ctrack );
177 }
178 else if( MKV_IS_ID( el, KaxCueClusterPosition ) )
179 {
180 KaxCueClusterPosition &ccpos = *(KaxCueClusterPosition*)el;
181
182 ccpos.ReadData( es.I_O() );
183 idx.i_position = segment->GetGlobalPosition( uint64( ccpos ) );
184 }
185 else if( MKV_IS_ID( el, KaxCueBlockNumber ) )
186 {
187 KaxCueBlockNumber &cbnum = *(KaxCueBlockNumber*)el;
188
189 cbnum.ReadData( es.I_O() );
190 idx.i_block_number = uint32( cbnum );
191 }
192#if LIBMATROSKA_VERSION >= 0x010401
193 else if( MKV_IS_ID( el, KaxCueRelativePosition ) )
194 {
195 /* For future use */
196 }
197 else if( MKV_IS_ID( el, KaxCueDuration ) )
198 {
199 /* For future use */
200 }
201#endif
202 else
203 {
204 msg_Dbg( &sys.demuxer, " * Unknown (%s)", typeid(*el).name() );
205 }
206 }
207 }
208 catch(...)
209 {
210 ep->Up();
211 msg_Err( &sys.demuxer, "Error while reading %s", typeid(*el).name() );
212 b_invalid_cue = true;
213 break;
214 }
215 ep->Up();
216 }
217 else
218 {
219 msg_Dbg( &sys.demuxer, " * Unknown (%s)", typeid(*el).name() );
220 }
221 }
222 ep->Up();
223
224#if 0
225 msg_Dbg( &sys.demuxer, " * added time=%"PRId64" pos=%"PRId64
226 " track=%d bnum=%d", idx.i_time, idx.i_position,
227 idx.i_track, idx.i_block_number );
228#endif
229 if( likely( !b_invalid_cue ) )
230 {
231 i_index++;
232 if( i_index >= i_index_max )
233 {
234 i_index_max += 1024;
235 p_indexes = (mkv_index_t*)xrealloc( p_indexes,
236 sizeof( mkv_index_t ) * i_index_max );
237 }
238 }
239#undef idx
240 }
241 else
242 {
243 msg_Dbg( &sys.demuxer, " * Unknown (%s)", typeid(*el).name() );
244 }
245 }
246 delete ep;
247 b_cues = true;
248 msg_Dbg( &sys.demuxer, "| - loading cues done." );
249}

276SimpleTag * matroska_segment_c::ParseSimpleTags( KaxTagSimple *tag, int target_type )
277{
278 EbmlElement *el;
279 EbmlParser *ep = new EbmlParser( &es, tag, &sys.demuxer );
280 SimpleTag * p_simple = new SimpleTag;
281 size_t max_size = tag->GetSize();
282 size_t size = 0;
283
284 if( !p_simple )
285 {
286 msg_Err( &sys.demuxer, "Couldn't allocate memory for Simple Tag... ignoring it");
287 return NULL;
288 }
289
290 if( !sys.meta )
291 sys.meta = vlc_meta_New();
292
293 msg_Dbg( &sys.demuxer, "| + Simple Tag ");
294 try
295 {
296 while( ( el = ep->Get() ) != NULL && size < max_size)
297 {
298 if( unlikely( el->GetSize() >= SIZE_MAX ) )
299 {
300 msg_Err( &sys.demuxer, "Error %s too big ignoring the tag", typeid(*el).name() );
301 delete ep;
302 delete p_simple;
303 return NULL;
304 }
305 if( MKV_IS_ID( el, KaxTagName ) )
306 {
307 KaxTagName &key = *(KaxTagName*)el;
308 key.ReadData( es.I_O(), SCOPE_ALL_DATA );
309 p_simple->psz_tag_name = strdup( UTFstring( key ).GetUTF8().c_str() );
310 }
311 else if( MKV_IS_ID( el, KaxTagString ) )
312 {
313 KaxTagString &value = *(KaxTagString*)el;
314 value.ReadData( es.I_O(), SCOPE_ALL_DATA );
315 p_simple->p_value = strdup( UTFstring( value ).GetUTF8().c_str() );
316 }
317 else if( MKV_IS_ID( el, KaxTagLangue ) )
318 {
319 KaxTagLangue &language = *(KaxTagLangue*) el;
320 language.ReadData( es.I_O(), SCOPE_ALL_DATA );
321 p_simple->psz_lang = strdup( string( language ).c_str());
322 }
323 else if( MKV_IS_ID( el, KaxTagDefault ) )
324 {
325 KaxTagDefault & dft = *(KaxTagDefault*) el;
326 dft.ReadData( es.I_O(), SCOPE_ALL_DATA );
327 p_simple->b_default = (bool) uint8( dft );
328 }
329 /*Tags can be nested*/
330 else if( MKV_IS_ID( el, KaxTagSimple) )
331 {
332 SimpleTag * p_st = ParseSimpleTags( (KaxTagSimple*)el, target_type );
333 if( p_st )
334 p_simple->sub_tags.push_back( p_st );
335 }
336 /*TODO Handle binary tags*/
337 size += el->HeadSize() + el->GetSize();
338 }
339 }
340 catch(...)
341 {
342 msg_Err( &sys.demuxer, "Error while reading Tag ");
343 delete ep;
344 delete p_simple;
345 return NULL;
346 }
347 delete ep;
348
349 if( !p_simple->psz_tag_name || !p_simple->p_value )
350 {
351 msg_Warn( &sys.demuxer, "Invalid MKV SimpleTag found.");
352 delete p_simple;
353 return NULL;
354 }
355 for( int i = 0; metadata_map[i].key; i++ )
356 {
357 if( !strcmp( p_simple->psz_tag_name, metadata_map[i].key ) &&
358 (metadata_map[i].target_type == 0 || target_type == metadata_map[i].target_type ) )
359 {
360 vlc_meta_Set( sys.meta, metadata_map[i].type, p_simple->p_value );
361 msg_Dbg( &sys.demuxer, "| | + Meta %s: %s", p_simple->psz_tag_name, p_simple->p_value);
362 goto done;
363 }
364 }
365 msg_Dbg( &sys.demuxer, "| | + Meta %s: %s", p_simple->psz_tag_name, p_simple->p_value);
366 vlc_meta_AddExtra( sys.meta, p_simple->psz_tag_name, p_simple->p_value);
367done:
368 return p_simple;
369}
370
371#define PARSE_TAG( type ) \
372 do { \
373 msg_Dbg( &sys.demuxer, "| + " type ); \
374 ep->Down(); \
375 while( ( el = ep->Get() ) != NULL ) \
376 { \
377 msg_Dbg( &sys.demuxer, "| | + Unknown (%s)", typeid( *el ).name() ); \
378 } \
379 ep->Up(); } while( 0 )

382void matroska_segment_c::LoadTags( KaxTags *tags )
383{
384 /* Master elements */
385 EbmlParser *ep = new EbmlParser( &es, tags, &sys.demuxer );
386 EbmlElement *el;
387
388 while( ( el = ep->Get() ) != NULL )
389 {
390 if( MKV_IS_ID( el, KaxTag ) )
391 {
392 Tag * p_tag = new Tag;
393 if(!p_tag)
394 {
395 msg_Err( &sys.demuxer,"Couldn't allocate memory for tag... ignoring it");
396 continue;
397 }
398 msg_Dbg( &sys.demuxer, "+ Tag" );
399 ep->Down();
400 int target_type = 50;
401 while( ( el = ep->Get() ) != NULL )
402 {
403 if( MKV_IS_ID( el, KaxTagTargets ) )
404 {
405 msg_Dbg( &sys.demuxer, "| + Targets" );
406 ep->Down();
407 while( ( el = ep->Get() ) != NULL )
408 {
409 try
410 {
411 if( unlikely( el->GetSize() >= SIZE_MAX ) )
412 {
413 msg_Err( &sys.demuxer, "Invalid size while reading tag");
414 break;
415 }
416 if( MKV_IS_ID( el, KaxTagTargetTypeValue ) )
417 {
418 KaxTagTargetTypeValue &value = *(KaxTagTargetTypeValue*)el;
419 value.ReadData( es.I_O() );
420
421 msg_Dbg( &sys.demuxer, "| | + TargetTypeValue: %u", uint32(value));
422 target_type = uint32(value);
423 }
424 if( MKV_IS_ID( el, KaxTagTrackUID ) )
425 {
426 p_tag->i_tag_type = TRACK_UID;
427 KaxTagTrackUID &uid = *(KaxTagTrackUID*) el;
428 uid.ReadData( es.I_O() );
429 p_tag->i_uid = uint64( uid );
430 msg_Dbg( &sys.demuxer, "| | + TrackUID: %"PRIu64, p_tag->i_uid);
431
432 }
433 if( MKV_IS_ID( el, KaxTagEditionUID ) )
434 {
435 p_tag->i_tag_type = EDITION_UID;
436 KaxTagEditionUID &uid = *(KaxTagEditionUID*) el;
437 uid.ReadData( es.I_O() );
438 p_tag->i_uid = uint64( uid );
439 msg_Dbg( &sys.demuxer, "| | + EditionUID: %"PRIu64, p_tag->i_uid);
440 }
441 if( MKV_IS_ID( el, KaxTagChapterUID ) )
442 {
443 p_tag->i_tag_type = CHAPTER_UID;
444 KaxTagChapterUID &uid = *(KaxTagChapterUID*) el;
445 uid.ReadData( es.I_O() );
446 p_tag->i_uid = uint64( uid );
447 msg_Dbg( &sys.demuxer, "| | + ChapterUID: %"PRIu64, p_tag->i_uid);
448 }
449 if( MKV_IS_ID( el, KaxTagAttachmentUID ) )
450 {
451 p_tag->i_tag_type = ATTACHMENT_UID;
452 KaxTagAttachmentUID &uid = *(KaxTagAttachmentUID*) el;
453 uid.ReadData( es.I_O() );
454 p_tag->i_uid = uint64( uid );
455 msg_Dbg( &sys.demuxer, "| | + AttachmentUID: %"PRIu64, p_tag->i_uid);
456 }
457 }
458 catch(...)
459 {
460 msg_Err( &sys.demuxer, "Error while reading tag");
461 break;
462 }
463 }
464 ep->Up();
465 }
466 else if( MKV_IS_ID( el, KaxTagSimple ) )
467 {
468 SimpleTag * p_simple =
469 ParseSimpleTags( static_cast<KaxTagSimple*>( el ),
470 target_type );
471 if( p_simple )
472 p_tag->simple_tags.push_back( p_simple );
473 }
474#if 0 // not valid anymore
475 else if( MKV_IS_ID( el, KaxTagGeneral ) )
476 PARSE_TAG( "General" );
477 else if( MKV_IS_ID( el, KaxTagGenres ) )
478 PARSE_TAG( "Genres" );
479 else if( MKV_IS_ID( el, KaxTagAudioSpecific ) )
480 PARSE_TAG( "Audio Specific" );
481 else if( MKV_IS_ID( el, KaxTagImageSpecific ) )
482 PARSE_TAG( "Images Specific" );
483 else if( MKV_IS_ID( el, KaxTagMultiComment ) )
484 {
485 msg_Dbg( &sys.demuxer, "| + Multi Comment" );
486 }
487 else if( MKV_IS_ID( el, KaxTagMultiCommercial ) )
488 {
489 msg_Dbg( &sys.demuxer, "| + Multi Commercial" );
490 }
491 else if( MKV_IS_ID( el, KaxTagMultiDate ) )
492 {
493 msg_Dbg( &sys.demuxer, "| + Multi Date" );
494 }
495 else if( MKV_IS_ID( el, KaxTagMultiEntity ) )
496 {
497 msg_Dbg( &sys.demuxer, "| + Multi Entity" );
498 }
499 else if( MKV_IS_ID( el, KaxTagMultiIdentifier ) )
500 {
501 msg_Dbg( &sys.demuxer, "| + Multi Identifier" );
502 }
503 else if( MKV_IS_ID( el, KaxTagMultiLegal ) )
504 {
505 msg_Dbg( &sys.demuxer, "| + Multi Legal" );
506 }
507 else if( MKV_IS_ID( el, KaxTagMultiTitle ) )
508 {
509 msg_Dbg( &sys.demuxer, "| + Multi Title" );
510 }
511#endif
512 else
513 {
514 msg_Dbg( &sys.demuxer, "| + LoadTag Unknown (%s)", typeid( *el ).name() );
515 }
516 }
517 ep->Up();
518 this->tags.push_back(p_tag);
519 }
520 else
521 {
522 msg_Dbg( &sys.demuxer, "+ Unknown (%s)", typeid( *el ).name() );
523 }
524 }
525 delete ep;
526
527 msg_Dbg( &sys.demuxer, "loading tags done." );
528}

607bool matroska_segment_c::PreloadFamily( const matroska_segment_c & of_segment )
608{
609 if ( b_preloaded )
610 return false;
611
612 for (size_t i=0; i<families.size(); i++)
613 {
614 for (size_t j=0; j<of_segment.families.size(); j++)
615 {
616 if ( *(families[i]) == *(of_segment.families[j]) )
617 return Preload( );
618 }
619 }
620
621 return false;
622}

624bool matroska_segment_c::CompareSegmentUIDs( const matroska_segment_c * p_item_a, const matroska_segment_c * p_item_b )
625{
626 EbmlBinary *p_tmp;
627
628 if ( p_item_a == NULL || p_item_b == NULL )
629 return false;
630
631 p_tmp = (EbmlBinary *)p_item_a->p_segment_uid;
632 if ( p_item_b->p_prev_segment_uid != NULL
633 && *p_tmp == *p_item_b->p_prev_segment_uid )
634 return true;
635
636 p_tmp = (EbmlBinary *)p_item_a->p_next_segment_uid;
637 if ( !p_tmp )
638 return false;
639
640 if ( p_item_b->p_segment_uid != NULL
641 && *p_tmp == *p_item_b->p_segment_uid )
642 return true;
643
644 if ( p_item_b->p_prev_segment_uid != NULL
645 && *p_tmp == *p_item_b->p_prev_segment_uid )
646 return true;
647
648 return false;
649}

651bool matroska_segment_c::Preload( )
652{
653 if ( b_preloaded )
654 return false;
655
656 EbmlElement *el = NULL;
657
658 ep->Reset( &sys.demuxer );
659
660 while( ( el = ep->Get() ) != NULL )
661 {
662 if( MKV_IS_ID( el, KaxSeekHead ) )
663 {
664 /* Multiple allowed */
665 /* We bail at 10, to prevent possible recursion */
666 msg_Dbg( &sys.demuxer, "| + Seek head" );
667 if( i_seekhead_count < 10 )
668 {
669 i_seekhead_position = (int64_t) es.I_O().getFilePointer();
670 ParseSeekHead( static_cast<KaxSeekHead*>( el ) );
671 }
672 }
673 else if( MKV_IS_ID( el, KaxInfo ) )
674 {
675 /* Multiple allowed, mandatory */
676 msg_Dbg( &sys.demuxer, "| + Information" );
677 if( i_info_position < 0 ) // FIXME
678 ParseInfo( static_cast<KaxInfo*>( el ) );
679 i_info_position = (int64_t) es.I_O().getFilePointer();
680 }
681 else if( MKV_IS_ID( el, KaxTracks ) )
682 {
683 /* Multiple allowed */
684 msg_Dbg( &sys.demuxer, "| + Tracks" );
685 if( i_tracks_position < 0 ) // FIXME
686 ParseTracks( static_cast<KaxTracks*>( el ) );
687 if ( tracks.size() == 0 )
688 {
689 msg_Err( &sys.demuxer, "No tracks supported" );
690 return false;
691 }
692 i_tracks_position = (int64_t) es.I_O().getFilePointer();
693 }
694 else if( MKV_IS_ID( el, KaxCues ) )
695 {
696 msg_Dbg( &sys.demuxer, "| + Cues" );
697 if( i_cues_position < 0 )
698 LoadCues( static_cast<KaxCues*>( el ) );
699 i_cues_position = (int64_t) es.I_O().getFilePointer();
700 }
701 else if( MKV_IS_ID( el, KaxCluster ) )
702 {
703 msg_Dbg( &sys.demuxer, "| + Cluster" );
704
705 cluster = (KaxCluster*)el;
706
707 i_cluster_pos = i_start_pos = cluster->GetElementPosition();
708 ParseCluster( );
709
710 ep->Down();
711 /* stop pre-parsing the stream */
712 break;
713 }
714 else if( MKV_IS_ID( el, KaxAttachments ) )
715 {
716 msg_Dbg( &sys.demuxer, "| + Attachments" );
717 if( i_attachments_position < 0 )
718 ParseAttachments( static_cast<KaxAttachments*>( el ) );
719 i_attachments_position = (int64_t) es.I_O().getFilePointer();
720 }
721 else if( MKV_IS_ID( el, KaxChapters ) )
722 {
723 msg_Dbg( &sys.demuxer, "| + Chapters" );
724 if( i_chapters_position < 0 )
725 ParseChapters( static_cast<KaxChapters*>( el ) );
726 i_chapters_position = (int64_t) es.I_O().getFilePointer();
727 }
728 else if( MKV_IS_ID( el, KaxTag ) )
729 {
730 msg_Dbg( &sys.demuxer, "| + Tags" );
731 /*FIXME if( i_tags_position < 0)
732 LoadTags( static_cast<KaxTags*>( el ) );*/
733 i_tags_position = (int64_t) es.I_O().getFilePointer();
734 }
735 else if( MKV_IS_ID( el, EbmlVoid ) )
736 msg_Dbg( &sys.demuxer, "| + Void" );
737 else
738 msg_Dbg( &sys.demuxer, "| + Preload Unknown (%s)", typeid(*el).name() );
739 }
740
741 ComputeTrackPriority();
742
743 b_preloaded = true;
744
745 return true;
746}

847void matroska_segment_c::Seek( mtime_t i_date, mtime_t i_time_offset, int64_t i_global_position )
848{
849 KaxBlock *block;
850 KaxSimpleBlock *simpleblock;
851 int64_t i_block_duration;
852 size_t i_track;
853 int64_t i_seek_position = i_start_pos;
854 int64_t i_seek_time = i_start_time;
855 mtime_t i_pts = 0;
856 spoint *p_first = NULL;
857 spoint *p_last = NULL;
858 int i_cat;
859 bool b_has_key = false;
860
861 for( size_t i = 0; i < tracks.size(); i++)
862 tracks[i]->i_last_dts = VLC_TS_INVALID;
863
864 if( i_global_position >= 0 )
865 {
866 /* Special case for seeking in files with no cues */
867 EbmlElement *el = NULL;
868
869 /* Start from the last known index instead of the beginning eachtime */
870 if( i_index == 0)
871 es.I_O().setFilePointer( i_start_pos, seek_beginning );
872 else
873 es.I_O().setFilePointer( p_indexes[ i_index - 1 ].i_position,
874 seek_beginning );
875 delete ep;
876 ep = new EbmlParser( &es, segment, &sys.demuxer );
877 cluster = NULL;
878
879 while( ( el = ep->Get() ) != NULL )
880 {
881 if( MKV_IS_ID( el, KaxCluster ) )
882 {
883 cluster = (KaxCluster *)el;
884 i_cluster_pos = cluster->GetElementPosition();
885 if( i_index == 0 ||
886 ( i_index > 0 &&
887 p_indexes[i_index - 1].i_position < (int64_t)cluster->GetElementPosition() ) )
888 {
889 ParseCluster(false);
890 IndexAppendCluster( cluster );
891 }
892 if( es.I_O().getFilePointer() >= (unsigned) i_global_position )
893 break;
894 }
895 }
896 }
897
898 /* Don't try complex seek if we seek to 0 */
899 if( i_date == 0 && i_time_offset == 0 )
900 {
901 es_out_Control( sys.demuxer.out, ES_OUT_SET_PCR, VLC_TS_0 );
902 es_out_Control( sys.demuxer.out, ES_OUT_SET_NEXT_DISPLAY_TIME,
903 INT64_C(0) );
904 es.I_O().setFilePointer( i_start_pos );
905
906 delete ep;
907 ep = new EbmlParser( &es, segment, &sys.demuxer );
908 cluster = NULL;
909 sys.i_start_pts = 0;
910 sys.i_pts = 0;
911 sys.i_pcr = 0;
912 return;
913 }
914
915 int i_idx = 0;
916 if ( i_index > 0 )
917 {
918
919 for( ; i_idx < i_index; i_idx++ )
920 if( p_indexes[i_idx].i_time + i_time_offset > i_date )
921 break;
922
923 if( i_idx > 0 )
924 i_idx--;
925
926 i_seek_position = p_indexes[i_idx].i_position;
927 i_seek_time = p_indexes[i_idx].i_time;
928 }
929
930 msg_Dbg( &sys.demuxer, "seek got %"PRId64" - %"PRId64, i_seek_time, i_seek_position );
931
932 es.I_O().setFilePointer( i_seek_position, seek_beginning );
933
934 delete ep;
935 ep = new EbmlParser( &es, segment, &sys.demuxer );
936 cluster = NULL;
937
938 sys.i_start_pts = i_date;
939
940 /* now parse until key frame */
941 const int es_types[3] = { VIDEO_ES, AUDIO_ES, SPU_ES };
942 i_cat = es_types[0];
943 mtime_t i_seek_preroll = 0;
944 for( int i = 0; i < 2; i_cat = es_types[++i] )
945 {
946 for( i_track = 0; i_track < tracks.size(); i_track++ )
947 {
948 if( tracks[i_track]->i_seek_preroll )
949 {
950 bool b_enabled;
951 if( es_out_Control( sys.demuxer.out,
952 ES_OUT_GET_ES_STATE,
953 tracks[i_track]->p_es,
954 &b_enabled ) == VLC_SUCCESS &&
955 b_enabled )
956 i_seek_preroll = __MAX( i_seek_preroll,
957 tracks[i_track]->i_seek_preroll );
958 }
959 if( tracks[i_track]->fmt.i_cat == i_cat )
960 {
961 spoint * seekpoint = new spoint(i_track, i_seek_time, i_seek_position, i_seek_position);
962 if( unlikely( !seekpoint ) )
963 {
964 for( spoint * sp = p_first; sp; )
965 {
966 spoint * tmp = sp;
967 sp = sp->p_next;
968 delete tmp;
969 }
970 return;
971 }
972 if( unlikely( !p_first ) )
973 {
974 p_first = seekpoint;
975 p_last = seekpoint;
976 }
977 else
978 {
979 p_last->p_next = seekpoint;
980 p_last = seekpoint;
981 }
982 }
983 }
984 if( likely( p_first ) )
985 break;
986 }
987 /*Neither video nor audio track... no seek further*/
988 if( unlikely( !p_first ) )
989 {
990 es_out_Control( sys.demuxer.out, ES_OUT_SET_PCR, i_date );
991 es_out_Control( sys.demuxer.out, ES_OUT_SET_NEXT_DISPLAY_TIME, i_date );
992 return;
993 }
994 i_date -= i_seek_preroll;
995 for(;;)
996 {
997 do
998 {
999 bool b_key_picture;
1000 bool b_discardable_picture;
1001 if( BlockGet( block, simpleblock, &b_key_picture, &b_discardable_picture, &i_block_duration ) )
1002 {
1003 msg_Warn( &sys.demuxer, "cannot get block EOF?" );
1004 while( p_first )
1005 {
1006 spoint *tmp = p_first;
1007 p_first = p_first->p_next;
1008 delete tmp;
1009 }
1010 return;
1011 }
1012
1013 /* check if block's track is in our list */
1014 for( i_track = 0; i_track < tracks.size(); i_track++ )
1015 {
1016 if( (simpleblock && tracks[i_track]->i_number == simpleblock->TrackNum()) ||
1017 (block && tracks[i_track]->i_number == block->TrackNum()) )
1018 break;
1019 }
1020
1021 if( simpleblock )
1022 i_pts = sys.i_chapter_time + simpleblock->GlobalTimecode() / (mtime_t) 1000;
1023 else
1024 i_pts = sys.i_chapter_time + block->GlobalTimecode() / (mtime_t) 1000;
1025 if( i_track < tracks.size() )
1026 {
1027 if( tracks[i_track]->fmt.i_cat == i_cat && b_key_picture )
1028 {
1029 /* get the seekpoint */
1030 spoint * sp;
1031 for( sp = p_first; sp; sp = sp->p_next )
1032 if( sp->i_track == i_track )
1033 break;
1034
1035 sp->i_date = i_pts;
1036 if( simpleblock )
1037 sp->i_seek_pos = simpleblock->GetElementPosition();
1038 else
1039 sp->i_seek_pos = i_block_pos;
1040 sp->i_cluster_pos = i_cluster_pos;
1041 b_has_key = true;
1042 }
1043 }
1044
1045 delete block;
1046 } while( i_pts < i_date );
1047 if( b_has_key || !i_idx )
1048 break;
1049
1050 /* No key picture was found in the cluster seek to previous seekpoint */
1051 i_date = i_time_offset + p_indexes[i_idx].i_time;
1052 i_idx--;
1053 i_pts = 0;
1054 es.I_O().setFilePointer( p_indexes[i_idx].i_position );
1055 delete ep;
1056 ep = new EbmlParser( &es, segment, &sys.demuxer );
1057 cluster = NULL;
1058 }
1059
1060 /* rewind to the last I img */
1061 spoint * p_min;
1062 for( p_min = p_first, p_last = p_first; p_last; p_last = p_last->p_next )
1063 if( p_last->i_date < p_min->i_date )
1064 p_min = p_last;
1065
1066 sys.i_pcr = sys.i_pts = p_min->i_date;
1067 es_out_Control( sys.demuxer.out, ES_OUT_SET_PCR, VLC_TS_0 + sys.i_pcr );
1068 es_out_Control( sys.demuxer.out, ES_OUT_SET_NEXT_DISPLAY_TIME, i_date );
1069 cluster = (KaxCluster *) ep->UnGet( p_min->i_seek_pos, p_min->i_cluster_pos );
1070
1071 /* hack use BlockGet to get the cluster then goto the wanted block */
1072 if ( !cluster )
1073 {
1074 bool b_key_picture;
1075 bool b_discardable_picture;
1076 BlockGet( block, simpleblock, &b_key_picture, &b_discardable_picture, &i_block_duration );
1077 delete block;
1078 cluster = (KaxCluster *) ep->UnGet( p_min->i_seek_pos, p_min->i_cluster_pos );
1079 }
1080
1081 while( p_first )
1082 {
1083 p_min = p_first;
1084 p_first = p_first->p_next;
1085 delete p_min;
1086 }
1087}

1112void matroska_segment_c::ComputeTrackPriority()
1113{
1114 bool b_has_default_video = false;
1115 bool b_has_default_audio = false;
1116 /* check for default */
1117 for(size_t i_track = 0; i_track < tracks.size(); i_track++)
1118 {
1119 mkv_track_t *p_tk = tracks[i_track];
1120 es_format_t *p_fmt = &p_tk->fmt;
1121 if( p_fmt->i_cat == VIDEO_ES )
1122 b_has_default_video |=
1123 p_tk->b_enabled && ( p_tk->b_default || p_tk->b_forced );
1124 else if( p_fmt->i_cat == AUDIO_ES )
1125 b_has_default_audio |=
1126 p_tk->b_enabled && ( p_tk->b_default || p_tk->b_forced );
1127 }
1128
1129 for( size_t i_track = 0; i_track < tracks.size(); i_track++ )
1130 {
1131 mkv_track_t *p_tk = tracks[i_track];
1132 es_format_t *p_fmt = &p_tk->fmt;
1133
1134 if( unlikely( p_fmt->i_cat == UNKNOWN_ES || !p_tk->psz_codec ) )
1135 {
1136 msg_Warn( &sys.demuxer, "invalid track[%d, n=%d]", (int)i_track, p_tk->i_number );
1137 p_tk->p_es = NULL;
1138 continue;
1139 }
1140 else if( unlikely( !b_has_default_video && p_fmt->i_cat == VIDEO_ES ) )
1141 {
1142 p_tk->b_default = true;
1143 b_has_default_video = true;
1144 }
1145 else if( unlikely( !b_has_default_audio && p_fmt->i_cat == AUDIO_ES ) )
1146 {
1147 p_tk->b_default = true;
1148 b_has_default_audio = true;
1149 }
1150 if( unlikely( !p_tk->b_enabled ) )
1151 p_tk->fmt.i_priority = ES_PRIORITY_NOT_SELECTABLE;
1152 else if( p_tk->b_forced )
1153 p_tk->fmt.i_priority = ES_PRIORITY_SELECTABLE_MIN + 2;
1154 else if( p_tk->b_default )
1155 p_tk->fmt.i_priority = ES_PRIORITY_SELECTABLE_MIN + 1;
1156 else
1157 p_tk->fmt.i_priority = ES_PRIORITY_SELECTABLE_MIN;
1158
1159 /* Avoid multivideo tracks when unnecessary */
1160 if( p_tk->fmt.i_cat == VIDEO_ES )
1161 p_tk->fmt.i_priority--;
1162 }
1163}

1165bool matroska_segment_c::Select( mtime_t i_start_time )
1166{
1167 /* add all es */
1168 msg_Dbg( &sys.demuxer, "found %d es", (int)tracks.size() );
1169
1170 for( size_t i_track = 0; i_track < tracks.size(); i_track++ )
1171 {
1172 mkv_track_t *p_tk = tracks[i_track];
1173 es_format_t *p_fmt = &p_tk->fmt;
1174
1175 if( unlikely( p_fmt->i_cat == UNKNOWN_ES || !p_tk->psz_codec ) )
1176 {
1177 msg_Warn( &sys.demuxer, "invalid track[%d, n=%d]", (int)i_track, p_tk->i_number );
1178 p_tk->p_es = NULL;
1179 continue;
1180 }
1181
1182 if( !p_tk->p_es )
1183 p_tk->p_es = es_out_Add( sys.demuxer.out, &p_tk->fmt );
1184
1185 /* Turn on a subtitles track if it has been flagged as default -
1186 * but only do this if no subtitles track has already been engaged,
1187 * either by an earlier 'default track' (??) or by default
1188 * language choice behaviour.
1189 */
1190 if( p_tk->b_default || p_tk->b_forced )
1191 {
1192 es_out_Control( sys.demuxer.out,
1193 ES_OUT_SET_ES_DEFAULT,
1194 p_tk->p_es );
1195 }
1196 }
1197 es_out_Control( sys.demuxer.out, ES_OUT_SET_NEXT_DISPLAY_TIME, i_start_time );
1198
1199 sys.i_start_pts = i_start_time;
1200 // reset the stream reading to the first cluster of the segment used
1201 es.I_O().setFilePointer( i_start_pos );
1202
1203 delete ep;
1204 ep = new EbmlParser( &es, segment, &sys.demuxer );
1205
1206 return true;
1207}

1225int matroska_segment_c::BlockGet( KaxBlock * & pp_block, KaxSimpleBlock * & pp_simpleblock, bool *pb_key_picture, bool *pb_discardable_picture, int64_t *pi_duration )
1226{
1227 pp_simpleblock = NULL;
1228 pp_block = NULL;
1229
1230 *pb_key_picture = true;
1231 *pb_discardable_picture = false;
1232 size_t i_tk;
1233 *pi_duration = 0;
1234
1235 for( ;; )
1236 {
1237 EbmlElement *el = NULL;
1238 int i_level;
1239
1240 if ( ep == NULL )
1241 return VLC_EGENERIC;
1242
1243 if( pp_simpleblock != NULL || ((el = ep->Get()) == NULL && pp_block != NULL) )
1244 {
1245 /* Check blocks validity to protect againts broken files */
1246 if( BlockFindTrackIndex( &i_tk, pp_block , pp_simpleblock ) )
1247 {
1248 ep->Unkeep();
1249 pp_simpleblock = NULL;
1250 pp_block = NULL;
1251 continue;
1252 }
1253 if( pp_simpleblock != NULL )
1254 {
1255 *pb_key_picture = pp_simpleblock->IsKeyframe();
1256 *pb_discardable_picture = pp_simpleblock->IsDiscardable();
1257 }
1258 /* We have block group let's check if the picture is a keyframe */
1259 else if( *pb_key_picture )
1260 {
1261 switch(tracks[i_tk]->fmt.i_codec)
1262 {
1263 case VLC_CODEC_THEORA:
1264 {
1265 DataBuffer *p_data = &pp_block->GetBuffer(0);
1266 size_t sz = p_data->Size();
1267 const uint8_t * p_buff = p_data->Buffer();
1268 /* if the second bit of a Theora frame is 1
1269 it's not a keyframe */
1270 if( sz && p_buff )
1271 {
1272 if( p_buff[0] & 0x40 )
1273 *pb_key_picture = false;
1274 }
1275 else
1276 *pb_key_picture = false;
1277 break;
1278 }
1279 }
1280 }
1281
1282 /* update the index */
1283#define idx p_indexes[i_index - 1]
1284 if( i_index > 0 && idx.i_time == -1 )
1285 {
1286 if ( pp_simpleblock != NULL )
1287 idx.i_time = pp_simpleblock->GlobalTimecode() / (mtime_t)1000;
1288 else
1289 idx.i_time = (*pp_block).GlobalTimecode() / (mtime_t)1000;
1290 idx.b_key = *pb_key_picture;
1291 }
1292#undef idx
1293 return VLC_SUCCESS;
1294 }
1295
1296 i_level = ep->GetLevel();
1297
1298 if( el == NULL )
1299 {
1300 if( i_level > 1 )
1301 {
1302 ep->Up();
1303 continue;
1304 }
1305 msg_Warn( &sys.demuxer, "EOF" );
1306 return VLC_EGENERIC;
1307 }
1308
1309 /* Verify that we are still inside our cluster
1310 * It can happens whith broken files and when seeking
1311 * without index */
1312 if( i_level > 1 )
1313 {
1314 if( cluster && !ep->IsTopPresent( cluster ) )
1315 {
1316 msg_Warn( &sys.demuxer, "Unexpected escape from current cluster" );
1317 cluster = NULL;
1318 }
1319 if( !cluster )
1320 continue;
1321 }
1322
1323 /* do parsing */
1324 try
1325 {
1326 switch ( i_level )
1327 {
1328 case 1:
1329 if( MKV_IS_ID( el, KaxCluster ) )
1330 {
1331 cluster = (KaxCluster*)el;
1332 i_cluster_pos = cluster->GetElementPosition();
1333
1334 // reset silent tracks
1335 for (size_t i=0; i<tracks.size(); i++)
1336 {
1337 tracks[i]->b_silent = false;
1338 }
1339
1340 ep->Down();
1341 }
1342 else if( MKV_IS_ID( el, KaxCues ) )
1343 {
1344 msg_Warn( &sys.demuxer, "find KaxCues FIXME" );
1345 return VLC_EGENERIC;
1346 }
1347 else
1348 {
1349 msg_Dbg( &sys.demuxer, "unknown (%s)", typeid( el ).name() );
1350 }
1351 break;
1352 case 2:
1353 if( unlikely( el->GetSize() >= SIZE_MAX ) )
1354 {
1355 msg_Err( &sys.demuxer, "Error while reading %s... upping level", typeid(*el).name());
1356 ep->Up();
1357 break;
1358 }
1359 if( MKV_IS_ID( el, KaxClusterTimecode ) )
1360 {
1361 KaxClusterTimecode &ctc = *(KaxClusterTimecode*)el;
1362
1363 ctc.ReadData( es.I_O(), SCOPE_ALL_DATA );
1364 cluster->InitTimecode( uint64( ctc ), i_timescale );
1365
1366 /* add it to the index */
1367 if( i_index == 0 ||
1368 ( i_index > 0 &&
1369 p_indexes[i_index - 1].i_position < (int64_t)cluster->GetElementPosition() ) )
1370 IndexAppendCluster( cluster );
1371 }
1372 else if( MKV_IS_ID( el, KaxClusterSilentTracks ) )
1373 {
1374 ep->Down();
1375 }
1376 else if( MKV_IS_ID( el, KaxBlockGroup ) )
1377 {
1378 i_block_pos = el->GetElementPosition();
1379 ep->Down();
1380 }
1381 else if( MKV_IS_ID( el, KaxSimpleBlock ) )
1382 {
1383 pp_simpleblock = (KaxSimpleBlock*)el;
1384
1385 pp_simpleblock->ReadData( es.I_O() );
1386 pp_simpleblock->SetParent( *cluster );
1387 }
1388 break;
1389 case 3:
1390 if( unlikely( el->GetSize() >= SIZE_MAX ) )
1391 {
1392 msg_Err( &sys.demuxer, "Error while reading %s... upping level", typeid(*el).name());
1393 ep->Up();
1394 ep->Unkeep();
1395 pp_simpleblock = NULL;
1396 pp_block = NULL;
1397 break;
1398 }
1399 if( MKV_IS_ID( el, KaxBlock ) )
1400 {
1401 pp_block = (KaxBlock*)el;
1402
1403 pp_block->ReadData( es.I_O() );
1404 pp_block->SetParent( *cluster );
1405
1406 ep->Keep();
1407 }
1408 else if( MKV_IS_ID( el, KaxBlockDuration ) )
1409 {
1410 KaxBlockDuration &dur = *(KaxBlockDuration*)el;
1411
1412 dur.ReadData( es.I_O() );
1413 *pi_duration = uint64( dur );
1414 }
1415 else if( MKV_IS_ID( el, KaxReferenceBlock ) )
1416 {
1417 KaxReferenceBlock &ref = *(KaxReferenceBlock*)el;
1418
1419 ref.ReadData( es.I_O() );
1420
1421 if( *pb_key_picture )
1422 *pb_key_picture = false;
1423 else if( int64( ref ) > 0 )
1424 *pb_discardable_picture = true;
1425 }
1426 else if( MKV_IS_ID( el, KaxClusterSilentTrackNumber ) )
1427 {
1428 KaxClusterSilentTrackNumber &track_num = *(KaxClusterSilentTrackNumber*)el;
1429 track_num.ReadData( es.I_O() );
1430 // find the track
1431 for (size_t i=0; i<tracks.size(); i++)
1432 {
1433 if ( tracks[i]->i_number == uint32(track_num))
1434 {
1435 tracks[i]->b_silent = true;
1436 break;
1437 }
1438 }
1439 }
1440#if LIBMATROSKA_VERSION >= 0x010401
1441 else if( MKV_IS_ID( el, KaxDiscardPadding ) )
1442 {
1443 KaxDiscardPadding &dp = *(KaxDiscardPadding*) el;
1444 dp.ReadData( es.I_O() );
1445 if ( *pi_duration < int64(dp) )
1446 *pi_duration = 0;
1447 else
1448 *pi_duration -= int64(dp);
1449 }
1450#endif
1451 break;
1452 default:
1453 msg_Err( &sys.demuxer, "invalid level = %d", i_level );
1454 return VLC_EGENERIC;
1455 }
1456 }
1457 catch(...)
1458 {
1459 msg_Err( &sys.demuxer, "Error while reading %s... upping level", typeid(*el).name());
1460 ep->Up();
1461 ep->Unkeep();
1462 pp_simpleblock = NULL;
1463 pp_block = NULL;
1464 }
1465 }
