125/*****************************************************************************
126 * Open: check file and initializes ASF structures
127 *****************************************************************************/
128static int Open( vlc_object_t * p_this )
129{
130 demux_t *p_demux = (demux_t *)p_this;
131 demux_sys_t *p_sys;
132 guid_t guid;
133 const uint8_t *p_peek;
134
135 /* A little test to see if it could be a asf stream */
136 if( stream_Peek( p_demux->s, &p_peek, 16 ) < 16 ) return VLC_EGENERIC;
137
138 ASF_GetGUID( &guid, p_peek );
139 if( !guidcmp( &guid, &asf_object_header_guid ) ) return VLC_EGENERIC;
140
141 /* Set p_demux fields */
142 p_demux->pf_demux = Demux;
143 p_demux->pf_control = Control;
144 p_demux->p_sys = p_sys = calloc( 1, sizeof( demux_sys_t ) );
145
146 /* Load the headers */
147 if( DemuxInit( p_demux ) )
148 {
149 free( p_sys );
150 return VLC_EGENERIC;
151 }
152 return VLC_SUCCESS;
153}

155/*****************************************************************************
156 * Demux: read packet and send them to decoders
157 *****************************************************************************/
158static int Demux( demux_t *p_demux )
159{
160 demux_sys_t *p_sys = p_demux->p_sys;
161
162 for( int i=0; i<ES_CATEGORY_COUNT; i++ )
163 {
164 if ( p_sys->i_access_selected_track[i] > 0 )
165 {
166 es_out_Control( p_demux->out, ES_OUT_SET_ES_STATE,
167 p_sys->track[p_sys->i_access_selected_track[i]]->p_es, true );
168 p_sys->i_access_selected_track[i] = 0;
169 }
170 }
171
172 /* Get selected tracks, especially for computing PCR */
173 for( int i=0; i<MAX_ASF_TRACKS; i++ )
174 {
175 asf_track_t *tk = p_sys->track[i];
176 if ( !tk ) continue;
177 if ( tk->p_es )
178 es_out_Control( p_demux->out, ES_OUT_GET_ES_STATE, tk->p_es, & tk->b_selected );
179 else
180 tk->b_selected = false;
181 }
182
183 for( ;; )
184 {
185 const uint8_t *p_peek;
186 mtime_t i_length;
187 mtime_t i_time_begin = GetMoviePTS( p_sys );
188 int i_result;
189
190 if( !vlc_object_alive (p_demux) )
191 break;
192#if 0
193 /* FIXME: returns EOF too early for some mms streams */
194 if( p_sys->i_data_end >= 0 &&
195 stream_Tell( p_demux->s ) >= p_sys->i_data_end )
196 return 0; /* EOF */
197#endif
198
199 /* Check if we have concatenated files */
200 if( stream_Peek( p_demux->s, &p_peek, 16 ) == 16 )
201 {
202 guid_t guid;
203
204 ASF_GetGUID( &guid, p_peek );
205 if( guidcmp( &guid, &asf_object_header_guid ) )
206 {
207 msg_Warn( p_demux, "found a new ASF header" );
208 /* We end this stream */
209 DemuxEnd( p_demux );
210
211 /* And we prepare to read the next one */
212 if( DemuxInit( p_demux ) )
213 {
214 msg_Err( p_demux, "failed to load the new header" );
215 dialog_Fatal( p_demux, _("Could not demux ASF stream"), "%s",
216 _("VLC failed to load the ASF header.") );
217 return 0;
218 }
219 es_out_Control( p_demux->out, ES_OUT_RESET_PCR );
220 continue;
221 }
222 }
223
224 /* Read and demux a packet */
225 if( ( i_result = DemuxPacket( p_demux ) ) <= 0 )
226 {
227 FlushRemainingPackets( p_demux );
228 return i_result;
229 }
230 if( i_time_begin == -1 )
231 {
232 i_time_begin = GetMoviePTS( p_sys );
233 }
234 else
235 {
236 i_length = GetMoviePTS( p_sys ) - i_time_begin;
237 if( i_length < 0 || i_length >= 40 * 1000 ) break;
238 }
239 }
240
241 /* Set the PCR */
242 /* WARN: Don't move it before the end of the whole chunk */
243 p_sys->i_time = GetMoviePTS( p_sys );
244 if( p_sys->i_time >= 0 )
245 {
246#ifdef ASF_DEBUG
247 msg_Dbg( p_demux, "Demux Loop Setting PCR to %"PRId64, VLC_TS_0 + p_sys->i_time );
248#endif
249 es_out_Control( p_demux->out, ES_OUT_SET_PCR, VLC_TS_0 + p_sys->i_time );
250 }
251
252 return 1;
253}

267/*****************************************************************************
268 * WaitKeyframe: computes the number of frames to wait for a keyframe
269 *****************************************************************************/
270static void WaitKeyframe( demux_t *p_demux )
271{
272 demux_sys_t *p_sys = p_demux->p_sys;
273 if ( ! p_sys->i_seek_track )
274 {
275 for ( int i=0; i<MAX_ASF_TRACKS; i++ )
276 {
277 asf_track_t *tk = p_sys->track[i];
278 if ( tk && tk->p_sp && tk->i_cat == VIDEO_ES && tk->b_selected )
279 {
280 p_sys->i_seek_track = tk->p_sp->i_stream_number;
281 break;
282 }
283 }
284 }
285
286 if ( p_sys->i_seek_track )
287 {
288 /* Skip forward at least 1 min */
289 asf_track_t *tk = p_sys->track[p_sys->i_seek_track];
290 if ( tk->p_esp && tk->p_esp->i_average_time_per_frame )
291 {
292 /* 1 min if fastseek, otherwise 5 sec */
293 /* That's a guess for bandwidth */
294 uint64_t i_maxwaittime = ( p_sys->b_canfastseek ) ? 600000000 : 50000000;
295 i_maxwaittime /= tk->p_esp->i_average_time_per_frame;
296 p_sys->i_wait_keyframe = __MIN( i_maxwaittime, UINT_MAX );
297 }
298 else
299 {
300 p_sys->i_wait_keyframe = ( p_sys->b_canfastseek ) ? 25 * 30 : 25 * 5;
301 }
302 }
303 else
304 {
305 p_sys->i_wait_keyframe = 0;
306 }
307
308}

327static int SeekIndex( demux_t *p_demux, mtime_t i_date, float f_pos )
328{
329 demux_sys_t *p_sys = p_demux->p_sys;
330 asf_object_index_t *p_index;
331
332 msg_Dbg( p_demux, "seek with index: %i seconds, position %f",
333 i_date >= 0 ? (int)(i_date/1000000) : -1, f_pos );
334
335 if( i_date < 0 )
336 i_date = p_sys->i_length * f_pos;
337
338 p_sys->i_preroll_start = i_date - (int64_t) p_sys->p_fp->i_preroll;
339 if ( p_sys->i_preroll_start < 0 ) p_sys->i_preroll_start = 0;
340
341 p_index = ASF_FindObject( p_sys->p_root, &asf_object_simple_index_guid, 0 );
342
343 uint64_t i_entry = p_sys->i_preroll_start * 10 / p_index->i_index_entry_time_interval;
344 if( i_entry >= p_index->i_index_entry_count )
345 {
346 msg_Warn( p_demux, "Incomplete index" );
347 return VLC_EGENERIC;
348 }
349
350 WaitKeyframe( p_demux );
351
352 uint64_t i_offset = (uint64_t)p_index->index_entry[i_entry].i_packet_number *
353 p_sys->p_fp->i_min_data_packet_size;
354
355 if ( stream_Seek( p_demux->s, i_offset + p_sys->i_data_begin ) == VLC_SUCCESS )
356 {
357 es_out_Control( p_demux->out, ES_OUT_SET_NEXT_DISPLAY_TIME, VLC_TS_0 + i_date );
358 return VLC_SUCCESS;
359 }
360 else return VLC_EGENERIC;
361}

363static void SeekPrepare( demux_t *p_demux )
364{
365 demux_sys_t *p_sys = p_demux->p_sys;
366
367 p_sys->i_time = VLC_TS_INVALID;
368 p_sys->i_preroll_start = ASF_PREROLL_FROM_CURRENT;
369 for( int i = 0; i < MAX_ASF_TRACKS ; i++ )
370 {
371 asf_track_t *tk = p_sys->track[i];
372 if( !tk )
373 continue;
374
375 tk->i_time = -1;
376 if( tk->p_frame )
377 block_ChainRelease( tk->p_frame );
378 tk->p_frame = NULL;
379 }
380
381 es_out_Control( p_demux->out, ES_OUT_RESET_PCR );
382}

384/*****************************************************************************
385 * Control:
386 *****************************************************************************/
387static int Control( demux_t *p_demux, int i_query, va_list args )
388{
389 demux_sys_t *p_sys = p_demux->p_sys;
390 vlc_meta_t *p_meta;
391 int64_t i64, *pi64;
392 int i;
393 double f, *pf;
394
395 switch( i_query )
396 {
397 case DEMUX_GET_LENGTH:
398 pi64 = (int64_t*)va_arg( args, int64_t * );
399 *pi64 = p_sys->i_length;
400 return VLC_SUCCESS;
401
402 case DEMUX_GET_TIME:
403 pi64 = (int64_t*)va_arg( args, int64_t * );
404 if( p_sys->i_time < 0 ) return VLC_EGENERIC;
405 *pi64 = p_sys->i_time;
406 return VLC_SUCCESS;
407
408 case DEMUX_SET_TIME:
409 if ( p_sys->p_fp &&
410 ! ( p_sys->p_fp->i_flags & ASF_FILE_PROPERTIES_SEEKABLE ) )
411 return VLC_EGENERIC;
412
413 SeekPrepare( p_demux );
414
415 if( p_sys->b_index && p_sys->i_length > 0 )
416 {
417 va_list acpy;
418 va_copy( acpy, args );
419 i64 = (int64_t)va_arg( acpy, int64_t );
420 va_end( acpy );
421
422 if( !SeekIndex( p_demux, i64, -1 ) )
423 return VLC_SUCCESS;
424 }
425 return SeekPercent( p_demux, i_query, args );
426
427 case DEMUX_SET_ES:
428 {
429 i = (int)va_arg( args, int );
430 int i_ret;
431 if ( i >= 0 )
432 {
433 i++; /* video/audio-es variable starts 0 */
434 msg_Dbg( p_demux, "Requesting access to enable stream %d", i );
435 i_ret = stream_Control( p_demux->s, STREAM_SET_PRIVATE_ID_STATE, i, true );
436 }
437 else
438 { /* i contains -1 * es_category */
439 msg_Dbg( p_demux, "Requesting access to disable stream %d", i );
440 i_ret = stream_Control( p_demux->s, STREAM_SET_PRIVATE_ID_STATE, i, false );
441 }
442
443 if ( i_ret == VLC_SUCCESS )
444 {
445 SeekPrepare( p_demux );
446 p_sys->i_seek_track = 0;
447 WaitKeyframe( p_demux );
448 }
449 return i_ret;
450 }
451
452 case DEMUX_GET_POSITION:
453 if( p_sys->i_time < 0 ) return VLC_EGENERIC;
454 if( p_sys->i_length > 0 )
455 {
456 pf = (double*)va_arg( args, double * );
457 *pf = p_sys->i_time / (double)p_sys->i_length;
458 return VLC_SUCCESS;
459 }
460 return demux_vaControlHelper( p_demux->s,
461 __MIN( INT64_MAX, p_sys->i_data_begin ),
462 __MIN( INT64_MAX, p_sys->i_data_end ),
463 __MIN( INT64_MAX, p_sys->i_bitrate ),
464 __MIN( INT16_MAX, p_sys->p_fp->i_min_data_packet_size ),
465 i_query, args );
466
467 case DEMUX_SET_POSITION:
468 if ( p_sys->p_fp &&
469 ! ( p_sys->p_fp->i_flags & ASF_FILE_PROPERTIES_SEEKABLE ) )
470 return VLC_EGENERIC;
471
472 SeekPrepare( p_demux );
473
474 if( p_sys->b_index && p_sys->i_length > 0 )
475 {
476 va_list acpy;
477 va_copy( acpy, args );
478 f = (double)va_arg( acpy, double );
479 va_end( acpy );
480
481 if( !SeekIndex( p_demux, -1, f ) )
482 return VLC_SUCCESS;
483 }
484 return SeekPercent( p_demux, i_query, args );
485
486 case DEMUX_GET_META:
487 p_meta = (vlc_meta_t*)va_arg( args, vlc_meta_t* );
488 vlc_meta_Merge( p_meta, p_sys->meta );
489 return VLC_SUCCESS;
490
491 case DEMUX_CAN_SEEK:
492 if ( p_sys->p_fp &&
493 ! ( p_sys->p_fp->i_flags & ASF_FILE_PROPERTIES_SEEKABLE ) )
494 {
495 bool *pb_bool = (bool*)va_arg( args, bool * );
496 *pb_bool = false;
497 return VLC_SUCCESS;
498 }
499 // ft
500
501 default:
502 return demux_vaControlHelper( p_demux->s,
503 __MIN( INT64_MAX, p_sys->i_data_begin ),
504 __MIN( INT64_MAX, p_sys->i_data_end),
505 __MIN( INT64_MAX, p_sys->i_bitrate ),
506 ( p_sys->p_fp ) ? __MIN( INT_MAX, p_sys->p_fp->i_min_data_packet_size ) : 1,
507 i_query, args );
508 }
509}

511/*****************************************************************************
512 *
513 *****************************************************************************/
514static mtime_t GetMoviePTS( demux_sys_t *p_sys )
515{
516 mtime_t i_time = -1;
517 int i;
518 /* As some tracks might have been deselected by access, the PCR might
519 * stop updating */
520 for( i = 0; i < MAX_ASF_TRACKS ; i++ )
521 {
522 const asf_track_t *tk = p_sys->track[i];
523
524 if( tk && tk->p_es && tk->b_selected )
525 {
526 /* Skip discrete tracks */
527 if ( tk->i_cat != VIDEO_ES && tk->i_cat != AUDIO_ES )
528 continue;
529
530 /* We need to have all ES seen once, as they might have lower DTS */
531 if ( tk->i_time + (int64_t)p_sys->p_fp->i_preroll * 1000 < 0 )
532 {
533 /* early fail */
534 return -1;
535 }
536 else if ( tk->i_time > -1 && ( i_time == -1 || i_time > tk->i_time ) )
537 {
538 i_time = tk->i_time;
539 }
540 }
541 }
542
543 return i_time;
544}

546static inline int GetValue2b(uint32_t *var, const uint8_t *p, unsigned int *skip, int left, int bits)
547{
548 switch(bits&0x03)
549 {
550 case 1:
551 if (left < 1)
552 return -1;
553 *var = p[*skip]; *skip += 1;
554 return 0;
555 case 2:
556 if (left < 2)
557 return -1;
558 *var = GetWLE(&p[*skip]); *skip += 2;
559 return 0;
560 case 3:
561 if (left < 4)
562 return -1;
563 *var = GetDWLE(&p[*skip]); *skip += 4;
564 return 0;
565 case 0:
566 default:
567 return 0;
568 }
569}

586static void SendPacket(demux_t *p_demux, asf_track_t *tk)
587{
588 demux_sys_t *p_sys = p_demux->p_sys;
589
590 block_t *p_gather = block_ChainGather( tk->p_frame );
591
592 if( p_sys->i_time < VLC_TS_0 && tk->i_time > VLC_TS_INVALID )
593 {
594 p_sys->i_time = tk->i_time;
595 es_out_Control( p_demux->out, ES_OUT_SET_PCR, VLC_TS_0 + p_sys->i_time );
596#ifdef ASF_DEBUG
597 msg_Dbg( p_demux, " setting PCR to %"PRId64, VLC_TS_0 + p_sys->i_time );
598#endif
599 }
600
601#ifdef ASF_DEBUG
602 msg_Dbg( p_demux, " sending packet dts %"PRId64" pts %"PRId64" pcr %"PRId64, p_gather->i_dts, p_gather->i_pts, p_sys->i_time );
603#endif
604 es_out_Send( p_demux->out, tk->p_es, p_gather );
605 tk->p_frame = NULL;
606}

608static int DemuxSubPayload(demux_t *p_demux, asf_track_t *tk,
609 uint32_t i_sub_payload_data_length, mtime_t i_pts, mtime_t i_dts,
610 uint32_t i_media_object_offset, bool b_keyframe )
611{
612 /* FIXME I don't use i_media_object_number, sould I ? */
613 if( tk->p_frame && i_media_object_offset == 0 )
614 SendPacket(p_demux, tk);
615
616 block_t *p_frag = stream_Block( p_demux->s, i_sub_payload_data_length );
617 if( p_frag == NULL ) {
618 msg_Warn( p_demux, "cannot read data" );
619 return -1;
620 }
621
622 p_frag->i_pts = VLC_TS_0 + i_pts;
623 p_frag->i_dts = VLC_TS_0 + i_dts;
624 if ( b_keyframe )
625 p_frag->i_flags |= BLOCK_FLAG_TYPE_I;
626
627 block_ChainAppend( &tk->p_frame, p_frag );
628
629 return 0;
630}

632static uint32_t SkipBytes( stream_t *s, uint32_t i_bytes )
633{
634 int i_read;
635 int i_to_read = __MIN(i_bytes, INT_MAX);
636 uint32_t i_bytes_read = 0;
637
638 while( i_bytes )
639 {
640 i_read = stream_Read( s, NULL, i_to_read );
641 i_bytes -= i_read;
642 i_bytes_read += i_read;
643 if ( i_read < i_to_read || i_bytes == 0 )
644 {
645 /* end of stream */
646 return i_bytes_read;
647 }
648 i_to_read = __MIN(i_bytes, INT_MAX);
649 }
650
651 return i_bytes_read;
652}

654static void ParsePayloadExtensions(demux_t *p_demux, asf_track_t *tk,
655 const struct asf_packet_t *pkt,
656 uint32_t i_length, bool *b_keyframe )
657{
658 if ( !tk || !tk->p_esp || !tk->p_esp->p_ext ) return;
659 const uint8_t *p_data = pkt->p_peek + pkt->i_skip + 8;
660 i_length -= 8;
661 uint16_t i_payload_extensions_size;
662 asf_payload_extension_system_t *p_ext = NULL;
663
664 /* Extensions always come in the declared order */
665 for ( int i=0; i< tk->p_esp->i_payload_extension_system_count; i++ )
666 {
667 p_ext = &tk->p_esp->p_ext[i];
668 if ( p_ext->i_data_size == 0xFFFF ) /* Variable length extension data */
669 {
670 if ( i_length < 2 ) return;
671 i_payload_extensions_size = GetWLE( p_data );
672 p_data += 2;
673 i_length -= 2;
674 i_payload_extensions_size = 0;
675 }
676 else
677 {
678 i_payload_extensions_size = p_ext->i_data_size;
679 }
680
681 if ( i_length < i_payload_extensions_size ) return;
682
683 if ( guidcmp( &p_ext->i_extension_id, &mfasf_sampleextension_outputcleanpoint_guid ) )
684 {
685 if ( i_payload_extensions_size != sizeof(uint8_t) ) goto sizeerror;
686 *b_keyframe |= *p_data;
687 }
688 else if ( guidcmp( &p_ext->i_extension_id, &asf_dvr_sampleextension_videoframe_guid ) )
689 {
690 if ( i_payload_extensions_size != sizeof(uint32_t) ) goto sizeerror;
691
692 uint32_t i_val = GetDWLE( p_data );
693 /* Valid keyframe must be a split frame start fragment */
694 *b_keyframe = i_val & ASF_EXTENSION_VIDEOFRAME_NEWFRAME;
695 if ( *b_keyframe )
696 {
697 /* And flagged as IFRAME */
698 *b_keyframe |= ( ( i_val & ASF_EXTENSION_VIDEOFRAME_TYPE_MASK )
699 == ASF_EXTENSION_VIDEOFRAME_IFRAME );
700 }
701 }
702 else if ( guidcmp( &p_ext->i_extension_id, &mfasf_sampleextension_pixelaspectratio_guid ) )
703 {
704 if ( i_payload_extensions_size != sizeof(uint16_t) ) goto sizeerror;
705
706 uint8_t i_ratio_x = *p_data;
707 uint8_t i_ratio_y = *(p_data + 1);
708 if ( tk->p_fmt->video.i_sar_num != i_ratio_x || tk->p_fmt->video.i_sar_den != i_ratio_y )
709 {
710 /* Only apply if origin pixel size >= 1x1, due to broken yacast */
711 if ( tk->p_fmt->video.i_height * i_ratio_x > tk->p_fmt->video.i_width * i_ratio_y )
712 {
713 vout_thread_t *p_vout = input_GetVout( p_demux->p_input );
714 if ( p_vout )
715 {
716 msg_Info( p_demux, "Changing aspect ratio to %i/%i", i_ratio_x, i_ratio_y );
717 vout_ChangeAspectRatio( p_vout, i_ratio_x, i_ratio_y );
718 vlc_object_release( p_vout );
719 }
720 }
721 tk->p_fmt->video.i_sar_num = i_ratio_x;
722 tk->p_fmt->video.i_sar_den = i_ratio_y;
723 }
724 }
725 i_length -= i_payload_extensions_size;
726 p_data += i_payload_extensions_size;
727 }
728
729 return;
730
731sizeerror:
732 msg_Warn( p_demux, "Unknown extension " GUID_FMT " data size of %u",
733 GUID_PRINT( p_ext->i_extension_id ), i_payload_extensions_size );
734}

736static int DemuxPayload(demux_t *p_demux, struct asf_packet_t *pkt, int i_payload)
737{
738#ifndef ASF_DEBUG
739 VLC_UNUSED( i_payload );
740#endif
741 demux_sys_t *p_sys = p_demux->p_sys;
742
743 if( ! pkt->left || pkt->i_skip >= pkt->left )
744 return -1;
745
746 bool b_packet_keyframe = pkt->p_peek[pkt->i_skip] >> 7;
747 uint8_t i_stream_number = pkt->p_peek[pkt->i_skip++] & 0x7f;
748
749 uint32_t i_media_object_number = 0;
750 if (GetValue2b(&i_media_object_number, pkt->p_peek, &pkt->i_skip, pkt->left - pkt->i_skip, pkt->property >> 4) < 0)
751 return -1;
752 uint32_t i_media_object_offset = 0;
753 if (GetValue2b(&i_media_object_offset, pkt->p_peek, &pkt->i_skip, pkt->left - pkt->i_skip, pkt->property >> 2) < 0)
754 return -1;
755 uint32_t i_replicated_data_length = 0;
756 if (GetValue2b(&i_replicated_data_length, pkt->p_peek, &pkt->i_skip, pkt->left - pkt->i_skip, pkt->property) < 0)
757 return -1;
758
759 mtime_t i_base_pts;
760 uint8_t i_pts_delta = 0;
761 uint32_t i_payload_data_length = 0;
762 uint32_t i_temp_payload_length = 0;
763 p_sys->p_fp->i_preroll = __MIN( p_sys->p_fp->i_preroll, INT64_MAX );
764
765 /* First packet, in case we do not have index to guess preroll start time */
766 if ( p_sys->i_preroll_start == ASF_PREROLL_FROM_CURRENT )
767 p_sys->i_preroll_start = pkt->send_time * 1000;
768
769 /* Non compressed */
770 if( i_replicated_data_length > 7 ) // should be at least 8 bytes
771 {
772 /* Followed by 2 optional DWORDS, offset in media and presentation time */
773 i_base_pts = (mtime_t)GetDWLE( pkt->p_peek + pkt->i_skip + 4 );
774
775 /* Parsing extensions, See 7.3.1 */
776 ParsePayloadExtensions( p_demux, p_sys->track[i_stream_number], pkt,
777 i_replicated_data_length, &b_packet_keyframe );
778 i_base_pts -= p_sys->p_fp->i_preroll;
779 pkt->i_skip += i_replicated_data_length;
780
781 if( ! pkt->left || pkt->i_skip >= pkt->left )
782 return -1;
783 }
784 else if ( i_replicated_data_length == 0 )
785 {
786 /* optional DWORDS missing */
787 i_base_pts = (mtime_t)pkt->send_time;
788 i_base_pts -= p_sys->p_fp->i_preroll;
789 }
790 /* Compressed payload */
791 else if( i_replicated_data_length == 1 )
792 {
793 /* i_media_object_offset is presentation time */
794 /* Next byte is Presentation Time Delta */
795 i_pts_delta = pkt->p_peek[pkt->i_skip];
796 i_base_pts = (mtime_t)i_media_object_offset;
797 i_base_pts -= p_sys->p_fp->i_preroll;
798 pkt->i_skip++;
799 i_media_object_offset = 0;
800 }
801 else
802 {
803 /* >1 && <8 Invalid replicated length ! */
804 msg_Warn( p_demux, "Invalid replicated data length detected." );
805 i_payload_data_length = pkt->length - pkt->padding_length - pkt->i_skip;
806 goto skip;
807 }
808
809 bool b_preroll_done = ( pkt->send_time > (p_sys->i_preroll_start/1000 + p_sys->p_fp->i_preroll) );
810
811 if (i_base_pts < 0) i_base_pts = 0; // FIXME?
812 i_base_pts *= 1000;
813
814 if( pkt->multiple ) {
815 if (GetValue2b(&i_temp_payload_length, pkt->p_peek, &pkt->i_skip, pkt->left - pkt->i_skip, pkt->length_type) < 0)
816 return -1;
817 } else
818 i_temp_payload_length = pkt->length - pkt->padding_length - pkt->i_skip;
819
820 i_payload_data_length = i_temp_payload_length;
821
822#ifdef ASF_DEBUG
823 msg_Dbg( p_demux,
824 "payload(%d) stream_number:%"PRIu8" media_object_number:%d media_object_offset:%"PRIu32" replicated_data_length:%"PRIu32" payload_data_length %"PRIu32,
825 i_payload + 1, i_stream_number, i_media_object_number,
826 i_media_object_offset, i_replicated_data_length, i_payload_data_length );
827 msg_Dbg( p_demux,
828 " pts=%"PRId64" st=%"PRIu32, i_base_pts, pkt->send_time );
829#endif
830
831 if( ! i_payload_data_length || i_payload_data_length > pkt->left )
832 {
833 msg_Dbg( p_demux, " payload length problem %d %"PRIu32" %"PRIu32, pkt->multiple, i_payload_data_length, pkt->left );
834 return -1;
835 }
836
837 asf_track_t *tk = p_sys->track[i_stream_number];
838 if( tk == NULL )
839 {
840 msg_Warn( p_demux, "undeclared stream[Id 0x%x]", i_stream_number );
841 goto skip;
842 }
843
844 if( p_sys->i_wait_keyframe )
845 {
846 if ( i_stream_number == p_sys->i_seek_track )
847 {
848 if ( !b_packet_keyframe )
849 {
850 p_sys->i_wait_keyframe--;
851 goto skip;
852 }
853 else
854 p_sys->i_wait_keyframe = 0;
855 }
856 else
857 goto skip;
858 }
859
860 if( !tk->p_es )
861 goto skip;
862
863 bool b_hugedelay = ( p_sys->p_fp->i_preroll * 1000 > CLOCK_FREQ * 3 );
864
865 if ( b_preroll_done || b_hugedelay )
866 {
867 if ( !b_hugedelay )
868 {
869 tk->i_time = INT64_C(1000) * pkt->send_time;
870 tk->i_time -= p_sys->p_fp->i_preroll * 1000;
871 tk->i_time -= tk->p_sp->i_time_offset * 10;
872 }
873 else
874 tk->i_time = i_base_pts;
875 }
876
877 uint32_t i_subpayload_count = 0;
878 while (i_payload_data_length)
879 {
880 uint32_t i_sub_payload_data_length = i_payload_data_length;
881 if( i_replicated_data_length == 1 )
882 {
883 i_sub_payload_data_length = pkt->p_peek[pkt->i_skip++];
884 i_payload_data_length--;
885 }
886
887 SkipBytes( p_demux->s, pkt->i_skip );
888
889 mtime_t i_payload_pts = i_base_pts + (mtime_t)i_pts_delta * i_subpayload_count * 1000;
890 i_payload_pts -= tk->p_sp->i_time_offset * 10;
891 mtime_t i_payload_dts = INT64_C(1000) * pkt->send_time;
892 i_payload_dts -= p_sys->p_fp->i_preroll * 1000;
893 i_payload_dts -= tk->p_sp->i_time_offset * 10;
894
895 if ( b_hugedelay )
896 i_payload_dts = i_base_pts;
897
898 if ( i_sub_payload_data_length &&
899 DemuxSubPayload(p_demux, tk, i_sub_payload_data_length,
900 i_payload_pts, i_payload_dts, i_media_object_offset,
901 b_packet_keyframe ) < 0)
902 return -1;
903
904 if ( pkt->left > pkt->i_skip + i_sub_payload_data_length )
905 pkt->left -= pkt->i_skip + i_sub_payload_data_length;
906 else
907 pkt->left = 0;
908 pkt->i_skip = 0;
909 if( pkt->left > 0 )
910 {
911 int i_return = stream_Peek( p_demux->s, &pkt->p_peek, __MIN(pkt->left, INT_MAX) );
912 if ( i_return <= 0 || (unsigned int) i_return < __MIN(pkt->left, INT_MAX) )
913 {
914 msg_Warn( p_demux, "cannot peek, EOF ?" );
915 return -1;
916 }
917 }
918
919 if ( i_sub_payload_data_length <= i_payload_data_length )
920 i_payload_data_length -= i_sub_payload_data_length;
921 else
922 i_payload_data_length = 0;
923
924 i_subpayload_count++;
925 }
926
927 return 0;
928
929skip:
930 pkt->i_skip += i_payload_data_length;
931 return 0;
932}

934static int DemuxPacket( demux_t *p_demux )
935{
936 demux_sys_t *p_sys = p_demux->p_sys;
937
938 uint32_t i_data_packet_min = p_sys->p_fp->i_min_data_packet_size;
939
940 const uint8_t *p_peek;
941 int i_return = stream_Peek( p_demux->s, &p_peek,i_data_packet_min );
942 if( i_return <= 0 || ((unsigned int) i_return) < i_data_packet_min )
943 {
944 msg_Warn( p_demux, "cannot peek while getting new packet, EOF ?" );
945 return 0;
946 }
947 unsigned int i_skip = 0;
948
949 /* *** parse error correction if present *** */
950 if( p_peek[0]&0x80 )
951 {
952 unsigned int i_error_correction_data_length = p_peek[0] & 0x0f;
953 unsigned int i_opaque_data_present = ( p_peek[0] >> 4 )& 0x01;
954 unsigned int i_error_correction_length_type = ( p_peek[0] >> 5 ) & 0x03;
955 i_skip += 1; // skip error correction flags
956
957 if( i_error_correction_length_type != 0x00 ||
958 i_opaque_data_present != 0 ||
959 i_error_correction_data_length != 0x02 )
960 {
961 goto loop_error_recovery;
962 }
963
964 i_skip += i_error_correction_data_length;
965 }
966 else
967 msg_Warn( p_demux, "no error correction" );
968
969 /* sanity check */
970 if( i_skip + 2 >= i_data_packet_min )
971 goto loop_error_recovery;
972
973 struct asf_packet_t pkt;
974 int i_packet_flags = p_peek[i_skip]; i_skip++;
975 pkt.property = p_peek[i_skip]; i_skip++;
976 pkt.multiple = !!(i_packet_flags&0x01);
977
978 pkt.length = i_data_packet_min;
979 pkt.padding_length = 0;
980
981 if (GetValue2b(&pkt.length, p_peek, &i_skip, i_data_packet_min - i_skip, i_packet_flags >> 5) < 0)
982 goto loop_error_recovery;
983 uint32_t i_packet_sequence;
984 if (GetValue2b(&i_packet_sequence, p_peek, &i_skip, i_data_packet_min - i_skip, i_packet_flags >> 1) < 0)
985 goto loop_error_recovery;
986 if (GetValue2b(&pkt.padding_length, p_peek, &i_skip, i_data_packet_min - i_skip, i_packet_flags >> 3) < 0)
987 goto loop_error_recovery;
988
989 if( pkt.padding_length > pkt.length )
990 {
991 msg_Warn( p_demux, "Too large padding: %"PRIu32, pkt.padding_length );
992 goto loop_error_recovery;
993 }
994
995 if( pkt.length < i_data_packet_min )
996 {
997 /* if packet length too short, there is extra padding */
998 pkt.padding_length += i_data_packet_min - pkt.length;
999 pkt.length = i_data_packet_min;
1000 }
1001
1002 pkt.send_time = GetDWLE( p_peek + i_skip ); i_skip += 4;
1003 /* uint16_t i_packet_duration = GetWLE( p_peek + i_skip ); */ i_skip += 2;
1004
1005 i_return = stream_Peek( p_demux->s, &p_peek, pkt.length );
1006 if( i_return <= 0 || pkt.length == 0 || (unsigned int)i_return < pkt.length )
1007 {
1008 msg_Warn( p_demux, "cannot peek, EOF ?" );
1009 return 0;
1010 }
1011
1012 int i_payload_count = 1;
1013 pkt.length_type = 0x02; //unused
1014 if( pkt.multiple )
1015 {
1016 i_payload_count = p_peek[i_skip] & 0x3f;
1017 pkt.length_type = ( p_peek[i_skip] >> 6 )&0x03;
1018 i_skip++;
1019 }
1020
1021#ifdef ASF_DEBUG
1022 msg_Dbg(p_demux, "%d payloads", i_payload_count);
1023#endif
1024
1025 pkt.i_skip = i_skip;
1026 pkt.p_peek = p_peek;
1027 pkt.left = pkt.length;
1028
1029 for( int i_payload = 0; i_payload < i_payload_count ; i_payload++ )
1030 if (DemuxPayload(p_demux, &pkt, i_payload) < 0)
1031 {
1032 msg_Warn( p_demux, "payload err %d / %d", i_payload + 1, i_payload_count );
1033 return 0;
1034 }
1035
1036 if( pkt.left > 0 )
1037 {
1038#ifdef ASF_DEBUG
1039 if( pkt.left > pkt.padding_length )
1040 msg_Warn( p_demux, "Didn't read %"PRIu32" bytes in the packet",
1041 pkt.left - pkt.padding_length );
1042 else if( pkt.left < pkt.padding_length )
1043 msg_Warn( p_demux, "Read %"PRIu32" too much bytes in the packet",
1044 pkt.padding_length - pkt.left );
1045#endif
1046 int i_return = stream_Read( p_demux->s, NULL, pkt.left );
1047 if( i_return < 0 || (unsigned int) i_return < pkt.left )
1048 {
1049 msg_Err( p_demux, "cannot skip data, EOF ?" );
1050 return 0;
1051 }
1052 }
1053
1054 return 1;
1055
1056loop_error_recovery:
1057 msg_Warn( p_demux, "unsupported packet header" );
1058 if( p_sys->p_fp->i_min_data_packet_size != p_sys->p_fp->i_max_data_packet_size )
1059 {
1060 msg_Err( p_demux, "unsupported packet header, fatal error" );
1061 return -1;
1062 }
1063 i_return = stream_Read( p_demux->s, NULL, i_data_packet_min );
1064 if( i_return <= 0 || (unsigned int) i_return != i_data_packet_min )
1065 {
1066 msg_Warn( p_demux, "cannot skip data, EOF ?" );
1067 return 0;
1068 }
1069
1070 return 1;
1071}

1225 ASF_fillup_es_bitrate_priorities_ex( p_sys, p_hdr_ext, &fmt_priorities_bitrate_ex );
1226 }
1227
1228 for( unsigned i_stream = 0; i_stream < p_sys->i_track; i_stream++ )
1229 {
1230 asf_track_t *tk;
1231 asf_object_stream_properties_t *p_sp;
1232 asf_object_extended_stream_properties_t *p_esp;
1233 bool b_access_selected;
1234
1235 p_sp = ASF_FindObject( p_sys->p_root->p_hdr,
1236 &asf_object_stream_properties_guid,
1237 i_stream );
1238 p_esp = NULL;
1239
1240 tk = p_sys->track[p_sp->i_stream_number] = malloc( sizeof( asf_track_t ) );
1241 memset( tk, 0, sizeof( asf_track_t ) );
1242
1243 tk->i_time = -1;
1244 tk->p_sp = p_sp;
1245 tk->p_es = NULL;
1246 tk->p_esp = NULL;
1247 tk->p_frame = NULL;
1248
1249 if ( strncmp( p_demux->psz_access, "mms", 3 ) )
1250 {
1251 /* Check (not mms) if this track is selected (ie will receive data) */
1252 if( !stream_Control( p_demux->s, STREAM_GET_PRIVATE_ID_STATE,
1253 (int) p_sp->i_stream_number, &b_access_selected ) &&
1254 !b_access_selected )
1255 {
1256 tk->i_cat = UNKNOWN_ES;
1257 msg_Dbg( p_demux, "ignoring not selected stream(ID:%u) (by access)",
1258 p_sp->i_stream_number );
1259 continue;
1260 }
1261 }
1262
1263 /* Find the associated extended_stream_properties if any */
1264 if( p_hdr_ext )
1265 {
1266 int i_ext_stream = ASF_CountObject( p_hdr_ext,
1267 &asf_object_extended_stream_properties_guid );
1268 for( int i = 0; i < i_ext_stream; i++ )
1269 {
1270 asf_object_t *p_tmp =
1271 ASF_FindObject( p_hdr_ext,
1272 &asf_object_extended_stream_properties_guid, i );
1273 if( p_tmp->ext_stream.i_stream_number == p_sp->i_stream_number )
1274 {
1275 p_esp = &p_tmp->ext_stream;
1276 tk->p_esp = p_esp;
1277 break;
1278 }
1279 }
1280 }
1281
1282 es_format_t fmt;
1283
1284 if( guidcmp( &p_sp->i_stream_type, &asf_object_stream_type_audio ) &&
1285 p_sp->i_type_specific_data_length >= sizeof( WAVEFORMATEX ) - 2 )
1286 {
1287 uint8_t *p_data = p_sp->p_type_specific_data;
1288 int i_format;
1289
1290 es_format_Init( &fmt, AUDIO_ES, 0 );
1291 i_format = GetWLE( &p_data[0] );
1292 wf_tag_to_fourcc( i_format, &fmt.i_codec, NULL );
1293
1294 GET_CHECKED( fmt.audio.i_channels, GetWLE( &p_data[2] ),
1295 255, uint16_t );
1296 GET_CHECKED( fmt.audio.i_rate, GetDWLE( &p_data[4] ),
1297 UINT_MAX, uint32_t );
1298 GET_CHECKED( fmt.i_bitrate, GetDWLE( &p_data[8] ) * 8,
1299 UINT_MAX, uint32_t );
1300 fmt.audio.i_blockalign = GetWLE( &p_data[12] );
1301 fmt.audio.i_bitspersample = GetWLE( &p_data[14] );
1302
1303 if( p_sp->i_type_specific_data_length > sizeof( WAVEFORMATEX ) &&
1304 i_format != WAVE_FORMAT_MPEGLAYER3 &&
1305 i_format != WAVE_FORMAT_MPEG )
1306 {
1307 GET_CHECKED( fmt.i_extra, __MIN( GetWLE( &p_data[16] ),
1308 p_sp->i_type_specific_data_length -
1309 sizeof( WAVEFORMATEX ) ),
1310 INT_MAX, uint32_t );
1311 fmt.p_extra = malloc( fmt.i_extra );
1312 memcpy( fmt.p_extra, &p_data[sizeof( WAVEFORMATEX )],
1313 fmt.i_extra );
1314 }
1315
1316 msg_Dbg( p_demux, "added new audio stream(codec:0x%x,ID:%d)",
1317 GetWLE( p_data ), p_sp->i_stream_number );
1318 }
1319 else if( guidcmp( &p_sp->i_stream_type,
1320 &asf_object_stream_type_video ) &&
1321 p_sp->i_type_specific_data_length >= 11 +
1322 sizeof( VLC_BITMAPINFOHEADER ) )
1323 {
1324 uint8_t *p_data = &p_sp->p_type_specific_data[11];
1325
1326 es_format_Init( &fmt, VIDEO_ES,
1327 VLC_FOURCC( p_data[16], p_data[17],
1328 p_data[18], p_data[19] ) );
1329
1330 GET_CHECKED( fmt.video.i_width, GetDWLE( p_data + 4 ),
1331 UINT_MAX, uint32_t );
1332 GET_CHECKED( fmt.video.i_height, GetDWLE( p_data + 8 ),
1333 UINT_MAX, uint32_t );
1334
1335 if( p_esp && p_esp->i_average_time_per_frame > 0 )
1336 {
1337 fmt.video.i_frame_rate = 10000000;
1338 GET_CHECKED( fmt.video.i_frame_rate_base,
1339 p_esp->i_average_time_per_frame,
1340 UINT_MAX, uint64_t );
1341 }
1342
1343 if( fmt.i_codec == VLC_FOURCC( 'D','V','R',' ') )
1344 {
1345 /* DVR-MS special ASF */
1346 fmt.i_codec = VLC_FOURCC( 'm','p','g','2' ) ;
1347 fmt.b_packetized = false;
1348 }
1349
1350 if( p_sp->i_type_specific_data_length > 11 +
1351 sizeof( VLC_BITMAPINFOHEADER ) )
1352 {
1353 GET_CHECKED( fmt.i_extra, __MIN( GetDWLE( p_data ),
1354 p_sp->i_type_specific_data_length - 11 -
1355 sizeof( VLC_BITMAPINFOHEADER ) ),
1356 UINT_MAX, uint32_t );
1357 fmt.p_extra = malloc( fmt.i_extra );
1358 memcpy( fmt.p_extra, &p_data[sizeof( VLC_BITMAPINFOHEADER )],
1359 fmt.i_extra );
1360 }
1361
1362 /* Look for an aspect ratio */
1363 if( p_sys->p_root->p_metadata )
1364 {
1365 asf_object_metadata_t *p_meta = p_sys->p_root->p_metadata;
1366 unsigned int i_aspect_x = 0, i_aspect_y = 0;
1367 uint32_t i;
1368 for( i = 0; i < p_meta->i_record_entries_count; i++ )
1369 {
1370 if( !strcmp( p_meta->record[i].psz_name, "AspectRatioX" ) )
1371 {
1372 if( (!i_aspect_x && !p_meta->record[i].i_stream) ||
1373 p_meta->record[i].i_stream ==
1374 p_sp->i_stream_number )
1375 GET_CHECKED( i_aspect_x, p_meta->record[i].i_val,
1376 UINT_MAX, uint64_t );
1377 }
1378 if( !strcmp( p_meta->record[i].psz_name, "AspectRatioY" ) )
1379 {
1380 if( (!i_aspect_y && !p_meta->record[i].i_stream) ||
1381 p_meta->record[i].i_stream ==
1382 p_sp->i_stream_number )
1383 GET_CHECKED( i_aspect_y, p_meta->record[i].i_val,
1384 UINT_MAX, uint64_t );
1385 }
1386 }
1387
1388 if( i_aspect_x && i_aspect_y )
1389 {
1390 fmt.video.i_sar_num = i_aspect_x;
1391 fmt.video.i_sar_den = i_aspect_y;
1392 }
1393 }
1394
1395 /* If there is a video track then use the index for seeking */
1396 p_sys->b_index = b_index;
1397
1398 msg_Dbg( p_demux, "added new video stream(ID:%d)",
1399 p_sp->i_stream_number );
1400 }
1401 else if( guidcmp( &p_sp->i_stream_type, &asf_object_extended_stream_header ) &&
1402 p_sp->i_type_specific_data_length >= 64 )
1403 {
1404 /* Now follows a 64 byte header of which we don't know much */
1405 guid_t *p_ref = (guid_t *)p_sp->p_type_specific_data;
1406 uint8_t *p_data = p_sp->p_type_specific_data + 64;
1407 unsigned int i_data = p_sp->i_type_specific_data_length - 64;
1408
1409 msg_Dbg( p_demux, "Ext stream header detected. datasize = %d", p_sp->i_type_specific_data_length );
1410 if( guidcmp( p_ref, &asf_object_extended_stream_type_audio ) &&
1411 i_data >= sizeof( WAVEFORMATEX ) - 2)
1412 {
1413 uint16_t i_format;
1414 es_format_Init( &fmt, AUDIO_ES, 0 );
1415 i_format = GetWLE( &p_data[0] );
1416 if( i_format == 0 )
1417 fmt.i_codec = VLC_CODEC_A52;
1418 else
1419 wf_tag_to_fourcc( i_format, &fmt.i_codec, NULL );
1420 GET_CHECKED( fmt.audio.i_channels, GetWLE( &p_data[2] ),
1421 255, uint16_t );
1422 GET_CHECKED( fmt.audio.i_rate, GetDWLE( &p_data[4] ),
1423 UINT_MAX, uint32_t );
1424 GET_CHECKED( fmt.i_bitrate, GetDWLE( &p_data[8] ) * 8,
1425 UINT_MAX, uint32_t );
1426 fmt.audio.i_blockalign = GetWLE( &p_data[12] );
1427 fmt.audio.i_bitspersample = GetWLE( &p_data[14] );
1428 fmt.b_packetized = true;
1429
1430 if( p_sp->i_type_specific_data_length > sizeof( WAVEFORMATEX ) &&
1431 i_format != WAVE_FORMAT_MPEGLAYER3 &&
1432 i_format != WAVE_FORMAT_MPEG )
1433 {
1434 GET_CHECKED( fmt.i_extra, __MIN( GetWLE( &p_data[16] ),
1435 p_sp->i_type_specific_data_length -
1436 sizeof( WAVEFORMATEX ) ),
1437 INT_MAX, uint32_t );
1438 fmt.p_extra = malloc( fmt.i_extra );
1439 memcpy( fmt.p_extra, &p_data[sizeof( WAVEFORMATEX )],
1440 fmt.i_extra );
1441 }
1442
1443 msg_Dbg( p_demux, "added new audio stream (codec:0x%x,ID:%d)",
1444 i_format, p_sp->i_stream_number );
1445 }
1446 else
1447 {
1448 es_format_Init( &fmt, UNKNOWN_ES, 0 );
1449 }
1450 }
1451 else
1452 {
1453 es_format_Init( &fmt, UNKNOWN_ES, 0 );
1454 }
1455
1456 tk->i_cat = fmt.i_cat;
1457 if( fmt.i_cat != UNKNOWN_ES )
1458 {
1459 if( p_esp && p_languages &&
1460 p_esp->i_language_index < p_languages->i_language )
1461 {
1462 fmt.psz_language = strdup( p_languages->ppsz_language[p_esp->i_language_index] );
1463 char *p;
1464 if( fmt.psz_language && (p = strchr( fmt.psz_language, '-' )) )
1465 *p = '\0';
1466 }
1467
1468 /* Set our priority so we won't get multiple videos */
1469 int i_priority = ES_PRIORITY_SELECTABLE_MIN;
1470 for( uint16_t i = 0; i < fmt_priorities_ex.i_count; i++ )
1471 {
1472 if ( fmt_priorities_ex.pi_stream_numbers[i] == p_sp->i_stream_number )
1473 {
1474 i_priority = ES_PRIORITY_NOT_DEFAULTABLE;
1475 break;
1476 }
1477 }
1478 for( uint16_t i = 0; i < fmt_priorities_bitrate_ex.i_count; i++ )
1479 {
1480 if ( fmt_priorities_bitrate_ex.pi_stream_numbers[i] == p_sp->i_stream_number )
1481 {
1482 i_priority = ES_PRIORITY_NOT_DEFAULTABLE;
1483 break;
1484 }
1485 }
1486 fmt.i_priority = i_priority;
1487
1488 if ( i_stream <= INT_MAX )
1489 fmt.i_id = i_stream;
1490 else
1491 msg_Warn( p_demux, "Can't set fmt.i_id to match stream id %u", i_stream );
1492
1493 if ( fmt.i_cat == VIDEO_ES )
1494 {
1495 /* Backup our video format */
1496 tk->p_fmt = malloc( sizeof( es_format_t ) );
1497 if ( tk->p_fmt )
1498 es_format_Copy( tk->p_fmt, &fmt );
1499 }
1500
1501 tk->p_es = es_out_Add( p_demux->out, &fmt );
1502
1503 if( !stream_Control( p_demux->s, STREAM_GET_PRIVATE_ID_STATE,
1504 (int) p_sp->i_stream_number, &b_access_selected ) &&
1505 b_access_selected )
1506 {
1507 p_sys->i_access_selected_track[fmt.i_cat] = p_sp->i_stream_number;
1508 }
1509
1510 }
1511 else
1512 {
1513 msg_Dbg( p_demux, "ignoring unknown stream(ID:%d)",
1514 p_sp->i_stream_number );
1515 }
1516
1517 es_format_Clean( &fmt );
1518 }
1519
1520 free( fmt_priorities_ex.pi_stream_numbers );
1521 free( fmt_priorities_bitrate_ex.pi_stream_numbers );
1522
1523 p_sys->i_data_begin = p_sys->p_root->p_data->i_object_pos + 50;
1524 if( p_sys->p_root->p_data->i_object_size != 0 )
1525 { /* local file */
1526 p_sys->i_data_end = p_sys->p_root->p_data->i_object_pos +
1527 p_sys->p_root->p_data->i_object_size;
1528 p_sys->i_data_end = __MIN( (uint64_t)stream_Size( p_demux->s ), p_sys->i_data_end );
1529 }
1530 else
1531 { /* live/broacast */
1532 p_sys->i_data_end = 0;
1533 }
1534
1535 /* go to first packet */
1536 stream_Seek( p_demux->s, p_sys->i_data_begin );
1537
1538 /* try to calculate movie time */
1539 if( p_sys->p_fp->i_data_packets_count > 0 )
1540 {
1541 uint64_t i_count;
1542 uint64_t i_size = stream_Size( p_demux->s );
1543
1544 if( p_sys->i_data_end > 0 && i_size > p_sys->i_data_end )
1545 {
1546 i_size = p_sys->i_data_end;
1547 }
1548
1549 /* real number of packets */
1550 i_count = ( i_size - p_sys->i_data_begin ) /
1551 p_sys->p_fp->i_min_data_packet_size;
1552
1553 /* calculate the time duration in micro-s */
1554 p_sys->i_length = (mtime_t)p_sys->p_fp->i_play_duration / 10 *
1555 (mtime_t)i_count /
1556 (mtime_t)p_sys->p_fp->i_data_packets_count - p_sys->p_fp->i_preroll * 1000;
1557 if( p_sys->i_length < 0 )
1558 p_sys->i_length = 0;
1559
1560 if( p_sys->i_length > 0 )
1561 {
1562 p_sys->i_bitrate = 8 * i_size * 1000000 / p_sys->i_length;
1563 }
1564 }
1565
1566 /* Create meta information */
1567 p_sys->meta = vlc_meta_New();
1568
1569 asf_object_content_description_t *p_cd;
1570 if( ( p_cd = ASF_FindObject( p_sys->p_root->p_hdr,
1571 &asf_object_content_description_guid, 0 ) ) )
1572 {
1573 if( p_cd->psz_title && *p_cd->psz_title )
1574 {
1575 vlc_meta_SetTitle( p_sys->meta, p_cd->psz_title );
1576 }
1577 if( p_cd->psz_artist && *p_cd->psz_artist )
1578 {
1579 vlc_meta_SetArtist( p_sys->meta, p_cd->psz_artist );
1580 }
1581 if( p_cd->psz_copyright && *p_cd->psz_copyright )
1582 {
1583 vlc_meta_SetCopyright( p_sys->meta, p_cd->psz_copyright );
1584 }
1585 if( p_cd->psz_description && *p_cd->psz_description )
1586 {
1587 vlc_meta_SetDescription( p_sys->meta, p_cd->psz_description );
1588 }
1589 if( p_cd->psz_rating && *p_cd->psz_rating )
1590 {
1591 vlc_meta_SetRating( p_sys->meta, p_cd->psz_rating );
1592 }
1593 }
1594 /// \tood Fix Child meta for ASF tracks
1595#if 0
1596 for( i_stream = 0, i = 0; i < MAX_ASF_TRACKS; i++ )
1597 {
1598 asf_object_codec_list_t *p_cl = ASF_FindObject( p_sys->p_root->p_hdr,
1599 &asf_object_codec_list_guid, 0 );
1600
1601 if( p_sys->track[i] )
1602 {
1603 vlc_meta_t *tk = vlc_meta_New();
1604 TAB_APPEND( p_sys->meta->i_track, p_sys->meta->track, tk );
1605
1606 if( p_cl && i_stream < p_cl->i_codec_entries_count )
1607 {
1608 if( p_cl->codec[i_stream].psz_name &&
1609 *p_cl->codec[i_stream].psz_name )
1610 {
1611 vlc_meta_Add( tk, VLC_META_CODEC_NAME,
1612 p_cl->codec[i_stream].psz_name );
1613 }
1614 if( p_cl->codec[i_stream].psz_description &&
1615 *p_cl->codec[i_stream].psz_description )
1616 {
1617 vlc_meta_Add( tk, VLC_META_CODEC_DESCRIPTION,
1618 p_cl->codec[i_stream].psz_description );
1619 }
1620 }
1621 i_stream++;
1622 }
1623 }
1624#endif
1625 return VLC_SUCCESS;
1626
1627error:
1628 ASF_FreeObjectRoot( p_demux->s, p_sys->p_root );
1629 return VLC_EGENERIC;
