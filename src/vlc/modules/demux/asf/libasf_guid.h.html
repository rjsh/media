266/****************************************************************************
267 * GUID functions
268 ****************************************************************************/
269static inline void ASF_GetGUID( guid_t *p_guid, const uint8_t *p_data )
270{
271 p_guid->Data1 = GetDWLE( p_data );
272 p_guid->Data2 = GetWLE( p_data + 4);
273 p_guid->Data3 = GetWLE( p_data + 6);
274 memcpy( p_guid->Data4, p_data + 8, 8 );
275}
