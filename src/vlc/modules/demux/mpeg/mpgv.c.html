52/*****************************************************************************

55struct demux_sys_t
56{
57 bool b_start;
58
59 es_out_id_t *p_es;
60
61 decoder_t *p_packetizer;
62};

64static int Demux( demux_t * );
65static int Control( demux_t *, int, va_list );

67#define MPGV_PACKET_SIZE 4096
68
69/*****************************************************************************
70 * Open: initializes demux structures
71 *****************************************************************************/
72static int Open( vlc_object_t * p_this )
73{
74 demux_t *p_demux = (demux_t*)p_this;
75 demux_sys_t *p_sys;
76 bool b_forced = false;
77
78 const uint8_t *p_peek;
79
80 es_format_t fmt;
81
82 if( stream_Peek( p_demux->s, &p_peek, 4 ) < 4 )
83 {
84 msg_Err( p_demux, "cannot peek" );
85 return VLC_EGENERIC;
86 }
87
88 if( p_demux->b_force )
89 b_forced = true;
90
91 if( p_peek[0] != 0x00 || p_peek[1] != 0x00 || p_peek[2] != 0x01 )
92 {
93 if( !b_forced ) return VLC_EGENERIC;
94
95 msg_Err( p_demux, "this doesn't look like an MPEG ES stream, continuing" );
96 }
97
98 if( p_peek[3] > 0xb9 )
99 {
100 if( !b_forced ) return VLC_EGENERIC;
101 msg_Err( p_demux, "this seems to be a system stream (PS plug-in ?), but continuing" );
102 }
103
104 p_demux->pf_demux = Demux;
105 p_demux->pf_control= Control;
106 p_demux->p_sys = p_sys = malloc( sizeof( demux_sys_t ) );
107 p_sys->b_start = true;
108 p_sys->p_es = NULL;
109
110 /* Load the mpegvideo packetizer */
111 es_format_Init( &fmt, VIDEO_ES, VLC_CODEC_MPGV );
112 p_sys->p_packetizer = demux_PacketizerNew( p_demux, &fmt, "mpeg video" );
113 if( !p_sys->p_packetizer )
114 {
115 free( p_sys );
116 return VLC_EGENERIC;
117 }
118
119 /* create the output */
120 es_format_Init( &fmt, VIDEO_ES, VLC_CODEC_MPGV );
121 p_sys->p_es = es_out_Add( p_demux->out, &fmt );
122
123 return VLC_SUCCESS;
124}

126/*****************************************************************************
127 * Close: frees unused data
128 *****************************************************************************/
129static void Close( vlc_object_t * p_this )
130{
131 demux_t *p_demux = (demux_t*)p_this;
132 demux_sys_t *p_sys = p_demux->p_sys;
133
134 demux_PacketizerDestroy( p_sys->p_packetizer );
135 free( p_sys );
136}

138/*****************************************************************************
139 * Demux: reads and demuxes data packets
140 *****************************************************************************
141 * Returns -1 in case of error, 0 in case of EOF, 1 otherwise
142 *****************************************************************************/
143static int Demux( demux_t *p_demux )
144{
145 demux_sys_t *p_sys = p_demux->p_sys;
146 block_t *p_block_in, *p_block_out;
147
148 if( ( p_block_in = stream_Block( p_demux->s, MPGV_PACKET_SIZE ) ) == NULL )
149 {
150 return 0;
151 }
152
153 if( p_sys->b_start )
154 {
155 p_block_in->i_pts =
156 p_block_in->i_dts = VLC_TS_0;
157 }
158 else
159 {
160 p_block_in->i_pts =
161 p_block_in->i_dts = VLC_TS_INVALID;
162 }
163
164 while( (p_block_out = p_sys->p_packetizer->pf_packetize( p_sys->p_packetizer, &p_block_in )) )
165 {
166 p_sys->b_start = false;
167
168 while( p_block_out )
169 {
170 block_t *p_next = p_block_out->p_next;
171
172 es_out_Control( p_demux->out, ES_OUT_SET_PCR, p_block_out->i_dts );
173
174 p_block_out->p_next = NULL;
175 es_out_Send( p_demux->out, p_sys->p_es, p_block_out );
176
177 p_block_out = p_next;
178 }
179 }
180 return 1;
181}

183/*****************************************************************************
184 * Control:
185 *****************************************************************************/
186static int Control( demux_t *p_demux, int i_query, va_list args )
187{
188 /* demux_sys_t *p_sys = p_demux->p_sys; */
189 /* FIXME calculate the bitrate */
190 if( i_query == DEMUX_SET_TIME )
191 return VLC_EGENERIC;
192 else
193 return demux_vaControlHelper( p_demux->s,
194 0, -1,
195 0, 1, i_query, args );
196}
