28static inline void hevc_skip_profile_tiers_level( bs_t * bs, int32_t max_sub_layer_minus1 )
29{
30 uint8_t sub_layer_profile_present_flag[8];
31 uint8_t sub_layer_level_present_flag[8];
32
33 /* skipping useless fields of the VPS see https://www.itu.int/rec/dologin_pub.asp?lang=e&id=T-REC-H.265-201304-I!!PDF-E&type=item */
34 bs_skip( bs, 2 + 1 + 5 + 32 + 1 + 1 + 1 + 1 + 44 + 8 );
35
36 for( int32_t i = 0; i < max_sub_layer_minus1; i++ )
37 {
38 sub_layer_profile_present_flag[i] = bs_read1( bs );
39 sub_layer_level_present_flag[i] = bs_read1( bs );
40 }
41
42 if(max_sub_layer_minus1 > 0)
43 bs_skip( bs, (8 - max_sub_layer_minus1) * 2 );
44
45 for( int32_t i = 0; i < max_sub_layer_minus1; i++ )
46 {
47 if( sub_layer_profile_present_flag[i] )
48 bs_skip( bs, 2 + 1 + 5 + 32 + 1 + 1 + 1 + 1 + 44 );
49 if( sub_layer_level_present_flag[i] )
50 bs_skip( bs, 8 );
51 }
52}

54static inline uint32_t read_ue( bs_t * bs )
55{
56 int32_t i = 0;
57
58 while( bs_read1( bs ) == 0 && bs->p < bs->p_end && i < 32 )
59 i++;
60
61 return (1 << i) - 1 + bs_read( bs, i );
62}

65static inline size_t nal_decode(uint8_t * p_src, uint8_t * p_dst, size_t i_size)
66{
67 size_t j = 0;
68 for (size_t i = 0; i < i_size; i++) {
69 if (i < i_size - 3 &&
70 p_src[i] == 0 && p_src[i+1] == 0 && p_src[i+2] == 3) {
71 p_dst[j++] = 0;
72 p_dst[j++] = 0;
73 i += 2;
74 continue;
75 }
76 p_dst[j++] = p_src[i];
77 }
78 return j;
79}
