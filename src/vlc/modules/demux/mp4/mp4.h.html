35/* Contain all information about a chunk */
36typedef struct
37{
38 uint64_t i_offset; /* absolute position of this chunk in the file */
39 uint32_t i_sample_description_index; /* index for SampleEntry to use */
40 uint32_t i_sample_count; /* how many samples in this chunk */
41 uint32_t i_sample_first; /* index of the first sample in this chunk */
42 uint32_t i_sample; /* index of the next sample to read in this chunk */
43
44 /* now provide way to calculate pts, dts, and offset without too
45 much memory and with fast access */
46
47 /* with this we can calculate dts/pts without waste memory */
48 uint64_t i_first_dts; /* DTS of the first sample */
49 uint64_t i_last_dts; /* DTS of the last sample */
50 uint32_t *p_sample_count_dts;
51 uint32_t *p_sample_delta_dts; /* dts delta */
52
53 uint32_t *p_sample_count_pts;
54 int32_t *p_sample_offset_pts; /* pts-dts */
55
56 uint8_t **p_sample_data; /* set when b_fragmented is true */
57 uint32_t *p_sample_size;
58 /* TODO if needed add pts
59 but quickly *add* support for edts and seeking */
60
61} mp4_chunk_t;
62
63 /* Contain all needed information for read all track with vlc */
64typedef struct
65{
66 unsigned int i_track_ID;/* this should be unique */
67
68 int b_ok; /* The track is usable */
69 int b_enable; /* is the trak enable by default */
70 bool b_selected; /* is the trak being played */
71 bool b_chapter; /* True when used for chapter only */
72
73 bool b_mac_encoding;
74
75 es_format_t fmt;
76 es_out_id_t *p_es;
77
78 /* display size only ! */
79 int i_width;
80 int i_height;
81 float f_rotation;
82
83 /* more internal data */
84 uint32_t i_timescale; /* time scale for this track only */
85 uint16_t current_qid; /* Smooth Streaming quality level ID */
86
87 /* elst */
88 int i_elst; /* current elst */
89 int64_t i_elst_time; /* current elst start time (in movie time scale)*/
90 MP4_Box_t *p_elst; /* elst (could be NULL) */
91
92 /* give the next sample to read, i_chunk is to find quickly where
93 the sample is located */
94 uint32_t i_sample; /* next sample to read */
95 uint32_t i_chunk; /* chunk where next sample is stored */
96 /* total count of chunk and sample */
97 uint32_t i_chunk_count;
98 uint32_t i_sample_count;
99
100 mp4_chunk_t *chunk; /* always defined for each chunk */
101 mp4_chunk_t *cchunk; /* current chunk if b_fragmented is true */
102
103 /* sample size, p_sample_size defined only if i_sample_size == 0
104 else i_sample_size is size for all sample */
105 uint32_t i_sample_size;
106 uint32_t *p_sample_size; /* XXX perhaps add file offset if take
107 too much time to do sumations each time*/
108
109 uint32_t i_sample_first; /* i_sample_first value
110 of the next chunk */
111 uint64_t i_first_dts; /* i_first_dts value
112 of the next chunk */
113
114 MP4_Box_t *p_stbl; /* will contain all timing information */
115 MP4_Box_t *p_stsd; /* will contain all data to initialize decoder */
116 MP4_Box_t *p_sample;/* point on actual sdsd */
117
118 bool b_drms;
119 bool b_has_non_empty_cchunk;
120 bool b_codec_need_restart;
121 void *p_drms;
122 MP4_Box_t *p_skcr;
123
124 mtime_t i_time;
125
126 struct
127 {
128 /* for moof parsing */
129 MP4_Box_t *p_traf;
130 MP4_Box_t *p_tfhd;
131 MP4_Box_t *p_trun;
132 uint64_t i_traf_base_offset;
133 } context;
134
135} mp4_track_t;
136
137typedef struct mp4_fragment_t mp4_fragment_t;
138struct mp4_fragment_t
139{
140 uint64_t i_chunk_range_min_offset;
141 uint64_t i_chunk_range_max_offset;
142 uint64_t i_duration;
143 MP4_Box_t *p_moox;
144 mp4_fragment_t *p_next;
145};
