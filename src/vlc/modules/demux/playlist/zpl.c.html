48/*****************************************************************************
49 * Import_ZPL: main import function
50 *****************************************************************************/
51int Import_ZPL( vlc_object_t *p_this )
52{
53 demux_t *p_demux = (demux_t *)p_this;
54
55 if(! ( demux_IsPathExtension( p_demux, ".zpl" ) || demux_IsForced( p_demux, "zpl" )))
56 return VLC_EGENERIC;
57
58 STANDARD_DEMUX_INIT_MSG( "found valid ZPL playlist" );
59 p_demux->p_sys->psz_prefix = FindPrefix( p_demux );
60
61 return VLC_SUCCESS;
62}

65/*****************************************************************************
66 * Deactivate: frees unused data

68void Close_ZPL( vlc_object_t *p_this )
69{
70 demux_t *p_demux = (demux_t *)p_this;
71 free( p_demux->p_sys->psz_prefix );
72 free( p_demux->p_sys );
73}

75static int Demux( demux_t *p_demux )
76{
77 char *psz_line;
78
79 mtime_t i_duration = -1;
80 char *psz_title = NULL, *psz_genre = NULL, *psz_tracknum = NULL,
81 *psz_language = NULL, *psz_artist = NULL, *psz_album = NULL,
82 *psz_date = NULL, *psz_publisher = NULL, *psz_encodedby = NULL,
83 *psz_description = NULL, *psz_url = NULL, *psz_copyright = NULL,
84 *psz_mrl = NULL;
85
86 input_item_t *p_current_input = GetCurrentItem(p_demux);
87
88 psz_line = stream_ReadLine( p_demux->s );
89 char *psz_parse = psz_line;
90
91 /* Skip leading tabs and spaces */
92 while( *psz_parse == ' ' || *psz_parse == '\t' ||
93 *psz_parse == '\n' || *psz_parse == '\r' )
94 psz_parse++;
95
96 /* if the 1st line is "AC", skip it */
97 /* TODO: using this information ? */
98 if( !strncasecmp( psz_parse, "AC", strlen( "AC" ) ) )
99 {
100 free( psz_line );
101 psz_line = stream_ReadLine( p_demux->s );
102 }
103
104 input_item_node_t *p_subitems = input_item_node_Create( p_current_input );
105
106 /* Loop on all lines */
107 while( psz_line )
108 {
109 psz_parse = psz_line;
110
111 /* Skip leading tabs and spaces */
112 while( *psz_parse == ' ' || *psz_parse == '\t' ||
113 *psz_parse == '\n' || *psz_parse == '\r' )
114 psz_parse++;
115
116 /* filename */
117 if( !strncasecmp( psz_parse, "NM", strlen( "NM" ) ) )
118 {
119 char *psz_tabvalue = ParseTabValue( psz_parse );
120 if( !EMPTY_STR(psz_tabvalue) )
121 {
122 free( psz_mrl );
123 psz_mrl = ProcessMRL( psz_tabvalue, p_demux->p_sys->psz_prefix );
124 }
125 free( psz_tabvalue );
126 }
127
128 /* duration */
129 else if( !strncasecmp( psz_parse, "DR", strlen( "DR" ) ) )
130 {
131 char *psz_tabvalue = ParseTabValue( psz_parse );
132 if( !EMPTY_STR(psz_tabvalue) )
133 {
134 int i_parsed_duration = atoi( psz_tabvalue );
135 if( i_parsed_duration >= 0 )
136 i_duration = i_parsed_duration * INT64_C(1000);
137 }
138 free( psz_tabvalue );
139 }
140
141#define PARSE(tag,variable) \
142 else if( !strncasecmp( psz_parse, tag, strlen( tag ) ) ) \
143 { \
144 free( variable ); \
145 variable = ParseTabValue( psz_parse ); \
146 }
147
148 PARSE( "TT", psz_title )
149 PARSE( "TG", psz_genre )
150 PARSE( "TR", psz_tracknum )
151 PARSE( "TL", psz_language )
152 PARSE( "TA", psz_artist )
153 PARSE( "TB", psz_album )
154 PARSE( "TY", psz_date )
155 PARSE( "TH", psz_publisher )
156 PARSE( "TE", psz_encodedby )
157 PARSE( "TC", psz_description )
158 PARSE( "TU", psz_url )
159 PARSE( "TO", psz_copyright )
160
161#undef PARSE
162
163 /* force a duration ? */
164 else if( !strncasecmp( psz_parse, "FD", strlen( "FD" ) ) )
165 {}
166
167 /* end of file entry */
168 else if( !strncasecmp( psz_parse, "BR!", strlen( "BR!" ) ) )
169 {
170 /* create the input item */
171 input_item_t *p_input = input_item_NewExt( psz_mrl,
172 psz_title, 0, NULL, 0, i_duration );
173 input_item_node_AppendItem( p_subitems, p_input );
174 FREENULL( psz_mrl );
175 FREENULL( psz_title );
176 i_duration = -1;
177
178#define SET(variable, type) \
179 if( !EMPTY_STR(variable) ) \
180 { \
181 input_item_Set##type( p_input, variable ); \
182 } \
183 FREENULL( variable );
184 /* set the meta */
185 SET( psz_genre, Genre );
186 SET( psz_tracknum, TrackNum );
187 SET( psz_language, Language );
188 SET( psz_artist, Artist );
189 SET( psz_album, Album );
190 SET( psz_date, Date );
191 SET( psz_encodedby, EncodedBy );
192 SET( psz_description, Description );
193 SET( psz_copyright, Copyright );
194 SET( psz_url, URL );
195 SET( psz_publisher, Publisher );
196#undef SET
197
198 vlc_gc_decref( p_input );
199 }
200 else
201 msg_Warn( p_demux, "invalid line '%s'", psz_parse );
202
203 /* Fetch another line */
204 free( psz_line );
205 psz_line = stream_ReadLine( p_demux->s );
206 }
207
208 input_item_node_PostAndDelete( p_subitems );
209
210 vlc_gc_decref(p_current_input);
211
212 // Free everything if the file is wrongly formated
213 free( psz_title );
214 free( psz_genre );
215 free( psz_tracknum );
216 free( psz_language );
217 free( psz_artist );
218 free( psz_album );
219 free( psz_date );
220 free( psz_publisher );
221 free( psz_encodedby );
222 free( psz_description );
223 free( psz_url );
224 free( psz_copyright );
225 free( psz_mrl );
226
227 var_Destroy( p_demux, "zpl-extvlcopt" );
228 return 0; /* Needed for correct operation of go back */
229}
230
231static char* ParseTabValue(char* psz_string)
232{
233 int i_len = strlen( psz_string );
234 if(i_len <= 3 )
235 return NULL;
236 char* psz_value = calloc( i_len, 1 );
237 if( ! psz_value )
238 return NULL;
239
240 sscanf( psz_string,"%*[^=]=%[^\r\t\n]", psz_value );
241
242 return psz_value;
243}
