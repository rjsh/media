72/**
73 * Import_RAM: main import function
74 * @param p_this: this demux object
75 * @return VLC_SUCCESS if everything is okay
76 */
77int Import_RAM( vlc_object_t *p_this )
78{
79 demux_t *p_demux = (demux_t *)p_this;
80 const uint8_t *p_peek;
81
82 if(! demux_IsPathExtension( p_demux, ".ram" ) ||
83 demux_IsPathExtension( p_demux, ".rm" ) )
84 return VLC_EGENERIC;
85
86 /* Many Real Media Files are misdetected */
87 if( stream_Peek( p_demux->s, &p_peek, 4 ) < 4 )
88 return VLC_EGENERIC;
89 if( !memcmp( p_peek, ".ra", 3 ) || !memcmp( p_peek, ".RMF", 4 ) )
90 {
91 return VLC_EGENERIC;
92 }
93
94 STANDARD_DEMUX_INIT_MSG( "found valid RAM playlist" );
95 p_demux->p_sys->psz_prefix = FindPrefix( p_demux );
96
97 return VLC_SUCCESS;
98}

109}

111/**
112 * Skips blanks in a given buffer
113 * @param s: input string
114 * @param i_strlen: length of the buffer
115 */
116static const char *SkipBlanks( const char *s, size_t i_strlen )
117{
118 while( i_strlen > 0 ) {
119 switch( *s )
120 {
121 case ' ':
122 case '\t':
123 case '\r':
124 case '\n':
125 --i_strlen;
126 ++s;
127 break;
128 default:
129 i_strlen = 0;
130 }
131 }
132 return s;
133}

135/**
136 * Converts a time of format hour:minutes:sec.fraction to seconds
137 * @param s: input string
138 * @param i_strlen: length of the buffer
139 * @return time in seconds
140 */
141static int ParseTime( const char *s, size_t i_strlen)
142{
143 // need to parse hour:minutes:sec.fraction string
144 int result = 0;
145 int val;
146 const char *end = s + i_strlen;
147 // skip leading spaces if any
148 s = SkipBlanks(s, i_strlen);
149
150 val = 0;
151 while( (s < end) && isdigit((unsigned char)*s) )
152 {
153 int newval = val*10 + (*s - '0');
154 if( newval < val )
155 {
156 // overflow
157 val = 0;
158 break;
159 }
160 val = newval;
161 ++s;
162 }
163 result = val;
164 s = SkipBlanks(s, end-s);
165 if( *s == ':' )
166 {
167 ++s;
168 s = SkipBlanks(s, end-s);
169 result = result * 60;
170 val = 0;
171 while( (s < end) && isdigit((unsigned char)*s) )
172 {
173 int newval = val*10 + (*s - '0');
174 if( newval < val )
175 {
176 // overflow
177 val = 0;
178 break;
179 }
180 val = newval;
181 ++s;
182 }
183 result += val;
184 s = SkipBlanks(s, end-s);
185 if( *s == ':' )
186 {
187 ++s;
188 s = SkipBlanks(s, end-s);
189 result = result * 60;
190 val = 0;
191 while( (s < end) && isdigit((unsigned char)*s) )
192 {
193 int newval = val*10 + (*s - '0');
194 if( newval < val )
195 {
196 // overflow
197 val = 0;
198 break;
199 }
200 val = newval;
201 ++s;
202 }
203 result += val;
204 // TODO: one day, we may need to parse fraction for sub-second resolution
205 }
206 }
207 return result;
208}

210/**
211 * Main demux callback function
212 * @param p_demux: this demux object
213 */
214static int Demux( demux_t *p_demux )
215{
216 char *psz_line;
217 char *psz_artist = NULL, *psz_album = NULL, *psz_genre = NULL, *psz_year = NULL;
218 char *psz_author = NULL, *psz_title = NULL, *psz_copyright = NULL, *psz_cdnum = NULL, *psz_comments = NULL;
219 mtime_t i_duration = -1;
220 const char **ppsz_options = NULL;
221 int i_options = 0, i_start = 0, i_stop = 0;
222 bool b_cleanup = false;
223 input_item_t *p_input;
224
225 input_item_t *p_current_input = GetCurrentItem(p_demux);
226
227 input_item_node_t *p_subitems = input_item_node_Create( p_current_input );
228
229 psz_line = stream_ReadLine( p_demux->s );
230 while( psz_line )
231 {
232 char *psz_parse = psz_line;
233
234 /* Skip leading tabs and spaces */
235 while( *psz_parse == ' ' || *psz_parse == '\t' ||
236 *psz_parse == '\n' || *psz_parse == '\r' ) psz_parse++;
237
238 if( *psz_parse == '#' )
239 {
240 /* Ignore comments */
241 }
242 else if( *psz_parse )
243 {
244 char *psz_mrl, *psz_option_next, *psz_option;
245 char *psz_param, *psz_value;
246
247 /* Get the MRL from the file. Note that this might contain parameters of form ?param1=value1&param2=value2 in a RAM file */
248 psz_mrl = ProcessMRL( psz_parse, p_demux->p_sys->psz_prefix );
249
250 b_cleanup = true;
251 if ( !psz_mrl ) goto error;
252
253 /* We have the MRL, now we have to check for options and parse them from MRL */
254 psz_option = strchr( psz_mrl, '?' ); /* Look for start of options */
255 if( psz_option )
256 {
257 /* Remove options from MRL
258 because VLC can't get the file otherwise */
259 *psz_option = '\0';
260 psz_option++;
261 psz_option_next = psz_option;
262 while( 1 ) /* Process each option */
263 {
264 /* Look for end of first option which maybe a & or \0 */
265 psz_option = psz_option_next;
266 psz_option_next = strchr( psz_option, '&' );
267 if( psz_option_next )
268 {
269 *psz_option_next = '\0';
270 psz_option_next++;
271 }
272 else
273 psz_option_next = strchr( psz_option, '\0' );
274 /* Quit if options are over */
275 if( psz_option_next == psz_option )
276 break;
277
278 /* Parse out param and value */
279 psz_param = psz_option;
280 psz_value = strchr( psz_option, '=' );
281 if( psz_value == NULL )
282 break;
283 *psz_value = '\0';
284 psz_value++;
285
286 /* Take action based on parameter value in the below if else structure */
287 /* TODO: Remove any quotes surrounding values if required */
288 if( !strcmp( psz_param, "clipinfo" ) )
289 {
290 ParseClipInfo( psz_value, &psz_artist, &psz_title,
291 &psz_album, &psz_genre, &psz_year,
292 &psz_cdnum, &psz_comments ); /* clipinfo has various sub parameters, which is parsed by this function */
293 }
294 else if( !strcmp( psz_param, "author" ) )
295 {
296 psz_author = decode_URI_duplicate(psz_value);
297 EnsureUTF8( psz_author );
298 }
299 else if( !strcmp( psz_param, "start" )
300 && strncmp( psz_mrl, "rtsp", 4 ) /* Our rtsp-real or our real demuxer is wrong */ )
301 {
302 i_start = ParseTime( psz_value, strlen( psz_value ) );
303 char *temp;
304 if( i_start )
305 {
306 if( asprintf( &temp, ":start-time=%d", i_start ) != -1 )
307 INSERT_ELEM( ppsz_options, i_options, i_options, temp );
308 }
309 }
310 else if( !strcmp( psz_param, "end" ) )
311 {
312 i_stop = ParseTime( psz_value, strlen( psz_value ) );
313 char *temp;
314 if( i_stop )
315 {
316 if( asprintf( &temp, ":stop-time=%d", i_stop ) != -1 )
317 INSERT_ELEM( ppsz_options, i_options, i_options, temp );
318 }
319 }
320 else if( !strcmp( psz_param, "title" ) )
321 {
322 psz_title = decode_URI_duplicate(psz_value);
323 EnsureUTF8( psz_title );
324 }
325 else if( !strcmp( psz_param, "copyright" ) )
326 {
327 psz_copyright = decode_URI_duplicate(psz_value);
328 EnsureUTF8( psz_copyright );
329 }
330 else
331 { /* TODO: insert option anyway? Currently ignores*/
332 /* INSERT_ELEM( ppsz_options, i_options, i_options, psz_option ); */
333 }
334 }
335 }
336
337 /* Create the input item and pump in all the options into playlist item */
338 p_input = input_item_NewExt( psz_mrl, psz_title, i_options, ppsz_options, 0, i_duration );
339
340 if( !EMPTY_STR( psz_artist ) ) input_item_SetArtist( p_input, psz_artist );
341 if( !EMPTY_STR( psz_author ) ) input_item_SetPublisher( p_input, psz_author );
342 if( !EMPTY_STR( psz_title ) ) input_item_SetTitle( p_input, psz_title );
343 if( !EMPTY_STR( psz_copyright ) ) input_item_SetCopyright( p_input, psz_copyright );
344 if( !EMPTY_STR( psz_album ) ) input_item_SetAlbum( p_input, psz_album );
345 if( !EMPTY_STR( psz_genre ) ) input_item_SetGenre( p_input, psz_genre );
346 if( !EMPTY_STR( psz_year ) ) input_item_SetDate( p_input, psz_year );
347 if( !EMPTY_STR( psz_cdnum ) ) input_item_SetTrackNum( p_input, psz_cdnum );
348 if( !EMPTY_STR( psz_comments ) ) input_item_SetDescription( p_input, psz_comments );
349
350 input_item_node_AppendItem( p_subitems, p_input );
351 vlc_gc_decref( p_input );
352 free( psz_mrl );
353 }
354
355 error:
356 /* Fetch another line */
357 free( psz_line );
358 psz_line = stream_ReadLine( p_demux->s );
359 if( !psz_line ) b_cleanup = true;
360
361 if( b_cleanup )
362 {
363 /* Cleanup state */
364 while( i_options-- ) free( (char*)ppsz_options[i_options] );
365 FREENULL( ppsz_options );
366 FREENULL( psz_artist );
367 FREENULL( psz_title );
368 FREENULL( psz_author );
369 FREENULL( psz_copyright );
370 FREENULL( psz_album );
371 FREENULL( psz_genre );
372 FREENULL( psz_year );
373 FREENULL( psz_cdnum );
374 FREENULL( psz_comments );
375 i_options = 0;
376 i_duration = -1;
377 i_start = 0;
378 i_stop = 0;
379 b_cleanup = false;
380 }
381 }
382 input_item_node_PostAndDelete( p_subitems );
383 vlc_gc_decref(p_current_input);
384 var_Destroy( p_demux, "m3u-extvlcopt" );
385 return 0; /* Needed for correct operation of go back */
386}

388/**
389 * Parses clipinfo parameter
390 * @param psz_clipinfo: string containing the clipinfo parameter along with quotes
391 * @param ppsz_artist: Buffer to store artist name
392 * @param ppsz_title: Buffer to store title
393 * @param ppsz_album: Buffer to store album
394 * @param ppsz_genre: Buffer to store genre
395 * @param ppsz_year: Buffer to store year
396 * @param ppsz_cdnum: Buffer to store cdnum
397 * @param ppsz_comments: Buffer to store comments
398 */
399static void ParseClipInfo( const char *psz_clipinfo, char **ppsz_artist, char **ppsz_title,
400 char **ppsz_album, char **ppsz_genre, char **ppsz_year,
401 char **ppsz_cdnum, char **ppsz_comments )
402{
403 char *psz_option_next, *psz_option_start, *psz_param, *psz_value, *psz_suboption;
404 char *psz_temp_clipinfo = strdup( psz_clipinfo );
405 psz_option_start = strchr( psz_temp_clipinfo, '"' );
406 if( !psz_option_start )
407 {
408 free( psz_temp_clipinfo );
409 return;
410 }
411
412 psz_option_start++;
413 psz_option_next = psz_option_start;
414 while( 1 ) /* Process each sub option */
415 {
416 /* Get the sub option */
417 psz_option_start = psz_option_next;
418 psz_option_next = strchr( psz_option_start, '|' );
419 if( psz_option_next )
420 *psz_option_next = '\0';
421 else
422 psz_option_next = strchr( psz_option_start, '"' );
423 if( psz_option_next )
424 *psz_option_next = '\0';
425 else
426 psz_option_next = strchr( psz_option_start, '\0' );
427 if( psz_option_next == psz_option_start )
428 break;
429
430 psz_suboption = strdup( psz_option_start );
431 if( !psz_suboption )
432 break;
433
434 /* Parse out param and value */
435 psz_param = psz_suboption;
436 if( strchr( psz_suboption, '=' ) )
437 {
438 psz_value = strchr( psz_suboption, '=' ) + 1;
439 *( strchr( psz_suboption, '=' ) ) = '\0';
440 }
441 else
442 break;
443 /* Put into args */
444 if( !strcmp( psz_param, "artist name" ) )
445 *ppsz_artist = decode_URI_duplicate( psz_value );
446 else if( !strcmp( psz_param, "title" ) )
447 *ppsz_title = decode_URI_duplicate( psz_value );
448 else if( !strcmp( psz_param, "album name" ) )
449 *ppsz_album = decode_URI_duplicate( psz_value );
450 else if( !strcmp( psz_param, "genre" ) )
451 *ppsz_genre = decode_URI_duplicate( psz_value );
452 else if( !strcmp( psz_param, "year" ) )
453 *ppsz_year = decode_URI_duplicate( psz_value );
454 else if( !strcmp( psz_param, "cdnum" ) )
455 *ppsz_cdnum = decode_URI_duplicate( psz_value );
456 else if( !strcmp( psz_param, "comments" ) )
457 *ppsz_comments = decode_URI_duplicate( psz_value );
458
459 free( psz_suboption );
460 psz_option_next++;
461 }
462
463 free( psz_temp_clipinfo );
464}
