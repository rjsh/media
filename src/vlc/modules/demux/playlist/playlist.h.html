85#define STANDARD_DEMUX_INIT_MSG( msg ) do { \
86 DEMUX_INIT_COMMON(); \
87 msg_Dbg( p_demux, "%s", msg ); } while(0)
88
89#define DEMUX_BY_EXTENSION_MSG( ext, msg ) \
90 demux_t *p_demux = (demux_t *)p_this; \
91 if( !demux_IsPathExtension( p_demux, ext ) ) \
92 return VLC_EGENERIC; \
93 STANDARD_DEMUX_INIT_MSG( msg );
94
95#define DEMUX_BY_EXTENSION_OR_FORCED_MSG( ext, module, msg ) \
96 demux_t *p_demux = (demux_t *)p_this; \
97 if( !demux_IsPathExtension( p_demux, ext ) && !demux_IsForced( p_demux, module ) ) \
98 return VLC_EGENERIC; \
99 STANDARD_DEMUX_INIT_MSG( msg );
100
101#define DEMUX_BY_EXTENSION_OR_MIMETYPE( ext, mime, msg ) \
102 demux_t *p_demux = (demux_t *)p_this; \
103 char* demux_mimetype = stream_ContentType( p_demux->s ); \
104 if(!( demux_IsPathExtension( p_demux, ext ) || (demux_mimetype && !strcasecmp( mime, demux_mimetype )) )) { \
105 free( demux_mimetype ); \
106 return VLC_EGENERIC; \
107 } \
108 free( demux_mimetype ); \
109 STANDARD_DEMUX_INIT_MSG( msg );

113 msg_Dbg( p_demux, "not enough data" ); return VLC_EGENERIC; } } while(0)
