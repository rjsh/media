56struct demux_sys_t
57{
58 es_out_id_t *p_es;
59
60 int64_t i_data_offset;
61 unsigned int i_data_size;
62 unsigned int i_block_frames;
63 unsigned int i_frame_size;
64 unsigned int i_bitrate;
65
66 date_t pts;
67};
68
69typedef struct xa_header_t
70{
71 char xa_id[4];
72 uint32_t iSize;
73
74 uint16_t wFormatTag;
75 uint16_t nChannels;
76 uint32_t nSamplesPerSec;
77 uint32_t nAvgBytesPerSec;
78 uint16_t nBlockAlign;
79 uint16_t wBitsPerSample;
80} xa_header_t;
81
82#define FRAME_LENGTH 28 /* samples per frame */
83
84/*****************************************************************************
85 * Open: check file and initializes structures
86 *****************************************************************************/
87static int Open( vlc_object_t * p_this )
88{
89 demux_t *p_demux = (demux_t*)p_this;
90 demux_sys_t *p_sys;
91 xa_header_t p_xa;
92 const uint8_t *p_buf;
93
94 /* XA file heuristic */
95 if( stream_Peek( p_demux->s, &p_buf, sizeof( p_xa ) )
96 < (signed)sizeof( p_xa ) )
97 return VLC_EGENERIC;
98
99 memcpy( &p_xa, p_buf, sizeof( p_xa ) );
100 if( ( strncmp( p_xa.xa_id, "XAI", 4 )
101 && strncmp( p_xa.xa_id, "XAJ", 4 ) )
102 || ( GetWLE( &p_xa.wFormatTag ) != 0x0001)
103 || ( GetWLE( &p_xa.wBitsPerSample ) != 16) )
104 return VLC_EGENERIC;
105
106 p_sys = malloc( sizeof( demux_sys_t ) );
107 if( unlikely( p_sys == NULL ) )
108 return VLC_ENOMEM;
109
110 p_demux->pf_demux = Demux;
111 p_demux->pf_control = Control;
112 p_demux->p_sys = p_sys;
113 p_sys->p_es = NULL;
114
115 /* skip XA header -- cannot fail */
116 stream_Read( p_demux->s, NULL, sizeof( p_xa ) );
117
118 es_format_t fmt;
119 es_format_Init( &fmt, AUDIO_ES, VLC_FOURCC('X','A','J',0) );
120
121 msg_Dbg( p_demux, "assuming EA ADPCM audio codec" );
122 fmt.audio.i_rate = GetDWLE( &p_xa.nSamplesPerSec );
123 fmt.audio.i_bytes_per_frame = 15 * GetWLE( &p_xa.nChannels );
124 fmt.audio.i_frame_length = FRAME_LENGTH;
125
126 fmt.audio.i_channels = GetWLE ( &p_xa.nChannels );
127 fmt.audio.i_blockalign = fmt.audio.i_bytes_per_frame;
128 fmt.audio.i_bitspersample = 16;
129 fmt.i_bitrate = (fmt.audio.i_rate * fmt.audio.i_bytes_per_frame * 8)
130 / fmt.audio.i_frame_length;
131
132 p_sys->i_data_offset = stream_Tell( p_demux->s );
133 /* FIXME: better computation */
134 p_sys->i_data_size = p_xa.iSize * 15 / 56;
135 /* How many frames per block (1:1 is too CPU intensive) */
136 p_sys->i_block_frames = fmt.audio.i_rate / (FRAME_LENGTH * 20) + 1;
137 p_sys->i_frame_size = fmt.audio.i_bytes_per_frame;
138 p_sys->i_bitrate = fmt.i_bitrate;
139
140 msg_Dbg( p_demux, "fourcc: %4.4s, channels: %d, "
141 "freq: %d Hz, bitrate: %dKo/s, blockalign: %d",
142 (char *)&fmt.i_codec, fmt.audio.i_channels, fmt.audio.i_rate,
143 fmt.i_bitrate / 8192, fmt.audio.i_blockalign );
144
145 p_sys->p_es = es_out_Add( p_demux->out, &fmt );
146
147 date_Init( &p_sys->pts, fmt.audio.i_rate, 1 );
148 date_Set( &p_sys->pts, VLC_TS_0 );
149
150 return VLC_SUCCESS;
151}

153/*****************************************************************************
154 * Demux: read packet and send them to decoders
155 *****************************************************************************
156 * Returns -1 in case of error, 0 in case of EOF, 1 otherwise
157 *****************************************************************************/
158static int Demux( demux_t *p_demux )
159{
160 demux_sys_t *p_sys = p_demux->p_sys;
161 block_t *p_block;
162 int64_t i_offset;
163 unsigned i_frames = p_sys->i_block_frames;
164
165 i_offset = stream_Tell( p_demux->s );
166
167 if( p_sys->i_data_size > 0 &&
168 i_offset >= p_sys->i_data_offset + p_sys->i_data_size )
169 {
170 /* EOF */
171 return 0;
172 }
173
174 p_block = stream_Block( p_demux->s, p_sys->i_frame_size * i_frames );
175 if( p_block == NULL )
176 {
177 msg_Warn( p_demux, "cannot read data" );
178 return 0;
179 }
180
181 i_frames = p_block->i_buffer / p_sys->i_frame_size;
182 p_block->i_dts = p_block->i_pts = date_Get( &p_sys->pts );
183 es_out_Control( p_demux->out, ES_OUT_SET_PCR, p_block->i_pts );
184 es_out_Send( p_demux->out, p_sys->p_es, p_block );
185
186 date_Increment( &p_sys->pts, i_frames * FRAME_LENGTH );
187
188 return 1;
189}

191/*****************************************************************************
192 * Close: frees unused data
193 *****************************************************************************/
194static void Close ( vlc_object_t * p_this )
195{
196 demux_sys_t *p_sys = ((demux_t *)p_this)->p_sys;
197
198 free( p_sys );
199}
200
201/*****************************************************************************
202 * Control:
203 *****************************************************************************/
204static int Control( demux_t *p_demux, int i_query, va_list args )
205{
206 demux_sys_t *p_sys = p_demux->p_sys;
207
208 return demux_vaControlHelper( p_demux->s, p_sys->i_data_offset,
209 p_sys->i_data_size ? p_sys->i_data_offset
210 + p_sys->i_data_size : -1,
211 p_sys->i_bitrate, p_sys->i_frame_size,
212 i_query, args );
213}
