80struct demux_sys_t
81{
82 es_out_id_t *p_es;
83 es_format_t fmt;
84 unsigned int i_frame_size;
85 unsigned int i_frame_samples;
86 unsigned int i_seek_step;
87 date_t pts;
88};

98/*****************************************************************************
99 * Open: initializes raw audio demuxer
100 *****************************************************************************/
101static int Open( vlc_object_t * p_this )
102{
103 demux_t *p_demux = (demux_t*)p_this;
104 demux_sys_t *p_sys;
105
106 p_demux->p_sys = p_sys = malloc( sizeof( demux_sys_t ) );
107 if( !p_sys )
108 return VLC_ENOMEM;
109
110 es_format_Init( &p_sys->fmt, AUDIO_ES, 0 );
111
112 char *psz_fourcc = var_CreateGetString( p_demux, "rawaud-fourcc" );
113 p_sys->fmt.i_codec = vlc_fourcc_GetCodecFromString( AUDIO_ES, psz_fourcc );
114 free( psz_fourcc );
115
116 if( !p_sys->fmt.i_codec )
117 {
118 msg_Err( p_demux, "rawaud-fourcc must be a 4 character string");
119 es_format_Clean( &p_sys->fmt );
120 free( p_sys );
121 return VLC_EGENERIC;
122 }
123
124 // get the bits per sample ratio based on codec
125 switch( p_sys->fmt.i_codec )
126 {
127
128 case VLC_CODEC_FL64:
129 p_sys->fmt.audio.i_bitspersample = 64;
130 break;
131
132 case VLC_CODEC_FL32:
133 case VLC_CODEC_S32L:
134 case VLC_CODEC_S32B:
135 p_sys->fmt.audio.i_bitspersample = 32;
136 break;
137
138 case VLC_CODEC_S24L:
139 case VLC_CODEC_S24B:
140 p_sys->fmt.audio.i_bitspersample = 24;
141 break;
142
143 case VLC_CODEC_S16L:
144 case VLC_CODEC_S16B:
145 p_sys->fmt.audio.i_bitspersample = 16;
146 break;
147
148 case VLC_CODEC_S8:
149 case VLC_CODEC_U8:
150 p_sys->fmt.audio.i_bitspersample = 8;
151 break;
152
153 default:
154 msg_Err( p_demux, "unknown fourcc format %4.4s",
155 (char *)&p_sys->fmt.i_codec);
156 es_format_Clean( &p_sys->fmt );
157 free( p_sys );
158 return VLC_EGENERIC;
159
160 }
161
162
163 p_sys->fmt.psz_language = var_CreateGetString( p_demux, "rawaud-lang" );
164 p_sys->fmt.audio.i_channels = var_CreateGetInteger( p_demux, "rawaud-channels" );
165 p_sys->fmt.audio.i_rate = var_CreateGetInteger( p_demux, "rawaud-samplerate" );
166
167 if( p_sys->fmt.audio.i_rate <= 0 )
168 {
169 msg_Err( p_demux, "invalid sample rate");
170 es_format_Clean( &p_sys->fmt );
171 free( p_sys );
172 return VLC_EGENERIC;
173 }
174
175 if( p_sys->fmt.audio.i_channels <= 0 )
176 {
177 msg_Err( p_demux, "invalid number of channels");
178 es_format_Clean( &p_sys->fmt );
179 free( p_sys );
180 return VLC_EGENERIC;
181 }
182
183 p_sys->fmt.i_bitrate = p_sys->fmt.audio.i_rate *
184 p_sys->fmt.audio.i_channels *
185 p_sys->fmt.audio.i_bitspersample;
186
187 msg_Dbg( p_demux,
188 "format initialized: channels=%d , samplerate=%d Hz, fourcc=%4.4s, bits per sample = %d, bitrate = %d bit/s",
189 p_sys->fmt.audio.i_channels,
190 p_sys->fmt.audio.i_rate,
191 (char*)&p_sys->fmt.i_codec,
192 p_sys->fmt.audio.i_bitspersample,
193 p_sys->fmt.i_bitrate);
194
195 /* add the es */
196 p_sys->p_es = es_out_Add( p_demux->out, &p_sys->fmt );
197 msg_Dbg( p_demux, "elementary stream added");
198
199 /* initialize timing */
200 date_Init( &p_sys->pts, p_sys->fmt.audio.i_rate, 1 );
201 date_Set( &p_sys->pts, 0 );
202
203 /* calculate 50ms frame size/time */
204 p_sys->i_frame_samples = __MAX( p_sys->fmt.audio.i_rate / 20, 1 );
205 p_sys->i_seek_step = p_sys->fmt.audio.i_channels *
206 ( (p_sys->fmt.audio.i_bitspersample + 7) / 8 );
207 p_sys->i_frame_size = p_sys->i_frame_samples * p_sys->i_seek_step;
208 msg_Dbg( p_demux, "frame size is %d bytes ", p_sys->i_frame_size);
209
210 p_demux->pf_demux = Demux;
211 p_demux->pf_control = Control;
212 return VLC_SUCCESS;
213}

215/*****************************************************************************
216 * Close: frees unused data
217 *****************************************************************************/
218static void Close( vlc_object_t *p_this )
219{
220 demux_t *p_demux = (demux_t*)p_this;
221 demux_sys_t *p_sys = p_demux->p_sys;
222
223 es_format_Clean( &p_sys->fmt );
224 free( p_sys );
225}

227/*****************************************************************************
228 * Demux: reads and demuxes data packets
229 *****************************************************************************
230 * Returns -1 in case of error, 0 in case of EOF, 1 otherwise
231 *****************************************************************************/
232static int Demux( demux_t *p_demux )
233{
234 demux_sys_t *p_sys = p_demux->p_sys;
235 block_t *p_block;
236
237 if( ( p_block = stream_Block( p_demux->s, p_sys->i_frame_size ) ) == NULL )
238 {
239 /* EOF */
240 return 0;
241 }
242
243 p_block->i_dts =
244 p_block->i_pts = VLC_TS_0 + date_Get( &p_sys->pts );
245
246 es_out_Control( p_demux->out, ES_OUT_SET_PCR, p_block->i_pts );
247 es_out_Send( p_demux->out, p_sys->p_es, p_block );
248
249 date_Increment( &p_sys->pts, p_sys->i_frame_samples );
250
251 return 1;
252}

254/*****************************************************************************
255 * Control:
256 *****************************************************************************/
257static int Control( demux_t *p_demux, int i_query, va_list args )
258{
259 demux_sys_t *p_sys = p_demux->p_sys;
260
261 return demux_vaControlHelper( p_demux->s, 0, -1,
262 p_sys->fmt.i_bitrate, p_sys->i_seek_step, i_query, args );
263}
