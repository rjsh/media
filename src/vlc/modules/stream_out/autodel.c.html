54/*****************************************************************************
55 * Local prototypes
56 *****************************************************************************/
57static sout_stream_id_sys_t *Add ( sout_stream_t *, es_format_t * );
58static int Del ( sout_stream_t *, sout_stream_id_sys_t * );
59static int Send ( sout_stream_t *, sout_stream_id_sys_t *, block_t * );
60
61struct sout_stream_id_sys_t
62{
63 sout_stream_id_sys_t *id;
64 es_format_t fmt;
65 mtime_t i_last;
66 bool b_error;
67};
68
69struct sout_stream_sys_t
70{
71 sout_stream_id_sys_t **pp_es;
72 int i_es_num;
73};
74
75/*****************************************************************************
76 * Open:
77 *****************************************************************************/
78static int Open( vlc_object_t *p_this )
79{
80 sout_stream_t *p_stream = (sout_stream_t*)p_this;
81 sout_stream_sys_t *p_sys;
82
83 p_sys = malloc( sizeof( sout_stream_sys_t ) );
84
85 if( !p_stream->p_next )
86 {
87 msg_Err( p_stream, "cannot create chain" );
88 free( p_sys );
89 return VLC_EGENERIC;
90 }
91 p_sys->pp_es = NULL;
92 p_sys->i_es_num = 0;
93
94 p_stream->pf_add = Add;
95 p_stream->pf_del = Del;
96 p_stream->pf_send = Send;
97
98 p_stream->p_sys = p_sys;
99
100 return VLC_SUCCESS;
101}
102
103/*****************************************************************************
104 * Close:
105 *****************************************************************************/
106static void Close( vlc_object_t * p_this )
107{
108 sout_stream_t *p_stream = (sout_stream_t*)p_this;
109 sout_stream_sys_t *p_sys = (sout_stream_sys_t *)p_stream->p_sys;
110
111 free( p_sys );
112}
113
114static sout_stream_id_sys_t * Add( sout_stream_t *p_stream, es_format_t *p_fmt )
115{
116 sout_stream_sys_t *p_sys = (sout_stream_sys_t *)p_stream->p_sys;
117 sout_stream_id_sys_t *p_es = malloc( sizeof(sout_stream_id_sys_t) );
118
119 p_es->fmt = *p_fmt;
120 p_es->id = NULL;
121 p_es->i_last = VLC_TS_INVALID;
122 p_es->b_error = false;
123 TAB_APPEND( p_sys->i_es_num, p_sys->pp_es, p_es );
124
125 return p_es;
126}
127
128static int Del( sout_stream_t *p_stream, sout_stream_id_sys_t *p_es )
129{
130 sout_stream_sys_t *p_sys = (sout_stream_sys_t *)p_stream->p_sys;
131 sout_stream_id_sys_t *id = p_es->id;
132
133 TAB_REMOVE( p_sys->i_es_num, p_sys->pp_es, p_es );
134 free( p_es );
135
136 if ( id != NULL )
137 return p_stream->p_next->pf_del( p_stream->p_next, id );
138 else
139 return VLC_SUCCESS;
140}
141
142static int Send( sout_stream_t *p_stream, sout_stream_id_sys_t *p_es,
143 block_t *p_buffer )
144{
145 sout_stream_sys_t *p_sys = (sout_stream_sys_t *)p_stream->p_sys;
146 mtime_t i_current = mdate();
147 int i;
148
149 p_es->i_last = p_buffer->i_dts;
150 if ( !p_es->id && !p_es->b_error )
151 {
152 p_es->id = p_stream->p_next->pf_add( p_stream->p_next, &p_es->fmt );
153 if ( p_es->id == NULL )
154 {
155 p_es->b_error = true;
156 msg_Err( p_stream, "couldn't create chain for id %d",
157 p_es->fmt.i_id );
158 }
159 }
160
161 if ( !p_es->b_error )
162 p_stream->p_next->pf_send( p_stream->p_next, p_es->id, p_buffer );
163 else
164 block_ChainRelease( p_buffer );
165
166 for ( i = 0; i < p_sys->i_es_num; i++ )
167 {
168 if ( p_sys->pp_es[i]->id != NULL
169 && (p_sys->pp_es[i]->fmt.i_cat == VIDEO_ES
170 || p_sys->pp_es[i]->fmt.i_cat == AUDIO_ES)
171 && p_sys->pp_es[i]->i_last < i_current )
172 {
173 p_stream->p_next->pf_del( p_stream->p_next, p_sys->pp_es[i]->id );
174 p_sys->pp_es[i]->id = NULL;
175 }
176 }
177
178 return VLC_SUCCESS;
179}
