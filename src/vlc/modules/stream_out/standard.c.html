141{

143 return VLC_SUCCESS;

151static void create_SDP(sout_stream_t *p_stream, sout_access_out_t *p_access)
152{
153 sout_stream_sys_t *p_sys = p_stream->p_sys;
154
155 static const struct addrinfo hints = {
156 .ai_family = AF_UNSPEC,
157 .ai_socktype = SOCK_DGRAM,
158 .ai_protocol = 0,
159 .ai_flags = AI_NUMERICHOST | AI_NUMERICSERV | AI_IDN,
160 };
161 char *shost = var_GetNonEmptyString (p_access, "src-addr");
162 char *dhost = var_GetNonEmptyString (p_access, "dst-addr");
163 int sport = var_GetInteger (p_access, "src-port");
164 int dport = var_GetInteger (p_access, "dst-port");
165 struct sockaddr_storage src, dst;
166 socklen_t srclen = 0, dstlen = 0;
167 struct addrinfo *res;
168
169 if (!vlc_getaddrinfo (dhost, dport, &hints, &res))
170 {
171 memcpy (&dst, res->ai_addr, dstlen = res->ai_addrlen);
172 freeaddrinfo (res);
173 }
174
175 if (!vlc_getaddrinfo (shost, sport, &hints, &res))
176 {
177 memcpy (&src, res->ai_addr, srclen = res->ai_addrlen);
178 freeaddrinfo (res);
179 }
180
181 char *head = vlc_sdp_Start (VLC_OBJECT (p_stream), SOUT_CFG_PREFIX,
182 (struct sockaddr *)&src, srclen,
183 (struct sockaddr *)&dst, dstlen);
184 free (shost);
185
186 if (head != NULL)
187 {
188 char *psz_sdp = NULL;
189 if (asprintf (&psz_sdp, "%s"
190 "m=video %d udp mpeg\r\n", head, dport) == -1)
191 psz_sdp = NULL;
192 free (head);
193
194 /* Register the SDP with the SAP thread */
195 if (psz_sdp)
196 {
197 msg_Dbg (p_stream, "Generated SDP:\n%s", psz_sdp);
198 p_sys->p_session =
199 sout_AnnounceRegisterSDP (p_stream, psz_sdp, dhost);
200 free( psz_sdp );
201 }
202 }
203 free (dhost);
204}

206static const char *getMuxFromAlias( const char *psz_alias )
207{
208 static struct { const char alias[6]; const char mux[32]; } mux_alias[] =
209 {
210 { "avi", "avi" },
211 { "ogg", "ogg" },
212 { "ogm", "ogg" },
213 { "ogv", "ogg" },
214 { "flac","raw" },
215 { "mp3", "raw" },
216 { "mp4", "mp4" },
217 { "mov", "mov" },
218 { "moov","mov" },
219 { "asf", "asf" },
220 { "wma", "asf" },
221 { "wmv", "asf" },
222 { "trp", "ts" },
223 { "ts", "ts" },
224 { "mpg", "ps" },
225 { "mpeg","ps" },
226 { "ps", "ps" },
227 { "mpeg1","mpeg1" },
228 { "wav", "wav" },
229 { "flv", "avformat{mux=flv}" },
230 { "mkv", "avformat{mux=matroska}"},
231 { "webm", "avformat{mux=webm}"},
232 };
233
234 if( !psz_alias )
235 return NULL;
236
237 for( size_t i = 0; i < sizeof mux_alias / sizeof *mux_alias; i++ )
238 if( !strcasecmp( psz_alias, mux_alias[i].alias ) )
239 return mux_alias[i].mux;
240
241 return NULL;
242}

244static int fixAccessMux( sout_stream_t *p_stream, char **ppsz_mux,
245 char **ppsz_access, const char *psz_url )
246{
247 char *psz_mux = *ppsz_mux;
248 char *psz_access = *ppsz_access;
249 if( !psz_mux )
250 {
251 const char *psz_ext = psz_url ? strrchr( psz_url, '.' ) : NULL;
252 if( psz_ext )
253 psz_ext++; /* use extension */
254 const char *psz_mux_byext = getMuxFromAlias( psz_ext );
255
256 if( !psz_access )
257 {
258 if( !psz_mux_byext )
259 {
260 msg_Err( p_stream, "no access _and_ no muxer" );
261 return 1;
262 }
263
264 msg_Warn( p_stream,
265 "no access _and_ no muxer, extension gives file/%s",
266 psz_mux_byext );
267 *ppsz_access = strdup("file");
268 *ppsz_mux = strdup(psz_mux_byext);
269 }
270 else
271 {
272 if( !strncmp( psz_access, "mmsh", 4 ) )
273 *ppsz_mux = strdup("asfh");
274 else if (!strcmp (psz_access, "udp"))
275 *ppsz_mux = strdup("ts");
276 else if( psz_mux_byext )
277 *ppsz_mux = strdup(psz_mux_byext);
278 else
279 {
280 msg_Err( p_stream, "no mux specified or found by extension" );
281 return 1;
282 }
283 }
284 }
285 else if( !psz_access )
286 {
287 if( !strncmp( psz_mux, "asfh", 4 ) )
288 *ppsz_access = strdup("mmsh");
289 else /* default file */
290 *ppsz_access = strdup("file");
291 }
292 return 0;
293}
294
295static void checkAccessMux( sout_stream_t *p_stream, char *psz_access,
296 char *psz_mux )
297{
298 if( !strncmp( psz_access, "mmsh", 4 ) && strncmp( psz_mux, "asfh", 4 ) )
299 msg_Err( p_stream, "mmsh output is only valid with asfh mux" );
300 else if( strncmp( psz_access, "file", 4 ) &&
301 ( !strncmp( psz_mux, "mov", 3 ) || !strncmp( psz_mux, "mp4", 3 ) ) )
302 msg_Err( p_stream, "mov and mp4 mux are only valid with file output" );
303 else if( !strncmp( psz_access, "udp", 3 ) )
304 {
305 if( !strncmp( psz_mux, "ffmpeg", 6 ) || !strncmp( psz_mux, "avformat", 8 ) )
306 { /* why would you use ffmpeg's ts muxer ? YOU DON'T LOVE VLC ??? */
307 char *psz_ffmpeg_mux = var_CreateGetString( p_stream, "sout-avformat-mux" );
308 if( !psz_ffmpeg_mux || strncmp( psz_ffmpeg_mux, "mpegts", 6 ) )
309 msg_Err( p_stream, "UDP output is only valid with TS mux" );
310 free( psz_ffmpeg_mux );
311 }
312 else if( strncmp( psz_mux, "ts", 2 ) )
313 msg_Err( p_stream, "UDP output is only valid with TS mux" );
314 }
315}

317/*****************************************************************************
318 * Open:
319 *****************************************************************************/
320static int Open( vlc_object_t *p_this )
321{
322 sout_stream_t *p_stream = (sout_stream_t*)p_this;
323 sout_stream_sys_t *p_sys;
324 char *psz_mux, *psz_access, *psz_url;
325 sout_access_out_t *p_access;
326 int ret = VLC_EGENERIC;
327
328 config_ChainParse( p_stream, SOUT_CFG_PREFIX, ppsz_sout_options,
329 p_stream->p_cfg );
330
331 psz_mux = var_GetNonEmptyString( p_stream, SOUT_CFG_PREFIX "mux" );
332
333 psz_access = var_GetNonEmptyString( p_stream, SOUT_CFG_PREFIX "access" );
334 if( !psz_access )
335 psz_access = strdup(p_stream->psz_name);
336
337 psz_url = var_GetNonEmptyString( p_stream, SOUT_CFG_PREFIX "dst" );
338 if (!psz_url)
339 {
340 char *psz_bind = var_GetNonEmptyString( p_stream, SOUT_CFG_PREFIX "bind" );
341 if( psz_bind )
342 {
343 char *psz_path = var_GetNonEmptyString( p_stream, SOUT_CFG_PREFIX "path" );
344 if( psz_path )
345 {
346 if( asprintf( &psz_url, "%s/%s", psz_bind, psz_path ) == -1 )
347 psz_url = NULL;
348 free(psz_bind);
349 free( psz_path );
350 }
351 else
352 psz_url = psz_bind;
353 }
354 }
355
356 p_sys = p_stream->p_sys = malloc( sizeof( sout_stream_sys_t) );
357 if( !p_sys )
358 {
359 ret = VLC_ENOMEM;
360 goto end;
361 }
362 p_sys->p_session = NULL;
363
364 if( fixAccessMux( p_stream, &psz_mux, &psz_access, psz_url ) )
365 goto end;
366
367 checkAccessMux( p_stream, psz_access, psz_mux );
368
369 p_access = sout_AccessOutNew( p_stream, psz_access, psz_url );
370 if( p_access == NULL )
371 {
372 msg_Err( p_stream, "no suitable sout access module for `%s/%s://%s'",
373 psz_access, psz_mux, psz_url );
374 goto end;
375 }
376
377 p_sys->p_mux = sout_MuxNew( p_stream->p_sout, psz_mux, p_access );
378 if( !p_sys->p_mux )
379 {
380 const char *psz_mux_guess = getMuxFromAlias( psz_mux );
381 if( psz_mux_guess && strcmp( psz_mux_guess, psz_mux ) )
382 {
383 msg_Dbg( p_stream, "Couldn't open mux `%s', trying `%s' instead",
384 psz_mux, psz_mux_guess );
385 p_sys->p_mux = sout_MuxNew( p_stream->p_sout, psz_mux_guess, p_access );
386 }
387
388 if( !p_sys->p_mux )
389 {
390 msg_Err( p_stream, "no suitable sout mux module for `%s/%s://%s'",
391 psz_access, psz_mux, psz_url );
392
393 sout_AccessOutDelete( p_access );
394 goto end;
395 }
396 }
397
398 if( var_GetBool( p_stream, SOUT_CFG_PREFIX"sap" ) )
399 create_SDP( p_stream, p_access );
400
401 p_stream->pf_add = Add;
402 p_stream->pf_del = Del;
403 p_stream->pf_send = Send;
404 if( !sout_AccessOutCanControlPace( p_access ) )
405 p_stream->pace_nocontrol = true;
406
407 ret = VLC_SUCCESS;
408
409 msg_Dbg( p_this, "using `%s/%s://%s'", psz_access, psz_mux, psz_url );
410
411end:
412 if( ret != VLC_SUCCESS )
413 free( p_sys );
414 free( psz_access );
415 free( psz_mux );
416 free( psz_url );
417
418 return ret;
419}
420
421/*****************************************************************************
422 * Close:
423 *****************************************************************************/
424static void Close( vlc_object_t * p_this )
425{
426 sout_stream_t *p_stream = (sout_stream_t*)p_this;
427 sout_stream_sys_t *p_sys = p_stream->p_sys;
428 sout_access_out_t *p_access = p_sys->p_mux->p_access;
429
430 if( p_sys->p_session != NULL )
431 sout_AnnounceUnRegister( p_stream, p_sys->p_session );
432
433 sout_MuxDelete( p_sys->p_mux );
434 sout_AccessOutDelete( p_access );
435
436 free( p_sys );
437}
