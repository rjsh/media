37/*****************************************************************************
38 * Exported prototypes
39 *****************************************************************************/
40static int Open ( vlc_object_t * );
41static void Close ( vlc_object_t * );
42
43static sout_stream_id_sys_t *Add ( sout_stream_t *, es_format_t * );
44static int Del ( sout_stream_t *, sout_stream_id_sys_t * );
45static int Send( sout_stream_t *, sout_stream_id_sys_t *, block_t* );
46
47/*****************************************************************************
48 * Module descriptor
49 *****************************************************************************/
50vlc_module_begin ()
51 set_description( N_("Dummy stream output") )
52 set_capability( "sout stream", 50 )
53 add_shortcut( "dummy", "drop" )
54 set_callbacks( Open, Close )
55vlc_module_end ()
56
57/*****************************************************************************
58 * Open:
59 *****************************************************************************/
60static int Open( vlc_object_t *p_this )
61{
62 sout_stream_t *p_stream = (sout_stream_t*)p_this;
63
64 p_stream->pf_add = Add;
65 p_stream->pf_del = Del;
66 p_stream->pf_send = Send;
67
68 p_stream->p_sys = NULL;
69
70 return VLC_SUCCESS;
71}
72
73/*****************************************************************************
74 * Close:
75 *****************************************************************************/
76static void Close( vlc_object_t * p_this )
77{
78 (void)p_this;
79}
80
81static sout_stream_id_sys_t *Add( sout_stream_t *p_stream, es_format_t *p_fmt )
82{
83 VLC_UNUSED(p_stream); VLC_UNUSED(p_fmt);
84 return malloc( 1 );
85}
86
87static int Del( sout_stream_t *p_stream, sout_stream_id_sys_t *id )
88{
89 VLC_UNUSED(p_stream);
90 free( id );
91
92 return VLC_SUCCESS;
93}
94
95static int Send( sout_stream_t *p_stream, sout_stream_id_sys_t *id,
96 block_t *p_buffer )
97{
98 (void)p_stream; (void)id;
99 block_ChainRelease( p_buffer );
100 return VLC_SUCCESS;
101}
