24/*****************************************************************************
25 * How to use it
26 *****************************************************************************
27 *
28 * You should use this module in combination with the transcode module, to get
29 * raw datas from it. This module does not make any conversion at all, so you
30 * need to use the transcode module for this purpose.
31 *
32 * For example, you can use smem as it :
33 * --sout="#transcode{vcodec=RV24,acodec=s16l}:smem{smem-options}"
34 *
35 * Into each lock function (audio and video), you will have all the information
36 * you need to allocate a buffer, so that this module will copy data in it.
37 *
38 * the video-data and audio-data pointers will be passed to lock/unlock function
39 *
40 ******************************************************************************/

140struct sout_stream_id_sys_t
141{
142 es_format_t* format;
143 void *p_data;
144};
145
146struct sout_stream_sys_t
147{
148 vlc_mutex_t *p_lock;
149 void ( *pf_video_prerender_callback ) ( void* p_video_data, uint8_t** pp_pixel_buffer, size_t size );
150 void ( *pf_audio_prerender_callback ) ( void* p_audio_data, uint8_t** pp_pcm_buffer, size_t size );
151 void ( *pf_video_postrender_callback ) ( void* p_video_data, uint8_t* p_pixel_buffer, int width, int height, int pixel_pitch, size_t size, mtime_t pts );
152 void ( *pf_audio_postrender_callback ) ( void* p_audio_data, uint8_t* p_pcm_buffer, unsigned int channels, unsigned int rate, unsigned int nb_samples, unsigned int bits_per_sample, size_t size, mtime_t pts );
153 bool time_sync;
154};

156/*****************************************************************************
157 * Open:
158 *****************************************************************************/
159static int Open( vlc_object_t *p_this )
160{
161 char* psz_tmp;
162 sout_stream_t *p_stream = (sout_stream_t*)p_this;
163 sout_stream_sys_t *p_sys;
164
165 p_sys = calloc( 1, sizeof( sout_stream_sys_t ) );
166 if( !p_sys )
167 return VLC_ENOMEM;
168 p_stream->p_sys = p_sys;
169
170 config_ChainParse( p_stream, SOUT_CFG_PREFIX, ppsz_sout_options,
171 p_stream->p_cfg );
172
173 p_sys->time_sync = var_GetBool( p_stream, SOUT_CFG_PREFIX "time-sync" );
174
175 psz_tmp = var_GetString( p_stream, SOUT_PREFIX_VIDEO "prerender-callback" );
176 p_sys->pf_video_prerender_callback = (void (*) (void *, uint8_t**, size_t))(intptr_t)atoll( psz_tmp );
177 free( psz_tmp );
178
179 psz_tmp = var_GetString( p_stream, SOUT_PREFIX_AUDIO "prerender-callback" );
180 p_sys->pf_audio_prerender_callback = (void (*) (void* , uint8_t**, size_t))(intptr_t)atoll( psz_tmp );
181 free( psz_tmp );
182
183 psz_tmp = var_GetString( p_stream, SOUT_PREFIX_VIDEO "postrender-callback" );
184 p_sys->pf_video_postrender_callback = (void (*) (void*, uint8_t*, int, int, int, size_t, mtime_t))(intptr_t)atoll( psz_tmp );
185 free( psz_tmp );
186
187 psz_tmp = var_GetString( p_stream, SOUT_PREFIX_AUDIO "postrender-callback" );
188 p_sys->pf_audio_postrender_callback = (void (*) (void*, uint8_t*, unsigned int, unsigned int, unsigned int, unsigned int, size_t, mtime_t))(intptr_t)atoll( psz_tmp );
189 free( psz_tmp );
190
191 /* Setting stream out module callbacks */
192 p_stream->pf_add = Add;
193 p_stream->pf_del = Del;
194 p_stream->pf_send = Send;
195 p_stream->pace_nocontrol = p_sys->time_sync;
196
197 return VLC_SUCCESS;
198}

209static sout_stream_id_sys_t *Add( sout_stream_t *p_stream, es_format_t *p_fmt )
210{
211 sout_stream_id_sys_t *id = NULL;
212
213 if ( p_fmt->i_cat == VIDEO_ES )
214 id = AddVideo( p_stream, p_fmt );
215 else if ( p_fmt->i_cat == AUDIO_ES )
216 id = AddAudio( p_stream, p_fmt );
217 return id;
218}
219
220static sout_stream_id_sys_t *AddVideo( sout_stream_t *p_stream, es_format_t *p_fmt )
221{
222 char* psz_tmp;
223 sout_stream_id_sys_t *id;
224 int i_bits_per_pixel;
225
226 switch( p_fmt->i_codec )
227 {
228 case VLC_CODEC_RGB32:
229 case VLC_CODEC_RGBA:
230 case VLC_CODEC_ARGB:
231 i_bits_per_pixel = 32;
232 break;
233 case VLC_CODEC_I444:
234 case VLC_CODEC_RGB24:
235 i_bits_per_pixel = 24;
236 break;
237 case VLC_CODEC_RGB16:
238 case VLC_CODEC_RGB15:
239 case VLC_CODEC_RGB8:
240 case VLC_CODEC_I422:
241 i_bits_per_pixel = 16;
242 break;
243 case VLC_CODEC_YV12:
244 case VLC_CODEC_I420:
245 i_bits_per_pixel = 12;
246 break;
247 case VLC_CODEC_RGBP:
248 i_bits_per_pixel = 8;
249 break;
250 default:
251 i_bits_per_pixel = 0;
252 msg_Dbg( p_stream, "non raw video format detected (%4.4s), buffers will contain compressed video", (char *)&p_fmt->i_codec );
253 break;
254 }
255
256 id = calloc( 1, sizeof( sout_stream_id_sys_t ) );
257 if( !id )
258 return NULL;
259
260 psz_tmp = var_GetString( p_stream, SOUT_PREFIX_VIDEO "data" );
261 id->p_data = (void *)( intptr_t )atoll( psz_tmp );
262 free( psz_tmp );
263
264 id->format = p_fmt;
265 id->format->video.i_bits_per_pixel = i_bits_per_pixel;
266 return id;
267}

269static sout_stream_id_sys_t *AddAudio( sout_stream_t *p_stream, es_format_t *p_fmt )
270{
271 char* psz_tmp;
272 sout_stream_id_sys_t* id;
273 int i_bits_per_sample = aout_BitsPerSample( p_fmt->i_codec );
274
275 if( !i_bits_per_sample )
276 {
277 msg_Err( p_stream, "Smem does only support raw audio format" );
278 return NULL;
279 }
280
281 id = calloc( 1, sizeof( sout_stream_id_sys_t ) );
282 if( !id )
283 return NULL;
284
285 psz_tmp = var_GetString( p_stream, SOUT_PREFIX_AUDIO "data" );
286 id->p_data = (void *)( intptr_t )atoll( psz_tmp );
287 free( psz_tmp );
288
289 id->format = p_fmt;
290 id->format->audio.i_bitspersample = i_bits_per_sample;
291 return id;
292}

301static int Send( sout_stream_t *p_stream, sout_stream_id_sys_t *id,
302 block_t *p_buffer )
303{
304 if ( id->format->i_cat == VIDEO_ES )
305 return SendVideo( p_stream, id, p_buffer );
306 else if ( id->format->i_cat == AUDIO_ES )
307 return SendAudio( p_stream, id, p_buffer );
308 return VLC_SUCCESS;
309}
310
311static int SendVideo( sout_stream_t *p_stream, sout_stream_id_sys_t *id,
312 block_t *p_buffer )
313{
314 sout_stream_sys_t *p_sys = p_stream->p_sys;
315 size_t i_size = p_buffer->i_buffer;
316 uint8_t* p_pixels = NULL;
317
318 /* Calling the prerender callback to get user buffer */
319 p_sys->pf_video_prerender_callback( id->p_data, &p_pixels, i_size );
320
321 if (!p_pixels)
322 {
323 msg_Err( p_stream, "No buffer given!" );
324 block_ChainRelease( p_buffer );
325 return VLC_EGENERIC;
326 }
327
328 /* Copying data into user buffer */
329 memcpy( p_pixels, p_buffer->p_buffer, i_size );
330 /* Calling the postrender callback to tell the user his buffer is ready */
331 p_sys->pf_video_postrender_callback( id->p_data, p_pixels,
332 id->format->video.i_width, id->format->video.i_height,
333 id->format->video.i_bits_per_pixel, i_size, p_buffer->i_pts );
334 block_ChainRelease( p_buffer );
335 return VLC_SUCCESS;
336}
337
338static int SendAudio( sout_stream_t *p_stream, sout_stream_id_sys_t *id,
339 block_t *p_buffer )
340{
341 sout_stream_sys_t *p_sys = p_stream->p_sys;
342 int i_size;
343 uint8_t* p_pcm_buffer = NULL;
344 int i_samples = 0;
345
346 i_size = p_buffer->i_buffer;
347 if (id->format->audio.i_channels <= 0)
348 {
349 msg_Warn( p_stream, "No buffer given!" );
350 block_ChainRelease( p_buffer );
351 return VLC_EGENERIC;
352 }
353
354 i_samples = i_size / ( ( id->format->audio.i_bitspersample / 8 ) * id->format->audio.i_channels );
355 /* Calling the prerender callback to get user buffer */
356 p_sys->pf_audio_prerender_callback( id->p_data, &p_pcm_buffer, i_size );
357 if (!p_pcm_buffer)
358 {
359 msg_Err( p_stream, "No buffer given!" );
360 block_ChainRelease( p_buffer );
361 return VLC_EGENERIC;
362 }
363
364 /* Copying data into user buffer */
365 memcpy( p_pcm_buffer, p_buffer->p_buffer, i_size );
366 /* Calling the postrender callback to tell the user his buffer is ready */
367 p_sys->pf_audio_postrender_callback( id->p_data, p_pcm_buffer,
368 id->format->audio.i_channels, id->format->audio.i_rate, i_samples,
369 id->format->audio.i_bitspersample, i_size, p_buffer->i_pts );
370 block_ChainRelease( p_buffer );
371 return VLC_SUCCESS;
372}
