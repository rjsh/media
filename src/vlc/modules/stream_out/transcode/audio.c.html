39{

41 AOUT_CHAN_CENTER,
42 AOUT_CHAN_LEFT | AOUT_CHAN_RIGHT,
43 AOUT_CHAN_LFE | AOUT_CHAN_LEFT | AOUT_CHAN_RIGHT,

55 | AOUT_CHAN_LFE,

58static int audio_update_format( decoder_t *p_dec )
59{
60 aout_FormatPrepare( &p_dec->fmt_out.audio );
61 return 0;
62}

64static int transcode_audio_initialize_filters( sout_stream_t *p_stream, sout_stream_id_sys_t *id,
65 sout_stream_sys_t *p_sys, audio_sample_format_t *fmt_last )
66{
67 /* Load user specified audio filters */
68 /* XXX: These variable names come kinda out of nowhere... */
69 var_Create( p_stream, "audio-time-stretch", VLC_VAR_BOOL );
70 var_Create( p_stream, "audio-filter", VLC_VAR_STRING );
71 if( p_sys->psz_af )
72 var_SetString( p_stream, "audio-filter", p_sys->psz_af );
73 id->p_af_chain = aout_FiltersNew( p_stream, fmt_last,
74 &id->p_encoder->fmt_in.audio, NULL );
75 var_Destroy( p_stream, "audio-filter" );
76 var_Destroy( p_stream, "audio-time-stretch" );
77 if( id->p_af_chain == NULL )
78 {
79 msg_Err( p_stream, "Unable to initialize audio filters" );
80 module_unneed( id->p_encoder, id->p_encoder->p_module );
81 id->p_encoder->p_module = NULL;
82 module_unneed( id->p_decoder, id->p_decoder->p_module );
83 id->p_decoder->p_module = NULL;
84 return VLC_EGENERIC;
85 }
86 id->fmt_audio.i_rate = fmt_last->i_rate;
87 id->fmt_audio.i_physical_channels = fmt_last->i_physical_channels;
88 return VLC_SUCCESS;
89}

91static int transcode_audio_initialize_encoder( sout_stream_id_sys_t *id, sout_stream_t *p_stream )
92{
93 sout_stream_sys_t *p_sys = p_stream->p_sys;
94 /* Initialization of encoder format structures */
95 es_format_Init( &id->p_encoder->fmt_in, id->p_decoder->fmt_in.i_cat,
96 id->p_decoder->fmt_out.i_codec );
97 id->p_encoder->fmt_in.audio.i_format = id->p_decoder->fmt_out.i_codec;
98 id->p_encoder->fmt_in.audio.i_rate = id->p_encoder->fmt_out.audio.i_rate;
99 id->p_encoder->fmt_in.audio.i_physical_channels =
100 id->p_encoder->fmt_out.audio.i_physical_channels;
101 aout_FormatPrepare( &id->p_encoder->fmt_in.audio );
102
103 id->p_encoder->p_cfg = p_stream->p_sys->p_audio_cfg;
104 id->p_encoder->p_module =
105 module_need( id->p_encoder, "encoder", p_sys->psz_aenc, true );
106 /* p_sys->i_acodec = 0 if there isn't acodec defined */
107 if( !id->p_encoder->p_module && p_sys->i_acodec )
108 {
109 msg_Err( p_stream, "cannot find audio encoder (module:%s fourcc:%4.4s). "
110 "Take a look few lines earlier to see possible reason.",
111 p_sys->psz_aenc ? p_sys->psz_aenc : "any",
112 (char *)&p_sys->i_acodec );
113 module_unneed( id->p_decoder, id->p_decoder->p_module );
114 id->p_decoder->p_module = NULL;
115 return VLC_EGENERIC;
116 }
117
118 id->p_encoder->fmt_out.i_codec =
119 vlc_fourcc_GetCodec( AUDIO_ES, id->p_encoder->fmt_out.i_codec );
120
121 /* Fix input format */
122 id->p_encoder->fmt_in.audio.i_format = id->p_encoder->fmt_in.i_codec;
123 if( !id->p_encoder->fmt_in.audio.i_physical_channels
124 || !id->p_encoder->fmt_in.audio.i_original_channels )
125 {
126 if( id->p_encoder->fmt_in.audio.i_channels < (sizeof(pi_channels_maps) / sizeof(*pi_channels_maps)) )
127 id->p_encoder->fmt_in.audio.i_physical_channels =
128 id->p_encoder->fmt_in.audio.i_original_channels =
129 pi_channels_maps[id->p_encoder->fmt_in.audio.i_channels];
130 }
131 aout_FormatPrepare( &id->p_encoder->fmt_in.audio );
132
133 return VLC_SUCCESS;
134}

136int transcode_audio_new( sout_stream_t *p_stream,
137 sout_stream_id_sys_t *id )
138{
139 sout_stream_sys_t *p_sys = p_stream->p_sys;
140 audio_sample_format_t fmt_last;
141
142 /*
143 * Open decoder
144 */
145
146 /* Initialization of decoder structures */
147 id->p_decoder->fmt_out = id->p_decoder->fmt_in;
148 id->p_decoder->fmt_out.i_extra = 0;
149 id->p_decoder->fmt_out.p_extra = 0;
150 id->p_decoder->pf_decode_audio = NULL;
151 id->p_decoder->pf_aout_format_update = audio_update_format;
152 /* id->p_decoder->p_cfg = p_sys->p_audio_cfg; */
153 id->p_decoder->p_module =
154 module_need( id->p_decoder, "decoder", "$codec", false );
155 if( !id->p_decoder->p_module )
156 {
157 msg_Err( p_stream, "cannot find audio decoder" );
158 return VLC_EGENERIC;
159 }
160 /* decoders don't set audio.i_format, but audio filters use it */
161 id->p_decoder->fmt_out.audio.i_format = id->p_decoder->fmt_out.i_codec;
162 aout_FormatPrepare( &id->p_decoder->fmt_out.audio );
163 fmt_last = id->p_decoder->fmt_out.audio;
164 /* Fix AAC SBR changing number of channels and sampling rate */
165 if( !(id->p_decoder->fmt_in.i_codec == VLC_CODEC_MP4A &&
166 fmt_last.i_rate != id->p_encoder->fmt_in.audio.i_rate &&
167 fmt_last.i_channels != id->p_encoder->fmt_in.audio.i_channels) )
168 fmt_last.i_rate = id->p_decoder->fmt_in.audio.i_rate;
169
170 /*
171 * Open encoder
172 */
173 if( transcode_audio_initialize_encoder( id, p_stream ) == VLC_EGENERIC )
174 return VLC_EGENERIC;
175
176 if( unlikely( transcode_audio_initialize_filters( p_stream, id, p_sys,
177 &fmt_last ) != VLC_SUCCESS ) )
178 return VLC_EGENERIC;
179
180 return VLC_SUCCESS;
181}

183void transcode_audio_close( sout_stream_id_sys_t *id )
184{
185 /* Close decoder */
186 if( id->p_decoder->p_module )
187 module_unneed( id->p_decoder, id->p_decoder->p_module );
188 id->p_decoder->p_module = NULL;
189
190 if( id->p_decoder->p_description )
191 vlc_meta_Delete( id->p_decoder->p_description );
192 id->p_decoder->p_description = NULL;
193
194 /* Close encoder */
195 if( id->p_encoder->p_module )
196 module_unneed( id->p_encoder, id->p_encoder->p_module );
197 id->p_encoder->p_module = NULL;
198
199 /* Close filters */
200 if( id->p_af_chain != NULL )
201 aout_FiltersDelete( (vlc_object_t *)NULL, id->p_af_chain );
202}

204int transcode_audio_process( sout_stream_t *p_stream,
205 sout_stream_id_sys_t *id,
206 block_t *in, block_t **out )
207{
208 sout_stream_sys_t *p_sys = p_stream->p_sys;
209 block_t *p_block, *p_audio_buf;
210 *out = NULL;
211
212 if( unlikely( in == NULL ) )
213 {
214 block_t *p_block;
215 do {
216 p_block = id->p_encoder->pf_encode_audio(id->p_encoder, NULL );
217 block_ChainAppend( out, p_block );
218 } while( p_block );
219 return VLC_SUCCESS;
220 }
221
222 while( (p_audio_buf = id->p_decoder->pf_decode_audio( id->p_decoder,
223 &in )) )
224 {
225 if( unlikely( !id->p_encoder->p_module ) )
226 {
227 /* Complete destination format */
228 id->p_encoder->fmt_out.i_codec = p_sys->i_acodec;
229 id->p_encoder->fmt_out.audio.i_rate = p_sys->i_sample_rate > 0 ?
230 p_sys->i_sample_rate : id->p_decoder->fmt_out.audio.i_rate;
231 id->p_encoder->fmt_out.i_bitrate = p_sys->i_abitrate;
232 id->p_encoder->fmt_out.audio.i_bitspersample =
233 id->p_decoder->fmt_out.audio.i_bitspersample;
234 id->p_encoder->fmt_out.audio.i_channels = p_sys->i_channels > 0 ?
235 p_sys->i_channels : id->p_decoder->fmt_out.audio.i_channels;
236
237 id->p_encoder->fmt_in.audio.i_original_channels =
238 id->p_encoder->fmt_out.audio.i_original_channels =
239 id->p_decoder->fmt_out.audio.i_physical_channels;
240
241 id->p_encoder->fmt_in.audio.i_physical_channels =
242 id->p_encoder->fmt_out.audio.i_physical_channels =
243 pi_channels_maps[id->p_encoder->fmt_out.audio.i_channels];
244
245 if( transcode_audio_initialize_encoder( id, p_stream ) )
246 {
247 msg_Err( p_stream, "cannot create audio chain" );
248 return VLC_EGENERIC;
249 }
250 if( unlikely( transcode_audio_initialize_filters( p_stream, id, p_sys,
251 &id->p_decoder->fmt_out.audio ) != VLC_SUCCESS ) )
252 return VLC_EGENERIC;
253 date_Init( &id->next_input_pts, id->p_decoder->fmt_out.audio.i_rate, 1 );
254 date_Set( &id->next_input_pts, p_audio_buf->i_pts );
255 }
256
257 /* Check if audio format has changed, and filters need reinit */
258 if( unlikely( ( id->p_decoder->fmt_out.audio.i_rate != id->fmt_audio.i_rate ) ||
259 ( id->p_decoder->fmt_out.audio.i_physical_channels != id->fmt_audio.i_physical_channels ) ) )
260 {
261 msg_Info( p_stream, "Audio changed, trying to reinitialize filters" );
262 if( id->p_af_chain != NULL )
263 aout_FiltersDelete( (vlc_object_t *)NULL, id->p_af_chain );
264
265 /* decoders don't set audio.i_format, but audio filters use it */
266 id->p_decoder->fmt_out.audio.i_format = id->p_decoder->fmt_out.i_codec;
267 aout_FormatPrepare( &id->p_decoder->fmt_out.audio );
268
269 if( transcode_audio_initialize_filters( p_stream, id, p_sys,
270 &id->p_decoder->fmt_out.audio ) != VLC_SUCCESS )
271 return VLC_EGENERIC;
272
273 /* Set next_input_pts to run with new samplerate */
274 date_Init( &id->next_input_pts, id->fmt_audio.i_rate, 1 );
275 date_Set( &id->next_input_pts, p_audio_buf->i_pts );
276 }
277
278 if( p_sys->b_master_sync )
279 {
280 mtime_t i_pts = date_Get( &id->next_input_pts );
281 mtime_t i_drift = 0;
282
283 if( likely( p_audio_buf->i_pts != VLC_TS_INVALID ) )
284 i_drift = p_audio_buf->i_pts - i_pts;
285
286 if ( unlikely(i_drift > MASTER_SYNC_MAX_DRIFT
287 || i_drift < -MASTER_SYNC_MAX_DRIFT) )
288 {
289 msg_Dbg( p_stream,
290 "audio drift is too high (%"PRId64"), resetting master sync",
291 i_drift );
292 date_Set( &id->next_input_pts, p_audio_buf->i_pts );
293 i_pts = date_Get( &id->next_input_pts );
294 if( likely(p_audio_buf->i_pts != VLC_TS_INVALID ) )
295 i_drift = p_audio_buf->i_pts - i_pts;
296 }
297 p_sys->i_master_drift = i_drift;
298 date_Increment( &id->next_input_pts, p_audio_buf->i_nb_samples );
299 }
300
301 p_audio_buf->i_dts = p_audio_buf->i_pts;
302
303 /* Run filter chain */
304 p_audio_buf = aout_FiltersPlay( id->p_af_chain, p_audio_buf,
305 INPUT_RATE_DEFAULT );
306 if( !p_audio_buf )
307 abort();
308
309 p_audio_buf->i_dts = p_audio_buf->i_pts;
310
311 p_block = id->p_encoder->pf_encode_audio( id->p_encoder, p_audio_buf );
312
313 block_ChainAppend( out, p_block );
314 block_Release( p_audio_buf );
315 }
316
317 return VLC_SUCCESS;
318}

320bool transcode_audio_add( sout_stream_t *p_stream, es_format_t *p_fmt,
321 sout_stream_id_sys_t *id )
322{
323 sout_stream_sys_t *p_sys = p_stream->p_sys;
324
325 msg_Dbg( p_stream,
326 "creating audio transcoding from fcc=`%4.4s' to fcc=`%4.4s'",
327 (char*)&p_fmt->i_codec, (char*)&p_sys->i_acodec );
328
329 /* Complete destination format */
330 id->p_encoder->fmt_out.i_codec = p_sys->i_acodec;
331 id->p_encoder->fmt_out.audio.i_rate = p_sys->i_sample_rate > 0 ?
332 p_sys->i_sample_rate : p_fmt->audio.i_rate;
333 id->p_encoder->fmt_out.i_bitrate = p_sys->i_abitrate;
334 id->p_encoder->fmt_out.audio.i_bitspersample =
335 p_fmt->audio.i_bitspersample;
336 id->p_encoder->fmt_out.audio.i_channels = p_sys->i_channels > 0 ?
337 p_sys->i_channels : p_fmt->audio.i_channels;
338
339 id->p_encoder->fmt_in.audio.i_original_channels =
340 id->p_encoder->fmt_out.audio.i_original_channels =
341 id->p_decoder->fmt_out.audio.i_physical_channels;
342
343 id->p_encoder->fmt_in.audio.i_physical_channels =
344 id->p_encoder->fmt_out.audio.i_physical_channels =
345 pi_channels_maps[id->p_encoder->fmt_out.audio.i_channels];
346
347 /* Build decoder -> filter -> encoder chain */
348 if( transcode_audio_new( p_stream, id ) == VLC_EGENERIC )
349 {
350 msg_Err( p_stream, "cannot create audio chain" );
351 return false;
352 }
353
354 /* Open output stream */
355 id->id = sout_StreamIdAdd( p_stream->p_next, &id->p_encoder->fmt_out );
356 id->b_transcode = true;
357
358 if( !id->id )
359 {
360 transcode_audio_close( id );
361 return false;
362 }
363
364 /* Reinit encoder again later on, when all information from decoders
365 * is available. */
366 if( id->p_encoder->p_module )
367 {
368 module_unneed( id->p_encoder, id->p_encoder->p_module );
369 id->p_encoder->p_module = NULL;
370 if( id->p_encoder->fmt_out.p_extra )
371 {
372 free( id->p_encoder->fmt_out.p_extra );
373 id->p_encoder->fmt_out.p_extra = NULL;
374 id->p_encoder->fmt_out.i_extra = 0;
375 }
376 if( id->p_af_chain != NULL )
377 aout_FiltersDelete( (vlc_object_t *)NULL, id->p_af_chain );
378 id->p_af_chain = NULL;
379 }
380 return true;
381}
