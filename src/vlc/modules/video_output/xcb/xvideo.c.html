106/**
107 * Check that the X server supports the XVideo extension.
108 */
109static bool CheckXVideo (vout_display_t *vd, xcb_connection_t *conn)
110{
111 xcb_xv_query_extension_reply_t *r;
112 xcb_xv_query_extension_cookie_t ck = xcb_xv_query_extension (conn);
113 bool ok = false;
114
115 /* We need XVideo 2.2 for PutImage */
116 r = xcb_xv_query_extension_reply (conn, ck, NULL);
117 if (r == NULL)
118 msg_Dbg (vd, "XVideo extension not available");
119 else
120 if (r->major != 2)
121 msg_Dbg (vd, "XVideo extension v%"PRIu8".%"PRIu8" unknown",
122 r->major, r->minor);
123 else
124 if (r->minor < 2)
125 msg_Dbg (vd, "XVideo extension v%"PRIu8".%"PRIu8" too old",
126 r->major, r->minor);
127 else
128 {
129 msg_Dbg (vd, "using XVideo extension v%"PRIu8".%"PRIu8,
130 r->major, r->minor);
131 ok = true;
132 }
133 free (r);
134 return ok;
135}

137static vlc_fourcc_t ParseFormat (vlc_object_t *obj,
138 const xcb_xv_image_format_info_t *restrict f)
139{
140 switch (f->type)
141 {
142 case XCB_XV_IMAGE_FORMAT_INFO_TYPE_RGB:
143 switch (f->num_planes)
144 {
145 case 1:
146 switch (popcount (f->red_mask | f->green_mask | f->blue_mask))
147 {
148 case 24:
149 if (f->bpp == 32 && f->depth == 32)
150 return VLC_CODEC_ARGB;
151 if (f->bpp == 32 && f->depth == 24)
152 return VLC_CODEC_RGB32;
153 if (f->bpp == 24 && f->depth == 24)
154 return VLC_CODEC_RGB24;
155 break;
156 case 16:
157 if (f->byte_order != ORDER)
158 return 0; /* Mixed endian! */
159 if (f->bpp == 16 && f->depth == 16)
160 return VLC_CODEC_RGB16;
161 break;
162 case 15:
163 if (f->byte_order != ORDER)
164 return 0; /* Mixed endian! */
165 if (f->bpp == 16 && f->depth == 15)
166 return VLC_CODEC_RGB15;
167 break;
168 case 12:
169 if (f->bpp == 16 && f->depth == 12)
170 return VLC_CODEC_RGB12;
171 break;
172 case 8:
173 if (f->bpp == 8 && f->depth == 8)
174 return VLC_CODEC_RGB8;
175 break;
176 }
177 break;
178 }
179 msg_Err (obj, "unknown XVideo RGB format %"PRIx32" (%.4s)",
180 f->id, f->guid);
181 msg_Dbg (obj, " %"PRIu8" planes, %"PRIu8" bits/pixel, "
182 "depth %"PRIu8, f->num_planes, f->bpp, f->depth);
183 break;
184
185 case XCB_XV_IMAGE_FORMAT_INFO_TYPE_YUV:
186 if (f->u_sample_bits != f->v_sample_bits
187 || f->vhorz_u_period != f->vhorz_v_period
188 || f->vvert_u_period != f->vvert_v_period
189 || f->y_sample_bits != 8 || f->u_sample_bits != 8
190 || f->vhorz_y_period != 1 || f->vvert_y_period != 1)
191 goto bad;
192 switch (f->num_planes)
193 {
194 case 1:
195 switch (f->bpp)
196 {
197 /*untested: case 24:
198 if (f->vhorz_u_period == 1 && f->vvert_u_period == 1)
199 return VLC_CODEC_I444;
200 break;*/
201 case 16:
202 if (f->vhorz_u_period == 2 && f->vvert_u_period == 1)
203 {
204 if (!strcmp ((const char *)f->vcomp_order, "YUYV"))
205 return VLC_CODEC_YUYV;
206 if (!strcmp ((const char *)f->vcomp_order, "UYVY"))
207 return VLC_CODEC_UYVY;
208 }
209 break;
210 }
211 break;
212 case 3:
213 switch (f->bpp)
214 {
215 case 12:
216 if (f->vhorz_u_period == 2 && f->vvert_u_period == 2)
217 {
218 if (!strcmp ((const char *)f->vcomp_order, "YVU"))
219 return VLC_CODEC_YV12;
220 if (!strcmp ((const char *)f->vcomp_order, "YUV"))
221 return VLC_CODEC_I420;
222 }
223 }
224 break;
225 }
226 bad:
227 msg_Err (obj, "unknown XVideo YUV format %"PRIx32" (%.4s)", f->id,
228 f->guid);
229 msg_Dbg (obj, " %"PRIu8" planes, %"PRIu32" bits/pixel, "
230 "%"PRIu32"/%"PRIu32"/%"PRIu32" bits/sample", f->num_planes,
231 f->bpp, f->y_sample_bits, f->u_sample_bits, f->v_sample_bits);
232 msg_Dbg (obj, " period: %"PRIu32"/%"PRIu32"/%"PRIu32"x"
233 "%"PRIu32"/%"PRIu32"/%"PRIu32,
234 f->vhorz_y_period, f->vhorz_u_period, f->vhorz_v_period,
235 f->vvert_y_period, f->vvert_u_period, f->vvert_v_period);
236 msg_Warn (obj, " order: %.32s", f->vcomp_order);
237 break;
238 }
239 return 0;
240}

242static bool BetterFormat (vlc_fourcc_t a, const vlc_fourcc_t *tab,
243 unsigned *rankp)
244{
245 for (unsigned i = 0, max = *rankp; i < max && tab[i] != 0; i++)
246 if (tab[i] == a)
247 {
248 *rankp = i;
249 return true;
250 }
251 return false;
252}

254static xcb_xv_query_image_attributes_reply_t *
255FindFormat (vlc_object_t *obj, xcb_connection_t *conn, video_format_t *fmt,
256 const xcb_xv_adaptor_info_t *a, uint32_t *idp)
257{
258 /* Order chromas by preference */
259 vlc_fourcc_t tab[7];
260 const vlc_fourcc_t *chromav = tab;
261
262 vlc_fourcc_t chroma = var_InheritInteger (obj, "xvideo-format-id");
263 if (chroma != 0) /* Forced chroma */
264 {
265 tab[0] = chroma;
266 tab[1] = 0;
267 }
268 else if (vlc_fourcc_IsYUV (fmt->i_chroma)) /* YUV chroma */
269 {
270 chromav = vlc_fourcc_GetYUVFallback (fmt->i_chroma);
271 }
272 else /* RGB chroma */
273 {
274 tab[0] = fmt->i_chroma;
275 tab[1] = VLC_CODEC_RGB32;
276 tab[2] = VLC_CODEC_RGB24;
277 tab[3] = VLC_CODEC_RGB16;
278 tab[4] = VLC_CODEC_RGB15;
279 tab[5] = VLC_CODEC_YUYV;
280 tab[6] = 0;
281 }
282
283 /* Get available image formats */
284 xcb_xv_list_image_formats_reply_t *list =
285 xcb_xv_list_image_formats_reply (conn,
286 xcb_xv_list_image_formats (conn, a->base_id), NULL);
287 if (list == NULL)
288 return NULL;
289
290 /* Check available XVideo chromas */
291 xcb_xv_query_image_attributes_reply_t *attr = NULL;
292 unsigned rank = UINT_MAX;
293
294 for (const xcb_xv_image_format_info_t *f =
295 xcb_xv_list_image_formats_format (list),
296 *f_end =
297 f + xcb_xv_list_image_formats_format_length (list);
298 f < f_end;
299 f++)
300 {
301 chroma = ParseFormat (obj, f);
302 if (chroma == 0)
303 continue;
304
305 /* Oink oink! */
306 if ((chroma == VLC_CODEC_I420 || chroma == VLC_CODEC_YV12)
307 && a->name_size >= 4
308 && !memcmp ("OMAP", xcb_xv_adaptor_info_name (a), 4))
309 {
310 msg_Dbg (obj, "skipping slow I420 format");
311 continue; /* OMAP framebuffer sucks at YUV 4:2:0 */
312 }
313
314 if (!BetterFormat (chroma, chromav, &rank))
315 continue;
316
317 /* VLC pads scanline to 16 pixels internally */
318 unsigned width = (fmt->i_width + 31) & ~31;
319 unsigned height = (fmt->i_height + 15) & ~15;
320 xcb_xv_query_image_attributes_reply_t *i;
321 i = xcb_xv_query_image_attributes_reply (conn,
322 xcb_xv_query_image_attributes (conn, a->base_id, f->id,
323 width, height), NULL);
324 if (i == NULL)
325 continue;
326
327 if (i->width != width || i->height != height)
328 {
329 msg_Warn (obj, "incompatible size %ux%u -> %"PRIu32"x%"PRIu32,
330 fmt->i_width, fmt->i_height,
331 i->width, i->height);
332 var_Create (obj->p_libvlc, "xvideo-res-error", VLC_VAR_BOOL);
333 if (!var_GetBool (obj->p_libvlc, "xvideo-res-error"))
334 {
335 dialog_FatalWait (obj, _("Video acceleration not available"),
336 _("The XVideo rendering acceleration driver does not "
337 "support the required resolution of %ux%u pixels but "
338 "%"PRIu32"x%"PRIu32" pixels instead.\n"
339 "Acceleration will thus be disabled. Performance may "
340 "be degraded severely if the resolution is large."),
341 width, height, i->width, i->height);
342 var_SetBool (obj->p_libvlc, "xvideo-res-error", true);
343 }
344 free (i);
345 continue;
346 }
347
348 fmt->i_chroma = chroma;
349 if (f->type == XCB_XV_IMAGE_FORMAT_INFO_TYPE_RGB)
350 {
351 fmt->i_rmask = f->red_mask;
352 fmt->i_gmask = f->green_mask;
353 fmt->i_bmask = f->blue_mask;
354 }
355 *idp = f->id;
356 free (attr);
357 attr = i;
358 if (rank == 0)
359 break; /* shortcut for perfect match */
360 }
361
362 free (list);
363 return attr;
364}

367/**
368 * Probe the X server.
369 */
370static int Open (vlc_object_t *obj)
371{
372 vout_display_t *vd = (vout_display_t *)obj;
373 vout_display_sys_t *p_sys;
374
375 if (!var_InheritBool (obj, "overlay"))
376 return VLC_EGENERIC;
377 p_sys = malloc (sizeof (*p_sys));
378 if (p_sys == NULL)
379 return VLC_ENOMEM;
380
381 vd->sys = p_sys;
382
383 /* Connect to X */
384 xcb_connection_t *conn;
385 const xcb_screen_t *screen;
386 uint16_t width, height;
387 p_sys->embed = XCB_parent_Create (vd, &conn, &screen, &width, &height);
388 if (p_sys->embed == NULL)
389 {
390 free (p_sys);
391 return VLC_EGENERIC;
392 }
393
394 p_sys->conn = conn;
395 p_sys->att = NULL;
396 p_sys->pool = NULL;
397
398 if (!CheckXVideo (vd, conn))
399 {
400 msg_Warn (vd, "Please enable XVideo 2.2 for faster video display");
401 goto error;
402 }
403
404 p_sys->window = xcb_generate_id (conn);
405 xcb_pixmap_t pixmap = xcb_generate_id (conn);
406
407 /* Cache adaptors infos */
408 xcb_xv_query_adaptors_reply_t *adaptors =
409 xcb_xv_query_adaptors_reply (conn,
410 xcb_xv_query_adaptors (conn, p_sys->embed->handle.xid), NULL);
411 if (adaptors == NULL)
412 goto error;
413
414 int adaptor_selected = var_InheritInteger (obj, "xvideo-adaptor");
415 int adaptor_current = -1;
416
417 /* */
418 video_format_t fmt;
419 vout_display_place_t place;
420
421 p_sys->port = 0;
422 vout_display_PlacePicture (&place, &vd->source, vd->cfg, false);
423 p_sys->width = place.width;
424 p_sys->height = place.height;
425
426 xcb_xv_adaptor_info_iterator_t it;
427 for (it = xcb_xv_query_adaptors_info_iterator (adaptors);
428 it.rem > 0;
429 xcb_xv_adaptor_info_next (&it))
430 {
431 const xcb_xv_adaptor_info_t *a = it.data;
432
433 adaptor_current++;
434 if (adaptor_selected != -1 && adaptor_selected != adaptor_current)
435 continue;
436 if (!(a->type & XCB_XV_TYPE_INPUT_MASK)
437 || !(a->type & XCB_XV_TYPE_IMAGE_MASK))
438 continue;
439
440 /* Look for an image format */
441 video_format_ApplyRotation(&fmt, &vd->fmt);
442 free (p_sys->att);
443 p_sys->att = FindFormat (obj, conn, &fmt, a, &p_sys->id);
444 if (p_sys->att == NULL) /* No acceptable image formats */
445 continue;
446
447 /* Grab a port */
448 for (unsigned i = 0; i < a->num_ports; i++)
449 {
450 xcb_xv_port_t port = a->base_id + i;
451 xcb_xv_grab_port_reply_t *gr =
452 xcb_xv_grab_port_reply (conn,
453 xcb_xv_grab_port (conn, port, XCB_CURRENT_TIME), NULL);
454 uint8_t result = gr ? gr->result : 0xff;
455
456 free (gr);
457 if (result == 0)
458 {
459 p_sys->port = port;
460 goto grabbed_port;
461 }
462 msg_Dbg (vd, "cannot grab port %"PRIu32": Xv error %"PRIu8, port,
463 result);
464 }
465 continue; /* No usable port */
466
467 grabbed_port:
468 /* Found port - initialize selected format */
469 msg_Dbg (vd, "using adaptor %.*s", (int)a->name_size,
470 xcb_xv_adaptor_info_name (a));
471 msg_Dbg (vd, "using port %"PRIu32, p_sys->port);
472 msg_Dbg (vd, "using image format 0x%"PRIx32, p_sys->id);
473
474 /* Look for an X11 visual, create a window */
475 xcb_xv_format_t *f = xcb_xv_adaptor_info_formats (a);
476 for (uint_fast16_t i = a->num_formats; i > 0; i--, f++)
477 {
478 if (f->depth != screen->root_depth)
479 continue; /* this would fail anyway */
480
481 uint32_t mask =
482 XCB_CW_BACK_PIXMAP |
483 XCB_CW_BACK_PIXEL |
484 XCB_CW_BORDER_PIXMAP |
485 XCB_CW_BORDER_PIXEL |
486 XCB_CW_EVENT_MASK |
487 XCB_CW_COLORMAP;
488 const uint32_t list[] = {
489 /* XCB_CW_BACK_PIXMAP */
490 pixmap,
491 /* XCB_CW_BACK_PIXEL */
492 screen->black_pixel,
493 /* XCB_CW_BORDER_PIXMAP */
494 pixmap,
495 /* XCB_CW_BORDER_PIXEL */
496 screen->black_pixel,
497 /* XCB_CW_EVENT_MASK */
498 XCB_EVENT_MASK_VISIBILITY_CHANGE,
499 /* XCB_CW_COLORMAP */
500 screen->default_colormap,
501 };
502 xcb_void_cookie_t c;
503
504 xcb_create_pixmap (conn, f->depth, pixmap, screen->root, 1, 1);
505 c = xcb_create_window_checked (conn, f->depth, p_sys->window,
506 p_sys->embed->handle.xid, place.x, place.y,
507 place.width, place.height, 0, XCB_WINDOW_CLASS_INPUT_OUTPUT,
508 f->visual, mask, list);
509 xcb_map_window (conn, p_sys->window);
510
511 if (!XCB_error_Check (vd, conn, "cannot create X11 window", c))
512 {
513 msg_Dbg (vd, "using X11 visual ID 0x%"PRIx32
514 " (depth: %"PRIu8")", f->visual, f->depth);
515 msg_Dbg (vd, "using X11 window 0x%08"PRIx32, p_sys->window);
516 goto created_window;
517 }
518 }
519 xcb_xv_ungrab_port (conn, p_sys->port, XCB_CURRENT_TIME);
520 p_sys->port = 0;
521 msg_Dbg (vd, "no usable X11 visual");
522 continue; /* No workable XVideo format (visual/depth) */
523
524 created_window:
525 break;
526 }
527 free (adaptors);
528 if (!p_sys->port)
529 {
530 msg_Err (vd, "no available XVideo adaptor");
531 goto error;
532 }
533
534 /* Create graphic context */
535 p_sys->gc = xcb_generate_id (conn);
536 xcb_create_gc (conn, p_sys->gc, p_sys->window, 0, NULL);
537 msg_Dbg (vd, "using X11 graphic context 0x%08"PRIx32, p_sys->gc);
538
539 /* Disable color keying if applicable */
540 {
541 xcb_intern_atom_reply_t *r =
542 xcb_intern_atom_reply (conn,
543 xcb_intern_atom (conn, 1, 21, "XV_AUTOPAINT_COLORKEY"), NULL);
544 if (r != NULL && r->atom != 0)
545 xcb_xv_set_port_attribute(conn, p_sys->port, r->atom, 1);
546 free(r);
547 }
548
549 /* Colour space */
550 {
551 xcb_intern_atom_reply_t *r =
552 xcb_intern_atom_reply (conn,
553 xcb_intern_atom (conn, 1, 13, "XV_ITURBT_709"), NULL);
554 if (r != NULL && r->atom != 0)
555 xcb_xv_set_port_attribute(conn, p_sys->port, r->atom,
556 fmt.i_height > 576);
557 free(r);
558 }
559
560 /* Create cursor */
561 p_sys->cursor = XCB_cursor_Create (conn, screen);
562
563 p_sys->shm = XCB_shm_Check (obj, conn);
564 p_sys->visible = false;
565
566 /* */
567 vout_display_info_t info = vd->info;
568 info.has_pictures_invalid = false;
569 info.has_event_thread = true;
570
571 /* Setup vout_display_t once everything is fine */
572 p_sys->swap_uv = vlc_fourcc_AreUVPlanesSwapped (fmt.i_chroma,
573 vd->fmt.i_chroma);
574 if (p_sys->swap_uv)
575 fmt.i_chroma = vd->fmt.i_chroma;
576 vd->fmt = fmt;
577 vd->info = info;
578
579 vd->pool = Pool;
580 vd->prepare = NULL;
581 vd->display = Display;
582 vd->control = Control;
583 vd->manage = Manage;
584
585 /* */
586 bool is_fullscreen = vd->cfg->is_fullscreen;
587 if (is_fullscreen && vout_window_SetFullScreen (p_sys->embed, true))
588 is_fullscreen = false;
589 vout_display_SendEventFullscreen (vd, is_fullscreen);
590 vout_display_SendEventDisplaySize (vd, width, height, is_fullscreen);
591
592 return VLC_SUCCESS;
593
594error:
595 Close (obj);
596 return VLC_EGENERIC;
597}

600/**
601 * Disconnect from the X server.
602 */
603static void Close (vlc_object_t *obj)
604{
605 vout_display_t *vd = (vout_display_t *)obj;
606 vout_display_sys_t *p_sys = vd->sys;
607
608 if (p_sys->pool)
609 picture_pool_Delete (p_sys->pool);
610
611 /* show the default cursor */
612 xcb_change_window_attributes (p_sys->conn, p_sys->embed->handle.xid, XCB_CW_CURSOR,
613 &(uint32_t) { XCB_CURSOR_NONE });
614 xcb_flush (p_sys->conn);
615
616 free (p_sys->att);
617 xcb_disconnect (p_sys->conn);
618 vout_display_DeleteWindow (vd, p_sys->embed);
619 free (p_sys);
620}

622static void PoolAlloc (vout_display_t *vd, unsigned requested_count)
623{
624 vout_display_sys_t *p_sys = vd->sys;
625
626 const uint32_t *pitches= xcb_xv_query_image_attributes_pitches (p_sys->att);
627 const uint32_t *offsets= xcb_xv_query_image_attributes_offsets (p_sys->att);
628 const unsigned num_planes= __MIN(p_sys->att->num_planes, PICTURE_PLANE_MAX);
629 p_sys->data_size = p_sys->att->data_size;
630
631 picture_resource_t res = { NULL };
632 for (unsigned i = 0; i < num_planes; i++)
633 {
634 uint32_t data_size;
635 data_size = (i < num_planes - 1) ? offsets[i+1] : p_sys->data_size;
636
637 res.p[i].i_lines = (data_size - offsets[i]) / pitches[i];
638 res.p[i].i_pitch = pitches[i];
639 }
640
641 picture_t *pic_array[MAX_PICTURES];
642 requested_count = __MIN(requested_count, MAX_PICTURES);
643
644 unsigned count;
645 for (count = 0; count < requested_count; count++)
646 {
647 xcb_shm_seg_t seg = p_sys->shm ? xcb_generate_id (p_sys->conn) : 0;
648
649 if (XCB_picture_Alloc (vd, &res, p_sys->data_size, p_sys->conn, seg))
650 break;
651
652 /* Allocate further planes as specified by XVideo */
653 /* We assume that offsets[0] is zero */
654 for (unsigned i = 1; i < num_planes; i++)
655 res.p[i].p_pixels = res.p[0].p_pixels + offsets[i];
656
657 if (p_sys->swap_uv)
658 { /* YVU: swap U and V planes */
659 uint8_t *buf = res.p[2].p_pixels;
660 res.p[2].p_pixels = res.p[1].p_pixels;
661 res.p[1].p_pixels = buf;
662 }
663
664 pic_array[count] = XCB_picture_NewFromResource (&vd->fmt, &res);
665 if (unlikely(pic_array[count] == NULL))
666 break;
667 }
668 xcb_flush (p_sys->conn);
669
670 if (count == 0)
671 return;
672
673 p_sys->pool = picture_pool_New (count, pic_array);
674 if (unlikely(p_sys->pool == NULL))
675 while (count > 0)
676 picture_Release(pic_array[--count]);
677}

692/**
693 * Sends an image to the X server.
694 */
695static void Display (vout_display_t *vd, picture_t *pic, subpicture_t *subpicture)
696{
697 vout_display_sys_t *p_sys = vd->sys;
698 xcb_shm_seg_t segment = XCB_picture_GetSegment(pic);
699 xcb_void_cookie_t ck;
700 video_format_t fmt;
701
702 if (!p_sys->visible)
703 goto out;
704
705 video_format_ApplyRotation(&fmt, &vd->source);
706
707 if (segment)
708 ck = xcb_xv_shm_put_image_checked (p_sys->conn, p_sys->port,
709 p_sys->window, p_sys->gc, segment, p_sys->id, 0,
710 /* Src: */ fmt.i_x_offset, fmt.i_y_offset,
711 fmt.i_visible_width, fmt.i_visible_height,
712 /* Dst: */ 0, 0, p_sys->width, p_sys->height,
713 /* Memory: */ pic->p->i_pitch / pic->p->i_pixel_pitch,
714 pic->p->i_lines, false);
715 else
716 ck = xcb_xv_put_image_checked (p_sys->conn, p_sys->port, p_sys->window,
717 p_sys->gc, p_sys->id,
718 fmt.i_x_offset, fmt.i_y_offset,
719 fmt.i_visible_width, fmt.i_visible_height,
720 0, 0, p_sys->width, p_sys->height,
721 pic->p->i_pitch / pic->p->i_pixel_pitch,
722 pic->p->i_lines,
723 p_sys->data_size, pic->p->p_pixels);
724
725 /* Wait for reply. See x11.c for rationale. */
726 xcb_generic_error_t *e = xcb_request_check (p_sys->conn, ck);
727 if (e != NULL)
728 {
729 msg_Dbg (vd, "%s: X11 error %d", "cannot put image", e->error_code);
730 free (e);
731 }
732out:
733 picture_Release (pic);
734 (void)subpicture;
735}

737static int Control (vout_display_t *vd, int query, va_list ap)
738{
739 vout_display_sys_t *p_sys = vd->sys;
740
741 switch (query)
742 {
743 case VOUT_DISPLAY_CHANGE_FULLSCREEN:
744 {
745 const vout_display_cfg_t *c = va_arg (ap, const vout_display_cfg_t *);
746 return vout_window_SetFullScreen (p_sys->embed, c->is_fullscreen);
747 }
748
749 case VOUT_DISPLAY_CHANGE_DISPLAY_SIZE:
750 case VOUT_DISPLAY_CHANGE_DISPLAY_FILLED:
751 case VOUT_DISPLAY_CHANGE_ZOOM:
752 case VOUT_DISPLAY_CHANGE_SOURCE_ASPECT:
753 case VOUT_DISPLAY_CHANGE_SOURCE_CROP:
754 {
755 const vout_display_cfg_t *cfg;
756 const video_format_t *source;
757
758 if (query == VOUT_DISPLAY_CHANGE_SOURCE_ASPECT
759 || query == VOUT_DISPLAY_CHANGE_SOURCE_CROP)
760 {
761 source = (const video_format_t *)va_arg (ap, const video_format_t *);
762 cfg = vd->cfg;
763 }
764 else
765 {
766 source = &vd->source;
767 cfg = (const vout_display_cfg_t*)va_arg (ap, const vout_display_cfg_t *);
768 }
769
770 if (query == VOUT_DISPLAY_CHANGE_DISPLAY_SIZE && va_arg (ap, int))
771 {
772 vout_window_SetSize (p_sys->embed,
773 cfg->display.width, cfg->display.height);
774 return VLC_EGENERIC; /* Always fail. See x11.c for rationale. */
775 }
776
777 vout_display_place_t place;
778 vout_display_PlacePicture (&place, source, cfg, false);
779 p_sys->width = place.width;
780 p_sys->height = place.height;
781
782 /* Move the picture within the window */
783 const uint32_t values[] = { place.x, place.y,
784 place.width, place.height, };
785 xcb_configure_window (p_sys->conn, p_sys->window,
786 XCB_CONFIG_WINDOW_X | XCB_CONFIG_WINDOW_Y
787 | XCB_CONFIG_WINDOW_WIDTH | XCB_CONFIG_WINDOW_HEIGHT,
788 values);
789 xcb_flush (p_sys->conn);
790 return VLC_SUCCESS;
791 }
792 case VOUT_DISPLAY_CHANGE_WINDOW_STATE:
793 {
794 unsigned state = va_arg (ap, unsigned);
795 return vout_window_SetState (p_sys->embed, state);
796 }
797
798 /* Hide the mouse. It will be send when
799 * vout_display_t::info.b_hide_mouse is false */
800 case VOUT_DISPLAY_HIDE_MOUSE:
801 xcb_change_window_attributes (p_sys->conn, p_sys->embed->handle.xid,
802 XCB_CW_CURSOR, &(uint32_t){ p_sys->cursor });
803 xcb_flush (p_sys->conn);
804 return VLC_SUCCESS;
805 case VOUT_DISPLAY_RESET_PICTURES:
806 assert(0);
807 default:
808 msg_Err (vd, "Unknown request in XCB vout display");
809 return VLC_EGENERIC;
810 }
811}

820static int EnumAdaptors (vlc_object_t *obj, const char *var,
821 int64_t **vp, char ***tp)
822{
823 /* Connect to X */
824 char *display = var_InheritString (obj, "x11-display");
825 xcb_connection_t *conn;
826 int snum;
827
828 conn = xcb_connect (display, &snum);
829 free (display);
830 if (xcb_connection_has_error (conn) /*== NULL*/)
831 return -1;
832
833 /* Find configured screen */
834 const xcb_setup_t *setup = xcb_get_setup (conn);
835 const xcb_screen_t *scr = NULL;
836 for (xcb_screen_iterator_t i = xcb_setup_roots_iterator (setup);
837 i.rem > 0; xcb_screen_next (&i))
838 {
839 if (snum == 0)
840 {
841 scr = i.data;
842 break;
843 }
844 snum--;
845 }
846
847 if (scr == NULL)
848 {
849 xcb_disconnect (conn);
850 return -1;
851 }
852
853 xcb_xv_query_adaptors_reply_t *adaptors =
854 xcb_xv_query_adaptors_reply (conn,
855 xcb_xv_query_adaptors (conn, scr->root), NULL);
856 xcb_disconnect (conn);
857 if (adaptors == NULL)
858 return -1;
859
860 xcb_xv_adaptor_info_iterator_t it;
861 size_t n = 0;
862
863 for (it = xcb_xv_query_adaptors_info_iterator (adaptors);
864 it.rem > 0;
865 xcb_xv_adaptor_info_next (&it))
866 {
867 const xcb_xv_adaptor_info_t *a = it.data;
868
869 if ((a->type & XCB_XV_TYPE_INPUT_MASK)
870 && (a->type & XCB_XV_TYPE_IMAGE_MASK))
871 n++;
872 }
873
874 int64_t *values = xmalloc ((n + 1) * sizeof (*values));
875 char **texts = xmalloc ((n + 1) * sizeof (*texts));
876 *vp = values;
877 *tp = texts;
878
879 *(values++) = -1;
880 *(texts++) = strdup (N_("Auto"));
881
882 for (it = xcb_xv_query_adaptors_info_iterator (adaptors), n = -1;
883 it.rem > 0;
884 xcb_xv_adaptor_info_next (&it))
885 {
886 const xcb_xv_adaptor_info_t *a = it.data;
887
888 n++;
889
890 if (!(a->type & XCB_XV_TYPE_INPUT_MASK)
891 || !(a->type & XCB_XV_TYPE_IMAGE_MASK))
892 continue;
893
894 *(values++) = n;
895 *(texts++) = strndup (xcb_xv_adaptor_info_name (a), a->name_size);
896 }
897 free (adaptors);
898 (void) obj; (void) var;
899 return values - *vp;
900}
