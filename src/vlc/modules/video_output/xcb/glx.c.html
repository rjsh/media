58struct vout_display_sys_t
59{
60 xcb_connection_t *conn; /**< XCB connection */
61 vout_window_t *embed; /* VLC window (when windowed) */
62 vlc_gl_t *gl;
63
64 xcb_cursor_t cursor; /* blank cursor */
65 bool visible; /* whether to draw */
66
67 vout_display_opengl_t *vgl;
68 picture_pool_t *pool; /* picture pool */
69};

77/**
78 * Probe the X server.
79 */
80static int Open (vlc_object_t *obj)
81{
82 vout_display_t *vd = (vout_display_t *)obj;
83 vout_display_sys_t *sys = malloc (sizeof (*sys));
84
85 if (sys == NULL)
86 return VLC_ENOMEM;
87
88 sys->vgl = NULL;
89 sys->pool = NULL;
90
91 /* Get window, connect to X server (via XCB) */
92 xcb_connection_t *conn;
93 const xcb_screen_t *scr;
94 uint16_t width, height;
95 sys->embed = XCB_parent_Create (vd, &conn, &scr, &width, &height);
96 if (sys->embed == NULL)
97 {
98 free (sys);
99 return VLC_EGENERIC;
100 }
101
102 sys->conn = conn;
103 sys->gl = vlc_gl_Create (sys->embed, VLC_OPENGL, "glx");
104 if (sys->gl == NULL)
105 goto error;
106
107 const vlc_fourcc_t *spu_chromas;
108
109 if (vlc_gl_MakeCurrent (sys->gl))
110 goto error;
111
112 sys->vgl = vout_display_opengl_New (&vd->fmt, &spu_chromas, sys->gl);
113 vlc_gl_ReleaseCurrent (sys->gl);
114 if (sys->vgl == NULL)
115 goto error;
116
117 sys->cursor = XCB_cursor_Create (conn, scr);
118 sys->visible = false;
119
120 /* Setup vout_display_t once everything is fine */
121 vd->sys = sys;
122 vd->info.has_pictures_invalid = false;
123 vd->info.has_event_thread = true;
124 vd->info.subpicture_chromas = spu_chromas;
125 vd->pool = Pool;
126 vd->prepare = PictureRender;
127 vd->display = PictureDisplay;
128 vd->control = Control;
129 vd->manage = Manage;
130
131 /* */
132 bool is_fullscreen = vd->cfg->is_fullscreen;
133 if (is_fullscreen && vout_window_SetFullScreen (sys->embed, true))
134 is_fullscreen = false;
135 vout_display_SendEventFullscreen (vd, is_fullscreen);
136 vout_display_SendEventDisplaySize (vd, width, height, is_fullscreen);
137
138 return VLC_SUCCESS;
139
140error:
141 if (sys->gl != NULL)
142 vlc_gl_Destroy (sys->gl);
143 xcb_disconnect (sys->conn);
144 vout_display_DeleteWindow (vd, sys->embed);
145 free (sys);
146 return VLC_EGENERIC;
147}

150/**
151 * Disconnect from the X server.
152 */
153static void Close (vlc_object_t *obj)
154{
155 vout_display_t *vd = (vout_display_t *)obj;
156 vout_display_sys_t *sys = vd->sys;
157
158 vlc_gl_MakeCurrent (sys->gl);
159 vout_display_opengl_Delete (sys->vgl);
160 vlc_gl_ReleaseCurrent (sys->gl);
161 vlc_gl_Destroy (sys->gl);
162
163 /* show the default cursor */
164 xcb_change_window_attributes (sys->conn, sys->embed->handle.xid,
165 XCB_CW_CURSOR, &(uint32_t) { XCB_CURSOR_NONE });
166 xcb_flush (sys->conn);
167 xcb_disconnect (sys->conn);
168
169 vout_display_DeleteWindow (vd, sys->embed);
170 free (sys);
171}

173/**
174 * Return a direct buffer
175 */
176static picture_pool_t *Pool (vout_display_t *vd, unsigned requested_count)
177{
178 vout_display_sys_t *sys = vd->sys;
179
180 if (!sys->pool)
181 {
182 vlc_gl_MakeCurrent (sys->gl);
183 sys->pool = vout_display_opengl_GetPool (sys->vgl, requested_count);
184 vlc_gl_ReleaseCurrent (sys->gl);
185 }
186 return sys->pool;
187}

189static void PictureRender (vout_display_t *vd, picture_t *pic, subpicture_t *subpicture)
190{
191 vout_display_sys_t *sys = vd->sys;
192
193 vlc_gl_MakeCurrent (sys->gl);
194 vout_display_opengl_Prepare (sys->vgl, pic, subpicture);
195 vlc_gl_ReleaseCurrent (sys->gl);
196}
197
198static void PictureDisplay (vout_display_t *vd, picture_t *pic, subpicture_t *subpicture)
199{
200 vout_display_sys_t *sys = vd->sys;
201
202 vlc_gl_MakeCurrent (sys->gl);
203 vout_display_opengl_Display (sys->vgl, &vd->source);
204 vlc_gl_ReleaseCurrent (sys->gl);
205
206 picture_Release (pic);
207 if (subpicture)
208 subpicture_Delete(subpicture);
209}

211static int Control (vout_display_t *vd, int query, va_list ap)
212{
213 vout_display_sys_t *sys = vd->sys;
214
215 switch (query)
216 {
217 case VOUT_DISPLAY_CHANGE_FULLSCREEN:
218 {
219 const vout_display_cfg_t *c = va_arg (ap, const vout_display_cfg_t *);
220 return vout_window_SetFullScreen (sys->embed, c->is_fullscreen);
221 }
222
223 case VOUT_DISPLAY_CHANGE_WINDOW_STATE:
224 {
225 unsigned state = va_arg (ap, unsigned);
226 return vout_window_SetState (sys->embed, state);
227 }
228
229 case VOUT_DISPLAY_CHANGE_DISPLAY_SIZE:
230 case VOUT_DISPLAY_CHANGE_DISPLAY_FILLED:
231 case VOUT_DISPLAY_CHANGE_ZOOM:
232 case VOUT_DISPLAY_CHANGE_SOURCE_ASPECT:
233 case VOUT_DISPLAY_CHANGE_SOURCE_CROP:
234 {
235 const vout_display_cfg_t *cfg;
236 const video_format_t *source;
237
238 if (query == VOUT_DISPLAY_CHANGE_SOURCE_ASPECT
239 || query == VOUT_DISPLAY_CHANGE_SOURCE_CROP)
240 {
241 source = (const video_format_t *)va_arg (ap, const video_format_t *);
242 cfg = vd->cfg;
243 }
244 else
245 {
246 source = &vd->source;
247 cfg = (const vout_display_cfg_t*)va_arg (ap, const vout_display_cfg_t *);
248 }
249
250 /* */
251 if (query == VOUT_DISPLAY_CHANGE_DISPLAY_SIZE && va_arg (ap, int))
252 {
253 vout_window_SetSize (sys->embed,
254 cfg->display.width, cfg->display.height);
255 return VLC_EGENERIC; /* Always fail. See x11.c for rationale. */
256 }
257
258 vout_display_place_t place;
259 vout_display_PlacePicture (&place, source, cfg, false);
260
261 vlc_gl_MakeCurrent (sys->gl);
262 glViewport (place.x, place.y, place.width, place.height);
263 vlc_gl_ReleaseCurrent (sys->gl);
264 return VLC_SUCCESS;
265 }
266
267 /* Hide the mouse. It will be send when
268 * vout_display_t::info.b_hide_mouse is false */
269 case VOUT_DISPLAY_HIDE_MOUSE:
270 xcb_change_window_attributes (sys->conn, sys->embed->handle.xid,
271 XCB_CW_CURSOR, &(uint32_t){ sys->cursor });
272 xcb_flush (sys->conn);
273 return VLC_SUCCESS;
274
275 case VOUT_DISPLAY_GET_OPENGL:
276 {
277 vlc_gl_t **gl = va_arg (ap, vlc_gl_t **);
278 *gl = sys->gl;
279 return VLC_SUCCESS;
280 }
281
282 case VOUT_DISPLAY_RESET_PICTURES:
283 assert (0);
284 default:
285 msg_Err (vd, "Unknown request in XCB vout display");
286 return VLC_EGENERIC;
287 }
288}

291{
292 vout_display_sys_t *sys = vd->sys;

294 XCB_Manage (vd, sys->conn, &sys->visible);
