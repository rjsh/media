186/*****************************************************************************
187 * Open:
188 *****************************************************************************/
189static int Open( vlc_object_t *p_this )
190{
191 sout_mux_t *p_mux = (sout_mux_t*)p_this;
192 sout_mux_sys_t *p_sys;
193
194 msg_Dbg( p_mux, "asf muxer opened" );
195 config_ChainParse( p_mux, SOUT_CFG_PREFIX, ppsz_sout_options, p_mux->p_cfg );
196
197 p_mux->pf_control = Control;
198 p_mux->pf_addstream = AddStream;
199 p_mux->pf_delstream = DelStream;
200 p_mux->pf_mux = Mux;
201
202 p_mux->p_sys = p_sys = malloc( sizeof( sout_mux_sys_t ) );
203 if( !p_sys )
204 return VLC_ENOMEM;
205 p_sys->b_asf_http = p_mux->psz_mux && !strcmp( p_mux->psz_mux, "asfh" );
206 if( p_sys->b_asf_http )
207 {
208 msg_Dbg( p_mux, "creating asf stream to be used with mmsh" );
209 }
210 p_sys->pk = NULL;
211 p_sys->i_pk_used = 0;
212 p_sys->i_pk_frame = 0;
213 p_sys->i_dts_first =
214 p_sys->i_dts_last = VLC_TS_INVALID;
215 p_sys->i_preroll_time = 2000;
216 p_sys->i_bitrate = 0;
217 p_sys->i_bitrate_override = 0;
218 p_sys->i_seq = 0;
219 p_sys->p_tracks = vlc_array_new();
220
221 p_sys->b_write_header = true;
222 p_sys->i_packet_size = var_InheritInteger( p_mux, "sout-asf-packet-size" );
223 p_sys->i_bitrate_override = var_InheritInteger( p_mux, "sout-asf-bitrate-override" );
224 msg_Dbg( p_mux, "Packet size %d", p_sys->i_packet_size);
225 if (p_sys->i_bitrate_override)
226 msg_Dbg( p_mux, "Bitrate override %"PRId64, p_sys->i_bitrate_override);
227 p_sys->i_packet_count= 0;
228
229 /* Generate a random fid */
230 p_sys->fid.Data1 = 0xbabac001;
231 vlc_rand_bytes(&p_sys->fid.Data2, sizeof(p_sys->fid.Data2));
232 vlc_rand_bytes(&p_sys->fid.Data3, sizeof(p_sys->fid.Data3));
233 vlc_rand_bytes(p_sys->fid.Data4, sizeof(p_sys->fid.Data4));
234
235 /* Meta data */
236 p_sys->psz_title = var_GetString( p_mux, SOUT_CFG_PREFIX "title" );
237 p_sys->psz_author = var_GetString( p_mux, SOUT_CFG_PREFIX "author" );
238 p_sys->psz_copyright = var_GetString( p_mux, SOUT_CFG_PREFIX "copyright" );
239 p_sys->psz_comment = var_GetString( p_mux, SOUT_CFG_PREFIX "comment" );
240 p_sys->psz_rating = var_GetString( p_mux, SOUT_CFG_PREFIX "rating" );
241
242 msg_Dbg( p_mux, "meta data: title='%s', author='%s', copyright='%s', "
243 "comment='%s', rating='%s'",
244 p_sys->psz_title, p_sys->psz_author, p_sys->psz_copyright,
245 p_sys->psz_comment, p_sys->psz_rating );
246
247 return VLC_SUCCESS;
248}

250/*****************************************************************************
251 * Close:
252 *****************************************************************************/
253static void Close( vlc_object_t * p_this )
254{
255 sout_mux_t *p_mux = (sout_mux_t*)p_this;
256 sout_mux_sys_t *p_sys = p_mux->p_sys;
257 block_t *out;
258
259 msg_Dbg( p_mux, "Asf muxer closed" );
260
261 /* Flush last packet if any */
262 if( (out = asf_packet_flush( p_mux ) ) )
263 {
264 sout_AccessOutWrite( p_mux->p_access, out );
265 }
266
267 if( ( out = asf_stream_end_create( p_mux ) ) )
268 {
269 sout_AccessOutWrite( p_mux->p_access, out );
270 }
271
272 /* rewrite header */
273 if( sout_AccessOutSeek( p_mux->p_access, 0 ) == VLC_SUCCESS )
274 {
275 out = asf_header_create( p_mux, false );
276 sout_AccessOutWrite( p_mux->p_access, out );
277 }
278
279
280 for( int i = 0; i < vlc_array_count( p_sys->p_tracks ); i++ )
281 {
282 asf_track_t *track = (asf_track_t *)vlc_array_item_at_index( p_sys->p_tracks, i );
283 free( track->p_extra );
284 es_format_Clean( &track->fmt );
285 }
286
287 vlc_array_clear( p_sys->p_tracks );
288
289 vlc_array_destroy( p_sys->p_tracks );
290
291 free( p_sys->psz_title );
292 free( p_sys->psz_author );
293 free( p_sys->psz_copyright );
294 free( p_sys->psz_comment );
295 free( p_sys->psz_rating );
296 free( p_sys );
297}

299/*****************************************************************************
300 * Capability:
301 *****************************************************************************/
302static int Control( sout_mux_t *p_mux, int i_query, va_list args )
303{
304 sout_mux_sys_t *p_sys = p_mux->p_sys;
305 bool *pb_bool;
306 char **ppsz;
307
308 switch( i_query )
309 {
310 case MUX_CAN_ADD_STREAM_WHILE_MUXING:
311 pb_bool = (bool*)va_arg( args, bool * );
312 if( p_sys->b_asf_http ) *pb_bool = true;
313 else *pb_bool = false;
314 return VLC_SUCCESS;
315
316 case MUX_GET_ADD_STREAM_WAIT:
317 pb_bool = (bool*)va_arg( args, bool * );
318 *pb_bool = true;
319 return VLC_SUCCESS;
320
321 case MUX_GET_MIME:
322 ppsz = (char**)va_arg( args, char ** );
323 if( p_sys->b_asf_http )
324 *ppsz = strdup( "video/x-ms-asf-stream" );
325 else
326 *ppsz = strdup( "video/x-ms-asf" );
327 return VLC_SUCCESS;
328
329 default:
330 return VLC_EGENERIC;
331 }
332}

334/*****************************************************************************
335 * AddStream:
336 *****************************************************************************/
337static int AddStream( sout_mux_t *p_mux, sout_input_t *p_input )
338{
339 sout_mux_sys_t *p_sys = p_mux->p_sys;
340 asf_track_t *tk;
341 bo_t bo;
342
343 msg_Dbg( p_mux, "adding input" );
344 if( vlc_array_count( p_sys->p_tracks ) >= MAX_ASF_TRACKS )
345 {
346 msg_Dbg( p_mux, "cannot add this track (too much tracks)" );
347 return VLC_EGENERIC;
348 }
349
350 tk = p_input->p_sys = malloc( sizeof( asf_track_t ) );
351 memset( tk, 0, sizeof( *tk ) );
352 tk->i_cat = p_input->p_fmt->i_cat;
353 tk->i_sequence = 0;
354 tk->b_audio_correction = 0;
355 tk->b_extended = false;
356
357 switch( tk->i_cat )
358 {
359 case AUDIO_ES:
360 {
361 int i_blockalign = p_input->p_fmt->audio.i_blockalign;
362 int i_bitspersample = p_input->p_fmt->audio.i_bitspersample;
363 int i_extra = 0;
364
365 switch( p_input->p_fmt->i_codec )
366 {
367 case VLC_CODEC_A52:
368 tk->i_tag = WAVE_FORMAT_A52;
369 tk->psz_name = "A/52";
370 i_bitspersample = 0;
371 break;
372 case VLC_CODEC_MP4A:
373 tk->i_tag = WAVE_FORMAT_AAC;
374 tk->psz_name = "MPEG-4 Audio";
375 i_bitspersample = 0;
376 break;
377 case VLC_CODEC_MP3:
378 tk->psz_name = "MPEG Audio Layer 3";
379 tk->i_tag = WAVE_FORMAT_MPEGLAYER3;
380 i_bitspersample = 0;
381 i_blockalign = 1;
382 i_extra = 12;
383 break;
384 case VLC_CODEC_MPGA:
385 tk->psz_name = "MPEG Audio Layer 1/2";
386 tk->i_tag = WAVE_FORMAT_MPEG;
387 i_bitspersample = 0;
388 i_blockalign = 1;
389 i_extra = 22;
390 break;
391 case VLC_CODEC_WMA1:
392 tk->psz_name = "Windows Media Audio v1";
393 tk->i_tag = WAVE_FORMAT_WMA1;
394 tk->b_audio_correction = true;
395 break;
396 case VLC_CODEC_WMA2:
397 tk->psz_name= "Windows Media Audio (v2) 7, 8 and 9 Series";
398 tk->i_tag = WAVE_FORMAT_WMA2;
399 tk->b_audio_correction = true;
400 break;
401 case VLC_CODEC_WMAP:
402 tk->psz_name = "Windows Media Audio 9 Professional";
403 tk->i_tag = WAVE_FORMAT_WMAP;
404 tk->b_audio_correction = true;
405 break;
406 case VLC_CODEC_WMAL:
407 tk->psz_name = "Windows Media Audio 9 Lossless";
408 tk->i_tag = WAVE_FORMAT_WMAL;
409 tk->b_audio_correction = true;
410 break;
411 /* raw codec */
412 case VLC_CODEC_U8:
413 tk->psz_name = "Raw audio 8bits";
414 tk->i_tag = WAVE_FORMAT_PCM;
415 i_blockalign= p_input->p_fmt->audio.i_channels;
416 i_bitspersample = 8;
417 break;
418 case VLC_CODEC_S16L:
419 tk->psz_name = "Raw audio 16bits";
420 tk->i_tag = WAVE_FORMAT_PCM;
421 i_blockalign= 2 * p_input->p_fmt->audio.i_channels;
422 i_bitspersample = 16;
423 break;
424 case VLC_CODEC_S24L:
425 tk->psz_name = "Raw audio 24bits";
426 tk->i_tag = WAVE_FORMAT_PCM;
427 i_blockalign= 3 * p_input->p_fmt->audio.i_channels;
428 i_bitspersample = 24;
429 break;
430 case VLC_CODEC_S32L:
431 tk->psz_name = "Raw audio 32bits";
432 tk->i_tag = WAVE_FORMAT_PCM;
433 i_blockalign= 4 * p_input->p_fmt->audio.i_channels;
434 i_bitspersample = 32;
435 break;
436 default:
437 return VLC_EGENERIC;
438 }
439
440 tk->i_extra = sizeof( WAVEFORMATEX ) +
441 p_input->p_fmt->i_extra + i_extra;
442 tk->p_extra = malloc( tk->i_extra );
443 if( !tk->p_extra )
444 return VLC_ENOMEM;
445 bo_init( &bo, tk->p_extra, tk->i_extra );
446 bo_addle_u16( &bo, tk->i_tag );
447 bo_addle_u16( &bo, p_input->p_fmt->audio.i_channels );
448 bo_addle_u32( &bo, p_input->p_fmt->audio.i_rate );
449 bo_addle_u32( &bo, p_input->p_fmt->i_bitrate / 8 );
450 bo_addle_u16( &bo, i_blockalign );
451 tk->i_blockalign = i_blockalign;
452 bo_addle_u16( &bo, i_bitspersample );
453 if( p_input->p_fmt->i_extra > 0 )
454 {
455 bo_addle_u16( &bo, p_input->p_fmt->i_extra );
456 bo_add_mem ( &bo, p_input->p_fmt->p_extra,
457 p_input->p_fmt->i_extra );
458 }
459 else
460 {
461 bo_addle_u16( &bo, i_extra );
462 if( tk->i_tag == WAVE_FORMAT_MPEGLAYER3 )
463 {
464 msg_Dbg( p_mux, "adding mp3 header" );
465 bo_addle_u16( &bo, 1 ); /* wId */
466 bo_addle_u32( &bo, 2 ); /* fdwFlags */
467 bo_addle_u16( &bo, 1152 ); /* nBlockSize */
468 bo_addle_u16( &bo, 1 ); /* nFramesPerBlock */
469 bo_addle_u16( &bo, 1393 ); /* nCodecDelay */
470 }
471 else if( tk->i_tag == WAVE_FORMAT_MPEG )
472 {
473 msg_Dbg( p_mux, "adding mp2 header" );
474 bo_addle_u16( &bo, 2 ); /* fwHeadLayer */
475 bo_addle_u32( &bo, p_input->p_fmt->i_bitrate );
476 bo_addle_u16( &bo, p_input->p_fmt->audio.i_channels == 2 ?1:8 );
477 bo_addle_u16( &bo, 0 ); /* fwHeadModeExt */
478 bo_addle_u16( &bo, 1 ); /* wHeadEmphasis */
479 bo_addle_u16( &bo, 16 ); /* fwHeadFlags */
480 bo_addle_u32( &bo, 0 ); /* dwPTSLow */
481 bo_addle_u32( &bo, 0 ); /* dwPTSHigh */
482 }
483 }
484
485 if( p_input->p_fmt->i_bitrate > 24000 )
486 {
487 p_sys->i_bitrate += p_input->p_fmt->i_bitrate;
488 }
489 else
490 {
491 p_sys->i_bitrate += 128000;
492 }
493 if (p_sys->i_bitrate_override)
494 p_sys->i_bitrate = p_sys->i_bitrate_override;
495 break;
496 }
497 case VIDEO_ES:
498 {
499 const es_format_t *p_fmt = p_input->p_fmt;
500 uint8_t *p_codec_extra = NULL;
501 int i_codec_extra = 0;
502
503 if( p_input->p_fmt->i_codec == VLC_CODEC_MP4V )
504 {
505 tk->psz_name = "MPEG-4 Video";
506 tk->i_fourcc = VLC_FOURCC( 'M', 'P', '4', 'S' );
507 }
508 else if( p_input->p_fmt->i_codec == VLC_CODEC_DIV3 )
509 {
510 tk->psz_name = "MSMPEG-4 V3 Video";
511 tk->i_fourcc = VLC_FOURCC( 'M', 'P', '4', '3' );
512 }
513 else if( p_input->p_fmt->i_codec == VLC_CODEC_DIV2 )
514 {
515 tk->psz_name = "MSMPEG-4 V2 Video";
516 tk->i_fourcc = VLC_FOURCC( 'M', 'P', '4', '2' );
517 }
518 else if( p_input->p_fmt->i_codec == VLC_CODEC_DIV1 )
519 {
520 tk->psz_name = "MSMPEG-4 V1 Video";
521 tk->i_fourcc = VLC_FOURCC( 'M', 'P', 'G', '4' );
522 }
523 else if( p_input->p_fmt->i_codec == VLC_CODEC_WMV1 )
524 {
525 tk->psz_name = "Windows Media Video 7";
526 tk->i_fourcc = VLC_FOURCC( 'W', 'M', 'V', '1' );
527 }
528 else if( p_input->p_fmt->i_codec == VLC_CODEC_WMV2 )
529 {
530 tk->psz_name = "Windows Media Video 8";
531 tk->i_fourcc = VLC_FOURCC( 'W', 'M', 'V', '2' );
532 }
533 else if( p_input->p_fmt->i_codec == VLC_CODEC_WMV3 )
534 {
535 tk->psz_name = "Windows Media Video 9";
536 tk->i_fourcc = VLC_FOURCC( 'W', 'M', 'V', '3' );
537 tk->b_extended = true;
538 }
539 else if( p_input->p_fmt->i_codec == VLC_CODEC_VC1 )
540 {
541 tk->psz_name = "Windows Media Video 9 Advanced Profile";
542 tk->i_fourcc = VLC_FOURCC( 'W', 'V', 'C', '1' );
543 tk->b_extended = true;
544
545 if( p_fmt->i_extra > 0 )
546 {
547 p_codec_extra = malloc( 1 + p_fmt->i_extra );
548 if( p_codec_extra )
549 {
550 i_codec_extra = 1 + p_fmt->i_extra;
551 p_codec_extra[0] = 0x01;
552 memcpy( &p_codec_extra[1], p_fmt->p_extra, p_fmt->i_extra );
553 }
554 }
555 }
556 else if( p_input->p_fmt->i_codec == VLC_CODEC_H264 )
557 {
558 tk->psz_name = "H.264/MPEG-4 AVC";
559 tk->i_fourcc = VLC_FOURCC('h','2','6','4');
560 }
561 else
562 {
563 tk->psz_name = _("Unknown Video");
564 tk->i_fourcc = p_input->p_fmt->i_original_fourcc ?: p_input->p_fmt->i_codec;
565 }
566 if( !i_codec_extra && p_fmt->i_extra > 0 )
567 {
568 p_codec_extra = malloc( p_fmt->i_extra );
569 if( p_codec_extra )
570 {
571 i_codec_extra = p_fmt->i_extra;
572 memcpy( p_codec_extra, p_fmt->p_extra, p_fmt->i_extra );
573 }
574 }
575
576 tk->i_extra = 11 + sizeof( VLC_BITMAPINFOHEADER ) + i_codec_extra;
577 tk->p_extra = malloc( tk->i_extra );
578 if( !tk->p_extra )
579 {
580 free( p_codec_extra );
581 return VLC_ENOMEM;
582 }
583 bo_init( &bo, tk->p_extra, tk->i_extra );
584 bo_addle_u32( &bo, p_input->p_fmt->video.i_width );
585 bo_addle_u32( &bo, p_input->p_fmt->video.i_height );
586 bo_add_u8 ( &bo, 0x02 ); /* flags */
587 bo_addle_u16( &bo, sizeof( VLC_BITMAPINFOHEADER ) + i_codec_extra );
588 bo_addle_u32( &bo, sizeof( VLC_BITMAPINFOHEADER ) + i_codec_extra );
589 bo_addle_u32( &bo, p_input->p_fmt->video.i_width );
590 bo_addle_u32( &bo, p_input->p_fmt->video.i_height );
591 bo_addle_u16( &bo, 1 );
592 bo_addle_u16( &bo, 24 );
593 bo_add_mem( &bo, (uint8_t*)&tk->i_fourcc, 4 );
594 bo_addle_u32( &bo, 0 );
595 bo_addle_u32( &bo, 0 );
596 bo_addle_u32( &bo, 0 );
597 bo_addle_u32( &bo, 0 );
598 bo_addle_u32( &bo, 0 );
599 if( i_codec_extra > 0 )
600 {
601 bo_add_mem( &bo, p_codec_extra, i_codec_extra );
602 free( p_codec_extra );
603 }
604
605 if( p_input->p_fmt->i_bitrate > 50000 )
606 {
607 p_sys->i_bitrate += p_input->p_fmt->i_bitrate;
608 }
609 else
610 {
611 p_sys->i_bitrate += 512000;
612 }
613 if( p_sys->i_bitrate_override )
614 p_sys->i_bitrate = p_sys->i_bitrate_override;
615 break;
616 }
617 default:
618 msg_Err(p_mux, "unhandled track type" );
619 return VLC_EGENERIC;
620 }
621
622 es_format_Copy( &tk->fmt, p_input->p_fmt );
623
624 vlc_array_append( p_sys->p_tracks, (void *)tk);
625 tk->i_id = vlc_array_index_of_item( p_sys->p_tracks, tk ) + 1;
626
627
628 if( p_sys->b_asf_http )
629 p_sys->b_write_header = true;
630
631 return VLC_SUCCESS;
632}

634/*****************************************************************************
635 * DelStream:
636 *****************************************************************************/
637static int DelStream( sout_mux_t *p_mux, sout_input_t *p_input )
638{
639 /* if bitrate ain't defined in commandline, reduce it when tracks are deleted
640 */
641 sout_mux_sys_t *p_sys = p_mux->p_sys;
642 asf_track_t *tk = p_input->p_sys;
643 msg_Dbg( p_mux, "removing input" );
644 if(!p_sys->i_bitrate_override)
645 {
646 if( tk->i_cat == AUDIO_ES )
647 {
648 if( p_input->p_fmt->i_bitrate > 24000 )
649 p_sys->i_bitrate -= p_input->p_fmt->i_bitrate;
650 else
651 p_sys->i_bitrate -= 128000;
652 }
653 else if(tk->i_cat == VIDEO_ES )
654 {
655 if( p_input->p_fmt->i_bitrate > 50000 )
656 p_sys->i_bitrate -= p_input->p_fmt->i_bitrate;
657 else
658 p_sys->i_bitrate -= 512000;
659 }
660 }
661
662 if( p_sys->b_asf_http )
663 {
664 vlc_array_remove( p_sys->p_tracks, vlc_array_index_of_item( p_sys->p_tracks, (void *)tk ) );
665 p_sys->b_write_header = true;
666 }
667 return VLC_SUCCESS;
668}

670/*****************************************************************************
671 * Mux:
672 *****************************************************************************/
673static int Mux( sout_mux_t *p_mux )
674{
675 sout_mux_sys_t *p_sys = p_mux->p_sys;
676
677 if( p_sys->b_write_header )
678 {
679 block_t *out = asf_header_create( p_mux, true );
680
681 out->i_flags |= BLOCK_FLAG_HEADER;
682 sout_AccessOutWrite( p_mux->p_access, out );
683
684 p_sys->b_write_header = false;
685 }
686
687 for( ;; )
688 {
689 sout_input_t *p_input;
690 asf_track_t *tk;
691 mtime_t i_dts;
692 block_t *data;
693 block_t *pk;
694
695 int i_stream = sout_MuxGetStream( p_mux, 1, &i_dts );
696 if( i_stream < 0 )
697 {
698 /* not enough data */
699 return VLC_SUCCESS;
700 }
701
702 if( p_sys->i_dts_first <= VLC_TS_INVALID )
703 {
704 p_sys->i_dts_first = i_dts;
705 }
706 if( p_sys->i_dts_last < i_dts )
707 {
708 p_sys->i_dts_last = i_dts;
709 }
710
711 p_input = p_mux->pp_inputs[i_stream];
712 tk = (asf_track_t*)p_input->p_sys;
713
714 data = block_FifoGet( p_input->p_fifo );
715
716 /* Convert VC1 to ASF special format */
717 if( tk->i_fourcc == VLC_FOURCC( 'W', 'V', 'C', '1' ) )
718 {
719 while( data->i_buffer >= 4 &&
720 ( data->p_buffer[0] != 0x00 || data->p_buffer[1] != 0x00 ||
721 data->p_buffer[2] != 0x01 ||
722 ( data->p_buffer[3] != 0x0D && data->p_buffer[3] != 0x0C ) ) )
723 {
724 data->i_buffer--;
725 data->p_buffer++;
726 }
727 if( data->i_buffer >= 4 )
728 {
729 data->i_buffer -= 4;
730 data->p_buffer += 4;
731 }
732 }
733
734 if( ( pk = asf_packet_create( p_mux, tk, data ) ) )
735 {
736 sout_AccessOutWrite( p_mux->p_access, pk );
737 }
738 }
739
740 return VLC_SUCCESS;
741}

839static block_t *asf_header_create( sout_mux_t *p_mux, bool b_broadcast )
840{
841 sout_mux_sys_t *p_sys = p_mux->p_sys;
842 asf_track_t *tk;
843 mtime_t i_duration = 0;
844 int i_size, i_header_ext_size, i;
845 int i_ci_size, i_cm_size = 0, i_cd_size = 0;
846 block_t *out;
847 bo_t bo;
848 tk=NULL;
849
850 msg_Dbg( p_mux, "Asf muxer creating header" );
851
852 if( p_sys->i_dts_first > VLC_TS_INVALID )
853 {
854 i_duration = p_sys->i_dts_last - p_sys->i_dts_first;
855 if( i_duration < 0 ) i_duration = 0;
856 }
857
858 /* calculate header size */
859 i_size = 30 + 104;
860 i_ci_size = 44;
861 for( i = 0; i < vlc_array_count( p_sys->p_tracks ); i++ )
862 {
863 tk = vlc_array_item_at_index( p_sys->p_tracks, i );
864 /* update also track-id */
865 tk->i_id = i + 1;
866
867 i_size += 78 + tk->i_extra;
868 i_ci_size += 8 + 2 * strlen( tk->psz_name );
869 if( tk->i_cat == AUDIO_ES ) i_ci_size += 4;
870 else if( tk->i_cat == VIDEO_ES ) i_ci_size += 6;
871
872 /* Error correction data field */
873 if( tk->b_audio_correction ) i_size += 8;
874 }
875
876 /* size of the content description object */
877 if( *p_sys->psz_title || *p_sys->psz_author || *p_sys->psz_copyright ||
878 *p_sys->psz_comment || *p_sys->psz_rating )
879 {
880 i_cd_size = 34 + 2 * ( strlen( p_sys->psz_title ) + 1 +
881 strlen( p_sys->psz_author ) + 1 +
882 strlen( p_sys->psz_copyright ) + 1 +
883 strlen( p_sys->psz_comment ) + 1 +
884 strlen( p_sys->psz_rating ) + 1 );
885 }
886
887 i_header_ext_size = 46;
888
889 /* size of the metadata object */
890 for( i = 0; i < vlc_array_count( p_sys->p_tracks ); i++ )
891 {
892 const asf_track_t *p_track = vlc_array_item_at_index( p_sys->p_tracks, i );
893 if( p_track->i_cat == VIDEO_ES &&
894 p_track->fmt.video.i_sar_num != 0 &&
895 p_track->fmt.video.i_sar_den != 0 )
896 {
897 i_cm_size = 26 + 2 * (16 + 2 * sizeof("AspectRatio?"));
898 }
899 if( p_track->b_extended )
900 i_header_ext_size += 88;
901
902 }
903
904 i_header_ext_size += i_cm_size;
905
906 i_size += i_ci_size + i_cd_size + i_header_ext_size ;
907
908 if( p_sys->b_asf_http )
909 {
910 out = block_Alloc( i_size + 50 + 12 );
911 bo_init( &bo, out->p_buffer, i_size + 50 + 12 );
912 asf_chunk_add( &bo, 0x4824, i_size + 50, 0xc00, p_sys->i_seq++ );
913 }
914 else
915 {
916 out = block_Alloc( i_size + 50 );
917 bo_init( &bo, out->p_buffer, i_size + 50 );
918 }
919
920 /* header object */
921 bo_add_guid ( &bo, &asf_object_header_guid );
922 bo_addle_u64( &bo, i_size );
923 bo_addle_u32( &bo, 2 + vlc_array_count( p_sys->p_tracks ) + 1 +
924 (i_cd_size ? 1 : 0) + (i_cm_size ? 1 : 0) );
925 bo_add_u8 ( &bo, 1 );
926 bo_add_u8 ( &bo, 2 );
927
928 /* sub object */
929
930 /* file properties */
931 bo_add_guid ( &bo, &asf_object_file_properties_guid );
932 bo_addle_u64( &bo, 104 );
933 bo_add_guid ( &bo, &p_sys->fid );
934 bo_addle_u64( &bo, i_size + 50 + p_sys->i_packet_count *
935 p_sys->i_packet_size ); /* file size */
936 bo_addle_u64( &bo, 0 ); /* creation date */
937 bo_addle_u64( &bo, b_broadcast ? 0xffffffffLL : p_sys->i_packet_count );
938 bo_addle_u64( &bo, i_duration * 10 ); /* play duration (100ns) */
939 bo_addle_u64( &bo, i_duration * 10 ); /* send duration (100ns) */
940 bo_addle_u64( &bo, p_sys->i_preroll_time ); /* preroll duration (ms) */
941 bo_addle_u32( &bo, b_broadcast ? 0x01 : 0x02 /* seekable */ ); /* flags */
942 bo_addle_u32( &bo, p_sys->i_packet_size ); /* packet size min */
943 bo_addle_u32( &bo, p_sys->i_packet_size ); /* packet size max */
944 /* NOTE: According to p6-9 of the ASF specification the bitrate cannot be 0,
945 * therefor apply this workaround to make sure it is not 0. If the bitrate is
946 * 0 the file will play in WMP11, but not in Sliverlight and WMP12 */
947 bo_addle_u32( &bo, p_sys->i_bitrate > 0 ? p_sys->i_bitrate : 1 ); /* maxbitrate */
948
949 /* header extension */
950 bo_add_guid ( &bo, &asf_object_header_extension_guid );
951 bo_addle_u64( &bo, i_header_ext_size );
952 bo_add_guid ( &bo, &asf_guid_reserved_1 );
953 bo_addle_u16( &bo, 6 );
954 bo_addle_u32( &bo, i_header_ext_size - 46 );
955
956 /* extended stream properties */
957 for( i = 0; i < vlc_array_count( p_sys->p_tracks ); i++ )
958 {
959 const asf_track_t *p_track = vlc_array_item_at_index( p_sys->p_tracks, i );
960 const es_format_t *p_fmt = &p_track->fmt;
961
962 if( !p_track->b_extended )
963 continue;
964
965 uint64_t i_avg_duration = 0;
966 if( p_fmt->i_cat == VIDEO_ES &&
967 p_fmt->video.i_frame_rate > 0 && p_fmt->video.i_frame_rate_base > 0 )
968 i_avg_duration = ( INT64_C(10000000) * p_fmt->video.i_frame_rate_base +
969 p_fmt->video.i_frame_rate/2 ) / p_fmt->video.i_frame_rate;
970
971 bo_add_guid ( &bo, &asf_object_extended_stream_properties_guid );
972 bo_addle_u64( &bo, 88 );
973 bo_addle_u64( &bo, 0 );
974 bo_addle_u64( &bo, 0 );
975 bo_addle_u32( &bo, p_fmt->i_bitrate ); /* Bitrate */
976 bo_addle_u32( &bo, p_sys->i_preroll_time ); /* Buffer size */
977 bo_addle_u32( &bo, 0 ); /* Initial buffer fullness */
978 bo_addle_u32( &bo, p_fmt->i_bitrate ); /* Alternate Bitrate */
979 bo_addle_u32( &bo, 0 ); /* Alternate Buffer size */
980 bo_addle_u32( &bo, 0 ); /* Alternate Initial buffer fullness */
981 bo_addle_u32( &bo, 0 ); /* Maximum object size (0 = unknown) */
982 bo_addle_u32( &bo, 0x02 ); /* Flags (seekable) */
983 bo_addle_u16( &bo, p_track->i_id ); /* Stream number */
984 bo_addle_u16( &bo, 0 ); /* Stream language index */
985 bo_addle_u64( &bo, i_avg_duration ); /* Average time per frame */
986 bo_addle_u16( &bo, 0 ); /* Stream name count */
987 bo_addle_u16( &bo, 0 ); /* Payload extension system count */
988 }
989
990 /* metadata object (part of header extension) */
991 if( i_cm_size )
992 {
993 unsigned int i_dst_num, i_dst_den;
994
995 asf_track_t *tk = NULL;
996 for( i = 0; i < vlc_array_count( p_sys->p_tracks ); i++ )
997 {
998 tk = vlc_array_item_at_index( p_sys->p_tracks, i );
999 if( tk->i_cat == VIDEO_ES &&
1000 tk->fmt.video.i_sar_num != 0 &&
1001 tk->fmt.video.i_sar_den != 0 )
1002 break;
1003 }
1004 assert( tk != NULL );
1005
1006 vlc_ureduce( &i_dst_num, &i_dst_den,
1007 tk->fmt.video.i_sar_num,
1008 tk->fmt.video.i_sar_den, 0 );
1009
1010 msg_Dbg( p_mux, "pixel aspect-ratio: %i/%i", i_dst_num, i_dst_den );
1011
1012 bo_add_guid ( &bo, &asf_object_metadata_guid );
1013 bo_addle_u64( &bo, i_cm_size );
1014 bo_addle_u16( &bo, 2 ); /* description records count */
1015 /* 1st description record */
1016 bo_addle_u16( &bo, 0 ); /* reserved */
1017 bo_addle_u16( &bo, tk->i_id ); /* stream number (0 for the whole file) */
1018 bo_addle_u16( &bo, 2 * sizeof("AspectRatioX") ); /* name length */
1019 bo_addle_u16( &bo, 0x3 /* DWORD */ ); /* data type */
1020 bo_addle_u32( &bo, 4 ); /* data length */
1021 bo_addle_str16_nosize( &bo, "AspectRatioX" );
1022 bo_addle_u32( &bo, i_dst_num ); /* data */
1023 /* 2nd description record */
1024 bo_addle_u16( &bo, 0 ); /* reserved */
1025 bo_addle_u16( &bo, tk->i_id ); /* stream number (0 for the whole file) */
1026 bo_addle_u16( &bo, 2 * sizeof("AspectRatioY") ); /* name length */
1027 bo_addle_u16( &bo, 0x3 /* DWORD */ ); /* data type */
1028 bo_addle_u32( &bo, 4 ); /* data length */
1029 bo_addle_str16_nosize( &bo, "AspectRatioY" );
1030 bo_addle_u32( &bo, i_dst_den ); /* data */
1031 }
1032
1033 /* content description header */
1034 if( i_cd_size > 0 )
1035 {
1036 bo_add_guid ( &bo, &asf_object_content_description_guid );
1037 bo_addle_u64( &bo, i_cd_size );
1038 bo_addle_u16( &bo, 2 * strlen( p_sys->psz_title ) + 2 );
1039 bo_addle_u16( &bo, 2 * strlen( p_sys->psz_author ) + 2 );
1040 bo_addle_u16( &bo, 2 * strlen( p_sys->psz_copyright ) + 2 );
1041 bo_addle_u16( &bo, 2 * strlen( p_sys->psz_comment ) + 2 );
1042 bo_addle_u16( &bo, 2 * strlen( p_sys->psz_rating ) + 2 );
1043
1044 bo_addle_str16_nosize( &bo, p_sys->psz_title );
1045 bo_addle_str16_nosize( &bo, p_sys->psz_author );
1046 bo_addle_str16_nosize( &bo, p_sys->psz_copyright );
1047 bo_addle_str16_nosize( &bo, p_sys->psz_comment );
1048 bo_addle_str16_nosize( &bo, p_sys->psz_rating );
1049 }
1050
1051 /* stream properties */
1052 for( i = 0; i < vlc_array_count( p_sys->p_tracks ); i++ )
1053 {
1054 tk = vlc_array_item_at_index( p_sys->p_tracks, i);
1055
1056 bo_add_guid ( &bo, &asf_object_stream_properties_guid );
1057 bo_addle_u64( &bo, 78 + tk->i_extra + (tk->b_audio_correction ? 8:0) );
1058
1059 if( tk->i_cat == AUDIO_ES )
1060 {
1061 bo_add_guid( &bo, &asf_object_stream_type_audio );
1062 if( tk->b_audio_correction )
1063 bo_add_guid( &bo, &asf_guid_audio_conceal_spread );
1064 else
1065 bo_add_guid( &bo, &asf_no_error_correction_guid );
1066 }
1067 else if( tk->i_cat == VIDEO_ES )
1068 {
1069 bo_add_guid( &bo, &asf_object_stream_type_video );
1070 bo_add_guid( &bo, &asf_no_error_correction_guid );
1071 }
1072 bo_addle_u64( &bo, 0 ); /* time offset */
1073 bo_addle_u32( &bo, tk->i_extra );
1074 /* correction data length */
1075 bo_addle_u32( &bo, tk->b_audio_correction ? 8 : 0 );
1076 bo_addle_u16( &bo, tk->i_id ); /* stream number */
1077 bo_addle_u32( &bo, 0 );
1078 bo_add_mem ( &bo, tk->p_extra, tk->i_extra );
1079
1080 /* Error correction data field */
1081 if( tk->b_audio_correction )
1082 {
1083 bo_add_u8( &bo, 0x1 ); /* span */
1084 bo_addle_u16( &bo, tk->i_blockalign ); /* virtual packet length */
1085 bo_addle_u16( &bo, tk->i_blockalign ); /* virtual chunck length */
1086 bo_addle_u16( &bo, 1 ); /* silence length */
1087 bo_add_u8( &bo, 0x0 ); /* data */
1088 }
1089 }
1090
1091 /* Codec Infos */
1092 bo_add_guid ( &bo, &asf_object_codec_list_guid );
1093 bo_addle_u64( &bo, i_ci_size );
1094 bo_add_guid ( &bo, &asf_guid_reserved_2 );
1095 bo_addle_u32( &bo, vlc_array_count( p_sys->p_tracks ) );
1096 for( i = 0; i < vlc_array_count( p_sys->p_tracks ); i++ )
1097 {
1098 tk = vlc_array_item_at_index( p_sys->p_tracks ,i);
1099
1100 if( tk->i_cat == VIDEO_ES ) bo_addle_u16( &bo, 1 /* video */ );
1101 else if( tk->i_cat == AUDIO_ES ) bo_addle_u16( &bo, 2 /* audio */ );
1102 else bo_addle_u16( &bo, 0xFFFF /* unknown */ );
1103
1104 bo_addle_str16( &bo, tk->psz_name );
1105 bo_addle_u16( &bo, 0 );
1106 if( tk->i_cat == AUDIO_ES )
1107 {
1108 bo_addle_u16( &bo, 2 );
1109 bo_addle_u16( &bo, tk->i_tag );
1110 }
1111 else if( tk->i_cat == VIDEO_ES )
1112 {
1113 bo_addle_u16( &bo, 4 );
1114 bo_add_mem ( &bo, (uint8_t*)&tk->i_fourcc, 4 );
1115 }
1116 }
1117
1118 /* data object */
1119 bo_add_guid ( &bo, &asf_object_data_guid );
1120 bo_addle_u64( &bo, 50 + p_sys->i_packet_count * p_sys->i_packet_size );
1121 bo_add_guid ( &bo, &p_sys->fid );
1122 bo_addle_u64( &bo, p_sys->i_packet_count );
1123 bo_addle_u16( &bo, 0x101 );
1124
1125 return out;
1126}

1128/****************************************************************************
1129 *
1130 ****************************************************************************/
1131static block_t *asf_packet_flush( sout_mux_t *p_mux )
1132{
1133 sout_mux_sys_t *p_sys = p_mux->p_sys;
1134 int i_pad, i_preheader = p_sys->b_asf_http ? 12 : 0;
1135 block_t *pk;
1136 bo_t bo;
1137
1138 if( !p_sys->pk ) return 0;
1139
1140 i_pad = p_sys->i_packet_size - p_sys->i_pk_used;
1141 memset( p_sys->pk->p_buffer + p_sys->i_pk_used, 0, i_pad );
1142
1143 bo_init( &bo, p_sys->pk->p_buffer, 14 + i_preheader );
1144
1145 if( p_sys->b_asf_http )
1146 asf_chunk_add( &bo, 0x4424, p_sys->i_packet_size, 0x0, p_sys->i_seq++);
1147
1148 bo_add_u8 ( &bo, 0x82 );
1149 bo_addle_u16( &bo, 0 );
1150 bo_add_u8( &bo, 0x11 );
1151 bo_add_u8( &bo, 0x5d );
1152 bo_addle_u16( &bo, i_pad );
1153 bo_addle_u32( &bo, (p_sys->i_pk_dts - p_sys->i_dts_first) / 1000 +
1154 p_sys->i_preroll_time );
1155 bo_addle_u16( &bo, 0 /* data->i_length */ );
1156 bo_add_u8( &bo, 0x80 | p_sys->i_pk_frame );
1157
1158 pk = p_sys->pk;
1159 p_sys->pk = NULL;
1160
1161 p_sys->i_packet_count++;
1162
1163 return pk;
1164}

1166static block_t *asf_packet_create( sout_mux_t *p_mux,
1167 asf_track_t *tk, block_t *data )
1168{
1169 sout_mux_sys_t *p_sys = p_mux->p_sys;
1170
1171 int i_data = data->i_buffer;
1172 int i_pos = 0;
1173 uint8_t *p_data= data->p_buffer;
1174 block_t *first = NULL, **last = &first;
1175 int i_preheader = p_sys->b_asf_http ? 12 : 0;
1176
1177 while( i_pos < i_data )
1178 {
1179 bo_t bo;
1180 int i_payload;
1181
1182 if( p_sys->pk == NULL )
1183 {
1184 p_sys->pk = block_Alloc( p_sys->i_packet_size + i_preheader );
1185 /* reserve 14 bytes for the packet header */
1186 p_sys->i_pk_used = 14 + i_preheader;
1187 p_sys->i_pk_frame = 0;
1188 p_sys->i_pk_dts = data->i_dts;
1189 }
1190
1191 bo_init( &bo, &p_sys->pk->p_buffer[p_sys->i_pk_used],
1192 p_sys->i_packet_size - p_sys->i_pk_used );
1193
1194 /* add payload (header size = 17) */
1195 i_payload = __MIN( i_data - i_pos,
1196 p_sys->i_packet_size - p_sys->i_pk_used - 17 );
1197
1198 if( tk->b_audio_correction && p_sys->i_pk_frame && i_payload < i_data )
1199 {
1200 /* Don't know why yet but WMP doesn't like splitted WMA packets */
1201 *last = asf_packet_flush( p_mux );
1202 last = &(*last)->p_next;
1203 continue;
1204 }
1205
1206 bo_add_u8 ( &bo, !(data->i_flags & BLOCK_FLAG_TYPE_P ||
1207 data->i_flags & BLOCK_FLAG_TYPE_B) ?
1208 0x80 | tk->i_id : tk->i_id );
1209 bo_add_u8 ( &bo, tk->i_sequence );
1210 bo_addle_u32( &bo, i_pos );
1211 bo_add_u8 ( &bo, 0x08 ); /* flags */
1212 bo_addle_u32( &bo, i_data );
1213 bo_addle_u32( &bo, (data->i_dts - p_sys->i_dts_first) / 1000 +
1214 p_sys->i_preroll_time );
1215 bo_addle_u16( &bo, i_payload );
1216 bo_add_mem ( &bo, &p_data[i_pos], i_payload );
1217 i_pos += i_payload;
1218 p_sys->i_pk_used += 17 + i_payload;
1219
1220 p_sys->i_pk_frame++;
1221
1222 if( p_sys->i_pk_used + 17 >= p_sys->i_packet_size )
1223 {
1224 /* Not enough data for another payload, flush the packet */
1225 *last = asf_packet_flush( p_mux );
1226 last = &(*last)->p_next;
1227 }
1228 }
1229
1230 tk->i_sequence++;
1231 block_Release( data );
1232
1233 return first;
1234}

1236static block_t *asf_stream_end_create( sout_mux_t *p_mux )
1237{
1238 sout_mux_sys_t *p_sys = p_mux->p_sys;
1239
1240 block_t *out = NULL;
1241 bo_t bo;
1242
1243 if( p_sys->b_asf_http )
1244 {
1245 out = block_Alloc( 12 );
1246 bo_init( &bo, out->p_buffer, 12 );
1247 asf_chunk_add( &bo, 0x4524, 0, 0x00, p_sys->i_seq++ );
1248 }
1249 else
1250 {
1251 /* Create index */
1252 out = block_Alloc( 56 );
1253 bo_init( &bo, out->p_buffer, 56 );
1254 bo_add_guid ( &bo, &asf_object_index_guid );
1255 bo_addle_u64( &bo, 56 );
1256 bo_add_guid ( &bo, &p_sys->fid );
1257 bo_addle_u64( &bo, 10000000 );
1258 bo_addle_u32( &bo, 5 );
1259 bo_addle_u32( &bo, 0 );
1260 }
1261
1262 return out;
1263}
