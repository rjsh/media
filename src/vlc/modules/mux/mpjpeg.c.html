63/* This pseudo-random sequence is unlikely to ever happen */
64/* This should be the same as in src/network/httpd.c */
65#define BOUNDARY "7b3cc56e5f51db803f790dad720ed50a"

67/*****************************************************************************
68 * Open:
69 *****************************************************************************/
70static int Open( vlc_object_t *p_this )
71{
72 sout_mux_t *p_mux = (sout_mux_t*)p_this;
73
74 msg_Dbg( p_mux, "Multipart jpeg muxer opened" );
75
76 p_mux->pf_control = Control;
77 p_mux->pf_addstream = AddStream;
78 p_mux->pf_delstream = DelStream;
79 p_mux->pf_mux = Mux;
80 p_mux->p_sys = NULL;
81
82 return VLC_SUCCESS;
83}

96static int Control( sout_mux_t *p_mux, int i_query, va_list args )
97{
98 VLC_UNUSED(p_mux);
99 bool *pb_bool;
100 char **ppsz;
101
102 switch( i_query )
103 {
104 case MUX_CAN_ADD_STREAM_WHILE_MUXING:
105 pb_bool = (bool*)va_arg( args, bool * );
106 *pb_bool = true;
107 return VLC_SUCCESS;
108
109 case MUX_GET_ADD_STREAM_WAIT:
110 pb_bool = (bool*)va_arg( args, bool * );
111 *pb_bool = true;
112 return VLC_SUCCESS;
113
114 case MUX_GET_MIME:
115 ppsz = (char**)va_arg( args, char ** );
116 *ppsz = strdup( "multipart/x-mixed-replace; boundary="BOUNDARY );
117 return VLC_SUCCESS;
118
119 default:
120 return VLC_EGENERIC;
121 }
122}

124static int AddStream( sout_mux_t *p_mux, sout_input_t *p_input )
125{
126 if( p_mux->i_nb_inputs > 1 )
127 {
128 msg_Dbg( p_mux, "only 1 input allowed" );
129 return VLC_EGENERIC;
130 }
131
132 msg_Dbg( p_mux, "adding input" );
133 if( p_input->p_fmt->i_codec != VLC_CODEC_MJPG )
134 return VLC_EGENERIC;
135
136 return VLC_SUCCESS;
137}
138
139static int DelStream( sout_mux_t *p_mux, sout_input_t *p_input )
140{
141 VLC_UNUSED(p_input);
142 msg_Dbg( p_mux, "removing input" );
143 return VLC_SUCCESS;
144}

146static int Mux( sout_mux_t *p_mux )
147{
148 block_fifo_t *p_fifo;
149
150 if( !p_mux->i_nb_inputs ) return VLC_SUCCESS;
151
152 p_fifo = p_mux->pp_inputs[0]->p_fifo;
153
154 while( block_FifoCount( p_fifo ) > 0 )
155 {
156 static const char psz_hfmt[] = "\r\n"
157 "--"BOUNDARY"\r\n"
158 "Content-Type: image/jpeg\r\n"
159 "Content-Length: %zu\r\n"
160 "\r\n";
161 block_t *p_data = block_FifoGet( p_fifo );
162 block_t *p_header = block_Alloc( sizeof( psz_hfmt ) + 20 );
163
164 if( p_header == NULL ) /* uho! */
165 {
166 block_Release( p_data );
167 continue;
168 }
169
170 p_header->i_buffer =
171 snprintf( (char *)p_header->p_buffer, p_header->i_buffer,
172 psz_hfmt, p_data->i_buffer );
173 p_header->i_flags |= BLOCK_FLAG_HEADER;
174 sout_AccessOutWrite( p_mux->p_access, p_header );
175 sout_AccessOutWrite( p_mux->p_access, p_data );
176 }
177
178 return VLC_SUCCESS;
179}
