249static inline void BufferChainInit ( sout_buffer_chain_t *c )
250{
251 c->i_depth = 0;
252 c->p_first = NULL;
253 c->pp_last = &c->p_first;
254}
255
256static inline void BufferChainAppend( sout_buffer_chain_t *c, block_t *b )
257{
258 *c->pp_last = b;
259 c->i_depth++;
260
261 while( b->p_next )
262 {
263 b = b->p_next;
264 c->i_depth++;
265 }
266 c->pp_last = &b->p_next;
267}
268
269static inline block_t *BufferChainGet( sout_buffer_chain_t *c )
270{
271 block_t *b = c->p_first;
272
273 if( b )
274 {
275 c->i_depth--;
276 c->p_first = b->p_next;
277
278 if( c->p_first == NULL )
279 {
280 c->pp_last = &c->p_first;
281 }
282
283 b->p_next = NULL;
284 }
285 return b;
286}
287
288static inline block_t *BufferChainPeek( sout_buffer_chain_t *c )
289{
290 block_t *b = c->p_first;
291
292 return b;
293}
294
295static inline void BufferChainClean( sout_buffer_chain_t *c )
296{
297 block_t *b;
298
299 while( ( b = BufferChainGet( c ) ) )
300 {
301 block_Release( b );
302 }
303 BufferChainInit( c );
304}

394/* Reserve a pid and return it */
395static int AllocatePID( sout_mux_sys_t *p_sys, int i_cat )
396{
397 int i_pid;
398 if ( i_cat == VIDEO_ES && p_sys->i_pid_video )
399 {
400 i_pid = p_sys->i_pid_video;
401 p_sys->i_pid_video = 0;
402 }
403 else if ( i_cat == AUDIO_ES && p_sys->i_pid_audio )
404 {
405 i_pid = p_sys->i_pid_audio;
406 p_sys->i_pid_audio = 0;
407 }
408 else if ( i_cat == SPU_ES && p_sys->i_pid_spu )
409 {
410 i_pid = p_sys->i_pid_spu;
411 p_sys->i_pid_spu = 0;
412 }
413 else
414 {
415 i_pid = ++p_sys->i_pid_free;
416 }
417 return i_pid;
418}

453static csa_t *csaSetup( vlc_object_t *p_this )
454{
455 sout_mux_t *p_mux = (sout_mux_t*)p_this;
456 sout_mux_sys_t *p_sys = p_mux->p_sys;
457 char *csack = var_CreateGetNonEmptyStringCommand( p_mux, SOUT_CFG_PREFIX "csa-ck" );
458 if( !csack )
459 return NULL;
460
461 csa_t *csa = csa_New();
462
463 if( csa_SetCW( p_this, csa, csack, true ) )
464 {
465 free(csack);
466 csa_Delete( csa );
467 return NULL;
468 }
469
470 vlc_mutex_init( &p_sys->csa_lock );
471 p_sys->b_crypt_audio = var_GetBool( p_mux, SOUT_CFG_PREFIX "crypt-audio" );
472 p_sys->b_crypt_video = var_GetBool( p_mux, SOUT_CFG_PREFIX "crypt-video" );
473
474 char *csa2ck = var_CreateGetNonEmptyStringCommand( p_mux, SOUT_CFG_PREFIX "csa2-ck");
475 if (!csa2ck || csa_SetCW( p_this, csa, csa2ck, false ) )
476 csa_SetCW( p_this, csa, csack, false );
477 free(csa2ck);
478
479 var_Create( p_mux, SOUT_CFG_PREFIX "csa-use", VLC_VAR_STRING | VLC_VAR_DOINHERIT | VLC_VAR_ISCOMMAND );
480 var_AddCallback( p_mux, SOUT_CFG_PREFIX "csa-use", ActiveKeyCallback, NULL );
481 var_AddCallback( p_mux, SOUT_CFG_PREFIX "csa-ck", ChangeKeyCallback, (void *)1 );
482 var_AddCallback( p_mux, SOUT_CFG_PREFIX "csa2-ck", ChangeKeyCallback, NULL );
483
484 vlc_value_t use_val;
485 var_Get( p_mux, SOUT_CFG_PREFIX "csa-use", &use_val );
486 if ( var_Set( p_mux, SOUT_CFG_PREFIX "csa-use", use_val ) )
487 var_SetString( p_mux, SOUT_CFG_PREFIX "csa-use", "odd" );
488 free( use_val.psz_string );
489
490 p_sys->i_csa_pkt_size = var_GetInteger( p_mux, SOUT_CFG_PREFIX "csa-pkt" );
491 if( p_sys->i_csa_pkt_size < 12 || p_sys->i_csa_pkt_size > 188 )
492 {
493 msg_Err( p_mux, "wrong packet size %d specified",
494 p_sys->i_csa_pkt_size );
495 p_sys->i_csa_pkt_size = 188;
496 }
497
498 msg_Dbg( p_mux, "encrypting %d bytes of packet", p_sys->i_csa_pkt_size );
499
500 free(csack);
501
502 return csa;
503}

505/*****************************************************************************
506 * Open:
507 *****************************************************************************/
508static int Open( vlc_object_t *p_this )
509{
510 sout_mux_t *p_mux =(sout_mux_t*)p_this;
511 sout_mux_sys_t *p_sys = NULL;
512
513 config_ChainParse( p_mux, SOUT_CFG_PREFIX, ppsz_sout_options, p_mux->p_cfg );
514
515 p_sys = calloc( 1, sizeof( sout_mux_sys_t ) );
516 if( !p_sys )
517 return VLC_ENOMEM;
518 p_sys->i_num_pmt = 1;
519
520
521#if (DVBPSI_VERSION_INT >= DVBPSI_VERSION_WANTED(1,0,0))
522 p_sys->p_dvbpsi = dvbpsi_new( &dvbpsi_messages, DVBPSI_MSG_DEBUG );
523 if( !p_sys->p_dvbpsi )
524 {
525 free( p_sys );
526 return VLC_ENOMEM;
527 }
528 p_sys->p_dvbpsi->p_sys = (void *) p_mux;
529#endif
530
531 p_sys->b_es_id_pid = var_GetBool( p_mux, SOUT_CFG_PREFIX "es-id-pid" );
532
533 /*
534 fetch string of pmts. Here's a sample: --sout-ts-muxpmt="0x451,0x200,0x28a,0x240,,0x450,0x201,0x28b,0x241,,0x452,0x202,0x28c,0x242"
535 This would mean 0x451, 0x200, 0x28a, 0x240 would fall under one pmt (program), 0x450,0x201,0x28b,0x241 would fall under another
536 */
537 char *muxpmt = var_GetNonEmptyString(p_mux, SOUT_CFG_PREFIX "muxpmt");
538 for (char *psz = muxpmt; psz; )
539 {
540 char *psz_next;
541 uint16_t i_pid = strtoul( psz, &psz_next, 0 );
542 psz = *psz_next ? &psz_next[1] : NULL;
543
544 if ( i_pid == 0 )
545 {
546 if ( ++p_sys->i_num_pmt > MAX_PMT )
547 {
548 msg_Err( p_mux, "Number of PMTs > %d)", MAX_PMT );
549 p_sys->i_num_pmt = MAX_PMT;
550 }
551 }
552 else
553 {
554 p_sys->pmtmap[p_sys->i_pmtslots].i_pid = i_pid;
555 p_sys->pmtmap[p_sys->i_pmtslots].i_prog = p_sys->i_num_pmt - 1;
556 if ( ++p_sys->i_pmtslots >= MAX_PMT_PID )
557 {
558 msg_Err( p_mux, "Number of pids in PMT > %d", MAX_PMT_PID );
559 p_sys->i_pmtslots = MAX_PMT_PID - 1;
560 }
561 }
562 }
563 /* Now sort according to pids for fast search later on */
564 qsort( (void *)p_sys->pmtmap, p_sys->i_pmtslots,
565 sizeof(pmt_map_t), pmtcompare );
566 free(muxpmt);
567
568 unsigned short subi[3];
569 vlc_rand_bytes(subi, sizeof(subi));
570 p_sys->i_pat_version_number = nrand48(subi) & 0x1f;
571
572 vlc_value_t val;
573 var_Get( p_mux, SOUT_CFG_PREFIX "tsid", &val );
574 if ( val.i_int )
575 p_sys->i_tsid = val.i_int;
576 else
577 p_sys->i_tsid = nrand48(subi) & 0xffff;
578
579 p_sys->i_netid = nrand48(subi) & 0xffff;
580
581 var_Get( p_mux, SOUT_CFG_PREFIX "netid", &val );
582 if ( val.i_int )
583 p_sys->i_netid = val.i_int;
584
585 p_sys->i_pmt_version_number = nrand48(subi) & 0x1f;
586 p_sys->sdt.i_pid = 0x11;
587
588 char *sdtdesc = var_GetNonEmptyString( p_mux, SOUT_CFG_PREFIX "sdtdesc" );
589
590 /* Syntax is provider_sdt1,service_name_sdt1,provider_sdt2,service_name_sdt2... */
591 if( sdtdesc )
592 {
593 p_sys->b_sdt = true;
594
595 char *psz_sdttoken = sdtdesc;
596
597 for (int i = 0; i < MAX_PMT * 2 && psz_sdttoken; i++)
598 {
599 sdt_desc_t *sdt = &p_sys->sdt_descriptors[i/2];
600 char *psz_end = strchr( psz_sdttoken, ',' );
601 if ( psz_end )
602 *psz_end++ = '\0';
603
604 if (i % 2)
605 sdt->psz_service_name = strdup(psz_sdttoken);
606 else
607 sdt->psz_provider = strdup(psz_sdttoken);
608
609 psz_sdttoken = psz_end;
610 }
611 free(sdtdesc);
612 }
613
614 p_sys->b_data_alignment = var_GetBool( p_mux, SOUT_CFG_PREFIX "alignment" );
615
616 char *pgrpmt = var_GetNonEmptyString(p_mux, SOUT_CFG_PREFIX "program-pmt");
617 if( pgrpmt )
618 {
619 char *psz = pgrpmt;
620 char *psz_next = psz;
621
622 for (int i = 0; psz; )
623 {
624 uint16_t i_pid = strtoul( psz, &psz_next, 0 );
625 if( psz_next[0] != '\0' )
626 psz = &psz_next[1];
627 else
628 psz = NULL;
629
630 if( i_pid == 0 )
631 {
632 if( i >= MAX_PMT )
633 msg_Err( p_mux, "Number of PMTs > maximum (%d)", MAX_PMT );
634 }
635 else
636 {
637 p_sys->i_pmt_program_number[i] = i_pid;
638 i++;
639 }
640 }
641 free(pgrpmt);
642 }
643 else
644 {
645 /* Option not specified, use 1, 2, 3... */
646 for (unsigned i = 0; i < p_sys->i_num_pmt; i++ )
647 p_sys->i_pmt_program_number[i] = i + 1;
648 }
649
650 var_Get( p_mux, SOUT_CFG_PREFIX "pid-pmt", &val );
651 if( !val.i_int ) /* Does this make any sense? */
652 val.i_int = 0x42;
653 for (unsigned i = 0; i < p_sys->i_num_pmt; i++ )
654 p_sys->pmt[i].i_pid = val.i_int + i;
655
656 p_sys->i_pid_free = p_sys->pmt[p_sys->i_num_pmt - 1].i_pid + 1;
657
658 p_sys->i_pid_video = var_GetInteger( p_mux, SOUT_CFG_PREFIX "pid-video" );
659 if ( p_sys->i_pid_video > p_sys->i_pid_free )
660 {
661 p_sys->i_pid_free = p_sys->i_pid_video + 1;
662 }
663
664 p_sys->i_pid_audio = var_GetInteger( p_mux, SOUT_CFG_PREFIX "pid-audio" );
665 if ( p_sys->i_pid_audio > p_sys->i_pid_free )
666 {
667 p_sys->i_pid_free = p_sys->i_pid_audio + 1;
668 }
669
670 p_sys->i_pid_spu = var_GetInteger( p_mux, SOUT_CFG_PREFIX "pid-spu" );
671 if ( p_sys->i_pid_spu > p_sys->i_pid_free )
672 {
673 p_sys->i_pid_free = p_sys->i_pid_spu + 1;
674 }
675
676 p_sys->i_pcr_pid = 0x1fff;
677
678 /* Allow to create constrained stream */
679 p_sys->i_bitrate_min = var_GetInteger( p_mux, SOUT_CFG_PREFIX "bmin" );
680
681 p_sys->i_bitrate_max = var_GetInteger( p_mux, SOUT_CFG_PREFIX "bmax" );
682
683 if( p_sys->i_bitrate_min > 0 && p_sys->i_bitrate_max > 0 &&
684 p_sys->i_bitrate_min > p_sys->i_bitrate_max )
685 {
686 msg_Err( p_mux, "incompatible minimum and maximum bitrate, "
687 "disabling bitrate control" );
688 p_sys->i_bitrate_min = 0;
689 p_sys->i_bitrate_max = 0;
690 }
691 if( p_sys->i_bitrate_min > 0 || p_sys->i_bitrate_max > 0 )
692 {
693 msg_Err( p_mux, "bmin and bmax no more supported "
694 "(if you need them report it)" );
695 }
696
697 var_Get( p_mux, SOUT_CFG_PREFIX "shaping", &val );
698 p_sys->i_shaping_delay = val.i_int * 1000;
699 if( p_sys->i_shaping_delay <= 0 )
700 {
701 msg_Err( p_mux,
702 "invalid shaping (%"PRId64"ms) resetting to 200ms",
703 p_sys->i_shaping_delay / 1000 );
704 p_sys->i_shaping_delay = 200000;
705 }
706
707 var_Get( p_mux, SOUT_CFG_PREFIX "pcr", &val );
708 p_sys->i_pcr_delay = val.i_int * 1000;
709 if( p_sys->i_pcr_delay <= 0 ||
710 p_sys->i_pcr_delay >= p_sys->i_shaping_delay )
711 {
712 msg_Err( p_mux,
713 "invalid pcr delay (%"PRId64"ms) resetting to 70ms",
714 p_sys->i_pcr_delay / 1000 );
715 p_sys->i_pcr_delay = 70000;
716 }
717
718 var_Get( p_mux, SOUT_CFG_PREFIX "dts-delay", &val );
719 p_sys->i_dts_delay = val.i_int * 1000;
720
721 msg_Dbg( p_mux, "shaping=%"PRId64" pcr=%"PRId64" dts_delay=%"PRId64,
722 p_sys->i_shaping_delay, p_sys->i_pcr_delay, p_sys->i_dts_delay );
723
724 p_sys->b_use_key_frames = var_GetBool( p_mux, SOUT_CFG_PREFIX "use-key-frames" );
725
726 p_sys->csa = csaSetup(p_this);
727
728 p_mux->pf_control = Control;
729 p_mux->pf_addstream = AddStream;
730 p_mux->pf_delstream = DelStream;
731 p_mux->pf_mux = Mux;
732 p_mux->p_sys = p_sys;
733
734 return VLC_SUCCESS;
735}

874/*****************************************************************************
875 * AddStream: called for each stream addition
876 *****************************************************************************/
877static int AddStream( sout_mux_t *p_mux, sout_input_t *p_input )
878{
879 sout_mux_sys_t *p_sys = p_mux->p_sys;
880 ts_stream_t *p_stream;
881
882 p_input->p_sys = p_stream = calloc( 1, sizeof( ts_stream_t ) );
883 if( !p_stream )
884 goto oom;
885
886 if ( p_sys->b_es_id_pid )
887 p_stream->i_pid = p_input->p_fmt->i_id & 0x1fff;
888 else
889 p_stream->i_pid = AllocatePID( p_sys, p_input->p_fmt->i_cat );
890
891 p_stream->i_codec = p_input->p_fmt->i_codec;
892
893 p_stream->i_stream_type = -1;
894 switch( p_input->p_fmt->i_codec )
895 {
896 /* VIDEO */
897
898 case VLC_CODEC_MPGV:
899 case VLC_CODEC_MP2V:
900 case VLC_CODEC_MP1V:
901 /* TODO: do we need to check MPEG-I/II ? */
902 p_stream->i_stream_type = 0x02;
903 p_stream->i_stream_id = 0xe0;
904 break;
905 case VLC_CODEC_MP4V:
906 p_stream->i_stream_type = 0x10;
907 p_stream->i_stream_id = 0xe0;
908 p_stream->i_es_id = p_stream->i_pid;
909 break;
910 case VLC_CODEC_HEVC:
911 p_stream->i_stream_type = 0x24;
912 p_stream->i_stream_id = 0xe0;
913 break;
914 case VLC_CODEC_H264:
915 p_stream->i_stream_type = 0x1b;
916 p_stream->i_stream_id = 0xe0;
917 break;
918 /* XXX dirty dirty but somebody want crapy MS-codec XXX */
919 case VLC_CODEC_H263I:
920 case VLC_CODEC_H263:
921 case VLC_CODEC_WMV3:
922 case VLC_CODEC_WMV2:
923 case VLC_CODEC_WMV1:
924 case VLC_CODEC_DIV3:
925 case VLC_CODEC_DIV2:
926 case VLC_CODEC_DIV1:
927 case VLC_CODEC_MJPG:
928 p_stream->i_stream_type = 0xa0; /* private */
929 p_stream->i_stream_id = 0xa0; /* beurk */
930 p_stream->i_bih_codec = p_input->p_fmt->i_codec;
931 p_stream->i_bih_width = p_input->p_fmt->video.i_width;
932 p_stream->i_bih_height = p_input->p_fmt->video.i_height;
933 break;
934 case VLC_CODEC_DIRAC:
935 /* stream_id makes use of stream_id_extension */
936 p_stream->i_stream_id = (PES_EXTENDED_STREAM_ID << 8) | 0x60;
937 p_stream->i_stream_type = 0xd1;
938 break;
939
940 /* AUDIO */
941
942 case VLC_CODEC_MPGA:
943 case VLC_CODEC_MP3:
944 p_stream->i_stream_type =
945 p_input->p_fmt->audio.i_rate >= 32000 ? 0x03 : 0x04;
946 p_stream->i_stream_id = 0xc0;
947 break;
948 case VLC_CODEC_A52:
949 p_stream->i_stream_type = 0x81;
950 p_stream->i_stream_id = 0xbd;
951 break;
952 case VLC_CODEC_EAC3:
953 p_stream->i_stream_type = 0x06;
954 p_stream->i_stream_id = 0xbd;
955 break;
956 case VLC_CODEC_DVD_LPCM:
957 p_stream->i_stream_type = 0x83;
958 p_stream->i_stream_id = 0xbd;
959 break;
960 case VLC_CODEC_DTS:
961 p_stream->i_stream_type = 0x06;
962 p_stream->i_stream_id = 0xbd;
963 break;
964 case VLC_CODEC_MP4A:
965 /* XXX: make that configurable in some way when LOAS
966 * is implemented for AAC in TS */
967 //p_stream->i_stream_type = 0x11; /* LOAS/LATM */
968 p_stream->i_stream_type = 0x0f; /* ADTS */
969 p_stream->i_stream_id = 0xc0;
970 p_stream->i_es_id = p_stream->i_pid;
971 break;
972
973 /* TEXT */
974
975 case VLC_CODEC_SPU:
976 p_stream->i_stream_type = 0x82;
977 p_stream->i_stream_id = 0xbd;
978 break;
979 case VLC_CODEC_SUBT:
980 p_stream->i_stream_type = 0x12;
981 p_stream->i_stream_id = 0xfa;
982 p_sys->i_mpeg4_streams++;
983 p_stream->i_es_id = p_stream->i_pid;
984 break;
985 case VLC_CODEC_DVBS:
986 p_stream->i_stream_type = 0x06;
987 p_stream->i_es_id = p_input->p_fmt->subs.dvb.i_id;
988 p_stream->i_stream_id = 0xbd;
989 break;
990 case VLC_CODEC_TELETEXT:
991 p_stream->i_stream_type = 0x06;
992 p_stream->i_stream_id = 0xbd; /* FIXME */
993 break;
994 }
995
996 if (p_stream->i_stream_type == -1)
997 {
998 free( p_stream );
999 return VLC_EGENERIC;
1000 }
1001
1002 p_stream->i_langs = 1 + p_input->p_fmt->i_extra_languages;
1003 p_stream->lang = calloc(1, p_stream->i_langs * 4);
1004 if( !p_stream->lang )
1005 goto oom;
1006
1007 msg_Dbg( p_mux, "adding input codec=%4.4s pid=%d",
1008 (char*)&p_stream->i_codec, p_stream->i_pid );
1009
1010 for (int i = 0; i < p_stream->i_langs; i++) {
1011 char *lang = (i == 0)
1012 ? p_input->p_fmt->psz_language
1013 : p_input->p_fmt->p_extra_languages[i-1].psz_language;
1014
1015 if (!lang)
1016 continue;
1017
1018 const char *code = GetIso639_2LangCode(lang);
1019 if (*code)
1020 {
1021 memcpy(&p_stream->lang[i*4], code, 3);
1022 p_stream->lang[i*4+3] = 0x00; /* audio type: 0x00 undefined */
1023 msg_Dbg( p_mux, " - lang=%3.3s", &p_stream->lang[i*4] );
1024 }
1025 }
1026
1027 /* Create decoder specific info for subt */
1028 if( p_stream->i_codec == VLC_CODEC_SUBT )
1029 {
1030 p_stream->i_extra = 55;
1031 p_stream->p_extra = malloc( p_stream->i_extra );
1032 if (!p_stream->p_extra)
1033 goto oom;
1034
1035 uint8_t *p = p_stream->p_extra;
1036 p[0] = 0x10; /* textFormat, 0x10 for 3GPP TS 26.245 */
1037 p[1] = 0x00; /* flags: 1b: associated video info flag
1038 3b: reserved
1039 1b: duration flag
1040 3b: reserved */
1041 p[2] = 52; /* remaining size */
1042
1043 p += 3;
1044
1045 p[0] = p[1] = p[2] = p[3] = 0; p+=4; /* display flags */
1046 *p++ = 0; /* horizontal justification (-1: left, 0 center, 1 right) */
1047 *p++ = 1; /* vertical justification (-1: top, 0 center, 1 bottom) */
1048
1049 p[0] = p[1] = p[2] = 0x00; p+=3;/* background rgb */
1050 *p++ = 0xff; /* background a */
1051
1052 p[0] = p[1] = 0; p += 2; /* text box top */
1053 p[0] = p[1] = 0; p += 2; /* text box left */
1054 p[0] = p[1] = 0; p += 2; /* text box bottom */
1055 p[0] = p[1] = 0; p += 2; /* text box right */
1056
1057 p[0] = p[1] = 0; p += 2; /* start char */
1058 p[0] = p[1] = 0; p += 2; /* end char */
1059 p[0] = p[1] = 0; p += 2; /* default font id */
1060
1061 *p++ = 0; /* font style flags */
1062 *p++ = 12; /* font size */
1063
1064 p[0] = p[1] = p[2] = 0x00; p+=3;/* foreground rgb */
1065 *p++ = 0x00; /* foreground a */
1066
1067 p[0] = p[1] = p[2] = 0; p[3] = 22; p += 4;
1068 memcpy( p, "ftab", 4 ); p += 4;
1069 *p++ = 0; *p++ = 1; /* entry count */
1070 p[0] = p[1] = 0; p += 2; /* font id */
1071 *p++ = 9; /* font name length */
1072 memcpy( p, "Helvetica", 9 ); /* font name */
1073 }
1074 else
1075 {
1076 /* Copy extra data (VOL for MPEG-4 and extra BitMapInfoHeader for VFW */
1077 es_format_t *fmt = p_input->p_fmt;
1078 if( fmt->i_extra > 0 )
1079 {
1080 p_stream->i_extra = fmt->i_extra;
1081 p_stream->p_extra = malloc( fmt->i_extra );
1082 if( !p_stream->p_extra )
1083 goto oom;
1084
1085 memcpy( p_stream->p_extra, fmt->p_extra, fmt->i_extra );
1086 }
1087 }
1088
1089 /* Init pes chain */
1090 BufferChainInit( &p_stream->chain_pes );
1091
1092 /* We only change PMT version (PAT isn't changed) */
1093 p_sys->i_pmt_version_number = ( p_sys->i_pmt_version_number + 1 )%32;
1094
1095 /* Update pcr_pid */
1096 if( p_input->p_fmt->i_cat != SPU_ES &&
1097 ( p_sys->i_pcr_pid == 0x1fff || p_input->p_fmt->i_cat == VIDEO_ES ) )
1098 {
1099 if( p_sys->p_pcr_input )
1100 {
1101 /* There was already a PCR stream, so clean context */
1102 /* FIXME */
1103 }
1104 p_sys->i_pcr_pid = p_stream->i_pid;
1105 p_sys->p_pcr_input = p_input;
1106
1107 msg_Dbg( p_mux, "new PCR PID is %d", p_sys->i_pcr_pid );
1108 }
1109
1110 return VLC_SUCCESS;
1111
1112oom:
1113 if(p_stream)
1114 {
1115 free(p_stream->lang);
1116 free(p_stream);
1117 }
1118 return VLC_ENOMEM;
1119}

1180 if ( pid > 0 && pid == p_stream->i_pid )
1181 {
1182 p_sys->i_pid_video = pid;
1183 msg_Dbg( p_mux, "freeing video PID %d", pid);
1184 }
1185 pid = var_GetInteger( p_mux, SOUT_CFG_PREFIX "pid-audio" );
1186 if ( pid > 0 && pid == p_stream->i_pid )
1187 {
1188 p_sys->i_pid_audio = pid;
1189 msg_Dbg( p_mux, "freeing audio PID %d", pid);
1190 }
1191 pid = var_GetInteger( p_mux, SOUT_CFG_PREFIX "pid-spu" );
1192 if ( pid > 0 && pid == p_stream->i_pid )
1193 {
1194 p_sys->i_pid_spu = pid;
1195 msg_Dbg( p_mux, "freeing spu PID %d", pid);
1196 }
1197
1198 free( p_stream );
1199
1200 /* We only change PMT version (PAT isn't changed) */
1201 p_sys->i_pmt_version_number++;
1202 p_sys->i_pmt_version_number %= 32;
1203
1204 return VLC_SUCCESS;

1220static bool MuxStreams(sout_mux_t *p_mux )
1221{
1222 sout_mux_sys_t *p_sys = p_mux->p_sys;
1223 ts_stream_t *p_pcr_stream = (ts_stream_t*)p_sys->p_pcr_input->p_sys;
1224
1225 sout_buffer_chain_t chain_ts;
1226 mtime_t i_shaping_delay = p_pcr_stream->b_key_frame
1227 ? p_pcr_stream->i_pes_length
1228 : p_sys->i_shaping_delay;
1229
1230 bool b_ok = true;
1231
1232 /* Accumulate enough data in the pcr stream (>i_shaping_delay) */
1233 /* Accumulate enough data in all other stream ( >= length of pcr)*/
1234 for (int i = -1; !b_ok || i < p_mux->i_nb_inputs; i++ )
1235 {
1236 if (i == p_mux->i_nb_inputs)
1237 {
1238 /* get enough PES packet for all input */
1239 b_ok = true;
1240 i = -1;
1241 }
1242 sout_input_t *p_input;
1243
1244 if( i == -1 )
1245 p_input = p_sys->p_pcr_input;
1246 else if( p_mux->pp_inputs[i]->p_sys == p_pcr_stream )
1247 continue;
1248 else
1249 p_input = p_mux->pp_inputs[i];
1250 ts_stream_t *p_stream = (ts_stream_t*)p_input->p_sys;
1251
1252 if( ( p_stream != p_pcr_stream ||
1253 p_stream->i_pes_length >= i_shaping_delay ) &&
1254 p_stream->i_pes_dts + p_stream->i_pes_length >=
1255 p_pcr_stream->i_pes_dts + p_pcr_stream->i_pes_length )
1256 continue;
1257
1258 /* Need more data */
1259 if( block_FifoCount( p_input->p_fifo ) <= 1 )
1260 {
1261 if( ( p_input->p_fmt->i_cat == AUDIO_ES ) ||
1262 ( p_input->p_fmt->i_cat == VIDEO_ES ) )
1263 {
1264 /* We need more data */
1265 return true;
1266 }
1267 else if( block_FifoCount( p_input->p_fifo ) <= 0 )
1268 {
1269 /* spu, only one packet is needed */
1270 continue;
1271 }
1272 else if( p_input->p_fmt->i_cat == SPU_ES )
1273 {
1274 /* Don't mux the SPU yet if it is too early */
1275 block_t *p_spu = block_FifoShow( p_input->p_fifo );
1276
1277 int64_t i_spu_delay = p_spu->i_dts - p_sys->first_dts - p_pcr_stream->i_pes_dts;
1278 if( ( i_spu_delay > i_shaping_delay ) &&
1279 ( i_spu_delay < INT64_C(100000000) ) )
1280 continue;
1281
1282 if ( ( i_spu_delay >= INT64_C(100000000) ) ||
1283 ( i_spu_delay < INT64_C(10000) ) )
1284 {
1285 BufferChainClean( &p_stream->chain_pes );
1286 p_stream->i_pes_dts = 0;
1287 p_stream->i_pes_used = 0;
1288 p_stream->i_pes_length = 0;
1289 continue;
1290 }
1291 }
1292 }
1293 b_ok = false;
1294
1295 block_t *p_data;
1296 if( p_stream == p_pcr_stream || p_sys->b_data_alignment
1297 || ((p_input->p_fmt->i_codec != VLC_CODEC_MPGA ) &&
1298 (p_input->p_fmt->i_codec != VLC_CODEC_MP3) ) )
1299 {
1300 p_data = block_FifoGet( p_input->p_fifo );
1301 if (p_data->i_pts <= VLC_TS_INVALID)
1302 p_data->i_pts = p_data->i_dts;
1303
1304 if( p_input->p_fmt->i_codec == VLC_CODEC_MP4A )
1305 p_data = Add_ADTS( p_data, p_input->p_fmt );
1306 }
1307 else
1308 p_data = FixPES( p_mux, p_input->p_fifo );
1309
1310 if( block_FifoCount( p_input->p_fifo ) > 0 &&
1311 p_input->p_fmt->i_cat != SPU_ES )
1312 {
1313 block_t *p_next = block_FifoShow( p_input->p_fifo );
1314 p_data->i_length = p_next->i_dts - p_data->i_dts;
1315 }
1316 else if( p_input->p_fmt->i_codec !=
1317 VLC_CODEC_SUBT )
1318 p_data->i_length = 1000;
1319
1320 if (p_sys->first_dts == 0)
1321 p_sys->first_dts = p_data->i_dts;
1322
1323 p_data->i_dts -= p_sys->first_dts;
1324 p_data->i_pts -= p_sys->first_dts;
1325
1326 if( ( p_pcr_stream->i_pes_dts > 0 &&
1327 p_data->i_dts - 10000000 > p_pcr_stream->i_pes_dts +
1328 p_pcr_stream->i_pes_length ) ||
1329 p_data->i_dts < p_stream->i_pes_dts ||
1330 ( p_stream->i_pes_dts > 0 &&
1331 p_input->p_fmt->i_cat != SPU_ES &&
1332 p_data->i_dts - 10000000 > p_stream->i_pes_dts +
1333 p_stream->i_pes_length ) )
1334 {
1335 msg_Warn( p_mux, "packet with too strange dts "
1336 "(dts=%"PRId64",old=%"PRId64",pcr=%"PRId64")",
1337 p_data->i_dts, p_stream->i_pes_dts,
1338 p_pcr_stream->i_pes_dts );
1339 block_Release( p_data );
1340
1341 BufferChainClean( &p_stream->chain_pes );
1342 p_stream->i_pes_dts = 0;
1343 p_stream->i_pes_used = 0;
1344 p_stream->i_pes_length = 0;
1345
1346 if( p_input->p_fmt->i_cat != SPU_ES )
1347 {
1348 BufferChainClean( &p_pcr_stream->chain_pes );
1349 p_pcr_stream->i_pes_dts = 0;
1350 p_pcr_stream->i_pes_used = 0;
1351 p_pcr_stream->i_pes_length = 0;
1352 }
1353
1354 continue;
1355 }
1356
1357 int i_header_size = 0;
1358 int i_max_pes_size = 0;
1359 int b_data_alignment = 0;
1360 if( p_input->p_fmt->i_cat == SPU_ES ) switch (p_input->p_fmt->i_codec)
1361 {
1362 case VLC_CODEC_SUBT:
1363 /* Prepend header */
1364 p_data = block_Realloc( p_data, 2, p_data->i_buffer );
1365 p_data->p_buffer[0] = ( (p_data->i_buffer - 2) >> 8) & 0xff;
1366 p_data->p_buffer[1] = ( (p_data->i_buffer - 2) ) & 0xff;
1367
1368 /* remove trailling \0 if any */
1369 if( p_data->i_buffer > 2 && !p_data->p_buffer[p_data->i_buffer-1] )
1370 p_data->i_buffer--;
1371
1372 /* Append a empty sub (sub text only) */
1373 if( p_data->i_length > 0 &&
1374 ( p_data->i_buffer != 1 || *p_data->p_buffer != ' ' ) )
1375 {
1376 block_t *p_spu = block_Alloc( 3 );
1377
1378 p_spu->i_dts = p_data->i_dts + p_data->i_length;
1379 p_spu->i_pts = p_spu->i_dts;
1380 p_spu->i_length = 1000;
1381
1382 p_spu->p_buffer[0] = 0;
1383 p_spu->p_buffer[1] = 1;
1384 p_spu->p_buffer[2] = ' ';
1385
1386 EStoPES( &p_spu, p_spu, p_input->p_fmt,
1387 p_stream->i_stream_id, 1, 0, 0, 0 );
1388 p_data->p_next = p_spu;
1389 }
1390 break;
1391
1392 case VLC_CODEC_TELETEXT:
1393 /* EN 300 472 */
1394 i_header_size = 0x24;
1395 b_data_alignment = 1;
1396 break;
1397
1398 case VLC_CODEC_DVBS:
1399 /* EN 300 743 */
1400 b_data_alignment = 1;
1401 break;
1402 }
1403 else if( p_data->i_length < 0 || p_data->i_length > 2000000 )
1404 {
1405 /* FIXME choose a better value, but anyway we
1406 * should never have to do that */
1407 p_data->i_length = 1000;
1408 }
1409
1410 p_stream->i_pes_length += p_data->i_length;
1411 if( p_stream->i_pes_dts == 0 )
1412 {
1413 p_stream->i_pes_dts = p_data->i_dts;
1414 }
1415
1416 /* Convert to pes */
1417 if( p_stream->i_stream_id == 0xa0 && p_data->i_pts <= 0 )
1418 {
1419 /* XXX yes I know, it's awful, but it's needed,
1420 * so don't remove it ... */
1421 p_data->i_pts = p_data->i_dts;
1422 }
1423
1424 if( p_input->p_fmt->i_codec == VLC_CODEC_DIRAC )
1425 {
1426 b_data_alignment = 1;
1427 /* dirac pes packets should be unbounded in
1428 * length, specify a suitibly large max size */
1429 i_max_pes_size = INT_MAX;
1430 }
1431
1432 EStoPES ( &p_data, p_data, p_input->p_fmt, p_stream->i_stream_id,
1433 1, b_data_alignment, i_header_size,
1434 i_max_pes_size );
1435
1436 BufferChainAppend( &p_stream->chain_pes, p_data );
1437
1438 if( p_sys->b_use_key_frames && p_stream == p_pcr_stream
1439 && (p_data->i_flags & BLOCK_FLAG_TYPE_I)
1440 && !(p_data->i_flags & BLOCK_FLAG_NO_KEYFRAME)
1441 && (p_stream->i_pes_length > 400000) )
1442 {
1443 i_shaping_delay = p_stream->i_pes_length;
1444 p_stream->b_key_frame = 1;
1445 }
1446 }
1447
1448 /* save */
1449 const mtime_t i_pcr_length = p_pcr_stream->i_pes_length;
1450 p_pcr_stream->b_key_frame = 0;
1451
1452 /* msg_Dbg( p_mux, "starting muxing %lldms", i_pcr_length / 1000 ); */
1453 /* 2: calculate non accurate total size of muxed ts */
1454 int i_packet_count = 0;
1455 for (int i = 0; i < p_mux->i_nb_inputs; i++ )
1456 {
1457 ts_stream_t *p_stream = (ts_stream_t*)p_mux->pp_inputs[i]->p_sys;
1458
1459 /* False for pcr stream but it will be enough to do PCR algo */
1460 for (block_t *p_pes = p_stream->chain_pes.p_first; p_pes != NULL;
1461 p_pes = p_pes->p_next )
1462 {
1463 int i_size = p_pes->i_buffer;
1464 if( p_pes->i_dts + p_pes->i_length >
1465 p_pcr_stream->i_pes_dts + p_pcr_stream->i_pes_length )
1466 {
1467 mtime_t i_frag = p_pcr_stream->i_pes_dts +
1468 p_pcr_stream->i_pes_length - p_pes->i_dts;
1469 if( i_frag < 0 )
1470 {
1471 /* Next stream */
1472 break;
1473 }
1474 i_size = p_pes->i_buffer * i_frag / p_pes->i_length;
1475 }
1476 i_packet_count += ( i_size + 183 ) / 184;
1477 }
1478 }
1479 /* add overhead for PCR (not really exact) */
1480 i_packet_count += (8 * i_pcr_length / p_sys->i_pcr_delay + 175) / 176;
1481
1482 /* 3: mux PES into TS */
1483 BufferChainInit( &chain_ts );
1484 /* append PAT/PMT -> FIXME with big pcr delay it won't have enough pat/pmt */
1485 bool pat_was_previous = true; //This is to prevent unnecessary double PAT/PMT insertions
1486 GetPAT( p_mux, &chain_ts );
1487 GetPMT( p_mux, &chain_ts );
1488 int i_packet_pos = 0;
1489 i_packet_count += chain_ts.i_depth;
1490 /* msg_Dbg( p_mux, "estimated pck=%d", i_packet_count ); */
1491
1492 const mtime_t i_pcr_dts = p_pcr_stream->i_pes_dts;
1493 for (;;)
1494 {
1495 int i_stream = -1;
1496 mtime_t i_dts = 0;
1497 ts_stream_t *p_stream;
1498
1499 /* Select stream (lowest dts) */
1500 for (int i = 0; i < p_mux->i_nb_inputs; i++ )
1501 {
1502 p_stream = (ts_stream_t*)p_mux->pp_inputs[i]->p_sys;
1503
1504 if( p_stream->i_pes_dts == 0 )
1505 {
1506 continue;
1507 }
1508
1509 if( i_stream == -1 || p_stream->i_pes_dts < i_dts )
1510 {
1511 i_stream = i;
1512 i_dts = p_stream->i_pes_dts;
1513 }
1514 }
1515 if( i_stream == -1 || i_dts > i_pcr_dts + i_pcr_length )
1516 {
1517 break;
1518 }
1519 p_stream = (ts_stream_t*)p_mux->pp_inputs[i_stream]->p_sys;
1520 sout_input_t *p_input = p_mux->pp_inputs[i_stream];
1521
1522 /* do we need to issue pcr */
1523 bool b_pcr = false;
1524 if( p_stream == p_pcr_stream &&
1525 i_pcr_dts + i_packet_pos * i_pcr_length / i_packet_count >=
1526 p_sys->i_pcr + p_sys->i_pcr_delay )
1527 {
1528 b_pcr = true;
1529 p_sys->i_pcr = i_pcr_dts + i_packet_pos *
1530 i_pcr_length / i_packet_count;
1531 }
1532
1533 /* Build the TS packet */
1534 block_t *p_ts = TSNew( p_mux, p_stream, b_pcr );
1535 if( p_sys->csa != NULL &&
1536 (p_input->p_fmt->i_cat != AUDIO_ES || p_sys->b_crypt_audio) &&
1537 (p_input->p_fmt->i_cat != VIDEO_ES || p_sys->b_crypt_video) )
1538 {
1539 p_ts->i_flags |= BLOCK_FLAG_SCRAMBLED;
1540 }
1541 i_packet_pos++;
1542
1543 /* Write PAT/PMT before every keyframe if use-key-frames is enabled,
1544 * this helps to do segmenting with livehttp-output so it can cut segment
1545 * and start new one with pat,pmt,keyframe*/
1546 if( ( p_sys->b_use_key_frames ) && ( p_ts->i_flags & BLOCK_FLAG_TYPE_I ) )
1547 {
1548 if( likely( !pat_was_previous ) )
1549 {
1550 int startcount = chain_ts.i_depth;
1551 GetPAT( p_mux, &chain_ts );
1552 GetPMT( p_mux, &chain_ts );
1553 SetHeader( &chain_ts, startcount );
1554 i_packet_count += (chain_ts.i_depth - startcount );
1555 } else {
1556 SetHeader( &chain_ts, 0); //We just inserted pat/pmt,so just flag it instead of adding new one
1557 }
1558 }
1559 pat_was_previous = false;
1560
1561 /* */
1562 BufferChainAppend( &chain_ts, p_ts );
1563 }
1564
1565 /* 4: date and send */
1566 TSSchedule( p_mux, &chain_ts, i_pcr_length, i_pcr_dts );
1567 return false;
1568}

1594#define STD_PES_PAYLOAD 170
1595static block_t *FixPES( sout_mux_t *p_mux, block_fifo_t *p_fifo )
1596{
1597 VLC_UNUSED(p_mux);
1598 block_t *p_data;
1599 size_t i_size;
1600
1601 p_data = block_FifoShow( p_fifo );
1602 i_size = p_data->i_buffer;
1603
1604 if( i_size == STD_PES_PAYLOAD )
1605 {
1606 return block_FifoGet( p_fifo );
1607 }
1608 else if( i_size > STD_PES_PAYLOAD )
1609 {
1610 block_t *p_new = block_Alloc( STD_PES_PAYLOAD );
1611 memcpy( p_new->p_buffer, p_data->p_buffer, STD_PES_PAYLOAD );
1612 p_new->i_pts = p_data->i_pts;
1613 p_new->i_dts = p_data->i_dts;
1614 p_new->i_length = p_data->i_length * STD_PES_PAYLOAD
1615 / p_data->i_buffer;
1616 p_data->i_buffer -= STD_PES_PAYLOAD;
1617 p_data->p_buffer += STD_PES_PAYLOAD;
1618 p_data->i_pts += p_new->i_length;
1619 p_data->i_dts += p_new->i_length;
1620 p_data->i_length -= p_new->i_length;
1621 p_data->i_flags |= BLOCK_FLAG_NO_KEYFRAME;
1622 return p_new;
1623 }
1624 else
1625 {
1626 block_t *p_next;
1627 int i_copy;
1628
1629 p_data = block_FifoGet( p_fifo );
1630 p_data = block_Realloc( p_data, 0, STD_PES_PAYLOAD );
1631 p_next = block_FifoShow( p_fifo );
1632 if ( p_data->i_flags & BLOCK_FLAG_NO_KEYFRAME )
1633 {
1634 p_data->i_flags &= ~BLOCK_FLAG_NO_KEYFRAME;
1635 p_data->i_pts = p_next->i_pts;
1636 p_data->i_dts = p_next->i_dts;
1637 }
1638 i_copy = __MIN( STD_PES_PAYLOAD - i_size, p_next->i_buffer );
1639
1640 memcpy( &p_data->p_buffer[i_size], p_next->p_buffer, i_copy );
1641 p_next->i_pts += p_next->i_length * i_copy / p_next->i_buffer;
1642 p_next->i_dts += p_next->i_length * i_copy / p_next->i_buffer;
1643 p_next->i_length -= p_next->i_length * i_copy / p_next->i_buffer;
1644 p_next->i_buffer -= i_copy;
1645 p_next->p_buffer += i_copy;
1646 p_next->i_flags |= BLOCK_FLAG_NO_KEYFRAME;
1647
1648 if( !p_next->i_buffer )
1649 {
1650 p_next = block_FifoGet( p_fifo );
1651 block_Release( p_next );
1652 }
1653 return p_data;
1654 }

1657static block_t *Add_ADTS( block_t *p_data, es_format_t *p_fmt )
1658{
1659#define ADTS_HEADER_SIZE 7 /* CRC needs 2 more bytes */
1660
1661 uint8_t *p_extra = p_fmt->p_extra;
1662
1663 if( !p_data || p_fmt->i_extra < 2 || !p_extra )
1664 return p_data; /* no data to construct the headers */
1665
1666 size_t frame_length = p_data->i_buffer + ADTS_HEADER_SIZE;
1667 int i_index = ( (p_extra[0] << 1) | (p_extra[1] >> 7) ) & 0x0f;
1668 int i_profile = (p_extra[0] >> 3) - 1; /* i_profile < 4 */
1669
1670 if( i_index == 0x0f && p_fmt->i_extra < 5 )
1671 return p_data; /* not enough data */
1672
1673 int i_channels = (p_extra[i_index == 0x0f ? 4 : 1] >> 3) & 0x0f;
1674
1675 /* keep a copy in case block_Realloc() fails */
1676 block_t *p_bak_block = block_Duplicate( p_data );
1677 if( !p_bak_block ) /* OOM, block_Realloc() is likely to lose our block */
1678 return p_data; /* the frame isn't correct but that's the best we have */
1679
1680 block_t *p_new_block = block_Realloc( p_data, ADTS_HEADER_SIZE,
1681 p_data->i_buffer );
1682 if( !p_new_block )
1683 return p_bak_block; /* OOM, send the (incorrect) original frame */
1684
1685 block_Release( p_bak_block ); /* we don't need the copy anymore */
1686
1687
1688 uint8_t *p_buffer = p_new_block->p_buffer;
1689
1690 /* fixed header */
1691 p_buffer[0] = 0xff;
1692 p_buffer[1] = 0xf1; /* 0xf0 | 0x00 | 0x00 | 0x01 */
1693 p_buffer[2] = (i_profile << 6) | ((i_index & 0x0f) << 2) | ((i_channels >> 2) & 0x01) ;
1694 p_buffer[3] = (i_channels << 6) | ((frame_length >> 11) & 0x03);
1695
1696 /* variable header (starts at last 2 bits of 4th byte) */
1697
1698 int i_fullness = 0x7ff; /* 0x7ff means VBR */
1699 /* XXX: We should check if it's CBR or VBR, but no known implementation
1700 * do that, and it's a pain to calculate this field */
1701
1702 p_buffer[4] = frame_length >> 3;
1703 p_buffer[5] = ((frame_length & 0x07) << 5) | ((i_fullness >> 6) & 0x1f);
1704 p_buffer[6] = ((i_fullness & 0x3f) << 2) /* | 0xfc */;
1705
1706 return p_new_block;
1707}

1709static void TSSchedule( sout_mux_t *p_mux, sout_buffer_chain_t *p_chain_ts,
1710 mtime_t i_pcr_length, mtime_t i_pcr_dts )
1711{
1712 sout_mux_sys_t *p_sys = p_mux->p_sys;
1713 sout_buffer_chain_t new_chain;
1714 int i_packet_count = p_chain_ts->i_depth;
1715
1716 BufferChainInit( &new_chain );
1717
1718 if ( i_pcr_length <= 0 )
1719 {
1720 i_pcr_length = i_packet_count;
1721 }
1722
1723 for (int i = 0; i < i_packet_count; i++ )
1724 {
1725 block_t *p_ts = BufferChainGet( p_chain_ts );
1726 mtime_t i_new_dts = i_pcr_dts + i_pcr_length * i / i_packet_count;
1727
1728 BufferChainAppend( &new_chain, p_ts );
1729
1730 if (!p_ts->i_dts || p_ts->i_dts + p_sys->i_dts_delay * 2/3 >= i_new_dts)
1731 continue;
1732
1733 mtime_t i_max_diff = i_new_dts - p_ts->i_dts;
1734 mtime_t i_cut_dts = p_ts->i_dts;
1735
1736 p_ts = BufferChainPeek( p_chain_ts );
1737 i++;
1738 i_new_dts = i_pcr_dts + i_pcr_length * i / i_packet_count;
1739 while ( p_ts != NULL && i_new_dts - p_ts->i_dts >= i_max_diff )
1740 {
1741 p_ts = BufferChainGet( p_chain_ts );
1742 i_max_diff = i_new_dts - p_ts->i_dts;
1743 i_cut_dts = p_ts->i_dts;
1744 BufferChainAppend( &new_chain, p_ts );
1745
1746 p_ts = BufferChainPeek( p_chain_ts );
1747 i++;
1748 i_new_dts = i_pcr_dts + i_pcr_length * i / i_packet_count;
1749 }
1750 msg_Dbg( p_mux, "adjusting rate at %"PRId64"/%"PRId64" (%d/%d)",
1751 i_cut_dts - i_pcr_dts, i_pcr_length, new_chain.i_depth,
1752 p_chain_ts->i_depth );
1753 if ( new_chain.i_depth )
1754 TSDate( p_mux, &new_chain, i_cut_dts - i_pcr_dts, i_pcr_dts );
1755 if ( p_chain_ts->i_depth )
1756 TSSchedule( p_mux, p_chain_ts, i_pcr_dts + i_pcr_length - i_cut_dts,
1757 i_cut_dts );
1758 return;
1759 }
1760
1761 if ( new_chain.i_depth )
1762 TSDate( p_mux, &new_chain, i_pcr_length, i_pcr_dts );
1763}

1765static void TSDate( sout_mux_t *p_mux, sout_buffer_chain_t *p_chain_ts,
1766 mtime_t i_pcr_length, mtime_t i_pcr_dts )
1767{
1768 sout_mux_sys_t *p_sys = p_mux->p_sys;
1769 int i_packet_count = p_chain_ts->i_depth;
1770
1771 if ( i_pcr_length / 1000 > 0 )
1772 {
1773 int i_bitrate = ((uint64_t)i_packet_count * 188 * 8000)
1774 / (uint64_t)(i_pcr_length / 1000);
1775 if ( p_sys->i_bitrate_max && p_sys->i_bitrate_max < i_bitrate )
1776 {
1777 msg_Warn( p_mux, "max bitrate exceeded at %"PRId64
1778 " (%d bi/s for %d pkt in %"PRId64" us)",
1779 i_pcr_dts + p_sys->i_shaping_delay * 3 / 2 - mdate(),
1780 i_bitrate, i_packet_count, i_pcr_length);
1781 }
1782 }
1783 else
1784 {
1785 /* This shouldn't happen, but happens in some rare heavy load
1786 * and packet losses conditions. */
1787 i_pcr_length = i_packet_count;
1788 }
1789
1790 /* msg_Dbg( p_mux, "real pck=%d", i_packet_count ); */
1791 for (int i = 0; i < i_packet_count; i++ )
1792 {
1793 block_t *p_ts = BufferChainGet( p_chain_ts );
1794 mtime_t i_new_dts = i_pcr_dts + i_pcr_length * i / i_packet_count;
1795
1796 p_ts->i_dts = i_new_dts;
1797 p_ts->i_length = i_pcr_length / i_packet_count;
1798
1799 if( p_ts->i_flags & BLOCK_FLAG_CLOCK )
1800 {
1801 /* msg_Dbg( p_mux, "pcr=%lld ms", p_ts->i_dts / 1000 ); */
1802 TSSetPCR( p_ts, p_ts->i_dts - p_sys->i_dts_delay );
1803 }
1804 if( p_ts->i_flags & BLOCK_FLAG_SCRAMBLED )
1805 {
1806 vlc_mutex_lock( &p_sys->csa_lock );
1807 csa_Encrypt( p_sys->csa, p_ts->p_buffer, p_sys->i_csa_pkt_size );
1808 vlc_mutex_unlock( &p_sys->csa_lock );
1809 }
1810
1811 /* latency */
1812 p_ts->i_dts += p_sys->i_shaping_delay * 3 / 2;
1813
1814 sout_AccessOutWrite( p_mux->p_access, p_ts );
1815 }

1818static block_t *TSNew( sout_mux_t *p_mux, ts_stream_t *p_stream,
1819 bool b_pcr )
1820{
1821 VLC_UNUSED(p_mux);
1822 block_t *p_pes = p_stream->chain_pes.p_first;
1823
1824 bool b_new_pes = false;
1825 bool b_adaptation_field = false;
1826
1827 int i_payload_max = 184 - ( b_pcr ? 8 : 0 );
1828
1829 if( p_stream->i_pes_used <= 0 )
1830 {
1831 b_new_pes = true;
1832 }
1833 int i_payload = __MIN( (int)p_pes->i_buffer - p_stream->i_pes_used,
1834 i_payload_max );
1835
1836 if( b_pcr || i_payload < i_payload_max )
1837 {
1838 b_adaptation_field = true;
1839 }
1840
1841 block_t *p_ts = block_Alloc( 188 );
1842
1843 if (b_new_pes && !(p_pes->i_flags & BLOCK_FLAG_NO_KEYFRAME) && p_pes->i_flags & BLOCK_FLAG_TYPE_I)
1844 {
1845 p_ts->i_flags |= BLOCK_FLAG_TYPE_I;
1846 }
1847
1848 p_ts->i_dts = p_pes->i_dts;
1849
1850 p_ts->p_buffer[0] = 0x47;
1851 p_ts->p_buffer[1] = ( b_new_pes ? 0x40 : 0x00 ) |
1852 ( ( p_stream->i_pid >> 8 )&0x1f );
1853 p_ts->p_buffer[2] = p_stream->i_pid & 0xff;
1854 p_ts->p_buffer[3] = ( b_adaptation_field ? 0x30 : 0x10 ) |
1855 p_stream->i_continuity_counter;
1856
1857 p_stream->i_continuity_counter = (p_stream->i_continuity_counter+1)%16;
1858 p_stream->b_discontinuity = p_pes->i_flags & BLOCK_FLAG_DISCONTINUITY;
1859
1860 if( b_adaptation_field )
1861 {
1862 int i_stuffing = i_payload_max - i_payload;
1863 if( b_pcr )
1864 {
1865 p_ts->i_flags |= BLOCK_FLAG_CLOCK;
1866
1867 p_ts->p_buffer[4] = 7 + i_stuffing;
1868 p_ts->p_buffer[5] = 0x10; /* flags */
1869 if( p_stream->b_discontinuity )
1870 {
1871 p_ts->p_buffer[5] |= 0x80; /* flag TS dicontinuity */
1872 p_stream->b_discontinuity = false;
1873 }
1874 p_ts->p_buffer[6] = 0 &0xff;
1875 p_ts->p_buffer[7] = 0 &0xff;
1876 p_ts->p_buffer[8] = 0 &0xff;
1877 p_ts->p_buffer[9] = 0 &0xff;
1878 p_ts->p_buffer[10]= ( 0 &0x80 ) | 0x7e;
1879 p_ts->p_buffer[11]= 0;
1880
1881 for (int i = 12; i < 12 + i_stuffing; i++ )
1882 {
1883 p_ts->p_buffer[i] = 0xff;
1884 }
1885 }
1886 else
1887 {
1888 p_ts->p_buffer[4] = i_stuffing - 1;
1889 if( i_stuffing > 1 )
1890 {
1891 p_ts->p_buffer[5] = 0x00;
1892 for (int i = 6; i < 6 + i_stuffing - 2; i++ )
1893 {
1894 p_ts->p_buffer[i] = 0xff;
1895 }
1896 }
1897 }
1898 }
1899
1900 /* copy payload */
1901 memcpy( &p_ts->p_buffer[188 - i_payload],
1902 &p_pes->p_buffer[p_stream->i_pes_used], i_payload );
1903
1904 p_stream->i_pes_used += i_payload;
1905 p_stream->i_pes_dts = p_pes->i_dts + p_pes->i_length *
1906 p_stream->i_pes_used / p_pes->i_buffer;
1907 p_stream->i_pes_length -= p_pes->i_length * i_payload / p_pes->i_buffer;
1908
1909 if( p_stream->i_pes_used >= (int)p_pes->i_buffer )
1910 {
1911 block_Release(BufferChainGet( &p_stream->chain_pes ));
1912
1913 p_pes = p_stream->chain_pes.p_first;
1914 p_stream->i_pes_length = 0;
1915 if( p_pes )
1916 {
1917 p_stream->i_pes_dts = p_pes->i_dts;
1918 while( p_pes )
1919 {
1920 p_stream->i_pes_length += p_pes->i_length;
1921 p_pes = p_pes->p_next;
1922 }
1923 }
1924 else
1925 {
1926 p_stream->i_pes_dts = 0;
1927 }
1928 p_stream->i_pes_used = 0;
1929 }
1930
1931 return p_ts;
1932}

1945static void PEStoTS( sout_buffer_chain_t *c, block_t *p_pes,
1946 ts_stream_t *p_stream )
1947{
1948 /* get PES total size */
1949 uint8_t *p_data = p_pes->p_buffer;
1950 int i_size = p_pes->i_buffer;
1951
1952 bool b_new_pes = true;
1953
1954 for (;;)
1955 {
1956 /* write header
1957 * 8b 0x47 sync byte
1958 * 1b transport_error_indicator
1959 * 1b payload_unit_start
1960 * 1b transport_priority
1961 * 13b pid
1962 * 2b transport_scrambling_control
1963 * 2b if adaptation_field 0x03 else 0x01
1964 * 4b continuity_counter
1965 */
1966
1967 int i_copy = __MIN( i_size, 184 );
1968 bool b_adaptation_field = i_size < 184;
1969 block_t *p_ts = block_Alloc( 188 );
1970
1971 p_ts->p_buffer[0] = 0x47;
1972 p_ts->p_buffer[1] = ( b_new_pes ? 0x40 : 0x00 )|
1973 ( ( p_stream->i_pid >> 8 )&0x1f );
1974 p_ts->p_buffer[2] = p_stream->i_pid & 0xff;
1975 p_ts->p_buffer[3] = ( b_adaptation_field ? 0x30 : 0x10 )|
1976 p_stream->i_continuity_counter;
1977
1978 b_new_pes = false;
1979 p_stream->i_continuity_counter = (p_stream->i_continuity_counter+1)%16;
1980
1981 if( b_adaptation_field )
1982 {
1983 int i_stuffing = 184 - i_copy;
1984
1985 p_ts->p_buffer[4] = i_stuffing - 1;
1986 if( i_stuffing > 1 )
1987 {
1988 p_ts->p_buffer[5] = 0x00;
1989 if( p_stream->b_discontinuity )
1990 {
1991 p_ts->p_buffer[5] |= 0x80;
1992 p_stream->b_discontinuity = false;
1993 }
1994 for (int i = 6; i < 6 + i_stuffing - 2; i++ )
1995 {
1996 p_ts->p_buffer[i] = 0xff;
1997 }
1998 }
1999 }
2000 /* copy payload */
2001 memcpy( &p_ts->p_buffer[188 - i_copy], p_data, i_copy );
2002 p_data += i_copy;
2003 i_size -= i_copy;
2004
2005 BufferChainAppend( c, p_ts );
2006
2007 if( i_size <= 0 )
2008 {
2009 block_t *p_next = p_pes->p_next;
2010
2011 p_pes->p_next = NULL;
2012 block_Release( p_pes );
2013 if( p_next == NULL )
2014 return;
2015
2016 b_new_pes = true;
2017 p_pes = p_next;
2018 i_size = p_pes->i_buffer;
2019 p_data = p_pes->p_buffer;
2020 }
2021 }
