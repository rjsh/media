172/*****************************************************************************
173 * Open: probe the packetizer and return score
174 * When opening after demux, the packetizer is only loaded AFTER the decoder
175 * That means that what you set in fmt_out is ignored by the decoder in this special case
176 *****************************************************************************/
177static int Open( vlc_object_t *p_this )
178{
179 decoder_t *p_dec = (decoder_t*)p_this;
180 decoder_sys_t *p_sys;
181 int i;
182
183 if( p_dec->fmt_in.i_codec != VLC_CODEC_H264 )
184 return VLC_EGENERIC;
185 if( p_dec->fmt_in.i_original_fourcc == VLC_FOURCC( 'a', 'v', 'c', '1') &&
186 p_dec->fmt_in.i_extra < 7 )
187 return VLC_EGENERIC;
188
189 /* Allocate the memory needed to store the decoder's structure */
190 if( ( p_dec->p_sys = p_sys = malloc( sizeof(decoder_sys_t) ) ) == NULL )
191 {
192 return VLC_ENOMEM;
193 }
194
195 packetizer_Init( &p_sys->packetizer,
196 p_h264_startcode, sizeof(p_h264_startcode),
197 p_h264_startcode, 1, 5,
198 PacketizeReset, PacketizeParse, PacketizeValidate, p_dec );
199
200 p_sys->b_slice = false;
201 p_sys->p_frame = NULL;
202 p_sys->b_frame_sps = false;
203 p_sys->b_frame_pps = false;
204
205 p_sys->b_header= false;
206 p_sys->b_sps = false;
207 p_sys->b_pps = false;
208 for( i = 0; i < SPS_MAX; i++ )
209 p_sys->pp_sps[i] = NULL;
210 for( i = 0; i < PPS_MAX; i++ )
211 p_sys->pp_pps[i] = NULL;
212 p_sys->i_recovery_frames = -1;
213
214 p_sys->slice.i_nal_type = -1;
215 p_sys->slice.i_nal_ref_idc = -1;
216 p_sys->slice.i_idr_pic_id = -1;
217 p_sys->slice.i_frame_num = -1;
218 p_sys->slice.i_frame_type = 0;
219 p_sys->slice.i_pic_parameter_set_id = -1;
220 p_sys->slice.i_field_pic_flag = 0;
221 p_sys->slice.i_bottom_field_flag = -1;
222 p_sys->slice.i_pic_order_cnt_lsb = -1;
223 p_sys->slice.i_delta_pic_order_cnt_bottom = -1;
224
225 p_sys->i_frame_dts = VLC_TS_INVALID;
226 p_sys->i_frame_pts = VLC_TS_INVALID;
227
228 /* Setup properties */
229 es_format_Copy( &p_dec->fmt_out, &p_dec->fmt_in );
230 p_dec->fmt_out.i_codec = VLC_CODEC_H264;
231
232 if( p_dec->fmt_in.i_original_fourcc == VLC_FOURCC( 'a', 'v', 'c', '1' ) )
233 {
234 /* This type of stream is produced by mp4 and matroska
235 * when we want to store it in another streamformat, you need to convert
236 * The fmt_in.p_extra should ALWAYS contain the avcC
237 * The fmt_out.p_extra should contain all the SPS and PPS with 4 byte startcodes */
238 uint8_t *p = &((uint8_t*)p_dec->fmt_in.p_extra)[4];
239 int i_sps, i_pps;
240 bool b_dummy;
241 int i;
242
243 /* Parse avcC */
244 p_sys->i_avcC_length_size = 1 + ((*p++)&0x03);
245
246 /* Read SPS */
247 i_sps = (*p++)&0x1f;
248 for( i = 0; i < i_sps; i++ )
249 {
250 uint16_t i_length = GetWBE( p ); p += 2;
251 if( i_length >
252 (uint8_t*)p_dec->fmt_in.p_extra + p_dec->fmt_in.i_extra - p )
253 {
254 return VLC_EGENERIC;
255 }
256 block_t *p_sps = CreateAnnexbNAL( p_dec, p, i_length );
257 if( !p_sps )
258 return VLC_EGENERIC;
259 ParseNALBlock( p_dec, &b_dummy, p_sps );
260 p += i_length;
261 }
262 /* Read PPS */
263 i_pps = *p++;
264 for( i = 0; i < i_pps; i++ )
265 {
266 uint16_t i_length = GetWBE( p ); p += 2;
267 if( i_length >
268 (uint8_t*)p_dec->fmt_in.p_extra + p_dec->fmt_in.i_extra - p )
269 {
270 return VLC_EGENERIC;
271 }
272 block_t *p_pps = CreateAnnexbNAL( p_dec, p, i_length );
273 if( !p_pps )
274 return VLC_EGENERIC;
275 ParseNALBlock( p_dec, &b_dummy, p_pps );
276 p += i_length;
277 }
278 msg_Dbg( p_dec, "avcC length size=%d, sps=%d, pps=%d",
279 p_sys->i_avcC_length_size, i_sps, i_pps );
280
281 if( !p_sys->b_sps || !p_sys->b_pps )
282 return VLC_EGENERIC;
283
284 /* FIXME: FFMPEG isn't happy at all if you leave this */
285 if( p_dec->fmt_out.i_extra > 0 )
286 free( p_dec->fmt_out.p_extra );
287 p_dec->fmt_out.i_extra = 0;
288 p_dec->fmt_out.p_extra = NULL;
289
290 /* Set the new extradata */
291 for( i = 0; i < SPS_MAX; i++ )
292 {
293 if( p_sys->pp_sps[i] )
294 p_dec->fmt_out.i_extra += p_sys->pp_sps[i]->i_buffer;
295 }
296 for( i = 0; i < PPS_MAX; i++ )
297 {
298 if( p_sys->pp_pps[i] )
299 p_dec->fmt_out.i_extra += p_sys->pp_pps[i]->i_buffer;
300 }
301 p_dec->fmt_out.p_extra = malloc( p_dec->fmt_out.i_extra );
302 if( p_dec->fmt_out.p_extra )
303 {
304 uint8_t *p_dst = p_dec->fmt_out.p_extra;
305
306 for( i = 0; i < SPS_MAX; i++ )
307 {
308 if( p_sys->pp_sps[i] )
309 {
310 memcpy( p_dst, p_sys->pp_sps[i]->p_buffer, p_sys->pp_sps[i]->i_buffer );
311 p_dst += p_sys->pp_sps[i]->i_buffer;
312 }
313 }
314 for( i = 0; i < PPS_MAX; i++ )
315 {
316 if( p_sys->pp_pps[i] )
317 {
318 memcpy( p_dst, p_sys->pp_pps[i]->p_buffer, p_sys->pp_pps[i]->i_buffer );
319 p_dst += p_sys->pp_pps[i]->i_buffer;
320 }
321 }
322 p_sys->b_header = true;
323 }
324 else
325 {
326 p_dec->fmt_out.i_extra = 0;
327 }
328
329 /* Set callback */
330 p_dec->pf_packetize = PacketizeAVC1;
331 /* TODO CC ? */
332 }
333 else
334 {
335 /* This type of stream contains data with 3 of 4 byte startcodes
336 * The fmt_in.p_extra MAY contain SPS/PPS with 4 byte startcodes
337 * The fmt_out.p_extra should be the same */
338
339 /* Set callback */
340 p_dec->pf_packetize = Packetize;
341 p_dec->pf_get_cc = GetCc;
342
343 /* */
344 p_sys->i_cc_pts = VLC_TS_INVALID;
345 p_sys->i_cc_dts = VLC_TS_INVALID;
346 p_sys->i_cc_flags = 0;
347 cc_Init( &p_sys->cc );
348 cc_Init( &p_sys->cc_next );
349
350 /* */
351 if( p_dec->fmt_in.i_extra > 0 )
352 packetizer_Header( &p_sys->packetizer,
353 p_dec->fmt_in.p_extra, p_dec->fmt_in.i_extra );
354 }
355
356 return VLC_SUCCESS;
357}

359/*****************************************************************************
360 * Close: clean up the packetizer
361 *****************************************************************************/
362static void Close( vlc_object_t *p_this )
363{
364 decoder_t *p_dec = (decoder_t*)p_this;
365 decoder_sys_t *p_sys = p_dec->p_sys;
366 int i;
367
368 if( p_sys->p_frame )
369 block_ChainRelease( p_sys->p_frame );
370 for( i = 0; i < SPS_MAX; i++ )
371 {
372 if( p_sys->pp_sps[i] )
373 block_Release( p_sys->pp_sps[i] );
374 }
375 for( i = 0; i < PPS_MAX; i++ )
376 {
377 if( p_sys->pp_pps[i] )
378 block_Release( p_sys->pp_pps[i] );
379 }
380 packetizer_Clean( &p_sys->packetizer );
381
382 if( p_dec->pf_get_cc )
383 {
384 cc_Exit( &p_sys->cc_next );
385 cc_Exit( &p_sys->cc );
386 }
387
388 free( p_sys );
389}

391/****************************************************************************

396static block_t *Packetize( decoder_t *p_dec, block_t **pp_block )

398 decoder_sys_t *p_sys = p_dec->p_sys;
399
400 return packetizer_Packetize( &p_sys->packetizer, pp_block );
401}

403/****************************************************************************
404 * PacketizeAVC1: Takes VCL blocks of data and creates annexe B type NAL stream
405 * Will always use 4 byte 0 0 0 1 startcodes
406 * Will prepend a SPS and PPS before each keyframe
407 ****************************************************************************/
408static block_t *PacketizeAVC1( decoder_t *p_dec, block_t **pp_block )
409{
410 decoder_sys_t *p_sys = p_dec->p_sys;
411 block_t *p_block;
412 block_t *p_ret = NULL;
413 uint8_t *p;
414
415 if( !pp_block || !*pp_block )
416 return NULL;
417 if( (*pp_block)->i_flags&(BLOCK_FLAG_DISCONTINUITY|BLOCK_FLAG_CORRUPTED) )
418 {
419 block_Release( *pp_block );
420 return NULL;
421 }
422
423 p_block = *pp_block;
424 *pp_block = NULL;
425
426 for( p = p_block->p_buffer; p < &p_block->p_buffer[p_block->i_buffer]; )
427 {
428 block_t *p_pic;
429 bool b_dummy;
430 int i_size = 0;
431 int i;
432
433 for( i = 0; i < p_sys->i_avcC_length_size; i++ )
434 {
435 i_size = (i_size << 8) | (*p++);
436 }
437
438 if( i_size <= 0 ||
439 i_size > ( p_block->p_buffer + p_block->i_buffer - p ) )
440 {
441 msg_Err( p_dec, "Broken frame : size %d is too big", i_size );
442 break;
443 }
444
445 block_t *p_part = CreateAnnexbNAL( p_dec, p, i_size );
446 if( !p_part )
447 break;
448
449 p_part->i_dts = p_block->i_dts;
450 p_part->i_pts = p_block->i_pts;
451
452 /* Parse the NAL */
453 if( ( p_pic = ParseNALBlock( p_dec, &b_dummy, p_part ) ) )
454 {
455 block_ChainAppend( &p_ret, p_pic );
456 }
457 p += i_size;
458 }
459 block_Release( p_block );
460
461 return p_ret;
462}

464/*****************************************************************************
465 * GetCc:
466 *****************************************************************************/
467static block_t *GetCc( decoder_t *p_dec, bool pb_present[4] )
468{
469 decoder_sys_t *p_sys = p_dec->p_sys;
470 block_t *p_cc;
471
472 for( int i = 0; i < 4; i++ )
473 pb_present[i] = p_sys->cc.pb_present[i];
474
475 if( p_sys->cc.i_data <= 0 )
476 return NULL;
477
478 p_cc = block_Alloc( p_sys->cc.i_data);
479 if( p_cc )
480 {
481 memcpy( p_cc->p_buffer, p_sys->cc.p_data, p_sys->cc.i_data );
482 p_cc->i_dts =
483 p_cc->i_pts = p_sys->cc.b_reorder ? p_sys->i_cc_pts : p_sys->i_cc_dts;
484 p_cc->i_flags = ( p_sys->cc.b_reorder ? p_sys->i_cc_flags : BLOCK_FLAG_TYPE_P ) & BLOCK_FLAG_TYPE_MASK;
485 }
486 cc_Flush( &p_sys->cc );
487 return p_cc;
488}

493static void PacketizeReset( void *p_private, bool b_broken )
494{
495 decoder_t *p_dec = p_private;
496 decoder_sys_t *p_sys = p_dec->p_sys;
497
498 if( b_broken )
499 {
500 if( p_sys->p_frame )
501 block_ChainRelease( p_sys->p_frame );
502 p_sys->p_frame = NULL;
503 p_sys->b_frame_sps = false;
504 p_sys->b_frame_pps = false;
505 p_sys->slice.i_frame_type = 0;
506 p_sys->b_slice = false;
507 }
508 p_sys->i_frame_pts = VLC_TS_INVALID;
509 p_sys->i_frame_dts = VLC_TS_INVALID;
510}
511static block_t *PacketizeParse( void *p_private, bool *pb_ts_used, block_t *p_block )
512{
513 decoder_t *p_dec = p_private;
514
515 /* Remove trailing 0 bytes */
516 while( p_block->i_buffer > 5 && p_block->p_buffer[p_block->i_buffer-1] == 0x00 )
517 p_block->i_buffer--;
518
519 return ParseNALBlock( p_dec, pb_ts_used, p_block );
520}

522{
523 VLC_UNUSED(p_private);
524 VLC_UNUSED(p_au);
525 return VLC_SUCCESS;
526}

528static block_t *CreateAnnexbNAL( decoder_t *p_dec, const uint8_t *p, int i_size )
529{
530 block_t *p_nal;
531
532 p_nal = block_Alloc( 4 + i_size );
533 if( !p_nal ) return NULL;
534
535 /* Add start code */
536 p_nal->p_buffer[0] = 0x00;
537 p_nal->p_buffer[1] = 0x00;
538 p_nal->p_buffer[2] = 0x00;
539 p_nal->p_buffer[3] = 0x01;
540
541 /* Copy nalu */
542 memcpy( &p_nal->p_buffer[4], p, i_size );
543
544 VLC_UNUSED(p_dec);
545 return p_nal;
546}

548static void CreateDecodedNAL( uint8_t **pp_ret, int *pi_ret,
549 const uint8_t *src, int i_src )
550{
551 const uint8_t *end = &src[i_src];
552 uint8_t *dst = malloc( i_src );
553
554 *pp_ret = dst;
555
556 if( dst )
557 {
558 while( src < end )
559 {
560 if( src < end - 3 && src[0] == 0x00 && src[1] == 0x00 &&
561 src[2] == 0x03 )
562 {
563 *dst++ = 0x00;
564 *dst++ = 0x00;
565
566 src += 3;
567 continue;
568 }
569 *dst++ = *src++;
570 }
571 }
572 *pi_ret = dst - *pp_ret;
573}

575static inline int bs_read_ue( bs_t *s )
576{
577 int i = 0;
578
579 while( bs_read1( s ) == 0 && s->p < s->p_end && i < 32 )
580 {
581 i++;
582 }
583 return( ( 1 << i) - 1 + bs_read( s, i ) );
584}
585
586static inline int bs_read_se( bs_t *s )
587{
588 int val = bs_read_ue( s );
589
590 return val&0x01 ? (val+1)/2 : -(val/2);
591}

593/*****************************************************************************
594 * ParseNALBlock: parses annexB type NALs
595 * All p_frag blocks are required to start with 0 0 0 1 4-byte startcode
596 *****************************************************************************/
597static block_t *ParseNALBlock( decoder_t *p_dec, bool *pb_ts_used, block_t *p_frag )
598{
599 decoder_sys_t *p_sys = p_dec->p_sys;
600 block_t *p_pic = NULL;
601
602 const int i_nal_ref_idc = (p_frag->p_buffer[4] >> 5)&0x03;
603 const int i_nal_type = p_frag->p_buffer[4]&0x1f;
604 const mtime_t i_frag_dts = p_frag->i_dts;
605 const mtime_t i_frag_pts = p_frag->i_pts;
606
607 if( p_sys->b_slice && ( !p_sys->b_sps || !p_sys->b_pps ) )
608 {
609 block_ChainRelease( p_sys->p_frame );
610 msg_Warn( p_dec, "waiting for SPS/PPS" );
611
612 /* Reset context */
613 p_sys->slice.i_frame_type = 0;
614 p_sys->p_frame = NULL;
615 p_sys->b_frame_sps = false;
616 p_sys->b_frame_pps = false;
617 p_sys->b_slice = false;
618 cc_Flush( &p_sys->cc_next );
619 }
620
621 if( ( !p_sys->b_sps || !p_sys->b_pps ) &&
622 i_nal_type >= NAL_SLICE && i_nal_type <= NAL_SLICE_IDR )
623 {
624 p_sys->b_slice = true;
625 /* Fragment will be discarded later on */
626 }
627 else if( i_nal_type >= NAL_SLICE && i_nal_type <= NAL_SLICE_IDR )
628 {
629 slice_t slice;
630 bool b_new_picture;
631
632 ParseSlice( p_dec, &b_new_picture, &slice, i_nal_ref_idc, i_nal_type, p_frag );
633
634 /* */
635 if( b_new_picture && p_sys->b_slice )
636 p_pic = OutputPicture( p_dec );
637
638 /* */
639 p_sys->slice = slice;
640 p_sys->b_slice = true;
641 }
642 else if( i_nal_type == NAL_SPS )
643 {
644 if( p_sys->b_slice )
645 p_pic = OutputPicture( p_dec );
646 p_sys->b_frame_sps = true;
647
648 PutSPS( p_dec, p_frag );
649
650 /* Do not append the SPS because we will insert it on keyframes */
651 p_frag = NULL;
652 }
653 else if( i_nal_type == NAL_PPS )
654 {
655 if( p_sys->b_slice )
656 p_pic = OutputPicture( p_dec );
657 p_sys->b_frame_pps = true;
658
659 PutPPS( p_dec, p_frag );
660
661 /* Do not append the PPS because we will insert it on keyframes */
662 p_frag = NULL;
663 }
664 else if( i_nal_type == NAL_AU_DELIMITER ||
665 i_nal_type == NAL_SEI ||
666 ( i_nal_type >= 13 && i_nal_type <= 18 ) )
667 {
668 if( p_sys->b_slice )
669 p_pic = OutputPicture( p_dec );
670
671 /* Parse SEI for CC support */
672 if( i_nal_type == NAL_SEI )
673 {
674 ParseSei( p_dec, p_frag );
675 }
676 else if( i_nal_type == NAL_AU_DELIMITER )
677 {
678 if( p_sys->p_frame && (p_sys->p_frame->i_flags & BLOCK_FLAG_PRIVATE_AUD) )
679 {
680 block_Release( p_frag );
681 p_frag = NULL;
682 }
683 else
684 {
685 p_frag->i_flags |= BLOCK_FLAG_PRIVATE_AUD;
686 }
687 }
688 }
689
690 /* Append the block */
691 if( p_frag )
692 block_ChainAppend( &p_sys->p_frame, p_frag );
693
694 *pb_ts_used = false;
695 if( p_sys->i_frame_dts <= VLC_TS_INVALID &&
696 p_sys->i_frame_pts <= VLC_TS_INVALID )
697 {
698 p_sys->i_frame_dts = i_frag_dts;
699 p_sys->i_frame_pts = i_frag_pts;
700 *pb_ts_used = true;
701 }
702 return p_pic;
703}

705static block_t *OutputPicture( decoder_t *p_dec )
706{
707 decoder_sys_t *p_sys = p_dec->p_sys;
708 block_t *p_pic;
709
710 if ( !p_sys->b_header && p_sys->i_recovery_frames != -1 )
711 {
712 if( p_sys->i_recovery_frames == 0 )
713 {
714 msg_Dbg( p_dec, "Recovery from SEI recovery point complete" );
715 p_sys->b_header = true;
716 }
717 --p_sys->i_recovery_frames;
718 }
719
720 if( !p_sys->b_header && p_sys->i_recovery_frames == -1 &&
721 p_sys->slice.i_frame_type != BLOCK_FLAG_TYPE_I)
722 return NULL;
723
724 const bool b_sps_pps_i = p_sys->slice.i_frame_type == BLOCK_FLAG_TYPE_I &&
725 p_sys->b_sps &&
726 p_sys->b_pps;
727 if( b_sps_pps_i || p_sys->b_frame_sps || p_sys->b_frame_pps )
728 {
729 block_t *p_head = NULL;
730 if( p_sys->p_frame->i_flags & BLOCK_FLAG_PRIVATE_AUD )
731 {
732 p_head = p_sys->p_frame;
733 p_sys->p_frame = p_sys->p_frame->p_next;
734 }
735
736 block_t *p_list = NULL;
737 for( int i = 0; i < SPS_MAX && (b_sps_pps_i || p_sys->b_frame_sps); i++ )
738 {
739 if( p_sys->pp_sps[i] )
740 block_ChainAppend( &p_list, block_Duplicate( p_sys->pp_sps[i] ) );
741 }
742 for( int i = 0; i < PPS_MAX && (b_sps_pps_i || p_sys->b_frame_pps); i++ )
743 {
744 if( p_sys->pp_pps[i] )
745 block_ChainAppend( &p_list, block_Duplicate( p_sys->pp_pps[i] ) );
746 }
747 if( b_sps_pps_i && p_list )
748 p_sys->b_header = true;
749
750 if( p_head )
751 p_head->p_next = p_list;
752 else
753 p_head = p_list;
754 block_ChainAppend( &p_head, p_sys->p_frame );
755
756 p_pic = block_ChainGather( p_head );
757 }
758 else
759 {
760 p_pic = block_ChainGather( p_sys->p_frame );
761 }
762 p_pic->i_dts = p_sys->i_frame_dts;
763 p_pic->i_pts = p_sys->i_frame_pts;
764 p_pic->i_length = 0; /* FIXME */
765 p_pic->i_flags |= p_sys->slice.i_frame_type;
766 p_pic->i_flags &= ~BLOCK_FLAG_PRIVATE_AUD;
767 if( !p_sys->b_header )
768 p_pic->i_flags |= BLOCK_FLAG_PREROLL;
769
770 p_sys->slice.i_frame_type = 0;
771 p_sys->p_frame = NULL;
772 p_sys->i_frame_dts = VLC_TS_INVALID;
773 p_sys->i_frame_pts = VLC_TS_INVALID;
774 p_sys->b_frame_sps = false;
775 p_sys->b_frame_pps = false;
776 p_sys->b_slice = false;
777
778 /* CC */
779 p_sys->i_cc_pts = p_pic->i_pts;
780 p_sys->i_cc_dts = p_pic->i_dts;
781 p_sys->i_cc_flags = p_pic->i_flags;
782
783 p_sys->cc = p_sys->cc_next;
784 cc_Flush( &p_sys->cc_next );
785
786 return p_pic;
787}

789static void PutSPS( decoder_t *p_dec, block_t *p_frag )
790{
791 decoder_sys_t *p_sys = p_dec->p_sys;
792
793 uint8_t *pb_dec = NULL;
794 int i_dec = 0;
795 bs_t s;
796 int i_tmp;
797 int i_sps_id;
798
799 CreateDecodedNAL( &pb_dec, &i_dec, &p_frag->p_buffer[5],
800 p_frag->i_buffer - 5 );
801
802 bs_init( &s, pb_dec, i_dec );
803 int i_profile_idc = bs_read( &s, 8 );
804 p_dec->fmt_out.i_profile = i_profile_idc;
805 /* Skip constraint_set0123, reserved(4) */
806 bs_skip( &s, 1+1+1+1 + 4 );
807 p_dec->fmt_out.i_level = bs_read( &s, 8 );
808 /* sps id */
809 i_sps_id = bs_read_ue( &s );
810 if( i_sps_id >= SPS_MAX || i_sps_id < 0 )
811 {
812 msg_Warn( p_dec, "invalid SPS (sps_id=%d)", i_sps_id );
813 free( pb_dec );
814 block_Release( p_frag );
815 return;
816 }
817
818 if( i_profile_idc == 100 || i_profile_idc == 110 ||
819 i_profile_idc == 122 || i_profile_idc == 244 ||
820 i_profile_idc == 44 || i_profile_idc == 83 ||
821 i_profile_idc == 86 )
822 {
823 /* chroma_format_idc */
824 const int i_chroma_format_idc = bs_read_ue( &s );
825 if( i_chroma_format_idc == 3 )
826 bs_skip( &s, 1 ); /* separate_colour_plane_flag */
827 /* bit_depth_luma_minus8 */
828 bs_read_ue( &s );
829 /* bit_depth_chroma_minus8 */
830 bs_read_ue( &s );
831 /* qpprime_y_zero_transform_bypass_flag */
832 bs_skip( &s, 1 );
833 /* seq_scaling_matrix_present_flag */
834 i_tmp = bs_read( &s, 1 );
835 if( i_tmp )
836 {
837 for( int i = 0; i < ((3 != i_chroma_format_idc) ? 8 : 12); i++ )
838 {
839 /* seq_scaling_list_present_flag[i] */
840 i_tmp = bs_read( &s, 1 );
841 if( !i_tmp )
842 continue;
843 const int i_size_of_scaling_list = (i < 6 ) ? 16 : 64;
844 /* scaling_list (...) */
845 int i_lastscale = 8;
846 int i_nextscale = 8;
847 for( int j = 0; j < i_size_of_scaling_list; j++ )
848 {
849 if( i_nextscale != 0 )
850 {
851 /* delta_scale */
852 i_tmp = bs_read_se( &s );
853 i_nextscale = ( i_lastscale + i_tmp + 256 ) % 256;
854 /* useDefaultScalingMatrixFlag = ... */
855 }
856 /* scalinglist[j] */
857 i_lastscale = ( i_nextscale == 0 ) ? i_lastscale : i_nextscale;
858 }
859 }
860 }
861 }
862
863 /* Skip i_log2_max_frame_num */
864 p_sys->i_log2_max_frame_num = bs_read_ue( &s );
865 if( p_sys->i_log2_max_frame_num > 12)
866 p_sys->i_log2_max_frame_num = 12;
867 /* Read poc_type */
868 p_sys->i_pic_order_cnt_type = bs_read_ue( &s );
869 if( p_sys->i_pic_order_cnt_type == 0 )
870 {
871 /* skip i_log2_max_poc_lsb */
872 p_sys->i_log2_max_pic_order_cnt_lsb = bs_read_ue( &s );
873 if( p_sys->i_log2_max_pic_order_cnt_lsb > 12 )
874 p_sys->i_log2_max_pic_order_cnt_lsb = 12;
875 }
876 else if( p_sys->i_pic_order_cnt_type == 1 )
877 {
878 int i_cycle;
879 /* skip b_delta_pic_order_always_zero */
880 p_sys->i_delta_pic_order_always_zero_flag = bs_read( &s, 1 );
881 /* skip i_offset_for_non_ref_pic */
882 bs_read_se( &s );
883 /* skip i_offset_for_top_to_bottom_field */
884 bs_read_se( &s );
885 /* read i_num_ref_frames_in_poc_cycle */
886 i_cycle = bs_read_ue( &s );
887 if( i_cycle > 256 ) i_cycle = 256;
888 while( i_cycle > 0 )
889 {
890 /* skip i_offset_for_ref_frame */
891 bs_read_se(&s );
892 i_cycle--;
893 }
894 }
895 /* i_num_ref_frames */
896 bs_read_ue( &s );
897 /* b_gaps_in_frame_num_value_allowed */
898 bs_skip( &s, 1 );
899
900 /* Read size */
901 p_dec->fmt_out.video.i_width = 16 * ( bs_read_ue( &s ) + 1 );
902 p_dec->fmt_out.video.i_height = 16 * ( bs_read_ue( &s ) + 1 );
903
904 /* b_frame_mbs_only */
905 p_sys->b_frame_mbs_only = bs_read( &s, 1 );
906 p_dec->fmt_out.video.i_height *= ( 2 - p_sys->b_frame_mbs_only );
907 if( p_sys->b_frame_mbs_only == 0 )
908 {
909 bs_skip( &s, 1 );
910 }
911 /* b_direct8x8_inference */
912 bs_skip( &s, 1 );
913
914 /* crop */
915 i_tmp = bs_read( &s, 1 );
916 if( i_tmp )
917 {
918 /* left */
919 bs_read_ue( &s );
920 /* right */
921 bs_read_ue( &s );
922 /* top */
923 bs_read_ue( &s );
924 /* bottom */
925 bs_read_ue( &s );
926 }
927
928 /* vui */
929 i_tmp = bs_read( &s, 1 );
930 if( i_tmp )
931 {
932 /* read the aspect ratio part if any */
933 i_tmp = bs_read( &s, 1 );
934 if( i_tmp )
935 {
936 static const struct { int w, h; } sar[17] =
937 {
938 { 0, 0 }, { 1, 1 }, { 12, 11 }, { 10, 11 },
939 { 16, 11 }, { 40, 33 }, { 24, 11 }, { 20, 11 },
940 { 32, 11 }, { 80, 33 }, { 18, 11 }, { 15, 11 },
941 { 64, 33 }, { 160,99 }, { 4, 3 }, { 3, 2 },
942 { 2, 1 },
943 };
944 int i_sar = bs_read( &s, 8 );
945 int w, h;
946
947 if( i_sar < 17 )
948 {
949 w = sar[i_sar].w;
950 h = sar[i_sar].h;
951 }
952 else if( i_sar == 255 )
953 {
954 w = bs_read( &s, 16 );
955 h = bs_read( &s, 16 );
956 }
957 else
958 {
959 w = 0;
960 h = 0;
961 }
962
963 if( w != 0 && h != 0 )
964 {
965 p_dec->fmt_out.video.i_sar_num = w;
966 p_dec->fmt_out.video.i_sar_den = h;
967 }
968 else
969 {
970 p_dec->fmt_out.video.i_sar_num = 1;
971 p_dec->fmt_out.video.i_sar_den = 1;
972 }
973 }
974 }
975
976 free( pb_dec );
977
978 /* We have a new SPS */
979 if( !p_sys->b_sps )
980 msg_Dbg( p_dec, "found NAL_SPS (sps_id=%d)", i_sps_id );
981 p_sys->b_sps = true;
982
983 if( p_sys->pp_sps[i_sps_id] )
984 block_Release( p_sys->pp_sps[i_sps_id] );
985 p_sys->pp_sps[i_sps_id] = p_frag;
986}

988static void PutPPS( decoder_t *p_dec, block_t *p_frag )
989{
990 decoder_sys_t *p_sys = p_dec->p_sys;
991 bs_t s;
992 int i_pps_id;
993 int i_sps_id;
994
995 bs_init( &s, &p_frag->p_buffer[5], p_frag->i_buffer - 5 );
996 i_pps_id = bs_read_ue( &s ); // pps id
997 i_sps_id = bs_read_ue( &s ); // sps id
998 if( i_pps_id >= PPS_MAX || i_sps_id >= SPS_MAX )
999 {
1000 msg_Warn( p_dec, "invalid PPS (pps_id=%d sps_id=%d)", i_pps_id, i_sps_id );
1001 block_Release( p_frag );
1002 return;
1003 }
1004 bs_skip( &s, 1 ); // entropy coding mode flag
1005 p_sys->i_pic_order_present_flag = bs_read( &s, 1 );

1007
1008 /* We have a new PPS */

1010 msg_Dbg( p_dec, "found NAL_PPS (pps_id=%d sps_id=%d)", i_pps_id, i_sps_id );
1011 p_sys->b_pps = true;
1012
1013 if( p_sys->pp_pps[i_pps_id] )
1014 block_Release( p_sys->pp_pps[i_pps_id] );
1015 p_sys->pp_pps[i_pps_id] = p_frag;
1016}

1018static void ParseSlice( decoder_t *p_dec, bool *pb_new_picture, slice_t *p_slice,
1019 int i_nal_ref_idc, int i_nal_type, const block_t *p_frag )
1020{
1021 decoder_sys_t *p_sys = p_dec->p_sys;
1022 uint8_t *pb_dec;
1023 int i_dec;
1024 int i_slice_type;
1025 slice_t slice;
1026 bs_t s;
1027
1028 /* do not convert the whole frame */
1029 CreateDecodedNAL( &pb_dec, &i_dec, &p_frag->p_buffer[5],
1030 __MIN( p_frag->i_buffer - 5, 60 ) );
1031 bs_init( &s, pb_dec, i_dec );
1032
1033 /* first_mb_in_slice */
1034 /* int i_first_mb = */ bs_read_ue( &s );
1035
1036 /* slice_type */
1037 switch( (i_slice_type = bs_read_ue( &s )) )
1038 {
1039 case 0: case 5:
1040 slice.i_frame_type = BLOCK_FLAG_TYPE_P;
1041 break;
1042 case 1: case 6:
1043 slice.i_frame_type = BLOCK_FLAG_TYPE_B;
1044 break;
1045 case 2: case 7:
1046 slice.i_frame_type = BLOCK_FLAG_TYPE_I;
1047 break;
1048 case 3: case 8: /* SP */
1049 slice.i_frame_type = BLOCK_FLAG_TYPE_P;
1050 break;
1051 case 4: case 9:
1052 slice.i_frame_type = BLOCK_FLAG_TYPE_I;
1053 break;
1054 default:
1055 slice.i_frame_type = 0;
1056 break;
1057 }
1058
1059 /* */
1060 slice.i_nal_type = i_nal_type;
1061 slice.i_nal_ref_idc = i_nal_ref_idc;
1062
1063 slice.i_pic_parameter_set_id = bs_read_ue( &s );
1064 slice.i_frame_num = bs_read( &s, p_sys->i_log2_max_frame_num + 4 );
1065
1066 slice.i_field_pic_flag = 0;
1067 slice.i_bottom_field_flag = -1;
1068 if( !p_sys->b_frame_mbs_only )
1069 {
1070 /* field_pic_flag */
1071 slice.i_field_pic_flag = bs_read( &s, 1 );
1072 if( slice.i_field_pic_flag )
1073 slice.i_bottom_field_flag = bs_read( &s, 1 );
1074 }
1075
1076 slice.i_idr_pic_id = p_sys->slice.i_idr_pic_id;
1077 if( slice.i_nal_type == NAL_SLICE_IDR )
1078 slice.i_idr_pic_id = bs_read_ue( &s );
1079
1080 slice.i_pic_order_cnt_lsb = -1;
1081 slice.i_delta_pic_order_cnt_bottom = -1;
1082 slice.i_delta_pic_order_cnt0 = 0;
1083 slice.i_delta_pic_order_cnt1 = 0;
1084 if( p_sys->i_pic_order_cnt_type == 0 )
1085 {
1086 slice.i_pic_order_cnt_lsb = bs_read( &s, p_sys->i_log2_max_pic_order_cnt_lsb + 4 );
1087 if( p_sys->i_pic_order_present_flag && !slice.i_field_pic_flag )
1088 slice.i_delta_pic_order_cnt_bottom = bs_read_se( &s );
1089 }
1090 else if( (p_sys->i_pic_order_cnt_type == 1) &&
1091 (!p_sys->i_delta_pic_order_always_zero_flag) )
1092 {
1093 slice.i_delta_pic_order_cnt0 = bs_read_se( &s );
1094 if( p_sys->i_pic_order_present_flag && !slice.i_field_pic_flag )
1095 slice.i_delta_pic_order_cnt1 = bs_read_se( &s );
1096 }
1097 free( pb_dec );
1098
1099 /* Detection of the first VCL NAL unit of a primary coded picture
1100 * (cf. 7.4.1.2.4) */
1101 bool b_pic = false;
1102 if( slice.i_frame_num != p_sys->slice.i_frame_num ||
1103 slice.i_pic_parameter_set_id != p_sys->slice.i_pic_parameter_set_id ||
1104 slice.i_field_pic_flag != p_sys->slice.i_field_pic_flag ||
1105 slice.i_nal_ref_idc != p_sys->slice.i_nal_ref_idc )
1106 b_pic = true;
1107 if( (slice.i_bottom_field_flag != -1) &&
1108 (p_sys->slice.i_bottom_field_flag != -1) &&
1109 (slice.i_bottom_field_flag != p_sys->slice.i_bottom_field_flag) )
1110 b_pic = true;
1111 if( p_sys->i_pic_order_cnt_type == 0 &&
1112 ( slice.i_pic_order_cnt_lsb != p_sys->slice.i_pic_order_cnt_lsb ||
1113 slice.i_delta_pic_order_cnt_bottom != p_sys->slice.i_delta_pic_order_cnt_bottom ) )
1114 b_pic = true;
1115 else if( p_sys->i_pic_order_cnt_type == 1 &&
1116 ( slice.i_delta_pic_order_cnt0 != p_sys->slice.i_delta_pic_order_cnt0 ||
1117 slice.i_delta_pic_order_cnt1 != p_sys->slice.i_delta_pic_order_cnt1 ) )
1118 b_pic = true;
1119 if( ( slice.i_nal_type == NAL_SLICE_IDR || p_sys->slice.i_nal_type == NAL_SLICE_IDR ) &&
1120 ( slice.i_nal_type != p_sys->slice.i_nal_type || slice.i_idr_pic_id != p_sys->slice.i_idr_pic_id ) )
1121 b_pic = true;
1122
1123 /* */
1124 *pb_new_picture = b_pic;
1125 *p_slice = slice;
1126}

1128static void ParseSei( decoder_t *p_dec, block_t *p_frag )
1129{
1130 decoder_sys_t *p_sys = p_dec->p_sys;
1131 uint8_t *pb_dec;
1132 int i_dec;
1133
1134 /* */
1135 CreateDecodedNAL( &pb_dec, &i_dec, &p_frag->p_buffer[5], p_frag->i_buffer - 5 );
1136 if( !pb_dec )
1137 return;
1138
1139 /* The +1 is for rbsp trailing bits */
1140 for( int i_used = 0; i_used+1 < i_dec; )
1141 {
1142 /* Read type */
1143 int i_type = 0;
1144 while( i_used+1 < i_dec )
1145 {
1146 const int i_byte = pb_dec[i_used++];
1147 i_type += i_byte;
1148 if( i_byte != 0xff )
1149 break;
1150 }
1151 /* Read size */
1152 int i_size = 0;
1153 while( i_used+1 < i_dec )
1154 {
1155 const int i_byte = pb_dec[i_used++];
1156 i_size += i_byte;
1157 if( i_byte != 0xff )
1158 break;
1159 }
1160 /* Check room */
1161 if( i_used + i_size + 1 > i_dec )
1162 break;
1163
1164 /* Look for user_data_registered_itu_t_t35 */
1165 if( i_type == 4 )
1166 {
1167 static const uint8_t p_dvb1_data_start_code[] = {
1168 0xb5,
1169 0x00, 0x31,
1170 0x47, 0x41, 0x39, 0x34
1171 };
1172 const int i_t35 = i_size;
1173 const uint8_t *p_t35 = &pb_dec[i_used];
1174
1175 /* Check for we have DVB1_data() */
1176 if( i_t35 >= 5 &&
1177 !memcmp( p_t35, p_dvb1_data_start_code, sizeof(p_dvb1_data_start_code) ) )
1178 {
1179 cc_Extract( &p_sys->cc_next, true, &p_t35[3], i_t35 - 3 );
1180 }
1181 }
1182
1183 /* Look for SEI recovery point */
1184 if( i_type == 6 )
1185 {
1186 bs_t s;
1187 const int i_rec = i_size;
1188 const uint8_t *p_rec = &pb_dec[i_used];
1189
1190 bs_init( &s, p_rec, i_rec );
1191 int i_recovery_frames = bs_read_ue( &s );
1192 //bool b_exact_match = bs_read( &s, 1 );
1193 //bool b_broken_link = bs_read( &s, 1 );
1194 //int i_changing_slice_group = bs_read( &s, 2 );
1195 if( !p_sys->b_header )
1196 {
1197 msg_Dbg( p_dec, "Seen SEI recovery point, %d recovery frames", i_recovery_frames );
1198 if ( p_sys->i_recovery_frames == -1 || i_recovery_frames < p_sys->i_recovery_frames )
1199 p_sys->i_recovery_frames = i_recovery_frames;
1200 }
1201 }
1202
1203 i_used += i_size;
1204 }
1205
1206 free( pb_dec );
1207}
