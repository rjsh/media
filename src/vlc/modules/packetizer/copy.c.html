67/*****************************************************************************
68 * Open: probe the packetizer and return score
69 *****************************************************************************
70 * Tries to launch a decoder and return score so that the interface is able
71 * to choose.
72 *****************************************************************************/
73static int Open( vlc_object_t *p_this )
74{
75 decoder_t *p_dec = (decoder_t*)p_this;
76 decoder_sys_t *p_sys;
77
78 if( p_dec->fmt_in.i_cat != AUDIO_ES &&
79 p_dec->fmt_in.i_cat != VIDEO_ES &&
80 p_dec->fmt_in.i_cat != SPU_ES )
81 {
82 msg_Err( p_dec, "invalid ES type" );
83 return VLC_EGENERIC;
84 }
85
86 if( p_dec->fmt_in.i_cat == SPU_ES )
87 p_dec->pf_packetize = PacketizeSub;
88 else
89 p_dec->pf_packetize = Packetize;
90
91 /* Create the output format */
92 es_format_Copy( &p_dec->fmt_out, &p_dec->fmt_in );
93
94 /* Fix the value of the fourcc for audio */
95 if( p_dec->fmt_in.i_cat == AUDIO_ES )
96 {
97 p_dec->fmt_out.i_codec = vlc_fourcc_GetCodecAudio( p_dec->fmt_in.i_codec,
98 p_dec->fmt_in.audio.i_bitspersample );
99 if( !p_dec->fmt_out.i_codec )
100 {
101 msg_Err( p_dec, "unknown raw audio sample size" );
102 return VLC_EGENERIC;
103 }
104 }
105
106 p_dec->p_sys = p_sys = malloc( sizeof(*p_sys) );
107 p_sys->p_block = NULL;
108 switch( p_dec->fmt_in.i_codec )
109 {
110 case VLC_CODEC_WMV3:
111 p_sys->pf_parse = ParseWMV3;
112 break;
113 default:
114 p_sys->pf_parse = NULL;
115 break;
116 }
117
118 return VLC_SUCCESS;
119}

121/*****************************************************************************
122 * Close:
123 *****************************************************************************/
124static void Close( vlc_object_t *p_this )
125{
126 decoder_t *p_dec = (decoder_t*)p_this;
127
128 if( p_dec->p_sys->p_block )
129 {
130 block_ChainRelease( p_dec->p_sys->p_block );
131 }
132
133 es_format_Clean( &p_dec->fmt_out );
134 free( p_dec->p_sys );
135}

137/*****************************************************************************
138 * Packetize: packetize an unit (here copy a complete block )
139 *****************************************************************************/
140static block_t *Packetize ( decoder_t *p_dec, block_t **pp_block )
141{
142 block_t *p_block;
143 block_t *p_ret = p_dec->p_sys->p_block;
144
145 if( pp_block == NULL || *pp_block == NULL )
146 return NULL;
147 if( (*pp_block)->i_flags&(BLOCK_FLAG_DISCONTINUITY|BLOCK_FLAG_CORRUPTED) )
148 {
149 block_Release( *pp_block );
150 return NULL;
151 }
152
153 p_block = *pp_block;
154 *pp_block = NULL;
155
156 if( p_block->i_dts <= VLC_TS_INVALID )
157 {
158 p_block->i_dts = p_block->i_pts;
159 }
160
161 if( p_block->i_dts <= VLC_TS_INVALID )
162 {
163 msg_Dbg( p_dec, "need valid dts" );
164 block_Release( p_block );
165 return NULL;
166 }
167
168 if( p_ret != NULL && p_block->i_pts > p_ret->i_pts )
169 {
170 if (p_dec->fmt_in.i_codec != VLC_CODEC_OPUS)
171 p_ret->i_length = p_block->i_pts - p_ret->i_pts;
172 }
173 p_dec->p_sys->p_block = p_block;
174
175 if( p_ret && p_dec->p_sys->pf_parse )
176 p_dec->p_sys->pf_parse( p_dec, p_ret );
177 return p_ret;
178}

180/*****************************************************************************
181 * PacketizeSub: packetize an unit (here copy a complete block )
182 *****************************************************************************/
183static block_t *PacketizeSub( decoder_t *p_dec, block_t **pp_block )
184{
185 block_t *p_block;
186
187 if( pp_block == NULL || *pp_block == NULL )
188 return NULL;
189 if( (*pp_block)->i_flags&(BLOCK_FLAG_DISCONTINUITY|BLOCK_FLAG_CORRUPTED) )
190 {
191 block_Release( *pp_block );
192 return NULL;
193 }
194
195 p_block = *pp_block;
196 *pp_block = NULL;
197
198 if( p_block->i_dts <= VLC_TS_INVALID )
199 {
200 p_block->i_dts = p_block->i_pts;
201 }
202
203 if( p_block->i_dts <= VLC_TS_INVALID )
204 {
205 msg_Dbg( p_dec, "need valid dts" );
206 block_Release( p_block );
207 return NULL;
208 }
209
210 return p_block;
211}

213/* Parse WMV3 packet and extract frame type information */
214static void ParseWMV3( decoder_t *p_dec, block_t *p_block )
215{
216 bs_t s;
217
218 /* Parse Sequence header */
219 bs_init( &s, p_dec->fmt_in.p_extra, p_dec->fmt_in.i_extra );
220 if( bs_read( &s, 2 ) == 3 )
221 return;
222 bs_skip( &s, 22 );
223 const bool b_range_reduction = bs_read( &s, 1 );
224 const bool b_has_frames = bs_read( &s, 3 ) > 0;
225 bs_skip( &s, 2 );
226 const bool b_frame_interpolation = bs_read( &s, 1 );
227 if( bs_eof( &s ) )
228 return;
229
230 /* Parse frame type */
231 bs_init( &s, p_block->p_buffer, p_block->i_buffer );
232 bs_skip( &s, b_frame_interpolation +
233 2 +
234 b_range_reduction );
235
236 p_block->i_flags &= ~BLOCK_FLAG_TYPE_MASK;
237 if( bs_read( &s, 1 ) )
238 p_block->i_flags |= BLOCK_FLAG_TYPE_P;
239 else if( !b_has_frames || bs_read( &s, 1 ) )
240 p_block->i_flags |= BLOCK_FLAG_TYPE_I;
241 else
242 p_block->i_flags |= BLOCK_FLAG_TYPE_B;
243}
