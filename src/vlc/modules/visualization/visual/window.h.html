24#ifndef _WINDOW_H_
25#define _WINDOW_H_
26
27#include <vlc_common.h>
28
29/* Window type enum */
30enum _enum_window_type { NONE, HANN, FLATTOP, BLACKMANHARRIS, KAISER };
31
32/* Window context structure */
33struct _struct_window_context {
34
35 /* Storage for window function values */
36 float * pf_window_table;
37
38 /* Buffer size for the window */
39 int i_buffer_size;
40};
41
42typedef enum _enum_window_type window_type;
43
44/* Window parameter structure */
45struct _struct_window_param {
46
47 /* Window type */
48 window_type wind_type;
49
50 /* Kaiser window parameter */
51 float f_kaiser_alpha;
52};
53
54/* Prototypes for the window function */
55typedef struct _struct_window_context window_context;
56typedef struct _struct_window_param window_param;
57void window_get_param( vlc_object_t * p_aout, window_param * p_param );
58bool window_init( int i_buffer_size, window_param * p_param,
59 window_context * p_ctx );
60void window_scale_in_place( int16_t * p_buffer, window_context * p_ctx );
61void window_close( window_context * p_ctx );
62
63/* Macro for defining a new window context */
64#define DEFINE_WIND_CONTEXT(name) window_context name = {NULL, 0}
65
66#endif /* _WINDOW_H_ */
