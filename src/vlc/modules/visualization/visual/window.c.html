34/* Flat top window coefficients */
35#define FT_A0 1.000f
36#define FT_A1 1.930f
37#define FT_A2 1.290f
38#define FT_A3 0.388f
39#define FT_A4 0.028f
40
41/* Blackman-Harris window coefficients */
42#define BH_A0 0.35875f
43#define BH_A1 0.48829f
44#define BH_A2 0.14128f
45#define BH_A3 0.01168f
46
47/*
48 * The modified Bessel function I0(x). See Chapter 6 of the "Numerical Recipes
49 * in C: The Art of Scientific Computing" book at
50 * http://www.aip.de/groups/soe/local/numres/bookcpdf/c6-6.pdf
51 */
52static float bessi0( float x )
53{
54 float ax, ans;
55 double y; /* Accumulate polynomials in double precision. */
56 if( ( ax = fabsf( x ) ) < 3.75f ) /* Polynomial fit. */
57 {
58 y = x / 3.75;
59 y *= y;
60 ans = 1.0 + y * ( 3.5156229 + y * ( 3.0899424 + y * ( 1.2067492
61 + y * ( 0.2659732 + y * ( 0.360768e-1
62 + y * 0.45813e-2 ) ) ) ) );
63 }
64 else
65 {
66 y = 3.75 / ax;
67 ans = ( exp( ax ) / sqrt( ax ) ) * ( 0.39894228 + y * ( 0.1328592e-1
68 + y * ( 0.225319e-2 + y * ( -0.157565e-2 + y * ( 0.916281e-2
69 + y * ( -0.2057706e-1 + y * ( 0.2635537e-1 + y * ( -0.1647633e-1
70 + y * 0.392377e-2 ) ) ) ) ) ) ) );
71 }
72 return ans;
73}

75/*
76 * Obtain the window type from the window type variable.
77 */
78void window_get_param( vlc_object_t * p_aout, window_param * p_param )
79{
80 /* Fetch Kaiser parameter */
81 p_param->f_kaiser_alpha = var_InheritFloat( p_aout, "effect-kaiser-param" );
82
83 /* Fetch window type */
84 char * psz_preset = var_InheritString( p_aout, "effect-fft-window" );
85 if( !psz_preset )
86 {
87 goto no_preset;
88 }
89
90 for( int i = 0; i < NB_WINDOWS; i++ )
91 {
92 if( !strcasecmp( psz_preset, window_list[i] ) )
93 {
94 p_param->wind_type = i;
95 return;
96 }
97 }
98
99no_preset:
100 msg_Warn( p_aout, "No matching window preset found; using rectangular "
101 "window (i.e. no window)" );
102 p_param->wind_type = NONE;
103}

105/*
106 * Initialization routine - sets up a lookup table for scaling a sample of data
107 * by window data. If the lookup table is successfully allocated, its memory
108 * location and its specified size are stored at the specified memory location
109 * of the internal context.
110 * Returns true if initialization succeeded and returns false otherwise.
111 * The internal context should be freed when it is finished with, by
112 * window_close().
113 */
114bool window_init( int i_buffer_size, window_param * p_param,
115 window_context * p_ctx )
116{
117 float * pf_table = NULL;
118 window_type wind_type = p_param->wind_type;
119
120 if( wind_type != HANN && wind_type != FLATTOP
121 && wind_type != BLACKMANHARRIS
122 && wind_type != KAISER )
123 {
124 /* Assume a rectangular window (i.e. no window) */
125 i_buffer_size = 0;
126 goto exit;
127 }
128
129 pf_table = malloc( i_buffer_size * sizeof( *pf_table ) );
130 if( !pf_table )
131 {
132 /* Memory allocation failed */
133 return false;
134 }
135
136 int i_buffer_size_minus_1 = i_buffer_size - 1;
137 switch( wind_type )
138 {
139 case HANN:
140 /* Hann window */
141 for( int i = 0; i < i_buffer_size; i++ )
142 {
143 float f_val = (float) i / (float) i_buffer_size_minus_1;
144 pf_table[i] = 0.5f - 0.5f * cosf( 2.0f * (float) M_PI * f_val );
145 }
146 break;
147 case FLATTOP:
148 /* Flat top window */
149 for( int i = 0; i < i_buffer_size; i++ )
150 {
151 float f_val = (float) i / (float) i_buffer_size_minus_1;
152 pf_table[i] = FT_A0
153 - FT_A1 * cosf( 2.0f * (float) M_PI * f_val )
154 + FT_A2 * cosf( 4.0f * (float) M_PI * f_val )
155 - FT_A3 * cosf( 6.0f * (float) M_PI * f_val )
156 + FT_A4 * cosf( 8.0f * (float) M_PI * f_val );
157 }
158 break;
159 case BLACKMANHARRIS:
160 /* Blackman-Harris window */
161 for( int i = 0; i < i_buffer_size; i++ )
162 {
163 float f_val = (float) i / (float) i_buffer_size_minus_1;
164 pf_table[i] = BH_A0
165 - BH_A1 * cosf( 2.0f * (float) M_PI * f_val )
166 + BH_A2 * cosf( 4.0f * (float) M_PI * f_val )
167 - BH_A3 * cosf( 6.0f * (float) M_PI * f_val );
168 }
169 break;
170 case KAISER:
171 {
172 /* Kaiser window */
173 float f_pialph = (float) M_PI * p_param->f_kaiser_alpha;
174 float f_bessi0_pialph = bessi0( f_pialph );
175 for( int i = 0; i < i_buffer_size; i++ )
176 {
177 float f_val = (float) i / (float) i_buffer_size_minus_1;
178 float f_term_to_square = 2.0f * f_val - 1.0f;
179 float f_sqd_term = f_term_to_square * f_term_to_square;
180 float f_sqr_term = sqrtf( 1.0f - f_sqd_term );
181 pf_table[i] = bessi0( f_pialph * f_sqr_term ) / f_bessi0_pialph;
182 }
183 break;
184 }
185 default:
186 /* We should not reach here */
187 assert(0);
188 break;
189 }
190
191exit:
192 p_ctx->pf_window_table = pf_table;
193 p_ctx->i_buffer_size = i_buffer_size;
194 return true;
195}

197/*
198 * Perform an in-place scaling of the input buffer by the window data
199 * referenced from the specified context.
200 */
201void window_scale_in_place( int16_t * p_buffer, window_context * p_ctx )
202{
203 for( int i = 0; i < p_ctx->i_buffer_size; i++ )
204 {
205 p_buffer[i] *= p_ctx->pf_window_table[i];
206 }
207}
208
209/*
210 * Free the context.
211 */
212void window_close( window_context * p_ctx )
213{
214 if( p_ctx->pf_window_table )
215 {
216 free( p_ctx->pf_window_table );
217 p_ctx->pf_window_table = NULL;
218 p_ctx->i_buffer_size = 0;
219 }
