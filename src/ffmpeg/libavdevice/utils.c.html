19#include "internal.h"
20#include "libavutil/opt.h"
21#include "libavformat/avformat.h"
22
23int ff_alloc_input_device_context(AVFormatContext **avctx, AVInputFormat *iformat, const char *format)
24{
25 AVFormatContext *s;
26 int ret = 0;
27
28 *avctx = NULL;
29 if (!iformat && !format)
30 return AVERROR(EINVAL);
31 if (!(s = avformat_alloc_context()))
32 return AVERROR(ENOMEM);
33
34 if (!iformat)
35 iformat = av_find_input_format(format);
36 if (!iformat || !iformat->priv_class || !AV_IS_INPUT_DEVICE(iformat->priv_class->category)) {
37 ret = AVERROR(EINVAL);
38 goto error;
39 }
40 s->iformat = iformat;
41 if (s->iformat->priv_data_size > 0) {
42 s->priv_data = av_mallocz(s->iformat->priv_data_size);
43 if (!s->priv_data) {
44 ret = AVERROR(ENOMEM);
45 goto error;
46 }
47 if (s->iformat->priv_class) {
48 *(const AVClass**)s->priv_data= s->iformat->priv_class;
49 av_opt_set_defaults(s->priv_data);
50 }
51 } else
52 s->priv_data = NULL;
53
54 *avctx = s;
55 return 0;
56 error:
57 avformat_free_context(s);
58 return ret;
59}
