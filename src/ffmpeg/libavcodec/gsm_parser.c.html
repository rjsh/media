21/**
22 * @file
23 * GSM audio parser
24 *
25 * Splits packets into individual blocks.
26 */
27
28#include "libavutil/avassert.h"
29#include "parser.h"
30#include "gsm.h"
31
32typedef struct GSMParseContext {
33 ParseContext pc;
34 int block_size;
35 int duration;
36 int remaining;
37} GSMParseContext;
38
39static int gsm_parse(AVCodecParserContext *s1, AVCodecContext *avctx,
40 const uint8_t **poutbuf, int *poutbuf_size,
41 const uint8_t *buf, int buf_size)
42{
43 GSMParseContext *s = s1->priv_data;
44 ParseContext *pc = &s->pc;
45 int next;
46
47 if (!s->block_size) {
48 switch (avctx->codec_id) {
49 case AV_CODEC_ID_GSM:
50 s->block_size = GSM_BLOCK_SIZE;
51 s->duration = GSM_FRAME_SIZE;
52 break;
53 case AV_CODEC_ID_GSM_MS:
54 s->block_size = avctx->block_align ? avctx->block_align
55 : GSM_MS_BLOCK_SIZE;
56 s->duration = GSM_FRAME_SIZE * 2;
57 break;
58 default:
59 av_assert0(0);
60 }
61 }
62
63 if (!s->remaining)
64 s->remaining = s->block_size;
65 if (s->remaining <= buf_size) {
66 next = s->remaining;
67 s->remaining = 0;
68 } else {
69 next = END_NOT_FOUND;
70 s->remaining -= buf_size;
71 }
72
73 if (ff_combine_frame(pc, next, &buf, &buf_size) < 0 || !buf_size) {
74 *poutbuf = NULL;
75 *poutbuf_size = 0;
76 return buf_size;
77 }
78
79 s1->duration = s->duration;
80
81 *poutbuf = buf;
82 *poutbuf_size = buf_size;
83 return next;
84}
85
86AVCodecParser ff_gsm_parser = {
87 .codec_ids = { AV_CODEC_ID_GSM, AV_CODEC_ID_GSM_MS },
88 .priv_data_size = sizeof(GSMParseContext),
89 .parser_parse = gsm_parse,
90 .parser_close = ff_parse_close,
91};
