27/**
28 * Parse MPEG-4 audio configuration for ALS object type.
29 * @param[in] gb bit reader context
30 * @param[in] c MPEG4AudioConfig structure to fill
31 * @return on success 0 is returned, otherwise a value < 0
32 */
33static int parse_config_ALS(GetBitContext *gb, MPEG4AudioConfig *c)
34{
35 if (get_bits_left(gb) < 112)
36 return -1;
37
38 if (get_bits_long(gb, 32) != MKBETAG('A','L','S','\0'))
39 return -1;
40
41 // override AudioSpecificConfig channel configuration and sample rate
42 // which are buggy in old ALS conformance files
43 c->sample_rate = get_bits_long(gb, 32);
44
45 // skip number of samples
46 skip_bits_long(gb, 32);
47
48 // read number of channels
49 c->chan_config = 0;
50 c->channels = get_bits(gb, 16) + 1;
51
52 return 0;
53}

55/* XXX: make sure to update the copies in the different encoders if you change
56 * this table */
57const int avpriv_mpeg4audio_sample_rates[16] = {
58 96000, 88200, 64000, 48000, 44100, 32000,
59 24000, 22050, 16000, 12000, 11025, 8000, 7350
60};
61
62const uint8_t ff_mpeg4audio_channels[8] = {
63 0, 1, 2, 3, 4, 5, 6, 8
64};
65
66static inline int get_object_type(GetBitContext *gb)
67{
68 int object_type = get_bits(gb, 5);
69 if (object_type == AOT_ESCAPE)
70 object_type = 32 + get_bits(gb, 6);
71 return object_type;
72}
73
74static inline int get_sample_rate(GetBitContext *gb, int *index)
75{
76 *index = get_bits(gb, 4);
77 return *index == 0x0f ? get_bits(gb, 24) :
78 avpriv_mpeg4audio_sample_rates[*index];
79}

81int avpriv_mpeg4audio_get_config(MPEG4AudioConfig *c, const uint8_t *buf,
82 int bit_size, int sync_extension)
83{
84 GetBitContext gb;
85 int specific_config_bitindex, ret;
86
87 if (bit_size <= 0)
88 return AVERROR_INVALIDDATA;
89
90 ret = init_get_bits(&gb, buf, bit_size);
91 if (ret < 0)
92 return ret;
93
94 c->object_type = get_object_type(&gb);
95 c->sample_rate = get_sample_rate(&gb, &c->sampling_index);
96 c->chan_config = get_bits(&gb, 4);
97 if (c->chan_config < FF_ARRAY_ELEMS(ff_mpeg4audio_channels))
98 c->channels = ff_mpeg4audio_channels[c->chan_config];
99 c->sbr = -1;
100 c->ps = -1;
101 if (c->object_type == AOT_SBR || (c->object_type == AOT_PS &&
102 // check for W6132 Annex YYYY draft MP3onMP4
103 !(show_bits(&gb, 3) & 0x03 && !(show_bits(&gb, 9) & 0x3F)))) {
104 if (c->object_type == AOT_PS)
105 c->ps = 1;
106 c->ext_object_type = AOT_SBR;
107 c->sbr = 1;
108 c->ext_sample_rate = get_sample_rate(&gb, &c->ext_sampling_index);
109 c->object_type = get_object_type(&gb);
110 if (c->object_type == AOT_ER_BSAC)
111 c->ext_chan_config = get_bits(&gb, 4);
112 } else {
113 c->ext_object_type = AOT_NULL;
114 c->ext_sample_rate = 0;
115 }
116 specific_config_bitindex = get_bits_count(&gb);
117
118 if (c->object_type == AOT_ALS) {
119 skip_bits(&gb, 5);
120 if (show_bits_long(&gb, 24) != MKBETAG('\0','A','L','S'))
121 skip_bits_long(&gb, 24);
122
123 specific_config_bitindex = get_bits_count(&gb);
124
125 if (parse_config_ALS(&gb, c))
126 return -1;
127 }
128
129 if (c->ext_object_type != AOT_SBR && sync_extension) {
130 while (get_bits_left(&gb) > 15) {
131 if (show_bits(&gb, 11) == 0x2b7) { // sync extension
132 get_bits(&gb, 11);
133 c->ext_object_type = get_object_type(&gb);
134 if (c->ext_object_type == AOT_SBR && (c->sbr = get_bits1(&gb)) == 1) {
135 c->ext_sample_rate = get_sample_rate(&gb, &c->ext_sampling_index);
136 if (c->ext_sample_rate == c->sample_rate)
137 c->sbr = -1;
138 }
139 if (get_bits_left(&gb) > 11 && get_bits(&gb, 11) == 0x548)
140 c->ps = get_bits1(&gb);
141 break;
142 } else
143 get_bits1(&gb); // skip 1 bit
144 }
145 }
146
147 //PS requires SBR
148 if (!c->sbr)
149 c->ps = 0;
150 //Limit implicit PS to the HE-AACv2 Profile
151 if ((c->ps == -1 && c->object_type != AOT_AAC_LC) || c->channels & ~0x01)
152 c->ps = 0;
153
154 return specific_config_bitindex;
155}

157static av_always_inline unsigned int copy_bits(PutBitContext *pb,
158 GetBitContext *gb,
159 int bits)
160{
161 unsigned int el = get_bits(gb, bits);
162 put_bits(pb, bits, el);
163 return el;
164}

166int avpriv_copy_pce_data(PutBitContext *pb, GetBitContext *gb)
167{
168 int five_bit_ch, four_bit_ch, comment_size, bits;
169 int offset = put_bits_count(pb);
170
171 copy_bits(pb, gb, 10); //Tag, Object Type, Frequency
172 five_bit_ch = copy_bits(pb, gb, 4); //Front
173 five_bit_ch += copy_bits(pb, gb, 4); //Side
174 five_bit_ch += copy_bits(pb, gb, 4); //Back
175 four_bit_ch = copy_bits(pb, gb, 2); //LFE
176 four_bit_ch += copy_bits(pb, gb, 3); //Data
177 five_bit_ch += copy_bits(pb, gb, 4); //Coupling
178 if (copy_bits(pb, gb, 1)) //Mono Mixdown
179 copy_bits(pb, gb, 4);
180 if (copy_bits(pb, gb, 1)) //Stereo Mixdown
181 copy_bits(pb, gb, 4);
182 if (copy_bits(pb, gb, 1)) //Matrix Mixdown
183 copy_bits(pb, gb, 3);
184 for (bits = five_bit_ch*5+four_bit_ch*4; bits > 16; bits -= 16)
185 copy_bits(pb, gb, 16);
186 if (bits)
187 copy_bits(pb, gb, bits);
188 avpriv_align_put_bits(pb);
189 align_get_bits(gb);
190 comment_size = copy_bits(pb, gb, 8);
191 for (; comment_size > 0; comment_size--)
192 copy_bits(pb, gb, 8);
193
194 return put_bits_count(pb) - offset;
195}
