30static void srt_to_ass(AVCodecContext *avctx, AVBPrint *dst,
31 const char *in, int x1, int y1, int x2, int y2)
32{
33 if (x1 >= 0 && y1 >= 0) {
34 /* XXX: here we rescale coordinate assuming they are in DVD resolution
35 * (720x480) since we don't have anything better */
36
37 if (x2 >= 0 && y2 >= 0 && (x2 != x1 || y2 != y1) && x2 >= x1 && y2 >= y1) {
38 /* text rectangle defined, write the text at the center of the rectangle */
39 const int cx = x1 + (x2 - x1)/2;
40 const int cy = y1 + (y2 - y1)/2;
41 const int scaled_x = cx * ASS_DEFAULT_PLAYRESX / 720;
42 const int scaled_y = cy * ASS_DEFAULT_PLAYRESY / 480;
43 av_bprintf(dst, "{\\an5}{\\pos(%d,%d)}", scaled_x, scaled_y);
44 } else {
45 /* only the top left corner, assume the text starts in that corner */
46 const int scaled_x = x1 * ASS_DEFAULT_PLAYRESX / 720;
47 const int scaled_y = y1 * ASS_DEFAULT_PLAYRESY / 480;
48 av_bprintf(dst, "{\\an1}{\\pos(%d,%d)}", scaled_x, scaled_y);
49 }
50 }
51
52 ff_htmlmarkup_to_ass(avctx, dst, in);
53}

55static int srt_decode_frame(AVCodecContext *avctx,
56 void *data, int *got_sub_ptr, AVPacket *avpkt)
57{
58 AVSubtitle *sub = data;
59 AVBPrint buffer;
60 int x1 = -1, y1 = -1, x2 = -1, y2 = -1;
61 int size, ret;
62 const uint8_t *p = av_packet_get_side_data(avpkt, AV_PKT_DATA_SUBTITLE_POSITION, &size);
63 FFASSDecoderContext *s = avctx->priv_data;
64
65 if (p && size == 16) {
66 x1 = AV_RL32(p );
67 y1 = AV_RL32(p + 4);
68 x2 = AV_RL32(p + 8);
69 y2 = AV_RL32(p + 12);
70 }
71
72 if (avpkt->size <= 0)
73 return avpkt->size;
74
75 av_bprint_init(&buffer, 0, AV_BPRINT_SIZE_UNLIMITED);
76
77 srt_to_ass(avctx, &buffer, avpkt->data, x1, y1, x2, y2);
78 ret = ff_ass_add_rect(sub, buffer.str, s->readorder++, 0, NULL, NULL);
79 av_bprint_finalize(&buffer, NULL);
80 if (ret < 0)
81 return ret;
82
83 *got_sub_ptr = sub->num_rects > 0;
84 return avpkt->size;
85}
