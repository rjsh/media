38/* motion vector table */
39typedef struct MVTable {
40 int n;
41 const uint16_t *table_mv_code;
42 const uint8_t *table_mv_bits;
43 const uint8_t *table_mvx;
44 const uint8_t *table_mvy;
45 uint16_t *table_mv_index; /* encoding: convert mv to index in table_mv */
46 VLC vlc; /* decoding: vlc */
47} MVTable;
