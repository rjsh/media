36struct AACISError {
37 int pass; /* 1 if dist2 <= dist1 */
38 int phase; /* -1 or +1 */
39 float error; /* fabs(dist1 - dist2) */
40 float dist1; /* From original coeffs */
41 float dist2; /* From IS'd coeffs */
42 float ener01;
43};
