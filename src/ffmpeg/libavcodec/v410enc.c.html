28static av_cold int v410_encode_init(AVCodecContext *avctx)
29{
30 if (avctx->width & 1) {
31 av_log(avctx, AV_LOG_ERROR, "v410 requires width to be even.\n");
32 return AVERROR_INVALIDDATA;
33 }
34
35 return 0;
36}
37
38static int v410_encode_frame(AVCodecContext *avctx, AVPacket *pkt,
39 const AVFrame *pic, int *got_packet)
40{
41 uint8_t *dst;
42 uint16_t *y, *u, *v;
43 uint32_t val;
44 int i, j, ret;
45
46 if ((ret = ff_alloc_packet2(avctx, pkt, avctx->width * avctx->height * 4,
47 avctx->width * avctx->height * 4)) < 0)
48 return ret;
49 dst = pkt->data;
50
51#if FF_API_CODED_FRAME
52FF_DISABLE_DEPRECATION_WARNINGS
53 avctx->coded_frame->key_frame = 1;
54 avctx->coded_frame->pict_type = AV_PICTURE_TYPE_I;
55FF_ENABLE_DEPRECATION_WARNINGS
56#endif
57
58 y = (uint16_t *)pic->data[0];
59 u = (uint16_t *)pic->data[1];
60 v = (uint16_t *)pic->data[2];
61
62 for (i = 0; i < avctx->height; i++) {
63 for (j = 0; j < avctx->width; j++) {
64 val = u[j] << 2;
65 val |= y[j] << 12;
66 val |= (uint32_t) v[j] << 22;
67 AV_WL32(dst, val);
68 dst += 4;
69 }
70 y += pic->linesize[0] >> 1;
71 u += pic->linesize[1] >> 1;
72 v += pic->linesize[2] >> 1;
73 }
74
75 pkt->flags |= AV_PKT_FLAG_KEY;
76 *got_packet = 1;
77 return 0;
78}

80AVCodec ff_v410_encoder = {
81 .name = "v410",
82 .long_name = NULL_IF_CONFIG_SMALL("Uncompressed 4:4:4 10-bit"),
83 .type = AVMEDIA_TYPE_VIDEO,
84 .id = AV_CODEC_ID_V410,
85 .init = v410_encode_init,
86 .encode2 = v410_encode_frame,
87 .pix_fmts = (const enum AVPixelFormat[]){ AV_PIX_FMT_YUV444P10, AV_PIX_FMT_NONE },
88};
