53/** Custom channel map locations bitmask
54 * Other channels described in documentation:
55 * Lc/Rc pair, Lrs/Rrs pair, Ts, Lsd/Rsd pair,
56 * Lw/Rw pair, Lvh/Rvh pair, Cvh, Reserved, LFE2
57 */
58enum CustomChannelMapLocation{
59 AC3_CHMAP_L= 1<<(15-0),
60 AC3_CHMAP_C= 1<<(15-1),
61 AC3_CHMAP_R= 1<<(15-2),
62 AC3_CHMAP_L_SUR= 1<<(15-3),
63 AC3_CHMAP_R_SUR = 1<<(15-4),
64 AC3_CHMAP_C_SUR= 1<<(15-7),
65 AC3_CHMAP_LFE = 1<<(15-15)
66};
