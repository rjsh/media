78/* butterfly operator */
79#define BF(a, b, c, s)\
80{\
81 tmp0 = val##a + val##b;\
82 tmp1 = val##a - val##b;\
83 val##a = tmp0;\
84 val##b = MULH3(tmp1, c, 1<<(s));\
85}
86
87#define BF0(a, b, c, s)\
88{\
89 tmp0 = tab[a] + tab[b];\
90 tmp1 = tab[a] - tab[b];\
91 val##a = tmp0;\
92 val##b = MULH3(tmp1, c, 1<<(s));\
93}
94
95#define BF1(a, b, c, d)\
96{\
97 BF(a, b, COS4_0, 1);\
98 BF(c, d,-COS4_0, 1);\
99 val##c += val##d;\
100}
101
102#define BF2(a, b, c, d)\
103{\
104 BF(a, b, COS4_0, 1);\
105 BF(c, d,-COS4_0, 1);\
106 val##c += val##d;\
107 val##a += val##c;\
108 val##c += val##b;\
109 val##b += val##d;\
110}
111
112#define ADD(a, b) val##a += val##b
