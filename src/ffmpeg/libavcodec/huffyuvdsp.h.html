19#ifndef AVCODEC_HUFFYUVDSP_H
20#define AVCODEC_HUFFYUVDSP_H
21
22#include <stdint.h>
23#include "config.h"
24
25#if HAVE_BIGENDIAN
26#define B 3
27#define G 2
28#define R 1
29#define A 0
30#else
31#define B 0
32#define G 1
33#define R 2
34#define A 3
35#endif
36
37typedef struct HuffYUVDSPContext {
38 void (*add_bytes)(uint8_t *dst /* align 16 */, uint8_t *src /* align 16 */,
39 intptr_t w);
40 void (*add_hfyu_median_pred)(uint8_t *dst, const uint8_t *top,
41 const uint8_t *diff, intptr_t w,
42 int *left, int *left_top);
43 int (*add_hfyu_left_pred)(uint8_t *dst, const uint8_t *src,
44 intptr_t w, int left);
45 void (*add_hfyu_left_pred_bgr32)(uint8_t *dst, const uint8_t *src,
46 intptr_t w, uint8_t *left);
47} HuffYUVDSPContext;
48
49void ff_huffyuvdsp_init(HuffYUVDSPContext *c);
50void ff_huffyuvdsp_init_ppc(HuffYUVDSPContext *c);
51void ff_huffyuvdsp_init_x86(HuffYUVDSPContext *c);
52
53#endif /* AVCODEC_HUFFYUVDSP_H */
