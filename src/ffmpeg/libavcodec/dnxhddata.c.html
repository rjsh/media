1139int ff_dnxhd_find_cid(AVCodecContext *avctx, int bit_depth)
1140{
1141 int i, j;
1142 int mbs = avctx->bit_rate / 1000000;
1143
1144 if (avctx->profile != FF_PROFILE_DNXHD)
1145 return dnxhd_find_hr_cid(avctx);
1146
1147 if (!mbs)
1148 return 0;
1149 for (i = 0; i < FF_ARRAY_ELEMS(ff_dnxhd_cid_table); i++) {
1150 const CIDEntry *cid = &ff_dnxhd_cid_table[i];
1151 int interlaced = cid->flags & DNXHD_INTERLACED ? 1 : 0;
1152 if (cid->width == avctx->width && cid->height == avctx->height &&
1153 interlaced == !!(avctx->flags & AV_CODEC_FLAG_INTERLACED_DCT) &&
1154 !(cid->flags & DNXHD_444) && cid->bit_depth == bit_depth) {
1155 if (avctx->strict_std_compliance > FF_COMPLIANCE_EXPERIMENTAL &&
1156 cid->flags & DNXHD_MBAFF) {
1157 av_log(avctx, AV_LOG_WARNING, "Profile selected is experimental\n");
1158 continue;
1159 }
1160 for (j = 0; j < FF_ARRAY_ELEMS(cid->bit_rates); j++) {
1161 if (cid->bit_rates[j] == mbs)
1162 return cid->cid;
1163 }
1164 }
1165 }
1166 return 0;
1167}
