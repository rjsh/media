19#ifndef AVCODEC_BLOCKDSP_H
20#define AVCODEC_BLOCKDSP_H
21
22#include <stdint.h>
23
24#include "avcodec.h"
25#include "version.h"
26
27/* add and put pixel (decoding)
28 * Block sizes for op_pixels_func are 8x4,8x8 16x8 16x16.
29 * h for op_pixels_func is limited to { width / 2, width },
30 * but never larger than 16 and never smaller than 4. */
31typedef void (*op_fill_func)(uint8_t *block /* align width (8 or 16) */,
32 uint8_t value, int line_size, int h);
33
34typedef struct BlockDSPContext {
35 void (*clear_block)(int16_t *block /* align 16 */);
36 void (*clear_blocks)(int16_t *blocks /* align 16 */);
37
38 op_fill_func fill_block_tab[2];
39} BlockDSPContext;
40
41void ff_blockdsp_init(BlockDSPContext *c, AVCodecContext *avctx);
42
43void ff_blockdsp_init_alpha(BlockDSPContext *c);
44void ff_blockdsp_init_arm(BlockDSPContext *c);
45void ff_blockdsp_init_ppc(BlockDSPContext *c);
46void ff_blockdsp_init_x86(BlockDSPContext *c, AVCodecContext *avctx);
47void ff_blockdsp_init_mips(BlockDSPContext *c);
48
49#endif /* AVCODEC_BLOCKDSP_H */
