19#include "libavutil/attributes.h"
20#include "ttadsp.h"
21#include "config.h"

23static void tta_filter_process_c(int32_t *qm, int32_t *dx, int32_t *dl,
24 int32_t *error, int32_t *in, int32_t shift,
25 int32_t round) {
26 if (*error < 0) {
27 qm[0] -= dx[0]; qm[1] -= dx[1]; qm[2] -= dx[2]; qm[3] -= dx[3];
28 qm[4] -= dx[4]; qm[5] -= dx[5]; qm[6] -= dx[6]; qm[7] -= dx[7];
29 } else if (*error > 0) {
30 qm[0] += dx[0]; qm[1] += dx[1]; qm[2] += dx[2]; qm[3] += dx[3];
31 qm[4] += dx[4]; qm[5] += dx[5]; qm[6] += dx[6]; qm[7] += dx[7];
32 }
33
34 round += dl[0] * qm[0] + dl[1] * qm[1] + dl[2] * qm[2] + dl[3] * qm[3] +
35 dl[4] * qm[4] + dl[5] * qm[5] + dl[6] * qm[6] + dl[7] * qm[7];
36
37 dx[0] = dx[1]; dx[1] = dx[2]; dx[2] = dx[3]; dx[3] = dx[4];
38 dl[0] = dl[1]; dl[1] = dl[2]; dl[2] = dl[3]; dl[3] = dl[4];
39
40 dx[4] = ((dl[4] >> 30) | 1);
41 dx[5] = ((dl[5] >> 30) | 2) & ~1;
42 dx[6] = ((dl[6] >> 30) | 2) & ~1;
43 dx[7] = ((dl[7] >> 30) | 4) & ~3;
44
45 *error = *in;
46 *in += (round >> shift);
47
48 dl[4] = -dl[5]; dl[5] = -dl[6];
49 dl[6] = *in - dl[7]; dl[7] = *in;
50 dl[5] += dl[6]; dl[4] += dl[5];
51}
52
53av_cold void ff_ttadsp_init(TTADSPContext *c)
54{
55 c->filter_process = tta_filter_process_c;
56
57 if (ARCH_X86)
58 ff_ttadsp_init_x86(c);
59}
