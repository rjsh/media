23#ifndef AVCODEC_MOTIONPIXELS_TABLEGEN_H
24#define AVCODEC_MOTIONPIXELS_TABLEGEN_H
25
26#include <stdint.h>
27#include "libavutil/attributes.h"
28
29typedef struct YuvPixel {
30 int8_t y, v, u;
31} YuvPixel;
32
33static int mp_yuv_to_rgb(int y, int v, int u, int clip_rgb) {
34 const uint8_t *cm = ff_crop_tab + MAX_NEG_CROP;
35 int r, g, b;
36
37 r = (1000 * y + 701 * v) / 1000;
38 g = (1000 * y - 357 * v - 172 * u) / 1000;
39 b = (1000 * y + 886 * u) / 1000;
40 if (clip_rgb)
41 return ((cm[r * 8] & 0xF8) << 7) | ((cm[g * 8] & 0xF8) << 2) | (cm[b * 8] >> 3);
42 if ((unsigned)r < 32 && (unsigned)g < 32 && (unsigned)b < 32)
43 return (r << 10) | (g << 5) | b;
44 return 1 << 15;
45}

47#if CONFIG_HARDCODED_TABLES
48#define motionpixels_tableinit()
49#include "libavcodec/motionpixels_tables.h"
50#else
51static YuvPixel mp_rgb_yuv_table[1 << 15];
52
53static av_cold void mp_set_zero_yuv(YuvPixel *p)
54{
55 int i, j;
56
57 for (i = 0; i < 31; ++i) {
58 for (j = 31; j > i; --j)
59 if (!(p[j].u | p[j].v | p[j].y))
60 p[j] = p[j - 1];
61 for (j = 0; j < 31 - i; ++j)
62 if (!(p[j].u | p[j].v | p[j].y))
63 p[j] = p[j + 1];
64 }
65}
66
67static av_cold void mp_build_rgb_yuv_table(YuvPixel *p)
68{
69 int y, v, u, i;
70
71 for (y = 0; y <= 31; ++y)
72 for (v = -31; v <= 31; ++v)
73 for (u = -31; u <= 31; ++u) {
74 i = mp_yuv_to_rgb(y, v, u, 0);
75 if (i < (1 << 15) && !(p[i].u | p[i].v | p[i].y)) {
76 p[i].y = y;
77 p[i].v = v;
78 p[i].u = u;
79 }
80 }
81 for (i = 0; i < 1024; ++i)
82 mp_set_zero_yuv(p + i * 32);
83}
84
85static av_cold void motionpixels_tableinit(void)
86{
87 if (!mp_rgb_yuv_table[0].u)
88 mp_build_rgb_yuv_table(mp_rgb_yuv_table);
89}
90#endif /* CONFIG_HARDCODED_TABLES */
