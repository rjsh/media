93uint16_t ff_mqc_qe [2 * 47];
94uint8_t ff_mqc_nlps[2 * 47];
95uint8_t ff_mqc_nmps[2 * 47];
96
97void av_cold ff_mqc_init_context_tables(void)
98{
99 int i;
100 for (i = 0; i < 47; i++) {
101 ff_mqc_qe[2 * i] =
102 ff_mqc_qe[2 * i + 1] = cx_states[i].qe;
103
104 ff_mqc_nlps[2 * i] = 2 * cx_states[i].nlps + cx_states[i].sw;
105 ff_mqc_nlps[2 * i + 1] = 2 * cx_states[i].nlps + 1 - cx_states[i].sw;
106 ff_mqc_nmps[2 * i] = 2 * cx_states[i].nmps;
107 ff_mqc_nmps[2 * i + 1] = 2 * cx_states[i].nmps + 1;
108 }
109}
110
111void ff_mqc_init_contexts(MqcState *mqc)
112{
113 memset(mqc->cx_states, 0, sizeof(mqc->cx_states));
114 mqc->cx_states[MQC_CX_UNI] = 2 * 46;
115 mqc->cx_states[MQC_CX_RL] = 2 * 3;
116 mqc->cx_states[0] = 2 * 4;
117}
