28static av_cold int decode_init(AVCodecContext *avctx) {
29 avctx->pix_fmt = AV_PIX_FMT_PAL8;
30 return 0;
31}
32
33static const uint8_t tc_offsets[9] = { 0, 1, 3, 4, 6, 7, 9, 10, 11 };
34static const uint8_t tc_muls[9] = { 10, 6, 10, 6, 10, 10, 10, 10, 1 };
35
36static int64_t parse_timecode(const uint8_t *buf, int64_t packet_time) {
37 int i;
38 int64_t ms = 0;
39 if (buf[2] != ':' || buf[5] != ':' || buf[8] != '.')
40 return AV_NOPTS_VALUE;
41 for (i = 0; i < sizeof(tc_offsets); i++) {
42 uint8_t c = buf[tc_offsets[i]] - '0';
43 if (c > 9) return AV_NOPTS_VALUE;
44 ms = (ms + c) * tc_muls[i];
45 }
46 return ms - packet_time;
47}
48
49static int decode_frame(AVCodecContext *avctx, void *data, int *data_size,
50 AVPacket *avpkt) {
51 const uint8_t *buf = avpkt->data;
52 int buf_size = avpkt->size;
53 AVSubtitle *sub = data;
54 const uint8_t *buf_end = buf + buf_size;
55 uint8_t *bitmap;
56 int w, h, x, y, i;
57 int64_t packet_time = 0;
58 GetBitContext gb;
59 int has_alpha = avctx->codec_tag == MKTAG('D','X','S','A');
60
61 // check that at least header fits
62 if (buf_size < 27 + 7 * 2 + 4 * (3 + has_alpha)) {
63 av_log(avctx, AV_LOG_ERROR, "coded frame size %d too small\n", buf_size);
64 return -1;
65 }
66
67 // read start and end time
68 if (buf[0] != '[' || buf[13] != '-' || buf[26] != ']') {
69 av_log(avctx, AV_LOG_ERROR, "invalid time code\n");
70 return -1;
71 }
72 if (avpkt->pts != AV_NOPTS_VALUE)
73 packet_time = av_rescale_q(avpkt->pts, AV_TIME_BASE_Q, (AVRational){1, 1000});
74 sub->start_display_time = parse_timecode(buf + 1, packet_time);
75 sub->end_display_time = parse_timecode(buf + 14, packet_time);
76 buf += 27;
77
78 // read header
79 w = bytestream_get_le16(&buf);
80 h = bytestream_get_le16(&buf);
81 if (av_image_check_size(w, h, 0, avctx) < 0)
82 return -1;
83 x = bytestream_get_le16(&buf);
84 y = bytestream_get_le16(&buf);
85 // skip bottom right position, it gives no new information
86 bytestream_get_le16(&buf);
87 bytestream_get_le16(&buf);
88 // The following value is supposed to indicate the start offset
89 // (relative to the palette) of the data for the second field,
90 // however there are files in which it has a bogus value and thus
91 // we just ignore it
92 bytestream_get_le16(&buf);
93
94 // allocate sub and set values
95 sub->rects = av_mallocz(sizeof(*sub->rects));
96 if (!sub->rects)
97 return AVERROR(ENOMEM);
98
99 sub->rects[0] = av_mallocz(sizeof(*sub->rects[0]));
100 if (!sub->rects[0]) {
101 av_freep(&sub->rects);
102 return AVERROR(ENOMEM);
103 }
104 sub->rects[0]->x = x; sub->rects[0]->y = y;
105 sub->rects[0]->w = w; sub->rects[0]->h = h;
106 sub->rects[0]->type = SUBTITLE_BITMAP;
107 sub->rects[0]->linesize[0] = w;
108 sub->rects[0]->data[0] = av_malloc(w * h);
109 sub->rects[0]->nb_colors = 4;
110 sub->rects[0]->data[1] = av_mallocz(AVPALETTE_SIZE);
111 if (!sub->rects[0]->data[0] || !sub->rects[0]->data[1]) {
112 av_freep(&sub->rects[0]->data[1]);
113 av_freep(&sub->rects[0]->data[0]);
114 av_freep(&sub->rects[0]);
115 av_freep(&sub->rects);
116 return AVERROR(ENOMEM);
117 }
118 sub->num_rects = 1;
119
120 // read palette
121 for (i = 0; i < sub->rects[0]->nb_colors; i++)
122 ((uint32_t*)sub->rects[0]->data[1])[i] = bytestream_get_be24(&buf);
123
124 if (!has_alpha) {
125 // make all except background (first entry) non-transparent
126 for (i = 1; i < sub->rects[0]->nb_colors; i++)
127 ((uint32_t *)sub->rects[0]->data[1])[i] |= 0xff000000;
128 } else {
129 for (i = 0; i < sub->rects[0]->nb_colors; i++)
130 ((uint32_t *)sub->rects[0]->data[1])[i] |= *buf++ << 24;
131 }
132
133#if FF_API_AVPICTURE
134FF_DISABLE_DEPRECATION_WARNINGS
135{
136 AVSubtitleRect *rect;
137 int j;
138 rect = sub->rects[0];
139 for (j = 0; j < 4; j++) {
140 rect->pict.data[j] = rect->data[j];
141 rect->pict.linesize[j] = rect->linesize[j];
142 }
143}
144FF_ENABLE_DEPRECATION_WARNINGS
145#endif
146
147 // process RLE-compressed data
148 init_get_bits(&gb, buf, (buf_end - buf) * 8);
149 bitmap = sub->rects[0]->data[0];
150 for (y = 0; y < h; y++) {
151 // interlaced: do odd lines
152 if (y == (h + 1) / 2) bitmap = sub->rects[0]->data[0] + w;
153 for (x = 0; x < w; ) {
154 int log2 = ff_log2_tab[show_bits(&gb, 8)];
155 int run = get_bits(&gb, 14 - 4 * (log2 >> 1));
156 int color = get_bits(&gb, 2);
157 run = FFMIN(run, w - x);
158 // run length 0 means till end of row
159 if (!run) run = w - x;
160 memset(bitmap, color, run);
161 bitmap += run;
162 x += run;
163 }
164 // interlaced, skip every second line
165 bitmap += w;
166 align_get_bits(&gb);
167 }
168 *data_size = 1;
169 return buf_size;
170}

175 .type = AVMEDIA_TYPE_SUBTITLE,

177 .init = decode_init,
178 .decode = decode_frame,
