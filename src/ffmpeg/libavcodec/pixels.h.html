19#ifndef AVCODEC_PIXELS_H
20#define AVCODEC_PIXELS_H
21
22#include <stddef.h>
23#include <stdint.h>
24
25/* pixel operations */
26#define CALL_2X_PIXELS_MACRO(STATIC, a, b, n) \
27STATIC void a(uint8_t *block, const uint8_t *pixels, \
28 ptrdiff_t line_size, int h) \
29{ \
30 b(block, pixels, line_size, h); \
31 b(block + n, pixels + n, line_size, h); \
32}
33
34#define CALL_2X_PIXELS(a, b, n) CALL_2X_PIXELS_MACRO(static, a, b, n)
35#define CALL_2X_PIXELS_EXPORT(a, b, n) CALL_2X_PIXELS_MACRO(, a, b, n)
36
37#endif /* AVCODEC_PIXELS_H */
