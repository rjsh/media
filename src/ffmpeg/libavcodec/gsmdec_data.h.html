22#ifndef AVCODEC_GSMDEC_DATA_H
23#define AVCODEC_GSMDEC_DATA_H

25#include <stdint.h>
26#include "avcodec.h"
27
28typedef struct GSMContext {
29 // Contains first 120 elements from the previous frame
30 // (used by long_term_synth according to the "lag"),
31 // then in the following 160 elements the current
32 // frame is constructed.
33 int16_t ref_buf[280];
34 int v[9];
35 int lar[2][8];
36 int lar_idx;
37 int msr;
38} GSMContext;

41extern const uint8_t ff_gsm_requant_tab[4][8];
42extern const int16_t ff_gsm_dequant_tab[64][8];
43
44extern const int* const ff_gsm_apcm_bits[][4];
45
46#endif /* AVCODEC_GSMDEC_DATA_H */
