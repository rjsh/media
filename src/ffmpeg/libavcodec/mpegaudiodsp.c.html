21#include "config.h"
22#include "libavutil/attributes.h"
23#include "mpegaudiodsp.h"
24#include "dct.h"
25#include "dct32.h"
26
27av_cold void ff_mpadsp_init(MPADSPContext *s)
28{
29 DCTContext dct;
30
31 ff_dct_init(&dct, 5, DCT_II);
32 ff_init_mpadsp_tabs_float();
33 ff_init_mpadsp_tabs_fixed();
34
35 s->apply_window_float = ff_mpadsp_apply_window_float;
36 s->apply_window_fixed = ff_mpadsp_apply_window_fixed;
37
38 s->dct32_float = dct.dct32;
39 s->dct32_fixed = ff_dct32_fixed;
40
41 s->imdct36_blocks_float = ff_imdct36_blocks_float;
42 s->imdct36_blocks_fixed = ff_imdct36_blocks_fixed;
43
44 if (ARCH_AARCH64) ff_mpadsp_init_aarch64(s);
45 if (ARCH_ARM) ff_mpadsp_init_arm(s);
46 if (ARCH_PPC) ff_mpadsp_init_ppc(s);
47 if (ARCH_X86) ff_mpadsp_init_x86(s);
48 if (HAVE_MIPSFPU) ff_mpadsp_init_mipsfpu(s);
49 if (HAVE_MIPSDSP) ff_mpadsp_init_mipsdsp(s);
50}
