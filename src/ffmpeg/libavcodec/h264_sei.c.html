37static const uint8_t sei_num_clock_ts_table[9] = {
38 1, 1, 1, 2, 2, 3, 3, 2, 3

41void ff_h264_sei_uninit(H264SEIContext *h)
42{
43 h->recovery_point.recovery_frame_cnt = -1;
44
45 h->picture_timing.dpb_output_delay = 0;
46 h->picture_timing.cpb_removal_delay = -1;
47
48 h->buffering_period.present = 0;
49 h->frame_packing.present = 0;
50 h->display_orientation.present = 0;
51 h->afd.present = 0;
52
53 h->a53_caption.a53_caption_size = 0;
54 av_freep(&h->a53_caption.a53_caption);
55}

57static int decode_picture_timing(H264SEIPictureTiming *h, GetBitContext *gb,
58 const H264ParamSets *ps, void *logctx)
59{
60 int i;
61 const SPS *sps = ps->sps;
62
63 for (i = 0; i<MAX_SPS_COUNT; i++)
64 if ((!sps || !sps->log2_max_frame_num) && ps->sps_list[i])
65 sps = (const SPS *)ps->sps_list[i]->data;
66
67 if (!sps) {
68 av_log(logctx, AV_LOG_ERROR, "SPS unavailable in decode_picture_timing\n");
69 return AVERROR_PS_NOT_FOUND;
70 }
71
72 if (sps->nal_hrd_parameters_present_flag ||
73 sps->vcl_hrd_parameters_present_flag) {
74 h->cpb_removal_delay = get_bits_long(gb, sps->cpb_removal_delay_length);
75 h->dpb_output_delay = get_bits_long(gb, sps->dpb_output_delay_length);
76 }
77 if (sps->pic_struct_present_flag) {
78 unsigned int i, num_clock_ts;
79
80 h->pic_struct = get_bits(gb, 4);
81 h->ct_type = 0;
82
83 if (h->pic_struct > SEI_PIC_STRUCT_FRAME_TRIPLING)
84 return AVERROR_INVALIDDATA;
85
86 num_clock_ts = sei_num_clock_ts_table[h->pic_struct];
87
88 for (i = 0; i < num_clock_ts; i++) {
89 if (get_bits(gb, 1)) { /* clock_timestamp_flag */
90 unsigned int full_timestamp_flag;
91
92 h->ct_type |= 1 << get_bits(gb, 2);
93 skip_bits(gb, 1); /* nuit_field_based_flag */
94 skip_bits(gb, 5); /* counting_type */
95 full_timestamp_flag = get_bits(gb, 1);
96 skip_bits(gb, 1); /* discontinuity_flag */
97 skip_bits(gb, 1); /* cnt_dropped_flag */
98 skip_bits(gb, 8); /* n_frames */
99 if (full_timestamp_flag) {
100 skip_bits(gb, 6); /* seconds_value 0..59 */
101 skip_bits(gb, 6); /* minutes_value 0..59 */
102 skip_bits(gb, 5); /* hours_value 0..23 */
103 } else {
104 if (get_bits(gb, 1)) { /* seconds_flag */
105 skip_bits(gb, 6); /* seconds_value range 0..59 */
106 if (get_bits(gb, 1)) { /* minutes_flag */
107 skip_bits(gb, 6); /* minutes_value 0..59 */
108 if (get_bits(gb, 1)) /* hours_flag */
109 skip_bits(gb, 5); /* hours_value 0..23 */
110 }
111 }
112 }
113 if (sps->time_offset_length > 0)
114 skip_bits(gb,
115 sps->time_offset_length); /* time_offset */
116 }
117 }
118
119 av_log(logctx, AV_LOG_DEBUG, "ct_type:%X pic_struct:%d\n",
120 h->ct_type, h->pic_struct);
121 }
122 return 0;
123}

125static int decode_registered_user_data_afd(H264SEIAFD *h, GetBitContext *gb, int size)
126{
127 int flag;
128
129 if (size-- < 1)
130 return AVERROR_INVALIDDATA;
131 skip_bits(gb, 1); // 0
132 flag = get_bits(gb, 1); // active_format_flag
133 skip_bits(gb, 6); // reserved
134
135 if (flag) {
136 if (size-- < 1)
137 return AVERROR_INVALIDDATA;
138 skip_bits(gb, 4); // reserved
139 h->active_format_description = get_bits(gb, 4);
140 h->present = 1;
141 }
142
143 return 0;
144}

146static int decode_registered_user_data_closed_caption(H264SEIA53Caption *h,
147 GetBitContext *gb, void *logctx,
148 int size)
149{
150 int flag;
151 int user_data_type_code;
152 int cc_count;
153
154 if (size < 3)
155 return AVERROR(EINVAL);
156
157 user_data_type_code = get_bits(gb, 8);
158 if (user_data_type_code == 0x3) {
159 skip_bits(gb, 1); // reserved
160
161 flag = get_bits(gb, 1); // process_cc_data_flag
162 if (flag) {
163 skip_bits(gb, 1); // zero bit
164 cc_count = get_bits(gb, 5);
165 skip_bits(gb, 8); // reserved
166 size -= 2;
167
168 if (cc_count && size >= cc_count * 3) {
169 const uint64_t new_size = (h->a53_caption_size + cc_count
170 * UINT64_C(3));
171 int i, ret;
172
173 if (new_size > INT_MAX)
174 return AVERROR(EINVAL);
175
176 /* Allow merging of the cc data from two fields. */
177 ret = av_reallocp(&h->a53_caption, new_size);
178 if (ret < 0)
179 return ret;
180
181 for (i = 0; i < cc_count; i++) {
182 h->a53_caption[h->a53_caption_size++] = get_bits(gb, 8);
183 h->a53_caption[h->a53_caption_size++] = get_bits(gb, 8);
184 h->a53_caption[h->a53_caption_size++] = get_bits(gb, 8);
185 }
186
187 skip_bits(gb, 8); // marker_bits
188 }
189 }
190 } else {
191 int i;
192 for (i = 0; i < size - 1; i++)
193 skip_bits(gb, 8);
194 }
195
196 return 0;
197}

199static int decode_registered_user_data(H264SEIContext *h, GetBitContext *gb,
200 void *logctx, int size)
201{
202 uint32_t country_code;
203 uint32_t user_identifier;
204
205 if (size < 7)
206 return AVERROR_INVALIDDATA;
207 size -= 7;
208
209 country_code = get_bits(gb, 8); // itu_t_t35_country_code
210 if (country_code == 0xFF) {
211 skip_bits(gb, 8); // itu_t_t35_country_code_extension_byte
212 size--;
213 }
214
215 /* itu_t_t35_payload_byte follows */
216 skip_bits(gb, 8); // terminal provider code
217 skip_bits(gb, 8); // terminal provider oriented code
218 user_identifier = get_bits_long(gb, 32);
219
220 switch (user_identifier) {
221 case MKBETAG('D', 'T', 'G', '1'): // afd_data
222 return decode_registered_user_data_afd(&h->afd, gb, size);
223 case MKBETAG('G', 'A', '9', '4'): // closed captions
224 return decode_registered_user_data_closed_caption(&h->a53_caption, gb,
225 logctx, size);
226 default:
227 skip_bits(gb, size * 8);
228 break;
229 }
230
231 return 0;
232}

234static int decode_unregistered_user_data(H264SEIUnregistered *h, GetBitContext *gb,
235 void *logctx, int size)
236{
237 uint8_t *user_data;
238 int e, build, i;
239
240 if (size < 16 || size >= INT_MAX - 16)
241 return AVERROR_INVALIDDATA;
242
243 user_data = av_malloc(16 + size + 1);
244 if (!user_data)
245 return AVERROR(ENOMEM);
246
247 for (i = 0; i < size + 16; i++)
248 user_data[i] = get_bits(gb, 8);
249
250 user_data[i] = 0;
251 e = sscanf(user_data + 16, "x264 - core %d", &build);
252 if (e == 1 && build > 0)
253 h->x264_build = build;
254 if (e == 1 && build == 1 && !strncmp(user_data+16, "x264 - core 0000", 16))
255 h->x264_build = 67;
256
257 if (strlen(user_data + 16) > 0)
258 av_log(logctx, AV_LOG_DEBUG, "user data:\"%s\"\n", user_data + 16);
259
260 av_free(user_data);
261 return 0;
262}

264static int decode_recovery_point(H264SEIRecoveryPoint *h, GetBitContext *gb)
265{
266 h->recovery_frame_cnt = get_ue_golomb_long(gb);
267
268 /* 1b exact_match_flag,
269 * 1b broken_link_flag,
270 * 2b changing_slice_group_idc */
271 skip_bits(gb, 4);
272
273 return 0;
274}

276static int decode_buffering_period(H264SEIBufferingPeriod *h, GetBitContext *gb,
277 const H264ParamSets *ps, void *logctx)
278{
279 unsigned int sps_id;
280 int sched_sel_idx;
281 const SPS *sps;
282
283 sps_id = get_ue_golomb_31(gb);
284 if (sps_id > 31 || !ps->sps_list[sps_id]) {
285 av_log(logctx, AV_LOG_ERROR,
286 "non-existing SPS %d referenced in buffering period\n", sps_id);
287 return sps_id > 31 ? AVERROR_INVALIDDATA : AVERROR_PS_NOT_FOUND;
288 }
289 sps = (const SPS*)ps->sps_list[sps_id]->data;
290
291 // NOTE: This is really so duplicated in the standard... See H.264, D.1.1
292 if (sps->nal_hrd_parameters_present_flag) {
293 for (sched_sel_idx = 0; sched_sel_idx < sps->cpb_cnt; sched_sel_idx++) {
294 h->initial_cpb_removal_delay[sched_sel_idx] =
295 get_bits_long(gb, sps->initial_cpb_removal_delay_length);
296 // initial_cpb_removal_delay_offset
297 skip_bits(gb, sps->initial_cpb_removal_delay_length);
298 }
299 }
300 if (sps->vcl_hrd_parameters_present_flag) {
301 for (sched_sel_idx = 0; sched_sel_idx < sps->cpb_cnt; sched_sel_idx++) {
302 h->initial_cpb_removal_delay[sched_sel_idx] =
303 get_bits_long(gb, sps->initial_cpb_removal_delay_length);
304 // initial_cpb_removal_delay_offset
305 skip_bits(gb, sps->initial_cpb_removal_delay_length);
306 }
307 }
308
309 h->present = 1;
310 return 0;
311}

313static int decode_frame_packing_arrangement(H264SEIFramePacking *h,
314 GetBitContext *gb)
315{
316 h->frame_packing_arrangement_id = get_ue_golomb_long(gb);
317 h->frame_packing_arrangement_cancel_flag = get_bits1(gb);
318 h->present = !h->frame_packing_arrangement_cancel_flag;
319
320 if (h->present) {
321 h->frame_packing_arrangement_type = get_bits(gb, 7);
322 h->quincunx_sampling_flag = get_bits1(gb);
323 h->content_interpretation_type = get_bits(gb, 6);
324
325 // the following skips: spatial_flipping_flag, frame0_flipped_flag,
326 // field_views_flag, current_frame_is_frame0_flag,
327 // frame0_self_contained_flag, frame1_self_contained_flag
328 skip_bits(gb, 6);
329
330 if (!h->quincunx_sampling_flag && h->frame_packing_arrangement_type != 5)
331 skip_bits(gb, 16); // frame[01]_grid_position_[xy]
332 skip_bits(gb, 8); // frame_packing_arrangement_reserved_byte
333 h->frame_packing_arrangement_repetition_period = get_ue_golomb_long(gb);
334 }
335 skip_bits1(gb); // frame_packing_arrangement_extension_flag
336
337 return 0;
338}

340static int decode_display_orientation(H264SEIDisplayOrientation *h,
341 GetBitContext *gb)
342{
343 h->present = !get_bits1(gb);
344
345 if (h->present) {
346 h->hflip = get_bits1(gb); // hor_flip
347 h->vflip = get_bits1(gb); // ver_flip
348
349 h->anticlockwise_rotation = get_bits(gb, 16);
350 get_ue_golomb_long(gb); // display_orientation_repetition_period
351 skip_bits1(gb); // display_orientation_extension_flag
352 }
353
354 return 0;
355}

357static int decode_green_metadata(H264SEIGreenMetaData *h, GetBitContext *gb)
358{
359 h->green_metadata_type = get_bits(gb, 8);
360
361 if (h->green_metadata_type == 0) {
362 h->period_type = get_bits(gb, 8);
363
364 if (h->period_type == 2)
365 h->num_seconds = get_bits(gb, 16);
366 else if (h->period_type == 3)
367 h->num_pictures = get_bits(gb, 16);
368
369 h->percent_non_zero_macroblocks = get_bits(gb, 8);
370 h->percent_intra_coded_macroblocks = get_bits(gb, 8);
371 h->percent_six_tap_filtering = get_bits(gb, 8);
372 h->percent_alpha_point_deblocking_instance = get_bits(gb, 8);
373
374 } else if (h->green_metadata_type == 1) {
375 h->xsd_metric_type = get_bits(gb, 8);
376 h->xsd_metric_value = get_bits(gb, 16);
377 }
378
379 return 0;
380}

382int ff_h264_sei_decode(H264SEIContext *h, GetBitContext *gb,
383 const H264ParamSets *ps, void *logctx)
384{
385 int master_ret = 0;
386
387 while (get_bits_left(gb) > 16 && show_bits(gb, 16)) {
388 int type = 0;
389 unsigned size = 0;
390 unsigned next;
391 int ret = 0;
392
393 do {
394 if (get_bits_left(gb) < 8)
395 return AVERROR_INVALIDDATA;
396 type += show_bits(gb, 8);
397 } while (get_bits(gb, 8) == 255);
398
399 do {
400 if (get_bits_left(gb) < 8)
401 return AVERROR_INVALIDDATA;
402 size += show_bits(gb, 8);
403 } while (get_bits(gb, 8) == 255);
404
405 if (size > get_bits_left(gb) / 8) {
406 av_log(logctx, AV_LOG_ERROR, "SEI type %d size %d truncated at %d\n",
407 type, 8*size, get_bits_left(gb));
408 return AVERROR_INVALIDDATA;
409 }
410 next = get_bits_count(gb) + 8 * size;
411
412 switch (type) {
413 case SEI_TYPE_PIC_TIMING: // Picture timing SEI
414 ret = decode_picture_timing(&h->picture_timing, gb, ps, logctx);
415 break;
416 case SEI_TYPE_USER_DATA_REGISTERED:
417 ret = decode_registered_user_data(h, gb, logctx, size);
418 break;
419 case SEI_TYPE_USER_DATA_UNREGISTERED:
420 ret = decode_unregistered_user_data(&h->unregistered, gb, logctx, size);
421 break;
422 case SEI_TYPE_RECOVERY_POINT:
423 ret = decode_recovery_point(&h->recovery_point, gb);
424 break;
425 case SEI_TYPE_BUFFERING_PERIOD:
426 ret = decode_buffering_period(&h->buffering_period, gb, ps, logctx);
427 break;
428 case SEI_TYPE_FRAME_PACKING:
429 ret = decode_frame_packing_arrangement(&h->frame_packing, gb);
430 break;
431 case SEI_TYPE_DISPLAY_ORIENTATION:
432 ret = decode_display_orientation(&h->display_orientation, gb);
433 break;
434 case SEI_TYPE_GREEN_METADATA:
435 ret = decode_green_metadata(&h->green_metadata, gb);
436 break;
437 default:
438 av_log(logctx, AV_LOG_DEBUG, "unknown SEI type %d\n", type);
439 }
440 if (ret < 0 && ret != AVERROR_PS_NOT_FOUND)
441 return ret;
442 if (ret < 0)
443 master_ret = ret;
444
445 skip_bits_long(gb, next - get_bits_count(gb));
446
447 // FIXME check bits here
448 align_get_bits(gb);
449 }
450
451 return master_ret;
452}

454const char *ff_h264_sei_stereo_mode(const H264SEIFramePacking *h)
455{
456 if (h->frame_packing_arrangement_cancel_flag == 0) {
457 switch (h->frame_packing_arrangement_type) {
458 case SEI_FPA_TYPE_CHECKERBOARD:
459 if (h->content_interpretation_type == 2)
460 return "checkerboard_rl";
461 else
462 return "checkerboard_lr";
463 case SEI_FPA_TYPE_INTERLEAVE_COLUMN:
464 if (h->content_interpretation_type == 2)
465 return "col_interleaved_rl";
466 else
467 return "col_interleaved_lr";
468 case SEI_FPA_TYPE_INTERLEAVE_ROW:
469 if (h->content_interpretation_type == 2)
470 return "row_interleaved_rl";
471 else
472 return "row_interleaved_lr";
473 case SEI_FPA_TYPE_SIDE_BY_SIDE:
474 if (h->content_interpretation_type == 2)
475 return "right_left";
476 else
477 return "left_right";
478 case SEI_FPA_TYPE_TOP_BOTTOM:
479 if (h->content_interpretation_type == 2)
480 return "bottom_top";
481 else
482 return "top_bottom";
483 case SEI_FPA_TYPE_INTERLEAVE_TEMPORAL:
484 if (h->content_interpretation_type == 2)
485 return "block_rl";
486 else
487 return "block_lr";
488 case SEI_FPA_TYPE_2D:
489 default:
490 return "mono";
491 }
492 } else if (h->frame_packing_arrangement_cancel_flag == 1) {
493 return "mono";
494 } else {
495 return NULL;
496 }
