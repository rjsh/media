26static inline void copy_block2(uint8_t *dst, const uint8_t *src, int dstStride, int srcStride, int h)
27{
28 int i;
29 for(i=0; i<h; i++)
30 {
31 AV_COPY16U(dst, src);
32 dst+=dstStride;
33 src+=srcStride;
34 }
35}
