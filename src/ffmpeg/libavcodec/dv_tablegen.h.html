23#ifndef AVCODEC_DV_TABLEGEN_H
24#define AVCODEC_DV_TABLEGEN_H
25
26#include <stdint.h>
27#include "libavutil/attributes.h"
28
29#include "dvdata.h"
30
31#if CONFIG_SMALL
32#define DV_VLC_MAP_RUN_SIZE 15
33#define DV_VLC_MAP_LEV_SIZE 23
34#else
35#define DV_VLC_MAP_RUN_SIZE 64
36#define DV_VLC_MAP_LEV_SIZE 512 // FIXME sign was removed so this should be /2 but needs check
37#endif
38
39/* VLC encoding lookup table */
40typedef struct dv_vlc_pair {
41 uint32_t vlc;
42 uint32_t size;
43} dv_vlc_pair;
44
45#if CONFIG_HARDCODED_TABLES
46#define dv_vlc_map_tableinit()
47#include "libavcodec/dv_tables.h"
48#else
49static struct dv_vlc_pair dv_vlc_map[DV_VLC_MAP_RUN_SIZE][DV_VLC_MAP_LEV_SIZE];
50
51static av_cold void dv_vlc_map_tableinit(void)
52{
53 int i, j;
54 for (i = 0; i < NB_DV_VLC - 1; i++) {
55 if (ff_dv_vlc_run[i] >= DV_VLC_MAP_RUN_SIZE)
56 continue;
57#if CONFIG_SMALL
58 if (ff_dv_vlc_level[i] >= DV_VLC_MAP_LEV_SIZE)
59 continue;
60#endif
61
62 if (dv_vlc_map[ff_dv_vlc_run[i]][ff_dv_vlc_level[i]].size != 0)
63 continue;
64
65 dv_vlc_map[ff_dv_vlc_run[i]][ff_dv_vlc_level[i]].vlc =
66 ff_dv_vlc_bits[i] << (!!ff_dv_vlc_level[i]);
67 dv_vlc_map[ff_dv_vlc_run[i]][ff_dv_vlc_level[i]].size =
68 ff_dv_vlc_len[i] + (!!ff_dv_vlc_level[i]);
69 }
70 for (i = 0; i < DV_VLC_MAP_RUN_SIZE; i++) {
71#if CONFIG_SMALL
72 for (j = 1; j < DV_VLC_MAP_LEV_SIZE; j++) {
73 if (dv_vlc_map[i][j].size == 0) {
74 dv_vlc_map[i][j].vlc = dv_vlc_map[0][j].vlc |
75 (dv_vlc_map[i - 1][0].vlc <<
76 dv_vlc_map[0][j].size);
77 dv_vlc_map[i][j].size = dv_vlc_map[i - 1][0].size +
78 dv_vlc_map[0][j].size;
79 }
80 }
81#else
82 for (j = 1; j < DV_VLC_MAP_LEV_SIZE / 2; j++) {
83 if (dv_vlc_map[i][j].size == 0) {
84 dv_vlc_map[i][j].vlc = dv_vlc_map[0][j].vlc |
85 (dv_vlc_map[i - 1][0].vlc <<
86 dv_vlc_map[0][j].size);
87 dv_vlc_map[i][j].size = dv_vlc_map[i - 1][0].size +
88 dv_vlc_map[0][j].size;
89 }
90 dv_vlc_map[i][((uint16_t) (-j)) & 0x1ff].vlc = dv_vlc_map[i][j].vlc | 1;
91 dv_vlc_map[i][((uint16_t) (-j)) & 0x1ff].size = dv_vlc_map[i][j].size;
92 }
93#endif
94 }
95}
96#endif /* CONFIG_HARDCODED_TABLES */

98#endif /* AVCODEC_DV_TABLEGEN_H */
