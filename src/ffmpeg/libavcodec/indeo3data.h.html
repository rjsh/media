236/**
237 * Pack two delta values (a,b) into one 16-bit word
238 * according with endianness of the host machine.
239 */
240#if HAVE_BIGENDIAN
241#define PD(a,b) (((a) * (1 << 8)) + (b))
242#else
243#define PD(a,b) (((b) * (1 << 8)) + (a))
244#endif
245
246/**
247 * Expand a pair of delta values (a,b)
248 * into two/four delta entries.
249 */
250#define E2(a, b) PD(a, b), PD(-(a), -(b))
251#define E4(a, b) PD(a, b), PD(-(a), -(b)), PD(b, a), PD(-(b), -(a))
