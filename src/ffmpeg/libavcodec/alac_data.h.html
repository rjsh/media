24#include <stdint.h>

26enum AlacRawDataBlockType {
27 /* At the moment, only SCE, CPE, LFE, and END are recognized. */
28 TYPE_SCE,
29 TYPE_CPE,
30 TYPE_CCE,
31 TYPE_LFE,
32 TYPE_DSE,
33 TYPE_PCE,
34 TYPE_FIL,
35 TYPE_END
36};

40extern const uint8_t ff_alac_channel_layout_offsets[ALAC_MAX_CHANNELS][ALAC_MAX_CHANNELS];

42extern const uint64_t ff_alac_channel_layouts[ALAC_MAX_CHANNELS + 1];
43
44extern const enum AlacRawDataBlockType ff_alac_channel_elements[ALAC_MAX_CHANNELS][5];
