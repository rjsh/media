34typedef struct TargaContext {
35 AVClass *class;
36
37 int rle;
38} TargaContext;

40/**
41 * RLE compress the image, with maximum size of out_size
42 * @param outbuf Output buffer
43 * @param out_size Maximum output size
44 * @param pic Image to compress
45 * @param bpp Bytes per pixel
46 * @param w Image width
47 * @param h Image height
48 * @return Size of output in bytes, or -1 if larger than out_size
49 */
50static int targa_encode_rle(uint8_t *outbuf, int out_size, const AVFrame *pic,
51 int bpp, int w, int h)
52{
53 int y,ret;
54 uint8_t *out;
55
56 out = outbuf;
57
58 for(y = 0; y < h; y ++) {
59 ret = ff_rle_encode(out, out_size, pic->data[0] + pic->linesize[0] * y, bpp, w, 0x7f, 0, -1, 0);
60 if(ret == -1){
61 return -1;
62 }
63 out+= ret;
64 out_size -= ret;
65 }
66
67 return out - outbuf;
68}

70static int targa_encode_normal(uint8_t *outbuf, const AVFrame *pic, int bpp, int w, int h)
71{
72 int i, n = bpp * w;
73 uint8_t *out = outbuf;
74 uint8_t *ptr = pic->data[0];
75
76 for(i=0; i < h; i++) {
77 memcpy(out, ptr, n);
78 out += n;
79 ptr += pic->linesize[0];
80 }
81
82 return out - outbuf;
83}

85static int targa_encode_frame(AVCodecContext *avctx, AVPacket *pkt,
86 const AVFrame *p, int *got_packet)
87{
88 TargaContext *s = avctx->priv_data;
89 int bpp, picsize, datasize = -1, ret, i;
90 uint8_t *out;
91
92 if(avctx->width > 0xffff || avctx->height > 0xffff) {
93 av_log(avctx, AV_LOG_ERROR, "image dimensions too large\n");
94 return AVERROR(EINVAL);
95 }
96 picsize = av_image_get_buffer_size(avctx->pix_fmt,
97 avctx->width, avctx->height, 1);
98 if ((ret = ff_alloc_packet2(avctx, pkt, picsize + 45, 0)) < 0)
99 return ret;
100
101 /* zero out the header and only set applicable fields */
102 memset(pkt->data, 0, 12);
103 AV_WL16(pkt->data+12, avctx->width);
104 AV_WL16(pkt->data+14, avctx->height);
105 /* image descriptor byte: origin is always top-left, bits 0-3 specify alpha */
106 pkt->data[17] = 0x20 | (avctx->pix_fmt == AV_PIX_FMT_BGRA ? 8 : 0);
107
108 out = pkt->data + 18; /* skip past the header we write */
109
110 avctx->bits_per_coded_sample = av_get_bits_per_pixel(av_pix_fmt_desc_get(avctx->pix_fmt));
111 switch(avctx->pix_fmt) {
112 case AV_PIX_FMT_PAL8: {
113 int pal_bpp = 24; /* Only write 32bit palette if there is transparency information */
114 for (i = 0; i < 256; i++)
115 if (AV_RN32(p->data[1] + 4 * i) >> 24 != 0xFF) {
116 pal_bpp = 32;
117 break;
118 }
119 pkt->data[1] = 1; /* palette present */
120 pkt->data[2] = TGA_PAL; /* uncompressed palettised image */
121 pkt->data[6] = 1; /* palette contains 256 entries */
122 pkt->data[7] = pal_bpp; /* palette contains pal_bpp bit entries */
123 pkt->data[16] = 8; /* bpp */
124 for (i = 0; i < 256; i++)
125 if (pal_bpp == 32) {
126 AV_WL32(pkt->data + 18 + 4 * i, *(uint32_t *)(p->data[1] + i * 4));
127 } else {
128 AV_WL24(pkt->data + 18 + 3 * i, *(uint32_t *)(p->data[1] + i * 4));
129 }
130 out += 32 * pal_bpp; /* skip past the palette we just output */
131 break;
132 }
133 case AV_PIX_FMT_GRAY8:
134 pkt->data[2] = TGA_BW; /* uncompressed grayscale image */
135 avctx->bits_per_coded_sample = 0x28;
136 pkt->data[16] = 8; /* bpp */
137 break;
138 case AV_PIX_FMT_RGB555LE:
139 pkt->data[2] = TGA_RGB; /* uncompressed true-color image */
140 avctx->bits_per_coded_sample =
141 pkt->data[16] = 16; /* bpp */
142 break;
143 case AV_PIX_FMT_BGR24:
144 pkt->data[2] = TGA_RGB; /* uncompressed true-color image */
145 pkt->data[16] = 24; /* bpp */
146 break;
147 case AV_PIX_FMT_BGRA:
148 pkt->data[2] = TGA_RGB; /* uncompressed true-color image */
149 pkt->data[16] = 32; /* bpp */
150 break;
151 default:
152 av_log(avctx, AV_LOG_ERROR, "Pixel format '%s' not supported.\n",
153 av_get_pix_fmt_name(avctx->pix_fmt));
154 return AVERROR(EINVAL);
155 }
156 bpp = pkt->data[16] >> 3;
157
158
159#if FF_API_CODER_TYPE
160FF_DISABLE_DEPRECATION_WARNINGS
161 if (avctx->coder_type == FF_CODER_TYPE_RAW)
162 s->rle = 0;
163FF_ENABLE_DEPRECATION_WARNINGS
164#endif
165
166 /* try RLE compression */
167 if (s->rle)
168 datasize = targa_encode_rle(out, picsize, p, bpp, avctx->width, avctx->height);
169
170 /* if that worked well, mark the picture as RLE compressed */
171 if(datasize >= 0)
172 pkt->data[2] |= TGA_RLE;
173
174 /* if RLE didn't make it smaller, go back to no compression */
175 else datasize = targa_encode_normal(out, p, bpp, avctx->width, avctx->height);
176
177 out += datasize;
178
179 /* The standard recommends including this section, even if we don't use
180 * any of the features it affords. TODO: take advantage of the pixel
181 * aspect ratio and encoder ID fields available? */
182 memcpy(out, "\0\0\0\0\0\0\0\0TRUEVISION-XFILE.", 26);
183
184 pkt->size = out + 26 - pkt->data;
185 pkt->flags |= AV_PKT_FLAG_KEY;
186 *got_packet = 1;
187
188 return 0;
189}

191static av_cold int targa_encode_init(AVCodecContext *avctx)
192{
193#if FF_API_CODED_FRAME
194FF_DISABLE_DEPRECATION_WARNINGS
195 avctx->coded_frame->key_frame = 1;
196 avctx->coded_frame->pict_type = AV_PICTURE_TYPE_I;
197FF_ENABLE_DEPRECATION_WARNINGS
198#endif
199
200 return 0;
201}
