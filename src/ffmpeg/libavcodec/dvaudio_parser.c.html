19/**
20 * @file
21 * Ulead DV audio parser
22 *
23 * Determines the duration for each packet.
24 */
25
26#include "parser.h"
27#include "dvaudio.h"
28
29static int dvaudio_parse(AVCodecParserContext *s1, AVCodecContext *avctx,
30 const uint8_t **poutbuf, int *poutbuf_size,
31 const uint8_t *buf, int buf_size)
32{
33 if (buf_size >= 248)
34 s1->duration = dv_get_audio_sample_count(buf + 244, avctx->block_align == 8640);
35
36 /* always return the full packet. this parser isn't doing any splitting or
37 combining, only packet analysis */
38 *poutbuf = buf;
39 *poutbuf_size = buf_size;
40 return buf_size;
41}
42
43AVCodecParser ff_dvaudio_parser = {
44 .codec_ids = { AV_CODEC_ID_DVAUDIO },
45 .parser_parse = dvaudio_parse,
46};
