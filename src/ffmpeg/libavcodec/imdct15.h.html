19#ifndef AVCODEC_IMDCT15_H
20#define AVCODEC_IMDCT15_H
21
22#include <stddef.h>
23
24#include "avfft.h"

26typedef struct IMDCT15Context {
27 int fft_n;
28 int len2;
29 int len4;
30
31 FFTComplex *tmp;
32
33 FFTComplex *twiddle_exptab;
34
35 FFTComplex *exptab[6];

37 /**
38 * Calculate the middle half of the iMDCT

41 ptrdiff_t src_stride, float scale);
42} IMDCT15Context;

49/**
50 * Free an iMDCT.

52void ff_imdct15_uninit(IMDCT15Context **s);

54
55void ff_imdct15_init_aarch64(IMDCT15Context *s);

57#endif /* AVCODEC_IMDCT15_H */
