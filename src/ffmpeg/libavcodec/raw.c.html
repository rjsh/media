284unsigned int avcodec_pix_fmt_to_codec_tag(enum AVPixelFormat fmt)
285{
286 const PixelFormatTag *tags = ff_raw_pix_fmt_tags;
287 while (tags->pix_fmt >= 0) {
288 if (tags->pix_fmt == fmt)
289 return tags->fourcc;
290 tags++;
291 }
292 return 0;
293}
