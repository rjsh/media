39typedef struct TqiContext {
40 AVCodecContext *avctx;
41 GetBitContext gb;
42 BlockDSPContext bdsp;
43 BswapDSPContext bsdsp;
44 IDCTDSPContext idsp;
45 ScanTable intra_scantable;
46
47 void *bitstream_buf;
48 unsigned int bitstream_buf_size;
49
50 int mb_x, mb_y;
51 uint16_t intra_matrix[64];
52 int last_dc[3];
53
54 DECLARE_ALIGNED(16, int16_t, block)[6][64];
55} TqiContext;
56
57static av_cold int tqi_decode_init(AVCodecContext *avctx)
58{
59 TqiContext *t = avctx->priv_data;
60
61 ff_blockdsp_init(&t->bdsp, avctx);
62 ff_bswapdsp_init(&t->bsdsp);
63 ff_idctdsp_init(&t->idsp, avctx);
64 ff_init_scantable_permutation(t->idsp.idct_permutation, FF_IDCT_PERM_NONE);
65 ff_init_scantable(t->idsp.idct_permutation, &t->intra_scantable, ff_zigzag_direct);
66
67 avctx->framerate = (AVRational){ 15, 1 };
68 avctx->pix_fmt = AV_PIX_FMT_YUV420P;
69 ff_mpeg12_init_vlcs();
70 return 0;
71}
72
73static int tqi_decode_mb(TqiContext *t, int16_t (*block)[64])
74{
75 int n;
76
77 t->bdsp.clear_blocks(block[0]);
78 for (n = 0; n < 6; n++) {
79 int ret = ff_mpeg1_decode_block_intra(&t->gb,
80 t->intra_matrix,
81 t->intra_scantable.permutated,
82 t->last_dc, block[n], n, 1);
83 if (ret < 0) {
84 av_log(t->avctx, AV_LOG_ERROR, "ac-tex damaged at %d %d\n",
85 t->mb_x, t->mb_y);
86 return -1;
87 }
88 }
89
90 return 0;
91}

93static inline void tqi_idct_put(AVCodecContext *avctx, AVFrame *frame,
94 int16_t (*block)[64])
95{
96 TqiContext *t = avctx->priv_data;
97 int linesize = frame->linesize[0];
98 uint8_t *dest_y = frame->data[0] + t->mb_y * 16 * linesize + t->mb_x * 16;
99 uint8_t *dest_cb = frame->data[1] + t->mb_y * 8 * frame->linesize[1] + t->mb_x * 8;
100 uint8_t *dest_cr = frame->data[2] + t->mb_y * 8 * frame->linesize[2] + t->mb_x * 8;
101
102 ff_ea_idct_put_c(dest_y , linesize, block[0]);
103 ff_ea_idct_put_c(dest_y + 8, linesize, block[1]);
104 ff_ea_idct_put_c(dest_y + 8*linesize , linesize, block[2]);
105 ff_ea_idct_put_c(dest_y + 8*linesize + 8, linesize, block[3]);
106
107 if (!(avctx->flags & AV_CODEC_FLAG_GRAY)) {
108 ff_ea_idct_put_c(dest_cb, frame->linesize[1], block[4]);
109 ff_ea_idct_put_c(dest_cr, frame->linesize[2], block[5]);
110 }
111}

113static void tqi_calculate_qtable(TqiContext *t, int quant)
114{
115 const int qscale = (215 - 2*quant)*5;
116 int i;
117
118 t->intra_matrix[0] = (ff_inv_aanscales[0] * ff_mpeg1_default_intra_matrix[0]) >> 11;
119 for(i=1; i<64; i++)
120 t->intra_matrix[i] = (ff_inv_aanscales[i] * ff_mpeg1_default_intra_matrix[i] * qscale + 32) >> 14;
121}

123static int tqi_decode_frame(AVCodecContext *avctx,
124 void *data, int *got_frame,
125 AVPacket *avpkt)
126{
127 const uint8_t *buf = avpkt->data;
128 int buf_size = avpkt->size;
129 const uint8_t *buf_end = buf+buf_size;
130 TqiContext *t = avctx->priv_data;
131 AVFrame *frame = data;
132 int ret, w, h;
133
134 t->avctx = avctx;
135
136 w = AV_RL16(&buf[0]);
137 h = AV_RL16(&buf[2]);
138 tqi_calculate_qtable(t, buf[4]);
139 buf += 8;
140
141 ret = ff_set_dimensions(avctx, w, h);
142 if (ret < 0)
143 return ret;
144
145 if ((ret = ff_get_buffer(avctx, frame, 0)) < 0)
146 return ret;
147
148 av_fast_padded_malloc(&t->bitstream_buf, &t->bitstream_buf_size,
149 buf_end - buf);
150 if (!t->bitstream_buf)
151 return AVERROR(ENOMEM);
152 t->bsdsp.bswap_buf(t->bitstream_buf, (const uint32_t *) buf,
153 (buf_end - buf) / 4);
154 init_get_bits(&t->gb, t->bitstream_buf, 8 * (buf_end - buf));
155
156 t->last_dc[0] =
157 t->last_dc[1] =
158 t->last_dc[2] = 0;
159 for (t->mb_y = 0; t->mb_y < (h + 15) / 16; t->mb_y++) {
160 for (t->mb_x = 0; t->mb_x < (w + 15) / 16; t->mb_x++) {
161 if (tqi_decode_mb(t, t->block) < 0)
162 goto end;
163 tqi_idct_put(avctx, frame, t->block);
164 }
165 }
166 end:
167
168 *got_frame = 1;
169 return buf_size;
170}
171
172static av_cold int tqi_decode_end(AVCodecContext *avctx)
173{
174 TqiContext *t = avctx->priv_data;
175 av_freep(&t->bitstream_buf);
176 return 0;
177}
