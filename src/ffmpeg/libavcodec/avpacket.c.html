33void av_init_packet(AVPacket *pkt)
34{
35 pkt->pts = AV_NOPTS_VALUE;
36 pkt->dts = AV_NOPTS_VALUE;
37 pkt->pos = -1;
38 pkt->duration = 0;
39#if FF_API_CONVERGENCE_DURATION
40FF_DISABLE_DEPRECATION_WARNINGS
41 pkt->convergence_duration = 0;
42FF_ENABLE_DEPRECATION_WARNINGS
43#endif
44 pkt->flags = 0;
45 pkt->stream_index = 0;
46 pkt->buf = NULL;
47 pkt->side_data = NULL;
48 pkt->side_data_elems = 0;
49}

51AVPacket *av_packet_alloc(void)

53 AVPacket *pkt = av_mallocz(sizeof(AVPacket));
54 if (!pkt)

57 av_packet_unref(pkt);

71static int packet_alloc(AVBufferRef **buf, int size)
72{
73 int ret;
74 if (size < 0 || size >= INT_MAX - AV_INPUT_BUFFER_PADDING_SIZE)
75 return AVERROR(EINVAL);
76
77 ret = av_buffer_realloc(buf, size + AV_INPUT_BUFFER_PADDING_SIZE);
78 if (ret < 0)
79 return ret;
80
81 memset((*buf)->data + size, 0, AV_INPUT_BUFFER_PADDING_SIZE);
82
83 return 0;
84}

86int av_new_packet(AVPacket *pkt, int size)
87{
88 AVBufferRef *buf = NULL;
89 int ret = packet_alloc(&buf, size);
90 if (ret < 0)
91 return ret;
92
93 av_init_packet(pkt);
94 pkt->buf = buf;
95 pkt->data = buf->data;
96 pkt->size = size;
97
98 return 0;
99}

109int av_grow_packet(AVPacket *pkt, int grow_by)
110{
111 int new_size;
112 av_assert0((unsigned)pkt->size <= INT_MAX - AV_INPUT_BUFFER_PADDING_SIZE);
113 if ((unsigned)grow_by >
114 INT_MAX - (pkt->size + AV_INPUT_BUFFER_PADDING_SIZE))
115 return -1;
116
117 new_size = pkt->size + grow_by + AV_INPUT_BUFFER_PADDING_SIZE;
118 if (pkt->buf) {
119 size_t data_offset;
120 uint8_t *old_data = pkt->data;
121 if (pkt->data == NULL) {
122 data_offset = 0;
123 pkt->data = pkt->buf->data;
124 } else {
125 data_offset = pkt->data - pkt->buf->data;
126 if (data_offset > INT_MAX - new_size)
127 return -1;
128 }
129
130 if (new_size + data_offset > pkt->buf->size) {
131 int ret = av_buffer_realloc(&pkt->buf, new_size + data_offset);
132 if (ret < 0) {
133 pkt->data = old_data;
134 return ret;
135 }
136 pkt->data = pkt->buf->data + data_offset;
137 }
138 } else {
139 pkt->buf = av_buffer_alloc(new_size);
140 if (!pkt->buf)
141 return AVERROR(ENOMEM);
142 if (pkt->size > 0)
143 memcpy(pkt->buf->data, pkt->data, pkt->size);
144 pkt->data = pkt->buf->data;
145 }
146 pkt->size += grow_by;
147 memset(pkt->data + pkt->size, 0, AV_INPUT_BUFFER_PADDING_SIZE);
148
149 return 0;
150}

152int av_packet_from_data(AVPacket *pkt, uint8_t *data, int size)
153{
154 if (size >= INT_MAX - AV_INPUT_BUFFER_PADDING_SIZE)
155 return AVERROR(EINVAL);
156
157 pkt->buf = av_buffer_create(data, size + AV_INPUT_BUFFER_PADDING_SIZE,
158 av_buffer_default_free, NULL, 0);
159 if (!pkt->buf)
160 return AVERROR(ENOMEM);
161
162 pkt->data = data;
163 pkt->size = size;
164
165 return 0;
166}

197/* Makes duplicates of data, side_data, but does not copy any other fields */
198static int copy_packet_data(AVPacket *pkt, const AVPacket *src, int dup)
199{
200 pkt->data = NULL;
201 pkt->side_data = NULL;
202 pkt->side_data_elems = 0;
203 if (pkt->buf) {
204 AVBufferRef *ref = av_buffer_ref(src->buf);
205 if (!ref)
206 return AVERROR(ENOMEM);
207 pkt->buf = ref;
208 pkt->data = ref->data;
209 } else {
210 DUP_DATA(pkt->data, src->data, pkt->size, 1, ALLOC_BUF);
211 }
212 if (src->side_data_elems && dup) {
213 pkt->side_data = src->side_data;
214 pkt->side_data_elems = src->side_data_elems;
215 }
216 if (src->side_data_elems && !dup) {
217 return av_copy_packet_side_data(pkt, src);
218 }
219 return 0;
220
221failed_alloc:
222 av_packet_unref(pkt);
223 return AVERROR(ENOMEM);
224}

226int av_copy_packet_side_data(AVPacket *pkt, const AVPacket *src)
227{
228 if (src->side_data_elems) {
229 int i;
230 DUP_DATA(pkt->side_data, src->side_data,
231 src->side_data_elems * sizeof(*src->side_data), 0, ALLOC_MALLOC);
232 if (src != pkt) {
233 memset(pkt->side_data, 0,
234 src->side_data_elems * sizeof(*src->side_data));
235 }
236 for (i = 0; i < src->side_data_elems; i++) {
237 DUP_DATA(pkt->side_data[i].data, src->side_data[i].data,
238 src->side_data[i].size, 1, ALLOC_MALLOC);
239 pkt->side_data[i].size = src->side_data[i].size;
240 pkt->side_data[i].type = src->side_data[i].type;
241 }
242 }
243 pkt->side_data_elems = src->side_data_elems;
244 return 0;
245
246failed_alloc:
247 av_packet_unref(pkt);
248 return AVERROR(ENOMEM);
249}
250FF_ENABLE_DEPRECATION_WARNINGS
251#endif

253int av_dup_packet(AVPacket *pkt)
254{
255 AVPacket tmp_pkt;
256
257 if (!pkt->buf && pkt->data) {
258 tmp_pkt = *pkt;
259 return copy_packet_data(pkt, &tmp_pkt, 1);
260 }
261 return 0;
262}

264int av_copy_packet(AVPacket *dst, const AVPacket *src)

295int av_packet_add_side_data(AVPacket *pkt, enum AVPacketSideDataType type,
296 uint8_t *data, size_t size)
297{
298 int elems = pkt->side_data_elems;
299
300 if ((unsigned)elems + 1 > INT_MAX / sizeof(*pkt->side_data))
301 return AVERROR(ERANGE);
302
303 pkt->side_data = av_realloc(pkt->side_data,
304 (elems + 1) * sizeof(*pkt->side_data));
305 if (!pkt->side_data)
306 return AVERROR(ENOMEM);
307
308 pkt->side_data[elems].data = data;
309 pkt->side_data[elems].size = size;
310 pkt->side_data[elems].type = type;
311 pkt->side_data_elems++;
312
313 return 0;
314}

317uint8_t *av_packet_new_side_data(AVPacket *pkt, enum AVPacketSideDataType type,
318 int size)
319{
320 int ret;
321 uint8_t *data;
322
323 if ((unsigned)size > INT_MAX - AV_INPUT_BUFFER_PADDING_SIZE)
324 return NULL;
325 data = av_mallocz(size + AV_INPUT_BUFFER_PADDING_SIZE);
326 if (!data)
327 return NULL;
328
329 ret = av_packet_add_side_data(pkt, type, data, size);
330 if (ret < 0) {
331 av_freep(&data);
332 return NULL;
333 }
334
335 return data;
336}

338uint8_t *av_packet_get_side_data(AVPacket *pkt, enum AVPacketSideDataType type,
339 int *size)
340{
341 int i;
342
343 for (i = 0; i < pkt->side_data_elems; i++) {
344 if (pkt->side_data[i].type == type) {
345 if (size)
346 *size = pkt->side_data[i].size;
347 return pkt->side_data[i].data;
348 }
349 }
350 return NULL;
351}

378#define FF_MERGE_MARKER 0x8c4d9d108e25e9feULL

380int av_packet_merge_side_data(AVPacket *pkt){
381 if(pkt->side_data_elems){
382 AVBufferRef *buf;
383 int i;
384 uint8_t *p;
385 uint64_t size= pkt->size + 8LL + AV_INPUT_BUFFER_PADDING_SIZE;
386 AVPacket old= *pkt;
387 for (i=0; i<old.side_data_elems; i++) {
388 size += old.side_data[i].size + 5LL;
389 }
390 if (size > INT_MAX)
391 return AVERROR(EINVAL);
392 buf = av_buffer_alloc(size);
393 if (!buf)
394 return AVERROR(ENOMEM);
395 pkt->buf = buf;
396 pkt->data = p = buf->data;
397 pkt->size = size - AV_INPUT_BUFFER_PADDING_SIZE;
398 bytestream_put_buffer(&p, old.data, old.size);
399 for (i=old.side_data_elems-1; i>=0; i--) {
400 bytestream_put_buffer(&p, old.side_data[i].data, old.side_data[i].size);
401 bytestream_put_be32(&p, old.side_data[i].size);
402 *p++ = old.side_data[i].type | ((i==old.side_data_elems-1)*128);
403 }
404 bytestream_put_be64(&p, FF_MERGE_MARKER);
405 av_assert0(p-pkt->data == pkt->size);
406 memset(p, 0, AV_INPUT_BUFFER_PADDING_SIZE);
407 av_packet_unref(&old);
408 pkt->side_data_elems = 0;
409 pkt->side_data = NULL;
410 return 1;
411 }
412 return 0;
413}

415int av_packet_split_side_data(AVPacket *pkt){
416 if (!pkt->side_data_elems && pkt->size >12 && AV_RB64(pkt->data + pkt->size - 8) == FF_MERGE_MARKER){
417 int i;
418 unsigned int size;
419 uint8_t *p;
420
421 p = pkt->data + pkt->size - 8 - 5;
422 for (i=1; ; i++){
423 size = AV_RB32(p);
424 if (size>INT_MAX - 5 || p - pkt->data < size)
425 return 0;
426 if (p[4]&128)
427 break;
428 if (p - pkt->data < size + 5)
429 return 0;
430 p-= size+5;
431 }
432
433 pkt->side_data = av_malloc_array(i, sizeof(*pkt->side_data));
434 if (!pkt->side_data)
435 return AVERROR(ENOMEM);
436
437 p= pkt->data + pkt->size - 8 - 5;
438 for (i=0; ; i++){
439 size= AV_RB32(p);
440 av_assert0(size<=INT_MAX - 5 && p - pkt->data >= size);
441 pkt->side_data[i].data = av_mallocz(size + AV_INPUT_BUFFER_PADDING_SIZE);
442 pkt->side_data[i].size = size;
443 pkt->side_data[i].type = p[4]&127;
444 if (!pkt->side_data[i].data)
445 return AVERROR(ENOMEM);
446 memcpy(pkt->side_data[i].data, p-size, size);
447 pkt->size -= size + 5;
448 if(p[4]&128)
449 break;
450 p-= size+5;
451 }
452 pkt->size -= 8;
453 pkt->side_data_elems = i+1;
454 return 1;
455 }
456 return 0;
457}

459uint8_t *av_packet_pack_dictionary(AVDictionary *dict, int *size)
460{
461 AVDictionaryEntry *t = NULL;
462 uint8_t *data = NULL;
463 *size = 0;
464
465 if (!dict)
466 return NULL;
467
468 while ((t = av_dict_get(dict, "", t, AV_DICT_IGNORE_SUFFIX))) {
469 const size_t keylen = strlen(t->key);
470 const size_t valuelen = strlen(t->value);
471 const size_t new_size = *size + keylen + 1 + valuelen + 1;
472 uint8_t *const new_data = av_realloc(data, new_size);
473
474 if (!new_data)
475 goto fail;
476 data = new_data;
477 if (new_size > INT_MAX)
478 goto fail;
479
480 memcpy(data + *size, t->key, keylen + 1);
481 memcpy(data + *size + keylen + 1, t->value, valuelen + 1);
482
483 *size = new_size;
484 }
485
486 return data;
487
488fail:
489 av_freep(&data);
490 *size = 0;
491 return NULL;
492}

494int av_packet_unpack_dictionary(const uint8_t *data, int size, AVDictionary **dict)
495{
496 const uint8_t *end = data + size;
497 int ret = 0;
498
499 if (!dict || !data || !size)
500 return ret;
501 if (size && end[-1])
502 return AVERROR_INVALIDDATA;
503 while (data < end) {
504 const uint8_t *key = data;
505 const uint8_t *val = data + strlen(key) + 1;
506
507 if (val >= end)
508 return AVERROR_INVALIDDATA;
509
510 ret = av_dict_set(dict, key, val, 0);
511 if (ret < 0)
512 break;
513 data = val + strlen(val) + 1;
514 }
515
516 return ret;
517}

535int av_packet_copy_props(AVPacket *dst, const AVPacket *src)
536{
537 int i;
538
539 dst->pts = src->pts;
540 dst->dts = src->dts;
541 dst->pos = src->pos;
542 dst->duration = src->duration;
543#if FF_API_CONVERGENCE_DURATION
544FF_DISABLE_DEPRECATION_WARNINGS
545 dst->convergence_duration = src->convergence_duration;
546FF_ENABLE_DEPRECATION_WARNINGS
547#endif
548 dst->flags = src->flags;
549 dst->stream_index = src->stream_index;
550
551 for (i = 0; i < src->side_data_elems; i++) {
552 enum AVPacketSideDataType type = src->side_data[i].type;
553 int size = src->side_data[i].size;
554 uint8_t *src_data = src->side_data[i].data;
555 uint8_t *dst_data = av_packet_new_side_data(dst, type, size);
556
557 if (!dst_data) {
558 av_packet_free_side_data(dst);
559 return AVERROR(ENOMEM);
560 }
561 memcpy(dst_data, src_data, size);
562 }
563
564 return 0;
565}

576int av_packet_ref(AVPacket *dst, const AVPacket *src)
577{
578 int ret;
579
580 ret = av_packet_copy_props(dst, src);
581 if (ret < 0)
582 return ret;
583
584 if (!src->buf) {
585 ret = packet_alloc(&dst->buf, src->size);
586 if (ret < 0)
587 goto fail;
588 if (src->size)
589 memcpy(dst->buf->data, src->data, src->size);
590
591 dst->data = dst->buf->data;
592 } else {
593 dst->buf = av_buffer_ref(src->buf);
594 if (!dst->buf) {
595 ret = AVERROR(ENOMEM);
596 goto fail;
597 }
598 dst->data = src->data;
599 }
600
601 dst->size = src->size;
602
603 return 0;
604fail:
605 av_packet_free_side_data(dst);
606 return ret;
607}

630void av_packet_rescale_ts(AVPacket *pkt, AVRational src_tb, AVRational dst_tb)
631{
632 if (pkt->pts != AV_NOPTS_VALUE)
633 pkt->pts = av_rescale_q(pkt->pts, src_tb, dst_tb);
634 if (pkt->dts != AV_NOPTS_VALUE)
635 pkt->dts = av_rescale_q(pkt->dts, src_tb, dst_tb);
636 if (pkt->duration > 0)
637 pkt->duration = av_rescale_q(pkt->duration, src_tb, dst_tb);
638#if FF_API_CONVERGENCE_DURATION
639FF_DISABLE_DEPRECATION_WARNINGS
640 if (pkt->convergence_duration > 0)
641 pkt->convergence_duration = av_rescale_q(pkt->convergence_duration, src_tb, dst_tb);
642FF_ENABLE_DEPRECATION_WARNINGS
643#endif
644}

646int ff_side_data_set_encoder_stats(AVPacket *pkt, int quality, int64_t *error, int error_count, int pict_type)
647{
648 uint8_t *side_data;
649 int side_data_size;
650 int i;
651
652 side_data = av_packet_get_side_data(pkt, AV_PKT_DATA_QUALITY_STATS, &side_data_size);
653 if (!side_data) {
654 side_data_size = 4+4+8*error_count;
655 side_data = av_packet_new_side_data(pkt, AV_PKT_DATA_QUALITY_STATS,
656 side_data_size);
657 }
658
659 if (!side_data || side_data_size < 4+4+8*error_count)
660 return AVERROR(ENOMEM);
661
662 AV_WL32(side_data , quality );
663 side_data[4] = pict_type;
664 side_data[5] = error_count;
665 for (i = 0; i<error_count; i++)
666 AV_WL64(side_data+8 + 8*i , error[i]);
667
668 return 0;
669}
