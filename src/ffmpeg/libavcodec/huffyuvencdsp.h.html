19#ifndef AVCODEC_HUFFYUVENCDSP_H

22#include <stdint.h>

24typedef struct HuffYUVEncDSPContext {
25 void (*diff_bytes)(uint8_t *dst /* align 16 */,
26 const uint8_t *src1 /* align 16 */,
27 const uint8_t *src2 /* align 1 */,
28 intptr_t w);
29 /**
30 * Subtract HuffYUV's variant of median prediction.
31 * Note, this might read from src1[-1], src2[-1].
32 */
33 void (*sub_hfyu_median_pred)(uint8_t *dst, const uint8_t *src1,
34 const uint8_t *src2, intptr_t w,
35 int *left, int *left_top);
36} HuffYUVEncDSPContext;
37
38void ff_huffyuvencdsp_init(HuffYUVEncDSPContext *c);
39void ff_huffyuvencdsp_init_x86(HuffYUVEncDSPContext *c);
