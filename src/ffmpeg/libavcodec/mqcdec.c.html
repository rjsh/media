30static void bytein(MqcState *mqc)
31{
32 if (*mqc->bp == 0xff) {
33 if (*(mqc->bp + 1) > 0x8f)
34 mqc->c++;
35 else {
36 mqc->bp++;
37 mqc->c += 2 + 0xfe00 - (*mqc->bp << 9);
38 }
39 } else {
40 mqc->bp++;
41 mqc->c += 1 + 0xff00 - (*mqc->bp << 8);
42 }
43}
44
45static int exchange(MqcState *mqc, uint8_t *cxstate, int lps)
46{
47 int d;
48 if ((mqc->a < ff_mqc_qe[*cxstate]) ^ (!lps)) {
49 if (lps)
50 mqc->a = ff_mqc_qe[*cxstate];
51 d = *cxstate & 1;
52 *cxstate = ff_mqc_nmps[*cxstate];
53 } else {
54 if (lps)
55 mqc->a = ff_mqc_qe[*cxstate];
56 d = 1 - (*cxstate & 1);
57 *cxstate = ff_mqc_nlps[*cxstate];
58 }
59 // do RENORMD: see ISO/IEC 15444-1:2002 Â§C.3.3
60 do {
61 if (!(mqc->c & 0xff)) {
62 mqc->c -= 0x100;
63 bytein(mqc);
64 }
65 mqc->a += mqc->a;
66 mqc->c += mqc->c;
67 } while (!(mqc->a & 0x8000));
68 return d;
69}

71void ff_mqc_initdec(MqcState *mqc, uint8_t *bp, int raw, int reset)
72{
73 mqc->raw = raw;
74 if (reset)
75 ff_mqc_init_contexts(mqc);
76 mqc->bp = bp;
77 mqc->c = (*mqc->bp ^ 0xff) << 16;
78 bytein(mqc);
79 mqc->c = mqc->c << 7;
80 mqc->a = 0x8000;
81}

83static int mqc_decode_bypass(MqcState *mqc) {
84 int bit = !(mqc->c & 0x40000000);
85 if (!(mqc->c & 0xff)) {
86 mqc->c -= 0x100;
87 bytein(mqc);
88 }
89 mqc->c += mqc->c;
90 return bit;
91}
92
93int ff_mqc_decode(MqcState *mqc, uint8_t *cxstate)
94{
95 if (mqc->raw)
96 return mqc_decode_bypass(mqc);
97 mqc->a -= ff_mqc_qe[*cxstate];
98 if ((mqc->c >> 16) < mqc->a) {
99 if (mqc->a & 0x8000)
100 return *cxstate & 1;
101 else
102 return exchange(mqc, cxstate, 0);
103 } else {
104 mqc->c -= mqc->a << 16;
105 return exchange(mqc, cxstate, 1);
106 }
107}
