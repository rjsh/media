30/* bitrate is in kb/s */
31int ff_mpa_l2_select_table(int bitrate, int nb_channels, int freq, int lsf)
32{
33 int ch_bitrate, table;
34
35 ch_bitrate = bitrate / nb_channels;
36 if (!lsf) {
37 if ((freq == 48000 && ch_bitrate >= 56) ||
38 (ch_bitrate >= 56 && ch_bitrate <= 80))
39 table = 0;
40 else if (freq != 48000 && ch_bitrate >= 96)
41 table = 1;
42 else if (freq != 32000 && ch_bitrate <= 48)
43 table = 2;
44 else
45 table = 3;
46 } else {
47 table = 4;
48 }
49 return table;
50}
