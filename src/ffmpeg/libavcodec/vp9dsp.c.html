24#include "libavutil/avassert.h"
25#include "libavutil/common.h"
26#include "vp9dsp.h"
27
28av_cold void ff_vp9dsp_init(VP9DSPContext *dsp, int bpp, int bitexact)
29{
30 if (bpp == 8) {
31 ff_vp9dsp_init_8(dsp);
32 } else if (bpp == 10) {
33 ff_vp9dsp_init_10(dsp);
34 } else {
35 av_assert0(bpp == 12);
36 ff_vp9dsp_init_12(dsp);
37 }
38
39 if (ARCH_X86) ff_vp9dsp_init_x86(dsp, bpp, bitexact);
40 if (ARCH_MIPS) ff_vp9dsp_init_mips(dsp, bpp);
41}
