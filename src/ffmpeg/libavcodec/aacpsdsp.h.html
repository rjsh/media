21#ifndef AVCODEC_AACPSDSP_H
22#define AVCODEC_AACPSDSP_H
23
24#include "aac_defines.h"
25
26#define PS_QMF_TIME_SLOTS 32
27#define PS_AP_LINKS 3
28#define PS_MAX_AP_DELAY 5
29
30typedef struct PSDSPContext {
31 void (*add_squares)(INTFLOAT *dst, const INTFLOAT (*src)[2], int n);
32 void (*mul_pair_single)(INTFLOAT (*dst)[2], INTFLOAT (*src0)[2], INTFLOAT *src1,
33 int n);
34 void (*hybrid_analysis)(INTFLOAT (*out)[2], INTFLOAT (*in)[2],
35 const INTFLOAT (*filter)[8][2],
36 int stride, int n);
37 void (*hybrid_analysis_ileave)(INTFLOAT (*out)[32][2], INTFLOAT L[2][38][64],
38 int i, int len);
39 void (*hybrid_synthesis_deint)(INTFLOAT out[2][38][64], INTFLOAT (*in)[32][2],
40 int i, int len);
41 void (*decorrelate)(INTFLOAT (*out)[2], INTFLOAT (*delay)[2],
42 INTFLOAT (*ap_delay)[PS_QMF_TIME_SLOTS+PS_MAX_AP_DELAY][2],
43 const INTFLOAT phi_fract[2], const INTFLOAT (*Q_fract)[2],
44 const INTFLOAT *transient_gain,
45 INTFLOAT g_decay_slope,
46 int len);
47 void (*stereo_interpolate[2])(INTFLOAT (*l)[2], INTFLOAT (*r)[2],
48 INTFLOAT h[2][4], INTFLOAT h_step[2][4],
49 int len);
50} PSDSPContext;
51
52void AAC_RENAME(ff_psdsp_init)(PSDSPContext *s);
53void ff_psdsp_init_arm(PSDSPContext *s);
54void ff_psdsp_init_mips(PSDSPContext *s);
55void ff_psdsp_init_x86(PSDSPContext *s);
56
57#endif /* AVCODEC_AACPSDSP_H */
