23#include "avcodec.h"
24#include "internal.h"
25#include "bytestream.h"
26
27static int encode_frame(AVCodecContext *avctx, AVPacket *pkt,
28 const AVFrame *pic, int *got_packet)
29{
30 int i, j, ret;
31 int aligned_width = FFALIGN(avctx->width,
32 avctx->codec_id == AV_CODEC_ID_R10K ? 1 : 64);
33 int pad = (aligned_width - avctx->width) * 4;
34 uint8_t *src_line;
35 uint8_t *dst;
36
37 if ((ret = ff_alloc_packet2(avctx, pkt, 4 * aligned_width * avctx->height, 0)) < 0)
38 return ret;
39
40 src_line = pic->data[0];
41 dst = pkt->data;
42
43 for (i = 0; i < avctx->height; i++) {
44 uint16_t *src = (uint16_t *)src_line;
45 for (j = 0; j < avctx->width; j++) {
46 uint32_t pixel;
47 uint16_t r = *src++ >> 6;
48 uint16_t g = *src++ >> 6;
49 uint16_t b = *src++ >> 6;
50 if (avctx->codec_id == AV_CODEC_ID_R210)
51 pixel = (r << 20) | (g << 10) | b;
52 else
53 pixel = (r << 22) | (g << 12) | (b << 2);
54 if (avctx->codec_id == AV_CODEC_ID_AVRP)
55 bytestream_put_le32(&dst, pixel);
56 else
57 bytestream_put_be32(&dst, pixel);
58 }
59 memset(dst, 0, pad);
60 dst += pad;
61 src_line += pic->linesize[0];
62 }
63
64 pkt->flags |= AV_PKT_FLAG_KEY;
65 *got_packet = 1;
66 return 0;
67}
