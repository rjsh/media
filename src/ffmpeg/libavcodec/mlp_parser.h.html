27#ifndef AVCODEC_MLP_PARSER_H
28#define AVCODEC_MLP_PARSER_H
29
30#include "get_bits.h"
31
32typedef struct MLPHeaderInfo
33{
34 int stream_type; ///< 0xBB for MLP, 0xBA for TrueHD
35 int header_size; ///< Size of the major sync header, in bytes
36
37 int group1_bits; ///< The bit depth of the first substream
38 int group2_bits; ///< Bit depth of the second substream (MLP only)
39
40 int group1_samplerate; ///< Sample rate of first substream
41 int group2_samplerate; ///< Sample rate of second substream (MLP only)
42
43 int channel_arrangement;
44
45 int channel_modifier_thd_stream0; ///< Channel modifier for substream 0 of TrueHD streams ("2-channel presentation")
46 int channel_modifier_thd_stream1; ///< Channel modifier for substream 1 of TrueHD streams ("6-channel presentation")
47 int channel_modifier_thd_stream2; ///< Channel modifier for substream 2 of TrueHD streams ("8-channel presentation")
48
49 int channels_mlp; ///< Channel count for MLP streams
50 int channels_thd_stream1; ///< Channel count for substream 1 of TrueHD streams ("6-channel presentation")
51 int channels_thd_stream2; ///< Channel count for substream 2 of TrueHD streams ("8-channel presentation")
52 uint64_t channel_layout_mlp; ///< Channel layout for MLP streams
53 uint64_t channel_layout_thd_stream1; ///< Channel layout for substream 1 of TrueHD streams ("6-channel presentation")
54 uint64_t channel_layout_thd_stream2; ///< Channel layout for substream 2 of TrueHD streams ("8-channel presentation")
55
56 int access_unit_size; ///< Number of samples per coded frame
57 int access_unit_size_pow2; ///< Next power of two above number of samples per frame
58
59 int is_vbr; ///< Stream is VBR instead of CBR
60 int peak_bitrate; ///< Peak bitrate for VBR, actual bitrate (==peak) for CBR
61
62 int num_substreams; ///< Number of substreams within stream
63} MLPHeaderInfo;
64

66int ff_mlp_read_major_sync(void *log, MLPHeaderInfo *mh, GetBitContext *gb);
67uint64_t ff_truehd_layout(int chanmap);
68
69extern const uint64_t ff_mlp_layout[32];
70
71#endif /* AVCODEC_MLP_PARSER_H */
