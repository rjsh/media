74static av_always_inline int wmv2_get_cbp_table_index(MpegEncContext *s, int cbp_index)
75{
76 static const uint8_t map[3][3] = {
77 { 0, 2, 1 },
78 { 1, 0, 2 },
79 { 2, 1, 0 },
80 };
81
82 return map[(s->qscale > 10) + (s->qscale > 20)][cbp_index];
83}
