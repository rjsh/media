22#ifndef AVCODEC_VC2ENC_DWT_H
23#define AVCODEC_VC2ENC_DWT_H
24
25#include <stddef.h>
26#include <stdint.h>
27
28typedef int32_t dwtcoef;
29
30enum VC2TransformType {
31 VC2_TRANSFORM_9_7 = 0, /* Deslauriers-Dubuc (9,7) */
32 VC2_TRANSFORM_5_3 = 1, /* LeGall (5,3) */
33 VC2_TRANSFORM_13_7 = 2, /* Deslauriers-Dubuc (13,7) */
34 VC2_TRANSFORM_HAAR = 3, /* Haar without shift */
35 VC2_TRANSFORM_HAAR_S = 4, /* Haar with 1 shift/lvl */
36 VC2_TRANSFORM_FIDEL = 5, /* Fidelity filter */
37 VC2_TRANSFORM_9_7_I = 6, /* Daubechies (9,7) */
38
39 VC2_TRANSFORMS_NB
40};
41
42typedef struct VC2TransformContext {
43 dwtcoef *buffer;
44 void (*vc2_subband_dwt[VC2_TRANSFORMS_NB])(struct VC2TransformContext *t,
45 dwtcoef *data, ptrdiff_t stride,
46 int width, int height);
47} VC2TransformContext;
48
49int ff_vc2enc_init_transforms(VC2TransformContext *t, int p_width, int p_height);
50void ff_vc2enc_free_transforms(VC2TransformContext *t);
51
52#endif /* AVCODEC_VC2ENC_DWT_H */
