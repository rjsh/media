151static av_always_inline int fold(int diff, int bits)
152{
153 if (bits == 8)
154 diff = (int8_t)diff;
155 else {
156 diff += 1 << (bits - 1);
157 diff = av_mod_uintp2(diff, bits);
158 diff -= 1 << (bits - 1);
159 }
160
161 return diff;
162}

164static inline void update_vlc_state(VlcState *const state, const int v)
165{
166 int drift = state->drift;
167 int count = state->count;
168 state->error_sum += FFABS(v);
169 drift += v;
170
171 if (count == 128) { // FIXME: variable
172 count >>= 1;
173 drift >>= 1;
174 state->error_sum >>= 1;
175 }
176 count++;
177
178 if (drift <= -count) {
179 if (state->bias > -128)
180 state->bias--;
181
182 drift += count;
183 if (drift <= -count)
184 drift = -count + 1;
185 } else if (drift > 0) {
186 if (state->bias < 127)
187 state->bias++;
188
189 drift -= count;
190 if (drift > 0)
191 drift = 0;
192 }
193
194 state->drift = drift;
195 state->count = count;
196}
