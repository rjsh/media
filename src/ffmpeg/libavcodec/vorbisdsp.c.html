25{

27
28 if (ARCH_AARCH64)
29 ff_vorbisdsp_init_aarch64(dsp);
30 if (ARCH_ARM)
31 ff_vorbisdsp_init_arm(dsp);
32 if (ARCH_PPC)
33 ff_vorbisdsp_init_ppc(dsp);
34 if (ARCH_X86)
35 ff_vorbisdsp_init_x86(dsp);
36}
