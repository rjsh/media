21#ifndef AVCODEC_DIRAC_DWT_H
22#define AVCODEC_DIRAC_DWT_H
23
24#include <stdint.h>
25
26typedef int DWTELEM;
27typedef short IDWTELEM;
28
29#define MAX_DWT_SUPPORT 8
30#define MAX_DECOMPOSITIONS 8
31
32typedef struct DWTCompose {
33 uint8_t *b[MAX_DWT_SUPPORT];
34 int y;
35} DWTCompose;
36
37typedef struct DWTPlane {
38 int width;
39 int height;
40 int stride;
41 uint8_t *buf;
42 uint8_t *buf_base;
43 uint8_t *tmp;
44} DWTPlane;
45
46struct DWTContext;
47
48// Possible prototypes for vertical_compose functions
49typedef void (*vertical_compose_2tap)(uint8_t *b0, uint8_t *b1, int width);
50typedef void (*vertical_compose_3tap)(uint8_t *b0, uint8_t *b1, uint8_t *b2, int width);
51typedef void (*vertical_compose_5tap)(uint8_t *b0, uint8_t *b1, uint8_t *b2, uint8_t *b3, uint8_t *b4, int width);
52typedef void (*vertical_compose_9tap)(uint8_t *dst, uint8_t *b[8], int width);
53
54typedef struct DWTContext {
55 uint8_t *buffer;
56 uint8_t *temp;
57 int width;
58 int height;
59 int stride;
60 int decomposition_count;
61 int support;
62
63 void (*spatial_compose)(struct DWTContext *cs, int level, int width, int height, int stride);
64 void (*vertical_compose_l0)(void);
65 void (*vertical_compose_h0)(void);
66 void (*vertical_compose_l1)(void);
67 void (*vertical_compose_h1)(void);
68 void (*vertical_compose)(void); ///< one set of lowpass and highpass combined
69 void (*horizontal_compose)(uint8_t *b, uint8_t *tmp, int width);
70
71 DWTCompose cs[MAX_DECOMPOSITIONS];
72} DWTContext;
73
74enum dwt_type {
75 DWT_SNOW_DAUB9_7,
76 DWT_SNOW_LEGALL5_3,
77 DWT_DIRAC_DD9_7,
78 DWT_DIRAC_LEGALL5_3,
79 DWT_DIRAC_DD13_7,
80 DWT_DIRAC_HAAR0,
81 DWT_DIRAC_HAAR1,
82 DWT_DIRAC_FIDELITY,
83 DWT_DIRAC_DAUB9_7,
84 DWT_NUM_TYPES
85};

94// shared stuff for simd optimizations
95#define COMPOSE_53iL0(b0, b1, b2)\
96 (b1 - ((b0 + b2 + 2) >> 2))
97
98#define COMPOSE_DIRAC53iH0(b0, b1, b2)\
99 (b1 + ((b0 + b2 + 1) >> 1))
100
101#define COMPOSE_DD97iH0(b0, b1, b2, b3, b4)\
102 (b2 + ((-b0 + 9*b1 + 9*b3 - b4 + 8) >> 4))
103
104#define COMPOSE_DD137iL0(b0, b1, b2, b3, b4)\
105 (b2 - ((-b0 + 9*b1 + 9*b3 - b4 + 16) >> 5))
106
107#define COMPOSE_HAARiL0(b0, b1)\
108 (b0 - ((b1 + 1) >> 1))
109
110#define COMPOSE_HAARiH0(b0, b1)\
111 (b0 + b1)
112
113#define COMPOSE_FIDELITYiL0(b0, b1, b2, b3, b4, b5, b6, b7, b8)\
114 (b4 - ((-8*(b0+b8) + 21*(b1+b7) - 46*(b2+b6) + 161*(b3+b5) + 128) >> 8))
115
116#define COMPOSE_FIDELITYiH0(b0, b1, b2, b3, b4, b5, b6, b7, b8)\
117 (b4 + ((-2*(b0+b8) + 10*(b1+b7) - 25*(b2+b6) + 81*(b3+b5) + 128) >> 8))
118
119#define COMPOSE_DAUB97iL1(b0, b1, b2)\
120 (b1 - ((1817*(b0 + b2) + 2048) >> 12))
121
122#define COMPOSE_DAUB97iH1(b0, b1, b2)\
123 (b1 - (( 113*(b0 + b2) + 64) >> 7))
124
125#define COMPOSE_DAUB97iL0(b0, b1, b2)\
126 (b1 + (( 217*(b0 + b2) + 2048) >> 12))
127
128#define COMPOSE_DAUB97iH0(b0, b1, b2)\
129 (b1 + ((6497*(b0 + b2) + 2048) >> 12))
