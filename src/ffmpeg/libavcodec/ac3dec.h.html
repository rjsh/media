22/**
23 * @file
24 * Common code between the AC-3 and E-AC-3 decoders.
25 *
26 * Summary of MDCT Coefficient Grouping:
27 * The individual MDCT coefficient indices are often referred to in the
28 * (E-)AC-3 specification as frequency bins. These bins are grouped together
29 * into subbands of 12 coefficients each. The subbands are grouped together
30 * into bands as defined in the bitstream by the band structures, which
31 * determine the number of bands and the size of each band. The full spectrum
32 * of 256 frequency bins is divided into 1 DC bin + 21 subbands = 253 bins.
33 * This system of grouping coefficients is used for channel bandwidth, stereo
34 * rematrixing, channel coupling, enhanced coupling, and spectral extension.
35 *
36 * +-+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+-+
37 * |1| |12| | [12|12|12|12] | | | | | | | | | | | | |3|
38 * +-+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+--+-+
39 * ~~~ ~~~~ ~~~~~~~~~~~~~ ~~~
40 * | | | |
41 * | | | 3 unused frequency bins--+
42 * | | |
43 * | | +--1 band containing 4 subbands
44 * | |
45 * | +--1 subband of 12 frequency bins
46 * |
47 * +--DC frequency bin
48 */
49

72 AVCodecContext *avctx; ///< parent context
73 GetBitContext gbc; ///< bitstream reader

75///@name Bit stream information
76///@{
77 int frame_type; ///< frame type (strmtyp)
78 int substreamid; ///< substream identification
79 int frame_size; ///< current frame size, in bytes
80 int bit_rate; ///< stream bit rate, in bits-per-second
81 int sample_rate; ///< sample frequency, in Hz
82 int num_blocks; ///< number of audio blocks
83 int bitstream_id; ///< bitstream id (bsid)
84 int bitstream_mode; ///< bitstream mode (bsmod)
85 int channel_mode; ///< channel mode (acmod)
86 int lfe_on; ///< lfe channel in use
87 int dialog_normalization[2]; ///< dialog level in dBFS (dialnorm)
88 int compression_exists[2]; ///< compression field is valid for frame (compre)
89 int compression_gain[2]; ///< gain to apply for heavy compression (compr)
90 int channel_map; ///< custom channel map
91 int preferred_downmix; ///< Preferred 2-channel downmix mode (dmixmod)
92 int center_mix_level; ///< Center mix level index
93 int center_mix_level_ltrt; ///< Center mix level index for Lt/Rt (ltrtcmixlev)
94 int surround_mix_level; ///< Surround mix level index
95 int surround_mix_level_ltrt; ///< Surround mix level index for Lt/Rt (ltrtsurmixlev)
96 int lfe_mix_level_exists; ///< indicates if lfemixlevcod is specified (lfemixlevcode)
97 int lfe_mix_level; ///< LFE mix level index (lfemixlevcod)
98 int eac3; ///< indicates if current frame is E-AC-3
99 int eac3_frame_dependent_found; ///< bitstream has E-AC-3 dependent frame(s)
100 int eac3_subsbtreamid_found; ///< bitstream has E-AC-3 additional substream(s)
101 int dolby_surround_mode; ///< dolby surround mode (dsurmod)
102 int dolby_surround_ex_mode; ///< dolby surround ex mode (dsurexmod)
103 int dolby_headphone_mode; ///< dolby headphone mode (dheadphonmod)
104///@}
105
106 int preferred_stereo_downmix;
107 float ltrt_center_mix_level;
108 float ltrt_surround_mix_level;
109 float loro_center_mix_level;
110 float loro_surround_mix_level;
111 int target_level; ///< target level in dBFS
112 float level_gain[2];
113
114///@name Frame syntax parameters
115 int snr_offset_strategy; ///< SNR offset strategy (snroffststr)
116 int block_switch_syntax; ///< block switch syntax enabled (blkswe)
117 int dither_flag_syntax; ///< dither flag syntax enabled (dithflage)
118 int bit_allocation_syntax; ///< bit allocation model syntax enabled (bamode)
119 int fast_gain_syntax; ///< fast gain codes enabled (frmfgaincode)
120 int dba_syntax; ///< delta bit allocation syntax enabled (dbaflde)
121 int skip_syntax; ///< skip field syntax enabled (skipflde)
122 ///@}
123
124///@name Standard coupling
125 int cpl_in_use[AC3_MAX_BLOCKS]; ///< coupling in use (cplinu)
126 int cpl_strategy_exists[AC3_MAX_BLOCKS];///< coupling strategy exists (cplstre)
127 int channel_in_cpl[AC3_MAX_CHANNELS]; ///< channel in coupling (chincpl)
128 int phase_flags_in_use; ///< phase flags in use (phsflginu)
129 int phase_flags[AC3_MAX_CPL_BANDS]; ///< phase flags (phsflg)
130 int num_cpl_bands; ///< number of coupling bands (ncplbnd)
131 uint8_t cpl_band_sizes[AC3_MAX_CPL_BANDS]; ///< number of coeffs in each coupling band
132 int firstchincpl; ///< first channel in coupling
133 int first_cpl_coords[AC3_MAX_CHANNELS]; ///< first coupling coordinates states (firstcplcos)
134 int cpl_coords[AC3_MAX_CHANNELS][AC3_MAX_CPL_BANDS]; ///< coupling coordinates (cplco)
135///@}
136
137///@name Spectral extension
138///@{
139 int spx_in_use; ///< spectral extension in use (spxinu)
140 uint8_t channel_uses_spx[AC3_MAX_CHANNELS]; ///< channel uses spectral extension (chinspx)
141 int8_t spx_atten_code[AC3_MAX_CHANNELS]; ///< spx attenuation code (spxattencod)
142 int spx_src_start_freq; ///< spx start frequency bin
143 int spx_dst_end_freq; ///< spx end frequency bin
144 int spx_dst_start_freq; ///< spx starting frequency bin for copying (copystartmant)
145 ///< the copy region ends at the start of the spx region.
146 int num_spx_bands; ///< number of spx bands (nspxbnds)
147 uint8_t spx_band_sizes[SPX_MAX_BANDS]; ///< number of bins in each spx band
148 uint8_t first_spx_coords[AC3_MAX_CHANNELS]; ///< first spx coordinates states (firstspxcos)
149 INTFLOAT spx_noise_blend[AC3_MAX_CHANNELS][SPX_MAX_BANDS]; ///< spx noise blending factor (nblendfact)
150 INTFLOAT spx_signal_blend[AC3_MAX_CHANNELS][SPX_MAX_BANDS];///< spx signal blending factor (sblendfact)
151///@}
152
153///@name Adaptive hybrid transform
154 int channel_uses_aht[AC3_MAX_CHANNELS]; ///< channel AHT in use (chahtinu)
155 int pre_mantissa[AC3_MAX_CHANNELS][AC3_MAX_COEFS][AC3_MAX_BLOCKS]; ///< pre-IDCT mantissas
156///@}
157
158///@name Channel
159 int fbw_channels; ///< number of full-bandwidth channels
160 int channels; ///< number of total channels
161 int lfe_ch; ///< index of LFE channel
162 SHORTFLOAT downmix_coeffs[AC3_MAX_CHANNELS][2]; ///< stereo downmix coefficients
163 int downmixed; ///< indicates if coeffs are currently downmixed
164 int output_mode; ///< output channel configuration
165 int out_channels; ///< number of output channels
166///@}
167
168///@name Dynamic range
169 INTFLOAT dynamic_range[2]; ///< dynamic range
170 INTFLOAT drc_scale; ///< percentage of dynamic range compression to be applied
171 int heavy_compression; ///< apply heavy compression
172 INTFLOAT heavy_dynamic_range[2]; ///< heavy dynamic range compression
173///@}
174
175///@name Bandwidth
176 int start_freq[AC3_MAX_CHANNELS]; ///< start frequency bin (strtmant)
177 int end_freq[AC3_MAX_CHANNELS]; ///< end frequency bin (endmant)
178///@}
179
180///@name Rematrixing
181 int num_rematrixing_bands; ///< number of rematrixing bands (nrematbnd)
182 int rematrixing_flags[4]; ///< rematrixing flags (rematflg)
183///@}
184
185///@name Exponents
186 int num_exp_groups[AC3_MAX_CHANNELS]; ///< Number of exponent groups (nexpgrp)
187 int8_t dexps[AC3_MAX_CHANNELS][AC3_MAX_COEFS]; ///< decoded exponents
188 int exp_strategy[AC3_MAX_BLOCKS][AC3_MAX_CHANNELS]; ///< exponent strategies (expstr)
189///@}
190
191///@name Bit allocation
192 AC3BitAllocParameters bit_alloc_params; ///< bit allocation parameters
193 int first_cpl_leak; ///< first coupling leak state (firstcplleak)
194 int snr_offset[AC3_MAX_CHANNELS]; ///< signal-to-noise ratio offsets (snroffst)
195 int fast_gain[AC3_MAX_CHANNELS]; ///< fast gain values/SMR's (fgain)
196 uint8_t bap[AC3_MAX_CHANNELS][AC3_MAX_COEFS]; ///< bit allocation pointers
197 int16_t psd[AC3_MAX_CHANNELS][AC3_MAX_COEFS]; ///< scaled exponents
198 int16_t band_psd[AC3_MAX_CHANNELS][AC3_CRITICAL_BANDS]; ///< interpolated exponents
199 int16_t mask[AC3_MAX_CHANNELS][AC3_CRITICAL_BANDS]; ///< masking curve values
200 int dba_mode[AC3_MAX_CHANNELS]; ///< delta bit allocation mode
201 int dba_nsegs[AC3_MAX_CHANNELS]; ///< number of delta segments
202 uint8_t dba_offsets[AC3_MAX_CHANNELS][8]; ///< delta segment offsets
203 uint8_t dba_lengths[AC3_MAX_CHANNELS][8]; ///< delta segment lengths
204 uint8_t dba_values[AC3_MAX_CHANNELS][8]; ///< delta values for each segment
205///@}
206
207///@name Zero-mantissa dithering
208 int dither_flag[AC3_MAX_CHANNELS]; ///< dither flags (dithflg)
209 AVLFG dith_state; ///< for dither generation
210///@}
211
212///@name IMDCT
213 int block_switch[AC3_MAX_CHANNELS]; ///< block switch flags (blksw)
214 FFTContext imdct_512; ///< for 512 sample IMDCT
215 FFTContext imdct_256; ///< for 256 sample IMDCT
216///@}
217
218///@name Optimization
219 BswapDSPContext bdsp;
220#if USE_FIXED
221 AVFixedDSPContext *fdsp;
222#else
223 AVFloatDSPContext *fdsp;
224#endif
225 AC3DSPContext ac3dsp;
226 FmtConvertContext fmt_conv; ///< optimized conversion functions
227///@}
228
229 SHORTFLOAT *outptr[AC3_MAX_CHANNELS];
230 INTFLOAT *xcfptr[AC3_MAX_CHANNELS];
231 INTFLOAT *dlyptr[AC3_MAX_CHANNELS];
232
233///@name Aligned arrays
234 DECLARE_ALIGNED(16, int, fixed_coeffs)[AC3_MAX_CHANNELS][AC3_MAX_COEFS]; ///< fixed-point transform coefficients
235 DECLARE_ALIGNED(32, INTFLOAT, transform_coeffs)[AC3_MAX_CHANNELS][AC3_MAX_COEFS]; ///< transform coefficients
236 DECLARE_ALIGNED(32, INTFLOAT, delay)[AC3_MAX_CHANNELS][AC3_BLOCK_SIZE]; ///< delay - added to the next block

240 DECLARE_ALIGNED(32, uint8_t, input_buffer)[AC3_FRAME_BUFFER_SIZE + AV_INPUT_BUFFER_PADDING_SIZE]; ///< temp buffer to prevent overread
241///@}
