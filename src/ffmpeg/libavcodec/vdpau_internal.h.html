35/** Extract VdpVideoSurface from an AVFrame */
36static inline uintptr_t ff_vdpau_get_surface_id(AVFrame *pic)
37{
38 return (uintptr_t)pic->data[3];
39}

41union VDPAUPictureInfo {
42 VdpPictureInfoH264 h264;
43 VdpPictureInfoMPEG1Or2 mpeg;
44 VdpPictureInfoVC1 vc1;
45 VdpPictureInfoMPEG4Part2 mpeg4;
46#ifdef VDP_DECODER_PROFILE_H264_HIGH_444_PREDICTIVE
47 VdpPictureInfoH264Predictive h264_predictive;
48#endif
49#ifdef VDP_DECODER_PROFILE_HEVC_MAIN
50 VdpPictureInfoHEVC hevc;
51#endif
52};

54typedef struct VDPAUHWContext {
55 AVVDPAUContext context;
56 VdpDevice device;
57 VdpGetProcAddress *get_proc_address;
58 char reset;
59 unsigned char flags;
60} VDPAUHWContext;
61
62typedef struct VDPAUContext {
63 /**
64 * VDPAU device handle
65 */
66 VdpDevice device;
67
68 /**
69 * VDPAU decoder handle
70 */
71 VdpDecoder decoder;
72
73 /**
74 * VDPAU device driver
75 */
76 VdpGetProcAddress *get_proc_address;
77
78 /**
79 * VDPAU decoder render callback
80 */
81 VdpDecoderRender *render;
82
83 uint32_t width;
84 uint32_t height;
85} VDPAUContext;
86
87struct vdpau_picture_context {
88 /**
89 * VDPAU picture information.
90 */
91 union VDPAUPictureInfo info;
92
93 /**
94 * Allocated size of the bitstream_buffers table.
95 */
96 int bitstream_buffers_allocated;
97
98 /**
99 * Useful bitstream buffers in the bitstream buffers table.
100 */
101 int bitstream_buffers_used;
102
103 /**
104 * Table of bitstream buffers.
105 */
106 VdpBitstreamBuffer *bitstream_buffers;
107};
