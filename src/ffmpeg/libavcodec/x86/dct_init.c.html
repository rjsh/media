19#include "config.h"
20#include "libavutil/attributes.h"
21#include "libavutil/cpu.h"
22#include "libavutil/x86/cpu.h"
23#include "libavcodec/dct.h"
24
25void ff_dct32_float_sse(FFTSample *out, const FFTSample *in);
26void ff_dct32_float_sse2(FFTSample *out, const FFTSample *in);
27void ff_dct32_float_avx(FFTSample *out, const FFTSample *in);
28
29av_cold void ff_dct_init_x86(DCTContext *s)
30{
31 int cpu_flags = av_get_cpu_flags();
32
33#if ARCH_X86_32
34 if (EXTERNAL_SSE(cpu_flags))
35 s->dct32 = ff_dct32_float_sse;
36#endif
37 if (EXTERNAL_SSE2(cpu_flags))
38 s->dct32 = ff_dct32_float_sse2;
39 if (EXTERNAL_AVX_FAST(cpu_flags))
40 s->dct32 = ff_dct32_float_avx;
41}
