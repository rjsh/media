32void ff_vector_clip_int32_mmx(int32_t *dst, const int32_t *src,
33 int32_t min, int32_t max, unsigned int len);

43av_cold void ff_audiodsp_init_x86(AudioDSPContext *c)
44{
45 int cpu_flags = av_get_cpu_flags();
46
47 if (EXTERNAL_MMX(cpu_flags))
48 c->vector_clip_int32 = ff_vector_clip_int32_mmx;
49
50 if (EXTERNAL_MMXEXT(cpu_flags))
51 c->scalarproduct_int16 = ff_scalarproduct_int16_mmxext;
52
53 if (EXTERNAL_SSE(cpu_flags))
54 c->vector_clipf = ff_vector_clipf_sse;
55
56 if (EXTERNAL_SSE2(cpu_flags)) {
57 c->scalarproduct_int16 = ff_scalarproduct_int16_sse2;
58 if (cpu_flags & AV_CPU_FLAG_ATOM)
59 c->vector_clip_int32 = ff_vector_clip_int32_int_sse2;
60 else
61 c->vector_clip_int32 = ff_vector_clip_int32_sse2;
62 }
63
64 if (EXTERNAL_SSE4(cpu_flags))
65 c->vector_clip_int32 = ff_vector_clip_int32_sse4;
66}
