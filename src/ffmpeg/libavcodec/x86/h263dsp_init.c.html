21#include <stdint.h>
22
23#include "libavutil/attributes.h"
24#include "libavutil/cpu.h"
25#include "libavutil/x86/cpu.h"
26#include "libavcodec/h263dsp.h"
27
28void ff_h263_h_loop_filter_mmx(uint8_t *src, int stride, int qscale);
29void ff_h263_v_loop_filter_mmx(uint8_t *src, int stride, int qscale);
30
31av_cold void ff_h263dsp_init_x86(H263DSPContext *c)
32{
33 int cpu_flags = av_get_cpu_flags();
34
35 if (EXTERNAL_MMX(cpu_flags)) {
36 c->h263_h_loop_filter = ff_h263_h_loop_filter_mmx;
37 c->h263_v_loop_filter = ff_h263_v_loop_filter_mmx;
38 }
39}
