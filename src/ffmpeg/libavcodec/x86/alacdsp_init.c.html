19#include "libavutil/attributes.h"
20#include "libavutil/x86/cpu.h"

22#include "config.h"

24void ff_alac_decorrelate_stereo_sse4(int32_t *buffer[2], int nb_samples,
25 int decorr_shift, int decorr_left_weight);
26void ff_alac_append_extra_bits_stereo_sse2(int32_t *buffer[2], int32_t *extra_bits_buffer[2],
27 int extra_bits, int channels, int nb_samples);
28void ff_alac_append_extra_bits_mono_sse2(int32_t *buffer[2], int32_t *extra_bits_buffer[2],
29 int extra_bits, int channels, int nb_samples);
30
31av_cold void ff_alacdsp_init_x86(ALACDSPContext *c)
32{
33#if HAVE_YASM
34 int cpu_flags = av_get_cpu_flags();
35
36 if (EXTERNAL_SSE2(cpu_flags)) {
37 c->append_extra_bits[0] = ff_alac_append_extra_bits_mono_sse2;
38 c->append_extra_bits[1] = ff_alac_append_extra_bits_stereo_sse2;
39 }
40 if (EXTERNAL_SSE4(cpu_flags)) {
41 c->decorrelate_stereo = ff_alac_decorrelate_stereo_sse4;
42 }
43#endif /* HAVE_YASM */
44}
