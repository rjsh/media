180#define SET_HPEL_FUNCS_EXT(PFX, IDX, SIZE, CPU) \
181 if (HAVE_MMX_EXTERNAL) \
182 c->PFX ## _pixels_tab IDX [0] = PFX ## _pixels ## SIZE ## _ ## CPU;
183
184#if HAVE_MMX_INLINE
185#define SET_HPEL_FUNCS(PFX, IDX, SIZE, CPU) \
186 do { \
187 SET_HPEL_FUNCS_EXT(PFX, IDX, SIZE, CPU) \
188 c->PFX ## _pixels_tab IDX [1] = PFX ## _pixels ## SIZE ## _x2_ ## CPU; \
189 c->PFX ## _pixels_tab IDX [2] = PFX ## _pixels ## SIZE ## _y2_ ## CPU; \
190 c->PFX ## _pixels_tab IDX [3] = PFX ## _pixels ## SIZE ## _xy2_ ## CPU; \
191 } while (0)
192#else
193#define SET_HPEL_FUNCS(PFX, IDX, SIZE, CPU) \
194 do { \
195 SET_HPEL_FUNCS_EXT(PFX, IDX, SIZE, CPU) \
196 } while (0)
197#endif

199static void hpeldsp_init_mmx(HpelDSPContext *c, int flags, int cpu_flags)
200{
201 SET_HPEL_FUNCS(put, [0], 16, mmx);
202 SET_HPEL_FUNCS(put_no_rnd, [0], 16, mmx);
203 SET_HPEL_FUNCS(avg, [0], 16, mmx);
204 SET_HPEL_FUNCS(avg_no_rnd, , 16, mmx);
205 SET_HPEL_FUNCS(put, [1], 8, mmx);
206 SET_HPEL_FUNCS(put_no_rnd, [1], 8, mmx);
207 if (HAVE_MMX_EXTERNAL) {
208 c->avg_pixels_tab[1][0] = ff_avg_pixels8_mmx;
209 c->avg_pixels_tab[1][1] = ff_avg_pixels8_x2_mmx;
210 }
211#if HAVE_MMX_INLINE
212 c->avg_pixels_tab[1][2] = avg_pixels8_y2_mmx;
213 c->avg_pixels_tab[1][3] = ff_avg_pixels8_xy2_mmx;
214#endif
215}

217static void hpeldsp_init_mmxext(HpelDSPContext *c, int flags, int cpu_flags)
218{
219#if HAVE_MMXEXT_EXTERNAL
220 c->put_pixels_tab[0][1] = ff_put_pixels16_x2_mmxext;
221 c->put_pixels_tab[0][2] = put_pixels16_y2_mmxext;
222
223 c->avg_pixels_tab[0][0] = avg_pixels16_mmxext;
224 c->avg_pixels_tab[0][1] = avg_pixels16_x2_mmxext;
225 c->avg_pixels_tab[0][2] = avg_pixels16_y2_mmxext;
226 c->avg_pixels_tab[0][3] = avg_pixels16_xy2_mmxext;
227
228 c->put_pixels_tab[1][1] = ff_put_pixels8_x2_mmxext;
229 c->put_pixels_tab[1][2] = ff_put_pixels8_y2_mmxext;
230
231 c->avg_pixels_tab[1][0] = ff_avg_pixels8_mmxext;
232 c->avg_pixels_tab[1][1] = ff_avg_pixels8_x2_mmxext;
233 c->avg_pixels_tab[1][2] = ff_avg_pixels8_y2_mmxext;
234 c->avg_pixels_tab[1][3] = ff_avg_pixels8_xy2_mmxext;
235
236 if (!(flags & AV_CODEC_FLAG_BITEXACT)) {
237 c->put_no_rnd_pixels_tab[0][1] = put_no_rnd_pixels16_x2_mmxext;
238 c->put_no_rnd_pixels_tab[0][2] = put_no_rnd_pixels16_y2_mmxext;
239 c->put_no_rnd_pixels_tab[1][1] = ff_put_no_rnd_pixels8_x2_mmxext;
240 c->put_no_rnd_pixels_tab[1][2] = ff_put_no_rnd_pixels8_y2_mmxext;
241
242 c->avg_pixels_tab[0][3] = avg_approx_pixels16_xy2_mmxext;
243 c->avg_pixels_tab[1][3] = ff_avg_approx_pixels8_xy2_mmxext;
244 }
245
246 if (CONFIG_VP3_DECODER && flags & AV_CODEC_FLAG_BITEXACT) {
247 c->put_no_rnd_pixels_tab[1][1] = ff_put_no_rnd_pixels8_x2_exact_mmxext;
248 c->put_no_rnd_pixels_tab[1][2] = ff_put_no_rnd_pixels8_y2_exact_mmxext;
249 }
250#endif /* HAVE_MMXEXT_EXTERNAL */
251}

317av_cold void ff_hpeldsp_init_x86(HpelDSPContext *c, int flags)
318{
319 int cpu_flags = av_get_cpu_flags();
320
321 if (INLINE_MMX(cpu_flags))
322 hpeldsp_init_mmx(c, flags, cpu_flags);
323
324 if (EXTERNAL_AMD3DNOW(cpu_flags))
325 hpeldsp_init_3dnow(c, flags, cpu_flags);
326
327 if (EXTERNAL_MMXEXT(cpu_flags))
328 hpeldsp_init_mmxext(c, flags, cpu_flags);
329
330 if (EXTERNAL_SSE2(cpu_flags))
331 hpeldsp_init_sse2(c, flags, cpu_flags);
332
333 if (EXTERNAL_SSSE3(cpu_flags))
334 hpeldsp_init_ssse3(c, flags, cpu_flags);
335}
