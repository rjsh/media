21#include "libavutil/attributes.h"
22#include "libavutil/cpu.h"
23#include "libavutil/x86/cpu.h"
24#include "libavcodec/pixblockdsp.h"
25
26void ff_get_pixels_mmx(int16_t *block, const uint8_t *pixels, ptrdiff_t line_size);
27void ff_get_pixels_sse2(int16_t *block, const uint8_t *pixels, ptrdiff_t line_size);
28void ff_diff_pixels_mmx(int16_t *block, const uint8_t *s1, const uint8_t *s2,
29 int stride);
30void ff_diff_pixels_sse2(int16_t *block, const uint8_t *s1, const uint8_t *s2,
31 int stride);
32
33av_cold void ff_pixblockdsp_init_x86(PixblockDSPContext *c,
34 AVCodecContext *avctx,
35 unsigned high_bit_depth)
36{
37 int cpu_flags = av_get_cpu_flags();
38
39 if (EXTERNAL_MMX(cpu_flags)) {
40 if (!high_bit_depth)
41 c->get_pixels = ff_get_pixels_mmx;
42 c->diff_pixels = ff_diff_pixels_mmx;
43 }
44
45 if (EXTERNAL_SSE2(cpu_flags)) {
46 if (!high_bit_depth)
47 c->get_pixels = ff_get_pixels_sse2;
48 c->diff_pixels = ff_diff_pixels_sse2;
49 }
50}
