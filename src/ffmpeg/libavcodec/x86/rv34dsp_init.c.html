22#include "libavutil/attributes.h"
23#include "libavutil/cpu.h"
24#include "libavutil/x86/cpu.h"
25#include "libavcodec/rv34dsp.h"
26
27void ff_rv34_idct_dc_mmxext(int16_t *block);
28void ff_rv34_idct_dc_noround_mmxext(int16_t *block);
29void ff_rv34_idct_dc_add_mmx(uint8_t *dst, ptrdiff_t stride, int dc);
30void ff_rv34_idct_dc_add_sse4(uint8_t *dst, ptrdiff_t stride, int dc);
31void ff_rv34_idct_add_mmxext(uint8_t *dst, ptrdiff_t stride, int16_t *block);
32
33av_cold void ff_rv34dsp_init_x86(RV34DSPContext* c)
34{
35 int cpu_flags = av_get_cpu_flags();
36
37 if (EXTERNAL_MMX(cpu_flags))
38 c->rv34_idct_dc_add = ff_rv34_idct_dc_add_mmx;
39 if (EXTERNAL_MMXEXT(cpu_flags)) {
40 c->rv34_inv_transform_dc = ff_rv34_idct_dc_noround_mmxext;
41 c->rv34_idct_add = ff_rv34_idct_add_mmxext;
42 }
43 if (EXTERNAL_SSE4(cpu_flags))
44 c->rv34_idct_dc_add = ff_rv34_idct_dc_add_sse4;
45}
