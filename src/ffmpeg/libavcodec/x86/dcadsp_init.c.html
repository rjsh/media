24#define LFE_FIR_FLOAT_FUNC(opt) \
25void ff_lfe_fir0_float_##opt(float *pcm_samples, int32_t *lfe_samples, \
26 const float *filter_coeff, ptrdiff_t npcmblocks); \
27void ff_lfe_fir1_float_##opt(float *pcm_samples, int32_t *lfe_samples, \
28 const float *filter_coeff, ptrdiff_t npcmblocks);
29
30LFE_FIR_FLOAT_FUNC(sse)
31LFE_FIR_FLOAT_FUNC(sse2)
32LFE_FIR_FLOAT_FUNC(sse3)
33LFE_FIR_FLOAT_FUNC(avx)
34LFE_FIR_FLOAT_FUNC(fma3)
35
36av_cold void ff_dcadsp_init_x86(DCADSPContext *s)
37{
38 int cpu_flags = av_get_cpu_flags();
39
40 if (ARCH_X86_32 && EXTERNAL_SSE(cpu_flags))
41 s->lfe_fir_float[0] = ff_lfe_fir0_float_sse;
42 if (EXTERNAL_SSE2(cpu_flags))
43 s->lfe_fir_float[0] = ff_lfe_fir0_float_sse2;
44 if (EXTERNAL_SSE3(cpu_flags))
45 s->lfe_fir_float[1] = ff_lfe_fir1_float_sse3;
46 if (EXTERNAL_AVX(cpu_flags)) {
47 s->lfe_fir_float[0] = ff_lfe_fir0_float_avx;
48 s->lfe_fir_float[1] = ff_lfe_fir1_float_avx;
49 }
50 if (EXTERNAL_FMA3(cpu_flags))
51 s->lfe_fir_float[0] = ff_lfe_fir0_float_fma3;
52}
