21#include "libavcodec/takdsp.h"
22#include "libavutil/x86/cpu.h"
23#include "config.h"
24
25void ff_tak_decorrelate_ls_sse2(int32_t *p1, int32_t *p2, int length);
26void ff_tak_decorrelate_sr_sse2(int32_t *p1, int32_t *p2, int length);
27void ff_tak_decorrelate_sm_sse2(int32_t *p1, int32_t *p2, int length);
28void ff_tak_decorrelate_sf_sse4(int32_t *p1, int32_t *p2, int length, int dshift, int dfactor);
29
30av_cold void ff_takdsp_init_x86(TAKDSPContext *c)
31{
32#if HAVE_YASM
33 int cpu_flags = av_get_cpu_flags();
34
35 if (EXTERNAL_SSE2(cpu_flags)) {
36 c->decorrelate_ls = ff_tak_decorrelate_ls_sse2;
37 c->decorrelate_sr = ff_tak_decorrelate_sr_sse2;
38 c->decorrelate_sm = ff_tak_decorrelate_sm_sse2;
39 }
40
41 if (EXTERNAL_SSE4(cpu_flags)) {
42 c->decorrelate_sf = ff_tak_decorrelate_sf_sse4;
43 }
44#endif
45}
