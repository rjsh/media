22#include "libavutil/attributes.h"
23#include "libavutil/common.h"
24#include "libavutil/x86/cpu.h"
25#include "libavcodec/pngdsp.h"
26
27void ff_add_png_paeth_prediction_mmxext(uint8_t *dst, uint8_t *src,
28 uint8_t *top, int w, int bpp);
29void ff_add_png_paeth_prediction_ssse3(uint8_t *dst, uint8_t *src,
30 uint8_t *top, int w, int bpp);
31void ff_add_bytes_l2_mmx (uint8_t *dst, uint8_t *src1,
32 uint8_t *src2, int w);
33void ff_add_bytes_l2_sse2(uint8_t *dst, uint8_t *src1,
34 uint8_t *src2, int w);
35
36av_cold void ff_pngdsp_init_x86(PNGDSPContext *dsp)
37{
38 int cpu_flags = av_get_cpu_flags();
39
40#if ARCH_X86_32
41 if (EXTERNAL_MMX(cpu_flags))
42 dsp->add_bytes_l2 = ff_add_bytes_l2_mmx;
43#endif
44 if (EXTERNAL_MMXEXT(cpu_flags))
45 dsp->add_paeth_prediction = ff_add_png_paeth_prediction_mmxext;
46 if (EXTERNAL_SSE2(cpu_flags))
47 dsp->add_bytes_l2 = ff_add_bytes_l2_sse2;
48 if (EXTERNAL_SSSE3(cpu_flags))
49 dsp->add_paeth_prediction = ff_add_png_paeth_prediction_ssse3;
50}
