217av_cold void ff_vp9dsp_init_x86(VP9DSPContext *dsp, int bpp, int bitexact)
218{
219#if HAVE_YASM
220 int cpu_flags;
221
222 if (bpp == 10) {
223 ff_vp9dsp_init_10bpp_x86(dsp, bitexact);
224 return;
225 } else if (bpp == 12) {
226 ff_vp9dsp_init_12bpp_x86(dsp, bitexact);
227 return;
228 }
229
230 cpu_flags = av_get_cpu_flags();
231
232#define init_lpf(opt) do { \
233 dsp->loop_filter_16[0] = ff_vp9_loop_filter_h_16_16_##opt; \
234 dsp->loop_filter_16[1] = ff_vp9_loop_filter_v_16_16_##opt; \
235 dsp->loop_filter_mix2[0][0][0] = ff_vp9_loop_filter_h_44_16_##opt; \
236 dsp->loop_filter_mix2[0][0][1] = ff_vp9_loop_filter_v_44_16_##opt; \
237 dsp->loop_filter_mix2[0][1][0] = ff_vp9_loop_filter_h_48_16_##opt; \
238 dsp->loop_filter_mix2[0][1][1] = ff_vp9_loop_filter_v_48_16_##opt; \
239 dsp->loop_filter_mix2[1][0][0] = ff_vp9_loop_filter_h_84_16_##opt; \
240 dsp->loop_filter_mix2[1][0][1] = ff_vp9_loop_filter_v_84_16_##opt; \
241 dsp->loop_filter_mix2[1][1][0] = ff_vp9_loop_filter_h_88_16_##opt; \
242 dsp->loop_filter_mix2[1][1][1] = ff_vp9_loop_filter_v_88_16_##opt; \
243} while (0)
244
245#define init_ipred(sz, opt, t, e) \
246 dsp->intra_pred[TX_##sz##X##sz][e##_PRED] = ff_vp9_ipred_##t##_##sz##x##sz##_##opt
247
248#define ff_vp9_ipred_hd_4x4_ssse3 ff_vp9_ipred_hd_4x4_mmxext
249#define ff_vp9_ipred_vl_4x4_ssse3 ff_vp9_ipred_vl_4x4_mmxext
250#define init_dir_tm_ipred(sz, opt) do { \
251 init_ipred(sz, opt, dl, DIAG_DOWN_LEFT); \
252 init_ipred(sz, opt, dr, DIAG_DOWN_RIGHT); \
253 init_ipred(sz, opt, hd, HOR_DOWN); \
254 init_ipred(sz, opt, vl, VERT_LEFT); \
255 init_ipred(sz, opt, hu, HOR_UP); \
256 init_ipred(sz, opt, tm, TM_VP8); \
257 init_ipred(sz, opt, vr, VERT_RIGHT); \
258} while (0)
259#define init_dir_tm_h_ipred(sz, opt) do { \
260 init_dir_tm_ipred(sz, opt); \
261 init_ipred(sz, opt, h, HOR); \
262} while (0)
263#define init_dc_ipred(sz, opt) do { \
264 init_ipred(sz, opt, dc, DC); \
265 init_ipred(sz, opt, dc_left, LEFT_DC); \
266 init_ipred(sz, opt, dc_top, TOP_DC); \
267} while (0)
268#define init_all_ipred(sz, opt) do { \
269 init_dc_ipred(sz, opt); \
270 init_dir_tm_h_ipred(sz, opt); \
271} while (0)
272
273 if (EXTERNAL_MMX(cpu_flags)) {
274 init_fpel_func(4, 0, 4, put, , mmx);
275 init_fpel_func(3, 0, 8, put, , mmx);
276 if (!bitexact) {
277 dsp->itxfm_add[4 /* lossless */][DCT_DCT] =
278 dsp->itxfm_add[4 /* lossless */][ADST_DCT] =
279 dsp->itxfm_add[4 /* lossless */][DCT_ADST] =
280 dsp->itxfm_add[4 /* lossless */][ADST_ADST] = ff_vp9_iwht_iwht_4x4_add_mmx;
281 }
282 init_ipred(8, mmx, v, VERT);
283 }
284
285 if (EXTERNAL_MMXEXT(cpu_flags)) {
286 dsp->loop_filter_8[0][0] = ff_vp9_loop_filter_h_4_8_mmxext;
287 dsp->loop_filter_8[0][1] = ff_vp9_loop_filter_v_4_8_mmxext;
288 dsp->loop_filter_8[1][0] = ff_vp9_loop_filter_h_8_8_mmxext;
289 dsp->loop_filter_8[1][1] = ff_vp9_loop_filter_v_8_8_mmxext;
290 init_subpel2(4, 0, 4, put, 8, mmxext);
291 init_subpel2(4, 1, 4, avg, 8, mmxext);
292 init_fpel_func(4, 1, 4, avg, _8, mmxext);
293 init_fpel_func(3, 1, 8, avg, _8, mmxext);
294 dsp->itxfm_add[TX_4X4][DCT_DCT] = ff_vp9_idct_idct_4x4_add_mmxext;
295 init_dc_ipred(4, mmxext);
296 init_dc_ipred(8, mmxext);
297 init_dir_tm_ipred(4, mmxext);
298 }
299
300 if (EXTERNAL_SSE(cpu_flags)) {
301 init_fpel_func(2, 0, 16, put, , sse);
302 init_fpel_func(1, 0, 32, put, , sse);
303 init_fpel_func(0, 0, 64, put, , sse);
304 init_ipred(16, sse, v, VERT);
305 init_ipred(32, sse, v, VERT);
306 }
307
308 if (EXTERNAL_SSE2(cpu_flags)) {
309 init_subpel3_8to64(0, put, 8, sse2);
310 init_subpel3_8to64(1, avg, 8, sse2);
311 init_fpel_func(2, 1, 16, avg, _8, sse2);
312 init_fpel_func(1, 1, 32, avg, _8, sse2);
313 init_fpel_func(0, 1, 64, avg, _8, sse2);
314 init_lpf(sse2);
315 dsp->itxfm_add[TX_4X4][ADST_DCT] = ff_vp9_idct_iadst_4x4_add_sse2;
316 dsp->itxfm_add[TX_4X4][DCT_ADST] = ff_vp9_iadst_idct_4x4_add_sse2;
317 dsp->itxfm_add[TX_4X4][ADST_ADST] = ff_vp9_iadst_iadst_4x4_add_sse2;
318 dsp->itxfm_add[TX_8X8][DCT_DCT] = ff_vp9_idct_idct_8x8_add_sse2;
319 dsp->itxfm_add[TX_8X8][ADST_DCT] = ff_vp9_idct_iadst_8x8_add_sse2;
320 dsp->itxfm_add[TX_8X8][DCT_ADST] = ff_vp9_iadst_idct_8x8_add_sse2;
321 dsp->itxfm_add[TX_8X8][ADST_ADST] = ff_vp9_iadst_iadst_8x8_add_sse2;
322 dsp->itxfm_add[TX_16X16][DCT_DCT] = ff_vp9_idct_idct_16x16_add_sse2;
323 dsp->itxfm_add[TX_16X16][ADST_DCT] = ff_vp9_idct_iadst_16x16_add_sse2;
324 dsp->itxfm_add[TX_16X16][DCT_ADST] = ff_vp9_iadst_idct_16x16_add_sse2;
325 dsp->itxfm_add[TX_16X16][ADST_ADST] = ff_vp9_iadst_iadst_16x16_add_sse2;
326 dsp->itxfm_add[TX_32X32][ADST_ADST] =
327 dsp->itxfm_add[TX_32X32][ADST_DCT] =
328 dsp->itxfm_add[TX_32X32][DCT_ADST] =
329 dsp->itxfm_add[TX_32X32][DCT_DCT] = ff_vp9_idct_idct_32x32_add_sse2;
330 init_dc_ipred(16, sse2);
331 init_dc_ipred(32, sse2);
332 init_dir_tm_h_ipred(8, sse2);
333 init_dir_tm_h_ipred(16, sse2);
334 init_dir_tm_h_ipred(32, sse2);
335 init_ipred(4, sse2, h, HOR);
336 }
337
338 if (EXTERNAL_SSSE3(cpu_flags)) {
339 init_subpel3(0, put, 8, ssse3);
340 init_subpel3(1, avg, 8, ssse3);
341 dsp->itxfm_add[TX_4X4][DCT_DCT] = ff_vp9_idct_idct_4x4_add_ssse3;
342 dsp->itxfm_add[TX_4X4][ADST_DCT] = ff_vp9_idct_iadst_4x4_add_ssse3;
343 dsp->itxfm_add[TX_4X4][DCT_ADST] = ff_vp9_iadst_idct_4x4_add_ssse3;
344 dsp->itxfm_add[TX_4X4][ADST_ADST] = ff_vp9_iadst_iadst_4x4_add_ssse3;
345 dsp->itxfm_add[TX_8X8][DCT_DCT] = ff_vp9_idct_idct_8x8_add_ssse3;
346 dsp->itxfm_add[TX_8X8][ADST_DCT] = ff_vp9_idct_iadst_8x8_add_ssse3;
347 dsp->itxfm_add[TX_8X8][DCT_ADST] = ff_vp9_iadst_idct_8x8_add_ssse3;
348 dsp->itxfm_add[TX_8X8][ADST_ADST] = ff_vp9_iadst_iadst_8x8_add_ssse3;
349 dsp->itxfm_add[TX_16X16][DCT_DCT] = ff_vp9_idct_idct_16x16_add_ssse3;
350 dsp->itxfm_add[TX_16X16][ADST_DCT] = ff_vp9_idct_iadst_16x16_add_ssse3;
351 dsp->itxfm_add[TX_16X16][DCT_ADST] = ff_vp9_iadst_idct_16x16_add_ssse3;
352 dsp->itxfm_add[TX_16X16][ADST_ADST] = ff_vp9_iadst_iadst_16x16_add_ssse3;
353 dsp->itxfm_add[TX_32X32][ADST_ADST] =
354 dsp->itxfm_add[TX_32X32][ADST_DCT] =
355 dsp->itxfm_add[TX_32X32][DCT_ADST] =
356 dsp->itxfm_add[TX_32X32][DCT_DCT] = ff_vp9_idct_idct_32x32_add_ssse3;
357 init_lpf(ssse3);
358 init_all_ipred(4, ssse3);
359 init_all_ipred(8, ssse3);
360 init_all_ipred(16, ssse3);
361 init_all_ipred(32, ssse3);
362 }
363
364 if (EXTERNAL_AVX(cpu_flags)) {
365 dsp->itxfm_add[TX_8X8][DCT_DCT] = ff_vp9_idct_idct_8x8_add_avx;
366 dsp->itxfm_add[TX_8X8][ADST_DCT] = ff_vp9_idct_iadst_8x8_add_avx;
367 dsp->itxfm_add[TX_8X8][DCT_ADST] = ff_vp9_iadst_idct_8x8_add_avx;
368 dsp->itxfm_add[TX_8X8][ADST_ADST] = ff_vp9_iadst_iadst_8x8_add_avx;
369 dsp->itxfm_add[TX_16X16][DCT_DCT] = ff_vp9_idct_idct_16x16_add_avx;
370 dsp->itxfm_add[TX_16X16][ADST_DCT] = ff_vp9_idct_iadst_16x16_add_avx;
371 dsp->itxfm_add[TX_16X16][DCT_ADST] = ff_vp9_iadst_idct_16x16_add_avx;
372 dsp->itxfm_add[TX_16X16][ADST_ADST] = ff_vp9_iadst_iadst_16x16_add_avx;
373 dsp->itxfm_add[TX_32X32][ADST_ADST] =
374 dsp->itxfm_add[TX_32X32][ADST_DCT] =
375 dsp->itxfm_add[TX_32X32][DCT_ADST] =
376 dsp->itxfm_add[TX_32X32][DCT_DCT] = ff_vp9_idct_idct_32x32_add_avx;
377 init_lpf(avx);
378 init_dir_tm_h_ipred(8, avx);
379 init_dir_tm_h_ipred(16, avx);
380 init_dir_tm_h_ipred(32, avx);
381 }
382 if (EXTERNAL_AVX_FAST(cpu_flags)) {
383 init_fpel_func(1, 0, 32, put, , avx);
384 init_fpel_func(0, 0, 64, put, , avx);
385 init_ipred(32, avx, v, VERT);
386 }
387
388 if (EXTERNAL_AVX2_FAST(cpu_flags)) {
389 init_fpel_func(1, 1, 32, avg, _8, avx2);
390 init_fpel_func(0, 1, 64, avg, _8, avx2);
391 if (ARCH_X86_64) {
392#if ARCH_X86_64 && HAVE_AVX2_EXTERNAL
393 dsp->itxfm_add[TX_16X16][DCT_DCT] = ff_vp9_idct_idct_16x16_add_avx2;
394 dsp->itxfm_add[TX_32X32][DCT_DCT] = ff_vp9_idct_idct_32x32_add_avx2;
395 init_subpel3_32_64(0, put, 8, avx2);
396 init_subpel3_32_64(1, avg, 8, avx2);
397#endif
398 }
399 init_dc_ipred(32, avx2);
400 init_ipred(32, avx2, h, HOR);
401 init_ipred(32, avx2, tm, TM_VP8);
402 }
