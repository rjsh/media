21#include "libavcodec/ttadsp.h"
22#include "libavutil/x86/cpu.h"
23#include "config.h"
24
25void ff_tta_filter_process_ssse3(int32_t *qm, int32_t *dx, int32_t *dl,
26 int32_t *error, int32_t *in, int32_t shift,
27 int32_t round);
28void ff_tta_filter_process_sse4(int32_t *qm, int32_t *dx, int32_t *dl,
29 int32_t *error, int32_t *in, int32_t shift,
30 int32_t round);
31
32av_cold void ff_ttadsp_init_x86(TTADSPContext *c)
33{
34#if HAVE_YASM
35 int cpu_flags = av_get_cpu_flags();
36
37 if (EXTERNAL_SSSE3(cpu_flags))
38 c->filter_process = ff_tta_filter_process_ssse3;
39 if (EXTERNAL_SSE4(cpu_flags))
40 c->filter_process = ff_tta_filter_process_sse4;
41#endif
42}
