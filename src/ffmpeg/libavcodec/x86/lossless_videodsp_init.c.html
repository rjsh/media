35void ff_llviddsp_init_x86(LLVidDSPContext *c, AVCodecContext *avctx)
36{
37 int cpu_flags = av_get_cpu_flags();
38 const AVPixFmtDescriptor *pix_desc = av_pix_fmt_desc_get(avctx->pix_fmt);
39
40 if (EXTERNAL_MMX(cpu_flags)) {
41 c->add_int16 = ff_add_int16_mmx;
42 c->diff_int16 = ff_diff_int16_mmx;
43 }
44
45 if (EXTERNAL_MMXEXT(cpu_flags) && pix_desc->comp[0].depth<16) {
46 c->add_hfyu_median_pred_int16 = ff_add_hfyu_median_pred_int16_mmxext;
47 c->sub_hfyu_median_pred_int16 = ff_sub_hfyu_median_pred_int16_mmxext;
48 }
49
50 if (EXTERNAL_SSE2(cpu_flags)) {
51 c->add_int16 = ff_add_int16_sse2;
52 c->diff_int16 = ff_diff_int16_sse2;
53 }
54
55 if (EXTERNAL_SSSE3(cpu_flags)) {
56 c->add_hfyu_left_pred_int16 = ff_add_hfyu_left_pred_int16_ssse3;
57 }
58
59 if (EXTERNAL_SSE4(cpu_flags)) {
60 c->add_hfyu_left_pred_int16 = ff_add_hfyu_left_pred_int16_sse4;
61 }
62}
