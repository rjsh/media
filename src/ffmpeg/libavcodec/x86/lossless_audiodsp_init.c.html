25 const int16_t *v3,

37
38av_cold void ff_llauddsp_init_x86(LLAudDSPContext *c)
39{
40#if HAVE_YASM
41 int cpu_flags = av_get_cpu_flags();
42
43 if (EXTERNAL_MMXEXT(cpu_flags))
44 c->scalarproduct_and_madd_int16 = ff_scalarproduct_and_madd_int16_mmxext;
45
46 if (EXTERNAL_SSE2(cpu_flags))
47 c->scalarproduct_and_madd_int16 = ff_scalarproduct_and_madd_int16_sse2;
48
49 if (EXTERNAL_SSSE3(cpu_flags) &&
50 !(cpu_flags & (AV_CPU_FLAG_SSE42 | AV_CPU_FLAG_3DNOW))) // cachesplit
51 c->scalarproduct_and_madd_int16 = ff_scalarproduct_and_madd_int16_ssse3;
52
53 if (EXTERNAL_SSE4(cpu_flags))
54 c->scalarproduct_and_madd_int32 = ff_scalarproduct_and_madd_int32_sse4;
55#endif
56}
