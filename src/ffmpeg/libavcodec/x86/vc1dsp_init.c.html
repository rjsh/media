36#define LOOP_FILTER(EXT) \
37void ff_vc1_v_loop_filter4_ ## EXT(uint8_t *src, int stride, int pq); \
38void ff_vc1_h_loop_filter4_ ## EXT(uint8_t *src, int stride, int pq); \
39void ff_vc1_v_loop_filter8_ ## EXT(uint8_t *src, int stride, int pq); \
40void ff_vc1_h_loop_filter8_ ## EXT(uint8_t *src, int stride, int pq); \
41\
42static void vc1_v_loop_filter16_ ## EXT(uint8_t *src, int stride, int pq) \
43{ \
44 ff_vc1_v_loop_filter8_ ## EXT(src, stride, pq); \
45 ff_vc1_v_loop_filter8_ ## EXT(src+8, stride, pq); \
46} \
47\
48static void vc1_h_loop_filter16_ ## EXT(uint8_t *src, int stride, int pq) \
49{ \
50 ff_vc1_h_loop_filter8_ ## EXT(src, stride, pq); \
51 ff_vc1_h_loop_filter8_ ## EXT(src+8*stride, stride, pq); \
52}

61static void vc1_h_loop_filter16_sse4(uint8_t *src, int stride, int pq)
62{
63 ff_vc1_h_loop_filter8_sse4(src, stride, pq);
64 ff_vc1_h_loop_filter8_sse4(src+8*stride, stride, pq);
65}

105av_cold void ff_vc1dsp_init_x86(VC1DSPContext *dsp)
106{
107 int cpu_flags = av_get_cpu_flags();
108
109 if (HAVE_6REGS && INLINE_MMX(cpu_flags) && EXTERNAL_MMX(cpu_flags))
110 ff_vc1dsp_init_mmx(dsp);
111
112 if (HAVE_6REGS && INLINE_MMXEXT(cpu_flags) && EXTERNAL_MMXEXT(cpu_flags))
113 ff_vc1dsp_init_mmxext(dsp);
114
115#define ASSIGN_LF(EXT) \
116 dsp->vc1_v_loop_filter4 = ff_vc1_v_loop_filter4_ ## EXT; \
117 dsp->vc1_h_loop_filter4 = ff_vc1_h_loop_filter4_ ## EXT; \
118 dsp->vc1_v_loop_filter8 = ff_vc1_v_loop_filter8_ ## EXT; \
119 dsp->vc1_h_loop_filter8 = ff_vc1_h_loop_filter8_ ## EXT; \
120 dsp->vc1_v_loop_filter16 = vc1_v_loop_filter16_ ## EXT; \
121 dsp->vc1_h_loop_filter16 = vc1_h_loop_filter16_ ## EXT
122
123#if HAVE_YASM
124 if (EXTERNAL_MMX(cpu_flags)) {
125 dsp->put_no_rnd_vc1_chroma_pixels_tab[0] = ff_put_vc1_chroma_mc8_nornd_mmx;
126
127 dsp->put_vc1_mspel_pixels_tab[1][0] = put_vc1_mspel_mc00_8_mmx;
128 dsp->put_vc1_mspel_pixels_tab[0][0] = put_vc1_mspel_mc00_16_mmx;
129 dsp->avg_vc1_mspel_pixels_tab[1][0] = avg_vc1_mspel_mc00_8_mmx;
130 dsp->avg_vc1_mspel_pixels_tab[0][0] = avg_vc1_mspel_mc00_16_mmx;
131 }
132 if (EXTERNAL_AMD3DNOW(cpu_flags)) {
133 dsp->avg_no_rnd_vc1_chroma_pixels_tab[0] = ff_avg_vc1_chroma_mc8_nornd_3dnow;
134 }
135 if (EXTERNAL_MMXEXT(cpu_flags)) {
136 ASSIGN_LF(mmxext);
137 dsp->avg_no_rnd_vc1_chroma_pixels_tab[0] = ff_avg_vc1_chroma_mc8_nornd_mmxext;
138
139 dsp->avg_vc1_mspel_pixels_tab[1][0] = avg_vc1_mspel_mc00_8_mmxext;
140 dsp->avg_vc1_mspel_pixels_tab[0][0] = avg_vc1_mspel_mc00_16_mmxext;
141
142 dsp->vc1_inv_trans_8x8_dc = ff_vc1_inv_trans_8x8_dc_mmxext;
143 dsp->vc1_inv_trans_4x8_dc = ff_vc1_inv_trans_4x8_dc_mmxext;
144 dsp->vc1_inv_trans_8x4_dc = ff_vc1_inv_trans_8x4_dc_mmxext;
145 dsp->vc1_inv_trans_4x4_dc = ff_vc1_inv_trans_4x4_dc_mmxext;
146 }
147 if (EXTERNAL_SSE2(cpu_flags)) {
148 dsp->vc1_v_loop_filter8 = ff_vc1_v_loop_filter8_sse2;
149 dsp->vc1_h_loop_filter8 = ff_vc1_h_loop_filter8_sse2;
150 dsp->vc1_v_loop_filter16 = vc1_v_loop_filter16_sse2;
151 dsp->vc1_h_loop_filter16 = vc1_h_loop_filter16_sse2;
152
153 dsp->put_vc1_mspel_pixels_tab[0][0] = put_vc1_mspel_mc00_16_sse2;
154 dsp->avg_vc1_mspel_pixels_tab[0][0] = avg_vc1_mspel_mc00_16_sse2;
155 }
156 if (EXTERNAL_SSSE3(cpu_flags)) {
157 ASSIGN_LF(ssse3);
158 dsp->put_no_rnd_vc1_chroma_pixels_tab[0] = ff_put_vc1_chroma_mc8_nornd_ssse3;
159 dsp->avg_no_rnd_vc1_chroma_pixels_tab[0] = ff_avg_vc1_chroma_mc8_nornd_ssse3;
160 }
161 if (EXTERNAL_SSE4(cpu_flags)) {
162 dsp->vc1_h_loop_filter8 = ff_vc1_h_loop_filter8_sse4;
163 dsp->vc1_h_loop_filter16 = vc1_h_loop_filter16_sse4;
164 }
165#endif /* HAVE_YASM */
166}
