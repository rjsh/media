28/* Input permutation for the simple_idct_mmx */
29static const uint8_t simple_mmx_permutation[64] = {
30 0x00, 0x08, 0x04, 0x09, 0x01, 0x0C, 0x05, 0x0D,
31 0x10, 0x18, 0x14, 0x19, 0x11, 0x1C, 0x15, 0x1D,
32 0x20, 0x28, 0x24, 0x29, 0x21, 0x2C, 0x25, 0x2D,
33 0x12, 0x1A, 0x16, 0x1B, 0x13, 0x1E, 0x17, 0x1F,
34 0x02, 0x0A, 0x06, 0x0B, 0x03, 0x0E, 0x07, 0x0F,
35 0x30, 0x38, 0x34, 0x39, 0x31, 0x3C, 0x35, 0x3D,
36 0x22, 0x2A, 0x26, 0x2B, 0x23, 0x2E, 0x27, 0x2F,
37 0x32, 0x3A, 0x36, 0x3B, 0x33, 0x3E, 0x37, 0x3F,
38};
39
40static const uint8_t idct_sse2_row_perm[8] = { 0, 4, 1, 5, 2, 6, 3, 7 };
41
42av_cold int ff_init_scantable_permutation_x86(uint8_t *idct_permutation,
43 enum idct_permutation_type perm_type)
44{
45 int i;
46
47 switch (perm_type) {
48 case FF_IDCT_PERM_SIMPLE:
49 for (i = 0; i < 64; i++)
50 idct_permutation[i] = simple_mmx_permutation[i];
51 return 1;
52 case FF_IDCT_PERM_SSE2:
53 for (i = 0; i < 64; i++)
54 idct_permutation[i] = (i & 0x38) | idct_sse2_row_perm[i & 7];
55 return 1;
56 }
57
58 return 0;
59}

61av_cold void ff_idctdsp_init_x86(IDCTDSPContext *c, AVCodecContext *avctx,
62 unsigned high_bit_depth)
63{
64 int cpu_flags = av_get_cpu_flags();
65
66 if (INLINE_MMX(cpu_flags)) {
67 if (!high_bit_depth &&
68 avctx->lowres == 0 &&
69 (avctx->idct_algo == FF_IDCT_AUTO ||
70 avctx->idct_algo == FF_IDCT_SIMPLEAUTO ||
71 avctx->idct_algo == FF_IDCT_SIMPLEMMX)) {
72 c->idct_put = ff_simple_idct_put_mmx;
73 c->idct_add = ff_simple_idct_add_mmx;
74 c->idct = ff_simple_idct_mmx;
75 c->perm_type = FF_IDCT_PERM_SIMPLE;
76 }
77 }
78 if (EXTERNAL_MMX(cpu_flags)) {
79 c->put_signed_pixels_clamped = ff_put_signed_pixels_clamped_mmx;
80 c->put_pixels_clamped = ff_put_pixels_clamped_mmx;
81 c->add_pixels_clamped = ff_add_pixels_clamped_mmx;
82 }
83 if (EXTERNAL_SSE2(cpu_flags)) {
84 c->put_signed_pixels_clamped = ff_put_signed_pixels_clamped_sse2;
85 c->put_pixels_clamped = ff_put_pixels_clamped_sse2;
86 c->add_pixels_clamped = ff_add_pixels_clamped_sse2;
87 }
88
89 if (ARCH_X86_64 && avctx->lowres == 0) {
90 if (avctx->bits_per_raw_sample == 10 &&
91 (avctx->idct_algo == FF_IDCT_AUTO ||
92 avctx->idct_algo == FF_IDCT_SIMPLEAUTO ||
93 avctx->idct_algo == FF_IDCT_SIMPLE)) {
94 if (EXTERNAL_SSE2(cpu_flags)) {
95 c->idct_put = ff_simple_idct10_put_sse2;
96 c->idct_add = NULL;
97 c->idct = ff_simple_idct10_sse2;
98 c->perm_type = FF_IDCT_PERM_TRANSPOSE;
99
100 }
101 if (EXTERNAL_AVX(cpu_flags)) {
102 c->idct_put = ff_simple_idct10_put_avx;
103 c->idct_add = NULL;
104 c->idct = ff_simple_idct10_avx;
105 c->perm_type = FF_IDCT_PERM_TRANSPOSE;
106 }
107 }
108
109 if (avctx->bits_per_raw_sample == 12 &&
110 (avctx->idct_algo == FF_IDCT_AUTO ||
111 avctx->idct_algo == FF_IDCT_SIMPLEMMX)) {
112 if (EXTERNAL_SSE2(cpu_flags)) {
113 c->idct_put = ff_simple_idct12_put_sse2;
114 c->idct_add = NULL;
115 c->idct = ff_simple_idct12_sse2;
116 c->perm_type = FF_IDCT_PERM_TRANSPOSE;
117 }
118 if (EXTERNAL_AVX(cpu_flags)) {
119 c->idct_put = ff_simple_idct12_put_avx;
120 c->idct_add = NULL;
121 c->idct = ff_simple_idct12_avx;
122 c->perm_type = FF_IDCT_PERM_TRANSPOSE;
123 }
124 }
125 }
