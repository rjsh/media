47av_cold void ff_vp3dsp_init_x86(VP3DSPContext *c, int flags)
48{
49 int cpu_flags = av_get_cpu_flags();
50
51 if (EXTERNAL_MMX(cpu_flags)) {
52 c->put_no_rnd_pixels_l2 = ff_put_vp_no_rnd_pixels8_l2_mmx;
53#if ARCH_X86_32
54 c->idct_put = ff_vp3_idct_put_mmx;
55 c->idct_add = ff_vp3_idct_add_mmx;
56#endif
57 }
58
59 if (EXTERNAL_MMXEXT(cpu_flags)) {
60 c->idct_dc_add = ff_vp3_idct_dc_add_mmxext;
61
62 if (!(flags & AV_CODEC_FLAG_BITEXACT)) {
63 c->v_loop_filter = ff_vp3_v_loop_filter_mmxext;
64 c->h_loop_filter = ff_vp3_h_loop_filter_mmxext;
65 }
66 }
67
68 if (EXTERNAL_SSE2(cpu_flags)) {
69 c->idct_put = ff_vp3_idct_put_sse2;
70 c->idct_add = ff_vp3_idct_add_sse2;
71 }
72}
