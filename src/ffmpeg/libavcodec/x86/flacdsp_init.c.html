32#define DECORRELATE_FUNCS(fmt, opt) \
33void ff_flac_decorrelate_ls_##fmt##_##opt(uint8_t **out, int32_t **in, int channels, \
34 int len, int shift); \
35void ff_flac_decorrelate_rs_##fmt##_##opt(uint8_t **out, int32_t **in, int channels, \
36 int len, int shift); \
37void ff_flac_decorrelate_ms_##fmt##_##opt(uint8_t **out, int32_t **in, int channels, \
38 int len, int shift); \
39void ff_flac_decorrelate_indep2_##fmt##_##opt(uint8_t **out, int32_t **in, int channels, \
40 int len, int shift); \
41void ff_flac_decorrelate_indep4_##fmt##_##opt(uint8_t **out, int32_t **in, int channels, \
42 int len, int shift); \
43void ff_flac_decorrelate_indep6_##fmt##_##opt(uint8_t **out, int32_t **in, int channels, \
44 int len, int shift); \
45void ff_flac_decorrelate_indep8_##fmt##_##opt(uint8_t **out, int32_t **in, int channels, \
46 int len, int shift)
47
48DECORRELATE_FUNCS(16, sse2);
49DECORRELATE_FUNCS(16, avx);
50DECORRELATE_FUNCS(32, sse2);
51DECORRELATE_FUNCS(32, avx);

53av_cold void ff_flacdsp_init_x86(FLACDSPContext *c, enum AVSampleFormat fmt, int channels,
54 int bps)
55{
56#if HAVE_YASM
57 int cpu_flags = av_get_cpu_flags();
58
59#if CONFIG_FLAC_DECODER
60 if (EXTERNAL_SSE2(cpu_flags)) {
61 if (fmt == AV_SAMPLE_FMT_S16) {
62 if (channels == 2)
63 c->decorrelate[0] = ff_flac_decorrelate_indep2_16_sse2;
64 else if (channels == 4)
65 c->decorrelate[0] = ff_flac_decorrelate_indep4_16_sse2;
66 else if (channels == 6)
67 c->decorrelate[0] = ff_flac_decorrelate_indep6_16_sse2;
68 else if (ARCH_X86_64 && channels == 8)
69 c->decorrelate[0] = ff_flac_decorrelate_indep8_16_sse2;
70 c->decorrelate[1] = ff_flac_decorrelate_ls_16_sse2;
71 c->decorrelate[2] = ff_flac_decorrelate_rs_16_sse2;
72 c->decorrelate[3] = ff_flac_decorrelate_ms_16_sse2;
73 } else if (fmt == AV_SAMPLE_FMT_S32) {
74 if (channels == 2)
75 c->decorrelate[0] = ff_flac_decorrelate_indep2_32_sse2;
76 else if (channels == 4)
77 c->decorrelate[0] = ff_flac_decorrelate_indep4_32_sse2;
78 else if (channels == 6)
79 c->decorrelate[0] = ff_flac_decorrelate_indep6_32_sse2;
80 else if (ARCH_X86_64 && channels == 8)
81 c->decorrelate[0] = ff_flac_decorrelate_indep8_32_sse2;
82 c->decorrelate[1] = ff_flac_decorrelate_ls_32_sse2;
83 c->decorrelate[2] = ff_flac_decorrelate_rs_32_sse2;
84 c->decorrelate[3] = ff_flac_decorrelate_ms_32_sse2;
85 }
86 }
87 if (EXTERNAL_SSE4(cpu_flags)) {
88 c->lpc32 = ff_flac_lpc_32_sse4;
89 }
90 if (EXTERNAL_AVX(cpu_flags)) {
91 if (fmt == AV_SAMPLE_FMT_S16) {
92 if (ARCH_X86_64 && channels == 8)
93 c->decorrelate[0] = ff_flac_decorrelate_indep8_16_avx;
94 } else if (fmt == AV_SAMPLE_FMT_S32) {
95 if (channels == 4)
96 c->decorrelate[0] = ff_flac_decorrelate_indep4_32_avx;
97 else if (channels == 6)
98 c->decorrelate[0] = ff_flac_decorrelate_indep6_32_avx;
99 else if (ARCH_X86_64 && channels == 8)
100 c->decorrelate[0] = ff_flac_decorrelate_indep8_32_avx;
101 }
102 }
103 if (EXTERNAL_XOP(cpu_flags)) {
104 c->lpc32 = ff_flac_lpc_32_xop;
105 }
106#endif
107
108#if CONFIG_FLAC_ENCODER
109 if (EXTERNAL_SSE4(cpu_flags)) {
110 if (CONFIG_GPL)
111 c->lpc16_encode = ff_flac_enc_lpc_16_sse4;
112 }
113#endif
114#endif /* HAVE_YASM */
115}
