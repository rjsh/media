21#include "config.h"
22#include "libavutil/attributes.h"
23#include "libavutil/cpu.h"
24#include "libavutil/x86/cpu.h"
25#include "libavcodec/vorbisdsp.h"
26
27void ff_vorbis_inverse_coupling_3dnow(float *mag, float *ang,
28 intptr_t blocksize);
29void ff_vorbis_inverse_coupling_sse(float *mag, float *ang,
30 intptr_t blocksize);
31
32av_cold void ff_vorbisdsp_init_x86(VorbisDSPContext *dsp)
33{
34 int cpu_flags = av_get_cpu_flags();
35
36#if ARCH_X86_32
37 if (EXTERNAL_AMD3DNOW(cpu_flags))
38 dsp->vorbis_inverse_coupling = ff_vorbis_inverse_coupling_3dnow;
39#endif /* ARCH_X86_32 */
40 if (EXTERNAL_SSE(cpu_flags))
41 dsp->vorbis_inverse_coupling = ff_vorbis_inverse_coupling_sse;
42}
