19#include "libavutil/x86/cpu.h"
20#include "libavcodec/v210enc.h"

22void ff_v210_planar_pack_8_ssse3(const uint8_t *y, const uint8_t *u,
23 const uint8_t *v, uint8_t *dst,
24 ptrdiff_t width);
25void ff_v210_planar_pack_8_avx(const uint8_t *y, const uint8_t *u,
26 const uint8_t *v, uint8_t *dst, ptrdiff_t width);
27void ff_v210_planar_pack_8_avx2(const uint8_t *y, const uint8_t *u,
28 const uint8_t *v, uint8_t *dst, ptrdiff_t width);
29void ff_v210_planar_pack_10_ssse3(const uint16_t *y, const uint16_t *u,
30 const uint16_t *v, uint8_t *dst,
31 ptrdiff_t width);
32void ff_v210_planar_pack_10_avx2(const uint16_t *y, const uint16_t *u,
33 const uint16_t *v, uint8_t *dst,
34 ptrdiff_t width);
35
36av_cold void ff_v210enc_init_x86(V210EncContext *s)
37{
38 int cpu_flags = av_get_cpu_flags();
39
40 if (EXTERNAL_SSSE3(cpu_flags)) {
41 s->pack_line_8 = ff_v210_planar_pack_8_ssse3;
42 s->pack_line_10 = ff_v210_planar_pack_10_ssse3;
43 }
44
45 if (EXTERNAL_AVX(cpu_flags))
46 s->pack_line_8 = ff_v210_planar_pack_8_avx;
47
48 if (EXTERNAL_AVX2(cpu_flags)) {
49 s->sample_factor_8 = 2;
50 s->pack_line_8 = ff_v210_planar_pack_8_avx2;
51 s->sample_factor_10 = 2;
52 s->pack_line_10 = ff_v210_planar_pack_10_avx2;
53 }
54}
