68av_cold void ff_vp9dsp_init_16bpp_x86(VP9DSPContext *dsp)
69{
70#if HAVE_YASM
71 int cpu_flags = av_get_cpu_flags();
72
73 if (EXTERNAL_MMX(cpu_flags)) {
74 init_fpel_func(4, 0, 8, put, , mmx);
75 init_ipred_func(v, VERT, 4, 16, mmx);
76 }
77
78 if (EXTERNAL_MMXEXT(cpu_flags)) {
79 init_fpel_func(4, 1, 8, avg, _16, mmxext);
80 init_ipred_func(h, HOR, 4, 16, mmxext);
81 init_ipred_func(dc, DC, 4, 16, mmxext);
82 init_ipred_func(dc_top, TOP_DC, 4, 16, mmxext);
83 init_ipred_func(dc_left, LEFT_DC, 4, 16, mmxext);
84 }
85
86 if (EXTERNAL_SSE(cpu_flags)) {
87 init_fpel_func(3, 0, 16, put, , sse);
88 init_fpel_func(2, 0, 32, put, , sse);
89 init_fpel_func(1, 0, 64, put, , sse);
90 init_fpel_func(0, 0, 128, put, , sse);
91 init_8_16_32_ipred_funcs(v, VERT, 16, sse);
92 }
93
94 if (EXTERNAL_SSE2(cpu_flags)) {
95 init_fpel_func(3, 1, 16, avg, _16, sse2);
96 init_fpel_func(2, 1, 32, avg, _16, sse2);
97 init_fpel_func(1, 1, 64, avg, _16, sse2);
98 init_fpel_func(0, 1, 128, avg, _16, sse2);
99 init_8_16_32_ipred_funcs(h, HOR, 16, sse2);
100 init_8_16_32_ipred_funcs(dc, DC, 16, sse2);
101 init_8_16_32_ipred_funcs(dc_top, TOP_DC, 16, sse2);
102 init_8_16_32_ipred_funcs(dc_left, LEFT_DC, 16, sse2);
103 init_ipred_funcs(dl, DIAG_DOWN_LEFT, 16, sse2);
104 init_ipred_funcs(dr, DIAG_DOWN_RIGHT, 16, sse2);
105 init_ipred_funcs(vl, VERT_LEFT, 16, sse2);
106 init_ipred_funcs(vr, VERT_RIGHT, 16, sse2);
107 init_ipred_funcs(hu, HOR_UP, 16, sse2);
108 init_ipred_funcs(hd, HOR_DOWN, 16, sse2);
109 }
110
111 if (EXTERNAL_SSSE3(cpu_flags)) {
112 init_ipred_funcs(dl, DIAG_DOWN_LEFT, 16, ssse3);
113 init_ipred_funcs(dr, DIAG_DOWN_RIGHT, 16, ssse3);
114 init_ipred_funcs(vl, VERT_LEFT, 16, ssse3);
115 init_ipred_funcs(vr, VERT_RIGHT, 16, ssse3);
116 init_ipred_funcs(hu, HOR_UP, 16, ssse3);
117 init_ipred_funcs(hd, HOR_DOWN, 16, ssse3);
118 }
119
120 if (EXTERNAL_AVX_FAST(cpu_flags)) {
121 init_fpel_func(2, 0, 32, put, , avx);
122 init_fpel_func(1, 0, 64, put, , avx);
123 init_fpel_func(0, 0, 128, put, , avx);
124 init_ipred_funcs(dl, DIAG_DOWN_LEFT, 16, avx);
125 init_ipred_funcs(dr, DIAG_DOWN_RIGHT, 16, avx);
126 init_ipred_funcs(vl, VERT_LEFT, 16, avx);
127 init_ipred_funcs(vr, VERT_RIGHT, 16, avx);
128 init_ipred_funcs(hu, HOR_UP, 16, avx);
129 init_ipred_funcs(hd, HOR_DOWN, 16, avx);
130 }
131
132 if (EXTERNAL_AVX2_FAST(cpu_flags)) {
133 init_fpel_func(2, 1, 32, avg, _16, avx2);
134 init_fpel_func(1, 1, 64, avg, _16, avx2);
135 init_fpel_func(0, 1, 128, avg, _16, avx2);
136 }
137
138#endif /* HAVE_YASM */
