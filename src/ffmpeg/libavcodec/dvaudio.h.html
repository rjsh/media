19#ifndef AVCODEC_DVAUDIO_H
20#define AVCODEC_DVAUDIO_H
21
22#include <stdint.h>
23
24static inline int dv_get_audio_sample_count(const uint8_t *buffer, int dsf)
25{
26 int samples = buffer[0] & 0x3f; /* samples in this frame - min samples */
27
28 switch ((buffer[3] >> 3) & 0x07) {
29 case 0:
30 return samples + (dsf ? 1896 : 1580);
31 case 1:
32 return samples + (dsf ? 1742 : 1452);
33 case 2:
34 default:
35 return samples + (dsf ? 1264 : 1053);
36 }
37}
38
39#endif /* AVCODEC_DVAUDIO_H */
