55/**
56 * AMRNB unpacked data subframe
57 */
58typedef struct AMRNBSubframe {
59 uint16_t p_lag; ///< index to decode the pitch lag
60 uint16_t p_gain; ///< index to decode the pitch gain
61 uint16_t fixed_gain; ///< index to decode the fixed gain factor, for MODE_12k2 and MODE_7k95
62 uint16_t pulses[10]; ///< pulses: 10 for MODE_12k2, 7 for MODE_10k2, and index and sign for others
63} AMRNBSubframe;

65/**
66 * AMRNB unpacked data frame
67 */
68typedef struct AMRNBFrame {
69 uint16_t lsf[5]; ///< lsf parameters: 5 parameters for MODE_12k2, only 3 for other modes
70 AMRNBSubframe subframe[4]; ///< unpacked data for each subframe
71} AMRNBFrame;

73/** The index of a frame parameter */
74#define AMR_BIT(field) (offsetof(AMRNBFrame, field))
75/** The index of a subframe-specific parameter */
76#define AMR_OF(frame_num, variable) AMR_BIT(subframe[frame_num].variable)
