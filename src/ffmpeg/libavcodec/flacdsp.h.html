19#ifndef AVCODEC_FLACDSP_H
20#define AVCODEC_FLACDSP_H
21
22#include <stdint.h>
23#include "libavutil/samplefmt.h"
24
25typedef struct FLACDSPContext {
26 void (*decorrelate[4])(uint8_t **out, int32_t **in, int channels,
27 int len, int shift);
28 void (*lpc16)(int32_t *samples, const int coeffs[32], int order,
29 int qlevel, int len);
30 void (*lpc32)(int32_t *samples, const int coeffs[32], int order,
31 int qlevel, int len);
32 void (*lpc16_encode)(int32_t *res, const int32_t *smp, int len, int order,
33 const int32_t coefs[32], int shift);
34 void (*lpc32_encode)(int32_t *res, const int32_t *smp, int len, int order,
35 const int32_t coefs[32], int shift);
36} FLACDSPContext;
37
38void ff_flacdsp_init(FLACDSPContext *c, enum AVSampleFormat fmt, int channels, int bps);
39void ff_flacdsp_init_arm(FLACDSPContext *c, enum AVSampleFormat fmt, int channels, int bps);
40void ff_flacdsp_init_x86(FLACDSPContext *c, enum AVSampleFormat fmt, int channels, int bps);
41
42#endif /* AVCODEC_FLACDSP_H */
