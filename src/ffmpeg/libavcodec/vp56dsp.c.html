29/* Gives very similar result than the vp6 version except in a few cases */
30static int vp5_adjust(int v, int t)
31{
32 int s2, s1 = v >> 31;
33 v ^= s1;
34 v -= s1;
35 v *= v < 2*t;
36 v -= t;
37 s2 = v >> 31;
38 v ^= s2;
39 v -= s2;
40 v = t - v;
41 v += s1;
42 v ^= s1;
43 return v;
44}
45
46static int vp6_adjust(int v, int t)
47{
48 int V = v, s = v >> 31;
49 V ^= s;
50 V -= s;
51 if (V-t-1 >= (unsigned)(t-1))
52 return v;
53 V = 2*t - V;
54 V += s;
55 V ^= s;
56 return V;
57}

60#define VP56_EDGE_FILTER(pfx, suf, pix_inc, line_inc) \
61static void pfx##_edge_filter_##suf(uint8_t *yuv, int stride, int t) \
62{ \
63 int pix2_inc = 2 * pix_inc; \
64 int i, v; \
65 \
66 for (i=0; i<12; i++) { \
67 v = (yuv[-pix2_inc] + 3*(yuv[0]-yuv[-pix_inc]) - yuv[pix_inc] + 4)>>3;\
68 v = pfx##_adjust(v, t); \
69 yuv[-pix_inc] = av_clip_uint8(yuv[-pix_inc] + v); \
70 yuv[0] = av_clip_uint8(yuv[0] - v); \
71 yuv += line_inc; \
72 } \
73}

77VP56_EDGE_FILTER(vp6, hor, 1, stride)

80av_cold void ff_vp56dsp_init(VP56DSPContext *s, enum AVCodecID codec)
81{
82 if (codec == AV_CODEC_ID_VP5) {
83 s->edge_filter_hor = vp5_edge_filter_hor;
84 s->edge_filter_ver = vp5_edge_filter_ver;
85 } else {
86 s->edge_filter_hor = vp6_edge_filter_hor;
87 s->edge_filter_ver = vp6_edge_filter_ver;
88
89 if (CONFIG_VP6_DECODER) {
90 s->vp6_filter_diag4 = ff_vp6_filter_diag4_c;
91
92 if (ARCH_ARM)
93 ff_vp6dsp_init_arm(s, codec);
94 if (ARCH_X86)
95 ff_vp6dsp_init_x86(s, codec);
96 }
97 }
