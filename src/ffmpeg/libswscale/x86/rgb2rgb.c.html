147av_cold void rgb2rgb_init_x86(void)
148{
149#if HAVE_INLINE_ASM
150 int cpu_flags = av_get_cpu_flags();
151
152 if (INLINE_MMX(cpu_flags))
153 rgb2rgb_init_mmx();
154 if (INLINE_AMD3DNOW(cpu_flags))
155 rgb2rgb_init_3dnow();
156 if (INLINE_MMXEXT(cpu_flags))
157 rgb2rgb_init_mmxext();
158 if (INLINE_SSE2(cpu_flags))
159 rgb2rgb_init_sse2();
160 if (INLINE_AVX(cpu_flags))
161 rgb2rgb_init_avx();
162#endif /* HAVE_INLINE_ASM */
163}
