54av_cold void ff_bwdif_init_x86(BWDIFContext *bwdif)
55{
56 int cpu_flags = av_get_cpu_flags();
57 int bit_depth = (!bwdif->csp) ? 8 : bwdif->csp->comp[0].depth;
58
59 if (bit_depth <= 8) {
60#if ARCH_X86_32
61 if (EXTERNAL_MMXEXT(cpu_flags))
62 bwdif->filter_line = ff_bwdif_filter_line_mmxext;
63#endif /* ARCH_X86_32 */
64 if (EXTERNAL_SSE2(cpu_flags))
65 bwdif->filter_line = ff_bwdif_filter_line_sse2;
66 if (EXTERNAL_SSSE3(cpu_flags))
67 bwdif->filter_line = ff_bwdif_filter_line_ssse3;
68 } else if (bit_depth <= 12) {
69#if ARCH_X86_32
70 if (EXTERNAL_MMXEXT(cpu_flags))
71 bwdif->filter_line = ff_bwdif_filter_line_12bit_mmxext;
72#endif /* ARCH_X86_32 */
73 if (EXTERNAL_SSE2(cpu_flags))
74 bwdif->filter_line = ff_bwdif_filter_line_12bit_sse2;
75 if (EXTERNAL_SSSE3(cpu_flags))
76 bwdif->filter_line = ff_bwdif_filter_line_12bit_ssse3;
77 }
78}
