26#define BLEND_FUNC(name, opt) \
27void ff_blend_##name##_##opt(const uint8_t *top, ptrdiff_t top_linesize, \
28 const uint8_t *bottom, ptrdiff_t bottom_linesize, \
29 uint8_t *dst, ptrdiff_t dst_linesize, \
30 ptrdiff_t width, ptrdiff_t height, \
31 struct FilterParams *param, double *values, int starty);

53av_cold void ff_blend_init_x86(FilterParams *param, int is_16bit)
54{
55 int cpu_flags = av_get_cpu_flags();
56
57 if (EXTERNAL_SSE2(cpu_flags) && param->opacity == 1 && !is_16bit) {
58 switch (param->mode) {
59 case BLEND_ADDITION: param->blend = ff_blend_addition_sse2; break;
60 case BLEND_ADDITION128: param->blend = ff_blend_addition128_sse2; break;
61 case BLEND_AND: param->blend = ff_blend_and_sse2; break;
62 case BLEND_AVERAGE: param->blend = ff_blend_average_sse2; break;
63 case BLEND_DARKEN: param->blend = ff_blend_darken_sse2; break;
64 case BLEND_DIFFERENCE128: param->blend = ff_blend_difference128_sse2; break;
65 case BLEND_DIVIDE: param->blend = ff_blend_divide_sse2; break;
66 case BLEND_HARDMIX: param->blend = ff_blend_hardmix_sse2; break;
67 case BLEND_LIGHTEN: param->blend = ff_blend_lighten_sse2; break;
68 case BLEND_MULTIPLY: param->blend = ff_blend_multiply_sse2; break;
69 case BLEND_OR: param->blend = ff_blend_or_sse2; break;
70 case BLEND_PHOENIX: param->blend = ff_blend_phoenix_sse2; break;
71 case BLEND_SCREEN: param->blend = ff_blend_screen_sse2; break;
72 case BLEND_SUBTRACT: param->blend = ff_blend_subtract_sse2; break;
73 case BLEND_XOR: param->blend = ff_blend_xor_sse2; break;
74 case BLEND_DIFFERENCE: param->blend = ff_blend_difference_sse2; break;
75 case BLEND_NEGATION: param->blend = ff_blend_negation_sse2; break;
76 }
77 }
78 if (EXTERNAL_SSSE3(cpu_flags) && param->opacity == 1 && !is_16bit) {
79 switch (param->mode) {
80 case BLEND_DIFFERENCE: param->blend = ff_blend_difference_ssse3; break;
81 case BLEND_NEGATION: param->blend = ff_blend_negation_ssse3; break;
82 }
83 }
84}
