19#include "libavutil/attributes.h"
20#include "libavutil/cpu.h"

26int ff_pullup_filter_diff_mmx(const uint8_t *a, const uint8_t *b, ptrdiff_t s);
27int ff_pullup_filter_comb_mmx(const uint8_t *a, const uint8_t *b, ptrdiff_t s);
28int ff_pullup_filter_var_mmx (const uint8_t *a, const uint8_t *b, ptrdiff_t s);
29
30av_cold void ff_pullup_init_x86(PullupContext *s)
31{
32#if HAVE_YASM
33 int cpu_flags = av_get_cpu_flags();
34
35 if (EXTERNAL_MMX(cpu_flags)) {
36 s->diff = ff_pullup_filter_diff_mmx;
37 s->comb = ff_pullup_filter_comb_mmx;
38 s->var = ff_pullup_filter_var_mmx;
39 }
40#endif
41}
