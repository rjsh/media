30void ff_lowpass_line_sse2(uint8_t *dstp, ptrdiff_t linesize,
31 const uint8_t *srcp,
32 const uint8_t *srcp_above,
33 const uint8_t *srcp_below);
34void ff_lowpass_line_avx (uint8_t *dstp, ptrdiff_t linesize,
35 const uint8_t *srcp,
36 const uint8_t *srcp_above,
37 const uint8_t *srcp_below);
38
39av_cold void ff_tinterlace_init_x86(TInterlaceContext *s)
40{
41 int cpu_flags = av_get_cpu_flags();
42
43 if (EXTERNAL_SSE2(cpu_flags))
44 s->lowpass_line = ff_lowpass_line_sse2;
45 if (EXTERNAL_AVX(cpu_flags))
46 s->lowpass_line = ff_lowpass_line_avx;
47}
