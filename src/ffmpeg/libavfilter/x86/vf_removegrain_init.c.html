50av_cold void ff_removegrain_init_x86(RemoveGrainContext *rg)
51{
52#if CONFIG_GPL
53 int cpu_flags = av_get_cpu_flags();
54 int i;
55
56 for (i = 0; i < rg->nb_planes; i++) {
57 if (EXTERNAL_SSE2(cpu_flags))
58 switch (rg->mode[i]) {
59 case 1: rg->fl[i] = ff_rg_fl_mode_1_sse2; break;
60 case 10: rg->fl[i] = ff_rg_fl_mode_10_sse2; break;
61 case 11: /* fall through */
62 case 12: rg->fl[i] = ff_rg_fl_mode_11_12_sse2; break;
63 case 13: /* fall through */
64 case 14: rg->fl[i] = ff_rg_fl_mode_13_14_sse2; break;
65 case 19: rg->fl[i] = ff_rg_fl_mode_19_sse2; break;
66 case 20: rg->fl[i] = ff_rg_fl_mode_20_sse2; break;
67 case 21: rg->fl[i] = ff_rg_fl_mode_21_sse2; break;
68 case 22: rg->fl[i] = ff_rg_fl_mode_22_sse2; break;
69#if ARCH_X86_64
70 case 2: rg->fl[i] = ff_rg_fl_mode_2_sse2; break;
71 case 3: rg->fl[i] = ff_rg_fl_mode_3_sse2; break;
72 case 4: rg->fl[i] = ff_rg_fl_mode_4_sse2; break;
73 case 5: rg->fl[i] = ff_rg_fl_mode_5_sse2; break;
74 case 6: rg->fl[i] = ff_rg_fl_mode_6_sse2; break;
75 case 7: rg->fl[i] = ff_rg_fl_mode_7_sse2; break;
76 case 8: rg->fl[i] = ff_rg_fl_mode_8_sse2; break;
77 case 9: rg->fl[i] = ff_rg_fl_mode_9_sse2; break;
78 case 15: /* fall through */
79 case 16: rg->fl[i] = ff_rg_fl_mode_15_16_sse2; break;
80 case 17: rg->fl[i] = ff_rg_fl_mode_17_sse2; break;
81 case 18: rg->fl[i] = ff_rg_fl_mode_18_sse2; break;
82 case 23: rg->fl[i] = ff_rg_fl_mode_23_sse2; break;
83 case 24: rg->fl[i] = ff_rg_fl_mode_24_sse2; break;
84#endif /* ARCH_x86_64 */
85 }
86 }
87#endif /* CONFIG_GPL */
88}
