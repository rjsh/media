30 const int16_t *coef, int linesize);

32void ff_w3fdif_simple_high_sse2(int32_t *work_line,

46void ff_w3fdif_scale_sse2(uint8_t *out_pixel, const int32_t *work_pixel, int linesize);
47
48av_cold void ff_w3fdif_init_x86(W3FDIFDSPContext *dsp)
49{
50 int cpu_flags = av_get_cpu_flags();
51
52 if (EXTERNAL_SSE2(cpu_flags)) {
53 dsp->filter_simple_low = ff_w3fdif_simple_low_sse2;
54 dsp->filter_simple_high = ff_w3fdif_simple_high_sse2;
55 dsp->filter_complex_low = ff_w3fdif_complex_low_sse2;
56 dsp->filter_scale = ff_w3fdif_scale_sse2;
57 }
58
59 if (ARCH_X86_64 && EXTERNAL_SSE2(cpu_flags)) {
60 dsp->filter_complex_high = ff_w3fdif_complex_high_sse2;
61 }
62}
