28#define OPENCL_REGISTER_KERNEL_CODE(X, x) \
29 { \
30 if (CONFIG_##X##_FILTER) { \
31 av_opencl_register_kernel_code(ff_kernel_##x##_opencl); \
32 } \
33 }
