30AVFrame *ff_null_get_audio_buffer(AVFilterLink *link, int nb_samples)
31{
32 return ff_get_audio_buffer(link->dst->outputs[0], nb_samples);
33}
34
35AVFrame *ff_default_get_audio_buffer(AVFilterLink *link, int nb_samples)
36{
37 AVFrame *frame = av_frame_alloc();
38 int channels = link->channels;
39 int ret;
40
41 av_assert0(channels == av_get_channel_layout_nb_channels(link->channel_layout) || !av_get_channel_layout_nb_channels(link->channel_layout));
42
43 if (!frame)
44 return NULL;
45
46 frame->nb_samples = nb_samples;
47 frame->format = link->format;
48 av_frame_set_channels(frame, link->channels);
49 frame->channel_layout = link->channel_layout;
50 frame->sample_rate = link->sample_rate;
51 ret = av_frame_get_buffer(frame, 0);
52 if (ret < 0) {
53 av_frame_free(&frame);
54 return NULL;
55 }
56
57 av_samples_set_silence(frame->extended_data, 0, nb_samples, channels,
58 link->format);
59
60
61 return frame;
62}
63
64AVFrame *ff_get_audio_buffer(AVFilterLink *link, int nb_samples)
65{
66 AVFrame *ret = NULL;
67
68 if (link->dstpad->get_audio_buffer)
69 ret = link->dstpad->get_audio_buffer(link, nb_samples);
70
71 if (!ret)
72 ret = ff_default_get_audio_buffer(link, nb_samples);
73
74 return ret;
75}
