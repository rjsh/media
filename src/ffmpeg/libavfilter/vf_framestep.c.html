26#include "libavutil/opt.h"
27#include "avfilter.h"
28#include "internal.h"
29#include "video.h"
30
31typedef struct NullContext {
32 const AVClass *class;
33 int frame_step;
34} FrameStepContext;
35
36#define OFFSET(x) offsetof(FrameStepContext, x)
37#define FLAGS AV_OPT_FLAG_FILTERING_PARAM|AV_OPT_FLAG_VIDEO_PARAM
38
39static const AVOption framestep_options[] = {
40 { "step", "set frame step", OFFSET(frame_step), AV_OPT_TYPE_INT, {.i64=1}, 1, INT_MAX, FLAGS},
41 { NULL },
42};
43
44AVFILTER_DEFINE_CLASS(framestep);
45
46static int config_output_props(AVFilterLink *outlink)
47{
48 AVFilterContext *ctx = outlink->src;
49 FrameStepContext *framestep = ctx->priv;
50 AVFilterLink *inlink = ctx->inputs[0];
51
52 outlink->frame_rate =
53 av_div_q(inlink->frame_rate, (AVRational){framestep->frame_step, 1});
54
55 av_log(ctx, AV_LOG_VERBOSE, "step:%d frame_rate:%d/%d(%f) -> frame_rate:%d/%d(%f)\n",
56 framestep->frame_step,
57 inlink->frame_rate.num, inlink->frame_rate.den, av_q2d(inlink->frame_rate),
58 outlink->frame_rate.num, outlink->frame_rate.den, av_q2d(outlink->frame_rate));
59 return 0;
60}
61
62static int filter_frame(AVFilterLink *inlink, AVFrame *ref)
63{
64 FrameStepContext *framestep = inlink->dst->priv;
65
66 if (!(inlink->frame_count % framestep->frame_step)) {
67 return ff_filter_frame(inlink->dst->outputs[0], ref);
68 } else {
69 av_frame_free(&ref);
70 return 0;
71 }
72}
73
74static const AVFilterPad framestep_inputs[] = {
75 {
76 .name = "default",
77 .type = AVMEDIA_TYPE_VIDEO,
78 .filter_frame = filter_frame,
79 },
80 { NULL }
81};
82
83static const AVFilterPad framestep_outputs[] = {
84 {
85 .name = "default",
86 .type = AVMEDIA_TYPE_VIDEO,
87 .config_props = config_output_props,
88 },
89 { NULL }
90};
91
92AVFilter ff_vf_framestep = {
93 .name = "framestep",
94 .description = NULL_IF_CONFIG_SMALL("Select one frame every N frames."),
95 .priv_size = sizeof(FrameStepContext),
96 .priv_class = &framestep_class,
97 .inputs = framestep_inputs,
98 .outputs = framestep_outputs,
99 .flags = AVFILTER_FLAG_SUPPORT_TIMELINE_GENERIC,
100};
