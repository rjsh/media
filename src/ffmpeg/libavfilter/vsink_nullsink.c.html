19#include "avfilter.h"
20#include "internal.h"
21#include "libavutil/internal.h"

23static int filter_frame(AVFilterLink *link, AVFrame *frame)
24{
25 av_frame_free(&frame);
26 return 0;
27}
28
29static const AVFilterPad avfilter_vsink_nullsink_inputs[] = {
30 {
31 .name = "default",
32 .type = AVMEDIA_TYPE_VIDEO,
33 .filter_frame = filter_frame,
34 },
35 { NULL },
36};
37
38AVFilter ff_vsink_nullsink = {
39 .name = "nullsink",
40 .description = NULL_IF_CONFIG_SMALL("Do absolutely nothing with the input video."),
41
42 .priv_size = 0,
43
44 .inputs = avfilter_vsink_nullsink_inputs,
45 .outputs = NULL,
46};
