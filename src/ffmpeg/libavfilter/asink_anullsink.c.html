25static int null_filter_frame(AVFilterLink *link, AVFrame *frame)
26{
27 av_frame_free(&frame);
28 return 0;
29}
30
31static const AVFilterPad avfilter_asink_anullsink_inputs[] = {
32 {
33 .name = "default",
34 .type = AVMEDIA_TYPE_AUDIO,
35 .filter_frame = null_filter_frame,
36 },
37 { NULL },
38};
39
40AVFilter ff_asink_anullsink = {
41 .name = "anullsink",
42 .description = NULL_IF_CONFIG_SMALL("Do absolutely nothing with the input audio."),
43
44 .priv_size = 0,
45
46 .inputs = avfilter_asink_anullsink_inputs,
47 .outputs = NULL,
48};
