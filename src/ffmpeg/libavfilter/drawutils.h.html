46#define MAX_PLANES 4
47
48typedef struct FFDrawContext {
49 const struct AVPixFmtDescriptor *desc;
50 enum AVPixelFormat format;
51 unsigned nb_planes;
52 int pixelstep[MAX_PLANES]; /*< offset between pixels */
53 uint8_t comp_mask[MAX_PLANES]; /*< bitmask of used non-alpha components */
54 uint8_t hsub[MAX_PLANES]; /*< horizontal subsampling */
55 uint8_t vsub[MAX_PLANES]; /*< vertical subsampling */
56 uint8_t hsub_max;
57 uint8_t vsub_max;
58 unsigned flags;
59} FFDrawContext;
60
61typedef struct FFDrawColor {
62 uint8_t rgba[4];
63 union {
64 uint32_t u32[4];
65 uint16_t u16[8];
66 uint8_t u8[16];
67 } comp[MAX_PLANES];
68} FFDrawColor;
69
70/**
71 * Process alpha pixel component.
72 */
73#define FF_DRAW_PROCESS_ALPHA 1
74
75/**
76 * Init a draw context.
77 *
78 * Only a limited number of pixel formats are supported, if format is not
79 * supported the function will return an error.
80 * flags is combination of FF_DRAW_* flags.
81 * @return 0 for success, < 0 for error
82 */
83int ff_draw_init(FFDrawContext *draw, enum AVPixelFormat format, unsigned flags);
84
85/**
86 * Prepare a color.
87 */
88void ff_draw_color(FFDrawContext *draw, FFDrawColor *color, const uint8_t rgba[4]);
89
90/**
91 * Copy a rectangle from an image to another.
92 *
93 * The coordinates must be as even as the subsampling requires.
94 */
95void ff_copy_rectangle2(FFDrawContext *draw,
96 uint8_t *dst[], int dst_linesize[],
97 uint8_t *src[], int src_linesize[],
98 int dst_x, int dst_y, int src_x, int src_y,
99 int w, int h);
100
101/**
102 * Fill a rectangle with an uniform color.
103 *
104 * The coordinates must be as even as the subsampling requires.
105 * The color needs to be inited with ff_draw_color.
106 */
107void ff_fill_rectangle(FFDrawContext *draw, FFDrawColor *color,
108 uint8_t *dst[], int dst_linesize[],
109 int dst_x, int dst_y, int w, int h);
110
111/**
112 * Blend a rectangle with an uniform color.
113 */
114void ff_blend_rectangle(FFDrawContext *draw, FFDrawColor *color,
115 uint8_t *dst[], int dst_linesize[],
116 int dst_w, int dst_h,
117 int x0, int y0, int w, int h);
118
119/**
120 * Blend an alpha mask with an uniform color.
121 *
122 * @param draw draw context
123 * @param color color for the overlay;
124 * @param dst destination image
125 * @param dst_linesize line stride of the destination
126 * @param dst_w width of the destination image
127 * @param dst_h height of the destination image
128 * @param mask mask
129 * @param mask_linesize line stride of the mask
130 * @param mask_w width of the mask
131 * @param mask_h height of the mask
132 * @param l2depth log2 of depth of the mask (0 for 1bpp, 3 for 8bpp)
133 * @param endianness bit order of the mask (0: MSB to the left)
134 * @param x0 horizontal position of the overlay
135 * @param y0 vertical position of the overlay
136 */
137void ff_blend_mask(FFDrawContext *draw, FFDrawColor *color,
138 uint8_t *dst[], int dst_linesize[], int dst_w, int dst_h,
139 const uint8_t *mask, int mask_linesize, int mask_w, int mask_h,
140 int l2depth, unsigned endianness, int x0, int y0);
141
142/**
143 * Round a dimension according to subsampling.
144 *
145 * @param draw draw context
146 * @param sub_dir 0 for horizontal, 1 for vertical
147 * @param round_dir 0 nearest, -1 round down, +1 round up
148 * @param value value to round
149 * @return the rounded value
150 */
151int ff_draw_round_to_sub(FFDrawContext *draw, int sub_dir, int round_dir,
152 int value);
153
154/**
155 * Return the list of pixel formats supported by the draw functions.
156 *
157 * The flags are the same as ff_draw_init, i.e., none currently.
158 */
159AVFilterFormats *ff_draw_supported_pixel_formats(unsigned flags);
