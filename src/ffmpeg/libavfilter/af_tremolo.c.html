26typedef struct TremoloContext {
27 const AVClass *class;
28 double freq;
29 double depth;
30 double *table;
31 int index;
32} TremoloContext;
33
34#define OFFSET(x) offsetof(TremoloContext, x)
35#define FLAGS AV_OPT_FLAG_AUDIO_PARAM|AV_OPT_FLAG_FILTERING_PARAM

45static int filter_frame(AVFilterLink *inlink, AVFrame *in)
46{
47 AVFilterContext *ctx = inlink->dst;
48 AVFilterLink *outlink = ctx->outputs[0];
49 TremoloContext *s = ctx->priv;
50 const double *src = (const double *)in->data[0];
51 const int channels = inlink->channels;
52 const int nb_samples = in->nb_samples;
53 AVFrame *out;
54 double *dst;
55 int n, c;
56
57 if (av_frame_is_writable(in)) {
58 out = in;
59 } else {
60 out = ff_get_audio_buffer(inlink, in->nb_samples);
61 if (!out) {
62 av_frame_free(&in);
63 return AVERROR(ENOMEM);
64 }
65 av_frame_copy_props(out, in);
66 }
67 dst = (double *)out->data[0];
68
69 for (n = 0; n < nb_samples; n++) {
70 for (c = 0; c < channels; c++)
71 dst[c] = src[c] * s->table[s->index];
72 dst += channels;
73 src += channels;
74 s->index++;
75 if (s->index >= inlink->sample_rate / s->freq)
76 s->index = 0;
77 }
78
79 if (in != out)
80 av_frame_free(&in);
81
82 return ff_filter_frame(outlink, out);
83}

85static int query_formats(AVFilterContext *ctx)
86{
87 AVFilterFormats *formats;
88 AVFilterChannelLayouts *layouts;
89 static const enum AVSampleFormat sample_fmts[] = {
90 AV_SAMPLE_FMT_DBL,
91 AV_SAMPLE_FMT_NONE
92 };
93 int ret;
94
95 layouts = ff_all_channel_counts();
96 if (!layouts)
97 return AVERROR(ENOMEM);
98 ret = ff_set_common_channel_layouts(ctx, layouts);
99 if (ret < 0)
100 return ret;
101
102 formats = ff_make_format_list(sample_fmts);
103 if (!formats)
104 return AVERROR(ENOMEM);
105 ret = ff_set_common_formats(ctx, formats);
106 if (ret < 0)
107 return ret;
108
109 formats = ff_all_samplerates();
110 if (!formats)
111 return AVERROR(ENOMEM);
112 return ff_set_common_samplerates(ctx, formats);
113}

115static av_cold void uninit(AVFilterContext *ctx)
116{
117 TremoloContext *s = ctx->priv;
118 av_freep(&s->table);
119}
120
121static int config_input(AVFilterLink *inlink)
122{
123 AVFilterContext *ctx = inlink->dst;
124 TremoloContext *s = ctx->priv;
125 const double offset = 1. - s->depth / 2.;
126 int i;
127
128 s->table = av_malloc_array(inlink->sample_rate / s->freq, sizeof(*s->table));
129 if (!s->table)
130 return AVERROR(ENOMEM);
131
132 for (i = 0; i < inlink->sample_rate / s->freq; i++) {
133 double env = s->freq * i / inlink->sample_rate;
134 env = sin(2 * M_PI * fmod(env + 0.25, 1.0));
135 s->table[i] = env * (1 - fabs(offset)) + offset;
136 }
137
138 s->index = 0;
139
140 return 0;
141}
