23#include "libavutil/lls.h"
24#include "libavutil/x86/cpu.h"
25
26void ff_update_lls_sse2(LLSModel *m, const double *var);
27void ff_update_lls_avx(LLSModel *m, const double *var);
28void ff_update_lls_fma3(LLSModel *m, const double *var);
29double ff_evaluate_lls_sse2(LLSModel *m, const double *var, int order);
30
31av_cold void ff_init_lls_x86(LLSModel *m)
32{
33 int cpu_flags = av_get_cpu_flags();
34 if (EXTERNAL_SSE2(cpu_flags)) {
35 m->update_lls = ff_update_lls_sse2;
36 if (m->indep_count >= 4)
37 m->evaluate_lls = ff_evaluate_lls_sse2;
38 }
39 if (EXTERNAL_AVX_FAST(cpu_flags)) {
40 m->update_lls = ff_update_lls_avx;
41 }
42 if (EXTERNAL_FMA3_FAST(cpu_flags)) {
43 m->update_lls = ff_update_lls_fma3;
44 }
45}
