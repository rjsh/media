19#ifndef AVUTIL_CPU_INTERNAL_H
20#define AVUTIL_CPU_INTERNAL_H
21
22#include "cpu.h"
23
24#define CPUEXT_SUFFIX(flags, suffix, cpuext) \
25 (HAVE_ ## cpuext ## suffix && ((flags) & AV_CPU_FLAG_ ## cpuext))
26
27#define CPUEXT_SUFFIX_FAST2(flags, suffix, cpuext, slow_cpuext) \
28 (HAVE_ ## cpuext ## suffix && ((flags) & AV_CPU_FLAG_ ## cpuext) && \
29 !((flags) & AV_CPU_FLAG_ ## slow_cpuext ## SLOW))
30
31#define CPUEXT_SUFFIX_SLOW2(flags, suffix, cpuext, slow_cpuext) \
32 (HAVE_ ## cpuext ## suffix && ((flags) & AV_CPU_FLAG_ ## cpuext) && \
33 ((flags) & AV_CPU_FLAG_ ## slow_cpuext ## SLOW))
34
35#define CPUEXT_SUFFIX_FAST(flags, suffix, cpuext) CPUEXT_SUFFIX_FAST2(flags, suffix, cpuext, cpuext)
36#define CPUEXT_SUFFIX_SLOW(flags, suffix, cpuext) CPUEXT_SUFFIX_SLOW2(flags, suffix, cpuext, cpuext)
37
38#define CPUEXT(flags, cpuext) CPUEXT_SUFFIX(flags, , cpuext)
39#define CPUEXT_FAST(flags, cpuext) CPUEXT_SUFFIX_FAST(flags, , cpuext)
40#define CPUEXT_SLOW(flags, cpuext) CPUEXT_SUFFIX_SLOW(flags, , cpuext)
41
42int ff_get_cpu_flags_aarch64(void);
43int ff_get_cpu_flags_arm(void);
44int ff_get_cpu_flags_ppc(void);
45int ff_get_cpu_flags_x86(void);
46
47#endif /* AVUTIL_CPU_INTERNAL_H */
