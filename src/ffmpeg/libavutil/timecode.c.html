34int av_timecode_adjust_ntsc_framenum2(int framenum, int fps)
35{
36 /* only works for NTSC 29.97 and 59.94 */
37 int drop_frames = 0;
38 int d, m, frames_per_10mins;
39
40 if (fps == 30) {
41 drop_frames = 2;
42 frames_per_10mins = 17982;
43 } else if (fps == 60) {
44 drop_frames = 4;
45 frames_per_10mins = 35964;
46 } else
47 return framenum;
48
49 d = framenum / frames_per_10mins;
50 m = framenum % frames_per_10mins;
51
52 return framenum + 9 * drop_frames * d + drop_frames * ((m - drop_frames) / (frames_per_10mins / 10));
53}

55uint32_t av_timecode_get_smpte_from_framenum(const AVTimecode *tc, int framenum)
56{
57 unsigned fps = tc->fps;
58 int drop = !!(tc->flags & AV_TIMECODE_FLAG_DROPFRAME);
59 int hh, mm, ss, ff;
60
61 framenum += tc->start;
62 if (drop)
63 framenum = av_timecode_adjust_ntsc_framenum2(framenum, tc->fps);
64 ff = framenum % fps;
65 ss = framenum / fps % 60;
66 mm = framenum / (fps*60) % 60;
67 hh = framenum / (fps*3600) % 24;
68 return 0 << 31 | // color frame flag (0: unsync mode, 1: sync mode)
69 drop << 30 | // drop frame flag (0: non drop, 1: drop)
70 (ff / 10) << 28 | // tens of frames
71 (ff % 10) << 24 | // units of frames
72 0 << 23 | // PC (NTSC) or BGF0 (PAL)
73 (ss / 10) << 20 | // tens of seconds
74 (ss % 10) << 16 | // units of seconds
75 0 << 15 | // BGF0 (NTSC) or BGF2 (PAL)
76 (mm / 10) << 12 | // tens of minutes
77 (mm % 10) << 8 | // units of minutes
78 0 << 7 | // BGF2 (NTSC) or PC (PAL)
79 0 << 6 | // BGF1
80 (hh / 10) << 4 | // tens of hours
81 (hh % 10); // units of hours
82}

84char *av_timecode_make_string(const AVTimecode *tc, char *buf, int framenum)
85{
86 int fps = tc->fps;
87 int drop = tc->flags & AV_TIMECODE_FLAG_DROPFRAME;
88 int hh, mm, ss, ff, neg = 0;
89
90 framenum += tc->start;
91 if (drop)
92 framenum = av_timecode_adjust_ntsc_framenum2(framenum, fps);
93 if (framenum < 0) {
94 framenum = -framenum;
95 neg = tc->flags & AV_TIMECODE_FLAG_ALLOWNEGATIVE;
96 }
97 ff = framenum % fps;
98 ss = framenum / fps % 60;
99 mm = framenum / (fps*60) % 60;
100 hh = framenum / (fps*3600);
101 if (tc->flags & AV_TIMECODE_FLAG_24HOURSMAX)
102 hh = hh % 24;
103 snprintf(buf, AV_TIMECODE_STR_SIZE, "%s%02d:%02d:%02d%c%02d",
104 neg ? "-" : "",
105 hh, mm, ss, drop ? ';' : ':', ff);
106 return buf;
107}

109static unsigned bcd2uint(uint8_t bcd)
110{
111 unsigned low = bcd & 0xf;
112 unsigned high = bcd >> 4;
113 if (low > 9 || high > 9)
114 return 0;
115 return low + 10*high;
116}

118char *av_timecode_make_smpte_tc_string(char *buf, uint32_t tcsmpte, int prevent_df)
119{
120 unsigned hh = bcd2uint(tcsmpte & 0x3f); // 6-bit hours
121 unsigned mm = bcd2uint(tcsmpte>>8 & 0x7f); // 7-bit minutes
122 unsigned ss = bcd2uint(tcsmpte>>16 & 0x7f); // 7-bit seconds
123 unsigned ff = bcd2uint(tcsmpte>>24 & 0x3f); // 6-bit frames
124 unsigned drop = tcsmpte & 1<<30 && !prevent_df; // 1-bit drop if not arbitrary bit
125 snprintf(buf, AV_TIMECODE_STR_SIZE, "%02u:%02u:%02u%c%02u",
126 hh, mm, ss, drop ? ';' : ':', ff);
127 return buf;
128}

130char *av_timecode_make_mpeg_tc_string(char *buf, uint32_t tc25bit)
131{
132 snprintf(buf, AV_TIMECODE_STR_SIZE, "%02u:%02u:%02u%c%02u",
133 tc25bit>>19 & 0x1f, // 5-bit hours
134 tc25bit>>13 & 0x3f, // 6-bit minutes
135 tc25bit>>6 & 0x3f, // 6-bit seconds
136 tc25bit & 1<<24 ? ';' : ':', // 1-bit drop flag
137 tc25bit & 0x3f); // 6-bit frames
138 return buf;
139}

141static int check_fps(int fps)
142{
143 int i;
144 static const int supported_fps[] = {
145 24, 25, 30, 48, 50, 60, 100, 120, 150,
146 };
147
148 for (i = 0; i < FF_ARRAY_ELEMS(supported_fps); i++)
149 if (fps == supported_fps[i])
150 return 0;
151 return -1;
152}

154static int check_timecode(void *log_ctx, AVTimecode *tc)
155{
156 if ((int)tc->fps <= 0) {
157 av_log(log_ctx, AV_LOG_ERROR, "Timecode frame rate must be specified\n");
158 return AVERROR(EINVAL);
159 }
160 if ((tc->flags & AV_TIMECODE_FLAG_DROPFRAME) && tc->fps != 30 && tc->fps != 60) {
161 av_log(log_ctx, AV_LOG_ERROR, "Drop frame is only allowed with 30000/1001 or 60000/1001 FPS\n");
162 return AVERROR(EINVAL);
163 }
164 if (check_fps(tc->fps) < 0) {
165 av_log(log_ctx, AV_LOG_WARNING, "Using non-standard frame rate %d/%d\n",
166 tc->rate.num, tc->rate.den);
167 }
168 return 0;
169}

171static int fps_from_frame_rate(AVRational rate)
172{
173 if (!rate.den || !rate.num)
174 return -1;
175 return (rate.num + rate.den/2) / rate.den;
176}
177
178int av_timecode_check_frame_rate(AVRational rate)
179{
180 return check_fps(fps_from_frame_rate(rate));
181}

183int av_timecode_init(AVTimecode *tc, AVRational rate, int flags, int frame_start, void *log_ctx)
184{
185 memset(tc, 0, sizeof(*tc));
186 tc->start = frame_start;
187 tc->flags = flags;
188 tc->rate = rate;
189 tc->fps = fps_from_frame_rate(rate);
190 return check_timecode(log_ctx, tc);
191}

193int av_timecode_init_from_string(AVTimecode *tc, AVRational rate, const char *str, void *log_ctx)
194{
195 char c;
196 int hh, mm, ss, ff, ret;
197
198 if (sscanf(str, "%d:%d:%d%c%d", &hh, &mm, &ss, &c, &ff) != 5) {
199 av_log(log_ctx, AV_LOG_ERROR, "Unable to parse timecode, "
200 "syntax: hh:mm:ss[:;.]ff\n");
201 return AVERROR_INVALIDDATA;
202 }
203
204 memset(tc, 0, sizeof(*tc));
205 tc->flags = c != ':' ? AV_TIMECODE_FLAG_DROPFRAME : 0; // drop if ';', '.', ...
206 tc->rate = rate;
207 tc->fps = fps_from_frame_rate(rate);
208
209 ret = check_timecode(log_ctx, tc);
210 if (ret < 0)
211 return ret;
212
213 tc->start = (hh*3600 + mm*60 + ss) * tc->fps + ff;
214 if (tc->flags & AV_TIMECODE_FLAG_DROPFRAME) { /* adjust frame number */
215 int tmins = 60*hh + mm;
216 tc->start -= 2 * (tmins - tmins/10);
217 }
218 return 0;
219}
