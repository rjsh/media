21#ifndef AVUTIL_MEM_INTERNAL_H
22#define AVUTIL_MEM_INTERNAL_H
23
24#include "avassert.h"
25#include "mem.h"
26
27static inline int ff_fast_malloc(void *ptr, unsigned int *size, size_t min_size, int zero_realloc)
28{
29 void *val;
30
31 memcpy(&val, ptr, sizeof(val));
32 if (min_size <= *size) {
33 av_assert0(val || !min_size);
34 return 0;
35 }
36 min_size = FFMAX(min_size + min_size / 16 + 32, min_size);
37 av_freep(ptr);
38 val = zero_realloc ? av_mallocz(min_size) : av_malloc(min_size);
39 memcpy(ptr, &val, sizeof(val));
40 if (!val)
41 min_size = 0;
42 *size = min_size;
43 return 1;
44}
45#endif /* AVUTIL_MEM_INTERNAL_H */
