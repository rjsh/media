26#define IA(x) ((x) >> 24)
27#define IB(x) (((x) >> 16) & 0xff)
28#define IC(x) (((x) >> 8) & 0xff)
29#define ID(x) ((x) & 0xff)
30
31#define LR(x, c) (((x) << (c)) | ((x) >> (32 - (c))))
32
33#define F3(l, r, i) \
34 do { \
35 I = LR(cs->Km[i] - r, cs->Kr[i]); \
36 f = ((S1[IA(I)] + S2[IB(I)]) ^ S3[IC(I)]) - S4[ID(I)]; \
37 l = f ^ l; \
38 } while (0)
39
40#define F2(l, r, i) \
41 do { \
42 I = LR(cs->Km[i] ^ r, cs->Kr[i]); \
43 f = ((S1[IA(I)] - S2[IB(I)]) + S3[IC(I)]) ^ S4[ID(I)]; \
44 l = f ^ l; \
45 } while (0)
46
47#define F1(l, r, i) \
48 do { \
49 I = LR(cs->Km[i] + r, cs->Kr[i]); \
50 f = ((S1[IA(I)] ^ S2[IB(I)]) - S3[IC(I)]) + S4[ID(I)]; \
51 l = f ^ l; \
52 } while (0)

54#define COMPUTE_Z \
55 do { \
56 z[0] = x[0] ^ S5[IB(x[3])] ^ S6[ID(x[3])] ^ S7[IA(x[3])] ^ S8[IC(x[3])] ^ S7[IA(x[2])]; \
57 z[1] = x[2] ^ S5[IA(z[0])] ^ S6[IC(z[0])] ^ S7[IB(z[0])] ^ S8[ID(z[0])] ^ S8[IC(x[2])]; \
58 z[2] = x[3] ^ S5[ID(z[1])] ^ S6[IC(z[1])] ^ S7[IB(z[1])] ^ S8[IA(z[1])] ^ S5[IB(x[2])]; \
59 z[3] = x[1] ^ S5[IC(z[2])] ^ S6[IB(z[2])] ^ S7[ID(z[2])] ^ S8[IA(z[2])] ^ S6[ID(x[2])]; \
60 } while (0)
61
62#define COMPUTE_X \
63 do { \
64 x[0] = z[2] ^ S5[IB(z[1])] ^ S6[ID(z[1])] ^ S7[IA(z[1])] ^ S8[IC(z[1])] ^ S7[IA(z[0])]; \
65 x[1] = z[0] ^ S5[IA(x[0])] ^ S6[IC(x[0])] ^ S7[IB(x[0])] ^ S8[ID(x[0])] ^ S8[IC(z[0])]; \
66 x[2] = z[1] ^ S5[ID(x[1])] ^ S6[IC(x[1])] ^ S7[IB(x[1])] ^ S8[IA(x[1])] ^ S5[IB(z[0])]; \
67 x[3] = z[3] ^ S5[IC(x[2])] ^ S6[IB(x[2])] ^ S7[ID(x[2])] ^ S8[IA(x[2])] ^ S6[ID(z[0])]; \
68 } while (0)

359static void generate_round_keys(int rnds, uint32_t* K, uint32_t* x, uint32_t* z)
360{
361 COMPUTE_Z;
362
363 K[1] = S5[IA(z[2])] ^ S6[IB(z[2])] ^ S7[ID(z[1])] ^ S8[IC(z[1])] ^ S5[IC(z[0])];
364 K[2] = S5[IC(z[2])] ^ S6[ID(z[2])] ^ S7[IB(z[1])] ^ S8[IA(z[1])] ^ S6[IC(z[1])];
365 K[3] = S5[IA(z[3])] ^ S6[IB(z[3])] ^ S7[ID(z[0])] ^ S8[IC(z[0])] ^ S7[IB(z[2])];
366 K[4] = S5[IC(z[3])] ^ S6[ID(z[3])] ^ S7[IB(z[0])] ^ S8[IA(z[0])] ^ S8[IA(z[3])];
367
368 COMPUTE_X;
369
370 K[5] = S5[ID(x[0])] ^ S6[IC(x[0])] ^ S7[IA(x[3])] ^ S8[IB(x[3])] ^ S5[IA(x[2])];
371 K[6] = S5[IB(x[0])] ^ S6[IA(x[0])] ^ S7[IC(x[3])] ^ S8[ID(x[3])] ^ S6[IB(x[3])];
372 K[7] = S5[ID(x[1])] ^ S6[IC(x[1])] ^ S7[IA(x[2])] ^ S8[IB(x[2])] ^ S7[ID(x[0])];
373 K[8] = S5[IB(x[1])] ^ S6[IA(x[1])] ^ S7[IC(x[2])] ^ S8[ID(x[2])] ^ S8[ID(x[1])];
374
375 COMPUTE_Z;
376
377 K[9] = S5[ID(z[0])] ^ S6[IC(z[0])] ^ S7[IA(z[3])] ^ S8[IB(z[3])] ^ S5[IB(z[2])];
378 K[10] = S5[IB(z[0])] ^ S6[IA(z[0])] ^ S7[IC(z[3])] ^ S8[ID(z[3])] ^ S6[IA(z[3])];
379 K[11] = S5[ID(z[1])] ^ S6[IC(z[1])] ^ S7[IA(z[2])] ^ S8[IB(z[2])] ^ S7[IC(z[0])];
380 K[12] = S5[IB(z[1])] ^ S6[IA(z[1])] ^ S7[IC(z[2])] ^ S8[ID(z[2])] ^ S8[IC(z[1])];
381
382 COMPUTE_X;
383
384 if (rnds == 16) {
385 K[13] = S5[IA(x[2])] ^ S6[IB(x[2])] ^ S7[ID(x[1])] ^ S8[IC(x[1])] ^ S5[ID(x[0])];
386 K[14] = S5[IC(x[2])] ^ S6[ID(x[2])] ^ S7[IB(x[1])] ^ S8[IA(x[1])] ^ S6[ID(x[1])];
387 K[15] = S5[IA(x[3])] ^ S6[IB(x[3])] ^ S7[ID(x[0])] ^ S8[IC(x[0])] ^ S7[IA(x[2])];
388 K[16] = S5[IC(x[3])] ^ S6[ID(x[3])] ^ S7[IB(x[0])] ^ S8[IA(x[0])] ^ S8[IB(x[3])];
389 }

392static void encipher(AVCAST5* cs, uint8_t* dst, const uint8_t* src)
393{
394 uint32_t r, l, f, I;
395 l = AV_RB32(src);
396 r = AV_RB32(src + 4);
397 F1(l, r, 1);
398 F2(r, l, 2);
399 F3(l, r, 3);
400 F1(r, l, 4);
401 F2(l, r, 5);
402 F3(r, l, 6);
403 F1(l, r, 7);
404 F2(r, l, 8);
405 F3(l, r, 9);
406 F1(r, l, 10);
407 F2(l, r, 11);
408 F3(r, l, 12);
409 if (cs->rounds == 16) {
410 F1(l, r, 13);
411 F2(r, l, 14);
412 F3(l, r, 15);
413 F1(r, l, 16);
414 }
415 AV_WB32(dst, r);
416 AV_WB32(dst + 4, l);
417}

419static void decipher(AVCAST5* cs, uint8_t* dst, const uint8_t* src, uint8_t *iv)
420{
421 uint32_t f, I, r, l;
422 l = AV_RB32(src);
423 r = AV_RB32(src + 4);
424 if (cs->rounds == 16) {
425 F1(l, r, 16);
426 F3(r, l, 15);
427 F2(l, r, 14);
428 F1(r, l, 13);
429 }
430 F3(l, r, 12);
431 F2(r, l, 11);
432 F1(l, r, 10);
433 F3(r, l, 9);
434 F2(l, r, 8);
435 F1(r, l, 7);
436 F3(l, r, 6);
437 F2(r, l, 5);
438 F1(l, r, 4);
439 F3(r, l, 3);
440 F2(l, r, 2);
441 F1(r, l, 1);
442 if (iv) {
443 r ^= AV_RB32(iv);
444 l ^= AV_RB32(iv + 4);
445 memcpy(iv, src, 8);
446 }
447 AV_WB32(dst, r);
448 AV_WB32(dst + 4, l);
449}

456av_cold int av_cast5_init(AVCAST5* cs, const uint8_t *key, int key_bits)
457{
458 uint8_t newKey[16];
459 int i;
460 uint32_t p[4], q[4];
461 if (key_bits % 8 || key_bits < 40 || key_bits > 128)
462 return AVERROR(EINVAL);
463 memset(newKey, 0, sizeof(newKey));
464 memcpy(newKey, key, key_bits >> 3);
465
466 cs->rounds = key_bits <= 80 ? 12 : 16;
467 for (i = 0; i < 4; i++)
468 q[i] = AV_RB32(newKey + (4 * i));
469 generate_round_keys(cs->rounds, cs->Km, q, p);
470 generate_round_keys(cs->rounds, cs->Kr, q, p);
471 for (i = 0; i <= cs->rounds; i++)
472 cs->Kr[i] = cs->Kr[i] & 0x1f;
473 return 0;
474}

477{
478 int i;
479 while (count--) {
480 if (decrypt) {
481 decipher(cs, dst, src, iv);
482 } else {
483 if (iv) {
484 for (i = 0; i < 8; i++)
485 dst[i] = src[i] ^ iv[i];
486 encipher(cs, dst, dst);
487 memcpy(iv, dst, 8);
488 } else {
489 encipher(cs, dst, src);
490 }
491 }
492 src = src + 8;
493 dst = dst + 8;
494 }
