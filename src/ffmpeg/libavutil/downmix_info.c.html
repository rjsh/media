21#include "downmix_info.h"
22#include "frame.h"
23
24AVDownmixInfo *av_downmix_info_update_side_data(AVFrame *frame)
25{
26 AVFrameSideData *side_data;
27
28 side_data = av_frame_get_side_data(frame, AV_FRAME_DATA_DOWNMIX_INFO);
29
30 if (!side_data)
31 side_data = av_frame_new_side_data(frame, AV_FRAME_DATA_DOWNMIX_INFO,
32 sizeof(AVDownmixInfo));
33
34 if (!side_data)
35 return NULL;
36
37 return (AVDownmixInfo*)side_data->data;
38}
