310int av_crc_init(AVCRC *ctx, int le, int bits, uint32_t poly, int ctx_size)
311{
312 unsigned i, j;
313 uint32_t c;
314
315 if (bits < 8 || bits > 32 || poly >= (1LL << bits))
316 return AVERROR(EINVAL);
317 if (ctx_size != sizeof(AVCRC) * 257 && ctx_size != sizeof(AVCRC) * 1024)
318 return AVERROR(EINVAL);
319
320 for (i = 0; i < 256; i++) {
321 if (le) {
322 for (c = i, j = 0; j < 8; j++)
323 c = (c >> 1) ^ (poly & (-(c & 1)));
324 ctx[i] = c;
325 } else {
326 for (c = i << 24, j = 0; j < 8; j++)
327 c = (c << 1) ^ ((poly << (32 - bits)) & (((int32_t) c) >> 31));
328 ctx[i] = av_bswap32(c);
329 }
330 }
331 ctx[256] = 1;
332#if !CONFIG_SMALL
333 if (ctx_size >= sizeof(AVCRC) * 1024)
334 for (i = 0; i < 256; i++)
335 for (j = 0; j < 3; j++)
336 ctx[256 * (j + 1) + i] =
337 (ctx[256 * j + i] >> 8) ^ ctx[ctx[256 * j + i] & 0xFF];
338#endif
339
340 return 0;
341}

343const AVCRC *av_crc_get_table(AVCRCId crc_id)
344{
345#if !CONFIG_HARDCODED_TABLES
346 if (!av_crc_table[crc_id][FF_ARRAY_ELEMS(av_crc_table[crc_id]) - 1])
347 if (av_crc_init(av_crc_table[crc_id],
348 av_crc_table_params[crc_id].le,
349 av_crc_table_params[crc_id].bits,
350 av_crc_table_params[crc_id].poly,
351 sizeof(av_crc_table[crc_id])) < 0)
352 return NULL;
353#endif
354 return av_crc_table[crc_id];
355}

357uint32_t av_crc(const AVCRC *ctx, uint32_t crc,
358 const uint8_t *buffer, size_t length)
359{
360 const uint8_t *end = buffer + length;
361
362#if !CONFIG_SMALL
363 if (!ctx[256]) {
364 while (((intptr_t) buffer & 3) && buffer < end)
365 crc = ctx[((uint8_t) crc) ^ *buffer++] ^ (crc >> 8);
366
367 while (buffer < end - 3) {
368 crc ^= av_le2ne32(*(const uint32_t *) buffer); buffer += 4;
369 crc = ctx[3 * 256 + ( crc & 0xFF)] ^
370 ctx[2 * 256 + ((crc >> 8 ) & 0xFF)] ^
371 ctx[1 * 256 + ((crc >> 16) & 0xFF)] ^
372 ctx[0 * 256 + ((crc >> 24) )];
373 }
374 }
375#endif
376 while (buffer < end)
377 crc = ctx[((uint8_t) crc) ^ *buffer++] ^ (crc >> 8);
378
379 return crc;
380}
