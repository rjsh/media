49void av_force_cpu_flags(int arg){
50 if ( (arg & ( AV_CPU_FLAG_3DNOW |
51 AV_CPU_FLAG_3DNOWEXT |
52 AV_CPU_FLAG_MMXEXT |
53 AV_CPU_FLAG_SSE |
54 AV_CPU_FLAG_SSE2 |
55 AV_CPU_FLAG_SSE2SLOW |
56 AV_CPU_FLAG_SSE3 |
57 AV_CPU_FLAG_SSE3SLOW |
58 AV_CPU_FLAG_SSSE3 |
59 AV_CPU_FLAG_SSE4 |
60 AV_CPU_FLAG_SSE42 |
61 AV_CPU_FLAG_AVX |
62 AV_CPU_FLAG_AVXSLOW |
63 AV_CPU_FLAG_XOP |
64 AV_CPU_FLAG_FMA3 |
65 AV_CPU_FLAG_FMA4 |
66 AV_CPU_FLAG_AVX2 ))
67 && !(arg & AV_CPU_FLAG_MMX)) {
68 av_log(NULL, AV_LOG_WARNING, "MMX implied by specified flags\n");
69 arg |= AV_CPU_FLAG_MMX;
70 }
71
72 flags = arg;
73 checked = arg != -1;
74}
75
76int av_get_cpu_flags(void)
77{
78 if (checked)
79 return flags;
80
81 if (ARCH_AARCH64)
82 flags = ff_get_cpu_flags_aarch64();
83 if (ARCH_ARM)
84 flags = ff_get_cpu_flags_arm();
85 if (ARCH_PPC)
86 flags = ff_get_cpu_flags_ppc();
87 if (ARCH_X86)
88 flags = ff_get_cpu_flags_x86();
89
90 checked = 1;
91 return flags;
92}
93
94void av_set_cpu_flags_mask(int mask)
95{
96 checked = 0;
97 flags = av_get_cpu_flags() & mask;
98 checked = 1;
99}

256int av_cpu_count(void)
257{
258 static volatile int printed;
259
260 int nb_cpus = 1;
261#if HAVE_WINRT
262 SYSTEM_INFO sysinfo;
263#endif
264#if HAVE_SCHED_GETAFFINITY && defined(CPU_COUNT)
265 cpu_set_t cpuset;
266
267 CPU_ZERO(&cpuset);
268
269 if (!sched_getaffinity(0, sizeof(cpuset), &cpuset))
270 nb_cpus = CPU_COUNT(&cpuset);
271#elif HAVE_GETPROCESSAFFINITYMASK
272 DWORD_PTR proc_aff, sys_aff;
273 if (GetProcessAffinityMask(GetCurrentProcess(), &proc_aff, &sys_aff))
274 nb_cpus = av_popcount64(proc_aff);
275#elif HAVE_SYSCTL && defined(HW_NCPU)
276 int mib[2] = { CTL_HW, HW_NCPU };
277 size_t len = sizeof(nb_cpus);
278
279 if (sysctl(mib, 2, &nb_cpus, &len, NULL, 0) == -1)
280 nb_cpus = 0;
281#elif HAVE_SYSCONF && defined(_SC_NPROC_ONLN)
282 nb_cpus = sysconf(_SC_NPROC_ONLN);
283#elif HAVE_SYSCONF && defined(_SC_NPROCESSORS_ONLN)
284 nb_cpus = sysconf(_SC_NPROCESSORS_ONLN);
285#elif HAVE_WINRT
286 GetNativeSystemInfo(&sysinfo);
287 nb_cpus = sysinfo.dwNumberOfProcessors;
288#endif
289
290 if (!printed) {
291 av_log(NULL, AV_LOG_DEBUG, "detected %d logical cores\n", nb_cpus);
292 printed = 1;
293 }
294
295 return nb_cpus;
296}
