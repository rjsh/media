26static int xvag_probe(AVProbeData *p)
27{
28 if (memcmp(p->buf, "XVAG", 4) ||
29 memcmp(p->buf+32, "fmat", 4))
30 return 0;
31
32 return AVPROBE_SCORE_MAX;
33}

35static int xvag_read_header(AVFormatContext *s)
36{
37 unsigned offset, big_endian, codec;
38 AVStream *st;
39
40 avio_skip(s->pb, 4);
41
42 st = avformat_new_stream(s, NULL);
43 if (!st)
44 return AVERROR(ENOMEM);
45
46 st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
47
48 offset = avio_rl32(s->pb);
49 big_endian = offset > av_bswap32(offset);
50 if (big_endian) {
51 offset = av_bswap32(offset);
52 avio_skip(s->pb, 28);
53 codec = avio_rb32(s->pb);
54 st->codecpar->channels = avio_rb32(s->pb);
55 avio_skip(s->pb, 4);
56 st->duration = avio_rb32(s->pb);
57 avio_skip(s->pb, 8);
58 st->codecpar->sample_rate = avio_rb32(s->pb);
59 } else {
60 avio_skip(s->pb, 28);
61 codec = avio_rl32(s->pb);
62 st->codecpar->channels = avio_rl32(s->pb);
63 avio_skip(s->pb, 4);
64 st->duration = avio_rl32(s->pb);
65 avio_skip(s->pb, 8);
66 st->codecpar->sample_rate = avio_rl32(s->pb);
67 }
68
69 if (st->codecpar->sample_rate <= 0)
70 return AVERROR_INVALIDDATA;
71 if (st->codecpar->channels <= 0)
72 return AVERROR_INVALIDDATA;
73
74 switch (codec) {
75 case 0x1c:
76 st->codecpar->codec_id = AV_CODEC_ID_ADPCM_PSX;
77 st->codecpar->block_align = 16 * st->codecpar->channels;
78 break;
79 default:
80 avpriv_request_sample(s, "codec %X", codec);
81 return AVERROR_PATCHWELCOME;
82 };
83
84 avio_skip(s->pb, offset - avio_tell(s->pb));
85
86 if (avio_rb16(s->pb) == 0xFFFB) {
87 st->codecpar->codec_id = AV_CODEC_ID_MP3;
88 st->codecpar->block_align = 0x1000;
89 st->need_parsing = AVSTREAM_PARSE_FULL_RAW;
90 }
91
92 avio_skip(s->pb, -2);
93 avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
94
95 return 0;
96}

98static int xvag_read_packet(AVFormatContext *s, AVPacket *pkt)
99{
100 AVCodecParameters *par = s->streams[0]->codecpar;
101
102 return av_get_packet(s->pb, pkt, par->block_align);
103}

105AVInputFormat ff_xvag_demuxer = {
106 .name = "xvag",
107 .long_name = NULL_IF_CONFIG_SMALL("Sony PS3 XVAG"),
108 .read_probe = xvag_probe,
109 .read_header = xvag_read_header,
110 .read_packet = xvag_read_packet,
111 .extensions = "xvag",
112};
