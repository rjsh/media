69static int mp3_read_probe(AVProbeData *p)
70{
71 int max_frames, first_frames = 0;
72 int whole_used = 0;
73 int frames, ret;
74 uint32_t header;
75 const uint8_t *buf, *buf0, *buf2, *end;
76
77 buf0 = p->buf;
78 end = p->buf + p->buf_size - sizeof(uint32_t);
79 while(buf0 < end && !*buf0)
80 buf0++;
81
82 max_frames = 0;
83 buf = buf0;
84
85 for(; buf < end; buf= buf2+1) {
86 buf2 = buf;
87 for(frames = 0; buf2 < end; frames++) {
88 MPADecodeHeader h;
89
90 header = AV_RB32(buf2);
91 ret = avpriv_mpegaudio_decode_header(&h, header);
92 if (ret != 0)
93 break;
94 buf2 += h.frame_size;
95 }
96 max_frames = FFMAX(max_frames, frames);
97 if(buf == buf0) {
98 first_frames= frames;
99 if (buf2 == end + sizeof(uint32_t))
100 whole_used = 1;
101 }
102 }
103 // keep this in sync with ac3 probe, both need to avoid
104 // issues with MPEG-files!
105 if (first_frames>=7) return AVPROBE_SCORE_EXTENSION + 1;
106 else if(max_frames>200)return AVPROBE_SCORE_EXTENSION;
107 else if(max_frames>=4 && max_frames >= p->buf_size/10000) return AVPROBE_SCORE_EXTENSION / 2;
108 else if(ff_id3v2_match(buf0, ID3v2_DEFAULT_MAGIC) && 2*ff_id3v2_tag_len(buf0) >= p->buf_size)
109 return p->buf_size < PROBE_BUF_MAX ? AVPROBE_SCORE_EXTENSION / 4 : AVPROBE_SCORE_EXTENSION - 2;
110 else if(first_frames > 1 && whole_used) return 5;
111 else if(max_frames>=1 && max_frames >= p->buf_size/10000) return 1;
112 else return 0;
113//mpegps_mp3_unrecognized_format.mpg has max_frames=3
114}

116static void read_xing_toc(AVFormatContext *s, int64_t filesize, int64_t duration)
117{
118 int i;
119 MP3DecContext *mp3 = s->priv_data;
120 int fast_seek = s->flags & AVFMT_FLAG_FAST_SEEK;
121 int fill_index = (mp3->usetoc || fast_seek) && duration > 0;
122
123 if (!filesize &&
124 !(filesize = avio_size(s->pb))) {
125 av_log(s, AV_LOG_WARNING, "Cannot determine file size, skipping TOC table.\n");
126 fill_index = 0;
127 }
128
129 for (i = 0; i < XING_TOC_COUNT; i++) {
130 uint8_t b = avio_r8(s->pb);
131 if (fill_index)
132 av_add_index_entry(s->streams[0],
133 av_rescale(b, filesize, 256),
134 av_rescale(i, duration, XING_TOC_COUNT),
135 0, 0, AVINDEX_KEYFRAME);
136 }
137 if (fill_index)
138 mp3->xing_toc = 1;
139}

141static void mp3_parse_info_tag(AVFormatContext *s, AVStream *st,
142 MPADecodeHeader *c, uint32_t spf)
143{
144#define LAST_BITS(k, n) ((k) & ((1 << (n)) - 1))
145#define MIDDLE_BITS(k, m, n) LAST_BITS((k) >> (m), ((n) - (m)))
146
147 uint16_t crc;
148 uint32_t v;
149
150 char version[10];
151
152 uint32_t peak = 0;
153 int32_t r_gain = INT32_MIN, a_gain = INT32_MIN;
154
155 MP3DecContext *mp3 = s->priv_data;
156 static const int64_t xing_offtbl[2][2] = {{32, 17}, {17,9}};
157 uint64_t fsize = avio_size(s->pb);
158 fsize = fsize >= avio_tell(s->pb) ? fsize - avio_tell(s->pb) : 0;
159
160 /* Check for Xing / Info tag */
161 avio_skip(s->pb, xing_offtbl[c->lsf == 1][c->nb_channels == 1]);
162 v = avio_rb32(s->pb);
163 mp3->is_cbr = v == MKBETAG('I', 'n', 'f', 'o');
164 if (v != MKBETAG('X', 'i', 'n', 'g') && !mp3->is_cbr)
165 return;
166
167 v = avio_rb32(s->pb);
168 if (v & XING_FLAG_FRAMES)
169 mp3->frames = avio_rb32(s->pb);
170 if (v & XING_FLAG_SIZE)
171 mp3->header_filesize = avio_rb32(s->pb);
172 if (fsize && mp3->header_filesize) {
173 uint64_t min, delta;
174 min = FFMIN(fsize, mp3->header_filesize);
175 delta = FFMAX(fsize, mp3->header_filesize) - min;
176 if (fsize > mp3->header_filesize && delta > min >> 4) {
177 mp3->frames = 0;
178 av_log(s, AV_LOG_WARNING,
179 "invalid concatenated file detected - using bitrate for duration\n");
180 } else if (delta > min >> 4) {
181 av_log(s, AV_LOG_WARNING,
182 "filesize and duration do not match (growing file?)\n");
183 }
184 }
185 if (v & XING_FLAG_TOC)
186 read_xing_toc(s, mp3->header_filesize, av_rescale_q(mp3->frames,
187 (AVRational){spf, c->sample_rate},
188 st->time_base));
189 /* VBR quality */
190 if (v & XING_FLAC_QSCALE)
191 avio_rb32(s->pb);
192
193 /* Encoder short version string */
194 memset(version, 0, sizeof(version));
195 avio_read(s->pb, version, 9);
196
197 /* Info Tag revision + VBR method */
198 avio_r8(s->pb);
199
200 /* Lowpass filter value */
201 avio_r8(s->pb);
202
203 /* ReplayGain peak */
204 v = avio_rb32(s->pb);
205 peak = av_rescale(v, 100000, 1 << 23);
206
207 /* Radio ReplayGain */
208 v = avio_rb16(s->pb);
209
210 if (MIDDLE_BITS(v, 13, 15) == 1) {
211 r_gain = MIDDLE_BITS(v, 0, 8) * 10000;
212
213 if (v & (1 << 9))
214 r_gain *= -1;
215 }
216
217 /* Audiophile ReplayGain */
218 v = avio_rb16(s->pb);
219
220 if (MIDDLE_BITS(v, 13, 15) == 2) {
221 a_gain = MIDDLE_BITS(v, 0, 8) * 10000;
222
223 if (v & (1 << 9))
224 a_gain *= -1;
225 }
226
227 /* Encoding flags + ATH Type */
228 avio_r8(s->pb);
229
230 /* if ABR {specified bitrate} else {minimal bitrate} */
231 avio_r8(s->pb);
232
233 /* Encoder delays */
234 v= avio_rb24(s->pb);
235 if(AV_RB32(version) == MKBETAG('L', 'A', 'M', 'E')
236 || AV_RB32(version) == MKBETAG('L', 'a', 'v', 'f')
237 || AV_RB32(version) == MKBETAG('L', 'a', 'v', 'c')
238 ) {
239
240 mp3->start_pad = v>>12;
241 mp3-> end_pad = v&4095;
242 st->start_skip_samples = mp3->start_pad + 528 + 1;
243 if (mp3->frames) {
244 st->first_discard_sample = -mp3->end_pad + 528 + 1 + mp3->frames * (int64_t)spf;
245 st->last_discard_sample = mp3->frames * (int64_t)spf;
246 }
247 if (!st->start_time)
248 st->start_time = av_rescale_q(st->start_skip_samples,
249 (AVRational){1, c->sample_rate},
250 st->time_base);
251 av_log(s, AV_LOG_DEBUG, "pad %d %d\n", mp3->start_pad, mp3-> end_pad);
252 }
253
254 /* Misc */
255 avio_r8(s->pb);
256
257 /* MP3 gain */
258 avio_r8(s->pb);
259
260 /* Preset and surround info */
261 avio_rb16(s->pb);
262
263 /* Music length */
264 avio_rb32(s->pb);
265
266 /* Music CRC */
267 avio_rb16(s->pb);
268
269 /* Info Tag CRC */
270 crc = ffio_get_checksum(s->pb);
271 v = avio_rb16(s->pb);
272
273 if (v == crc) {
274 ff_replaygain_export_raw(st, r_gain, peak, a_gain, 0);
275 av_dict_set(&st->metadata, "encoder", version, 0);
276 }
277}

279static void mp3_parse_vbri_tag(AVFormatContext *s, AVStream *st, int64_t base)
280{
281 uint32_t v;
282 MP3DecContext *mp3 = s->priv_data;
283
284 /* Check for VBRI tag (always 32 bytes after end of mpegaudio header) */
285 avio_seek(s->pb, base + 4 + 32, SEEK_SET);
286 v = avio_rb32(s->pb);
287 if (v == MKBETAG('V', 'B', 'R', 'I')) {
288 /* Check tag version */
289 if (avio_rb16(s->pb) == 1) {
290 /* skip delay and quality */
291 avio_skip(s->pb, 4);
292 mp3->header_filesize = avio_rb32(s->pb);
293 mp3->frames = avio_rb32(s->pb);
294 }
295 }
296}

298/**
299 * Try to find Xing/Info/VBRI tags and compute duration from info therein
300 */
301static int mp3_parse_vbr_tags(AVFormatContext *s, AVStream *st, int64_t base)
302{
303 uint32_t v, spf;
304 MPADecodeHeader c;
305 int vbrtag_size = 0;
306 MP3DecContext *mp3 = s->priv_data;
307 int ret;
308
309 ffio_init_checksum(s->pb, ff_crcA001_update, 0);
310
311 v = avio_rb32(s->pb);
312
313 ret = avpriv_mpegaudio_decode_header(&c, v);
314 if (ret < 0)
315 return ret;
316 else if (ret == 0)
317 vbrtag_size = c.frame_size;
318 if(c.layer != 3)
319 return -1;
320
321 spf = c.lsf ? 576 : 1152; /* Samples per frame, layer 3 */
322
323 mp3->frames = 0;
324 mp3->header_filesize = 0;
325
326 mp3_parse_info_tag(s, st, &c, spf);
327 mp3_parse_vbri_tag(s, st, base);
328
329 if (!mp3->frames && !mp3->header_filesize)
330 return -1;
331
332 /* Skip the vbr tag frame */
333 avio_seek(s->pb, base + vbrtag_size, SEEK_SET);
334
335 if (mp3->frames)
336 st->duration = av_rescale_q(mp3->frames, (AVRational){spf, c.sample_rate},
337 st->time_base);
338 if (mp3->header_filesize && mp3->frames && !mp3->is_cbr)
339 st->codecpar->bit_rate = av_rescale(mp3->header_filesize, 8 * c.sample_rate, mp3->frames * (int64_t)spf);
340
341 return 0;
342}

344static int mp3_read_header(AVFormatContext *s)
345{
346 MP3DecContext *mp3 = s->priv_data;
347 AVStream *st;
348 int64_t off;
349 int ret;
350 int i;
351
352 st = avformat_new_stream(s, NULL);
353 if (!st)
354 return AVERROR(ENOMEM);
355
356 st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
357 st->codecpar->codec_id = AV_CODEC_ID_MP3;
358 st->need_parsing = AVSTREAM_PARSE_FULL_RAW;
359 st->start_time = 0;
360
361 // lcm of all mp3 sample rates
362 avpriv_set_pts_info(st, 64, 1, 14112000);
363
364 s->pb->maxsize = -1;
365 off = avio_tell(s->pb);
366
367 if (!av_dict_get(s->metadata, "", NULL, AV_DICT_IGNORE_SUFFIX))
368 ff_id3v1_read(s);
369
370 if(s->pb->seekable)
371 mp3->filesize = avio_size(s->pb);
372
373 if (mp3_parse_vbr_tags(s, st, off) < 0)
374 avio_seek(s->pb, off, SEEK_SET);
375
376 ret = ff_replaygain_export(st, s->metadata);
377 if (ret < 0)
378 return ret;
379
380 off = avio_tell(s->pb);
381 for (i = 0; i < 64 * 1024; i++) {
382 uint32_t header, header2;
383 int frame_size;
384 if (!(i&1023))
385 ffio_ensure_seekback(s->pb, i + 1024 + 4);
386 frame_size = check(s->pb, off + i, &header);
387 if (frame_size > 0) {
388 ret = avio_seek(s->pb, off, SEEK_SET);
389 if (ret < 0)
390 return ret;
391 ffio_ensure_seekback(s->pb, i + 1024 + frame_size + 4);
392 ret = check(s->pb, off + i + frame_size, &header2);
393 if (ret >= 0 &&
394 (header & SAME_HEADER_MASK) == (header2 & SAME_HEADER_MASK))
395 {
396 av_log(s, i > 0 ? AV_LOG_INFO : AV_LOG_VERBOSE, "Skipping %d bytes of junk at %"PRId64".\n", i, off);
397 ret = avio_seek(s->pb, off + i, SEEK_SET);
398 if (ret < 0)
399 return ret;
400 break;
401 } else if (ret == CHECK_SEEK_FAILED) {
402 av_log(s, AV_LOG_ERROR, "Invalid frame size (%d): Could not seek to %"PRId64".\n", frame_size, off + i + frame_size);
403 return AVERROR(EINVAL);
404 }
405 } else if (frame_size == CHECK_SEEK_FAILED) {
406 av_log(s, AV_LOG_ERROR, "Failed to read frame size: Could not seek to %"PRId64".\n", (int64_t) (i + 1024 + frame_size + 4));
407 return AVERROR(EINVAL);
408 }
409 ret = avio_seek(s->pb, off, SEEK_SET);
410 if (ret < 0)
411 return ret;
412 }
413
414 // the seek index is relative to the end of the xing vbr headers
415 for (i = 0; i < st->nb_index_entries; i++)
416 st->index_entries[i].pos += avio_tell(s->pb);
417
418 /* the parameters will be extracted from the compressed bitstream */
419 return 0;
420}

424static int mp3_read_packet(AVFormatContext *s, AVPacket *pkt)
425{
426 MP3DecContext *mp3 = s->priv_data;
427 int ret, size;
428 int64_t pos;
429
430 size= MP3_PACKET_SIZE;
431 pos = avio_tell(s->pb);
432 if(mp3->filesize > ID3v1_TAG_SIZE && pos < mp3->filesize)
433 size= FFMIN(size, mp3->filesize - pos);
434
435 ret= av_get_packet(s->pb, pkt, size);
436 if (ret <= 0) {
437 if(ret<0)
438 return ret;
439 return AVERROR_EOF;
440 }
441
442 pkt->flags &= ~AV_PKT_FLAG_CORRUPT;
443 pkt->stream_index = 0;
444
445 return ret;
446}

448#define SEEK_WINDOW 4096
449
450static int check(AVIOContext *pb, int64_t pos, uint32_t *ret_header)
451{
452 int64_t ret = avio_seek(pb, pos, SEEK_SET);
453 uint8_t header_buf[4];
454 unsigned header;
455 MPADecodeHeader sd;
456 if (ret < 0)
457 return CHECK_SEEK_FAILED;
458
459 ret = avio_read(pb, &header_buf[0], 4);
460 if (ret < 0)
461 return CHECK_SEEK_FAILED;
462
463 header = AV_RB32(&header_buf[0]);
464 if (ff_mpa_check_header(header) < 0)
465 return CHECK_WRONG_HEADER;
466 if (avpriv_mpegaudio_decode_header(&sd, header) == 1)
467 return CHECK_WRONG_HEADER;
468
469 if (ret_header)
470 *ret_header = header;
471 return sd.frame_size;
472}

474static int64_t mp3_sync(AVFormatContext *s, int64_t target_pos, int flags)
475{
476 int dir = (flags&AVSEEK_FLAG_BACKWARD) ? -1 : 1;
477 int64_t best_pos;
478 int best_score, i, j;
479 int64_t ret;
480
481 avio_seek(s->pb, FFMAX(target_pos - SEEK_WINDOW, 0), SEEK_SET);
482 ret = avio_seek(s->pb, target_pos, SEEK_SET);
483 if (ret < 0)
484 return ret;
485
486#define MIN_VALID 3
487 best_pos = target_pos;
488 best_score = 999;
489 for(i=0; i<SEEK_WINDOW; i++) {
490 int64_t pos = target_pos + (dir > 0 ? i - SEEK_WINDOW/4 : -i);
491 int64_t candidate = -1;
492 int score = 999;
493
494 if (pos < 0)
495 continue;
496
497 for(j=0; j<MIN_VALID; j++) {
498 ret = check(s->pb, pos, NULL);
499 if(ret < 0) {
500 if (ret == CHECK_WRONG_HEADER) {
501 break;
502 } else if (ret == CHECK_SEEK_FAILED) {
503 av_log(s, AV_LOG_ERROR, "Could not seek to %"PRId64".\n", pos);
504 return AVERROR(EINVAL);
505 }
506 }
507 if ((target_pos - pos)*dir <= 0 && abs(MIN_VALID/2-j) < score) {
508 candidate = pos;
509 score = abs(MIN_VALID/2-j);
510 }
511 pos += ret;
512 }
513 if (best_score > score && j == MIN_VALID) {
514 best_pos = candidate;
515 best_score = score;
516 if(score == 0)
517 break;
518 }
519 }
520
521 return avio_seek(s->pb, best_pos, SEEK_SET);
522}

524static int mp3_seek(AVFormatContext *s, int stream_index, int64_t timestamp,
525 int flags)
526{
527 MP3DecContext *mp3 = s->priv_data;
528 AVIndexEntry *ie, ie1;
529 AVStream *st = s->streams[0];
530 int64_t best_pos;
531 int fast_seek = s->flags & AVFMT_FLAG_FAST_SEEK;
532 int64_t filesize = mp3->header_filesize;
533
534 if (filesize <= 0) {
535 int64_t size = avio_size(s->pb);
536 if (size > 0 && size > s->internal->data_offset)
537 filesize = size - s->internal->data_offset;
538 }
539
540 if (mp3->xing_toc && (mp3->usetoc || (fast_seek && !mp3->is_cbr))) {
541 int64_t ret = av_index_search_timestamp(st, timestamp, flags);
542
543 // NOTE: The MP3 TOC is not a precise lookup table. Accuracy is worse
544 // for bigger files.
545 av_log(s, AV_LOG_WARNING, "Using MP3 TOC to seek; may be imprecise.\n");
546
547 if (ret < 0)
548 return ret;
549
550 ie = &st->index_entries[ret];
551 } else if (fast_seek && st->duration > 0 && filesize > 0) {
552 if (!mp3->is_cbr)
553 av_log(s, AV_LOG_WARNING, "Using scaling to seek VBR MP3; may be imprecise.\n");
554
555 ie = &ie1;
556 timestamp = av_clip64(timestamp, 0, st->duration);
557 ie->timestamp = timestamp;
558 ie->pos = av_rescale(timestamp, filesize, st->duration) + s->internal->data_offset;
559 } else {
560 return -1; // generic index code
561 }
562
563 best_pos = mp3_sync(s, ie->pos, flags);
564 if (best_pos < 0)
565 return best_pos;
566
567 if (mp3->is_cbr && ie == &ie1 && mp3->frames) {
568 int frame_duration = av_rescale(st->duration, 1, mp3->frames);
569 ie1.timestamp = frame_duration * av_rescale(best_pos - s->internal->data_offset, mp3->frames, mp3->header_filesize);
570 }
571
572 ff_update_cur_dts(s, st, ie->timestamp);
573 return 0;
574}
