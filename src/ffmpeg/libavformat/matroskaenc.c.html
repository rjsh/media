176static int ebml_id_size(unsigned int id)
177{
178 return (av_log2(id + 1) - 1) / 7 + 1;
179}
180
181static void put_ebml_id(AVIOContext *pb, unsigned int id)
182{
183 int i = ebml_id_size(id);
184 while (i--)
185 avio_w8(pb, (uint8_t)(id >> (i * 8)));
186}

188/**
189 * Write an EBML size meaning "unknown size".
190 *
191 * @param bytes The number of bytes the size should occupy (maximum: 8).
192 */
193static void put_ebml_size_unknown(AVIOContext *pb, int bytes)
194{
195 av_assert0(bytes <= 8);
196 avio_w8(pb, 0x1ff >> bytes);
197 ffio_fill(pb, 0xff, bytes - 1);
198}

200/**
201 * Calculate how many bytes are needed to represent a given number in EBML.
202 */
203static int ebml_num_size(uint64_t num)
204{
205 int bytes = 1;
206 while ((num + 1) >> bytes * 7)
207 bytes++;
208 return bytes;
209}
210
211/**
212 * Write a number in EBML variable length format.
213 *
214 * @param bytes The number of bytes that need to be used to write the number.
215 * If zero, any number of bytes can be used.
216 */
217static void put_ebml_num(AVIOContext *pb, uint64_t num, int bytes)
218{
219 int i, needed_bytes = ebml_num_size(num);
220
221 // sizes larger than this are currently undefined in EBML
222 av_assert0(num < (1ULL << 56) - 1);
223
224 if (bytes == 0)
225 // don't care how many bytes are used, so use the min
226 bytes = needed_bytes;
227 // the bytes needed to write the given size would exceed the bytes
228 // that we need to use, so write unknown size. This shouldn't happen.
229 av_assert0(bytes >= needed_bytes);
230
231 num |= 1ULL << bytes * 7;
232 for (i = bytes - 1; i >= 0; i--)
233 avio_w8(pb, (uint8_t)(num >> i * 8));
234}
235
236static void put_ebml_uint(AVIOContext *pb, unsigned int elementid, uint64_t val)
237{
238 int i, bytes = 1;
239 uint64_t tmp = val;
240 while (tmp >>= 8)
241 bytes++;
242
243 put_ebml_id(pb, elementid);
244 put_ebml_num(pb, bytes, 0);
245 for (i = bytes - 1; i >= 0; i--)
246 avio_w8(pb, (uint8_t)(val >> i * 8));
247}
248
249static void put_ebml_sint(AVIOContext *pb, unsigned int elementid, int64_t val)
250{
251 int i, bytes = 1;
252 uint64_t tmp = 2*(val < 0 ? val^-1 : val);
253
254 while (tmp>>=8) bytes++;
255
256 put_ebml_id(pb, elementid);
257 put_ebml_num(pb, bytes, 0);
258 for (i = bytes - 1; i >= 0; i--)
259 avio_w8(pb, (uint8_t)(val >> i * 8));
260}

262static void put_ebml_float(AVIOContext *pb, unsigned int elementid, double val)
263{
264 put_ebml_id(pb, elementid);
265 put_ebml_num(pb, 8, 0);
266 avio_wb64(pb, av_double2int(val));
267}

283/**
284 * Write a void element of a given size. Useful for reserving space in
285 * the file to be written to later.
286 *
287 * @param size The number of bytes to reserve, which must be at least 2.
288 */
289static void put_ebml_void(AVIOContext *pb, uint64_t size)
290{
291 int64_t currentpos = avio_tell(pb);
292
293 av_assert0(size >= 2);
294
295 put_ebml_id(pb, EBML_ID_VOID);
296 // we need to subtract the length needed to store the size from the
297 // size we need to reserve so 2 cases, we use 8 bytes to store the
298 // size if possible, 1 byte otherwise
299 if (size < 10)
300 put_ebml_num(pb, size - 2, 0);
301 else
302 put_ebml_num(pb, size - 9, 8);
303 ffio_fill(pb, 0, currentpos + size - avio_tell(pb));
304}

306static ebml_master start_ebml_master(AVIOContext *pb, unsigned int elementid,
307 uint64_t expectedsize)
308{
309 int bytes = expectedsize ? ebml_num_size(expectedsize) : 8;
310 put_ebml_id(pb, elementid);
311 put_ebml_size_unknown(pb, bytes);
312 return (ebml_master) {avio_tell(pb), bytes };
313}
314
315static void end_ebml_master(AVIOContext *pb, ebml_master master)
316{
317 int64_t pos = avio_tell(pb);
318
319 if (avio_seek(pb, master.pos - master.sizebytes, SEEK_SET) < 0)
320 return;
321 put_ebml_num(pb, pos - master.pos, master.sizebytes);
322 avio_seek(pb, pos, SEEK_SET);
323}

325static int start_ebml_master_crc32(AVIOContext *pb, AVIOContext **dyn_cp, ebml_master *master,
326 unsigned int elementid, uint64_t expectedsize)
327{
328 int ret;
329
330 if ((ret = avio_open_dyn_buf(dyn_cp)) < 0)
331 return ret;
332
333 if (pb->seekable)
334 *master = start_ebml_master(pb, elementid, expectedsize);
335 else
336 *master = start_ebml_master(*dyn_cp, elementid, expectedsize);
337
338 return 0;
339}

341static void end_ebml_master_crc32(AVIOContext *pb, AVIOContext **dyn_cp, MatroskaMuxContext *mkv,
342 ebml_master master)
343{
344 uint8_t *buf, crc[4];
345 int size;
346
347 if (pb->seekable) {
348 size = avio_close_dyn_buf(*dyn_cp, &buf);
349 if (mkv->write_crc && mkv->mode != MODE_WEBM) {
350 AV_WL32(crc, av_crc(av_crc_get_table(AV_CRC_32_IEEE_LE), UINT32_MAX, buf, size) ^ UINT32_MAX);
351 put_ebml_binary(pb, EBML_ID_CRC32, crc, sizeof(crc));
352 }
353 avio_write(pb, buf, size);
354 end_ebml_master(pb, master);
355 } else {
356 end_ebml_master(*dyn_cp, master);
357 size = avio_close_dyn_buf(*dyn_cp, &buf);
358 avio_write(pb, buf, size);
359 }
360 av_free(buf);
361 *dyn_cp = NULL;
362}

391/**
392 * Initialize a mkv_seekhead element to be ready to index level 1 Matroska
393 * elements. If a maximum number of elements is specified, enough space
394 * will be reserved at the current file location to write a seek head of
395 * that size.
396 *
397 * @param segment_offset The absolute offset to the position in the file
398 * where the segment begins.
399 * @param numelements The maximum number of elements that will be indexed
400 * by this seek head, 0 if unlimited.
401 */
402static mkv_seekhead *mkv_start_seekhead(AVIOContext *pb, int64_t segment_offset,
403 int numelements)
404{
405 mkv_seekhead *new_seekhead = av_mallocz(sizeof(mkv_seekhead));
406 if (!new_seekhead)
407 return NULL;
408
409 new_seekhead->segment_offset = segment_offset;
410
411 if (numelements > 0) {
412 new_seekhead->filepos = avio_tell(pb);
413 // 21 bytes max for a seek entry, 10 bytes max for the SeekHead ID
414 // and size, 6 bytes for a CRC32 element, and 3 bytes to guarantee
415 // that an EBML void element will fit afterwards
416 new_seekhead->reserved_size = numelements * MAX_SEEKENTRY_SIZE + 19;
417 new_seekhead->max_entries = numelements;
418 put_ebml_void(pb, new_seekhead->reserved_size);
419 }
420 return new_seekhead;
421}

423static int mkv_add_seekhead_entry(mkv_seekhead *seekhead, unsigned int elementid, uint64_t filepos)
424{
425 mkv_seekhead_entry *entries = seekhead->entries;
426
427 // don't store more elements than we reserved space for
428 if (seekhead->max_entries > 0 && seekhead->max_entries <= seekhead->num_entries)
429 return -1;
430
431 entries = av_realloc_array(entries, seekhead->num_entries + 1, sizeof(mkv_seekhead_entry));
432 if (!entries)
433 return AVERROR(ENOMEM);
434 seekhead->entries = entries;
435
436 seekhead->entries[seekhead->num_entries].elementid = elementid;
437 seekhead->entries[seekhead->num_entries++].segmentpos = filepos - seekhead->segment_offset;
438
439 return 0;
440}

442/**
443 * Write the seek head to the file and free it. If a maximum number of
444 * elements was specified to mkv_start_seekhead(), the seek head will
445 * be written at the location reserved for it. Otherwise, it is written
446 * at the current location in the file.
447 *
448 * @return The file offset where the seekhead was written,
449 * -1 if an error occurred.
450 */
451static int64_t mkv_write_seekhead(AVIOContext *pb, MatroskaMuxContext *mkv)
452{
453 AVIOContext *dyn_cp;
454 mkv_seekhead *seekhead = mkv->main_seekhead;
455 ebml_master metaseek, seekentry;
456 int64_t currentpos;
457 int i;
458
459 currentpos = avio_tell(pb);
460
461 if (seekhead->reserved_size > 0) {
462 if (avio_seek(pb, seekhead->filepos, SEEK_SET) < 0) {
463 currentpos = -1;
464 goto fail;
465 }
466 }
467
468 if (start_ebml_master_crc32(pb, &dyn_cp, &metaseek, MATROSKA_ID_SEEKHEAD,
469 seekhead->reserved_size) < 0) {
470 currentpos = -1;
471 goto fail;
472 }
473
474 for (i = 0; i < seekhead->num_entries; i++) {
475 mkv_seekhead_entry *entry = &seekhead->entries[i];
476
477 seekentry = start_ebml_master(dyn_cp, MATROSKA_ID_SEEKENTRY, MAX_SEEKENTRY_SIZE);
478
479 put_ebml_id(dyn_cp, MATROSKA_ID_SEEKID);
480 put_ebml_num(dyn_cp, ebml_id_size(entry->elementid), 0);
481 put_ebml_id(dyn_cp, entry->elementid);
482
483 put_ebml_uint(dyn_cp, MATROSKA_ID_SEEKPOSITION, entry->segmentpos);
484 end_ebml_master(dyn_cp, seekentry);
485 }
486 end_ebml_master_crc32(pb, &dyn_cp, mkv, metaseek);
487
488 if (seekhead->reserved_size > 0) {
489 uint64_t remaining = seekhead->filepos + seekhead->reserved_size - avio_tell(pb);
490 put_ebml_void(pb, remaining);
491 avio_seek(pb, currentpos, SEEK_SET);
492
493 currentpos = seekhead->filepos;
494 }
495fail:
496 av_freep(&mkv->main_seekhead->entries);
497 av_freep(&mkv->main_seekhead);
498
499 return currentpos;
500}

513 int64_t cluster_pos, int64_t relative_pos, int64_t duration)
514{

521 if (!entries)
522 return AVERROR(ENOMEM);
523 cues->entries = entries;
524
525 cues->entries[cues->num_entries].pts = ts;
526 cues->entries[cues->num_entries].stream_idx = stream;
527 cues->entries[cues->num_entries].tracknum = tracknum;
528 cues->entries[cues->num_entries].cluster_pos = cluster_pos - cues->segment_offset;
529 cues->entries[cues->num_entries].relative_pos = relative_pos;
530 cues->entries[cues->num_entries++].duration = duration;
531
532 return 0;
533}

535static int64_t mkv_write_cues(AVFormatContext *s, mkv_cues *cues, mkv_track *tracks, int num_tracks)
536{
537 MatroskaMuxContext *mkv = s->priv_data;
538 AVIOContext *dyn_cp, *pb = s->pb;
539 ebml_master cues_element;
540 int64_t currentpos;
541 int i, j, ret;
542
543 currentpos = avio_tell(pb);
544 ret = start_ebml_master_crc32(pb, &dyn_cp, &cues_element, MATROSKA_ID_CUES, 0);
545 if (ret < 0)
546 return ret;
547
548 for (i = 0; i < cues->num_entries; i++) {
549 ebml_master cuepoint, track_positions;
550 mkv_cuepoint *entry = &cues->entries[i];
551 uint64_t pts = entry->pts;
552 int ctp_nb = 0;
553
554 // Calculate the number of entries, so we know the element size
555 for (j = 0; j < num_tracks; j++)
556 tracks[j].has_cue = 0;
557 for (j = 0; j < cues->num_entries - i && entry[j].pts == pts; j++) {
558 int tracknum = entry[j].stream_idx;
559 av_assert0(tracknum>=0 && tracknum<num_tracks);
560 if (tracks[tracknum].has_cue && s->streams[tracknum]->codecpar->codec_type != AVMEDIA_TYPE_SUBTITLE)
561 continue;
562 tracks[tracknum].has_cue = 1;
563 ctp_nb ++;
564 }
565
566 cuepoint = start_ebml_master(dyn_cp, MATROSKA_ID_POINTENTRY, MAX_CUEPOINT_SIZE(ctp_nb));
567 put_ebml_uint(dyn_cp, MATROSKA_ID_CUETIME, pts);
568
569 // put all the entries from different tracks that have the exact same
570 // timestamp into the same CuePoint
571 for (j = 0; j < num_tracks; j++)
572 tracks[j].has_cue = 0;
573 for (j = 0; j < cues->num_entries - i && entry[j].pts == pts; j++) {
574 int tracknum = entry[j].stream_idx;
575 av_assert0(tracknum>=0 && tracknum<num_tracks);
576 if (tracks[tracknum].has_cue && s->streams[tracknum]->codecpar->codec_type != AVMEDIA_TYPE_SUBTITLE)
577 continue;
578 tracks[tracknum].has_cue = 1;
579 track_positions = start_ebml_master(dyn_cp, MATROSKA_ID_CUETRACKPOSITION, MAX_CUETRACKPOS_SIZE);
580 put_ebml_uint(dyn_cp, MATROSKA_ID_CUETRACK , entry[j].tracknum );
581 put_ebml_uint(dyn_cp, MATROSKA_ID_CUECLUSTERPOSITION , entry[j].cluster_pos);
582 put_ebml_uint(dyn_cp, MATROSKA_ID_CUERELATIVEPOSITION, entry[j].relative_pos);
583 if (entry[j].duration != -1)
584 put_ebml_uint(dyn_cp, MATROSKA_ID_CUEDURATION , entry[j].duration);
585 end_ebml_master(dyn_cp, track_positions);
586 }
587 i += j - 1;
588 end_ebml_master(dyn_cp, cuepoint);
589 }
590 end_ebml_master_crc32(pb, &dyn_cp, mkv, cues_element);
591
592 return currentpos;
593}

595static int put_xiph_codecpriv(AVFormatContext *s, AVIOContext *pb, AVCodecParameters *par)
596{
597 const uint8_t *header_start[3];
598 int header_len[3];
599 int first_header_size;
600 int j;
601
602 if (par->codec_id == AV_CODEC_ID_VORBIS)
603 first_header_size = 30;
604 else
605 first_header_size = 42;
606
607 if (avpriv_split_xiph_headers(par->extradata, par->extradata_size,
608 first_header_size, header_start, header_len) < 0) {
609 av_log(s, AV_LOG_ERROR, "Extradata corrupt.\n");
610 return -1;
611 }
612
613 avio_w8(pb, 2); // number packets - 1
614 for (j = 0; j < 2; j++) {
615 put_xiph_size(pb, header_len[j]);
616 }
617 for (j = 0; j < 3; j++)
618 avio_write(pb, header_start[j], header_len[j]);
619
620 return 0;
621}

632static int put_flac_codecpriv(AVFormatContext *s,
633 AVIOContext *pb, AVCodecParameters *par)
634{
635 int write_comment = (par->channel_layout &&
636 !(par->channel_layout & ~0x3ffffULL) &&
637 !ff_flac_is_native_layout(par->channel_layout));
638 int ret = ff_flac_write_header(pb, par->extradata, par->extradata_size,
639 !write_comment);
640
641 if (ret < 0)
642 return ret;
643
644 if (write_comment) {
645 const char *vendor = (s->flags & AVFMT_FLAG_BITEXACT) ?
646 "Lavf" : LIBAVFORMAT_IDENT;
647 AVDictionary *dict = NULL;
648 uint8_t buf[32], *data, *p;
649 int64_t len;
650
651 snprintf(buf, sizeof(buf), "0x%"PRIx64, par->channel_layout);
652 av_dict_set(&dict, "WAVEFORMATEXTENSIBLE_CHANNEL_MASK", buf, 0);
653
654 len = ff_vorbiscomment_length(dict, vendor);
655 if (len >= ((1<<24) - 4))
656 return AVERROR(EINVAL);
657
658 data = av_malloc(len + 4);
659 if (!data) {
660 av_dict_free(&dict);
661 return AVERROR(ENOMEM);
662 }
663
664 data[0] = 0x84;
665 AV_WB24(data + 1, len);
666
667 p = data + 4;

913static int mkv_write_stereo_mode(AVFormatContext *s, AVIOContext *pb,
914 AVStream *st, int mode, int *h_width, int *h_height)
915{
916 int i;
917 int ret = 0;
918 AVDictionaryEntry *tag;
919 MatroskaVideoStereoModeType format = MATROSKA_VIDEO_STEREOMODE_TYPE_NB;
920
921 *h_width = 1;
922 *h_height = 1;
923 // convert metadata into proper side data and add it to the stream
924 if ((tag = av_dict_get(st->metadata, "stereo_mode", NULL, 0)) ||
925 (tag = av_dict_get( s->metadata, "stereo_mode", NULL, 0))) {
926 int stereo_mode = atoi(tag->value);
927
928 for (i=0; i<MATROSKA_VIDEO_STEREOMODE_TYPE_NB; i++)
929 if (!strcmp(tag->value, ff_matroska_video_stereo_mode[i])){
930 stereo_mode = i;
931 break;
932 }
933
934 if (stereo_mode < MATROSKA_VIDEO_STEREOMODE_TYPE_NB &&
935 stereo_mode != 10 && stereo_mode != 12) {
936 int ret = ff_mkv_stereo3d_conv(st, stereo_mode);
937 if (ret < 0)
938 return ret;
939 }
940 }
941
942 // iterate to find the stereo3d side data
943 for (i = 0; i < st->nb_side_data; i++) {
944 AVPacketSideData sd = st->side_data[i];
945 if (sd.type == AV_PKT_DATA_STEREO3D) {
946 AVStereo3D *stereo = (AVStereo3D *)sd.data;
947
948 switch (stereo->type) {
949 case AV_STEREO3D_2D:
950 format = MATROSKA_VIDEO_STEREOMODE_TYPE_MONO;
951 break;
952 case AV_STEREO3D_SIDEBYSIDE:
953 format = (stereo->flags & AV_STEREO3D_FLAG_INVERT)
954 ? MATROSKA_VIDEO_STEREOMODE_TYPE_RIGHT_LEFT
955 : MATROSKA_VIDEO_STEREOMODE_TYPE_LEFT_RIGHT;
956 *h_width = 2;
957 break;
958 case AV_STEREO3D_TOPBOTTOM:
959 format = MATROSKA_VIDEO_STEREOMODE_TYPE_TOP_BOTTOM;
960 if (stereo->flags & AV_STEREO3D_FLAG_INVERT)
961 format--;
962 *h_height = 2;
963 break;
964 case AV_STEREO3D_CHECKERBOARD:
965 format = MATROSKA_VIDEO_STEREOMODE_TYPE_CHECKERBOARD_LR;
966 if (stereo->flags & AV_STEREO3D_FLAG_INVERT)
967 format--;
968 break;
969 case AV_STEREO3D_LINES:
970 format = MATROSKA_VIDEO_STEREOMODE_TYPE_ROW_INTERLEAVED_LR;
971 if (stereo->flags & AV_STEREO3D_FLAG_INVERT)
972 format--;
973 *h_height = 2;
974 break;
975 case AV_STEREO3D_COLUMNS:
976 format = MATROSKA_VIDEO_STEREOMODE_TYPE_COL_INTERLEAVED_LR;
977 if (stereo->flags & AV_STEREO3D_FLAG_INVERT)
978 format--;
979 *h_width = 2;
980 break;
981 case AV_STEREO3D_FRAMESEQUENCE:
982 format = MATROSKA_VIDEO_STEREOMODE_TYPE_BOTH_EYES_BLOCK_LR;
983 if (stereo->flags & AV_STEREO3D_FLAG_INVERT)
984 format++;
985 break;
986 }
987 break;
988 }
989 }
990
991 if (format == MATROSKA_VIDEO_STEREOMODE_TYPE_NB)
992 return ret;
993
994 // if webm, do not write unsupported modes
995 if ((mode == MODE_WEBM &&
996 format > MATROSKA_VIDEO_STEREOMODE_TYPE_TOP_BOTTOM &&
997 format != MATROSKA_VIDEO_STEREOMODE_TYPE_RIGHT_LEFT)
998 || format >= MATROSKA_VIDEO_STEREOMODE_TYPE_NB) {
999 av_log(s, AV_LOG_ERROR,
1000 "The specified stereo mode is not valid.\n");
1001 format = MATROSKA_VIDEO_STEREOMODE_TYPE_NB;
1002 return AVERROR(EINVAL);
1003 }
1004
1005 // write StereoMode if format is valid
1006 put_ebml_uint(pb, MATROSKA_ID_VIDEOSTEREOMODE, format);
1007
1008 return ret;
1009}

1011static int mkv_write_track(AVFormatContext *s, MatroskaMuxContext *mkv,
1012 int i, AVIOContext *pb, int default_stream_exists)
1013{

1016 ebml_master subinfo, track;
1017 int native_id = 0;
1018 int qt_id = 0;
1019 int bit_depth = av_get_bits_per_sample(par->codec_id);
1020 int sample_rate = par->sample_rate;
1021 int output_sample_rate = 0;
1022 int display_width_div = 1;
1023 int display_height_div = 1;
1024 int j, ret;
1025 AVDictionaryEntry *tag;
1026
1027 if (par->codec_type == AVMEDIA_TYPE_ATTACHMENT) {
1028 mkv->have_attachments = 1;
1029 return 0;
1030 }
1031
1032 if (par->codec_type == AVMEDIA_TYPE_AUDIO) {
1033 if (!bit_depth && par->codec_id != AV_CODEC_ID_ADPCM_G726) {
1034 if (par->bits_per_raw_sample)
1035 bit_depth = par->bits_per_raw_sample;
1036 else
1037 bit_depth = av_get_bytes_per_sample(par->format) << 3;
1038 }
1039 if (!bit_depth)
1040 bit_depth = par->bits_per_coded_sample;
1041 }
1042
1043 if (par->codec_id == AV_CODEC_ID_AAC) {
1044 ret = get_aac_sample_rates(s, par, &sample_rate, &output_sample_rate);
1045 if (ret < 0)
1046 return ret;
1047 }
1048
1049 track = start_ebml_master(pb, MATROSKA_ID_TRACKENTRY, 0);
1050 put_ebml_uint (pb, MATROSKA_ID_TRACKNUMBER,
1051 mkv->is_dash ? mkv->dash_track_number : i + 1);
1052 put_ebml_uint (pb, MATROSKA_ID_TRACKUID,
1053 mkv->is_dash ? mkv->dash_track_number : i + 1);

1057 put_ebml_string(pb, MATROSKA_ID_TRACKNAME, tag->value);

1061 } else if (tag && tag->value) {
1062 put_ebml_string(pb, MATROSKA_ID_TRACKLANGUAGE, tag->value);

1066 // if we need to clear it.

1080 native_id = MATROSKA_TRACK_TYPE_METADATA;
1081 } else if (st->disposition & AV_DISPOSITION_METADATA) {
1082 codec_id = "D_WEBVTT/METADATA";
1083 native_id = MATROSKA_TRACK_TYPE_METADATA;
1084 } else {
1085 codec_id = "D_WEBVTT/SUBTITLES";
1086 native_id = MATROSKA_TRACK_TYPE_SUBTITLE;
1087 }
1088 put_ebml_string(pb, MATROSKA_ID_CODECID, codec_id);
1089 } else {
1090 // look for a codec ID string specific to mkv to use,
1091 // if none are found, use AVI codes
1092 for (j = 0; ff_mkv_codec_tags[j].id != AV_CODEC_ID_NONE; j++) {
1093 if (ff_mkv_codec_tags[j].id == par->codec_id) {
1094 put_ebml_string(pb, MATROSKA_ID_CODECID, ff_mkv_codec_tags[j].str);
1095 native_id = 1;
1096 break;
1097 }
1098 }
1099 if (par->codec_id == AV_CODEC_ID_RAWVIDEO && !par->codec_tag) {
1100 if (mkv->allow_raw_vfw) {
1101 native_id = 0;
1102 } else {
1103 av_log(s, AV_LOG_ERROR, "Raw RGB is not supported Natively in Matroska, you can use AVI or NUT or\n"
1104 "If you would like to store it anyway using VFW mode, enable allow_raw_vfw (-allow_raw_vfw 1)\n");
1105 return AVERROR(EINVAL);
1106 }
1107 }
1108 }
1109
1110 if (par->codec_type == AVMEDIA_TYPE_AUDIO && par->initial_padding && par->codec_id == AV_CODEC_ID_OPUS) {
1111 int64_t codecdelay = av_rescale_q(par->initial_padding,
1112 (AVRational){ 1, 48000 },
1113 (AVRational){ 1, 1000000000 });
1114 if (codecdelay < 0) {
1115 av_log(s, AV_LOG_ERROR, "Initial padding is invalid\n");

1117 }
1118// mkv->tracks[i].ts_offset = av_rescale_q(par->initial_padding,
1119// (AVRational){ 1, par->sample_rate },
1120// st->time_base);
1121
1122 put_ebml_uint(pb, MATROSKA_ID_CODECDELAY, codecdelay);
1123 }
1124 if (par->codec_id == AV_CODEC_ID_OPUS) {
1125 put_ebml_uint(pb, MATROSKA_ID_SEEKPREROLL, OPUS_SEEK_PREROLL);
1126 }

1129 par->codec_id == AV_CODEC_ID_VP9 ||
1130 par->codec_id == AV_CODEC_ID_OPUS ||
1131 par->codec_id == AV_CODEC_ID_VORBIS ||
1132 par->codec_id == AV_CODEC_ID_WEBVTT)) {
1133 av_log(s, AV_LOG_ERROR,
1134 "Only VP8 or VP9 video and Vorbis or Opus audio and WebVTT subtitles are supported for WebM.\n");

1146 else

1148
1149 if (!native_id &&
1150 ff_codec_get_tag(ff_codec_movvideo_tags, par->codec_id) &&
1151 ((!ff_codec_get_tag(ff_codec_bmp_tags, par->codec_id) && par->codec_id != AV_CODEC_ID_RAWVIDEO) ||
1152 par->codec_id == AV_CODEC_ID_SVQ1 ||

1186
1187 // write DisplayWidth and DisplayHeight, they contain the size of
1188 // a single source view and/or the display aspect ratio
1189 if (st->sample_aspect_ratio.num) {
1190 int64_t d_width = av_rescale(par->width, st->sample_aspect_ratio.num, st->sample_aspect_ratio.den);
1191 if (d_width > INT_MAX) {
1192 av_log(s, AV_LOG_ERROR, "Overflow in display width\n");
1193 return AVERROR(EINVAL);
1194 }
1195 if (d_width != par->width || display_width_div != 1 || display_height_div != 1) {
1196 put_ebml_uint(pb, MATROSKA_ID_VIDEODISPLAYWIDTH , d_width / display_width_div);
1197 put_ebml_uint(pb, MATROSKA_ID_VIDEODISPLAYHEIGHT, par->height / display_height_div);
1198 }
1199 } else if (display_width_div != 1 || display_height_div != 1) {
1200 put_ebml_uint(pb, MATROSKA_ID_VIDEODISPLAYWIDTH , par->width / display_width_div);
1201 put_ebml_uint(pb, MATROSKA_ID_VIDEODISPLAYHEIGHT, par->height / display_height_div);
1202 } else
1203 put_ebml_uint(pb, MATROSKA_ID_VIDEODISPLAYUNIT, MATROSKA_VIDEO_DISPLAYUNIT_UNKNOWN);
1204
1205 if (par->codec_id == AV_CODEC_ID_RAWVIDEO) {
1206 uint32_t color_space = av_le2ne32(par->codec_tag);
1207 put_ebml_binary(pb, MATROSKA_ID_VIDEOCOLORSPACE, &color_space, sizeof(color_space));
1208 }
1209 if (s->strict_std_compliance <= FF_COMPLIANCE_UNOFFICIAL) {
1210 ret = mkv_write_video_color(pb, par, st);
1211 if (ret < 0)
1212 return ret;
1213 }
1214 end_ebml_master(pb, subinfo);
1215 break;
1216
1217 case AVMEDIA_TYPE_AUDIO:
1218 put_ebml_uint(pb, MATROSKA_ID_TRACKTYPE, MATROSKA_TRACK_TYPE_AUDIO);
1219
1220 if (!native_id)
1221 // no mkv-specific ID, use ACM mode
1222 put_ebml_string(pb, MATROSKA_ID_CODECID, "A_MS/ACM");
1223
1224 subinfo = start_ebml_master(pb, MATROSKA_ID_TRACKAUDIO, 0);
1225 put_ebml_uint (pb, MATROSKA_ID_AUDIOCHANNELS , par->channels);
1226 put_ebml_float (pb, MATROSKA_ID_AUDIOSAMPLINGFREQ, sample_rate);
1227 if (output_sample_rate)
1228 put_ebml_float(pb, MATROSKA_ID_AUDIOOUTSAMPLINGFREQ, output_sample_rate);
1229 if (bit_depth)
1230 put_ebml_uint(pb, MATROSKA_ID_AUDIOBITDEPTH, bit_depth);
1231 end_ebml_master(pb, subinfo);
1232 break;
1233
1234 case AVMEDIA_TYPE_SUBTITLE:
1235 if (!native_id) {
1236 av_log(s, AV_LOG_ERROR, "Subtitle codec %d is not supported.\n", par->codec_id);
1237 return AVERROR(ENOSYS);
1238 }
1239
1240 if (mkv->mode != MODE_WEBM || par->codec_id != AV_CODEC_ID_WEBVTT)
1241 native_id = MATROSKA_TRACK_TYPE_SUBTITLE;
1242
1243 put_ebml_uint(pb, MATROSKA_ID_TRACKTYPE, native_id);
1244 break;
1245 default:
1246 av_log(s, AV_LOG_ERROR, "Only audio, video, and subtitles are supported for Matroska.\n");
1247 return AVERROR(EINVAL);
1248 }
1249
1250 if (mkv->mode != MODE_WEBM || par->codec_id != AV_CODEC_ID_WEBVTT) {
1251 ret = mkv_write_codecprivate(s, pb, par, native_id, qt_id);
1252 if (ret < 0)
1253 return ret;
1254 }
1255
1256 end_ebml_master(pb, track);
1257
1258 return 0;
1259}

1343static int mkv_write_simpletag(AVIOContext *pb, AVDictionaryEntry *t)
1344{

1355 *p = 0;

1357 p = key;

1359 if (*p == ' ')

1451static int mkv_write_tags(AVFormatContext *s)
1452{
1453 MatroskaMuxContext *mkv = s->priv_data;
1454 int i, ret;
1455
1456 ff_metadata_conv_ctx(s, ff_mkv_metadata_conv, NULL);
1457
1458 if (mkv_check_tag(s->metadata, 0)) {
1459 ret = mkv_write_tag(s, s->metadata, 0, 0, &mkv->tags);
1460 if (ret < 0) return ret;
1461 }
1462
1463 for (i = 0; i < s->nb_streams; i++) {
1464 AVStream *st = s->streams[i];
1465
1466 if (st->codecpar->codec_type == AVMEDIA_TYPE_ATTACHMENT)
1467 continue;
1468
1469 if (!mkv_check_tag(st->metadata, MATROSKA_ID_TAGTARGETS_TRACKUID))
1470 continue;
1471
1472 ret = mkv_write_tag(s, st->metadata, MATROSKA_ID_TAGTARGETS_TRACKUID, i + 1, &mkv->tags);
1473 if (ret < 0) return ret;
1474 }
1475
1476 if (s->pb->seekable && !mkv->is_live) {
1477 for (i = 0; i < s->nb_streams; i++) {
1478 AVIOContext *pb;
1479 AVStream *st = s->streams[i];
1480 ebml_master tag_target;
1481 ebml_master tag;
1482
1483 if (st->codecpar->codec_type == AVMEDIA_TYPE_ATTACHMENT)
1484 continue;
1485
1486 mkv_write_tag_targets(s, MATROSKA_ID_TAGTARGETS_TRACKUID, i + 1, &mkv->tags, &tag_target);
1487 pb = mkv->tags_bc;
1488
1489 tag = start_ebml_master(pb, MATROSKA_ID_SIMPLETAG, 0);
1490 put_ebml_string(pb, MATROSKA_ID_TAGNAME, "DURATION");
1491 mkv->stream_duration_offsets[i] = avio_tell(pb);
1492
1493 // Reserve space to write duration as a 20-byte string.
1494 // 2 (ebml id) + 1 (data size) + 20 (data)
1495 put_ebml_void(pb, 23);
1496 end_ebml_master(pb, tag);
1497 end_ebml_master(pb, tag_target);
1498 }
1499 }
1500
1501 for (i = 0; i < s->nb_chapters; i++) {
1502 AVChapter *ch = s->chapters[i];
1503
1504 if (!mkv_check_tag(ch->metadata, MATROSKA_ID_TAGTARGETS_CHAPTERUID))
1505 continue;
1506
1507 ret = mkv_write_tag(s, ch->metadata, MATROSKA_ID_TAGTARGETS_CHAPTERUID, ch->id + mkv->chapter_id_offset, &mkv->tags);
1508 if (ret < 0) return ret;
1509 }
1510
1511 if (mkv->have_attachments) {
1512 for (i = 0; i < mkv->attachments->num_entries; i++) {
1513 mkv_attachment *attachment = &mkv->attachments->entries[i];
1514 AVStream *st = s->streams[attachment->stream_idx];
1515
1516 if (!mkv_check_tag(st->metadata, MATROSKA_ID_TAGTARGETS_ATTACHUID))
1517 continue;
1518
1519 ret = mkv_write_tag(s, st->metadata, MATROSKA_ID_TAGTARGETS_ATTACHUID, attachment->fileuid, &mkv->tags);
1520 if (ret < 0)
1521 return ret;
1522 }
1523 }
1524
1525 if (mkv->tags.pos) {
1526 if (s->pb->seekable && !mkv->is_live)
1527 put_ebml_void(s->pb, avio_tell(mkv->tags_bc) + ((mkv->write_crc && mkv->mode != MODE_WEBM) ? 2 /* ebml id + data size */ + 4 /* CRC32 */ : 0));
1528 else
1529 end_ebml_master_crc32(s->pb, &mkv->tags_bc, mkv, mkv->tags);
1530 }
1531 return 0;
1532}

1534static int mkv_write_attachments(AVFormatContext *s)
1535{
1536 MatroskaMuxContext *mkv = s->priv_data;
1537 AVIOContext *dyn_cp, *pb = s->pb;
1538 ebml_master attachments;
1539 AVLFG c;
1540 int i, ret;
1541
1542 if (!mkv->have_attachments)
1543 return 0;
1544
1545 mkv->attachments = av_mallocz(sizeof(*mkv->attachments));
1546 if (!mkv->attachments)
1547 return ret;
1548
1549 av_lfg_init(&c, av_get_random_seed());
1550
1551 ret = mkv_add_seekhead_entry(mkv->main_seekhead, MATROSKA_ID_ATTACHMENTS, avio_tell(pb));

1555 if (ret < 0) return ret;

1558 AVStream *st = s->streams[i];
1559 ebml_master attached_file;
1560 mkv_attachment *attachment = mkv->attachments->entries;
1561 AVDictionaryEntry *t;
1562 const char *mimetype = NULL;
1563 uint32_t fileuid;
1564
1565 if (st->codecpar->codec_type != AVMEDIA_TYPE_ATTACHMENT)
1566 continue;
1567
1568 attachment = av_realloc_array(attachment, mkv->attachments->num_entries + 1, sizeof(mkv_attachment));
1569 if (!attachment)

1587 if (ff_mkv_mime_tags[i].id == st->codecpar->codec_id) {

1592 if (ff_mkv_image_mime_tags[i].id == st->codecpar->codec_id) {
1593 mimetype = ff_mkv_image_mime_tags[i].str;
1594 break;
1595 }
1596 }

1598 av_log(s, AV_LOG_ERROR, "Attachment stream %d has no mimetype tag and "
1599 "it cannot be deduced from the codec id.\n", i);
1600 return AVERROR(EINVAL);
1601 }
1602
1603 if (s->flags & AVFMT_FLAG_BITEXACT) {
1604 struct AVSHA *sha = av_sha_alloc();

1606 if (!sha)
1607 return AVERROR(ENOMEM);
1608 av_sha_init(sha, 160);
1609 av_sha_update(sha, st->codecpar->extradata, st->codecpar->extradata_size);
1610 av_sha_final(sha, digest);
1611 av_free(sha);
1612 fileuid = AV_RL32(digest);
1613 } else {
1614 fileuid = av_lfg_get(&c);
1615 }
1616 av_log(s, AV_LOG_VERBOSE, "Using %.8"PRIx32" for attachment %d\n",
1617 fileuid, mkv->attachments->num_entries);
1618
1619 put_ebml_string(dyn_cp, MATROSKA_ID_FILEMIMETYPE, mimetype);
1620 put_ebml_binary(dyn_cp, MATROSKA_ID_FILEDATA, st->codecpar->extradata, st->codecpar->extradata_size);
1621 put_ebml_uint(dyn_cp, MATROSKA_ID_FILEUID, fileuid);
1622 end_ebml_master(dyn_cp, attached_file);
1623
1624 mkv->attachments->entries[mkv->attachments->num_entries].stream_idx = i;
1625 mkv->attachments->entries[mkv->attachments->num_entries++].fileuid = fileuid;
1626 }
1627 end_ebml_master_crc32(pb, &dyn_cp, mkv, attachments);
1628
1629 return 0;
1630}

1700 put_ebml_uint (pb, EBML_ID_EBMLREADVERSION , 1);

1709 mkv->segment_offset = avio_tell(pb);
1710
1711 // we write 2 seek heads - one at the end of the file to point to each
1712 // cluster, and one at the beginning to point to all other level one
1713 // elements (including the seek head at the end of the file), which
1714 // isn't more than 10 elements if we only write one of each other
1715 // currently defined level 1 element
1716 mkv->main_seekhead = mkv_start_seekhead(pb, mkv->segment_offset, 10);
1717 if (!mkv->main_seekhead) {
1718 ret = AVERROR(ENOMEM);
1719 goto fail;
1720 }
1721
1722 ret = mkv_add_seekhead_entry(mkv->main_seekhead, MATROSKA_ID_INFO, avio_tell(pb));
1723 if (ret < 0) goto fail;
1724
1725 ret = start_ebml_master_crc32(pb, &mkv->info_bc, &mkv->info, MATROSKA_ID_INFO, 0);
1726 if (ret < 0)
1727 return ret;
1728 pb = mkv->info_bc;
1729
1730 put_ebml_uint(pb, MATROSKA_ID_TIMECODESCALE, 1000000);
1731 if ((tag = av_dict_get(s->metadata, "title", NULL, 0)))
1732 put_ebml_string(pb, MATROSKA_ID_TITLE, tag->value);
1733 if (!(s->flags & AVFMT_FLAG_BITEXACT)) {
1734 put_ebml_string(pb, MATROSKA_ID_MUXINGAPP, LIBAVFORMAT_IDENT);
1735 if ((tag = av_dict_get(s->metadata, "encoding_tool", NULL, 0)))
1736 put_ebml_string(pb, MATROSKA_ID_WRITINGAPP, tag->value);
1737 else
1738 put_ebml_string(pb, MATROSKA_ID_WRITINGAPP, LIBAVFORMAT_IDENT);
1739
1740 if (mkv->mode != MODE_WEBM) {
1741 uint32_t segment_uid[4];
1742 AVLFG lfg;
1743
1744 av_lfg_init(&lfg, av_get_random_seed());
1745
1746 for (i = 0; i < 4; i++)
1747 segment_uid[i] = av_lfg_get(&lfg);
1748
1749 put_ebml_binary(pb, MATROSKA_ID_SEGMENTUID, segment_uid, 16);
1750 }
1751 } else {
1752 const char *ident = "Lavf";
1753 put_ebml_string(pb, MATROSKA_ID_MUXINGAPP , ident);
1754 put_ebml_string(pb, MATROSKA_ID_WRITINGAPP, ident);
1755 }
1756
1757 if (ff_parse_creation_time_metadata(s, &creation_time, 0) > 0) {
1758 // Adjust time so it's relative to 2001-01-01 and convert to nanoseconds.
1759 int64_t date_utc = (creation_time - 978307200000000LL) * 1000;
1760 uint8_t date_utc_buf[8];
1761 AV_WB64(date_utc_buf, date_utc);
1762 put_ebml_binary(pb, MATROSKA_ID_DATEUTC, date_utc_buf, 8);
1763 }
1764
1765 // reserve space for the duration
1766 mkv->duration = 0;
1767 mkv->duration_offset = avio_tell(pb);
1768 if (!mkv->is_live) {
1769 int64_t metadata_duration = get_metadata_duration(s);
1770
1771 if (s->duration > 0) {
1772 int64_t scaledDuration = av_rescale(s->duration, 1000, AV_TIME_BASE);
1773 put_ebml_float(pb, MATROSKA_ID_DURATION, scaledDuration);
1774 av_log(s, AV_LOG_DEBUG, "Write early duration from recording time = %" PRIu64 "\n", scaledDuration);
1775 } else if (metadata_duration > 0) {
1776 int64_t scaledDuration = av_rescale(metadata_duration, 1000, AV_TIME_BASE);
1777 put_ebml_float(pb, MATROSKA_ID_DURATION, scaledDuration);
1778 av_log(s, AV_LOG_DEBUG, "Write early duration from metadata = %" PRIu64 "\n", scaledDuration);
1779 } else {
1780 put_ebml_void(pb, 11); // assumes double-precision float to be written
1781 }
1782 }
1783 if (s->pb->seekable)
1784 put_ebml_void(s->pb, avio_tell(pb) + ((mkv->write_crc && mkv->mode != MODE_WEBM) ? 2 /* ebml id + data size */ + 4 /* CRC32 */ : 0));
1785 else
1786 end_ebml_master_crc32(s->pb, &mkv->info_bc, mkv, mkv->info);
1787 pb = s->pb;
1788
1789 // initialize stream_duration fields
1790 mkv->stream_durations = av_mallocz(s->nb_streams * sizeof(int64_t));
1791 mkv->stream_duration_offsets = av_mallocz(s->nb_streams * sizeof(int64_t));
1792
1793 ret = mkv_write_tracks(s);
1794 if (ret < 0)
1795 goto fail;
1796
1797 for (i = 0; i < s->nb_chapters; i++)
1798 mkv->chapter_id_offset = FFMAX(mkv->chapter_id_offset, 1LL - s->chapters[i]->id);
1799
1800 if (mkv->mode != MODE_WEBM) {
1801 ret = mkv_write_chapters(s);
1802 if (ret < 0)
1803 goto fail;
1804
1805 ret = mkv_write_attachments(s);
1806 if (ret < 0)
1807 goto fail;
1808
1809 ret = mkv_write_tags(s);
1810 if (ret < 0)
1811 goto fail;
1812 }
1813
1814 if (!s->pb->seekable && !mkv->is_live)
1815 mkv_write_seekhead(pb, mkv);
1816
1817 mkv->cues = mkv_start_cues(mkv->segment_offset);
1818 if (!mkv->cues) {
1819 ret = AVERROR(ENOMEM);
1820 goto fail;
1821 }
1822 if (pb->seekable && mkv->reserve_cues_space) {
1823 mkv->cues_pos = avio_tell(pb);
1824 put_ebml_void(pb, mkv->reserve_cues_space);
1825 }
1826
1827 av_init_packet(&mkv->cur_audio_pkt);
1828 mkv->cur_audio_pkt.size = 0;
1829 mkv->cluster_pos = -1;
1830
1831 avio_flush(pb);
1832
1833 // start a new cluster every 5 MB or 5 sec, or 32k / 1 sec for streaming or
1834 // after 4k and on a keyframe
1835 if (pb->seekable) {
1836 if (mkv->cluster_time_limit < 0)
1837 mkv->cluster_time_limit = 5000;
1838 if (mkv->cluster_size_limit < 0)
1839 mkv->cluster_size_limit = 5 * 1024 * 1024;
1840 } else {
1841 if (mkv->cluster_time_limit < 0)
1842 mkv->cluster_time_limit = 1000;
1843 if (mkv->cluster_size_limit < 0)
1844 mkv->cluster_size_limit = 32 * 1024;
1845 }
1846
1847 return 0;
1848fail:
1849 mkv_free(mkv);
1850 return ret;
1851}

1864static int mkv_strip_wavpack(const uint8_t *src, uint8_t **pdst, int *size)
1865{
1866 uint8_t *dst;
1867 int srclen = *size;
1868 int offset = 0;
1869 int ret;
1870
1871 dst = av_malloc(srclen);
1872 if (!dst)
1873 return AVERROR(ENOMEM);
1874
1875 while (srclen >= WV_HEADER_SIZE) {
1876 WvHeader header;
1877
1878 ret = ff_wv_parse_header(&header, src);
1879 if (ret < 0)
1880 goto fail;
1881 src += WV_HEADER_SIZE;
1882 srclen -= WV_HEADER_SIZE;
1883
1884 if (srclen < header.blocksize) {
1885 ret = AVERROR_INVALIDDATA;
1886 goto fail;
1887 }
1888
1889 if (header.initial) {
1890 AV_WL32(dst + offset, header.samples);
1891 offset += 4;
1892 }
1893 AV_WL32(dst + offset, header.flags);
1894 AV_WL32(dst + offset + 4, header.crc);
1895 offset += 8;
1896
1897 if (!(header.initial && header.final)) {
1898 AV_WL32(dst + offset, header.blocksize);
1899 offset += 4;
1900 }
1901
1902 memcpy(dst + offset, src, header.blocksize);
1903 src += header.blocksize;
1904 srclen -= header.blocksize;
1905 offset += header.blocksize;
1906 }
1907
1908 *pdst = dst;
1909 *size = offset;
1910
1911 return 0;
1912fail:
1913 av_freep(&dst);
1914 return ret;
1915}

1917static void mkv_write_block(AVFormatContext *s, AVIOContext *pb,
1918 unsigned int blockid, AVPacket *pkt, int keyframe)
1919{
1920 MatroskaMuxContext *mkv = s->priv_data;
1921 AVCodecParameters *par = s->streams[pkt->stream_index]->codecpar;
1922 uint8_t *data = NULL, *side_data = NULL;
1923 int offset = 0, size = pkt->size, side_data_size = 0;
1924 int64_t ts = mkv->tracks[pkt->stream_index].write_dts ? pkt->dts : pkt->pts;
1925 uint64_t additional_id = 0;
1926 int64_t discard_padding = 0;
1927 uint8_t track_number = (mkv->is_dash ? mkv->dash_track_number : (pkt->stream_index + 1));
1928 ebml_master block_group, block_additions, block_more;
1929
1930 av_log(s, AV_LOG_DEBUG, "Writing block at offset %" PRIu64 ", size %d, "
1931 "pts %" PRId64 ", dts %" PRId64 ", duration %" PRId64 ", keyframe %d\n",
1932 avio_tell(pb), pkt->size, pkt->pts, pkt->dts, pkt->duration,
1933 keyframe != 0);
1934 if (par->codec_id == AV_CODEC_ID_H264 && par->extradata_size > 0 &&
1935 (AV_RB24(par->extradata) == 1 || AV_RB32(par->extradata) == 1))
1936 ff_avc_parse_nal_units_buf(pkt->data, &data, &size);
1937 else if (par->codec_id == AV_CODEC_ID_HEVC && par->extradata_size > 6 &&
1938 (AV_RB24(par->extradata) == 1 || AV_RB32(par->extradata) == 1))
1939 /* extradata is Annex B, assume the bitstream is too and convert it */
1940 ff_hevc_annexb2mp4_buf(pkt->data, &data, &size, 0, NULL);
1941 else if (par->codec_id == AV_CODEC_ID_WAVPACK) {
1942 int ret = mkv_strip_wavpack(pkt->data, &data, &size);
1943 if (ret < 0) {
1944 av_log(s, AV_LOG_ERROR, "Error stripping a WavPack packet.\n");
1945 return;
1946 }
1947 } else
1948 data = pkt->data;
1949
1950 if (par->codec_id == AV_CODEC_ID_PRORES && size >= 8) {
1951 /* Matroska specification requires to remove the first QuickTime atom
1952 */
1953 size -= 8;
1954 offset = 8;
1955 }
1956
1957 side_data = av_packet_get_side_data(pkt,
1958 AV_PKT_DATA_SKIP_SAMPLES,
1959 &side_data_size);
1960
1961 if (side_data && side_data_size >= 10) {
1962 discard_padding = av_rescale_q(AV_RL32(side_data + 4),
1963 (AVRational){1, par->sample_rate},
1964 (AVRational){1, 1000000000});
1965 }
1966
1967 side_data = av_packet_get_side_data(pkt,
1968 AV_PKT_DATA_MATROSKA_BLOCKADDITIONAL,
1969 &side_data_size);
1970 if (side_data) {
1971 additional_id = AV_RB64(side_data);
1972 side_data += 8;
1973 side_data_size -= 8;
1974 }
1975
1976 if ((side_data_size && additional_id == 1) || discard_padding) {
1977 block_group = start_ebml_master(pb, MATROSKA_ID_BLOCKGROUP, 0);
1978 blockid = MATROSKA_ID_BLOCK;
1979 }
1980
1981 put_ebml_id(pb, blockid);
1982 put_ebml_num(pb, size + 4, 0);
1983 // this assumes stream_index is less than 126
1984 avio_w8(pb, 0x80 | track_number);
1985 avio_wb16(pb, ts - mkv->cluster_pts);
1986 avio_w8(pb, (blockid == MATROSKA_ID_SIMPLEBLOCK && keyframe) ? (1 << 7) : 0);
1987 avio_write(pb, data + offset, size);
1988 if (data != pkt->data)
1989 av_free(data);
1990
1991 if (blockid == MATROSKA_ID_BLOCK && !keyframe) {
1992 put_ebml_sint(pb, MATROSKA_ID_BLOCKREFERENCE,
1993 mkv->last_track_timestamp[track_number - 1]);
1994 }
1995 mkv->last_track_timestamp[track_number - 1] = ts - mkv->cluster_pts;
1996
1997 if (discard_padding) {
1998 put_ebml_sint(pb, MATROSKA_ID_DISCARDPADDING, discard_padding);
1999 }
2000
2001 if (side_data_size && additional_id == 1) {
2002 block_additions = start_ebml_master(pb, MATROSKA_ID_BLOCKADDITIONS, 0);
2003 block_more = start_ebml_master(pb, MATROSKA_ID_BLOCKMORE, 0);
2004 put_ebml_uint(pb, MATROSKA_ID_BLOCKADDID, 1);
2005 put_ebml_id(pb, MATROSKA_ID_BLOCKADDITIONAL);
2006 put_ebml_num(pb, side_data_size, 0);
2007 avio_write(pb, side_data, side_data_size);
2008 end_ebml_master(pb, block_more);
2009 end_ebml_master(pb, block_additions);
2010 }
2011 if ((side_data_size && additional_id == 1) || discard_padding) {
2012 end_ebml_master(pb, block_group);
2013 }
2014}

2072static int mkv_write_packet_internal(AVFormatContext *s, AVPacket *pkt, int add_cue)
2073{
2074 MatroskaMuxContext *mkv = s->priv_data;
2075 AVIOContext *pb = s->pb;
2076 AVCodecParameters *par = s->streams[pkt->stream_index]->codecpar;
2077 int keyframe = !!(pkt->flags & AV_PKT_FLAG_KEY);
2078 int duration = pkt->duration;
2079 int ret;
2080 int64_t ts = mkv->tracks[pkt->stream_index].write_dts ? pkt->dts : pkt->pts;
2081 int64_t relative_packet_pos;
2082 int dash_tracknum = mkv->is_dash ? mkv->dash_track_number : pkt->stream_index + 1;
2083
2084 if (ts == AV_NOPTS_VALUE) {
2085 av_log(s, AV_LOG_ERROR, "Can't write packet with unknown timestamp\n");
2086 return AVERROR(EINVAL);
2087 }
2088 ts += mkv->tracks[pkt->stream_index].ts_offset;
2089
2090 if (mkv->cluster_pos != -1) {
2091 int64_t cluster_time = ts - mkv->cluster_pts + mkv->tracks[pkt->stream_index].ts_offset;
2092 if ((int16_t)cluster_time != cluster_time) {
2093 av_log(s, AV_LOG_WARNING, "Starting new cluster due to timestamp\n");
2094 mkv_start_new_cluster(s, pkt);
2095 }
2096 }
2097
2098 if (mkv->cluster_pos == -1) {
2099 mkv->cluster_pos = avio_tell(s->pb);
2100 ret = start_ebml_master_crc32(s->pb, &mkv->dyn_bc, &mkv->cluster, MATROSKA_ID_CLUSTER, 0);
2101 if (ret < 0)
2102 return ret;
2103 put_ebml_uint(mkv->dyn_bc, MATROSKA_ID_CLUSTERTIMECODE, FFMAX(0, ts));
2104 mkv->cluster_pts = FFMAX(0, ts);
2105 }
2106 pb = mkv->dyn_bc;
2107
2108 relative_packet_pos = avio_tell(s->pb) - mkv->cluster.pos + avio_tell(pb);
2109
2110 if (par->codec_type != AVMEDIA_TYPE_SUBTITLE) {
2111 mkv_write_block(s, pb, MATROSKA_ID_SIMPLEBLOCK, pkt, keyframe);
2112 if (s->pb->seekable && (par->codec_type == AVMEDIA_TYPE_VIDEO && keyframe || add_cue)) {
2113 ret = mkv_add_cuepoint(mkv->cues, pkt->stream_index, dash_tracknum, ts, mkv->cluster_pos, relative_packet_pos, -1);
2114 if (ret < 0) return ret;
2115 }
2116 } else {
2117 if (par->codec_id == AV_CODEC_ID_WEBVTT) {
2118 duration = mkv_write_vtt_blocks(s, pb, pkt);
2119 } else {
2120 ebml_master blockgroup = start_ebml_master(pb, MATROSKA_ID_BLOCKGROUP,
2121 mkv_blockgroup_size(pkt->size));
2122
2123#if FF_API_CONVERGENCE_DURATION
2124FF_DISABLE_DEPRECATION_WARNINGS
2125 /* For backward compatibility, prefer convergence_duration. */
2126 if (pkt->convergence_duration > 0) {
2127 duration = pkt->convergence_duration;
2128 }
2129FF_ENABLE_DEPRECATION_WARNINGS
2130#endif
2131 /* All subtitle blocks are considered to be keyframes. */
2132 mkv_write_block(s, pb, MATROSKA_ID_BLOCK, pkt, 1);
2133 put_ebml_uint(pb, MATROSKA_ID_BLOCKDURATION, duration);
2134 end_ebml_master(pb, blockgroup);
2135 }
2136
2137 if (s->pb->seekable) {
2138 ret = mkv_add_cuepoint(mkv->cues, pkt->stream_index, dash_tracknum, ts,
2139 mkv->cluster_pos, relative_packet_pos, duration);
2140 if (ret < 0)
2141 return ret;
2142 }
2143 }
2144
2145 mkv->duration = FFMAX(mkv->duration, ts + duration);
2146
2147 if (mkv->stream_durations)
2148 mkv->stream_durations[pkt->stream_index] =
2149 FFMAX(mkv->stream_durations[pkt->stream_index], ts + duration);
2150
2151 return 0;
2152}

2154static int mkv_write_packet(AVFormatContext *s, AVPacket *pkt)
2155{
2156 MatroskaMuxContext *mkv = s->priv_data;
2157 int codec_type = s->streams[pkt->stream_index]->codecpar->codec_type;
2158 int keyframe = !!(pkt->flags & AV_PKT_FLAG_KEY);
2159 int cluster_size;
2160 int64_t cluster_time;
2161 int ret;
2162 int start_new_cluster;
2163
2164 if (mkv->tracks[pkt->stream_index].write_dts)
2165 cluster_time = pkt->dts - mkv->cluster_pts;
2166 else
2167 cluster_time = pkt->pts - mkv->cluster_pts;
2168 cluster_time += mkv->tracks[pkt->stream_index].ts_offset;
2169
2170 // start a new cluster every 5 MB or 5 sec, or 32k / 1 sec for streaming or
2171 // after 4k and on a keyframe
2172 cluster_size = avio_tell(mkv->dyn_bc);
2173
2174 if (mkv->is_dash && codec_type == AVMEDIA_TYPE_VIDEO) {
2175 // WebM DASH specification states that the first block of every cluster
2176 // has to be a key frame. So for DASH video, we only create a cluster
2177 // on seeing key frames.
2178 start_new_cluster = keyframe;
2179 } else if (mkv->is_dash && codec_type == AVMEDIA_TYPE_AUDIO &&
2180 (mkv->cluster_pos == -1 ||
2181 cluster_time > mkv->cluster_time_limit)) {
2182 // For DASH audio, we create a Cluster based on cluster_time_limit
2183 start_new_cluster = 1;
2184 } else if (!mkv->is_dash &&
2185 (cluster_size > mkv->cluster_size_limit ||
2186 cluster_time > mkv->cluster_time_limit ||
2187 (codec_type == AVMEDIA_TYPE_VIDEO && keyframe &&
2188 cluster_size > 4 * 1024))) {
2189 start_new_cluster = 1;
2190 } else {
2191 start_new_cluster = 0;
2192 }
2193
2194 if (mkv->cluster_pos != -1 && start_new_cluster) {
2195 mkv_start_new_cluster(s, pkt);
2196 }
2197
2198 if (!mkv->cluster_pos)
2199 avio_write_marker(s->pb,
2200 av_rescale_q(pkt->dts, s->streams[pkt->stream_index]->time_base, AV_TIME_BASE_Q),
2201 keyframe && (mkv->have_video ? codec_type == AVMEDIA_TYPE_VIDEO : 1) ? AVIO_DATA_MARKER_SYNC_POINT : AVIO_DATA_MARKER_BOUNDARY_POINT);
2202
2203 // check if we have an audio packet cached
2204 if (mkv->cur_audio_pkt.size > 0) {
2205 // for DASH audio, a CuePoint has to be added when there is a new cluster.
2206 ret = mkv_write_packet_internal(s, &mkv->cur_audio_pkt,
2207 mkv->is_dash ? start_new_cluster : 0);
2208 av_packet_unref(&mkv->cur_audio_pkt);
2209 if (ret < 0) {
2210 av_log(s, AV_LOG_ERROR,
2211 "Could not write cached audio packet ret:%d\n", ret);
2212 return ret;
2213 }
2214 }
2215
2216 // buffer an audio packet to ensure the packet containing the video
2217 // keyframe's timecode is contained in the same cluster for WebM
2218 if (codec_type == AVMEDIA_TYPE_AUDIO) {
2219 ret = av_packet_ref(&mkv->cur_audio_pkt, pkt);
2220 } else
2221 ret = mkv_write_packet_internal(s, pkt, 0);
2222 return ret;
2223}

2246static int mkv_write_trailer(AVFormatContext *s)
