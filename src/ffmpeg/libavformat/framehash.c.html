23int ff_framehash_write_header(AVFormatContext *s)
24{
25 int i;
26
27 if (s->nb_streams && !(s->flags & AVFMT_FLAG_BITEXACT))
28 avio_printf(s->pb, "#software: %s\n", LIBAVFORMAT_IDENT);
29 for (i = 0; i < s->nb_streams; i++) {
30 AVStream *st = s->streams[i];
31 AVCodecParameters *avctx = st->codecpar;
32 avio_printf(s->pb, "#tb %d: %d/%d\n", i, st->time_base.num, st->time_base.den);
33 avio_printf(s->pb, "#media_type %d: %s\n", i, av_get_media_type_string(avctx->codec_type));
34 avio_printf(s->pb, "#codec_id %d: %s\n", i, avcodec_get_name(avctx->codec_id));
35 switch (avctx->codec_type) {
36 case AVMEDIA_TYPE_AUDIO:
37 avio_printf(s->pb, "#sample_rate %d: %d\n", i,avctx->sample_rate);
38 avio_printf(s->pb, "#channel_layout %d: %"PRIx64"\n", i,avctx->channel_layout);
39 break;
40 case AVMEDIA_TYPE_VIDEO:
41 avio_printf(s->pb, "#dimensions %d: %dx%d\n", i, avctx->width, avctx->height);
42 avio_printf(s->pb, "#sar %d: %d/%d\n", i, st->sample_aspect_ratio.num, st->sample_aspect_ratio.den);
43 break;
44 }
45 avio_flush(s->pb);
46 }
47 return 0;
48}
