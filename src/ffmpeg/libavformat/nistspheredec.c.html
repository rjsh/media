28static int nist_probe(AVProbeData *p)
29{
30 if (AV_RL64(p->buf) == AV_RL64("NIST_1A\x0a"))
31 return AVPROBE_SCORE_MAX;
32 return 0;
33}

35static int nist_read_header(AVFormatContext *s)
36{
37 char buffer[256], coding[32] = "pcm", format[32] = "01";
38 int bps = 0, be = 0;
39 int32_t header_size = -1;
40 AVStream *st;
41
42 st = avformat_new_stream(s, NULL);
43 if (!st)
44 return AVERROR(ENOMEM);
45
46 st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
47
48 ff_get_line(s->pb, buffer, sizeof(buffer));
49 ff_get_line(s->pb, buffer, sizeof(buffer));
50 sscanf(buffer, "%"SCNd32, &header_size);
51 if (header_size <= 0)
52 return AVERROR_INVALIDDATA;
53
54 while (!avio_feof(s->pb)) {
55 ff_get_line(s->pb, buffer, sizeof(buffer));
56
57 if (avio_tell(s->pb) >= header_size)
58 return AVERROR_INVALIDDATA;
59
60 if (!memcmp(buffer, "end_head", 8)) {
61 if (!st->codecpar->bits_per_coded_sample)
62 st->codecpar->bits_per_coded_sample = bps << 3;
63
64 if (!av_strcasecmp(coding, "pcm")) {
65 if (st->codecpar->codec_id == AV_CODEC_ID_NONE)
66 st->codecpar->codec_id = ff_get_pcm_codec_id(st->codecpar->bits_per_coded_sample,
67 0, be, 0xFFFF);
68 } else if (!av_strcasecmp(coding, "alaw")) {
69 st->codecpar->codec_id = AV_CODEC_ID_PCM_ALAW;
70 } else if (!av_strcasecmp(coding, "ulaw") ||
71 !av_strcasecmp(coding, "mu-law")) {
72 st->codecpar->codec_id = AV_CODEC_ID_PCM_MULAW;
73 } else if (!av_strncasecmp(coding, "pcm,embedded-shorten", 20)) {
74 st->codecpar->codec_id = AV_CODEC_ID_SHORTEN;
75 if (ff_alloc_extradata(st->codecpar, 1))
76 st->codecpar->extradata[0] = 1;
77 } else {
78 avpriv_request_sample(s, "coding %s", coding);
79 }
80
81 avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
82
83 st->codecpar->block_align = st->codecpar->bits_per_coded_sample * st->codecpar->channels / 8;
84
85 if (avio_tell(s->pb) > header_size)
86 return AVERROR_INVALIDDATA;
87
88 avio_skip(s->pb, header_size - avio_tell(s->pb));
89
90 return 0;
91 } else if (!memcmp(buffer, "channel_count", 13)) {
92 sscanf(buffer, "%*s %*s %"SCNd32, &st->codecpar->channels);
93 } else if (!memcmp(buffer, "sample_byte_format", 18)) {
94 sscanf(buffer, "%*s %*s %31s", format);
95
96 if (!av_strcasecmp(format, "01")) {
97 be = 0;
98 } else if (!av_strcasecmp(format, "10")) {
99 be = 1;
100 } else if (!av_strcasecmp(format, "mu-law")) {
101 st->codecpar->codec_id = AV_CODEC_ID_PCM_MULAW;
102 } else if (av_strcasecmp(format, "1")) {
103 avpriv_request_sample(s, "sample byte format %s", format);
104 return AVERROR_PATCHWELCOME;
105 }
106 } else if (!memcmp(buffer, "sample_coding", 13)) {
107 sscanf(buffer, "%*s %*s %31s", coding);
108 } else if (!memcmp(buffer, "sample_count", 12)) {
109 sscanf(buffer, "%*s %*s %"SCNd64, &st->duration);
110 } else if (!memcmp(buffer, "sample_n_bytes", 14)) {
111 sscanf(buffer, "%*s %*s %"SCNd32, &bps);
112 } else if (!memcmp(buffer, "sample_rate", 11)) {
113 sscanf(buffer, "%*s %*s %"SCNd32, &st->codecpar->sample_rate);
114 } else if (!memcmp(buffer, "sample_sig_bits", 15)) {
115 sscanf(buffer, "%*s %*s %"SCNd32, &st->codecpar->bits_per_coded_sample);
116 } else {
117 char key[32], value[32];
118 if (sscanf(buffer, "%31s %*s %31s", key, value) == 2) {
119 av_dict_set(&s->metadata, key, value, AV_DICT_APPEND);
120 } else {
121 av_log(s, AV_LOG_ERROR, "Failed to parse '%s' as metadata\n", buffer);
122 }
123 }
124 }
125
126 return AVERROR_EOF;
127}
