26#include "avformat.h"
27#include "internal.h"
28#include "subtitles.h"
29
30typedef struct {
31 FFDemuxSubtitlesQueue q;
32} SubViewer1Context;
33
34static int subviewer1_probe(AVProbeData *p)
35{
36 const unsigned char *ptr = p->buf;
37
38 if (strstr(ptr, "******** START SCRIPT ********"))
39 return AVPROBE_SCORE_EXTENSION;
40 return 0;
41}
42
43static int subviewer1_read_header(AVFormatContext *s)
44{
45 int delay = 0;
46 AVPacket *sub = NULL;
47 SubViewer1Context *subviewer1 = s->priv_data;
48 AVStream *st = avformat_new_stream(s, NULL);
49
50 if (!st)
51 return AVERROR(ENOMEM);
52 avpriv_set_pts_info(st, 64, 1, 1);
53 st->codecpar->codec_type = AVMEDIA_TYPE_SUBTITLE;
54 st->codecpar->codec_id = AV_CODEC_ID_SUBVIEWER1;
55
56 while (!avio_feof(s->pb)) {
57 char line[4096];
58 int len = ff_get_line(s->pb, line, sizeof(line));
59 int hh, mm, ss;
60
61 if (!len)
62 break;
63
64 if (!strncmp(line, "[DELAY]", 7)) {
65 ff_get_line(s->pb, line, sizeof(line));
66 sscanf(line, "%d", &delay);
67 }
68
69 if (sscanf(line, "[%d:%d:%d]", &hh, &mm, &ss) == 3) {
70 const int64_t pos = avio_tell(s->pb);
71 int64_t pts_start = hh*3600LL + mm*60LL + ss + delay;
72
73 len = ff_get_line(s->pb, line, sizeof(line));
74 line[strcspn(line, "\r\n")] = 0;
75 if (!*line) {
76 if (sub)
77 sub->duration = pts_start - sub->pts;
78 } else {
79 sub = ff_subtitles_queue_insert(&subviewer1->q, line, len, 0);
80 if (!sub)
81 return AVERROR(ENOMEM);
82 sub->pos = pos;
83 sub->pts = pts_start;
84 sub->duration = -1;
85 }
86 }
87 }
88
89 ff_subtitles_queue_finalize(s, &subviewer1->q);
90 return 0;
91}
92
93static int subviewer1_read_packet(AVFormatContext *s, AVPacket *pkt)
94{
95 SubViewer1Context *subviewer1 = s->priv_data;
96 return ff_subtitles_queue_read_packet(&subviewer1->q, pkt);
97}
98
99static int subviewer1_read_seek(AVFormatContext *s, int stream_index,
100 int64_t min_ts, int64_t ts, int64_t max_ts, int flags)
101{
102 SubViewer1Context *subviewer1 = s->priv_data;
103 return ff_subtitles_queue_seek(&subviewer1->q, s, stream_index,
104 min_ts, ts, max_ts, flags);
105}
106
107static int subviewer1_read_close(AVFormatContext *s)
108{
109 SubViewer1Context *subviewer1 = s->priv_data;
110 ff_subtitles_queue_clean(&subviewer1->q);
111 return 0;
112}

114AVInputFormat ff_subviewer1_demuxer = {
115 .name = "subviewer1",
116 .long_name = NULL_IF_CONFIG_SMALL("SubViewer v1 subtitle format"),
117 .priv_data_size = sizeof(SubViewer1Context),
118 .read_probe = subviewer1_probe,
119 .read_header = subviewer1_read_header,
120 .read_packet = subviewer1_read_packet,
121 .read_seek2 = subviewer1_read_seek,
122 .read_close = subviewer1_read_close,
123 .extensions = "sub",
124};
