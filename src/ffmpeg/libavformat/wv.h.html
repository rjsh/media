21#ifndef AVFORMAT_WV_H
22#define AVFORMAT_WV_H
23
24#include <stdint.h>
25
26#define WV_HEADER_SIZE 32
27
28#define WV_FLAG_INITIAL_BLOCK (1 << 11)
29#define WV_FLAG_FINAL_BLOCK (1 << 12)
30
31// specs say that maximum block size is 1Mb
32#define WV_BLOCK_LIMIT 1048576
33
34typedef struct WvHeader {
35 uint32_t blocksize; //< size of the block data (excluding the header)
36 uint16_t version; //< bitstream version
37 uint32_t total_samples; //< total number of samples in the stream
38 uint32_t block_idx; //< index of the first sample in this block
39 uint32_t samples; //< number of samples in this block
40 uint32_t flags;
41 uint32_t crc;
42
43 int initial, final;
44} WvHeader;
45
46/**
47 * Parse a WavPack block header.
48 *
49 * @param wv this struct will be filled with parse header information
50 * @param data header data, must be WV_HEADER_SIZE bytes long
51 *
52 * @return 0 on success, a negative AVERROR code on failure
53 */
54int ff_wv_parse_header(WvHeader *wv, const uint8_t *data);
55
56#endif /* AVFORMAT_WV_H */
