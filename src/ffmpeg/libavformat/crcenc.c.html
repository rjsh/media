22#include <inttypes.h>
23
24#include "libavutil/adler32.h"
25#include "avformat.h"
26
27typedef struct CRCState {
28 uint32_t crcval;
29} CRCState;
30
31static int crc_write_header(struct AVFormatContext *s)
32{
33 CRCState *crc = s->priv_data;
34
35 /* init CRC */
36 crc->crcval = 1;
37
38 return 0;
39}
40
41static int crc_write_packet(struct AVFormatContext *s, AVPacket *pkt)
42{
43 CRCState *crc = s->priv_data;
44 crc->crcval = av_adler32_update(crc->crcval, pkt->data, pkt->size);
45 return 0;
46}
47
48static int crc_write_trailer(struct AVFormatContext *s)
49{
50 CRCState *crc = s->priv_data;
51 char buf[64];
52
53 snprintf(buf, sizeof(buf), "CRC=0x%08"PRIx32"\n", crc->crcval);
54 avio_write(s->pb, buf, strlen(buf));
55
56 return 0;
57}
58
59AVOutputFormat ff_crc_muxer = {
60 .name = "crc",
61 .long_name = NULL_IF_CONFIG_SMALL("CRC testing"),
62 .priv_data_size = sizeof(CRCState),
63 .audio_codec = AV_CODEC_ID_PCM_S16LE,
64 .video_codec = AV_CODEC_ID_RAWVIDEO,
65 .write_header = crc_write_header,
66 .write_packet = crc_write_packet,
67 .write_trailer = crc_write_trailer,
68 .flags = AVFMT_NOTIMESTAMPS,
69};
