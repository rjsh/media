384int ff_mov_iso639_to_lang(const char lang[4], int mp4)
385{
386 int i, code = 0;
387
388 /* old way, only for QT? */
389 for (i = 0; lang[0] && !mp4 && i < FF_ARRAY_ELEMS(mov_mdhd_language_map); i++) {
390 if (!strcmp(lang, mov_mdhd_language_map[i]))
391 return i;
392 }
393 /* XXX:can we do that in mov too? */
394 if (!mp4)
395 return -1;
396 /* handle undefined as such */
397 if (lang[0] == '\0')
398 lang = "und";
399 /* 5 bits ASCII */
400 for (i = 0; i < 3; i++) {
401 uint8_t c = lang[i];
402 c -= 0x60;
403 if (c > 0x1f)
404 return -1;
405 code <<= 5;
406 code |= c;
407 }
408 return code;
409}

411int ff_mov_lang_to_iso639(unsigned code, char to[4])
412{
413 int i;
414 memset(to, 0, 4);
415 /* is it the mangled iso code? */
416 /* see http://www.geocities.com/xhelmboyx/quicktime/formats/mp4-layout.txt */
417 if (code >= 0x400 && code != 0x7fff) {
418 for (i = 2; i >= 0; i--) {
419 to[i] = 0x60 + (code & 0x1f);
420 code >>= 5;
421 }
422 return 1;
423 }
424 /* old fashion apple lang code */
425 if (code >= FF_ARRAY_ELEMS(mov_mdhd_language_map))
426 return 0;
427 if (!mov_mdhd_language_map[code][0])
428 return 0;
429 memcpy(to, mov_mdhd_language_map[code], 4);
430 return 1;
431}

433int ff_mp4_read_descr_len(AVIOContext *pb)
434{
435 int len = 0;
436 int count = 4;
437 while (count--) {
438 int c = avio_r8(pb);
439 len = (len << 7) | (c & 0x7f);
440 if (!(c & 0x80))
441 break;
442 }
443 return len;
444}

451 av_log(fc, AV_LOG_TRACE, "MPEG-4 description: tag=0x%02x len=%d\n", *tag, len);

455void ff_mp4_parse_es_descr(AVIOContext *pb, int *es_id)
456{
457 int flags;
458 if (es_id) *es_id = avio_rb16(pb);
459 else avio_rb16(pb);
460 flags = avio_r8(pb);
461 if (flags & 0x80) //streamDependenceFlag
462 avio_rb16(pb);
463 if (flags & 0x40) { //URL_Flag
464 int len = avio_r8(pb);
465 avio_skip(pb, len);
466 }
467 if (flags & 0x20) //OCRstreamFlag
468 avio_rb16(pb);
469}

480int ff_mp4_read_dec_config_descr(AVFormatContext *fc, AVStream *st, AVIOContext *pb)
481{
482 enum AVCodecID codec_id;
483 unsigned v;
484 int len, tag;
485 int ret;
486 int object_type_id = avio_r8(pb);
487 avio_r8(pb); /* stream type */
488 avio_rb24(pb); /* buffer size db */
489
490 v = avio_rb32(pb);
491 // TODO: fix this
492 //if (v < INT32_MAX)
493 // st->codecpar->rc_max_rate = v;
494
495 st->codecpar->bit_rate = avio_rb32(pb); /* avg bitrate */
496
497 codec_id= ff_codec_get_id(ff_mp4_obj_type, object_type_id);
498 if (codec_id)
499 st->codecpar->codec_id = codec_id;
500 av_log(fc, AV_LOG_TRACE, "esds object type id 0x%02x\n", object_type_id);
501 len = ff_mp4_read_descr(fc, pb, &tag);
502 if (tag == MP4DecSpecificDescrTag) {
503 av_log(fc, AV_LOG_TRACE, "Specific MPEG-4 header len=%d\n", len);
504 if (!len || (uint64_t)len > (1<<30))
505 return -1;
506 av_free(st->codecpar->extradata);
507 if ((ret = ff_get_extradata(fc, st->codecpar, pb, len)) < 0)
508 return ret;
509 if (st->codecpar->codec_id == AV_CODEC_ID_AAC) {
510 MPEG4AudioConfig cfg = {0};
511 avpriv_mpeg4audio_get_config(&cfg, st->codecpar->extradata,
512 st->codecpar->extradata_size * 8, 1);
513 st->codecpar->channels = cfg.channels;
514 if (cfg.object_type == 29 && cfg.sampling_index < 3) // old mp3on4
515 st->codecpar->sample_rate = avpriv_mpa_freq_tab[cfg.sampling_index];
516 else if (cfg.ext_sample_rate)
517 st->codecpar->sample_rate = cfg.ext_sample_rate;
518 else
519 st->codecpar->sample_rate = cfg.sample_rate;
520 av_log(fc, AV_LOG_TRACE, "mp4a config channels %d obj %d ext obj %d "
521 "sample rate %d ext sample rate %d\n", st->codecpar->channels,
522 cfg.object_type, cfg.ext_object_type,
523 cfg.sample_rate, cfg.ext_sample_rate);
524 if (!(st->codecpar->codec_id = ff_codec_get_id(mp4_audio_types,
525 cfg.object_type)))
526 st->codecpar->codec_id = AV_CODEC_ID_AAC;
527 }
528 }
529 return 0;
530}
