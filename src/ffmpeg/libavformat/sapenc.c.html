35struct SAPState {
36 uint8_t *ann;
37 int ann_size;
38 URLContext *ann_fd;
39 int64_t last_time;
40};

42static int sap_write_close(AVFormatContext *s)
43{
44 struct SAPState *sap = s->priv_data;
45 int i;
46
47 for (i = 0; i < s->nb_streams; i++) {
48 AVFormatContext *rtpctx = s->streams[i]->priv_data;
49 if (!rtpctx)
50 continue;
51 av_write_trailer(rtpctx);
52 avio_closep(&rtpctx->pb);
53 avformat_free_context(rtpctx);
54 s->streams[i]->priv_data = NULL;
55 }
56
57 if (sap->last_time && sap->ann && sap->ann_fd) {
58 sap->ann[0] |= 4; /* Session deletion*/
59 ffurl_write(sap->ann_fd, sap->ann, sap->ann_size);
60 }
61
62 av_freep(&sap->ann);
63 if (sap->ann_fd)
64 ffurl_close(sap->ann_fd);
65 ff_network_close();
66 return 0;
67}

69static int sap_write_header(AVFormatContext *s)
70{
71 struct SAPState *sap = s->priv_data;
72 char host[1024], path[1024], url[1024], announce_addr[50] = "";
73 char *option_list;
74 int port = 9875, base_port = 5004, i, pos = 0, same_port = 0, ttl = 255;
75 AVFormatContext **contexts = NULL;
76 int ret = 0;
77 struct sockaddr_storage localaddr;
78 socklen_t addrlen = sizeof(localaddr);
79 int udp_fd;
80 AVDictionaryEntry* title = av_dict_get(s->metadata, "title", NULL, 0);
81
82 if (!ff_network_init())
83 return AVERROR(EIO);
84
85 /* extract hostname and port */
86 av_url_split(NULL, 0, NULL, 0, host, sizeof(host), &base_port,
87 path, sizeof(path), s->filename);
88 if (base_port < 0)
89 base_port = 5004;
90
91 /* search for options */
92 option_list = strrchr(path, '?');
93 if (option_list) {
94 char buf[50];
95 if (av_find_info_tag(buf, sizeof(buf), "announce_port", option_list)) {
96 port = strtol(buf, NULL, 10);
97 }
98 if (av_find_info_tag(buf, sizeof(buf), "same_port", option_list)) {
99 same_port = strtol(buf, NULL, 10);
100 }
101 if (av_find_info_tag(buf, sizeof(buf), "ttl", option_list)) {
102 ttl = strtol(buf, NULL, 10);
103 }
104 if (av_find_info_tag(buf, sizeof(buf), "announce_addr", option_list)) {
105 av_strlcpy(announce_addr, buf, sizeof(announce_addr));
106 }
107 }
108
109 if (!announce_addr[0]) {
110 struct addrinfo hints = { 0 }, *ai = NULL;
111 hints.ai_family = AF_UNSPEC;
112 if (getaddrinfo(host, NULL, &hints, &ai)) {
113 av_log(s, AV_LOG_ERROR, "Unable to resolve %s\n", host);
114 ret = AVERROR(EIO);
115 goto fail;
116 }
117 if (ai->ai_family == AF_INET) {
118 /* Also known as sap.mcast.net */
119 av_strlcpy(announce_addr, "224.2.127.254", sizeof(announce_addr));
120#if HAVE_STRUCT_SOCKADDR_IN6
121 } else if (ai->ai_family == AF_INET6) {
122 /* With IPv6, you can use the same destination in many different
123 * multicast subnets, to choose how far you want it routed.
124 * This one is intended to be routed globally. */
125 av_strlcpy(announce_addr, "ff0e::2:7ffe", sizeof(announce_addr));
126#endif
127 } else {
128 freeaddrinfo(ai);
129 av_log(s, AV_LOG_ERROR, "Host %s resolved to unsupported "
130 "address family\n", host);
131 ret = AVERROR(EIO);
132 goto fail;
133 }
134 freeaddrinfo(ai);
135 }
136
137 contexts = av_mallocz_array(s->nb_streams, sizeof(AVFormatContext*));
138 if (!contexts) {
139 ret = AVERROR(ENOMEM);
140 goto fail;
141 }
142
143 if (s->start_time_realtime == 0 || s->start_time_realtime == AV_NOPTS_VALUE)
144 s->start_time_realtime = av_gettime();
145 for (i = 0; i < s->nb_streams; i++) {
146 URLContext *fd;
147
148 ff_url_join(url, sizeof(url), "rtp", NULL, host, base_port,
149 "?ttl=%d", ttl);
150 if (!same_port)
151 base_port += 2;
152 ret = ffurl_open_whitelist(&fd, url, AVIO_FLAG_WRITE,
153 &s->interrupt_callback, NULL,
154 s->protocol_whitelist, s->protocol_blacklist, NULL);
155 if (ret) {
156 ret = AVERROR(EIO);
157 goto fail;
158 }
159 ret = ff_rtp_chain_mux_open(&contexts[i], s, s->streams[i], fd, 0, i);
160 if (ret < 0)
161 goto fail;
162 s->streams[i]->priv_data = contexts[i];
163 s->streams[i]->time_base = contexts[i]->streams[0]->time_base;
164 av_strlcpy(contexts[i]->filename, url, sizeof(contexts[i]->filename));
165 }
166
167 if (s->nb_streams > 0 && title)
168 av_dict_set(&contexts[0]->metadata, "title", title->value, 0);
169
170 ff_url_join(url, sizeof(url), "udp", NULL, announce_addr, port,
171 "?ttl=%d&connect=1", ttl);
172 ret = ffurl_open_whitelist(&sap->ann_fd, url, AVIO_FLAG_WRITE,
173 &s->interrupt_callback, NULL,
174 s->protocol_whitelist, s->protocol_blacklist, NULL);
175 if (ret) {
176 ret = AVERROR(EIO);
177 goto fail;
178 }
179
180 udp_fd = ffurl_get_file_handle(sap->ann_fd);
181 if (getsockname(udp_fd, (struct sockaddr*) &localaddr, &addrlen)) {
182 ret = AVERROR(EIO);
183 goto fail;
184 }
185 if (localaddr.ss_family != AF_INET
186#if HAVE_STRUCT_SOCKADDR_IN6
187 && localaddr.ss_family != AF_INET6
188#endif
189 ) {
190 av_log(s, AV_LOG_ERROR, "Unsupported protocol family\n");
191 ret = AVERROR(EIO);
192 goto fail;
193 }
194 sap->ann_size = 8192;
195 sap->ann = av_mallocz(sap->ann_size);
196 if (!sap->ann) {
197 ret = AVERROR(EIO);
198 goto fail;
199 }
200 sap->ann[pos] = (1 << 5);
201#if HAVE_STRUCT_SOCKADDR_IN6
202 if (localaddr.ss_family == AF_INET6)
203 sap->ann[pos] |= 0x10;
204#endif
205 pos++;
206 sap->ann[pos++] = 0; /* Authentication length */
207 AV_WB16(&sap->ann[pos], av_get_random_seed());
208 pos += 2;
209 if (localaddr.ss_family == AF_INET) {
210 memcpy(&sap->ann[pos], &((struct sockaddr_in*)&localaddr)->sin_addr,
211 sizeof(struct in_addr));
212 pos += sizeof(struct in_addr);
213#if HAVE_STRUCT_SOCKADDR_IN6
214 } else {
215 memcpy(&sap->ann[pos], &((struct sockaddr_in6*)&localaddr)->sin6_addr,
216 sizeof(struct in6_addr));
217 pos += sizeof(struct in6_addr);
218#endif
219 }
220
221 av_strlcpy(&sap->ann[pos], "application/sdp", sap->ann_size - pos);
222 pos += strlen(&sap->ann[pos]) + 1;
223
224 if (av_sdp_create(contexts, s->nb_streams, &sap->ann[pos],
225 sap->ann_size - pos)) {
226 ret = AVERROR_INVALIDDATA;
227 goto fail;
228 }
229 av_freep(&contexts);
230 av_log(s, AV_LOG_VERBOSE, "SDP:\n%s\n", &sap->ann[pos]);
231 pos += strlen(&sap->ann[pos]);
232 sap->ann_size = pos;
233
234 if (sap->ann_size > sap->ann_fd->max_packet_size) {
235 av_log(s, AV_LOG_ERROR, "Announcement too large to send in one "
236 "packet\n");
237 goto fail;
238 }
239
240 return 0;
241
242fail:
243 av_free(contexts);
244 sap_write_close(s);
245 return ret;
246}

248static int sap_write_packet(AVFormatContext *s, AVPacket *pkt)
249{
250 AVFormatContext *rtpctx;
251 struct SAPState *sap = s->priv_data;
252 int64_t now = av_gettime_relative();
253
254 if (!sap->last_time || now - sap->last_time > 5000000) {
255 int ret = ffurl_write(sap->ann_fd, sap->ann, sap->ann_size);
256 /* Don't abort even if we get "Destination unreachable" */
257 if (ret < 0 && ret != AVERROR(ECONNREFUSED))
258 return ret;
259 sap->last_time = now;
260 }
261 rtpctx = s->streams[pkt->stream_index]->priv_data;
262 return ff_write_chained(rtpctx, 0, pkt, s, 0);
263}

265AVOutputFormat ff_sap_muxer = {
266 .name = "sap",
267 .long_name = NULL_IF_CONFIG_SMALL("SAP output"),
268 .priv_data_size = sizeof(struct SAPState),
269 .audio_codec = AV_CODEC_ID_AAC,
270 .video_codec = AV_CODEC_ID_MPEG4,
271 .write_header = sap_write_header,
272 .write_packet = sap_write_packet,
273 .write_trailer = sap_write_close,
274 .flags = AVFMT_NOFILE | AVFMT_GLOBALHEADER,
275};
