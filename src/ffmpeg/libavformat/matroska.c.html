151int ff_mkv_stereo3d_conv(AVStream *st, MatroskaVideoStereoModeType stereo_mode)
152{
153 AVPacketSideData *sd, *tmp;
154 AVStereo3D *stereo;
155
156 stereo = av_stereo3d_alloc();
157 if (!stereo)
158 return AVERROR(ENOMEM);
159
160 tmp = av_realloc_array(st->side_data, st->nb_side_data + 1, sizeof(*tmp));
161 if (!tmp) {
162 av_freep(&stereo);
163 return AVERROR(ENOMEM);
164 }
165 st->side_data = tmp;
166 st->nb_side_data++;
167
168 sd = &st->side_data[st->nb_side_data - 1];
169 sd->type = AV_PKT_DATA_STEREO3D;
170 sd->data = (uint8_t *)stereo;
171 sd->size = sizeof(*stereo);
172
173 // note: the missing breaks are intentional
174 switch (stereo_mode) {
175 case MATROSKA_VIDEO_STEREOMODE_TYPE_MONO:
176 stereo->type = AV_STEREO3D_2D;
177 break;
178 case MATROSKA_VIDEO_STEREOMODE_TYPE_RIGHT_LEFT:
179 stereo->flags |= AV_STEREO3D_FLAG_INVERT;
180 case MATROSKA_VIDEO_STEREOMODE_TYPE_LEFT_RIGHT:
181 stereo->type = AV_STEREO3D_SIDEBYSIDE;
182 break;
183 case MATROSKA_VIDEO_STEREOMODE_TYPE_BOTTOM_TOP:
184 stereo->flags |= AV_STEREO3D_FLAG_INVERT;
185 case MATROSKA_VIDEO_STEREOMODE_TYPE_TOP_BOTTOM:
186 stereo->type = AV_STEREO3D_TOPBOTTOM;
187 break;
188 case MATROSKA_VIDEO_STEREOMODE_TYPE_CHECKERBOARD_RL:
189 stereo->flags |= AV_STEREO3D_FLAG_INVERT;
190 case MATROSKA_VIDEO_STEREOMODE_TYPE_CHECKERBOARD_LR:
191 stereo->type = AV_STEREO3D_CHECKERBOARD;
192 break;
193 case MATROSKA_VIDEO_STEREOMODE_TYPE_ROW_INTERLEAVED_RL:
194 stereo->flags |= AV_STEREO3D_FLAG_INVERT;
195 case MATROSKA_VIDEO_STEREOMODE_TYPE_ROW_INTERLEAVED_LR:
196 stereo->type = AV_STEREO3D_LINES;
197 break;
198 case MATROSKA_VIDEO_STEREOMODE_TYPE_COL_INTERLEAVED_RL:
199 stereo->flags |= AV_STEREO3D_FLAG_INVERT;
200 case MATROSKA_VIDEO_STEREOMODE_TYPE_COL_INTERLEAVED_LR:
201 stereo->type = AV_STEREO3D_COLUMNS;
202 break;
203 case MATROSKA_VIDEO_STEREOMODE_TYPE_BOTH_EYES_BLOCK_RL:
204 stereo->flags |= AV_STEREO3D_FLAG_INVERT;
205 case MATROSKA_VIDEO_STEREOMODE_TYPE_BOTH_EYES_BLOCK_LR:
206 stereo->type = AV_STEREO3D_FRAMESEQUENCE;
207 break;
208 }
209
210 return 0;
211}
