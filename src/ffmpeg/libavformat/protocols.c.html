75const AVClass *ff_urlcontext_child_class_next(const AVClass *prev)
76{
77 int i;
78
79 /* find the protocol that corresponds to prev */
80 for (i = 0; prev && url_protocols[i]; i++) {
81 if (url_protocols[i]->priv_data_class == prev) {
82 i++;
83 break;
84 }
85 }
86
87 /* find next protocol with priv options */
88 for (; url_protocols[i]; i++)
89 if (url_protocols[i]->priv_data_class)
90 return url_protocols[i]->priv_data_class;
91 return NULL;
92}

95const char *avio_enum_protocols(void **opaque, int output)
96{
97 const URLProtocol **p = *opaque;
98
99 p = p ? p + 1 : url_protocols;
100 *opaque = p;
101 if (!*p) {
102 *opaque = NULL;
103 return NULL;
104 }
105 if ((output && (*p)->url_write) || (!output && (*p)->url_read))
106 return (*p)->name;
107 return avio_enum_protocols(opaque, output);
108}

110const URLProtocol **ffurl_get_protocols(const char *whitelist,
111 const char *blacklist)
112{
113 const URLProtocol **ret;
114 int i, ret_idx = 0;
115
116 ret = av_mallocz_array(FF_ARRAY_ELEMS(url_protocols), sizeof(*ret));
117 if (!ret)
118 return NULL;
119
120 for (i = 0; url_protocols[i]; i++) {
121 const URLProtocol *up = url_protocols[i];
122
123 if (whitelist && *whitelist && !av_match_name(up->name, whitelist))
124 continue;
125 if (blacklist && *blacklist && av_match_name(up->name, blacklist))
126 continue;
127
128 ret[ret_idx++] = up;
129 }
130
131 return ret;
132}
