20#include "avformat.h"
21#include "internal.h"
22#include "libavutil/intreadwrite.h"
23
24typedef struct IVFEncContext {
25 unsigned frame_cnt;
26 uint64_t last_pts, sum_delta_pts;
27} IVFEncContext;
28
29static int ivf_write_header(AVFormatContext *s)
30{
31 AVCodecParameters *par;
32 AVIOContext *pb = s->pb;
33
34 if (s->nb_streams != 1) {
35 av_log(s, AV_LOG_ERROR, "Format supports only exactly one video stream\n");
36 return AVERROR(EINVAL);
37 }
38 par = s->streams[0]->codecpar;
39 if (par->codec_type != AVMEDIA_TYPE_VIDEO ||
40 !(par->codec_id == AV_CODEC_ID_VP8 || par->codec_id == AV_CODEC_ID_VP9)) {
41 av_log(s, AV_LOG_ERROR, "Currently only VP8 and VP9 are supported!\n");
42 return AVERROR(EINVAL);
43 }
44 avio_write(pb, "DKIF", 4);
45 avio_wl16(pb, 0); // version
46 avio_wl16(pb, 32); // header length
47 avio_wl32(pb, par->codec_tag ? par->codec_tag : par->codec_id == AV_CODEC_ID_VP9 ? AV_RL32("VP90") : AV_RL32("VP80"));
48 avio_wl16(pb, par->width);
49 avio_wl16(pb, par->height);
50 avio_wl32(pb, s->streams[0]->time_base.den);
51 avio_wl32(pb, s->streams[0]->time_base.num);
52 avio_wl64(pb, 0xFFFFFFFFFFFFFFFFULL);
53
54 return 0;
55}
56
57static int ivf_write_packet(AVFormatContext *s, AVPacket *pkt)
58{
59 AVIOContext *pb = s->pb;
60 IVFEncContext *ctx = s->priv_data;
61
62 avio_wl32(pb, pkt->size);
63 avio_wl64(pb, pkt->pts);
64 avio_write(pb, pkt->data, pkt->size);
65 if (ctx->frame_cnt)
66 ctx->sum_delta_pts += pkt->pts - ctx->last_pts;
67 ctx->frame_cnt++;
68 ctx->last_pts = pkt->pts;
69
70 return 0;
71}
72
73static int ivf_write_trailer(AVFormatContext *s)
74{
75 AVIOContext *pb = s->pb;
76 IVFEncContext *ctx = s->priv_data;
77
78 if (pb->seekable && ctx->frame_cnt > 1) {
79 size_t end = avio_tell(pb);
80
81 avio_seek(pb, 24, SEEK_SET);
82 avio_wl64(pb, ctx->frame_cnt * ctx->sum_delta_pts / (ctx->frame_cnt - 1));
83 avio_seek(pb, end, SEEK_SET);
84 }
85
86 return 0;
87}

89static int ivf_check_bitstream(struct AVFormatContext *s, const AVPacket *pkt)
90{
91 int ret = 1;
92 AVStream *st = s->streams[pkt->stream_index];
93
94 if (st->codecpar->codec_id == AV_CODEC_ID_VP9)
95 ret = ff_stream_add_bitstream_filter(st, "vp9_superframe", NULL);
96
97 return ret;
98}

103 { AV_CODEC_ID_NONE, 0 }
