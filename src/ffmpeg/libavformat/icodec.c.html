34typedef struct {
35 int offset;
36 int size;
37 int nb_pal;
38} IcoImage;

40typedef struct {
41 int current_image;
42 int nb_images;
43 IcoImage * images;
44} IcoDemuxContext;

46static int probe(AVProbeData *p)
47{
48 unsigned i, frames, checked = 0;
49
50 if (p->buf_size < 22 || AV_RL16(p->buf) || AV_RL16(p->buf + 2) != 1)
51 return 0;
52 frames = AV_RL16(p->buf + 4);
53 if (!frames)
54 return 0;
55 for (i = 0; i < frames && i * 16 + 22 <= p->buf_size; i++) {
56 unsigned offset;
57 if (AV_RL16(p->buf + 10 + i * 16) & ~1)
58 return FFMIN(i, AVPROBE_SCORE_MAX / 4);
59 if (p->buf[13 + i * 16])
60 return FFMIN(i, AVPROBE_SCORE_MAX / 4);
61 if (AV_RL32(p->buf + 14 + i * 16) < 40)
62 return FFMIN(i, AVPROBE_SCORE_MAX / 4);
63 offset = AV_RL32(p->buf + 18 + i * 16);
64 if (offset < 22)
65 return FFMIN(i, AVPROBE_SCORE_MAX / 4);
66 if (offset > p->buf_size - 8)
67 continue;
68 if (p->buf[offset] != 40 && AV_RB64(p->buf + offset) != PNGSIG)
69 return FFMIN(i, AVPROBE_SCORE_MAX / 4);
70 checked++;
71 }
72
73 if (checked < frames)
74 return AVPROBE_SCORE_MAX / 4 + FFMIN(checked, 1);
75 return AVPROBE_SCORE_MAX / 2 + 1;
76}

78static int read_header(AVFormatContext *s)
79{
80 IcoDemuxContext *ico = s->priv_data;
81 AVIOContext *pb = s->pb;
82 int i, codec;
83
84 avio_skip(pb, 4);
85 ico->nb_images = avio_rl16(pb);
86
87 ico->images = av_malloc_array(ico->nb_images, sizeof(IcoImage));
88 if (!ico->images)
89 return AVERROR(ENOMEM);
90
91 for (i = 0; i < ico->nb_images; i++) {
92 AVStream *st;
93 int tmp;
94
95 if (avio_seek(pb, 6 + i * 16, SEEK_SET) < 0)
96 break;
97
98 st = avformat_new_stream(s, NULL);
99 if (!st)
100 return AVERROR(ENOMEM);
101
102 st->codecpar->codec_type = AVMEDIA_TYPE_VIDEO;
103 st->codecpar->width = avio_r8(pb);
104 st->codecpar->height = avio_r8(pb);
105 ico->images[i].nb_pal = avio_r8(pb);
106 if (ico->images[i].nb_pal == 255)
107 ico->images[i].nb_pal = 0;
108
109 avio_skip(pb, 5);
110
111 ico->images[i].size = avio_rl32(pb);
112 ico->images[i].offset = avio_rl32(pb);
113
114 if (avio_seek(pb, ico->images[i].offset, SEEK_SET) < 0)
115 break;
116
117 codec = avio_rl32(pb);
118 switch (codec) {
119 case MKTAG(0x89, 'P', 'N', 'G'):
120 st->codecpar->codec_id = AV_CODEC_ID_PNG;
121 st->codecpar->width = 0;
122 st->codecpar->height = 0;
123 break;
124 case 40:
125 if (ico->images[i].size < 40)
126 return AVERROR_INVALIDDATA;
127 st->codecpar->codec_id = AV_CODEC_ID_BMP;
128 tmp = avio_rl32(pb);
129 if (tmp)
130 st->codecpar->width = tmp;
131 tmp = avio_rl32(pb);
132 if (tmp)
133 st->codecpar->height = tmp / 2;
134 break;
135 default:
136 avpriv_request_sample(s, "codec %d", codec);
137 return AVERROR_INVALIDDATA;
138 }
139 }
140
141 return 0;
142}

144static int read_packet(AVFormatContext *s, AVPacket *pkt)
145{
146 IcoDemuxContext *ico = s->priv_data;
147 IcoImage *image;
148 AVIOContext *pb = s->pb;
149 AVStream *st = s->streams[0];
150 int ret;
151
152 if (ico->current_image >= ico->nb_images)
153 return AVERROR_EOF;
154
155 image = &ico->images[ico->current_image];
156
157 if ((ret = avio_seek(pb, image->offset, SEEK_SET)) < 0)
158 return ret;
159
160 if (s->streams[ico->current_image]->codecpar->codec_id == AV_CODEC_ID_PNG) {
161 if ((ret = av_get_packet(pb, pkt, image->size)) < 0)
162 return ret;
163 } else {
164 uint8_t *buf;
165 if ((ret = av_new_packet(pkt, 14 + image->size)) < 0)
166 return ret;
167 buf = pkt->data;
168
169 /* add BMP header */
170 bytestream_put_byte(&buf, 'B');
171 bytestream_put_byte(&buf, 'M');
172 bytestream_put_le32(&buf, pkt->size);
173 bytestream_put_le16(&buf, 0);
174 bytestream_put_le16(&buf, 0);
175 bytestream_put_le32(&buf, 0);
176
177 if ((ret = avio_read(pb, buf, image->size)) < 0)
178 return ret;
179
180 st->codecpar->bits_per_coded_sample = AV_RL16(buf + 14);
181
182 if (AV_RL32(buf + 32))
183 image->nb_pal = AV_RL32(buf + 32);
184
185 if (st->codecpar->bits_per_coded_sample <= 8 && !image->nb_pal) {
186 image->nb_pal = 1 << st->codecpar->bits_per_coded_sample;
187 AV_WL32(buf + 32, image->nb_pal);
188 }
189
190 AV_WL32(buf - 4, 14 + 40 + image->nb_pal * 4);
191 AV_WL32(buf + 8, AV_RL32(buf + 8) / 2);
192 }
193
194 pkt->stream_index = ico->current_image++;
195 pkt->flags |= AV_PKT_FLAG_KEY;
196
197 return 0;
198}
