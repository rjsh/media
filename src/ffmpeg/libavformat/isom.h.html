194typedef struct MOVContext {

199 int found_moov; ///< 'moov' atom has been found
200 int found_mdat; ///< 'mdat' atom has been found
201 int found_hdlr_mdta; ///< 'hdlr' atom with type 'mdta' has been found
202 int trak_index; ///< Index of the current 'trak'
203 char **meta_keys;
204 unsigned meta_keys_count;
205 DVDemuxContext *dv_demux;
206 AVFormatContext *dv_fctx;
207 int isom; ///< 1 if file is ISO Media (mp4/3gp)
208 MOVFragment fragment; ///< current fragment in moof atom
209 MOVTrackExt *trex_data;
210 unsigned trex_count;
211 int itunes_metadata; ///< metadata are itunes style
212 int handbrake_version;
213 int *chapter_tracks;
214 unsigned int nb_chapter_tracks;
215 int use_absolute_path;
216 int ignore_editlist;
217 int ignore_chapters;
218 int seek_individually;
219 int64_t next_root_atom; ///< offset of the next root atom
220 int export_all;
221 int export_xmp;
222 int *bitrates; ///< bitrates read before streams creation
223 int bitrates_count;
224 int moov_retry;
225 int use_mfra_for;
226 int has_looked_for_mfra;
227 MOVFragmentIndex** fragment_index_data;
228 unsigned fragment_index_count;
229 int fragment_index_complete;
230 int atom_depth;
231 unsigned int aax_mode; ///< 'aax' file has been detected
232 uint8_t file_key[20];
233 uint8_t file_iv[20];
234 void *activation_bytes;
235 int activation_bytes_size;
236 void *audible_fixed_key;
237 int audible_fixed_key_size;
238 struct AVAES *aes_decrypt;
239 uint8_t *decryption_key;
240 int decryption_key_len;
241 int enable_drefs;
242} MOVContext;
