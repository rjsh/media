33#define URL_PROTOCOL_FLAG_NESTED_SCHEME 1 /*< The protocol name can be the first part of a nested protocol scheme */
34#define URL_PROTOCOL_FLAG_NETWORK 2 /*< The protocol uses network */

36extern const AVClass ffurl_context_class;
37
38typedef struct URLContext {
39 const AVClass *av_class; /**< information for av_log(). Set by url_open(). */
40 const struct URLProtocol *prot;
41 void *priv_data;
42 char *filename; /**< specified URL */
43 int flags;
44 int max_packet_size; /**< if non zero, the stream is packetized with this max packet size */
45 int is_streamed; /**< true if streamed (no seek possible), default = false */
46 int is_connected;
47 AVIOInterruptCB interrupt_callback;
48 int64_t rw_timeout; /**< maximum time to wait for (network) read/write operation completion, in mcs */
49 const char *protocol_whitelist;
50 const char *protocol_blacklist;
51} URLContext;
52
53typedef struct URLProtocol {
54 const char *name;
55 int (*url_open)( URLContext *h, const char *url, int flags);
56 /**
57 * This callback is to be used by protocols which open further nested
58 * protocols. options are then to be passed to ffurl_open()/ffurl_connect()
59 * for those nested protocols.
60 */
61 int (*url_open2)(URLContext *h, const char *url, int flags, AVDictionary **options);
62 int (*url_accept)(URLContext *s, URLContext **c);
63 int (*url_handshake)(URLContext *c);
64
65 /**
66 * Read data from the protocol.
67 * If data is immediately available (even less than size), EOF is
68 * reached or an error occurs (including EINTR), return immediately.
69 * Otherwise:
70 * In non-blocking mode, return AVERROR(EAGAIN) immediately.
71 * In blocking mode, wait for data/EOF/error with a short timeout (0.1s),
72 * and return AVERROR(EAGAIN) on timeout.
73 * Checking interrupt_callback, looping on EINTR and EAGAIN and until
74 * enough data has been read is left to the calling function; see
75 * retry_transfer_wrapper in avio.c.
76 */
77 int (*url_read)( URLContext *h, unsigned char *buf, int size);
78 int (*url_write)(URLContext *h, const unsigned char *buf, int size);
79 int64_t (*url_seek)( URLContext *h, int64_t pos, int whence);
80 int (*url_close)(URLContext *h);
81 int (*url_read_pause)(URLContext *h, int pause);
82 int64_t (*url_read_seek)(URLContext *h, int stream_index,
83 int64_t timestamp, int flags);
84 int (*url_get_file_handle)(URLContext *h);
85 int (*url_get_multi_file_handle)(URLContext *h, int **handles,
86 int *numhandles);
87 int (*url_shutdown)(URLContext *h, int flags);
88 int priv_data_size;
89 const AVClass *priv_data_class;
90 int flags;
91 int (*url_check)(URLContext *h, int mask);
92 int (*url_open_dir)(URLContext *h);
93 int (*url_read_dir)(URLContext *h, AVIODirEntry **next);
94 int (*url_close_dir)(URLContext *h);
95 int (*url_delete)(URLContext *h);
96 int (*url_move)(URLContext *h_src, URLContext *h_dst);
97 const char *default_whitelist;
98} URLProtocol;

100/**
101 * Create a URLContext for accessing to the resource indicated by
102 * url, but do not initiate the connection yet.
103 *
104 * @param puc pointer to the location where, in case of success, the
105 * function puts the pointer to the created URLContext
106 * @param flags flags which control how the resource indicated by url
107 * is to be opened
108 * @param int_cb interrupt callback to use for the URLContext, may be
109 * NULL
110 * @return >= 0 in case of success, a negative value corresponding to an
111 * AVERROR code in case of failure
112 */
113int ffurl_alloc(URLContext **puc, const char *filename, int flags,
114 const AVIOInterruptCB *int_cb);
115
116/**
117 * Connect an URLContext that has been allocated by ffurl_alloc
118 *
119 * @param options A dictionary filled with options for nested protocols,
120 * i.e. it will be passed to url_open2() for protocols implementing it.
121 * This parameter will be destroyed and replaced with a dict containing options
122 * that were not found. May be NULL.
123 */
124int ffurl_connect(URLContext *uc, AVDictionary **options);
125
126/**
127 * Create an URLContext for accessing to the resource indicated by
128 * url, and open it.
129 *
130 * @param puc pointer to the location where, in case of success, the
131 * function puts the pointer to the created URLContext
132 * @param flags flags which control how the resource indicated by url
133 * is to be opened
134 * @param int_cb interrupt callback to use for the URLContext, may be
135 * NULL
136 * @param options A dictionary filled with protocol-private options. On return
137 * this parameter will be destroyed and replaced with a dict containing options
138 * that were not found. May be NULL.
139 * @param parent An enclosing URLContext, whose generic options should
140 * be applied to this URLContext as well.
141 * @return >= 0 in case of success, a negative value corresponding to an
142 * AVERROR code in case of failure
143 */
144int ffurl_open_whitelist(URLContext **puc, const char *filename, int flags,
145 const AVIOInterruptCB *int_cb, AVDictionary **options,
146 const char *whitelist, const char* blacklist,
147 URLContext *parent);
148
149int ffurl_open(URLContext **puc, const char *filename, int flags,
150 const AVIOInterruptCB *int_cb, AVDictionary **options);
151
152/**
153 * Accept an URLContext c on an URLContext s
154 *
155 * @param s server context
156 * @param c client context, must be unallocated.
157 * @return >= 0 on success, ff_neterrno() on failure.
158 */
159int ffurl_accept(URLContext *s, URLContext **c);
160
161/**
162 * Perform one step of the protocol handshake to accept a new client.
163 * See avio_handshake() for details.
164 * Implementations should try to return decreasing values.
165 * If the protocol uses an underlying protocol, the underlying handshake is
166 * usually the first step, and the return value can be:
167 * (largest value for this protocol) + (return value from other protocol)
168 *
169 * @param c the client context
170 * @return >= 0 on success or a negative value corresponding
171 * to an AVERROR code on failure
172 */
173int ffurl_handshake(URLContext *c);
174
175/**
176 * Read up to size bytes from the resource accessed by h, and store
177 * the read bytes in buf.
178 *
179 * @return The number of bytes actually read, or a negative value
180 * corresponding to an AVERROR code in case of error. A value of zero
181 * indicates that it is not possible to read more from the accessed
182 * resource (except if the value of the size argument is also zero).
183 */
184int ffurl_read(URLContext *h, unsigned char *buf, int size);
185
186/**
187 * Read as many bytes as possible (up to size), calling the
188 * read function multiple times if necessary.
189 * This makes special short-read handling in applications
190 * unnecessary, if the return value is < size then it is
191 * certain there was either an error or the end of file was reached.
192 */
193int ffurl_read_complete(URLContext *h, unsigned char *buf, int size);
194
195/**
196 * Write size bytes from buf to the resource accessed by h.
197 *
198 * @return the number of bytes actually written, or a negative value
199 * corresponding to an AVERROR code in case of failure
200 */
201int ffurl_write(URLContext *h, const unsigned char *buf, int size);
202
203/**
204 * Change the position that will be used by the next read/write
205 * operation on the resource accessed by h.
206 *
207 * @param pos specifies the new position to set
208 * @param whence specifies how pos should be interpreted, it must be
209 * one of SEEK_SET (seek from the beginning), SEEK_CUR (seek from the
210 * current position), SEEK_END (seek from the end), or AVSEEK_SIZE
211 * (return the filesize of the requested resource, pos is ignored).
212 * @return a negative value corresponding to an AVERROR code in case
213 * of failure, or the resulting file position, measured in bytes from
214 * the beginning of the file. You can use this feature together with
215 * SEEK_CUR to read the current file position.
216 */
217int64_t ffurl_seek(URLContext *h, int64_t pos, int whence);
218
219/**
220 * Close the resource accessed by the URLContext h, and free the
221 * memory used by it. Also set the URLContext pointer to NULL.
222 *
223 * @return a negative value if an error condition occurred, 0
224 * otherwise
225 */
226int ffurl_closep(URLContext **h);
227int ffurl_close(URLContext *h);
228
229/**
230 * Return the filesize of the resource accessed by h, AVERROR(ENOSYS)
231 * if the operation is not supported by h, or another negative value
232 * corresponding to an AVERROR error code in case of failure.
233 */
234int64_t ffurl_size(URLContext *h);
235
236/**
237 * Return the file descriptor associated with this URL. For RTP, this
238 * will return only the RTP file descriptor, not the RTCP file descriptor.
239 *
240 * @return the file descriptor associated with this URL, or <0 on error.
241 */
242int ffurl_get_file_handle(URLContext *h);
243
244/**
245 * Return the file descriptors associated with this URL.
246 *
247 * @return 0 on success or <0 on error.
248 */
249int ffurl_get_multi_file_handle(URLContext *h, int **handles, int *numhandles);

251/**
252 * Signal the URLContext that we are done reading or writing the stream.
253 *
254 * @param h pointer to the resource

258 * @return a negative value if an error condition occurred, 0

261int ffurl_shutdown(URLContext *h, int flags);

273/**
274 * Assemble a URL string from components. This is the reverse operation
275 * of av_url_split.
276 *
277 * Note, this requires networking to be initialized, so the caller must
278 * ensure ff_network_init has been called.
279 *
280 * @see av_url_split
281 *
282 * @param str the buffer to fill with the url
283 * @param size the size of the str buffer
284 * @param proto the protocol identifier, if null, the separator
285 * after the identifier is left out, too
286 * @param authorization an optional authorization string, may be null.
287 * An empty string is treated the same as a null string.
288 * @param hostname the host name string
289 * @param port the port number, left out from the string if negative
290 * @param fmt a generic format string for everything to add after the
291 * host/port, may be null
292 * @return the number of characters written to the destination buffer
293 */
294int ff_url_join(char *str, int size, const char *proto,
295 const char *authorization, const char *hostname,
296 int port, const char *fmt, ...) av_printf_format(7, 8);

298/**
299 * Convert a relative url into an absolute url, given a base url.
300 *
301 * @param buf the buffer where output absolute url is written
302 * @param size the size of buf
303 * @param base the base url, may be equal to buf.
304 * @param rel the new url, which is interpreted relative to base
305 */
306void ff_make_absolute_url(char *buf, int size, const char *base,
307 const char *rel);

318/**
319 * Construct a list of protocols matching a given whitelist and/or blacklist.
320 *
321 * @param whitelist a comma-separated list of allowed protocol names or NULL. If
322 * this is a non-empty string, only protocols in this list will
323 * be included.
324 * @param blacklist a comma-separated list of forbidden protocol names or NULL.
325 * If this is a non-empty string, all protocols in this list
326 * will be excluded.
327 *
328 * @return a NULL-terminated array of matching protocols. The array must be
329 * freed by the caller.
330 */
331const URLProtocol **ffurl_get_protocols(const char *whitelist,
332 const char *blacklist);
