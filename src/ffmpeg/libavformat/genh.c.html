26typedef struct GENHDemuxContext {
27 unsigned dsp_int_type;
28 unsigned interleave_size;
29} GENHDemuxContext;

31static int genh_probe(AVProbeData *p)
32{
33 if (AV_RL32(p->buf) != MKTAG('G','E','N','H'))
34 return 0;
35 if (AV_RL32(p->buf+4) <= 0 || AV_RL32(p->buf+4) > 0xFFFF) // channels
36 return 0;
37
38 return AVPROBE_SCORE_MAX / 3 * 2;
39}

41static int genh_read_header(AVFormatContext *s)
42{
43 unsigned start_offset, header_size, codec, coef_type, coef[2];
44 GENHDemuxContext *c = s->priv_data;
45 av_unused unsigned coef_splitted[2];
46 int align, ch, ret;
47 AVStream *st;
48
49 avio_skip(s->pb, 4);
50
51 st = avformat_new_stream(s, NULL);
52 if (!st)
53 return AVERROR(ENOMEM);
54
55 st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
56 st->codecpar->channels = avio_rl32(s->pb);
57 if (st->codecpar->channels <= 0)
58 return AVERROR_INVALIDDATA;
59 if (st->codecpar->channels == 1)
60 st->codecpar->channel_layout = AV_CH_LAYOUT_MONO;
61 else if (st->codecpar->channels == 2)
62 st->codecpar->channel_layout = AV_CH_LAYOUT_STEREO;
63 align =
64 c->interleave_size = avio_rl32(s->pb);
65 if (align < 0 || align > INT_MAX / st->codecpar->channels)
66 return AVERROR_INVALIDDATA;
67 st->codecpar->block_align = align * st->codecpar->channels;
68 st->codecpar->sample_rate = avio_rl32(s->pb);
69 avio_skip(s->pb, 4);
70 st->duration = avio_rl32(s->pb);
71
72 codec = avio_rl32(s->pb);
73 switch (codec) {
74 case 0: st->codecpar->codec_id = AV_CODEC_ID_ADPCM_PSX; break;
75 case 1:
76 case 11: st->codecpar->bits_per_coded_sample = 4;
77 st->codecpar->block_align = 36 * st->codecpar->channels;
78 st->codecpar->codec_id = AV_CODEC_ID_ADPCM_IMA_WAV; break;
79 case 2: st->codecpar->codec_id = AV_CODEC_ID_ADPCM_DTK; break;
80 case 3: st->codecpar->codec_id = st->codecpar->block_align > 0 ?
81 AV_CODEC_ID_PCM_S16BE_PLANAR :
82 AV_CODEC_ID_PCM_S16BE; break;
83 case 4: st->codecpar->codec_id = st->codecpar->block_align > 0 ?
84 AV_CODEC_ID_PCM_S16LE_PLANAR :
85 AV_CODEC_ID_PCM_S16LE; break;
86 case 5: st->codecpar->codec_id = st->codecpar->block_align > 0 ?
87 AV_CODEC_ID_PCM_S8_PLANAR :
88 AV_CODEC_ID_PCM_S8; break;
89 case 6: st->codecpar->codec_id = AV_CODEC_ID_SDX2_DPCM; break;
90 case 7: ret = ff_alloc_extradata(st->codecpar, 2);
91 if (ret < 0)
92 return ret;
93 AV_WL16(st->codecpar->extradata, 3);
94 st->codecpar->codec_id = AV_CODEC_ID_ADPCM_IMA_WS; break;
95 case 10: st->codecpar->codec_id = AV_CODEC_ID_ADPCM_AICA; break;
96 case 12: st->codecpar->codec_id = AV_CODEC_ID_ADPCM_THP; break;
97 case 13: st->codecpar->codec_id = AV_CODEC_ID_PCM_U8; break;
98 case 17: st->codecpar->codec_id = AV_CODEC_ID_ADPCM_IMA_QT; break;
99 default:
100 avpriv_request_sample(s, "codec %d", codec);
101 return AVERROR_PATCHWELCOME;
102 }
103
104 start_offset = avio_rl32(s->pb);
105 header_size = avio_rl32(s->pb);
106
107 if (header_size > start_offset)
108 return AVERROR_INVALIDDATA;
109
110 if (header_size == 0)
111 start_offset = 0x800;
112
113 coef[0] = avio_rl32(s->pb);
114 coef[1] = avio_rl32(s->pb);
115 c->dsp_int_type = avio_rl32(s->pb);
116 coef_type = avio_rl32(s->pb);
117 coef_splitted[0] = avio_rl32(s->pb);
118 coef_splitted[1] = avio_rl32(s->pb);
119
120 if (st->codecpar->codec_id == AV_CODEC_ID_ADPCM_THP) {
121 if (st->codecpar->channels > 2) {
122 avpriv_request_sample(s, "channels %d>2", st->codecpar->channels);
123 return AVERROR_PATCHWELCOME;
124 }
125
126 ff_alloc_extradata(st->codecpar, 32 * st->codecpar->channels);
127 for (ch = 0; ch < st->codecpar->channels; ch++) {
128 if (coef_type & 1) {
129 avpriv_request_sample(s, "coef_type & 1");
130 return AVERROR_PATCHWELCOME;
131 } else {
132 avio_seek(s->pb, coef[ch], SEEK_SET);
133 avio_read(s->pb, st->codecpar->extradata + 32 * ch, 32);
134 }
135 }
136
137 if (c->dsp_int_type == 1) {
138 st->codecpar->block_align = 8 * st->codecpar->channels;
139 if (c->interleave_size != 1 &&
140 c->interleave_size != 2 &&
141 c->interleave_size != 4)
142 return AVERROR_INVALIDDATA;
143 }
144 }
145
146 avio_skip(s->pb, start_offset - avio_tell(s->pb));
147
148 avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
149
150 return 0;
151}

153static int genh_read_packet(AVFormatContext *s, AVPacket *pkt)
154{
155 AVCodecParameters *par = s->streams[0]->codecpar;
156 GENHDemuxContext *c = s->priv_data;
157 int ret;
158
159 if (c->dsp_int_type == 1 && par->codec_id == AV_CODEC_ID_ADPCM_THP &&
160 par->channels > 1) {
161 int i, ch;
162
163 if (avio_feof(s->pb))
164 return AVERROR_EOF;
165 ret = av_new_packet(pkt, 8 * par->channels);
166 if (ret < 0)
167 return ret;
168 for (i = 0; i < 8 / c->interleave_size; i++) {
169 for (ch = 0; ch < par->channels; ch++) {
170 pkt->data[ch * 8 + i*c->interleave_size+0] = avio_r8(s->pb);
171 pkt->data[ch * 8 + i*c->interleave_size+1] = avio_r8(s->pb);
172 }
173 }
174 ret = 0;
175 } else if (par->codec_id == AV_CODEC_ID_SDX2_DPCM) {
176 ret = av_get_packet(s->pb, pkt, par->block_align * 1024);
177
178 } else {
179 ret = av_get_packet(s->pb, pkt, par->block_align ? par->block_align : 1024 * par->channels);
180 }
181
182 pkt->stream_index = 0;
183 return ret;
184}
