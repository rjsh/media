21#include "libavutil/attributes.h"
22#include "avformat.h"

25#define RTP_G726_HANDLER(bitrate) \
26static av_cold int g726_ ## bitrate ##_init(AVFormatContext *s, int st_index, \
27 PayloadContext *data) \
28{ \
29 AVStream *stream = s->streams[st_index]; \
30 AVCodecParameters *par = stream->codecpar; \
31\
32 par->bits_per_coded_sample = bitrate/8; \
33 par->bit_rate = par->bits_per_coded_sample * par->sample_rate; \
34\
35 return 0; \
36} \
37\
38RTPDynamicProtocolHandler ff_g726_ ## bitrate ## _dynamic_handler = { \
39 .enc_name = "AAL2-G726-" #bitrate, \
40 .codec_type = AVMEDIA_TYPE_AUDIO, \
41 .codec_id = AV_CODEC_ID_ADPCM_G726, \
42 .init = g726_ ## bitrate ## _init, \
43}; \
44RTPDynamicProtocolHandler ff_g726le_ ## bitrate ## _dynamic_handler = { \
45 .enc_name = "G726-" #bitrate, \
46 .codec_type = AVMEDIA_TYPE_AUDIO, \
47 .codec_id = AV_CODEC_ID_ADPCM_G726LE, \
48 .init = g726_ ## bitrate ## _init, \
49}
