141int ff_id3v2_match(const uint8_t *buf, const char *magic)
142{
143 return buf[0] == magic[0] &&
144 buf[1] == magic[1] &&
145 buf[2] == magic[2] &&
146 buf[3] != 0xff &&
147 buf[4] != 0xff &&
148 (buf[6] & 0x80) == 0 &&
149 (buf[7] & 0x80) == 0 &&
150 (buf[8] & 0x80) == 0 &&
151 (buf[9] & 0x80) == 0;
152}
153
154int ff_id3v2_tag_len(const uint8_t *buf)
155{
156 int len = ((buf[6] & 0x7f) << 21) +
157 ((buf[7] & 0x7f) << 14) +
158 ((buf[8] & 0x7f) << 7) +
159 (buf[9] & 0x7f) +
160 ID3v2_HEADER_SIZE;
161 if (buf[5] & 0x10)
162 len += ID3v2_HEADER_SIZE;
163 return len;
164}

166static unsigned int get_size(AVIOContext *s, int len)
167{
168 int v = 0;
169 while (len--)
170 v = (v << 7) + (avio_r8(s) & 0x7F);
171 return v;
172}

174static unsigned int size_to_syncsafe(unsigned int size)
175{
176 return (((size) & (0x7f << 0)) >> 0) +
177 (((size) & (0x7f << 8)) >> 1) +
178 (((size) & (0x7f << 16)) >> 2) +
179 (((size) & (0x7f << 24)) >> 3);
180}

182/* No real verification, only check that the tag consists of
183 * a combination of capital alpha-numerical characters */
184static int is_tag(const char *buf, unsigned int len)
185{
186 if (!len)
187 return 0;
188
189 while (len--)
190 if ((buf[len] < 'A' ||
191 buf[len] > 'Z') &&
192 (buf[len] < '0' ||
193 buf[len] > '9'))
194 return 0;
195
196 return 1;
197}

199/**
200 * Return 1 if the tag of length len at the given offset is valid, 0 if not, -1 on error
201 */
202static int check_tag(AVIOContext *s, int offset, unsigned int len)
203{
204 char tag[4];
205
206 if (len > 4 ||
207 avio_seek(s, offset, SEEK_SET) < 0 ||
208 avio_read(s, tag, len) < (int)len)
209 return -1;
210 else if (!AV_RB32(tag) || is_tag(tag, len))
211 return 1;
212
213 return 0;
214}

229/**
230 * Decode characters to UTF-8 according to encoding type. The decoded buffer is
231 * always null terminated. Stop reading when either *maxread bytes are read from
232 * pb or U+0000 character is found.
233 *
234 * @param dst Pointer where the address of the buffer with the decoded bytes is
235 * stored. Buffer must be freed by caller.
236 * @param maxread Pointer to maximum number of characters to read from the
237 * AVIOContext. After execution the value is decremented by the number of bytes
238 * actually read.
239 * @returns 0 if no error occurred, dst is uninitialized on error
240 */
241static int decode_str(AVFormatContext *s, AVIOContext *pb, int encoding,
242 uint8_t **dst, int *maxread)
243{
244 int ret;
245 uint8_t tmp;
246 uint32_t ch = 1;
247 int left = *maxread;
248 unsigned int (*get)(AVIOContext*) = avio_rb16;
249 AVIOContext *dynbuf;
250
251 if ((ret = avio_open_dyn_buf(&dynbuf)) < 0) {
252 av_log(s, AV_LOG_ERROR, "Error opening memory stream\n");
253 return ret;
254 }
255
256 switch (encoding) {
257 case ID3v2_ENCODING_ISO8859:
258 while (left && ch) {
259 ch = avio_r8(pb);
260 PUT_UTF8(ch, tmp, avio_w8(dynbuf, tmp);)
261 left--;
262 }
263 break;
264
265 case ID3v2_ENCODING_UTF16BOM:
266 if ((left -= 2) < 0) {
267 av_log(s, AV_LOG_ERROR, "Cannot read BOM value, input too short\n");
268 ffio_free_dyn_buf(&dynbuf);
269 *dst = NULL;
270 return AVERROR_INVALIDDATA;
271 }
272 switch (avio_rb16(pb)) {
273 case 0xfffe:
274 get = avio_rl16;
275 case 0xfeff:
276 break;
277 default:
278 av_log(s, AV_LOG_ERROR, "Incorrect BOM value\n");
279 ffio_free_dyn_buf(&dynbuf);
280 *dst = NULL;
281 *maxread = left;
282 return AVERROR_INVALIDDATA;
283 }
284 // fall-through
285
286 case ID3v2_ENCODING_UTF16BE:
287 while ((left > 1) && ch) {
288 GET_UTF16(ch, ((left -= 2) >= 0 ? get(pb) : 0), break;)
289 PUT_UTF8(ch, tmp, avio_w8(dynbuf, tmp);)
290 }
291 if (left < 0)
292 left += 2; /* did not read last char from pb */
293 break;
294
295 case ID3v2_ENCODING_UTF8:
296 while (left && ch) {
297 ch = avio_r8(pb);
298 avio_w8(dynbuf, ch);
299 left--;
300 }
301 break;
302 default:
303 av_log(s, AV_LOG_WARNING, "Unknown encoding\n");
304 }
305
306 if (ch)
307 avio_w8(dynbuf, 0);
308
309 avio_close_dyn_buf(dynbuf, dst);
310 *maxread = left;
311
312 return 0;
313}

315/**
316 * Parse a text tag.
317 */
318static void read_ttag(AVFormatContext *s, AVIOContext *pb, int taglen,
319 AVDictionary **metadata, const char *key)
320{
321 uint8_t *dst;
322 int encoding, dict_flags = AV_DICT_DONT_OVERWRITE | AV_DICT_DONT_STRDUP_VAL;
323 unsigned genre;
324
325 if (taglen < 1)
326 return;
327
328 encoding = avio_r8(pb);
329 taglen--; /* account for encoding type byte */
330
331 if (decode_str(s, pb, encoding, &dst, &taglen) < 0) {
332 av_log(s, AV_LOG_ERROR, "Error reading frame %s, skipped\n", key);
333 return;
334 }
335
336 if (!(strcmp(key, "TCON") && strcmp(key, "TCO")) &&
337 (sscanf(dst, "(%d)", &genre) == 1 || sscanf(dst, "%d", &genre) == 1) &&
338 genre <= ID3v1_GENRE_MAX) {
339 av_freep(&dst);
340 dst = av_strdup(ff_id3v1_genre_str[genre]);
341 } else if (!(strcmp(key, "TXXX") && strcmp(key, "TXX"))) {
342 /* dst now contains the key, need to get value */
343 key = dst;
344 if (decode_str(s, pb, encoding, &dst, &taglen) < 0) {
345 av_log(s, AV_LOG_ERROR, "Error reading frame %s, skipped\n", key);
346 av_freep(&key);
347 return;
348 }
349 dict_flags |= AV_DICT_DONT_STRDUP_KEY;
350 } else if (!*dst)
351 av_freep(&dst);
352
353 if (dst)
354 av_dict_set(metadata, key, dst, dict_flags);
355}

357static void read_uslt(AVFormatContext *s, AVIOContext *pb, int taglen,
358 AVDictionary **metadata)
359{
360 uint8_t lang[4];
361 uint8_t *descriptor = NULL; // 'Content descriptor'
362 uint8_t *text = NULL;
363 char *key = NULL;
364 int encoding;
365 int ok = 0;
366
367 if (taglen < 1)
368 goto error;
369
370 encoding = avio_r8(pb);
371 taglen--;
372
373 if (avio_read(pb, lang, 3) < 3)
374 goto error;
375 lang[3] = '\0';
376 taglen -= 3;
377
378 if (decode_str(s, pb, encoding, &descriptor, &taglen) < 0)
379 goto error;
380
381 if (decode_str(s, pb, encoding, &text, &taglen) < 0)
382 goto error;
383
384 // FFmpeg does not support hierarchical metadata, so concatenate the keys.
385 key = av_asprintf("lyrics-%s%s%s", descriptor[0] ? (char *)descriptor : "",
386 descriptor[0] ? "-" : "",
387 lang);
388 if (!key)
389 goto error;
390
391 av_dict_set(metadata, key, text, 0);
392
393 ok = 1;
394error:
395 if (!ok)
396 av_log(s, AV_LOG_ERROR, "Error reading lyrics, skipped\n");
397 av_free(descriptor);
398 av_free(text);
399 av_free(key);
400}

402/**
403 * Parse a comment tag.
404 */
405static void read_comment(AVFormatContext *s, AVIOContext *pb, int taglen,
406 AVDictionary **metadata)
407{
408 const char *key = "comment";
409 uint8_t *dst;
410 int encoding, dict_flags = AV_DICT_DONT_OVERWRITE | AV_DICT_DONT_STRDUP_VAL;
411 av_unused int language;
412
413 if (taglen < 4)
414 return;
415
416 encoding = avio_r8(pb);
417 language = avio_rl24(pb);
418 taglen -= 4;
419
420 if (decode_str(s, pb, encoding, &dst, &taglen) < 0) {
421 av_log(s, AV_LOG_ERROR, "Error reading comment frame, skipped\n");
422 return;
423 }
424
425 if (dst && !*dst)
426 av_freep(&dst);
427
428 if (dst) {
429 key = (const char *) dst;
430 dict_flags |= AV_DICT_DONT_STRDUP_KEY;
431 }
432
433 if (decode_str(s, pb, encoding, &dst, &taglen) < 0) {
434 av_log(s, AV_LOG_ERROR, "Error reading comment frame, skipped\n");
435 if (dict_flags & AV_DICT_DONT_STRDUP_KEY)
436 av_freep((void*)&key);
437 return;
438 }
439
440 if (dst)
441 av_dict_set(metadata, key, (const char *) dst, dict_flags);
442}

444/**
445 * Parse GEOB tag into a ID3v2ExtraMetaGEOB struct.
446 */
447static void read_geobtag(AVFormatContext *s, AVIOContext *pb, int taglen,
448 const char *tag, ID3v2ExtraMeta **extra_meta,
449 int isv34)
450{
451 ID3v2ExtraMetaGEOB *geob_data = NULL;
452 ID3v2ExtraMeta *new_extra = NULL;
453 char encoding;
454 unsigned int len;
455
456 if (taglen < 1)
457 return;
458
459 geob_data = av_mallocz(sizeof(ID3v2ExtraMetaGEOB));
460 if (!geob_data) {
461 av_log(s, AV_LOG_ERROR, "Failed to alloc %"SIZE_SPECIFIER" bytes\n",
462 sizeof(ID3v2ExtraMetaGEOB));
463 return;
464 }
465
466 new_extra = av_mallocz(sizeof(ID3v2ExtraMeta));
467 if (!new_extra) {
468 av_log(s, AV_LOG_ERROR, "Failed to alloc %"SIZE_SPECIFIER" bytes\n",
469 sizeof(ID3v2ExtraMeta));
470 goto fail;
471 }
472
473 /* read encoding type byte */
474 encoding = avio_r8(pb);
475 taglen--;
476
477 /* read MIME type (always ISO-8859) */
478 if (decode_str(s, pb, ID3v2_ENCODING_ISO8859, &geob_data->mime_type,
479 &taglen) < 0 ||
480 taglen <= 0)
481 goto fail;
482
483 /* read file name */
484 if (decode_str(s, pb, encoding, &geob_data->file_name, &taglen) < 0 ||
485 taglen <= 0)
486 goto fail;
487
488 /* read content description */
489 if (decode_str(s, pb, encoding, &geob_data->description, &taglen) < 0 ||
490 taglen < 0)
491 goto fail;
492
493 if (taglen) {
494 /* save encapsulated binary data */
495 geob_data->data = av_malloc(taglen);
496 if (!geob_data->data) {
497 av_log(s, AV_LOG_ERROR, "Failed to alloc %d bytes\n", taglen);
498 goto fail;
499 }
500 if ((len = avio_read(pb, geob_data->data, taglen)) < taglen)
501 av_log(s, AV_LOG_WARNING,
502 "Error reading GEOB frame, data truncated.\n");
503 geob_data->datasize = len;
504 } else {
505 geob_data->data = NULL;
506 geob_data->datasize = 0;
507 }
508
509 /* add data to the list */
510 new_extra->tag = "GEOB";
511 new_extra->data = geob_data;
512 new_extra->next = *extra_meta;
513 *extra_meta = new_extra;
514
515 return;
516
517fail:
518 av_log(s, AV_LOG_ERROR, "Error reading frame %s, skipped\n", tag);
519 free_geobtag(geob_data);
520 av_free(new_extra);
521 return;
522}

524static int is_number(const char *str)
525{
526 while (*str >= '0' && *str <= '9')
527 str++;
528 return !*str;
529}

531static AVDictionaryEntry *get_date_tag(AVDictionary *m, const char *tag)
532{
533 AVDictionaryEntry *t;
534 if ((t = av_dict_get(m, tag, NULL, AV_DICT_MATCH_CASE)) &&
535 strlen(t->value) == 4 && is_number(t->value))
536 return t;
537 return NULL;
538}

540static void merge_date(AVDictionary **m)
541{
542 AVDictionaryEntry *t;
543 char date[17] = { 0 }; // YYYY-MM-DD hh:mm
544
545 if (!(t = get_date_tag(*m, "TYER")) &&
546 !(t = get_date_tag(*m, "TYE")))
547 return;
548 av_strlcpy(date, t->value, 5);
549 av_dict_set(m, "TYER", NULL, 0);
550 av_dict_set(m, "TYE", NULL, 0);
551
552 if (!(t = get_date_tag(*m, "TDAT")) &&
553 !(t = get_date_tag(*m, "TDA")))
554 goto finish;
555 snprintf(date + 4, sizeof(date) - 4, "-%.2s-%.2s", t->value + 2, t->value);
556 av_dict_set(m, "TDAT", NULL, 0);
557 av_dict_set(m, "TDA", NULL, 0);
558
559 if (!(t = get_date_tag(*m, "TIME")) &&
560 !(t = get_date_tag(*m, "TIM")))
561 goto finish;
562 snprintf(date + 10, sizeof(date) - 10,
563 " %.2s:%.2s", t->value, t->value + 2);
564 av_dict_set(m, "TIME", NULL, 0);
565 av_dict_set(m, "TIM", NULL, 0);
566
567finish:
568 if (date[0])
569 av_dict_set(m, "date", date, 0);
570}

580static void rstrip_spaces(char *buf)
581{
582 size_t len = strlen(buf);
583 while (len > 0 && buf[len - 1] == ' ')
584 buf[--len] = 0;
585}

587static void read_apic(AVFormatContext *s, AVIOContext *pb, int taglen,
588 const char *tag, ID3v2ExtraMeta **extra_meta,
589 int isv34)
590{
591 int enc, pic_type;
592 char mimetype[64];
593 const CodecMime *mime = ff_id3v2_mime_tags;
594 enum AVCodecID id = AV_CODEC_ID_NONE;
595 ID3v2ExtraMetaAPIC *apic = NULL;
596 ID3v2ExtraMeta *new_extra = NULL;
597 int64_t end = avio_tell(pb) + taglen;
598
599 if (taglen <= 4 || (!isv34 && taglen <= 6))
600 goto fail;
601
602 new_extra = av_mallocz(sizeof(*new_extra));
603 apic = av_mallocz(sizeof(*apic));
604 if (!new_extra || !apic)
605 goto fail;
606
607 enc = avio_r8(pb);
608 taglen--;
609
610 /* mimetype */
611 if (isv34) {
612 taglen -= avio_get_str(pb, taglen, mimetype, sizeof(mimetype));
613 } else {
614 avio_read(pb, mimetype, 3);
615 mimetype[3] = 0;
616 taglen -= 3;
617 }
618
619 while (mime->id != AV_CODEC_ID_NONE) {
620 if (!av_strncasecmp(mime->str, mimetype, sizeof(mimetype))) {
621 id = mime->id;
622 break;
623 }
624 mime++;
625 }
626 if (id == AV_CODEC_ID_NONE) {
627 av_log(s, AV_LOG_WARNING,
628 "Unknown attached picture mimetype: %s, skipping.\n", mimetype);
629 goto fail;
630 }
631 apic->id = id;
632
633 /* picture type */
634 pic_type = avio_r8(pb);
635 taglen--;
636 if (pic_type < 0 || pic_type >= FF_ARRAY_ELEMS(ff_id3v2_picture_types)) {
637 av_log(s, AV_LOG_WARNING, "Unknown attached picture type %d.\n",
638 pic_type);
639 pic_type = 0;
640 }
641 apic->type = ff_id3v2_picture_types[pic_type];
642
643 /* description and picture data */
644 if (decode_str(s, pb, enc, &apic->description, &taglen) < 0) {
645 av_log(s, AV_LOG_ERROR,
646 "Error decoding attached picture description.\n");
647 goto fail;
648 }
649
650 apic->buf = av_buffer_alloc(taglen + AV_INPUT_BUFFER_PADDING_SIZE);
651 if (!apic->buf || !taglen || avio_read(pb, apic->buf->data, taglen) != taglen)
652 goto fail;
653 memset(apic->buf->data + taglen, 0, AV_INPUT_BUFFER_PADDING_SIZE);
654
655 new_extra->tag = "APIC";
656 new_extra->data = apic;
657 new_extra->next = *extra_meta;
658 *extra_meta = new_extra;
659
660 // The description must be unique, and some ID3v2 tag writers add spaces
661 // to write several APIC entries with the same description.
662 rstrip_spaces(apic->description);
663
664 return;
665
666fail:
667 if (apic)
668 free_apic(apic);
669 av_freep(&new_extra);
670 avio_seek(pb, end, SEEK_SET);
671}

673static void read_chapter(AVFormatContext *s, AVIOContext *pb, int len, const char *ttag, ID3v2ExtraMeta **extra_meta, int isv34)
674{
675 AVRational time_base = {1, 1000};
676 uint32_t start, end;
677 AVChapter *chapter;
678 uint8_t *dst = NULL;
679 int taglen;
680 char tag[5];
681
682 if (!s) {
683 /* We should probably just put the chapter data to extra_meta here
684 * and do the AVFormatContext-needing part in a separate
685 * ff_id3v2_parse_apic()-like function. */
686 av_log(NULL, AV_LOG_DEBUG, "No AVFormatContext, skipped ID3 chapter data\n");
687 return;
688 }
689
690 if (decode_str(s, pb, 0, &dst, &len) < 0)
691 return;
692 if (len < 16)
693 return;
694
695 start = avio_rb32(pb);
696 end = avio_rb32(pb);
697 avio_skip(pb, 8);
698
699 chapter = avpriv_new_chapter(s, s->nb_chapters + 1, time_base, start, end, dst);
700 if (!chapter) {
701 av_free(dst);
702 return;
703 }
704
705 len -= 16;
706 while (len > 10) {
707 if (avio_read(pb, tag, 4) < 4)
708 goto end;
709 tag[4] = 0;
710 taglen = avio_rb32(pb);
711 avio_skip(pb, 2);
712 len -= 10;
713 if (taglen < 0 || taglen > len)
714 goto end;
715 if (tag[0] == 'T')
716 read_ttag(s, pb, taglen, &chapter->metadata, tag);
717 else
718 avio_skip(pb, taglen);
719 len -= taglen;
720 }
721
722 ff_metadata_conv(&chapter->metadata, NULL, ff_id3v2_34_metadata_conv);
723 ff_metadata_conv(&chapter->metadata, NULL, ff_id3v2_4_metadata_conv);
724end:
725 av_free(dst);
726}

736static void read_priv(AVFormatContext *s, AVIOContext *pb, int taglen,
737 const char *tag, ID3v2ExtraMeta **extra_meta, int isv34)
738{
739 ID3v2ExtraMeta *meta;
740 ID3v2ExtraMetaPRIV *priv;
741
742 meta = av_mallocz(sizeof(*meta));
743 priv = av_mallocz(sizeof(*priv));
744
745 if (!meta || !priv)
746 goto fail;
747
748 if (decode_str(s, pb, ID3v2_ENCODING_ISO8859, &priv->owner, &taglen) < 0)
749 goto fail;
750
751 priv->data = av_malloc(taglen);
752 if (!priv->data)
753 goto fail;
754
755 priv->datasize = taglen;
756
757 if (avio_read(pb, priv->data, priv->datasize) != priv->datasize)
758 goto fail;
759
760 meta->tag = "PRIV";
761 meta->data = priv;
762 meta->next = *extra_meta;
763 *extra_meta = meta;
764
765 return;
766
767fail:
768 if (priv)
769 free_priv(priv);
770 av_freep(&meta);
771}

773typedef struct ID3v2EMFunc {
774 const char *tag3;
775 const char *tag4;
776 void (*read)(AVFormatContext *s, AVIOContext *pb, int taglen,
777 const char *tag, ID3v2ExtraMeta **extra_meta,
778 int isv34);
779 void (*free)(void *obj);
780} ID3v2EMFunc;

782static const ID3v2EMFunc id3v2_extra_meta_funcs[] = {
783 { "GEO", "GEOB", read_geobtag, free_geobtag },
784 { "PIC", "APIC", read_apic, free_apic },
785 { "CHAP","CHAP", read_chapter, NULL },
786 { "PRIV","PRIV", read_priv, free_priv },
787 { NULL }
788};

790/**
791 * Get the corresponding ID3v2EMFunc struct for a tag.
792 * @param isv34 Determines if v2.2 or v2.3/4 strings are used
793 * @return A pointer to the ID3v2EMFunc struct if found, NULL otherwise.
794 */
795static const ID3v2EMFunc *get_extra_meta_func(const char *tag, int isv34)
796{
797 int i = 0;
798 while (id3v2_extra_meta_funcs[i].tag3) {
799 if (tag && !memcmp(tag,
800 (isv34 ? id3v2_extra_meta_funcs[i].tag4 :
801 id3v2_extra_meta_funcs[i].tag3),
802 (isv34 ? 4 : 3)))
803 return &id3v2_extra_meta_funcs[i];
804 i++;
805 }
806 return NULL;
807}

809static void id3v2_parse(AVIOContext *pb, AVDictionary **metadata,
810 AVFormatContext *s, int len, uint8_t version,
811 uint8_t flags, ID3v2ExtraMeta **extra_meta)
812{
813 int isv34, unsync;
814 unsigned tlen;
815 char tag[5];
816 int64_t next, end = avio_tell(pb) + len;
817 int taghdrlen;
818 const char *reason = NULL;
819 AVIOContext pb_local;
820 AVIOContext *pbx;
821 unsigned char *buffer = NULL;
822 int buffer_size = 0;
823 const ID3v2EMFunc *extra_func = NULL;
824 unsigned char *uncompressed_buffer = NULL;
825 av_unused int uncompressed_buffer_size = 0;
826
827 av_log(s, AV_LOG_DEBUG, "id3v2 ver:%d flags:%02X len:%d\n", version, flags, len);
828
829 switch (version) {
830 case 2:
831 if (flags & 0x40) {
832 reason = "compression";
833 goto error;
834 }
835 isv34 = 0;
836 taghdrlen = 6;
837 break;
838
839 case 3:
840 case 4:
841 isv34 = 1;
842 taghdrlen = 10;
843 break;
844
845 default:
846 reason = "version";
847 goto error;
848 }
849
850 unsync = flags & 0x80;
851
852 if (isv34 && flags & 0x40) { /* Extended header present, just skip over it */
853 int extlen = get_size(pb, 4);
854 if (version == 4)
855 /* In v2.4 the length includes the length field we just read. */
856 extlen -= 4;
857
858 if (extlen < 0) {
859 reason = "invalid extended header length";
860 goto error;
861 }
862 avio_skip(pb, extlen);
863 len -= extlen + 4;
864 if (len < 0) {
865 reason = "extended header too long.";
866 goto error;
867 }
868 }
869
870 while (len >= taghdrlen) {
871 unsigned int tflags = 0;
872 int tunsync = 0;
873 int tcomp = 0;
874 int tencr = 0;
875 unsigned long av_unused dlen;
876
877 if (isv34) {
878 if (avio_read(pb, tag, 4) < 4)
879 break;
880 tag[4] = 0;
881 if (version == 3) {
882 tlen = avio_rb32(pb);
883 } else {
884 /* some encoders incorrectly uses v3 sizes instead of syncsafe ones
885 * so check the next tag to see which one to use */
886 tlen = avio_rb32(pb);
887 if (tlen > 0x7f) {
888 if (tlen < len) {
889 int64_t cur = avio_tell(pb);
890
891 if (ffio_ensure_seekback(pb, 2 /* tflags */ + tlen + 4 /* next tag */))
892 break;
893
894 if (check_tag(pb, cur + 2 + size_to_syncsafe(tlen), 4) == 1)
895 tlen = size_to_syncsafe(tlen);
896 else if (check_tag(pb, cur + 2 + tlen, 4) != 1)
897 break;
898 avio_seek(pb, cur, SEEK_SET);
899 } else
900 tlen = size_to_syncsafe(tlen);
901 }
902 }
903 tflags = avio_rb16(pb);
904 tunsync = tflags & ID3v2_FLAG_UNSYNCH;
905 } else {
906 if (avio_read(pb, tag, 3) < 3)
907 break;
908 tag[3] = 0;
909 tlen = avio_rb24(pb);
910 }
911 if (tlen > (1<<28))
912 break;
913 len -= taghdrlen + tlen;
914
915 if (len < 0)
916 break;
917
918 next = avio_tell(pb) + tlen;
919
920 if (!tlen) {
921 if (tag[0])
922 av_log(s, AV_LOG_DEBUG, "Invalid empty frame %s, skipping.\n",
923 tag);
924 continue;
925 }
926
927 if (tflags & ID3v2_FLAG_DATALEN) {
928 if (tlen < 4)
929 break;
930 dlen = avio_rb32(pb);
931 tlen -= 4;
932 } else
933 dlen = tlen;
934
935 tcomp = tflags & ID3v2_FLAG_COMPRESSION;
936 tencr = tflags & ID3v2_FLAG_ENCRYPTION;
937
938 /* skip encrypted tags and, if no zlib, compressed tags */
939 if (tencr || (!CONFIG_ZLIB && tcomp)) {
940 const char *type;
941 if (!tcomp)
942 type = "encrypted";
943 else if (!tencr)
944 type = "compressed";
945 else
946 type = "encrypted and compressed";
947
948 av_log(s, AV_LOG_WARNING, "Skipping %s ID3v2 frame %s.\n", type, tag);
949 avio_skip(pb, tlen);
950 /* check for text tag or supported special meta tag */
951 } else if (tag[0] == 'T' ||
952 !memcmp(tag, "USLT", 4) ||
953 !memcmp(tag, "COMM", 4) ||
954 (extra_meta &&
955 (extra_func = get_extra_meta_func(tag, isv34)))) {
956 pbx = pb;
957
958 if (unsync || tunsync || tcomp) {
959 av_fast_malloc(&buffer, &buffer_size, tlen);
960 if (!buffer) {
961 av_log(s, AV_LOG_ERROR, "Failed to alloc %d bytes\n", tlen);
962 goto seek;
963 }
964 }
965 if (unsync || tunsync) {
966 int64_t end = avio_tell(pb) + tlen;
967 uint8_t *b;
968
969 b = buffer;
970 while (avio_tell(pb) < end && b - buffer < tlen && !pb->eof_reached) {
971 *b++ = avio_r8(pb);
972 if (*(b - 1) == 0xff && avio_tell(pb) < end - 1 &&
973 b - buffer < tlen &&
974 !pb->eof_reached ) {
975 uint8_t val = avio_r8(pb);
976 *b++ = val ? val : avio_r8(pb);
977 }
978 }
979 ffio_init_context(&pb_local, buffer, b - buffer, 0, NULL, NULL, NULL,
980 NULL);
981 tlen = b - buffer;
982 pbx = &pb_local; // read from sync buffer
983 }
984
985#if CONFIG_ZLIB
986 if (tcomp) {
987 int err;
988
989 av_log(s, AV_LOG_DEBUG, "Compresssed frame %s tlen=%d dlen=%ld\n", tag, tlen, dlen);
990
991 av_fast_malloc(&uncompressed_buffer, &uncompressed_buffer_size, dlen);
992 if (!uncompressed_buffer) {
993 av_log(s, AV_LOG_ERROR, "Failed to alloc %ld bytes\n", dlen);
994 goto seek;
995 }
996
997 if (!(unsync || tunsync)) {
998 err = avio_read(pb, buffer, tlen);
999 if (err < 0) {
1000 av_log(s, AV_LOG_ERROR, "Failed to read compressed tag\n");
1001 goto seek;
1002 }
1003 tlen = err;
1004 }
1005
1006 err = uncompress(uncompressed_buffer, &dlen, buffer, tlen);
1007 if (err != Z_OK) {
1008 av_log(s, AV_LOG_ERROR, "Failed to uncompress tag: %d\n", err);
1009 goto seek;
1010 }
1011 ffio_init_context(&pb_local, uncompressed_buffer, dlen, 0, NULL, NULL, NULL, NULL);
1012 tlen = dlen;
1013 pbx = &pb_local; // read from sync buffer
1014 }
1015#endif
1016 if (tag[0] == 'T')
1017 /* parse text tag */
1018 read_ttag(s, pbx, tlen, metadata, tag);
1019 else if (!memcmp(tag, "USLT", 4))
1020 read_uslt(s, pbx, tlen, metadata);
1021 else if (!memcmp(tag, "COMM", 4))
1022 read_comment(s, pbx, tlen, metadata);
1023 else
1024 /* parse special meta tag */
1025 extra_func->read(s, pbx, tlen, tag, extra_meta, isv34);
1026 } else if (!tag[0]) {
1027 if (tag[1])
1028 av_log(s, AV_LOG_WARNING, "invalid frame id, assuming padding\n");
1029 avio_skip(pb, tlen);
1030 break;
1031 }
1032 /* Skip to end of tag */
1033seek:
1034 avio_seek(pb, next, SEEK_SET);
1035 }
1036
1037 /* Footer preset, always 10 bytes, skip over it */
1038 if (version == 4 && flags & 0x10)
1039 end += 10;
1040
1041error:
1042 if (reason)
1043 av_log(s, AV_LOG_INFO, "ID3v2.%d tag skipped, cannot handle %s\n",
1044 version, reason);
1045 avio_seek(pb, end, SEEK_SET);
1046 av_free(buffer);
1047 av_free(uncompressed_buffer);
1048 return;
1049}

1051static void id3v2_read_internal(AVIOContext *pb, AVDictionary **metadata,
1052 AVFormatContext *s, const char *magic,
1053 ID3v2ExtraMeta **extra_meta, int64_t max_search_size)
1054{
1055 int len, ret;
1056 uint8_t buf[ID3v2_HEADER_SIZE];
1057 int found_header;
1058 int64_t start, off;
1059
1060 if (max_search_size && max_search_size < ID3v2_HEADER_SIZE)
1061 return;
1062
1063 start = avio_tell(pb);
1064 do {
1065 /* save the current offset in case there's nothing to read/skip */
1066 off = avio_tell(pb);
1067 if (max_search_size && off - start >= max_search_size - ID3v2_HEADER_SIZE) {
1068 avio_seek(pb, off, SEEK_SET);
1069 break;
1070 }
1071
1072 ret = avio_read(pb, buf, ID3v2_HEADER_SIZE);
1073 if (ret != ID3v2_HEADER_SIZE) {
1074 avio_seek(pb, off, SEEK_SET);
1075 break;
1076 }
1077 found_header = ff_id3v2_match(buf, magic);
1078 if (found_header) {
1079 /* parse ID3v2 header */
1080 len = ((buf[6] & 0x7f) << 21) |
1081 ((buf[7] & 0x7f) << 14) |
1082 ((buf[8] & 0x7f) << 7) |
1083 (buf[9] & 0x7f);
1084 id3v2_parse(pb, metadata, s, len, buf[3], buf[5], extra_meta);
1085 } else {
1086 avio_seek(pb, off, SEEK_SET);
1087 }
1088 } while (found_header);
1089 ff_metadata_conv(metadata, NULL, ff_id3v2_34_metadata_conv);
1090 ff_metadata_conv(metadata, NULL, id3v2_2_metadata_conv);
1091 ff_metadata_conv(metadata, NULL, ff_id3v2_4_metadata_conv);
1092 merge_date(metadata);
1093}

1096 const char *magic, ID3v2ExtraMeta **extra_meta)
1097{
1098 id3v2_read_internal(pb, metadata, NULL, magic, extra_meta, 0);
1099}

1107void ff_id3v2_free_extra_meta(ID3v2ExtraMeta **extra_meta)
1108{
1109 ID3v2ExtraMeta *current = *extra_meta, *next;
1110 const ID3v2EMFunc *extra_func;
1111
1112 while (current) {
1113 if ((extra_func = get_extra_meta_func(current->tag, 1)))
1114 extra_func->free(current->data);
1115 next = current->next;
1116 av_freep(&current);
1117 current = next;
1118 }
1119
1120 *extra_meta = NULL;
1121}
