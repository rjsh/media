27static int adp_probe(AVProbeData *p)
28{
29 int i, changes = 0;
30 char last = 0;
31
32 if (p->buf_size < 32)
33 return 0;
34
35 for (i = 0; i < p->buf_size - 3; i+=32) {
36 if (p->buf[i] != p->buf[i+2] || p->buf[i+1] != p->buf[i+3])
37 return 0;
38 if (p->buf[i] != last)
39 changes++;
40 last = p->buf[i];
41 }
42 if (changes <= 1)
43 return 0;
44
45 return p->buf_size < 260 ? 1 : AVPROBE_SCORE_MAX / 4;
46}

48static int adp_read_header(AVFormatContext *s)
49{
50 AVStream *st;
51
52 st = avformat_new_stream(s, NULL);
53 if (!st)
54 return AVERROR(ENOMEM);
55
56 st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
57 st->codecpar->codec_id = AV_CODEC_ID_ADPCM_DTK;
58 st->codecpar->channel_layout = AV_CH_LAYOUT_STEREO;
59 st->codecpar->channels = 2;
60 st->codecpar->sample_rate = 48000;
61 st->start_time = 0;
62 if (s->pb->seekable)
63 st->duration = av_get_audio_frame_duration2(st->codecpar, avio_size(s->pb));
64
65 avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
66
67 return 0;
68}

70static int adp_read_packet(AVFormatContext *s, AVPacket *pkt)
71{
72 int ret, size = 1024;
73
74 if (avio_feof(s->pb))
75 return AVERROR_EOF;
76
77 ret = av_get_packet(s->pb, pkt, size);
78
79 if (ret != size) {
80 if (ret < 0) {
81 av_packet_unref(pkt);
82 return ret;
83 }
84 av_shrink_packet(pkt, ret);
85 }
86 pkt->stream_index = 0;
87
88 return ret;
89}

94 .read_probe = adp_probe,

96 .read_packet = adp_read_packet,
97 .extensions = "adp,dtk",
