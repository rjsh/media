21#ifndef AVFORMAT_AVLANGUAGE_H
22#define AVFORMAT_AVLANGUAGE_H
23
24#include "libavutil/attributes.h"
25#include "libavformat/version.h"
26
27/**
28 * Known language codespaces
29 */
30enum AVLangCodespace {
31 AV_LANG_ISO639_2_BIBL, /** 3-char bibliographic language codes as per ISO-IEC 639-2 */
32 AV_LANG_ISO639_2_TERM, /** 3-char terminological language codes as per ISO-IEC 639-2 */
33 AV_LANG_ISO639_1 /** 2-char code of language as per ISO/IEC 639-1 */
34};
35
36/**
37 * Convert a language code to a target codespace. The source codespace is guessed.
38 * @return NULL if the provided lang is null or invalid.
39 */
40const char *ff_convert_lang_to(const char *lang, enum AVLangCodespace target_codespace);
41#if LIBAVFORMAT_VERSION_MAJOR < 58
42attribute_deprecated
43const char *av_convert_lang_to(const char *lang, enum AVLangCodespace target_codespace);
44#endif
45
46#endif /* AVFORMAT_AVLANGUAGE_H */
