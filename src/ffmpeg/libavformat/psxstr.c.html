22/**
23 * @file
24 * PSX STR file demuxer
25 * by Mike Melanson (melanson@pcisys.net)
26 * This module handles streams that have been ripped from Sony Playstation
27 * CD games. This demuxer can handle either raw STR files (which are just
28 * concatenations of raw compact disc sectors) or STR files with 0x2C-byte
29 * RIFF headers, followed by CD sectors.
30 */
31

38#define RIFF_TAG MKTAG('R', 'I', 'F', 'F')
39#define CDXA_TAG MKTAG('C', 'D', 'X', 'A')
40
41#define RAW_CD_SECTOR_SIZE 2352
42#define RAW_CD_SECTOR_DATA_SIZE 2304
43#define VIDEO_DATA_CHUNK_SIZE 0x7E0
44#define VIDEO_DATA_HEADER_SIZE 0x38
45#define RIFF_HEADER_SIZE 0x2C
46
47#define CDXA_TYPE_MASK 0x0E
48#define CDXA_TYPE_DATA 0x08
49#define CDXA_TYPE_AUDIO 0x04
50#define CDXA_TYPE_VIDEO 0x02
51
52#define STR_MAGIC (0x80010160)
53
54typedef struct StrChannel {
55 /* video parameters */
56 int video_stream_index;
57 AVPacket tmp_pkt;
58
59 /* audio parameters */
60 int audio_stream_index;
61} StrChannel;
62
63typedef struct StrDemuxContext {
64
65 /* a STR file can contain up to 32 channels of data */
66 StrChannel channels[32];
67} StrDemuxContext;

69static const uint8_t sync_header[12] = {0x00,0xff,0xff,0xff,0xff,0xff,0xff,0xff,0xff,0xff,0xff,0x00};
70
71static int str_probe(AVProbeData *p)
72{
73 const uint8_t *sector= p->buf;
74 const uint8_t *end= sector + p->buf_size;
75 int aud=0, vid=0;
76
77 if (p->buf_size < RAW_CD_SECTOR_SIZE)
78 return 0;
79
80 if ((AV_RL32(&p->buf[0]) == RIFF_TAG) &&
81 (AV_RL32(&p->buf[8]) == CDXA_TAG)) {
82
83 /* RIFF header seen; skip 0x2C bytes */
84 sector += RIFF_HEADER_SIZE;
85 }
86
87 while (end - sector >= RAW_CD_SECTOR_SIZE) {
88 /* look for CD sync header (00, 0xFF x 10, 00) */
89 if (memcmp(sector,sync_header,sizeof(sync_header)))
90 return 0;
91
92 if (sector[0x11] >= 32)
93 return 0;
94
95 switch (sector[0x12] & CDXA_TYPE_MASK) {
96 case CDXA_TYPE_DATA:
97 case CDXA_TYPE_VIDEO: {
98 int current_sector = AV_RL16(&sector[0x1C]);
99 int sector_count = AV_RL16(&sector[0x1E]);
100 int frame_size = AV_RL32(&sector[0x24]);
101
102 if(!( frame_size>=0
103 && current_sector < sector_count
104 && sector_count*VIDEO_DATA_CHUNK_SIZE >=frame_size)){
105 return 0;
106 }
107 vid++;
108
109 }
110 break;
111 case CDXA_TYPE_AUDIO:
112 if(sector[0x13]&0x2A)
113 return 0;
114 aud++;
115 break;
116 default:
117 if(sector[0x12] & CDXA_TYPE_MASK)
118 return 0;
119 }
120 sector += RAW_CD_SECTOR_SIZE;
121 }
122 /* MPEG files (like those ripped from VCDs) can also look like this;
123 * only return half certainty */
124 if(vid+aud > 3) return AVPROBE_SCORE_EXTENSION;
125 else if(vid+aud) return 1;
126 else return 0;
127}

129static int str_read_header(AVFormatContext *s)
130{
131 AVIOContext *pb = s->pb;
132 StrDemuxContext *str = s->priv_data;
133 unsigned char sector[RAW_CD_SECTOR_SIZE];
134 int start;
135 int i;
136
137 /* skip over any RIFF header */
138 if (avio_read(pb, sector, RIFF_HEADER_SIZE) != RIFF_HEADER_SIZE)
139 return AVERROR(EIO);
140 if (AV_RL32(&sector[0]) == RIFF_TAG)
141 start = RIFF_HEADER_SIZE;
142 else
143 start = 0;
144
145 avio_seek(pb, start, SEEK_SET);
146
147 for(i=0; i<32; i++){
148 str->channels[i].video_stream_index=
149 str->channels[i].audio_stream_index= -1;
150 }
151
152 s->ctx_flags |= AVFMTCTX_NOHEADER;
153
154 return 0;
155}

157static int str_read_packet(AVFormatContext *s,
158 AVPacket *ret_pkt)
159{
160 AVIOContext *pb = s->pb;
161 StrDemuxContext *str = s->priv_data;
162 unsigned char sector[RAW_CD_SECTOR_SIZE];
163 int channel;
164 AVPacket *pkt;
165 AVStream *st;
166
167 while (1) {
168
169 if (avio_read(pb, sector, RAW_CD_SECTOR_SIZE) != RAW_CD_SECTOR_SIZE)
170 return AVERROR(EIO);
171
172 channel = sector[0x11];
173 if (channel >= 32)
174 return AVERROR_INVALIDDATA;
175
176 switch (sector[0x12] & CDXA_TYPE_MASK) {
177
178 case CDXA_TYPE_DATA:
179 case CDXA_TYPE_VIDEO:
180 {
181
182 int current_sector = AV_RL16(&sector[0x1C]);
183 int sector_count = AV_RL16(&sector[0x1E]);
184 int frame_size = AV_RL32(&sector[0x24]);
185
186 if(!( frame_size>=0
187 && current_sector < sector_count
188 && sector_count*VIDEO_DATA_CHUNK_SIZE >=frame_size)){
189 av_log(s, AV_LOG_ERROR, "Invalid parameters %d %d %d\n", current_sector, sector_count, frame_size);
190 break;
191 }
192
193 if(str->channels[channel].video_stream_index < 0){
194 /* allocate a new AVStream */
195 st = avformat_new_stream(s, NULL);
196 if (!st)
197 return AVERROR(ENOMEM);
198 avpriv_set_pts_info(st, 64, 1, 15);
199
200 str->channels[channel].video_stream_index = st->index;
201
202 st->codecpar->codec_type = AVMEDIA_TYPE_VIDEO;
203 st->codecpar->codec_id = AV_CODEC_ID_MDEC;
204 st->codecpar->codec_tag = 0; /* no fourcc */
205 st->codecpar->width = AV_RL16(&sector[0x28]);
206 st->codecpar->height = AV_RL16(&sector[0x2A]);
207 }
208
209 /* if this is the first sector of the frame, allocate a pkt */
210 pkt = &str->channels[channel].tmp_pkt;
211
212 if(pkt->size != sector_count*VIDEO_DATA_CHUNK_SIZE){
213 if(pkt->data)
214 av_log(s, AV_LOG_ERROR, "mismatching sector_count\n");
215 av_packet_unref(pkt);
216 if (av_new_packet(pkt, sector_count*VIDEO_DATA_CHUNK_SIZE))
217 return AVERROR(EIO);
218 memset(pkt->data, 0, sector_count*VIDEO_DATA_CHUNK_SIZE);
219
220 pkt->pos= avio_tell(pb) - RAW_CD_SECTOR_SIZE;
221 pkt->stream_index =
222 str->channels[channel].video_stream_index;
223 }
224
225 memcpy(pkt->data + current_sector*VIDEO_DATA_CHUNK_SIZE,
226 sector + VIDEO_DATA_HEADER_SIZE,
227 VIDEO_DATA_CHUNK_SIZE);
228
229 if (current_sector == sector_count-1) {
230 pkt->size= frame_size;
231 *ret_pkt = *pkt;
232 pkt->data= NULL;
233 pkt->size= -1;
234 pkt->buf = NULL;
235 return 0;
236 }
237
238 }
239 break;
240
241 case CDXA_TYPE_AUDIO:
242 if(str->channels[channel].audio_stream_index < 0){
243 int fmt = sector[0x13];
244 /* allocate a new AVStream */
245 st = avformat_new_stream(s, NULL);
246 if (!st)
247 return AVERROR(ENOMEM);
248
249 str->channels[channel].audio_stream_index = st->index;
250
251 st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
252 st->codecpar->codec_id = AV_CODEC_ID_ADPCM_XA;
253 st->codecpar->codec_tag = 0; /* no fourcc */
254 if (fmt & 1) {
255 st->codecpar->channels = 2;
256 st->codecpar->channel_layout = AV_CH_LAYOUT_STEREO;
257 } else {
258 st->codecpar->channels = 1;
259 st->codecpar->channel_layout = AV_CH_LAYOUT_MONO;
260 }
261 st->codecpar->sample_rate = (fmt&4)?18900:37800;
262 // st->codecpar->bit_rate = 0; //FIXME;
263 st->codecpar->block_align = 128;
264
265 avpriv_set_pts_info(st, 64, 18 * 224 / st->codecpar->channels,
266 st->codecpar->sample_rate);
267 st->start_time = 0;
268 }
269 pkt = ret_pkt;
270 if (av_new_packet(pkt, 2304))
271 return AVERROR(EIO);
272 memcpy(pkt->data,sector+24,2304);
273
274 pkt->stream_index =
275 str->channels[channel].audio_stream_index;
276 pkt->duration = 1;
277 return 0;
278 default:
279 av_log(s, AV_LOG_WARNING, "Unknown sector type %02X\n", sector[0x12]);
280 /* drop the sector and move on */
281 break;
282 }
283
284 if (avio_feof(pb))
285 return AVERROR(EIO);
286 }
287}
288
289static int str_read_close(AVFormatContext *s)
290{
291 StrDemuxContext *str = s->priv_data;
292 int i;
293 for(i=0; i<32; i++){
294 if(str->channels[i].tmp_pkt.data)
295 av_packet_unref(&str->channels[i].tmp_pkt);
296 }
297
298 return 0;
299}

301AVInputFormat ff_str_demuxer = {
302 .name = "psxstr",
303 .long_name = NULL_IF_CONFIG_SMALL("Sony Playstation STR"),
304 .priv_data_size = sizeof(StrDemuxContext),
305 .read_probe = str_probe,
306 .read_header = str_read_header,
307 .read_packet = str_read_packet,
308 .read_close = str_read_close,
309 .flags = AVFMT_NO_BYTE_SEEK,
310};
