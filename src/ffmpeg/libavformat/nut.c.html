238void ff_nut_reset_ts(NUTContext *nut, AVRational time_base, int64_t val)
239{
240 int i;
241 for (i = 0; i < nut->avf->nb_streams; i++)
242 nut->stream[i].last_pts =
243 av_rescale_rnd(val,
244 time_base.num * (int64_t)nut->stream[i].time_base->den,
245 time_base.den * (int64_t)nut->stream[i].time_base->num,
246 AV_ROUND_DOWN);
247}
248
249int64_t ff_lsb2full(StreamContext *stream, int64_t lsb)
250{
251 int64_t mask = (1ULL << stream->msb_pts_shift) - 1;
252 int64_t delta = stream->last_pts - mask / 2;
253 return ((lsb - delta) & mask) + delta;
254}

256int ff_nut_sp_pos_cmp(const void *a, const void *b)
257{
258 const Syncpoint *va = a, *vb = b;
259 return ((va->pos - vb->pos) >> 32) - ((vb->pos - va->pos) >> 32);
260}
261
262int ff_nut_sp_pts_cmp(const void *a, const void *b)
263{
264 const Syncpoint *va = a, *vb = b;
265 return ((va->ts - vb->ts) >> 32) - ((vb->ts - va->ts) >> 32);
266}

268int ff_nut_add_sp(NUTContext *nut, int64_t pos, int64_t back_ptr, int64_t ts)
269{
270 Syncpoint *sp = av_mallocz(sizeof(Syncpoint));
271 struct AVTreeNode *node = av_tree_node_alloc();
272
273 if (!sp || !node) {
274 av_freep(&sp);
275 av_freep(&node);
276 return AVERROR(ENOMEM);
277 }
278
279 nut->sp_count++;
280
281 sp->pos = pos;
282 sp->back_ptr = back_ptr;
283 sp->ts = ts;
284 av_tree_insert(&nut->syncpoints, sp, ff_nut_sp_pos_cmp, &node);
285 if (node) {
286 av_free(sp);
287 av_free(node);
288 }
289
290 return 0;
291}

293static int enu_free(void *opaque, void *elem)
294{
295 av_free(elem);
296 return 0;
297}
298
299void ff_nut_free_sp(NUTContext *nut)
300{
301 if (nut->syncpoints) {
302 av_tree_enumerate(nut->syncpoints, NULL, NULL, enu_free);
303 av_tree_destroy(nut->syncpoints);
304 }
