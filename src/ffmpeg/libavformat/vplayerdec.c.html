21/**
22 * @file
23 * VPlayer subtitles format demuxer
24 */
25
26#include "avformat.h"
27#include "internal.h"
28#include "subtitles.h"
29
30typedef struct {
31 FFDemuxSubtitlesQueue q;
32} VPlayerContext;
33
34static int vplayer_probe(AVProbeData *p)
35{
36 char c;
37 const unsigned char *ptr = p->buf;
38
39 if ((sscanf(ptr, "%*3d:%*2d:%*2d.%*2d%c", &c) == 1 ||
40 sscanf(ptr, "%*3d:%*2d:%*2d%c", &c) == 1) && strchr(": =", c))
41 return AVPROBE_SCORE_MAX;
42 return 0;
43}
44
45static int64_t read_ts(char **line)
46{
47 char c;
48 int hh, mm, ss, ms, n, len;
49
50 if (((n = sscanf(*line, "%d:%d:%d.%d%c%n", &hh, &mm, &ss, &ms, &c, &len)) >= 5 ||
51 (n = sscanf(*line, "%d:%d:%d%c%n", &hh, &mm, &ss, &c, &len)) >= 4) && strchr(": =", c)) {
52 *line += len;
53 return (hh*3600LL + mm*60LL + ss) * 100LL + (n < 5 ? 0 : ms);
54 }
55 return AV_NOPTS_VALUE;
56}
57
58static int vplayer_read_header(AVFormatContext *s)
59{
60 VPlayerContext *vplayer = s->priv_data;
61 AVStream *st = avformat_new_stream(s, NULL);
62
63 if (!st)
64 return AVERROR(ENOMEM);
65 avpriv_set_pts_info(st, 64, 1, 100);
66 st->codecpar->codec_type = AVMEDIA_TYPE_SUBTITLE;
67 st->codecpar->codec_id = AV_CODEC_ID_VPLAYER;
68
69 while (!avio_feof(s->pb)) {
70 char line[4096];
71 char *p = line;
72 const int64_t pos = avio_tell(s->pb);
73 int len = ff_get_line(s->pb, line, sizeof(line));
74 int64_t pts_start;
75
76 if (!len)
77 break;
78
79 line[strcspn(line, "\r\n")] = 0;
80
81 pts_start = read_ts(&p);
82 if (pts_start != AV_NOPTS_VALUE) {
83 AVPacket *sub;
84
85 sub = ff_subtitles_queue_insert(&vplayer->q, p, strlen(p), 0);
86 if (!sub)
87 return AVERROR(ENOMEM);
88 sub->pos = pos;
89 sub->pts = pts_start;
90 sub->duration = -1;
91 }
92 }
93
94 ff_subtitles_queue_finalize(s, &vplayer->q);
95 return 0;
96}
97
98static int vplayer_read_packet(AVFormatContext *s, AVPacket *pkt)
99{
100 VPlayerContext *vplayer = s->priv_data;
101 return ff_subtitles_queue_read_packet(&vplayer->q, pkt);
102}
103
104static int vplayer_read_seek(AVFormatContext *s, int stream_index,
105 int64_t min_ts, int64_t ts, int64_t max_ts, int flags)
106{
107 VPlayerContext *vplayer = s->priv_data;
108 return ff_subtitles_queue_seek(&vplayer->q, s, stream_index,
109 min_ts, ts, max_ts, flags);
110}
111
112static int vplayer_read_close(AVFormatContext *s)
113{
114 VPlayerContext *vplayer = s->priv_data;
115 ff_subtitles_queue_clean(&vplayer->q);
116 return 0;
117}

119AVInputFormat ff_vplayer_demuxer = {
120 .name = "vplayer",
121 .long_name = NULL_IF_CONFIG_SMALL("VPlayer subtitles"),
122 .priv_data_size = sizeof(VPlayerContext),
123 .read_probe = vplayer_probe,
124 .read_header = vplayer_read_header,
125 .read_packet = vplayer_read_packet,
126 .read_seek2 = vplayer_read_seek,
127 .read_close = vplayer_read_close,
128 .extensions = "txt",
129};
