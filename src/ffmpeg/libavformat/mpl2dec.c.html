21/**
22 * @file
23 * MPL2 subtitles format demuxer
24 */
25
26#include "avformat.h"
27#include "internal.h"
28#include "subtitles.h"
29
30typedef struct {
31 FFDemuxSubtitlesQueue q;
32} MPL2Context;
33
34static int mpl2_probe(AVProbeData *p)
35{
36 int i;
37 char c;
38 int64_t start, end;
39 const unsigned char *ptr = p->buf;
40 const unsigned char *ptr_end = ptr + p->buf_size;
41
42 for (i = 0; i < 2; i++) {
43 if (sscanf(ptr, "[%"SCNd64"][%"SCNd64"]%c", &start, &end, &c) != 3 &&
44 sscanf(ptr, "[%"SCNd64"][]%c", &start, &c) != 2)
45 return 0;
46 ptr += ff_subtitles_next_line(ptr);
47 if (ptr >= ptr_end)
48 return 0;
49 }
50 return AVPROBE_SCORE_MAX;
51}
52
53static int read_ts(char **line, int64_t *pts_start, int *duration)
54{
55 char c;
56 int len;
57 int64_t end;
58
59 if (sscanf(*line, "[%"SCNd64"][]%c%n",
60 pts_start, &c, &len) >= 2) {
61 *duration = -1;
62 *line += len - 1;
63 return 0;
64 }
65 if (sscanf(*line, "[%"SCNd64"][%"SCNd64"]%c%n",
66 pts_start, &end, &c, &len) >= 3) {
67 *duration = end - *pts_start;
68 *line += len - 1;
69 return 0;
70 }
71 return -1;
72}

74static int mpl2_read_header(AVFormatContext *s)
75{
76 MPL2Context *mpl2 = s->priv_data;
77 AVStream *st = avformat_new_stream(s, NULL);
78 int res = 0;
79
80 if (!st)
81 return AVERROR(ENOMEM);
82 avpriv_set_pts_info(st, 64, 1, 10);
83 st->codecpar->codec_type = AVMEDIA_TYPE_SUBTITLE;
84 st->codecpar->codec_id = AV_CODEC_ID_MPL2;
85
86 while (!avio_feof(s->pb)) {
87 char line[4096];
88 char *p = line;
89 const int64_t pos = avio_tell(s->pb);
90 int len = ff_get_line(s->pb, line, sizeof(line));
91 int64_t pts_start;
92 int duration;
93
94 if (!len)
95 break;
96
97 line[strcspn(line, "\r\n")] = 0;
98
99 if (!read_ts(&p, &pts_start, &duration)) {
100 AVPacket *sub;
101
102 sub = ff_subtitles_queue_insert(&mpl2->q, p, strlen(p), 0);
103 if (!sub)
104 return AVERROR(ENOMEM);
105 sub->pos = pos;
106 sub->pts = pts_start;
107 sub->duration = duration;
108 }
109 }
110
111 ff_subtitles_queue_finalize(s, &mpl2->q);
112 return res;
113}
114
115static int mpl2_read_packet(AVFormatContext *s, AVPacket *pkt)
116{
117 MPL2Context *mpl2 = s->priv_data;
118 return ff_subtitles_queue_read_packet(&mpl2->q, pkt);
119}
120
121static int mpl2_read_seek(AVFormatContext *s, int stream_index,
122 int64_t min_ts, int64_t ts, int64_t max_ts, int flags)
123{
124 MPL2Context *mpl2 = s->priv_data;
125 return ff_subtitles_queue_seek(&mpl2->q, s, stream_index,
126 min_ts, ts, max_ts, flags);
127}
128
129static int mpl2_read_close(AVFormatContext *s)
130{
131 MPL2Context *mpl2 = s->priv_data;
132 ff_subtitles_queue_clean(&mpl2->q);
133 return 0;
134}

136AVInputFormat ff_mpl2_demuxer = {
137 .name = "mpl2",
138 .long_name = NULL_IF_CONFIG_SMALL("MPL2 subtitles"),
139 .priv_data_size = sizeof(MPL2Context),
140 .read_probe = mpl2_probe,
141 .read_header = mpl2_read_header,
142 .read_packet = mpl2_read_packet,
143 .read_seek2 = mpl2_read_seek,
144 .read_close = mpl2_read_close,
145 .extensions = "txt,mpl2",
146};
