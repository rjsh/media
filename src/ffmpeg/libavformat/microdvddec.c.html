32typedef struct {
33 const AVClass *class;
34 FFDemuxSubtitlesQueue q;
35 AVRational frame_rate;
36} MicroDVDContext;

39static int microdvd_probe(AVProbeData *p)
40{
41 unsigned char c;
42 const uint8_t *ptr = p->buf;
43 int i;
44
45 if (AV_RB24(ptr) == 0xEFBBBF)
46 ptr += 3; /* skip UTF-8 BOM */
47
48 for (i=0; i<3; i++) {
49 if (sscanf(ptr, "{%*d}{}%c", &c) != 1 &&
50 sscanf(ptr, "{%*d}{%*d}%c", &c) != 1 &&
51 sscanf(ptr, "{DEFAULT}{}%c", &c) != 1)
52 return 0;
53 ptr += ff_subtitles_next_line(ptr);
54 }
55 return AVPROBE_SCORE_MAX;
56}
57
58static int64_t get_pts(const char *buf)
59{
60 int frame;
61 char c;
62
63 if (sscanf(buf, "{%d}{%c", &frame, &c) == 2)
64 return frame;
65 return AV_NOPTS_VALUE;
66}
67
68static int get_duration(const char *buf)
69{
70 int frame_start, frame_end;
71
72 if (sscanf(buf, "{%d}{%d}", &frame_start, &frame_end) == 2)
73 return frame_end - frame_start;
74 return -1;
75}

77static const char *bom = "\xEF\xBB\xBF";

79static int microdvd_read_header(AVFormatContext *s)
80{
81 AVRational pts_info = (AVRational){ 2997, 125 }; /* default: 23.976 fps */
82 MicroDVDContext *microdvd = s->priv_data;
83 AVStream *st = avformat_new_stream(s, NULL);
84 int i = 0;
85 char line_buf[MAX_LINESIZE];
86 int has_real_fps = 0;
87
88 if (!st)
89 return AVERROR(ENOMEM);
90
91 while (!avio_feof(s->pb)) {
92 char *p;
93 AVPacket *sub;
94 int64_t pos = avio_tell(s->pb);
95 int len = ff_get_line(s->pb, line_buf, sizeof(line_buf));
96 char *line = line_buf;
97
98 if (!strncmp(line, bom, 3))
99 line += 3;
100 p = line;
101
102 if (!len)
103 break;
104 line[strcspn(line, "\r\n")] = 0;
105 if (i++ < 3) {
106 int frame;
107 double fps;
108 char c;
109
110 if ((sscanf(line, "{%d}{}%6lf", &frame, &fps) == 2 ||
111 sscanf(line, "{%d}{%*d}%6lf", &frame, &fps) == 2)
112 && frame <= 1 && fps > 3 && fps < 100) {
113 pts_info = av_d2q(fps, 100000);
114 has_real_fps = 1;
115 continue;
116 }
117 if (!st->codecpar->extradata && sscanf(line, "{DEFAULT}{}%c", &c) == 1) {
118 st->codecpar->extradata = av_strdup(line + 11);
119 if (!st->codecpar->extradata)
120 return AVERROR(ENOMEM);
121 st->codecpar->extradata_size = strlen(st->codecpar->extradata) + 1;
122 continue;
123 }
124 }
125#define SKIP_FRAME_ID \
126 p = strchr(p, '}'); \
127 if (!p) { \
128 av_log(s, AV_LOG_WARNING, "Invalid event \"%s\"" \
129 " at line %d\n", line, i); \
130 continue; \
131 } \
132 p++
133 SKIP_FRAME_ID;
134 SKIP_FRAME_ID;
135 if (!*p)
136 continue;
137 sub = ff_subtitles_queue_insert(&microdvd->q, p, strlen(p), 0);
138 if (!sub)
139 return AVERROR(ENOMEM);
140 sub->pos = pos;
141 sub->pts = get_pts(line);
142 sub->duration = get_duration(line);
143 }
144 ff_subtitles_queue_finalize(s, &microdvd->q);
145 if (has_real_fps) {
146 /* export the FPS info only if set in the file */
147 microdvd->frame_rate = pts_info;
148 } else if (microdvd->frame_rate.num) {
149 /* fallback on user specified frame rate */
150 pts_info = microdvd->frame_rate;
151 }
152 avpriv_set_pts_info(st, 64, pts_info.den, pts_info.num);
153 st->codecpar->codec_type = AVMEDIA_TYPE_SUBTITLE;
154 st->codecpar->codec_id = AV_CODEC_ID_MICRODVD;
155 return 0;
156}

158static int microdvd_read_packet(AVFormatContext *s, AVPacket *pkt)
159{
160 MicroDVDContext *microdvd = s->priv_data;
161 return ff_subtitles_queue_read_packet(&microdvd->q, pkt);
162}
163
164static int microdvd_read_seek(AVFormatContext *s, int stream_index,
165 int64_t min_ts, int64_t ts, int64_t max_ts, int flags)
166{
167 MicroDVDContext *microdvd = s->priv_data;
168 return ff_subtitles_queue_seek(&microdvd->q, s, stream_index,
169 min_ts, ts, max_ts, flags);
170}
171
172static int microdvd_read_close(AVFormatContext *s)
173{
174 MicroDVDContext *microdvd = s->priv_data;
175 ff_subtitles_queue_clean(&microdvd->q);
176 return 0;
177}

180#define OFFSET(x) offsetof(MicroDVDContext, x)
181#define SD AV_OPT_FLAG_SUBTITLE_PARAM|AV_OPT_FLAG_DECODING_PARAM
182static const AVOption microdvd_options[] = {
183 { "subfps", "set the movie frame rate fallback", OFFSET(frame_rate), AV_OPT_TYPE_RATIONAL, {.dbl=0}, 0, INT_MAX, SD },
184 { NULL }
185};
