22#include "avformat.h"
23#include "internal.h"

25static int svag_probe(AVProbeData *p)
26{
27 if (memcmp(p->buf, "Svag", 4))
28 return 0;
29
30 return AVPROBE_SCORE_MAX;
31}
32
33static int svag_read_header(AVFormatContext *s)
34{
35 unsigned size, align;
36 AVStream *st;
37
38 avio_skip(s->pb, 4);
39
40 st = avformat_new_stream(s, NULL);
41 if (!st)
42 return AVERROR(ENOMEM);
43
44 size = avio_rl32(s->pb);
45 st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
46 st->codecpar->codec_id = AV_CODEC_ID_ADPCM_PSX;
47 st->codecpar->sample_rate = avio_rl32(s->pb);
48 if (st->codecpar->sample_rate <= 0)
49 return AVERROR_INVALIDDATA;
50 st->codecpar->channels = avio_rl32(s->pb);
51 if (st->codecpar->channels <= 0 || st->codecpar->channels > 8)
52 return AVERROR_INVALIDDATA;
53 st->duration = size / (16 * st->codecpar->channels) * 28;
54 align = avio_rl32(s->pb);
55 if (align <= 0 || align > INT_MAX / st->codecpar->channels)
56 return AVERROR_INVALIDDATA;
57 st->codecpar->block_align = align * st->codecpar->channels;
58 avio_skip(s->pb, 0x800 - avio_tell(s->pb));
59 avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
60
61 return 0;
62}
63
64static int svag_read_packet(AVFormatContext *s, AVPacket *pkt)
65{
66 AVCodecParameters *par = s->streams[0]->codecpar;
67
68 return av_get_packet(s->pb, pkt, par->block_align);
69}

71AVInputFormat ff_svag_demuxer = {
72 .name = "svag",
73 .long_name = NULL_IF_CONFIG_SMALL("Konami PS2 SVAG"),
74 .read_probe = svag_probe,
75 .read_header = svag_read_header,
76 .read_packet = svag_read_packet,
77 .extensions = "svag",
78};
