37struct speex_params {
38 int packet_size;
39 int final_packet_duration;
40 int seq;
41};
42
43static int speex_header(AVFormatContext *s, int idx) {
44 struct ogg *ogg = s->priv_data;
45 struct ogg_stream *os = ogg->streams + idx;
46 struct speex_params *spxp = os->private;
47 AVStream *st = s->streams[idx];
48 uint8_t *p = os->buf + os->pstart;
49
50 if (!spxp) {
51 spxp = av_mallocz(sizeof(*spxp));
52 if (!spxp)
53 return AVERROR(ENOMEM);
54 os->private = spxp;
55 }
56
57 if (spxp->seq > 1)
58 return 0;
59
60 if (spxp->seq == 0) {
61 int frames_per_packet;
62 st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
63 st->codecpar->codec_id = AV_CODEC_ID_SPEEX;
64
65 if (os->psize < 68) {
66 av_log(s, AV_LOG_ERROR, "speex packet too small\n");
67 return AVERROR_INVALIDDATA;
68 }
69
70 st->codecpar->sample_rate = AV_RL32(p + 36);
71 st->codecpar->channels = AV_RL32(p + 48);
72 if (st->codecpar->channels < 1 || st->codecpar->channels > 2) {
73 av_log(s, AV_LOG_ERROR, "invalid channel count. Speex must be mono or stereo.\n");
74 return AVERROR_INVALIDDATA;
75 }
76 st->codecpar->channel_layout = st->codecpar->channels == 1 ? AV_CH_LAYOUT_MONO :
77 AV_CH_LAYOUT_STEREO;
78
79 spxp->packet_size = AV_RL32(p + 56);
80 frames_per_packet = AV_RL32(p + 64);
81 if (frames_per_packet)
82 spxp->packet_size *= frames_per_packet;
83
84 if (ff_alloc_extradata(st->codecpar, os->psize) < 0)
85 return AVERROR(ENOMEM);
86 memcpy(st->codecpar->extradata, p, st->codecpar->extradata_size);
87
88 avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
89 } else
90 ff_vorbis_stream_comment(s, st, p, os->psize);
91
92 spxp->seq++;
93 return 1;
94}

96static int ogg_page_packets(struct ogg_stream *os)
97{
98 int i;
99 int packets = 0;
100 for (i = 0; i < os->nsegs; i++)
101 if (os->segments[i] < 255)
102 packets++;
103 return packets;
104}
105
106static int speex_packet(AVFormatContext *s, int idx)
107{
108 struct ogg *ogg = s->priv_data;
109 struct ogg_stream *os = ogg->streams + idx;
110 struct speex_params *spxp = os->private;
111 int packet_size = spxp->packet_size;
112
113 if (os->flags & OGG_FLAG_EOS && os->lastpts != AV_NOPTS_VALUE &&
114 os->granule > 0) {
115 /* first packet of final page. we have to calculate the final packet
116 duration here because it is the only place we know the next-to-last
117 granule position. */
118 spxp->final_packet_duration = os->granule - os->lastpts -
119 packet_size * (ogg_page_packets(os) - 1);
120 }
121
122 if (!os->lastpts && os->granule > 0)
123 /* first packet */
124 os->lastpts = os->lastdts = os->granule - packet_size *
125 ogg_page_packets(os);
126 if (os->flags & OGG_FLAG_EOS && os->segp == os->nsegs &&
127 spxp->final_packet_duration)
128 /* final packet */
129 os->pduration = spxp->final_packet_duration;
130 else
131 os->pduration = packet_size;
132
133 return 0;
134}

136const struct ogg_codec ff_speex_codec = {

138 .magicsize = 8,
139 .header = speex_header,
140 .packet = speex_packet,
141 .nb_header = 2,
142};
