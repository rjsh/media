41static enum AVCodecID aiff_codec_get_id(int bps)
42{
43 if (bps <= 8)
44 return AV_CODEC_ID_PCM_S8;
45 if (bps <= 16)
46 return AV_CODEC_ID_PCM_S16BE;
47 if (bps <= 24)
48 return AV_CODEC_ID_PCM_S24BE;
49 if (bps <= 32)
50 return AV_CODEC_ID_PCM_S32BE;
51
52 /* bigger than 32 isn't allowed */
53 return AV_CODEC_ID_NONE;
54}

56/* returns the size of the found tag */
57static int get_tag(AVIOContext *pb, uint32_t * tag)
58{
59 int size;
60
61 if (avio_feof(pb))
62 return AVERROR(EIO);
63
64 *tag = avio_rl32(pb);
65 size = avio_rb32(pb);
66
67 if (size < 0)
68 size = 0x7fffffff;
69
70 return size;
71}

73/* Metadata string read */
74static void get_meta(AVFormatContext *s, const char *key, int size)
75{
76 uint8_t *str = av_malloc(size+1);
77
78 if (str) {
79 int res = avio_read(s->pb, str, size);
80 if (res < 0){
81 av_free(str);
82 return;
83 }
84 size += (size&1)-res;
85 str[res] = 0;
86 av_dict_set(&s->metadata, key, str, AV_DICT_DONT_STRDUP_VAL);
87 }else
88 size+= size&1;
89
90 avio_skip(s->pb, size);
91}

93/* Returns the number of sound data frames or negative on error */
94static int get_aiff_header(AVFormatContext *s, int size,
95 unsigned version)
96{
97 AVIOContext *pb = s->pb;
98 AVCodecParameters *par = s->streams[0]->codecpar;
99 AIFFInputContext *aiff = s->priv_data;
100 int exp;
101 uint64_t val;
102 int sample_rate;
103 unsigned int num_frames;
104
105 if (size & 1)
106 size++;
107 par->codec_type = AVMEDIA_TYPE_AUDIO;
108 par->channels = avio_rb16(pb);
109 num_frames = avio_rb32(pb);
110 par->bits_per_coded_sample = avio_rb16(pb);
111
112 exp = avio_rb16(pb) - 16383 - 63;
113 val = avio_rb64(pb);
114 if (exp <-63 || exp >63) {
115 av_log(s, AV_LOG_ERROR, "exp %d is out of range\n", exp);
116 return AVERROR_INVALIDDATA;
117 }
118 if (exp >= 0)
119 sample_rate = val << exp;
120 else
121 sample_rate = (val + (1ULL<<(-exp-1))) >> -exp;
122 par->sample_rate = sample_rate;
123 size -= 18;
124
125 /* get codec id for AIFF-C */
126 if (size < 4) {
127 version = AIFF;
128 } else if (version == AIFF_C_VERSION1) {
129 par->codec_tag = avio_rl32(pb);
130 par->codec_id = ff_codec_get_id(ff_codec_aiff_tags, par->codec_tag);
131 if (par->codec_id == AV_CODEC_ID_NONE) {
132 char tag[32];
133 av_get_codec_tag_string(tag, sizeof(tag), par->codec_tag);
134 avpriv_request_sample(s, "unknown or unsupported codec tag: %s", tag);
135 }
136 size -= 4;
137 }
138
139 if (version != AIFF_C_VERSION1 || par->codec_id == AV_CODEC_ID_PCM_S16BE) {
140 par->codec_id = aiff_codec_get_id(par->bits_per_coded_sample);
141 par->bits_per_coded_sample = av_get_bits_per_sample(par->codec_id);
142 aiff->block_duration = 1;
143 } else {
144 switch (par->codec_id) {
145 case AV_CODEC_ID_PCM_F32BE:
146 case AV_CODEC_ID_PCM_F64BE:
147 case AV_CODEC_ID_PCM_S16LE:
148 case AV_CODEC_ID_PCM_ALAW:
149 case AV_CODEC_ID_PCM_MULAW:
150 aiff->block_duration = 1;
151 break;
152 case AV_CODEC_ID_ADPCM_IMA_QT:
153 par->block_align = 34 * par->channels;
154 break;
155 case AV_CODEC_ID_MACE3:
156 par->block_align = 2 * par->channels;
157 break;
158 case AV_CODEC_ID_ADPCM_G726LE:
159 par->bits_per_coded_sample = 5;
160 case AV_CODEC_ID_ADPCM_IMA_WS:
161 case AV_CODEC_ID_ADPCM_G722:
162 case AV_CODEC_ID_MACE6:
163 case AV_CODEC_ID_SDX2_DPCM:
164 par->block_align = 1 * par->channels;
165 break;
166 case AV_CODEC_ID_GSM:
167 par->block_align = 33;
168 break;
169 default:
170 aiff->block_duration = 1;
171 break;
172 }
173 if (par->block_align > 0)
174 aiff->block_duration = av_get_audio_frame_duration2(par,
175 par->block_align);
176 }
177
178 /* Block align needs to be computed in all cases, as the definition
179 * is specific to applications -> here we use the WAVE format definition */
180 if (!par->block_align)
181 par->block_align = (av_get_bits_per_sample(par->codec_id) * par->channels) >> 3;
182
183 if (aiff->block_duration) {
184 par->bit_rate = par->sample_rate * (par->block_align << 3) /
185 aiff->block_duration;
186 }
187
188 /* Chunk is over */
189 if (size)
190 avio_skip(pb, size);
191
192 return num_frames;
193}

195static int aiff_probe(AVProbeData *p)
196{
197 /* check file header */
198 if (p->buf[0] == 'F' && p->buf[1] == 'O' &&
199 p->buf[2] == 'R' && p->buf[3] == 'M' &&
200 p->buf[8] == 'A' && p->buf[9] == 'I' &&
201 p->buf[10] == 'F' && (p->buf[11] == 'F' || p->buf[11] == 'C'))
202 return AVPROBE_SCORE_MAX;
203 else
204 return 0;
205}

207/* aiff input */
208static int aiff_read_header(AVFormatContext *s)
209{
210 int ret, size, filesize;
211 int64_t offset = 0, position;
212 uint32_t tag;
213 unsigned version = AIFF_C_VERSION1;
214 AVIOContext *pb = s->pb;
215 AVStream * st;
216 AIFFInputContext *aiff = s->priv_data;
217 ID3v2ExtraMeta *id3v2_extra_meta = NULL;
218
219 /* check FORM header */
220 filesize = get_tag(pb, &tag);
221 if (filesize < 0 || tag != MKTAG('F', 'O', 'R', 'M'))
222 return AVERROR_INVALIDDATA;
223
224 /* AIFF data type */
225 tag = avio_rl32(pb);
226 if (tag == MKTAG('A', 'I', 'F', 'F')) /* Got an AIFF file */
227 version = AIFF;
228 else if (tag != MKTAG('A', 'I', 'F', 'C')) /* An AIFF-C file then */
229 return AVERROR_INVALIDDATA;
230
231 filesize -= 4;
232
233 st = avformat_new_stream(s, NULL);
234 if (!st)
235 return AVERROR(ENOMEM);
236
237 while (filesize > 0) {
238 /* parse different chunks */
239 size = get_tag(pb, &tag);
240
241 if (size == AVERROR_EOF && offset > 0 && st->codecpar->block_align) {
242 av_log(s, AV_LOG_WARNING, "header parser hit EOF\n");
243 goto got_sound;
244 }
245 if (size < 0)
246 return size;
247
248 filesize -= size + 8;
249
250 switch (tag) {
251 case MKTAG('C', 'O', 'M', 'M'): /* Common chunk */
252 /* Then for the complete header info */
253 st->nb_frames = get_aiff_header(s, size, version);
254 if (st->nb_frames < 0)
255 return st->nb_frames;
256 if (offset > 0) // COMM is after SSND
257 goto got_sound;
258 break;
259 case MKTAG('I', 'D', '3', ' '):
260 position = avio_tell(pb);
261 ff_id3v2_read(s, ID3v2_DEFAULT_MAGIC, &id3v2_extra_meta, size);
262 if (id3v2_extra_meta)
263 if ((ret = ff_id3v2_parse_apic(s, &id3v2_extra_meta)) < 0) {
264 ff_id3v2_free_extra_meta(&id3v2_extra_meta);
265 return ret;
266 }
267 ff_id3v2_free_extra_meta(&id3v2_extra_meta);
268 if (position + size > avio_tell(pb))
269 avio_skip(pb, position + size - avio_tell(pb));
270 break;
271 case MKTAG('F', 'V', 'E', 'R'): /* Version chunk */
272 version = avio_rb32(pb);
273 break;
274 case MKTAG('N', 'A', 'M', 'E'): /* Sample name chunk */
275 get_meta(s, "title" , size);
276 break;
277 case MKTAG('A', 'U', 'T', 'H'): /* Author chunk */
278 get_meta(s, "author" , size);
279 break;
280 case MKTAG('(', 'c', ')', ' '): /* Copyright chunk */
281 get_meta(s, "copyright", size);
282 break;
283 case MKTAG('A', 'N', 'N', 'O'): /* Annotation chunk */
284 get_meta(s, "comment" , size);
285 break;
286 case MKTAG('S', 'S', 'N', 'D'): /* Sampled sound chunk */
287 aiff->data_end = avio_tell(pb) + size;
288 offset = avio_rb32(pb); /* Offset of sound data */
289 avio_rb32(pb); /* BlockSize... don't care */
290 offset += avio_tell(pb); /* Compute absolute data offset */
291 if (st->codecpar->block_align && !pb->seekable) /* Assume COMM already parsed */
292 goto got_sound;
293 if (!pb->seekable) {
294 av_log(s, AV_LOG_ERROR, "file is not seekable\n");
295 return -1;
296 }
297 avio_skip(pb, size - 8);
298 break;
299 case MKTAG('w', 'a', 'v', 'e'):
300 if ((uint64_t)size > (1<<30))
301 return -1;
302 if (ff_get_extradata(s, st->codecpar, pb, size) < 0)
303 return AVERROR(ENOMEM);
304 if ( (st->codecpar->codec_id == AV_CODEC_ID_QDMC || st->codecpar->codec_id == AV_CODEC_ID_QDM2)
305 && size>=12*4 && !st->codecpar->block_align) {
306 st->codecpar->block_align = AV_RB32(st->codecpar->extradata+11*4);
307 aiff->block_duration = AV_RB32(st->codecpar->extradata+9*4);
308 } else if (st->codecpar->codec_id == AV_CODEC_ID_QCELP) {
309 char rate = 0;
310 if (size >= 25)
311 rate = st->codecpar->extradata[24];
312 switch (rate) {
313 case 'H': // RATE_HALF
314 st->codecpar->block_align = 17;
315 break;
316 case 'F': // RATE_FULL
317 default:
318 st->codecpar->block_align = 35;
319 }
320 aiff->block_duration = 160;
321 st->codecpar->bit_rate = st->codecpar->sample_rate * (st->codecpar->block_align << 3) /
322 aiff->block_duration;
323 }
324 break;
325 case MKTAG('C','H','A','N'):
326 if(ff_mov_read_chan(s, pb, st, size) < 0)
327 return AVERROR_INVALIDDATA;
328 break;
329 case 0:
330 if (offset > 0 && st->codecpar->block_align) // COMM && SSND
331 goto got_sound;
332 default: /* Jump */
333 if (size & 1) /* Always even aligned */
334 size++;
335 avio_skip(pb, size);
336 }
337 }
338
339got_sound:
340 if (!st->codecpar->block_align && st->codecpar->codec_id == AV_CODEC_ID_QCELP) {
341 av_log(s, AV_LOG_WARNING, "qcelp without wave chunk, assuming full rate\n");
342 st->codecpar->block_align = 35;
343 } else if (!st->codecpar->block_align) {
344 av_log(s, AV_LOG_ERROR, "could not find COMM tag or invalid block_align value\n");
345 return -1;
346 }
347
348 /* Now positioned, get the sound data start and end */
349 avpriv_set_pts_info(st, 64, 1, st->codecpar->sample_rate);
350 st->start_time = 0;
351 st->duration = st->nb_frames * aiff->block_duration;
352
353 /* Position the stream at the first block */
354 avio_seek(pb, offset, SEEK_SET);
355
356 return 0;
357}

359#define MAX_SIZE 4096
360
361static int aiff_read_packet(AVFormatContext *s,
362 AVPacket *pkt)
363{
364 AVStream *st = s->streams[0];
365 AIFFInputContext *aiff = s->priv_data;
366 int64_t max_size;
367 int res, size;
368
369 /* calculate size of remaining data */
370 max_size = aiff->data_end - avio_tell(s->pb);
371 if (max_size <= 0)
372 return AVERROR_EOF;
373
374 if (!st->codecpar->block_align) {
375 av_log(s, AV_LOG_ERROR, "block_align not set\n");
376 return AVERROR_INVALIDDATA;
377 }
378
379 /* Now for that packet */
380 switch (st->codecpar->codec_id) {
381 case AV_CODEC_ID_ADPCM_IMA_QT:
382 case AV_CODEC_ID_GSM:
383 case AV_CODEC_ID_QDM2:
384 case AV_CODEC_ID_QCELP:
385 size = st->codecpar->block_align;
386 break;
387 default:
388 size = st->codecpar->block_align ? (MAX_SIZE / st->codecpar->block_align) * st->codecpar->block_align : MAX_SIZE;
389 }
390 size = FFMIN(max_size, size);
391 res = av_get_packet(s->pb, pkt, size);
392 if (res < 0)
393 return res;
394
395 if (size >= st->codecpar->block_align)
396 pkt->flags &= ~AV_PKT_FLAG_CORRUPT;
397 /* Only one stream in an AIFF file */
398 pkt->stream_index = 0;
399 pkt->duration = (res / st->codecpar->block_align) * aiff->block_duration;
400 return 0;
401}
