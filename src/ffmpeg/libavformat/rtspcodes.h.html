125#define RTSP_STATUS_CODE2STRING(x) (\
126x >= 100 && x < FF_ARRAY_ELEMS(rtsp_status_strings) && rtsp_status_strings[x] \
127)? rtsp_status_strings[x] : NULL

129enum RTSPMethod {
130 DESCRIBE,
131 ANNOUNCE,
132 OPTIONS,
133 SETUP,
134 PLAY,
135 PAUSE,
136 TEARDOWN,
137 GET_PARAMETER,
138 SET_PARAMETER,
139 REDIRECT,
140 RECORD,
141 UNKNOWN = -1,
142};

144static inline int ff_rtsp_averror(enum RTSPStatusCode status_code, int default_averror)
145{
146 return ff_http_averror(status_code, default_averror);
147}
