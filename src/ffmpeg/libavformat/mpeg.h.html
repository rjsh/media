64/**
65 * Parse MPEG-PES five-byte timestamp
66 */
67static inline int64_t ff_parse_pes_pts(const uint8_t *buf) {
68 return (int64_t)(*buf & 0x0e) << 29 |
69 (AV_RB16(buf+1) >> 1) << 15 |
70 AV_RB16(buf+3) >> 1;
71}
