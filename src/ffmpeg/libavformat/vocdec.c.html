26static int voc_probe(AVProbeData *p)
27{
28 int version, check;
29
30 if (memcmp(p->buf, ff_voc_magic, sizeof(ff_voc_magic) - 1))
31 return 0;
32 version = AV_RL16(p->buf + 22);
33 check = AV_RL16(p->buf + 24);
34 if (~version + 0x1234 != check)
35 return 10;
36
37 return AVPROBE_SCORE_MAX;
38}

40static int voc_read_header(AVFormatContext *s)
41{
42 VocDecContext *voc = s->priv_data;
43 AVIOContext *pb = s->pb;
44 int header_size;
45
46 avio_skip(pb, 20);
47 header_size = avio_rl16(pb) - 22;
48 if (header_size != 4) {
49 av_log(s, AV_LOG_ERROR, "unknown header size: %d\n", header_size);
50 return AVERROR(ENOSYS);
51 }
52 avio_skip(pb, header_size);
53
54 s->ctx_flags |= AVFMTCTX_NOHEADER;
55
56 voc->remaining_size = 0;
57 return 0;
58}

60static int voc_read_packet(AVFormatContext *s, AVPacket *pkt)
61{
62 if (!s->nb_streams) {
63 AVStream *st = avformat_new_stream(s, NULL);
64 if (!st)
65 return AVERROR(ENOMEM);
66 st->codecpar->codec_type = AVMEDIA_TYPE_AUDIO;
67 }
68 return ff_voc_get_packet(s, pkt, s->streams[0], 0);
69}

71static int voc_read_seek(AVFormatContext *s, int stream_index,
72 int64_t timestamp, int flags)
73{
74 VocDecContext *voc = s->priv_data;
75 AVStream *st;
76 int index;
77
78 if (s->nb_streams < 1) {
79 av_log(s, AV_LOG_ERROR, "cannot seek while no stream was found yet\n");
80 return AVERROR(EINVAL);
81 }
82
83 st = s->streams[stream_index];
84 index = av_index_search_timestamp(st, timestamp, flags);
85
86 if (index >= 0 && index < st->nb_index_entries - 1) {
87 AVIndexEntry *e = &st->index_entries[index];
88 avio_seek(s->pb, e->pos, SEEK_SET);
89 voc->pts = e->timestamp;
90 voc->remaining_size = e->size;
91 return 0;
92 } else if (st->nb_index_entries && st->index_entries[0].timestamp <= timestamp) {
93 AVIndexEntry *e = &st->index_entries[st->nb_index_entries - 1];
94 // prepare context for seek_frame_generic()
95 voc->pts = e->timestamp;
96 voc->remaining_size = e->size;
97 }
98 return -1;
99}
