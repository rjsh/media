48/* AV/C Mask macros */
49#define AVC1394_MASK_START(x) ((x) & 0xF0000000)
50#define AVC1394_MASK_CTYPE(x) ((x) & 0x0F000000)
51#define AVC1394_MASK_RESPONSE(x) ((x) & 0x0F000000)
52#define AVC1394_MASK_SUBUNIT_TYPE(x) ((x) & 0x00F80000)
53#define AVC1394_MASK_SUBUNIT_ID(x) ((x) & 0x00070000)
54#define AVC1394_MASK_OPCODE(x) ((x) & 0x0000FF00)
55#define AVC1394_MASK_OPERAND0(x) ((x) & 0x000000FF)
56#define AVC1394_MASK_OPERAND(x, n) ((x) & (0xFF000000 >> ((((n)-1)%4)*8)))
57#define AVC1394_MASK_RESPONSE_OPERAND(x, n) ((x) & (0xFF000000 >> (((n)%4)*8)))
58
59/* AV/C Mask and shift macros */
60#define AVC1394_GET_CTYPE(x) (((x) & 0x0F000000) >> 24)
61#define AVC1394_GET_RESPONSE(x) (((x) & 0x0F000000) >> 24)
62#define AVC1394_GET_SUBUNIT_TYPE(x) (((x) & 0x00F80000) >> 19)
63#define AVC1394_GET_SUBUNIT_ID(x) (((x) & 0x00070000) >> 16)
64#define AVC1394_GET_OPCODE(x) (((x) & 0x0000FF00) >> 8)
65#define AVC1394_GET_OPERAND0(x) ((x) & 0x000000FF)
66#define AVC1394_GET_OPERAND(x, n) (((x) & (0xFF000000 >> ((((n)-1)%4)*8))) >> ((((n)-1)%4)*8))
67#define AVC1394_GET_RESPONSE_OPERAND(x, n) (((x) & (0xFF000000 >> (((n)%4)*8))) >> (((3-(n))%4)*8))
