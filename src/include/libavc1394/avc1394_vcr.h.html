22#ifndef AVC1394_VCR_H
23#define AVC1394_VCR_H 1
24
25#include <libraw1394/raw1394.h>
26
27#ifdef __cplusplus
28extern "C" {
29#endif
30
31/* ##### Check to see if device is playing ##### */
32int
33avc1394_vcr_is_playing(raw1394handle_t handle, nodeid_t node);
34
35/* ##### Check to see if the device is recording ##### */
36int
37avc1394_vcr_is_recording(raw1394handle_t handle, nodeid_t node);
38
39/* ##### Tell the device to PLAY FORWARD ##### */
40/* issue twice in a row to play forward at the slowest rate */
41void
42avc1394_vcr_play(raw1394handle_t handle, nodeid_t node);
43
44/* ##### Tell the device to PLAY in REVERSE ##### */
45/* issue twice in a row to play in reverse at the slowest rate */
46void
47avc1394_vcr_reverse(raw1394handle_t handle, nodeid_t node);
48
49/* ##### exercise the trick play modes! ##### */
50/* speed is a signed percentage value from -14 to +14 */
51void
52avc1394_vcr_trick_play(raw1394handle_t handle, nodeid_t node, int speed);
53
54/* ##### Tell the device to STOP ##### */
55void
56avc1394_vcr_stop(raw1394handle_t handle, nodeid_t node);
57
58/* ##### Tell the device to REWIND ##### */
59void
60avc1394_vcr_rewind(raw1394handle_t handle, nodeid_t node);
61
62/* ##### Tell the device to PAUSE ##### */
63void
64avc1394_vcr_pause(raw1394handle_t handle, nodeid_t node);
65
66/* ##### Tell the device to FAST FORWARD ##### */
67void
68avc1394_vcr_forward(raw1394handle_t handle, nodeid_t node);
69
70/* ##### Tell the device goto NEXT frame ##### */
71void
72avc1394_vcr_next(raw1394handle_t handle, nodeid_t node);
73
74/* ##### Tell the device goto NEXT index point ##### */
75void
76avc1394_vcr_next_index(raw1394handle_t handle, nodeid_t node);
77
78/* ##### Tell the device goto PREVIOUS frame ##### */
79void
80avc1394_vcr_previous(raw1394handle_t handle, nodeid_t node);
81
82/* ##### Tell the device goto PREVIOUS index point ##### */
83void
84avc1394_vcr_previous_index(raw1394handle_t handle, nodeid_t node);
85
86/* ##### Tell the device to EJECT (usually not supported) ##### */
87void
88avc1394_vcr_eject(raw1394handle_t handle, nodeid_t node);
89
90/* ##### Tell the device to RECORD ##### */
91void
92avc1394_vcr_record(raw1394handle_t handle, nodeid_t node);
93
94/* ##### Ask the device for the status ##### */
95quadlet_t
96avc1394_vcr_status(raw1394handle_t handle, nodeid_t node);
97
98/* Get a textual description of the status */
99char *
100avc1394_vcr_decode_status(quadlet_t response);
101
102/* Get the time code on tape in format HH:MM:SS:FF */
103/* This version allocates memory for the string, and
104 the caller is required to free it. */
105char *
106avc1394_vcr_get_timecode(raw1394handle_t handle, nodeid_t node)
107__attribute__ ((deprecated));
108/* This version requires a pre-allocated output string of at least 12
109 characters. */
110int
111avc1394_vcr_get_timecode2(raw1394handle_t handle, nodeid_t node, char *output);
112
113/* Go to the time code on tape in format HH:MM:SS:FF */
114void
115avc1394_vcr_seek_timecode(raw1394handle_t handle, nodeid_t node, char *timecode);
116
117#ifdef __cplusplus
118}
119#endif
120
121#endif
