84/**
85 * @ref jack_options_t bits
86 */
87enum JackOptions {
88
89 /**
90 * Null value to use when no option bits are needed.
91 */
92 JackNullOption = 0x00,
93
94 /**
95 * Do not automatically start the JACK server when it is not
96 * already running. This option is always selected if
97 * \$JACK_NO_START_SERVER is defined in the calling process
98 * environment.
99 */
100 JackNoStartServer = 0x01,
101
102 /**
103 * Use the exact client name requested. Otherwise, JACK
104 * automatically generates a unique one, if needed.
105 */
106 JackUseExactName = 0x02,
107
108 /**
109 * Open with optional <em>(char *) server_name</em> parameter.
110 */
111 JackServerName = 0x04,
112
113 /**
114 * Load internal client from optional <em>(char *)
115 * load_name</em>. Otherwise use the @a client_name.
116 */
117 JackLoadName = 0x08,
118
119 /**
120 * Pass optional <em>(char *) load_init</em> string to the
121 * jack_initialize() entry point of an internal client.
122 */
123 JackLoadInit = 0x10,
124
125 /**
126 * pass a SessionID Token this allows the sessionmanager to identify the client again.
127 */
128 JackSessionID = 0x20
129};
130
131/** Valid options for opening an external client. */
132#define JackOpenOptions (JackSessionID|JackServerName|JackNoStartServer|JackUseExactName)
133
134/** Valid options for loading an internal client. */
135#define JackLoadOptions (JackLoadInit|JackLoadName|JackUseExactName)

137/**
138 * Options for several JACK operations, formed by OR-ing together the
139 * relevant @ref JackOptions bits.
140 */
141typedef enum JackOptions jack_options_t;

143/**
144 * @ref jack_status_t bits
145 */
146enum JackStatus {
147
148 /**
149 * Overall operation failed.
150 */
151 JackFailure = 0x01,
152
153 /**
154 * The operation contained an invalid or unsupported option.
155 */
156 JackInvalidOption = 0x02,
157
158 /**
159 * The desired client name was not unique. With the @ref
160 * JackUseExactName option this situation is fatal. Otherwise,
161 * the name was modified by appending a dash and a two-digit
162 * number in the range "-01" to "-99". The
163 * jack_get_client_name() function will return the exact string
164 * that was used. If the specified @a client_name plus these
165 * extra characters would be too long, the open fails instead.
166 */
167 JackNameNotUnique = 0x04,
168
169 /**
170 * The JACK server was started as a result of this operation.
171 * Otherwise, it was running already. In either case the caller
172 * is now connected to jackd, so there is no race condition.
173 * When the server shuts down, the client will find out.
174 */
175 JackServerStarted = 0x08,
176
177 /**
178 * Unable to connect to the JACK server.
179 */
180 JackServerFailed = 0x10,
181
182 /**
183 * Communication error with the JACK server.
184 */
185 JackServerError = 0x20,
186
187 /**
188 * Requested client does not exist.
189 */
190 JackNoSuchClient = 0x40,
191
192 /**
193 * Unable to load internal client
194 */
195 JackLoadFailure = 0x80,
196
197 /**
198 * Unable to initialize client
199 */
200 JackInitFailure = 0x100,
201
202 /**
203 * Unable to access shared memory
204 */
205 JackShmFailure = 0x200,
206
207 /**
208 * Client's protocol version does not match
209 */
210 JackVersionError = 0x400,
211
212 /*
213 * BackendError
214 */
215 JackBackendError = 0x800,
216
217 /*
218 * Client is being shutdown against its will
219 */
220 JackClientZombie = 0x1000
221};

333/**
334 * Prototype for the @a bufsize_callback that is invoked whenever the
335 * JACK engine buffer size changes. Although this function is called
336 * in the JACK process thread, the normal process cycle is suspended
337 * during its operation, causing a gap in the audio flow. So, the @a
338 * bufsize_callback can allocate storage, touch memory not previously
339 * referenced, and perform other operations that are not realtime
340 * safe.
341 *
342 * @param nframes buffer size
343 * @param arg pointer supplied by jack_set_buffer_size_callback().
344 *
345 * @return zero on success, non-zero on error
346 */
347typedef int (*JackBufferSizeCallback)(jack_nframes_t nframes, void *arg);

447/**
448 * A port has a set of flags that are formed by OR-ing together the
449 * desired values from the list below. The flags "JackPortIsInput" and
450 * "JackPortIsOutput" are mutually exclusive and it is an error to use
451 * them both.
452 */
453enum JackPortFlags {
454
455 /**
456 * if JackPortIsInput is set, then the port can receive
457 * data.
458 */
459 JackPortIsInput = 0x1,
460
461 /**
462 * if JackPortIsOutput is set, then data can be read from
463 * the port.
464 */
465 JackPortIsOutput = 0x2,
466
467 /**
468 * if JackPortIsPhysical is set, then the port corresponds
469 * to some kind of physical I/O connector.
470 */
471 JackPortIsPhysical = 0x4,
472
473 /**
474 * if JackPortCanMonitor is set, then a call to
475 * jack_port_request_monitor() makes sense.
476 *
477 * Precisely what this means is dependent on the client. A typical
478 * result of it being called with TRUE as the second argument is
479 * that data that would be available from an output port (with
480 * JackPortIsPhysical set) is sent to a physical output connector
481 * as well, so that it can be heard/seen/whatever.
482 *
483 * Clients that do not control physical interfaces
484 * should never create ports with this bit set.
485 */
486 JackPortCanMonitor = 0x8,
487
488 /**
489 * JackPortIsTerminal means:
490 *
491 * for an input port: the data received by the port
492 * will not be passed on or made
493 * available at any other port
494 *
495 * for an output port: the data available at the port
496 * does not originate from any other port
497 *
498 * Audio synthesizers, I/O hardware interface clients, HDR
499 * systems are examples of clients that would set this flag for
500 * their ports.
501 */
502 JackPortIsTerminal = 0x10
503};
