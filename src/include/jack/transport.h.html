47/**
48 * Transport states.

50typedef enum {

53 JackTransportStopped = 0, /**< Transport halted */
54 JackTransportRolling = 1, /**< Transport playing */
55 JackTransportLooping = 2, /**< For OLD_TRANSPORT, now ignored */
56 JackTransportStarting = 3 /**< Waiting for sync ready */

58} jack_transport_state_t;

70 JackAudioVideoRatio = 0x80, /**< audio frames per video frame */
71 JackVideoFrameOffset = 0x100 /**< frame offset of first video frame */

78/**
79 * Struct for transport position information.
80 */
81typedef struct {
82
83 /*@{*/
84 /** @name Server-set fields
85 * these cannot be set from clients; the server sets them */
86 jack_unique_t unique_1; /**< unique ID */
87 jack_time_t usecs; /**< microsecond timestamp that is guaranteed to be
88 monotonic, but not neccessarily
89 linear.
90
91 The absolute value is
92 implementation-dependent (i.e. it
93 could be wall-clock, time since
94 jack started, uptime, etc). */
95 jack_nframes_t frame_rate; /**< current frame rate, in frames per second */
96 /*}@*/
97
98 /*@{*/
99 /** @name Mandatory fields
100 */
101 jack_nframes_t frame; /**< frame number, always present/required.
102
103 This is the frame number on the
104 transport timeline, which is not
105 the same as what @ref
106 jack_frame_time returns. */
107 jack_position_bits_t valid; /**< which other fields are valid, as a
108 bitmask constructed from values in
109 \ref jack_position_bits_t */
110 /*}@*/
111
112 /*@{*/
113 /** @name JackPositionBBT fields
114 * Bar:Beat.Tick-related information.
115 *
116 * Applications that support
117 * JackPositionBBT are encouraged to also fill the JackBBTFrameOffset
118 */
119 int32_t bar; /**< current bar
120
121 Should be >0: the first bar is
122 bar '1'. */
123 int32_t beat; /**< current beat-within-bar
124
125 Should be >0 and <=beats_per_bar:
126 the first beat is beat '1'.
127 */
128 int32_t tick; /**< current tick-within-beat
129
130 Should be >0 and <=ticks_per_beat:
131 the first tick is tick '0'. */
132 double bar_start_tick; /**< number of ticks that have elapsed
133 between frame 0 and the first beat
134 of the current measure. */
135
136 float beats_per_bar; /**< time signature "numerator" */
137 float beat_type; /**< time signature "denominator" */
138 double ticks_per_beat; /**< number of ticks within a bar.
139
140 Usually a moderately large integer
141 with many denominators, such as
142 1920.0 */
143 double beats_per_minute; /**< BPM, quantized to block size. This
144 means when the tempo is not constant
145 within this block, the BPM value should
146 adapted to compensate for this. This
147 is different from most fields in this
148 struct, which specify the value at
149 the beginning of the block rather
150 than an average.*/
151 /*}@*/
152
153 /*@{*/
154 /** @name JackPositionTimecode fields
155 * EXPERIMENTAL: could change */
156 double frame_time; /**< current time in seconds */
157 double next_time; /**< next sequential frame_time
158 (unless repositioned) */
159 /*}@*/
160
161 /*@{*/
162 /* JackBBTFrameOffset fields */
163 jack_nframes_t bbt_offset; /**< frame offset for the BBT fields
164 (the given bar, beat, and tick
165 values actually refer to a time
166 frame_offset frames before the
167 start of the cycle), should
168 be assumed to be 0 if
169 JackBBTFrameOffset is not
170 set. If JackBBTFrameOffset is
171 set and this value is zero, the BBT
172 time refers to the first frame of this
173 cycle. If the value is positive,
174 the BBT time refers to a frame that
175 many frames before the start of the
176 cycle. */
177 /*}@*/
178
179 /*@{*/
180 /* JACK video positional data
181 * EXPERIMENTAL: could change */
182 float audio_frames_per_video_frame; /**< number of audio frames
183 per video frame. Should be assumed
184 zero if JackAudioVideoRatio is not
185 set. If JackAudioVideoRatio is set
186 and the value is zero, no video
187 data exists within the JACK graph */
188
189 jack_nframes_t video_offset; /**< audio frame at which the first video
190 frame in this cycle occurs. Should
191 be assumed to be 0 if JackVideoFrameOffset
192 is not set. If JackVideoFrameOffset is
193 set, but the value is zero, there is
194 no video frame within this cycle. */
195 /*}@*/
196
197 /*@{*/
198 /** @name Other fields */
199 /* For binary compatibility, new fields should be allocated from
200 * this padding area with new valid bits controlling access, so
201 * the existing structure size and offsets are preserved. */
202 int32_t padding[7];
203 /*}@*/
204
205 /* When (unique_1 == unique_2) the contents are consistent. */
206 jack_unique_t unique_2; /**< unique ID */
207
208} POST_PACKED_STRUCTURE jack_position_t;

215/**
216 * Called by the timebase master to release itself from that
217 * responsibility.
218 *
219 * If the timebase master releases the timebase or leaves the JACK
220 * graph for any reason, the JACK engine takes over at the start of
221 * the next process cycle. The transport state does not change. If
222 * rolling, it continues to play, with frame numbers as the only
223 * available position information.
224 *
225 * @see jack_set_timebase_callback
226 *
227 * @param client the JACK client structure.
228 *
229 * @return 0 on success, otherwise a non-zero error code.
230 */
231int jack_release_timebase (jack_client_t *client) JACK_OPTIONAL_WEAK_EXPORT;

233/**
234 * Prototype for the @a sync_callback defined by @ref slowsyncclients
235 * "slow-sync clients". When the client is active, this callback is
236 * invoked just before process() in the same thread. This occurs once
237 * after registration, then subsequently whenever some client requests
238 * a new position, or the transport enters the ::JackTransportStarting
239 * state. This realtime function must not wait.
240 *
241 * The transport @a state will be:
242 *
243 * - ::JackTransportStopped when a new position is requested;
244 * - ::JackTransportStarting when the transport is waiting to start;
245 * - ::JackTransportRolling when the timeout has expired, and the
246 * position is now a moving target.
247 *
248 * @param state current transport state.
249 * @param pos new transport position.
250 * @param arg the argument supplied by jack_set_sync_callback().
251 *
252 * @return TRUE (non-zero) when ready to roll.
253 */
254typedef int (*JackSyncCallback)(jack_transport_state_t state,
255 jack_position_t *pos,
256 void *arg);
257
258/**
259 * Register (or unregister) as a @ref slowsyncclients "slow-sync client", that cannot
260 * respond immediately to transport position changes.
261 *
262 * The @a sync_callback will be invoked at the first available
263 * opportunity after its registration is complete. If the client is
264 * currently active this will be the following process cycle,
265 * otherwise it will be the first cycle after calling jack_activate().
266 * After that, it runs according to the ::JackSyncCallback rules.
267 * Clients that don't set a @a sync_callback are assumed to be ready
268 * immediately any time the transport wants to start.
269 *
270 * @param client the JACK client structure.
271 * @param sync_callback is a realtime function that returns TRUE when
272 * the client is ready. Setting @a sync_callback to NULL declares that
273 * this client no longer requires slow-sync processing.
274 * @param arg an argument for the @a sync_callback function.
275 *
276 * @return 0 on success, otherwise a non-zero error code.
277 */
278int jack_set_sync_callback (jack_client_t *client,
279 JackSyncCallback sync_callback,
280 void *arg) JACK_OPTIONAL_WEAK_EXPORT;
281
282/**
283 * Set the timeout value for @ref slowsyncclients "slow-sync clients".
284 *
285 * This timeout prevents unresponsive slow-sync clients from
286 * completely halting the transport mechanism. The default is two
287 * seconds. When the timeout expires, the transport starts rolling,
288 * even if some slow-sync clients are still unready. The @a
289 * sync_callbacks of these clients continue being invoked, giving them
290 * a chance to catch up.
291 *
292 * @see jack_set_sync_callback
293 *
294 * @param client the JACK client structure.
295 * @param timeout is delay (in microseconds) before the timeout expires.
296 *
297 * @return 0 on success, otherwise a non-zero error code.
298 */
299int jack_set_sync_timeout (jack_client_t *client,
300 jack_time_t timeout) JACK_OPTIONAL_WEAK_EXPORT;
301
302/**
303 * Prototype for the @a timebase_callback used to provide extended
304 * position information. Its output affects all of the following
305 * process cycle. This realtime function must not wait.
306 *
307 * This function is called immediately after process() in the same
308 * thread whenever the transport is rolling, or when any client has
309 * requested a new position in the previous cycle. The first cycle
310 * after jack_set_timebase_callback() is also treated as a new
311 * position, or the first cycle after jack_activate() if the client
312 * had been inactive.
313 *
314 * The timebase master may not use its @a pos argument to set @a
315 * pos->frame. To change position, use jack_transport_reposition() or
316 * jack_transport_locate(). These functions are realtime-safe, the @a
317 * timebase_callback can call them directly.
318 *
319 * @param state current transport state.
320 * @param nframes number of frames in current period.
321 * @param pos address of the position structure for the next cycle; @a
322 * pos->frame will be its frame number. If @a new_pos is FALSE, this
323 * structure contains extended position information from the current
324 * cycle. If TRUE, it contains whatever was set by the requester.
325 * The @a timebase_callback's task is to update the extended
326 * information here.
327 * @param new_pos TRUE (non-zero) for a newly requested @a pos, or for
328 * the first cycle after the @a timebase_callback is defined.
329 * @param arg the argument supplied by jack_set_timebase_callback().
330 */
331typedef void (*JackTimebaseCallback)(jack_transport_state_t state,
332 jack_nframes_t nframes,
333 jack_position_t *pos,
334 int new_pos,
335 void *arg);
336
337/**
338 * Register as timebase master for the JACK subsystem.
339 *
340 * The timebase master registers a callback that updates extended
341 * position information such as beats or timecode whenever necessary.
342 * Without this extended information, there is no need for this
343 * function.
344 *
345 * There is never more than one master at a time. When a new client
346 * takes over, the former @a timebase_callback is no longer called.
347 * Taking over the timebase may be done conditionally, so it fails if
348 * there was a master already.
349 *
350 * The method may be called whether the client has been activated or not.
351 *
352 * @param client the JACK client structure.
353 * @param conditional non-zero for a conditional request.
354 * @param timebase_callback is a realtime function that returns
355 * position information.
356 * @param arg an argument for the @a timebase_callback function.
357 *
358 * @return
359 * - 0 on success;
360 * - EBUSY if a conditional request fails because there was already a
361 * timebase master;
362 * - other non-zero error code.
363 */
364int jack_set_timebase_callback (jack_client_t *client,
365 int conditional,
366 JackTimebaseCallback timebase_callback,
367 void *arg) JACK_OPTIONAL_WEAK_EXPORT;
368
369/**
370 * Reposition the transport to a new frame number.
371 *
372 * May be called at any time by any client. The new position takes
373 * effect in two process cycles. If there are @ref slowsyncclients
374 * "slow-sync clients" and the transport is already rolling, it will
375 * enter the ::JackTransportStarting state and begin invoking their @a
376 * sync_callbacks until ready. This function is realtime-safe.
377 *
378 * @see jack_transport_reposition, jack_set_sync_callback
379 *
380 * @param client the JACK client structure.
381 * @param frame frame number of new transport position.
382 *
383 * @return 0 if valid request, non-zero otherwise.
384 */
385int jack_transport_locate (jack_client_t *client,
386 jack_nframes_t frame) JACK_OPTIONAL_WEAK_EXPORT;
387
388/**
389 * Query the current transport state and position.
390 *
391 * This function is realtime-safe, and can be called from any thread.
392 * If called from the process thread, @a pos corresponds to the first
393 * frame of the current cycle and the state returned is valid for the
394 * entire cycle.
395 *
396 * @param client the JACK client structure.
397 * @param pos pointer to structure for returning current transport
398 * position; @a pos->valid will show which fields contain valid data.
399 * If @a pos is NULL, do not return position information.
400 *
401 * @return Current transport state.
402 */
403jack_transport_state_t jack_transport_query (const jack_client_t *client,
404 jack_position_t *pos) JACK_OPTIONAL_WEAK_EXPORT;

406/**
407 * Return an estimate of the current transport frame,
408 * including any time elapsed since the last transport

413jack_nframes_t jack_get_current_transport_frame (const jack_client_t *client) JACK_OPTIONAL_WEAK_EXPORT;
414
415/**
416 * Request a new transport position.
417 *
418 * May be called at any time by any client. The new position takes
419 * effect in two process cycles. If there are @ref slowsyncclients
420 * "slow-sync clients" and the transport is already rolling, it will
421 * enter the ::JackTransportStarting state and begin invoking their @a
422 * sync_callbacks until ready. This function is realtime-safe.
423 *
424 * @see jack_transport_locate, jack_set_sync_callback
425 *
426 * @param client the JACK client structure.
427 * @param pos requested new transport position. Fill pos->valid to specify
428 * which fields should be taken into account. If you mark a set of fields
429 * as valid, you are expected to fill them all.
430 *
431 * @return 0 if valid request, EINVAL if position structure rejected.
432 */
433int jack_transport_reposition (jack_client_t *client,
434 const jack_position_t *pos) JACK_OPTIONAL_WEAK_EXPORT;
435
436/**
437 * Start the JACK transport rolling.
438 *
439 * Any client can make this request at any time. It takes effect no
440 * sooner than the next process cycle, perhaps later if there are
441 * @ref slowsyncclients "slow-sync clients". This function is realtime-safe.
442 *
443 * @see jack_set_sync_callback
444 *
445 * @param client the JACK client structure.
446 */
447void jack_transport_start (jack_client_t *client) JACK_OPTIONAL_WEAK_EXPORT;
448
449/**
450 * Stop the JACK transport.
451 *
452 * Any client can make this request at any time. It takes effect on
453 * the next process cycle. This function is realtime-safe.
454 *
455 * @param client the JACK client structure.
456 */
457void jack_transport_stop (jack_client_t *client) JACK_OPTIONAL_WEAK_EXPORT;
