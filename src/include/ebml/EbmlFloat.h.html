36#ifndef LIBEBML_FLOAT_H
37#define LIBEBML_FLOAT_H
38
39#include "EbmlTypes.h"
40#include "EbmlElement.h"
41
42START_LIBEBML_NAMESPACE
43
44/*!
45 \class EbmlFloat
46 \brief Handle all operations on a float EBML element
47*/
48class EBML_DLL_API EbmlFloat : public EbmlElement {
49 public:
50 enum Precision {
51 FLOAT_32
52 ,FLOAT_64
53 };
54
55 EbmlFloat(const Precision prec = FLOAT_32);
56 EbmlFloat(const double DefaultValue, const Precision prec = FLOAT_32);
57 EbmlFloat(const EbmlFloat & ElementToClone);
58
59 virtual bool ValidateSize() const
60 {
61 return (GetSize() == 4 || GetSize() == 8);
62 }
63
64 filepos_t RenderData(IOCallback & output, bool bForceRender, bool bWithDefault = false);
65 filepos_t ReadData(IOCallback & input, ScopeMode ReadFully = SCOPE_ALL_DATA);
66 filepos_t UpdateSize(bool bWithDefault = false, bool bForceRender = false);
67
68 void SetPrecision(const EbmlFloat::Precision prec = FLOAT_32)
69 {
70 if (prec == FLOAT_64)
71 SetSize_(8);
72 else
73 SetSize_(4); // default size
74 }
75
76
77// EbmlFloat & operator=(const float NewValue) { Value = NewValue; return *this;}
78 EbmlFloat & operator=(const double NewValue) { Value = NewValue; SetValueIsSet(); return *this;}
79
80 virtual bool IsSmallerThan(const EbmlElement *Cmp) const;
81
82 operator float() const;
83 operator double() const;
84
85 EbmlFloat &SetValue(double NewValue);
86 double GetValue() const;
87
88 void SetDefaultValue(double);
89
90 double DefaultVal() const;
91
92 bool IsDefaultValue() const {
93 return (DefaultISset() && Value == DefaultValue);
94 }
95
96#if defined(EBML_STRICT_API)
97 private:
98#else
99 protected:
100#endif
101 double Value; /// The actual value of the element
102 double DefaultValue;
103};
104
105END_LIBEBML_NAMESPACE
106
107#endif // LIBEBML_FLOAT_H
