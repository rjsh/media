55/*!
56 \class EbmlMaster
57 \brief Handle all operations on an EBML element that contains other EBML elements
58*/
59class EBML_DLL_API EbmlMaster : public EbmlElement {
60 public:
61 EbmlMaster(const EbmlSemanticContext & aContext, bool bSizeIsKnown = true);
62 EbmlMaster(const EbmlMaster & ElementToClone);
63 virtual bool ValidateSize() const {return true;}
64 /*!
65 \warning be carefull to clear the memory allocated in the ElementList elsewhere
66 */
67 virtual ~EbmlMaster();
68
69 filepos_t RenderData(IOCallback & output, bool bForceRender, bool bWithDefault = false);
70 filepos_t ReadData(IOCallback & input, ScopeMode ReadFully);
71 filepos_t UpdateSize(bool bWithDefault = false, bool bForceRender = false);
72
73 /*!
74 \brief Set wether the size is finite (size is known in advance when writing, or infinite size is not known on writing)
75 */
76 bool SetSizeInfinite(bool aIsInfinite = true) {SetSizeIsFinite(!aIsInfinite); return true;}
77
78 bool PushElement(EbmlElement & element);
79 uint64 GetSize() const {
80 if (IsFiniteSize())
81 return EbmlElement::GetSize();
82 else
83 return (0-1);
84 }
85
86 uint64 GetDataStart() const {
87 return GetElementPosition() + EBML_ID_LENGTH((const EbmlId&)*this) + CodedSizeLength(GetSize(), GetSizeLength(), IsFiniteSize());
88 }
89
90 /*!
91 \brief find the element corresponding to the ID of the element, NULL if not found
92 */
93 EbmlElement *FindElt(const EbmlCallbacks & Callbacks) const;
94 /*!
95 \brief find the first element corresponding to the ID of the element
96 */
97 EbmlElement *FindFirstElt(const EbmlCallbacks & Callbacks, bool bCreateIfNull);
98 EbmlElement *FindFirstElt(const EbmlCallbacks & Callbacks) const;
99
100 /*!
101 \brief find the element of the same type of PasElt following in the list of elements
102 */
103 EbmlElement *FindNextElt(const EbmlElement & PastElt, bool bCreateIfNull);
104 EbmlElement *FindNextElt(const EbmlElement & PastElt) const;
105 EbmlElement *AddNewElt(const EbmlCallbacks & Callbacks);
106
107 /*!
108 \brief add an element at a specified location
109 */
110 bool InsertElement(EbmlElement & element, size_t position = 0);
111 bool InsertElement(EbmlElement & element, const EbmlElement & before);
112
113 /*!
114 \brief Read the data and keep the known children
115 */
116 void Read(EbmlStream & inDataStream, const EbmlSemanticContext & Context, int & UpperEltFound, EbmlElement * & FoundElt, bool AllowDummyElt, ScopeMode ReadFully = SCOPE_ALL_DATA);
117
118 /*!
119 \brief sort Data when they can
120 */
121 void Sort();
122
123 size_t ListSize() const {return ElementList.size();}
124 std::vector<EbmlElement *> const &GetElementList() const {return ElementList;}
125 std::vector<EbmlElement *> &GetElementList() {return ElementList;}
126
127 inline EBML_MASTER_ITERATOR begin() {return ElementList.begin();}
128 inline EBML_MASTER_ITERATOR end() {return ElementList.end();}
129 inline EBML_MASTER_RITERATOR rbegin() {return ElementList.rbegin();}
130 inline EBML_MASTER_RITERATOR rend() {return ElementList.rend();}
131 inline EBML_MASTER_CONST_ITERATOR begin() const {return ElementList.begin();}
132 inline EBML_MASTER_CONST_ITERATOR end() const {return ElementList.end();}
133 inline EBML_MASTER_CONST_RITERATOR rbegin() const {return ElementList.rbegin();}
134 inline EBML_MASTER_CONST_RITERATOR rend() const {return ElementList.rend();}
135
136 EbmlElement * operator[](unsigned int position) {return ElementList[position];}
137 const EbmlElement * operator[](unsigned int position) const {return ElementList[position];}
138
139 bool IsDefaultValue() const {
140 return (ElementList.size() == 0);
141 }
142 virtual bool IsMaster() const {return true;}
143
144 /*!
145 \brief verify that all mandatory elements are present
146 \note usefull after reading or before writing
147 */
148 bool CheckMandatory() const;
149
150 /*!
151 \brief Remove an element from the list of the master
152 */
153 void Remove(size_t Index);
154 void Remove(EBML_MASTER_ITERATOR & Itr);
155 void Remove(EBML_MASTER_RITERATOR & Itr);
156
157 /*!
158 \brief remove all elements, even the mandatory ones
159 */
160 void RemoveAll() {ElementList.clear();}
161
162 /*!
163 \brief facility for Master elements to write only the head and force the size later
164 \warning
165 */
166 filepos_t WriteHead(IOCallback & output, int SizeLength, bool bWithDefault = false);
167
168 void EnableChecksum(bool bIsEnabled = true) { bChecksumUsed = bIsEnabled; }
169 bool HasChecksum() const {return bChecksumUsed;}
170 bool VerifyChecksum() const;
171 uint32 GetCrc32() const {return Checksum.GetCrc32();}
172 void ForceChecksum(uint32 NewChecksum) {
173 Checksum.ForceCrc32(NewChecksum);
174 bChecksumUsed = true;
175 }
176
177 /*!
178 \brief drill down all sub-elements, finding any missing elements
179 */
180 std::vector<std::string> FindAllMissingElements();
181
182#if defined(EBML_STRICT_API)
183 private:
184#else
185 protected:
186#endif
187 std::vector<EbmlElement *> ElementList;
188
189 const EbmlSemanticContext & Context;
190
191 bool bChecksumUsed;
192 EbmlCrc32 Checksum;
193
194 private:
195 /*!
196 \brief Add all the mandatory elements to the list
197 */
198 bool ProcessMandatory();
199};

202template <typename Type>
203Type & GetChild(EbmlMaster & Master)
204{
205 return *(static_cast<Type *>(Master.FindFirstElt(EBML_INFO(Type), true)));
206}

208// MyDocType = GetChild<EDocType>(TestHead);
209
210template <typename Type>
211Type * FindChild(EbmlMaster & Master)
212{
213 return static_cast<Type *>(Master.FindFirstElt(EBML_INFO(Type), false));
214}
215
216template <typename Type>
217Type & GetNextChild(EbmlMaster & Master, const Type & PastElt)
218{
219 return *(static_cast<Type *>(Master.FindNextElt(PastElt, true)));
220}
221
222template <typename Type>
223Type & AddNewChild(EbmlMaster & Master)
224{
225 return *(static_cast<Type *>(Master.AddNewElt(EBML_INFO(Type))));
226}
