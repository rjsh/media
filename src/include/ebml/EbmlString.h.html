36#ifndef LIBEBML_STRING_H
37#define LIBEBML_STRING_H
38
39#include <string>
40
41#include "EbmlTypes.h"
42#include "EbmlElement.h"
43
44START_LIBEBML_NAMESPACE
45
46/*!
47 \class EbmlString
48 \brief Handle all operations on a printable string EBML element
49*/
50class EBML_DLL_API EbmlString : public EbmlElement {
51 public:
52 EbmlString();
53 EbmlString(const std::string & aDefaultValue);
54 EbmlString(const EbmlString & ElementToClone);
55
56 virtual ~EbmlString() {}
57
58 virtual bool ValidateSize() const {return IsFiniteSize();} // any size is possible
59 filepos_t RenderData(IOCallback & output, bool bForceRender, bool bWithDefault = false);
60 filepos_t ReadData(IOCallback & input, ScopeMode ReadFully = SCOPE_ALL_DATA);
61 filepos_t UpdateSize(bool bWithDefault = false, bool bForceRender = false);
62
63 EbmlString & operator=(const std::string &);
64 operator const std::string &() const;
65
66 EbmlString &SetValue(std::string const &NewValue);
67 std::string GetValue() const;
68
69 void SetDefaultValue(std::string &);
70
71 const std::string & DefaultVal() const;
72
73 bool IsDefaultValue() const {
74 return (DefaultISset() && Value == DefaultValue);
75 }
76
77#if defined(EBML_STRICT_API)
78 private:
79#else
80 protected:
81#endif
82 std::string Value; /// The actual value of the element
83 std::string DefaultValue;
84};
85
86END_LIBEBML_NAMESPACE
87
88#endif // LIBEBML_STRING_H
