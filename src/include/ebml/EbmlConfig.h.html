36#ifndef LIBEBML_CONFIG_H
37#define LIBEBML_CONFIG_H
38
39#if defined(__linux__)
40#include <endian.h>
41#if __BYTE_ORDER == __LITTLE_ENDIAN
42#undef WORDS_BIGENDIAN
43#elif __BYTE_ORDER == __BIG_ENDIAN
44#define WORDS_BIGENDIAN 1
45#endif
46#else
47// automatic endianess detection working on GCC
48#if !defined(WORDS_BIGENDIAN)
49#if (defined (__arm__) && ! defined (__ARMEB__)) || defined (__i386__) || defined (__i860__) || defined (__ns32000__) || defined (__vax__) || defined (__amd64__) || defined (__x86_64__)
50#undef WORDS_BIGENDIAN
51#elif defined (__sparc__) || defined (__alpha__) || defined (__PPC__) || defined (__mips__) || defined (__ppc__) || defined (__BIG_ENDIAN__)
52#define WORDS_BIGENDIAN 1
53#else
54// not automatically detected, put it yourself
55#undef WORDS_BIGENDIAN // for my testing platform (x86)
56#endif
57#endif // not autoconf
58#endif
59
60#define LIBEBML_NAMESPACE libebml
61#if defined(NO_NAMESPACE) // for older GCC
62# define START_LIBEBML_NAMESPACE
63# define END_LIBEBML_NAMESPACE
64#else // NO_NAMESPACE
65# define START_LIBEBML_NAMESPACE namespace LIBEBML_NAMESPACE {
66# define END_LIBEBML_NAMESPACE };
67#endif // NO_NAMESPACE
68
69
70// There are special implementations for certain platforms. For example on Windows
71// we use the Win32 file API. here we set the appropriate macros.
72#if defined(_WIN32)||defined(WIN32)
73
74# if defined(EBML_DLL)
75# if defined(EBML_DLL_EXPORT)
76# define EBML_DLL_API __declspec(dllexport)
77# else // EBML_DLL_EXPORT
78# define EBML_DLL_API __declspec(dllimport)
79# endif // EBML_DLL_EXPORT
80# else // EBML_DLL
81# define EBML_DLL_API
82# endif // EBML_DLL
83
84# ifdef _MSC_VER
85# pragma warning(disable:4786) // length of internal identifiers
86# endif // _MSC_VER
87#else
88# define EBML_DLL_API
89#endif // WIN32 || _WIN32
90
91
92#ifndef countof
93#define countof(x) (sizeof(x)/sizeof(x[0]))
94#endif
95
96
97// The LIBEBML_DEBUG symbol is defined, when we are creating a debug build. In this
98// case the debug logging code is compiled in.
99#if (defined(DEBUG)||defined(_DEBUG)) && !defined(LIBEBML_DEBUG)
100#define LIBEBML_DEBUG
101#endif
102
103// For compilers that don't define __TIMESTAMP__ (e.g. gcc 2.95, gcc 3.2)
104#ifndef __TIMESTAMP__
105#define __TIMESTAMP__ __DATE__ " " __TIME__
106#endif
107
108#ifdef __GNUC__
109#define EBML_PRETTYLONGINT(c) (c ## ll)
110#else // __GNUC__
111#define EBML_PRETTYLONGINT(c) (c)
112#endif // __GNUC__
113
114#if defined(__BORLANDC__) && __BORLANDC__ >= 0x0581 //Borland C++ Builder 2006 preview
115 #include <stdlib.h> //malloc(), free()
116 #include <memory.h> //memcpy()
117#endif //__BORLANDC__
118
119#endif // LIBEBML_CONFIG_H
