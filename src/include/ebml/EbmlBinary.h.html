53/*!
54 \class EbmlBinary
55 \brief Handle all operations on an EBML element that contains "unknown" binary data
56
57 \todo handle fix sized elements (like UID of CodecID)
58*/
59class EBML_DLL_API EbmlBinary : public EbmlElement {
60 public:
61 EbmlBinary();
62 EbmlBinary(const EbmlBinary & ElementToClone);
63 virtual ~EbmlBinary(void);
64
65 virtual bool ValidateSize() const {return IsFiniteSize() && GetSize() < 0x7FFFFFFF;} // we don't mind about what's inside
66
67 filepos_t RenderData(IOCallback & output, bool bForceRender, bool bWithDefault = false);
68 filepos_t ReadData(IOCallback & input, ScopeMode ReadFully = SCOPE_ALL_DATA);
69 filepos_t UpdateSize(bool bWithDefault = false, bool bForceRender = false);
70
71 void SetBuffer(const binary *Buffer, const uint32 BufferSize) {
72 Data = (binary *) Buffer;
73 SetSize_(BufferSize);
74 SetValueIsSet();
75 }
76
77 binary *GetBuffer() const {return Data;}
78
79 void CopyBuffer(const binary *Buffer, const uint32 BufferSize) {
80 if (Data != NULL)
81 free(Data);
82 Data = (binary *)malloc(BufferSize * sizeof(binary));
83 memcpy(Data, Buffer, BufferSize);
84 SetSize_(BufferSize);
85 SetValueIsSet();
86 }
87
88 operator const binary &() const;
89
90 bool IsDefaultValue() const {
91 return false;
92 }
93
94 bool operator==(const EbmlBinary & ElementToCompare) const;
95
96#if defined(EBML_STRICT_API)
97 private:
98#else
99 protected:
100#endif
101 binary *Data; // the binary data inside the element
102};
