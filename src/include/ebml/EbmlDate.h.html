34#ifndef LIBEBML_DATE_H
35#define LIBEBML_DATE_H
36
37#include "EbmlTypes.h"
38#include "EbmlElement.h"
39
40START_LIBEBML_NAMESPACE
41
42/*!
43 \class EbmlDate
44 \brief Handle all operations related to an EBML date
45*/
46class EBML_DLL_API EbmlDate : public EbmlElement {
47 public:
48 EbmlDate() :EbmlElement(8, false), myDate(0) {}
49 EbmlDate(const EbmlDate & ElementToClone);
50
51 /*!
52 \brief set the date with a UNIX/C/EPOCH form
53 \param NewDate UNIX/C date in UTC (no timezone)
54 */
55 void SetEpochDate(int64 NewDate) {myDate = (NewDate - UnixEpochDelay) * 1000000000; SetValueIsSet();}
56 EbmlDate &SetValue(int64 NewValue) {SetEpochDate(NewValue); return *this;}
57
58 /*!
59 \brief get the date with a UNIX/C/EPOCH form
60 \note the date is in UTC (no timezone)
61 */
62 int64 GetEpochDate() const {return int64(myDate/1000000000 + UnixEpochDelay);}
63 int64 GetValue() const {return GetEpochDate();}
64
65 virtual bool ValidateSize() const {return IsFiniteSize() && ((GetSize() == 8) || (GetSize() == 0));}
66
67 /*!
68 \note no Default date handled
69 */
70 filepos_t UpdateSize(bool /* bWithDefault = false */, bool /* bForceRender = false */) {
71 if(!ValueIsSet())
72 SetSize_(0);
73 else
74 SetSize_(8);
75 return GetSize();
76 }
77
78 virtual bool IsSmallerThan(const EbmlElement *Cmp) const;
79
80 filepos_t ReadData(IOCallback & input, ScopeMode ReadFully = SCOPE_ALL_DATA);
81
82 bool IsDefaultValue() const {
83 return false;
84 }
85
86#if defined(EBML_STRICT_API)
87 private:
88#else
89 protected:
90#endif
91 filepos_t RenderData(IOCallback & output, bool bForceRender, bool bWithDefault = false);
92
93 int64 myDate; ///< internal format of the date
94
95 static const uint64 UnixEpochDelay;
96};
97
98END_LIBEBML_NAMESPACE
99
100#endif // LIBEBML_DATE_H
