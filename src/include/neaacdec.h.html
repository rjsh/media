78/* object types for AAC */
79#define MAIN 1
80#define LC 2
81#define SSR 3
82#define LTP 4
83#define HE_AAC 5
84#define ER_LC 17
85#define ER_LTP 19
86#define LD 23
87#define DRM_ER_LC 27 /* special object type for DRM */
88
89/* header types */
90#define RAW 0
91#define ADIF 1
92#define ADTS 2
93#define LATM 3
94
95/* SBR signalling */
96#define NO_SBR 0
97#define SBR_UPSAMPLED 1
98#define SBR_DOWNSAMPLED 2
99#define NO_SBR_UPSAMPLED 3
100
101/* library output formats */
102#define FAAD_FMT_16BIT 1
103#define FAAD_FMT_24BIT 2
104#define FAAD_FMT_32BIT 3
105#define FAAD_FMT_FLOAT 4
106#define FAAD_FMT_FIXED FAAD_FMT_FLOAT
107#define FAAD_FMT_DOUBLE 5
108
109/* Capabilities */
110#define LC_DEC_CAP (1<<0) /* Can decode LC */
111#define MAIN_DEC_CAP (1<<1) /* Can decode MAIN */
112#define LTP_DEC_CAP (1<<2) /* Can decode LTP */
113#define LD_DEC_CAP (1<<3) /* Can decode LD */
114#define ERROR_RESILIENCE_CAP (1<<4) /* Can decode ER */
115#define FIXED_POINT_CAP (1<<5) /* Fixed point */
116
117/* Channel definitions */
118#define FRONT_CHANNEL_CENTER (1)
119#define FRONT_CHANNEL_LEFT (2)
120#define FRONT_CHANNEL_RIGHT (3)
121#define SIDE_CHANNEL_LEFT (4)
122#define SIDE_CHANNEL_RIGHT (5)
123#define BACK_CHANNEL_LEFT (6)
124#define BACK_CHANNEL_RIGHT (7)
125#define BACK_CHANNEL_CENTER (8)
126#define LFE_CHANNEL (9)
127#define UNKNOWN_CHANNEL (0)
128
129/* DRM channel definitions */
130#define DRMCH_MONO 1
131#define DRMCH_STEREO 2
132#define DRMCH_SBR_MONO 3
133#define DRMCH_SBR_STEREO 4
134#define DRMCH_SBR_PS_STEREO 5

137/* A decode call can eat up to FAAD_MIN_STREAMSIZE bytes per decoded channel,
138 so at least so much bytes per channel should be available in this stream */
139#define FAAD_MIN_STREAMSIZE 768 /* 6144 bits/channel */

144typedef struct mp4AudioSpecificConfig
145{
146 /* Audio Specific Info */
147 unsigned char objectTypeIndex;
148 unsigned char samplingFrequencyIndex;
149 unsigned long samplingFrequency;
150 unsigned char channelsConfiguration;
151
152 /* GA Specific Info */
153 unsigned char frameLengthFlag;
154 unsigned char dependsOnCoreCoder;
155 unsigned short coreCoderDelay;
156 unsigned char extensionFlag;
157 unsigned char aacSectionDataResilienceFlag;
158 unsigned char aacScalefactorDataResilienceFlag;
159 unsigned char aacSpectralDataResilienceFlag;
160 unsigned char epConfig;
161
162 char sbr_present_flag;
163 char forceUpSampling;
164 char downSampledSBR;
165} mp4AudioSpecificConfig;

167typedef struct NeAACDecConfiguration
168{
169 unsigned char defObjectType;
170 unsigned long defSampleRate;
171 unsigned char outputFormat;
172 unsigned char downMatrix;
173 unsigned char useOldADTSFormat;
174 unsigned char dontUpSampleImplicitSBR;
175} NeAACDecConfiguration, *NeAACDecConfigurationPtr;

177typedef struct NeAACDecFrameInfo
178{
179 unsigned long bytesconsumed;
180 unsigned long samples;
181 unsigned char channels;
182 unsigned char error;
183 unsigned long samplerate;
184
185 /* SBR: 0: off, 1: on; upsample, 2: on; downsampled, 3: off; upsampled */
186 unsigned char sbr;
187
188 /* MPEG-4 ObjectType */
189 unsigned char object_type;
190
191 /* AAC header type; MP4 will be signalled as RAW also */
192 unsigned char header_type;
193
194 /* multichannel configuration */
195 unsigned char num_front_channels;
196 unsigned char num_side_channels;
197 unsigned char num_back_channels;
198 unsigned char num_lfe_channels;
199 unsigned char channel_position[64];
200
201 /* PS: 0: off, 1: on */
202 unsigned char ps;
203} NeAACDecFrameInfo;
