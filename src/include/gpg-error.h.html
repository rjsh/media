48/* The GnuPG project consists of many components. Error codes are
49 exchanged between all components. The common error codes and their
50 user-presentable descriptions are kept into a shared library to
51 allow adding new error codes and components without recompiling any
52 of the other components. The interface will not change in a
53 backward incompatible way.
54
55 An error code together with an error source build up an error
56 value. As the error value is been passed from one component to
57 another, it preserver the information about the source and nature
58 of the error.
59
60 A component of the GnuPG project can define the following macros to
61 tune the behaviour of the library:
62
63 GPG_ERR_SOURCE_DEFAULT: Define to an error source of type
64 gpg_err_source_t to make that source the default for gpg_error().
65 Otherwise GPG_ERR_SOURCE_UNKNOWN is used as default.
66
67 GPG_ERR_ENABLE_GETTEXT_MACROS: Define to provide macros to map the
68 internal gettext API to standard names. This has only an effect on
69 Windows platforms. */
70

521/* The error value type gpg_error_t. */
522
523/* We would really like to use bit-fields in a struct, but using
524 structs as return values can cause binary compatibility issues, in
525 particular if you want to do it effeciently (also see
526 -freg-struct-return option to GCC). */
527typedef unsigned int gpg_error_t;
528
529/* We use the lowest 16 bits of gpg_error_t for error codes. The 16th
530 bit indicates system errors. */
531#define GPG_ERR_CODE_MASK (GPG_ERR_CODE_DIM - 1)
532
533/* Bits 17 to 24 are reserved. */
534
535/* We use the upper 7 bits of gpg_error_t for error sources. */
536#define GPG_ERR_SOURCE_MASK (GPG_ERR_SOURCE_DIM - 1)
537#define GPG_ERR_SOURCE_SHIFT 24
538
539/* The highest bit is reserved. It shouldn't be used to prevent
540 potential negative numbers when transmitting error values as
541 text. */

581/* Construct an error value from an error code and source. Within a
582 subsystem, use gpg_error. */
583static GPG_ERR_INLINE gpg_error_t
584gpg_err_make (gpg_err_source_t source, gpg_err_code_t code)
585{
586 return code == GPG_ERR_NO_ERROR ? GPG_ERR_NO_ERROR
587 : (((source & GPG_ERR_SOURCE_MASK) << GPG_ERR_SOURCE_SHIFT)
588 | (code & GPG_ERR_CODE_MASK));
589}

593 file to specify a default source for gpg_error. */

595#define GPG_ERR_SOURCE_DEFAULT GPG_ERR_SOURCE_UNKNOWN
596#endif

599gpg_error (gpg_err_code_t code)
600{
