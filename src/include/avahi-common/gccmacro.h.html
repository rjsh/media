23/** \file gccmacro.h Defines some macros for GCC extensions */
24
25#include <avahi-common/cdecl.h>
26
27AVAHI_C_DECL_BEGIN
28
29#if defined(__GNUC__) && (__GNUC__ >= 4) && (__GNUC_MINOR__ >= 3)
30#define AVAHI_GCC_ALLOC_SIZE(x) __attribute__ ((__alloc_size__(x)))
31#define AVAHI_GCC_ALLOC_SIZE2(x,y) __attribute__ ((__alloc_size__(x,y)))
32#else
33/** Macro for usage of GCC's alloc_size attribute */
34#define AVAHI_GCC_ALLOC_SIZE(x)
35#define AVAHI_GCC_ALLOC_SIZE2(x,y)
36#endif
37
38#if defined(__GNUC__) && (__GNUC__ >= 4)
39#define AVAHI_GCC_SENTINEL __attribute__ ((sentinel))
40#else
41/** Macro for usage of GCC's sentinel compilation warnings */
42#define AVAHI_GCC_SENTINEL
43#endif
44
45#ifdef __GNUC__
46#define AVAHI_GCC_PRINTF_ATTR(a,b) __attribute__ ((format (printf, a, b)))
47#else
48/** Macro for usage of GCC's printf compilation warnings */
49#define AVAHI_GCC_PRINTF_ATTR(a,b)
50#endif
51
52/** Same as AVAHI_GCC_PRINTF_ATTR but hard coded to arguments 1 and 2 */
53#define AVAHI_GCC_PRINTF_ATTR12 AVAHI_GCC_PRINTF_ATTR(1,2)
54
55/** Same as AVAHI_GCC_PRINTF_ATTR but hard coded to arguments 2 and 3 */
56#define AVAHI_GCC_PRINTF_ATTR23 AVAHI_GCC_PRINTF_ATTR(2,3)
57
58#ifdef __GNUC__
59#define AVAHI_GCC_NORETURN __attribute__((noreturn))
60#else
61/** Macro for no-return functions */
62#define AVAHI_GCC_NORETURN
63#endif
64
65#ifdef __GNUC__
66#define AVAHI_GCC_UNUSED __attribute__ ((unused))
67#else
68/** Macro for not used parameter */
69#define AVAHI_GCC_UNUSED
70#endif
71
72AVAHI_C_DECL_END
73
74#endif
