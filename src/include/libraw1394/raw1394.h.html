100enum raw1394_iso_disposition {
101 /* continue on to the next packet */
102 RAW1394_ISO_OK = 0,
103
104 /* no error, but return from raw1394_loop_iterate() immediately */
105 RAW1394_ISO_DEFER = 1,
106
107 /* return from raw1394_loop_iterate() immediately, which will return an error */
108 RAW1394_ISO_ERROR = 2,
109
110 /* return from raw1394_loop_iterate() immediately, and stop receiving packets */
111 RAW1394_ISO_STOP = 3,
112
113 /* (transmission only) - like ISO_STOP, but don't wait for the buffer to empty */
114 RAW1394_ISO_STOP_NOSYNC = 4,
115
116 /* (transmission only) - use this if the data you need for the packet was not ready yet
117 * The packet queueing will retry at a later point
118 */
119 RAW1394_ISO_AGAIN = 5,
120};

306/**
307 * raw1394_iso_recv_flush - flush all already received iso packets from kernel into user space
308 * @handle: libraw1394 handle
309 *
310 * If you specified an irq_interval > 1 in
311 * iso_recv_init, you won't be notified for every single iso packet, but
312 * for groups of them. Now e.g. if irq_interval is 100, and you were just
313 * notified about iso packets and after them only 20 more packets arrived,
314 * no notification will be generated (20 < 100). In the case that you know
315 * that there should be more packets at this moment, you can call this
316 * function and all iso packets which are already received by the kernel
317 * will be flushed out to user space.
318 *
319 * Returns: 0 on success or -1 on failure (sets errno)
320 **/
321int raw1394_iso_recv_flush(raw1394handle_t handle);

349/**
350 * raw1394_read_cycle_timer_and_clock - get the current value of the cycle timer
351 * @handle: libraw1394 handle
352 * @cycle_timer: buffer for Isochronous Cycle Timer
353 * @local_time: buffer for local system time in microseconds
354 * @clk_id: clock from which to get the system time
355 *
356 * Simultaneously reads the cycle timer register together with a system clock.
357 *
358 * Format of @cycle_timer, from MSB to LSB: 7 bits cycleSeconds (seconds, or
359 * number of cycleCount rollovers), 13 bits cycleCount (isochronous cycles, or
360 * cycleOffset rollovers), 12 bits cycleOffset (24.576 MHz clock ticks, not
361 * provided on some hardware). The union of cycleSeconds and cycleCount is the
362 * current cycle number. The nominal duration of a cycle is 125 microseconds.
363 *
364 * @clk_id chooses the system clock as in the clock_gettime() function.
365 * At least the clocks CLOCK_REALTIME, CLOCK_MONOTONIC, and CLOCK_MONOTONIC_RAW
366 * are supported by the underlying kernel call.
367 *
368 * In case of CLOCK_REALTIME, @local_time are microseconds since the Epoch.
369 * CLOCK_REALTIME is subject to resets; CLOCK_MONOTONIC is subject to gradual
370 * adjustments; CLOCK_MONOTONIC_RAW is the bare hardware clock.
371 *
372 * Returns: the error code of the ioctl, or 0 if successful.
373 *
374 * History: New function in libraw1394 v2.1.0.
375 **/
376int raw1394_read_cycle_timer_and_clock(raw1394handle_t handle,
377 u_int32_t *cycle_timer, u_int64_t *local_time, clockid_t clk_id);

379typedef int raw1394_errcode_t;
380#define raw1394_make_errcode(ack, rcode) (((ack) << 16) | rcode)
381#define raw1394_internal_err(errcode) ((errcode) < 0)
382#define raw1394_get_ack(errcode) ((errcode) >> 16)
383#define raw1394_get_rcode(errcode) ((errcode) & 0xf)
384#define raw1394_get_internal(errcode) (errcode)

386/**
387 * raw1394_get_errcode - return error code of async transaction
388 * @handle: libraw1394 handle
389 *
390 * Some macros are available to extract information from the error code,
391 * raw1394_errcode_to_errno() can be used to convert it to an errno number of
392 * roughly the same meaning.
393 *
394 * Returns: the error code of the last raw1394_read(), raw1394_write(),
395 * raw1394_lock(). The error code is either an internal
396 * error (i.e. not a bus error) or a combination of acknowledge code and
397 * response code, as appropriate.
398 *
399 **/
400raw1394_errcode_t raw1394_get_errcode(raw1394handle_t handle);
401
402/**
403 * raw1394_errcode_to_errno - convert libraw1394 errcode to errno
404 * @errcode: the error code to convert
405 *
406 * The error code as retrieved by raw1394_get_errcode() is converted into a
407 * roughly equivalent errno number and returned. %0xdead is returned for an
408 * illegal errcode.
409 *
410 * It is intended to be used to decide what to do (retry, give up, report error)
411 * for those programs that aren't interested in details, since these get lost in
412 * the conversion. However the returned errnos are equivalent in source code
413 * meaning only, the associated text of e.g. perror() is not necessarily
414 * meaningful.
415 *
416 * Returns: %EAGAIN (retrying might succeed, also generation number
417 * mismatch), %EREMOTEIO (other node had internal problems), %EPERM (operation
418 * not allowed on this address, e.g. write on read-only location), %EINVAL
419 * (invalid argument) and %EFAULT (invalid pointer).
420 **/
421int raw1394_errcode_to_errno(raw1394_errcode_t errcode);

423/**
424 * raw1394_new_handle - create new handle
425 *
426 * Creates and returns a new handle which can (after being set up) control one
427 * port. It is not allowed to use the same handle in multiple threads or forked
428 * processes. It is allowed to create and use multiple handles, however. Use
429 * one handle per thread which needs it in the multithreaded case.
430 *
431 * The default device node is /dev/raw1394, but one can override the default
432 * by setting environment variable RAW1394DEV. However, if RAW1394DEV points to
433 * a non-existant or invalid device node, then it also attempts to open the
434 * default device node.
435 *
436 * Returns: the created handle or %NULL when initialization fails. In the latter
437 * case errno either contains some OS specific error code or EPROTO if
438 * libraw1394 and raw1394 don't support each other's protocol versions.
439 **/
440raw1394handle_t raw1394_new_handle(void);

590 * @handle: libraw1394 handle
591 * @port: port to connect to (corresponds to index of struct raw1394_portinfo)
592 *
593 * This function connects the handle to the port given (as queried with
594 * raw1394_get_port_info()). If successful, raw1394_get_port_info() and
