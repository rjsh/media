11/*!\defgroup vp8 VP8
12 * \ingroup codecs
13 * VP8 is vpx's newest video compression algorithm that uses motion
14 * compensated prediction, Discrete Cosine Transform (DCT) coding of the
15 * prediction error signal and context dependent entropy coding techniques
16 * based on arithmetic principles. It features:
17 * - YUV 4:2:0 image format
18 * - Macro-block based coding (16x16 luma plus two 8x8 chroma)
19 * - 1/4 (1/8) pixel accuracy motion compensated prediction
20 * - 4x4 DCT transform
21 * - 128 level linear quantizer
22 * - In loop deblocking filter
23 * - Context-based entropy coding
24 *
25 * @{
26 */
27/*!\file
28 * \brief Provides controls common to both the VP8 encoder and decoder.
29 */

40/*!\brief Control functions
41 *
42 * The set of macros define the control functions of VP8 interface
43 */
44enum vp8_com_control_id {
45 VP8_SET_REFERENCE = 1, /**< pass in an external frame into decoder to be used as reference frame */
46 VP8_COPY_REFERENCE = 2, /**< get a copy of reference frame from the decoder */
47 VP8_SET_POSTPROC = 3, /**< set the decoder's post processing settings */
48 VP8_SET_DBG_COLOR_REF_FRAME = 4, /**< set the reference frames to color for each macroblock */
49 VP8_SET_DBG_COLOR_MB_MODES = 5, /**< set which macro block modes to color */
50 VP8_SET_DBG_COLOR_B_MODES = 6, /**< set which blocks modes to color */
51 VP8_SET_DBG_DISPLAY_MV = 7, /**< set which motion vector modes to draw */
52
53 /* TODO(jkoleszar): The encoder incorrectly reuses some of these values (5+)
54 * for its control ids. These should be migrated to something like the
55 * VP8_DECODER_CTRL_ID_START range next time we're ready to break the ABI.
56 */
57 VP9_GET_REFERENCE = 128, /**< get a pointer to a reference frame */
58 VP8_COMMON_CTRL_ID_MAX,
59 VP8_DECODER_CTRL_ID_START = 256
60};
61
62/*!\brief post process flags
63 *
64 * The set of macros define VP8 decoder post processing flags
65 */
66enum vp8_postproc_level {
67 VP8_NOFILTERING = 0,
68 VP8_DEBLOCK = 1 << 0,
69 VP8_DEMACROBLOCK = 1 << 1,
70 VP8_ADDNOISE = 1 << 2,
71 VP8_DEBUG_TXT_FRAME_INFO = 1 << 3, /**< print frame information */
72 VP8_DEBUG_TXT_MBLK_MODES = 1 << 4, /**< print macro block modes over each macro block */
73 VP8_DEBUG_TXT_DC_DIFF = 1 << 5, /**< print dc diff for each macro block */
74 VP8_DEBUG_TXT_RATE_INFO = 1 << 6, /**< print video rate info (encoder only) */
75 VP8_MFQE = 1 << 10
76};
77
78/*!\brief post process flags
79 *
80 * This define a structure that describe the post processing settings. For
81 * the best objective measure (using the PSNR metric) set post_proc_flag
82 * to VP8_DEBLOCK and deblocking_level to 1.
83 */
84
85typedef struct vp8_postproc_cfg {
86 int post_proc_flag; /**< the types of post processing to be done, should be combination of "vp8_postproc_level" */
87 int deblocking_level; /**< the strength of deblocking, valid range [0, 16] */
88 int noise_level; /**< the strength of additive noise, valid range [0, 16] */
89} vp8_postproc_cfg_t;
90
91/*!\brief reference frame type
92 *
93 * The set of macros define the type of VP8 reference frames
94 */
95typedef enum vpx_ref_frame_type {
96 VP8_LAST_FRAME = 1,
97 VP8_GOLD_FRAME = 2,
98 VP8_ALTR_FRAME = 4
99} vpx_ref_frame_type_t;
100
101/*!\brief reference frame data struct
102 *
103 * Define the data struct to access vp8 reference frames.
104 */
105typedef struct vpx_ref_frame {
106 vpx_ref_frame_type_t frame_type; /**< which reference frame */
107 vpx_image_t img; /**< reference frame data in image format */
108} vpx_ref_frame_t;
109
110/*!\brief VP9 specific reference frame data struct
111 *
112 * Define the data struct to access vp9 reference frames.
113 */
114typedef struct vp9_ref_frame {
115 int idx; /**< frame index to get (input) */
116 vpx_image_t img; /**< img structure to populate (output) */
117} vp9_ref_frame_t;
