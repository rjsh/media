21#ifndef __NOTIFY_VERSION_H__
22#define __NOTIFY_VERSION_H__
23
24/* compile time version
25 */
26#define NOTIFY_VERSION_MAJOR (0)
27#define NOTIFY_VERSION_MINOR (7)
28#define NOTIFY_VERSION_MICRO (6)
29
30/* check whether a version equal to or greater than
31 * major.minor.micro is present.
32 */
33#define NOTIFY_CHECK_VERSION(major,minor,micro) \
34 (NOTIFY_VERSION_MAJOR > (major) || \
35 (NOTIFY_VERSION_MAJOR == (major) && NOTIFY_VERSION_MINOR > (minor)) || \
36 (NOTIFY_VERSION_MAJOR == (major) && NOTIFY_VERSION_MINOR == (minor) && \
37 NOTIFY_VERSION_MICRO >= (micro)))
38
39
40#endif /* __NOTIFY_VERSION_H__ */
