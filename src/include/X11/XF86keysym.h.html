1/*
2 * XFree86 vendor specific keysyms.
3 *
4 * The XFree86 keysym range is 0x10080001 - 0x1008FFFF.
5 *
6 * When adding new entries, the xc/lib/XKeysymDB file should also be
7 * updated to make the new entries visible to Xlib.
8 */
9
10/*
11 * ModeLock
12 *
13 * This one is old, and not really used any more since XKB offers this
14 * functionality.
15 */
16
17#define XF86XK_ModeLock 0x1008FF01 /* Mode Switch Lock */
18
19/*
20 * Note, 0x1008FF07 - 0x1008FF0F are free and should be used for misc new
21 * keysyms that don't fit into any of the groups below.
22 *
23 * 0x1008FF64, 0x1008FF6F, 0x1008FF71, 0x1008FF83 are no longer used,
24 * and should be used first for new keysyms.
25 *
26 * Check in keysymdef.h for generic symbols before adding new XFree86-specific
27 * symbols here.
28 *
29 * X.Org will not be adding to the XF86 set of keysyms, though they have
30 * been adopted and are considered a "standard" part of X keysym definitions.
31 * XFree86 never properly commented these keysyms, so we have done our
32 * best to explain the semantic meaning of these keys.
33 *
34 * XFree86 has removed their mail archives of the period, that might have
35 * shed more light on some of these definitions. Until/unless we resurrect
36 * these archives, these are from memory and usage.
37 */
38
