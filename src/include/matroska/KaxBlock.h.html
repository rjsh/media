55class MATROSKA_DLL_API DataBuffer {
56 protected:
57 binary * myBuffer;
58 uint32 mySize;
59 bool bValidValue;
60 bool (*myFreeBuffer)(const DataBuffer & aBuffer); // method to free the internal buffer
61 bool bInternalBuffer;
62
63 public:
64 DataBuffer(binary * aBuffer, uint32 aSize, bool (*aFreeBuffer)(const DataBuffer & aBuffer) = NULL, bool _bInternalBuffer = false)
65 :myBuffer(NULL)
66 ,mySize(aSize)
67 ,bValidValue(true)
68 ,myFreeBuffer(aFreeBuffer)
69 ,bInternalBuffer(_bInternalBuffer)
70 {
71 if (bInternalBuffer)
72 {
73 myBuffer = new (std::nothrow) binary[mySize];
74 if (myBuffer == NULL)
75 bValidValue = false;
76 else
77 memcpy(myBuffer, aBuffer, mySize);
78 }
79 else
80 myBuffer = aBuffer;
81 }
82
83 virtual ~DataBuffer() {}
84 virtual binary * Buffer() {assert(bValidValue); return myBuffer;}
85 virtual uint32 & Size() {return mySize;};
86 virtual const binary * Buffer() const {assert(bValidValue); return myBuffer;}
87 virtual uint32 Size() const {return mySize;};
88 bool FreeBuffer(const DataBuffer & aBuffer) {
89 bool bResult = true;
90 if (myBuffer != NULL && bValidValue) {
91 if (myFreeBuffer != NULL)
92 bResult = myFreeBuffer(aBuffer);
93 if (bInternalBuffer)
94 delete [] myBuffer;
95 myBuffer = NULL;
96 mySize = 0;
97 bValidValue = false;
98 }
99 return bResult;
100 }
101
102 virtual DataBuffer * Clone();
103};

105class MATROSKA_DLL_API SimpleDataBuffer : public DataBuffer {
106 public:
107 SimpleDataBuffer(binary * aBuffer, uint32 aSize, uint32 aOffset, bool (*aFreeBuffer)(const DataBuffer & aBuffer) = myFreeBuffer)
108 :DataBuffer(aBuffer + aOffset, aSize, aFreeBuffer)
109 ,Offset(aOffset)
110 ,BaseBuffer(aBuffer)
111 {}
112 virtual ~SimpleDataBuffer() {}
113
114 DataBuffer * Clone() {return new SimpleDataBuffer(*this);}
115
116 protected:
117 uint32 Offset;
118 binary * BaseBuffer;
119
120 static bool myFreeBuffer(const DataBuffer & aBuffer)
121 {
122 binary *_Buffer = static_cast<const SimpleDataBuffer*>(&aBuffer)->BaseBuffer;
123 if (_Buffer != NULL)
124 free(_Buffer);
125 return true;
126 }
127
128 SimpleDataBuffer(const SimpleDataBuffer & ToClone);
129};
