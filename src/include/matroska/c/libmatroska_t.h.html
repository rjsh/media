36 \brief Misc type definitions for the C API of libmatroska
37
38 \note These types should be compiler/language independant (just platform dependant)
39 \todo recover the sized types (uint16, int32, etc) here too (or maybe here only)
40*/
41
42#ifndef _LIBMATROSKA_T_H_INCLUDED_
43#define _LIBMATROSKA_T_H_INCLUDED_
44
45#include "ebml/c/libebml_t.h"
46
47#ifdef __cplusplus
48extern "C" {
49#endif
50
51/*!
52 \enum track_type
53*/
54typedef enum track_type {
55 track_video = 0x01, ///< Rectangle-shaped non-transparent pictures aka video
56 track_audio = 0x02, ///< Anything you can hear
57 track_complex = 0x03, ///< Audio and video in same track, used by DV
58
59 track_logo = 0x10, ///< Overlay-pictures, displayed over video
60 track_subtitle = 0x11, ///< Text-subtitles. One track contains one language and only one track can be active (player-side configuration)
61 track_buttons = 0x12, ///< Buttons meta-infos.
62
63 track_control = 0x20 ///< Control-codes for menus and other stuff
64} track_type;
65
66/*!
67 \enum matroska_error_t
68 \brief a callback that the library use to inform of errors happening
69 \note this should be used by the libmatroska internals
70*/
71typedef enum {
72 error_null_pointer ///< NULL pointer where something else is expected
73} matroska_error_t;
74
75typedef void *matroska_stream;
76
77/*!
78 \var void* matroska_id
79 \brief UID used to access an Matroska file instance
80*/
81typedef void* matroska_id;
82/*!
83 \var void* matroska_track
84 \brief UID used to access a track
85*/
86typedef void* matroska_track;
87/*!
88 \var char* c_string
89 \brief C-String, ie a buffer with characters terminated by \0
90*/
91typedef char* c_string;
92/*!
93 \var unsigned int matroska_file_mode
94 \brief A bit buffer, each bit representing a value for the file opening
95 \todo replace the unsigned int by a sized type (8 or 16 bits)
96*/
97typedef char * matroska_file_mode;
98/*!
99 \var void (*matroska_error_callback)(matroska_error_t error_code, char* error_message)
100 \brief a callback that the library use to inform of errors happening
101*/
102typedef void (*matroska_error_callback)(matroska_error_t error_code, char* error_message);
103
104#ifdef __cplusplus
105}
106#endif
107
108#endif /* _LIBMATROSKA_T_H_INCLUDED_ */
