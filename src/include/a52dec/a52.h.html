24#ifndef A52_H
25#define A52_H
26
27#ifndef LIBA52_DOUBLE
28typedef float sample_t;
29#else
30typedef double sample_t;
31#endif
32
33typedef struct {
34 uint8_t bai; /* fine SNR offset, fast gain */
35 uint8_t deltbae; /* delta bit allocation exists */
36 int8_t deltba[50]; /* per-band delta bit allocation */
37} ba_t;
38
39typedef struct {
40 uint8_t exp[256]; /* decoded channel exponents */
41 int8_t bap[256]; /* derived channel bit allocation */
42} expbap_t;
43
44typedef struct {
45 sample_t real;
46 sample_t imag;
47} complex_t;
48
49typedef struct {
50 uint8_t fscod; /* sample rate */
51 uint8_t halfrate; /* halfrate factor */
52 uint8_t acmod; /* coded channels */
53 uint8_t lfeon; /* coded lfe channel */
54 sample_t clev; /* centre channel mix level */
55 sample_t slev; /* surround channels mix level */
56
57 int output; /* type of output */
58 sample_t level; /* output level */
59 sample_t bias; /* output bias */
60
61 int dynrnge; /* apply dynamic range */
62 sample_t dynrng; /* dynamic range */
63 void * dynrngdata; /* dynamic range callback funtion and data */
64 sample_t (* dynrngcall) (sample_t range, void * dynrngdata);
65
66 uint8_t chincpl; /* channel coupled */
67 uint8_t phsflginu; /* phase flags in use (stereo only) */
68 uint8_t cplstrtmant; /* coupling channel start mantissa */
69 uint8_t cplendmant; /* coupling channel end mantissa */
70 uint32_t cplbndstrc; /* coupling band structure */
71 sample_t cplco[5][18]; /* coupling coordinates */
72
73 /* derived information */
74 uint8_t cplstrtbnd; /* coupling start band (for bit allocation) */
75 uint8_t ncplbnd; /* number of coupling bands */
76
77 uint8_t rematflg; /* stereo rematrixing */
78
79 uint8_t endmant[5]; /* channel end mantissa */
80
81 uint16_t bai; /* bit allocation information */
82
83 uint32_t * buffer_start;
84 uint16_t lfsr_state; /* dither state */
85 uint32_t bits_left;
86 uint32_t current_word;
87
88 uint8_t csnroffst; /* coarse SNR offset */
89 ba_t cplba; /* coupling bit allocation parameters */
90 ba_t ba[5]; /* channel bit allocation parameters */
91 ba_t lfeba; /* lfe bit allocation parameters */
92
93 uint8_t cplfleak; /* coupling fast leak init */
94 uint8_t cplsleak; /* coupling slow leak init */
95
96 expbap_t cpl_expbap;
97 expbap_t fbw_expbap[5];
98 expbap_t lfe_expbap;
99
100 sample_t * samples;
101 int downmixed;
102
103 /* Root values for IFFT */
104 sample_t * roots16; // size 3
105 sample_t * roots32; // size 7
106 sample_t * roots64; // size 15
107 sample_t * roots128; // size 31
108
109 /* Twiddle factors for IMDCT */
110 complex_t * pre1; // size 128
111 complex_t * post1; // size 64
112 complex_t * pre2; // size 64
113 complex_t * post2; // size 32
114
115 sample_t * a52_imdct_window; // size 256
116} a52_state_t;
117
118#define A52_CHANNEL 0
119#define A52_MONO 1
120#define A52_STEREO 2
121#define A52_3F 3
122#define A52_2F1R 4
123#define A52_3F1R 5
124#define A52_2F2R 6
125#define A52_3F2R 7
126#define A52_CHANNEL1 8
127#define A52_CHANNEL2 9
128#define A52_DOLBY 10
129#define A52_CHANNEL_MASK 15
130
131#define A52_LFE 16
132#define A52_ADJUST_LEVEL 32
133
134a52_state_t * a52_init (uint32_t mm_accel);
135sample_t * a52_samples (a52_state_t * state);
136int a52_syncinfo (uint8_t * buf, int * flags,
137 int * sample_rate, int * bit_rate);
138int a52_frame (a52_state_t * state, uint8_t * buf, int * flags,
139 sample_t * level, sample_t bias);
140void a52_dynrng (a52_state_t * state,
141 sample_t (* call) (sample_t, void *), void * data);
142int a52_block (a52_state_t * state);
143void a52_free (a52_state_t * state);
144
145#endif /* A52_H */
