46typedef void *ChromaprintContext;
47
48#define CHROMAPRINT_VERSION_MAJOR 1
49#define CHROMAPRINT_VERSION_MINOR 1
50#define CHROMAPRINT_VERSION_PATCH 0
51
52enum ChromaprintAlgorithm {
53 CHROMAPRINT_ALGORITHM_TEST1 = 0,
54 CHROMAPRINT_ALGORITHM_TEST2,
55 CHROMAPRINT_ALGORITHM_TEST3,
56 CHROMAPRINT_ALGORITHM_TEST4
57};
58
59#define CHROMAPRINT_ALGORITHM_DEFAULT CHROMAPRINT_ALGORITHM_TEST2
60
61/**
62 * Return the version number of Chromaprint.
63 */
64CHROMAPRINT_API const char *chromaprint_get_version(void);
65
66/**
67 * Allocate and initialize the Chromaprint context.
68 *
69 * Note that when Chromaprint is compiled with FFTW, this function is
70 * not reentrant and you need to call it only from one thread at a time.
71 * This is not a problem when using FFmpeg or vDSP.
72 *
73 * Parameters:
74 * - version: Version of the fingerprint algorithm, use
75 * CHROMAPRINT_ALGORITHM_DEFAULT for the default
76 * algorithm
77 *
78 * Returns:
79 * - Chromaprint context pointer
80 */
81CHROMAPRINT_API ChromaprintContext *chromaprint_new(int algorithm);
82
83/**
84 * Deallocate the Chromaprint context.
85 *
86 * Note that when Chromaprint is compiled with FFTW, this function is
87 * not reentrant and you need to call it only from one thread at a time.
88 * This is not a problem when using FFmpeg or vDSP.
89 *
90 * Parameters:
91 * - ctx: Chromaprint context pointer
92 */
93CHROMAPRINT_API void chromaprint_free(ChromaprintContext *ctx);
94
95/**
96 * Return the fingerprint algorithm this context is configured to use.
97 */
98CHROMAPRINT_API int chromaprint_get_algorithm(ChromaprintContext *ctx);
99
100/**
101 * Set a configuration option for the selected fingerprint algorithm.
102 *
103 * NOTE: DO NOT USE THIS FUNCTION IF YOU ARE PLANNING TO USE
104 * THE GENERATED FINGERPRINTS WITH THE ACOUSTID SERVICE.
105 *
106 * Parameters:
107 * - ctx: Chromaprint context pointer
108 * - name: option name
109 * - value: option value
110 *
111 * Possible options:
112 * - silence_threshold: threshold for detecting silence, 0-32767
113 *
114 * Returns:
115 * - 0 on error, 1 on success
116 */
117CHROMAPRINT_API int chromaprint_set_option(ChromaprintContext *ctx, const char *name, int value);
118
119/**
120 * Restart the computation of a fingerprint with a new audio stream.
121 *
122 * Parameters:
123 * - ctx: Chromaprint context pointer
124 * - sample_rate: sample rate of the audio stream (in Hz)
125 * - num_channels: numbers of channels in the audio stream (1 or 2)
126 *
127 * Returns:
128 * - 0 on error, 1 on success
129 */
130CHROMAPRINT_API int chromaprint_start(ChromaprintContext *ctx, int sample_rate, int num_channels);
131
132/**
133 * Send audio data to the fingerprint calculator.
134 *
135 * Parameters:
136 * - ctx: Chromaprint context pointer
137 * - data: raw audio data, should point to an array of 16-bit signed
138 * integers in native byte-order
139 * - size: size of the data buffer (in samples)
140 *
141 * Returns:
142 * - 0 on error, 1 on success
143 */
144CHROMAPRINT_API int chromaprint_feed(ChromaprintContext *ctx, void *data, int size);
145
146/**
147 * Process any remaining buffered audio data and calculate the fingerprint.
148 *
149 * Parameters:
150 * - ctx: Chromaprint context pointer
151 *
152 * Returns:
153 * - 0 on error, 1 on success
154 */
155CHROMAPRINT_API int chromaprint_finish(ChromaprintContext *ctx);
156
157/**
158 * Return the calculated fingerprint as a compressed string.
159 *
160 * The caller is responsible for freeing the returned pointer using
161 * chromaprint_dealloc().
162 *
163 * Parameters:
164 * - ctx: Chromaprint context pointer
165 * - fingerprint: pointer to a pointer, where a pointer to the allocated array
166 * will be stored
167 *
168 * Returns:
169 * - 0 on error, 1 on success
170 */
171CHROMAPRINT_API int chromaprint_get_fingerprint(ChromaprintContext *ctx, char **fingerprint);
172
173/**
174 * Return the calculated fingerprint as an array of 32-bit integers.
175 *
176 * The caller is responsible for freeing the returned pointer using
177 * chromaprint_dealloc().
178 *
179 * Parameters:
180 * - ctx: Chromaprint context pointer
181 * - fingerprint: pointer to a pointer, where a pointer to the allocated array
182 * will be stored
183 * - size: number of items in the returned raw fingerprint
184 *
185 * Returns:
186 * - 0 on error, 1 on success
187 */
188CHROMAPRINT_API int chromaprint_get_raw_fingerprint(ChromaprintContext *ctx, void **fingerprint, int *size);
189
190/**
191 * Compress and optionally base64-encode a raw fingerprint
192 *
193 * The caller is responsible for freeing the returned pointer using
194 * chromaprint_dealloc().
195 *
196 * Parameters:
197 * - fp: pointer to an array of 32-bit integers representing the raw
198 * fingerprint to be encoded
199 * - size: number of items in the raw fingerprint
200 * - algorithm: Chromaprint algorithm version which was used to generate the
201 * raw fingerprint
202 * - encoded_fp: pointer to a pointer, where the encoded fingerprint will be
203 * stored
204 * - encoded_size: size of the encoded fingerprint in bytes
205 * - base64: Whether to return binary data or base64-encoded ASCII data. The
206 * compressed fingerprint will be encoded using base64 with the
207 * URL-safe scheme if you set this parameter to 1. It will return
208 * binary data if it's 0.
209 *
210 * Returns:
211 * - 0 on error, 1 on success
212 */
213CHROMAPRINT_API int chromaprint_encode_fingerprint(void *fp, int size, int algorithm, void **encoded_fp, int *encoded_size, int base64);
214
215/**
216 * Uncompress and optionally base64-decode an encoded fingerprint
217 *
218 * The caller is responsible for freeing the returned pointer using
219 * chromaprint_dealloc().
220 *
221 * Parameters:
222 * - encoded_fp: Pointer to an encoded fingerprint
223 * - encoded_size: Size of the encoded fingerprint in bytes
224 * - fp: Pointer to a pointer, where the decoded raw fingerprint (array
225 * of 32-bit integers) will be stored
226 * - size: Number of items in the returned raw fingerprint
227 * - algorithm: Chromaprint algorithm version which was used to generate the
228 * raw fingerprint
229 * - base64: Whether the encoded_fp parameter contains binary data or
230 * base64-encoded ASCII data. If 1, it will base64-decode the data
231 * before uncompressing the fingerprint.
232 *
233 * Returns:
234 * - 0 on error, 1 on success
235 */
236CHROMAPRINT_API int chromaprint_decode_fingerprint(void *encoded_fp, int encoded_size, void **fp, int *size, int *algorithm, int base64);
237
238/**
239 * Free memory allocated by any function from the Chromaprint API.
240 *
241 * Parameters:
242 * - ptr: Pointer to be deallocated
243 */
244CHROMAPRINT_API void chromaprint_dealloc(void *ptr);
