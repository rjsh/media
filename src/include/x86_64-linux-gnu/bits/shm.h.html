22#include <bits/types.h>

24/* Permission flag for shmget. */
25#define SHM_R 0400 /* or S_IRUGO from <linux/stat.h> */
26#define SHM_W 0200 /* or S_IWUGO from <linux/stat.h> */
27
28/* Flags for `shmat'. */
29#define SHM_RDONLY 010000 /* attach read-only else read-write */
30#define SHM_RND 020000 /* round attach address to SHMLBA */
31#define SHM_REMAP 040000 /* take-over region on attach */
32#define SHM_EXEC 0100000 /* execution access */
33
34/* Commands for `shmctl'. */
35#define SHM_LOCK 11 /* lock segment (root only) */
36#define SHM_UNLOCK 12 /* unlock segment (root only) */

38__BEGIN_DECLS
39
40/* Segment low boundary address multiple. */
41#define SHMLBA (__getpagesize ())
42extern int __getpagesize (void) __THROW __attribute__ ((__const__));
43
44
45/* Type to count number of attaches. */
46typedef __syscall_ulong_t shmatt_t;
47
48/* Data structure describing a shared memory segment. */
49struct shmid_ds
50 {
51 struct ipc_perm shm_perm; /* operation permission struct */
52 size_t shm_segsz; /* size of segment in bytes */
53 __time_t shm_atime; /* time of last shmat() */
54#ifndef __x86_64__
55 unsigned long int __glibc_reserved1;
56#endif
57 __time_t shm_dtime; /* time of last shmdt() */
58#ifndef __x86_64__
59 unsigned long int __glibc_reserved2;
60#endif
61 __time_t shm_ctime; /* time of last change by shmctl() */
62#ifndef __x86_64__
63 unsigned long int __glibc_reserved3;
64#endif
65 __pid_t shm_cpid; /* pid of creator */
66 __pid_t shm_lpid; /* pid of last shmop */
67 shmatt_t shm_nattch; /* number of current attaches */
68 __syscall_ulong_t __glibc_reserved4;
69 __syscall_ulong_t __glibc_reserved5;
70 };

74/* ipcs ctl commands */
75# define SHM_STAT 13
76# define SHM_INFO 14

78/* shm_mode upper byte flags */
79# define SHM_DEST 01000 /* segment will be destroyed on last detach */
80# define SHM_LOCKED 02000 /* segment will not be swapped */
81# define SHM_HUGETLB 04000 /* segment is mapped via hugetlb */
82# define SHM_NORESERVE 010000 /* don't check for reservations */

85 {
86 __syscall_ulong_t shmmax;

95 };

97struct shm_info
98 {
99 int used_ids;
100 __syscall_ulong_t shm_tot; /* total allocated shm */

102 __syscall_ulong_t shm_swp; /* total swapped shm */
103 __syscall_ulong_t swap_attempts;
104 __syscall_ulong_t swap_successes;
105 };
