30/**
31 * @file shell.h
32 * @brief Command shell interface
33 *
34 * The shell interface allows you to send simple textual commands to
35 * the synthesizer, to parse a command file, or to read commands
36 * from the stdin or other input streams.
37 */

42FLUIDSYNTH_API char* fluid_get_userconf(char* buf, int len);
43FLUIDSYNTH_API char* fluid_get_sysconf(char* buf, int len);

45/**
46 * Command handler function prototype.
47 * @param data User defined data
48 * @param ac Argument count
49 * @param av Array of string arguments
50 * @param out Output stream to send response to
51 * @return Should return #FLUID_OK on success, #FLUID_FAILED otherwise
52 */
53typedef int (*fluid_cmd_func_t)(void* data, int ac, char** av, fluid_ostream_t out);
54
55/**
56 * Shell command information structure.
57 */
58typedef struct {
59 char* name; /**< The name of the command, as typed in the shell */
60 char* topic; /**< The help topic group of this command */
61 fluid_cmd_func_t handler; /**< Pointer to the handler for this command */
62 void* data; /**< User data passed to the handler */
63 char* help; /**< A help string */
64} fluid_cmd_t;
