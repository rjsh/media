29/**
30 * @file sfont.h
31 * @brief SoundFont plugins
32 *
33 * It is possible to add new SoundFont loaders to the
34 * synthesizer. The API uses a couple of "interfaces" (structures
35 * with callback functions): #fluid_sfloader_t, #fluid_sfont_t, and
36 * #fluid_preset_t. This API allows for virtual SoundFont files to be loaded
37 * and synthesized, which may not actually be SoundFont files, as long as they
38 * can be represented by the SoundFont synthesis model.
39 *
40 * To add a new SoundFont loader to the synthesizer, call
41 * fluid_synth_add_sfloader() and pass a pointer to an
42 * fluid_sfloader_t structure. The important callback function in
43 * this structure is "load", which should try to load a file and
44 * returns a #fluid_sfont_t structure, or NULL if it fails.
45 *
46 * The #fluid_sfont_t structure contains a callback to obtain the
47 * name of the SoundFont. It contains two functions to iterate
48 * though the contained presets, and one function to obtain a
49 * preset corresponding to a bank and preset number. This
50 * function should return a #fluid_preset_t structure.
51 *
52 * The #fluid_preset_t structure contains some functions to obtain
53 * information from the preset (name, bank, number). The most
54 * important callback is the noteon function. The noteon function
55 * should call fluid_synth_alloc_voice() for every sample that has
56 * to be played. fluid_synth_alloc_voice() expects a pointer to a
57 * #fluid_sample_t structure and returns a pointer to the opaque
58 * #fluid_voice_t structure. To set or increment the values of a
59 * generator, use fluid_voice_gen_set() or fluid_voice_gen_incr(). When you are
60 * finished initializing the voice call fluid_voice_start() to
61 * start playing the synthesis voice.
62 */
63
64/**
65 * Some notification enums for presets and samples.
66 */
67enum {
68 FLUID_PRESET_SELECTED, /**< Preset selected notify */
69 FLUID_PRESET_UNSELECTED, /**< Preset unselected notify */
70 FLUID_SAMPLE_DONE /**< Sample no longer needed notify */
71};

74/**
75 * SoundFont loader structure.
76 */
77struct _fluid_sfloader_t {
78 void* data; /**< User defined data pointer */
79
80 /**
81 * The free method should free the memory allocated for the loader in
82 * addition to any private data.
83 * @param loader SoundFont loader
84 * @return Should return 0 if no error occured, non-zero otherwise
85 */
86 int (*free)(fluid_sfloader_t* loader);
87
88 /**
89 * Method to load an instrument file (does not actually need to be a real file name,
90 * could be another type of string identifier that the \a loader understands).
91 * @param loader SoundFont loader
92 * @param filename File name or other string identifier
93 * @return The loaded instrument file (SoundFont) or NULL if an error occured.
94 */
95 fluid_sfont_t* (*load)(fluid_sfloader_t* loader, const char* filename);
96};
97
98/**
99 * Virtual SoundFont instance structure.
100 */
101struct _fluid_sfont_t {
102 void* data; /**< User defined data */
103 unsigned int id; /**< SoundFont ID */
104
105 /**
106 * Method to free a virtual SoundFont bank.
107 * @param sfont Virtual SoundFont to free.
108 * @return Should return 0 when it was able to free all resources or non-zero
109 * if some of the samples could not be freed because they are still in use,
110 * in which case the free will be tried again later, until success.
111 */
112 int (*free)(fluid_sfont_t* sfont);
113
114 /**
115 * Method to return the name of a virtual SoundFont.
116 * @param sfont Virtual SoundFont
117 * @return The name of the virtual SoundFont.
118 */
119 char* (*get_name)(fluid_sfont_t* sfont);
120
121 /**
122 * Get a virtual SoundFont preset by bank and program numbers.
123 * @param sfont Virtual SoundFont
124 * @param bank MIDI bank number (0-16384)
125 * @param prenum MIDI preset number (0-127)
126 * @return Should return an allocated virtual preset or NULL if it could not
127 * be found.
128 */
129 fluid_preset_t* (*get_preset)(fluid_sfont_t* sfont, unsigned int bank, unsigned int prenum);
130
131 /**
132 * Start virtual SoundFont preset iteration method.
133 * @param sfont Virtual SoundFont
134 *
135 * Starts/re-starts virtual preset iteration in a SoundFont.
136 */
137 void (*iteration_start)(fluid_sfont_t* sfont);
138
139 /**
140 * Virtual SoundFont preset iteration function.
141 * @param sfont Virtual SoundFont
142 * @param preset Caller supplied preset to fill in with current preset information
143 * @return 0 when no more presets are available, 1 otherwise
144 *
145 * Should store preset information to the caller supplied \a preset structure
146 * and advance the internal iteration state to the next preset for subsequent
147 * calls.
148 */
149 int (*iteration_next)(fluid_sfont_t* sfont, fluid_preset_t* preset);
150};
151
152#define fluid_sfont_get_id(_sf) ((_sf)->id)

154/**
155 * Virtual SoundFont preset.
156 */
157struct _fluid_preset_t {
158 void* data; /**< User supplied data */
159 fluid_sfont_t* sfont; /**< Parent virtual SoundFont */
160
161 /**
162 * Method to free a virtual SoundFont preset.
163 * @param preset Virtual SoundFont preset
164 * @return Should return 0
165 */
166 int (*free)(fluid_preset_t* preset);
167
168 /**
169 * Method to get a virtual SoundFont preset name.
170 * @param preset Virtual SoundFont preset
171 * @return Should return the name of the preset. The returned string must be
172 * valid for the duration of the virtual preset (or the duration of the
173 * SoundFont, in the case of preset iteration).
174 */
175 char* (*get_name)(fluid_preset_t* preset);
176
177 /**
178 * Method to get a virtual SoundFont preset MIDI bank number.
179 * @param preset Virtual SoundFont preset
180 * @param return The bank number of the preset
181 */
182 int (*get_banknum)(fluid_preset_t* preset);
183
184 /**
185 * Method to get a virtual SoundFont preset MIDI program number.
186 * @param preset Virtual SoundFont preset
187 * @param return The program number of the preset
188 */
189 int (*get_num)(fluid_preset_t* preset);
190
191 /**
192 * Method to handle a noteon event (synthesize the instrument).
193 * @param preset Virtual SoundFont preset
194 * @param synth Synthesizer instance
195 * @param chan MIDI channel number of the note on event
196 * @param key MIDI note number (0-127)
197 * @param vel MIDI velocity (0-127)
198 * @return #FLUID_OK on success (0) or #FLUID_FAILED (-1) otherwise
199 *
200 * This method may be called from within synthesis context and therefore
201 * should be as efficient as possible and not perform any operations considered
202 * bad for realtime audio output (memory allocations and other OS calls).
203 *
204 * Call fluid_synth_alloc_voice() for every sample that has
205 * to be played. fluid_synth_alloc_voice() expects a pointer to a
206 * #fluid_sample_t structure and returns a pointer to the opaque
207 * #fluid_voice_t structure. To set or increment the values of a
208 * generator, use fluid_voice_gen_set() or fluid_voice_gen_incr(). When you are
209 * finished initializing the voice call fluid_voice_start() to
210 * start playing the synthesis voice. Starting with FluidSynth 1.1.0 all voices
211 * created will be started at the same time.
212 */
213 int (*noteon)(fluid_preset_t* preset, fluid_synth_t* synth, int chan, int key, int vel);
214
215 /**
216 * Virtual SoundFont preset notify method.
217 * @param preset Virtual SoundFont preset
218 * @param reason #FLUID_PRESET_SELECTED or #FLUID_PRESET_UNSELECTED
219 * @param chan MIDI channel number
220 * @return Should return #FLUID_OK
221 *
222 * Implement this optional method if the preset needs to be notified about
223 * preset select and unselect events.
224 *
225 * This method may be called from within synthesis context and therefore
226 * should be as efficient as possible and not perform any operations considered
227 * bad for realtime audio output (memory allocations and other OS calls).
228 */
229 int (*notify)(fluid_preset_t* preset, int reason, int chan);
230};

232/**
233 * Virtual SoundFont sample.
234 */
235struct _fluid_sample_t
236{
237 char name[21]; /**< Sample name */
238 unsigned int start; /**< Start index */
239 unsigned int end; /**< End index, index of last valid sample point (contrary to SF spec) */
240 unsigned int loopstart; /**< Loop start index */
241 unsigned int loopend; /**< Loop end index, first point following the loop (superimposed on loopstart) */
242 unsigned int samplerate; /**< Sample rate */
243 int origpitch; /**< Original pitch (MIDI note number, 0-127) */
244 int pitchadj; /**< Fine pitch adjustment (+/- 99 cents) */
245 int sampletype; /**< Values: #FLUID_SAMPLETYPE_MONO, FLUID_SAMPLETYPE_RIGHT, FLUID_SAMPLETYPE_LEFT, FLUID_SAMPLETYPE_ROM */
246 int valid; /**< Should be TRUE if sample data is valid, FALSE otherwise (in which case it will not be synthesized) */
247 short* data; /**< Pointer to the sample's data */
248
249 int amplitude_that_reaches_noise_floor_is_valid; /**< Indicates if \a amplitude_that_reaches_noise_floor is valid (TRUE), set to FALSE initially to calculate. */
250 double amplitude_that_reaches_noise_floor; /**< The amplitude at which the sample's loop will be below the noise floor. For voice off optimization, calculated automatically. */
251
252 unsigned int refcount; /**< Count of voices using this sample (use #fluid_sample_refcount to access this field) */
253
254 /**
255 * Implement this function to receive notification when sample is no longer used.
256 * @param sample Virtual SoundFont sample
257 * @param reason #FLUID_SAMPLE_DONE only currently
258 * @return Should return #FLUID_OK
259 */
260 int (*notify)(fluid_sample_t* sample, int reason);
261
262 void* userdata; /**< User defined data */
263};

266#define fluid_sample_refcount(_sample) ((_sample)->refcount) /**< Get the reference count of a sample. Should only be called from within synthesis context (noteon method for example) */

269#define FLUID_SAMPLETYPE_MONO 1 /**< Flag for #fluid_sample_t \a sampletype field for mono samples */
270#define FLUID_SAMPLETYPE_RIGHT 2 /**< Flag for #fluid_sample_t \a sampletype field for right samples of a stereo pair */
271#define FLUID_SAMPLETYPE_LEFT 4 /**< Flag for #fluid_sample_t \a sampletype field for left samples of a stereo pair */
272#define FLUID_SAMPLETYPE_LINKED 8 /**< Flag for #fluid_sample_t \a sampletype field, not used currently */
273#define FLUID_SAMPLETYPE_ROM 0x8000 /**< Flag for #fluid_sample_t \a sampletype field, ROM sample, causes sample to be ignored */
