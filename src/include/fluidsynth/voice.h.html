41/**
42 * Enum used with fluid_voice_add_mod() to specify how to handle duplicate modulators.
43 */
44enum fluid_voice_add_mod {
45 FLUID_VOICE_OVERWRITE, /**< Overwrite any existing matching modulator */
46 FLUID_VOICE_ADD, /**< Add (sum) modulator amounts */
47 FLUID_VOICE_DEFAULT /**< For default modulators only, no need to check for duplicates */
48};
