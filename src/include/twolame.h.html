66/** MPEG modes */
67 typedef enum {
68 TWOLAME_AUTO_MODE = -1,
69 /**< Choose Mode Automatically */
70 TWOLAME_STEREO = 0, /**< Stereo */
71 TWOLAME_JOINT_STEREO,
72 /**< Joint Stereo */
73 TWOLAME_DUAL_CHANNEL,
74 /**< Dual Channel */
75 TWOLAME_MONO, /**< Mono */
76 TWOLAME_NOT_SET
77 } TWOLAME_MPEG_mode;
78
79
80/** MPEG Version.
81 *
82 * MPEG2 is for Lower Sampling Frequencies - LSF < 32000.
83 */
84 typedef enum {
85 TWOLAME_MPEG2 = 0,
86 /**< MPEG-2 - for sample rates less than 32k */
87 TWOLAME_MPEG1 /**< MPEG-1 */
88 } TWOLAME_MPEG_version;
89
90
91/** Padding types. */
92 typedef enum {
93 TWOLAME_PAD_NO = 0, /**< No Padding */
94 TWOLAME_PAD_ALL /**< Pad all frames */
95// TWOLAME_PAD_ADJUST // unsupported by twolame
96 } TWOLAME_Padding;
97
98/** Emphasis types. */
99 typedef enum {
100 TWOLAME_EMPHASIS_N = 0,
101 /**< No Emphasis */
102 TWOLAME_EMPHASIS_5 = 1,
103 /**< 50/15 ms */
104 // reserved
105 TWOLAME_EMPHASIS_C = 3
106 /**< CCIT J.17 */
107 } TWOLAME_Emphasis;

179/** Encode some 16-bit PCM audio to MP2.
180 *
181 * Takes 16-bit PCM audio samples from seperate left and right
182 * buffers and places encoded audio into mp2buffer.
183 *
184 * \param glopts twolame options pointer
185 * \param leftpcm Left channel audio samples
186 * \param rightpcm Right channel audio samples
187 * \param num_samples Number of samples per channel
188 * \param mp2buffer Buffer to place encoded audio into
189 * \param mp2buffer_size Size of the output buffer
190 * \return The number of bytes put in output buffer
191 * or a negative value on error
192 */
193 DLL_EXPORT int twolame_encode_buffer(twolame_options * glopts,
194 const short int leftpcm[],
195 const short int rightpcm[],
196 int num_samples,
197 unsigned char *mp2buffer, int mp2buffer_size);
198

200/** Encode some 16-bit PCM audio to MP2.
201 *
202 * Takes interleaved 16-bit PCM audio samples from a single
203 * buffer and places encoded audio into mp2buffer.
204 *
205 * \param glopts twolame options pointer
206 * \param pcm Audio samples for left AND right channels
207 * \param num_samples Number of samples per channel
208 * \param mp2buffer Buffer to place encoded audio into
209 * \param mp2buffer_size Size of the output buffer
210 * \return The number of bytes put in output buffer
211 * or a negative value on error
212 */
213 DLL_EXPORT int twolame_encode_buffer_interleaved(twolame_options * glopts,
214 const short int pcm[],
215 int num_samples,
216 unsigned char *mp2buffer, int mp2buffer_size);

219/** Encode some 32-bit PCM audio to MP2.
220 *
221 * Takes 32-bit floating point PCM audio samples from seperate
222 * left and right buffers and places encoded audio into mp2buffer.
223 *
224 * Note: the 32-bit samples are currently scaled down to
225 * 16-bit samples internally.
226 *
227 * \param glopts twolame options pointer
228 * \param leftpcm Left channel audio samples
229 * \param rightpcm Right channel audio samples
230 * \param num_samples Number of samples per channel
231 * \param mp2buffer Buffer to place encoded audio into
232 * \param mp2buffer_size Size of the output buffer
233 * \return The number of bytes put in output buffer
234 * or a negative value on error
235 */
236 DLL_EXPORT int twolame_encode_buffer_float32(twolame_options * glopts,
237 const float leftpcm[],
238 const float rightpcm[],
239 int num_samples,
240 unsigned char *mp2buffer, int mp2buffer_size);
