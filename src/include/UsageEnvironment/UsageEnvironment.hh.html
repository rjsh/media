51// An abstract base class, subclassed for each use of the library
52
53class UsageEnvironment {
54public:
55 void reclaim();
56
57 // task scheduler:
58 TaskScheduler& taskScheduler() const {return fScheduler;}
59
60 // result message handling:
61 typedef char const* MsgString;
62 virtual MsgString getResultMsg() const = 0;
63
64 virtual void setResultMsg(MsgString msg) = 0;
65 virtual void setResultMsg(MsgString msg1, MsgString msg2) = 0;
66 virtual void setResultMsg(MsgString msg1, MsgString msg2, MsgString msg3) = 0;
67 virtual void setResultErrMsg(MsgString msg, int err = 0) = 0;
68 // like setResultMsg(), except that an 'errno' message is appended. (If "err == 0", the "getErrno()" code is used instead.)
69
70 virtual void appendToResultMsg(MsgString msg) = 0;
71
72 virtual void reportBackgroundError() = 0;
73 // used to report a (previously set) error message within
74 // a background event
75
76 virtual void internalError(); // used to 'handle' a 'should not occur'-type error condition within the library.
77
78 // 'errno'
79 virtual int getErrno() const = 0;
80
81 // 'console' output:
82 virtual UsageEnvironment& operator<<(char const* str) = 0;
83 virtual UsageEnvironment& operator<<(int i) = 0;
84 virtual UsageEnvironment& operator<<(unsigned u) = 0;
85 virtual UsageEnvironment& operator<<(double d) = 0;
86 virtual UsageEnvironment& operator<<(void* p) = 0;
87
88 // a pointer to additional, optional, client-specific state
89 void* liveMediaPriv;
90 void* groupsockPriv;
91
92protected:
93 UsageEnvironment(TaskScheduler& scheduler); // abstract base class
94 virtual ~UsageEnvironment(); // we are deleted only by reclaim()
95
96private:
97 TaskScheduler& fScheduler;
98};

101typedef void TaskFunc(void* clientData);
102typedef void* TaskToken;
103typedef u_int32_t EventTriggerId;
104
105class TaskScheduler {
106public:
107 virtual ~TaskScheduler();
108
109 virtual TaskToken scheduleDelayedTask(int64_t microseconds, TaskFunc* proc,
110 void* clientData) = 0;
111 // Schedules a task to occur (after a delay) when we next
112 // reach a scheduling point.
113 // (Does not delay if "microseconds" <= 0)
114 // Returns a token that can be used in a subsequent call to
115 // unscheduleDelayedTask()
116
117 virtual void unscheduleDelayedTask(TaskToken& prevTask) = 0;
118 // (Has no effect if "prevTask" == NULL)
119 // Sets "prevTask" to NULL afterwards.
120
121 virtual void rescheduleDelayedTask(TaskToken& task,
122 int64_t microseconds, TaskFunc* proc,
123 void* clientData);
124 // Combines "unscheduleDelayedTask()" with "scheduleDelayedTask()"
125 // (setting "task" to the new task token).
126
127 // For handling socket operations in the background (from the event loop):
128 typedef void BackgroundHandlerProc(void* clientData, int mask);
129 // Possible bits to set in "mask". (These are deliberately defined
130 // the same as those in Tcl, to make a Tcl-based subclass easy.)
131 #define SOCKET_READABLE (1<<1)
132 #define SOCKET_WRITABLE (1<<2)
133 #define SOCKET_EXCEPTION (1<<3)
134 virtual void setBackgroundHandling(int socketNum, int conditionSet, BackgroundHandlerProc* handlerProc, void* clientData) = 0;
135 void disableBackgroundHandling(int socketNum) { setBackgroundHandling(socketNum, 0, NULL, NULL); }
136 virtual void moveSocketHandling(int oldSocketNum, int newSocketNum) = 0;
137 // Changes any socket handling for "oldSocketNum" so that occurs with "newSocketNum" instead.
138
139 virtual void doEventLoop(char* watchVariable = NULL) = 0;
140 // Causes further execution to take place within the event loop.
141 // Delayed tasks, background I/O handling, and other events are handled, sequentially (as a single thread of control).
142 // (If "watchVariable" is not NULL, then we return from this routine when *watchVariable != 0)
143
144 virtual EventTriggerId createEventTrigger(TaskFunc* eventHandlerProc) = 0;
145 // Creates a 'trigger' for an event, which - if it occurs - will be handled (from the event loop) using "eventHandlerProc".
146 // (Returns 0 iff no such trigger can be created (e.g., because of implementation limits on the number of triggers).)
147 virtual void deleteEventTrigger(EventTriggerId eventTriggerId) = 0;
148
149 virtual void triggerEvent(EventTriggerId eventTriggerId, void* clientData = NULL) = 0;
150 // Causes the (previously-registered) handler function for the specified event to be handled (from the event loop).
151 // The handler function is called with "clientData" as parameter.
152 // Note: This function (unlike other library functions) may be called from an external thread - to signal an external event.
153
154 // The following two functions are deprecated, and are provided for backwards-compatibility only:
155 void turnOnBackgroundReadHandling(int socketNum, BackgroundHandlerProc* handlerProc, void* clientData) {
156 setBackgroundHandling(socketNum, SOCKET_READABLE, handlerProc, clientData);
157 }
158 void turnOffBackgroundReadHandling(int socketNum) { disableBackgroundHandling(socketNum); }
159
160 virtual void internalError(); // used to 'handle' a 'should not occur'-type error condition within the library.
161
162protected:
163 TaskScheduler(); // abstract base class
164};
