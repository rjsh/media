33/**
34 * Retrieve a disc record from the CDDB server. This function
35 * requires that the category and disc ID of the provided disc
36 * structure are valid.
37 *
38 * If nothing goes wrong, the function will return 1 and the error
39 * code will be reset to:
40 * - #CDDB_ERR_OK:
41 * If everything went as planned.
42 *
43 * If there is a problem with reading data from the CDDB server one of
44 * the following error codes will be set:
45 * - #CDDB_ERR_DATA_MISSING:
46 * If some required data is missing from the given disc
47 * structure to execute this command.
48 * - #CDDB_ERR_DISC_NOT_FOUND:
49 * If the requested disc is not known by the CDDB server.
50 * - #CDDB_ERR_SERVER_ERROR:
51 * If the server encountered an error while trying to process your
52 * request.
53 * - #CDDB_ERR_UNKNOWN:
54 * If the server specified an unknown response code. Please
55 * report this as a libcddb bug.
56 *
57 * When there are problems with the connection to the CDDB server one
58 * of the following error codes will be set:
59 * - #CDDB_ERR_UNKNOWN_HOST_NAME:
60 * If there was an error when resolving the host name of the CDDB
61 * server.
62 * - #CDDB_ERR_CONNECT:
63 * If a connection to the CDDB server could not be established.
64 * This can be due to incorrect data about the location of the
65 * server (host name, port).
66 * - #CDDB_ERR_NOT_CONNECTED:
67 * If something when wrong in the process and you got
68 * disconnected. Retrying might succeed (but no guarantees).
69 * - #CDDB_ERR_PERMISSION_DENIED:
70 * If the server is up and running but denied the connection.
71 * This can occur when the server is too highly loaded or the
72 * handshake information (user name, ...) is considered to be
73 * invalid.
74 *
75 * @param c The CDDB connection structure.
76 * @param disc A non-null CDDB disc structure.
77 * @return 1 on succes, 0 on failure
78 */
79int cddb_read(cddb_conn_t *c, cddb_disc_t *disc);
80
81/**
82 * Query the CDDB database for a list of possible disc matches. This
83 * function requires that the disc ID and disc length of the provided
84 * disc structure are valid. The disc should also contain a number of
85 * tracks and for each track its frame offset on the CD should be
86 * valid.
87 *
88 * If there are multiple matches then only the first one will be
89 * returned by this function. For other matches you will have to use
90 * the #cddb_query_next function.
91 *
92 * @param c The CDDB connection structure.
93 * @param disc A non-null CDDB disc structure.
94 *
95 * @return The number of matches found or -1 on error.
96 */
97int cddb_query(cddb_conn_t *c, cddb_disc_t *disc);
98
99/**
100 * Returns the next match in a CDDB query result set. This function
101 * should be used in conjunction with #cddb_query.
102 *
103 * @param c The CDDB connection structure.
104 * @param disc A non-null CDDB disc structure.
105 */
106int cddb_query_next(cddb_conn_t *c, cddb_disc_t *disc);
107
108/**
109 * Perform a text search in the CDDB database. Instead of actually
110 * needing information about a real disc like in #cddb_query this
111 * function accept a string that is used for searching the database.
112 *
113 * If there are multiple matches then only the first one will be
114 * returned by this function. For other matches you will have to use
115 * the #cddb_search_next function.
116 *
117 * @param c The CDDB connection structure.
118 * @param disc A non-null CDDB disc structure.
119 * @param str The search string
120 *
121 * @return The number of matches found or -1 on error.
122 */
123int cddb_search(cddb_conn_t *c, cddb_disc_t *disc, const char *str);
124
125/**
126 * Returns the next match in a CDDB search result set. This function
127 * should be used in conjunction with #cddb_search.
128 *
129 * @param c The CDDB connection structure.
130 * @param disc A non-null CDDB disc structure.
131 */
132int cddb_search_next(cddb_conn_t *c, cddb_disc_t *disc);
133
134/**
135 * Perform a text search in the CDDB database. It uses the album
136 * command implemented on the freedb2.org servers. Either the album
137 * title or artist's name should be filled in, in the disc structure.
138 *
139 * If there are multiple matches then only the first one will be
140 * returned by this function. For other matches you will have to use
141 * the #cddb_album_next function.
142 *
143 * @param c The CDDB connection structure.
144 * @param disc A non-null CDDB disc structure.
145 *
146 * @return The number of matches found or -1 on error.
147 */
148int cddb_album(cddb_conn_t *c, cddb_disc_t *disc);
149
150/**
151 * Returns the next match in a CDDB album result set. This function
152 * should be used in conjunction with #cddb_album.
153 *
154 * @param c The CDDB connection structure.
155 * @param disc A non-null CDDB disc structure.
156 */
157int cddb_album_next(cddb_conn_t *c, cddb_disc_t *disc);
158
159/**
160 * Submit a new or updated disc to the CDDB database. This function
161 * requires that the disc ID, length, category, artist and title of
162 * the provided disc structure are valid. The disc should also
163 * contain a number of tracks and for each track its frame offset on
164 * the CD and title should be valid.
165 *
166 * @param c The CDDB connection structure.
167 * @param disc A non-null CDDB disc structure.
168 */
169int cddb_write(cddb_conn_t *c, cddb_disc_t *disc);
170
171/**
172 * Query the currently configured server for a list of mirrors.
173 * Accessing the list of mirror sites is done with the iterator
174 * functions #cddb_first_site and #cddb_next_site.
175 *
176 * @param c The CDDB connection structure.
177 */
178int cddb_sites(cddb_conn_t *c);
