22#ifndef CDDB_TRACK_H
23#define CDDB_TRACK_H 1
24
25#ifdef __cplusplus
26 extern "C" {
27#endif
28
29
30/**
31 * The CDDB track structure. Contains all information associated with
32 * a single CD track. This structure will be used to populate the
33 * tracks linked list of the cddb_disc_s structure.
34 */
35typedef struct cddb_track_s cddb_track_t;
36
37
38/* --- construction / destruction */
39
40
41/**
42 * Creates a new CDDB track structure.
43 *
44 * @return The CDDB track structure or NULL if memory allocation failed.
45 */
46cddb_track_t *cddb_track_new(void);
47
48/**
49 * Free all resources associated with the given CDDB track structure.
50 * The linked list pointer (next) will not be touched. So you have to
51 * make sure that no other tracks are attached to this one before
52 * calling this function.
53 *
54 * @param track The CDDB track structure.
55 */
56void cddb_track_destroy(cddb_track_t *track);
57
58/**
59 * Creates a clone of the given track.
60 *
61 * @param track The CDDB track structure.
62 */
63cddb_track_t *cddb_track_clone(const cddb_track_t *track);
64
65
66/* --- getters & setters --- */
67
68
69/**
70 * Get the number of this track. This track number starts counting at
71 * 1. If the track is invalid or the track number is not defined -1
72 * will be returned.
73 *
74 * @param track The CDDB track structure.
75 * @return The track number.
76 */
77int cddb_track_get_number(const cddb_track_t *track);
78
79/**
80 * Get the frame offset of this track on the disc. If the track is
81 * invalid -1 will be returned.
82 *
83 * @param track The CDDB track structure.
84 * @return The frame offset.
85 */
86int cddb_track_get_frame_offset(const cddb_track_t *track);
87
88/**
89 * Set the frame offset of this track on the disc.
90 *
91 * @param track The CDDB track structure.
92 * @param offset The frame offset.
93 * @return The frame offset.
94 */
95void cddb_track_set_frame_offset(cddb_track_t *track, int offset);
96
97/**
98 * Get the length of the track in seconds. If the track length is not
99 * defined this routine will try to calculate it using the frame
100 * offsets of the tracks and the total disc length. These
101 * calculations will do no rounding to the nearest second. So it is
102 * possible that the sum off all track lengths does not add up to the
103 * actual disc length. If the length can not be calculated -1 will be
104 * returned.
105 *
106 * @param track The CDDB track structure.
107 * @return The track length.
108 */
109int cddb_track_get_length(cddb_track_t *track);
110
111/**
112 * Set the length of the track. If no frame offset is yet known for
113 * this track, and it is part of a disc, then the frame offset will be
114 * calculated.
115 *
116 * @param track The CDDB track structure.
117 * @param length The track length in seconds.
118 */
119void cddb_track_set_length(cddb_track_t *track, int length);
120
121/**
122 * Get the track title. If the track is invalid or no title is set
123 * for this track then NULL will be returned.
124 *
125 * @param track The CDDB track structure.
126 * @return The track title.
127 */
128const char *cddb_track_get_title(const cddb_track_t *track);
129
130/**
131 * Set the track title. If the track already had a title, then the
132 * memory for that string will be freed. The new title will be copied
133 * into a new chunk of memory. If the given title is NULL, then the
134 * title of the track will be deleted.
135 *
136 * @param track The CDDB track structure.
137 * @param title The new track title.
138 */
139void cddb_track_set_title(cddb_track_t *track, const char *title);
140
141/**
142 * Append to the track title. If the track does not have a title yet,
143 * then a new one will be created from the given string, otherwise
144 * that string will be appended to the existing title.
145 *
146 * @param track The CDDB track structure.
147 * @param title Part of the track title.
148 */
149void cddb_track_append_title(cddb_track_t *track, const char *title);
150
151/**
152 * Get the track artist name. If there is no track artist defined,
153 * the disc artist will be returned. NULL will be returned if neither
154 * is defined.
155 *
156 * @param track The CDDB track structure.
157 */
158const char *cddb_track_get_artist(cddb_track_t *track);
159
160/**
161 * Set the track artist name. If the track already had an artist
162 * name, then the memory for that string will be freed. The new
163 * artist name will be copied into a new chunk of memory. If the given artist
164 * name is NULL, then the artist name of the track will be deleted.
165 *
166 * @param track The CDDB track structure.
167 * @param artist The new track artist name.
168 */
169void cddb_track_set_artist(cddb_track_t *track, const char *artist);
170
171/**
172 * Append to the track artist. If the track does not have an artist
173 * yet, then a new one will be created from the given string,
174 * otherwise that string will be appended to the existing artist.
175 *
176 * @param track The CDDB track structure.
177 * @param artist Part of the artist name.
178 */
179void cddb_track_append_artist(cddb_track_t *track, const char *artist);
180
181/**
182 * Get the extended track data. If no extended data is set for this
183 * track then NULL will be returned.
184 *
185 * @param track The CDDB track structure.
186 * @return The extended data.
187 */
188const char *cddb_track_get_ext_data(cddb_track_t *track);
189
190/**
191 * Set the extended data for the track. If the track already had
192 * extended data, then the memory for that string will be freed. The
193 * new extended data will be copied into a new chunk of memory. If
194 * the given extended data is NULL, then the existing data will be
195 * deleted.
196 *
197 * @param track The CDDB track structure.
198 * @param ext_data The new extended data.
199 */
200void cddb_track_set_ext_data(cddb_track_t *track, const char *ext_data);
201
202/**
203 * Append to the extended track data. If the track does not have an
204 * extended data section yet, then a new one will be created from the
205 * given string, otherwise that string will be appended to the
206 * existing data.
207 *
208 * @param track The CDDB track structure.
209 * @param ext_data Part of the extended track data.
210 */
211void cddb_track_append_ext_data(cddb_track_t *track, const char *ext_data);
212
213
214/* --- miscellaneous */
215
216
217/**
218 * Copy all data from one track to another. Any fields that are
219 * unavailable in the source track structure will not result in a
220 * reset of the same field in the destination track structure; e.g. if
221 * there is no title in the source track, but there is one in the
222 * destination track, then the destination's title will remain
223 * unchanged.
224 *
225 * @param dst The destination CDDB track structure.
226 * @param src The source CDDB track structure.
227 */
228void cddb_track_copy(cddb_track_t *dst, cddb_track_t *src);
229
230/**
231 * Prints information about the track on stdout. This is just a
232 * debugging routine to display the structure's content. It is used
233 * by cddb_disc_print to print the contents of a complete disc.
234 *
235 * @param track The CDDB track structure.
236 */
237void cddb_track_print(cddb_track_t *track);
238
239
240#ifdef __cplusplus
241 }
242#endif
243
244#endif /* CDDB_TRACK_H */
