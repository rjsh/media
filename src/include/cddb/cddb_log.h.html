22#ifndef CDDB_LOH_H
23#define CDDB_LOG_H
24
25#ifdef __cplusplus
26 extern "C" {
27#endif
28
29/**
30 * The different log levels supported by libcddb.
31 */
32typedef enum {
33 CDDB_LOG_DEBUG = 1, /**< Debug-level messages. */
34 CDDB_LOG_INFO, /**< Informational. */
35 CDDB_LOG_WARN, /**< Warning conditions. */
36 CDDB_LOG_ERROR, /**< Error conditions. */
37 CDDB_LOG_CRITICAL, /**< Critical conditions. */
38 CDDB_LOG_NONE = 99 /**< No log messages. */
39} cddb_log_level_t;
40
41
42/**
43 * This type defines the signature of a libcddb log handler. For
44 * every message being logged by libcddb, the handler will receive the
45 * log level and the message string.
46 *
47 * @see cddb_log_set_handler
48 * @see cddb_log_level_t
49 *
50 * @param level The log level.
51 * @param message The log message.
52 */
53typedef void (*cddb_log_handler_t)(cddb_log_level_t level, const char *message);
54
55/**
56 * Set a custom log handler for libcddb. The return value is the log
57 * handler being replaced. If the provided parameter is NULL, then
58 * the handler will be reset to the default handler.
59 *
60 * @see cddb_log_handler_t
61 *
62 * @param new_handler The new log handler.
63 * @return The previous log handler.
64 */
65cddb_log_handler_t cddb_log_set_handler(cddb_log_handler_t new_handler);
66
67/**
68 * Set the minimum log level. This function is only useful in
69 * conjunction with the default log handler. The default log handler
70 * will print any log messages that have a log level equal or higher
71 * than this minimum log level to stderr. By default the minimum log
72 * level is set to CDDB_LOG_WARN. This means that only warning, error
73 * and critical messages will be printed. You can silence the default
74 * log handler by setting the minimum log level to CDDB_LOG_NONE.
75 *
76 * @see cddb_log_level_t
77 *
78 * @param level The minimum log level.
79 */
80void cddb_log_set_level(cddb_log_level_t level);
81
82
83#ifdef __cplusplus
84 }
85#endif
86
87#endif /* CDDB_LOG_H */
