38typedef enum {
39 CACHE_OFF = 0, /**< do not use local CDDB cache, network
40 only */
41 CACHE_ON, /**< use local CDDB cache, if possible */
42 CACHE_ONLY /**< only use local CDDB cache, no network
43 access */
44} cddb_cache_mode_t;
45
46/**
47 * Forward declaration of opaque structure used for character set
48 * conversions.
49 */
50typedef struct cddb_iconv_s *cddb_iconv_t;
51
52/**
53 * An opaque structure for keeping state about the connection to a
54 * CDDB server.
55 */
56typedef struct cddb_conn_s cddb_conn_t;
57
58/**
59 * Which fields to use for the full text search is defined by one or
60 * more of the constants below.
61 */
62typedef enum {
63 SEARCH_NONE = 0, /**< no fields */
64 SEARCH_ARTIST = 1, /**< artist name field */
65 SEARCH_TITLE = 2, /**< disc title field */
66 SEARCH_TRACK = 4, /**< track title field */
67 SEARCH_OTHER = 8, /**< other fields */
68 SEARCH_ALL = ~0, /**< all fields */
69} cddb_search_t;
70
71/**
72 * Macro to be used for building the category search bit-string from
73 * the values of #cddb_cat_t.
74 */
75#define SEARCHCAT(c) (1 << (c))

81/**
82 * Creates a new CDDB connection structure. This structure will have
83 * to be passed to all libcddb functions. Default values will be used
84 * for the connection parameters allowing it to contact the CDDB
85 * server at freedb.org.
86 *
87 * @return The CDDB connection structure or NULL if something went wrong.
88 */
89cddb_conn_t *cddb_new(void);
90
91/**
92 * Free all resources associated with the given CDDB connection
93 * structure.
94 */
95void cddb_destroy(cddb_conn_t *c);
96

98/* --- getters & setters --- */
99
100
101/**
102 * Set the character set. By default the FreeDB server uses UTF-8 when
103 * providing CD data. When a character set is defined with this function
104 * any strings retrieved from or sent to the server will automatically be
105 * converted.
106 *
107 * @param c The connection structure.
108 * @param cs The character set that will be used.
109 * @return False if the specified character set is unknown, or no conversion
110 * from/to UTF-8 is available. True otherwise.
111 */
112int cddb_set_charset(cddb_conn_t *c, const char *cs);
113
114/**
115 * Change the size of the internal buffer.
116 *
117 * @param c The connection structure.
118 * @param size The new buffer size.
119 */
120void cddb_set_buf_size(cddb_conn_t *c, unsigned int size);
121
122/**
123 * Set all server details in one go through the use of a site structure. This
124 * function initializzes the server address, port, protocol and query path in
125 * case of HTTP.
126 *
127 * @see cddb_sites
128 * @see cddb_first_site
129 * @see cddb_next_site
130 *
131 * @param c The connection structure.
132 * @param site The site to use.
133 * @return Error code: CDDB_ERR_OK or CDDB_ERR_INVALID.
134 */
135cddb_error_t cddb_set_site(cddb_conn_t *c, const cddb_site_t *site);
136
137/**
138 * Get the host name of the CDDB server that is currently being used.
139 *
140 * @see cddb_set_server_name
141 *
142 * @param c The connection structure.
143 * @return The server host name.
144 */
145const char *cddb_get_server_name(const cddb_conn_t *c);
146
147/**
148 * Set the host name of the CDDB server. The default value for the
149 * server is 'freedb.org'.
150 *
151 * @see cddb_get_server_name
152 *
153 * @param c The connection structure.
154 * @param server The server host name.
155 */
156void cddb_set_server_name(cddb_conn_t *c, const char *server);
157
158/**
159 * Get the port of the CDDB server that is currently being used.
160 *
161 * @see cddb_set_server_port
162 *
163 * @param c The connection structure.
164 * @return The server port.
165 */
166unsigned int cddb_get_server_port(const cddb_conn_t *c);
167
168/**
169 * Set the port of the CDDB server. The default value is 888.
170 *
171 * @see cddb_get_server_port
172 *
173 * @param c The connection structure.
174 * @param port The server port.
175 */
176void cddb_set_server_port(cddb_conn_t *c, int port);
177
178/**
179 * Get the network time out value (in seconds).

183 * @param c The connection structure.
184 * @return The current time out in seconds.

186unsigned int cddb_get_timeout(const cddb_conn_t *c);

188/**
189 * Set the network time out value (in seconds). The default is 10

192 * @see cddb_get_timeout
193 *
194 * @param c The connection structure.

197void cddb_set_timeout(cddb_conn_t *c, unsigned int t);

199/**
200 * Get the URL path for querying a CDDB server through HTTP.

218void cddb_set_http_path_query(cddb_conn_t *c, const char *path);
219
220/**
221 * Get the URL path for submitting to a CDDB server through HTTP.
222 *
223 * @see cddb_set_http_path_submit
224 *
225 * @param c The connection structure.
226 * @return The URL path.

228const char *cddb_get_http_path_submit(const cddb_conn_t *c);
229
230/**
231 * Set the URL path for submitting to a CDDB server through HTTP. The

233 *
234 * @see cddb_get_http_path_submit
235 *
236 * @param c The connection structure.
237 * @param path The URL path.

239void cddb_set_http_path_submit(cddb_conn_t *c, const char *path);

241/**
242 * Returns true if the HTTP protocol is currently enabled and false if
243 * CDDBP is enabled.
244 *
245 * @see cddb_http_enable
246 * @see cddb_http_disable
247 *
248 * @param c The CDDB connection structure.
249 * @return True or false.
250 */
251unsigned int cddb_is_http_enabled(const cddb_conn_t *c);
252
253/**
254 * Enable HTTP tunneling to connect to the CDDB server. By default
255 * this option is disabled.
256 *
257 * @see cddb_is_http_enabled
258 * @see cddb_http_disable
259 *
260 * @param c The CDDB connection structure.
261 */
262void cddb_http_enable(cddb_conn_t *c);
263
264/**
265 * Disable HTTP tunneling to connect to the CDDB server. By default this
266 * option is disabled.
267 *
268 * @see cddb_is_http_enabled
269 * @see cddb_http_enable
270 *
271 * @param c The CDDB connection structure.
272 */
273void cddb_http_disable(cddb_conn_t *c);
274
275/**
276 * Returns true if the proxy support is currently enabled and false if
277 * it is not. This fucntion does not check whether HTTP is enabled.
278 * So it is possible that true will be returned while in reality the
279 * CDDBP protocol is being used (no proxy support).
280 *
281 * @see cddb_http_proxy_enable
282 * @see cddb_http_proxy_disable
283 *
284 * @param c The CDDB connection structure.
285 * @return True or false.
286 */
287unsigned int cddb_is_http_proxy_enabled(const cddb_conn_t *c);
288
289/**
290 * Enable HTTP tunneling through an HTTP proxy server to connect to
291 * the CDDB server. The usage of an HTTP proxy implies normal HTTP
292 * tunneling instead of connecting directly to the CDDB server. By
293 * default this option is disabled.
294 *
295 * @see cddb_is_http_proxy_enabled
296 * @see cddb_http_proxy_disable
297 *
298 * @param c The CDDB connection structure.
299 */
300void cddb_http_proxy_enable(cddb_conn_t *c);
301
302/**
303 * Disable HTTP tunneling through an HTTP proxy server to connect to
304 * the CDDB server. By default this option is disabled.
305 *
306 * @see cddb_is_http_proxy_enabled
307 * @see cddb_http_proxy_enable
308 *
309 * @param c The CDDB connection structure.
310 */
311void cddb_http_proxy_disable(cddb_conn_t *c);
312
313/**
314 * Get the host name of the HTTP proxy server.
315 *
316 * @see cddb_set_http_proxy_server_name
317 *
318 * @param c The connection structure.
319 * @return The proxy server host name.
320 */
321const char *cddb_get_http_proxy_server_name(const cddb_conn_t *c);
322
323/**
324 * Set the host name of the HTTP proxy server. There is no default
325 * value.
326 *
327 * @see cddb_get_http_proxy_server_name

329 * @param c The connection structure.
330 * @param server The server host name.
331 */
332void cddb_set_http_proxy_server_name(cddb_conn_t *c, const char *server);

334/**
335 * Get the port of the HTTP proxy server.
336 *
337 * @see cddb_set_http_proxy_server_port

339 * @param c The connection structure.
340 * @return The proxy server port.

342unsigned int cddb_get_http_proxy_server_port(const cddb_conn_t *c);
343
344/**
345 * Set the port of the HTTP proxy server. The default value is 8080.

349 * @param c The connection structure.

352void cddb_set_http_proxy_server_port(cddb_conn_t *c, int port);

371/**
372 * Set the HTTP proxy password which is used when Basic Authentication

386const char *cddb_get_http_proxy_password(const cddb_conn_t *c);

388/**
389 * Set the HTTP proxy user name and password in one go. These
390 * credentials are used when Basic Authentication is required. The
391 * advantage of using this function over setting the user name and
392 * password seperately is that the cleartext user name and password
393 * are not kept in memory longer than needed.
394 *
395 * @param c The connection structure.
396 * @param username The user name.
397 * @param passwd The password.
398 */
399void cddb_set_http_proxy_credentials(cddb_conn_t* c,
400 const char *username, const char* passwd);

402/**
403 * Get the error number returned by the last libcddb command.
404 *
405 * @param c The CDDB connection structure.
406 * @return The error number.
407 */
408cddb_error_t cddb_errno(const cddb_conn_t *c);
409
410/**
411 * Set the name and version of the client program overwriting the
412 * previous values. This function will make a copy of the provided
413 * strings. The defaults are 'libcddb' and the version number of the
414 * libcddb library in use. Both parameters must be valid strings. If
415 * any of teh strings is NULL, this fucntion will return without
416 * changing anything.
417 *
418 * @param c The connection structure.
419 * @param cname The name of the client program.
420 * @param cversion The version number of the client program.
421 */
422void cddb_set_client(cddb_conn_t *c, const char *cname, const char *cversion);
423
424/**
425 * Sets the user name and host name of the local machine. This
426 * function will parse out the user name and host name from the e-mail
427 * address.
428 *
429 * @param c The connection structure.
430 * @param email The e-mail address of the user.
431 */
432int cddb_set_email_address(cddb_conn_t *c, const char *email);
433
434/**
435 * Returns the current cache mode. This can be either on, off or
436 * cache only.
437 *
438 * @see CACHE_ON
439 * @see CACHE_ONLY
440 * @see CACHE_OFF
441 * @see cddb_cache_enable
442 * @see cddb_cache_only
443 * @see cddb_cache_disable
444 *
445 * @param c The connection structure.
446 */
447cddb_cache_mode_t cddb_cache_mode(const cddb_conn_t *c);
448
449/**
450 * Enable caching of CDDB entries locally. Caching is enabled by
451 * default. The cache directory can be changed with the
452 * cddb_cache_set_dir function.
453 *
454 * @see cddb_cache_mode
455 * @see cddb_cache_disable
456 * @see cddb_cache_only
457 *
458 * @param c The connection structure.
459 */
460void cddb_cache_enable(cddb_conn_t *c);
461
462/**
463 * Only use the local CDDB cache. Never contact a server to retrieve
464 * any data. The cache directory can be changed with the
465 * cddb_cache_set_dir function.
466 *
467 * @see cddb_cache_mode
468 * @see cddb_cache_enable
469 * @see cddb_cache_disable
470 *
471 * @param c The connection structure.
472 */
473void cddb_cache_only(cddb_conn_t *c);
474
475/**
476 * Disable caching of CDDB entries locally. All data will be fetched
477 * from a CDDB server everytime and the retrieved data will not be
478 * cached locally.
479 *
480 * @see cddb_cache_mode
481 * @see cddb_cache_enable
482 * @see cddb_cache_only
483 *
484 * @param c The connection structure.
485 */
486void cddb_cache_disable(cddb_conn_t *c);
487
488/**
489 * Return the directory currently being used for caching.
490 *
491 * @see cddb_cache_set_dir
492 *
493 * @param c The connection structure.
494 * @return The directory being used for caching.
495 */
496const char *cddb_cache_get_dir(const cddb_conn_t *c);
497
498/**
499 * Change the directory used for caching CDDB entries locally. The
500 * default location of the cached entries is a subdirectory
501 * (.cddbslave) of the user's home directory. If the first character
502 * of the directory is '~', then it will be expanded to the contents
503 * of $HOME.
504 *
505 * @see cddb_cache_get_dir
506 *
507 * @param c The connection structure.
508 * @param dir The directory to use for caching.
509 */
510int cddb_cache_set_dir(cddb_conn_t *c, const char *dir);
511
512/**
513 * Retrieve the first CDDB mirror site.
514 *
515 * @param c The connection structure.
516 * @return The first mirror site or NULL if not found.

518const cddb_site_t *cddb_first_site(cddb_conn_t *c);
519
520/**
521 * Retrieve the next CDDB mirror site.
522 *
523 * @param c The connection structure.

526const cddb_site_t *cddb_next_site(cddb_conn_t *c);
527
528/**
529 * Set the bit-string specifying which fields to examine when
530 * performing a text search. By default only the artist and disc
531 * title fields are searched.
532 *
533 * @param c The connection structure.
534 * @param fields A bitwise ORed set of values from #cddb_search_t.
535 */
536void cddb_search_set_fields(cddb_conn_t *c, unsigned int fields);
537
538/**
539 * Set the bit-string specifying which categories to examine when
540 * performing a text search. The #SEARCHCAT macro needs to be used to
541 * build the actual bit-string from individual categories. The
542 * #cddb_search_t values #SEARCH_NONE and #SEARCH_ALL are also valid.
543 * The example below shows some possible combinations. By default all
544 * categories are searched.
545 *
546 * @code
547 * unsigned int cats = SEARCHCAT(CDDB_CAT_ROCK) | SEARCHCAT(CDDB_CAT_MISC);
548 * unsigned int cats = SEARCH_ALL;
549 * unsigned int cats = SEARCH_NONE;
550 * @endcode
551 *
552 * @param c The connection structure.
553 * @param cats A bitwise ORed set of values from #SEARCHCAT(#cddb_cat_t).
554 */
555void cddb_search_set_categories(cddb_conn_t *c, unsigned int cats);
