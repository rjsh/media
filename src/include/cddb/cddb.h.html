47#define BIT(n) (1 << n)
48
49/**
50 * An enumeration of flags that influence the behaviour of the
51 * library. You can set or reset these flags using the
52 * #libcddb_set_flags and #libcddb_reset_flags functions.
53 */
54typedef enum {
55 CDDB_F_EMPTY_STR = BIT(0), /**< never return NULL pointer strings
56 (default), return an empty string
57 instead */
58 CDDB_F_NO_TRACK_ARTIST = BIT(1), /**< do not return the disc artist as the
59 track artist (default), return NULL
60 instead */
61} cddb_flag_t;
62
63/**
64 * Initializes the library. This is used to setup any globally used
65 * variables. The first time you create a new CDDB connection structure
66 * the library will automatically initialize itself. So, there is no
67 * need to explicitly call this function.
68 */
69void libcddb_init(void);
70
71/**
72 * Frees up any global (cross connection) resources. You should call
73 * this function before terminating your program. Using any library
74 * calls after shutting down are bound to give problems.
75 */
76void libcddb_shutdown(void);
77
78/**
79 * Set one or more flags that influence the library behvaiour
80 *
81 * @param flags A bitwise ORed set of values from #cddb_flag_t.
82 */
83void libcddb_set_flags(unsigned int flags);
84
85/**
86 * Reset one or more flags that influence the library behvaiour
87 *
88 * @param flags A bitwise ORed set of values from #cddb_flag_t.
89 */
90void libcddb_reset_flags(unsigned int flags);
