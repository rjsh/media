41/*****************************************************************************
42 * dvbpsi_eit_event_t
43 *****************************************************************************/
44/*!
45 * \struct dvbpsi_eit_event_s
46 * \brief EIT service description structure.
47 *
48 * This structure is used to store a decoded EIT event description.
49 * (ETSI EN 300 468 V1.5.1 section 5.2.4).
50 */
51/*!
52 * \typedef struct dvbpsi_eit_service_s dvbpsi_eit_event_t
53 * \brief dvbpsi_eit_event_t type definition.
54 */
55typedef struct dvbpsi_eit_event_s
56{
57 uint16_t i_event_id; /*!< event_id */
58 uint64_t i_start_time; /*!< start_time */
59 uint32_t i_duration; /*!< duration */
60 uint8_t i_running_status; /*!< Running status */
61 bool b_free_ca; /*!< Free CA mode flag */
62 uint16_t i_descriptors_length; /*!< Descriptors loop
63 length */
64 dvbpsi_descriptor_t * p_first_descriptor; /*!< First of the following
65 DVB descriptors */
66
67 struct dvbpsi_eit_event_s * p_next; /*!< next element of
68 the list */
69
70} dvbpsi_eit_event_t;
71
72/*****************************************************************************
73 * dvbpsi_eit_t
74 *****************************************************************************/
75/*!
76 * \struct dvbpsi_eit_s
77 * \brief EIT structure.
78 *
79 * This structure is used to store a decoded EIT.
80 * (ETSI EN 300 468 V1.5.1 section 5.2.4).
81 */
82/*!
83 * \typedef struct dvbpsi_eit_s dvbpsi_eit_t
84 * \brief dvbpsi_eit_t type definition.
85 */
86typedef struct dvbpsi_eit_s
87{
88 uint8_t i_table_id; /*!< table id */
89 uint16_t i_extension; /*!< subtable id */
90
91 uint8_t i_version; /*!< version_number */
92 bool b_current_next; /*!< current_next_indicator */
93 uint16_t i_ts_id; /*!< transport stream id */
94 uint16_t i_network_id; /*!< original network id */
95 uint8_t i_segment_last_section_number; /*!< segment last section number */
96 uint8_t i_last_table_id; /*!< last table id */
97
98 dvbpsi_eit_event_t *p_first_event; /*!< event information list */
99
100} dvbpsi_eit_t;
101

105/*!
106 * \typedef void (* dvbpsi_eit_callback)(void* p_cb_data, dvbpsi_eit_t* p_new_eit)
107 * \brief Callback type definition.
108 */
109typedef void (* dvbpsi_eit_callback)(void* p_cb_data, dvbpsi_eit_t* p_new_eit);

111/*****************************************************************************
112 * dvbpsi_AttachEIT
113 *****************************************************************************/
114/*!
115 * \fn bool dvbpsi_eit_attach(dvbpsi_t *p_dvbpsi, uint8_t i_table_id,
116 uint16_t i_extension, dvbpsi_eit_callback pf_callback,
117 void* p_cb_data)
118 * \brief Creation and initialization of a EIT decoder.
119 * \param p_dvbpsi pointer to Subtable demultiplexor to which the EIT decoder is attached.
120 * \param i_table_id Table ID, 0x4E, 0x4F, or 0x50-0x6F.
121 * \param i_extension Table ID extension, here service ID.
122 * \param pf_callback function to call back on new EIT.
123 * \param p_cb_data private data given in argument to the callback.
124 * \return true on success, false on failure
125 */
126bool dvbpsi_eit_attach(dvbpsi_t *p_dvbpsi, uint8_t i_table_id, uint16_t i_extension,
127 dvbpsi_eit_callback pf_callback, void* p_cb_data);
128
129/*****************************************************************************
130 * dvbpsi_eit_detach
131 *****************************************************************************/
132/*!
133 * \fn void dvbpsi_eit_detach(dvbpsi_t *p_dvbpsi, uint8_t i_table_id,
134 uint16_t i_extension)
135 * \brief Destroy a EIT decoder.
136 * \param p_dvbpsi dvbpsi handle pointing to Subtable demultiplexor to which the
137 eit decoder is attached.
138 * \param i_table_id Table ID, 0x4E, 0x4F, or 0x50-0x6F.
139 * \param i_extension Table ID extension, here service ID.
140 * \return nothing.
141 */
142void dvbpsi_eit_detach(dvbpsi_t *p_dvbpsi, uint8_t i_table_id, uint16_t i_extension);

156 * \param i_extension Table ID extension, here service ID.
157 * \param i_version EIT version
158 * \param b_current_next current next indicator
159 * \param i_ts_id transport stream ID
160 * \param i_network_id original network id
161 * \param i_segment_last_section_number segment_last_section_number
162 * \param i_last_table_id i_last_table_id
163 * \return nothing.
164 */
165void dvbpsi_eit_init(dvbpsi_eit_t* p_eit, uint8_t i_table_id,
166 uint16_t i_extension, uint8_t i_version,
167 bool b_current_next, uint16_t i_ts_id, uint16_t i_network_id,
168 uint8_t i_segment_last_section_number,
169 uint8_t i_last_table_id);

171/*!
172 * \fn dvbpsi_eit_t* dvbpsi_eit_new(uint8_t i_table_id, uint16_t i_extension,
173 uint8_t i_version, bool b_current_next,
174 uint16_t i_ts_id, uint16_t i_network_id, uint8_t i_segment_last_section_number,

185 * \return p_eit pointer to the EIT structure

188 uint8_t i_version, bool b_current_next,
189 uint16_t i_ts_id, uint16_t i_network_id,
190 uint8_t i_segment_last_section_number,
191 uint8_t i_last_table_id);
