33/*****************************************************************************
34 * dvbpsi_data_broadcast_id_dr_s
35 *****************************************************************************/
36/*!
37 * \struct dvbpsi_data_broadcast_id_dr_s
38 * \brief Data Broadcast id Descriptor
39 *
40 * This structure is used to store a decoded Data Broadcast id descriptor.
41 */
42/*!
43 * \typedef struct dvbpsi_data_broadcast_id_dr_s dvbpsi_data_broadcast_id_dr_t
44 * \brief dvbpsi_data_broadcast_id_dr_t type definition.
45 */
46typedef struct dvbpsi_data_broadcast_id_dr_s
47{
48 uint16_t i_data_broadcast_id; /*!< broadcast identifier */
49 uint8_t i_id_selector_len; /*!< length of selector data in bytes */
50 uint8_t *p_id_selector; /*!< pointer to selector data. Memory is allocated
51 right after sizeof struct, when freeing this
52 struct the private data is freed at the same time. */
53} dvbpsi_data_broadcast_id_dr_t;
54
55/*****************************************************************************
56 * dvbpsi_DecodeDataBroadcastIdDr
57 *****************************************************************************/
58/*!
59 * \fn dvbpsi_data_broadcast_id_dr_t *dvbpsi_DecodeDataBroadcastIdDr(
60 * dvbpsi_descriptor_t *p_descriptor)
61 * \brief Decode a Data broadcast id descriptor (tag 0x66)
62 * \param p_descriptor Raw descriptor to decode.
63 * \return NULL if the descriptor could not be decoded or a pointer to a
64 * dvbpsi_data_broadcast_id_dr_t structure.
65 */
66dvbpsi_data_broadcast_id_dr_t *dvbpsi_DecodeDataBroadcastIdDr(dvbpsi_descriptor_t *p_descriptor);
