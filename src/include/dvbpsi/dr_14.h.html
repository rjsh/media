33/*****************************************************************************
34 * dvbpsi_association_tag_dr_s
35 *****************************************************************************/
36/*!
37 * \struct dvbpsi_association_tag_dr_s
38 * \brief Data Broadcast id Descriptor
39 *
40 * This structure is used to store a decoded Association Tag descriptor.
41 */
42/*!
43 * \typedef struct dvbpsi_association_tag_dr_s dvbpsi_association_tag_dr_t
44 * \brief dvbpsi_association_tag_dr_t type definition.
45 */
46typedef struct dvbpsi_association_tag_dr_s
47{
48 uint16_t i_tag; /*!< association tag identifier */
49 uint16_t i_use; /*!< indicator if association tag identifier is in use */
50 uint8_t i_selector_len; /*!< length of selector data in bytes */
51 uint8_t *p_selector; /*!< pointer to selector. Memory is allocated
52 right after sizeof struct, when freeing this
53 struct the private data is freed at the same time. */
54 uint8_t i_private_data_len; /*!< length of private data segment in bytes */
55 uint8_t *p_private_data; /*!< pointer to private data. Memory is allocated
56 right after sizeof struct, when freeing this
57 struct the private data is freed at the same time. */
58} dvbpsi_association_tag_dr_t;
59
60/*****************************************************************************
61 * dvbpsi_DecodeAssociationTagDr
62 *****************************************************************************/
63/*!
64 * \fn dvbpsi_association_tag_dr_t *dvbpsi_DecodeAssociationTagDr(
65 * dvbpsi_descriptor_t *p_descriptor)
66 * \brief Decode a Association Tag descriptor (tag 0x14)
67 * \param p_descriptor Raw descriptor to decode.
68 * \return NULL if the descriptor could not be decoded or a pointer to a
69 * dvbpsi_association_tag_dr_t structure.
70 */
71dvbpsi_association_tag_dr_t *dvbpsi_DecodeAssociationTagDr(dvbpsi_descriptor_t *p_descriptor);
