24/*!
25 * \file <dr_0a.h>
26 * \author Arnaud de Bossoreille de Ribou <bozo@via.ecp.fr>
27 * \brief Application interface for the MPEG 2 "ISO 639 language"
28 * descriptor decoder and generator.
29 *
30 * Application interface for the MPEG 2 "ISO 639 language" descriptor
31 * decoder and generator. This descriptor's definition can be found in
32 * ISO/IEC 13818-1 section 2.6.18.
33 */
34
35#ifndef _DVBPSI_DR_0A_H_
36#define _DVBPSI_DR_0A_H_
37
38#ifdef __cplusplus
39extern "C" {
40#endif
41
42#define DR_0A_API_VER 2 /*!< descriptor 0a API version */
43
44/*****************************************************************************
45 * dvbpsi_iso639_dr_t
46 *****************************************************************************/
47/*!
48 * \struct dvbpsi_iso639_dr_s
49 * \brief "ISO 639 language" descriptor structure.
50 *
51 * This structure is used to store a decoded "ISO 639 language"
52 * descriptor. (ISO/IEC 13818-1 section 2.6.18).
53 */
54/*!
55 * \typedef struct dvbpsi_iso639_dr_s dvbpsi_iso639_dr_t
56 * \brief dvbpsi_iso639_dr_t type definition.
57 */
58typedef struct dvbpsi_iso639_dr_s
59{
60 uint8_t i_code_count; /*!< length of the i_iso_639_code
61 array */
62 struct {
63 iso_639_language_code_t iso_639_code; /*!< ISO_639_language_code */
64 uint8_t i_audio_type; /*!< audio_type */
65 } code[64]; /*!< list of ISO 639 codes */
66
67} dvbpsi_iso639_dr_t;
68
69
70/*****************************************************************************
71 * dvbpsi_DecodeISO639Dr
72 *****************************************************************************/
73/*!
74 * \fn dvbpsi_iso639_dr_t * dvbpsi_DecodeISO639Dr(
75 dvbpsi_descriptor_t * p_descriptor)
76 * \brief "ISO 639 language" descriptor decoder.
77 * \param p_descriptor pointer to the descriptor structure
78 * \return a pointer to a new "ISO 639 language" descriptor structure which
79 * contains the decoded data.
80 */
81dvbpsi_iso639_dr_t* dvbpsi_DecodeISO639Dr(dvbpsi_descriptor_t * p_descriptor);
82
83
84/*****************************************************************************
85 * dvbpsi_GenISO639Dr
86 *****************************************************************************/
87/*!
88 * \fn dvbpsi_descriptor_t * dvbpsi_GenISO639Dr(
89 dvbpsi_iso639_dr_t * p_decoded, bool b_duplicate)
90 * \brief "ISO 639 language" descriptor generator.
91 * \param p_decoded pointer to a decoded "ISO 639 language" descriptor
92 * structure
93 * \param b_duplicate if true then duplicate the p_decoded structure into
94 * the descriptor
95 * \return a pointer to a new descriptor structure which contains encoded data.
96 */
97dvbpsi_descriptor_t * dvbpsi_GenISO639Dr(dvbpsi_iso639_dr_t * p_decoded,
98 bool b_duplicate);
99
100
101#ifdef __cplusplus
102};
103#endif
104
