41/*****************************************************************************
42 * dvbpsi_sdt_service_t
43 *****************************************************************************/
44/*!
45 * \struct dvbpsi_sdt_service_s
46 * \brief SDT service description structure.
47 *
48 * This structure is used to store a decoded SDT service description.
49 * (ETSI EN 300 468 V1.4.1 section 5.2.3).
50 */
51/*!
52 * \typedef struct dvbpsi_sdt_service_s dvbpsi_sdt_service_t
53 * \brief dvbpsi_sdt_service_t type definition.
54 */
55typedef struct dvbpsi_sdt_service_s
56{
57 uint16_t i_service_id; /*!< service_id */
58 bool b_eit_schedule; /*!< EIT schedule flag */
59 bool b_eit_present; /*!< EIT present/following
60 flag */
61 uint8_t i_running_status; /*!< Running status */
62 bool b_free_ca; /*!< Free CA mode flag */
63 uint16_t i_descriptors_length; /*!< Descriptors loop
64 length */
65 dvbpsi_descriptor_t * p_first_descriptor; /*!< First of the following
66 DVB descriptors */
67
68 struct dvbpsi_sdt_service_s * p_next; /*!< next element of
69 the list */
70
71} dvbpsi_sdt_service_t;
72
73/*****************************************************************************
74 * dvbpsi_sdt_t
75 *****************************************************************************/
76/*!
77 * \struct dvbpsi_sdt_s
78 * \brief SDT structure.
79 *
80 * This structure is used to store a decoded SDT.
81 * (ETSI EN 300 468 V1.4.1 section 5.2.3).
82 */
83/*!
84 * \typedef struct dvbpsi_sdt_s dvbpsi_sdt_t
85 * \brief dvbpsi_sdt_t type definition.
86 */
87typedef struct dvbpsi_sdt_s
88{
89 /* PSI table members */
90 uint8_t i_table_id; /*!< table id */
91 uint16_t i_extension; /*!< subtable id, here transport_stream_id */
92
93 /* Table specific */
94 uint8_t i_version; /*!< version_number */
95 bool b_current_next; /*!< current_next_indicator */
96 uint16_t i_network_id; /*!< original network id */
97
98 dvbpsi_sdt_service_t * p_first_service; /*!< service description
99 list */
100
101} dvbpsi_sdt_t;
102
103/*****************************************************************************
104 * dvbpsi_sdt_callback
105 *****************************************************************************/
106/*!
107 * \typedef void (* dvbpsi_sdt_callback)(void* p_cb_data,
108 dvbpsi_sdt_t* p_new_sdt)
109 * \brief Callback type definition.
110 */
111typedef void (* dvbpsi_sdt_callback)(void* p_cb_data, dvbpsi_sdt_t* p_new_sdt);

113/*****************************************************************************
114 * dvbpsi_sdt_attach
115 *****************************************************************************/
116/*!
117 * \fn bool dvbpsi_sdt_attach(dvbpsi_t *p_dvbpsi, uint8_t i_table_id,
118 uint16_t i_extension, dvbpsi_sdt_callback pf_callback,
119 void* p_cb_data)
120 * \brief Creation and initialization of a SDT decoder. It is attached to p_dvbpsi.
121 * \param p_dvbpsi pointer to dvbpsi to hold decoder/demuxer structure
122 * \param i_table_id Table ID, 0x42 or 0x46.
123 * \param i_extension Table ID extension, here TS ID.
124 * \param pf_callback function to call back on new SDT.
125 * \param p_cb_data private data given in argument to the callback.
126 * \return true on success, false on failure
127 */
128bool dvbpsi_sdt_attach(dvbpsi_t *p_dvbpsi, uint8_t i_table_id, uint16_t i_extension,
129 dvbpsi_sdt_callback pf_callback, void* p_cb_data);
130
131/*****************************************************************************
132 * dvbpsi_sdt_detach
133 *****************************************************************************/
134/*!
135 * \fn void dvbpsi_sdt_detach(dvbpsi_t *p_dvbpsi, uint8_t i_table_id,
136 uint16_t i_extension)
137 * \brief Destroy a SDT decoder.
138 * \param p_dvbpsi pointer holding decoder/demuxer structure
139 * \param i_table_id Table ID, 0x42 or 0x46.
140 * \param i_extension Table ID extension, here TS ID.
141 * \return nothing.
142 */
143void dvbpsi_sdt_detach(dvbpsi_t *p_dvbpsi, uint8_t i_table_id, uint16_t i_extension);

149 * \fn void dvbpsi_sdt_init(dvbpsi_sdt_t* p_sdt, uint8_t i_table_id, uint16_t i_extension,
150 uint8_t i_version, bool b_current_next, uint16_t i_network_id)
151 * \brief Initialize a user-allocated dvbpsi_sdt_t structure.

156 * \param b_current_next current next indicator
157 * \param i_network_id original network id

161 uint8_t i_version, bool b_current_next, uint16_t i_network_id);
162
