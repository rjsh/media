23/*!
24 * \file <dr_49.h>

26 * \brief Application interface for the DVB "country availability"
27 * descriptor decoder and generator.
28 *
29 * Application interface for the DVB "country availability" descriptor
30 * decoder and generator. This descriptor's definition can be found in
31 * ETSI EN 300 468 section 6.2.10.
32 */
33
34#ifndef DR_49_H_
35#define DR_49_H_
36
37#ifdef __cplusplus
38extern "C" {
39#endif
40
41
42/*****************************************************************************
43 * dvbpsi_country_availability_dr_t
44 *****************************************************************************/
45/*!
46 * \struct dvbpsi_country_availability_dr_t
47 * \brief "country availability" descriptor structure.
48 *
49 * This structure is used to store a decoded "country availability"
50 * descriptor. (ETSI EN 300 468 section 6.2.10).
51 */
52/*!
53 * \typedef struct dvbpsi_country_availability_dr_s dvbpsi_country_availability_dr_t
54 * \brief dvbpsi_country_availability_dr_t type definition.
55 */
56/*!
57 * \struct dvbpsi_country_availability_dr_s
58 * \brief dvbpsi_country_availability_dr_s type definition @see dvbpsi_country_availability_dr_t
59 */
60typedef struct dvbpsi_country_availability_dr_s
61{
62 bool b_country_availability_flag; /*!< country availability flag */
63 uint8_t i_code_count; /*!< length of the i_iso_639_code
64 array */
65 struct {
66 iso_639_language_code_t iso_639_code; /*!< ISO_639 language code */
67 } code[84]; /*!< ISO_639_language_code array */
68
69} dvbpsi_country_availability_dr_t;
70
71/*****************************************************************************
72 * dvbpsi_DecodeCountryAvailabilityDr
73 *****************************************************************************/
74/*!
75 * \fn dvbpsi_country_availability_dr_t * dvbpsi_DecodeCountryAvailability(
76 dvbpsi_descriptor_t * p_descriptor)
77 * \brief "country availability" descriptor decoder.
78 * \param p_descriptor pointer to the descriptor structure
79 * \return a pointer to a new "country availability" descriptor structure
80 * which contains the decoded data.
81 */
82dvbpsi_country_availability_dr_t* dvbpsi_DecodeCountryAvailability(
83 dvbpsi_descriptor_t * p_descriptor);
84
85/*****************************************************************************
86 * dvbpsi_GenCountryAvailabilityDr
87 *****************************************************************************/
88/*!
89 * \fn dvbpsi_descriptor_t * dvbpsi_GenCountryAvailabilityDr(
90 dvbpsi_country_availability_dr_t * p_decoded,
91 bool b_duplicate)
92 * \brief "country availability" descriptor generator.
93 * \param p_decoded pointer to a decoded "country availability" descriptor
94 * structure
95 * \param b_duplicate if true then duplicate the p_decoded structure into
96 * the descriptor
97 * \return a pointer to a new descriptor structure which contains encoded data.
98 */
99dvbpsi_descriptor_t * dvbpsi_GenCountryAvailabilityDr(
100 dvbpsi_country_availability_dr_t * p_decoded,
101 bool b_duplicate);
102
103#ifdef __cplusplus
104};
105#endif
106
107#else
