24/*!
25 * \file dr_76.h

29
30#ifndef _DR_76_H
31#define _DR_76_H
32
33#ifdef __cplusplus
34extern "C" {
35#endif
36
37/*!
38 * \def CRID_TYPE_UNDEFINED
39 * \brief Content Resource Identifier Descriptor for undefined content
40 *
41 * \def CRID_TYPE_CONTENT
42 * \brief Content Resource Identifier Descriptor for type content
43 *
44 * \def CRID_TYPE_SERIES
45 * \brief Content Resource Identifier Descriptor for type series content
46 *
47 * \def CRID_TYPE_RECOMMENDATION
48 * \brief Content Resource Identifier Descriptor for type recommended content
49 */
50#define CRID_TYPE_UNDEFINED 0
51#define CRID_TYPE_CONTENT 1
52#define CRID_TYPE_SERIES 2
53#define CRID_TYPE_RECOMMENDATION 3
54
55/*!
56 * \def CRID_LOCATION_DESCRIPTOR
57 * \brief Content Resource Identifier Descriptor for location
58 *
59 * \def CRID_LOCATION_CIT
60 * \brief Content Resource Identifier Descriptor for CIT(?)
61 */
62#define CRID_LOCATION_DESCRIPTOR 0
63#define CRID_LOCATION_CIT 1
64
65/*****************************************************************************
66 * dvbpsi_lcn_entry_t
67 *****************************************************************************/
68/*!
69 * \struct dvbpsi_crid_entry_s
70 * \brief CRID Entry
71 *
72 * This structure is used to store a decoded CRID entry.
73 */
74/*!
75 * \typedef struct dvbpsi_crid_entry_s dvbpsi_crid_entry_t
76 * \brief dvbpsi_crid_entry_t type definition.
77 */
78typedef struct dvbpsi_crid_entry_s
79{
80 uint8_t i_type; /*!< content type */
81 uint8_t i_location; /*!< content location */
82 union
83 {
84 uint8_t path[253]; /*!< content path */
85 uint16_t ref; /*!< content reference */
86 } value; /*!< content specific value */
87}dvbpsi_crid_entry_t;
88
89/*****************************************************************************
90 * dvbpsi_content_id_dr_s
91 *****************************************************************************/
92/*!
93 * \struct dvbpsi_content_id_dr_s
94 * \brief Content Identifier Descriptor
95 *
96 * This structure is used to store a decoded Content Identifier descriptor.
97 */
98/*!
99 * \typedef struct dvbpsi_content_id_dr_s dvbpsi_content_id_dr_t
100 * \brief dvbpsi_content_id_dr_t type definition.
101 */
102typedef struct dvbpsi_content_id_dr_s
103{
104 uint8_t i_number_of_entries; /*!< Number of CRID entries present. */
105 dvbpsi_crid_entry_t p_entries[85];/*!< Array of CRID entries. */
106}dvbpsi_content_id_dr_t;
107
108/*****************************************************************************
109 * dvbpsi_DecodeLCNDr
110 *****************************************************************************/
111/*!
112 * \fn dvbpsi_content_id_dr_s dvbpsi_DecodeContentIdDr(dvbpsi_descriptor_t *p_descriptor)
113 * \brief Decode a Content Identifier descriptor (tag 0x76)
114 * \param p_descriptor Raw descriptor to decode.
115 * \return NULL if the descriptor could not be decoded or a pointer to a
116 * dvbpsi_content_id_dr_t structure.
117 */
118dvbpsi_content_id_dr_t *dvbpsi_DecodeContentIdDr(dvbpsi_descriptor_t *p_descriptor);
119
120#ifdef __cplusplus
121};
122#endif
123
124#endif
