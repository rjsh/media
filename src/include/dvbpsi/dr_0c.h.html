24/*!
25 * \file <dr_0c.h>
26 * \author Arnaud de Bossoreille de Ribou <bozo@via.ecp.fr>
27 * \brief Application interface for the MPEG 2 "multiplex buffer utilization"
28 * descriptor decoder and generator.
29 *
30 * Application interface for the MPEG 2 "multiplex buffer utilization"
31 * descriptor decoder and generator. This descriptor's definition can be found
32 * in ISO/IEC 13818-1 section 2.6.22.
33 */
34
35#ifndef _DVBPSI_DR_0C_H_
36#define _DVBPSI_DR_0C_H_
37
38#ifdef __cplusplus
39extern "C" {
40#endif
41
42
43/*****************************************************************************
44 * dvbpsi_mx_buff_utilization_dr_t
45 *****************************************************************************/
46/*!
47 * \struct dvbpsi_mx_buff_utilization_dr_s
48 * \brief "multiplex buffer utilization" descriptor structure.
49 *
50 * This structure is used to store a decoded "multiplex buffer utilization"
51 * descriptor. (ISO/IEC 13818-1 section 2.6.22).
52 */
53/*!
54 * \typedef struct dvbpsi_mx_buff_utilization_dr_s dvbpsi_mx_buff_utilization_dr_t
55 * \brief dvbpsi_systemclock_dr_t type definition.
56 */
57typedef struct dvbpsi_mx_buff_utilization_dr_s
58{
59 bool b_mdv_valid; /*!< mdv_valid_flag */
60 uint16_t i_mx_delay_variation; /*!< multiplex_delay_variation */
61 uint8_t i_mx_strategy; /*!< multiplex_strategy */
62
63} dvbpsi_mx_buff_utilization_dr_t;
64
65
66/*****************************************************************************
67 * dvbpsi_DecodeMxBuffUtilizationDr
68 *****************************************************************************/
69/*!
70 * \fn dvbpsi_mx_buff_utilization_dr_t * dvbpsi_DecodeMxBuffUtilizationDr(
71 dvbpsi_descriptor_t * p_descriptor)
72 * \brief "multiplex buffer utilization" descriptor decoder.
73 * \param p_descriptor pointer to the descriptor structure
74 * \return a pointer to a new "multiplex buffer utilization" descriptor
75 * structure which contains the decoded data.
76 */
77dvbpsi_mx_buff_utilization_dr_t* dvbpsi_DecodeMxBuffUtilizationDr(
78 dvbpsi_descriptor_t * p_descriptor);
79
80
81/*****************************************************************************
82 * dvbpsi_GenMxBuffUtilizationDr
83 *****************************************************************************/
84/*!
85 * \fn dvbpsi_descriptor_t * dvbpsi_GenMxBuffUtilizationDr(
86 dvbpsi_mx_buff_utilization_dr_t * p_decoded, bool b_duplicate)
87 * \brief "multiplex buffer utilization" descriptor generator.
88 * \param p_decoded pointer to a decoded "system clock" descriptor
89 * structure
90 * \param b_duplicate if true then duplicate the p_decoded structure into
91 * the descriptor
92 * \return a pointer to a new descriptor structure which contains encoded data.
93 */
94dvbpsi_descriptor_t * dvbpsi_GenMxBuffUtilizationDr(
95 dvbpsi_mx_buff_utilization_dr_t * p_decoded,
96 bool b_duplicate);
97
98
99#ifdef __cplusplus
100};
101#endif
