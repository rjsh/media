107/*!
108 * \def DVBPSI_DECODER(x)
109 * \brief Helper macro for casting a private decoder into a dvbpsi_decoder_t
110 */
111/* Clever cast borrowed from VLC */
112#if DVBPSI_GCC_VERSION(4,0)
113# ifndef __cplusplus
114# define DVBPSI_DECODER( x ) \
115 __builtin_choose_expr( \
116 __builtin_offsetof(__typeof__(*(x)), i_magic), \
117 (void)0, \
118 (dvbpsi_decoder_t *)(x))
119# else
120# define DVBPSI_DECODER( x ) \
121 ((dvbpsi_decoder_t *)(x) \
122 + 0 * __builtin_offsetof(__typeof__(*(x)), i_magic))
123# endif
124#else
125# define DVBPSI_DECODER(x) ((dvbpsi_decoder_t *)(x))
126#endif

128/*****************************************************************************
129 * dvbpsi_t
130 *****************************************************************************/
131/*!
132 * \struct dvbpsi_s
133 * \brief DVBPSI handle structure
134 *
135 * This structure provides a handle to libdvbpsi API and should be used instead
136 * of dvbpsi_decoder_t.
137 */
138/*!
139 * \typedef struct dvbpsi_s dvbpsi_t
140 * \brief dvbpsi_t type definition.
141 */
142struct dvbpsi_s
143{
144 dvbpsi_decoder_t *p_decoder; /*!< private pointer to
145 specific decoder */
146 /* Messages callback */
147 dvbpsi_message_cb pf_message; /*!< Log message callback */
148 enum dvbpsi_msg_level i_msg_level; /*!< Log level */
149
150 /* private data pointer for use by caller, not by libdvbpsi itself ! */
151 void *p_sys; /*!< pointer to private data
152 from caller. Do not use
153 from inside libdvbpsi. It
154 will crash any application. */
155};

157/*****************************************************************************
158 * dvbpsi_new
159 *****************************************************************************/
160/*!
161 * \fn dvbpsi_t *dvbpsi_new(dvbpsi_message_cb callback, enum dvbpsi_msg_level level)
162 * \brief Create a new dvbpsi_t handle to be used by PSI decoders or encoders
163 * \param callback message callback handler, if NULL then no errors, warnings
164 * or debug messages will be sent to the caller application
165 * \param level enum dvbpsi_msg_level for filtering logging messages
166 * \return pointer to dvbpsi_t malloced data
167 *
168 * Creates a handle to use with PSI decoder and encoder API functions. The
169 * handle must be freed with dvbpsi_delete().
170 */
171dvbpsi_t *dvbpsi_new(dvbpsi_message_cb callback, enum dvbpsi_msg_level level);

177 * \fn void dvbpsi_delete(dvbpsi_t *p_dvbpsi)

179 * \param p_dvbpsi pointer to dvbpsi_t malloced data
180 * \return nothing

182 * Delets a dvbpsi_t handle by calling free(handle). Make sure to detach any
183 * decoder of encoder before deleting the dvbpsi handle.

185void dvbpsi_delete(dvbpsi_t *p_dvbpsi);

222/*****************************************************************************
223 * DVBPSI_DECODER_COMMON
224 *****************************************************************************/
225/*!
226 * \def DVBPSI_DECODER_COMMON
227 * \brief Common members for all dvbpsi table decoders. These should be the first
228 * members of a table decoder struct, such that they can be casted to a
229 * dvbpsi_decoder_t type.
230 */
231#define DVBPSI_DECODER_COMMON \
232/*! \
233 * \name DVBPSI_DECODER_COMMON \
234 * these members are common for all dvbpsi_decoder_t \
235 */ \
236/**@{*/ \
237 uint8_t i_magic[3]; /*!< Reserved magic value */ \
238 bool b_complete_header; /*!< Flag for header completion */ \
239 bool b_discontinuity; /*!< Discontinuity flag */ \
240 bool b_current_valid; /*!< Current valid indicator */ \
241 uint8_t i_continuity_counter; /*!< Continuity counter */ \
242 uint8_t i_last_section_number;/*!< Last received section number */ \
243 dvbpsi_psi_section_t *p_current_section; /*!< Current section */ \
244 dvbpsi_psi_section_t *p_sections; /*!< List of received PSI sections */ \
245 dvbpsi_callback_gather_t pf_gather;/*!< PSI decoder's callback */ \
246 int i_section_max_size; /*!< Max size of a section for this decoder */ \
247 int i_need; /*!< Bytes needed */ \
248/**@}*/
249

265/*****************************************************************************
266 * dvbpsi_decoder_new
267 *****************************************************************************/
268/*!
269 * \fn void *dvbpsi_decoder_new(dvbpsi_callback_gather_t pf_gather,
270 * const int i_section_max_size, const bool b_discontinuity, const size_t psi_size);
271 * \brief Create a new (private) dvbpsi decoder.
272 * \param pf_gather pointer to gather function for PSI decoder.
273 * \param i_section_max_size Max size of a section for this decoder
274 * \param b_discontinuity Discontinuity flag
275 * \param psi_size size of new PSI struct, eg: sizeof(dvbpsi_pat_t)
276 * \return pointer to memory of size 'psi_size' that can be casted into a dvbpsi_decoder_t.
277 *
278 * Creates a void pointer that points to a private dvbpsi decoder struct (eg: dvbpsi_pat_t).
279 * The first elements in this newly created decoder should contain DVBPSI_DECODER_COMMON, which
280 * corresponds to a dvbpsi_decoder_t.
281 * Upon return the pointer should be casted to the correct type. Delete the pointer
282 * with @see dvbpsi_decoder_delete() function.
283 */
284void *dvbpsi_decoder_new(dvbpsi_callback_gather_t pf_gather,
285 const int i_section_max_size,
286 const bool b_discontinuity,
287 const size_t psi_size);
