23/*!
24 * \file <dr_4a.h>
25 * \author Corno Roberto <corno.roberto@gmail.com>
26 * \brief Application interface for the DVB "country availability"
27 * descriptor decoder and generator.
28 *
29 * Application interface for the DVB "linkage" descriptor
30 * decoder and generator. This descriptor's definition can be found in
31 * ETSI EN 300 468 section 6.2.19.
32 */
33
34#ifndef DR_4A_H_
35#define DR_4A_H_
36
37#ifdef __cplusplus
38extern "C" {
39#endif
40
41/*****************************************************************************
42 * dvbpsi_linkage_dr_t
43 *****************************************************************************/
44/*!
45 * \struct dvbpsi_linkage_dr_t
46 * \brief "linkage" descriptor structure.
47 *
48 * This structure is used to store a decoded "linkage"
49 * descriptor. (ETSI EN 300 468 section 6.2.10).
50 */
51/*!
52 * \typedef struct dvbpsi_linkage_dr_s dvbpsi_linkage_dr_t
53 * \brief dvbpsi_linkage_dr_t type definition.
54 */
55/*!
56 * \struct dvbpsi_linkage_dr_s
57 * \brief struct dvbpsi_linkage_dr_s @see dvbpsi_linkage_dr_t
58 */
59typedef struct dvbpsi_linkage_dr_s
60{
61 uint16_t i_transport_stream_id; /*!< transport stream id */
62 uint16_t i_original_network_id; /*!< original network id */
63 uint16_t i_service_id; /*!< service id */
64 uint8_t i_linkage_type; /*!< linkage type */
65
66 /* used if i_linkage_type ==0x08 */
67 uint8_t i_handover_type; /*!< hand-over type */
68 uint8_t i_origin_type; /*!< origin type */
69 /* used if i_linkage_type ==0x08 &&
70 * (i_handover_type ==0x01 || i_handover_type ==0x02|| i_handover_type ==0x03)
71 */
72 uint16_t i_network_id; /*!< network id */
73 /* used if i_linkage_type ==0x08 && i_origin_type ==0 */
74 uint16_t i_initial_service_id; /*!< initial service id */
75
76 /* used if i_linkage_type ==0x0D */
77 uint16_t i_target_event_id; /*!< target event id */
78 bool b_target_listed; /*!< target listed */
79 bool b_event_simulcast; /*!< event simulcast */
80
81 uint8_t i_private_data_length; /*!< length of the i_private_data
82 array */
83 uint8_t i_private_data[248]; /*!< private data */
84
85} dvbpsi_linkage_dr_t;
86
87/*****************************************************************************
88 * dvbpsi_DecodeLinkageDr
89 *****************************************************************************/
90/*!
91 * \fn dvbpsi_linkage_dr_t * dvbpsi_DecodeLinkageDr(
92 dvbpsi_descriptor_t * p_descriptor)
93 * \brief "linkage" descriptor decoder.
94 * \param p_descriptor pointer to the descriptor structure
95 * \return a pointer to a new "linkage" descriptor structure
96 * which contains the decoded data.
97 */
98dvbpsi_linkage_dr_t* dvbpsi_DecodeLinkageDr(dvbpsi_descriptor_t * p_descriptor);
99
100/*****************************************************************************
101 * dvbpsi_GenLinkageDr
102 *****************************************************************************/
103/*!
104 * \fn dvbpsi_descriptor_t *dvbpsi_GenLinkageDr(dvbpsi_linkage_dr_t *p_decoded,
105 bool b_duplicate);
106 * \brief "linkage" descriptor generator.
107 * \param p_decoded pointer to a decoded "linkage" descriptor structure
108 * \param b_duplicate if true then duplicate the p_decoded structure into
109 * the descriptor
110 * \return a pointer to a new descriptor structure which contains encoded data.
111 */
112dvbpsi_descriptor_t *dvbpsi_GenLinkageDr(dvbpsi_linkage_dr_t * p_decoded,
113 bool b_duplicate);
114
115#ifdef __cplusplus
116};
117#endif
