27/*!
28 * \file <tot.h>
29 * \author Johann Hanne
30 * \brief Application interface for the TDT/TOT decoder and the TDT/TOT generator.
31 *
32 * Application interface for the TDT (Time and Date Table)/TOT (Time Offset Table)
33 * decoder and the TDT/TOT generator. New decoded TDT/TOT tables are sent by
34 * callback to the application.
35 */
36
37#ifndef _DVBPSI_TOT_H_
38#define _DVBPSI_TOT_H_
39
40#ifdef __cplusplus
41extern "C" {
42#endif
43
44/*****************************************************************************
45 * dvbpsi_tot_t
46 *****************************************************************************/
47/*!
48 * \struct dvbpsi_tot_s
49 * \brief TDT/TOT structure.
50 *
51 * This structure is used to store a decoded TDT/TOT.
52 * (ETSI EN 300 468 section 5.2.5/5.2.6).
53 */
54/*!
55 * \typedef struct dvbpsi_tot_s dvbpsi_tot_t
56 * \brief dvbpsi_tot_t type definition.
57 */
58typedef struct dvbpsi_tot_s
59{
60 uint8_t i_table_id; /*!< table id */
61 uint16_t i_extension; /*!< subtable id */
62
63 /* Subtable specific */
64 uint8_t i_version; /*!< version_number */
65 bool b_current_next; /*!< current_next_indicator */
66
67 uint64_t i_utc_time; /*!< UTC_time */
68
69 dvbpsi_descriptor_t * p_first_descriptor; /*!< descriptor list */
70
71 uint32_t i_crc; /*!< CRC_32 (TOT only) */
72
73} __attribute__((packed)) dvbpsi_tot_t;
74
75/*****************************************************************************
76 * dvbpsi_tot_callback
77 *****************************************************************************/
78/*!
79 * \typedef void (* dvbpsi_tot_callback)(void* p_cb_data,
80 dvbpsi_tot_t* p_new_tot)
81 * \brief Callback type definition.
82 */
83typedef void (* dvbpsi_tot_callback)(void* p_cb_data, dvbpsi_tot_t* p_new_tot);
84
85/*****************************************************************************
86 * dvbpsi_tot_attach
87 *****************************************************************************/
88/*!
89 * \fn bool dvbpsi_tot_attach(dvbpsi_t* p_dvbpsi, uint8_t i_table_id, uint16_t i_extension,
90 dvbpsi_tot_callback pf_callback, void* p_cb_data)
91 * \brief Creation and initialization of a TDT/TOT decoder.
92 * \param p_dvbpsi dvbpsi handle pointing to Subtable demultiplexor to which the decoder is attached.
93 * \param i_table_id Table ID, usually 0x70
94 * \param i_extension Table ID extension, unused in the TDT/TOT
95 * \param pf_callback function to call back on new TDT/TOT.
96 * \param p_cb_data private data given in argument to the callback.
97 * \return true on success, false on failure
98 */
99bool dvbpsi_tot_attach(dvbpsi_t* p_dvbpsi, uint8_t i_table_id, uint16_t i_extension,
100 dvbpsi_tot_callback pf_callback, void* p_cb_data);
101
102/*****************************************************************************
103 * dvbpsi_tot_detach
104 *****************************************************************************/
105/*!
106 * \fn int dvbpsi_tot_detach(dvbpsi_t *p_dvbpsi, uint8_t i_table_id, uint16_t i_extension)
107 * \brief Destroy a TDT/TOT decoder.
108 * \param p_dvbpsi Subtable demultiplexor to which the decoder is attached.
109 * \param i_table_id Table ID, usually 0x70
110 * \param i_extension Table ID extension, unused in the TDT/TOT
111 * \return nothing.
112 */
113void dvbpsi_tot_detach(dvbpsi_t* p_dvbpsi, uint8_t i_table_id,
114 uint16_t i_extension);
115
116/*****************************************************************************
117 * dvbpsi_tot_init/dvbpsi_tot_new
118 *****************************************************************************/
119/*!
120 * \fn void dvbpsi_tot_init(dvbpsi_tot_t* p_tot, uint8_t i_table_id, uint16_t i_extension,
121 uint8_t i_version, bool b_current_next, uint64_t i_utc_time);
122 * \brief Initialize a user-allocated dvbpsi_tot_t structure.
123 * \param p_tot pointer to the TDT/TOT structure
124 * \param i_table_id Table ID, usually 0x70
125 * \param i_extension Table ID extension, unused in the TDT/TOT
126 * \param i_version SDT version
127 * \param b_current_next current next indicator
128 * \param i_utc_time the time in UTC
129 * \return nothing.
130 */
131void dvbpsi_tot_init(dvbpsi_tot_t* p_tot, uint8_t i_table_id, uint16_t i_extension,
132 uint8_t i_version, bool b_current_next, uint64_t i_utc_time);
133
134/*!
135 * \fn dvbpsi_tot_t *dvbpsi_tot_new(uint8_t i_table_id, uint16_t i_extension,
136 uint8_t i_version, bool b_current_next, uint64_t i_utc_time);
137 * \brief Allocate and initialize a new dvbpsi_tot_t structure.
138 * \param i_table_id Table ID, usually 0x70
139 * \param i_extension Table ID extension, unused in the TDT/TOT
140 * \param i_version SDT version
141 * \param b_current_next current next indicator
142 * \param i_utc_time the time in UTC
143 * \return p_tot pointer to the TDT/TOT structure
144 */
145dvbpsi_tot_t *dvbpsi_tot_new(uint8_t i_table_id, uint16_t i_extension, uint8_t i_version,
146 bool b_current_next, uint64_t i_utc_time);
147
148/*****************************************************************************
149 * dvbpsi_tot_empty/dvbpsi_tot_delete
150 *****************************************************************************/
151/*!
152 * \fn void dvbpsi_tot_empty(dvbpsi_tot_t* p_tot)
153 * \brief Clean a dvbpsi_tot_t structure.
154 * \param p_tot pointer to the TDT/TOT structure
155 * \return nothing.
156 */
157void dvbpsi_tot_empty(dvbpsi_tot_t* p_tot);
158
159/*!
160 * \fn dvbpsi_tot_delete(dvbpsi_tot_t* p_tot)
161 * \brief Clean and free a dvbpsi_tot_t structure.
162 * \param p_tot pointer to the TDT/TOT structure
163 * \return nothing.
164 */
165void dvbpsi_tot_delete(dvbpsi_tot_t* p_tot);
166
167/*****************************************************************************
168 * dvbpsi_tot_descriptor_add
169 *****************************************************************************/
170/*!
171 * \fn dvbpsi_descriptor_t* dvbpsi_tot_descriptor_add(dvbpsi_tot_t* p_tot,
172 uint8_t i_tag,
173 uint8_t i_length,
174 uint8_t* p_data)
175 * \brief Add a descriptor in the TOT.
176 * \param p_tot pointer to the TOT structure
177 * \param i_tag descriptor's tag
178 * \param i_length descriptor's length
179 * \param p_data descriptor's data
180 * \return a pointer to the added descriptor.
181 */
182dvbpsi_descriptor_t* dvbpsi_tot_descriptor_add(dvbpsi_tot_t* p_tot,
183 uint8_t i_tag, uint8_t i_length,
184 uint8_t* p_data);
185
186/*****************************************************************************
187 * dvbpsi_tot_sections_generate
188 *****************************************************************************/
189/*!
190 * \fn dvbpsi_psi_section_t* dvbpsi_tot_sections_generate(dvbpsi_t *p_dvbpsi, dvbpsi_tot_t* p_tot)
191 * \brief TDT/TOT generator
192 * \param p_dvbpsi handle to dvbpsi with attached decoder
193 * \param p_tot TDT/TOT structure

197 */
198dvbpsi_psi_section_t* dvbpsi_tot_sections_generate(dvbpsi_t* p_dvbpsi, dvbpsi_tot_t* p_tot);
