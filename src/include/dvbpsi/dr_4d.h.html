24/*!
25 * \file <dr_4d.h>
26 * \author Laurent Aimar <fenrir@via.ecp.fr>
27 * \brief Application interface for the "Short event" descriptor
28 * decoder and generator.
29 *
30 * Application interface for the "Short event" descriptor
31 * decoder and generator. This descriptor's definition can be found in
32 * ETSI EN 300 468 section 6.2.35.
33 */
34
35#ifndef _DVBPSI_DR_4D_H_
36#define _DVBPSI_DR_4D_H_
37
38#ifdef __cplusplus
39extern "C" {
40#endif
41
42
43/*****************************************************************************
44 * dvbpsi_short_event_dr_t
45 *****************************************************************************/
46/*!
47 * \struct dvbpsi_short_event_dr_s
48 * \brief "short event" descriptor structure.
49 *
50 * This structure is used to store a decoded "short event" descriptor.
51 */
52/*!
53 * \typedef struct dvbpsi_short_event_dr_s dvbpsi_short_event_dr_t
54 * \brief dvbpsi_short_event_dr_t type definition.
55 */
56typedef struct dvbpsi_short_event_dr_s
57{
58 uint8_t i_iso_639_code[3]; /*!< ISO 639 language code */
59 int i_event_name_length; /*!< length of event name */
60 uint8_t i_event_name[256]; /*!< "short event" name */
61 int i_text_length; /*!< text length */
62 uint8_t i_text[256]; /*!< "short event" text */
63
64} dvbpsi_short_event_dr_t;
65
66
67/*****************************************************************************
68 * dvbpsi_DecodeShortEventDr
69 *****************************************************************************/
70/*!
71 * \fn dvbpsi_short_event_dr_t * dvbpsi_DecodeShortEventDr(
72 dvbpsi_descriptor_t * p_descriptor)
73 * \brief "short event" descriptor decoder.
74 * \param p_descriptor pointer to the descriptor structure
75 * \return a pointer to a new "short event" descriptor structure which
76 * contains the decoded data.
77 */
78dvbpsi_short_event_dr_t* dvbpsi_DecodeShortEventDr(dvbpsi_descriptor_t * p_descriptor);
79
80
81/*****************************************************************************
82 * dvbpsi_GenShortEventDr
83 *****************************************************************************/
84/*!
85 * \fn dvbpsi_descriptor_t * dvbpsi_GenShortEventDr(
86 dvbpsi_short_event_dr_t * p_decoded, bool b_duplicate)
87 * \brief "short event" descriptor generator.
88 * \param p_decoded pointer to a decoded "video stream" descriptor structure
89 * \param b_duplicate if true then duplicate the p_decoded structure into
90 * the descriptor
91 * \return a pointer to a new descriptor structure which contains encoded data.
92 */
93dvbpsi_descriptor_t * dvbpsi_GenShortEventDr(dvbpsi_short_event_dr_t * p_decoded,
94 bool b_duplicate);
95
96
97#ifdef __cplusplus
98};
99#endif
100
