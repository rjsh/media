24/*!
25 * \file <dr_07.h>
26 * \author Arnaud de Bossoreille de Ribou <bozo@via.ecp.fr>
27 * \brief Application interface for the MPEG 2 "target background grid"
28 * descriptor decoder and generator.
29 *
30 * Application interface for the MPEG 2 "target background grid" descriptor
31 * decoder and generator. This descriptor's definition can be found in
32 * ISO/IEC 13818-1 section 2.6.12.
33 */
34
35#ifndef _DVBPSI_DR_07_H_
36#define _DVBPSI_DR_07_H_
37
38#ifdef __cplusplus
39extern "C" {
40#endif
41
42
43/*****************************************************************************
44 * dvbpsi_target_bg_grid_dr_t
45 *****************************************************************************/
46/*!
47 * \struct dvbpsi_target_bg_grid_dr_s
48 * \brief "target background grid" descriptor structure.
49 *
50 * This structure is used to store a decoded "target background grid"
51 * descriptor. (ISO/IEC 13818-1 section 2.6.12).
52 */
53/*!
54 * \typedef struct dvbpsi_target_bg_grid_dr_s dvbpsi_target_bg_grid_dr_t
55 * \brief dvbpsi_target_bg_grid_dr_t type definition.
56 */
57typedef struct dvbpsi_target_bg_grid_dr_s
58{
59 uint16_t i_horizontal_size; /*!< horizontal_size */
60 uint16_t i_vertical_size; /*!< vertical_size */
61 uint8_t i_pel_aspect_ratio; /*!< pel_aspect_ratio */
62
63} dvbpsi_target_bg_grid_dr_t;
64
65
66/*****************************************************************************
67 * dvbpsi_DecodeTargetBgGridDr
68 *****************************************************************************/
69/*!
70 * \fn dvbpsi_target_bg_grid_dr_t * dvbpsi_DecodeTargetBgGridDr(
71 dvbpsi_descriptor_t * p_descriptor)
72 * \brief "target background grid" descriptor decoder.
73 * \param p_descriptor pointer to the descriptor structure
74 * \return a pointer to a new "target background grid" descriptor structure
75 * which contains the decoded data.
76 */
77dvbpsi_target_bg_grid_dr_t* dvbpsi_DecodeTargetBgGridDr(
78 dvbpsi_descriptor_t * p_descriptor);
79
80
81/*****************************************************************************
82 * dvbpsi_GenTargetBgGridDr
83 *****************************************************************************/
84/*!
85 * \fn dvbpsi_descriptor_t * dvbpsi_GenTargetBgGridDr(
86 dvbpsi_target_bg_grid_dr_t * p_decoded, bool b_duplicate)
87 * \brief "target background grid" descriptor generator.
88 * \param p_decoded pointer to a decoded "target background grid" descriptor
89 * structure
90 * \param b_duplicate if true then duplicate the p_decoded structure into
91 * the descriptor
92 * \return a pointer to a new descriptor structure which contains encoded data.
93 */
94dvbpsi_descriptor_t * dvbpsi_GenTargetBgGridDr(
95 dvbpsi_target_bg_grid_dr_t * p_decoded,
96 bool b_duplicate);
97
98
99#ifdef __cplusplus
100};
101#endif
102
