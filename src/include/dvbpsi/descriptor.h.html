51/*****************************************************************************
52 * dvbpsi_descriptor_t
53 *****************************************************************************/
54/*!
55 * \struct dvbpsi_descriptor_s
56 * \brief Descriptor structure.
57 *
58 * This structure is used to store a descriptor.
59 * (ISO/IEC 13818-1 section 2.6).
60 */
61/*!
62 * \typedef struct dvbpsi_descriptor_s dvbpsi_descriptor_t
63 * \brief dvbpsi_descriptor_t type definition.
64 *
65 * The common descriptor header and its payload is contained in this
66 * structure. The payload is the raw descriptor data and its interpretation
67 * depends on the the p_descriptor::i_tag value.
68 *
69 * After passing the descriptor to a dvbpsi_DecodeXXXXDr function the raw
70 * descriptor data is interpreted and decoded into a descriptor specific
71 * structure. This structure is stored in the p_descriptor::p_decoded member
72 * by the dvbpsi_DecodeXXXXDr function.
73 *
74 * NOTE: It is mandatory to add a decoded descriptor to the 'p_decoded' member
75 * of this struct. Failing to do so will result in memory leakage when
76 * deleting descriptor with @see dvbpsi_DeleteDescriptor.
77 */
78typedef struct dvbpsi_descriptor_s
79{
80 uint8_t i_tag; /*!< descriptor_tag */
81 uint8_t i_length; /*!< descriptor_length */
82
83 uint8_t * p_data; /*!< content */
84
85 struct dvbpsi_descriptor_s * p_next; /*!< next element of
86 the list */
87
88 void * p_decoded; /*!< decoded descriptor */
89
90} dvbpsi_descriptor_t;
91
92/*****************************************************************************
93 * dvbpsi_NewDescriptor
94 *****************************************************************************/
95/*!
96 * \fn dvbpsi_descriptor_t* dvbpsi_NewDescriptor(uint8_t i_tag,
97 uint8_t i_length,
98 uint8_t* p_data)
99 * \brief Creation of a new dvbpsi_descriptor_t structure.
100 * \param i_tag descriptor's tag
101 * \param i_length descriptor's length
102 * \param p_data descriptor's data
103 * \return a pointer to the descriptor.
104 */
105dvbpsi_descriptor_t* dvbpsi_NewDescriptor(uint8_t i_tag, uint8_t i_length,
106 uint8_t* p_data);
107
108
109/*****************************************************************************
110 * dvbpsi_DeleteDescriptors
111 *****************************************************************************/
112/*!
113 * \fn void dvbpsi_DeleteDescriptors(dvbpsi_descriptor_t* p_descriptor)
114 * \brief Destruction of a dvbpsi_descriptor_t structure together with the decoded
115 * descriptor, if present.
116 * \param p_descriptor pointer to the first descriptor structure
117 * \return nothing.
118 */
119void dvbpsi_DeleteDescriptors(dvbpsi_descriptor_t* p_descriptor);
120
121/*****************************************************************************
122 * dvbpsi_AddDescriptor
123 *****************************************************************************/
124/*!
125 * \fn dvbpsi_descriptor_t *dvbpsi_AddDescriptor(dvbpsi_descriptor_t *p_list,
126 dvbpsi_descriptor_t *p_descriptor);
127 * \brief Add a descriptor to the end of descriptor list.
128 * \param p_list the first descriptor in the descriptor list.
129 * \param p_descriptor the descriptor to add to the list
130 * \return a pointer to the first element in the descriptor list.
131 */
132dvbpsi_descriptor_t *dvbpsi_AddDescriptor(dvbpsi_descriptor_t *p_list,
133 dvbpsi_descriptor_t *p_descriptor);

135/*****************************************************************************
136 * dvbpsi_CanDecodeAsDescriptor
137 *****************************************************************************/
138/*!
139 * \fn bool dvbpsi_CanDecodeAsDescriptor(dvbpsi_descriptor_t *p_descriptor, const uint8_t i_tag);
140 * \brief Checks if descriptor tag matches.
141 * \param p_descriptor pointer to descriptor allocated with @see dvbpsi_NewDescriptor
142 * \param i_tag descriptor tag to evaluate against
143 * \return true if descriptor can be decoded, false if not.
144 */
145bool dvbpsi_CanDecodeAsDescriptor(dvbpsi_descriptor_t *p_descriptor, const uint8_t i_tag);
146
147/*****************************************************************************
148 * dvbpsi_IsDescriptorDecoded
149 *****************************************************************************/
150/*!
151 * \fn bool dvbpsi_IsDescriptorDecoded(dvbpsi_descriptor_t *p_descriptor);
152 * \brief Checks if descriptor was already decoded.
153 * \param p_descriptor pointer to descriptor allocated with @see dvbpsi_NewDescriptor
154 * \return true if descriptor can be decoded, false if already decoded.
155 */
156bool dvbpsi_IsDescriptorDecoded(dvbpsi_descriptor_t *p_descriptor);
157
158/*****************************************************************************
159 * dvbpsi_DuplicateDecodedDescriptor
160 *****************************************************************************/
161/*!
162 * \fn void *dvbpsi_DuplicateDecodedDescriptor(void *p_decoded, ssize_t i_size);
163 * \brief Duplicate a decoded descriptor. The caller is responsible for releasing the associated memory.
164 * \param p_decoded pointer to decoded descriptor obtained with dvbpsi_Decode* function
165 * \param i_size the sizeof decoded descriptor
166 * \return pointer to duplicated descriptor, NULL on error.
167 */
168void *dvbpsi_DuplicateDecodedDescriptor(void *p_decoded, ssize_t i_size);
