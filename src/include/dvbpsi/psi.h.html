39/*****************************************************************************
40 * dvbpsi_psi_section_t
41 *****************************************************************************/
42/*!
43 * \struct dvbpsi_psi_section_s
44 * \brief PSI section structure.
45 *
46 * This structure is used to store a PSI section. The common information are
47 * decoded (ISO/IEC 13818-1 section 2.4.4.10).
48 *
49 * dvbpsi_psi_section_s::p_data stores the complete section including the
50 * header.
51 *
52 * When dvbpsi_psi_section_s::b_syntax_indicator == false,
53 * dvbpsi_psi_section_s::p_payload_start points immediately after the
54 * section_length field and dvbpsi_psi_section_s::p_payload_end points
55 * immediately after the end of the section (don't try to access this byte).
56 *
57 * When dvbpsi_psi_section_s::b_syntax_indicator != false,
58 * dvbpsi_psi_section_s::p_payload_start points immediately after the
59 * last_section_number field and dvbpsi_psi_section_s::p_payload_end points to
60 * the first byte of the CRC_32 field.
61 *
62 * When dvbpsi_psi_section_s::b_syntax_indicator == false
63 * dvbpsi_psi_section_s::i_extension, dvbpsi_psi_section_s::i_version,
64 * dvbpsi_psi_section_s::b_current_next, dvbpsi_psi_section_s::i_number,
65 * dvbpsi_psi_section_s::i_last_number, and dvbpsi_psi_section_s::i_crc are
66 * undefined.
67 */
68struct dvbpsi_psi_section_s
69{
70 /* non-specific section data */
71 uint8_t i_table_id; /*!< table_id */
72 bool b_syntax_indicator; /*!< section_syntax_indicator */
73 bool b_private_indicator; /*!< private_indicator */
74 uint16_t i_length; /*!< section_length */
75
76 /* used if b_syntax_indicator is true */
77 uint16_t i_extension; /*!< table_id_extension */
78 /*!< transport_stream_id for a
79 PAT section */
80 uint8_t i_version; /*!< version_number */
81 bool b_current_next; /*!< current_next_indicator */
82 uint8_t i_number; /*!< section_number */
83 uint8_t i_last_number; /*!< last_section_number */
84
85 /* non-specific section data */
86 /* the content is table-specific */
87 uint8_t * p_data; /*!< complete section */
88 uint8_t * p_payload_start; /*!< payload start */
89 uint8_t * p_payload_end; /*!< payload end */
90
91 /* used if b_syntax_indicator is true */
92 uint32_t i_crc; /*!< CRC_32 */
93
94 /* list handling */
95 struct dvbpsi_psi_section_s * p_next; /*!< next element of
96 the list */
97};

122 * dvbpsi_CheckPSISection
123 *****************************************************************************/
124/*!
125 * \fn bool dvbpsi_CheckPSISection(dvbpsi_t *p_dvbpsi, dvbpsi_psi_section_t *p_section,
126 const uint8_t table_id, const char *psz_table_name)
127 * \brief Check if PSI section has the expected table_id and it the syntax indicator
128 * is true.
129 * \param p_dvbpsi pointer to dvbpsi library handle
130 * \param p_section pointer to the PSI section structure
131 * \param table_id expected table id
132 * \param psz_table_name table name to use when reporting errors.
133 * \return boolean value (false if the section did not pass the tests).
134 */
135bool dvbpsi_CheckPSISection(dvbpsi_t *p_dvbpsi, dvbpsi_psi_section_t *p_section,
136 const uint8_t table_id, const char *psz_table_name);
137

141/*!
142 * \fn bool dvbpsi_ValidPSISection(dvbpsi_psi_section_t* p_section)
143 * \brief Validity check of a PSI section.
144 * \param p_section pointer to the PSI section structure
145 * \return boolean value (false if the section is not valid).
146 *
147 * Check the CRC_32 if the section has b_syntax_indicator set.
148 */
149bool dvbpsi_ValidPSISection(dvbpsi_psi_section_t* p_section);
