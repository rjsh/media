37/*****************************************************************************
38 * dvbpsi_frequency_list_dr_s
39 *****************************************************************************/
40/*!
41 * \struct dvbpsi_frequency_list_dr_s
42 * \brief Frequency List Descriptor
43 *
44 * This structure is used to store a decoded Frequency List descriptor.
45 */
46/*!
47 * \typedef struct dvbpsi_frequency_list_dr_s dvbpsi_frequency_list_dr_t
48 * \brief dvbpsi_frequency_list_dr_t type definition.
49 */
50typedef struct dvbpsi_frequency_list_dr_s
51{
52 uint8_t i_coding_type; /*!< Coding type, 1 = Satellite, 2 = Cable, 3 = Terrestrial */
53 uint8_t i_number_of_frequencies; /*!< Number of center frequencies present */
54 uint32_t p_center_frequencies[63]; /*!< Center frequency as defined by a delivery_system_descriptor */
55}dvbpsi_frequency_list_dr_t;
56
57/*****************************************************************************
58 * dvbpsi_DecodeFrequencyListDr
59 *****************************************************************************/
60/*!
61 * \fn dvbpsi_frequency_list_dr_t *dvbpsi_DecodeFrequencyListDr(
62 * dvbpsi_descriptor_t *p_descriptor)
63 * \brief Decode a Frequency List descriptor (tag 0x62)
64 * \param p_descriptor Raw descriptor to decode.
65 * \return NULL if the descriptor could not be decoded or a pointer to a
66 * dvbpsi_frequency_list_dr_t structure.
67 */
68dvbpsi_frequency_list_dr_t *dvbpsi_DecodeFrequencyListDr(dvbpsi_descriptor_t *p_descriptor);
69
70/*****************************************************************************
71 * dvbpsi_Bcd8ToUint32
72 *****************************************************************************/
73/*!
74 * \fn uint32_t dvbpsi_Bcd8ToUint32(uint32_t bcd)
75 * \brief Decode an 8 character BCD encoded number into a uint32_t.
76 * \param bcd BCD value to convert.
77 * \return Decoded value as a uint32_t.
78 */
79uint32_t dvbpsi_Bcd8ToUint32(uint32_t bcd);
