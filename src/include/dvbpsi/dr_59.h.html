25/*!
26 * \file <dr_59.h>
27 * \author Tristan Leteurtre <tristan.leteurtre@anevia.com>
28 * \brief DVB subtitling descriptor parsing.
29 *
30 * DVB subtitling descriptor parsing, according to ETSI EN 300 468
31 * section 6.2.36.
32 */
33
34#ifndef _DVBPSI_DR_59_H_
35#define _DVBPSI_DR_59_H_
36
37#ifdef __cplusplus
38extern "C" {
39#endif
40
41/*****************************************************************************
42 * dvbpsi_subtitle_t
43 *****************************************************************************/
44/*!
45 * \struct dvbpsi_subtitle_s
46 * \brief one subtitle structure.
47 *
48 * This structure is used since subtitling_descriptor will contain several
49 * subtitles
50 */
51/*!
52 * \typedef struct dvbpsi_subtitle_s dvbpsi_subtitle_t
53 * \brief dvbpsi_subtitle_t type definition.
54 */
55typedef struct dvbpsi_subtitle_s
56{
57 uint8_t i_iso6392_language_code[3]; /*!< ISO 6392 language code */
58 uint8_t i_subtitling_type; /*!< subtitling type */
59 uint16_t i_composition_page_id; /*!< composition page id */
60 uint16_t i_ancillary_page_id; /*!< ancillayr page id */
61
62} dvbpsi_subtitle_t;
63
64
65/*****************************************************************************
66 * dvbpsi_subtitling_dr_t
67 *****************************************************************************/
68/*!
69 * \struct dvbpsi_subtitling_dr_s
70 * \brief "subtitling" descriptor structure.
71 *
72 * This structure is used to store a decoded "subtitling"
73 * descriptor. (ETSI EN 300 468 section 6.2.30).
74 */
75/*!
76 * \typedef struct dvbpsi_subtitling_dr_s dvbpsi_subtitling_dr_t
77 * \brief dvbpsi_subtitling_dr_t type definition.
78 */
79typedef struct dvbpsi_subtitling_dr_s
80{
81 uint8_t i_subtitles_number; /*!< subtiles number */
82 dvbpsi_subtitle_t p_subtitle[20]; /*!< subtitles */
83
84} dvbpsi_subtitling_dr_t;
85
86
87/*****************************************************************************
88 * dvbpsi_DecodeSubtitlingDataDr
89 *****************************************************************************/
90/*!
91 * \fn dvbpsi_subtitling_dr_t * dvbpsi_DecodeSubtitlingDr(
92 dvbpsi_descriptor_t * p_descriptor)
93 * \brief "subtitling" descriptor decoder.
94 * \param p_descriptor pointer to the descriptor structure
95 * \return a pointer to a new "subtitling" descriptor structure
96 * which contains the decoded data.
97 */
98dvbpsi_subtitling_dr_t* dvbpsi_DecodeSubtitlingDr(
99 dvbpsi_descriptor_t * p_descriptor);
100
101
102/*****************************************************************************
103 * dvbpsi_GenSubtitlingDataDr
104 *****************************************************************************/
105/*!
106 * \fn dvbpsi_descriptor_t * dvbpsi_GenSubtitlingDr(
107 dvbpsi_subtitling_dr_t * p_decoded, bool b_duplicate)
108 * \brief "subtitling" descriptor generator.
109 * \param p_decoded pointer to a decoded "subtitling" descriptor
110 * structure
111 * \param b_duplicate if bool then duplicate the p_decoded structure into
112 * the descriptor
113 * \return a pointer to a new descriptor structure which contains encoded data.
114 */
115dvbpsi_descriptor_t * dvbpsi_GenSubtitlingDr(
116 dvbpsi_subtitling_dr_t * p_decoded,
117 bool b_duplicate);
118
119
120#ifdef __cplusplus
121};
122#endif
123
124#else
125#error "Multiple inclusions of dr_59.h"
