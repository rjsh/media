24/*!
25 * \file <dr_44.h>
26 * \author Ilkka Ollakka
27 * \brief Application interface for the DVB cable delivery system
28 * descriptor decoder and generator.
29 *
30 * Application interface for the DVB cable delivery system descriptor
31 * decoder and generator. This descriptor's definition can be found in
32 * ETSI EN 300 468 section 6.2.13.1.
33 * dr_44.h is heavily based on dr_43.h file
34 */
35
36#ifndef _DVBPSI_DR_44_H_
37#define _DVBPSI_DR_44_H_
38
39#ifdef __cplusplus
40extern "C" {
41#endif
42
43
44/*****************************************************************************
45 * dvbpsi_cable_deliv_sys_dr_t
46 *****************************************************************************/
47/*!
48 * \struct dvbpsi_cable_deliv_sys_dr_s
49 * \brief cable delivery system descriptor structure.
50 *
51 * This structure is used to store a decoded cable delivery system
52 * descriptor. (ETSI EN 300 468 section 6.2.13.1).
53 */
54/*!
55 * \typedef struct dvbpsi_cable_deliv_sys_dr_s dvbpsi_cable_deliv_sys_dr_t
56 * \brief dvbpsi_cable_deliv_sys_dr_t type definition.
57 */
58typedef struct dvbpsi_cable_deliv_sys_dr_s
59{
60 uint32_t i_frequency; /*!< frequency */
61 uint8_t i_modulation; /*!< modulation type */
62 uint32_t i_symbol_rate; /*!< symbol rate */
63 uint8_t i_fec_inner; /*!< FEC inner */
64 uint8_t i_fec_outer; /*!< FEC outer*/
65
66} dvbpsi_cable_deliv_sys_dr_t;
67
68
69/*****************************************************************************
70 * dvbpsi_DecodeCableDelivSysDr
71 *****************************************************************************/
72/*!
73 * \fn dvbpsi_cable_deliv_sys_dr_t * dvbpsi_DecodeCableDelivSysDr(
74 dvbpsi_descriptor_t * p_descriptor)
75 * \brief cable delivery system descriptor decoder.
76 * \param p_descriptor pointer to the descriptor structure
77 * \return a pointer to a new cable delivery system descriptor structure
78 * which contains the decoded data.
79 */
80dvbpsi_cable_deliv_sys_dr_t* dvbpsi_DecodeCableDelivSysDr(
81 dvbpsi_descriptor_t * p_descriptor);
82
83
84/*****************************************************************************
85 * dvbpsi_GenCableDelivSysDr
86 *****************************************************************************/
87/*!
88 * \fn dvbpsi_descriptor_t * dvbpsi_GenCableDelivSysDr(
89 dvbpsi_cable_deliv_sys_dr_t * p_decoded, bool b_duplicate)
90 * \brief cable delivery system descriptor generator.
91 * \param p_decoded pointer to a decoded cable delivery system descriptor
92 * descriptor structure
93 * \param b_duplicate if non zero then duplicate the p_decoded structure into
94 * the descriptor
95 * \return a pointer to a new descriptor structure which contains encoded data.
96 */
97dvbpsi_descriptor_t * dvbpsi_GenCableDelivSysDr(
98 dvbpsi_cable_deliv_sys_dr_t * p_decoded,
99 bool b_duplicate);
100
101
102#ifdef __cplusplus
103};
104#endif
