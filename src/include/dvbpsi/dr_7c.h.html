43/*!
44 * \enum dvbpsi_aac_profile_and_level_s
45 * \brief enumeration of AAC profile and levels as specified in ISO/IEC 13818-1:2007 table 2.71
46 */
47/*!
48 * \typedef enum dvbpsi_aac_profile_and_level_s dvbpsi_aac_profile_and_level_t
49 * \brief AAC profile and level as specified in ISO/IEC 13818-1:2007 table 2.71.
50 */
51typedef enum dvbpsi_aac_profile_and_level_s
52{
53 DVBPSI_AAC_PROFILE_RESERVED = 0x00, /*!< 0x00-0x0F Reserved */
54
55 DVBPSI_AAC_PROFILE_MAIN_LEVEL_1 = 0x10, /*!< Main profile, level 1 */
56 DVBPSI_AAC_PROFILE_MAIN_LEVEL_2 = 0x11, /*!< Main profile, level 2 */
57 DVBPSI_AAC_PROFILE_MAIN_LEVEL_3 = 0x12, /*!< Main profile, level 3 */
58 DVBPSI_AAC_PROFILE_MAIN_LEVEL_4 = 0x13, /*!< Main profile, level 4 */
59 /** 0x14-0x17 Reserved */
60 DVBPSI_AAC_PROFILE_SCALABLE_LEVEL_1 = 0x18, /*!< Scalable Profile, level 1 */
61 DVBPSI_AAC_PROFILE_SCALABLE_LEVEL_2 = 0x19, /*!< Scalable Profile, level 2 */
62 DVBPSI_AAC_PROFILE_SCALABLE_LEVEL_3 = 0x1A, /*!< Scalable Profile, level 3 */
63 DVBPSI_AAC_PROFILE_SCALABLE_LEVEL_4 = 0x1B, /*!< Scalable Profile, level 4 */
64 /** 0x1C-0x1F Reserved */
65 DVBPSI_AAC_PROFILE_SPEECH_LEVEL_1 = 0x20, /*!< Speech profile, level 1 */
66 DVBPSI_AAC_PROFILE_SPEECH_LEVEL_2 = 0x21, /*!< Speech profile, level 2 */
67 /** 0x22-0x27 Reserved */
68 DVBPSI_AAC_PROFILE_SYNTHESIS_LEVEL_1 = 0x28, /*!< Synthesis profile, level 1 */
69 DVBPSI_AAC_PROFILE_SYNTHESIS_LEVEL_2 = 0x29, /*!< Synthesis profile, level 2 */
70 DVBPSI_AAC_PROFILE_SYNTHESIS_LEVEL_3 = 0x2A, /*!< Synthesis profile, level 3 */
71 /** 0x2B-0x2F Reserved */
72 DVBPSI_AAC_PROFILE_HQ_LEVEL_1 = 0x30, /*!< High quality audio profile, level 1 */
73 DVBPSI_AAC_PROFILE_HQ_LEVEL_2 = 0x31, /*!< High quality audio profile, level 2 */
74 DVBPSI_AAC_PROFILE_HQ_LEVEL_3 = 0x32, /*!< High quality audio profile, level 3 */
75 DVBPSI_AAC_PROFILE_HQ_LEVEL_4 = 0x33, /*!< High quality audio profile, level 4 */
76 DVBPSI_AAC_PROFILE_HQ_LEVEL_5 = 0x34, /*!< High quality audio profile, level 5 */
77 DVBPSI_AAC_PROFILE_HQ_LEVEL_6 = 0x35, /*!< High quality audio profile, level 6 */
78 DVBPSI_AAC_PROFILE_HQ_LEVEL_7 = 0x36, /*!< High quality audio profile, level 7 */
79 DVBPSI_AAC_PROFILE_HQ_LEVEL_8 = 0x37, /*!< High quality audio profile, level 8 */
80 DVBPSI_AAC_PROFILE_LOW_DELAY_LEVEL_1 = 0x38, /*!< Low delay audio profile, level 1 */
81 DVBPSI_AAC_PROFILE_LOW_DELAY_LEVEL_2 = 0x39, /*!< Low delay audio profile, level 2 */
82 DVBPSI_AAC_PROFILE_LOW_DELAY_LEVEL_3 = 0x3A, /*!< Low delay audio profile, level 3 */
83 DVBPSI_AAC_PROFILE_LOW_DELAY_LEVEL_4 = 0x3B, /*!< Low delay audio profile, level 4 */
84 DVBPSI_AAC_PROFILE_LOW_DELAY_LEVEL_5 = 0x3C, /*!< Low delay audio profile, level 5 */
85 DVBPSI_AAC_PROFILE_LOW_DELAY_LEVEL_6 = 0x3D, /*!< Low delay audio profile, level 6 */
86 DVBPSI_AAC_PROFILE_LOW_DELAY_LEVEL_7 = 0x3E, /*!< Low delay audio profile, level 7 */
87 DVBPSI_AAC_PROFILE_LOW_DELAY_LEVEL_8 = 0x3F, /*!< Low delay audio profile, level 8 */
88 DVBPSI_AAC_PROFILE_NATURAL_LEVEL_1 = 0x40, /*!< Natural audio profile, level 1 */
89 DVBPSI_AAC_PROFILE_NATURAL_LEVEL_2 = 0x41, /*!< Natural audio profile, level 2 */
90 DVBPSI_AAC_PROFILE_NATURAL_LEVEL_3 = 0x42, /*!< Natural audio profile, level 3 */
91 DVBPSI_AAC_PROFILE_NATURAL_LEVEL_4 = 0x43, /*!< Natural audio profile, level 4 */
92 /** 0x44-0x47 Reserved */
93 DVBPSI_AAC_PROFILE_MOBILE_LEVEL_1 = 0x48, /*!< Mobile audio internetworking profile, level 1 */
94 DVBPSI_AAC_PROFILE_MOBILE_LEVEL_2 = 0x49, /*!< Mobile audio internetworking profile, level 2 */
95 DVBPSI_AAC_PROFILE_MOBILE_LEVEL_3 = 0x4A, /*!< Mobile audio internetworking profile, level 3 */
96 DVBPSI_AAC_PROFILE_MOBILE_LEVEL_4 = 0x4B, /*!< Mobile audio internetworking profile, level 4 */
97 DVBPSI_AAC_PROFILE_MOBILE_LEVEL_5 = 0x4C, /*!< Mobile audio internetworking profile, level 5 */
98 DVBPSI_AAC_PROFILE_MOBILE_LEVEL_6 = 0x4D, /*!< Mobile audio internetworking profile, level 6 */
99 /** 0x4E-0x4F Reserved */
100 DVBPSI_AAC_PROFILE_LEVEL_1 = 0x50, /*!< AAC profile, level 1 */
101 DVBPSI_AAC_PROFILE_LEVEL_2 = 0x51, /*!< AAC profile, level 2 */
102 DVBPSI_AAC_PROFILE_LEVEL_4 = 0x52, /*!< AAC profile, level 4 */
103 DVBPSI_AAC_PROFILE_LEVEL_5 = 0x53, /*!< AAC profile, level 5 */
104 /** 0x54-0x57 RESERVED */
105 DVBPSI_HE_AAC_PROFILE_LEVEL_2 = 0x58, /*!< High efficiency AAC profile, level 2 */
106 DVBPSI_HE_AAC_PROFILE_LEVEL_3 = 0x59, /*!< High efficiency AAC profile, level 3 */
107 DVBPSI_HE_AAC_PROFILE_LEVEL_4 = 0x5A, /*!< High efficiency AAC profile, level 4 */
108 DVBPSI_HE_AAC_PROFILE_LEVEL_5 = 0x5B /*!< High efficiency AAC profile, level 5 */
109 /** 0x5C-0xFF RESERVED */
110} dvbpsi_aac_profile_and_level_t;

112/*!
113 * \enum dvbpsi_aac_type_s
114 * \brief AAC type information as described in table 26, page 41 where column
115 *
116 * The AAC Audio descriptor is used to label the PIDs that carry AAC audio data.
117 * 'stream_content' has value 0x06 as specified in ISO/IEC 13818-1:2007.
118 */
119/*!
120 * \typedef enum dvbpsi_aac_type_s dvbpsi_aac_type_t
121 * \brief AAC type information as descripted in table 26, page 41 where column
122 * 'stream_content' has value 0x06.
123*/
124typedef enum dvbpsi_aac_type_s
125{
126 DVBPSI_AAC_RESERVED0 = 0x00, /*!< reserved for future use */
127 DVBPSI_HE_AAC_MONO = 0x01, /*!< HE-AAC audio, single mono channel */
128 DVBPSI_AAC_RESERVED1 = 0x02, /*!< reserved for future use */
129 DVBPSI_HE_AAC_STEREO = 0x03, /*!< HE-AAC audio, stereo */
130 DVBPSI_AAC_RESERVED2 = 0x04, /*!< reserved for future use */
131 DVBPSI_HE_AAC_SURROUND = 0x05, /*!< HE-AAC audio, surround sound */
132 DVBPSI_AAC_RESERVED3 = 0x06, /*!< 0x06 to 0x3F reserved for future use */
133 DVBPSI_HE_AAC_IMPAIRED = 0x40, /*!< HE-AAC audio description for
134 the visually impaired */
135 DVBPSI_HE_AAC_HEARING = 0x41, /*!< HE-AAC audio for the hard of hearing */
136 DVBPSI_HE_AAC_MIXED = 0x42, /*!< HE-AAC receiver-mixed supplementary
137 audio as per annex E of TS 101 154 */
138 DVBPSI_HE_AAC_V2_STEREO = 0x43, /*!< HE-AAC v2 audio, stereo */
139 DVBPSI_HE_AAC_V2_IMPAIRED = 0x44, /*!< HE-AAC v2 audio description for the
140 visually impaired */
141 DVBPSI_HE_AAC_V2_HEARING = 0x45, /*!< HE-AAC v2 audio for the hard of hearing */
142 DVBPSI_HE_AAC_V2_MIXED = 0x46, /*!< HE-AAC v2 receiver-mixed supplementary
143 audio as per annex E of TS 101 154 */
144 DVBPSI_HE_AAC_MIXED_IMPAIRED = 0x47, /*!< HE-AAC receiver mix audio description
145 for the visually impaired */
146 DVBPSI_HE_AAC_BROADCAST_MIXED_IMPAIRED = 0x48, /*!< HE-AAC broadcaster mix
147 audio description for the visually impaired */
148 DVBPSI_HE_AAC_V2_MIXED_IMPAIRED = 0x49, /*!< HE-AAC v2 receiver mix audio
149 description for the visually impaired */
150 DVBPSI_HE_AAC_V2_BROADCAST_MIXED_IMPAIRED = 0x4A, /*!< HE-AAC v2 broadcaster
151 mix audio description for the visually impaired */
152 DVBPSI_AAC_RESERVED4 = 0x4B, /*!< 0x4B to 0xAF reserved for future use */
153 DVBPSI_AAC_USER = 0xB0, /*!< 0xB0 to 0xFE user-defined */
154 DVBPSI_AAC_RESERVED5 = 0xFF /*!< reserved for future use */
155} dvbpsi_aac_type_t;
156
157/*!
158 * \struct dvbpsi_aac_dr_s
159 * \brief "AAC Audio" descriptor structure.
160 *
161 * The AAC Audio descriptor is used to label the PIDs that carry AAC audio data.
162 */
163/*!
164 * \typedef struct dvbpsi_aac_dr_s dvbpsi_aac_dr_t
165 * \brief dvbpsi_aac_dr_t type definition.
166 */
167typedef struct dvbpsi_aac_dr_s
168{
169 dvbpsi_aac_profile_and_level_t i_profile_and_level;
170 /*!< Specifies the Profile and Level used in MPEG-4 AAC,
171 MPEG-4 HE AAC or MPEG-4 HE AAC v2 according to
172 table 2-71 in ISO/IEC 13818-1:2007/AMD1 [18]. */
173
174 bool b_type; /*!< Indicates presence of AAC type information. */
175 dvbpsi_aac_type_t i_type; /*!< AAC type information, @see enum dvbpsi_aac_type_t */
176
177 uint8_t i_additional_info_length; /*!< Length of additional info bytes fields */
178 uint8_t *p_additional_info; /*!< additional info bytes field. Memory is allocated
179 right after sizeof struct, when freeing this struct
180 the additional info bytes field is freed at the same time. */
181
182} dvbpsi_aac_dr_t;
