30#ifndef _DR_13_H
31#define _DR_13_H
32
33/*****************************************************************************
34 * dvbpsi_carousel_id_dr_s
35 *****************************************************************************/
36/*!
37 * \struct dvbpsi_carousel_id_dr_s
38 * \brief Data Broadcast id Descriptor
39 *
40 * This structure is used to store a decoded Carsouel ID descriptor.
41 */
42/*!
43 * \typedef struct dvbpsi_carousel_id_dr_s dvbpsi_carousel_id_dr_t
44 * \brief dvbpsi_carousel_id_dr_t type definition.
45 */
46typedef struct dvbpsi_carousel_id_dr_s
47{
48 uint32_t i_carousel_id; /*!< carousel identifier */
49 uint8_t i_private_data_len; /*!< length of private data pointer in bytes */
50 uint8_t *p_private_data; /*!< memory is allocated right after sizeof struct,
51 when freeing this struct the private data is
52 freed at the same time. */
53} dvbpsi_carousel_id_dr_t;
54
55/*****************************************************************************
56 * dvbpsi_DecodeCarouselIdDr
57 *****************************************************************************/
58/*!
59 * \fn dvbpsi_carousel_id_dr_t *dvbpsi_DecodeCarouselIdDr(
60 * dvbpsi_descriptor_t *p_descriptor)
61 * \brief Decode a Carousel id descriptor (tag 0x13)
62 * \param p_descriptor Raw descriptor to decode.
63 * \return NULL if the descriptor could not be decoded or a pointer to a
64 * dvbpsi_carousel_id_dr_t structure.
65 */
66dvbpsi_carousel_id_dr_t *dvbpsi_DecodeCarouselIdDr(dvbpsi_descriptor_t *p_descriptor);
67
68#endif
