23/*!
24 * \file <dr_50.h>
25 * \author Corno Roberto <corno.roberto@gmail.com>
26 * \brief Application interface for the time shifted event
27 * descriptor decoder and generator.
28 *
29 * Application interface for the Component descriptor
30 * descriptor decoder and generator. This descriptor's definition can be found in
31 * ETSI EN 300 468 section 6.2.8.
32 */
33
34#ifndef DR_50_H_
35#define DR_50_H_
36
37#ifdef __cplusplus
38extern "C" {
39#endif
40
41/*****************************************************************************
42 * dvbpsi_component_dr_t
43 *****************************************************************************/
44/*!
45 * \struct dvbpsi_component_dr_t
46 * \brief "Component" descriptor structure.
47 *
48 * This structure is used to store a decoded "Component"
49 * descriptor. (ETSI EN 300 468 section 6.2.8).
50 */
51/*!
52 * \typedef struct dvbpsi_component_dr_s dvbpsi_component_dr_t
53 * \brief dvbpsi_component_dr_t type definition.
54 */
55/*!
56 * \struct dvbpsi_component_dr_s
57 * \brief struct dvbpsi_component_dr_s @see dvbpsi_component_dr_t
58 */
59typedef struct dvbpsi_component_dr_t
60{
61 uint8_t i_stream_content; /*!< stream content */
62 uint8_t i_component_type; /*!< component type */
63 uint8_t i_component_tag; /*!< component tag */
64 uint8_t i_iso_639_code[3]; /*!< 3 letter ISO 639 language code */
65 int i_text_length; /*!< text length */
66 uint8_t *i_text; /*!< text */
67
68} dvbpsi_component_dr_t;
69
70/*****************************************************************************
71 * dvbpsi_DecodeComponentDr
72 *****************************************************************************/
73/*!
74 * \fn dvbpsi_component_dr_t * dvbpsi_DecodeComponentDr(
75 dvbpsi_descriptor_t * p_descriptor)
76 * \brief "Component" descriptor decoder.
77 * \param p_descriptor pointer to the descriptor structure
78 * \return a pointer to a new "Component" descriptor structure
79 * which contains the decoded data.
80 */
81dvbpsi_component_dr_t* dvbpsi_DecodeComponentDr(dvbpsi_descriptor_t * p_descriptor);
82
83/*****************************************************************************
84 * dvbpsi_GenComponentDr
85 *****************************************************************************/
86/*!
87 * \fn dvbpsi_descriptor_t *dvbpsi_GenComponentDr(dvbpsi_component_dr_t *p_decoded,
88 bool b_duplicate);
89 * \brief "Component" descriptor generator.
90 * \param p_decoded pointer to a decoded "Component" descriptor structure
91 * \param b_duplicate if true then duplicate the p_decoded structure into
92 * the descriptor
93 * \return a pointer to a new descriptor structure which contains encoded data.
94 */
95dvbpsi_descriptor_t *dvbpsi_GenComponentDr(dvbpsi_component_dr_t * p_decoded,
96 bool b_duplicate);
97
98#ifdef __cplusplus
99};
100#endif
101
102#else
103#error "Multiple inclusions of dr_50.h"
104#endif /* DR_50_H_ */
