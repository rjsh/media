24/*!
25 * \file <dr_5a.h>
26 * \author Johann Hanne
27 * \brief Application interface for the DVB terrestrial delivery system
28 * descriptor decoder and generator.
29 *
30 * Application interface for the DVB terrestrial delivery system descriptor
31 * decoder and generator. This descriptor's definition can be found in
32 * ETSI EN 300 468 section 6.2.13.4.
33 */
34
35#ifndef _DVBPSI_DR_5A_H_
36#define _DVBPSI_DR_5A_H_
37
38#ifdef __cplusplus
39extern "C" {
40#endif
41
42
43/*****************************************************************************
44 * dvbpsi_terr_deliv_sys_dr_t
45 *****************************************************************************/
46/*!
47 * \struct dvbpsi_terr_deliv_sys_dr_s
48 * \brief terrestrial delivery system descriptor structure.
49 *
50 * This structure is used to store a decoded terrestrial delivery system
51 * descriptor. (ETSI EN 300 468 section 6.2.13.4).
52 */
53/*!
54 * \typedef struct dvbpsi_terr_deliv_sys_dr_s dvbpsi_terr_deliv_sys_dr_t
55 * \brief dvbpsi_terr_deliv_sys_dr_t type definition.
56 */
57typedef struct dvbpsi_terr_deliv_sys_dr_s
58{
59 uint32_t i_centre_frequency; /*!< centre frequency */
60 uint8_t i_bandwidth; /*!< bandwidth */
61 uint8_t i_priority; /*!< priority */
62 uint8_t i_time_slice_indicator; /*!< Time_Slicing_indicator */
63 uint8_t i_mpe_fec_indicator; /*!< MPE-FEC_indicator */
64 uint8_t i_constellation; /*!< constellation */
65 uint8_t i_hierarchy_information; /*!< hierarchy_information */
66 uint8_t i_code_rate_hp_stream; /*!< code_rate-HP_stream */
67 uint8_t i_code_rate_lp_stream; /*!< code_rate-LP_stream */
68 uint8_t i_guard_interval; /*!< guard_interval */
69 uint8_t i_transmission_mode; /*!< transmission_mode */
70 uint8_t i_other_frequency_flag; /*!< other_frequency_flag */
71
72} dvbpsi_terr_deliv_sys_dr_t;
73
74
75/*****************************************************************************
76 * dvbpsi_DecodeTerrDelivSysDr
77 *****************************************************************************/
78/*!
79 * \fn dvbpsi_terr_deliv_sys_dr_t * dvbpsi_DecodeTerrDelivSysDr(
80 dvbpsi_descriptor_t * p_descriptor)
81 * \brief terrestrial delivery system descriptor decoder.
82 * \param p_descriptor pointer to the descriptor structure
83 * \return a pointer to a new terrestrial delivery system descriptor structure
84 * which contains the decoded data.
85 */
86dvbpsi_terr_deliv_sys_dr_t* dvbpsi_DecodeTerrDelivSysDr(
87 dvbpsi_descriptor_t * p_descriptor);
88
89
90/*****************************************************************************
91 * dvbpsi_GenTerrDelivSysDr
92 *****************************************************************************/
93/*!
94 * \fn dvbpsi_descriptor_t * dvbpsi_GenTerrDelivSysDr(
95 dvbpsi_terr_deliv_sys_dr_t * p_decoded, bool b_duplicate)
96 * \brief terrestrial delivery system descriptor generator.
97 * \param p_decoded pointer to a decoded terrestrial delivery system descriptor
98 * descriptor structure
99 * \param b_duplicate if true then duplicate the p_decoded structure into
100 * the descriptor
101 * \return a pointer to a new descriptor structure which contains encoded data.
102 */
103dvbpsi_descriptor_t * dvbpsi_GenTerrDelivSysDr(
104 dvbpsi_terr_deliv_sys_dr_t * p_decoded,
105 bool b_duplicate);
106
107
108#ifdef __cplusplus
109};
110#endif
