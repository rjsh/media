24/*!
25 * \file <pat.h>
26 * \author Arnaud de Bossoreille de Ribou <bozo@via.ecp.fr>
27 * \brief Application interface for the PAT decoder and the PAT generator.
28 *
29 * Application interface for the PAT decoder and the PAT generator.
30 * New decoded PAT tables are sent by callback to the application.
31 */
32
33#ifndef _DVBPSI_PAT_H_
34#define _DVBPSI_PAT_H_
35
36#ifdef __cplusplus
37extern "C" {
38#endif
39
40/*****************************************************************************
41 * dvbpsi_pat_program_t
42 *****************************************************************************/
43/*!
44 * \struct dvbpsi_pat_program_s
45 * \brief PAT program structure.
46 *
47 * This structure is used to store a decoded PAT program.
48 * (ISO/IEC 13818-1 section 2.4.4.3).
49 */
50/*!
51 * \typedef struct dvbpsi_pat_program_s dvbpsi_pat_program_t
52 * \brief dvbpsi_pat_program_t type definition.
53 */
54typedef struct dvbpsi_pat_program_s
55{
56 uint16_t i_number; /*!< program_number */
57 uint16_t i_pid; /*!< PID of NIT/PMT */
58
59 struct dvbpsi_pat_program_s * p_next; /*!< next element of
60 the list */
61
62} dvbpsi_pat_program_t;
63
64
65/*****************************************************************************
66 * dvbpsi_pat_t
67 *****************************************************************************/
68/*!
69 * \struct dvbpsi_pat_s
70 * \brief PAT structure.
71 *
72 * This structure is used to store a decoded PAT.
73 * (ISO/IEC 13818-1 section 2.4.4.3).
74 */
75/*!
76 * \typedef struct dvbpsi_pat_s dvbpsi_pat_t
77 * \brief dvbpsi_pat_t type definition.
78 */
79typedef struct dvbpsi_pat_s
80{
81 uint16_t i_ts_id; /*!< transport_stream_id */
82 uint8_t i_version; /*!< version_number */
83 bool b_current_next; /*!< current_next_indicator */
84
85 dvbpsi_pat_program_t * p_first_program; /*!< program list */
86
87} dvbpsi_pat_t;
88
89
90/*****************************************************************************
91 * dvbpsi_pat_callback
92 *****************************************************************************/
93/*!
94 * \typedef void (* dvbpsi_pat_callback)(void* p_cb_data,
95 dvbpsi_pat_t* p_new_pat)
96 * \brief Callback type definition.
97 */
98typedef void (* dvbpsi_pat_callback)(void* p_cb_data, dvbpsi_pat_t* p_new_pat);
99
100/*****************************************************************************
101 * dvbpsi_pat_attach
102 *****************************************************************************/
103/*!
104 * \fn bool dvbpsi_pat_attach(dvbpsi_t *p_dvbpsi, dvbpsi_pat_callback pf_callback, void* p_cb_data)
105 * \brief Creation and initialization of a PAT decoder. The decoder will be attached to 'p_dvbpsi' argument.
106 * \param p_dvbpsi handle to dvbpsi with attached decoder
107 * \param pf_callback function to call back on new PAT
108 * \param p_cb_data private data given in argument to the callback
109 * \return true on success, false on failure
110 */
111bool dvbpsi_pat_attach(dvbpsi_t *p_dvbpsi, dvbpsi_pat_callback pf_callback,
112 void* p_cb_data);
113
114/*****************************************************************************
115 * dvbpsi_pat_detach
116 *****************************************************************************/
117/*!
118 * \fn void dvbpsi_pat_detach(dvbpsi_t *p_dvbpsi)
119 * \brief Destroy a PAT decoder.
120 * \param p_dvbpsi pointer to dvbpsi_t handle
121 * \return nothing.
122 *
123 * The handle isn't valid any more.
124 */
125void dvbpsi_pat_detach(dvbpsi_t *p_dvbpsi);
126
127/*****************************************************************************
128 * dvbpsi_pat_init/dvbpsi_pat_new
129 *****************************************************************************/
130/*!
131 * \fn void dvbpsi_pat_init(dvbpsi_pat_t* p_pat, uint16_t i_ts_id,
132 uint8_t i_version, bool b_current_next)
133 * \brief Initialize a user-allocated dvbpsi_pat_t structure.
134 * \param p_pat pointer to the PAT structure
135 * \param i_ts_id transport stream ID
136 * \param i_version PAT version
137 * \param b_current_next current next indicator
138 * \return nothing.
139 */
140void dvbpsi_pat_init(dvbpsi_pat_t* p_pat, uint16_t i_ts_id, uint8_t i_version,
141 bool b_current_next);
142
143/*!
144 * \fn dvbpsi_pat_t *dvbpsi_pat_new(uint16_t i_ts_id, uint8_t i_version,
145 * bool b_current_next);
146 * \brief Allocate and initialize a new dvbpsi_pat_t structure.
147 * \param i_ts_id transport stream ID
148 * \param i_version PAT version
149 * \param b_current_next current next indicator
150 * \return p_pat pointer to the PAT structure
151 */
152dvbpsi_pat_t *dvbpsi_pat_new(uint16_t i_ts_id, uint8_t i_version, bool b_current_next);
153
154/*****************************************************************************
155 * dvbpsi_pat_empty/dvbpsi_pat_delete
156 *****************************************************************************/
157/*!
158 * \fn void dvbpsi_pat_empty(dvbpsi_pat_t* p_pat)
159 * \brief Clean a dvbpsi_pat_t structure.
160 * \param p_pat pointer to the PAT structure
161 * \return nothing.
162 */
163void dvbpsi_pat_empty(dvbpsi_pat_t* p_pat);
164
165/*!
166 * \fn void dvbpsi_pat_delete(dvbpsi_pat_t *p_pat)
167 * \brief Clean and free a dvbpsi_pat_t structure.
168 * \param p_pat pointer to the PAT structure
169 * \return nothing.
170 */
171void dvbpsi_pat_delete(dvbpsi_pat_t *p_pat);
172
173/*****************************************************************************
174 * dvbpsi_pat_program_add
175 *****************************************************************************/
176/*!
177 * \fn dvbpsi_pat_program_t* dvbpsi_pat_program_add(dvbpsi_pat_t* p_pat,
178 uint16_t i_number,
179 uint16_t i_pid)
180 * \brief Add a program at the end of the PAT.
181 * \param p_pat pointer to the PAT structure
182 * \param i_number program number
183 * \param i_pid PID of the NIT/PMT
184 * \return a pointer to the added program.
185 */
186dvbpsi_pat_program_t* dvbpsi_pat_program_add(dvbpsi_pat_t* p_pat,
187 uint16_t i_number, uint16_t i_pid);
188
189/*****************************************************************************
190 * dvbpsi_pat_sections_generate
191 *****************************************************************************/
192/*!
193 * \fn dvbpsi_psi_section_t* dvbpsi_pat_sections_generate(dvbpsi_t *p_dvbpsi, dvbpsi_pat_t* p_pat,
194 int i_max_pps);
195 * \brief PAT generator.
196 * \param p_dvbpsi handle to dvbpsi with attached decoder
197 * \param p_pat pointer to the PAT structure
198 * \param i_max_pps limitation of the number of program in each section
199 * (max: 253).
200 * \return a pointer to the list of generated PSI sections.
201 *
202 * Generate PAT sections based on the dvbpsi_pat_t structure.
203 */
204dvbpsi_psi_section_t* dvbpsi_pat_sections_generate(dvbpsi_t *p_dvbpsi,
205 dvbpsi_pat_t* p_pat, int i_max_pps);
