466/*!
467 * \brief Inserts a new child \b Node before the existing child \b Node.
468 *
469 * \b refChild can be \c NULL, which inserts \b newChild at the
470 * end of the list of children. Note that the \b Node (or \b Nodes)
471 * in \b newChild must already be owned by the owner document (or have no
472 * owner at all) of \b nodeptr for insertion. If not, the \b Node
473 * (or \b Nodes) must be imported into the document using
474 * \b ixmlDocument_importNode. If \b newChild is already in the tree,
475 * it is removed first.
476 *
477 * \return An integer representing one of the following:
478 * \li \c IXML_SUCCESS: The operation completed successfully.
479 * \li \c IXML_INVALID_PARAMETER: Either \b nodeptr or
480 * \b newChild is \c NULL.
481 * \li \c IXML_HIERARCHY_REQUEST_ERR: The type of the \b Node
482 * does not allow children of the type of \b newChild.
483 * \li \c IXML_WRONG_DOCUMENT_ERR: \b newChild has an owner
484 * document that does not match the owner of \b nodeptr.
485 * \li \c IXML_NO_MODIFICATION_ALLOWED_ERR: \b nodeptr is
486 * read-only or the parent of the \b Node being inserted is
487 * read-only.
488 * \li \c IXML_NOT_FOUND_ERR: \b refChild is not a child of
489 * \b nodeptr.
490 */
491EXPORT_SPEC int ixmlNode_insertBefore(
492 /*! [in] The parent of the \b Node before which to insert the new child. */
493 IXML_Node *nodeptr,
494 /*! [in] The \b Node to insert into the tree. */
495 IXML_Node * newChild,
496 /*! [in] The reference child where the new \b Node should be inserted.
497 * The new \b Node will appear directly before the reference child. */
498 IXML_Node * refChild);

501/*!
502 * \brief Replaces an existing child \b Node with a new child \b Node in the
503 * list of children of a \b Node.
504 *
505 * If \b newChild is already in the tree, it will first be removed.
506 * \b returnNode will contain the \b oldChild \b Node, appropriately removed
507 * from the tree (i.e. it will no longer have an owner document).
508 *
509 * \return An integer representing one of the following:
510 * \li \c IXML_SUCCESS: The operation completed successfully.
511 * \li \c IXML_INVALID_PARAMTER: Either \b nodeptr, \b newChild,
512 * or \b oldChild is \c NULL.
513 * \li \c IXML_HIERARCHY_REQUEST_ERR: The \b newChild is not
514 * a type of \b Node that can be inserted into this tree or
515 * \b newChild is an ancestor of \b nodePtr.
516 * \li \c IXML_WRONG_DOCUMENT_ERR: \b newChild was created from
517 * a different document than \b nodeptr.
518 * \li \c IXML_NO_MODIFICATION_ALLOWED_ERR: \b nodeptr or
519 * its parent is read-only.
520 * \li \c IXML_NOT_FOUND_ERR: \b oldChild is not a child of
521 * \b nodeptr.
522 */
523EXPORT_SPEC int ixmlNode_replaceChild(
524 /*! [in] The parent of the \b Node which contains the child to replace. */
525 IXML_Node *nodeptr,
526 /*! [in] The child with which to replace \b oldChild. */
527 IXML_Node *newChild,
528 /*! [in] The child to replace with \b newChild. */
529 IXML_Node *oldChild,
530 /*! [out] Pointer to a \b Node to place the removed \b oldChild \b Node. */
531 IXML_Node **returnNode);
