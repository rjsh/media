65/*!
66 * \brief Converts an SDK error code into a string error message suitable for
67 * display. The memory returned from this function should NOT be freed.
68 *
69 * \return An ASCII text string representation of the error message associated
70 * with the error code or the string "Unknown error code"
71 */
72EXPORT_SPEC const char *UpnpGetErrorMessage(
73 /*! [in] The SDK error code to convert. */
74 int errorcode);
75
76
77/*!
78 * \brief Combines a base URL and a relative URL into a single absolute URL.
79 *
80 * The memory for \b AbsURL needs to be allocated by the caller and must
81 * be large enough to hold the \b BaseURL and \b RelURL combined.
82 *
83 * \return An integer representing one of the following:
84 * \li <tt>UPNP_E_SUCCESS</tt>: The operation completed successfully.
85 * \li <tt>UPNP_E_INVALID_PARAM</tt>: \b RelURL is <tt>NULL</tt>.
86 * \li <tt>UPNP_E_INVALID_URL</tt>: The \b BaseURL / \b RelURL
87 * combination does not form a valid URL.
88 * \li <tt>UPNP_E_OUTOF_MEMORY</tt>: Insufficient resources exist to
89 * complete this operation.
90 */
91EXPORT_SPEC int UpnpResolveURL(
92 /*! [in] The base URL to combine. */
93 const char *BaseURL,
94 /*! [in] The relative URL to \b BaseURL. */
95 const char *RelURL,
96 /*! [out] A pointer to a buffer to store the absolute URL. */
97 char *AbsURL);
98
99
100/*!
101 * \brief Combines a base URL and a relative URL into a single absolute URL.
102 *
103 * The memory for \b AbsURL becomes owned by the caller and should be freed
104 * later.
105 *
106 * \return An integer representing one of the following:
107 * \li <tt>UPNP_E_SUCCESS</tt>: The operation completed successfully.
108 * \li <tt>UPNP_E_INVALID_PARAM</tt>: \b RelURL is <tt>NULL</tt>.
109 * \li <tt>UPNP_E_INVALID_URL</tt>: The \b BaseURL / \b RelURL
110 * combination does not form a valid URL.
111 * \li <tt>UPNP_E_OUTOF_MEMORY</tt>: Insufficient resources exist to
112 * complete this operation.
113 */
114EXPORT_SPEC int UpnpResolveURL2(
115 /*! [in] The base URL to combine. */
116 const char *BaseURL,
117 /*! [in] The relative URL to \b BaseURL. */
118 const char *RelURL,
119 /*! [out] A pointer to a pointer to a buffer to store the
120 * absolute URL. Must be freed later by the caller. */
121 char **AbsURL);
122

124/*!
125 * \brief Creates an action request packet based on its input parameters
126 * (status variable name and value pair).
127 *
128 * Any number of input parameters can be passed to this function but every
129 * input variable name should have a matching value argument.
130 *
131 * It is a wrapper function that calls makeAction() function to create the
132 * action request.
133 *
134 * \return The action node of \b Upnp_Document type or <tt>NULL</tt> if the
135 * operation failed.
136 */
137EXPORT_SPEC IXML_Document *UpnpMakeAction(
138 /*! [in] Name of the action request or response. */
139 const char *ActionName,
140 /*! [in] The service type. */
141 const char *ServType,
142 /*! [in] Number of argument pairs to be passed. */
143 int NumArg,
144 /*! [in] pointer to the first argument. */
145 const char *Arg,
146 /*! [in] Argument list. */
147 ...);
148
149
150/*!
151 * \brief Ceates an action response packet based on its output parameters
152 * (status variable name and value pair).
153 *
154 * Any number of input parameters can be passed to this function but every
155 * output variable name should have a matching value argument.
156 *
157 * It is a wrapper function that calls makeAction() function to create the
158 * action request.
159 *
160 * \return The action node of \b Upnp_Document type or <tt>NULL</tt> if the
161 * operation failed.
162 */
163EXPORT_SPEC IXML_Document *UpnpMakeActionResponse(
164 /*! [in] The action name. */
165 const char *ActionName,
166 /*! [in] The service type.. */
167 const char *ServType,
168 /*! [in] The number of argument pairs passed. */
169 int NumArg,
170 /*! [in] The status variable name and value pair. */
171 const char *Arg,
172 /*! [in] Other status variable name and value pairs. */
173 ...);
174
175
176/*!
177 * \brief Adds the argument in the action request.
178 *
179 * This API is specially suitable inside a loop to add any number input
180 * parameters into an existing action. If no action document exists in the
181 * beginning then a <b>Upnp_Document variable initialized with <tt>NULL</tt></b>
182 * should be passed as a parameter.
183 *
184 * It is a wrapper function that calls addToAction() function to add the
185 * argument in the action request.
186 *
187 * \return An integer representing one of the following:
188 * \li <tt>UPNP_E_SUCCESS</tt>: The operation completed successfully.
189 * \li <tt>UPNP_E_INVALID_PARAM</tt>: One or more of the parameters are invalid.
190 * \li <tt>UPNP_E_OUTOF_MEMORY</tt>: Insufficient resources exist to
191 * complete this operation.
192 */
193EXPORT_SPEC int UpnpAddToAction(
194 /*! [in,out] A pointer to store the action document node. */
195 IXML_Document **ActionDoc,
196 /*! [in] The action name. */
197 const char *ActionName,
198 /*! [in] The service type. */
199 const char *ServType,
200 /*! [in] The status variable name. */
201 const char *ArgName,
202 /*! [in] The status variable value. */
203 const char *ArgVal);
204
205
206/*!
207 * \brief Creates an action response packet based on its output parameters
208 * (status variable name and value pair).
209 *
210 * This API is especially suitable inside a loop to add any number of input
211 * parameters into an existing action response. If no action document exists
212 * in the beginning, a \b Upnp_Document variable initialized with <tt>NULL</tt>
213 * should be passed as a parameter.
214 *
215 * It is a wrapper function that calls addToAction() function to add the
216 * argument in the action request.
217 *
218 * \return An integer representing one of the following:
219 * \li <tt>UPNP_E_SUCCESS</tt>: The operation completed successfully.
220 * \li <tt>UPNP_E_INVALID_PARAM</tt>: One or more of the parameters are invalid.
221 * \li <tt>UPNP_E_OUTOF_MEMORY</tt>: Insufficient resources exist to
222 * complete this operation.
223 */
224EXPORT_SPEC int UpnpAddToActionResponse(
225 /*! [in,out] Pointer to a document to store the action document node. */
226 IXML_Document **ActionResponse,
227 /*! [in] The action name. */
228 const char *ActionName,
229 /*! [in] The service type. */
230 const char *ServType,
231 /*! [in] The status variable name. */
232 const char *ArgName,
233 /*! [in] The status variable value. */
234 const char *ArgVal);
235
236
237/*!
238 * \brief Creates a property set message packet.
239 *
240 * Any number of input parameters can be passed to this function but every
241 * input variable name should have a matching value input argument.
242 *
243 * \return <tt>NULL</tt> on failure, or the property-set document node.
244 */
245EXPORT_SPEC IXML_Document *UpnpCreatePropertySet(
246 /*! [in] The number of argument pairs passed. */
247 int NumArg,
248 /*! [in] The status variable name and value pair. */
249 const char *Arg,
250 /*! [in] Variable sized list with the rest of the parameters. */
251 ...);
252
253
254/*!
255 * \brief Can be used when an application needs to transfer the status of many
256 * variables at once.
257 *
258 * It can be used (inside a loop) to add some extra status variables into an
259 * existing property set. If the application does not already have a property
260 * set document, the application should create a variable initialized with
261 * <tt>NULL</tt> and pass that as the first parameter.
262 *
263 * \return An integer representing one of the following:
264 * \li <tt>UPNP_E_SUCCESS</tt>: The operation completed successfully.
265 * \li <tt>UPNP_E_INVALID_PARAM</tt>: One or more of the parameters are invalid.
266 * \li <tt>UPNP_E_OUTOF_MEMORY</tt>: Insufficient resources exist to
267 * complete this operation.
268 */
269EXPORT_SPEC int UpnpAddToPropertySet(
270 /*! [in,out] A pointer to the document containing the property set document node. */
271 IXML_Document **PropSet,
272 /*! [in] The status variable name. */
273 const char *ArgName,
274 /*! [in] The status variable value. */
275 const char *ArgVal);
