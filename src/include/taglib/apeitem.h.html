26#ifndef TAGLIB_APEITEM_H
27#define TAGLIB_APEITEM_H
28
29#include "tbytevector.h"
30#include "tstring.h"
31#include "tstringlist.h"
32
33namespace TagLib {
34
35 namespace APE {
36
37 //! An implementation of APE-items
38
39 /*!
40 * This class provides the features of items in the APEv2 standard.
41 */
42 class TAGLIB_EXPORT Item
43 {
44 public:
45 /*!
46 * Enum of types an Item can have. The value of 3 is reserved.
47 */
48 enum ItemTypes {
49 //! Item contains text information coded in UTF-8
50 Text = 0,
51 //! Item contains binary information
52 Binary = 1,
53 //! Item is a locator of external stored information
54 Locator = 2
55 };
56 /*!
57 * Constructs an empty item.
58 */
59 Item();
60
61 /*!
62 * Constructs a text item with \a key and \a value.
63 */
64 // BIC: Remove this, StringList has a constructor from a single string
65 Item(const String &key, const String &value);
66
67 /*!
68 * Constructs a text item with \a key and \a values.
69 */
70 Item(const String &key, const StringList &values);
71
72 /*!
73 * Constructs an item with \a key and \a value.
74 * If \a binary is true a Binary item will be created, otherwise \a value will be interpreted as text
75 */
76 Item(const String &key, const ByteVector &value, bool binary);
77
78 /*!
79 * Construct an item as a copy of \a item.
80 */
81 Item(const Item &item);
82
83 /*!
84 * Destroys the item.
85 */
86 virtual ~Item();
87
88 /*!
89 * Copies the contents of \a item into this item.
90 */
91 Item &operator=(const Item &item);
92
93 /*!
94 * Returns the key.
95 */
96 String key() const;
97
98 /*!
99 * Returns the binary value.
100 * If the item type is not \a Binary, always returns an empty ByteVector.
101 */
102 ByteVector binaryData() const;
103
104 /*!
105 * Set the binary value to \a value
106 * The item's type will also be set to \a Binary
107 */
108 void setBinaryData(const ByteVector &value);
109
110#ifndef DO_NOT_DOCUMENT
111 /* Remove in next binary incompatible release */
112 ByteVector value() const;
113#endif
114
115 /*!
116 * Sets the key for the item to \a key.
117 */
118 void setKey(const String &key);
119
120 /*!
121 * Sets the text value of the item to \a value and clears any previous contents.
122 *
123 * \see toString()
124 */
125 void setValue(const String &value);
126
127 /*!
128 * Sets the text value of the item to the list of values in \a value and clears
129 * any previous contents.
130 *
131 * \see toStringList()
132 */
133 void setValues(const StringList &values);
134
135 /*!
136 * Appends \a value to create (or extend) the current list of text values.
137 *
138 * \see toString()
139 */
140 void appendValue(const String &value);
141
142 /*!
143 * Appends \a values to extend the current list of text values.
144 *
145 * \see toStringList()
146 */
147 void appendValues(const StringList &values);
148
149 /*!
150 * Returns the size of the full item.
151 */
152 int size() const;
153
154 /*!
155 * Returns the value as a single string. In case of multiple strings,
156 * the first is returned. If the data type is not \a Text, always returns
157 * an empty String.
158 */
159 String toString() const;
160
161#ifndef DO_NOT_DOCUMENT
162 /* Remove in next binary incompatible release */
163 StringList toStringList() const;
164#endif
165
166 /*!
167 * Returns the list of text values. If the data type is not \a Text, always
168 * returns an empty StringList.
169 */
170 StringList values() const;
171
172 /*!
173 * Render the item to a ByteVector.
174 */
175 ByteVector render() const;
176
177 /*!
178 * Parse the item from the ByteVector \a data.
179 */
180 void parse(const ByteVector& data);
181
182 /*!
183 * Set the item to read-only.
184 */
185 void setReadOnly(bool readOnly);
186
187 /*!
188 * Return true if the item is read-only.
189 */
190 bool isReadOnly() const;
191
192 /*!
193 * Sets the type of the item to \a type.
194 *
195 * \see ItemTypes
196 */
197 void setType(ItemTypes type);
198
199 /*!
200 * Returns the type of the item.
201 */
202 ItemTypes type() const;
203
204 /*!
205 * Returns if the item has any real content.
206 */
207 bool isEmpty() const;
208
209 private:
210 class ItemPrivate;
211 ItemPrivate *d;
212 };
213 }
214
215}
216
217#endif
