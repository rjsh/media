42 //! An ID3v2 text identification frame implementation
43
44 /*!
45 * This is an implementation of the most common type of ID3v2 frame -- text
46 * identification frames. There are a number of variations on this. Those
47 * enumerated in the ID3v2.4 standard are:
48 *
49 * <ul>
50 * <li><b>TALB</b> Album/Movie/Show title</li>
51 * <li><b>TBPM</b> BPM (beats per minute)</li>
52 * <li><b>TCOM</b> Composer</li>
53 * <li><b>TCON</b> Content type</li>
54 * <li><b>TCOP</b> Copyright message</li>
55 * <li><b>TDEN</b> Encoding time</li>
56 * <li><b>TDLY</b> Playlist delay</li>
57 * <li><b>TDOR</b> Original release time</li>
58 * <li><b>TDRC</b> Recording time</li>
59 * <li><b>TDRL</b> Release time</li>
60 * <li><b>TDTG</b> Tagging time</li>
61 * <li><b>TENC</b> Encoded by</li>
62 * <li><b>TEXT</b> Lyricist/Text writer</li>
63 * <li><b>TFLT</b> File type</li>
64 * <li><b>TIPL</b> Involved people list</li>
65 * <li><b>TIT1</b> Content group description</li>
66 * <li><b>TIT2</b> Title/songname/content description</li>
67 * <li><b>TIT3</b> Subtitle/Description refinement</li>
68 * <li><b>TKEY</b> Initial key</li>
69 * <li><b>TLAN</b> Language(s)</li>
70 * <li><b>TLEN</b> Length</li>
71 * <li><b>TMCL</b> Musician credits list</li>
72 * <li><b>TMED</b> Media type</li>
73 * <li><b>TMOO</b> Mood</li>
74 * <li><b>TOAL</b> Original album/movie/show title</li>
75 * <li><b>TOFN</b> Original filename</li>
76 * <li><b>TOLY</b> Original lyricist(s)/text writer(s)</li>
77 * <li><b>TOPE</b> Original artist(s)/performer(s)</li>
78 * <li><b>TOWN</b> File owner/licensee</li>
79 * <li><b>TPE1</b> Lead performer(s)/Soloist(s)</li>
80 * <li><b>TPE2</b> Band/orchestra/accompaniment</li>
81 * <li><b>TPE3</b> Conductor/performer refinement</li>
82 * <li><b>TPE4</b> Interpreted, remixed, or otherwise modified by</li>
83 * <li><b>TPOS</b> Part of a set</li>
84 * <li><b>TPRO</b> Produced notice</li>
85 * <li><b>TPUB</b> Publisher</li>
86 * <li><b>TRCK</b> Track number/Position in set</li>
87 * <li><b>TRSN</b> Internet radio station name</li>
88 * <li><b>TRSO</b> Internet radio station owner</li>
89 * <li><b>TSOA</b> Album sort order</li>
90 * <li><b>TSOP</b> Performer sort order</li>
91 * <li><b>TSOT</b> Title sort order</li>
92 * <li><b>TSRC</b> ISRC (international standard recording code)</li>
93 * <li><b>TSSE</b> Software/Hardware and settings used for encoding</li>
94 * <li><b>TSST</b> Set subtitle</li>
95 * </ul>
96 *
97 * The ID3v2 Frames document gives a description of each of these formats
98 * and the expected order of strings in each. ID3v2::Header::frameID() can
99 * be used to determine the frame type.
100 *
101 * \note If non-Latin1 compatible strings are used with this class, even if
102 * the text encoding is set to Latin1, the frame will be written using UTF8
103 * (with the encoding flag appropriately set in the output).
104 */
105
106 class TAGLIB_EXPORT TextIdentificationFrame : public Frame
107 {
108 friend class FrameFactory;
109
110 public:
111 /*!
112 * Construct an empty frame of type \a type. Uses \a encoding as the
113 * default text encoding.
114 *
115 * \note In this case you must specify the text encoding as it
116 * resolves the ambiguity between constructors.
117 *
118 * \note Please see the note in the class description regarding Latin1.
119 */
120 TextIdentificationFrame(const ByteVector &type, String::Type encoding);
121
122 /*!
123 * This is a dual purpose constructor. \a data can either be binary data
124 * that should be parsed or (at a minimum) the frame ID.
125 */
126 explicit TextIdentificationFrame(const ByteVector &data);
127
128 /*!
129 * This is a special factory method to create a TIPL (involved people list)
130 * frame from the given \a properties. Will parse key=[list of values] data
131 * into the TIPL format as specified in the ID3 standard.
132 */
133 static TextIdentificationFrame *createTIPLFrame(const PropertyMap &properties);
134
135 /*!
136 * This is a special factory method to create a TMCL (musician credits list)
137 * frame from the given \a properties. Will parse key=[list of values] data
138 * into the TMCL format as specified in the ID3 standard, where key should be
139 * of the form instrumentPrefix:instrument.
140 */
141 static TextIdentificationFrame *createTMCLFrame(const PropertyMap &properties);
142 /*!
143 * Destroys this TextIdentificationFrame instance.
144 */
145 virtual ~TextIdentificationFrame();
146
147 /*!
148 * Text identification frames are a list of string fields.
149 *
150 * This function will accept either a StringList or a String (using the
151 * StringList constructor that accepts a single String).
152 *
153 * \note This will not change the text encoding of the frame even if the
154 * strings passed in are not of the same encoding. Please use
155 * setEncoding(s.type()) if you wish to change the encoding of the frame.
156 */
157 void setText(const StringList &l);
158
159 // Reimplementations.
160
161 virtual void setText(const String &s);
162 virtual String toString() const;
163
164 /*!
165 * Returns the text encoding that will be used in rendering this frame.
166 * This defaults to the type that was either specified in the constructor
167 * or read from the frame when parsed.
168 *
169 * \note Please see the note in the class description regarding Latin1.
170 *
171 * \see setTextEncoding()
172 * \see render()
173 */
174 String::Type textEncoding() const;
175
176 /*!
177 * Sets the text encoding to be used when rendering this frame to
178 * \a encoding.
179 *
180 * \note Please see the note in the class description regarding Latin1.
181 *
182 * \see textEncoding()
183 * \see render()
184 */
185 void setTextEncoding(String::Type encoding);
186
187 /*!
188 * Returns a list of the strings in this frame.
189 */
190 StringList fieldList() const;
191
192 /*!
193 * Returns a KeyConversionMap mapping a role as it would be used in a PropertyMap
194 * to the corresponding key used in a TIPL ID3 frame to describe that role.
195 */
196 static const KeyConversionMap &involvedPeopleMap();
197
198 PropertyMap asProperties() const;
199
200 protected:
201 // Reimplementations.
202
203 virtual void parseFields(const ByteVector &data);
204 virtual ByteVector renderFields() const;
205
206 /*!
207 * The constructor used by the FrameFactory.
208 */
209 TextIdentificationFrame(const ByteVector &data, Header *h);
210
211 private:
212 TextIdentificationFrame(const TextIdentificationFrame &);
213 TextIdentificationFrame &operator=(const TextIdentificationFrame &);
214
215 /*!
216 * Parses the special structure of a TIPL frame
217 * Only the whitelisted roles "ARRANGER", "ENGINEER", "PRODUCER",
218 * "DJMIXER" (ID3: "DJ-MIX") and "MIXER" (ID3: "MIX") are allowed.
219 */
220 PropertyMap makeTIPLProperties() const;
221 /*!
222 * Parses the special structure of a TMCL frame.
223 */
224 PropertyMap makeTMCLProperties() const;
225 class TextIdentificationFramePrivate;
226 TextIdentificationFramePrivate *d;
227 };
228
229 /*!
230 * This is a specialization of text identification frames that allows for
231 * user defined entries. Each entry has a description in addition to the
232 * normal list of fields that a text identification frame has.
233 *
234 * This description identifies the frame and must be unique.
235 */
236
237 //! An ID3v2 custom text identification frame implementationx
238
239 class TAGLIB_EXPORT UserTextIdentificationFrame : public TextIdentificationFrame
240 {
241 friend class FrameFactory;
242
243 public:
244 /*!
245 * Constructs an empty user defined text identification frame. For this to be
246 * a useful frame both a description and text must be set.
247 */
248 explicit UserTextIdentificationFrame(String::Type encoding = String::Latin1);
249
250 /*!
251 * Creates a frame based on \a data.
252 */
253 explicit UserTextIdentificationFrame(const ByteVector &data);
254
255 /*!
256 * Creates a user defined text identification frame with the given \a description
257 * and \a values.
258 */
259 UserTextIdentificationFrame(const String &description, const StringList &values, String::Type encoding = String::UTF8);
260
261 virtual String toString() const;
262
263 /*!
264 * Returns the description for this frame.
265 */
266 String description() const;
267
268 /*!
269 * Sets the description of the frame to \a s. \a s must be unique. You can
270 * check for the presence of another user defined text frame of the same type
271 * using find() and testing for null.
272 */
273 void setDescription(const String &s);
274
275 StringList fieldList() const;
276 void setText(const String &text);
277 void setText(const StringList &fields);
278
279 /*!
280 * A UserTextIdentificationFrame is parsed into a PropertyMap as follows:
281 * - the key is the frame's description, uppercased
282 * - if the description contains '::', only the substring after that
283 * separator is considered as key (compatibility with exfalso)
284 * - if the above rules don't yield a valid key (e.g. containing non-ASCII
285 * characters), the returned map will contain an entry "TXXX/<description>"
286 * in its unsupportedData() list.
287 * - The values will be copies of the fieldList().
288 * - If the description() appears as value in fieldList(), it will be omitted
289 * in the value list, in order to be compatible with TagLib which copies
290 * the description() into the fieldList().
291 */
292 PropertyMap asProperties() const;
293
294 /*!
295 * Searches for the user defined text frame with the description \a description
296 * in \a tag. This returns null if no matching frames were found.
297 */
298 static UserTextIdentificationFrame *find(Tag *tag, const String &description);
299
300 private:
301 UserTextIdentificationFrame(const ByteVector &data, Header *h);
302 UserTextIdentificationFrame(const TextIdentificationFrame &);
303 UserTextIdentificationFrame &operator=(const UserTextIdentificationFrame &);
304
305 void checkFields();
306
307 class UserTextIdentificationFramePrivate;
308 UserTextIdentificationFramePrivate *d;
309 };
