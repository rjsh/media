26#ifndef TAGLIB_BYTEVECTOR_H
27#define TAGLIB_BYTEVECTOR_H
28
29#include "taglib.h"
30#include "taglib_export.h"
31
32#include <vector>
33#include <iostream>
34
35namespace TagLib {
36
37 //! A byte vector
38
39 /*!
40 * This class provides a byte vector with some methods that are useful for
41 * tagging purposes. Many of the search functions are tailored to what is
42 * useful for finding tag related paterns in a data array.
43 */
44
45 class TAGLIB_EXPORT ByteVector
46 {
47 public:
48#ifndef DO_NOT_DOCUMENT
49 typedef std::vector<char>::iterator Iterator;
50 typedef std::vector<char>::const_iterator ConstIterator;
51 typedef std::vector<char>::reverse_iterator ReverseIterator;
52 typedef std::vector<char>::const_reverse_iterator ConstReverseIterator;
53#endif
54
55 /*!
56 * Constructs an empty byte vector.
57 */
58 ByteVector();
59
60 /*!
61 * Construct a vector of size \a size with all values set to \a value by
62 * default.
63 */
64 ByteVector(uint size, char value = 0);
65
66 /*!
67 * Constructs a byte vector that is a copy of \a v.
68 */
69 ByteVector(const ByteVector &v);
70
71 /*!
72 * Constructs a byte vector that is a copy of \a v.
73 */
74 ByteVector(const ByteVector &v, uint offset, uint length);
75
76 /*!
77 * Constructs a byte vector that contains \a c.
78 */
79 ByteVector(char c);
80
81 /*!
82 * Constructs a byte vector that copies \a data for up to \a length bytes.
83 */
84 ByteVector(const char *data, uint length);
85
86 /*!
87 * Constructs a byte vector that copies \a data up to the first null
88 * byte. The behavior is undefined if \a data is not null terminated.
89 * This is particularly useful for constructing byte arrays from string
90 * constants.
91 */
92 ByteVector(const char *data);
93
94 /*!
95 * Destroys this ByteVector instance.
96 */
97 virtual ~ByteVector();
98
99 /*!
100 * Sets the data for the byte array using the first \a length bytes of \a data
101 */
102 ByteVector &setData(const char *data, uint length);
103
104 /*!
105 * Sets the data for the byte array copies \a data up to the first null
106 * byte. The behavior is undefined if \a data is not null terminated.
107 */
108 ByteVector &setData(const char *data);
109
110 /*!
111 * Returns a pointer to the internal data structure.
112 *
113 * \warning Care should be taken when modifying this data structure as it is
114 * easy to corrupt the ByteVector when doing so. Specifically, while the
115 * data may be changed, its length may not be.
116 */
117 char *data();
118
119 /*!
120 * Returns a pointer to the internal data structure which may not be modified.
121 */
122 const char *data() const;
123
124 /*!
125 * Returns a byte vector made up of the bytes starting at \a index and
126 * for \a length bytes. If \a length is not specified it will return the bytes
127 * from \a index to the end of the vector.
128 */
129 ByteVector mid(uint index, uint length = 0xffffffff) const;
130
131 /*!
132 * This essentially performs the same as operator[](), but instead of causing
133 * a runtime error if the index is out of bounds, it will return a null byte.
134 */
135 char at(uint index) const;
136
137 /*!
138 * Searches the ByteVector for \a pattern starting at \a offset and returns
139 * the offset. Returns -1 if the pattern was not found. If \a byteAlign is
140 * specified the pattern will only be matched if it starts on a byte divisible
141 * by \a byteAlign (starting from \a offset).
142 */
143 int find(const ByteVector &pattern, uint offset = 0, int byteAlign = 1) const;
144
145 /*!
146 * Searches the char for \a c starting at \a offset and returns
147 * the offset. Returns \a npos if the pattern was not found. If \a byteAlign is
148 * specified the pattern will only be matched if it starts on a byte divisible
149 * by \a byteAlign (starting from \a offset).
150 */
151 int find(char c, uint offset = 0, int byteAlign = 1) const;
152
153 /*!
154 * Searches the ByteVector for \a pattern starting from either the end of the
155 * vector or \a offset and returns the offset. Returns -1 if the pattern was
156 * not found. If \a byteAlign is specified the pattern will only be matched
157 * if it starts on a byte divisible by \a byteAlign (starting from \a offset).
158 */
159 int rfind(const ByteVector &pattern, uint offset = 0, int byteAlign = 1) const;
160
161 /*!
162 * Checks to see if the vector contains the \a pattern starting at position
163 * \a offset. Optionally, if you only want to search for part of the pattern
164 * you can specify an offset within the pattern to start from. Also, you can
165 * specify to only check for the first \a patternLength bytes of \a pattern with
166 * the \a patternLength argument.
167 */
168 bool containsAt(const ByteVector &pattern, uint offset, uint patternOffset = 0, uint patternLength = 0xffffffff) const;
169
170 /*!
171 * Returns true if the vector starts with \a pattern.
172 */
173 bool startsWith(const ByteVector &pattern) const;
174
175 /*!
176 * Returns true if the vector ends with \a pattern.
177 */
178 bool endsWith(const ByteVector &pattern) const;
179
180 /*!
181 * Replaces \a pattern with \a with and returns a reference to the ByteVector
182 * after the operation. This \e does modify the vector.
183 */
184 ByteVector &replace(const ByteVector &pattern, const ByteVector &with);
185
186 /*!
187 * Checks for a partial match of \a pattern at the end of the vector. It
188 * returns the offset of the partial match within the vector, or -1 if the
189 * pattern is not found. This method is particularly useful when searching for
190 * patterns that start in one vector and end in another. When combined with
191 * startsWith() it can be used to find a pattern that overlaps two buffers.
192 *
193 * \note This will not match the complete pattern at the end of the string; use
194 * endsWith() for that.
195 */
196 int endsWithPartialMatch(const ByteVector &pattern) const;
197
198 /*!
199 * Appends \a v to the end of the ByteVector.
200 */
201 ByteVector &append(const ByteVector &v);
202
203 /*!
204 * Clears the data.
205 */
206 ByteVector &clear();
207
208 /*!
209 * Returns the size of the array.
210 */
211 uint size() const;
212
213 /*!
214 * Resize the vector to \a size. If the vector is currently less than
215 * \a size, pad the remaining spaces with \a padding. Returns a reference
216 * to the resized vector.
217 */
218 ByteVector &resize(uint size, char padding = 0);
219
220 /*!
221 * Returns an Iterator that points to the front of the vector.
222 */
223 Iterator begin();
224
225 /*!
226 * Returns a ConstIterator that points to the front of the vector.
227 */
228 ConstIterator begin() const;
229
230 /*!
231 * Returns an Iterator that points to the back of the vector.
232 */
233 Iterator end();
234
235 /*!
236 * Returns a ConstIterator that points to the back of the vector.
237 */
238 ConstIterator end() const;
239
240 /*!
241 * Returns a ReverseIterator that points to the front of the vector.
242 */
243 ReverseIterator rbegin();
244
245 /*!
246 * Returns a ConstReverseIterator that points to the front of the vector.
247 */
248 ConstReverseIterator rbegin() const;
249
250 /*!
251 * Returns a ReverseIterator that points to the back of the vector.
252 */
253 ReverseIterator rend();
254
255 /*!
256 * Returns a ConstReverseIterator that points to the back of the vector.
257 */
258 ConstReverseIterator rend() const;
259
260 /*!
261 * Returns true if the vector is null.
262 *
263 * \note A vector may be empty without being null.
264 * \see isEmpty()
265 */
266 bool isNull() const;
267
268 /*!
269 * Returns true if the ByteVector is empty.
270 *
271 * \see size()
272 * \see isNull()
273 */
274 bool isEmpty() const;
275
276 /*!
277 * Returns a CRC checksum of the byte vector's data.
278 */
279 uint checksum() const;
280
281 /*!
282 * Converts the first 4 bytes of the vector to an unsigned integer.
283 *
284 * If \a mostSignificantByteFirst is true this will operate left to right
285 * evaluating the integer. For example if \a mostSignificantByteFirst is
286 * true then $00 $00 $00 $01 == 0x00000001 == 1, if false, $01 00 00 00 ==
287 * 0x01000000 == 1.
288 *
289 * \see fromUInt()
290 */
291 uint toUInt(bool mostSignificantByteFirst = true) const;
292
293 /*!
294 * Converts the 4 bytes at \a offset of the vector to an unsigned integer.
295 *
296 * If \a mostSignificantByteFirst is true this will operate left to right
297 * evaluating the integer. For example if \a mostSignificantByteFirst is
298 * true then $00 $00 $00 $01 == 0x00000001 == 1, if false, $01 00 00 00 ==
299 * 0x01000000 == 1.
300 *
301 * \see fromUInt()
302 */
303 uint toUInt(uint offset, bool mostSignificantByteFirst = true) const;
304
305 /*!
306 * Converts the \a length bytes at \a offset of the vector to an unsigned
307 * integer. If \a length is larger than 4, the excess is ignored.
308 *
309 * If \a mostSignificantByteFirst is true this will operate left to right
310 * evaluating the integer. For example if \a mostSignificantByteFirst is
311 * true then $00 $00 $00 $01 == 0x00000001 == 1, if false, $01 00 00 00 ==
312 * 0x01000000 == 1.
313 *
314 * \see fromUInt()
315 */
316 uint toUInt(uint offset, uint length, bool mostSignificantByteFirst = true) const;
317
318 /*!
319 * Converts the first 2 bytes of the vector to a (signed) short.
320 *
321 * If \a mostSignificantByteFirst is true this will operate left to right
322 * evaluating the integer. For example if \a mostSignificantByteFirst is
323 * true then $00 $01 == 0x0001 == 1, if false, $01 00 == 0x01000000 == 1.
324 *
325 * \see fromShort()
326 */
327 short toShort(bool mostSignificantByteFirst = true) const;
328
329 /*!
330 * Converts the 2 bytes at \a offset of the vector to a (signed) short.
331 *
332 * If \a mostSignificantByteFirst is true this will operate left to right
333 * evaluating the integer. For example if \a mostSignificantByteFirst is
334 * true then $00 $01 == 0x0001 == 1, if false, $01 00 == 0x01000000 == 1.
335 *
336 * \see fromShort()
337 */
338 short toShort(uint offset, bool mostSignificantByteFirst = true) const;
339
340 /*!
341 * Converts the first 2 bytes of the vector to a unsigned short.
342 *
343 * If \a mostSignificantByteFirst is true this will operate left to right
344 * evaluating the integer. For example if \a mostSignificantByteFirst is
345 * true then $00 $01 == 0x0001 == 1, if false, $01 00 == 0x01000000 == 1.
346 *
347 * \see fromShort()
348 */
349 unsigned short toUShort(bool mostSignificantByteFirst = true) const;
350
351 /*!
352 * Converts the 2 bytes at \a offset of the vector to a unsigned short.
353 *
354 * If \a mostSignificantByteFirst is true this will operate left to right
355 * evaluating the integer. For example if \a mostSignificantByteFirst is
356 * true then $00 $01 == 0x0001 == 1, if false, $01 00 == 0x01000000 == 1.
357 *
358 * \see fromShort()
359 */
360 unsigned short toUShort(uint offset, bool mostSignificantByteFirst = true) const;
361
362 /*!
363 * Converts the first 8 bytes of the vector to a (signed) long long.
364 *
365 * If \a mostSignificantByteFirst is true this will operate left to right
366 * evaluating the integer. For example if \a mostSignificantByteFirst is
367 * true then $00 00 00 00 00 00 00 01 == 0x0000000000000001 == 1,
368 * if false, $01 00 00 00 00 00 00 00 == 0x0100000000000000 == 1.
369 *
370 * \see fromUInt()
371 */
372 long long toLongLong(bool mostSignificantByteFirst = true) const;
373
374 /*!
375 * Converts the 8 bytes at \a offset of the vector to a (signed) long long.
376 *
377 * If \a mostSignificantByteFirst is true this will operate left to right
378 * evaluating the integer. For example if \a mostSignificantByteFirst is
379 * true then $00 00 00 00 00 00 00 01 == 0x0000000000000001 == 1,
380 * if false, $01 00 00 00 00 00 00 00 == 0x0100000000000000 == 1.
381 *
382 * \see fromUInt()
383 */
384 long long toLongLong(uint offset, bool mostSignificantByteFirst = true) const;
385
386 /*!
387 * Creates a 4 byte ByteVector based on \a value. If
388 * \a mostSignificantByteFirst is true, then this will operate left to right
389 * in building the ByteVector. For example if \a mostSignificantByteFirst is
390 * true then $00 00 00 01 == 0x00000001 == 1, if false, $01 00 00 00 ==
391 * 0x01000000 == 1.
392 *
393 * \see toUInt()
394 */
395 static ByteVector fromUInt(uint value, bool mostSignificantByteFirst = true);
396
397 /*!
398 * Creates a 2 byte ByteVector based on \a value. If
399 * \a mostSignificantByteFirst is true, then this will operate left to right
400 * in building the ByteVector. For example if \a mostSignificantByteFirst is
401 * true then $00 01 == 0x0001 == 1, if false, $01 00 == 0x0100 == 1.
402 *
403 * \see toShort()
404 */
405 static ByteVector fromShort(short value, bool mostSignificantByteFirst = true);
406
407 /*!
408 * Creates a 8 byte ByteVector based on \a value. If
409 * \a mostSignificantByteFirst is true, then this will operate left to right
410 * in building the ByteVector. For example if \a mostSignificantByteFirst is
411 * true then $00 00 00 01 == 0x0000000000000001 == 1, if false,
412 * $01 00 00 00 00 00 00 00 == 0x0100000000000000 == 1.
413 *
414 * \see toLongLong()
415 */
416 static ByteVector fromLongLong(long long value, bool mostSignificantByteFirst = true);
417
418 /*!
419 * Returns a ByteVector based on the CString \a s.
420 */
421 static ByteVector fromCString(const char *s, uint length = 0xffffffff);
422
423 /*!
424 * Returns a const refernence to the byte at \a index.
425 */
426 const char &operator[](int index) const;
427
428 /*!
429 * Returns a reference to the byte at \a index.
430 */
431 char &operator[](int index);
432
433 /*!
434 * Returns true if this ByteVector and \a v are equal.
435 */
436 bool operator==(const ByteVector &v) const;
437
438 /*!
439 * Returns true if this ByteVector and \a v are not equal.
440 */
441 bool operator!=(const ByteVector &v) const;
442
443 /*!
444 * Returns true if this ByteVector and the null terminated C string \a s
445 * contain the same data.
446 */
447 bool operator==(const char *s) const;
448
449 /*!
450 * Returns true if this ByteVector and the null terminated C string \a s
451 * do not contain the same data.
452 */
453 bool operator!=(const char *s) const;
454
455 /*!
456 * Returns true if this ByteVector is less than \a v. The value of the
457 * vectors is determined by evaluating the character from left to right, and
458 * in the event one vector is a superset of the other, the size is used.
459 */
460 bool operator<(const ByteVector &v) const;
461
462 /*!
463 * Returns true if this ByteVector is greater than \a v.
464 */
465 bool operator>(const ByteVector &v) const;
466
467 /*!
468 * Returns a vector that is \a v appended to this vector.
469 */
470 ByteVector operator+(const ByteVector &v) const;
471
472 /*!
473 * Copies ByteVector \a v.
474 */
475 ByteVector &operator=(const ByteVector &v);
476
477 /*!
478 * Copies ByteVector \a v.
479 */
480 ByteVector &operator=(char c);
481
482 /*!
483 * Copies ByteVector \a v.
484 */
485 ByteVector &operator=(const char *data);
486
487 /*!
488 * A static, empty ByteVector which is convenient and fast (since returning
489 * an empty or "null" value does not require instantiating a new ByteVector).
490 */
491 static ByteVector null;
492
493 /*!
494 * Returns a hex-encoded copy of the byte vector.
495 */
496 ByteVector toHex() const;
497
498 protected:
499 /*
500 * If this ByteVector is being shared via implicit sharing, do a deep copy
501 * of the data and separate from the shared members. This should be called
502 * by all non-const subclass members.
503 */
504 void detach();
505
506 private:
507 class ByteVectorPrivate;
508 ByteVectorPrivate *d;
509 };
510}
