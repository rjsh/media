61 //! A \e wide string class suitable for unicode.
62
63 /*!
64 * This is an implicitly shared \e wide string. For storage it uses
65 * TagLib::wstring, but as this is an <i>implementation detail</i> this of
66 * course could change. Strings are stored internally as UTF-16(without BOM/
67 * CPU byte order)
68 *
69 * The use of implicit sharing means that copying a string is cheap, the only
70 * \e cost comes into play when the copy is modified. Prior to that the string
71 * just has a pointer to the data of the \e parent String. This also makes
72 * this class suitable as a function return type.
73 *
74 * In addition to adding implicit sharing, this class keeps track of four
75 * possible encodings, which are the four supported by the ID3v2 standard.
76 */
77
78 class TAGLIB_EXPORT String
79 {
80 public:
81
82#ifndef DO_NOT_DOCUMENT
83 typedef std::basic_string<wchar>::iterator Iterator;
84 typedef std::basic_string<wchar>::const_iterator ConstIterator;
85#endif
86
87 /**
88 * The four types of string encodings supported by the ID3v2 specification.
89 * ID3v1 is assumed to be Latin1 and Ogg Vorbis comments use UTF8.
90 */
91 enum Type {
92 /*!
93 * IS08859-1, or <i>Latin1</i> encoding. 8 bit characters.
94 */
95 Latin1 = 0,
96 /*!
97 * UTF16 with a <i>byte order mark</i>. 16 bit characters.
98 */
99 UTF16 = 1,
100 /*!
101 * UTF16 <i>big endian</i>. 16 bit characters. This is the encoding used
102 * internally by TagLib.
103 */
104 UTF16BE = 2,
105 /*!
106 * UTF8 encoding. Characters are usually 8 bits but can be up to 32.
107 */
108 UTF8 = 3,
109 /*!
110 * UTF16 <i>little endian</i>. 16 bit characters.
111 */
112 UTF16LE = 4
113 };
114
115 /*!
116 * Constructs an empty String.
117 */
118 String();
119
120 /*!
121 * Make a shallow, implicitly shared, copy of \a s. Because this is
122 * implicitly shared, this method is lightweight and suitable for
123 * pass-by-value usage.
124 */
125 String(const String &s);
126
127 /*!
128 * Makes a deep copy of the data in \a s.
129 *
130 * \note This should only be used with the 8-bit codecs Latin1 and UTF8, when
131 * used with other codecs it will simply print a warning and exit.
132 */
133 String(const std::string &s, Type t = Latin1);
134
135 /*!
136 * Makes a deep copy of the data in \a s.
137 */
138 String(const wstring &s, Type t = WCharByteOrder);
139
140 /*!
141 * Makes a deep copy of the data in \a s.
142 */
143 String(const wchar_t *s, Type t = WCharByteOrder);
144
145 /*!
146 * Makes a deep copy of the data in \a c.
147 *
148 * \note This should only be used with the 8-bit codecs Latin1 and UTF8, when
149 * used with other codecs it will simply print a warning and exit.
150 */
151 String(char c, Type t = Latin1);
152
153 /*!
154 * Makes a deep copy of the data in \a c.
155 */
156 String(wchar_t c, Type t = Latin1);
157
158
159 /*!
160 * Makes a deep copy of the data in \a s.
161 *
162 * \note This should only be used with the 8-bit codecs Latin1 and UTF8, when
163 * used with other codecs it will simply print a warning and exit.
164 */
165 String(const char *s, Type t = Latin1);
166
167 /*!
168 * Makes a deep copy of the data in \a s.
169 *
170 * \note This should only be used with the 8-bit codecs Latin1 and UTF8, when
171 * used with other codecs it will simply print a warning and exit.
172 */
173 String(const ByteVector &v, Type t = Latin1);
174
175 /*!
176 * Destroys this String instance.
177 */
178 virtual ~String();
179
180 /*!
181 * Returns a deep copy of this String as an std::string. The returned string
182 * is encoded in UTF8 if \a unicode is true, otherwise Latin1.
183 *
184 * \see toCString()
185 */
186 std::string to8Bit(bool unicode = false) const;
187
188 /*!
189 * Returns a deep copy of this String as a wstring. The returned string is
190 * encoded in UTF-16 (without BOM/CPU byte order).
191 *
192 * \see toCWString()
193 */
194 wstring toWString() const;
195
196 /*!
197 * Creates and returns a standard C-style (null-terminated) version of this
198 * String. The returned string is encoded in UTF8 if \a unicode is true,
199 * otherwise Latin1.
200 *
201 * The returned string is still owned by this String and should not be deleted
202 * by the user.
203 *
204 * The returned pointer remains valid until this String instance is destroyed
205 * or toCString() is called again.
206 *
207 * \warning This however has the side effect that the returned string will remain
208 * in memory <b>in addition to</b> other memory that is consumed by this
209 * String instance. So, this method should not be used on large strings or
210 * where memory is critical. Consider using to8Bit() instead to avoid it.
211 *
212 * \see to8Bit()
213 */
214 const char *toCString(bool unicode = false) const;
215
216 /*!
217 * Returns a standard C-style (null-terminated) wide character version of
218 * this String. The returned string is encoded in UTF-16 (without BOM/CPU byte
219 * order).
220 *
221 * The returned string is still owned by this String and should not be deleted
222 * by the user.
223 *
224 * The returned pointer remains valid until this String instance is destroyed
225 * or any other method of this String is called.
226 *
227 * /note This returns a pointer to the String's internal data without any
228 * conversions.
229 *
230 * \see toWString()
231 */
232 const wchar_t *toCWString() const;
233
234 /*!
235 * Returns an iterator pointing to the beginning of the string.
236 */
237 Iterator begin();
238
239 /*!
240 * Returns a const iterator pointing to the beginning of the string.
241 */
242 ConstIterator begin() const;
243
244 /*!
245 * Returns an iterator pointing to the end of the string (the position
246 * after the last character).
247 */
248 Iterator end();
249
250 /*!
251 * Returns a const iterator pointing to the end of the string (the position
252 * after the last character).
253 */
254 ConstIterator end() const;
255
256 /*!
257 * Finds the first occurrence of pattern \a s in this string starting from
258 * \a offset. If the pattern is not found, -1 is returned.
259 */
260 int find(const String &s, int offset = 0) const;
261
262 /*!
263 * Finds the last occurrence of pattern \a s in this string, searched backwards,
264 * either from the end of the string or starting from \a offset. If the pattern
265 * is not found, -1 is returned.
266 */
267 int rfind(const String &s, int offset = -1) const;
268
269 /*!
270 * Splits the string on each occurrence of \a separator.
271 */
272 StringList split(const String &separator = " ") const;
273
274 /*!
275 * Returns true if the strings starts with the substring \a s.
276 */
277 bool startsWith(const String &s) const;
278
279 /*!
280 * Extract a substring from this string starting at \a position and
281 * continuing for \a n characters.
282 */
283 String substr(uint position, uint n = 0xffffffff) const;
284
285 /*!
286 * Append \a s to the current string and return a reference to the current
287 * string.
288 */
289 String &append(const String &s);
290
291 /*!
292 * Returns an upper case version of the string.
293 *
294 * \warning This only works for the characters in US-ASCII, i.e. A-Z.
295 */
296 String upper() const;
297
298 /*!
299 * Returns the size of the string.
300 */
301 uint size() const;
302
303 /*!
304 * Returns the length of the string. Equivalent to size().
305 */
306 uint length() const;
307
308 /*!
309 * Returns true if the string is empty.
310 *
311 * \see isNull()
312 */
313 bool isEmpty() const;
314
315 /*!
316 * Returns true if this string is null -- i.e. it is a copy of the
317 * String::null string.
318 *
319 * \note A string can be empty and not null.
320 * \see isEmpty()
321 */
322 bool isNull() const;
323
324 /*!
325 * Returns a ByteVector containing the string's data. If \a t is Latin1 or
326 * UTF8, this will return a vector of 8 bit characters, otherwise it will use
327 * 16 bit characters.
328 */
329 ByteVector data(Type t) const;
330
331 /*!
332 * Convert the string to an integer.
333 *
334 * Returns the integer if the conversion was successful or 0 if the
335 * string does not represent a number.
336 */
337 // BIC: merge with the method below
338 int toInt() const;
339
340 /*!
341 * Convert the string to an integer.
342 *
343 * If the conversion was successfull, it sets the value of \a *ok to
344 * true and returns the integer. Otherwise it sets \a *ok to false
345 * and the result is undefined.
346 */
347 int toInt(bool *ok) const;
348
349 /*!
350 * Returns a string with the leading and trailing whitespace stripped.
351 */
352 String stripWhiteSpace() const;
353
354 /*!
355 * Returns true if the file only uses characters required by Latin1.
356 */
357 bool isLatin1() const;
358
359 /*!
360 * Returns true if the file only uses characters required by (7-bit) ASCII.
361 */
362 bool isAscii() const;
363
364 /*!
365 * Converts the base-10 integer \a n to a string.
366 */
367 static String number(int n);
368
369 /*!
370 * Returns a reference to the character at position \a i.
371 */
372 wchar &operator[](int i);
373
374 /*!
375 * Returns a const reference to the character at position \a i.
376 */
377 const wchar &operator[](int i) const;
378
379 /*!
380 * Compares each character of the String with each character of \a s and
381 * returns true if the strings match.
382 */
383 bool operator==(const String &s) const;
384
385 /*!
386 * Compares each character of the String with each character of \a s and
387 * returns false if the strings match.
388 */
389 bool operator!=(const String &s) const;
390
391 /*!
392 * Appends \a s to the end of the String.
393 */
394 String &operator+=(const String &s);
395
396 /*!
397 * Appends \a s to the end of the String.
398 */
399 String &operator+=(const wchar_t* s);
400
401 /*!
402 * Appends \a s to the end of the String.
403 */
404 String &operator+=(const char* s);
405
406 /*!
407 * Appends \a s to the end of the String.
408 */
409 String &operator+=(wchar_t c);
410
411 /*!
412 * Appends \a c to the end of the String.
413 */
414 String &operator+=(char c);
415
416 /*!
417 * Performs a shallow, implicitly shared, copy of \a s, overwriting the
418 * String's current data.
419 */
420 String &operator=(const String &s);
421
422 /*!
423 * Performs a deep copy of the data in \a s.
424 */
425 String &operator=(const std::string &s);
426
427 /*!
428 * Performs a deep copy of the data in \a s.
429 */
430 String &operator=(const wstring &s);
431
432 /*!
433 * Performs a deep copy of the data in \a s.
434 */
435 String &operator=(const wchar_t *s);
436
437 /*!
438 * Performs a deep copy of the data in \a s.
439 */
440 String &operator=(char c);
441
442 /*!
443 * Performs a deep copy of the data in \a s.
444 */
445 String &operator=(wchar_t c);
446
447 /*!
448 * Performs a deep copy of the data in \a s.
449 */
450 String &operator=(const char *s);
451
452 /*!
453 * Performs a deep copy of the data in \a v.
454 */
455 String &operator=(const ByteVector &v);
456
457 /*!
458 * To be able to use this class in a Map, this operator needed to be
459 * implemented. Returns true if \a s is less than this string in a bytewise
460 * comparison.
461 */
462 bool operator<(const String &s) const;
463
464 /*!
465 * A null string provided for convenience.
466 */
467 static String null;
468
469 protected:
470 /*!
471 * If this String is being shared via implicit sharing, do a deep copy of the
472 * data and separate from the shared members. This should be called by all
473 * non-const subclass members.
474 */
475 void detach();
476
477 private:
478 /*!
479 * Converts a \e Latin-1 string into \e UTF-16(without BOM/CPU byte order)
480 * and copies it to the internal buffer.
481 */
482 void copyFromLatin1(const char *s, size_t length);
483
484 /*!
485 * Converts a \e UTF-8 string into \e UTF-16(without BOM/CPU byte order)
486 * and copies it to the internal buffer.
487 */
488 void copyFromUTF8(const char *s, size_t length);
489
490 /*!
491 * Converts a \e UTF-16 (with BOM), UTF-16LE or UTF16-BE string into
492 * \e UTF-16(without BOM/CPU byte order) and copies it to the internal buffer.
493 */
494 void copyFromUTF16(const wchar_t *s, size_t length, Type t);
495
496 /*!
497 * Converts a \e UTF-16 (with BOM), UTF-16LE or UTF16-BE string into
498 * \e UTF-16(without BOM/CPU byte order) and copies it to the internal buffer.
499 */
500 void copyFromUTF16(const char *s, size_t length, Type t);
501
502 /*!
503 * Indicates which byte order of UTF-16 is used to store strings internally.
504 *
505 * \note \e String::UTF16BE or \e String::UTF16LE
506 */
507 static const Type WCharByteOrder;
508
509 class StringPrivate;
510 StringPrivate *d;
511 };
512}
