44 //! An implementation of FLAC metadata
45
46 /*!
47 * This is implementation of FLAC metadata for non-Ogg FLAC files. At some
48 * point when Ogg / FLAC is more common there will be a similar implementation
49 * under the Ogg hiearchy.
50 *
51 * This supports ID3v1, ID3v2 and Xiph style comments as well as reading stream
52 * properties from the file.
53 */
54
55 namespace FLAC {
56
57 //! An implementation of TagLib::File with FLAC specific methods
58
59 /*!
60 * This implements and provides an interface for FLAC files to the
61 * TagLib::Tag and TagLib::AudioProperties interfaces by way of implementing
62 * the abstract TagLib::File API as well as providing some additional
63 * information specific to FLAC files.
64 */
65
66 class TAGLIB_EXPORT File : public TagLib::File
67 {
68 public:
69 /*!
70 * Constructs a FLAC file from \a file. If \a readProperties is true the
71 * file's audio properties will also be read.
72 *
73 * \note In the current implementation, \a propertiesStyle is ignored.
74 *
75 * \deprecated This constructor will be dropped in favor of the one below
76 * in a future version.
77 */
78 File(FileName file, bool readProperties = true,
79 Properties::ReadStyle propertiesStyle = Properties::Average);
80
81 /*!
82 * Constructs an APE file from \a file. If \a readProperties is true the
83 * file's audio properties will also be read.
84 *
85 * If this file contains and ID3v2 tag the frames will be created using
86 * \a frameFactory.
87 *
88 * \note In the current implementation, \a propertiesStyle is ignored.
89 */
90 // BIC: merge with the above constructor
91 File(FileName file, ID3v2::FrameFactory *frameFactory,
92 bool readProperties = true,
93 Properties::ReadStyle propertiesStyle = Properties::Average);
94
95 /*!
96 * Constructs a FLAC file from \a stream. If \a readProperties is true the
97 * file's audio properties will also be read.
98 *
99 * \note TagLib will *not* take ownership of the stream, the caller is
100 * responsible for deleting it after the File object.
101 *
102 * If this file contains and ID3v2 tag the frames will be created using
103 * \a frameFactory.
104 *
105 * \note In the current implementation, \a propertiesStyle is ignored.
106 */
107 // BIC: merge with the above constructor
108 File(IOStream *stream, ID3v2::FrameFactory *frameFactory,
109 bool readProperties = true,
110 Properties::ReadStyle propertiesStyle = Properties::Average);
111
112 /*!
113 * Destroys this instance of the File.
114 */
115 virtual ~File();
116
117 /*!
118 * Returns the Tag for this file. This will be a union of XiphComment,
119 * ID3v1 and ID3v2 tags.
120 *
121 * \see ID3v2Tag()
122 * \see ID3v1Tag()
123 * \see XiphComment()
124 */
125 virtual TagLib::Tag *tag() const;
126
127 /*!
128 * Implements the unified property interface -- export function.
129 * If the file contains more than one tag (e.g. XiphComment and ID3v1),
130 * only the first one (in the order XiphComment, ID3v2, ID3v1) will be
131 * converted to the PropertyMap.
132 */
133 PropertyMap properties() const;
134
135 void removeUnsupportedProperties(const StringList &);
136
137 /*!
138 * Implements the unified property interface -- import function.
139 * This always creates a Xiph comment, if none exists. The return value
140 * relates to the Xiph comment only.
141 * Ignores any changes to ID3v1 or ID3v2 comments since they are not allowed
142 * in the FLAC specification.
143 */
144 PropertyMap setProperties(const PropertyMap &);
145
146 /*!
147 * Returns the FLAC::Properties for this file. If no audio properties
148 * were read then this will return a null pointer.
149 */
150 virtual Properties *audioProperties() const;
151
152 /*!
153 * Save the file. This will primarily save the XiphComment, but
154 * will also keep any old ID3-tags up to date. If the file
155 * has no XiphComment, one will be constructed from the ID3-tags.
156 *
157 * This returns true if the save was successful.
158 */
159 virtual bool save();
160
161 /*!
162 * Returns a pointer to the ID3v2 tag of the file.
163 *
164 * If \a create is false (the default) this returns a null pointer
165 * if there is no valid ID3v2 tag. If \a create is true it will create
166 * an ID3v2 tag if one does not exist and returns a valid pointer.
167 *
168 * \note This may return a valid pointer regardless of whether or not the
169 * file on disk has an ID3v2 tag. Use hasID3v2Tag() to check if the file
170 * on disk actually has an ID3v2 tag.
171 *
172 * \note The Tag <b>is still</b> owned by the MPEG::File and should not be
173 * deleted by the user. It will be deleted when the file (object) is
174 * destroyed.
175 *
176 * \see hasID3v2Tag()
177 */
178 ID3v2::Tag *ID3v2Tag(bool create = false);
179
180 /*!
181 * Returns a pointer to the ID3v1 tag of the file.
182 *
183 * If \a create is false (the default) this returns a null pointer
184 * if there is no valid APE tag. If \a create is true it will create
185 * an APE tag if one does not exist and returns a valid pointer.
186 *
187 * \note This may return a valid pointer regardless of whether or not the
188 * file on disk has an ID3v1 tag. Use hasID3v1Tag() to check if the file
189 * on disk actually has an ID3v1 tag.
190 *
191 * \note The Tag <b>is still</b> owned by the MPEG::File and should not be
192 * deleted by the user. It will be deleted when the file (object) is
193 * destroyed.
194 *
195 * \see hasID3v1Tag()
196 */
197 ID3v1::Tag *ID3v1Tag(bool create = false);
198
199 /*!
200 * Returns a pointer to the XiphComment for the file.
201 *
202 * If \a create is false (the default) this returns a null pointer
203 * if there is no valid XiphComment. If \a create is true it will create
204 * a XiphComment if one does not exist and returns a valid pointer.
205 *
206 * \note This may return a valid pointer regardless of whether or not the
207 * file on disk has a XiphComment. Use hasXiphComment() to check if the
208 * file on disk actually has a XiphComment.
209 *
210 * \note The Tag <b>is still</b> owned by the FLAC::File and should not be
211 * deleted by the user. It will be deleted when the file (object) is
212 * destroyed.
213 *
214 * \see hasXiphComment()
215 */
216 Ogg::XiphComment *xiphComment(bool create = false);
217
218 /*!
219 * Set the ID3v2::FrameFactory to something other than the default. This
220 * can be used to specify the way that ID3v2 frames will be interpreted
221 * when
222 *
223 * \see ID3v2FrameFactory
224 */
225 void setID3v2FrameFactory(const ID3v2::FrameFactory *factory);
226
227 /*!
228 * Returns the block of data used by FLAC::Properties for parsing the
229 * stream properties.
230 *
231 * \deprecated This method will not be public in a future release.
232 */
233 ByteVector streamInfoData(); // BIC: remove
234
235 /*!
236 * Returns the length of the audio-stream, used by FLAC::Properties for
237 * calculating the bitrate.
238 *
239 * \deprecated This method will not be public in a future release.
240 */
241 long streamLength(); // BIC: remove
242
243 /*!
244 * Returns a list of pictures attached to the FLAC file.
245 */
246 List<Picture *> pictureList();
247
248 /*!
249 * Removes an attached picture. If \a del is true the picture's memory
250 * will be freed; if it is false, it must be deleted by the user.
251 */
252 void removePicture(Picture *picture, bool del = true);
253
254 /*!
255 * Remove all attached images.
256 */
257 void removePictures();
258
259 /*!
260 * Add a new picture to the file. The file takes ownership of the
261 * picture and will handle freeing its memory.
262 *
263 * \note The file will be saved only after calling save().
264 */
265 void addPicture(Picture *picture);
266
267 /*!
268 * Returns whether or not the file on disk actually has a XiphComment.
269 *
270 * \see xiphComment()
271 */
272 bool hasXiphComment() const;
273
274 /*!
275 * Returns whether or not the file on disk actually has an ID3v1 tag.
276 *
277 * \see ID3v1Tag()
278 */
279 bool hasID3v1Tag() const;
280
281 /*!
282 * Returns whether or not the file on disk actually has an ID3v2 tag.
283 *
284 * \see ID3v2Tag()
285 */
286 bool hasID3v2Tag() const;
287
288 private:
289 File(const File &);
290 File &operator=(const File &);
291
292 void read(bool readProperties, Properties::ReadStyle propertiesStyle);
293 void scan();
294 long findID3v2();
295 long findID3v1();
296 ByteVector xiphCommentData() const;
297 long findPaddingBreak(long nextPageOffset, long targetOffset, bool *isLast);
298
299 class FilePrivate;
300 FilePrivate *d;
301 };
302 }
