47 BMP = TypeBMP,
48 GIF = TypeGIF,
49 Unknown = TypeImplicit,
50 };
51

54
55 CoverArt(const CoverArt &item);

57
58 //! Format of the image
59 Format format() const;
60
61 //! The image data
62 ByteVector data() const;
63

66 CoverArtPrivate *d;
67 };

71 }
