60 //! An abstract class that provides operations on a sequence of bytes
61
62 class TAGLIB_EXPORT IOStream
63 {
64 public:
65 /*!
66 * Position in the file used for seeking.
67 */
68 enum Position {
69 //! Seek from the beginning of the file.
70 Beginning,
71 //! Seek from the current position in the file.
72 Current,
73 //! Seek from the end of the file.
74 End
75 };
76
77 IOStream();
78
79 /*!
80 * Destroys this IOStream instance.
81 */
82 virtual ~IOStream();
83
84 /*!
85 * Returns the stream name in the local file system encoding.
86 */
87 virtual FileName name() const = 0;
88
89 /*!
90 * Reads a block of size \a length at the current get pointer.
91 */
92 virtual ByteVector readBlock(ulong length) = 0;
93
94 /*!
95 * Attempts to write the block \a data at the current get pointer. If the
96 * file is currently only opened read only -- i.e. readOnly() returns true --
97 * this attempts to reopen the file in read/write mode.
98 *
99 * \note This should be used instead of using the streaming output operator
100 * for a ByteVector. And even this function is significantly slower than
101 * doing output with a char[].
102 */
103 virtual void writeBlock(const ByteVector &data) = 0;
104
105 /*!
106 * Insert \a data at position \a start in the file overwriting \a replace
107 * bytes of the original content.
108 *
109 * \note This method is slow since it requires rewriting all of the file
110 * after the insertion point.
111 */
112 virtual void insert(const ByteVector &data, ulong start = 0, ulong replace = 0) = 0;
113
114 /*!
115 * Removes a block of the file starting a \a start and continuing for
116 * \a length bytes.
117 *
118 * \note This method is slow since it involves rewriting all of the file
119 * after the removed portion.
120 */
121 virtual void removeBlock(ulong start = 0, ulong length = 0) = 0;
122
123 /*!
124 * Returns true if the file is read only (or if the file can not be opened).
125 */
126 virtual bool readOnly() const = 0;
127
128 /*!
129 * Since the file can currently only be opened as an argument to the
130 * constructor (sort-of by design), this returns if that open succeeded.
131 */
132 virtual bool isOpen() const = 0;
133
134 /*!
135 * Move the I/O pointer to \a offset in the stream from position \a p. This
136 * defaults to seeking from the beginning of the stream.
137 *
138 * \see Position
139 */
140 virtual void seek(long offset, Position p = Beginning) = 0;
141
142 /*!
143 * Reset the end-of-stream and error flags on the stream.
144 */
145 virtual void clear();
146
147 /*!
148 * Returns the current offset within the stream.
149 */
150 virtual long tell() const = 0;
151
152 /*!
153 * Returns the length of the stream.
154 */
155 virtual long length() = 0;
156
157 /*!
158 * Truncates the stream to a \a length.
159 */
160 virtual void truncate(long length) = 0;
161
162 private:
163 IOStream(const IOStream &);
164 IOStream &operator=(const IOStream &);
165 };
