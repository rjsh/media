50 typedef void (responseHandler)(RTSPClient* rtspClient,
51 int resultCode, char* resultString);
52 // A function that is called in response to a RTSP command. The parameters are as follows:
53 // "rtspClient": The "RTSPClient" object on which the original command was issued.
54 // "resultCode": If zero, then the command completed successfully. If non-zero, then the command did not complete
55 // successfully, and "resultCode" indicates the error, as follows:
56 // A positive "resultCode" is a RTSP error code (for example, 404 means "not found")
57 // A negative "resultCode" indicates a socket/network error; 0-"resultCode" is the standard "errno" code.
58 // "resultString": A ('\0'-terminated) string returned along with the response, or else NULL.
59 // In particular:
60 // "resultString" for a successful "DESCRIBE" command will be the media session's SDP description.
61 // "resultString" for a successful "OPTIONS" command will be a list of allowed commands.
62 // Note that this string can be present (i.e., not NULL) even if "resultCode" is non-zero - i.e., an error message.
63 // Also, "resultString" can be NULL, even if "resultCode" is zero (e.g., if the RTSP command succeeded, but without
64 // including an appropriate result header).
65 // Note also that this string is dynamically allocated, and must be freed by the handler (or the caller)
66 // - using "delete[]".
67
68 unsigned sendDescribeCommand(responseHandler* responseHandler, Authenticator* authenticator = NULL);
69 // Issues a RTSP "DESCRIBE" command, then returns the "CSeq" sequence number that was used in the command.
70 // The (programmer-supplied) "responseHandler" function is called later to handle the response
71 // (or is called immediately - with an error code - if the command cannot be sent).
72 // "authenticator" (optional) is used for access control. If you have username and password strings, you can use this by
73 // passing an actual parameter that you created by creating an "Authenticator(username, password) object".
74 // (Note that if you supply a non-NULL "authenticator" parameter, you need do this only for the first command you send.)
75
76 unsigned sendOptionsCommand(responseHandler* responseHandler, Authenticator* authenticator = NULL);
77 // Issues a RTSP "OPTIONS" command, then returns the "CSeq" sequence number that was used in the command.
78 // (The "responseHandler" and "authenticator" parameters are as described for "sendDescribeCommand".)
79

96 // Issues an aggregate RTSP "PLAY" command on "session", then returns the "CSeq" sequence number that was used in the command.

100 double start = 0.0f, double end = -1.0f, float scale = 1.0f,
101 Authenticator* authenticator = NULL);
102 // Issues a RTSP "PLAY" command on "subsession", then returns the "CSeq" sequence number that was used in the command.
103 // (Note: start=-1 means 'resume'; end=-1 means 'play to end')
104 // (The "responseHandler" and "authenticator" parameters are as described for "sendDescribeCommand".)
105
106 // Alternative forms of "sendPlayCommand()", used to send "PLAY" commands that include an 'absolute' time range:

148 void sendDummyUDPPackets(MediaSession& session, unsigned numDummyPackets = 2);
149 void sendDummyUDPPackets(MediaSubsession& subsession, unsigned numDummyPackets = 2);
150 // Sends short 'dummy' (i.e., non-RTP or RTCP) UDP packets towards the server, to increase
151 // the likelihood of RTP/RTCP packets from the server reaching us if we're behind a NAT.
152 // (If we requested RTP-over-TCP streaming, then these functions have no effect.)
153 // Our implementation automatically does this just prior to sending each "PLAY" command;
154 // You should not call these functions yourself unless you know what you're doing.
155
156 Boolean changeResponseHandler(unsigned cseq, responseHandler* newResponseHandler);
157 // Changes the response handler for the previously-performed command (whose operation returned "cseq").
158 // (To turn off any response handling for the command, use a "newResponseHandler" value of NULL. This might be done as part
159 // of an implementation of a 'timeout handler' on the command, for example.)
160 // This function returns True iff "cseq" was for a valid previously-performed command (whose response is still unhandled).

333// A simple server that creates a new "RTSPClient" object whenever a "REGISTER" request arrives (specifying the "rtsp://" URL
334// of a stream). The new "RTSPClient" object will be created with the specified URL, and passed to the provided handler function.
335
336typedef void onRTSPClientCreationFunc(RTSPClient* newRTSPClient, Boolean requestStreamingOverTCP);
337
338class HandlerServerForREGISTERCommand: public RTSPServer {
339public:
340 static HandlerServerForREGISTERCommand* createNew(UsageEnvironment& env, onRTSPClientCreationFunc* creationFunc,
341 Port ourPort = 0, UserAuthenticationDatabase* authDatabase = NULL,
342 int verbosityLevel = 0, char const* applicationName = NULL);
343 // If ourPort.num() == 0, we'll choose the port number ourself. (Use the following function to get it.)
344 portNumBits serverPortNum() const { return ntohs(fRTSPServerPort.num()); }
345
346protected:
347 HandlerServerForREGISTERCommand(UsageEnvironment& env, onRTSPClientCreationFunc* creationFunc, int ourSocket, Port ourPort,
348 UserAuthenticationDatabase* authDatabase, int verbosityLevel, char const* applicationName);
349 // called only by createNew();
350 virtual ~HandlerServerForREGISTERCommand();
351
352 virtual RTSPClient* createNewRTSPClient(char const* rtspURL, int verbosityLevel, char const* applicationName,
353 int socketNumToServer);
354 // This function - by default - creates a (base) "RTSPClient" object. If you want to create a subclass
355 // of "RTSPClient" instead, then subclass this class, and redefine this virtual function.
356
357protected: // redefined virtual functions
358 virtual char const* allowedCommandNames(); // we support "OPTIONS" and "REGISTER" only
359 virtual Boolean weImplementREGISTER(char const* proxyURLSuffix, char*& responseStr); // redefined to return True
360 virtual void implementCmd_REGISTER(char const* url, char const* urlSuffix, int socketToRemoteServer,
361 Boolean deliverViaTCP, char const* proxyURLSuffix);
362
363private:
364 onRTSPClientCreationFunc* fCreationFunc;
365 int fVerbosityLevel;
366 char* fApplicationName;
367};
