18// A class for generating MPEG-2 Transport Stream from one or more input
19// Elementary Stream data sources
20// C++ header
21
22#ifndef _MPEG2_TRANSPORT_STREAM_MULTIPLEXOR_HH
23#define _MPEG2_TRANSPORT_STREAM_MULTIPLEXOR_HH
24
25#ifndef _FRAMED_SOURCE_HH
26#include "FramedSource.hh"
27#endif
28#ifndef _MPEG_1OR2_DEMUX_HH
29#include "MPEG1or2Demux.hh" // for SCR
30#endif
31
32#define PID_TABLE_SIZE 256
33
34class MPEG2TransportStreamMultiplexor: public FramedSource {
35protected:
36 MPEG2TransportStreamMultiplexor(UsageEnvironment& env);
37 virtual ~MPEG2TransportStreamMultiplexor();
38
39 virtual void awaitNewBuffer(unsigned char* oldBuffer) = 0;
40 // implemented by subclasses
41
42 void handleNewBuffer(unsigned char* buffer, unsigned bufferSize,
43 int mpegVersion, MPEG1or2Demux::SCR scr);
44 // called by "awaitNewBuffer()"
45 // Note: For MPEG-4 video, set "mpegVersion" to 4; for H.264 video, set "mpegVersion" to 5.
46
47private:
48 // Redefined virtual functions:
49 virtual void doGetNextFrame();
50
51private:
52 void deliverDataToClient(u_int8_t pid, unsigned char* buffer, unsigned bufferSize,
53 unsigned& startPositionInBuffer);
54
55 void deliverPATPacket();
56 void deliverPMTPacket(Boolean hasChanged);
57
58 void setProgramStreamMap(unsigned frameSize);
59
60protected:
61 Boolean fHaveVideoStreams;
62
63private:
64 unsigned fOutgoingPacketCounter;
65 unsigned fProgramMapVersion;
66 u_int8_t fPreviousInputProgramMapVersion, fCurrentInputProgramMapVersion;
67 // These two fields are used if we see "program_stream_map"s in the input.
68 struct {
69 unsigned counter;
70 u_int8_t streamType; // for use in Program Maps
71 } fPIDState[PID_TABLE_SIZE];
72 u_int8_t fPCR_PID, fCurrentPID;
73 // Note: We map 8-bit stream_ids directly to PIDs
74 MPEG1or2Demux::SCR fPCR;
75 unsigned char* fInputBuffer;
76 unsigned fInputBufferSize, fInputBufferBytesUsed;
77 Boolean fIsFirstAdaptationField;
78};
