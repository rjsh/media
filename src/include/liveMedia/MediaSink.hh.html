28class MediaSink: public Medium {
29public:
30 static Boolean lookupByName(UsageEnvironment& env, char const* sinkName,
31 MediaSink*& resultSink);
32
33 typedef void (afterPlayingFunc)(void* clientData);
34 Boolean startPlaying(MediaSource& source,
35 afterPlayingFunc* afterFunc,
36 void* afterClientData);
37 virtual void stopPlaying();
38
39 // Test for specific types of sink:
40 virtual Boolean isRTPSink() const;
41
42 FramedSource* source() const {return fSource;}
43
44protected:
45 MediaSink(UsageEnvironment& env); // abstract base class
46 virtual ~MediaSink();
47
48 virtual Boolean sourceIsCompatibleWithUs(MediaSource& source);
49 // called by startPlaying()
50 virtual Boolean continuePlaying() = 0;
51 // called by startPlaying()
52
53 static void onSourceClosure(void* clientData); // can be used in "getNextFrame()" calls
54 void onSourceClosure();
55 // should be called (on ourselves) by continuePlaying() when it
56 // discovers that the source we're playing from has closed.
57
58 FramedSource* fSource;
59
60private:
61 // redefined virtual functions:
62 virtual Boolean isSink() const;
63
64private:
65 // The following fields are used when we're being played:
66 afterPlayingFunc* fAfterFunc;
67 void* fAfterClientData;
68};

70// A data structure that a sink may use for an output packet:
71class OutPacketBuffer {
72public:
73 OutPacketBuffer(unsigned preferredPacketSize, unsigned maxPacketSize);
74 ~OutPacketBuffer();
75
76 static unsigned maxSize;
77
78 unsigned char* curPtr() const {return &fBuf[fPacketStart + fCurOffset];}
79 unsigned totalBytesAvailable() const {
80 return fLimit - (fPacketStart + fCurOffset);
81 }
82 unsigned totalBufferSize() const { return fLimit; }
83 unsigned char* packet() const {return &fBuf[fPacketStart];}
84 unsigned curPacketSize() const {return fCurOffset;}
85
86 void increment(unsigned numBytes) {fCurOffset += numBytes;}
87
88 void enqueue(unsigned char const* from, unsigned numBytes);
89 void enqueueWord(u_int32_t word);
90 void insert(unsigned char const* from, unsigned numBytes, unsigned toPosition);
91 void insertWord(u_int32_t word, unsigned toPosition);
92 void extract(unsigned char* to, unsigned numBytes, unsigned fromPosition);
93 u_int32_t extractWord(unsigned fromPosition);
94
95 void skipBytes(unsigned numBytes);
96
97 Boolean isPreferredSize() const {return fCurOffset >= fPreferred;}
98 Boolean wouldOverflow(unsigned numBytes) const {
99 return (fCurOffset+numBytes) > fMax;
100 }
101 unsigned numOverflowBytes(unsigned numBytes) const {
102 return (fCurOffset+numBytes) - fMax;
103 }
104 Boolean isTooBigForAPacket(unsigned numBytes) const {
105 return numBytes > fMax;
106 }
107
108 void setOverflowData(unsigned overflowDataOffset,
109 unsigned overflowDataSize,
110 struct timeval const& presentationTime,
111 unsigned durationInMicroseconds);
112 unsigned overflowDataSize() const {return fOverflowDataSize;}
113 struct timeval overflowPresentationTime() const {return fOverflowPresentationTime;}
114 unsigned overflowDurationInMicroseconds() const {return fOverflowDurationInMicroseconds;}
115 Boolean haveOverflowData() const {return fOverflowDataSize > 0;}
116 void useOverflowData();
117
118 void adjustPacketStart(unsigned numBytes);
119 void resetPacketStart();
120 void resetOffset() { fCurOffset = 0; }
121 void resetOverflowData() { fOverflowDataOffset = fOverflowDataSize = 0; }
122
123private:
124 unsigned fPacketStart, fCurOffset, fPreferred, fMax, fLimit;
125 unsigned char* fBuf;
126
127 unsigned fOverflowDataOffset, fOverflowDataSize;
128 struct timeval fOverflowPresentationTime;
129 unsigned fOverflowDurationInMicroseconds;
130};
