34class MatroskaFile: public Medium {
35public:
36 typedef void (onCreationFunc)(MatroskaFile* newFile, void* clientData);
37 static void createNew(UsageEnvironment& env, char const* fileName, onCreationFunc* onCreation, void* onCreationClientData,
38 char const* preferredLanguage = "eng");
39 // Note: Unlike most "createNew()" functions, this one doesn't return a new object immediately. Instead, because this class
40 // requires file reading (to parse the Matroska 'Track' headers) before a new object can be initialized, the creation of a new
41 // object is signalled by calling - from the event loop - an 'onCreationFunc' that is passed as a parameter to "createNew()".
42
43 MatroskaTrack* lookup(unsigned trackNumber) const;
44
45 // Create a demultiplexor for extracting tracks from this file. (Separate clients will typically have separate demultiplexors.)
46 MatroskaDemux* newDemux();
47
48 // Parameters of the file ('Segment'); set when the file is parsed:
49 unsigned timecodeScale() { return fTimecodeScale; } // in nanoseconds
50 float segmentDuration() { return fSegmentDuration; } // in units of "timecodeScale()"
51 float fileDuration(); // in seconds
52
53 char const* fileName() const { return fFileName; }
54
55 unsigned chosenVideoTrackNumber() { return fChosenVideoTrackNumber; }
56 unsigned chosenAudioTrackNumber() { return fChosenAudioTrackNumber; }
57 unsigned chosenSubtitleTrackNumber() { return fChosenSubtitleTrackNumber; }
58
59private:
60 MatroskaFile(UsageEnvironment& env, char const* fileName, onCreationFunc* onCreation, void* onCreationClientData,
61 char const* preferredLanguage);
62 // called only by createNew()
63 virtual ~MatroskaFile();
64
65 static void handleEndOfTrackHeaderParsing(void* clientData);
66 void handleEndOfTrackHeaderParsing();
67
68 void addTrack(MatroskaTrack* newTrack, unsigned trackNumber);
69 void addCuePoint(double cueTime, u_int64_t clusterOffsetInFile, unsigned blockNumWithinCluster);
70 Boolean lookupCuePoint(double& cueTime, u_int64_t& resultClusterOffsetInFile, unsigned& resultBlockNumWithinCluster);
71 void printCuePoints(FILE* fid);
72
73 void removeDemux(MatroskaDemux* demux);
74
75private:
76 friend class MatroskaFileParser;
77 friend class MatroskaDemux;
78 char const* fFileName;
79 onCreationFunc* fOnCreation;
80 void* fOnCreationClientData;
81 char const* fPreferredLanguage;
82
83 unsigned fTimecodeScale; // in nanoseconds
84 float fSegmentDuration; // in units of "fTimecodeScale"
85 u_int64_t fSegmentDataOffset, fClusterOffset, fCuesOffset;
86
87 class MatroskaTrackTable* fTrackTable;
88 HashTable* fDemuxesTable;
89 class CuePoint* fCuePoints;
90 unsigned fChosenVideoTrackNumber, fChosenAudioTrackNumber, fChosenSubtitleTrackNumber;
91 class MatroskaFileParser* fParserForInitialization;
92};

94// We define our own track type codes as bits (powers of 2), so we can use the set of track types as a bitmap, representing a set:
95// (Note that MATROSKA_TRACK_TYPE_OTHER must be last, and have the largest value.)
96#define MATROSKA_TRACK_TYPE_VIDEO 0x01
97#define MATROSKA_TRACK_TYPE_AUDIO 0x02
98#define MATROSKA_TRACK_TYPE_SUBTITLE 0x04
99#define MATROSKA_TRACK_TYPE_OTHER 0x08
100
101class MatroskaTrack {
102public:
103 MatroskaTrack();
104 virtual ~MatroskaTrack();
105
106 // track parameters
107 unsigned trackNumber;
108 u_int8_t trackType;
109 Boolean isEnabled, isDefault, isForced;
110 unsigned defaultDuration;
111 char* name;
112 char* language;
113 char* codecID;
114 unsigned samplingFrequency;
115 unsigned numChannels;
116 char const* mimeType;
117 unsigned codecPrivateSize;
118 u_int8_t* codecPrivate;
119 Boolean codecPrivateUsesH264FormatForH265; // a hack specifically for H.265 video tracks
120 unsigned headerStrippedBytesSize;
121 u_int8_t* headerStrippedBytes;
122 unsigned subframeSizeSize; // 0 means: frames do not have subframes (the default behavior)
123 Boolean haveSubframes() const { return subframeSizeSize > 0; }
124};

126class MatroskaDemux: public Medium {
127public:
128 FramedSource* newDemuxedTrack();
129 FramedSource* newDemuxedTrack(unsigned& resultTrackNumber);
130 // Returns a new stream ("FramedSource" subclass) that represents the next preferred media
131 // track (video, audio, subtitle - in that order) from the file. (Preferred media tracks
132 // are based on the file's language preference.)
133 // This function returns NULL when no more media tracks exist.
134
135 FramedSource* newDemuxedTrackByTrackNumber(unsigned trackNumber);
136 // As above, but creates a new stream for a specific track number within the Matroska file.
137 // (You should not call this function more than once with the same track number.)
138
139 // Note: We assume that:
140 // - Every track created by "newDemuxedTrack()" is later read
141 // - All calls to "newDemuxedTrack()" are made before any track is read
142
143protected:
144 friend class MatroskaFile;
145 friend class MatroskaFileParser;
146 class MatroskaDemuxedTrack* lookupDemuxedTrack(unsigned trackNumber);
147
148 MatroskaDemux(MatroskaFile& ourFile); // we're created only by a "MatroskaFile" (a friend)
149 virtual ~MatroskaDemux();
150
151private:
152 friend class MatroskaDemuxedTrack;
153 void removeTrack(unsigned trackNumber);
154 void continueReading(); // called by a demuxed track to tell us that it has a pending read ("doGetNextFrame()")
155 void seekToTime(double& seekNPT);
156
157 static void handleEndOfFile(void* clientData);
158 void handleEndOfFile();
159
160private:
161 MatroskaFile& fOurFile;
162 class MatroskaFileParser* fOurParser;
163 HashTable* fDemuxedTracksTable;
164
165 // Used to implement "newServerMediaSubsession()":
166 u_int8_t fNextTrackTypeToCheck;
167};
