33class RTPSink: public MediaSink {
34public:
35 static Boolean lookupByName(UsageEnvironment& env, char const* sinkName,
36 RTPSink*& resultSink);
37
38 // used by RTSP servers:
39 Groupsock const& groupsockBeingUsed() const { return *(fRTPInterface.gs()); }
40 Groupsock& groupsockBeingUsed() { return *(fRTPInterface.gs()); }
41
42 unsigned char rtpPayloadType() const { return fRTPPayloadType; }
43 unsigned rtpTimestampFrequency() const { return fTimestampFrequency; }
44 void setRTPTimestampFrequency(unsigned freq) {
45 fTimestampFrequency = freq;
46 }
47 char const* rtpPayloadFormatName() const {return fRTPPayloadFormatName;}
48
49 unsigned numChannels() const { return fNumChannels; }
50
51 virtual char const* sdpMediaType() const; // for use in SDP m= lines
52 virtual char* rtpmapLine() const; // returns a string to be delete[]d
53 virtual char const* auxSDPLine();
54 // optional SDP line (e.g. a=fmtp:...)
55
56 u_int16_t currentSeqNo() const { return fSeqNo; }
57 u_int32_t presetNextTimestamp();
58 // ensures that the next timestamp to be used will correspond to
59 // the current 'wall clock' time.
60
61 RTPTransmissionStatsDB& transmissionStatsDB() const {
62 return *fTransmissionStatsDB;
63 }
64
65 Boolean nextTimestampHasBeenPreset() const { return fNextTimestampHasBeenPreset; }
66 Boolean& enableRTCPReports() { return fEnableRTCPReports; }
67
68 void getTotalBitrate(unsigned& outNumBytes, double& outElapsedTime);
69 // returns the number of bytes sent since the last time that we
70 // were called, and resets the counter.
71
72 struct timeval const& creationTime() const { return fCreationTime; }
73 struct timeval const& initialPresentationTime() const { return fInitialPresentationTime; }
74 struct timeval const& mostRecentPresentationTime() const { return fMostRecentPresentationTime; }
75 void resetPresentationTimes();
76
77 // Hacks to allow sending RTP over TCP (RFC 2236, section 10.12):
78 void setStreamSocket(int sockNum, unsigned char streamChannelId) {
79 fRTPInterface.setStreamSocket(sockNum, streamChannelId);
80 }
81 void addStreamSocket(int sockNum, unsigned char streamChannelId) {
82 fRTPInterface.addStreamSocket(sockNum, streamChannelId);
83 }
84 void removeStreamSocket(int sockNum, unsigned char streamChannelId) {
85 fRTPInterface.removeStreamSocket(sockNum, streamChannelId);
86 }
87
88protected:
89 RTPSink(UsageEnvironment& env,
90 Groupsock* rtpGS, unsigned char rtpPayloadType,
91 u_int32_t rtpTimestampFrequency,
92 char const* rtpPayloadFormatName,
93 unsigned numChannels);
94 // abstract base class
95
96 virtual ~RTPSink();
97
98 // used by RTCP:
99 friend class RTCPInstance;
100 friend class RTPTransmissionStats;
101 u_int32_t SSRC() const {return fSSRC;}
102 // later need a means of changing the SSRC if there's a collision #####
103 u_int32_t convertToRTPTimestamp(struct timeval tv);
104 unsigned packetCount() const {return fPacketCount;}
105 unsigned octetCount() const {return fOctetCount;}
106
107protected:
108 RTPInterface fRTPInterface;
109 unsigned char fRTPPayloadType;
110 unsigned fPacketCount, fOctetCount, fTotalOctetCount /*incl RTP hdr*/;
111 struct timeval fTotalOctetCountStartTime, fInitialPresentationTime, fMostRecentPresentationTime;
112 u_int32_t fCurrentTimestamp;
113 u_int16_t fSeqNo;
114
115private:
116 // redefined virtual functions:
117 virtual Boolean isRTPSink() const;
118
119private:
120 u_int32_t fSSRC, fTimestampBase;
121 unsigned fTimestampFrequency;
122 Boolean fNextTimestampHasBeenPreset;
123 Boolean fEnableRTCPReports; // whether RTCP "SR" reports should be sent for this sink (default: True)
124 char const* fRTPPayloadFormatName;
125 unsigned fNumChannels;
126 struct timeval fCreationTime;
127
128 RTPTransmissionStatsDB* fTransmissionStatsDB;
129};

134class RTPTransmissionStatsDB {
135public:
136 unsigned numReceivers() const { return fNumReceivers; }
137
138 class Iterator {
139 public:
140 Iterator(RTPTransmissionStatsDB& receptionStatsDB);
141 virtual ~Iterator();
142
143 RTPTransmissionStats* next();
144 // NULL if none
145
146 private:
147 HashTable::Iterator* fIter;
148 };
149
150 // The following is called whenever a RTCP RR packet is received:
151 void noteIncomingRR(u_int32_t SSRC, struct sockaddr_in const& lastFromAddress,
152 unsigned lossStats, unsigned lastPacketNumReceived,
153 unsigned jitter, unsigned lastSRTime, unsigned diffSR_RRTime);
154
155 // The following is called when a RTCP BYE packet is received:
156 void removeRecord(u_int32_t SSRC);
157
158 RTPTransmissionStats* lookup(u_int32_t SSRC) const;
159
160private: // constructor and destructor, called only by RTPSink:
161 friend class RTPSink;
162 RTPTransmissionStatsDB(RTPSink& rtpSink);
163 virtual ~RTPTransmissionStatsDB();
164
165private:
166 void add(u_int32_t SSRC, RTPTransmissionStats* stats);
167
168private:
169 friend class Iterator;
170 unsigned fNumReceivers;
171 RTPSink& fOurRTPSink;
172 HashTable* fTable;
173};

175class RTPTransmissionStats {
176public:
177 u_int32_t SSRC() const {return fSSRC;}
178 struct sockaddr_in const& lastFromAddress() const {return fLastFromAddress;}
179 unsigned lastPacketNumReceived() const {return fLastPacketNumReceived;}
180 unsigned firstPacketNumReported() const {return fFirstPacketNumReported;}
181 unsigned totNumPacketsLost() const {return fTotNumPacketsLost;}
182 unsigned jitter() const {return fJitter;}
183 unsigned lastSRTime() const { return fLastSRTime; }
184 unsigned diffSR_RRTime() const { return fDiffSR_RRTime; }
185 unsigned roundTripDelay() const;
186 // The round-trip delay (in units of 1/65536 seconds) computed from
187 // the most recently-received RTCP RR packet.
188 struct timeval timeCreated() const {return fTimeCreated;}
189 struct timeval lastTimeReceived() const {return fTimeReceived;}
190 void getTotalOctetCount(u_int32_t& hi, u_int32_t& lo);
191 void getTotalPacketCount(u_int32_t& hi, u_int32_t& lo);
192
193 // Information which requires at least two RRs to have been received:
194 unsigned packetsReceivedSinceLastRR() const;
195 u_int8_t packetLossRatio() const { return fPacketLossRatio; }
196 // as an 8-bit fixed-point number
197 int packetsLostBetweenRR() const;
198
199private:
200 // called only by RTPTransmissionStatsDB:
201 friend class RTPTransmissionStatsDB;
202 RTPTransmissionStats(RTPSink& rtpSink, u_int32_t SSRC);
203 virtual ~RTPTransmissionStats();
204
205 void noteIncomingRR(struct sockaddr_in const& lastFromAddress,
206 unsigned lossStats, unsigned lastPacketNumReceived,
207 unsigned jitter,
208 unsigned lastSRTime, unsigned diffSR_RRTime);
209
210private:
211 RTPSink& fOurRTPSink;
212 u_int32_t fSSRC;
213 struct sockaddr_in fLastFromAddress;
214 unsigned fLastPacketNumReceived;
215 u_int8_t fPacketLossRatio;
216 unsigned fTotNumPacketsLost;
217 unsigned fJitter;
218 unsigned fLastSRTime;
219 unsigned fDiffSR_RRTime;
220 struct timeval fTimeCreated, fTimeReceived;
221 Boolean fAtLeastTwoRRsHaveBeenReceived;
222 unsigned fOldLastPacketNumReceived;
223 unsigned fOldTotNumPacketsLost;
224 Boolean fFirstPacket;
225 unsigned fFirstPacketNumReported;
226 u_int32_t fLastOctetCount, fTotalOctetCount_hi, fTotalOctetCount_lo;
227 u_int32_t fLastPacketCount, fTotalPacketCount_hi, fTotalPacketCount_lo;
228};
