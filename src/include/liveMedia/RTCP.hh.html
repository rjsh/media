44class RTCPInstance: public Medium {
45public:
46 static RTCPInstance* createNew(UsageEnvironment& env, Groupsock* RTCPgs,
47 unsigned totSessionBW, /* in kbps */
48 unsigned char const* cname,
49 RTPSink* sink,
50 RTPSource const* source,
51 Boolean isSSMSource = False);
52
53 static Boolean lookupByName(UsageEnvironment& env, char const* instanceName,
54 RTCPInstance*& resultInstance);
55
56 unsigned numMembers() const;
57 unsigned totSessionBW() const { return fTotSessionBW; }
58
59 void setByeHandler(TaskFunc* handlerTask, void* clientData,
60 Boolean handleActiveParticipantsOnly = True);
61 // Assigns a handler routine to be called if a "BYE" arrives.
62 // The handler is called once only; for subsequent "BYE"s,
63 // "setByeHandler()" would need to be called again.
64 // If "handleActiveParticipantsOnly" is True, then the handler is called
65 // only if the SSRC is for a known sender (if we have a "RTPSource"),
66 // or if the SSRC is for a known receiver (if we have a "RTPSink").
67 // This prevents (for example) the handler for a multicast receiver being
68 // called if some other multicast receiver happens to exit.
69 // If "handleActiveParticipantsOnly" is False, then the handler is called
70 // for any incoming RTCP "BYE".
71 // (To remove an existing "BYE" handler, call "setByeHandler()" again, with a "handlerTask" of NULL.)
72 void setSRHandler(TaskFunc* handlerTask, void* clientData);
73 void setRRHandler(TaskFunc* handlerTask, void* clientData);
74 // Assigns a handler routine to be called if a "SR" or "RR"
75 // (respectively) arrives. Unlike "setByeHandler()", the handler will
76 // be called once for each incoming "SR" or "RR". (To turn off handling,
77 // call the function again with "handlerTask" (and "clientData") as NULL.
78 void setSpecificRRHandler(netAddressBits fromAddress, Port fromPort,
79 TaskFunc* handlerTask, void* clientData);
80 // Like "setRRHandler()", but applies only to "RR" packets that come from
81 // a specific source address and port. (Note that if both a specific
82 // and a general "RR" handler function is set, then both will be called.)
83 void unsetSpecificRRHandler(netAddressBits fromAddress, Port fromPort); // equivalent to setSpecificRRHandler(..., NULL, NULL);
84
85 Groupsock* RTCPgs() const { return fRTCPInterface.gs(); }
86
87 void setStreamSocket(int sockNum, unsigned char streamChannelId);
88 void addStreamSocket(int sockNum, unsigned char streamChannelId);
89 void removeStreamSocket(int sockNum, unsigned char streamChannelId) {
90 fRTCPInterface.removeStreamSocket(sockNum, streamChannelId);
91 }
92 // hacks to allow sending RTP over TCP (RFC 2236, section 10.12)
93
94 void setAuxilliaryReadHandler(AuxHandlerFunc* handlerFunc,
95 void* handlerClientData) {
96 fRTCPInterface.setAuxilliaryReadHandler(handlerFunc,
97 handlerClientData);
98 }
99
100protected:
101 RTCPInstance(UsageEnvironment& env, Groupsock* RTPgs, unsigned totSessionBW,
102 unsigned char const* cname,
103 RTPSink* sink, RTPSource const* source,
104 Boolean isSSMSource);
105 // called only by createNew()
106 virtual ~RTCPInstance();
107
108private:
109 // redefined virtual functions:
110 virtual Boolean isRTCPInstance() const;
111
112private:
113 Boolean addReport(Boolean alwaysAdd = False);
114 void addSR();
115 void addRR();
116 void enqueueCommonReportPrefix(unsigned char packetType, u_int32_t SSRC,
117 unsigned numExtraWords = 0);
118 void enqueueCommonReportSuffix();
119 void enqueueReportBlock(RTPReceptionStats* receptionStats);
120 void addSDES();
121 void addBYE();
122
123 void sendBuiltPacket();
124
125 static void onExpire(RTCPInstance* instance);
126 void onExpire1();
127
128 static void incomingReportHandler(RTCPInstance* instance, int /*mask*/);
129 void incomingReportHandler1();
130 void onReceive(int typeOfPacket, int totPacketSize, u_int32_t ssrc);
131
132private:
133 unsigned char* fInBuf;
134 unsigned fNumBytesAlreadyRead;
135 OutPacketBuffer* fOutBuf;
136 RTPInterface fRTCPInterface;
137 unsigned fTotSessionBW;
138 RTPSink* fSink;
139 RTPSource const* fSource;
140 Boolean fIsSSMSource;
141
142 SDESItem fCNAME;
143 RTCPMemberDatabase* fKnownMembers;
144 unsigned fOutgoingReportCount; // used for SSRC member aging
145
146 double fAveRTCPSize;
147 int fIsInitial;
148 double fPrevReportTime;
149 double fNextReportTime;
150 int fPrevNumMembers;
151
152 int fLastSentSize;
153 int fLastReceivedSize;
154 u_int32_t fLastReceivedSSRC;
155 int fTypeOfEvent;
156 int fTypeOfPacket;
157 Boolean fHaveJustSentPacket;
158 unsigned fLastPacketSentSize;
159
160 TaskFunc* fByeHandlerTask;
161 void* fByeHandlerClientData;
162 Boolean fByeHandleActiveParticipantsOnly;
163 TaskFunc* fSRHandlerTask;
164 void* fSRHandlerClientData;
165 TaskFunc* fRRHandlerTask;
166 void* fRRHandlerClientData;
167 AddressPortLookupTable* fSpecificRRHandlerTable;
168
169public: // because this stuff is used by an external "C" function
170 void schedule(double nextTime);
171 void reschedule(double nextTime);
172 void sendReport();
173 void sendBYE();
174 int typeOfEvent() {return fTypeOfEvent;}
175 int sentPacketSize() {return fLastSentSize;}
176 int packetType() {return fTypeOfPacket;}
177 int receivedPacketSize() {return fLastReceivedSize;}
178 int checkNewSSRC();
179 void removeLastReceivedSSRC();
180 void removeSSRC(u_int32_t ssrc, Boolean alsoRemoveStats);
181};
