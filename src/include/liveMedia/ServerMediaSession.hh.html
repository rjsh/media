18// A data structure that represents a session that consists of
19// potentially multiple (audio and/or video) sub-sessions
20// (This data structure is used for media *streamers* - i.e., servers.
21// For media receivers, use "MediaSession" instead.)
22// C++ header

42class ServerMediaSession: public Medium {
43public:
44 static ServerMediaSession* createNew(UsageEnvironment& env,
45 char const* streamName = NULL,
46 char const* info = NULL,
47 char const* description = NULL,
48 Boolean isSSM = False,
49 char const* miscSDPLines = NULL);
50
51 static Boolean lookupByName(UsageEnvironment& env,
52 char const* mediumName,
53 ServerMediaSession*& resultSession);
54
55 char* generateSDPDescription(); // based on the entire session
56 // Note: The caller is responsible for freeing the returned string
57
58 char const* streamName() const { return fStreamName; }
59
60 Boolean addSubsession(ServerMediaSubsession* subsession);
61 unsigned numSubsessions() const { return fSubsessionCounter; }
62
63 void testScaleFactor(float& scale); // sets "scale" to the actual supported scale
64 float duration() const;
65 // a result == 0 means an unbounded session (the default)
66 // a result < 0 means: subsession durations differ; the result is -(the largest).
67 // a result > 0 means: this is the duration of a bounded session
68
69 unsigned referenceCount() const { return fReferenceCount; }
70 void incrementReferenceCount() { ++fReferenceCount; }
71 void decrementReferenceCount() { if (fReferenceCount > 0) --fReferenceCount; }
72 Boolean& deleteWhenUnreferenced() { return fDeleteWhenUnreferenced; }
73
74 void deleteAllSubsessions();
75 // Removes and deletes all subsessions added by "addSubsession()", returning us to an 'empty' state
76 // Note: If you have already added this "ServerMediaSession" to a "RTSPServer" then, before calling this function,
77 // you must first close any client connections that use it,
78 // by calling "RTSPServer::closeAllClientSessionsForServerMediaSession()".
79
80protected:
81 ServerMediaSession(UsageEnvironment& env, char const* streamName,
82 char const* info, char const* description,
83 Boolean isSSM, char const* miscSDPLines);
84 // called only by "createNew()"
85
86 virtual ~ServerMediaSession();
87
88private: // redefined virtual functions
89 virtual Boolean isServerMediaSession() const;
90
91private:
92 Boolean fIsSSM;
93
94 // Linkage fields:
95 friend class ServerMediaSubsessionIterator;
96 ServerMediaSubsession* fSubsessionsHead;
97 ServerMediaSubsession* fSubsessionsTail;
98 unsigned fSubsessionCounter;
99
100 char* fStreamName;
101 char* fInfoSDPString;
102 char* fDescriptionSDPString;
103 char* fMiscSDPLines;
104 struct timeval fCreationTime;
105 unsigned fReferenceCount;
106 Boolean fDeleteWhenUnreferenced;
107};

124class ServerMediaSubsession: public Medium {
125public:
126 unsigned trackNumber() const { return fTrackNumber; }
127 char const* trackId();
128 virtual char const* sdpLines() = 0;
129 virtual void getStreamParameters(unsigned clientSessionId, // in
130 netAddressBits clientAddress, // in
131 Port const& clientRTPPort, // in
132 Port const& clientRTCPPort, // in
133 int tcpSocketNum, // in (-1 means use UDP, not TCP)
134 unsigned char rtpChannelId, // in (used if TCP)
135 unsigned char rtcpChannelId, // in (used if TCP)
136 netAddressBits& destinationAddress, // in out
137 u_int8_t& destinationTTL, // in out
138 Boolean& isMulticast, // out
139 Port& serverRTPPort, // out
140 Port& serverRTCPPort, // out
141 void*& streamToken // out
142 ) = 0;
143 virtual void startStream(unsigned clientSessionId, void* streamToken,
144 TaskFunc* rtcpRRHandler,
145 void* rtcpRRHandlerClientData,
146 unsigned short& rtpSeqNum,
147 unsigned& rtpTimestamp,
148 ServerRequestAlternativeByteHandler* serverRequestAlternativeByteHandler,
149 void* serverRequestAlternativeByteHandlerClientData) = 0;
150 virtual void pauseStream(unsigned clientSessionId, void* streamToken);
151 virtual void seekStream(unsigned clientSessionId, void* streamToken, double& seekNPT, double streamDuration, u_int64_t& numBytes);
152 // This routine is used to seek by relative (i.e., NPT) time.
153 // "streamDuration", if >0.0, specifies how much data to stream, past "seekNPT". (If <=0.0, all remaining data is streamed.)
154 // "numBytes" returns the size (in bytes) of the data to be streamed, or 0 if unknown or unlimited.
155 virtual void seekStream(unsigned clientSessionId, void* streamToken, char*& absStart, char*& absEnd);
156 // This routine is used to seek by 'absolute' time.
157 // "absStart" should be a string of the form "YYYYMMDDTHHMMSSZ" or "YYYYMMDDTHHMMSS.<frac>Z".
158 // "absEnd" should be either NULL (for no end time), or a string of the same form as "absStart".
159 // These strings may be modified in-place, or can be reassigned to a newly-allocated value (after delete[]ing the original).
160 virtual void nullSeekStream(unsigned clientSessionId, void* streamToken);
161 // Called whenever we're handling a "PLAY" command without a specified start time.
162 virtual void setStreamScale(unsigned clientSessionId, void* streamToken, float scale);
163 virtual float getCurrentNPT(void* streamToken);
164 virtual FramedSource* getStreamSource(void* streamToken);
165 virtual void deleteStream(unsigned clientSessionId, void*& streamToken);
166
167 virtual void testScaleFactor(float& scale); // sets "scale" to the actual supported scale
168 virtual float duration() const;
169 // returns 0 for an unbounded session (the default)
170 // returns > 0 for a bounded session
171 virtual void getAbsoluteTimeRange(char*& absStartTime, char*& absEndTime) const;
172 // Subclasses can reimplement this iff they support seeking by 'absolute' time.
173
174 // The following may be called by (e.g.) SIP servers, for which the
175 // address and port number fields in SDP descriptions need to be non-zero:
176 void setServerAddressAndPortForSDP(netAddressBits addressBits,
177 portNumBits portBits);
178
179protected: // we're a virtual base class
180 ServerMediaSubsession(UsageEnvironment& env);
181 virtual ~ServerMediaSubsession();
182
183 char const* rangeSDPLine() const;
184 // returns a string to be delete[]d
185
186 ServerMediaSession* fParentSession;
187 netAddressBits fServerAddressForSDP;
188 portNumBits fPortNumForSDP;
189
190private:
191 friend class ServerMediaSession;
192 friend class ServerMediaSubsessionIterator;
193 ServerMediaSubsession* fNext;
194
195 unsigned fTrackNumber; // within an enclosing ServerMediaSession
196 char const* fTrackId;
197};
