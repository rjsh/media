36class UserAuthenticationDatabase {
37public:
38 UserAuthenticationDatabase(char const* realm = NULL,
39 Boolean passwordsAreMD5 = False);
40 // If "passwordsAreMD5" is True, then each password stored into, or removed from,
41 // the database is actually the value computed
42 // by md5(<username>:<realm>:<actual-password>)
43 virtual ~UserAuthenticationDatabase();
44
45 virtual void addUserRecord(char const* username, char const* password);
46 virtual void removeUserRecord(char const* username);
47
48 virtual char const* lookupPassword(char const* username);
49 // returns NULL if the user name was not present
50
51 char const* realm() { return fRealm; }
52 Boolean passwordsAreMD5() { return fPasswordsAreMD5; }
53
54protected:
55 HashTable* fTable;
56 char* fRealm;

58};

105 typedef void (responseHandlerForREGISTER)(RTSPServer* rtspServer, unsigned requestId, int resultCode, char* resultString);
106 unsigned registerStream(ServerMediaSession* serverMediaSession,
107 char const* remoteClientNameOrAddress, portNumBits remoteClientPortNum,
108 responseHandlerForREGISTER* responseHandler,
109 char const* username = NULL, char const* password = NULL,
110 Boolean receiveOurStreamViaTCP = False,
111 char const* proxyURLSuffix = NULL);
112 // 'Register' the stream represented by "serverMediaSession" with the given remote client (specifed by name and port number).
113 // This is done using our custom "REGISTER" RTSP command.
114 // The function returns a unique number that can be used to identify the request; this number is also passed to "responseHandler".
115 // When a response is received from the remote client (or the "REGISTER" request fails), the specified response handler
116 // (if non-NULL) is called. (Note that the "resultString" passed to the handler was dynamically allocated,
117 // and should be delete[]d by the handler after use.)
118 // If "receiveOurStreamViaTCP" is True, then we're requesting that the remote client access our stream using RTP/RTCP-over-TCP.
119 // (Otherwise, the remote client may choose regular RTP/RTCP-over-UDP streaming.)
120 // "proxyURLSuffix" (optional) is used only when the remote client is also a proxy server.
121 // It tells the proxy server the suffix that it should use in its "rtsp://" URL (when front-end clients access the stream)
122
123 char* rtspURL(ServerMediaSession const* serverMediaSession, int clientSocket = -1) const;
124 // returns a "rtsp://" URL that could be used to access the
125 // specified session (which must already have been added to
126 // us using "addServerMediaSession()".
127 // This string is dynamically allocated; caller should delete[]
128 // (If "clientSocket" is non-negative, then it is used (by calling "getsockname()") to determine
129 // the IP address to be used in the URL.)
130 char* rtspURLPrefix(int clientSocket = -1) const;
131 // like "rtspURL()", except that it returns just the common prefix used by
132 // each session's "rtsp://" URL.
133 // This string is dynamically allocated; caller should delete[]
