38class OnDemandServerMediaSubsession: public ServerMediaSubsession {
39protected: // we're a virtual base class
40 OnDemandServerMediaSubsession(UsageEnvironment& env, Boolean reuseFirstSource,
41 portNumBits initialPortNum = 6970);
42 virtual ~OnDemandServerMediaSubsession();
43
44protected: // redefined virtual functions
45 virtual char const* sdpLines();
46 virtual void getStreamParameters(unsigned clientSessionId,
47 netAddressBits clientAddress,
48 Port const& clientRTPPort,
49 Port const& clientRTCPPort,
50 int tcpSocketNum,
51 unsigned char rtpChannelId,
52 unsigned char rtcpChannelId,
53 netAddressBits& destinationAddress,
54 u_int8_t& destinationTTL,
55 Boolean& isMulticast,
56 Port& serverRTPPort,
57 Port& serverRTCPPort,
58 void*& streamToken);
59 virtual void startStream(unsigned clientSessionId, void* streamToken,
60 TaskFunc* rtcpRRHandler,
61 void* rtcpRRHandlerClientData,
62 unsigned short& rtpSeqNum,
63 unsigned& rtpTimestamp,
64 ServerRequestAlternativeByteHandler* serverRequestAlternativeByteHandler,
65 void* serverRequestAlternativeByteHandlerClientData);
66 virtual void pauseStream(unsigned clientSessionId, void* streamToken);
67 virtual void seekStream(unsigned clientSessionId, void* streamToken, double& seekNPT, double streamDuration, u_int64_t& numBytes);
68 virtual void seekStream(unsigned clientSessionId, void* streamToken, char*& absStart, char*& absEnd);
69 virtual void nullSeekStream(unsigned clientSessionId, void* streamToken);
70 virtual void setStreamScale(unsigned clientSessionId, void* streamToken, float scale);
71 virtual float getCurrentNPT(void* streamToken);
72 virtual FramedSource* getStreamSource(void* streamToken);
73 virtual void deleteStream(unsigned clientSessionId, void*& streamToken);
74
75protected: // new virtual functions, possibly redefined by subclasses
76 virtual char const* getAuxSDPLine(RTPSink* rtpSink,
77 FramedSource* inputSource);
78 virtual void seekStreamSource(FramedSource* inputSource, double& seekNPT, double streamDuration, u_int64_t& numBytes);
79 // This routine is used to seek by relative (i.e., NPT) time.
80 // "streamDuration", if >0.0, specifies how much data to stream, past "seekNPT". (If <=0.0, all remaining data is streamed.)
81 // "numBytes" returns the size (in bytes) of the data to be streamed, or 0 if unknown or unlimited.
82 virtual void seekStreamSource(FramedSource* inputSource, char*& absStart, char*& absEnd);
83 // This routine is used to seek by 'absolute' time.
84 // "absStart" should be a string of the form "YYYYMMDDTHHMMSSZ" or "YYYYMMDDTHHMMSS.<frac>Z".
85 // "absEnd" should be either NULL (for no end time), or a string of the same form as "absStart".
86 // These strings may be modified in-place, or can be reassigned to a newly-allocated value (after delete[]ing the original).
87 virtual void setStreamSourceScale(FramedSource* inputSource, float scale);
88 virtual void closeStreamSource(FramedSource* inputSource);
89
90protected: // new virtual functions, defined by all subclasses
91 virtual FramedSource* createNewStreamSource(unsigned clientSessionId,
92 unsigned& estBitrate) = 0;
93 // "estBitrate" is the stream's estimated bitrate, in kbps
94 virtual RTPSink* createNewRTPSink(Groupsock* rtpGroupsock,
95 unsigned char rtpPayloadTypeIfDynamic,
96 FramedSource* inputSource) = 0;
97
98private:
99 void setSDPLinesFromRTPSink(RTPSink* rtpSink, FramedSource* inputSource,
100 unsigned estBitrate);
101 // used to implement "sdpLines()"
102
103protected:
104 char* fSDPLines;
105 HashTable* fDestinationsHashTable; // indexed by client session id
106
107private:
108 Boolean fReuseFirstSource;
109 portNumBits fInitialPortNum;
110 void* fLastStreamToken;
111 char fCNAME[100]; // for RTCP
112 friend class StreamState;
113};

116// A class that represents the state of an ongoing stream. This is used only internally, in the implementation of
117// "OnDemandServerMediaSubsession", but we expose the definition here, in case subclasses of "OnDemandServerMediaSubsession"
118// want to access it.
119
120class Destinations {
121public:
122 Destinations(struct in_addr const& destAddr,
123 Port const& rtpDestPort,
124 Port const& rtcpDestPort)
125 : isTCP(False), addr(destAddr), rtpPort(rtpDestPort), rtcpPort(rtcpDestPort) {
126 }
127 Destinations(int tcpSockNum, unsigned char rtpChanId, unsigned char rtcpChanId)
128 : isTCP(True), rtpPort(0) /*dummy*/, rtcpPort(0) /*dummy*/,
129 tcpSocketNum(tcpSockNum), rtpChannelId(rtpChanId), rtcpChannelId(rtcpChanId) {
130 }
131
132public:
133 Boolean isTCP;
134 struct in_addr addr;
135 Port rtpPort;
136 Port rtcpPort;
137 int tcpSocketNum;
138 unsigned char rtpChannelId, rtcpChannelId;
139};
