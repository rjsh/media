18// A special object which, when created, sends a custom RTSP "REGISTER" command to a specified client.
19// C++ header
20
21#ifndef _RTSP_REGISTER_SENDER_HH
22#define _RTSP_REGISTER_SENDER_HH
23
24#ifndef _RTSP_CLIENT_HH
25#include "RTSPClient.hh"
26#endif
27
28class RTSPRegisterSender: public RTSPClient {
29public:
30 static RTSPRegisterSender*
31 createNew(UsageEnvironment& env,
32 char const* remoteClientNameOrAddress, portNumBits remoteClientPortNum, char const* rtspURLToRegister,
33 RTSPClient::responseHandler* rtspResponseHandler, Authenticator* authenticator = NULL,
34 Boolean requestStreamingViaTCP = False, char const* proxyURLSuffix = NULL, Boolean reuseConnection = False,
35 int verbosityLevel = 0, char const* applicationName = NULL);
36
37 void grabConnection(int& sock, struct sockaddr_in& remoteAddress); // so that the socket doesn't get closed when we're deleted
38
39protected:
40 RTSPRegisterSender(UsageEnvironment& env,
41 char const* remoteClientNameOrAddress, portNumBits remoteClientPortNum, char const* rtspURLToRegister,
42 RTSPClient::responseHandler* rtspResponseHandler, Authenticator* authenticator,
43 Boolean requestStreamingViaTCP, char const* proxyURLSuffix, Boolean reuseConnection,
44 int verbosityLevel, char const* applicationName);
45 // called only by "createNew()"
46 virtual ~RTSPRegisterSender();
47
48 // Redefined virtual functions:
49 virtual Boolean setRequestFields(RequestRecord* request,
50 char*& cmdURL, Boolean& cmdURLWasAllocated,
51 char const*& protocolStr,
52 char*& extraHeaders, Boolean& extraHeadersWereAllocated);
53
54public: // Some compilers complain if this is "protected:"
55 // A subclass of "RTSPClient::RequestRecord", specific to our "REGISTER" command:
56 class RequestRecord_REGISTER: public RTSPClient::RequestRecord {
57 public:
58 RequestRecord_REGISTER(unsigned cseq, RTSPClient::responseHandler* rtspResponseHandler, char const* rtspURLToRegister,
59 Boolean reuseConnection, Boolean requestStreamingViaTCP, char const* proxyURLSuffix);
60 virtual ~RequestRecord_REGISTER();
61
62 char const* rtspURLToRegister() const { return fRTSPURLToRegister; }
63 Boolean reuseConnection() const { return fReuseConnection; }
64 Boolean requestStreamingViaTCP() const { return fRequestStreamingViaTCP; }
65 char const* proxyURLSuffix() const { return fProxyURLSuffix; }
66
67 private:
68 char* fRTSPURLToRegister;
69 Boolean fReuseConnection, fRequestStreamingViaTCP;
70 char* fProxyURLSuffix;
71 };
72
73private:
74 portNumBits fRemoteClientPortNum;
75};
