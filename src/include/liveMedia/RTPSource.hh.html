33class RTPSource: public FramedSource {
34public:
35 static Boolean lookupByName(UsageEnvironment& env, char const* sourceName,
36 RTPSource*& resultSource);
37
38 Boolean curPacketMarkerBit() const { return fCurPacketMarkerBit; }
39
40 unsigned char rtpPayloadFormat() const { return fRTPPayloadFormat; }
41
42 virtual Boolean hasBeenSynchronizedUsingRTCP();
43
44 Groupsock* RTPgs() const { return fRTPInterface.gs(); }
45
46 virtual void setPacketReorderingThresholdTime(unsigned uSeconds) = 0;
47
48 // used by RTCP:
49 u_int32_t SSRC() const { return fSSRC; }
50 // Note: This is *our* SSRC, not the SSRC in incoming RTP packets.
51 // later need a means of changing the SSRC if there's a collision #####
52
53 unsigned timestampFrequency() const {return fTimestampFrequency;}
54
55 RTPReceptionStatsDB& receptionStatsDB() const {
56 return *fReceptionStatsDB;
57 }
58
59 u_int32_t lastReceivedSSRC() const { return fLastReceivedSSRC; }
60 // Note: This is the SSRC in the most recently received RTP packet; not *our* SSRC
61
62 Boolean& enableRTCPReports() { return fEnableRTCPReports; }
63 Boolean const& enableRTCPReports() const { return fEnableRTCPReports; }
64
65 void setStreamSocket(int sockNum, unsigned char streamChannelId) {
66 // hack to allow sending RTP over TCP (RFC 2236, section 10.12)
67 fRTPInterface.setStreamSocket(sockNum, streamChannelId);
68 }
69
70 void setAuxilliaryReadHandler(AuxHandlerFunc* handlerFunc,
71 void* handlerClientData) {
72 fRTPInterface.setAuxilliaryReadHandler(handlerFunc,
73 handlerClientData);
74 }
75
76 // Note that RTP receivers will usually not need to call either of the following two functions, because
77 // RTP sequence numbers and timestamps are usually not useful to receivers.
78 // (Our implementation of RTP reception already does all needed handling of RTP sequence numbers and timestamps.)
79 u_int16_t curPacketRTPSeqNum() const { return fCurPacketRTPSeqNum; }
80private: friend class MediaSubsession; // "MediaSubsession" is the only outside class that ever needs to see RTP timestamps!
81 u_int32_t curPacketRTPTimestamp() const { return fCurPacketRTPTimestamp; }
82
83protected:
84 RTPSource(UsageEnvironment& env, Groupsock* RTPgs,
85 unsigned char rtpPayloadFormat, u_int32_t rtpTimestampFrequency);
86 // abstract base class
87 virtual ~RTPSource();
88
89protected:
90 RTPInterface fRTPInterface;
91 u_int16_t fCurPacketRTPSeqNum;
92 u_int32_t fCurPacketRTPTimestamp;
93 Boolean fCurPacketMarkerBit;
94 Boolean fCurPacketHasBeenSynchronizedUsingRTCP;
95 u_int32_t fLastReceivedSSRC;
96
97private:
98 // redefined virtual functions:
99 virtual Boolean isRTPSource() const;
100 virtual void getAttributes() const;
101
102private:
103 unsigned char fRTPPayloadFormat;
104 unsigned fTimestampFrequency;
105 u_int32_t fSSRC;
106 Boolean fEnableRTCPReports; // whether RTCP "RR" reports should be sent for this source (default: True)
107
108 RTPReceptionStatsDB* fReceptionStatsDB;
109};

114class RTPReceptionStatsDB {

116 unsigned totNumPacketsReceived() const { return fTotNumPacketsReceived; }

125 class Iterator {
126 public:
127 Iterator(RTPReceptionStatsDB& receptionStatsDB);
128 virtual ~Iterator();
129
130 RTPReceptionStats* next(Boolean includeInactiveSources = False);
131 // NULL if none
132
133 private:
134 HashTable::Iterator* fIter;
135 };
136
137 // The following is called whenever a RTP packet is received:
138 void noteIncomingPacket(u_int32_t SSRC, u_int16_t seqNum,
139 u_int32_t rtpTimestamp,
140 unsigned timestampFrequency,
141 Boolean useForJitterCalculation,
142 struct timeval& resultPresentationTime,
143 Boolean& resultHasBeenSyncedUsingRTCP,
144 unsigned packetSize /* payload only */);
145
146 // The following is called whenever a RTCP SR packet is received:
147 void noteIncomingSR(u_int32_t SSRC,
148 u_int32_t ntpTimestampMSW, u_int32_t ntpTimestampLSW,
149 u_int32_t rtpTimestamp);
150
151 // The following is called when a RTCP BYE packet is received:
152 void removeRecord(u_int32_t SSRC);
153
154 RTPReceptionStats* lookup(u_int32_t SSRC) const;
155
156protected: // constructor and destructor, called only by RTPSource:
157 friend class RTPSource;
158 RTPReceptionStatsDB();
159 virtual ~RTPReceptionStatsDB();
160
161protected:
162 void add(u_int32_t SSRC, RTPReceptionStats* stats);
163
164protected:
165 friend class Iterator;
166 unsigned fNumActiveSourcesSinceLastReset;
167
168private:
169 HashTable* fTable;
170 unsigned fTotNumPacketsReceived; // for all SSRCs
171};

173class RTPReceptionStats {
174public:
175 u_int32_t SSRC() const { return fSSRC; }
176 unsigned numPacketsReceivedSinceLastReset() const {
177 return fNumPacketsReceivedSinceLastReset;
178 }
179 unsigned totNumPacketsReceived() const { return fTotNumPacketsReceived; }
180 double totNumKBytesReceived() const;
181
182 unsigned totNumPacketsExpected() const {
183 return (fHighestExtSeqNumReceived - fBaseExtSeqNumReceived) + 1;
184 }
185
186 unsigned baseExtSeqNumReceived() const { return fBaseExtSeqNumReceived; }
187 unsigned lastResetExtSeqNumReceived() const {
188 return fLastResetExtSeqNumReceived;
189 }
190 unsigned highestExtSeqNumReceived() const {
191 return fHighestExtSeqNumReceived;
192 }
193
194 unsigned jitter() const;
195
196 unsigned lastReceivedSR_NTPmsw() const { return fLastReceivedSR_NTPmsw; }
197 unsigned lastReceivedSR_NTPlsw() const { return fLastReceivedSR_NTPlsw; }
198 struct timeval const& lastReceivedSR_time() const {
199 return fLastReceivedSR_time;
200 }
201
202 unsigned minInterPacketGapUS() const { return fMinInterPacketGapUS; }
203 unsigned maxInterPacketGapUS() const { return fMaxInterPacketGapUS; }
204 struct timeval const& totalInterPacketGaps() const {
205 return fTotalInterPacketGaps;
206 }
207
208protected:
209 // called only by RTPReceptionStatsDB:
210 friend class RTPReceptionStatsDB;
211 RTPReceptionStats(u_int32_t SSRC, u_int16_t initialSeqNum);
212 RTPReceptionStats(u_int32_t SSRC);
213 virtual ~RTPReceptionStats();
214
215private:
216 void noteIncomingPacket(u_int16_t seqNum, u_int32_t rtpTimestamp,
217 unsigned timestampFrequency,
218 Boolean useForJitterCalculation,
219 struct timeval& resultPresentationTime,
220 Boolean& resultHasBeenSyncedUsingRTCP,
221 unsigned packetSize /* payload only */);
222 void noteIncomingSR(u_int32_t ntpTimestampMSW, u_int32_t ntpTimestampLSW,
223 u_int32_t rtpTimestamp);
224 void init(u_int32_t SSRC);
225 void initSeqNum(u_int16_t initialSeqNum);
226 void reset();
227 // resets periodic stats (called each time they're used to
228 // generate a reception report)
229
230protected:
231 u_int32_t fSSRC;
232 unsigned fNumPacketsReceivedSinceLastReset;
233 unsigned fTotNumPacketsReceived;
234 u_int32_t fTotBytesReceived_hi, fTotBytesReceived_lo;
235 Boolean fHaveSeenInitialSequenceNumber;
236 unsigned fBaseExtSeqNumReceived;
237 unsigned fLastResetExtSeqNumReceived;
238 unsigned fHighestExtSeqNumReceived;
239 int fLastTransit; // used in the jitter calculation
240 u_int32_t fPreviousPacketRTPTimestamp;
241 double fJitter;
242 // The following are recorded whenever we receive a RTCP SR for this SSRC:
243 unsigned fLastReceivedSR_NTPmsw; // NTP timestamp (from SR), most-signif
244 unsigned fLastReceivedSR_NTPlsw; // NTP timestamp (from SR), least-signif
245 struct timeval fLastReceivedSR_time;
246 struct timeval fLastPacketReceptionTime;
247 unsigned fMinInterPacketGapUS, fMaxInterPacketGapUS;
248 struct timeval fTotalInterPacketGaps;
249
250private:
251 // Used to convert from RTP timestamp to 'wall clock' time:
252 Boolean fHasBeenSynchronized;
253 u_int32_t fSyncTimestamp;
254 struct timeval fSyncTime;
255};
