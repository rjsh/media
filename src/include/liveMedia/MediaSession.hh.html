18// A data structure that represents a session that consists of
19// potentially multiple (audio and/or video) sub-sessions
20// (This data structure is used for media *receivers* - i.e., clients.
21// For media streamers, use "ServerMediaSession" instead.)
22// C++ header
23
24/* NOTE: To support receiving your own custom RTP payload format, you must first define a new subclass of "MultiFramedRTPSource"
25 (or "BasicUDPSource") that implements it. Then define your own subclass of "MediaSession" and "MediaSubsession", as follows:
26 - In your subclass of "MediaSession" (named, for example, "myMediaSession"):
27 - Define and implement your own static member function
28 static myMediaSession* createNew(UsageEnvironment& env, char const* sdpDescription);
29 and call this - instead of "MediaSession::createNew()" - in your application, when you create a new "MediaSession" object.
30 - Reimplement the "createNewMediaSubsession()" virtual function, as follows:
31 MediaSubsession* myMediaSession::createNewMediaSubsession() { return new myMediaSubsession(*this); }
32 - In your subclass of "MediaSubsession" (named, for example, "myMediaSubsession"):
33 - Reimplement the "createSourceObjects()" virtual function, perhaps similar to this:
34 Boolean myMediaSubsession::createSourceObjects(int useSpecialRTPoffset) {
35 if (strcmp(fCodecName, "X-MY-RTP-PAYLOAD-FORMAT") == 0) {
36 // This subsession uses our custom RTP payload format:
37 fReadSource = fRTPSource = myRTPPayloadFormatRTPSource::createNew( <parameters> );
38 return True;
39 } else {
40 // This subsession uses some other RTP payload format - perhaps one that we already implement:
41 return ::createSourceObjects(useSpecialRTPoffset);
42 }
43 }
44*/

266 double getNormalPlayTime(struct timeval const& presentationTime);
267 // Computes the stream's "Normal Play Time" (NPT) from the given "presentationTime".
268 // (For the definition of "Normal Play Time", see RFC 2326, section 3.6.)
269 // This function is useful only if the "rtpInfo" structure was previously filled in
270 // (e.g., by a "RTP-Info:" header in a RTSP response).
271 // Also, for this function to work properly, the RTP stream's presentation times must (eventually) be
272 // synchronized via RTCP.
273 // (Note: If this function returns a negative number, then the result should be ignored by the caller.)
