35// A subclass of "RTSPClient", used to refer to the particular "ProxyServerMediaSession" object being used.
36// It is used only within the implementation of "ProxyServerMediaSession", but is defined here, in case developers wish to
37// subclass it.
38
39class ProxyRTSPClient: public RTSPClient {
40public:
41 ProxyRTSPClient(class ProxyServerMediaSession& ourServerMediaSession, char const* rtspURL,
42 char const* username, char const* password,
43 portNumBits tunnelOverHTTPPortNum, int verbosityLevel, int socketNumToServer);
44 virtual ~ProxyRTSPClient();
45
46 void continueAfterDESCRIBE(char const* sdpDescription);
47 void continueAfterLivenessCommand(int resultCode, Boolean serverSupportsGetParameter);
48 void continueAfterSETUP();
49
50private:
51 void reset();
52
53 Authenticator* auth() { return fOurAuthenticator; }
54
55 void scheduleLivenessCommand();
56 static void sendLivenessCommand(void* clientData);
57
58 void scheduleDESCRIBECommand();
59 static void sendDESCRIBE(void* clientData);
60
61 static void subsessionTimeout(void* clientData);
62 void handleSubsessionTimeout();
63
64private:
65 friend class ProxyServerMediaSession;
66 friend class ProxyServerMediaSubsession;
67 ProxyServerMediaSession& fOurServerMediaSession;
68 char* fOurURL;
69 Authenticator* fOurAuthenticator;
70 Boolean fStreamRTPOverTCP;
71 class ProxyServerMediaSubsession *fSetupQueueHead, *fSetupQueueTail;
72 unsigned fNumSetupsDone;
73 unsigned fNextDESCRIBEDelay; // in seconds
74 Boolean fServerSupportsGetParameter, fLastCommandWasPLAY;
75 TaskToken fLivenessCommandTask, fDESCRIBECommandTask, fSubsessionTimerTask;
76};

92class ProxyServerMediaSession: public ServerMediaSession {
93public:
94 static ProxyServerMediaSession* createNew(UsageEnvironment& env,
95 RTSPServer* ourRTSPServer, // Note: We can be used by just one "RTSPServer"
96 char const* inputStreamURL, // the "rtsp://" URL of the stream we'll be proxying
97 char const* streamName = NULL,
98 char const* username = NULL, char const* password = NULL,
99 portNumBits tunnelOverHTTPPortNum = 0,
100 // for streaming the *proxied* (i.e., back-end) stream
101 int verbosityLevel = 0,
102 int socketNumToServer = -1);
103 // Hack: "tunnelOverHTTPPortNum" == 0xFFFF (i.e., all-ones) means: Stream RTP/RTCP-over-TCP, but *not* using HTTP
104 // "verbosityLevel" == 1 means display basic proxy setup info; "verbosityLevel" == 2 means display RTSP client protocol also.
105 // If "socketNumToServer" is >= 0, then it is the socket number of an already-existing TCP connection to the server.
106 // (In this case, "inputStreamURL" must point to the socket's endpoint, so that it can be accessed via the socket.)
107
108 virtual ~ProxyServerMediaSession();
109
110 char const* url() const;
111
112 char describeCompletedFlag;
113 // initialized to 0; set to 1 when the back-end "DESCRIBE" completes.
114 // (This can be used as a 'watch variable' in "doEventLoop()".)
115 Boolean describeCompletedSuccessfully() const { return fClientMediaSession != NULL; }
116 // This can be used - along with "describeCompletdFlag" - to check whether the back-end "DESCRIBE" completed *successfully*.
117
118protected:
119 ProxyServerMediaSession(UsageEnvironment& env, RTSPServer* ourRTSPServer,
120 char const* inputStreamURL, char const* streamName,
121 char const* username, char const* password,
122 portNumBits tunnelOverHTTPPortNum, int verbosityLevel,
123 int socketNumToServer,
124 createNewProxyRTSPClientFunc* ourCreateNewProxyRTSPClientFunc
125 = defaultCreateNewProxyRTSPClientFunc);
126
127 // If you subclass "ProxyRTSPClient", then you will also need to define your own function
128 // - with signature "createNewProxyRTSPClientFunc" (see above) - that creates a new object
129 // of this subclass. You should also subclass "ProxyServerMediaSession" and, in your
130 // subclass's constructor, initialize the parent class (i.e., "ProxyServerMediaSession")
131 // constructor by passing your new function as the "ourCreateNewProxyRTSPClientFunc"
132 // parameter.
133
134protected:
135 RTSPServer* fOurRTSPServer;
136 ProxyRTSPClient* fProxyRTSPClient;
137 MediaSession* fClientMediaSession;
138
139private:
140 friend class ProxyRTSPClient;
141 friend class ProxyServerMediaSubsession;
142 void continueAfterDESCRIBE(char const* sdpDescription);
143 void resetDESCRIBEState(); // undoes what was done by "contineAfterDESCRIBE()"
144
145private:
146 int fVerbosityLevel;
147 class PresentationTimeSessionNormalizer* fPresentationTimeSessionNormalizer;
148 createNewProxyRTSPClientFunc* fCreateNewProxyRTSPClientFunc;
149};

152////////// PresentationTimeSessionNormalizer and PresentationTimeSubsessionNormalizer definitions //////////

154// The following two classes are used by proxies to convert incoming streams' presentation times into wall-clock-aligned
155// presentation times that are suitable for our "RTPSink"s (for the corresponding outgoing streams).
156// (For multi-subsession (i.e., audio+video) sessions, the outgoing streams' presentation times retain the same relative
157// separation as those of the incoming streams.)
158
159class PresentationTimeSubsessionNormalizer: public FramedFilter {
160public:
161 void setRTPSink(RTPSink* rtpSink) { fRTPSink = rtpSink; }
162
163private:
164 friend class PresentationTimeSessionNormalizer;
165 PresentationTimeSubsessionNormalizer(PresentationTimeSessionNormalizer& parent, FramedSource* inputSource, RTPSource* rtpSource,
166 char const* codecName, PresentationTimeSubsessionNormalizer* next);
167 // called only from within "PresentationTimeSessionNormalizer"
168 virtual ~PresentationTimeSubsessionNormalizer();
169
170 static void afterGettingFrame(void* clientData, unsigned frameSize,
171 unsigned numTruncatedBytes,
172 struct timeval presentationTime,
173 unsigned durationInMicroseconds);
174 void afterGettingFrame(unsigned frameSize,
175 unsigned numTruncatedBytes,
176 struct timeval presentationTime,
177 unsigned durationInMicroseconds);
178
179private: // redefined virtual functions:
180 virtual void doGetNextFrame();
181
182private:
183 PresentationTimeSessionNormalizer& fParent;
184 RTPSource* fRTPSource;
185 RTPSink* fRTPSink;
186 char const* fCodecName;
187 PresentationTimeSubsessionNormalizer* fNext;
188};
