115/** defines a type of curve */
116typedef enum kate_curve_type {
117 kate_curve_none, /**< no curve */
118 kate_curve_static, /**< a single point */
119 kate_curve_linear, /**< linear interpolation of line segments */
120 kate_curve_catmull_rom_spline, /**< Catmull-Rom spline, goes through each point (even the first and last) */
121 kate_curve_bezier_cubic_spline, /**< Bezier cubic spline, goes through the first and last points, but not others */
122 kate_curve_bspline /**< Cubic uniform B-spline with 3-multiplicity end knots (goes through each point) */
123} kate_curve_type;
