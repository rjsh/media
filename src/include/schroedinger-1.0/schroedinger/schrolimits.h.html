9/**
10 * Section:schrolimit:
11 *
12 * This section contains a number of arbitrary limits that are
13 * compiled into Schroedinger. These can be changed as necessary
14 * if they are unsuitable.
15 */
16/**
17 * SCHRO_LIMIT_TRANSFORM_DEPTH:
18 *
19 * The maximum transform depth that the decoder can handle.
20 */
21#define SCHRO_LIMIT_TRANSFORM_DEPTH 6
22
23/**
24 * SCHRO_LIMIT_ENCODER_TRANSFORM_DEPTH:
25 *
26 * The maximum transform depth that the encoder can handle. Must be
27 * less than SCHRO_LIMIT_TRANSFORM_DEPTH.
28 */
29#define SCHRO_LIMIT_ENCODER_TRANSFORM_DEPTH 6
30
31/**
32 * SCHRO_LIMIT_SUBBANDS:
33 *
34 * The maximum number of subbands.
35 */
36#define SCHRO_LIMIT_SUBBANDS (1+3*SCHRO_LIMIT_TRANSFORM_DEPTH)
37
38/**
39 * SCHRO_LIMIT_REFERENCE_FRAMES:
40 *
41 * The maximum number of active reference frames. In the encoder,
42 * the number of active reference frames may be much larger than in
43 * the resulting stream.
44 */
45#define SCHRO_LIMIT_REFERENCE_FRAMES 8
46
47/**
48 * SCHRO_LIMIT_FRAME_QUEUE_LENGTH:
49 *
50 * The maximum number of pictures being processed by the encoder.
51 */
52#define SCHRO_LIMIT_FRAME_QUEUE_LENGTH 40
53
54/**
55 * SCHRO_LIMIT_WIDTH:
56 *
57 * Maximum picture width.
58 */
59#define SCHRO_LIMIT_WIDTH 2048
60
61/**
62 * SCHRO_LIMIT_BLOCK_SIZE
63 *
64 * Maximum block size. Both length and separation must be less than
65 * or equal to this limit.
66 */
67#define SCHRO_LIMIT_BLOCK_SIZE 64
68
69SCHRO_END_DECLS
