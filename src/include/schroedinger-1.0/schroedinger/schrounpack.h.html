9typedef struct _SchroUnpack SchroUnpack;

11struct _SchroUnpack {
12 /* pointer to data that haven't been shifted into the shift register */
13 uint8_t *data;
14
15 /* number of bits remaining that haven't been shifted into the shift
16 * register */
17 int n_bits_left;
18
19 /* number of bits read by reader */
20 int n_bits_read;
21
22 /* chunk of bits we're currently frobbing, aligned to (1<<31) */
23 uint32_t shift_register;
24 int n_bits_in_shift_register;
25
26 /* bit pattern used after an error */
27 unsigned int guard_bit;
28
29 /* unpacking has overrun the end of the buffer */
30 int overrun;
31};
