74// A "Groupsock" is used to both send and receive packets.
75// As the name suggests, it was originally designed to send/receive
76// multicast, but it can send/receive unicast as well.
77
78class Groupsock: public OutputSocket {
79public:
80 Groupsock(UsageEnvironment& env, struct in_addr const& groupAddr,
81 Port port, u_int8_t ttl);
82 // used for a 'source-independent multicast' group
83 Groupsock(UsageEnvironment& env, struct in_addr const& groupAddr,
84 struct in_addr const& sourceFilterAddr,
85 Port port);
86 // used for a 'source-specific multicast' group
87 virtual ~Groupsock();
88
89 void changeDestinationParameters(struct in_addr const& newDestAddr,
90 Port newDestPort, int newDestTTL);
91 // By default, the destination address, port and ttl for
92 // outgoing packets are those that were specified in
93 // the constructor. This works OK for multicast sockets,
94 // but for unicast we usually want the destination port
95 // number, at least, to be different from the source port.
96 // (If a parameter is 0 (or ~0 for ttl), then no change made.)
97
98 // As a special case, we also allow multiple destinations (addresses & ports)
99 // (This can be used to implement multi-unicast.)
100 void addDestination(struct in_addr const& addr, Port const& port);
101 void removeDestination(struct in_addr const& addr, Port const& port);
102 void removeAllDestinations();
103
104 struct in_addr const& groupAddress() const {
105 return fIncomingGroupEId.groupAddress();
106 }
107 struct in_addr const& sourceFilterAddress() const {
108 return fIncomingGroupEId.sourceFilterAddress();
109 }
110
111 Boolean isSSM() const {
112 return fIncomingGroupEId.isSSM();
113 }
114
115 u_int8_t ttl() const { return fTTL; }
116
117 void multicastSendOnly(); // send, but don't receive any multicast packets
118
119 Boolean output(UsageEnvironment& env, u_int8_t ttl,
120 unsigned char* buffer, unsigned bufferSize,
121 DirectedNetInterface* interfaceNotToFwdBackTo = NULL);
122
123 DirectedNetInterfaceSet& members() { return fMembers; }
124
125 Boolean deleteIfNoMembers;
126 Boolean isSlave; // for tunneling
127
128 static NetInterfaceTrafficStats statsIncoming;
129 static NetInterfaceTrafficStats statsOutgoing;
130 static NetInterfaceTrafficStats statsRelayedIncoming;
131 static NetInterfaceTrafficStats statsRelayedOutgoing;
132 NetInterfaceTrafficStats statsGroupIncoming; // *not* static
133 NetInterfaceTrafficStats statsGroupOutgoing; // *not* static
134 NetInterfaceTrafficStats statsGroupRelayedIncoming; // *not* static
135 NetInterfaceTrafficStats statsGroupRelayedOutgoing; // *not* static
136
137 Boolean wasLoopedBackFromUs(UsageEnvironment& env,
138 struct sockaddr_in& fromAddress);
139
140public: // redefined virtual functions
141 virtual Boolean handleRead(unsigned char* buffer, unsigned bufferMaxSize,
142 unsigned& bytesRead,
143 struct sockaddr_in& fromAddress);
144
145private:
146 int outputToAllMembersExcept(DirectedNetInterface* exceptInterface,
147 u_int8_t ttlToFwd,
148 unsigned char* data, unsigned size,
149 netAddressBits sourceAddr);
150
151private:
152 GroupEId fIncomingGroupEId;
153 destRecord* fDests;
154 u_int8_t fTTL;
155 DirectedNetInterfaceSet fMembers;
156};
