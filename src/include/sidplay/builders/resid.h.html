18#ifndef _resid_h_
19#define _resid_h_
20
21/* Since ReSID is not part of this project we are actually
22 * creating a wrapper instead of implementing a SID emulation
23 */
24
25#include <vector>
26#include <sidplay/sidbuilder.h>
27#include <sidplay/event.h>
28
29
30/***************************************************************************
31 * ReSID Builder Class
32 ***************************************************************************/
33// Create the SID builder object
34class ReSIDBuilder: public sidbuilder
35{
36protected:
37 std::vector<sidemu *> sidobjs;
38
39private:
40 static const char *ERR_FILTER_DEFINITION;
41 char m_errorBuffer[100];
42 const char *m_error;
43
44public:
45 ReSIDBuilder (const char * const name);
46 ~ReSIDBuilder (void);
47 // true will give you the number of used devices.
48 // return values: 0 none, positive is used sids
49 // false will give you all available sids.
50 // return values: 0 endless, positive is available sids.
51 // use bool operator to determine error
52 uint devices (bool used);
53 uint create (uint sids);
54 sidemu *lock (c64env *env, sid2_model_t model);
55 void unlock (sidemu *device);
56 void remove (void);
57 const char *error (void) const { return m_error; }
58 const char *credits (void);
59
60 // Settings that effect all SIDs
61 void filter (bool enable);
62 void filter (const sid_filter_t *filter);
63 void sampling (uint_least32_t freq);
64};
65
66#endif // _resid_h_
