18#ifndef _sid2types_h_
19#define _sid2types_h_

21#include "sidtypes.h"
22#include "event.h"
23
24class sidbuilder;
25struct SidTuneInfo;
26
27#ifndef SIDPLAY2_DEFAULTS
28#define SIDPLAY2_DEFAULTS
29 // Maximum values
30 const uint_least8_t SID2_MAX_PRECISION = 16;
31 const uint_least8_t SID2_MAX_OPTIMISATION = 2;
32 // Delays <= MAX produce constant results.
33 // Delays > MAX produce random results
34 const uint_least16_t SID2_MAX_POWER_ON_DELAY = 0x1FFF;
35 // Default settings
36 const uint_least32_t SID2_DEFAULT_SAMPLING_FREQ = 44100;
37 const uint_least8_t SID2_DEFAULT_PRECISION = 16;
38 const uint_least8_t SID2_DEFAULT_OPTIMISATION = 1;
39 const bool SID2_DEFAULT_SID_SAMPLES = true; // Samples through sid
40 const uint_least16_t SID2_DEFAULT_POWER_ON_DELAY = SID2_MAX_POWER_ON_DELAY + 1;
41#endif // SIDPLAY2_DEFAULTS
42
43typedef enum {sid2_playing = 0, sid2_paused, sid2_stopped} sid2_player_t;
44typedef enum {sid2_left = 0, sid2_mono, sid2_stereo, sid2_right} sid2_playback_t;
45typedef enum {sid2_envPS = 0, sid2_envTP, sid2_envBS, sid2_envR,
46 sid2_envTR} sid2_env_t;
47typedef enum {SID2_MODEL_CORRECT, SID2_MOS6581, SID2_MOS8580} sid2_model_t;
48typedef enum {SID2_CLOCK_CORRECT, SID2_CLOCK_PAL, SID2_CLOCK_NTSC} sid2_clock_t;
49
50typedef enum
51{ // Soundcard sample format
52 SID2_LITTLE_SIGNED,
53 SID2_LITTLE_UNSIGNED,
54 SID2_BIG_SIGNED,
55 SID2_BIG_UNSIGNED
56} sid2_sample_t;
57
58/* Environment Modes
59sid2_envPS = Playsid
60sid2_envTP = Sidplay - Transparent Rom
61sid2_envBS = Sidplay - Bankswitching
62sid2_envR = Sidplay2 - Real C64 Environment
63*/
64
65struct sid2_config_t
66{
67 sid2_clock_t clockDefault; // Intended tune speed when unknown
68 bool clockForced;
69 sid2_clock_t clockSpeed; // User requested emulation speed
70 sid2_env_t environment;
71 bool forceDualSids;
72 bool emulateStereo;
73 uint_least32_t frequency;
74 uint_least8_t optimisation;
75 sid2_playback_t playback;
76 int precision;
77 sid2_model_t sidDefault; // Intended sid model when unknown
78 sidbuilder *sidEmulation;
79 sid2_model_t sidModel; // User requested sid model
80 bool sidSamples;
81 uint_least32_t leftVolume;
82 uint_least32_t rightVolume;
83 sid2_sample_t sampleFormat;
84 uint_least16_t powerOnDelay;
85 uint_least32_t sid2crcCount; // Max sid writes to form crc
86};
87
88struct sid2_info_t
89{
90 const char **credits;
91 uint channels;
92 uint_least16_t driverAddr;
93 uint_least16_t driverLength;
94 const char *name;
95 const SidTuneInfo *tuneInfo; // May not need this
96 const char *version;
97 // load, config and stop calls will reset this
98 // and remove all pending events! 10th sec resolution.
99 EventContext *eventContext;
100 uint maxsids;
101 sid2_env_t environment;
102 uint_least16_t powerOnDelay;
103 uint_least32_t sid2crc;
104 uint_least32_t sid2crcCount; // Number of sid writes forming crc
105};

107#endif // _sid2types_h_
