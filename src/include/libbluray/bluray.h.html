583 BD_EVENT_PLAYITEM = 7, /* current play item, 0...N-1 */
584 BD_EVENT_CHAPTER = 8, /* current chapter, 1...N */
585 BD_EVENT_PLAYMARK = 30, /* playmark reached */
586 BD_EVENT_END_OF_TITLE = 9,
587
588 /*
589 * stream selection
590 */
591
592 BD_EVENT_AUDIO_STREAM = 10, /* 1..32, 0xff = none */
593 BD_EVENT_IG_STREAM = 11, /* 1..32 */
594 BD_EVENT_PG_TEXTST_STREAM = 12, /* 1..255, 0xfff = none */
595 BD_EVENT_PIP_PG_TEXTST_STREAM = 13, /* 1..255, 0xfff = none */
596 BD_EVENT_SECONDARY_AUDIO_STREAM = 14, /* 1..32, 0xff = none */
597 BD_EVENT_SECONDARY_VIDEO_STREAM = 15, /* 1..32, 0xff = none */
598
599 BD_EVENT_PG_TEXTST = 16, /* 0 - disable, 1 - enable */
600 BD_EVENT_PIP_PG_TEXTST = 17, /* 0 - disable, 1 - enable */
601 BD_EVENT_SECONDARY_AUDIO = 18, /* 0 - disable, 1 - enable */
602 BD_EVENT_SECONDARY_VIDEO = 19, /* 0 - disable, 1 - enable */
603 BD_EVENT_SECONDARY_VIDEO_SIZE = 20, /* 0 - PIP, 0xf - fullscreen */
604
605 /*
606 * playback control
607 */
608
609 /* discontinuity in the stream (non-seamless connection). Reset demuxer PES buffers. */
610 BD_EVENT_DISCONTINUITY = 28, /* new timestamp (45 kHz) */
611
612 /* HDMV VM or JVM seeked the stream. Next read() will return data from new position. Flush all buffers. */
613 BD_EVENT_SEEK = 21,
614
615 /* still playback (pause) */
616 BD_EVENT_STILL = 22, /* 0 - off, 1 - on */
617
618 /* Still playback for n seconds (reached end of still mode play item).
619 * Playback continues by calling bd_read_skip_still(). */
620 BD_EVENT_STILL_TIME = 23, /* 0 = infinite ; 1...300 = seconds */
621
622 /* Play sound effect */
623 BD_EVENT_SOUND_EFFECT = 24, /* effect ID */
624
625 /*
626 * status
627 */
628
629 /* Nothing to do. Playlist is not playing, but title applet is running. */
630 BD_EVENT_IDLE = 29,
631
632 /* Pop-Up menu available */
633 BD_EVENT_POPUP = 25, /* 0 - no, 1 - yes */
634
635 /* Interactive menu visible */
636 BD_EVENT_MENU = 26, /* 0 - no, 1 - yes */
637
638 /* 3D */
639 BD_EVENT_STEREOSCOPIC_STATUS = 27, /* 0 - 2D, 1 - 3D */
