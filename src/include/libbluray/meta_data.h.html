20#if !defined(_META_DATA_H_)
21#define _META_DATA_H_
22
23#include <stdint.h>
24
25typedef struct meta_thumbnail {
26 char * path;
27 uint32_t xres;
28 uint32_t yres;
29} META_THUMBNAIL;
30
31typedef struct meta_title {
32 uint32_t title_number;
33 char * title_name;
34} META_TITLE;
35
36typedef struct meta_dl {
37 char language_code[4];
38 char * filename;
39 char * di_name;
40 char * di_alternative;
41 uint8_t di_num_sets;
42 uint8_t di_set_number;
43 uint32_t toc_count;
44 META_TITLE * toc_entries;
45 uint8_t thumb_count;
46 META_THUMBNAIL * thumbnails;
47} META_DL;
48
49typedef struct meta_root {
50 uint8_t dl_count;
51 META_DL * dl_entries;
52} META_ROOT;
53
54#endif // _META_DATA_H_
