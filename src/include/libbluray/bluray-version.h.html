20#ifndef BLURAY_VERSION_H_
21#define BLURAY_VERSION_H_
22
23#define BLURAY_VERSION_CODE(major, minor, micro) \
24 (((major) * 10000) + \
25 ((minor) * 100) + \
26 ((micro) * 1))
27
28#define BLURAY_VERSION_MAJOR 0
29#define BLURAY_VERSION_MINOR 5
30#define BLURAY_VERSION_MICRO 0
31
32#define BLURAY_VERSION_STRING "0.5.0"
33
34#define BLURAY_VERSION \
35 BLURAY_VERSION_CODE(BLURAY_VERSION_MAJOR, BLURAY_VERSION_MINOR, BLURAY_VERSION_MICRO)
36
37#endif /* BLURAY_VERSION_H_ */
