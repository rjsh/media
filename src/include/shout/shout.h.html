21#ifndef __LIBSHOUT_SHOUT_H__
22#define __LIBSHOUT_SHOUT_H__
23
24#include <sys/types.h>
25#ifdef WIN32
26#include <os.h>
27#endif
28
29#define SHOUTERR_SUCCESS (0)
30#define SHOUTERR_INSANE (-1)
31#define SHOUTERR_NOCONNECT (-2)
32#define SHOUTERR_NOLOGIN (-3)
33#define SHOUTERR_SOCKET (-4)
34#define SHOUTERR_MALLOC (-5)
35#define SHOUTERR_METADATA (-6)
36#define SHOUTERR_CONNECTED (-7)
37#define SHOUTERR_UNCONNECTED (-8)
38#define SHOUTERR_UNSUPPORTED (-9)
39
40#define SHOUTERR_BUSY (-10)
41
42#define SHOUT_FORMAT_OGG (0)
43#define SHOUT_FORMAT_MP3 (1)
44#define SHOUT_FORMAT_WEBM (2)
45/* backward-compatibility alias */
46#define SHOUT_FORMAT_VORBIS SHOUT_FORMAT_OGG
47
48#define SHOUT_PROTOCOL_HTTP (0)
49#define SHOUT_PROTOCOL_XAUDIOCAST (1)
50#define SHOUT_PROTOCOL_ICY (2)
51
52#define SHOUT_AI_BITRATE "bitrate"
53#define SHOUT_AI_SAMPLERATE "samplerate"
54#define SHOUT_AI_CHANNELS "channels"
55#define SHOUT_AI_QUALITY "quality"
56
57typedef struct shout shout_t;
58typedef struct _util_dict shout_metadata_t;
59
60#ifdef __cplusplus
61extern "C" {
62#endif
63
64/* initializes the shout library. Must be called before anything else */
65void shout_init(void);
66
67/* shuts down the shout library, deallocating any global storage. Don't call
68 * anything afterwards */
69void shout_shutdown(void);
70
71/* returns a static version string. Non-null parameters will be set to the
72 * value of the library major, minor, and patch levels, respectively */
73const char *shout_version(int *major, int *minor, int *patch);
74
75/* Allocates and sets up a new shout_t. Returns NULL if it can't get enough
76 * memory. The returns shout_t must be disposed of with shout_free. */
77shout_t *shout_new(void);
78
79/* Free all memory allocated by a shout_t */
80void shout_free(shout_t *self);
81
82/* Returns a statically allocated string describing the last shout error
83 * to occur. Only valid until the next libshout call on this shout_t */
84const char *shout_get_error(shout_t *self);
85
86/* Return the error code (e.g. SHOUTERR_SOCKET) for this shout instance */
87int shout_get_errno(shout_t *self);
88
89/* returns SHOUTERR_CONNECTED or SHOUTERR_UNCONNECTED */
90int shout_get_connected(shout_t *self);
91
92/* Parameter manipulation functions. libshout makes copies of all parameters,
93 * the caller may free its copies after giving them to libshout. May return
94 * SHOUTERR_MALLOC */
95
96int shout_set_host(shout_t *self, const char *host);
97const char *shout_get_host(shout_t *self);
98
99int shout_set_port(shout_t *self, unsigned short port);
100unsigned short shout_get_port(shout_t *self);
101
102int shout_set_password(shout_t *, const char *password);
103const char *shout_get_password(shout_t *self);
104
105int shout_set_mount(shout_t *self, const char *mount);
106const char *shout_get_mount(shout_t *self);
107
108int shout_set_name(shout_t *self, const char *name);
109const char *shout_get_name(shout_t *self);
110
111int shout_set_url(shout_t *self, const char *url);
112const char *shout_get_url(shout_t *self);
113
114int shout_set_genre(shout_t *self, const char *genre);
115const char *shout_get_genre(shout_t *self);
116
117int shout_set_user(shout_t *self, const char *username);
118const char *shout_get_user(shout_t *self);
119
120int shout_set_agent(shout_t *self, const char *username);
121const char *shout_get_agent(shout_t *self);
122
123int shout_set_description(shout_t *self, const char *description);
124const char *shout_get_description(shout_t *self);
125
126int shout_set_dumpfile(shout_t *self, const char *dumpfile);
127const char *shout_get_dumpfile(shout_t *self);
128
129int shout_set_audio_info(shout_t *self, const char *name, const char *value);
130const char *shout_get_audio_info(shout_t *self, const char *name);
131
132int shout_set_public(shout_t *self, unsigned int make_public);
133unsigned int shout_get_public(shout_t *self);
134
135/* takes a SHOUT_FORMAT_xxxx argument */
136int shout_set_format(shout_t *self, unsigned int format);
137unsigned int shout_get_format(shout_t *self);
138
139/* takes a SHOUT_PROTOCOL_xxxxx argument */
140int shout_set_protocol(shout_t *self, unsigned int protocol);
141unsigned int shout_get_protocol(shout_t *self);
142
143/* Instructs libshout to use nonblocking I/O. Must be called before
144 * shout_open (no switching back and forth midstream at the moment). */
145int shout_set_nonblocking(shout_t* self, unsigned int nonblocking);
146unsigned int shout_get_nonblocking(shout_t *self);
147
148/* Opens a connection to the server. All parameters must already be set */
149int shout_open(shout_t *self);
150
151/* Closes a connection to the server */
152int shout_close(shout_t *self);
153
154/* Send data to the server, parsing it for format specific timing info */
155int shout_send(shout_t *self, const unsigned char *data, size_t len);
156
157/* Send unparsed data to the server. Do not use this unless you know
158 * what you are doing.
159 * Returns the number of bytes written, or < 0 on error.
160 */
161ssize_t shout_send_raw(shout_t *self, const unsigned char *data, size_t len);
162
163/* return the number of bytes currently on the write queue (only makes sense in
164 * nonblocking mode). */
165ssize_t shout_queuelen(shout_t *self);
166
167/* Puts caller to sleep until it is time to send more data to the server */
168void shout_sync(shout_t *self);
169
170/* Amount of time in ms caller should wait before sending again */
171int shout_delay(shout_t *self);
172
173/* Sets MP3 metadata.
174 * Returns:
175 * SHOUTERR_SUCCESS
176 * SHOUTERR_UNSUPPORTED if format isn't MP3
177 * SHOUTERR_MALLOC
178 * SHOUTERR_INSANE
179 * SHOUTERR_NOCONNECT
180 * SHOUTERR_SOCKET
181 */
182int shout_set_metadata(shout_t *self, shout_metadata_t *metadata);
183
184/* Allocates a new metadata structure. Must be freed by shout_metadata_free. */
185shout_metadata_t *shout_metadata_new(void);
186
187/* Free resources allocated by shout_metadata_t */
188void shout_metadata_free(shout_metadata_t *self);
189
190/* Add a parameter to the metadata structure.
191 * Returns:
192 * SHOUTERR_SUCCESS on success
193 * SHOUTERR_INSANE if self isn't a valid shout_metadata_t* or name is null
194 * SHOUTERR_MALLOC if memory can't be allocated */
195int shout_metadata_add(shout_metadata_t *self, const char *name, const char *value);
196
197#ifdef __cplusplus
198}
199#endif
200
201/* --- Compiled features --- */
202
203#define SHOUT_THREADSAFE 1
204
205#endif /* __LIBSHOUT_SHOUT_H__ */
