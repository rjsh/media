41/*
42 * Output rendering
43 * Following is the rendering interface for X windows,
44 * to get the decode output surface to a X drawable
45 * It basically performs a de-interlacing (if needed),
46 * color space conversion and scaling to the destination
47 * rectangle
48 */
49VAStatus vaPutSurface (
50 VADisplay dpy,
51 VASurfaceID surface,
52 Drawable draw, /* X Drawable */
53 short srcx,
54 short srcy,
55 unsigned short srcw,
56 unsigned short srch,
57 short destx,
58 short desty,
59 unsigned short destw,
60 unsigned short desth,
61 VARectangle *cliprects, /* client supplied destination clip list */
62 unsigned int number_cliprects, /* number of clip rects in the clip list */
63 unsigned int flags /* PutSurface flags */
64);
65
66#ifdef __cplusplus
67}
