56/**
57 * \defgroup api_winsys_x11 X11 Window System Integration Layer
58 *
59 * The set of VDPAU functionality specific to usage with the X
60 * Window System.
61 *
62 * \section Driver Library Layout
63 *
64 * An X11-oriented VDPAU installation consists of the following
65 * components:
66 *
67 * - Header files. These files are located in the standard
68 * system header file path.
69 * - \c vdpau/vdpau.h
70 * - \c vdpau/vdpau_x11.h
71 * - The VDPAU wrapper library. These files are located in the
72 * standard system (possibly X11-specific) library path.
73 * - \c libvdpau.so.1 (runtime)
74 * - \c libvdpau.so (development)
75 * - Back-end driver files. These files are located in a
76 * system-defined library path, which is configurable at compile
77 * time but is typically /usr/lib/vdpau. Use `pkg-config
78 * --variable=moduledir vdpau` to locate the driver install path.
79 * - \c $moduledir/libvdpau_\%s.so.1
80 * For example:
81 * - \c /usr/lib/vdpau/libvdpau_nvidia.so.1
82 * - \c /usr/lib/vdpau/libvdpau_intel.so.1
83 * - \c /usr/lib/vdpau/libvdpau_ati.so.1
84 *
85 * The VDPAU wrapper library implements just one function; \ref
86 * vdp_device_create_x11. The wrapper will implement this function
87 * by dynamically loading the appropriate back-end driver file
88 * mentioned above. Long-term, the wrapper will use a
89 * VDPAU-specific X extension to determine which back-end driver
90 * to load. Currently, the wrapper library hard-codes the driver
91 * name as "nvidia", although this can be overridden using the
92 * environment variable VDPAU_DRIVER.
93 *
94 * The back-end driver is expected to implement a function named
95 * \b vdp_imp_device_create_x11. The wrapper will call this function to
96 * actually implement the \ref vdp_device_create_x11 application call.
97 *
98 * Note that it is theoretically possible for an application to
99 * create multiple \ref VdpDevice "VdpDevice" objects. In this
100 * case, the wrapper library may load multiple back-end drivers
101 * into the same application, and/or invoke a specific back-end
102 * driver's \b VdpImpDeviceCreateX11 multiple times. The wrapper
103 * libray imposes no policy regarding whether the application
104 * may instantiate multiple \ref VdpDevice "VdpDevice" objects for
105 * the same display and/or screen. However, back-end drivers are
106 * free to limit the number of \ref VdpDevice "VdpDevice" objects
107 * as required by their implementation.
108 *
109 * @{
110 */
111
112/**
113 * \brief Create a VdpDevice object for use with X11.
114 * \param[in] display The X Display that the VdpDevice VdpDevice
115 * will operate against.
116 * \param[in] screen The X screen that the VdpDevice will operate
117 * against.
118 * \param[out] device The new device's handle.
119 * \param[out] get_proc_address The get_proc_address entry point
120 * to use with this device.
121 * \return VdpStatus The completion status of the operation.
122 */
123typedef VdpStatus VdpDeviceCreateX11(
124 Display * display,
125 int screen,
126 /* output parameters follow */
127 VdpDevice * device,
128 VdpGetProcAddress * * get_proc_address
129);
130
131/**
132 * \brief Create a VdpDevice object for use with X11.
133 * This is an actual symbol of type \ref VdpDeviceCreateX11
134 *
135 */
136VdpDeviceCreateX11 vdp_device_create_x11;
137
138/**
139 * \brief Create a VdpPresentationQueueTarget for use with X11.
140 * \param[in] device The device that will contain the queue
141 * target.
142 * \param[in] drawable The X11 Drawable that the presentation
143 * queue will present into.
144 * \param[out] target The new queue target's handle.
145 * \return VdpStatus The completion status of the operation.
146 *
147 * Note: VDPAU expects to own the entire drawable for the duration of time
148 * that the presentation queue target exists. In particular,
149 * implementations may choose to manipulate client-visible X11 window state
150 * as required. As such, it is recommended that applications create a
151 * dedicated window for the presentation queue target, as a child
152 * (grand-child, ...) of their top-level application window.
153 *
154 * Applications may also create child-windows of the presentation queue
155 * target, which will cover any presented video in the normal fashion. VDPAU
156 * implementations will not manipulate such child windows in any fashion.
157 */
158typedef VdpStatus VdpPresentationQueueTargetCreateX11(
159 VdpDevice device,
160 Drawable drawable,
161 /* output parameters follow */
162 VdpPresentationQueueTarget * target
163);
