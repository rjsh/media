78/*
79@@ LUA_PATH_DEFAULT is the default path that Lua uses to look for
80@* Lua libraries.
81@@ LUA_CPATH_DEFAULT is the default path that Lua uses to look for
82@* C libraries.
83** CHANGE them if your machine has a non-conventional directory
84** hierarchy or if you want to install your libraries in
85** non-conventional directories.
86*/
87#if defined(_WIN32) /* { */
88/*
89** In Windows, any exclamation mark ('!') in the path is replaced by the
90** path of the directory of the executable file of the current process.
91*/
92#define LUA_LDIR "!\\lua\\"
93#define LUA_CDIR "!\\"
94#define LUA_PATH_DEFAULT \
95 LUA_LDIR"?.lua;" LUA_LDIR"?\\init.lua;" \
96 LUA_CDIR"?.lua;" LUA_CDIR"?\\init.lua;" ".\\?.lua"
97#define LUA_CPATH_DEFAULT \
98 LUA_CDIR"?.dll;" LUA_CDIR"loadall.dll;" ".\\?.dll"
99
100#else /* }{ */
101/* This defines DEB_HOST_MULTIARCH */
102#include "lua5.2-deb-multiarch.h"
103#define LUA_VDIR LUA_VERSION_MAJOR "." LUA_VERSION_MINOR "/"
104#define LUA_ROOT "/usr/local/"
105#define LUA_ROOT2 "/usr/"
106#define LUA_LDIR LUA_ROOT "share/lua/" LUA_VDIR
107#define LUA_LDIR2 LUA_ROOT2 "share/lua/" LUA_VDIR
108#define LUA_CDIR LUA_ROOT "lib/lua/" LUA_VDIR
109#define LUA_CDIR2 LUA_ROOT2 "lib/" DEB_HOST_MULTIARCH "/lua/" LUA_VDIR
110#define LUA_CDIR3 LUA_ROOT2 "lib/lua/" LUA_VDIR
111#define LUA_PATH_DEFAULT \
112 LUA_LDIR"?.lua;" LUA_LDIR"?/init.lua;" \
113 LUA_CDIR"?.lua;" LUA_CDIR"?/init.lua;" \
114 LUA_LDIR2"?.lua;" LUA_LDIR2"?/init.lua;" "./?.lua"
115#define LUA_CPATH_DEFAULT \
116 LUA_CDIR"?.so;" LUA_CDIR2"?.so;" \
117 LUA_CDIR3"?.so;" LUA_CDIR"loadall.so;" "./?.so"

170/*
171@@ LUAI_FUNC is a mark for all extern functions that are not to be
172@* exported to outside modules.
173@@ LUAI_DDEF and LUAI_DDEC are marks for all extern (const) variables
174@* that are not to be exported to outside modules (LUAI_DDEF for
175@* definitions and LUAI_DDEC for declarations).
176** CHANGE them if you need to mark them in some special way. Elf/gcc
177** (versions 3.2 and later) mark them as "hidden" to optimize access
178** when Lua is compiled as a shared library. Not all elf targets support
179** this attribute. Unfortunately, gcc does not offer a way to check
180** whether the target offers that support, and those without support
181** give a warning about it. To avoid these warnings, change to the
182** default definition.
183*/
184#if defined(__GNUC__) && ((__GNUC__*100 + __GNUC_MINOR__) >= 302) && \
185 defined(__ELF__) /* { */
186#define LUAI_FUNC __attribute__((visibility("hidden"))) extern
187#define LUAI_DDEC LUAI_FUNC
188#define LUAI_DDEF /* empty */
189
190#else /* }{ */
191#define LUAI_FUNC extern
192#define LUAI_DDEC extern
193#define LUAI_DDEF /* empty */
194#endif /* } */

320@@ LUAI_BITSINT defines the number of bits in an int.
321** CHANGE here if Lua cannot automatically detect the number of bits of
322** your machine. Probably you do not need to change this.
323*/
324/* avoid overflows in comparison */

336@@ LUA_INT32 is an signed integer with exactly 32 bits.
337@@ LUAI_UMEM is an unsigned integer big enough to count the total
338@* memory used by Lua.
339@@ LUAI_MEM is a signed integer big enough to count the total memory
340@* used by Lua.
341** CHANGE here if for some weird reason the default definitions are not
342** good enough for your machine. Probably you do not need to change
343** this.

421/*
422@@ lua_str2number converts a decimal numeric string to a number.
423@@ lua_strx2number converts an hexadecimal numeric string to a number.
424** In C99, 'strtod' does both conversions. C89, however, has no function
425** to convert floating hexadecimal strings to numbers. For these

481#if defined(LUA_NUMBER_DOUBLE) && !defined(LUA_ANSI) /* { */
482/*
483** The next definitions activate some tricks to speed up the
484** conversion from doubles to integer types, mainly to LUA_UNSIGNED.
485**
486@@ LUA_MSASMTRICK uses Microsoft assembler to avoid clashes with a
487** DirectX idiosyncrasy.
488**
489@@ LUA_IEEE754TRICK uses a trick that should work on any machine
490** using IEEE754 with a 32-bit integer type.
491**
492@@ LUA_IEEELL extends the trick to LUA_INTEGER; should only be
493** defined when LUA_INTEGER is a 32-bit integer.
494**
495@@ LUA_IEEEENDIAN is the endianness of doubles in your machine
496** (0 for little endian, 1 for big endian); if not defined, Lua will
497** check it dynamically for LUA_IEEE754TRICK (but not for LUA_NANTRICK).
498**
499@@ LUA_NANTRICK controls the use of a trick to pack all types into
500** a single double value, using NaN values to represent non-number
501** values. The trick only works on 32-bit machines (ints and pointers
502** are 32-bit values) with numbers represented as IEEE 754-2008 doubles
503** with conventional endianess (12345678 or 87654321), in CPUs that do
504** not produce signaling NaN values (all NaNs are quiet).
505*/
506
507/* Microsoft compiler on a Pentium (32 bit) ? */
508#if defined(LUA_WIN) && defined(_MSC_VER) && defined(_M_IX86) /* { */
509
510#define LUA_MSASMTRICK
511#define LUA_IEEEENDIAN 0
512#define LUA_NANTRICK
513
514
515/* pentium 32 bits? */
516#elif defined(__i386__) || defined(__i386) || defined(__X86__) /* }{ */
517
518#define LUA_IEEE754TRICK
519#define LUA_IEEELL
520#define LUA_IEEEENDIAN 0
521#define LUA_NANTRICK
522
523/* pentium 64 bits? */
524#elif defined(__x86_64) /* }{ */
525
526#define LUA_IEEE754TRICK
527#define LUA_IEEEENDIAN 0
528
529#elif defined(__POWERPC__) || defined(__ppc__) /* }{ */
530
531#define LUA_IEEE754TRICK
532#define LUA_IEEEENDIAN 1
533
534#else /* }{ */
535
536/* assume IEEE754 and a 32-bit integer type */
537#define LUA_IEEE754TRICK
538
539#endif /* } */
540
541#endif /* } */
