42#ifndef QMARGINS_H
43#define QMARGINS_H
44
45#include <QtCore/qnamespace.h>
46
47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Core)
52
53class QMargins
54{
55public:
56 QMargins();
57 QMargins(int left, int top, int right, int bottom);
58
59 bool isNull() const;
60
61 int left() const;
62 int top() const;
63 int right() const;
64 int bottom() const;
65
66 void setLeft(int left);
67 void setTop(int top);
68 void setRight(int right);
69 void setBottom(int bottom);
70
71private:
72 int m_left;
73 int m_top;
74 int m_right;
75 int m_bottom;
76
77 friend inline bool operator==(const QMargins &, const QMargins &);
78 friend inline bool operator!=(const QMargins &, const QMargins &);
79};
80
81Q_DECLARE_TYPEINFO(QMargins, Q_MOVABLE_TYPE);
82
83/*****************************************************************************
84 QMargins inline functions
85 *****************************************************************************/
86
87inline QMargins::QMargins()
88{ m_top = m_bottom = m_left = m_right = 0; }
89
90inline QMargins::QMargins(int aleft, int atop, int aright, int abottom)
91 : m_left(aleft), m_top(atop), m_right(aright), m_bottom(abottom) {}
92
93inline bool QMargins::isNull() const
94{ return m_left==0 && m_top==0 && m_right==0 && m_bottom==0; }
95
96inline int QMargins::left() const
97{ return m_left; }
98
99inline int QMargins::top() const
100{ return m_top; }
101
102inline int QMargins::right() const
103{ return m_right; }
104
105inline int QMargins::bottom() const
106{ return m_bottom; }
107
108
109inline void QMargins::setLeft(int aleft)
110{ m_left = aleft; }
111
112inline void QMargins::setTop(int atop)
113{ m_top = atop; }
114
115inline void QMargins::setRight(int aright)
116{ m_right = aright; }
117
118inline void QMargins::setBottom(int abottom)
119{ m_bottom = abottom; }
120
121inline bool operator==(const QMargins &m1, const QMargins &m2)
122{
123 return
124 m1.m_left == m2.m_left &&
125 m1.m_top == m2.m_top &&
126 m1.m_right == m2.m_right &&
127 m1.m_bottom == m2.m_bottom;
128}
129
130inline bool operator!=(const QMargins &m1, const QMargins &m2)
131{
132 return
133 m1.m_left != m2.m_left ||
134 m1.m_top != m2.m_top ||
135 m1.m_right != m2.m_right ||
136 m1.m_bottom != m2.m_bottom;
137}
