53#ifndef QT_NO_SIGNALMAPPER
54class QSignalMapperPrivate;
55
56class Q_CORE_EXPORT QSignalMapper : public QObject
57{
58 Q_OBJECT
59 Q_DECLARE_PRIVATE(QSignalMapper)
60public:
61 explicit QSignalMapper(QObject *parent = 0);
62 ~QSignalMapper();
63
64 void setMapping(QObject *sender, int id);
65 void setMapping(QObject *sender, const QString &text);
66 void setMapping(QObject *sender, QWidget *widget);
67 void setMapping(QObject *sender, QObject *object);
68 void removeMappings(QObject *sender);
69
70 QObject *mapping(int id) const;
71 QObject *mapping(const QString &text) const;
72 QObject *mapping(QWidget *widget) const;
73 QObject *mapping(QObject *object) const;
74
75Q_SIGNALS:
76 void mapped(int);
77 void mapped(const QString &);
78 void mapped(QWidget *);
79 void mapped(QObject *);
80
81public Q_SLOTS:
82 void map();
83 void map(QObject *sender);
84
85private:
86 Q_DISABLE_COPY(QSignalMapper)
87 Q_PRIVATE_SLOT(d_func(), void _q_senderDestroyed())
88
89#ifdef QT3_SUPPORT
90public:
91 QT3_SUPPORT_CONSTRUCTOR QSignalMapper(QObject *parent, const char *name);
92#endif
93};
