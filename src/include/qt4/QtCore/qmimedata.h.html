42#ifndef QMIMEDATA_H
43#define QMIMEDATA_H
44
45#include <QtCore/qvariant.h>
46#include <QtCore/qobject.h>
47
48QT_BEGIN_HEADER
49
50QT_BEGIN_NAMESPACE
51
52QT_MODULE(Core)
53
54class QUrl;
55class QMimeDataPrivate;
56
57class Q_CORE_EXPORT QMimeData : public QObject
58{
59 Q_OBJECT
60public:
61 QMimeData();
62 ~QMimeData();
63
64 QList<QUrl> urls() const;
65 void setUrls(const QList<QUrl> &urls);
66 bool hasUrls() const;
67
68 QString text() const;
69 void setText(const QString &text);
70 bool hasText() const;
71
72 QString html() const;
73 void setHtml(const QString &html);
74 bool hasHtml() const;
75
76 QVariant imageData() const;
77 void setImageData(const QVariant &image);
78 bool hasImage() const;
79
80 QVariant colorData() const;
81 void setColorData(const QVariant &color);
82 bool hasColor() const;
83
84 QByteArray data(const QString &mimetype) const;
85 void setData(const QString &mimetype, const QByteArray &data);
86 void removeFormat(const QString &mimetype);
87
88 virtual bool hasFormat(const QString &mimetype) const;
89 virtual QStringList formats() const;
90
91 void clear();
92protected:
93 virtual QVariant retrieveData(const QString &mimetype,
94 QVariant::Type preferredType) const;
95private:
96 Q_DISABLE_COPY(QMimeData)
97 Q_DECLARE_PRIVATE(QMimeData)
98};
99
100QT_END_NAMESPACE

102QT_END_HEADER

104#endif // QMIMEDATA_H
