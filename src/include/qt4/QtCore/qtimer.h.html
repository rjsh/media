42#ifndef QTIMER_H
43#define QTIMER_H
44
45#ifndef QT_NO_QOBJECT
46
47#include <QtCore/qbasictimer.h> // conceptual inheritance
48#include <QtCore/qobject.h>
49
50QT_BEGIN_HEADER
51
52QT_BEGIN_NAMESPACE
53
54QT_MODULE(Core)
55
56class Q_CORE_EXPORT QTimer : public QObject
57{
58 Q_OBJECT
59 Q_PROPERTY(bool singleShot READ isSingleShot WRITE setSingleShot)
60 Q_PROPERTY(int interval READ interval WRITE setInterval)
61 Q_PROPERTY(bool active READ isActive)
62public:
63 explicit QTimer(QObject *parent = 0);
64#ifdef QT3_SUPPORT
65 QT3_SUPPORT_CONSTRUCTOR QTimer(QObject *parent, const char *name);
66#endif
67 ~QTimer();
68
69 inline bool isActive() const { return id >= 0; }
70 int timerId() const { return id; }
71
72 void setInterval(int msec);
73 int interval() const { return inter; }
74
75 inline void setSingleShot(bool singleShot);
76 inline bool isSingleShot() const { return single; }
77
78 static void singleShot(int msec, QObject *receiver, const char *member);
79
80public Q_SLOTS:
81 void start(int msec);
82
83 void start();
84 void stop();
85
86#ifdef QT3_SUPPORT
87 inline QT_MOC_COMPAT void changeInterval(int msec) { start(msec); }
88 QT_MOC_COMPAT int start(int msec, bool sshot);
89#endif
90
91Q_SIGNALS:
92 void timeout();
93
94protected:
95 void timerEvent(QTimerEvent *);
96
97private:
98 Q_DISABLE_COPY(QTimer)
99
100 inline int startTimer(int){ return -1;}
101 inline void killTimer(int){}
102
103 int id, inter, del;
104 uint single : 1;
105 uint nulltimer : 1;
106};
107
108inline void QTimer::setSingleShot(bool asingleShot) { single = asingleShot; }
109
110QT_END_NAMESPACE
111
112QT_END_HEADER
113
114#endif // QT_NO_QOBJECT
115
116#endif // QTIMER_H
