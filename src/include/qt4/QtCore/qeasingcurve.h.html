42#ifndef QEASINGCURVE_H
43#define QEASINGCURVE_H
44
45#include <QtCore/qglobal.h>
46#include <QtCore/qobjectdefs.h>
47
48QT_BEGIN_HEADER
49
50QT_BEGIN_NAMESPACE
51
52QT_MODULE(Core)
53
54class QEasingCurvePrivate;
55class Q_CORE_EXPORT QEasingCurve
56{
57 Q_GADGET
58 Q_ENUMS(Type)
59public:
60 enum Type {
61 Linear,
62 InQuad, OutQuad, InOutQuad, OutInQuad,
63 InCubic, OutCubic, InOutCubic, OutInCubic,
64 InQuart, OutQuart, InOutQuart, OutInQuart,
65 InQuint, OutQuint, InOutQuint, OutInQuint,
66 InSine, OutSine, InOutSine, OutInSine,
67 InExpo, OutExpo, InOutExpo, OutInExpo,
68 InCirc, OutCirc, InOutCirc, OutInCirc,
69 InElastic, OutElastic, InOutElastic, OutInElastic,
70 InBack, OutBack, InOutBack, OutInBack,
71 InBounce, OutBounce, InOutBounce, OutInBounce,
72 InCurve, OutCurve, SineCurve, CosineCurve,
73 Custom, NCurveTypes
74 };
75
76 QEasingCurve(Type type = Linear);
77 QEasingCurve(const QEasingCurve &other);
78 ~QEasingCurve();
79
80 QEasingCurve &operator=(const QEasingCurve &other);
81 bool operator==(const QEasingCurve &other) const;
82 inline bool operator!=(const QEasingCurve &other) const
83 { return !(this->operator==(other)); }
84
85 qreal amplitude() const;
86 void setAmplitude(qreal amplitude);
87
88 qreal period() const;
89 void setPeriod(qreal period);
90
91 qreal overshoot() const;
92 void setOvershoot(qreal overshoot);
93
94 Type type() const;
95 void setType(Type type);
96 typedef qreal (*EasingFunction)(qreal progress);
97 void setCustomType(EasingFunction func);
98 EasingFunction customType() const;
99
100 qreal valueForProgress(qreal progress) const;
101private:
102 QEasingCurvePrivate *d_ptr;
103#ifndef QT_NO_DEBUG_STREAM
104 friend Q_CORE_EXPORT QDebug operator<<(QDebug debug, const QEasingCurve &item);
105#endif
106#ifndef QT_NO_DATASTREAM
107 friend Q_CORE_EXPORT QDataStream &operator<<(QDataStream &, const QEasingCurve&);
108 friend Q_CORE_EXPORT QDataStream &operator>>(QDataStream &, QEasingCurve &);
109#endif
110};
