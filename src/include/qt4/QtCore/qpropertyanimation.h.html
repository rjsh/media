53#ifndef QT_NO_ANIMATION
54
55class QPropertyAnimationPrivate;
56class Q_CORE_EXPORT QPropertyAnimation : public QVariantAnimation
57{
58 Q_OBJECT
59 Q_PROPERTY(QByteArray propertyName READ propertyName WRITE setPropertyName)
60 Q_PROPERTY(QObject* targetObject READ targetObject WRITE setTargetObject)
61
62public:
63 QPropertyAnimation(QObject *parent = 0);
64 QPropertyAnimation(QObject *target, const QByteArray &propertyName, QObject *parent = 0);
65 ~QPropertyAnimation();
66
67 QObject *targetObject() const;
68 void setTargetObject(QObject *target);
69
70 QByteArray propertyName() const;
71 void setPropertyName(const QByteArray &propertyName);
72
73protected:
74 bool event(QEvent *event);
75 void updateCurrentValue(const QVariant &value);
76 void updateState(QAbstractAnimation::State newState, QAbstractAnimation::State oldState);
77
78private:
79 Q_DISABLE_COPY(QPropertyAnimation)
80 Q_DECLARE_PRIVATE(QPropertyAnimation)
81};
