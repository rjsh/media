42#ifndef QSTRINGMATCHER_H
43#define QSTRINGMATCHER_H
44
45#include <QtCore/qstring.h>
46
47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Core)
52
53class QStringMatcherPrivate;
54
55class Q_CORE_EXPORT QStringMatcher
56{
57public:
58 QStringMatcher();
59 QStringMatcher(const QString &pattern,
60 Qt::CaseSensitivity cs = Qt::CaseSensitive);
61 QStringMatcher(const QChar *uc, int len,
62 Qt::CaseSensitivity cs = Qt::CaseSensitive);
63 QStringMatcher(const QStringMatcher &other);
64 ~QStringMatcher();
65
66 QStringMatcher &operator=(const QStringMatcher &other);
67
68 void setPattern(const QString &pattern);
69 void setCaseSensitivity(Qt::CaseSensitivity cs);
70
71 int indexIn(const QString &str, int from = 0) const;
72 int indexIn(const QChar *str, int length, int from = 0) const;
73 QString pattern() const;
74 inline Qt::CaseSensitivity caseSensitivity() const { return q_cs; }
75
76private:
77 QStringMatcherPrivate *d_ptr;
78 QString q_pattern;
79 Qt::CaseSensitivity q_cs;
80#ifdef Q_CC_RVCT
81// explicitly allow anonymous unions for RVCT to prevent compiler warnings
82# pragma push
83# pragma anon_unions
84#endif
85 struct Data {
86 uchar q_skiptable[256];
87 const QChar *uc;
88 int len;
89 };
90 union {
91 uint q_data[256];
92 Data p;
93 };
94#ifdef Q_CC_RVCT
95# pragma pop
96#endif
97};

99QT_END_NAMESPACE
