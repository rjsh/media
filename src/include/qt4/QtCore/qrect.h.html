207inline QRect::QRect(int aleft, int atop, int awidth, int aheight)
208{
209 x1 = aleft;
210 y1 = atop;
211 x2 = (aleft + awidth - 1);
212 y2 = (atop + aheight - 1);
213}

218 y1 = atopLeft.y();

223inline QRect::QRect(const QPoint &atopLeft, const QSize &asize)
224{
