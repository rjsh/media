42#ifndef QANIMATION_H
43#define QANIMATION_H
44
45#include <QtCore/qeasingcurve.h>
46#include <QtCore/qabstractanimation.h>
47#include <QtCore/qvector.h>
48#include <QtCore/qvariant.h>
49#include <QtCore/qpair.h>
50
51QT_BEGIN_HEADER
52
53QT_BEGIN_NAMESPACE
54
55QT_MODULE(Core)
56
57#ifndef QT_NO_ANIMATION
58
59class QVariantAnimationPrivate;
60class Q_CORE_EXPORT QVariantAnimation : public QAbstractAnimation
61{
62 Q_OBJECT
63 Q_PROPERTY(QVariant startValue READ startValue WRITE setStartValue)
64 Q_PROPERTY(QVariant endValue READ endValue WRITE setEndValue)
65 Q_PROPERTY(QVariant currentValue READ currentValue NOTIFY valueChanged)
66 Q_PROPERTY(int duration READ duration WRITE setDuration)
67 Q_PROPERTY(QEasingCurve easingCurve READ easingCurve WRITE setEasingCurve)
68
69public:
70 typedef QPair<qreal, QVariant> KeyValue;
71 typedef QVector<KeyValue> KeyValues;
72
73 QVariantAnimation(QObject *parent = 0);
74 ~QVariantAnimation();
75
76 QVariant startValue() const;
77 void setStartValue(const QVariant &value);
78
79 QVariant endValue() const;
80 void setEndValue(const QVariant &value);
81
82 QVariant keyValueAt(qreal step) const;
83 void setKeyValueAt(qreal step, const QVariant &value);
84
85 KeyValues keyValues() const;
86 void setKeyValues(const KeyValues &values);
87
88 QVariant currentValue() const;
89
90 int duration() const;
91 void setDuration(int msecs);
92
93 QEasingCurve easingCurve() const;
94 void setEasingCurve(const QEasingCurve &easing);
95
96 typedef QVariant (*Interpolator)(const void *from, const void *to, qreal progress);
97
98Q_SIGNALS:
99 void valueChanged(const QVariant &value);
100
101protected:
102 QVariantAnimation(QVariantAnimationPrivate &dd, QObject *parent = 0);
103 bool event(QEvent *event);
104
105 void updateCurrentTime(int);
106 void updateState(QAbstractAnimation::State newState, QAbstractAnimation::State oldState);
107
108 virtual void updateCurrentValue(const QVariant &value) = 0;
109 virtual QVariant interpolated(const QVariant &from, const QVariant &to, qreal progress) const;
110
111private:
112 template <typename T> friend void qRegisterAnimationInterpolator(QVariant (*func)(const T &, const T &, qreal));
113 static void registerInterpolator(Interpolator func, int interpolationType);
114
115 Q_DISABLE_COPY(QVariantAnimation)
116 Q_DECLARE_PRIVATE(QVariantAnimation)
117};
118
119template <typename T>
120void qRegisterAnimationInterpolator(QVariant (*func)(const T &from, const T &to, qreal progress)) {
121 QVariantAnimation::registerInterpolator(reinterpret_cast<QVariantAnimation::Interpolator>(func), qMetaTypeId<T>());
122}
123
124#endif //QT_NO_ANIMATION
125
126QT_END_NAMESPACE
127
128QT_END_HEADER
129
130#endif //QANIMATION_H
