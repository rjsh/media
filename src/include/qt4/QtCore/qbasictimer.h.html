42#ifndef QBASICTIMER_H
43#define QBASICTIMER_H
44
45#include <QtCore/qglobal.h>
46
47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Core)
52
53class QObject;
54
55class Q_CORE_EXPORT QBasicTimer
56{
57 int id;
58public:
59 inline QBasicTimer() : id(0) {}
60 inline ~QBasicTimer() { if (id) stop(); }
61
62 inline bool isActive() const { return id != 0; }
63 inline int timerId() const { return id; }
64
65 void start(int msec, QObject *obj);
66 void stop();
67};
68Q_DECLARE_TYPEINFO(QBasicTimer, Q_MOVABLE_TYPE);
69
70QT_END_NAMESPACE
71
72QT_END_HEADER
73
74#endif // QBASICTIMER_H
