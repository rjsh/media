56class QResourcePrivate;
57
58class Q_CORE_EXPORT QResource
59{
60public:
61 QResource(const QString &file=QString(), const QLocale &locale=QLocale());
62 ~QResource();
63
64 void setFileName(const QString &file);
65 QString fileName() const;
66 QString absoluteFilePath() const;
67
68 void setLocale(const QLocale &locale);
69 QLocale locale() const;
70
71 bool isValid() const;
72
73 bool isCompressed() const;
74 qint64 size() const;
75 const uchar *data() const;
76
77 static void addSearchPath(const QString &path);
78 static QStringList searchPaths();
79
80 static bool registerResource(const QString &rccFilename, const QString &resourceRoot=QString());
81 static bool unregisterResource(const QString &rccFilename, const QString &resourceRoot=QString());
82
83 static bool registerResource(const uchar *rccData, const QString &resourceRoot=QString());
84 static bool unregisterResource(const uchar *rccData, const QString &resourceRoot=QString());
85
86protected:
87 friend class QResourceFileEngine;
88 friend class QResourceFileEngineIterator;
89 bool isDir() const;
90 inline bool isFile() const { return !isDir(); }
91 QStringList children() const;
92
93protected:
94 QScopedPointer<QResourcePrivate> d_ptr;
95
96private:
97 Q_DECLARE_PRIVATE(QResource)
98};
