42#ifndef QEVENTLOOP_H
43#define QEVENTLOOP_H
44
45#include <QtCore/qobject.h>
46
47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Core)
52
53class QEventLoopPrivate;
54
55class Q_CORE_EXPORT QEventLoop : public QObject
56{
57 Q_OBJECT
58 Q_DECLARE_PRIVATE(QEventLoop)
59
60public:
61 explicit QEventLoop(QObject *parent = 0);
62 ~QEventLoop();
63
64 enum ProcessEventsFlag {
65 AllEvents = 0x00,
66 ExcludeUserInputEvents = 0x01,
67 ExcludeSocketNotifiers = 0x02,
68 WaitForMoreEvents = 0x04,
69#ifdef QT3_SUPPORT
70 ExcludeUserInput = ExcludeUserInputEvents,
71 WaitForMore = WaitForMoreEvents,
72#endif
73 X11ExcludeTimers = 0x08
74#ifdef QT_DEPRECATED
75 , DeferredDeletion = 0x10
76#endif
77 , EventLoopExec = 0x20
78 , DialogExec = 0x40
79 };
80 Q_DECLARE_FLAGS(ProcessEventsFlags, ProcessEventsFlag)
81
82 bool processEvents(ProcessEventsFlags flags = AllEvents);
83 void processEvents(ProcessEventsFlags flags, int maximumTime);
84
85 int exec(ProcessEventsFlags flags = AllEvents);
86 void exit(int returnCode = 0);
87 bool isRunning() const;
88
89 void wakeUp();
90
91public Q_SLOTS:
92 void quit();
93};
94
95Q_DECLARE_OPERATORS_FOR_FLAGS(QEventLoop::ProcessEventsFlags)
96
97QT_END_NAMESPACE
98
99QT_END_HEADER
100
101#endif // QEVENTLOOP_H
