45#include <QtGui/qwidget.h>
46#include <QtGui/qvalidator.h>

49
50QT_BEGIN_NAMESPACE
51
52QT_MODULE(Gui)
53
54#ifndef QT_NO_SPINBOX
55
56class QLineEdit;
57
58class QAbstractSpinBoxPrivate;
59class QStyleOptionSpinBox;
60
61class Q_GUI_EXPORT QAbstractSpinBox : public QWidget
62{
63 Q_OBJECT
64
65 Q_ENUMS(ButtonSymbols)
66 Q_ENUMS(CorrectionMode)
67 Q_PROPERTY(bool wrapping READ wrapping WRITE setWrapping)
68 Q_PROPERTY(bool frame READ hasFrame WRITE setFrame)
69 Q_PROPERTY(Qt::Alignment alignment READ alignment WRITE setAlignment)
70 Q_PROPERTY(bool readOnly READ isReadOnly WRITE setReadOnly)
71 Q_PROPERTY(ButtonSymbols buttonSymbols READ buttonSymbols WRITE setButtonSymbols)
72 Q_PROPERTY(QString specialValueText READ specialValueText WRITE setSpecialValueText)
73 Q_PROPERTY(QString text READ text)
74 Q_PROPERTY(bool accelerated READ isAccelerated WRITE setAccelerated)
75 Q_PROPERTY(CorrectionMode correctionMode READ correctionMode WRITE setCorrectionMode)
76 Q_PROPERTY(bool acceptableInput READ hasAcceptableInput)
77 Q_PROPERTY(bool keyboardTracking READ keyboardTracking WRITE setKeyboardTracking)
78public:
79 explicit QAbstractSpinBox(QWidget *parent = 0);
80 ~QAbstractSpinBox();
81
82 enum StepEnabledFlag { StepNone = 0x00, StepUpEnabled = 0x01,
83 StepDownEnabled = 0x02 };
84 Q_DECLARE_FLAGS(StepEnabled, StepEnabledFlag)
85
86 enum ButtonSymbols { UpDownArrows, PlusMinus, NoButtons };
87
88 ButtonSymbols buttonSymbols() const;
89 void setButtonSymbols(ButtonSymbols bs);
90
91 enum CorrectionMode { CorrectToPreviousValue, CorrectToNearestValue };
92
93 void setCorrectionMode(CorrectionMode cm);
94 CorrectionMode correctionMode() const;
95
96 bool hasAcceptableInput() const;
97 QString text() const;
98
99 QString specialValueText() const;
100 void setSpecialValueText(const QString &txt);
101
102 bool wrapping() const;
103 void setWrapping(bool w);
104
105 void setReadOnly(bool r);
106 bool isReadOnly() const;
107
108 void setKeyboardTracking(bool kt);
109 bool keyboardTracking() const;
110
111 void setAlignment(Qt::Alignment flag);
112 Qt::Alignment alignment() const;
113
114 void setFrame(bool);
115 bool hasFrame() const;
116
117 void setAccelerated(bool on);
118 bool isAccelerated() const;
119
120 QSize sizeHint() const;
121 QSize minimumSizeHint() const;
122 void interpretText();
123 bool event(QEvent *event);
124
125 QVariant inputMethodQuery(Qt::InputMethodQuery) const;
126
127 virtual QValidator::State validate(QString &input, int &pos) const;
128 virtual void fixup(QString &input) const;
129
130 virtual void stepBy(int steps);
131public Q_SLOTS:
132 void stepUp();
133 void stepDown();
134 void selectAll();
135 virtual void clear();
136protected:
137 void resizeEvent(QResizeEvent *event);
138 void keyPressEvent(QKeyEvent *event);
139 void keyReleaseEvent(QKeyEvent *event);
140#ifndef QT_NO_WHEELEVENT
141 void wheelEvent(QWheelEvent *event);
142#endif
143 void focusInEvent(QFocusEvent *event);
144 void focusOutEvent(QFocusEvent *event);
145 void contextMenuEvent(QContextMenuEvent *event);
146 void changeEvent(QEvent *event);
147 void closeEvent(QCloseEvent *event);
148 void hideEvent(QHideEvent *event);
149 void mousePressEvent(QMouseEvent *event);
150 void mouseReleaseEvent(QMouseEvent *event);
151 void mouseMoveEvent(QMouseEvent *event);
152 void timerEvent(QTimerEvent *event);
153 void paintEvent(QPaintEvent *event);
154 void showEvent(QShowEvent *event);
155 void initStyleOption(QStyleOptionSpinBox *option) const;
156
157 QLineEdit *lineEdit() const;
158 void setLineEdit(QLineEdit *edit);
159
160 virtual StepEnabled stepEnabled() const;
161Q_SIGNALS:
162 void editingFinished();
163protected:
164 QAbstractSpinBox(QAbstractSpinBoxPrivate &dd, QWidget *parent = 0);
165
166private:
167 Q_PRIVATE_SLOT(d_func(), void _q_editorTextChanged(const QString &))
168 Q_PRIVATE_SLOT(d_func(), void _q_editorCursorPositionChanged(int, int))
169
170 Q_DECLARE_PRIVATE(QAbstractSpinBox)
171 Q_DISABLE_COPY(QAbstractSpinBox)
172};
173Q_DECLARE_OPERATORS_FOR_FLAGS(QAbstractSpinBox::StepEnabled)
174
175#endif // QT_NO_SPINBOX
