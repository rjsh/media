42#ifndef QSTACKEDWIDGET_H

45#include <QtGui/qframe.h>

47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Gui)
52
53#ifndef QT_NO_STACKEDWIDGET
54
55class QStackedWidgetPrivate;
56
57class Q_GUI_EXPORT QStackedWidget : public QFrame
58{
59 Q_OBJECT
60
61 Q_PROPERTY(int currentIndex READ currentIndex WRITE setCurrentIndex NOTIFY currentChanged)
62 Q_PROPERTY(int count READ count)
63public:
64 explicit QStackedWidget(QWidget *parent=0);
65 ~QStackedWidget();
66
67 int addWidget(QWidget *w);
68 int insertWidget(int index, QWidget *w);
69 void removeWidget(QWidget *w);
70
71 QWidget *currentWidget() const;
72 int currentIndex() const;
73
74 int indexOf(QWidget *) const;
75 QWidget *widget(int) const;
76 int count() const;
77
78public Q_SLOTS:
79 void setCurrentIndex(int index);
80 void setCurrentWidget(QWidget *w);
81
82Q_SIGNALS:
83 void currentChanged(int);
84 void widgetRemoved(int index);
85
86protected:
87 bool event(QEvent *e);
88
89private:
90 Q_DISABLE_COPY(QStackedWidget)
91 Q_DECLARE_PRIVATE(QStackedWidget)
92};
93
94#endif // QT_NO_STACKEDWIDGET

96QT_END_NAMESPACE
