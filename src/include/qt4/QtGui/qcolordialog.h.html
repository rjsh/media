42#ifndef QCOLORDIALOG_H
43#define QCOLORDIALOG_H
44
45#include <QtGui/qdialog.h>
46
47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Gui)
52
53#ifndef QT_NO_COLORDIALOG
54
55class QColorDialogPrivate;
56
57class Q_GUI_EXPORT QColorDialog : public QDialog
58{
59 Q_OBJECT
60 Q_DECLARE_PRIVATE(QColorDialog)
61 Q_ENUMS(ColorDialogOption)
62 Q_PROPERTY(QColor currentColor READ currentColor WRITE setCurrentColor
63 NOTIFY currentColorChanged)
64 Q_PROPERTY(ColorDialogOptions options READ options WRITE setOptions)
65
66public:
67 enum ColorDialogOption {
68 ShowAlphaChannel = 0x00000001,
69 NoButtons = 0x00000002,
70 DontUseNativeDialog = 0x00000004
71 };
72
73 Q_DECLARE_FLAGS(ColorDialogOptions, ColorDialogOption)
74
75 explicit QColorDialog(QWidget *parent = 0);
76 explicit QColorDialog(const QColor &initial, QWidget *parent = 0);
77 ~QColorDialog();
78
79 void setCurrentColor(const QColor &color);
80 QColor currentColor() const;
81
82 QColor selectedColor() const;
83
84 void setOption(ColorDialogOption option, bool on = true);
85 bool testOption(ColorDialogOption option) const;
86 void setOptions(ColorDialogOptions options);
87 ColorDialogOptions options() const;
88
89#ifdef Q_NO_USING_KEYWORD
90 void open() { QDialog::open(); }
91#else
92 using QDialog::open;
93#endif
94 void open(QObject *receiver, const char *member);
95
96 void setVisible(bool visible);
97
98 // ### Qt 5: merge overloads with title = QString()
99 static QColor getColor(const QColor &initial, QWidget *parent, const QString &title,
100 ColorDialogOptions options = 0);
101 static QColor getColor(const QColor &initial = Qt::white, QWidget *parent = 0);
102
103 // obsolete
104 static QRgb getRgba(QRgb rgba = 0xffffffff, bool *ok = 0, QWidget *parent = 0);
105
106 // ### Qt 5: use QColor in signatures
107 static int customCount();
108 static QRgb customColor(int index);
109 static void setCustomColor(int index, QRgb color);
110 static void setStandardColor(int index, QRgb color);
111
112#ifdef QT3_SUPPORT
113 static QColor getColor(const QColor &init, QWidget *parent, const char *name)
114 { Q_UNUSED(name); return getColor(init, parent); }
115 static QRgb getRgba(QRgb rgba, bool *ok, QWidget *parent, const char *name)
116 { Q_UNUSED(name); return getRgba(rgba, ok, parent); }
117#endif
118
119Q_SIGNALS:
120 void currentColorChanged(const QColor &color);
121 void colorSelected(const QColor &color);
122
123protected:
124 void changeEvent(QEvent *event);
125 void done(int result);
126
127private:
128 Q_DISABLE_COPY(QColorDialog)
129
130 Q_PRIVATE_SLOT(d_func(), void _q_addCustom())
131 Q_PRIVATE_SLOT(d_func(), void _q_newHsv(int h, int s, int v))
132 Q_PRIVATE_SLOT(d_func(), void _q_newColorTypedIn(QRgb rgb))
133 Q_PRIVATE_SLOT(d_func(), void _q_newCustom(int, int))
134 Q_PRIVATE_SLOT(d_func(), void _q_newStandard(int, int))
135#if defined(Q_WS_MAC)
136 Q_PRIVATE_SLOT(d_func(), void _q_macRunNativeAppModalPanel())
137#endif
138
139 friend class QColorShower;
140};
141
142Q_DECLARE_OPERATORS_FOR_FLAGS(QColorDialog::ColorDialogOptions)
143
144#endif // QT_NO_COLORDIALOG
145
146QT_END_NAMESPACE
147
148QT_END_HEADER
149
150#endif // QCOLORDIALOG_H
