46#include <QtGui/qabstractslider.h>

51

54#ifndef QT_NO_DIAL
55
56class QDialPrivate;
57class QStyleOptionSlider;
58
59class Q_GUI_EXPORT QDial: public QAbstractSlider
60{
61 Q_OBJECT
62
63 Q_PROPERTY(bool wrapping READ wrapping WRITE setWrapping)
64 Q_PROPERTY(int notchSize READ notchSize)
65 Q_PROPERTY(qreal notchTarget READ notchTarget WRITE setNotchTarget)
66 Q_PROPERTY(bool notchesVisible READ notchesVisible WRITE setNotchesVisible)
67public:
68 explicit QDial(QWidget *parent = 0);
69
70 ~QDial();
71
72 bool wrapping() const;
73
74 int notchSize() const;
75
76 void setNotchTarget(double target);
77 qreal notchTarget() const;
78 bool notchesVisible() const;
79
80 QSize sizeHint() const;
81 QSize minimumSizeHint() const;
82
83public Q_SLOTS:
84 void setNotchesVisible(bool visible);
85 void setWrapping(bool on);
86
87protected:
88 bool event(QEvent *e);
89 void resizeEvent(QResizeEvent *re);
90 void paintEvent(QPaintEvent *pe);
91
92 void mousePressEvent(QMouseEvent *me);
93 void mouseReleaseEvent(QMouseEvent *me);
94 void mouseMoveEvent(QMouseEvent *me);
95
96 void sliderChange(SliderChange change);
97 void initStyleOption(QStyleOptionSlider *option) const;
98
99#ifdef QT3_SUPPORT
100public:
101 QT3_SUPPORT_CONSTRUCTOR QDial(int minValue, int maxValue, int pageStep, int value,
102 QWidget* parent = 0, const char* name = 0);
103 QT3_SUPPORT_CONSTRUCTOR QDial(QWidget *parent, const char *name);
104
105Q_SIGNALS:
106 QT_MOC_COMPAT void dialPressed();
107 QT_MOC_COMPAT void dialMoved(int value);
108 QT_MOC_COMPAT void dialReleased();
109#endif
110
111private:
112 Q_DECLARE_PRIVATE(QDial)
113 Q_DISABLE_COPY(QDial)
114};
115
116#endif // QT_NO_DIAL
117
118QT_END_NAMESPACE

120QT_END_HEADER
