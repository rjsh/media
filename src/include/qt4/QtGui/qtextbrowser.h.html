42#ifndef QTEXTBROWSER_H
43#define QTEXTBROWSER_H
44
45#include <QtGui/qtextedit.h>
46#include <QtCore/qurl.h>
47
48QT_BEGIN_HEADER
49
50QT_BEGIN_NAMESPACE
51
52QT_MODULE(Gui)
53
54#ifndef QT_NO_TEXTBROWSER
55
56class QTextBrowserPrivate;
57
58class Q_GUI_EXPORT QTextBrowser : public QTextEdit
59{
60 Q_OBJECT
61
62 Q_PROPERTY(QUrl source READ source WRITE setSource)
63 Q_OVERRIDE(bool modified SCRIPTABLE false)
64 Q_OVERRIDE(bool readOnly DESIGNABLE false SCRIPTABLE false)
65 Q_OVERRIDE(bool undoRedoEnabled DESIGNABLE false SCRIPTABLE false)
66 Q_PROPERTY(QStringList searchPaths READ searchPaths WRITE setSearchPaths)
67 Q_PROPERTY(bool openExternalLinks READ openExternalLinks WRITE setOpenExternalLinks)
68 Q_PROPERTY(bool openLinks READ openLinks WRITE setOpenLinks)
69
70public:
71 explicit QTextBrowser(QWidget* parent = 0);
72 virtual ~QTextBrowser();
73
74 QUrl source() const;
75
76 QStringList searchPaths() const;
77 void setSearchPaths(const QStringList &paths);
78
79 virtual QVariant loadResource(int type, const QUrl &name);
80
81 bool isBackwardAvailable() const;
82 bool isForwardAvailable() const;
83 void clearHistory();
84 QString historyTitle(int) const;
85 QUrl historyUrl(int) const;
86 int backwardHistoryCount() const;
87 int forwardHistoryCount() const;
88
89 bool openExternalLinks() const;
90 void setOpenExternalLinks(bool open);
91
92 bool openLinks() const;
93 void setOpenLinks(bool open);
94
95public Q_SLOTS:
96 virtual void setSource(const QUrl &name);
97 virtual void backward();
98 virtual void forward();
99 virtual void home();
100 virtual void reload();
101
102Q_SIGNALS:
103 void backwardAvailable(bool);
104 void forwardAvailable(bool);
105 void historyChanged();
106 void sourceChanged(const QUrl &);
107 void highlighted(const QUrl &);
108 void highlighted(const QString &);
109 void anchorClicked(const QUrl &);
110
111protected:
112 bool event(QEvent *e);
113 virtual void keyPressEvent(QKeyEvent *ev);
114 virtual void mouseMoveEvent(QMouseEvent *ev);
115 virtual void mousePressEvent(QMouseEvent *ev);
116 virtual void mouseReleaseEvent(QMouseEvent *ev);
117 virtual void focusOutEvent(QFocusEvent *ev);
118 virtual bool focusNextPrevChild(bool next);
119 virtual void paintEvent(QPaintEvent *e);
120
121#if defined(QT3_SUPPORT)
122public:
123 QT3_SUPPORT_CONSTRUCTOR QTextBrowser(QWidget *parent, const char *name);
124#endif
125
126private:
127 Q_DISABLE_COPY(QTextBrowser)
128 Q_DECLARE_PRIVATE(QTextBrowser)
129 Q_PRIVATE_SLOT(d_func(), void _q_documentModified())
130 Q_PRIVATE_SLOT(d_func(), void _q_activateAnchor(const QString &))
131 Q_PRIVATE_SLOT(d_func(), void _q_highlightLink(const QString &))
132};
133
134#endif // QT_NO_TEXTBROWSER
135
136QT_END_NAMESPACE
137
138QT_END_HEADER
139
140#endif // QTEXTBROWSER_H
