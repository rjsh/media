45#include <QtGui/qcolor.h>

54class QVariant;
55class QPenPrivate;
56class QBrush;
57class QPen;
58
59#ifndef QT_NO_DATASTREAM
60Q_GUI_EXPORT QDataStream &operator<<(QDataStream &, const QPen &);

62#endif
63
64class Q_GUI_EXPORT QPen
65{
66public:
67 QPen();
68 QPen(Qt::PenStyle);
69 QPen(const QColor &color);
70 QPen(const QBrush &brush, qreal width, Qt::PenStyle s = Qt::SolidLine,
71 Qt::PenCapStyle c = Qt::SquareCap, Qt::PenJoinStyle j = Qt::BevelJoin);
72 QPen(const QPen &pen);
73
74 ~QPen();
75
76 QPen &operator=(const QPen &pen);
77#ifdef Q_COMPILER_RVALUE_REFS
78 inline QPen &operator=(QPen &&other)
79 { qSwap(d, other.d); return *this; }
80#endif
81 inline void swap(QPen &other) { qSwap(d, other.d); }
82
83 Qt::PenStyle style() const;
84 void setStyle(Qt::PenStyle);
85
86 QVector<qreal> dashPattern() const;
87 void setDashPattern(const QVector<qreal> &pattern);
88
89 qreal dashOffset() const;
90 void setDashOffset(qreal doffset);
91
92 qreal miterLimit() const;
93 void setMiterLimit(qreal limit);
94
95 qreal widthF() const;
96 void setWidthF(qreal width);
97
98 int width() const;
99 void setWidth(int width);
100
101 QColor color() const;
102 void setColor(const QColor &color);
103
104 QBrush brush() const;
105 void setBrush(const QBrush &brush);
106
107 bool isSolid() const;
108
109 Qt::PenCapStyle capStyle() const;
110 void setCapStyle(Qt::PenCapStyle pcs);
111
112 Qt::PenJoinStyle joinStyle() const;
113 void setJoinStyle(Qt::PenJoinStyle pcs);
114
115 bool isCosmetic() const;
116 void setCosmetic(bool cosmetic);
117
118 bool operator==(const QPen &p) const;
119 inline bool operator!=(const QPen &p) const { return !(operator==(p)); }
120 operator QVariant() const;
121
122 bool isDetached();
123private:
124 friend Q_GUI_EXPORT QDataStream &operator>>(QDataStream &, QPen &);
125 friend Q_GUI_EXPORT QDataStream &operator<<(QDataStream &, const QPen &);
126
127 void detach();
128 class QPenPrivate *d;
129
130public:
131 typedef QPenPrivate * DataPtr;
132 inline DataPtr &data_ptr() { return d; }
133};
