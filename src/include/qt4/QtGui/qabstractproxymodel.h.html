53#ifndef QT_NO_PROXYMODEL
54
55class QAbstractProxyModelPrivate;
56class QItemSelection;
57
58class Q_GUI_EXPORT QAbstractProxyModel : public QAbstractItemModel
59{
60 Q_OBJECT
61
62public:
63 QAbstractProxyModel(QObject *parent = 0);
64 ~QAbstractProxyModel();
65
66 virtual void setSourceModel(QAbstractItemModel *sourceModel);
67 QAbstractItemModel *sourceModel() const;
68
69 virtual QModelIndex mapToSource(const QModelIndex &proxyIndex) const = 0;
70 virtual QModelIndex mapFromSource(const QModelIndex &sourceIndex) const = 0;
71
72 virtual QItemSelection mapSelectionToSource(const QItemSelection &selection) const;
73 virtual QItemSelection mapSelectionFromSource(const QItemSelection &selection) const;
74
75 bool submit();
76 void revert();
77
78 QVariant data(const QModelIndex &proxyIndex, int role = Qt::DisplayRole) const;
79 QVariant headerData(int section, Qt::Orientation orientation, int role) const;
80 QMap<int, QVariant> itemData(const QModelIndex &index) const;
81 Qt::ItemFlags flags(const QModelIndex &index) const;
82
83 bool setData(const QModelIndex &index, const QVariant &value, int role = Qt::EditRole);
84 bool setItemData(const QModelIndex& index, const QMap<int, QVariant> &roles);
85 bool setHeaderData(int section, Qt::Orientation orientation, const QVariant &value, int role = Qt::EditRole);
86
87 QModelIndex buddy(const QModelIndex &index) const;
88 bool canFetchMore(const QModelIndex &parent) const;
89 void fetchMore(const QModelIndex &parent);
90 void sort(int column, Qt::SortOrder order = Qt::AscendingOrder);
91 QSize span(const QModelIndex &index) const;
92 bool hasChildren(const QModelIndex &parent = QModelIndex()) const;
93
94 QMimeData* mimeData(const QModelIndexList &indexes) const;
95 QStringList mimeTypes() const;
96 Qt::DropActions supportedDropActions() const;
97
98protected:
99 QAbstractProxyModel(QAbstractProxyModelPrivate &, QObject *parent);
100
101private:
102 Q_DECLARE_PRIVATE(QAbstractProxyModel)
103 Q_DISABLE_COPY(QAbstractProxyModel)
104 Q_PRIVATE_SLOT(d_func(), void _q_sourceModelDestroyed())
105};
106
107#endif // QT_NO_PROXYMODEL

109QT_END_NAMESPACE

111QT_END_HEADER
