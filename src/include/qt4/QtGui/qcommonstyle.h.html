76 int pixelMetric(PixelMetric m, const QStyleOption *opt = 0, const QWidget *widget = 0) const;

87 void polish(QPalette &);

97protected:
98 QCommonStyle(QCommonStylePrivate &dd);
99
100private:
101 Q_DECLARE_PRIVATE(QCommonStyle)
102 Q_DISABLE_COPY(QCommonStyle)
103};
