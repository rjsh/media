42#ifndef QCLIPBOARD_H
43#define QCLIPBOARD_H
44
45#include <QtCore/qobject.h>
46
47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Gui)
52
53#ifndef QT_NO_CLIPBOARD
54
55class QMimeSource;
56class QMimeData;
57class QImage;
58class QPixmap;
59
60class QClipboardPrivate;
61
62class Q_GUI_EXPORT QClipboard : public QObject
63{
64 Q_OBJECT
65 Q_DECLARE_PRIVATE(QClipboard)
66private:
67 QClipboard(QObject *parent);
68 ~QClipboard();
69
70public:
71 enum Mode { Clipboard, Selection, FindBuffer, LastMode = FindBuffer };
72
73 void clear(Mode mode = Clipboard);
74
75 bool supportsSelection() const;
76 bool supportsFindBuffer() const;
77
78 bool ownsSelection() const;
79 bool ownsClipboard() const;
80 bool ownsFindBuffer() const;
81
82 QString text(Mode mode = Clipboard) const;
83 QString text(QString& subtype, Mode mode = Clipboard) const;
84 void setText(const QString &, Mode mode = Clipboard);
85
86#ifdef QT3_SUPPORT
87 QT3_SUPPORT QMimeSource *data(Mode mode = Clipboard) const;
88 QT3_SUPPORT void setData(QMimeSource*, Mode mode = Clipboard);
89#endif
90 const QMimeData *mimeData(Mode mode = Clipboard ) const;
91 void setMimeData(QMimeData *data, Mode mode = Clipboard);
92
93 QImage image(Mode mode = Clipboard) const;
94 QPixmap pixmap(Mode mode = Clipboard) const;
95 void setImage(const QImage &, Mode mode = Clipboard);
96 void setPixmap(const QPixmap &, Mode mode = Clipboard);
97
98Q_SIGNALS:
99 void changed(QClipboard::Mode mode);
100 void selectionChanged();
101 void findBufferChanged();
102 void dataChanged();
103private Q_SLOTS:
104 void ownerDestroyed();
105
106protected:
107 void connectNotify(const char *);
108 bool event(QEvent *);
109
110 friend class QApplication;
111 friend class QApplicationPrivate;
112 friend class QBaseApplication;
113 friend class QDragManager;
114 friend class QMimeSource;
115 friend class QPlatformClipboard;
116
117private:
118 Q_DISABLE_COPY(QClipboard)
119
120 bool supportsMode(Mode mode) const;
121 bool ownsMode(Mode mode) const;
122 void emitChanged(Mode mode);
123};
124
125#endif // QT_NO_CLIPBOARD
126
127QT_END_NAMESPACE
128
129QT_END_HEADER
130
131#endif // QCLIPBOARD_H
