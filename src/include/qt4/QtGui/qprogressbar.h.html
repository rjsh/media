42#ifndef QPROGRESSBAR_H
43#define QPROGRESSBAR_H
44
45#include <QtGui/qframe.h>
46
47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Gui)
52
53#ifndef QT_NO_PROGRESSBAR
54
55class QProgressBarPrivate;
56class QStyleOptionProgressBar;
57
58class Q_GUI_EXPORT QProgressBar : public QWidget
59{
60 Q_OBJECT
61 Q_ENUMS(Direction)
62 Q_PROPERTY(int minimum READ minimum WRITE setMinimum)
63 Q_PROPERTY(int maximum READ maximum WRITE setMaximum)
64 Q_PROPERTY(QString text READ text)
65 Q_PROPERTY(int value READ value WRITE setValue NOTIFY valueChanged)
66 Q_PROPERTY(Qt::Alignment alignment READ alignment WRITE setAlignment)
67 Q_PROPERTY(bool textVisible READ isTextVisible WRITE setTextVisible)
68 Q_PROPERTY(Qt::Orientation orientation READ orientation WRITE setOrientation)
69 Q_PROPERTY(bool invertedAppearance READ invertedAppearance WRITE setInvertedAppearance)
70 Q_PROPERTY(Direction textDirection READ textDirection WRITE setTextDirection)
71 Q_PROPERTY(QString format READ format WRITE setFormat)
72
73public:
74 enum Direction { TopToBottom, BottomToTop };
75
76 explicit QProgressBar(QWidget *parent = 0);
77
78 int minimum() const;
79 int maximum() const;
80
81 int value() const;
82
83 virtual QString text() const;
84 void setTextVisible(bool visible);
85 bool isTextVisible() const;
86
87 Qt::Alignment alignment() const;
88 void setAlignment(Qt::Alignment alignment);
89
90 QSize sizeHint() const;
91 QSize minimumSizeHint() const;
92
93 Qt::Orientation orientation() const;
94
95 void setInvertedAppearance(bool invert);
96 bool invertedAppearance(); //### Qt5 make const
97 bool invertedAppearance() const { return const_cast<QProgressBar *>(this)->invertedAppearance(); }
98 void setTextDirection(QProgressBar::Direction textDirection);
99 QProgressBar::Direction textDirection(); //### Qt5 make const
100 QProgressBar::Direction textDirection() const { return const_cast<QProgressBar *>(this)->textDirection(); }
101
102 void setFormat(const QString &format);
103 QString format() const;
104
105public Q_SLOTS:
106 void reset();
107 void setRange(int minimum, int maximum);
108 void setMinimum(int minimum);
109 void setMaximum(int maximum);
110 void setValue(int value);
111 void setOrientation(Qt::Orientation);
112
113Q_SIGNALS:
114 void valueChanged(int value);
115
116protected:
117 bool event(QEvent *e);
118 void paintEvent(QPaintEvent *);
119 void initStyleOption(QStyleOptionProgressBar *option) const;
120
121private:
122 Q_DECLARE_PRIVATE(QProgressBar)
123 Q_DISABLE_COPY(QProgressBar)
124};
125
126#endif // QT_NO_PROGRESSBAR
127
128QT_END_NAMESPACE
129
130QT_END_HEADER
131
132#endif // QPROGRESSBAR_H
