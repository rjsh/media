42#ifndef QGROUPBOX_H
43#define QGROUPBOX_H
44
45#include <QtGui/qframe.h>
46
47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Gui)
52
53#ifndef QT_NO_GROUPBOX
54
55class QGroupBoxPrivate;
56class QStyleOptionGroupBox;
57class Q_GUI_EXPORT QGroupBox : public QWidget
58{
59 Q_OBJECT
60
61 Q_PROPERTY(QString title READ title WRITE setTitle)
62 Q_PROPERTY(Qt::Alignment alignment READ alignment WRITE setAlignment)
63 Q_PROPERTY(bool flat READ isFlat WRITE setFlat)
64 Q_PROPERTY(bool checkable READ isCheckable WRITE setCheckable)
65 Q_PROPERTY(bool checked READ isChecked WRITE setChecked DESIGNABLE isCheckable NOTIFY toggled USER true)
66public:
67 explicit QGroupBox(QWidget* parent=0);
68 explicit QGroupBox(const QString &title, QWidget* parent=0);
69 ~QGroupBox();
70
71 QString title() const;
72 void setTitle(const QString &title);
73
74 Qt::Alignment alignment() const;
75 void setAlignment(int alignment);
76
77 QSize minimumSizeHint() const;
78
79 bool isFlat() const;
80 void setFlat(bool flat);
81 bool isCheckable() const;
82 void setCheckable(bool checkable);
83 bool isChecked() const;
84
85public Q_SLOTS:
86 void setChecked(bool checked);
87
88Q_SIGNALS:
89 void clicked(bool checked = false);
90 void toggled(bool);
91
92protected:
93 bool event(QEvent *event);
94 void childEvent(QChildEvent *event);
95 void resizeEvent(QResizeEvent *event);
96 void paintEvent(QPaintEvent *event);
97 void focusInEvent(QFocusEvent *event);
98 void changeEvent(QEvent *event);
99 void mousePressEvent(QMouseEvent *event);
100 void mouseMoveEvent(QMouseEvent *event);
101 void mouseReleaseEvent(QMouseEvent *event);
102 void initStyleOption(QStyleOptionGroupBox *option) const;
103
104#ifdef QT3_SUPPORT
105public:
106 QT3_SUPPORT_CONSTRUCTOR QGroupBox(QWidget* parent, const char* name);
107 QT3_SUPPORT_CONSTRUCTOR QGroupBox(const QString &title, QWidget* parent, const char* name);
108#endif
109
110private:
111 Q_DISABLE_COPY(QGroupBox)
112 Q_DECLARE_PRIVATE(QGroupBox)
113 Q_PRIVATE_SLOT(d_func(), void _q_setChildrenEnabled(bool b))
114};
115
116#endif // QT_NO_GROUPBOX
117
118QT_END_NAMESPACE
119
120QT_END_HEADER
121
122#endif // QGROUPBOX_H
