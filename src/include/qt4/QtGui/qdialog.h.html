42#ifndef QDIALOG_H
43#define QDIALOG_H
44
45#include <QtGui/qwidget.h>
46
47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Gui)
52
53class QPushButton;
54class QDialogPrivate;
55
56class Q_GUI_EXPORT QDialog : public QWidget
57{
58 Q_OBJECT
59 friend class QPushButton;
60
61 Q_PROPERTY(bool sizeGripEnabled READ isSizeGripEnabled WRITE setSizeGripEnabled)
62 Q_PROPERTY(bool modal READ isModal WRITE setModal)
63
64public:
65 explicit QDialog(QWidget *parent = 0, Qt::WindowFlags f = 0);
66#ifdef QT3_SUPPORT
67 QT3_SUPPORT_CONSTRUCTOR QDialog(QWidget *parent, const char *name, bool modal = false,
68 Qt::WindowFlags f = 0);
69#endif
70 ~QDialog();
71
72 enum DialogCode { Rejected, Accepted };
73
74 int result() const;
75
76 void setVisible(bool visible);
77
78 void setOrientation(Qt::Orientation orientation);
79 Qt::Orientation orientation() const;
80
81 void setExtension(QWidget* extension);
82 QWidget* extension() const;
83
84 QSize sizeHint() const;
85 QSize minimumSizeHint() const;
86
87 void setSizeGripEnabled(bool);
88 bool isSizeGripEnabled() const;
89
90 void setModal(bool modal);
91 void setResult(int r);
92
93Q_SIGNALS:
94 void finished(int result);
95 void accepted();
96 void rejected();
97
98public Q_SLOTS:
99 void open();
100 int exec();
101 virtual void done(int);
102 virtual void accept();
103 virtual void reject();
104
105 void showExtension(bool);
106
107protected:
108 QDialog(QDialogPrivate &, QWidget *parent, Qt::WindowFlags f = 0);
109
110#if defined(Q_WS_WINCE) || defined(Q_OS_SYMBIAN)
111 bool event(QEvent *e);
112#endif
113 void keyPressEvent(QKeyEvent *);
114 void closeEvent(QCloseEvent *);
115 void showEvent(QShowEvent *);
116 void resizeEvent(QResizeEvent *);
117#ifndef QT_NO_CONTEXTMENU
118 void contextMenuEvent(QContextMenuEvent *);
119#endif
120 bool eventFilter(QObject *, QEvent *);
121 void adjustPosition(QWidget*);
122private:
123 Q_DECLARE_PRIVATE(QDialog)
124 Q_DISABLE_COPY(QDialog)
125
126#if defined(Q_OS_SYMBIAN)
127 bool symbianAdjustedPosition();
128#endif
129
130
131#ifdef Q_WS_WINCE_WM
132 Q_PRIVATE_SLOT(d_func(), void _q_doneAction())
133#endif
134};
135
136QT_END_NAMESPACE
137
138QT_END_HEADER
139
140#endif // QDIALOG_H
