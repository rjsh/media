235inline int QHeaderView::logicalIndexAt(int ax, int ay) const
236{ return orientation() == Qt::Horizontal ? logicalIndexAt(ax) : logicalIndexAt(ay); }
237inline int QHeaderView::logicalIndexAt(const QPoint &apos) const
238{ return logicalIndexAt(apos.x(), apos.y()); }
239inline void QHeaderView::hideSection(int alogicalIndex)
240{ setSectionHidden(alogicalIndex, true); }
241inline void QHeaderView::showSection(int alogicalIndex)
242{ setSectionHidden(alogicalIndex, false); }
