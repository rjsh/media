42#ifndef QTREEWIDGETITEMITERATOR_H
43#define QTREEWIDGETITEMITERATOR_H
44
45#include <QtCore/qglobal.h>
46#include <QtCore/qscopedpointer.h>
47
48QT_BEGIN_HEADER
49
50QT_BEGIN_NAMESPACE
51
52QT_MODULE(Gui)
53
54#ifndef QT_NO_TREEWIDGET
55
56class QTreeWidget;
57class QTreeWidgetItem;
58class QTreeModel;
59
60class QTreeWidgetItemIteratorPrivate;
61class Q_GUI_EXPORT QTreeWidgetItemIterator
62{
63 friend class QTreeModel;
64
65public:
66 enum IteratorFlag {
67 All = 0x00000000,
68 Hidden = 0x00000001,
69 NotHidden = 0x00000002,
70 Selected = 0x00000004,
71 Unselected = 0x00000008,
72 Selectable = 0x00000010,
73 NotSelectable = 0x00000020,
74 DragEnabled = 0x00000040,
75 DragDisabled = 0x00000080,
76 DropEnabled = 0x00000100,
77 DropDisabled = 0x00000200,
78 HasChildren = 0x00000400,
79 NoChildren = 0x00000800,
80 Checked = 0x00001000,
81 NotChecked = 0x00002000,
82 Enabled = 0x00004000,
83 Disabled = 0x00008000,
84 Editable = 0x00010000,
85 NotEditable = 0x00020000,
86 UserFlag = 0x01000000 // The first flag that can be used by the user.
87 };
88 Q_DECLARE_FLAGS(IteratorFlags, IteratorFlag)
89
90 QTreeWidgetItemIterator(const QTreeWidgetItemIterator &it);
91 explicit QTreeWidgetItemIterator(QTreeWidget *widget, IteratorFlags flags = All);
92 explicit QTreeWidgetItemIterator(QTreeWidgetItem *item, IteratorFlags flags = All);
93 ~QTreeWidgetItemIterator();
94
95 QTreeWidgetItemIterator &operator=(const QTreeWidgetItemIterator &it);
96
97 QTreeWidgetItemIterator &operator++();
98 inline const QTreeWidgetItemIterator operator++(int);
99 inline QTreeWidgetItemIterator &operator+=(int n);
100
101 QTreeWidgetItemIterator &operator--();
102 inline const QTreeWidgetItemIterator operator--(int);
103 inline QTreeWidgetItemIterator &operator-=(int n);
104
105 inline QTreeWidgetItem *operator*() const;
106
107private:
108 bool matchesFlags(const QTreeWidgetItem *item) const;
109 QScopedPointer<QTreeWidgetItemIteratorPrivate> d_ptr;
110 QTreeWidgetItem *current;
111 IteratorFlags flags;
112 Q_DECLARE_PRIVATE(QTreeWidgetItemIterator)
113};
114
115inline const QTreeWidgetItemIterator QTreeWidgetItemIterator::operator++(int)
116{
117 QTreeWidgetItemIterator it = *this;
118 ++(*this);
119 return it;
120}
121
122inline const QTreeWidgetItemIterator QTreeWidgetItemIterator::operator--(int)
123{
124 QTreeWidgetItemIterator it = *this;
125 --(*this);
126 return it;
127}
128
129inline QTreeWidgetItemIterator &QTreeWidgetItemIterator::operator+=(int n)
130{
131 if (n < 0)
132 return (*this) -= (-n);
133 while (current && n--)
134 ++(*this);
135 return *this;
136}
137
138inline QTreeWidgetItemIterator &QTreeWidgetItemIterator::operator-=(int n)
139{
140 if (n < 0)
141 return (*this) += (-n);
142 while (current && n--)
143 --(*this);
144 return *this;
145}
146
147inline QTreeWidgetItem *QTreeWidgetItemIterator::operator*() const
148{
149 return current;
150}
151
152Q_DECLARE_OPERATORS_FOR_FLAGS(QTreeWidgetItemIterator::IteratorFlags)
153
