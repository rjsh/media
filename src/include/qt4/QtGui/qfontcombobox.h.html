42#ifndef QFONTCOMBOBOX_H
43#define QFONTCOMBOBOX_H
44
45#include <QtGui/qcombobox.h>
46#include <QtGui/qfontdatabase.h>
47
48#ifndef QT_NO_FONTCOMBOBOX
49
50QT_BEGIN_HEADER
51
52QT_BEGIN_NAMESPACE
53
54QT_MODULE(Gui)
55
56class QFontComboBoxPrivate;
57
58class Q_GUI_EXPORT QFontComboBox : public QComboBox
59{
60 Q_OBJECT
61 Q_FLAGS(FontFilters)
62 Q_PROPERTY(QFontDatabase::WritingSystem writingSystem READ writingSystem WRITE setWritingSystem)
63 Q_PROPERTY(FontFilters fontFilters READ fontFilters WRITE setFontFilters)
64 Q_PROPERTY(QFont currentFont READ currentFont WRITE setCurrentFont NOTIFY currentFontChanged)
65 Q_ENUMS(FontSelection)
66
67public:
68 explicit QFontComboBox(QWidget *parent = 0);
69 ~QFontComboBox();
70
71 void setWritingSystem(QFontDatabase::WritingSystem);
72 QFontDatabase::WritingSystem writingSystem() const;
73
74 enum FontFilter {
75 AllFonts = 0,
76 ScalableFonts = 0x1,
77 NonScalableFonts = 0x2,
78 MonospacedFonts = 0x4,
79 ProportionalFonts = 0x8
80 };
81 Q_DECLARE_FLAGS(FontFilters, FontFilter)
82
83 void setFontFilters(FontFilters filters);
84 FontFilters fontFilters() const;
85
86 QFont currentFont() const;
87 QSize sizeHint() const;
88
89public Q_SLOTS:
90 void setCurrentFont(const QFont &f);
91
92Q_SIGNALS:
93 void currentFontChanged(const QFont &f);
94
95protected:
96 bool event(QEvent *e);
97
98private:
99 Q_DISABLE_COPY(QFontComboBox)
100 Q_DECLARE_PRIVATE(QFontComboBox)
101 Q_PRIVATE_SLOT(d_func(), void _q_currentChanged(const QString &))
102 Q_PRIVATE_SLOT(d_func(), void _q_updateModel())
103};
104
105Q_DECLARE_OPERATORS_FOR_FLAGS(QFontComboBox::FontFilters)
106
107QT_END_NAMESPACE
108
109QT_END_HEADER
110
111#endif // QT_NO_FONTCOMBOBOX
112#endif
