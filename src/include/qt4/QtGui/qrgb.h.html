42#ifndef QRGB_H
43#define QRGB_H
44
45#include <QtCore/qglobal.h>
46
47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Gui)
52
53typedef unsigned int QRgb; // RGB triplet
54
55const QRgb RGB_MASK = 0x00ffffff; // masks RGB values
56
57Q_GUI_EXPORT_INLINE int qRed(QRgb rgb) // get red part of RGB
58{ return ((rgb >> 16) & 0xff); }
59
60Q_GUI_EXPORT_INLINE int qGreen(QRgb rgb) // get green part of RGB
61{ return ((rgb >> 8) & 0xff); }
62
63Q_GUI_EXPORT_INLINE int qBlue(QRgb rgb) // get blue part of RGB
64{ return (rgb & 0xff); }
65
66Q_GUI_EXPORT_INLINE int qAlpha(QRgb rgb) // get alpha part of RGBA
67{ return rgb >> 24; }
68
69Q_GUI_EXPORT_INLINE QRgb qRgb(int r, int g, int b)// set RGB value
70{ return (0xffu << 24) | ((r & 0xff) << 16) | ((g & 0xff) << 8) | (b & 0xff); }
71
72Q_GUI_EXPORT_INLINE QRgb qRgba(int r, int g, int b, int a)// set RGBA value
73{ return ((a & 0xff) << 24) | ((r & 0xff) << 16) | ((g & 0xff) << 8) | (b & 0xff); }
74
75Q_GUI_EXPORT_INLINE int qGray(int r, int g, int b)// convert R,G,B to gray 0..255
76{ return (r*11+g*16+b*5)/32; }
77
78Q_GUI_EXPORT_INLINE int qGray(QRgb rgb) // convert RGB to gray 0..255
79{ return qGray(qRed(rgb), qGreen(rgb), qBlue(rgb)); }
80
81Q_GUI_EXPORT_INLINE bool qIsGray(QRgb rgb)
82{ return qRed(rgb) == qGreen(rgb) && qRed(rgb) == qBlue(rgb); }
83
84QT_END_NAMESPACE
85
86QT_END_HEADER
87
88#endif // QRGB_H
