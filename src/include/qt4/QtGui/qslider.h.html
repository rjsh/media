42#ifndef QSLIDER_H
43#define QSLIDER_H
44
45#include <QtGui/qabstractslider.h>
46
47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Gui)
52
53#ifndef QT_NO_SLIDER
54
55class QSliderPrivate;
56class QStyleOptionSlider;
57class Q_GUI_EXPORT QSlider : public QAbstractSlider
58{
59 Q_OBJECT
60
61 Q_ENUMS(TickPosition)
62 Q_PROPERTY(TickPosition tickPosition READ tickPosition WRITE setTickPosition)
63 Q_PROPERTY(int tickInterval READ tickInterval WRITE setTickInterval)
64
65public:
66 enum TickPosition {
67 NoTicks = 0,
68 TicksAbove = 1,
69 TicksLeft = TicksAbove,
70 TicksBelow = 2,
71 TicksRight = TicksBelow,
72 TicksBothSides = 3
73
74#if defined(QT3_SUPPORT) && !defined(Q_MOC_RUN)
75 ,NoMarks = NoTicks,
76 Above = TicksAbove,
77 Left = TicksAbove,
78 Below = TicksBelow,
79 Right = TicksRight,
80 Both = TicksBothSides
81#endif
82 };
83
84 explicit QSlider(QWidget *parent = 0);
85 explicit QSlider(Qt::Orientation orientation, QWidget *parent = 0);
86
87 ~QSlider();
88
89 QSize sizeHint() const;
90 QSize minimumSizeHint() const;
91
92 void setTickPosition(TickPosition position);
93 TickPosition tickPosition() const;
94
95 void setTickInterval(int ti);
96 int tickInterval() const;
97
98 bool event(QEvent *event);
99
100protected:
101 void paintEvent(QPaintEvent *ev);
102 void mousePressEvent(QMouseEvent *ev);
103 void mouseReleaseEvent(QMouseEvent *ev);
104 void mouseMoveEvent(QMouseEvent *ev);
105 void initStyleOption(QStyleOptionSlider *option) const;
106
107#ifdef QT3_SUPPORT
108public:
109 QT3_SUPPORT_CONSTRUCTOR QSlider(QWidget *parent, const char *name);
110 QT3_SUPPORT_CONSTRUCTOR QSlider(Qt::Orientation, QWidget *parent, const char *name);
111 QT3_SUPPORT_CONSTRUCTOR QSlider(int minValue, int maxValue, int pageStep, int value,
112 Qt::Orientation orientation,
113 QWidget *parent = 0, const char *name = 0);
114 inline QT3_SUPPORT void setTickmarks(TickPosition position) { setTickPosition(position); }
115 inline QT3_SUPPORT TickPosition tickmarks() const { return tickPosition(); }
116public Q_SLOTS:
117 inline QT_MOC_COMPAT void addStep() { triggerAction(SliderSingleStepAdd); }
118 inline QT_MOC_COMPAT void subtractStep() { triggerAction(SliderSingleStepSub); }
119#endif
120
121private:
122 friend Q_GUI_EXPORT QStyleOptionSlider qt_qsliderStyleOption(QSlider *slider);
123
124 Q_DISABLE_COPY(QSlider)
125 Q_DECLARE_PRIVATE(QSlider)
126};
127
128#endif // QT_NO_SLIDER
129
130QT_END_NAMESPACE
131
132QT_END_HEADER
133
134#endif // QSLIDER_H
