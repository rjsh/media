42#ifndef QPROGRESSDIALOG_H
43#define QPROGRESSDIALOG_H
44
45#include <QtGui/qdialog.h>
46
47QT_BEGIN_HEADER
48
49QT_BEGIN_NAMESPACE
50
51QT_MODULE(Gui)
52
53#ifndef QT_NO_PROGRESSDIALOG
54
55class QPushButton;
56class QLabel;
57class QProgressBar;
58class QTimer;
59class QProgressDialogPrivate;
60
61class Q_GUI_EXPORT QProgressDialog : public QDialog
62{
63 Q_OBJECT
64 Q_DECLARE_PRIVATE(QProgressDialog)
65 Q_PROPERTY(bool wasCanceled READ wasCanceled)
66 Q_PROPERTY(int minimum READ minimum WRITE setMinimum)
67 Q_PROPERTY(int maximum READ maximum WRITE setMaximum)
68 Q_PROPERTY(int value READ value WRITE setValue)
69 Q_PROPERTY(bool autoReset READ autoReset WRITE setAutoReset)
70 Q_PROPERTY(bool autoClose READ autoClose WRITE setAutoClose)
71 Q_PROPERTY(int minimumDuration READ minimumDuration WRITE setMinimumDuration)
72 Q_PROPERTY(QString labelText READ labelText WRITE setLabelText)
73
74public:
75 explicit QProgressDialog(QWidget *parent = 0, Qt::WindowFlags flags = 0);
76 QProgressDialog(const QString &labelText, const QString &cancelButtonText,
77 int minimum, int maximum, QWidget *parent = 0, Qt::WindowFlags flags = 0);
78 ~QProgressDialog();
79
80 void setLabel(QLabel *label);
81 void setCancelButton(QPushButton *button);
82 void setBar(QProgressBar *bar);
83
84 bool wasCanceled() const;
85
86 int minimum() const;
87 int maximum() const;
88
89 int value() const;
90
91 QSize sizeHint() const;
92
93 QString labelText() const;
94 int minimumDuration() const;
95
96 void setAutoReset(bool reset);
97 bool autoReset() const;
98 void setAutoClose(bool close);
99 bool autoClose() const;
100
101#ifdef Q_NO_USING_KEYWORD
102#ifndef Q_QDOC
103 void open() { QDialog::open(); }
104#endif
105#else
106 using QDialog::open;
107#endif
108 void open(QObject *receiver, const char *member);
109
110public Q_SLOTS:
111 void cancel();
112 void reset();
113 void setMaximum(int maximum);
114 void setMinimum(int minimum);
115 void setRange(int minimum, int maximum);
116 void setValue(int progress);
117 void setLabelText(const QString &text);
118 void setCancelButtonText(const QString &text);
119 void setMinimumDuration(int ms);
120
121Q_SIGNALS:
122 void canceled();
123
124protected:
125 void resizeEvent(QResizeEvent *event);
126 void closeEvent(QCloseEvent *event);
127 void changeEvent(QEvent *event);
128 void showEvent(QShowEvent *event);
129
130protected Q_SLOTS:
131 void forceShow();
132
133private:
134 Q_DISABLE_COPY(QProgressDialog)
135
136 Q_PRIVATE_SLOT(d_func(), void _q_disconnectOnClose())
137};
138
139#endif // QT_NO_PROGRESSDIALOG
140
141QT_END_NAMESPACE
142
143QT_END_HEADER
144
145#endif // QPROGRESSDIALOG_H
