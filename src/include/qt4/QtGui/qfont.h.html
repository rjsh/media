347
348inline bool QFont::bold() const
349{ return weight() > Normal; }
350
351
352inline void QFont::setBold(bool enable)
353{ setWeight(enable ? Bold : Normal); }
354
355inline bool QFont::italic() const
356{
357 return (style() != StyleNormal);
358}
359
360inline void QFont::setItalic(bool b) {
361 setStyle(b ? StyleItalic : StyleNormal);
362}
