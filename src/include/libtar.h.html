38/* our version of the tar header structure */
39struct tar_header
40{
41 char name[100];
42 char mode[8];
43 char uid[8];
44 char gid[8];
45 char size[12];
46 char mtime[12];
47 char chksum[8];
48 char typeflag;
49 char linkname[100];
50 char magic[6];
51 char version[2];
52 char uname[32];

54 char devmajor[8];
55 char devminor[8];

60};

162/* determine file type */
163#define TH_ISREG(t) ((t)->th_buf.typeflag == REGTYPE \
164 || (t)->th_buf.typeflag == AREGTYPE \
165 || (t)->th_buf.typeflag == CONTTYPE \
166 || (S_ISREG((mode_t)oct_to_int((t)->th_buf.mode)) \
167 && (t)->th_buf.typeflag != LNKTYPE))
168#define TH_ISLNK(t) ((t)->th_buf.typeflag == LNKTYPE)
169#define TH_ISSYM(t) ((t)->th_buf.typeflag == SYMTYPE \
170 || S_ISLNK((mode_t)oct_to_int((t)->th_buf.mode)))
171#define TH_ISCHR(t) ((t)->th_buf.typeflag == CHRTYPE \
172 || S_ISCHR((mode_t)oct_to_int((t)->th_buf.mode)))
173#define TH_ISBLK(t) ((t)->th_buf.typeflag == BLKTYPE \
174 || S_ISBLK((mode_t)oct_to_int((t)->th_buf.mode)))
175#define TH_ISDIR(t) ((t)->th_buf.typeflag == DIRTYPE \
176 || S_ISDIR((mode_t)oct_to_int((t)->th_buf.mode)) \
177 || ((t)->th_buf.typeflag == AREGTYPE \
178 && ((t)->th_buf.name[strlen((t)->th_buf.name) - 1] == '/')))
179#define TH_ISFIFO(t) ((t)->th_buf.typeflag == FIFOTYPE \
180 || S_ISFIFO((mode_t)oct_to_int((t)->th_buf.mode)))
181#define TH_ISLONGNAME(t) ((t)->th_buf.typeflag == GNU_LONGNAME_TYPE)
182#define TH_ISLONGLINK(t) ((t)->th_buf.typeflag == GNU_LONGLINK_TYPE)
183
184/* decode tar header info */
185#define th_get_crc(t) oct_to_int((t)->th_buf.chksum)
186/* We cast from int (what oct_to_int() returns) to
187 unsigned int, to avoid unwieldy sign extensions
188 from occurring on systems where size_t is bigger than int,
189 since th_get_size() is often stored into a size_t. */
190#define th_get_size(t) ((unsigned int)oct_to_int((t)->th_buf.size))
191#define th_get_mtime(t) oct_to_int((t)->th_buf.mtime)
192#define th_get_devmajor(t) oct_to_int((t)->th_buf.devmajor)
193#define th_get_devminor(t) oct_to_int((t)->th_buf.devminor)
194#define th_get_linkname(t) ((t)->th_buf.gnu_longlink \
195 ? (t)->th_buf.gnu_longlink \
196 : (t)->th_buf.linkname)
197char *th_get_pathname(TAR *t);
198mode_t th_get_mode(TAR *t);
199uid_t th_get_uid(TAR *t);
200gid_t th_get_gid(TAR *t);
