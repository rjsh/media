48//***************************************************************************
49/// @brief MediaInfoList
50/// @version 0.7
51//***************************************************************************
52
53class MEDIAINFO_EXP MediaInfoList
54{
55public :
56 //Class
57 /// @brief Constructor
58 /// @param Count_Init optimization information: How many files do you plan to handle?
59 MediaInfoList (size_t Count_Init=64);
60 ~MediaInfoList ();
61
62 //Files
63 /// Open one or more files and collect information about them (technical information and tags)
64 /// @brief Open files
65 /// @param File Full name of file(s) to open \n
66 /// or Full name of folder(s) to open \n
67 /// @param Options: FileOption_Recursive = Recursive mode for folders \n
68 /// FileOption_Close = Close all already opened files before
69 /// @return Number of files successfuly added
70 size_t Open (const String &File, const fileoptions_t Options=FileOption_Nothing);
71 /// Open a stream and collect information about it (technical information and tags)
72 /// @brief Open a stream (Init)
73 /// @param File_Size Estimated file size
74 /// @param File_Offset Offset of the file (if we don't have the beginning of the file)
75 /// @retval 0 failed
76 /// @retval 1 succeed
77 size_t Open_Buffer_Init (ZenLib::int64u File_Size=(ZenLib::int64u)-1, ZenLib::int64u File_Offset=0);
78 /// Open a stream and collect information about it (technical information and tags)
79 /// @brief Open a stream (Continue)
80 /// @param FilePos File position
81 /// @param Buffer pointer to the stream
82 /// @param Buffer_Size Count of bytes to read
83 /// @retval 0 failed
84 /// @retval 1 succeed
85 size_t Open_Buffer_Continue (size_t FilePos, const ZenLib::int8u* Buffer, size_t Buffer_Size);
86 /// Open a stream and collect information about it (technical information and tags)
87 /// @brief Open a stream (Get the needed file Offset)
88 /// @param FilePos File position
89 /// @return the needed offset of the file \n
90 /// File size if no more bytes are needed
91 ZenLib::int64u Open_Buffer_Continue_GoTo_Get (size_t FilePos);
92 /// Open a stream and collect information about it (technical information and tags)
93 /// @brief Open a stream (Finalize)
94 /// @param FilePos File position
95 /// @retval 0 failed
96 /// @retval 1 succeed
97 size_t Open_Buffer_Finalize (size_t FilePos);
98 /// Save the file opened before with Open() (modifications of tags)
99 /// @brief Save the file
100 /// @param FilePos File position \n
101 /// (you can know the position in searching the filename with MediaInfoList::Get(FilePos, 0, 0, "CompleteName") )
102 /// @retval 0 failed
103 /// @retval 1 succeed
104 size_t Save (size_t FilePos);
105 /// (NOT IMPLEMENTED YET) Save all files opened before with Open() (modifications of tags)
106 /// @brief (NOT IMPLEMENTED YET) Save all files
107 /// @param FilePos File position \n
108 /// (you can know the position in searching the filename with MediaInfoList::Get(FilePos, 0, 0, "CompleteName") )
109 /// @retval Count of files saved
110 void Close (size_t FilePos=(size_t)-1);
111
112 /// Get all details about a file in one string
113 /// @brief Get all details about a file
114 /// @param FilePos File position \n
115 /// (you can know the position in searching the filename with MediaInfoList::Get(FilePos, 0, 0, "CompleteName") )
116 /// @param Reserved Deprecated, do not use it anymore
117 /// @pre You can change default presentation with Inform_Set()
118 /// @return Text with information about the file
119 String Inform (size_t FilePos=(size_t)-1, size_t Reserved=0);
120
121 //Get
122 /// Get a piece of information about a file (parameter is an integer)
123 /// @brief Get a piece of information about a file (parameter is an integer)
124 /// @param FilePos File position \n
125 /// (you can know the position in searching the filename with MediaInfoList::Get(FilePos, 0, 0, "CompleteName") )
126 /// @param StreamKind Kind of stream (general, video, audio...)
127 /// @param StreamNumber Stream number in Kind of stream (first, second...)
128 /// @param Parameter Parameter you are looking for in the stream (Codec, width, bitrate...), in integer format (first parameter, second parameter...)
129 /// @param KindOfInfo Kind of information you want about the parameter (the text, the measure, the help...)
130 /// @return a string about information you search \n
131 /// an empty string if there is a problem
132 String Get (size_t FilePos, stream_t StreamKind, size_t StreamNumber, size_t Parameter, info_t KindOfInfo=Info_Text); //Get info, FilePos=File position, StreamKind=General video audio text chapter, StreamNumber=stream number, PosInStream=parameter you want, KindOfInfo=name, text, measure, options, name (language), measure (language), info, how to
133 /// Get a piece of information about a file (parameter is a string)
134 /// @brief Get a piece of information about a file (parameter is a string)
135 /// @param FilePos File position \n
136 /// (you can know the position in searching the filename with MediaInfoList::Get(FilePos, 0, 0, "CompleteName") )
137 /// @param StreamKind Kind of stream (general, video, audio...)
138 /// @param StreamNumber Stream number in Kind of stream (first, second...)
139 /// @param Parameter Parameter you are looking for in the stream (Codec, width, bitrate...), in string format ("Codec", "Width"...) \n
140 /// See MediaInfo::Option("Info_Parameters") to have the full list
141 /// @param KindOfInfo Kind of information you want about the parameter (the text, the measure, the help...)
142 /// @param KindOfSearch Where to look for the parameter
143 /// @return a string about information you search \n
144 /// an empty string if there is a problem
145 String Get (size_t FilePos, stream_t StreamKind, size_t StreamNumber, const String &Parameter, info_t KindOfInfo=Info_Text, info_t KindOfSearch=Info_Name); //Get info, FilePos=File position, StreamKind=General video audio text chapter, StreamNumber=stream number, PosInStream=parameter you want, KindOfInfo=name text measure options name(language) measure(language) information how to, KindOfSearch=which Kind Of information Parameter must be searched?
146
147 //Set
148 /// (NOT IMPLEMENTED YET) Set a piece of information about a file (parameter is an int)
149 /// @brief (NOT IMPLEMENTED YET) Set a piece of information about a file (parameter is an int)
150 /// @warning Not yet implemented, do not use it
151 /// @param ToSet Piece of information
152 /// @param FilePos File position \n
153 /// (you can know the position in searching the filename with MediaInfoList::Get(FilePos, 0, 0, "CompleteName") )
154 /// @param StreamKind Kind of stream (general, video, audio...)
155 /// @param StreamNumber Stream number in Kind of stream (first, second...)
156 /// @param Parameter Parameter you are looking for in the stream (Codec, width, bitrate...), in integer format (first parameter, second parameter...)
157 /// @param OldValue The old value of the parameter \n if OldValue is empty and ToSet is filled: tag is added \n if OldValue is filled and ToSet is filled: tag is replaced \n if OldValue is filled and ToSet is empty: tag is deleted
158 /// @retval >0 succeed
159 /// @retval 0 failed
160 size_t Set (const String &ToSet, size_t FilePos, stream_t StreamKind, size_t StreamNumber, size_t Parameter, const String &OldValue=String()); //Get info, FilePos=File position, StreamKind=General video audio text chapter, StreamNumber=stream number, PosInStream=parameter you want, KindOfInfo=name, text, measure, options name(language) measure(language) information how to
161 /// @brief (NOT IMPLEMENTED YET) Get information about a file (parameter is a string)
162 /// @warning Not yet implemented, do not use it
163 /// @param ToSet Piece of information
164 /// @param FilePos File position \n
165 /// (you can know the position in searching the filename with MediaInfoList::Get(FilePos, 0, 0, "CompleteName") )
166 /// @param StreamKind Kind of stream (general, video, audio...)
167 /// @param StreamNumber Stream number in Kind of stream (first, second...)
168 /// @param Parameter Parameter you are looking for in the stream (Codec, width, bitrate...), in string format ("Codec", "Width"...) \n
169 /// See Option("Info_Parameters") to have the full list
170 /// @param OldValue The old value of the parameter \n if OldValue is empty and ToSet is filled: tag is added \n if OldValue is filled and ToSet is filled: tag is replaced \n if OldValue is filled and ToSet is empty: tag is deleted
171 /// @retval >0 succeed
172 /// @retval 0 failed
173 size_t Set (const String &ToSet, size_t FilePos, stream_t StreamKind, size_t StreamNumber, const String &Parameter, const String &OldValue=String()); //Get info, FilePos=File position, StreamKind=General video audio text chapter, StreamNumber=stream number, PosInStream=parameter you want, KindOfInfo=name text measure options name (language) measure (language) information how to, KindOfSearch=which Kind Of information Parameter must be searched?
174
175 //Output_Buffered
176 /// Output buffer retrieving, used for File_Duplicate option.
177 /// @brief Output buffer retrieving
178 /// @param FilePos File position
179 /// @param Output_Buffer_Size A pointer to the variable that receives the size of the buffer \n
180 /// Note: you must use all the size of the buffer before the next call to this procedure
181 /// @return A pointer on the output buffer, NULL if there is nothing in the buffer
182 char* Output_Buffer_Get (size_t FilePos, size_t &Output_Buffer_Size);
183
184 //Info
185 /// Configure or get information about MediaInfoLib
186 /// @param Option The name of option
187 /// @param Value The value of option
188 /// @return Depend of the option: by default "" (nothing) means No, other means Yes
189 /// @post Known options are: See MediaInfo::Option()
190 String Option (const String &Option, const String &Value=String());
191 /// Configure or get information about MediaInfoLib (static version)
192 /// @param Option The name of option
193 /// @param Value The value of option
194 /// @return Depend of the option: by default "" (nothing) means No, other means Yes
195 /// @post Known options are: See MediaInfo::Option()
196 static String Option_Static (const String &Option, const String &Value=String());
197 /// @brief (NOT IMPLEMENTED YET) Get the state of the library
198 /// @retval <1000 No information is available for the file yet
199 /// @retval >=1000_<5000 Only local (into the file) information is available, getting Internet information (titles only) is no finished yet
200 /// @retval 5000 (only if Internet connection is accepted) User interaction is needed (use Option() with "Internet_Title_Get") \n
201 /// Warning: even there is only one possible, user interaction (or the software) is needed
202 /// @retval >5000<=10000 Only local (into the file) information is available, getting Internet information (all) is no finished yet
203 /// @retval <10000 Done
204 size_t State_Get ();
205 /// @brief Count of streams, or count of piece of information in this stream
206 /// @param FilePos File position \n
207 /// (you can know the position in searching the filename with MediaInfoList::Get(FilePos, 0, 0, "CompleteName") )
208 /// @param StreamKind Kind of stream (general, video, audio...)
209 /// @param StreamNumber Stream number in this kind of stream (first, second...)
210 /// @return The count of fields for this stream kind / stream number if stream number is provided, else the count of streams for this stream kind
211 size_t Count_Get (size_t FilePos, stream_t StreamKind, size_t StreamNumber=(size_t)-1);
212 /// @brief Get the count of opened files
213 /// @return Count of files opened
214 size_t Count_Get ();
215
216private :
217 MediaInfoList_Internal* Internal;
218
219 //Constructor
220 MediaInfoList (const MediaInfoList&); // Prevent copy-construction
221 MediaInfoList& operator=(const MediaInfoList&); // Prevent assignment
222};
