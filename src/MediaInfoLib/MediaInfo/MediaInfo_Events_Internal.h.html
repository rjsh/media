37 inline void Events_PCR(int64u PCR, int64u &Event_PCR, char* Event_PCR_HR)
38 {
39 Event_PCR=PCR;
40 if (PCR!=(int64u)-1)
41 {
42 string PCR_HR=Ztring().Duration_From_Milliseconds(PCR/1000000).To_UTF8();
43 if (PCR_HR.size()==12)
44 strcpy(Event_PCR_HR, PCR_HR.c_str());
45 else
46 memset(Event_PCR_HR, 0x00, 13);
47 }
48 else
49 memset(Event_PCR_HR, 0x00, 13);
50 }

82 inline void Events_TimeCode(const TimeCode &Tc, MediaInfo_time_code &Event_TimeCode, char* Event_TimeCode_HR)
83 {
84 if (Tc.IsValid())
85 {
86 Event_TimeCode.Hours=Tc.Hours;
87 Event_TimeCode.Minutes=Tc.Minutes;
88 Event_TimeCode.Seconds=Tc.Seconds;
89 Event_TimeCode.Frames=Tc.Frames;
90 Event_TimeCode.FramesPerSecond=Tc.FramesPerSecond;
91 Event_TimeCode.DropFrame=Tc.DropFrame;
92 Event_TimeCode_HR[ 0]='0'+Tc.Hours/10;
93 Event_TimeCode_HR[ 1]='0'+Tc.Hours%10;
94 Event_TimeCode_HR[ 2]=':';
95 Event_TimeCode_HR[ 3]='0'+Tc.Minutes/10;
96 Event_TimeCode_HR[ 4]='0'+Tc.Minutes%10;
97 Event_TimeCode_HR[ 5]=':';
98 Event_TimeCode_HR[ 6]='0'+Tc.Seconds/10;
99 Event_TimeCode_HR[ 7]='0'+Tc.Seconds%10;
100 Event_TimeCode_HR[ 8]=Tc.DropFrame?';':':';
101 Event_TimeCode_HR[ 9]='0'+Tc.Frames/10;
102 Event_TimeCode_HR[10]='0'+Tc.Frames%10;
103 Event_TimeCode_HR[11]='\0';
104 Event_TimeCode_HR[12]='\0';
105 }
106 else
107 {
108 Event_TimeCode.Hours=(MediaInfo_int8u)-1;
109 Event_TimeCode.Minutes=(MediaInfo_int8u)-1;
110 Event_TimeCode.Seconds=(MediaInfo_int8u)-1;
111 Event_TimeCode.Frames=(MediaInfo_int8u)-1;
112 Event_TimeCode.FramesPerSecond=(MediaInfo_int8u)-1;
113 Event_TimeCode.DropFrame=(MediaInfo_int8u)-1;
114 memset(Event_TimeCode_HR, 0x00, 13);
115 }
116 }
