408

417 node->Set_Name(Parameter.c_str());
418 node->Pos = Pos==(int64u)-1 ? Pos : (File_Offset+Buffer_Offset+Pos);
419 node->Value.set_Option(GenericOption);
420 node->Value = Value;
421 Element[Element_Level].TraceNode.Current_Child = Element[Element_Level].TraceNode.Children.size();
422 Element[Element_Level].TraceNode.Children.push_back(node);
423 }
