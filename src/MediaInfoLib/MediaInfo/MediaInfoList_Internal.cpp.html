43MediaInfoList_Internal::MediaInfoList_Internal(size_t Count_Init)
44: Thread()
45{
46 CriticalSectionLocker CSL(CS);
47
48 //Initialisation
49 Info.reserve(Count_Init);
50 for (size_t Pos=0; Pos<Info.size(); Pos++)
51 Info[Pos]=NULL;
52 ToParse_AlreadyDone=0;
53 ToParse_Total=0;
54 CountValid=0;
55
56 //Threading
57 BlockMethod=0;
58 State=0;
59 IsInThread=false;
60}

73//---------------------------------------------------------------------------
74size_t MediaInfoList_Internal::Open(const String &File_Name, const fileoptions_t Options)
75{
76 //Option FileOption_Close
77 if (Options & FileOption_CloseAll)
78 Close(All);
79
80 //Option Recursive
81 //TODO
82
83 //Get all filenames
84 ZtringList List;
85 size_t Pos=File_Name.find(__T(':'));
86 if (Pos!=string::npos && Pos!=1)
87 List.push_back(File_Name);
88 else if (File::Exists(File_Name))
89 List.push_back(File_Name);
90 else
91 List=Dir::GetAllFileNames(File_Name, (Options&FileOption_NoRecursive)?Dir::Include_Files:((Dir::dirlist_t)(Dir::Include_Files|Dir::Parse_SubDirs)));
92
93 #if defined(MEDIAINFO_DIRECTORY_YES)
94 Reader_Directory().Directory_Cleanup(List);
95 #endif //defined(MEDIAINFO_DIRECTORY_YES)
96
97 //Registering files
98 CS.Enter();
99 if (ToParse.empty())
100 CountValid=0;
101 for (ZtringList::iterator L=List.begin(); L!=List.end(); ++L)
102 ToParse.push(*L);
103 ToParse_Total+=List.size();
104 if (ToParse_Total)
105 State=ToParse_AlreadyDone*10000/ToParse_Total;
106 else
107 State=10000;
108 CS.Leave();
109
110 //Parsing
111 if (BlockMethod==1)
112 {
113 CS.Enter();
114 if (!IsRunning()) //If already created, the routine will read the new files
115 {
116 RunAgain();
117 IsInThread=true;
118 }
119 CS.Leave();
120 return 0;
121 }
122 else
123 {
124 Entry(); //Normal parsing
125 return Count_Get();
126 }

129void MediaInfoList_Internal::Entry()
130{
131 if (ToParse_Total==0)
132 return;
133
134 for (;;)
135 {
136 CS.Enter();
137 if (!ToParse.empty())
138 {
139 Ztring FileName=ToParse.front();
140 ToParse.pop();
141 MediaInfo_Internal* MI=new MediaInfo_Internal();
142 for (std::map<String, String>::iterator Config_MediaInfo_Item=Config_MediaInfo_Items.begin(); Config_MediaInfo_Item!=Config_MediaInfo_Items.end(); ++Config_MediaInfo_Item)
143 MI->Option(Config_MediaInfo_Item->first, Config_MediaInfo_Item->second);
144 if (BlockMethod==1)
145 MI->Option(__T("Thread"), __T("1"));
146 Info.push_back(MI);
147 CS.Leave();
148 MI->Open(FileName);
149
150 if (BlockMethod==1)
151 {
152 while (MI->State_Get()<10000)
153 {
154 size_t A=MI->State_Get();
155 CS.Enter();
156 State=(ToParse_AlreadyDone*10000+A)/ToParse_Total;
157 CS.Leave();
158 if (IsTerminating())
159 {
160 break;
161 }
162 Yield();
163 }
164 }
165 CS.Enter();
166 ToParse_AlreadyDone++;
167
168 //Removing sequences of files from the list
169 if (!MI->Get(Stream_General, 0, General_CompleteName_Last).empty())
170 {
171 Ztring CompleteName_Begin=MI->Get(Stream_General, 0, General_CompleteName);
172 Ztring CompleteName_Last=MI->Get(Stream_General, 0, General_CompleteName_Last);
173 size_t Pos=0;
174 for (; Pos<CompleteName_Begin.size(); Pos++)
175 {
176 if (Pos>=CompleteName_Last.size())
177 break;
178 if (CompleteName_Begin[Pos]!=CompleteName_Last[Pos])
179 break;
180 }
181 if (Pos<CompleteName_Begin.size())
182 {
183 CompleteName_Begin.resize(Pos);
184 while (!ToParse.empty() && ToParse.front().find(CompleteName_Begin)==0)
185 {
186 ToParse.pop();
187 ToParse_Total--;
188 }
189 }
190 }
191
192 State=ToParse_AlreadyDone*10000/ToParse_Total;
193 //if ((ToParse_AlreadyDone%10)==0)
194 // printf("%f done (%i/%i %s)\n", ((float)State)/100, (int)ToParse_AlreadyDone, (int)ToParse_Total, Ztring(ToParse.front()).To_UTF8().c_str());
195 }
196 if (IsTerminating() || State==10000)
197 {
198 CS.Leave();
199 break;
200 }
201 CS.Leave();
202 Yield();
203 }

255//---------------------------------------------------------------------------
256void MediaInfoList_Internal::Close(size_t FilePos)
257{
258 if (IsRunning())
259 {
260 RequestTerminate();
261 while(IsExited())
262 Yield();
263 }
264
265 CriticalSectionLocker CSL(CS);
266 if (FilePos==Unlimited)
267 {
268 for (size_t Pos=0; Pos<Info.size(); Pos++)
269 {
270 delete Info[Pos]; Info[Pos]=NULL;
271 }
272 Info.clear();
273 }
274 else if (FilePos<Info.size())
275 {
276 delete Info[FilePos]; Info[FilePos]=NULL;
277 Info.erase(Info.begin()+FilePos);
278 }
279
280 ToParse_AlreadyDone=0;
281 ToParse_Total=0;
282}

551//---------------------------------------------------------------------------
552String MediaInfoList_Internal::Option (const String &Option, const String &Value)
553{
554 CriticalSectionLocker CSL(CS);
555 Ztring OptionLower=Option; OptionLower.MakeLowerCase();
556 if (Option==__T(""))
557 return __T("");
558 else if (OptionLower==__T("manguage_update"))
559 {
560 //Special case : Language_Update must update all MediaInfo classes
561 for (unsigned int Pos=0; Pos<Info.size(); Pos++)
562 if (Info[Pos])
563 Info[Pos]->Option(__T("language_update"), Value);
564
565 return __T("");
566 }
567 else if (OptionLower==__T("create_dummy"))
568 {
569 Info.resize(Info.size()+1);
570 Info[Info.size()-1]=new MediaInfo_Internal();
571 Info[Info.size()-1]->Option(Option, Value);
572 return __T("");
573 }
574 else if (OptionLower==__T("thread"))
575 {
576 BlockMethod=1;
577 return __T("");
578 }
579 else if (OptionLower.find(__T("file_"))==0)
580 {
581 Config_MediaInfo_Items[Option]=Value;
582 return __T("");
583 }
584 else
585 return MediaInfo::Option_Static(Option, Value);
586}

594//---------------------------------------------------------------------------
595size_t MediaInfoList_Internal::State_Get()
596{
597 CriticalSectionLocker CSL(CS);
598 if (State==10000)
599 {
600 //Pause();
601 IsInThread=false;
602 }
603
604 if (!Info.empty())
605 {
606 State=0;
607 for (size_t Pos=0; Pos<Info.size(); Pos++)
608 State+=Info[Pos]->State_Get();
609 State/=Info.size()+ToParse.size();
610 }
611
612 return State;
613}
