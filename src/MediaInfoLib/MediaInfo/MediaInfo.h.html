56 //File
57 /// Open a file and collect information about it (technical information and tags)
58 /// @brief Open a file
59 /// @param File_Name Full name of file to open
60 /// @retval 0 File not opened
61 /// @retval 1 File opened
62 size_t Open (const String &File_Name);
63 /// Open a Buffer (Begin and end of the stream) and collect information about it (technical information and tags)
64 /// @brief Open a buffer
65 /// @param Begin First bytes of the buffer
66 /// @param Begin_Size Size of Begin
67 /// @param End Last bytes of the buffer
68 /// @param End_Size Size of End
69 /// @param File_Size Total size of the file
70 /// @retval 0 File not opened
71 /// @retval 1 File opened
72 size_t Open (const ZenLib::int8u* Begin, size_t Begin_Size, const ZenLib::int8u* End=NULL, size_t End_Size=0, ZenLib::int64u File_Size=0);
73 /// Open a stream and collect information about it (technical information and tags)
74 /// @brief Open a stream (Init)
75 /// @param File_Size Estimated file size
76 /// @param File_Offset Offset of the file (if we don't have the beginning of the file)
77 /// @retval 0 File not opened
78 /// @retval 1 File opened
79 size_t Open_Buffer_Init (ZenLib::int64u File_Size=(ZenLib::int64u)-1, ZenLib::int64u File_Offset=0);
80 /// Open a stream and collect information about it (technical information and tags)
81 /// @brief Open a stream (Continue)
82 /// @param Buffer pointer to the stream
83 /// @param Buffer_Size Count of bytes to read
84 /// @return a bitfield \n
85 /// bit 0: Is Accepted (format is known)
86 /// bit 1: Is Filled (main data is collected)
87 /// bit 2: Is Updated (some data have beed updated, example: duration for a real time MPEG-TS stream)
88 /// bit 3: Is Finalized (No more data is needed, will not use further data)
89 /// bit 4-15: Reserved
90 /// bit 16-31: User defined
91 size_t Open_Buffer_Continue (const ZenLib::int8u* Buffer, size_t Buffer_Size);
92 /// Open a stream and collect information about it (technical information and tags)
93 /// @brief Open a stream (Get the needed file Offset)
94 /// @return the needed offset of the file \n
95 /// File size if no more bytes are needed
96 ZenLib::int64u Open_Buffer_Continue_GoTo_Get ();
97 /// Open a stream and collect information about it (technical information and tags)
98 /// @brief Open a stream (Finalize)
99 /// @retval 0 failed
100 /// @retval 1 succeed
101 size_t Open_Buffer_Finalize ();
102 /// If Open() is used in "PerPacket" mode, parse only one packet and return
103 /// @brief Read one packet (if "PerPacket" mode is set)
104 /// @return a bitfield \n
105 /// bit 0: A packet was read
106 size_t Open_NextPacket ();
107 /// (NOT IMPLEMENTED YET) Save the file opened before with Open() (modifications of tags)
108 /// @brief (NOT IMPLEMENTED YET) Save the file
109 /// @retval 0 failed
110 /// @retval 1 suceed
111 size_t Save ();
112 /// Close a file opened before with Open() (without saving)
113 /// @brief Close a file
114 /// @warning without have saved before, modifications are lost
115 void Close ();

125 //Get
126 /// Get a piece of information about a file (parameter is an integer)
127 /// @brief Get a piece of information about a file (parameter is an integer)
128 /// @param StreamKind Kind of stream (general, video, audio...)
129 /// @param StreamNumber Stream number in Kind of stream (first, second...)
130 /// @param Parameter Parameter you are looking for in the stream (Codec, width, bitrate...), in integer format (first parameter, second parameter...) \n
131 /// This integer is arbitarily assigned by the library, so its consistency should not be relied on, but is useful when looping through all the parameters
132 /// @param InfoKind Kind of information you want about the parameter (the text, the measure, the help...)
133 /// @return a string about information you search \n
134 /// an empty string if there is a problem
135 String Get (stream_t StreamKind, size_t StreamNumber, size_t Parameter, info_t InfoKind=Info_Text);
136 /// Get a piece of information about a file (parameter is a string)
137 /// @brief Get a piece of information about a file (parameter is a string)
138 /// @param StreamKind Kind of stream (general, video, audio...)
139 /// @param StreamNumber Stream number in Kind of stream (first, second...)
140 /// @param Parameter Parameter you are looking for in the stream (Codec, width, bitrate...), in string format ("Codec", "Width"...) \n
141 /// See MediaInfo::Option("Info_Parameters") to have the full list
142 /// @param InfoKind Kind of information you want about the parameter (the text, the measure, the help...)
143 /// @param SearchKind Where to look for the parameter
144 /// @return a string about information you search \n
145 /// an empty string if there is a problem
146 String Get (stream_t StreamKind, size_t StreamNumber, const String &Parameter, info_t InfoKind=Info_Text, info_t SearchKind=Info_Name);

184 //Info
185 /// Configure or get information about MediaInfoLib
186 /// @param Option The name of option
187 /// @param Value The value of option
188 /// @return Depend of the option: by default "" (nothing) means No, other means Yes
189 /// @post Known options are: \n
190 /// * (NOT IMPLEMENTED YET) "BlockMethod": Configure when Open Method must return (default or not command not understood: "1") \n
191 /// "0": Immediatly \n
192 /// "1": After geting local information \n
193 /// "2": When user interaction is needed, or whan Internet information is get
194 /// * "Complete": For debug, configure if MediaInfoLib::Inform() show all information (doesn't care of InfoOption_NoShow tag): shows all information if true, shows only useful for user information if false (No by default)\n
195 /// * "Complete_Get": return the state of "Complete" \n
196 /// * "Language": Configure language (default language, and this object); Value is Description of language (format: "Column1;Colum2\n...) \n
197 /// Column 1: Unique name ("Bytes", "Title") \n
198 /// Column 2: translation ("Octets", "Titre") \n
199 /// * "Language_Get": Get the language file in memory
200 /// * "Language_Update": Configure language of this object only (for optimisation); Value is Description of language (format: "Column1;Colum2\n...) \n
201 /// Column 1: Unique name ("Bytes", "Title") \n
202 /// Column 2: translation ("Octets", "Titre") \n
203 /// * "Inform": Configure custom text, See MediaInfoLib::Inform() function; Description of views (format: "Column1;Colum2...) \n
204 /// Column 1: code (11 lines: "General", "Video", "Audio", "Text", "Other", "Begin", "End", "Page_Begin", "Page_Middle", "Page_End") \n
205 /// Column 2: The text to show (exemple: "Audio: %FileName% is at %BitRate/String%") \n
206 /// * "ParseUnknownExtensions": Configure if MediaInfo parse files with unknown extension\n
207 /// * "ParseUnknownExtensions_Get": Get if MediaInfo parse files with unknown extension\n
208 /// * "ShowFiles": Configure if MediaInfo keep in memory files with specific kind of streams (or no streams); Value is Description of components (format: "Column1;Colum2\n...) \n\n
209 /// Column 1: code (available: "Nothing" for unknown format, "VideoAudio" for at least 1 video and 1 audio, "VideoOnly" for video streams only, "AudioOnly", "TextOnly") \n
210 /// Column 2: "" (nothing) not keeping, other for keeping
211 /// * (NOT IMPLEMENTED YET) "TagSeparator": Configure the separator if there are multiple same tags (" | " by default)\n
212 /// * (NOT IMPLEMENTED YET) "TagSeparator_Get": return the state of "TagSeparator" \n
213 /// * (NOT IMPLEMENTED YET) "Internet": Authorize Internet connection (Yes by default)
214 /// * (NOT IMPLEMENTED YET) "Internet_Title_Get": When State=5000, give all possible titles for this file (one per line) \n
215 /// Form: Author TagSeparator Title TagSeparator Year\n...
216 /// * (NOT IMPLEMENTED YET) "Internet_Title_Set": Set the Good title (same as given by Internet_Title_Get) \n
217 /// Form: Author TagSeparator Title TagSeparator Year
218 /// * "Info_Parameters": Information about what are known unique names for parameters \n
219 /// * "Info_Parameters_CSV": Information about what are known unique names for parameters, in CSV format \n
220 /// * "Info_Codecs": Information about which codec is known \n
221 /// * "Info_Version": Information about the version of MediaInfoLib
222 /// * "Info_Url": Information about where to find the last version
223 String Option (const String &Option, const String &Value=String());
224 /// Configure or get information about MediaInfoLib
225 /// @param Option The name of option
226 /// @param Value The value of option
227 /// @return Depend of the option: by default "" (nothing) means No, other means Yes
228 /// @post Known options are: See MediaInfo::Option()
229 static String Option_Static (const String &Option, const String &Value=String());
230 /// @brief (NOT IMPLEMENTED YET) Get the state of the library
231 /// @retval <1000 No information is available for the file yet
232 /// @retval >=1000_<5000 Only local (into the file) information is available, getting Internet information (titles only) is no finished yet
233 /// @retval 5000 (only if Internet connection is accepted) User interaction is needed (use Option() with "Internet_Title_Get") \n
234 /// Warning: even there is only one possible, user interaction (or the software) is needed
235 /// @retval >5000<=10000 Only local (into the file) information is available, getting Internet information (all) is no finished yet
236 /// @retval <10000 Done
237 size_t State_Get ();

241 /// @return The count of fields for this stream kind / stream number if stream number is provided, else the count of streams for this stream kind
