62//---------------------------------------------------------------------------
63MediaInfo::MediaInfo()
64{
65 Internal=new MediaInfo_Internal();
66
67 // FMA3 support in the 2013 CRT is broken on Vista and Windows 7 RTM (fixed in SP1).
68 // See https://connect.microsoft.com/VisualStudio/feedback/details/987093/x64-log-function-uses-vpsrlq-avx-instruction-without-regard-to-operating-system-so-it-crashes-on-vista-x64
69 // Hotfix: we disable it for MSVC2013.
70 #if defined(_MSC_VER) && _MSC_VER >= 1800 && _MSC_VER < 1900 && defined(_M_X64)
71 _set_FMA3_enable(0);
72 #endif
73}

203//---------------------------------------------------------------------------
204String MediaInfo::Option_Static (const String &Option, const String &Value)
205{
206 MEDIAINFO_DEBUG_STATIC(Debug+=__T("Option_Static, Option=");Debug+=Ztring(Option);Debug+=__T(", Value=");Debug+=Ztring(Value);)
207 MediaInfoLib::Config.Init(); //Initialize Configuration
208
209 if (Option==__T("Info_Capacities"))
210 {
211 return __T("Option disactivated for this version, will come back soon!");
212 //MediaInfo_Internal MI;
213 //return MI.Option(Option);
214 }
215 else if (Option==__T("Info_Version"))
216 {
217 Ztring ToReturn=MediaInfoLib::Config.Info_Version_Get();
218 if (MediaInfo_Internal::LibraryIsModified())
219 ToReturn+=__T(" modified");
220 return ToReturn;
221 }
222 else
223 return MediaInfoLib::Config.Option(Option, Value);
224}
