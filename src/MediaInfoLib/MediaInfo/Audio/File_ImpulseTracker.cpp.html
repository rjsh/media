34//---------------------------------------------------------------------------
35bool File_ImpulseTracker::FileHeader_Begin()
36{
37 //Element_Size
38 if (Buffer_Size<4)
39 return false; //Must wait for more data
40
41 if (CC4(Buffer)!=0x494D504D) //"IMPM"
42 {
43 Reject("Impulse Tracker");
44 return false;
45 }
46
47 //All should be OK...
48 return true;
49}
50
51//***************************************************************************
52// Buffer - Global
53//***************************************************************************
54
55//---------------------------------------------------------------------------
56void File_ImpulseTracker::Read_Buffer_Continue()
57{
58 //Parsing
59 Ztring SongName;
60 int16u OrdNum, InsNum, SmpNum, PatNum, Flags, Special;
61 int8u VersionMajor, VersionMinor, SoftwareVersionMajor, SoftwareVersionMinor, IS, TS;
62 bool Stereo;
63 Skip_B4( "Signature");
64
65 Get_Local(26, SongName, "Song name");
66 Skip_L1( "Unknown");
67 Skip_L1( "Unknown");
68 Get_L2 (OrdNum, "Orders count");
69 Get_L2 (InsNum, "Instruments count");
70 Get_L2 (SmpNum, "Samples count");
71 Get_L2 (PatNum, "Paterns count");
72 Get_L1 (SoftwareVersionMinor, "Cwt/v (Minor)");
73 Get_L1 (SoftwareVersionMajor, "Cwt/v (Major)");
74 Get_L1 (VersionMinor, "Cwt (Minor)");
75 Get_L1 (VersionMajor, "Cwt (Major)");
76 Get_L2 (Flags, "Flags");
77 Get_Flags (Flags, 0, Stereo, "Stereo");
78 Skip_Flags(Flags, 1, "Vol0MixOptimizations");
79 Skip_Flags(Flags, 2, "Use instruments/Samples");
80 Skip_Flags(Flags, 3, "Linear/Amiga slides");
81 Skip_Flags(Flags, 4, "Old/IT Effects");
82 Get_L2 (Special, "Special");
83 Skip_Flags(Special, 0, "Song Message attached");
84 Skip_L1( "Global volume");
85 Skip_L1( "Mix volume");
86 Get_L1 (IS, "Initial Speed");
87 Get_L1 (TS, "Initial Temp");
88 Skip_L1( "Panning separation between channels");
89 Skip_L1( "0");
90 Skip_L2( "Message Length");
91 Skip_L4( "Message Offset");
92 Skip_L1( "Unknown");
93 Skip_L1( "Unknown");
94 Skip_L1( "Unknown");
95 Skip_L1( "Unknown");
96 Skip_L1( "Unknown");
97 Skip_XX(64, "Chnl Pan");
98 Skip_XX(64, "Chnl Vol");
99 Skip_XX(OrdNum, "Orders");
100 Skip_XX(InsNum*4, "Instruments");
101 Skip_XX(SmpNum*4, "Samples");
102 Skip_XX(PatNum*4, "Patterns");
103
104 FILLING_BEGIN();
105 Accept("Impulse Tracker");
106
107 Fill(Stream_General, 0, General_Format, "Impulse Tracker");
108 Fill(Stream_General, 0, General_Format_Version, Ztring(__T("Version "))+Ztring::ToZtring(VersionMajor)+__T(".")+Ztring::ToZtring(VersionMinor/16)+Ztring::ToZtring(VersionMinor%16));
109 Fill(Stream_General, 0, General_Track, SongName);
110 Fill(Stream_General, 0, General_Encoded_Application, Ztring(__T("Impulse Tracker ")+Ztring::ToZtring(SoftwareVersionMajor)+__T(".")+Ztring::ToZtring(SoftwareVersionMinor/16)+Ztring::ToZtring(SoftwareVersionMinor%16)));
111 Fill(Stream_General, 0, "BPM", TS);
112
113 Stream_Prepare(Stream_Audio);
114 Fill(Stream_Audio, StreamPos_Last, Audio_Channel_s_, Stereo?2:1);
115
116 //No more need data
117 Finish("Impulse Tracker");
118 FILLING_END();
119}
