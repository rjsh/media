61//***************************************************************************
62// Buffer - File header
63//***************************************************************************
64
65//---------------------------------------------------------------------------
66bool File_Caf::FileHeader_Begin()
67{
68 //Synchro
69 if (3>Buffer_Size)
70 return false;
71 if (Buffer[0]!=0x63 //"caff"
72 || Buffer[1]!=0x61
73 || Buffer[2]!=0x66
74 || Buffer[3]!=0x66)
75 {
76 Reject();
77 return false;
78 }
79 if (8>Buffer_Size)
80 return false;
81
82 return true;
83}

89//---------------------------------------------------------------------------
90void File_Caf::FileHeader_Parse()
91{
92 //Parsing
93 int16u FileVersion;
94 Skip_C4( "FileType");
95 Get_B2 (FileVersion, "FileVersion");
96 Skip_B2( "FileFlags");
97
98 FILLING_BEGIN();
99 Accept();
100 Fill(Stream_General, 0, General_Format, "CAF");
101 Fill(Stream_General, 0, General_Format_Version, __T("Version ")+Ztring::ToZtring(FileVersion));
102 Stream_Prepare(Stream_Audio);
103
104 if (FileVersion!=1)
105 Finish(); //Version 0 or 2+ are not supported
106 FILLING_END();
107}

113//---------------------------------------------------------------------------
114void File_Caf::Header_Parse()
115{
116 //Parsing
117 int64u ChunkSize;
118 int32u ChunkType;
119 Get_B4 (ChunkType, "ChunkType");
120 Get_B8(ChunkSize, "ChunkSize");
121
122 //Filling
123 Header_Fill_Code2(ChunkType, Ztring().From_CC4(ChunkType));
124 Header_Fill_Size(12+ChunkSize);
125}

127//---------------------------------------------------------------------------
128void File_Caf::Data_Parse()
129{
130 if (Element_Code!=Elements::data && !Element_IsComplete_Get())
131 {
132 Element_WaitForMoreData();
133 return;
134 }
135
136 #define ELEMENT_CASE(_NAME, _DETAIL) \
137 case Elements::_NAME : Element_Name(_DETAIL); _NAME(); break;
138
139 //Parsing
140 switch (Element_Code)
141 {
142 ELEMENT_CASE(data, "Audio Data");
143 ELEMENT_CASE(desc, "Audio Description");
144 ELEMENT_CASE(free, "Free");
145 ELEMENT_CASE(info, "Information");
146 ELEMENT_CASE(kuki, "Magic Cookie");
147 ELEMENT_CASE(pakt, "Packet Table");
148 ELEMENT_CASE(uuid, "User-Defined Chunk");
149 default :
150 Skip_XX(Element_Size, "Data");
151 }
152}

158//---------------------------------------------------------------------------
159void File_Caf::data()
160{
161 //Parsing
162 Skip_XX(Element_Size, "Data");
163
164 Fill(Stream_Audio, 0, Retrieve(Stream_Audio, 0, Audio_Source_Duration).empty()?Audio_StreamSize:Audio_Source_StreamSize, Element_Size);
165
166 //TODO: put this code in the common section Streams_Finish_StreamOnly()
167 int64u BitRate=Retrieve(Stream_Audio, 0, "BitRate").To_int64u();
168 if (BitRate && Element_Size && Retrieve(Stream_Audio, 0, Audio_Source_Duration).empty() && Retrieve(Stream_Audio, 0, Audio_Duration).empty())
169 Fill(Stream_Audio, 0, Audio_Duration, Element_Size*8*1000/BitRate);
170}

172//---------------------------------------------------------------------------
173void File_Caf::desc()
174{
175 //Parsing
176 float64 SampleRate;
177 int32u FormatID, FormatFlags, BytesPerPacket, FramesPerPacket, ChannelsPerFrame, BitsPerChannel;
178 Get_BF8(SampleRate, "SampleRate");
179 Get_C4 (FormatID, "FormatID");
180 Get_B4 (FormatFlags, "FormatFlags");
181 Get_B4 (BytesPerPacket, "BytesPerPacket");
182 Get_B4 (FramesPerPacket, "FramesPerPacket");
183 Get_B4 (ChannelsPerFrame, "ChannelsPerFrame");
184 Get_B4 (BitsPerChannel, "BitsPerChannel");
185
186 FILLING_BEGIN();
187 if (SampleRate)
188 Fill(Stream_Audio, 0, Audio_SamplingRate, SampleRate);
189 CodecID_Fill(Ztring().From_CC4(FormatID), Stream_Audio, 0, InfoCodecID_Format_Mpeg4);
190 if (ChannelsPerFrame)
191 Fill(Stream_Audio, 0, Audio_Channel_s_, ChannelsPerFrame);
192 if (BitsPerChannel)
193 Fill(Stream_Audio, 0, Audio_BitDepth, BitsPerChannel);
194 if (BytesPerPacket && SampleRate && FramesPerPacket)
195 Fill(Stream_Audio, 0, Audio_BitRate, SampleRate*BytesPerPacket*8/FramesPerPacket);
196 FILLING_END();
197}

206//---------------------------------------------------------------------------
207void File_Caf::info()
208{
209 if (Element_Size<4)
210 return;
211
212 //Parsing
213 int32u NumEntries;
214 Get_B4 (NumEntries, "NumEntries");
215 ZtringList List;
216 std::map<Ztring, Ztring> ListList;
217 const int8u* Buffer_Max = Buffer+(size_t)(Buffer_Offset+Element_Size);
218 while (Element_Offset<Element_Size)
219 {
220 const int8u* Buffer_Begin = Buffer+(size_t)(Buffer_Offset+Element_Offset);
221 const int8u* Buffer_Middle = Buffer_Begin;
222 while (Buffer_Middle<Buffer_Max && *Buffer_Middle)
223 ++Buffer_Middle;
224 const int8u* Buffer_End = Buffer_Middle + 1;
225 while (Buffer_End<Buffer_Max && *Buffer_End)
226 ++Buffer_End;
227
228 Ztring Key, Value;
229 Get_UTF8(Buffer_Middle-Buffer_Begin, Key, "Key");
230 Skip_B1 ( "Zero");
231 Get_UTF8(Buffer_End-(Buffer_Middle+1), Value, "Value");
232 if (Buffer_End!=Buffer_Max)
233 Skip_B1 ( "Zero");
234
235 ListList[Key]=Value;
236 }
237
238 if (ListList.size()!=NumEntries)
239 return;
240
241 for (std::map<Ztring, Ztring>::iterator Item=ListList.begin(); Item!=ListList.end(); ++Item)
242 Fill(Stream_General, 0, Item->first.To_UTF8().c_str(), Item->second);
243}

252//---------------------------------------------------------------------------
253void File_Caf::pakt()
254{
255 //Parsing
256 int64u NumberPackets, NumberValidFrames;
257 int32u PrimingFrames, RemainderFrames;
258 Get_B8 (NumberPackets, "NumberPackets");
259 Get_B8 (NumberValidFrames, "NumberValidFrames");
260 Get_B4 (PrimingFrames, "PrimingFrames");
261 Get_B4 (RemainderFrames, "RemainderFrames");
262 Skip_XX(Element_Size-Element_Offset, "Packet sizes");
263
264 FILLING_BEGIN();
265 float64 SampleRate=Retrieve(Stream_Audio, 0, Audio_SamplingRate).To_float64();
266 Fill(Stream_Audio, 0, Audio_FrameCount, NumberPackets);
267 Fill(Stream_Audio, 0, Audio_Duration, NumberValidFrames/SampleRate*1000, 0);
268 if (PrimingFrames && RemainderFrames)
269 Fill(Stream_Audio, 0, Audio_Source_Duration, (PrimingFrames+NumberValidFrames+RemainderFrames)/SampleRate*1000, 0);
270 Fill(Stream_Audio, 0, Audio_Delay, PrimingFrames/SampleRate*1000, 0);
271 FILLING_END();
272}
