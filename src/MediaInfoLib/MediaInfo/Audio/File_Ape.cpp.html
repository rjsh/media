37int32u Ape_SamplesPerFrame(int16u Version, int16u CompressionLevel)
38{
39 if (Version>=3950)
40 return 73728*4;
41 else if (Version>=3900)
42 return 73728;
43 else if (Version>=3800 && CompressionLevel==4000)
44 return 73728;
45 else
46 return 9216;
47}
48
49//---------------------------------------------------------------------------
50const char* Ape_Codec_Settings(int16u Setting)
51{
52 switch (Setting)
53 {
54 case 1000 : return "Fast";
55 case 2000 : return "Normal";
56 case 3000 : return "High";
57 case 4000 : return "Extra-high";
58 case 5000 : return "Insane";
59 default : return "";
60 }
61}

79//---------------------------------------------------------------------------
80void File_Ape::Streams_Finish()
81{
82 //Filling
83 int64u CompressedSize=File_Size-TagsSize;
84 float32 CompressionRatio=((float32)UncompressedSize)/CompressedSize;
85 int64u BitRate=Duration?(CompressedSize*8*1000/Duration):0;
86
87 Fill(Stream_Audio, 0, Audio_Compression_Ratio, CompressionRatio);
88 Fill(Stream_Audio, 0, Audio_BitRate, BitRate);
89
90 File__Tags_Helper::Streams_Finish();
91}

97//---------------------------------------------------------------------------
98bool File_Ape::FileHeader_Begin()
99{
100 if (!File__Tags_Helper::FileHeader_Begin())
101 return false;
102
103 //Testing
104 if (Buffer_Size<Buffer_Offset+4)
105 return false;
106 if (Buffer[Buffer_Offset ]!=0x4D //"MAC "
107 || Buffer[Buffer_Offset+1]!=0x41
108 || Buffer[Buffer_Offset+2]!=0x43
109 || Buffer[Buffer_Offset+3]!=0x20)
110 {
111 File__Tags_Helper::Reject("APE");
112 return false;
113 }
114
115 return true;
116}
117
118//---------------------------------------------------------------------------
119void File_Ape::FileHeader_Parse()
120{
121 //Parsing
122 int32u SampleRate=0, TotalFrames=0, FinalFrameSamples=0, SamplesPerFrame=0, SeekElements;
123 int16u Version, CompressionLevel=0, Flags=0, Channels=0, Resolution=0;
124 Skip_C4( "Identifier");
125 Get_L2 (Version, "Version");
126 if (Version<3980) //<3.98
127 {
128 bool Resolution8=false, Resolution24=false, no_wav_header;
129 Get_L2 (CompressionLevel, "CompressionLevel"); Param_Info1(Ape_Codec_Settings(CompressionLevel));
130 Get_L2 (Flags, "FormatFlags");
131 Get_Flags (Flags, 0, Resolution8, "8-bit");
132 Skip_Flags(Flags, 1, "crc-32");
133 Skip_Flags(Flags, 2, "peak_level");
134 Get_Flags (Flags, 3, Resolution24, "24-bit");
135 Skip_Flags(Flags, 4, "seek_elements");
136 Get_Flags (Flags, 5, no_wav_header, "no_wav_header");
137 if (Resolution8)
138 Resolution=8;
139 else if (Resolution24)
140 Resolution=24;
141 else
142 Resolution=16;
143 Get_L2 (Channels, "Channels");
144 Get_L4 (SampleRate, "SampleRate");
145 Skip_L4( "WavHeaderDataBytes");
146 Skip_L4( "WavTerminatingBytes");
147 Get_L4 (TotalFrames, "TotalFrames");
148 Get_L4 (FinalFrameSamples, "FinalFrameSamples");
149 SamplesPerFrame=Ape_SamplesPerFrame(Version, CompressionLevel);
150 Skip_L4( "PeakLevel");
151 Get_L4 (SeekElements, "SeekElements");
152 if (!no_wav_header)
153 Skip_XX(44, "RIFF header");
154 Skip_XX(SeekElements*4, "Seek table");
155 }
156 else
157 {
158 Skip_L2( "Version_High");
159 Skip_L4( "DescriptorBytes");
160 Skip_L4( "HeaderBytes");
161 Skip_L4( "SeekTableBytes");
162 Skip_L4( "WavHeaderDataBytes");
163 Skip_L4( "APEFrameDataBytes");
164 Skip_L4( "APEFrameDataBytesHigh");
165 Skip_L4( "WavTerminatingDataBytes");
166 Skip_L16( "FileMD5");
167 Get_L2 (CompressionLevel, "CompressionLevel"); Param_Info1(Ape_Codec_Settings(CompressionLevel));
168 Get_L2 (Flags, "FormatFlags");
169 Get_L4 (SamplesPerFrame, "BlocksPerFrame");
170 Get_L4 (FinalFrameSamples, "FinalFrameBlocks");
171 Get_L4 (TotalFrames, "TotalFrames");
172 Get_L2 (Resolution, "BitsPerSample");
173 Get_L2 (Channels, "Channels");
174 Get_L4 (SampleRate, "SampleRate");
175 }
176
177 FILLING_BEGIN();
178 //Coherancy
179 int32u Samples=(TotalFrames-1)*SamplesPerFrame+FinalFrameSamples;
180 if (Samples==0 || SampleRate==0 || Channels==0 || Resolution==0)
181 {
182 File__Tags_Helper::Reject("APE");
183 return;
184 }
185
186 //Filling
187 File__Tags_Helper::Accept("APE");
188 File__Tags_Helper::Streams_Fill();
189
190 Duration=((int64u)Samples)*1000/SampleRate;
191 UncompressedSize=Samples*Channels*(Resolution/8);
192
193 File__Tags_Helper::Stream_Prepare(Stream_Audio);
194 Fill(Stream_Audio, 0, Audio_Format, "Monkey's Audio");
195 Fill(Stream_Audio, 0, Audio_Encoded_Library_Settings, Ape_Codec_Settings(CompressionLevel));
196 Fill(Stream_Audio, 0, Audio_Codec, "APE");
197 Fill(Stream_Audio, 0, Audio_BitDepth, Resolution);
198 Fill(Stream_Audio, 0, Audio_Channel_s_, Channels);
199 Fill(Stream_Audio, 0, Audio_SamplingRate, SampleRate);
200 Fill(Stream_Audio, 0, Audio_Duration, Duration);
201
202 //No more need data
203 File__Tags_Helper::Finish("APE");
204 FILLING_END();
205}
