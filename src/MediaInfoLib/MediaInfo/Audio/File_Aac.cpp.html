112//---------------------------------------------------------------------------
113void File_Aac::Streams_Fill()
114{
115 switch(Mode)
116 {
117 case Mode_LATM : Fill(Stream_General, 0, General_Format, "LATM"); if (IsSub) Fill(Stream_Audio, 0, Audio_MuxingMode, "LATM"); break;
118 default : ;
119 }
120
121 for (std::map<std::string, Ztring>::iterator Info=Infos_General.begin(); Info!=Infos_General.end(); ++Info)
122 Fill(Stream_General, 0, Info->first.c_str(), Info->second);
123 File__Tags_Helper::Stream_Prepare(Stream_Audio);
124 for (std::map<std::string, Ztring>::iterator Info=Infos.begin(); Info!=Infos.end(); ++Info)
125 Fill(Stream_Audio, StreamPos_Last, Info->first.c_str(), Info->second);
126
127 switch(Mode)
128 {
129 case Mode_ADTS : File__Tags_Helper::Streams_Fill(); break;
130 default : ;
131 }
132
133 Fill(Stream_Audio, StreamPos_Last, Audio_SamplesPerFrame, frame_length);
134}

136//---------------------------------------------------------------------------
137void File_Aac::Streams_Update()
138{
139 if (Frame_Count)
140 {
141 if (Mode==Mode_ADTS)
142 Infos["BitRate_Mode"].From_Local(adts_buffer_fullness_Is7FF?"VBR":"CBR");
143
144 #if MEDIAINFO_ADVANCED
145 switch(Mode)
146 {
147 case Mode_ADTS :
148 case Mode_LATM : if (Config->File_RiskyBitRateEstimation_Get() && !adts_buffer_fullness_Is7FF)
149 {
150 int64u BitRate=(Frequency_b/1024);
151 BitRate*=aac_frame_length_Total*8;
152 BitRate/=Frame_Count;
153
154 Fill(Stream_Audio, 0, Audio_BitRate, BitRate, 10, true);
155 }
156 break;
157 default : ;
158 }
159 #endif //MEDIAINFO_ADVANCED
160 }
161}
162
163//---------------------------------------------------------------------------
164void File_Aac::Streams_Finish()
165{
166 switch(Mode)
167 {
168 case Mode_ADIF :
169 case Mode_ADTS : File__Tags_Helper::Streams_Finish(); break;
170 default : ;
171 }
172
173 if (FrameSize_Min!=(int32u)-1 && FrameSize_Max)
174 {
175 if (FrameSize_Max>FrameSize_Min*1.02)
176 {
177 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "VBR", Unlimited, true, true);
178 if (Config->ParseSpeed>=1.0)
179 {
180 Fill(Stream_Audio, 0, Audio_BitRate_Minimum, ((float64)FrameSize_Min)/1024*48000*8, 0);
181 Fill(Stream_Audio, 0, Audio_BitRate_Maximum, ((float64)FrameSize_Max)/1024*48000*8, 0);
182 }
183 }
184 else if (Config->ParseSpeed>=1.0)
185 {
186 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "CBR");
187 }
188 }
189}

195//---------------------------------------------------------------------------
196bool File_Aac::FileHeader_Begin()
197{
198 switch(Mode)
199 {
200 case Mode_AudioSpecificConfig :
201 case Mode_ADIF :
202 MustSynchronize=false; break;
203 default : ; //Synchronization is requested, and this is the default
204 }
205
206 switch(Mode)
207 {
208 case Mode_Unknown :
209 case Mode_ADIF :
210 case Mode_ADTS :
211 break;
212 default : return true; //no file header test with other modes
213 }
214
215 //Tags
216 if (!File__Tags_Helper::FileHeader_Begin())
217 return false;
218
219 //Testing
220 if (Buffer_Size<4)
221 return false;
222 if (Buffer[0]==0x41 //"ADIF"
223 && Buffer[1]==0x44
224 && Buffer[2]==0x49
225 && Buffer[3]==0x46)
226 {
227 Mode=Mode_ADIF;
228 File__Tags_Helper::Accept("ADIF");
229 MustSynchronize=false;
230 }
231
232 return true;
233}

262//---------------------------------------------------------------------------
263void File_Aac::Read_Buffer_Continue()
264{
265 if (Element_Size==0)
266 return;
267
268 if (Frame_Count==0)
269 PTS_Begin=FrameInfo.PTS;
270
271 switch(Mode)
272 {
273 case Mode_AudioSpecificConfig : Read_Buffer_Continue_AudioSpecificConfig(); break;
274 case Mode_raw_data_block : Read_Buffer_Continue_raw_data_block(); break;
275 case Mode_ADIF :
276 case Mode_ADTS : File__Tags_Helper::Read_Buffer_Continue(); break;
277 default : ;
278 }
279}

281//---------------------------------------------------------------------------
282void File_Aac::Read_Buffer_Continue_AudioSpecificConfig()
283{
284 File__Analyze::Accept(); //We automaticly trust it
285
286 BS_Begin();
287 AudioSpecificConfig(0); //Up to the end of the block
288 BS_End();
289
290 Mode=Mode_raw_data_block; //Mode_AudioSpecificConfig only once
291}
292
293//---------------------------------------------------------------------------
294void File_Aac::Read_Buffer_Continue_raw_data_block()
295{
296 if (Frame_Count>Frame_Count_Valid)
297 {
298 Skip_XX(Element_Size, "Data");
299 return; //Parsing completely only the 1st frame
300 }
301
302 BS_Begin();
303 raw_data_block();
304 BS_End();
305 if (FrameIsAlwaysComplete && Element_Offset<Element_Size)
306 Skip_XX(Element_Size-Element_Offset, "Unknown");
307
308 FILLING_BEGIN();
309 //Counting
310 Frame_Count++;
311 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
312 Frame_Count_NotParsedIncluded++;
313 Element_Info1(Ztring::ToZtring(Frame_Count));
314
315 //Filling
316 if (!Status[IsAccepted])
317 File__Analyze::Accept();
318 if (Frame_Count>=Frame_Count_Valid)
319 {
320 //No more need data
321 if (Mode==Mode_LATM)
322 File__Analyze::Accept();
323 File__Analyze::Finish();
324 }
325 FILLING_END();
326}

333bool File_Aac::Synchronize()

335 switch (Mode)

344//---------------------------------------------------------------------------
345bool File_Aac::Synchronize_ADTS()
346{
347 //Tags
348 bool Tag_Found;
349 if (!File__Tags_Helper::Synchronize(Tag_Found))
350 return false;
351 if (Tag_Found)
352 return true;
353
354 //Synchronizing
355 while (Buffer_Offset+6<=Buffer_Size)
356 {
357 while (Buffer_Offset+6<=Buffer_Size && (Buffer[Buffer_Offset ]!=0xFF
358 || (Buffer[Buffer_Offset+1]&0xF6)!=0xF0))
359 Buffer_Offset++;
360
361 if (Buffer_Offset+6<=Buffer_Size)//Testing if size is coherant
362 {
363 //Testing next start, to be sure
364 int16u aac_frame_length=(CC3(Buffer+Buffer_Offset+3)>>5)&0x1FFF;
365 if (IsSub && Buffer_Offset+aac_frame_length==Buffer_Size)
366 break;
367 if (File_Offset+Buffer_Offset+aac_frame_length!=File_Size-File_EndTagSize)
368 {
369 //Padding
370 while (Buffer_Offset+aac_frame_length+2<=Buffer_Size && Buffer[Buffer_Offset+aac_frame_length]==0x00)
371 aac_frame_length++;
372
373 if (IsSub && Buffer_Offset+aac_frame_length==Buffer_Size)
374 break; //while()
375 if (Buffer_Offset+aac_frame_length+2>Buffer_Size)
376 return false; //Need more data
377
378 //Testing
379 if (aac_frame_length<=7 || (CC2(Buffer+Buffer_Offset+aac_frame_length)&0xFFF6)!=0xFFF0)
380 Buffer_Offset++;
381 else
382 {
383 //Testing next start, to be sure
384 if (Buffer_Offset+aac_frame_length+3+3>Buffer_Size)
385 return false; //Need more data
386 int16u aac_frame_length2=(CC3(Buffer+Buffer_Offset+aac_frame_length+3)>>5)&0x1FFF;
387 if (File_Offset+Buffer_Offset+aac_frame_length+aac_frame_length2!=File_Size-File_EndTagSize)
388 {
389 //Padding
390 while (Buffer_Offset+aac_frame_length+aac_frame_length2+2<=Buffer_Size && Buffer[Buffer_Offset+aac_frame_length+aac_frame_length2]==0x00)
391 aac_frame_length2++;
392
393 if (IsSub && Buffer_Offset+aac_frame_length+aac_frame_length2==Buffer_Size)
394 break; //while()
395 if (Buffer_Offset+aac_frame_length+aac_frame_length2+2>Buffer_Size)
396 return false; //Need more data
397
398 //Testing
399 if (aac_frame_length2<=7 || (CC2(Buffer+Buffer_Offset+aac_frame_length+aac_frame_length2)&0xFFF6)!=0xFFF0)
400 Buffer_Offset++;
401 else
402 {
403 //Testing next start, to be sure
404 if (Buffer_Offset+aac_frame_length+aac_frame_length2+3+3>Buffer_Size)
405 return false; //Need more data
406 int16u aac_frame_length3=(CC3(Buffer+Buffer_Offset+aac_frame_length+aac_frame_length2+3)>>5)&0x1FFF;
407 if (File_Offset+Buffer_Offset+aac_frame_length+aac_frame_length2+aac_frame_length3!=File_Size-File_EndTagSize)
408 {
409 //Padding
410 while (Buffer_Offset+aac_frame_length+aac_frame_length2+aac_frame_length3+2<=Buffer_Size && Buffer[Buffer_Offset+aac_frame_length+aac_frame_length2+aac_frame_length3]==0x00)
411 aac_frame_length3++;
412
413 if (IsSub && Buffer_Offset+aac_frame_length+aac_frame_length2+aac_frame_length3==Buffer_Size)
414 break; //while()
415 if (Buffer_Offset+aac_frame_length+aac_frame_length2+aac_frame_length3+2>Buffer_Size)
416 return false; //Need more data
417
418 //Testing
419 if (aac_frame_length3<=7 || (CC2(Buffer+Buffer_Offset+aac_frame_length+aac_frame_length2+aac_frame_length3)&0xFFF6)!=0xFFF0)
420 Buffer_Offset++;
421 else
422 break; //while()
423 }
424 else
425 break; //while()
426 }
427 }
428 else
429 break; //while()
430 }
431 }
432 else
433 break; //while()
434 }
435 }
436
437 //Parsing last bytes if needed
438 if (Buffer_Offset+6>Buffer_Size)
439 {
440 if (Buffer_Offset+5==Buffer_Size && (CC2(Buffer+Buffer_Offset)&0xFFF6)!=0xFFF0)
441 Buffer_Offset++;
442 if (Buffer_Offset+4==Buffer_Size && (CC2(Buffer+Buffer_Offset)&0xFFF6)!=0xFFF0)
443 Buffer_Offset++;
444 if (Buffer_Offset+3==Buffer_Size && (CC2(Buffer+Buffer_Offset)&0xFFF6)!=0xFFF0)
445 Buffer_Offset++;
446 if (Buffer_Offset+2==Buffer_Size && (CC2(Buffer+Buffer_Offset)&0xFFF6)!=0xFFF0)
447 Buffer_Offset++;
448 if (Buffer_Offset+1==Buffer_Size && CC1(Buffer+Buffer_Offset)!=0xFF)
449 Buffer_Offset++;
450 return false;
451 }
452
453 //Synched is OK
454 Mode=Mode_ADTS;
455 return true;
456}

458//---------------------------------------------------------------------------
459bool File_Aac::Synchronize_LATM()
460{
461 //Synchronizing
462 while (Buffer_Offset+3<=Buffer_Size)
463 {
464 while (Buffer_Offset+3<=Buffer_Size && (Buffer[Buffer_Offset ]!=0x56
465 || (Buffer[Buffer_Offset+1]&0xE0)!=0xE0))
466 Buffer_Offset++;
467
468 if (Buffer_Offset+3<=Buffer_Size)//Testing if size is coherant
469 {
470 //Testing next start, to be sure
471 int16u audioMuxLengthBytes=CC2(Buffer+Buffer_Offset+1)&0x1FFF;
472 if (IsSub && Buffer_Offset+3+audioMuxLengthBytes==Buffer_Size)
473 break;
474 if (File_Offset+Buffer_Offset+3+audioMuxLengthBytes!=File_Size)
475 {
476 if (Buffer_Offset+3+audioMuxLengthBytes+3>Buffer_Size)
477 return false; //Need more data
478
479 //Testing
480 if ((CC2(Buffer+Buffer_Offset+3+audioMuxLengthBytes)&0xFFE0)!=0x56E0)
481 Buffer_Offset++;
482 else
483 {
484 //Testing next start, to be sure
485 int16u audioMuxLengthBytes2=CC2(Buffer+Buffer_Offset+3+audioMuxLengthBytes+1)&0x1FFF;
486 if (File_Offset+Buffer_Offset+3+audioMuxLengthBytes+3+audioMuxLengthBytes2!=File_Size)
487 {
488 if (Buffer_Offset+3+audioMuxLengthBytes+3+audioMuxLengthBytes2+3>Buffer_Size)
489 return false; //Need more data
490
491 //Testing
492 if ((CC2(Buffer+Buffer_Offset+3+audioMuxLengthBytes+3+audioMuxLengthBytes2)&0xFFE0)!=0x56E0)
493 Buffer_Offset++;
494 else
495 break; //while()
496 }
497 else
498 break; //while()
499 }
500 }
501 else
502 break; //while()
503 }
504 }
505
506
507 //Synchronizing
508 while (Buffer_Offset+2<=Buffer_Size && (Buffer[Buffer_Offset ]!=0x56
509 || (Buffer[Buffer_Offset+1]&0xE0)!=0xE0))
510 Buffer_Offset++;
511 if (Buffer_Offset+2>=Buffer_Size)
512 return false;
513
514 //Synched is OK
515 Mode=Mode_LATM;
516 return true;
517}

679//---------------------------------------------------------------------------
680void File_Aac::Data_Parse()
681{
682 if (FrameSize_Min>Header_Size+Element_Size)
683 FrameSize_Min=Header_Size+Element_Size;
684 if (FrameSize_Max<Header_Size+Element_Size)
685 FrameSize_Max=Header_Size+Element_Size;
686
687 switch (Mode)
688 {
689 case Mode_ADTS : Data_Parse_ADTS(); break;
690 case Mode_LATM : Data_Parse_LATM(); break;
691 default : ; //No header
692 }
693
694 FILLING_BEGIN();
695 //Counting
696 if (File_Offset+Buffer_Offset+Element_Size==File_Size)
697 Frame_Count_Valid=Frame_Count; //Finish frames in case of there are less than Frame_Count_Valid frames
698
699 #if MEDIAINFO_ADVANCED
700 switch(Mode)
701 {
702 case Mode_LATM :
703 aac_frame_length_Total+=Element_Size;
704 break;
705 default : ;
706 }
707 #endif //MEDIAINFO_ADVANCED
708
709 if (!Status[IsAccepted])
710 File__Analyze::Accept();
711
712 //Filling
713 if (Frame_Count>=Frame_Count_Valid && Config->ParseSpeed<1.0)
714 {
715 //No more need data
716 switch (Mode)
717 {
718 case Mode_ADTS :
719 case Mode_LATM :
720 if (!Status[IsFilled])
721 {
722 Fill();
723 if (!IsSub)
724 File__Tags_Helper::Finish();
725 }
726 break;
727 default : ; //No header
728 }
729
730 }
731
732 TS_Add(frame_length);
733 FILLING_END();
734}
