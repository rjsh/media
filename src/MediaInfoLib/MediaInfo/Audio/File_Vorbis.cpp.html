38//---------------------------------------------------------------------------
39int8u ilog(int32u Value)
40{
41 int8u ToReturn=0;
42 while(Value)
43 {
44 ToReturn++;
45 Value>>=1;
46 }
47 return ToReturn;
48}

76//---------------------------------------------------------------------------
77void File_Vorbis::Identification()
78{
79 Element_Name("Identification");
80
81 //Parsing
82 int32u Version, SamplingRate, BitRate_Maximum, BitRate_Nominal, BitRate_Minimum;
83 int8u Channels;
84 Skip_B1 ( "Signature");
85 Skip_Local(6, "Signature");
86 Get_L4 (Version, "Version");
87 if (Version>0)
88 return; //Not supported
89 Get_L1 (Channels, "Channels");
90 Get_L4 (SamplingRate, "SamplingRate");
91 Get_L4 (BitRate_Maximum, "BitRate_Maximum");
92 Get_L4 (BitRate_Nominal, "BitRate_Nominal");
93 Get_L4 (BitRate_Minimum, "BitRate_Minimum");
94 BS_Begin();
95 Skip_BS(4, "BlockSize_0"); //2^Value
96 Skip_BS(4, "BlockSize_1"); //2^Value
97 BS_End();
98 Skip_L1( "Framing");
99
100 //Filling
101 FILLING_BEGIN();
102 Accept("Vorbis");
103
104 Stream_Prepare(Stream_Audio);
105 Fill(Stream_Audio, StreamPos_Last, Audio_Format, "Vorbis");
106 Fill(Stream_Audio, StreamPos_Last, Audio_Codec, "Vorbis");
107 if (BitRate_Maximum!=0 && BitRate_Maximum<0x80000000) //This is a signed value, and negative values are not OK
108 Fill(Stream_Audio, StreamPos_Last, Audio_BitRate_Maximum, BitRate_Maximum);
109 if (BitRate_Nominal!=0 && BitRate_Nominal<0x80000000) //This is a signed value, and negative values are not OK
110 Fill(Stream_Audio, StreamPos_Last, Audio_BitRate, BitRate_Nominal);
111 if (BitRate_Minimum!=0 && BitRate_Minimum<0x80000000) //This is a signed value, and negative values are not OK
112 Fill(Stream_Audio, StreamPos_Last, Audio_BitRate_Minimum, BitRate_Minimum);
113 if (BitRate_Nominal && BitRate_Maximum==BitRate_Nominal && BitRate_Nominal==BitRate_Minimum)
114 Fill(Stream_Audio, StreamPos_Last, Audio_BitRate_Mode, "CBR");
115 else
116 Fill(Stream_Audio, StreamPos_Last, Audio_BitRate_Mode, "VBR");
117 Fill(Stream_Audio, StreamPos_Last, Audio_Channel_s_, Channels);
118 Fill(Stream_Audio, StreamPos_Last, Audio_SamplingRate, SamplingRate);
119 FILLING_END();
120}

122//---------------------------------------------------------------------------
123void File_Vorbis::Setup()
124{
125 Element_Name("Setup");
126
127 //Parsing
128 Skip_Local(6, "Signature");
129 int32u codebook, codebook_dimensions, codebook_entries, ordered, codebook_lookup_type;
130 int8u vorbis_codebook_count;
131 Get_L1 (vorbis_codebook_count, "vorbis_codebook_count");
132 BS_Begin_LE(); //Vorbis bitstream is Little Endian
133 vorbis_codebook_count+=1;
134 for (int Pos=0; Pos<vorbis_codebook_count; Pos++)
135 {
136 Element_Begin1("codebook");
137 Get_T4 (24, codebook, "codebook");
138 if (codebook!=0x564342)
139 return;
140 Get_BT (16, codebook_dimensions, "codebook_dimensions");
141 Get_BT (24, codebook_entries, "codebook_entries");
142 Get_BT (1, ordered, "ordered");
143 if (!ordered)
144 {
145 int32u sparse;
146 Get_BT (1, sparse, "sparse");
147 for (int32u Pos2=0; Pos2<codebook_entries; Pos2++)
148 {
149 if (sparse)
150 {
151 int32u flag;
152 Get_BT (1, flag, "flag");
153 if (flag)
154 {
155 Info_BT(5, length, "length");
156 }
157 }
158 else
159 {
160 Info_BT(5, length, "length");
161 }
162 }
163 }
164 else
165 {
166 Skip_BT(5, "length");
167 int32u num;
168 for(int32u i=0; i<codebook_entries; )
169 {
170 Get_BT (ilog(codebook_entries-i), num, "num");
171 for(int32u j=0; j<num && i<codebook_entries; j++, i++);
172 }
173 }
174 Get_BT (4, codebook_lookup_type, "codebook_lookup_type");
175 if (codebook_lookup_type>2)
176 return; //Not decodable
177 if (codebook_lookup_type>0)
178 {
179 int8u codebook_value_bits;
180 Info_BT(32, codebook_minimum_value, "codebook_minimum_value");
181 Info_BT(32, codebook_delta_value, "codebook_delta_value");
182 Get_T1 ( 4, codebook_value_bits, "codebook_value_bits");
183 codebook_value_bits++;
184 Info_BT( 1, codebook_sequence_p, "codebook_sequence_p");
185 int32s vals;
186 if (codebook_lookup_type==1)
187 {
188 vals=(int32u)floor(pow((float)codebook_entries,1.f/codebook_dimensions));
189 for (;;)
190 {
191 int32u acc=1, acc1=1;
192 for(int32u i=0; i<codebook_dimensions; i++)
193 {
194 acc*=vals;
195 acc1*=vals+1;
196 }
197 if(acc<=codebook_entries && acc1>codebook_entries)
198 break;
199 else if(acc>codebook_entries)
200 vals--;
201 else
202 vals++;
203 }
204 }
205 else //codebook_lookup_type==2
206 vals=codebook_entries*codebook_dimensions;
207 int32u codebook_multiplicands;
208 for(int i=0; i<vals; i++)
209 Get_BT (codebook_value_bits, codebook_multiplicands, "codebook_multiplicands");
210 }
211 Element_End0();
212 }
213
214 //Time domain transforms
215 int32u vorbis_time_count;
216 Get_BT (6, vorbis_time_count, "vorbis_time_count");
217 for (int32u Pos=0; Pos<vorbis_time_count+1; Pos++)
218 Skip_BT(16, "zero");
219
220 //Floors
221 int32u vorbis_floor_count;
222 Get_BT (6, vorbis_floor_count, "vorbis_floor_count");
223 for (int32u Pos=0; Pos<vorbis_floor_count; Pos++)
224 {
225 int16u vorbis_floor_types;
226 Get_T2(16, vorbis_floor_types, "vorbis_floor_types");
227
228 FILLING_BEGIN();
229 Fill(Stream_Audio, 0, Audio_Format_Settings_Floor, vorbis_floor_types);
230 Fill(Stream_Audio, 0, Audio_Codec_Settings_Floor, vorbis_floor_types);
231 if (vorbis_floor_types==0)
232 {
233 Fill(Stream_Audio, 0, Audio_Format_Settings, "Floor0");
234 Fill(Stream_Audio, 0, Audio_Codec_Settings, "Floor0");
235 }
236 FILLING_END();
237 //Must continue parsing...
238 }
239 BS_End_LE();
240
241 Finish("Vorbis");
242}
