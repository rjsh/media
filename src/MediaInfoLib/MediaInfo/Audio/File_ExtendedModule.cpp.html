35bool File_ExtendedModule::FileHeader_Begin()
36{
37 //Element_Size
38 if (Buffer_Size<38)
39 return false; //Must wait for more data
40
41 if (CC8(Buffer)!=0x457874656E646564LL || CC8(Buffer+8)!=0x204D6F64756C653ALL //"Extended Module: "
42 || CC1(Buffer+16)!=0x20 || CC1(Buffer+37)!=0x1A)
43 {
44 Reject("Extended Module");
45 return false;
46 }
47
48 //All should be OK...
49 return true;
50}

52//***************************************************************************
53// Buffer - Global
54//***************************************************************************
55
56//---------------------------------------------------------------------------
57void File_ExtendedModule::Read_Buffer_Continue()
58{
59 //Parsing
60 Ztring ModuleName, TrackerName;
61 int32u HeaderSize;
62 int16u Length, Channels, Patterns, Instruments, Flags, Tempo, BPM;
63 int8u VersionMinor, VersionMajor;
64 Skip_String(17, "Signature");
65 Get_Local(20, ModuleName, "Module name");
66 Skip_L1( "0x1A");
67 Get_Local(20, TrackerName, "Tracker name");
68 Get_L1 (VersionMinor, "Version (minor)");
69 Get_L1 (VersionMajor, "Version (major)");
70 Get_L4 (HeaderSize, "Header size");
71 Get_L2 (Length, "Song Length");
72 Skip_L2( "Restart position");
73 Get_L2 (Channels, "Number of channels");
74 Get_L2 (Patterns, "Number of patterns");
75 Get_L2 (Instruments, "Number of instruments");
76 Get_L2 (Flags, "Flags");
77 Get_L2 (Tempo, "Tempo");
78 Get_L2 (BPM, "BPM");
79 Skip_XX(256, "Pattern order table");
80
81 FILLING_BEGIN();
82 Accept("Extended Module");
83
84 Fill(Stream_General, 0, General_Format, "Extended Module");
85 Fill(Stream_General, 0, General_Format_Version, Ztring::ToZtring(VersionMajor)+__T(".")+Ztring::ToZtring(VersionMinor/10)+Ztring::ToZtring(VersionMinor%10));
86 Fill(Stream_General, 0, General_Track, ModuleName.Trim(__T(' ')));
87 Fill(Stream_General, 0, General_Encoded_Application, TrackerName.Trim(__T(' ')));
88 Fill(Stream_General, 0, "Tempo", Tempo);
89 Fill(Stream_General, 0, "BPM", BPM);
90
91 Stream_Prepare(Stream_Audio);
92 Fill(Stream_Audio, 0, "Sampler, Channels", Channels);
93 Fill(Stream_Audio, 0, "Sampler, Patterns", Patterns);
94 Fill(Stream_Audio, 0, "Sampler, Instruments", Instruments);
95
96 //No more need data
97 Finish("Extended Module");
98 FILLING_END();
99}
