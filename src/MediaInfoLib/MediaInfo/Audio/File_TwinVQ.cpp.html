43const char* TwinVQ_samplerate(int32u samplerate)
44{
45 switch (samplerate)
46 {
47 case 11 : return "11025";
48 case 22 : return "22050";
49 case 44 : return "44100";
50 default : return "";
51 }
52}

68// Buffer - File header
69//***************************************************************************
70
71//---------------------------------------------------------------------------
72bool File_TwinVQ::FileHeader_Begin()
73{
74 //Testing
75 if (Buffer_Offset+4>Buffer_Size)
76 return false;
77 if (CC4(Buffer+Buffer_Offset)!=0x5457494E) //"TWIN"
78 {
79 Reject("TwinVQ");
80 return false;
81 }
82
83 //All should be OK...
84 return true;
85}
86
87//---------------------------------------------------------------------------
88void File_TwinVQ::FileHeader_Parse()
89{
90 //Parsing
91 Skip_C4( "magic");
92 Skip_Local(8, "version");
93 Skip_B4( "subchunks_size");
94
95 FILLING_BEGIN();
96 Accept("TwinVQ");
97
98 Stream_Prepare(Stream_Audio);
99 Fill(Stream_Audio, 0, Audio_Format, "TwinVQ");
100 Fill(Stream_Audio, 0, Audio_Codec, "TwinVQ");
101 FILLING_END();
102}

104//***************************************************************************
105// Buffer - Per element
106//***************************************************************************
107
108//---------------------------------------------------------------------------
109void File_TwinVQ::Header_Parse()
110{
111 //Parsing
112 int32u id, size;
113 Get_C4 (id, "id");
114 Get_B4 (size, "size");
115
116 //Filling
117 Header_Fill_Code(id, Ztring().From_CC4(id));
118 Header_Fill_Size(8+(id==Elements::DATA?0:size)); //DATA chunk indicates the end of the header, with no chunk size
119}
120
121//---------------------------------------------------------------------------
122void File_TwinVQ::Data_Parse()
123{
124 #define ELEMENT_CASE(_NAME, _DETAIL) \
125 case Elements::_NAME : Element_Info1(_DETAIL); _NAME(); break;
126
127 //Parsing
128 switch (Element_Code)
129 {
130 ELEMENT_CASE(_c__, "Copyright");
131 ELEMENT_CASE(AUTH, "Author");
132 ELEMENT_CASE(COMM, "Mandatory information");
133 ELEMENT_CASE(COMT, "Comment");
134 ELEMENT_CASE(DATA, "Data");
135 ELEMENT_CASE(DSIZ, "Data size");
136 ELEMENT_CASE(FILE, "Filename");
137 ELEMENT_CASE(NAME, "Song title");
138 default : Skip_XX(Element_Size, "Unknown");
139 }
140}

142//***************************************************************************
143// Elements
144//***************************************************************************
145
146//---------------------------------------------------------------------------
147void File_TwinVQ::COMM()
148{
149 //Parsing
150 int32u channel_mode, bitrate, samplerate;
151 Get_B4 (channel_mode, "channel_mode");
152 Get_B4 (bitrate, "bitrate");
153 Get_B4 (samplerate, "samplerate");
154 Skip_B4( "security_level");
155
156 //Filling
157 Fill(Stream_Audio, 0, Audio_Channel_s_, channel_mode+1);
158 Fill(Stream_Audio, 0, Audio_BitRate, bitrate*1000);
159 Fill(Stream_Audio, 0, Audio_SamplingRate, TwinVQ_samplerate(samplerate));
160 if (!IsSub && File_Size!=(int64u)-1)
161 Fill(Stream_Audio, 0, Audio_StreamSize, File_Size);
162}
