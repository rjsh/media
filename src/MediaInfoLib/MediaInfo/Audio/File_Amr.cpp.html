74//---------------------------------------------------------------------------
75bool File_Amr::FileHeader_Begin()
76{
77 if (!Codec.empty()) //Test of header only if it is a file --> The codec field is empty
78 return true;
79
80 //Testing
81 if (Buffer_Size<5)
82 return false; //Must wait for more data
83 if (CC5(Buffer)!=0x2321414D52LL) //"#!AMR"
84 {
85 Reject("AMR");
86 return false;
87 }
88
89 //All should be OK...
90 return true;
91}

93//***************************************************************************
94// Streams management
95//***************************************************************************
96
97//---------------------------------------------------------------------------
98void File_Amr::Streams_Fill()
99{
100 Stream_Prepare(Stream_Audio);
101 Fill(Stream_Audio, 0, Audio_Format, "AMR");
102 Fill(Stream_Audio, 0, Audio_Codec, "AMR");
103 if (!Codec.empty())
104 {
105 if (MediaInfoLib::Config.CodecID_Get(Stream_Audio, InfoCodecID_Format_Mpeg4, Codec, InfoCodecID_Profile)==__T("Narrow band"))
106 IsWB=false;
107 if (MediaInfoLib::Config.CodecID_Get(Stream_Audio, InfoCodecID_Format_Mpeg4, Codec, InfoCodecID_Profile)==__T("Wide band"))
108 IsWB=true;
109 Channels=1;
110 }
111 if (Channels==0)
112 return; //No more info
113
114 if (Header_Size!=(int64u)-1)
115 Fill(Stream_General, 0, General_StreamSize, Header_Size);
116 Fill(Stream_Audio, 0, Audio_Channel_s_, Channels);
117 if (IsWB)
118 {
119 Fill(Stream_Audio, 0, Audio_Format_Profile, "Wide band");
120 Fill(Stream_Audio, 0, Audio_Codec, "sawb", Unlimited, true, true);
121 if (Codec.empty()) //If there is a container, trusting the container sampling rate
122 Fill(Stream_Audio, 0, Audio_SamplingRate, 16000);
123 Fill(Stream_Audio, 0, Audio_BitDepth, 14);
124 //Fill(Stream_Audio, 0, Audio_InternetMediaType, "audio/AMR-WB", Unlimited, true, true);
125 }
126 else
127 {
128 Fill(Stream_Audio, 0, Audio_Format_Profile, "Narrow band");
129 Fill(Stream_Audio, 0, Audio_Codec, "samr", Unlimited, true, true);
130 if (Codec.empty()) //If there is a container, trusting the container sampling rate
131 Fill(Stream_Audio, 0, Audio_SamplingRate, 8000);
132 Fill(Stream_Audio, 0, Audio_BitDepth, 13);
133 if (FrameType!=(int8u)-1 && Amr_BitRate[FrameType] && FrameTypes.size()==1)
134 {
135 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "CBR");
136 Fill(Stream_Audio, 0, Audio_BitRate, Amr_BitRate[FrameType]);
137 Fill(Stream_General, 0, General_OverallBitRate, Amr_BitRate[FrameType]);
138 if (File_Size!=(int64u)-1)
139 {
140 Fill(Stream_Audio, 0, Audio_Duration, ((float32)(File_Size-Header_Size))*8*1000/Amr_BitRate[FrameType]);
141 }
142 }
143 }
144}

151//***************************************************************************
152// Buffer - Global

155//---------------------------------------------------------------------------
156void File_Amr::FileHeader_Parse()
157{
158 //From a container
159 if (!Codec.empty())
160 {
161 Accept("AMR");
162 Finish("AMR");
163 return;
164 }
165
166 //Parsing
167 int64u Signature;
168 Skip_C5( "Signature (Common)");
169 Peek_B8(Signature);
170 if ((Signature&0xFF00000000000000LL)==0x0A00000000000000LL) //\n
171 {
172 IsWB=false;
173 Channels=1;
174 }
175 else if ((Signature&0xFFFFFFFFFFFFFF00LL)==0x5F4D43312E300A00LL) //_MC1.0\n
176 {
177 IsWB=false;
178 Channels=2; //Or more, see later
179 }
180 else if ((Signature&0xFFFFFF0000000000LL)==0x2D57420000000000LL) //-WB
181 {
182 Skip_C3( "Signature (WB)");
183 IsWB=true;
184
185 Peek_B8(Signature);
186 if ((Signature&0xFF00000000000000LL)==0x0A00000000000000LL) //\n
187 {
188 Channels=1;
189 }
190 else if ((Signature&0xFFFFFFFFFFFFFF00LL)==0x5F4D43312E300A00LL) //_MC1.0\n
191 {
192 Channels=2; //Or more, see later
193 }
194 }
195 else
196 Channels=0;
197 Skip_B1( "Signature (Carriage return)");
198 /*
199 if (Channels==2) //Mutli-channels
200 {
201 BS_Begin();
202 Skip_S4(28, "Reserved");
203 Get_S1 ( 4, Channels, "Channels");
204 }
205 */
206 Header_Size=(int8u)Element_Offset;
207
208
209 FILLING_BEGIN();
210 Accept("AMR");
211
212 if (Channels!=1 || IsWB)
213 Finish("AMR");
214 FILLING_END();
215}

217//***************************************************************************
218// Buffer - Per element
219//***************************************************************************
220
221//---------------------------------------------------------------------------
222void File_Amr::Header_Parse()
223{
224 BS_Begin();
225 Skip_SB( "Frame Following");
226 Get_S1 ( 4, FrameType, "Frame Type");
227 Skip_SB( "Frame Quality");
228 Skip_SB( "Unknown");
229 Skip_SB( "Unknown");
230 BS_End();
231
232 //Filling
233 if (Amr_BitRate[FrameType]==0)
234 {
235 Finish("AMR");
236 return;
237 }
238 Header_Fill_Size(Amr_BitRate[FrameType]/400);
239 Header_Fill_Code(0, "frame");
240}
241
242//---------------------------------------------------------------------------
243void File_Amr::Data_Parse()
244{
245 Element_Info1(Frame_Number);
246
247 //Parsing
248 Skip_XX(Element_Size, "Data");
249
250
251 FILLING_BEGIN();
252 Frame_Number++;
253 FrameTypes[FrameType]++;
254 if (Frame_Number>=32)
255 Finish("AMR");
256 FILLING_END();
257}
