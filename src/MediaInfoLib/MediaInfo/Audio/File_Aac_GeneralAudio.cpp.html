68void File_Aac::GASpecificConfig ()
69{
70 //Parsing
71 Element_Begin1("GASpecificConfig");
72 bool frameLengthFlag, dependsOnCoreCoder, extensionFlag;
73 Get_SB ( frameLengthFlag, "frameLengthFlag");
74 frame_length=frameLengthFlag==0?1024:960; Param_Info2(frame_length, " bytes");
75 Get_SB ( dependsOnCoreCoder, "dependsOnCoreCoder");
76 if (dependsOnCoreCoder)
77 Skip_S2(14, "coreCoderDelay");
78 Get_SB ( extensionFlag, "extensionFlag");
79 if (channelConfiguration==0)
80 program_config_element();
81 if (audioObjectType==06 || audioObjectType==20)
82 Skip_S1(3, "layerNr");
83 if (extensionFlag)
84 {
85 bool extensionFlag3;
86 if (audioObjectType==22)
87 {
88 Skip_S1( 5, "numOfSubFrame");
89 Skip_S2(11, "layer_length");
90 }
91 if (audioObjectType==17
92 || audioObjectType==19
93 || audioObjectType==20
94 || audioObjectType==23)
95 {
96 Skip_SB( "aacSectionDataResilienceFlag");
97 Skip_SB( "aacScalefactorDataResilienceFlag");
98 Skip_SB( "aacSpectralDataResilienceFlag");
99 }
100 Get_SB ( extensionFlag3, "extensionFlag3");
101 if (extensionFlag3)
102 {
103 Skip_BS(Data_BS_Remain(), "Not implemented");
104 }
105 }
106 Element_End0();
107}

109//---------------------------------------------------------------------------
110void File_Aac::program_config_element()
111{
112 Element_Begin1("program_config_element");
113 Ztring comment_field_data;
114 int8u Channels=0, Channels_Front=0, Channels_Side=0, Channels_Back=0, Channels_LFE=0;
115 int8u num_front_channel_elements, num_side_channel_elements, num_back_channel_elements, num_lfe_channel_elements, num_assoc_data_elements, num_valid_cc_elements, comment_field_bytes;
116 int8u audioObjectType_Temp, sampling_frequency_index_Temp;
117 Skip_S1(4, "element_instance_tag");
118 Get_S1 (2, audioObjectType_Temp, "object_type"); audioObjectType_Temp++; Param_Info1(Aac_audioObjectType(audioObjectType_Temp));
119 Get_S1 (4, sampling_frequency_index_Temp, "sampling_frequency_index"); Param_Info1(Aac_sampling_frequency[sampling_frequency_index_Temp]);
120 Get_S1 (4, num_front_channel_elements, "num_front_channel_elements");
121 Get_S1 (4, num_side_channel_elements, "num_side_channel_elements");
122 Get_S1 (4, num_back_channel_elements, "num_back_channel_elements");
123 Get_S1 (2, num_lfe_channel_elements, "num_lfe_channel_elements");
124 Get_S1 (3, num_assoc_data_elements, "num_assoc_data_elements");
125 Get_S1 (4, num_valid_cc_elements, "num_valid_cc_elements");
126 TEST_SB_SKIP( "mono_mixdown_present");
127 Skip_S1(4, "mono_mixdown_element_number");
128 TEST_SB_END();
129 TEST_SB_SKIP( "stereo_mixdown_present");
130 Skip_S1(4, "stereo_mixdown_element_number");
131 TEST_SB_END();
132 TEST_SB_SKIP( "matrix_mixdown_idx_present");
133 Skip_S1(2, "matrix_mixdown_idx");
134 Skip_SB( "pseudo_surround_enable");
135 TEST_SB_END();
136 bool front1_element_is_cpe=false;
137 if (!num_side_channel_elements && num_back_channel_elements && num_back_channel_elements<3) // Hack: e.g. in case of 5.1,
138 {
139 num_side_channel_elements=num_back_channel_elements;
140 num_back_channel_elements=0;
141 }
142 for (int8u Pos=0; Pos<num_front_channel_elements; Pos++)
143 {
144 Element_Begin1("front_element");
145 bool front_element_is_cpe;
146 Get_SB ( front_element_is_cpe, "front_element_is_cpe");
147 Skip_S1(4, "front_element_tag_select");
148 if (front_element_is_cpe)
149 {
150 Channels_Front+=2;
151 Channels+=2;
152 if (Pos==0)
153 front1_element_is_cpe=true;
154 }
155 else
156 {
157 Channels_Front++;
158 Channels++;
159 }
160 Element_End0();
161 }
162 for (int8u Pos=0; Pos<num_side_channel_elements; Pos++)
163 {
164 Element_Begin1("side_element");
165 bool side_element_is_cpe;
166 Get_SB ( side_element_is_cpe, "side_element_is_cpe");
167 Skip_S1(4, "side_element_tag_select");
168 if (side_element_is_cpe)
169 {
170 Channels_Side+=2;
171 Channels+=2;
172 }
173 else
174 {
175 Channels_Side++;
176 Channels++;
177 }
178 Element_End0();
179 }
180 for (int8u Pos=0; Pos<num_back_channel_elements; Pos++)
181 {
182 Element_Begin1("back_element");
183 bool back_element_is_cpe;
184 Get_SB ( back_element_is_cpe, "back_element_is_cpe");
185 Skip_S1(4, "back_element_tag_select");
186 if (back_element_is_cpe)
187 {
188 Channels_Back+=2;
189 Channels+=2;
190 }
191 else
192 {
193 Channels_Back++;
194 Channels++;
195 }
196 Element_End0();
197 }
198 for (int8u Pos=0; Pos<num_lfe_channel_elements; Pos++)
199 {
200 Element_Begin1("lfe_element");
201 Skip_S1(4, "lfe_element_tag_select");
202 Channels_LFE++;
203 Channels++;
204 Element_End0();
205 }
206 for (int8u Pos=0; Pos<num_assoc_data_elements; Pos++)
207 {
208 Element_Begin1("assoc_data_element");
209 Skip_S1(4, "assoc_data_element_tag_select");
210 Element_End0();
211 }
212 for (int8u Pos=0; Pos<num_valid_cc_elements; Pos++)
213 {
214 Element_Begin1("valid_cc_element");
215 Skip_SB( "cc_element_is_ind_sw");
216 Skip_S1(4, "valid_cc_element_tag_select");
217 Element_End0();
218 }
219 BS_End(); //Byte align
220 Get_B1 (comment_field_bytes, "comment_field_bytes");
221 if (comment_field_bytes)
222 Get_Local(comment_field_bytes, comment_field_data, "comment_field_data");
223 BS_Begin(); //The stream needs continuity in the bitstream
224 Element_End0();
225
226 //Filling
227 Ztring Channels_Positions, Channels_Positions2, ChannelLayout;
228 switch (Channels_Front)
229 {
230 case 0 : break;
231 case 1 : Channels_Positions+=__T("Front: C"); ChannelLayout+=__T("C "); break;
232 case 2 : Channels_Positions+=__T("Front: L R"); ChannelLayout+=__T("L R "); break;
233 case 3 : Channels_Positions+=__T("Front: L C R"); ChannelLayout+=num_front_channel_elements==2?(front1_element_is_cpe?__T("L R C "):__T("C L R ")):__T("? ? ? "); break;
234 default : Channels_Positions+=__T("Front: "); Channels_Positions+=Ztring::ToZtring(Channels_Front); ChannelLayout+=__T("? "); //Which config?
235 }
236 switch (Channels_Side)
237 {
238 case 0 : break;
239 case 1 : Channels_Positions+=__T(", Side: C"); ChannelLayout+=__T("Cs "); break;
240 case 2 : Channels_Positions+=__T(", Side: L R"); ChannelLayout+=__T("Ls Rs "); break;
241 case 3 : Channels_Positions+=__T(", Side: L C R"); ChannelLayout+=__T("? ? ? "); break;
242 default : Channels_Positions+=__T(", Side: "); Channels_Positions+=Ztring::ToZtring(Channels_Side); ChannelLayout+=__T("? "); //Which config?
243 }
244 switch (Channels_Back)
245 {
246 case 0 : break;
247 case 1 : Channels_Positions+=__T(", Back: C"); ChannelLayout+=__T("Cs "); break;
248 case 2 : Channels_Positions+=__T(", Back: L R"); ChannelLayout+=__T("Rls Rrs "); break;
249 case 3 : Channels_Positions+=__T(", Back: L C R"); ChannelLayout+=__T("Rls Cs Rrs "); break;
250 default : Channels_Positions+=__T(", Back: "); Channels_Positions+=Ztring::ToZtring(Channels_Back); ChannelLayout+=__T("? "); //Which config?
251 }
252 switch (Channels_LFE)
253 {
254 case 0 : break;
255 case 1 : Channels_Positions+=__T(", LFE"); ChannelLayout+=__T("LFE "); break;
256 default : Channels_Positions+=__T(", LFE= "); Channels_Positions+=Ztring::ToZtring(Channels_LFE); ChannelLayout+=__T("? "); //Which config?
257 }
258 Channels_Positions2=Ztring::ToZtring(Channels_Front)+__T('/')
259 +Ztring::ToZtring(Channels_Side)+__T('/')
260 +Ztring::ToZtring(Channels_Back)
261 +(Channels_LFE?__T(".1"):__T(""));
262 if (!ChannelLayout.empty())
263 ChannelLayout.resize(ChannelLayout.size()-1);
264
265 FILLING_BEGIN();
266 //Integrity test
267 if (Aac_sampling_frequency[sampling_frequency_index_Temp]==0 || Channels>24) // TODO: full_2023548870.mp4 is buggy
268 {
269 Trusted_IsNot("sampling frequency / channels");
270 Skip_BS(Data_BS_Remain(), "(Unknown frequency)");
271 return;
272 }
273
274 if (audioObjectType==(int8u)-1)
275 audioObjectType=audioObjectType_Temp;
276 if (sampling_frequency_index==(int8u)-1)
277 sampling_frequency_index=sampling_frequency_index_Temp;
278
279 Infos_General["Comment"]=comment_field_data;
280
281 Infos["Format"].From_Local("AAC");
282 Infos["Format_Profile"].From_Local(Aac_Format_Profile(audioObjectType));
283 Infos["Codec"].From_Local(Aac_audioObjectType(audioObjectType));
284 Infos["SamplingRate"].From_Number(Aac_sampling_frequency[sampling_frequency_index]);
285 Infos["Channel(s)"].From_Number(Channels);
286 Infos["ChannelPositions"]=Channels_Positions;
287 Infos["ChannelPositions/String2"]=Channels_Positions2;
288 Infos["ChannelLayout"]=ChannelLayout;
289
290 if (!Infos["Format_Settings_SBR"].empty())
291 {
292 Infos["Format_Profile"]=__T("HE-AAC");
293 Ztring SamplingRate=Infos["SamplingRate"];
294 Infos["SamplingRate"].From_Number((extension_sampling_frequency_index==(int8u)-1)?(Frequency_b*2):extension_sampling_frequency, 10);
295 if (MediaInfoLib::Config.LegacyStreamDisplay_Get())
296 {
297 Infos["Format_Profile"]+=__T(" / LC");
298 Infos["SamplingRate"]+=__T(" / ")+SamplingRate;
299 }
300 Infos["Format_Settings_SBR"]=__T("Yes (Implicit)");
301 Infos["Codec"]=Ztring().From_Local(Aac_audioObjectType(audioObjectType))+__T("-SBR");
302 }
303
304 if (!Infos["Format_Settings_PS"].empty())
305 {
306 Infos["Format_Profile"]=__T("HE-AACv2");
307 Ztring Channels=Infos["Channel(s)"];
308 Ztring ChannelPositions=Infos["ChannelPositions"];
309 Ztring SamplingRate=Infos["SamplingRate"];
310 Infos["Channel(s)"]=__T("2");
311 Infos["ChannelPositions"]=__T("Front: L R");
312 if (MediaInfoLib::Config.LegacyStreamDisplay_Get())
313 {
314 Infos["Format_Profile"]+=__T(" / HE-AAC / LC");
315 Infos["Channel(s)"]+=__T(" / ")+Channels+__T(" / ")+Channels;
316 Infos["ChannelPositions"]+=__T(" / ")+ChannelPositions+__T(" / ")+ChannelPositions;
317 Infos["SamplingRate"]=Ztring().From_Number((extension_sampling_frequency_index==(int8u)-1)?(Frequency_b*2):extension_sampling_frequency, 10)+__T(" / ")+SamplingRate;
318 }
319 Infos["Format_Settings_PS"]=__T("Yes (Implicit)");
320 Ztring Codec=Retrieve(Stream_Audio, StreamPos_Last, Audio_Codec);
321 Infos["Codec"]=Ztring().From_Local(Aac_audioObjectType(audioObjectType))+__T("-SBR-PS");
322 }
323 FILLING_END();
324}

330//---------------------------------------------------------------------------
331void File_Aac::raw_data_block()
332{
333 if (sampling_frequency_index>=13)
334 {
335 Trusted_IsNot("(Problem)");
336 Skip_BS(Data_BS_Remain(), "(Problem)");
337 return;
338 }
339
340 if (audioObjectType!=2)
341 {
342 Skip_BS(Data_BS_Remain(), "Data");
343 return; //We test only AAC LC
344 }
345
346 //Parsing
347 Element_Begin1("raw_data_block");
348 int8u id_syn_ele=0, id_syn_ele_Previous;
349 do
350 {
351 Element_Begin0();
352 id_syn_ele_Previous=id_syn_ele;
353 Get_S1 (3, id_syn_ele, "id_syn_ele"); Param_Info1(Aac_id_syn_ele[id_syn_ele]); Element_Name(Aac_id_syn_ele[id_syn_ele]);
354
355 #if MEDIAINFO_TRACE
356 bool Trace_Activated_Save=Trace_Activated;
357 Trace_Activated=false; //It is too big, disabling trace for now for full AAC parsing
358 #endif //MEDIAINFO_TRACE
359
360 switch (id_syn_ele)
361 {
362 case 0x00 : single_channel_element(); break; //ID_SCE
363 case 0x01 : channel_pair_element(); break; //ID_CPE
364 case 0x02 : coupling_channel_element(); break; //ID_CCE
365 case 0x03 : lfe_channel_element(); break; //ID_LFE
366 case 0x04 : data_stream_element(); break; //ID_DSE
367 case 0x05 : program_config_element(); break; //ID_PCE
368 case 0x06 : fill_element(id_syn_ele_Previous); break; //ID_FIL
369 case 0x07 : break; //ID_END
370 default : ; //Can not happen
371 }
372
373 #if MEDIAINFO_TRACE
374 Trace_Activated=Trace_Activated_Save;
375 #endif //MEDIAINFO_TRACE
376
377 Element_End0();
378 }
379 while(Element_IsOK() && Data_BS_Remain() && id_syn_ele!=0x07); //ID_END
380 if (Element_IsOK() && Data_BS_Remain()%8)
381 Skip_S1(Data_BS_Remain()%8, "byte_alignment");
382 Element_End0();
383}

406 Element_Begin1("ms_mask");
407 for (int8u g=0; g<num_window_groups; g++)
408 {

414 Element_End0();
415 }
416 }
417 individual_channel_stream(common_window, false);
418 if (!Element_IsOK())
419 return;
420 individual_channel_stream(common_window, false);

423//---------------------------------------------------------------------------
424void File_Aac::ics_info()
425{
426 //Parsing
427 Element_Begin1("ics_info");
428 Skip_SB( "ics_reserved_bit");
429 Get_S1 (2, window_sequence, "window_sequence"); Param_Info1(Aac_window_sequence[window_sequence]);
430 Skip_SB( "window_shape");
431 if (window_sequence==2) //EIGHT_SHORT_SEQUENCE
432 {
433 Get_S1 (4, max_sfb, "max_sfb");
434 Get_S1 (7, scale_factor_grouping, "scale_factor_grouping");
435 }
436 else
437 {
438 bool predictor_data_present;
439 Get_S1 (6, max_sfb, "max_sfb");
440 Get_SB ( predictor_data_present, "predictor_data_present");
441 if (predictor_data_present)
442 {
443 if (audioObjectType==1) //AAC Main
444 {
445 bool predictor_reset;
446 Get_SB (predictor_reset, "predictor_reset");
447 if (predictor_reset)
448 Skip_S1(5, "predictor_reset_group_number");
449 int8u PRED_SFB_MAX=max_sfb;
450 if (PRED_SFB_MAX>Aac_PRED_SFB_MAX[sampling_frequency_index])
451 PRED_SFB_MAX=Aac_PRED_SFB_MAX[sampling_frequency_index];
452 for (int8u sfb=0; sfb<PRED_SFB_MAX; sfb++)
453 Skip_SB( "prediction_used[sfb]");
454 }
455 else
456 {
457 bool ltp_data_present;
458 Get_SB (ltp_data_present, "ltp_data_present");
459 if (ltp_data_present)
460 ltp_data();
461 if (common_window)
462 {
463 Get_SB (ltp_data_present, "ltp_data_present");
464 if (ltp_data_present)
465 ltp_data();
466 }
467 }
468 }
469 }
470 Element_End0();
471
472 //Calculation of windows
473 switch (window_sequence)
474 {
475 case 0 : //ONLY_LONG_SEQUENCE
476 case 1 : //LONG_START_SEQUENCE
477 case 3 : //LONG_STOP_SEQUENCE
478 num_windows=1;
479 num_window_groups=1;
480 window_group_length[0]=1;
481 num_swb=Aac_swb_offset_long_window[sampling_frequency_index]->num_swb;
482 for (int8u i=0; i<num_swb+1; i++)
483 {
484 if (Aac_swb_offset_long_window[sampling_frequency_index]->swb_offset[i]<frame_length)
485 swb_offset[i]=Aac_swb_offset_long_window[sampling_frequency_index]->swb_offset[i];
486 else
487 swb_offset[i]=frame_length;
488 sect_sfb_offset[0][i]=swb_offset[i];
489 }
490 break;
491 case 2 : //EIGHT_SHORT_SEQUENCE
492 num_windows=8;
493 num_window_groups=1;
494 window_group_length[0]=1;
495 num_swb=Aac_swb_offset_short_window[sampling_frequency_index]->num_swb;
496 for (int8u i=0; i<num_swb + 1; i++)
497 swb_offset[i] = Aac_swb_offset_short_window[sampling_frequency_index]->swb_offset[i];
498 swb_offset[num_swb] = frame_length/8;
499 for (int8u i=0; i<num_windows-1; i++)
500 {
501 if (!(scale_factor_grouping&(1<<(6-i))))
502 {
503 num_window_groups++;
504 window_group_length[num_window_groups-1]=1;
505 }
506 else
507 window_group_length[num_window_groups-1]++;
508 }
509 for (int g = 0; g < num_window_groups; g++)
510 {
511 int8u sect_sfb = 0;
512 int16u offset = 0;
513 for (int8u i=0; i<num_swb; i++)
514 {
515 int16u width = Aac_swb_offset_short_window[sampling_frequency_index]->swb_offset[i+1] - Aac_swb_offset_short_window[sampling_frequency_index]->swb_offset[i];
516 width *= window_group_length[g];
517 sect_sfb_offset[g][sect_sfb++] = offset;
518 offset += width;
519 }
520 sect_sfb_offset[g][sect_sfb] = offset;
521 }
522 break;
523 default: ;
524 }
525}

541//---------------------------------------------------------------------------
542void File_Aac::coupling_channel_element()
543{
544 //Parsing
545 int8u num_coupled_elements;
546 bool ind_sw_cce_flag;
547 Skip_S1(4, "element_instance_tag");
548 Get_SB ( ind_sw_cce_flag, "ind_sw_cce_flag");
549 Get_S1 (3, num_coupled_elements, "num_coupled_elements");
550 size_t num_gain_element_lists=0;
551 for (int8u c=0; c<num_coupled_elements+1; c++)
552 {
553 num_gain_element_lists++;
554 bool cc_target_is_cpe;
555 Get_SB ( cc_target_is_cpe, "cc_target_is_cpe[c]");
556 Skip_S1(4, "cc_target_tag_select[c]");
557 if (cc_target_is_cpe)
558 {
559 bool cc_l, cc_r;
560 Get_SB (cc_l, "cc_l[c]");
561 Get_SB (cc_r, "cc_r[c]");
562 if (cc_l && cc_r)
563 num_gain_element_lists++;
564 }
565 }
566 Skip_SB( "cc_domain");
567 Skip_SB( "gain_element_sign");
568 Skip_S1(2, "gain_element_scale");
569 individual_channel_stream(false, false);
570 if (!Element_IsOK())
571 return;
572 bool cge;
573 for (size_t c=1; c<num_gain_element_lists; c++)
574 {
575 if (ind_sw_cce_flag)
576 cge = true;
577 else
578 Get_SB (cge, "common_gain_element_present[c]");
579 if (cge)
580 hcod_sf( "hcod_sf[common_gain_element[c]]");
581 else
582 {
583 for (int g = 0; g < num_window_groups; g++)
584 {
585 for (int sfb=0; sfb<max_sfb; sfb++)
586 {
587 if (sfb_cb[g][sfb]) //Not ZERO_HCB
588 hcod_sf( "hcod_sf[dpcm_gain_element[c][g][sfb]]");
589 }
590 }
591 }
592 }
593}

624 Skip_S1(8, "[i]");
625 Element_End0();
626}

628//---------------------------------------------------------------------------
629void File_Aac::fill_element(int8u id_syn_ele)
630{
631 //Parsing
632 int8u count;
633 Get_S1 (4, count, "count");
634 size_t cnt=count;
635 if (count==15)
636 {
637 int8u esc_count;
