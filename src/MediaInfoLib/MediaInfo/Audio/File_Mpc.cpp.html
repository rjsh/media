118//***************************************************************************
119// Buffer - File header
120//***************************************************************************
121
122//---------------------------------------------------------------------------
123bool File_Mpc::FileHeader_Begin()
124{
125 //Tags
126 if (!File__Tags_Helper::FileHeader_Begin())
127 return false;
128
129 if (Buffer_Offset+4>Buffer_Size)
130 return false;
131
132 //Test
133 if (CC3(Buffer)!=0x4D502B || (CC1(Buffer+3)&0x0F)!=7) //"MP+" version 7
134 {
135 File__Tags_Helper::Reject("Musepack SV7");
136 return false;
137 }
138
139 return true;
140}
141
142//---------------------------------------------------------------------------
143void File_Mpc::FileHeader_Parse()
144{
145 //Parsing
146 Element_Begin1("SV7 header");
147 Ztring Encoder;
148 int32u FrameCount;
149 int16u TitleGain, AlbumGain;
150 int8u Profile, Link, SampleFreq, EncoderVersion;
151
152 Skip_C3( "Signature");
153 BS_Begin();
154 Skip_S1(4, "PNS");
155 Skip_S1(4, "Version");
156 BS_End();
157
158 Get_L4 (FrameCount, "FrameCount");
159
160 Skip_L2( "MaxLevel");
161 BS_Begin();
162 Get_S1 (4, Profile, "Profile"); Param_Info1(Mpc_Profile[Profile]);
163 Get_S1 (2, Link, "Link"); Param_Info1(Mpc_Link[Link]);
164 Get_S1 (2, SampleFreq, "SampleFreq"); Param_Info1(Mpc_SampleFreq[SampleFreq]);
165 Skip_SB( "IntensityStereo");
166 Skip_SB( "MidSideStereo");
167 Skip_S1(6, "MaxBand");
168 BS_End();
169
170 Skip_L2( "TitlePeak");
171 Get_L2 (TitleGain, "TitleGain"); Param_Info3(((float32)((int16s)TitleGain))/1000, " dB", 2);
172
173 Skip_L2( "AlbumPeak");
174 Get_L2 (AlbumGain, "AlbumGain"); Param_Info3(((float32)((int16s)TitleGain))/1000, " dB", 2);
175
176 BS_Begin();
177 Skip_S2(16, "unused");
178 Skip_S1( 4, "LastFrameLength (part 1)");
179 Skip_SB( "FastSeekingSafe");
180 Skip_S1( 3, "unused");
181 Skip_SB( "TrueGapless");
182 Skip_S1( 7, "LastFrameLength (part 2)");
183 BS_End();
184
185 Get_L1 (EncoderVersion, "EncoderVersion");
186 Encoder.From_Number(((float)EncoderVersion)/100, 2); if (EncoderVersion%10==0); else if (EncoderVersion%2==0) Encoder+=__T(" Beta"); else if (EncoderVersion%2==1) Encoder+=__T(" Alpha"); Param_Info1(Encoder);
187
188 Element_End0();
189
190 FILLING_BEGIN();
191 File__Tags_Helper::Accept("Musepack SV7");
192
193 File__Tags_Helper::Stream_Prepare(Stream_Audio);
194 Fill(Stream_Audio, 0, Audio_SamplingRate, Mpc_SampleFreq[SampleFreq]);
195 Fill(Stream_Audio, 0, Audio_SamplingCount, FrameCount*1152);
196 Fill(Stream_Audio, 0, Audio_Format, "Musepack SV7");
197 Fill(Stream_Audio, 0, Audio_Codec, "SV7");
198 Fill(Stream_Audio, 0, Audio_Codec_Settings, Mpc_Profile[Profile]);
199 Fill(Stream_Audio, 0, Audio_Encoded_Library, Encoder);
200 Fill(Stream_Audio, 0, Audio_BitDepth, 16); //MPC support only 16 bits
201 Fill(Stream_Audio, 0, Audio_Duration, ((int64u)FrameCount)*1152*1000/Mpc_SampleFreq[SampleFreq]);
202 if (FrameCount)
203 Fill(Stream_Audio, 0, Audio_BitRate, (File_Size-25)*8*Mpc_SampleFreq[SampleFreq]/FrameCount/1152);
204
205 //No more need data
206 File__Tags_Helper::Finish("Musepack SV7");
207 FILLING_END();
208}
