53void File_La::Streams_Finish()
54{
55 //Filling
56 int64u CompressedSize=File_Size-TagsSize;
57 float32 CompressionRatio=((float32)UncompressedSize)/CompressedSize;
58
59 Fill(Stream_Audio, 0, Audio_StreamSize, CompressedSize);
60 Fill(Stream_Audio, 0, Audio_Compression_Ratio, CompressionRatio);
61
62 File__Tags_Helper::Streams_Finish();
63}

65//***************************************************************************
66// Buffer - File header
67//***************************************************************************
68
69//---------------------------------------------------------------------------
70bool File_La::FileHeader_Begin()
71{
72 if (!File__Tags_Helper::FileHeader_Begin())
73 return false;
74
75 //Synchro
76 if (Buffer_Offset+2>Buffer_Size)
77 return false;
78 if (CC3(Buffer+Buffer_Offset)!=0x4C4130) //"LA0"
79 {
80 File__Tags_Helper::Reject("LA");
81 return false;
82 }
83
84 return true;
85}
86
87//---------------------------------------------------------------------------
88void File_La::FileHeader_Parse()
89{
90 //Parsing
91 Ztring Major, Minor;
92 int32u SampleRate, Samples, BytesPerSecond, UnCompSize, WAVEChunk, FmtSize, FmtChunk, CRC32;
93 int16u RawFormat, Channels, BytesPerSample, BitsPerSample;
94
95 Skip_Local(2, "signature");
96 Get_Local (1, Major, "major_version");
97 Get_Local (1, Minor, "minor_version");
98 Get_L4 (UnCompSize, "uncompressed_size");
99 Get_L4 (WAVEChunk, "chunk");
100 Skip_L4( "fmt_size");
101 Get_L4 (FmtChunk, "fmt_chunk");
102 Get_L4 (FmtSize, "fmt_size");
103 Get_L2 (RawFormat, "raw_format");
104 Get_L2 (Channels, "channels"); Param_Info2(Channels, " channel(s)");
105 Get_L4 (SampleRate, "sample_rate");
106 Get_L4 (BytesPerSecond, "bytes_per_second");
107 Get_L2 (BytesPerSample, "bytes_per_sample");
108 Get_L2 (BitsPerSample, "bits_per_sample");
109 Get_L4 (Samples, "samples");
110 Skip_L1( "flags");
111 Get_L4 (CRC32, "crc");
112
113 FILLING_BEGIN();
114 if (SampleRate==0)
115 return;
116 Duration=((int64u)Samples/Channels)*1000/SampleRate; // Seems that it's samples per channels otherwise Duration is doubled ??!!
117 if (Duration==0)
118 return;
119 UncompressedSize=((int64u)Samples)*Channels*(BitsPerSample/8);
120 if (UncompressedSize==0)
121 return;
122
123 File__Tags_Helper::Accept("LA");
124 Fill(Stream_General, 0, General_Format_Version, Major+__T('.')+Minor);
125
126 File__Tags_Helper::Stream_Prepare(Stream_Audio);
127 Fill(Stream_Audio, 0, Audio_Format, "LA");
128 Fill(Stream_Audio, 0, Audio_Codec, "LA");
129 Fill(Stream_Audio, 0, Audio_Format_Version, Major+__T('.')+Minor);
130 Fill(Stream_Audio, 0, Audio_BitDepth, BitsPerSample);
131 Fill(Stream_Audio, 0, Audio_Channel_s_, Channels);
132 Fill(Stream_Audio, 0, Audio_SamplingRate, SampleRate);
133 Fill(Stream_Audio, 0, Audio_Duration, Duration);
134
135 //No more need data
136 File__Tags_Helper::Finish("LA");
137 FILLING_END();
138}
