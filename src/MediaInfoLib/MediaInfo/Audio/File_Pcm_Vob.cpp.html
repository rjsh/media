56const char* Pcm_VOB_ChannelsPositions(int8u channel_assignment)
57{
58 switch (channel_assignment)
59 {
60 case 1 : return "Front: C"; //1 channel
61 case 3 : return "Front: L R"; //2 channels
62 case 4 : return "Front: L C R"; //3 channels
63 case 5 : return "Front: L R, LFE"; //3 channels
64 case 6 : return "Front: L C R, LFE"; //4 channels
65 case 7 : return "Front: L R, Side: L R"; //4 channels
66 case 8 : return "Front: L C R, Side: L R"; //5 channels
67 case 9 : return "Front: L C R, Side: L R, LFE"; //6 channels
68 case 10 : return "Front: L C R, Side: L R, Back: L R"; //7 channels
69 case 11 : return "Front: L C R, Side: L R, Back: L R, LFE"; //8 channels
70 default : return "";
71 }
72}
73
74//---------------------------------------------------------------------------
75const char* Pcm_VOB_ChannelsPositions2(int8u channel_assignment)
76{
77 switch (channel_assignment)
78 {
79 case 1 : return "1/0/0.0"; //1 channel
80 case 3 : return "2/0/0.0"; //2 channels
81 case 4 : return "3/0/0.0"; //3 channels
82 case 5 : return "2/0/0.1"; //3 channels
83 case 6 : return "3/0/0.1"; //4 channels
84 case 7 : return "2/2/0.0"; //4 channels
85 case 8 : return "3/2/0.0"; //5 channels
86 case 9 : return "3/2/0.1"; //6 channels
87 case 10 : return "3/2/2.0"; //7 channels
88 case 11 : return "3/2/2.1"; //8 channels
89 default : return "";
90 }
91}

110//---------------------------------------------------------------------------
111void File_Pcm_Vob::Streams_Fill()
112{
113 Stream_Prepare(Stream_Audio);
114 Fill(Stream_Audio, 0, Audio_Format, "PCM");
115 Fill(Stream_Audio, 0, Audio_Codec, "PCM");
116 Fill(Stream_Audio, 0, Audio_Codec_Family, "PCM");
117 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "CBR");
118
119 Fill(Stream_Audio, 0, Audio_BitDepth, Pcm_VOB_BitDepth[BitDepth]);
120 Fill(Stream_Audio, 0, Audio_SamplingRate, Pcm_VOB_Frequency[Frequency]);
121 Fill(Stream_Audio, 0, Audio_Channel_s_, NumberOfChannelsMinusOne+1);
122 Fill(Stream_Audio, 0, Audio_ChannelPositions, Pcm_VOB_ChannelsPositions(NumberOfChannelsMinusOne+1));
123 Fill(Stream_Audio, 0, Audio_ChannelPositions_String2, Pcm_VOB_ChannelsPositions2(NumberOfChannelsMinusOne+1));
124 Fill(Stream_Audio, 0, Audio_BitRate, Pcm_VOB_Frequency[Frequency]*(NumberOfChannelsMinusOne+1)*16);
125
126 //PCM Signed 16 bits Big Endian, Interleavement is for 2 samples*2 channels L0-1/L0-0/R0-1/R0-0/L1-1/L1-0/R1-1/R1-0/L0-2/R0-2/L1-2/R1-2, http://wiki.multimedia.cx/index.php?title=PCM
127 Fill(Stream_Audio, 0, Audio_Format_Settings, "Big");
128 Fill(Stream_Audio, 0, Audio_Format_Settings_Endianness, "Big");
129 Fill(Stream_Audio, 0, Audio_Codec_Settings, "Big");
130 Fill(Stream_Audio, 0, Audio_Codec_Settings_Endianness, "Big");
131 Fill(Stream_Audio, 0, Audio_Format_Settings, "Signed");
132 Fill(Stream_Audio, 0, Audio_Format_Settings_Sign, "Signed");
133 Fill(Stream_Audio, 0, Audio_Codec_Settings, "Signed");
134 Fill(Stream_Audio, 0, Audio_Codec_Settings_Sign, "Signed");
135}

137//***************************************************************************
138// Buffer - Global
139//***************************************************************************
140
141//---------------------------------------------------------------------------
142void File_Pcm_Vob::Read_Buffer_Continue()
143{
144 if (Buffer_Size==0)
145 return;
146
147 //Parsing
148 Skip_B1( "Frame number");
149 Skip_B2( "Bytes to skip (+1?)");
150 Skip_B1( "Unknown");
151 BS_Begin();
152 Get_S1 (2, BitDepth, "Bit depth"); Param_Info1(Pcm_VOB_BitDepth[BitDepth]);
153 Get_S1 (2, Frequency, "Frequency"); Param_Info1(Pcm_VOB_Frequency[Frequency]);
154 Skip_SB( "Unknown");
155 Get_S1 (3, NumberOfChannelsMinusOne, "Number of channels (minus 1)");
156 BS_End();
157 Skip_B1( "Start code");
158
159 #if MEDIAINFO_DEMUX
160 if (Config->Demux_PCM_20bitTo16bit_Get() && BitDepth==1) //20-bit
161 {
162 int8u* Info=new int8u[(size_t)((Element_Size-6)*4/5)];
163 size_t Info_Offset=0;
164
165 while (Element_Offset+5*(NumberOfChannelsMinusOne+1)<=Element_Size)
166 {
167 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
168
169 std::memcpy(Info+Info_Offset, Buffer+Buffer_Pos, 4*(NumberOfChannelsMinusOne+1));
170
171 Info_Offset+=4*(NumberOfChannelsMinusOne+1);
172 Element_Offset+=5*(NumberOfChannelsMinusOne+1);
173 }
174 Element_Offset=6;
175
176 FrameInfo.PTS=FrameInfo.DTS;
177 if (Pcm_VOB_Frequency[Frequency])
178 FrameInfo.DUR=(Element_Size-6)/5*1000000000/Pcm_VOB_Frequency[Frequency];
179 Demux_random_access=true;
180 Element_Code=(int64u)-1;
181 Demux(Info, Info_Offset, ContentType_MainStream);
182
183 delete[] Info;
184 }
185 else
186 {
187 Demux_Offset=Buffer_Offset+(size_t)Element_Size;
188 Buffer_Offset+=6; //Header is dropped
189 Demux_UnpacketizeContainer_Demux();
190 Buffer_Offset-=6;
191 }
192 #endif //MEDIAINFO_DEMUX
193
194 Skip_XX(Element_Size-6, "Data");
195
196 FILLING_BEGIN();
197 Frame_Count++;
198 Frame_Count_InThisBlock++;
199 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
200 Frame_Count_NotParsedIncluded++;
201 if (FrameInfo.DTS!=(int64u)-1 && FrameInfo.DUR!=(int64u)-1)
202 {
203 FrameInfo.DTS+=FrameInfo.DUR;
204 FrameInfo.PTS=FrameInfo.DTS;
205 }
206
207 if (!Status[IsAccepted])
208 {
209 Accept();
210 Finish();
211 }
212 FILLING_END();
213}
