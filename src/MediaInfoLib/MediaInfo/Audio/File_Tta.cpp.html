52//---------------------------------------------------------------------------
53void File_Tta::Streams_Finish()
54{
55 //Filling
56 int64u CompressedSize=File_Size-TagsSize;
57 float32 CompressionRatio=((float32)UncompressedSize)/CompressedSize;
58
59 Fill(Stream_Audio, 0, Audio_StreamSize, CompressedSize);
60 Fill(Stream_Audio, 0, Audio_Compression_Ratio, CompressionRatio);
61 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "VBR");
62
63 File__Tags_Helper::Streams_Finish();
64}

66//***************************************************************************
67// Buffer - File header
68//***************************************************************************
69
70//---------------------------------------------------------------------------
71bool File_Tta::FileHeader_Begin()
72{
73 if (!File__Tags_Helper::FileHeader_Begin())
74 return false;
75
76 //Synchro
77 if (Buffer_Offset+4>Buffer_Size)
78 return false;
79 if (CC4(Buffer+Buffer_Offset)!=0x54544131) //"TTA1"
80 {
81 File__Tags_Helper::Reject("TTA");
82 return false;
83 }
84
85 return true;
86}
87
88//---------------------------------------------------------------------------
89void File_Tta::FileHeader_Parse()
90{
91 //Parsing
92 int32u SampleRate, Samples, CRC32;
93 int16u AudioFormat, Channels, BitsPerSample;
94 Skip_C4( "Signature");
95 Get_L2 (AudioFormat, "AudioFormat");
96 Get_L2 (Channels, "NumChannels");
97 Get_L2 (BitsPerSample, "BitsPerSample");
98 Get_L4 (SampleRate, "SampleRate");
99 Get_L4 (Samples, "DataLength");
100 Get_L4 (CRC32, "CRC32");
101
102 FILLING_BEGIN();
103 if (SampleRate==0)
104 return;
105 Duration=((int64u)Samples)*1000/SampleRate;
106 if (Duration==0)
107 return;
108 UncompressedSize=((int64u)Samples)*Channels*(BitsPerSample/8);
109 if (UncompressedSize==0)
110 return;
111
112 //Filling data
113 File__Tags_Helper::Accept("TTA");
114
115 File__Tags_Helper::Stream_Prepare(Stream_Audio);
116 Fill(Stream_Audio, 0, Audio_Format, "TTA");
117 Fill(Stream_Audio, 0, Audio_Codec, "TTA ");
118 Fill(Stream_Audio, 0, Audio_BitDepth, BitsPerSample);
119 Fill(Stream_Audio, StreamPos_Last, Audio_Channel_s_, Channels);
120 Fill(Stream_Audio, StreamPos_Last, Audio_SamplingRate, SampleRate);
121 Fill(Stream_Audio, 0, Audio_Duration, Duration);
122 FILLING_END();
123
124 //No more need data
125 File__Tags_Helper::Finish("TTA");
126}
