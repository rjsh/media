35void File_Adpcm::Read_Buffer_Continue()
36{
37 //It is impossible to detect... Default is no detection, only filling
38
39 //Filling
40 Accept("ADPCM");
41
42 Stream_Prepare(Stream_Audio);
43 Fill(Stream_General, 0, Audio_Format, "ADPCM");
44 Fill(Stream_General, 0, Audio_Codec, "ADPCM");
45 Ztring Profile, Firm;
46 if (Codec==__T("alaw")) {Profile=__T("A-Law");}
47 else if (Codec==__T("ulaw")) {Profile=__T("U-Law");}
48 else if (Codec==__T("ima4")) { Firm=__T("IMA");}
49 else if (Codec==__T("6")) {Profile=__T("A-Law");}
50 else if (Codec==__T("7")) {Profile=__T("U-Law");}
51 else if (Codec==__T("102")) {Profile=__T("A-Law");}
52 else if (Codec==__T("171")) {Profile=__T("U-Law"); Firm=__T("Unisys");}
53 else if (Codec==__T("172")) {Profile=__T("A-Law"); Firm=__T("Unisys");}
54
55 if (!Profile.empty())
56 Fill(Stream_Audio, 0, Audio_Format_Profile, Profile);
57 if (!Firm.empty())
58 {
59 Fill(Stream_Audio, 0, Audio_Format_Settings, Firm);
60 Fill(Stream_Audio, 0, Audio_Format_Settings_Firm, Firm);
61 Fill(Stream_Audio, 0, Audio_Codec_Settings, Firm);
62 Fill(Stream_Audio, 0, Audio_Codec_Settings_Firm, Firm);
63 }
64 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "CBR");
65
66 //No more need data
67 Finish("ADPCM");
68}
