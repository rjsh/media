123//---------------------------------------------------------------------------
124void File_Aac::CelpHeader ()
125{
126 Element_Begin1("CelpHeader");
127 bool ExcitationMode;
128 Get_SB(ExcitationMode, "ExcitationMode");
129 Skip_SB( "SampleRateMode");
130 Skip_SB( "FineRateControl");
131 if (ExcitationMode == 1/*RPE*/)
132 {
133 Skip_S1(3, "RPE_Configuration");
134 }
135 if (ExcitationMode == 0/*MPE*/)
136 {
137 Skip_S1(5, "MPE_Configuration");
138 Skip_S1(2, "NumEnhLayers");
139 Skip_SB( "BandwidthScalabilityMode");
140 }
141 Element_End0();
142}

329//***************************************************************************
330// Audio Lossless
331//***************************************************************************
332
333//---------------------------------------------------------------------------
334void File_Aac::ALSSpecificConfig()
335{
336 //Not in spec, but something weird in the example I have
337 int32u Junk;
338 while (Data_BS_Remain())
339 {
340 Peek_S4(32, Junk);
341 if (Junk!=0x414C5300)
342 {
343 Skip_SB( "Unknown");
344 }
345 else
346 break;
347 }
348 if (Data_BS_Remain()==0)
349 return; //There is a problem
350
351 Element_Begin1("ALSSpecificConfig");
352 bool chan_config,chan_sort,crc_enabled,aux_data_enabled;
353 int32u samp_freq, samples;
354 int16u channels,frame_length;
355 int8u ra_flag,random_access, file_type;
356 Skip_BS(32,"als_id");
357 Get_BS (32, samp_freq, "samp_freq");
358 Get_BS (32, samples, "samples");
359 Get_S2 (16, channels, "channels"); Param_Info2(channels+1, " channel(s)");
360 Get_S1 (3, file_type, "file_type");
361 Skip_S1(3,"resolution");
362 Skip_SB("floating");
363 Skip_SB("msb_first");
364 Get_S2 (16,frame_length,"frame_length");
365 Get_S1 (8,random_access,"random_access");
366 Get_S1 (2,ra_flag,"ra_flag");
367 Skip_SB("adapt_order");
368 Skip_S1(2,"coef_table");
369 Skip_SB("long_term_prediction");
370 Skip_S2(10,"max_order");
371 Skip_S1(2,"block_switching");
372 Skip_SB("bgmc_mode");
373 Skip_SB("sb_part");
374 Skip_SB("joint_stereo");
375 Skip_SB("mc_coding");
376 Get_SB (chan_config,"chan_config");
377 Get_SB (chan_sort,"chan_sort");
378 Get_SB (crc_enabled,"crc_enabled");
379 Skip_SB("RLSLMS");
380 Skip_BS(5,"(reserved)");
381 Get_SB (aux_data_enabled,"aux_data_enabled");
382 if (chan_config)
383 Skip_S2(16,"chan_config_info");
384 if (chan_sort)
385 {
386 int16u ChBits=(int16u)ceil(log((double)(channels+1))/log((double)2));
387 for (int8u c=0; c<=channels; c++)
388 Skip_BS(ChBits, "chan_pos[c]");
389 }
390 if(Data_BS_Remain()%8)
391 Skip_S1(Data_BS_Remain()%8, "byte_align");
392 BS_End();
393 int32u header_size,trailer_size;
394 Get_B4(header_size, "header_size");
395 Get_B4(trailer_size, "trailer_size");
396 #ifdef MEDIAINFO_RIFF_YES
397 if (file_type==1) //WAVE file
398 {
399 Element_Begin1("orig_header");
400 File_Riff MI;
401 Open_Buffer_Init(&MI);
402 Open_Buffer_Continue(&MI, Buffer+Buffer_Offset+(size_t)Element_Offset, header_size);
403 Element_Offset+=header_size;
404 File__Analyze::Finish(&MI); //No merge of data, only for trace information, because this is the data about the decoded stream, not the encoded stream
405 Element_End0();
406 }
407 else
408 #endif //MEDIAINFO_RIFF_YES
409 Skip_XX(header_size, "orig_header[]");
410
411 Skip_XX(trailer_size, "orig_trailer[]");
412 if (crc_enabled)
413 Skip_B4( "crc");
414 if ((ra_flag == 2) && (random_access > 0))
415 for (int32u f=0; f<((samples-1)/(frame_length+1))+1; f++)
416 Skip_B4( "ra_unit_size[f]");
417 if (aux_data_enabled)
418 {
419 int32u aux_size;
420 Get_B4(aux_size, "aux_size");
421 Skip_XX(aux_size, "aux_data[]");
422 }
423 Element_End0();
424 BS_Begin(); //To be in sync with other objectTypes
425
426 FILLING_BEGIN();
427 //Filling
428 File__Analyze::Stream_Prepare(Stream_Audio);
429 Fill(Stream_Audio, StreamPos_Last, Audio_Channel_s_, channels+1);
430
431 //Forcing default confignuration (something weird in the example I have)
432 channelConfiguration=0;
433 sampling_frequency_index=(int8u)-1;
434 Frequency_b=samp_freq;
435 FILLING_END();
436}
