70//---------------------------------------------------------------------------
71const char* Wvpk_id(int8u ID)
72{
73 switch (ID)
74 {
75 case 0x00 : return "could be used to pad WavPack blocks";
76 case 0x02 : return "decorrelation terms & deltas";
77 case 0x03 : return "initial decorrelation weights";
78 case 0x04 : return "decorrelation sample history";
79 case 0x05 : return "initial entropy variables";
80 case 0x06 : return "entropy variables specific to hybrid mode";
81 case 0x07 : return "info needed for hybrid lossless (wvc) mode";
82 case 0x08 : return "specific info for floating point decode";
83 case 0x09 : return "specific info for decoding integers > 24 bits";
84 case 0x0A : return "normal compressed audio bitstream (wv file)";
85 case 0x0B : return "correction file bitstream (wvc file)";
86 case 0x0C : return "special extended bitstream for floating point data or integers > 24 bit";
87 case 0x0D : return "contains channel count and channel_mask";
88 case 0x21 : return "RIFF header for .wav files (before audio)";
89 case 0x22 : return "RIFF trailer for .wav files (after audio)";
90 case 0x25 : return "some encoding details for info purposes";
91 case 0x26 : return "16-byte MD5 sum of raw audio data";
92 case 0x27 : return "non-standard sampling rate info";
93 default: return "";
94 }

130//---------------------------------------------------------------------------
131void File_Wvpk::Streams_Finish()
132{
133 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "VBR");
134
135 //Specific case
136 if (FromMKV)
137 return;
138
139 //Duration
140 if (SamplingRate<15)
141 {
142 int64u Duration=(((int64u)(block_index_LastFrame+block_samples_LastFrame-block_index_FirstFrame))*1000/Wvpk_SamplingRate[SamplingRate]); //Don't forget the last frame with block_samples...
143 int64u CompressedSize=File_Size-TagsSize;
144 int64u UncompressedSize=Duration*(mono?1:2)*Wvpk_Resolution[(resolution1?1:0)*2+(resolution0?1:0)]*Wvpk_SamplingRate[SamplingRate]/8/1000;
145 float32 CompressionRatio=((float32)UncompressedSize)/CompressedSize;
146 Fill(Stream_Audio, 0, Audio_StreamSize, CompressedSize, 3, true);
147 Fill(Stream_Audio, 0, Audio_Duration, Duration, 10, true);
148 Fill(Stream_Audio, 0, Audio_Compression_Ratio, CompressionRatio, 3, true);
149 }
150
151 File__Tags_Helper::Streams_Finish();
152}

158//---------------------------------------------------------------------------
159bool File_Wvpk::FileHeader_Begin()
160{
161 //Element_Size
162 if (Buffer_Size<2)
163 return false; //Must wait for more data
164
165 if (CC2(Buffer)==0x4D5A) //"MZ"
166 {
167 File__Tags_Helper::Reject("WavPack");
168 return false; //Executable with WavPack data are currently not well supported --> It is preferable to set them as executable
169 }
170
171 //All should be OK...
172 return true;
173}

175//***************************************************************************
176// Buffer - Synchro
177//***************************************************************************
178
179//---------------------------------------------------------------------------
180bool File_Wvpk::Synchronize()
181{
182 //Specific case
183 if (FromMKV)
184 return true;
185
186 //Tags
187 bool Tag_Found;
188 if (!File__Tags_Helper::Synchronize(Tag_Found))
189 return false;
190 if (Tag_Found)
191 return true;
192
193 //Synchronizing
194 while (Buffer_Offset+4<=Buffer_Size)
195 {
196 while (Buffer_Offset+4<=Buffer_Size && (Buffer[Buffer_Offset ]!=0x77
197 || Buffer[Buffer_Offset+1]!=0x76
198 || Buffer[Buffer_Offset+2]!=0x70
199 || Buffer[Buffer_Offset+3]!=0x6B)) //"wvpk"
200 Buffer_Offset++;
201
202 if (Buffer_Offset+4<=Buffer_Size)//Testing if size is coherant
203 {
204 //Testing next start, to be sure
205 size_t Size=LittleEndian2int32u(Buffer+Buffer_Offset+4)+8;
206 if (Buffer_Offset+Size+4>Buffer_Size)
207 return false; //Need more data
208
209 //Testing
210 if (Buffer[Buffer_Offset+Size ]!=0x77
211 || Buffer[Buffer_Offset+Size+1]!=0x76
212 || Buffer[Buffer_Offset+Size+2]!=0x70
213 || Buffer[Buffer_Offset+Size+3]!=0x6B) //"wvpk"
214 Buffer_Offset++;
215 else
216 break; //while()
217 }
218 }
219
220 //Parsing last bytes if needed
221 if (Buffer_Offset+4>Buffer_Size)
222 {
223 if (Buffer_Offset+3==Buffer_Size && CC3(Buffer+Buffer_Offset)!=0x777670) //"wvp"
224 Buffer_Offset++;
225 if (Buffer_Offset+2==Buffer_Size && CC2(Buffer+Buffer_Offset)!=0x7776) //"wv"
226 Buffer_Offset++;
227 if (Buffer_Offset+1==Buffer_Size && CC1(Buffer+Buffer_Offset)!=0x77) //"w"
228 Buffer_Offset++;
229 return false;
230 }
231
232 //Synched is OK
233 return true;
234}

236//---------------------------------------------------------------------------
237bool File_Wvpk::Synched_Test()
238{
239 //Specific case
240 if (FromMKV)
241 return true;
242
243 //Tags
244 if (!File__Tags_Helper::Synched_Test())
245 return false;
246
247 //Must have enough buffer for having header
248 if (Buffer_Offset+3>Buffer_Size)
249 return false;
250
251 //Quick test of synchro
252 if (Buffer[Buffer_Offset ]!=0x77
253 || Buffer[Buffer_Offset+1]!=0x76
254 || Buffer[Buffer_Offset+2]!=0x70
255 || Buffer[Buffer_Offset+3]!=0x6B) //"wvpk"
256 Synched=false;
257
258 //We continue
259 return true;
260}

278//---------------------------------------------------------------------------
279void File_Wvpk::Header_Parse()
280{
281 //Specific cases
282 if (FromMKV)
283 {
284 Header_Fill_Size(Element_Size);
285 Header_Fill_Code(0, "Block");
286 return;
287 }
288
289 //Parsing
290 int32u ckSize;
291 Skip_C4( "ckID");
292 Get_L4 (ckSize, "ckSize");
293
294 //Filling
295 Header_Fill_Size(8+ckSize);
296 Header_Fill_Code(0, "Block");
297}

299//---------------------------------------------------------------------------
300void File_Wvpk::Data_Parse()
301{
302 //Specific
303 if (FromMKV && !FromMKV_CodecPrivateParsed)
304 {
305 //Parsing
306 Get_L2 (version, "version");
307
308 FILLING_BEGIN();
309 FromMKV_CodecPrivateParsed=true;
310 FILLING_END();
311 return;
312 }
313
314 //Counting
315 Frame_Count++;
316
317 //Parsing
318 Element_Begin1("Block Header");
319 if (!FromMKV)
320 Get_L2 (version, "version");
321 if (version/0x100==0x4)
322 {
323 while (Element_Offset<Element_Size)
324 {
325 int32u total_samples=(int32u)-1, block_index=(int32u)-1, block_samples=0, flags, blocksize=(int32u)-1;
326 bool initial_block=true, final_block=true;
327 if (!FromMKV)
328 {
329 Skip_L1( "track_no");
330 Skip_L1( "index_no");
331 Get_L4 (total_samples, "total_samples");
332 Get_L4 (block_index, "block_index");
333 }
334 if (!FromMKV || Element_Offset==0)
335 Get_L4 (block_samples, "block_samples");
336 if (block_samples!=0) //empty frames have other values empty
337 {
338 if (!FromMKV)
339 {
340 if (block_index==0) //Only the frame with block_index==0
341 total_samples_FirstFrame=total_samples; //Note: total_samples is not trustable for a cutted file
342 if (Frame_Count==1)
343 block_index_FirstFrame=block_index; //Save the block_index of the first block
344 block_index_LastFrame=block_index;
345 block_samples_LastFrame=block_samples;
346 }
347 Get_L4 (flags, "flags");
348 Get_Flags (flags, 0, resolution0, "resolution0");
349 Get_Flags (flags, 1, resolution1, "resolution1");
350 Get_Flags (flags, 2, mono, "mono");
351 Get_Flags (flags, 3, hybrid, "hybrid");
352 Get_Flags (flags, 4, joint_stereo, "joint stereo");
353 Get_Flags (flags, 5, cross_channel_decorrelation, "cross-channel decorrelation");
354 Skip_Flags(flags, 6, "hybrid noise shaping");
355 Skip_Flags(flags, 7, "floating point data");
356 Skip_Flags(flags, 8, "extended size integers");
357 Skip_Flags(flags, 9, "hybrid mode parameters control bitrate");
358 Skip_Flags(flags, 10, "hybrid noise balanced between channels");
359 Get_Flags (flags, 11, initial_block, "initial block in sequence");
360 Get_Flags (flags, 12, final_block, "final block in sequence");
361 Skip_Flags(flags, 13, "amount of data left-shift after decode");
362 Skip_Flags(flags, 14, "amount of data left-shift after decode");
363 Skip_Flags(flags, 15, "amount of data left-shift after decode");
364 Skip_Flags(flags, 16, "amount of data left-shift after decode");
365 Skip_Flags(flags, 17, "amount of data left-shift after decode");
366 Skip_Flags(flags, 18, "maximum magnitude of decoded data");
367 Skip_Flags(flags, 19, "maximum magnitude of decoded data");
368 Skip_Flags(flags, 20, "maximum magnitude of decoded data");
369 Skip_Flags(flags, 21, "maximum magnitude of decoded data");
370 Skip_Flags(flags, 22, "maximum magnitude of decoded data");
371 Skip_Flags(flags, 23, "sampling rate");
372 Skip_Flags(flags, 24, "sampling rate");
373 Skip_Flags(flags, 25, "sampling rate");
374 Skip_Flags(flags, 26, "sampling rate"); SamplingRate=(int8u)(((flags>>23)&0xF)); Param_Info1(Wvpk_SamplingRate[SamplingRate]);
375 Skip_Flags(flags, 27, "reserved");
376 Skip_Flags(flags, 28, "reserved");
377 Skip_Flags(flags, 29, "use IIR for negative hybrid noise shaping");
378 Skip_Flags(flags, 30, "false stereo");
379 Skip_Flags(flags, 31, "reserved");
380 }
381 else
382 {
383 Skip_L4( "flags (empty)");
384
385 //Counting
386 Frame_Count--; //This is not a real frame
387 }
388 Skip_L4( "crc");
389 Element_End0();
390
391 int64u End=Element_Size;
392 if (FromMKV && !(initial_block && final_block))
393 {
394 Get_L4 (blocksize, "blocksize");
395 End=Element_Offset+blocksize;
396 if (End>=Element_Size)
397 End=Element_Size;
398 }
399
400 //Sub-block
401 int8u id;
402 while (Element_Offset<End)
403 {
404 Element_Begin0();
405 int32u word_size;
406 bool large, odd_size;
407 BS_Begin();
408 Get_SB (large, "large");
409 Get_SB (odd_size, "odd_size");
410 Get_S1 (6, id, "id"); Element_Info1(Wvpk_id(id));
411 BS_End();
412 if (large)
413 {
414 Get_L3 (word_size, "word_size");
415 }
416 else
417 {
418 int8u word_size1;
419 Get_L1 (word_size1, "word_size");
420 word_size=word_size1;
421 }
422 if (word_size==0 && odd_size)
423 Size=0; //Problem!
424 else
425 Size=word_size*2-(odd_size?1:0);
426 Element_Name(Ztring().From_CC1(id));
427 switch (id)
428 {
429 case 0x07 : id_07(); break;
430 case 0x0D : id_0D(); break;
431 case 0x25 : id_25(); break;
432 default : if (word_size)
433 Skip_XX(Size, "data");
434 }
435 if (odd_size)
436 Skip_XX(1, "padding");
437 Element_End0();
438 }
439 }
440 }
441
442 //Filling
443 if (!Status[IsAccepted] && Frame_Count>=Frame_Count_Valid)
444 {
445 File__Tags_Helper::Accept("WavPack");
446 Data_Parse_Fill();
447 }
448}

450//---------------------------------------------------------------------------
451void File_Wvpk::Data_Parse_Fill()
452{
453 //Filling
454 File__Tags_Helper::Stream_Prepare(Stream_Audio);
455 Fill(Stream_Audio, 0, Audio_Format, "WavPack");
456 Ztring Version_Minor=Ztring::ToZtring(version%0x100);
457 if (Version_Minor.size()==1)
458 Version_Minor.insert(0, 1, __T('0'));
459 Fill(Stream_Audio, 0, Audio_Format_Profile, Ztring::ToZtring(version/0x100)+__T('.')+Version_Minor);
460 Fill(Stream_Audio, 0, Audio_Codec, "Wavpack");
461 Fill(Stream_Audio, 0, Audio_BitDepth, Wvpk_Resolution[(resolution1?1:0)*2+(resolution0?1:0)]);
462 Fill(Stream_Audio, StreamPos_Last, Audio_Channel_s_, num_channels?num_channels:(mono?1:2));
463 if (channel_mask)
464 {
465 Ztring Channels_Positions, Channels_Positions2;
466 if (channel_mask&0x00C7)
467 {
468 int8u Count=0;
469 Channels_Positions+=__T("Front:");
470 if (channel_mask&0x0001)
471 {
472 Channels_Positions+=__T(" L");
473 Count++;
474 }
475 if (channel_mask&0x0004)
476 {
477 Channels_Positions+=__T(" C");
478 Count++;
479 }
480 if (channel_mask&0x0040)
481 {
482 Channels_Positions+=__T(" C");
483 Count++;
484 }
485 if (channel_mask&0x0080)
486 {
487 Channels_Positions+=__T(" C");
488 Count++;
489 }
490 if (channel_mask&0x0002)
491 {
492 Channels_Positions+=__T(" R");
493 Count++;
494 }
495 Channels_Positions2+=Ztring::ToZtring(Count);
496 }
497 if (channel_mask&0x0600)
498 {
499 int8u Count=0;
500 if (!Channels_Positions.empty())
501 Channels_Positions+=__T(", ");
502 Channels_Positions+=__T("Side:");
503 if (channel_mask&0x0200)
504 {
505 Channels_Positions+=__T(" L");
506 Count++;
507 }
508 if (channel_mask&0x0400)
509 {
510 Channels_Positions+=__T(" R");
511 Count++;
512 }
513 Channels_Positions2+=__T('.')+Ztring::ToZtring(Count);
514 }
515 else
516 Channels_Positions2+=__T("/0");
517 if (channel_mask&0x0130)
518 {
519 int8u Count=0;
520 if (!Channels_Positions.empty())
521 Channels_Positions+=__T(", ");
522 Channels_Positions+=__T("Back:");
523 if (channel_mask&0x0010)
524 {
525 Channels_Positions+=__T(" L");
526 Count++;
527 }
528 if (channel_mask&0x0100)
529 {
530 Channels_Positions+=__T(" C");
531 Count++;
532 }
533 if (channel_mask&0x0020)
534 {
535 Channels_Positions+=__T(" R");
536 Count++;
537 }
538 Channels_Positions2+=__T('/')+Ztring::ToZtring(Count);
539 }
540 else
541 Channels_Positions2+=__T("/0");
542 if (channel_mask&0x0008)
543 {
544 if (!Channels_Positions.empty())
545 Channels_Positions+=__T(", ");
546 Channels_Positions+=__T("LFE");
547 Channels_Positions2+=__T(".1");
548 }
549 Fill(Stream_Audio, 0, Audio_ChannelPositions, Channels_Positions);
550 Fill(Stream_Audio, 0, Audio_ChannelPositions_String2, Channels_Positions2);
551 }
552
553 if (!FromMKV && SamplingRate<15)
554 {
555 Fill(Stream_Audio, StreamPos_Last, Audio_SamplingRate, Wvpk_SamplingRate[SamplingRate]);
556 if (total_samples_FirstFrame!=(int32u)-1) //--> this is a valid value
557 Fill(Stream_Audio, 0, Audio_Duration, ((int64u)total_samples_FirstFrame)*1000/Wvpk_SamplingRate[SamplingRate]);
558 }
559 Fill(Stream_Audio, 0, Audio_Format_Settings, hybrid?"Hybrid lossy":"Lossless");
560 Fill(Stream_Audio, 0, Audio_Codec_Settings, hybrid?"hybrid lossy":"lossless");
561 Fill(Stream_Audio, 0, Audio_Encoded_Library_Settings, Encoded_Library_Settings);
562
563 //No more need data
564 File__Tags_Helper::GoToFromEnd(512*1024, "WavPack");
565}

583//---------------------------------------------------------------------------
584void File_Wvpk::id_0D()
585{
586 //Parsing
587 Get_L1 (num_channels, "num_channels");
588 switch (Size)
589 {
590 case 1 :
591 break;
592 case 2 :
593 {
594 int8u channel_mask_1;
595 Get_L1 (channel_mask_1, "channel_mask");
596 channel_mask=channel_mask_1;
597 }
598 break;
599 case 3 :
600 {
601 int16u channel_mask_2;
602 Get_L2 (channel_mask_2, "channel_mask");
603 channel_mask=channel_mask_2;
604 }
605 break;
606 case 4 :
607 Get_L3 (channel_mask, "channel_mask");
608 break;
609 case 5 :
610 Get_L4 (channel_mask, "channel_mask");
611 break;
612 default : Skip_XX(Size, "unknown");
613 }
614}

620 int32u flags;

622 Get_L3 (flags, "flags");
623 Skip_Flags(flags, 0, "");
624 Skip_Flags(flags, 1, "fast mode");
625 Skip_Flags(flags, 2, "");
626 Skip_Flags(flags, 3, "high quality mode");
627
628 Skip_Flags(flags, 4, "very high quality mode");
629 Skip_Flags(flags, 5, "bitrate is kbps, not bits/sample");
630 Skip_Flags(flags, 6, "automatic noise shaping");
631 Skip_Flags(flags, 7, "shaping mode specified");
632
633 Skip_Flags(flags, 8, "joint-stereo mode specified");
634 Skip_Flags(flags, 9, "dynamic noise shaping");
635 Skip_Flags(flags, 10, "create executable");
636 Skip_Flags(flags, 11, "create correction file");
637
638 Skip_Flags(flags, 12, "maximize bybrid compression");
639 Skip_Flags(flags, 13, "");
640 Skip_Flags(flags, 14, "");
641 Skip_Flags(flags, 15, "calc noise in hybrid mode");
642
643 Skip_Flags(flags, 16, "lossy mode");
644 Skip_Flags(flags, 17, "extra processing mode");
645 Skip_Flags(flags, 18, "no wvx stream w/ floats & big ints");
646 Skip_Flags(flags, 19, "store MD5 signature");
647
648 Skip_Flags(flags, 20, "merge blocks of equal redundancy (for lossyWAV)");
649 Skip_Flags(flags, 21, "");
650 Skip_Flags(flags, 22, "");
651 Skip_Flags(flags, 23, "optimize for mono streams posing as stereo");
652 if (flags&0x20000 && Size>=4)
653 {
654 Get_L1(extra, "extra");
655 }
656
657 if ((flags&0x20000 && Size>4) || (!(flags&0x20000) && Size>3))
658 Skip_XX(Size-3-(flags&0x20000?1:0), "unknown");

660 //Filling
661 if (flags&0x000001)
662 Encoded_Library_Settings+=__T(" -?");

672 Encoded_Library_Settings+=__T(" -?(bitrate is kbps, not bits/sample)");
673 if (flags&0x000040)
674 Encoded_Library_Settings+=__T(" -?(automatic noise shaping)");
675 if (flags&0x000080)
676 Encoded_Library_Settings+=__T(" -sn");
677 if (flags&0x000100)
678 Encoded_Library_Settings+=__T(" -jn");
679 if (flags&0x000200)
680 Encoded_Library_Settings+=__T(" -use-dns");
681 if (flags&0x000400)
682 Encoded_Library_Settings+=__T(" -e");
683 if (flags&0x000800)
684 Encoded_Library_Settings+=__T(" -c");
685 if (flags&0x001000)
686 Encoded_Library_Settings+=__T(" -cc");
687 if (flags&0x002000)
688 Encoded_Library_Settings+=__T(" -?");
689 if (flags&0x004000)
690 Encoded_Library_Settings+=__T(" -?");
691 if (flags&0x008000)
692 Encoded_Library_Settings+=__T(" -n");
693 if (flags&0x010000)
694 Encoded_Library_Settings+=__T(" -?(lossy mode)");
695 if (flags&0x020000)
696 {
697 Encoded_Library_Settings+=__T(" -x");
698 if (extra)
699 Encoded_Library_Settings+=Ztring::ToZtring(extra);
700 }
701 if (flags&0x04000)
702 Encoded_Library_Settings+=__T(" -?");
703 if (flags&0x080000)
704 Encoded_Library_Settings+=__T(" -m");
705 if (flags&0x100000)
706 Encoded_Library_Settings+=__T(" --merge-blocks");
707 if (flags&0x200000)
708 Encoded_Library_Settings+=__T(" -?");
709 if (flags&0x400000)
710 Encoded_Library_Settings+=__T(" -?");
711 if (flags&0x800000)
712 Encoded_Library_Settings+=__T(" --optimize-mono");
713 if (!Encoded_Library_Settings.empty())
714 Encoded_Library_Settings.erase(Encoded_Library_Settings.begin());
715}
