41const char* OpenMG_CodecID_Format (int8u CodecID)
42{
43 switch (CodecID)
44 {
45 case 0 :
46 case 1 :
47 return "Atrac3";
48 case 3 : return "MPEG Audio";
49 case 4 : return "PCM";
50 case 5 : return "WMA";
51 default : return "";
52 }
53}
54
55//---------------------------------------------------------------------------
56const char* OpenMG_CodecID_Encryption (int8u CodecID)
57{
58 switch (CodecID)
59 {
60 case 1 : return "SDMI";
61 default : return "";
62 }
63}
64
65//---------------------------------------------------------------------------
66int32u OpenMG_SamplingRate (int8u SamplingRate_Code)
67{
68 switch (SamplingRate_Code)
69 {
70 case 0 : return 32000;
71 case 1 : return 44100;
72 case 2 : return 44800;
73 case 3 : return 88200;
74 case 4 : return 96000;
75 default : return 0;
76 }
77}
78
79//---------------------------------------------------------------------------
80int8u OpenMG_Channels (int8u Channels_Code)
81{
82 if (Channels_Code<=4)
83 return Channels_Code;
84 else
85 return Channels_Code+1; //+LFE
86}
87
88//---------------------------------------------------------------------------
89const char* OpenMG_ChannelPositions (int8u Channels_Code)
90{
91 switch (Channels_Code)
92 {
93 case 1 : return "Front: C";
94 case 2 : return "Front: L R";
95 case 3 : return "Front: L R, Side: C";
96 case 4 : return "Front: L R, Back: L R";
97 case 5 : return "Front: L C R, Side: L R, LFE";
98 case 6 : return "Front: L C R, Side: L R, Back: C, LFE";
99 case 7 : return "Front: L C R, Side: L R, Back: L R, LFE";
100 default : return "";
101 }
102}

135//***************************************************************************
136// Buffer - File header
137//***************************************************************************
138
139//---------------------------------------------------------------------------
140bool File_OpenMG::FileHeader_Begin()
141{
142 if (!File__Tags_Helper::FileHeader_Begin())
143 return false;
144
145 //Synchro
146 if (Buffer_Offset+3>Buffer_Size)
147 return false;
148
149 // Testing
150 if (Buffer[Buffer_Offset ]!=0x45 // "EA3"
151 || Buffer[Buffer_Offset+1]!=0x41
152 || Buffer[Buffer_Offset+2]!=0x33)
153 {
154 File__Tags_Helper::Reject();
155 return false;
156 }
157
158 return true;
159}
160
161//---------------------------------------------------------------------------
162void File_OpenMG::FileHeader_Parse()
163{
164 //Parsing
165 int16u Size, FrameSize=0;
166 int8u Flags, CodecID, SamplingRate_Code=0, Channels_Code=0;
167 bool JointStereo=false;
168 Skip_C3( "Code");
169 Get_B1 (Flags, "Flags");
170 Get_B2 (Size, "Size");
171 Skip_XX(26, "Unknown");
172 Get_B1 (CodecID, "Coded ID"); Param_Info1(OpenMG_CodecID_Format(CodecID));
173 if (CodecID<=1) //Atrac3
174 {
175 BS_Begin();
176 Skip_S1(7, "Unknown");
177 Get_SB ( JointStereo, "Joint Stereo");
178 Get_S1 (3, SamplingRate_Code, "Sampling Rate"); Param_Info2(OpenMG_SamplingRate(SamplingRate_Code), " Hz");
179 Get_S1 (3, Channels_Code, "Channels"); Param_Info2(OpenMG_Channels(Channels_Code), " channel(s)");
180 Get_S2 (10, FrameSize, "Frame size");
181 BS_End();
182 }
183 Skip_XX(Size-Element_Offset, "Unknown");
184
185 FILLING_BEGIN();
186 if (!Status[IsAccepted])
187 {
188 File__Tags_Helper::Accept();
189
190 Fill(Stream_Audio, 0, Audio_Format, OpenMG_CodecID_Format(CodecID));
191 Fill(Stream_Audio, 0, Audio_Encryption, OpenMG_CodecID_Encryption(CodecID));
192 int64u StreamSize=(int64u)-1;
193 if (File_Size!=(int64u)-1)
194 {
195 StreamSize=File_Size-(Buffer_Offset+Element_Size);
196 Fill(Stream_Audio, 0, Audio_StreamSize, StreamSize);
197 }
198 if (CodecID<=1) // Atrac3
199 {
200 Fill(Stream_Audio, 0, Audio_Channel_s_, OpenMG_Channels(Channels_Code));
201 Fill(Stream_Audio, 0, Audio_ChannelPositions, OpenMG_ChannelPositions(Channels_Code));
202 if (Channels_Code==1 && JointStereo)
203 Fill(Stream_Audio, 0, Audio_Format_Settings_Mode, "Joint Stereo");
204 Fill(Stream_Audio, 0, Audio_SamplingRate, OpenMG_SamplingRate(SamplingRate_Code));
205
206 if (CodecID==1) //Protected
207 FrameSize++; //Not sure
208 FrameSize<<=3; //8-byte blocks
209 int64u BitRate=OpenMG_SamplingRate(SamplingRate_Code)*FrameSize/256;
210 Fill(Stream_Audio, 0, Audio_BitRate, BitRate);
211 if (StreamSize!=(int64u)-1 && BitRate)
212 Fill(Stream_Audio, 0, Audio_Duration, StreamSize*8*1000/BitRate);
213 }
214 }
215 FILLING_END();
216}
