102//***************************************************************************
103// Elements
104//***************************************************************************
105
106//---------------------------------------------------------------------------
107void File_Opus::Identification()
108{
109 Element_Name("Identification");
110
111 //Parsing
112 Ztring opus_version, Channels_Positions;
113 int8u Opus_version_id, ch_count, ch_map;
114 int16u preskip;
115 int32u sample_rate;
116 Get_Local(8,opus_version, "opus_codec_id");
117 Get_L1 (Opus_version_id, "opus_version_id");
118 Get_L1 (ch_count, "channel_count");
119 Get_L2 (preskip, "preskip");
120 Get_L4 (sample_rate, "rate");
121 Skip_L2( "ouput_gain");
122 Get_L1 (ch_map, "channel_map");
123 if (ch_map)
124 {
125 Skip_L1( "Stream count (N)");
126 Skip_L1( "Two-channel stream count (M)");
127 for (int8u Pos=0; Pos<ch_count; Pos++)
128 Skip_L1( "Channel mapping");
129 }
130 if (Element_Offset<Element_Size)
131 Skip_XX(Element_Size-Element_Offset, "unknown");
132
133 //Filling
134 FILLING_BEGIN();
135 Accept("Opus");
136
137 Stream_Prepare(Stream_Audio);
138 Fill(Stream_Audio, 0, Audio_Format, "Opus");
139 Fill(Stream_Audio, 0, Audio_Codec, "Opus");
140
141 if (!opus_version.empty())
142 {
143 Fill(Stream_Audio, 0, Audio_SamplingRate, sample_rate?sample_rate:48000);
144 Fill(Stream_Audio, 0, Audio_Channel_s_, ch_count);
145 }
146
147 switch (ch_map)
148 {
149 case 0 : // Mono/Stereo
150 if (ch_count>2)
151 break; // Not in spec
152 // else it is as Vorbis specs, no break
153 case 1 : // Vorbis order
154 {
155 Ztring ChannelPositions; ChannelPositions.From_Local(Opus_ChannelPositions[ch_count]);
156 Ztring ChannelPositions2; ChannelPositions2.From_Local(Opus_ChannelPositions2[ch_count]);
157 if (ChannelPositions!=Retrieve(Stream_Audio, 0, Audio_ChannelPositions))
158 Fill(Stream_Audio, 0, Audio_ChannelPositions, ChannelPositions);
159 if (ChannelPositions2!=Retrieve(Stream_Audio, 0, Audio_ChannelPositions_String2))
160 Fill(Stream_Audio, 0, Audio_ChannelPositions_String2, ChannelPositions2);
161 }
162 default: ; //Unknown
163 }
164
165 FILLING_END();
166
167 //Filling
168 Identification_Done=true;
169}
170
171//---------------------------------------------------------------------------
172void File_Opus::Stream()
173{
174 Element_Name("Stream");
175
176 Skip_XX(Element_Size, "Data");
177
178 Finish("Opus");
179}
