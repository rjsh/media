44File_ChannelGrouping::File_ChannelGrouping()
45{
46 //Configuration
47 #if MEDIAINFO_EVENTS
48 ParserIDs[0]=MediaInfo_Parser_ChannelGrouping;
49 StreamIDs_Width[0]=0;
50 #endif //MEDIAINFO_EVENTS
51 #if MEDIAINFO_DEMUX
52 Demux_Level=2; //Container
53 #endif //MEDIAINFO_DEMUX
54 #if MEDIAINFO_TRACE
55 Trace_Layers_Update(0); //Container1
56 #endif //MEDIAINFO_TRACE
57 IsRawStream=true;
58
59 //In
60 BitDepth=0;
61 SamplingRate=0;
62 Endianness=0;
63 Aligned=false;
64 CanBePcm=false;
65 Common=NULL;
66 Channel_Pos=0;
67 Channel_Total=1;
68}
69
70File_ChannelGrouping::~File_ChannelGrouping()
71{
72 Common->Instances--;
73
74 if (Common->Instances==0)
75 {
76 for (size_t Pos=0; Pos<Common->Channels.size(); Pos++)
77 delete Common->Channels[Pos]; //Common->Channels[Pos]=NULL;
78 delete Common; //Common=NULL;
79 }
80}

86//---------------------------------------------------------------------------
87void File_ChannelGrouping::Streams_Fill()
88{
89 Fill(Stream_General, 0, General_Format, "ChannelGrouping");
90
91 if (Channel_Pos!=Common->Channels.size()-1)
92 return;
93
94 if (Common->Parsers.size()!=1 && CanBePcm) // Last parser is PCM, impossible to detect with another method if htere is only one block
95 {
96 for (size_t Pos=0; Pos<Common->Parsers.size()-1; Pos++)
97 delete Common->Parsers[Pos];
98 Common->Parsers.erase(Common->Parsers.begin(), Common->Parsers.begin()+Common->Parsers.size()-1);
99 Common->Parsers[0]->Accept();
100 Common->Parsers[0]->Fill();
101 }
102
103 if (Common->Parsers.size()!=1)
104 return;
105
106 Fill(Common->Parsers[0]);
107 Merge(*Common->Parsers[0]);
108}
109
110//---------------------------------------------------------------------------
111void File_ChannelGrouping::Streams_Finish()
112{
113 if (Channel_Pos!=Common->Channels.size()-1 || Common->Parsers.size()!=1)
114 return;
115
116 Finish(Common->Parsers[0]);
117}

123//---------------------------------------------------------------------------
124void File_ChannelGrouping::Read_Buffer_Init()
125{
126 if (Common==NULL)
127 {
128 //Common
129 Common=new common;
130 Common->Channels.resize(Channel_Total);
131 for (size_t Pos=0; Pos<Common->Channels.size(); Pos++)
132 Common->Channels[Pos]=new common::channel;
133 Element_Code=(int64u)-1;
134
135 //SMPTE ST 337
136 {
137 File_SmpteSt0337* Parser=new File_SmpteSt0337;
138 Parser->Container_Bits=BitDepth;
139 Parser->Endianness=Endianness;
140 Parser->Aligned=Aligned;
141 Common->Parsers.push_back(Parser);
142 }
143
144 //PCM
145 if (CanBePcm)
146 {
147 File_Pcm* Parser=new File_Pcm;
148 Parser->BitDepth=BitDepth;
149 Parser->Channels=Channel_Total;
150 Parser->SamplingRate=SamplingRate;
151 Parser->Endianness=Endianness;
152 Common->Parsers.push_back(Parser);
153 }
154
155 //for all parsers
156 for (size_t Pos=0; Pos<Common->Parsers.size(); Pos++)
157 {
158 #if MEDIAINFO_DEMUX
159 if (Config->Demux_Unpacketize_Get())
160 {
161 Common->Parsers[Pos]->Demux_UnpacketizeContainer=true;
162 Common->Parsers[Pos]->Demux_Level=2; //Container
163 Demux_Level=4; //Intermediate
164 }
165 #endif //MEDIAINFO_DEMUX
166 Open_Buffer_Init(Common->Parsers[Pos]);
167 }
168 }
169 Common->Instances++;
170 Common->Instances_Max++;
171}

173//---------------------------------------------------------------------------
174void File_ChannelGrouping::Read_Buffer_Continue()
175{
176 //Verifying that all instances are still present
177 if (Common->Instances!=Common->Instances_Max)
178 {
179 Reject();
180 return;
181 }
182
183 //Handling of multiple frames in one block
184 if (Buffer_Size==0)
185 {
186 Offsets_Stream.clear();
187 Offsets_Buffer.clear();
188 for (size_t Pos=0; Pos<Common->Parsers.size(); Pos++)
189 Open_Buffer_Continue(Common->Parsers[Pos], Common->MergedChannel.Buffer+Common->MergedChannel.Buffer_Offset, 0, false);
190 return;
191 }
192
193 //Demux
194 #if MEDIAINFO_DEMUX
195 Demux(Common->MergedChannel.Buffer+Common->MergedChannel.Buffer_Offset, Common->MergedChannel.Buffer_Size-Common->MergedChannel.Buffer_Offset, ContentType_MainStream);
196 #endif //MEDIAINFO_EVENTS
197
198 //Copying to Channel buffer
199 if (Common->Channels[Channel_Pos]->Buffer_Size+Buffer_Size>Common->Channels[Channel_Pos]->Buffer_Size_Max)
200 Common->Channels[Channel_Pos]->resize(Common->Channels[Channel_Pos]->Buffer_Size+Buffer_Size);
201 memcpy(Common->Channels[Channel_Pos]->Buffer+Common->Channels[Channel_Pos]->Buffer_Size, Buffer, Buffer_Size);
202 Common->Channels[Channel_Pos]->Buffer_Size+=Buffer_Size;
203 Common->Channels[Channel_Pos]->Offsets_Stream.insert(Common->Channels[Channel_Pos]->Offsets_Stream.begin(), Offsets_Stream.begin(), Offsets_Stream.end());
204 Offsets_Stream.clear();
205 Common->Channels[Channel_Pos]->Offsets_Buffer.insert(Common->Channels[Channel_Pos]->Offsets_Buffer.begin(), Offsets_Buffer.begin(), Offsets_Buffer.end());
206 Offsets_Buffer.clear();
207 Skip_XX(Buffer_Size, "Channel grouping data");
208 Common->Channel_Current++;
209 if (Common->Channel_Current>=Channel_Total)
210 Common->Channel_Current=0;
211
212 //Copying to merged channel
213 size_t Minimum=(size_t)-1;
214 for (size_t Pos=0; Pos<Common->Channels.size(); Pos++)
215 if (Minimum>Common->Channels[Pos]->Buffer_Size-Common->Channels[Pos]->Buffer_Offset)
216 Minimum=Common->Channels[Pos]->Buffer_Size-Common->Channels[Pos]->Buffer_Offset;
217 if (Minimum*8>=BitDepth)
218 {
219 for (size_t Pos=0; Pos<Common->Channels.size(); Pos++)
220 {
221 Common->MergedChannel.Offsets_Stream.insert(Common->MergedChannel.Offsets_Stream.end(), Common->Channels[Pos]->Offsets_Stream.begin(), Common->Channels[Pos]->Offsets_Stream.end());
222 Common->Channels[Pos]->Offsets_Stream.clear();
223 Common->MergedChannel.Offsets_Buffer.insert(Common->MergedChannel.Offsets_Buffer.end(), Common->Channels[Pos]->Offsets_Buffer.begin(), Common->Channels[Pos]->Offsets_Buffer.end());
224 Common->Channels[Pos]->Offsets_Buffer.clear();
225 }
226
227 while (Minimum*8>=BitDepth)
228 {
229 switch (BitDepth)
230 {
231 case 16:
232 // Source: 16XE / L3L2 L1L0 + R3R2 R1R0
233 // Dest : 16XE / L3L2 L1L0 R3R2 R1R0
234 for (size_t Pos=0; Pos<Common->Channels.size(); Pos++)
235 {
236 if (Common->MergedChannel.Buffer_Size+Minimum>Common->MergedChannel.Buffer_Size_Max)
237 Common->MergedChannel.resize(Common->MergedChannel.Buffer_Size+Minimum);
238 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Common->Channels[Pos]->Buffer[Common->Channels[Pos]->Buffer_Offset++];
239 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Common->Channels[Pos]->Buffer[Common->Channels[Pos]->Buffer_Offset++];
240 }
241 Minimum-=2;
242 break;
243 case 20:
244 // Source: 20BE / L4L3 L2L1 L0L4 L3L2 L1L0 + R4R3 R2R1 R0R4 R3R2 R1R0
245 // Dest : 20BE / L4L3 L2L1 L0R4 R3R2 R1R0 L4L3 L2L1 L0R4 R2R1 R1R0
246 if (Endianness=='B')
247 for (size_t Pos=0; Pos+1<Common->Channels.size(); Pos+=2)
248 {
249 if (Common->MergedChannel.Buffer_Size+Minimum*2>Common->MergedChannel.Buffer_Size_Max)
250 Common->MergedChannel.resize(Common->MergedChannel.Buffer_Size+Minimum*2);
251 int8u* Channel1=Common->Channels[Pos]->Buffer+Common->Channels[Pos]->Buffer_Offset;
252 int8u* Channel2=Common->Channels[Pos]->Buffer+Common->Channels[Pos]->Buffer_Offset;
253 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Channel1[0];
254 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Channel1[1];
255 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]=(Channel1[0]&0xF0) | (Channel2[0]>>4 );
256 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]=(Channel2[0]<<4 ) | (Channel2[1]>>4 );
257 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]=(Channel2[1]<<4 ) | (Channel2[2]>>4 );
258 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]=(Channel1[2]<<4 ) | (Channel1[3]>>4 );
259 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]=(Channel1[3]<<4 ) | (Channel1[4]>>4 );
260 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]=(Channel1[4]<<4 ) | (Channel2[2]&0x0F);
261 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Channel2[3];
262 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Channel2[4];
263 Common->Channels[Pos]->Buffer_Offset+=5;
264 Common->Channels[Pos+1]->Buffer_Offset+=5;
265 }
266 // Source: 20LE / L1L0 L3L2 L0L4 L2L1 L4L3 + R1R0 R3R2 R0R4 R2R1 R4R3
267 // Dest : 20LE / L1L0 L3L2 R0L4 R2R1 R4R3 L1L0 L3L2 R0L4 R2R1 R4R3
268 else
269 for (size_t Pos=0; Pos+1<Common->Channels.size(); Pos+=2)
270 {
271 if (Common->MergedChannel.Buffer_Size+Minimum*2>Common->MergedChannel.Buffer_Size_Max)
272 Common->MergedChannel.resize(Common->MergedChannel.Buffer_Size+Minimum*2);
273 int8u* Channel1=Common->Channels[Pos]->Buffer+Common->Channels[Pos]->Buffer_Offset;
274 int8u* Channel2=Common->Channels[Pos]->Buffer+Common->Channels[Pos]->Buffer_Offset;
275 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Channel1[0];
276 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Channel1[1];
277 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]=(Channel2[0]<<4 ) | (Channel1[2]&0x0F);
278 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]=(Channel2[1]<<4 ) | (Channel2[0]>>4 );
279 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]=(Channel2[2]<<4 ) | (Channel2[1]>>4 );
280 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]=(Channel1[3]<<4 ) | (Channel1[2]>>4 );
281 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]=(Channel1[4]<<4 ) | (Channel1[3]>>4 );
282 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]=(Channel2[2]&0xF0) | (Channel1[4]>>4 );
283 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Channel2[3];
284 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Channel2[4];
285 Common->Channels[Pos]->Buffer_Offset+=5;
286 Common->Channels[Pos+1]->Buffer_Offset+=5;
287 }
288 Minimum-=5; //2.5 twice
289 break;
290 case 24:
291 // Source: 24XE / L5L4 L3L2 L1L0 + R5R4 R3R2 R1R0
292 // Dest : 24XE / L5L4 L3L2 L1L0 R5R4 R3R2 R1R0
293 for (size_t Pos=0; Pos<Common->Channels.size(); Pos++)
294 {
295 if (Common->MergedChannel.Buffer_Size+Minimum>Common->MergedChannel.Buffer_Size_Max)
296 Common->MergedChannel.resize(Common->MergedChannel.Buffer_Size+Minimum);
297 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Common->Channels[Pos]->Buffer[Common->Channels[Pos]->Buffer_Offset++];
298 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Common->Channels[Pos]->Buffer[Common->Channels[Pos]->Buffer_Offset++];
299 Common->MergedChannel.Buffer[Common->MergedChannel.Buffer_Size++]= Common->Channels[Pos]->Buffer[Common->Channels[Pos]->Buffer_Offset++];
300 }
301 Minimum-=3;
302 break;
303 default: ;
304 // Not supported
305 Reject();
306 return;
307 }
308 }
309 }
310
311 if (Common->MergedChannel.Buffer_Size>Common->MergedChannel.Buffer_Offset)
312 {
313 for (size_t Pos=0; Pos<Common->Parsers.size(); Pos++)
314 {
315 if (FrameInfo_Next.DTS!=(int64u)-1)
316 Common->Parsers[Pos]->FrameInfo=FrameInfo_Next; //AES3 parse has its own buffer management
317 else if (FrameInfo.DTS!=(int64u)-1)
318 {
319 Common->Parsers[Pos]->FrameInfo=FrameInfo;
320 FrameInfo=frame_info();
321 }
322 Common->Parsers[Pos]->Offsets_Stream.insert(Common->Parsers[Pos]->Offsets_Stream.end(), Common->MergedChannel.Offsets_Stream.begin(), Common->MergedChannel.Offsets_Stream.end());
323 Common->Parsers[Pos]->Offsets_Buffer.insert(Common->Parsers[Pos]->Offsets_Buffer.end(), Common->MergedChannel.Offsets_Buffer.begin(), Common->MergedChannel.Offsets_Buffer.end());
324 for (size_t Offsets_Pos_Temp=Common->Parsers[Pos]->Offsets_Buffer.size()-Common->MergedChannel.Offsets_Buffer.size(); Offsets_Pos_Temp<Common->Parsers[Pos]->Offsets_Buffer.size(); Offsets_Pos_Temp++)
325 Common->Parsers[Pos]->Offsets_Buffer[Offsets_Pos_Temp]+=Common->Parsers[Pos]->Buffer_Size/Common->Channels.size();
326 Open_Buffer_Continue(Common->Parsers[Pos], Common->MergedChannel.Buffer+Common->MergedChannel.Buffer_Offset, Common->MergedChannel.Buffer_Size-Common->MergedChannel.Buffer_Offset, false);
327
328 //Multiple parsers
329 if (Common->Parsers.size()>1)
330 {
331 if (!Common->Parsers[Pos]->Status[IsAccepted] && Common->Parsers[Pos]->Status[IsFinished])
332 {
333 delete *(Common->Parsers.begin()+Pos);
334 Common->Parsers.erase(Common->Parsers.begin()+Pos);
335 Pos--;
336 }
337 else if (Common->Parsers.size()>1 && Common->Parsers[Pos]->Status[IsAccepted])
338 {
339 File__Analyze* Parser=Common->Parsers[Pos];
340 for (size_t Pos2=0; Pos2<Common->Parsers.size(); Pos2++)
341 {
342 if (Pos2!=Pos)
343 delete *(Common->Parsers.begin()+Pos2);
344 }
345 Common->Parsers.clear();
346 Common->Parsers.push_back(Parser);
347 }
348 }
349 }
350 Common->MergedChannel.Buffer_Offset=Common->MergedChannel.Buffer_Size;
351 Common->MergedChannel.Offsets_Stream.clear();
352 Common->MergedChannel.Offsets_Buffer.clear();
353 }
354 if (!Status[IsAccepted] && Common->Parsers.size()==1 && Common->Parsers[0]->Status[IsAccepted])
355 Accept();
356 if (!Status[IsFilled] && Common->Parsers.size()==1 && Common->Parsers[0]->Status[IsFilled])
357 Fill();
358 if (!Status[IsFinished] && Common->Parsers.size()==1 && Common->Parsers[0]->Status[IsFinished])
359 Finish();
360
361 //Optimize buffer
362 for (size_t Pos=0; Pos<Common->Channels.size(); Pos++)
363 Common->Channels[Pos]->optimize();
364 Common->MergedChannel.optimize();
365}

367//---------------------------------------------------------------------------
368void File_ChannelGrouping::Read_Buffer_Unsynched()
369{
370 for (size_t Pos=0; Pos<Common->Parsers.size(); Pos++)
371 if (Common->Parsers[Pos])
372 Common->Parsers[Pos]->Open_Buffer_Unsynch();
373
374 Common->MergedChannel.Buffer_Offset=0;
375 Common->MergedChannel.Buffer_Size=0;
376 for (size_t Pos=0; Pos<Common->Channels.size(); Pos++)
377 {
378 Common->Channels[Pos]->Buffer_Offset=0;
379 Common->Channels[Pos]->Buffer_Size=0;
380 }
381}
