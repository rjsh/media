192//---------------------------------------------------------------------------
193Ztring AC3_chanmap_ChannelPositions (int16u chanmap)
194{
195 Ztring Front;
196 Ztring Side;
197 Ztring Back;
198 Ztring More;
199
200 for (int8u Pos=0; Pos<16; Pos++)
201 {
202 if (chanmap&(1<<(15-Pos)))
203 {
204 switch (Pos)
205 {
206 case 0 : Front+=__T(" L"); break;
207 case 1 : Front+=__T(" C"); break;
208 case 2 : Front+=__T(" R"); break;
209 case 3 : Side+=__T(" L"); break;
210 case 4 : Side+=__T(" R"); break;
211 case 5 : {
212 bool HasR=false;
213 if (Front.find(__T(" R"))!=string::npos)
214 {
215 Front.resize(Front.size()-2);
216 HasR=true;
217 }
218 Front+=__T(" C C");
219 if (HasR)
220 Front+=__T(" R");
221 }
222 break;
223 case 6 : Back+=__T(" L R"); break;
224 case 7 : if (Back.empty())
225 Back=__T(" C");
226 else
227 Back=__T(" L C R");
228 break;
229 case 15 : More+=__T(", LFE");
230 break;
231 default: ;
232 }
233 }
234 }
235
236 Ztring ToReturn;
237 if (!Front.empty())
238 {
239 ToReturn+=__T("Front:")+Front;
240 }
241 if (!Side.empty())
242 {
243 if (!ToReturn.empty())
244 ToReturn+=__T(", ");
245 ToReturn+=__T("Side:")+Side;
246 }
247 if (!Back.empty())
248 {
249 if (!ToReturn.empty())
250 ToReturn+=__T(", ");
251 ToReturn+=__T("Back:")+Back;
252 }
253 ToReturn+=More;
254
255 return ToReturn;
256}

338//---------------------------------------------------------------------------
339int16u AC3_FrameSize_Get(int8u frmsizecod, int8u fscod)
340{
341 bool Padding=(frmsizecod%2)?true:false;
342 int16u frame_size_id=frmsizecod/2;
343
344 if (frame_size_id>26 || fscod>3)
345 return 0;
346
347 int16u FrameSize=AC3_FrameSize[frame_size_id][fscod];
348 if (fscod==1 && Padding)
349 FrameSize+=2; // frame lengths are padded by 1 word (16 bits) at 44100 Hz
350 return FrameSize;
351}

395int CRC16_Init(int16u *Table, int16u Polynomial)
396{
397 for (size_t Pos=0; Pos<256; Pos++)
398 {
399 Table[Pos]=(int16u)Pos<<8;
400
401 for(int8u bit=0; bit<8; bit++)
402 {
403 if (Table[Pos]&0x8000)
404 Table[Pos]=(Table[Pos]<<1)^Polynomial;
405 else
406 Table[Pos]=Table[Pos]<<1;
407 }
408 }
409 return 0;
410}
