593void File_DolbyE::Streams_Fill()
594{
595 Fill(Stream_General, 0, General_Format, "Dolby E");
596 for (int8u program=0; program<DolbyE_Programs[ProgramConfiguration]; program++)
597 {
598 Stream_Prepare(Stream_Audio);
599 Fill(Stream_Audio, StreamPos_Last, Audio_Format, "Dolby E");
600 if (DolbyE_Programs[ProgramConfiguration]>1)
601 Fill(Stream_Audio, StreamPos_Last, Audio_ID, Count_Get(Stream_Audio));
602 Fill(Stream_Audio, StreamPos_Last, Audio_Channel_s_, DolbyE_Channels_PerProgram(ProgramConfiguration, program));
603 Fill(Stream_Audio, StreamPos_Last, Audio_ChannelPositions, DolbyE_ChannelPositions_PerProgram(ProgramConfiguration, program));
604 Fill(Stream_Audio, StreamPos_Last, Audio_ChannelPositions_String2, DolbyE_ChannelPositions2_PerProgram(ProgramConfiguration, program));
605 Fill(Stream_Audio, StreamPos_Last, Audio_ChannelLayout, DolbyE_ChannelLayout_PerProgram(ProgramConfiguration, program));
606 Fill(Stream_Audio, StreamPos_Last, Audio_SamplingRate, 48000);
607 Fill(Stream_Audio, StreamPos_Last, Audio_BitDepth, BitDepth);
608 if (SMPTE_time_code_StartTimecode!=(int64u)-1)
609 {
610 Fill(StreamKind_Last, StreamPos_Last, Audio_Delay, SMPTE_time_code_StartTimecode);
611 Fill(StreamKind_Last, StreamPos_Last, Audio_Delay_Source, "Stream");
612 }
613
614 Fill(Stream_Audio, StreamPos_Last, Audio_FrameRate, Mpegv_frame_rate[FrameRate]);
615 if (FrameInfo.PTS!=(int64u)-1 && BitDepth)
616 {
617 float BitRate=(float)(96000*BitDepth);
618
619 if (GuardBand_Before_Initial)
620 {
621 float GuardBand_Before_Initial_Duration=GuardBand_Before_Initial*8/BitRate;
622 Fill(Stream_Audio, StreamPos_Last, "GuardBand_Before", GuardBand_Before_Initial_Duration, 9);
623 Fill(Stream_Audio, StreamPos_Last, "GuardBand_Before/String", Ztring::ToZtring(GuardBand_Before_Initial_Duration*1000000, 0)+Ztring().From_UTF8(" \xC2xB5s")); //0xC2 0xB5 = micro sign
624 (*Stream_More)[Stream_Audio][StreamPos_Last](Ztring().From_Local("GuardBand_Before"), Info_Options)=__T("N NT");
625 (*Stream_More)[Stream_Audio][StreamPos_Last](Ztring().From_Local("GuardBand_Before/String"), Info_Options)=__T("N NT");
626 }
627 if (GuardBand_Before_Initial)
628 {
629 float GuardBand_After_Initial_Duration=GuardBand_After_Initial*8/BitRate;
630 Fill(Stream_Audio, StreamPos_Last, "GuardBand_After", GuardBand_After_Initial_Duration, 9);
631 Fill(Stream_Audio, StreamPos_Last, "GuardBand_After/String", Ztring::ToZtring(GuardBand_After_Initial_Duration*1000000, 0)+Ztring().From_UTF8(" \xC2xB5s")); //0xC2 0xB5 = micro sign
632 (*Stream_More)[Stream_Audio][StreamPos_Last](Ztring().From_Local("GuardBand_After"), Info_Options)=__T("N NT");
633 (*Stream_More)[Stream_Audio][StreamPos_Last](Ztring().From_Local("GuardBand_After/String"), Info_Options)=__T("N NT");
634 }
635 }
636 }
637 Fill(Stream_General, 0, General_OverallBitRate, Element_Size*8*Mpegv_frame_rate[FrameRate], 0);
638}

640//***************************************************************************
641// Buffer - Synchro
642//***************************************************************************
643
644//---------------------------------------------------------------------------
645bool File_DolbyE::Synchronize()
646{
647 //Synchronizing
648 while (Buffer_Offset+32<=Buffer_Size)
649 {
650 if ((CC2(Buffer+Buffer_Offset_Temp)&0xFFFE)==0x078E) //16-bit
651 {
652 BitDepth=16;
653 ScrambledBitStream=(CC2(Buffer+Buffer_Offset)&0x0001)?true:false;
654 break; //while()
655 }
656 if ((CC3(Buffer+Buffer_Offset)&0xFFFFE0)==0x0788E0) //20-bit
657 {
658 BitDepth=20;
659 ScrambledBitStream=(CC3(Buffer+Buffer_Offset)&0x000010)?true:false;
660 break; //while()
661 }
662 if ((CC3(Buffer+Buffer_Offset)&0xFFFFFE)==0x07888E) //24-bit
663 {
664 BitDepth=24;
665 ScrambledBitStream=(CC3(Buffer+Buffer_Offset)&0x000001)?true:false;
666 break; //while()
667 }
668 Buffer_Offset++;
669 }
670
671 //Parsing last bytes if needed
672 if (Buffer_Offset+8>Buffer_Size)
673 return false;
674
675 //Synched
676 return true;
677}

679//---------------------------------------------------------------------------
680bool File_DolbyE::Synched_Test()
681{
682 //Must have enough buffer for having header
683 if (Buffer_Offset+32>Buffer_Size)
684 return false;
685
686 //Quick test of synchro
687 switch (BitDepth)
688 {
689 case 16 : if ((CC2(Buffer+Buffer_Offset)&0xFFFE )!=0x078E ) {Synched=false; return true;} break;
690 case 20 : if ((CC3(Buffer+Buffer_Offset)&0xFFFFE0)!=0x0788E0) {Synched=false; return true;} break;
691 case 24 : if ((CC3(Buffer+Buffer_Offset)&0xFFFFFE)!=0x07888E) {Synched=false; return true;} break;
692 default : ;
693 }
694
695 //We continue
696 return true;
697}

699//***************************************************************************
700// Buffer - Per element
701//***************************************************************************
702
703//---------------------------------------------------------------------------
704void File_DolbyE::Header_Parse()
705{
706 //Filling
707 if (IsSub)
708 Header_Fill_Size(Buffer_Size-Buffer_Offset);
709 else
710 {
711 //Looking for synchro
712 //Synchronizing
713 Buffer_Offset_Temp=Buffer_Offset+3;
714 if (BitDepth==16)
715 while (Buffer_Offset_Temp+2<=Buffer_Size)
716 {
717 if ((CC2(Buffer+Buffer_Offset_Temp)&0xFFFE)==0x078E) //16-bit
718 break; //while()
719 Buffer_Offset_Temp++;
720 }
721 if (BitDepth==20)
722 while (Buffer_Offset_Temp+3<=Buffer_Size)
723 {
724 if ((CC3(Buffer+Buffer_Offset_Temp)&0xFFFFE0)==0x0788E0) //20-bit
725 break; //while()
726 Buffer_Offset_Temp++;
727 }
728 if (BitDepth==24)
729 while (Buffer_Offset_Temp+3<=Buffer_Size)
730 {
731 if ((CC3(Buffer+Buffer_Offset_Temp)&0xFFFFFE)==0x07888E) //24-bit
732 break; //while()
733 Buffer_Offset_Temp++;
734 }
735
736 if (Buffer_Offset_Temp+(BitDepth>16?3:2)>Buffer_Size)
737 {
738 if (File_Offset+Buffer_Size==File_Size)
739 Buffer_Offset_Temp=Buffer_Size;
740 else
741 {
742 Element_WaitForMoreData();
743 return;
744 }
745 }
746
747 Header_Fill_Size(Buffer_Offset_Temp-Buffer_Offset);
748 }
749 Header_Fill_Code(0, "Frame");
750}

752//---------------------------------------------------------------------------
753void File_DolbyE::Data_Parse()
754{
755 //In case of scrambling
756 const int8u* Save_Buffer=NULL;
757 size_t Save_Buffer_Offset=0;
758 int64u Save_File_Offset=0;
759 if (ScrambledBitStream)
760 {
761 //We must change the buffer,
762 Save_Buffer=Buffer;
763 Save_Buffer_Offset=Buffer_Offset;
764 Save_File_Offset=File_Offset;
765 File_Offset+=Buffer_Offset;
766 Buffer_Offset=0;
767 Descrambled_Buffer=new int8u[(size_t)Element_Size];
768 std::memcpy(Descrambled_Buffer, Save_Buffer+Save_Buffer_Offset, (size_t)Element_Size);
769 Buffer=Descrambled_Buffer;
770 }
771
772 //Parsing
773 BS_Begin();
774 Block();
775 BS_End();
776
777 //In case of scrambling
778 if (ScrambledBitStream)
779 {
780 delete[] Buffer; Buffer=Save_Buffer;
781 Buffer_Offset=Save_Buffer_Offset;
782 File_Offset=Save_File_Offset;
783 }
784
785 FILLING_BEGIN();
786 if (!Status[IsAccepted])
787 {
788 Accept("Dolby E");
789
790 //Guard band
791 GuardBand_Before_Initial=GuardBand_Before;
792 GuardBand_After_Initial=GuardBand_After;
793 }
794 Frame_Count++;
795 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
796 Frame_Count_NotParsedIncluded++;
797 if (Mpegv_frame_rate[FrameRate])
798 FrameInfo.DUR=float64_int64s(1000000000/Mpegv_frame_rate[FrameRate]);
799 else
800 FrameInfo.DUR=(int64u)-1;
801 if (FrameInfo.DTS!=(int64u)-1)
802 FrameInfo.PTS=FrameInfo.DTS+=FrameInfo.DUR;
803 if (Frame_Count==1)
804 {
805 Finish("Dolby E");
806 }
807 FILLING_END();
808 if (Frame_Count==0 && Buffer_TotalBytes>Buffer_TotalBytes_FirstSynched_Max)
809 Reject("Dolby E");
810}

812//---------------------------------------------------------------------------
813void File_DolbyE::Block()
814{
815 //Parsing
816 Skip_S3(BitDepth, "Synchro");
817 if (ScrambledBitStream)
818 {
819 //We must change the buffer
820 switch (BitDepth)
821 {
822 case 16 :
823 if (!Descramble_16bit())
824 return;
825 break;
826 case 20 :
827 if (!Descramble_20bit())
828 return;
829 break;
830 case 24 :
831 if (!Descramble_24bit())
832 return;
833 break;
834 default : ;
835 }
836 }
837 Skip_S2(14, "Unknown");
838 Get_S1 ( 6, ProgramConfiguration, "Program configuration"); Param_Info1(DolbyE_ChannelPositions[ProgramConfiguration]);
839 Get_S1 ( 4, FrameRate, "Frame rate 1"); Param_Info3(Mpegv_frame_rate[FrameRate], " fps", 3);
840 Skip_S1( 4, "Frame rate 2? Always same as Frame rate 1");
841 Skip_S2(16, "Frame number?");
842 Element_Begin1("SMPTE time code?");
843 int8u Frames_Units, Frames_Tens, Seconds_Units, Seconds_Tens, Minutes_Units, Minutes_Tens, Hours_Units, Hours_Tens;
844 bool DropFrame;
845
846 Skip_S1(4, "BG8");
847 Skip_S1(4, "BG7");
848
849 Skip_SB( "BGF2 / Field Phase");
850 Skip_SB( "BGF1");
851 Get_S1 (2, Hours_Tens, "Hours (Tens)");
852 Get_S1 (4, Hours_Units, "Hours (Units)");
853
854 Skip_S1(4, "BG6");
855 Skip_S1(4, "BG5");
856
857 Skip_SB( "BGF0 / BGF2");
858 Get_S1 (3, Minutes_Tens, "Minutes (Tens)");
859 Get_S1 (4, Minutes_Units, "Minutes (Units)");
860
861 Skip_S1(4, "BG4");
862 Skip_S1(4, "BG3");
863
864 Skip_SB( "FP - Field Phase / BGF0");
865 Get_S1 (3, Seconds_Tens, "Seconds (Tens)");
866 Get_S1 (4, Seconds_Units, "Seconds (Units)");
867
868 Skip_S1(4, "BG2");
869 Skip_S1(4, "BG1");
870
871 Skip_SB( "CF - Color fame");
872 Get_SB ( DropFrame, "DP - Drop frame");
873 Get_S1 (2, Frames_Tens, "Frames (Tens)");
874 Get_S1 (4, Frames_Units, "Frames (Units)");
875
876 Skip_BS(Data_BS_Remain(), "Unknown");
877
878 if (Hours_Tens<3)
879 {
880 int64u TimeCode=(int64u)(Hours_Tens *10*60*60*1000
881 + Hours_Units *60*60*1000
882 + Minutes_Tens *10*60*1000
883 + Minutes_Units *60*1000
884 + Seconds_Tens *10*1000
885 + Seconds_Units *1000
886 + (Mpegv_frame_rate[FrameRate]?float64_int32s((Frames_Tens*10+Frames_Units)*1000/Mpegv_frame_rate[FrameRate]):0));
887
888 Element_Info1(Ztring().Duration_From_Milliseconds(TimeCode));
889
890 //TimeCode
891 if (SMPTE_time_code_StartTimecode==(int64u)-1)
892 SMPTE_time_code_StartTimecode=TimeCode;
893 }
894}

900//---------------------------------------------------------------------------
901bool File_DolbyE::Descramble_16bit ()
902{
903 int16u ScrambleMask;
904 Get_S2 (16, ScrambleMask, "Scramble mask");
905 int16u Size=((BigEndian2int16u(Buffer+Buffer_Offset+(size_t)Element_Size-Data_BS_Remain()/8)^ScrambleMask)>>2)&0x3FF;
906
907 if (Data_BS_Remain()<(size_t)((Size+1)*BitDepth)) //+1 for additional unknown word
908 return false; //There is a problem
909
910 int8u* Temp=Descrambled_Buffer+(size_t)Element_Size-Data_BS_Remain()/8;
911 for (int16u Pos=0; Pos<Size; Pos++)
912 int16u2BigEndian(Temp+Pos*2, BigEndian2int16u(Temp+Pos*2)^ScrambleMask);
913
914 return true;
915}

917//---------------------------------------------------------------------------
918bool File_DolbyE::Descramble_20bit ()
919{
920 int32u ScrambleMask;
921 Get_S3 (20, ScrambleMask, "Scramble mask");
922 int16u Size=((BigEndian2int16u(Buffer+Buffer_Offset+(size_t)Element_Size-Data_BS_Remain()/8)^(ScrambleMask>>4))>>2)&0x3FF;
923
924 if (Data_BS_Remain()<(size_t)((Size+1)*BitDepth)) //+1 for additional unknown word
925 return false; //There is a problem
926
927 int8u* Temp=Descrambled_Buffer+(size_t)Element_Size-Data_BS_Remain()/8;
928 int64u ScrambleMasks=(((int64u)ScrambleMask)<<20)|ScrambleMask;
929 bool Half;
930 if (Data_BS_Remain()%8)
931 {
932 Temp--;
933 int24u2BigEndian(Temp, BigEndian2int24u(Temp)^(ScrambleMask));
934 Half=true;
935 }
936 else
937 Half=false;
938 for (int16u Pos=0; Pos<Size-(Half?1:0); Pos+=2)
939 int40u2BigEndian(Temp+(Half?3:0)+Pos*5/2, BigEndian2int40u(Temp+(Half?3:0)+Pos*5/2)^ScrambleMasks);
940 if ((Size-((Size && Half)?1:0))%2==0)
941 int24u2BigEndian(Temp+(Half?3:0)+(Size-((Size && Half)?1:0))*5/2, BigEndian2int24u(Temp+(Half?3:0)+(Size-((Size && Half)?1:0))*5/2)^(((int32u)ScrambleMasks)<<4));
942
943 return true;
944}
