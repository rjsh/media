348void File_Mpega::Streams_Fill()
349{
350 //VBR detection without header
351 if (VBR_Frames==0)
352 {
353 //How much kinds of bitrates?
354 if (BitRate_Count.size()>1)
355 BitRate_Mode=__T("VBR");
356 }
357
358 File__Tags_Helper::Stream_Prepare(Stream_Audio);
359 Fill(Stream_Audio, 0, Audio_Format, "MPEG Audio");
360 Fill(Stream_Audio, 0, Audio_Format_Version, Mpega_Format_Profile_Version[ID]);
361 Fill(Stream_Audio, 0, Audio_Format_Profile, Mpega_Format_Profile_Layer[layer]);
362 if (mode && mode<4)
363 {
364 Fill(Stream_Audio, 0, Audio_Format_Settings, Mpega_Codec_Profile[mode]);
365 Fill(Stream_Audio, 0, Audio_Format_Settings_Mode, Mpega_Codec_Profile[mode]);
366 }
367 if (mode_extension && mode_extension<4)
368 {
369 Fill(Stream_Audio, 0, Audio_Format_Settings, Mpega_Codec_Profile_Extension[mode_extension]);
370 Fill(Stream_Audio, 0, Audio_Format_Settings_ModeExtension, Mpega_Codec_Profile_Extension[mode_extension]);
371 }
372 if (emphasis && emphasis<4)
373 {
374 Fill(Stream_Audio, 0, Audio_Format_Settings, Mpega_Emphasis[emphasis]);
375 Fill(Stream_Audio, 0, Audio_Format_Settings_Emphasis, Mpega_Emphasis[emphasis]);
376 }
377 Fill(Stream_Audio, 0, Audio_Codec, Ztring(Mpega_Version[ID])+Ztring(Mpega_Layer[layer]));
378 Fill(Stream_Audio, 0, Audio_Codec_String, Ztring(Mpega_Version_String[ID])+Ztring(Mpega_Layer_String[layer]), true);
379 Fill(Stream_Audio, 0, Audio_SamplingRate, Mpega_SamplingRate[ID][sampling_frequency]);
380 if (mode<4)
381 {
382 Fill(Stream_Audio, 0, Audio_Channel_s_, Mpega_Channels[mode]);
383 Fill(Stream_Audio, 0, Audio_Codec_Profile, Mpega_Codec_Profile[mode]);
384 }
385
386 //Bitrate, if CBR
387 if (VBR_Frames==0 && BitRate_Mode!=__T("VBR"))
388 {
389 BitRate_Mode=__T("CBR");
390 BitRate=Mpega_BitRate[ID][layer][bitrate_index]*1000;
391 Fill(Stream_General, 0, General_OverallBitRate, BitRate);
392 Fill(Stream_Audio, 0, Audio_BitRate, BitRate);
393 if (CalculateDelay && Buffer_TotalBytes_FirstSynched>10 && BitRate>0)
394 {
395 Fill(Stream_Audio, 0, Audio_Delay, Buffer_TotalBytes_FirstSynched*8*1000/BitRate, 0);
396 Fill(Stream_Audio, 0, Audio_Delay_Source, "Stream");
397 }
398 }
399
400 //Bitrate mode
401 Fill(Stream_Audio, 0, Audio_BitRate_Mode, BitRate_Mode);
402 Fill(Stream_Audio, 0, Audio_BitRate_Minimum, BitRate_Minimum);
403 Fill(Stream_Audio, 0, Audio_BitRate_Nominal, BitRate_Nominal);
404
405 //Tags
406 File__Tags_Helper::Streams_Fill();
407}

535}

566//---------------------------------------------------------------------------
567bool File_Mpega::Synchronize()
568{
569 //Tags
570 bool Tag_Found_Begin;
571 if (!File__Tags_Helper::Synchronize(Tag_Found_Begin))
572 return false;
573 if (Tag_Found_Begin)
574 return true;
575
576 //Synchronizing
577 while (Buffer_Offset+4<=Buffer_Size)
578 {
579 while (Buffer_Offset+4<=Buffer_Size)
580 {
581 if (Buffer[Buffer_Offset ]==0xFF
582 && (Buffer[Buffer_Offset+1]&0xE0)==0xE0
583 && (Buffer[Buffer_Offset+2]&0xF0)!=0xF0
584 && (Buffer[Buffer_Offset+2]&0x0C)!=0x0C)
585 break; //while()
586
587 //Tags
588 bool Tag_Found_Synchro;
589 if (!File__Tags_Helper::Synchronize(Tag_Found_Synchro))
590 return false;
591 if (Tag_Found_Synchro)
592 return true;
593
594 //Better detect MPEG-PS
595 if (Frame_Count==0
596 && Buffer[Buffer_Offset ]==0x00
597 && Buffer[Buffer_Offset+1]==0x00
598 && Buffer[Buffer_Offset+2]==0x01
599 && Buffer[Buffer_Offset+3]==0xBA)
600 {
601 MpegPsPattern_Count++;
602 if (MpegPsPattern_Count>=2)
603 {
604 File__Tags_Helper::Reject("MPEG Audio");
605 return false;
606 }
607 }
608
609 Buffer_Offset++;
610 }
611
612 if (Buffer_Offset+4<=Buffer_Size)//Testing if size is coherant
613 {
614 //Retrieving some info
615 int8u ID0 =(CC1(Buffer+Buffer_Offset+1)>>3)&0x03;
616 int8u layer0 =(CC1(Buffer+Buffer_Offset+1)>>1)&0x03;

935{

937 if (Element_Size==0)

939 Element_DoNotShow();

941 }

970 {
971 int16u Samples;
972 if (ID==3 && layer==3) //MPEG 1 layer 1
973 Samples=384;
974 else if ((ID==2 || ID==0) && layer==1) //MPEG 2 or 2.5 layer 3
975 Samples=576;
976 else
977 Samples=1152;
978
979 Frequency_b=Mpega_SamplingRate[ID][sampling_frequency];
980 TS_Add(Samples);
981 }
982
983 //LAME
984 if (Encoded_Library.empty() && (Frame_Count<Frame_Count_Valid || File_Offset+Buffer_Offset+Element_Size==File_Size-File_EndTagSize)) //Can be elsewhere... At the start, or end frame
985 Header_Encoders();
986
987 //Filling
988 BitRate_Count[Mpega_BitRate[ID][layer][bitrate_index]]++;
989 Channels_Count[mode]++;
990 Extension_Count[mode_extension]++;
991 Emphasis_Count[emphasis]++;
992
993 if (Status[IsFilled])
994 {

1013 break;

1035 break;

1038 if (CRC12_Calculated!=CRC12)
1039 break;

1042 Skip_XX(Element_Offset_S-Element_Offset, "data");
1043 BS_Begin();
1044 Element_Begin1("Surround");
1045 Skip_S2(12, "Sync");
1046 Skip_S1( 8, "Size");
1047 Skip_S2(12, "CRC12");
1048 BS_End();
1049 Skip_XX(Surround_Size-4, "data");

1052 //Filling
1053 Surround_Frames++;
1054 break;
1055 }
1056 }
1057
1058 if (Element_Offset<Element_Size)
1059 Skip_XX(Element_Size-Element_Offset, "next data");
1060
1061 FILLING_BEGIN();
1062 //Filling
1063 if (IsSub && BitRate_Count.size()>1 && !Encoded_Library.empty())

1065 if (!Status[IsAccepted])

1275 mode_Count.clear();

1278 }
