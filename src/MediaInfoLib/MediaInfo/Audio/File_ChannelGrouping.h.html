7//+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
8//
9// Some containers use mono stream for AES3 (Stereo) grouping
10// We need to group the 2-mono streams in one before sending
11// data to AES parser
12//
13//+++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++++
14
15//---------------------------------------------------------------------------
16#ifndef MediaInfo_File_ChannelGroupingH
17#define MediaInfo_File_ChannelGroupingH
18//---------------------------------------------------------------------------
19
20//---------------------------------------------------------------------------
21#include "MediaInfo/File__Analyze.h"
22#include <cstring>
23//---------------------------------------------------------------------------
24
25namespace MediaInfoLib
26{
27
28//***************************************************************************
29// Class File_ChannelGrouping
30//***************************************************************************
31
32class File_ChannelGrouping : public File__Analyze
33{
34public :
35 //In
36 int8u BitDepth;
37 int16u SamplingRate;
38 int8u Endianness;
39 bool Aligned;
40 bool CanBePcm;
41
42 struct common
43 {
44 struct channel
45 {
46 int8u* Buffer;
47 size_t Buffer_Offset;
48 size_t Buffer_Size;
49 size_t Buffer_Size_Max;
50 std::vector<int64u> Offsets_Stream;
51 std::vector<int64u> Offsets_Buffer;
52
53 channel()
54 {
55 Buffer=new int8u[32768];
56 Buffer_Offset=0;
57 Buffer_Size=0;
58 Buffer_Size_Max=32768;
59 }
60
61 ~channel()
62 {
63 delete[] Buffer; //Buffer=NULL;
64 }
65
66 void resize(size_t NewSize)
67 {
68 if (NewSize<Buffer_Size_Max)
69 return;
70 int8u* Old=Buffer;
71 if (NewSize<2*Buffer_Size_Max)
72 Buffer_Size_Max*=2;
73 else
74 Buffer_Size_Max=NewSize;
75 Buffer=new int8u[Buffer_Size_Max];
76 std::memcpy(Buffer, Old, Buffer_Size);
77 delete[] Old; //Old=NULL;
78 }
79
80 void optimize()
81 {
82 if (Buffer_Offset<Buffer_Size_Max/2)
83 return;
84 std::memcpy(Buffer, Buffer+Buffer_Offset, Buffer_Size-Buffer_Offset);
85 Buffer_Size-=Buffer_Offset;
86 Buffer_Offset=0;
87 }
88
89 };
90 vector<channel*> Channels;
91 channel MergedChannel;
92 size_t Channel_Current;
93 std::vector<File__Analyze*> Parsers;
94 size_t Instances;
95 size_t Instances_Max;
96
97 common()
98 {
99 Channel_Current=0;
100 Instances=0;
101 Instances_Max=0;
102 }
103
104 ~common()
105 {
106 for (size_t Pos=0; Pos<Parsers.size(); Pos++)
107 delete Parsers[Pos];
108 }
109 };
110 int64u StreamID;
111 common* Common;
112 int8u Channel_Pos;
113 int8u Channel_Total;
114
115 //Constructor/Destructor
116 File_ChannelGrouping();
117 ~File_ChannelGrouping();
118
119private :
120 //Streams management
121 void Streams_Fill();
122 void Streams_Finish();
123
124 //Buffer - Global
125 void Read_Buffer_Init ();
126 void Read_Buffer_Continue ();
127 void Read_Buffer_Unsynched ();
128};
129
130} //NameSpace
131
132#endif
