181//---------------------------------------------------------------------------
182void File_SmpteSt0337::Streams_Fill()
183{
184 if (Parser && Parser->Status[IsAccepted])
185 {
186 Fill(Parser);
187 Merge(*Parser);
188
189 int64u OverallBitRate=Parser->Retrieve(Stream_General, 0, General_OverallBitRate).To_int64u();
190 if (OverallBitRate)
191 {
192 OverallBitRate*=Element_Size; OverallBitRate/=Element_Size-Stream_Bits*4/8;
193 Fill(Stream_General, 0, General_OverallBitRate, Ztring::ToZtring(OverallBitRate)+__T(" / ")+Parser->Retrieve(Stream_General, 0, General_OverallBitRate));
194 }
195 if (Parser->Count_Get(Stream_Audio))
196 FrameRate=Retrieve(Stream_Audio, 0, Audio_FrameRate).To_float64();
197 }
198 else if (data_type!=(int8u)-1)
199 {
200 if (Retrieve(Stream_Audio, 0, Audio_Format).empty() && Smpte_St0337_data_type_StreamKind[data_type]!=Stream_Max)
201 {
202 Stream_Prepare(Smpte_St0337_data_type_StreamKind[data_type]);
203 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_Format), Smpte_St0337_data_type[data_type]);
204 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_Codec), Smpte_St0337_data_type[data_type]);
205 }
206 }
207
208 // Guard band
209 if (GuardBand_Before) // With guard band, there is big chances that AES3 bit rate is respected
210 {
211 Fill(Stream_General, 0, General_OverallBitRate, Container_Bits*2*48000);
212 if (!IsSub && File_Size!=(int64u)-1)
213 Fill(Stream_General, 0, General_Duration, ((float64)File_Size)*8/(Container_Bits*2*48000)*1000);
214 }
215
216 if (FrameRate && FrameSizes.size()==1)
217 {
218 Fill(Stream_General, 0, General_OverallBitRate, FrameSizes.begin()->first*Container_Bits*FrameRate, 0);
219 }
220
221 for (size_t Pos=0; Pos<Count_Get(StreamKind_Last); Pos++)
222 {
223 if (Endianness=='L' && Retrieve(StreamKind_Last, Pos, "Format_Settings_Endianness")==__T("Little"))
224 Endianness='B';
225 switch (Endianness)
226 {
227 case 'B' :
228 Fill(StreamKind_Last, Pos, "Format_Settings", "Big");
229 Fill(StreamKind_Last, Pos, "Format_Settings_Endianness", "Big", Unlimited, true, true);
230 break;
231 case 'L' :
232 Fill(StreamKind_Last, Pos, "Format_Settings", "Little");
233 Fill(StreamKind_Last, Pos, "Format_Settings_Endianness", "Little", Unlimited, true, true);
234 break;
235 default : ;
236 }
237 Fill(StreamKind_Last, Pos, "Format_Settings_Mode", Container_Bits);
238 if (Retrieve(StreamKind_Last, Pos, Fill_Parameter(StreamKind_Last, Generic_BitDepth)).empty())
239 Fill(StreamKind_Last, Pos, Fill_Parameter(StreamKind_Last, Generic_BitDepth), Stream_Bits);
240
241 if (IsSub)
242 Fill(StreamKind_Last, Pos, "MuxingMode", "SMPTE ST 337");
243 if (Retrieve(StreamKind_Last, Pos, Fill_Parameter(StreamKind_Last, Generic_BitRate_Mode))!=__T("CBR"))
244 Fill(StreamKind_Last, Pos, Fill_Parameter(StreamKind_Last, Generic_BitRate_Mode), "CBR");
245 if (File_Size!=(int64u)-1 && FrameSizes.size()==1)
246 Fill(StreamKind_Last, Pos, Fill_Parameter(StreamKind_Last, Generic_FrameCount), File_Size/FrameSizes.begin()->first);
247 if (Retrieve(StreamKind_Last, Pos, Fill_Parameter(StreamKind_Last, Generic_Duration)).empty())
248 Fill(StreamKind_Last, Pos, Fill_Parameter(StreamKind_Last, Generic_Duration), Retrieve(Stream_General, 0, General_Duration));
249 }

268//---------------------------------------------------------------------------
269#if MEDIAINFO_SEEK
270size_t File_SmpteSt0337::Read_Buffer_Seek (size_t Method, int64u Value, int64u ID)
271{
272 // Init
273 if (!Duration_Detected)
274 {
275 MediaInfo_Internal MI;
276 MI.Option(__T("File_KeepInfo"), __T("1"));
277 Ztring ParseSpeed_Save=MI.Option(__T("ParseSpeed_Get"), __T(""));
278 Ztring Demux_Save=MI.Option(__T("Demux_Get"), __T(""));
279 MI.Option(__T("ParseSpeed"), __T("0"));
280 MI.Option(__T("Demux"), Ztring());
281 size_t MiOpenResult=MI.Open(File_Name);
282 MI.Option(__T("ParseSpeed"), ParseSpeed_Save); // This is a global value, need to reset it. TODO: local value
283 MI.Option(__T("Demux"), Demux_Save); // This is a global value, need to reset it. TODO: local value
284 if (!MiOpenResult)
285 return 0;
286
287 FrameRate=MI.Get(Stream_Audio, 0, __T("FrameRate")).To_float64();
288
289 Duration_Detected=true;
290 }
291
292 // Parsing
293 switch (Method)
294 {
295 case 0 :
296 if (FrameRate)
297 {
298 float64 FrameSize=3072000/FrameRate;
299 int64u FrameCount=float64_int64s(Value/FrameSize);
300 Value=float64_int64s(FrameCount*FrameSize);
301 }
302 GoTo(Value);
303 Open_Buffer_Unsynch();
304 return 1;
305 case 1 :
306 return Read_Buffer_Seek(0, File_Size*Value/10000, ID);
307 case 2 : // Timestamp
308 {
309 if (FrameRate)
310 return (size_t)-1; // Not supported
311
312 {
313 float64 FrameSize=3072000/FrameRate;
314 Unsynch_Frame_Count=float64_int64s(((float64)Value)/1000000000*FrameRate);
315 GoTo(float64_int64s(Unsynch_Frame_Count*FrameSize));
316 Open_Buffer_Unsynch();
317 return 1;
318 }
319 }
320 case 3 : // FrameNumber
321 {
322 if (FrameRate)
323 return (size_t)-1; // Not supported
324
325 {
326 float64 FrameSize=3072000/FrameRate;
327 Unsynch_Frame_Count=Value;
328 GoTo(float64_int64s(Unsynch_Frame_Count*FrameSize));
329 Open_Buffer_Unsynch();
330 return 1;
331 }
332 }
333 default : return (size_t)-1; // Not supported
334 }
335}
336#endif // MEDIAINFO_SEEK

343bool File_SmpteSt0337::Synchronize()
344{
345 // Guard band
346 size_t Buffer_Offset_Base=Buffer_Offset;
347
348 // Synchronizing
349 while (Buffer_Offset+16<=Buffer_Size)
350 {
351 if (!Status[IsAccepted] && !IsSub && File_Offset_FirstSynched==(int64u)-1 && Buffer_TotalBytes+Buffer_Offset>=Buffer_TotalBytes_FirstSynched_Max)
352 {
353 Reject();
354 return false;
355 }
356
357 if ((Container_Bits==0 || Container_Bits==16) && (!Aligned || ((Buffer_TotalBytes+Buffer_Offset)%4)==0))
358 {
359 if (Buffer[Buffer_Offset ]==0xF8
360 && Buffer[Buffer_Offset+1]==0x72
361 && Buffer[Buffer_Offset+2]==0x4E
362 && Buffer[Buffer_Offset+3]==0x1F) // 16-bit, BE
363 {
364 Container_Bits=16;
365 Stream_Bits=16;
366 Endianness='B'; // BE
367 break; // while()
368 }
369 if (Buffer[Buffer_Offset ]==0x72
370 && Buffer[Buffer_Offset+1]==0xF8
371 && Buffer[Buffer_Offset+2]==0x1F
372 && Buffer[Buffer_Offset+3]==0x4E) // 16-bit, LE
373 {
374 Container_Bits=16;
375 Stream_Bits=16;
376 Endianness='L'; // LE
377 break; // while()
378 }
379 }
380 if ((Container_Bits==0 || Container_Bits==20) && (!Aligned || ((Buffer_TotalBytes+Buffer_Offset)%5)==0))
381 {
382 if (Buffer[Buffer_Offset ]==0x6F
383 && Buffer[Buffer_Offset+1]==0x87
384 && Buffer[Buffer_Offset+2]==0x25
385 && Buffer[Buffer_Offset+3]==0x4E
386 && Buffer[Buffer_Offset+4]==0x1F) // 20-bit, BE
387 {
388 Container_Bits=20;
389 Stream_Bits=20;
390 Endianness='B'; // BE
391 break; // while()
392 }
393 }
394 if ((Container_Bits==0 || Container_Bits==20) && (!Aligned || ((Buffer_TotalBytes+Buffer_Offset)%5)==0))
395 {
396 if (Buffer[Buffer_Offset ]==0x72
397 && Buffer[Buffer_Offset+1]==0xF8
398 && Buffer[Buffer_Offset+2]==0xF6
399 && Buffer[Buffer_Offset+3]==0xE1
400 && Buffer[Buffer_Offset+4]==0x54) // 20-bit, LE
401 {
402 Container_Bits=20;
403 Stream_Bits=20;
404 Endianness='L'; // BE
405 break; // while()
406 }
407 }
408 if ((Container_Bits==0 || Container_Bits==24) && (!Aligned || ((Buffer_TotalBytes+Buffer_Offset)%6)==0))
409 {
410 if (Buffer[Buffer_Offset ]==0x96
411 && Buffer[Buffer_Offset+1]==0xF8
412 && Buffer[Buffer_Offset+2]==0x72
413 && Buffer[Buffer_Offset+3]==0xA5
414 && Buffer[Buffer_Offset+4]==0x4E
415 && Buffer[Buffer_Offset+5]==0x1F) // 24-bit, BE
416 {
417 Container_Bits=24;
418 Stream_Bits=24;
419 Endianness='B'; // BE
420 break; // while()
421 }
422 if (Buffer[Buffer_Offset ]==0x72
423 && Buffer[Buffer_Offset+1]==0xF8
424 && Buffer[Buffer_Offset+2]==0x96
425 && Buffer[Buffer_Offset+3]==0x1F
426 && Buffer[Buffer_Offset+4]==0x4E
427 && Buffer[Buffer_Offset+5]==0xA5) // 24-bit, LE
428 {
429 Container_Bits=24;
430 Stream_Bits=24;
431 Endianness='L'; // LE
432 break; // while()
433 }
434 if (Buffer[Buffer_Offset ]==0x00
435 && Buffer[Buffer_Offset+1]==0xF8
436 && Buffer[Buffer_Offset+2]==0x72
437 && Buffer[Buffer_Offset+3]==0x00
438 && Buffer[Buffer_Offset+4]==0x4E

440 {
441 Container_Bits=24;

444 NullPadding_Size=1;

446 }

459 }

464 && Buffer[Buffer_Offset+4]==0xE1

466 {
467 Container_Bits=24;
468 Stream_Bits=20;
469 Endianness='B'; // BE
470 break; // while()
471 }
472 if (Buffer[Buffer_Offset ]==0x20
473 && Buffer[Buffer_Offset+1]==0x87
474 && Buffer[Buffer_Offset+2]==0x6F
475 && Buffer[Buffer_Offset+3]==0xF0
476 && Buffer[Buffer_Offset+4]==0xE1

479 Container_Bits=24;

488 && Buffer[Buffer_Offset+1]==0x00

501 }
502 if (Buffer[Buffer_Offset ]==0x00
503 && Buffer[Buffer_Offset+1]==0x00
504 && Buffer[Buffer_Offset+2]==0x72
505 && Buffer[Buffer_Offset+3]==0xF8
506 && Buffer[Buffer_Offset+4]==0x00
507 && Buffer[Buffer_Offset+5]==0x00

511 Container_Bits=32;

514 NullPadding_Size=2;
515 break; // while()
516 }
517 if (Buffer[Buffer_Offset ]==0x00
518 && Buffer[Buffer_Offset+1]==0x6F
519 && Buffer[Buffer_Offset+2]==0x87
520 && Buffer[Buffer_Offset+3]==0x20
521 && Buffer[Buffer_Offset+4]==0x00
522 && Buffer[Buffer_Offset+5]==0x54
523 && Buffer[Buffer_Offset+6]==0xE1
524 && Buffer[Buffer_Offset+7]==0xF0) // 20-bit in 32-bit, BE
525 {
526 Container_Bits=32;
527 Stream_Bits=20;
528 Endianness='B'; // BE
529 NullPadding_Size=1;
530 break; // while()
531 }
532 if (Buffer[Buffer_Offset ]==0x00
533 && Buffer[Buffer_Offset+1]==0x20
534 && Buffer[Buffer_Offset+2]==0x87
535 && Buffer[Buffer_Offset+3]==0x6F
536 && Buffer[Buffer_Offset+4]==0x00
537 && Buffer[Buffer_Offset+5]==0xF0
538 && Buffer[Buffer_Offset+6]==0xE1
539 && Buffer[Buffer_Offset+7]==0x54) // 20-bit in 32-bit, LE
540 {
541 Container_Bits=32;
542 Stream_Bits=20;
543 Endianness='L'; // LE
544 NullPadding_Size=1;
545 break; // while()
546 }
547 if (Buffer[Buffer_Offset ]==0x00
548 && Buffer[Buffer_Offset+1]==0x96
549 && Buffer[Buffer_Offset+2]==0xF8
550 && Buffer[Buffer_Offset+3]==0x72
551 && Buffer[Buffer_Offset+4]==0x00
552 && Buffer[Buffer_Offset+5]==0xA5
553 && Buffer[Buffer_Offset+6]==0x4E
554 && Buffer[Buffer_Offset+7]==0x1F) // 24-bit in 32-bit, BE
555 {
556 Container_Bits=32;
557 Stream_Bits=24;
558 Endianness='B'; // BE
559 NullPadding_Size=1;
560 break; // while()
561 }
562 if (Buffer[Buffer_Offset ]==0x00
563 && Buffer[Buffer_Offset+1]==0x72
564 && Buffer[Buffer_Offset+2]==0xF8
565 && Buffer[Buffer_Offset+3]==0x96
566 && Buffer[Buffer_Offset+4]==0x00
567 && Buffer[Buffer_Offset+5]==0x1F
568 && Buffer[Buffer_Offset+6]==0x4E
569 && Buffer[Buffer_Offset+7]==0xA5) // 24-bit in 32-bit, LE
570 {
571 Container_Bits=32;
572 Stream_Bits=24;
573 Endianness='L'; // LE
574 NullPadding_Size=1;
575 break; // while()
576 }
577 }
578
579 if (Container_Bits>=4 && Aligned)
580 Buffer_Offset+=Container_Bits/4;
581 else
582 Buffer_Offset++;
583 }
584
585 // Guard band
586 GuardBand_Before+=Buffer_Offset-Buffer_Offset_Base;
587
588 // Parsing last bytes if needed
589 if (Buffer_Offset+16>Buffer_Size)
590 {
591 return false;
592 }
593
594 if (!Status[IsAccepted])
595 Accept("SMPTE ST 337");
596
597 // Synched
598 return true;
599}

601//---------------------------------------------------------------------------
602bool File_SmpteSt0337::Synched_Test()
603{
604 // Skip NULL padding
605 size_t Buffer_Offset_Temp=Buffer_Offset;
606 if (Aligned)
607 {
608 if (Container_Bits==16)
609 {
610 while ((Buffer_TotalBytes+Buffer_Offset_Temp)%4) // Padding in part of the AES3 block
611 {
612 if (Buffer_Offset_Temp+1>Buffer_Size)
613 {
614 Element_WaitForMoreData();
615 return false;
616 }
617 if (Buffer[Buffer_Offset_Temp])
618 {
619 Trusted_IsNot("Bad sync");
620 return true;
621 }
622 Buffer_Offset_Temp++;
623 }
624 while(Buffer_Offset_Temp+4<=Buffer_Size && CC4(Buffer+Buffer_Offset_Temp)==0x00000000)
625 Buffer_Offset_Temp+=4;
626 if (Buffer_Offset_Temp+4>Buffer_Size)
627 {
628 Element_WaitForMoreData();
629 return false;
630 }
631 }
632 if (Container_Bits==20)
633 {
634 while ((Buffer_TotalBytes+Buffer_Offset_Temp)%5) // Padding in part of the AES3 block
635 {
636 if (Buffer_Offset_Temp+1>Buffer_Size)
637 {
638 Element_WaitForMoreData();
639 return false;
640 }
641 if (Buffer[Buffer_Offset_Temp])
642 {
643 Trusted_IsNot("Bad sync");
644 return true;
645 }
646 Buffer_Offset_Temp++;
647 }
648 while(Buffer_Offset_Temp+5<=Buffer_Size && CC5(Buffer+Buffer_Offset_Temp)==0x0000000000LL)
649 Buffer_Offset_Temp+=5;
650 if (Buffer_Offset_Temp+5>Buffer_Size)
651 {
652 Element_WaitForMoreData();
653 return false;
654 }
655 }
656 if (Container_Bits==24)
657 {
658 while ((Buffer_TotalBytes+Buffer_Offset_Temp)%6) // Padding in part of the AES3 block
659 {
660 if (Buffer_Offset_Temp+1>Buffer_Size)
661 {
662 Element_WaitForMoreData();
663 return false;
664 }
665 if (Buffer[Buffer_Offset_Temp])
666 {
667 Trusted_IsNot("Bad sync");
668 return true;
669 }
670 Buffer_Offset_Temp++;
671 }
672 while(Buffer_Offset_Temp+6<=Buffer_Size && CC6(Buffer+Buffer_Offset_Temp)==0x000000000000LL)
673 Buffer_Offset_Temp+=6;
674 if (Buffer_Offset_Temp+6>Buffer_Size)
675 {
676 Element_WaitForMoreData();
677 return false;
678 }
679 }
680 else if (Container_Bits==32)
681 {
682 while ((Buffer_TotalBytes+Buffer_Offset_Temp)%8) // Padding in part of the AES3 block
683 {
684 if (Buffer_Offset_Temp+1>Buffer_Size)
685 {
686 Element_WaitForMoreData();
687 return false;
688 }
689 if (Buffer[Buffer_Offset_Temp])
690 {
691 Trusted_IsNot("Bad sync");
692 return true;
693 }
694 Buffer_Offset_Temp++;
695 }
696 while(Buffer_Offset_Temp+8<=Buffer_Size && CC8(Buffer+Buffer_Offset_Temp)==0x0000000000000000LL)
697 Buffer_Offset_Temp+=8;
698 if (Buffer_Offset_Temp+8>Buffer_Size)
699 {
700 Element_WaitForMoreData();
701 return false;
702 }
703 }
704 }
705 else
706 {
707 while(Buffer_Offset_Temp+NullPadding_Size<Buffer_Size && !Buffer[Buffer_Offset_Temp+NullPadding_Size])
708 Buffer_Offset_Temp++;
709 if (Buffer_Offset_Temp+NullPadding_Size>=Buffer_Size)
710 {
711 Element_WaitForMoreData();
712 return false;
713 }
714 }
715
716 #if MEDIAINFO_TRACE
717 if (Buffer_Offset_Temp-Buffer_Offset)
718 {
719 Element_Size=Buffer_Offset_Temp-Buffer_Offset;
720 Skip_XX(Buffer_Offset_Temp-Buffer_Offset, "Guard band");
721
722 GuardBand_Before+=Buffer_Offset_Temp-Buffer_Offset;
723 }
724 #endif // MEDIAINFO_TRACE
725 Buffer_Offset=Buffer_Offset_Temp;
726
727 // Must have enough buffer for having header
728 if (Buffer_Offset+16>Buffer_Size)
729 return false;
730
731 // Quick test of synchro
732 switch (Endianness)
733 {
734 case 'B' :
735 switch (Container_Bits)
736 {
737 case 16 : if (CC4(Buffer+Buffer_Offset)!=0xF8724E1F) {Synched=false; return true;} break;
738 case 20 : if (CC5(Buffer+Buffer_Offset)!=0x6F87254E1FLL) {Synched=false; return true;} break;

807void File_SmpteSt0337::Header_Parse()
808{
809 // Parsing
810 int32u Size=0;
811 switch (Endianness)
812 {
813 case 'B' :
814 switch (Container_Bits)
815 {
816 case 16 : Size=BigEndian2int16u(Buffer+Buffer_Offset+6) ; break;
817 case 20 : Size=BigEndian2int24u(Buffer+Buffer_Offset+7)&0x0FFFFF; break;
818 case 24 :
819 switch (Stream_Bits)
820 {
821 case 16 : Size=BigEndian2int16u(Buffer+Buffer_Offset+9) ; break;
822 case 20 : Size=BigEndian2int24u(Buffer+Buffer_Offset+9)>>4; break;
823 case 24 : Size=BigEndian2int24u(Buffer+Buffer_Offset+9) ; break;
824 default : ;
825 }
826 break;
827 case 32 :
828 switch (Stream_Bits)
829 {
830 case 16 : Size=BigEndian2int16u(Buffer+Buffer_Offset+0xE) ; break;
831 case 20 : Size=BigEndian2int24u(Buffer+Buffer_Offset+0xD)>>4; break;
832 case 24 : Size=BigEndian2int24u(Buffer+Buffer_Offset+0xD) ; break;
833 default : ;
834 }
835 break;
836 default : ;
837 }
838 break;
839 case 'L' :
840 switch (Container_Bits)
841 {
842 case 16 : Size=LittleEndian2int16u(Buffer+Buffer_Offset+6) ; break;
843 case 20 : Size=LittleEndian2int24u(Buffer+Buffer_Offset+7)>>4; break;
844 case 24 :
845 switch (Stream_Bits)
846 {
847 case 16 : Size=LittleEndian2int16u(Buffer+Buffer_Offset+0xA) ; break;
848 case 20 : Size=LittleEndian2int24u(Buffer+Buffer_Offset+0x9)>>4; break;
849 case 24 : Size=LittleEndian2int24u(Buffer+Buffer_Offset+0x9) ; break;
850 default : ;
851 }
852 break;
853 case 32 :
854 switch (Stream_Bits)
855 {
856 case 16 : Size=LittleEndian2int16u(Buffer+Buffer_Offset+0xE) ; break;
857 case 20 : Size=LittleEndian2int24u(Buffer+Buffer_Offset+0xD)>>4; break;
858 case 24 : Size=LittleEndian2int24u(Buffer+Buffer_Offset+0xD) ; break;
859 default : ;
860 }
861 break;
862 default : ;
863 }
864 break;
865 default : ; // Should never happen
866 }
867
868 // Adaptation
869 if (Container_Bits!=Stream_Bits)
870 {
871 Size*=Container_Bits; Size/=Stream_Bits;
872 GuardBand_Before*=Container_Bits; GuardBand_Before/=Stream_Bits;
873 }
874
875 // Coherency test
876 if (!IsSub && !Status[IsAccepted])
877 {
878 size_t Offset=Buffer_Offset+(size_t)(Container_Bits*4/8+Size/8);
879 while (Offset<Buffer_Size && Buffer[Offset]==0x00)
880 Offset++;
881 if (Offset+Container_Bits/4>Buffer_Size)
882 {
883 Element_WaitForMoreData();
884 return;
885 }
886 Offset/=Container_Bits/4;
887 Offset*=Container_Bits/4;
888 bool IsOK=true;
889 for (int8u Pos=0; Pos<Container_Bits/4; Pos++)
890 if (Buffer[Buffer_Offset+Pos]!=Buffer[Offset+Pos])
891 {
892 IsOK=false;
893 break;
894 }
895 if (!IsOK)
896 {
897 Trusted_IsNot("Bad sync");
898 Buffer_Offset++;
899 return;
900 }
901 }
902
903 // Filling
904 Header_Fill_Size(Container_Bits*4/8+Size/8);
905 Header_Fill_Code(0, "SMPTE ST 337");
906
907 //Guard band
908 if (IsSub && FrameInfo.DTS!=(int64u)-1)
909 GuardBand_After+=Element_Size-(Container_Bits*4/8+Size/8);
910}

912//---------------------------------------------------------------------------
913void File_SmpteSt0337::Data_Parse()
914{
915 #if MEDIAINFO_DEMUX
916 FrameInfo.PTS=FrameInfo.DTS;
917 Demux_random_access=true;
918 Element_Code=(int64u)-1;
919 #endif //MEDIAINFO_DEMUX
920
921 // Adapting
922 const int8u* Save_Buffer=NULL;
923 size_t Save_Buffer_Offset=0;
924 size_t Save_Buffer_Size=0;
925 int64u Save_Element_Size=0;
926
927 if (Endianness=='L'|| Container_Bits!=Stream_Bits)
928 {
929 int8u* Info=new int8u[(size_t)Element_Size];
930 int8u* Info_Temp=Info;
931
932 if (Endianness=='L' && Container_Bits==16 && Stream_Bits==16)
933 {
934 // Source: 16LE / L1L0 L3L2 R1R0 R3R2
935 // Dest : 16BE / L3L2 L1L0 R3R2 R1R0
936 while (Element_Offset+4<=Element_Size)
937 {
938 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
939
940 *(Info_Temp++)= Buffer[Buffer_Pos+1] ;
941 *(Info_Temp++)= Buffer[Buffer_Pos ] ;
942 *(Info_Temp++)= Buffer[Buffer_Pos+3] ;
943 *(Info_Temp++)= Buffer[Buffer_Pos+2] ;
944
945 Element_Offset+=4;
946 }
947 if (Element_Offset+2<=Element_Size) // Only in half of the AES3 stream
948 {
949 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
950
951 *(Info_Temp++)= Buffer[Buffer_Pos+1] ;
952 *(Info_Temp++)= Buffer[Buffer_Pos ] ;
953
954 Element_Offset+=2;
955 }
956 }
957
958 if (Endianness=='L' && Container_Bits==20 && Stream_Bits==20)
959 {
960 // Source: 20LE / L1L0 L3L2 R0L4 R2R1 R4R3
961 // Dest : 20BE / L4L3 L2L1 L0R4 R3R2 R1R0
962 while (Element_Offset+5<=Element_Size)
963 {
964 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
965
966 *(Info_Temp++)=(Buffer[Buffer_Pos+2]<<4 ) | (Buffer[Buffer_Pos+1]>>4 );
967 *(Info_Temp++)=(Buffer[Buffer_Pos+1]<<4 ) | (Buffer[Buffer_Pos+0]>>4 );
968 *(Info_Temp++)=(Buffer[Buffer_Pos+0]<<4 ) | (Buffer[Buffer_Pos+4]>>4 );
969 *(Info_Temp++)=(Buffer[Buffer_Pos+4]<<4 ) | (Buffer[Buffer_Pos+3]>>4 );
970 *(Info_Temp++)=(Buffer[Buffer_Pos+3]<<4 ) | (Buffer[Buffer_Pos+2]>>4 );
971
972 Element_Offset+=5;
973 }
974 }
975
976 if (Endianness=='L' && Container_Bits==24 && Stream_Bits==16)
977 {
978 // Source: XXXX L1L0 L3L2 XXXX R1R0 R3R2
979 // Dest : 16BE / L3L2 L1L0 R3R2 R1R0
980 while (Element_Offset+6<=Element_Size)
981 {
982 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
983
984 *(Info_Temp++)= Buffer[Buffer_Pos+2] ;
985 *(Info_Temp++)= Buffer[Buffer_Pos+1] ;
986 *(Info_Temp++)= Buffer[Buffer_Pos+5] ;
987 *(Info_Temp++)= Buffer[Buffer_Pos+4] ;
988
989 Element_Offset+=6;
990 }
991 }
992
993 if (Endianness=='L' && Container_Bits==24 && Stream_Bits==20)
994 {
995 // Source: L0XX L2L1 L4L3 R0XX R2R1 R4R3
996 // Dest : 20BE / L4L3 L2L1 L0R4 R3R2 R1R0
997 while (Element_Offset+6<=Element_Size)
998 {
999 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
1000
1001 *(Info_Temp++)= Buffer[Buffer_Pos+2] ;
1002 *(Info_Temp++)= Buffer[Buffer_Pos+1] ;
1003 *(Info_Temp++)=(Buffer[Buffer_Pos ]&0xF0) | (Buffer[Buffer_Pos+5]>>4 );
1004 *(Info_Temp++)=(Buffer[Buffer_Pos+5]<<4 ) | (Buffer[Buffer_Pos+4]>>4 );
1005 *(Info_Temp++)=(Buffer[Buffer_Pos+4]<<4 ) | (Buffer[Buffer_Pos+3]>>4 );
1006
1007 Element_Offset+=6;
1008 }
1009 }
1010
1011 if (Endianness=='L' && Container_Bits==24 && Stream_Bits==24)
1012 {
1013 // Source: 24LE / L1L0 L3L2 L5L3 R1R0 R3R2 R5R4
1014 // Dest : 24BE / L5L3 L3L2 L1L0 R5R4 R3R2 R1R0
1015 while (Element_Offset+6<=Element_Size)
1016 {
1017 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
1018
1019 *(Info_Temp++)= Buffer[Buffer_Pos+2] ;
1020 *(Info_Temp++)= Buffer[Buffer_Pos+1] ;
1021 *(Info_Temp++)= Buffer[Buffer_Pos ] ;
1022 *(Info_Temp++)= Buffer[Buffer_Pos+5] ;
1023 *(Info_Temp++)= Buffer[Buffer_Pos+4] ;
1024 *(Info_Temp++)= Buffer[Buffer_Pos+3] ;
1025
1026 Element_Offset+=6;
1027 }
1028 }
1029
1030 if (Endianness=='L' && Container_Bits==32 && Stream_Bits==16)
1031 {
1032 // Source: XXXX XXXX L1L0 L3L2 XXXX XXXX R1R0 R3R2
1033 // Dest : 16BE / L3L2 L1L0 R3R2 R1R0
1034 while (Element_Offset+8<=Element_Size)
1035 {
1036 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
1037
1038 *(Info_Temp++)= Buffer[Buffer_Pos+3] ;
1039 *(Info_Temp++)= Buffer[Buffer_Pos+2] ;
1040 *(Info_Temp++)= Buffer[Buffer_Pos+7] ;
1041 *(Info_Temp++)= Buffer[Buffer_Pos+6] ;
1042
1043 Element_Offset+=8;
1044 }
1045 }
1046 if (Endianness=='L' && Container_Bits==32 && Stream_Bits==20)
1047 {
1048 // Source: XXXX L0XX L2L1 L4L3 XXXX R0XX R2R1 R4R3
1049 // Dest : 20BE / L4L3 L2L1 L0R4 R3R2 R1R0
1050 while (Element_Offset+8<=Element_Size)
1051 {
1052 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
1053
1054 *(Info_Temp++)= Buffer[Buffer_Pos+3] ;
1055 *(Info_Temp++)= Buffer[Buffer_Pos+2] ;
1056 *(Info_Temp++)=(Buffer[Buffer_Pos+1]&0xF0) | (Buffer[Buffer_Pos+7]>>4 );
1057 *(Info_Temp++)=(Buffer[Buffer_Pos+7]<<4 ) | (Buffer[Buffer_Pos+6]>>4 );
1058 *(Info_Temp++)=(Buffer[Buffer_Pos+6]<<4 ) | (Buffer[Buffer_Pos+5]>>4 );
1059
1060 Element_Offset+=8;
1061 }
1062 }
1063
1064 if (Endianness=='L' && Container_Bits==32 && Stream_Bits==24)
1065 {
1066 // Source: XXXX L1L0 L3L2 L5L3 XXXX R1R0 R3R2 R5R4
1067 // Dest : 24BE / L5L3 L3L2 L1L0 R5R4 R3R2 R1R0
1068 while (Element_Offset+8<=Element_Size)
1069 {
1070 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
1071
1072 *(Info_Temp++)= Buffer[Buffer_Pos+3] ;
1073 *(Info_Temp++)= Buffer[Buffer_Pos+2] ;
1074 *(Info_Temp++)= Buffer[Buffer_Pos+1] ;
1075 *(Info_Temp++)= Buffer[Buffer_Pos+7] ;
1076 *(Info_Temp++)= Buffer[Buffer_Pos+6] ;
1077 *(Info_Temp++)= Buffer[Buffer_Pos+5] ;
1078
1079 Element_Offset+=8;
1080 }
1081 }
1082
1083 if (Endianness=='B' && Container_Bits==24 && Stream_Bits==20)
1084 {
1085 // Source: L4L3 L2L1 L0XX R4R3 R2R1 R0XX
1086 // Dest : 20BE / L4L3 L2L1 L0R4 R3R2 R1R0
1087 while (Element_Offset+6<=Element_Size)
1088 {
1089 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
1090
1091 *(Info_Temp++)= Buffer[Buffer_Pos ] ;
1092 *(Info_Temp++)= Buffer[Buffer_Pos+1] ;
1093 *(Info_Temp++)=(Buffer[Buffer_Pos+2]&0xF0) | (Buffer[Buffer_Pos+3]>>4 );
1094 *(Info_Temp++)=(Buffer[Buffer_Pos+3]<<4 ) | (Buffer[Buffer_Pos+4]>>4 );
1095 *(Info_Temp++)=(Buffer[Buffer_Pos+4]<<4 ) | (Buffer[Buffer_Pos+5]>>4 );
1096
1097 Element_Offset+=6;
1098 }
1099 }
1100
1101 Save_Buffer=Buffer;
1102 Save_Buffer_Offset=Buffer_Offset;
1103 Save_Buffer_Size=Buffer_Size;
1104 Save_Element_Size=Element_Size;
1105 File_Offset+=Buffer_Offset;
1106 Buffer=Info;
1107 Buffer_Offset=0;
1108 Buffer_Size=Info_Temp-Info;
1109 Element_Offset=0;
1110 Element_Size=Buffer_Size;
1111 }
1112
1113 // Parsing
1114 int32u length_code;
1115 Element_Begin1("Header");
1116 BS_Begin();
1117 Skip_S3(Stream_Bits, "Pa");
1118 Skip_S3(Stream_Bits, "Pb");
1119 Element_Begin1("Pc");
1120 Skip_S1( 3, "data_stream_number");
1121 Skip_S1( 5, "data_type_dependent");
1122 Skip_SB( "error_flag");
1123 Info_S1( 2, data_mode, "data_mode"); Param_Info2(16+4*data_mode, " bits");
1124 Get_S1 ( 5, data_type, "data_type"); Param_Info1(Smpte_St0337_data_type[data_type]);
1125 if (Stream_Bits>16)
1126 Skip_S1( 4, "reserved");
1127 if (Stream_Bits>20)
1128 Skip_S1( 4, "reserved");
1129 Element_End0();
1130 Get_S3 (Stream_Bits, length_code, "length_code"); Param_Info2(length_code/8, " bytes");
1131 BS_End();
1132 Element_End0();
1133
1134 if (Parser==NULL)
1135 {
1136 switch(data_type)
1137 {
1138 // SMPTE ST338
1139 case 1 : // AC-3
1140 case 16 : // E-AC-3 (professional)
1141 case 21 : // E-AC-3 (consumer)
1142 Parser=new File_Ac3();
1143 ((File_Ac3*)Parser)->Frame_Count_Valid=2;
1144 #if MEDIAINFO_DEMUX
1145 if (Config->Demux_Unpacketize_Get())
1146 {
1147 Demux_UnpacketizeContainer=false; //No demux from this parser
1148 Demux_Level=4; //Intermediate
1149 Parser->Demux_Level=2; //Container
1150 Parser->Demux_UnpacketizeContainer=true;
1151 }
1152 #endif //MEDIAINFO_DEMUX
1153 break;
1154 case 4 : // MPEG-1 Layer 1
1155 case 5 : // MPEG-1 Layer 2/3, MPEG-2 Layer 1/2/3 without extension
1156 case 6 : // MPEG-2 Layer 1/2/3 with extension
1157 case 8 : // MPEG-2 Layer 1 low frequency
1158 case 9 : // MPEG-2 Layer 2/3 low frequency
1159 Parser=new File_Mpega();
1160 break;
1161 case 7 : // MPEG-2 AAC in ADTS
1162 case 19 : // MPEG-2 AAC in ADTS low frequency
1163 #if defined(MEDIAINFO_AAC_YES)
1164 Parser=new File_Aac();
1165 ((File_Aac*)Parser)->Mode=File_Aac::Mode_ADTS;
1166 #else
1167 {
1168 //Filling
1169 File__Analyze* Parser=new File_Unknown();
1170 Open_Buffer_Init(Parser);
1171 Parser->Stream_Prepare(Stream_Audio);
1172 Parser->Fill(Stream_Audio, 0, Audio_Format, "AAC");
1173 Parser->Fill(Stream_Audio, 0, Audio_MuxingMode, "ADTS");
1174 }
1175 #endif //defined(MEDIAINFO_AAC_YES)
1176 break;
1177 case 10 : // MPEG-4 AAC in ADTS or LATM
1178 case 11 : // MPEG-4 AAC in ADTS or LATM
1179 #if defined(MEDIAINFO_AAC_YES)
1180 Parser=new File_Aac();
1181 #else
1182 {
1183 //Filling
1184 File__Analyze* Parser=new File_Unknown();
1185 Open_Buffer_Init(Parser);
1186 Parser->Stream_Prepare(Stream_Audio);
1187 Parser->Fill(Stream_Audio, 0, Audio_Format, "AAC");
1188 }
1189 #endif //defined(MEDIAINFO_AAC_YES)
1190 break;
1191 case 28 : // Dolby E
1192 Parser=new File_DolbyE();
1193 break;
1194 default : ;
1195 }
1196
1197 if (Parser)
1198 {
1199 Open_Buffer_Init(Parser);
1200 }
1201 }
1202
1203 #if MEDIAINFO_DEMUX
1204 if (Save_Buffer)
1205 {
1206 std::swap(Buffer, Save_Buffer);
1207 std::swap(Buffer_Offset, Save_Buffer_Offset);
1208 std::swap(Buffer_Size, Save_Buffer_Size);
1209 std::swap(Element_Size, Save_Element_Size);
1210 File_Offset-=Buffer_Offset;
1211 }
1212
1213 if (data_type==28) //If Dolby E, we must demux the SMPTE ST 337 header too (TODO: add an option for forcing SMPTE ST 337 header)
1214 {
1215 int64u Demux_Element_Offset=Element_Offset;
1216 Element_Offset=0;
1217
1218 if (Container_Bits==20)
1219 {
1220 //We must pad to 24 bits
1221 int8u* Info2=new int8u[(size_t)Element_Size*6/5];
1222 size_t Info2_Offset=0;
1223 const int8u* Demux_Buffer=Buffer+Buffer_Offset;
1224 size_t Demux_Buffer_Size=(size_t)Element_Size;
1225 size_t Demux_Buffer_Pos=0;
1226
1227 // Source: 20LE L1L0 L3L2 R0L4 R2R1 R4R3
1228 // Dest : L0XX L2L1 L4L3 R0XX R2R1 R4R3
1229 while (Demux_Buffer_Pos+5<=Demux_Buffer_Size)
1230 {
1231 Info2[Info2_Offset+0]= Demux_Buffer[Demux_Buffer_Pos+0]<<4 ;
1232 Info2[Info2_Offset+1]=(Demux_Buffer[Demux_Buffer_Pos+1]<<4 ) | (Demux_Buffer[Demux_Buffer_Pos+0]>>4 );
1233 Info2[Info2_Offset+2]=(Demux_Buffer[Demux_Buffer_Pos+2]<<4 ) | (Demux_Buffer[Demux_Buffer_Pos+1]>>4 );
1234 Info2[Info2_Offset+3]= Demux_Buffer[Demux_Buffer_Pos+2]&0xF0 ;
1235 Info2[Info2_Offset+4]= Demux_Buffer[Demux_Buffer_Pos+3] ;
1236 Info2[Info2_Offset+5]= Demux_Buffer[Demux_Buffer_Pos+4] ;
1237
1238 Info2_Offset+=6;
1239 Demux_Buffer_Pos+=5;
1240 }
1241
1242 Demux(Info2, Info2_Offset, ContentType_MainStream);
1243
1244 delete[] Info2;
1245 }
1246 else
1247 Demux(Buffer+Buffer_Offset, (size_t)Element_Size, ContentType_MainStream);
1248
1249 Element_Offset=Demux_Element_Offset;
1250 }
1251 else
1252 Demux(Buffer+Buffer_Offset+Container_Bits/2, (size_t)(Element_Size-Container_Bits/2), ContentType_MainStream);
1253
1254 if (Save_Buffer)
1255 {
1256 File_Offset+=Buffer_Offset;
1257 std::swap(Buffer, Save_Buffer);
1258 std::swap(Buffer_Offset, Save_Buffer_Offset);
1259 std::swap(Buffer_Size, Save_Buffer_Size);
1260 std::swap(Element_Size, Save_Element_Size);
1261 }
1262 #endif //MEDIAINFO_DEMUX
1263
1264 if (Parser && !Parser->Status[IsFinished])
1265 {
1266 switch(data_type)
1267 {
1268 case 28 :
1269 ((File_DolbyE*)Parser)->GuardBand_Before+=GuardBand_Before;
1270 ((File_DolbyE*)Parser)->GuardBand_After+=GuardBand_After;
1271 break;
1272 default : ;
1273 }
1274
1275 Parser->FrameInfo=FrameInfo;
1276 Open_Buffer_Continue(Parser, Buffer+Buffer_Offset+(size_t)Element_Offset, (size_t)(Element_Size-Element_Offset));
1277 Element_Offset=Element_Size;
1278 #if MEDIAINFO_DEMUX
1279 FrameInfo.DUR=Parser->FrameInfo.DUR;
1280 if (FrameInfo.DUR!=(int64u)-1)
1281 FrameInfo.DTS+=FrameInfo.DUR;
1282 else
1283 FrameInfo.DTS=(int64u)-1;
1284 FrameInfo.PTS=FrameInfo.DTS;
1285 #endif // MEDIAINFO_DEMUX
1286 }
1287 else
1288 {
1289 Skip_XX(Element_Size-Element_Offset, "Data");
1290 }
1291
1292 FILLING_BEGIN();
1293 FrameSizes[IsSub?Buffer_Size:((GuardBand_Before+Element_Size)*Container_Bits/Stream_Bits)]++;
1294
1295 Frame_Count++;
1296 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
1297 Frame_Count_NotParsedIncluded++;
1298
1299 if (Parser==NULL || (Frame_Count>=2 && Parser->Status[IsFilled]))
1300 Fill("SMPTE ST 337");
1301 if (Parser==NULL || (Frame_Count>=2 && Parser->Status[IsFinished]))
1302 Finish("SMPTE ST 337");
1303 FILLING_END();
1304
1305 if (Save_Buffer)
1306 {
1307 delete[] Buffer;
1308 Buffer=Save_Buffer;
1309 Buffer_Offset=Save_Buffer_Offset;
1310 Buffer_Size=Save_Buffer_Size;
1311 File_Offset-=Buffer_Offset;
1312 }
1313
1314 // Guard band
1315 GuardBand_Before=0;
1316 GuardBand_After=0;
1317}
