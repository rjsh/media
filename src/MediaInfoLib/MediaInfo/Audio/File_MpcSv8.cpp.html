77//---------------------------------------------------------------------------
78bool File_MpcSv8::FileHeader_Begin()
79{
80 //Element_Size
81 if (Buffer_Size<4)
82 return false; //Must wait for more data
83
84 if (CC4(Buffer)!=0x4D50434B) //"MPCK"
85 {
86 File__Tags_Helper::Reject("Musepack SV8");
87 return false;
88 }
89
90 //All should be OK...
91 return true;
92}

94//***************************************************************************
95// Buffer - File header
96//***************************************************************************
97
98//---------------------------------------------------------------------------
99void File_MpcSv8::FileHeader_Parse()
100{
101 //Parsing
102 Skip_C4( "Magic Number");
103
104 FILLING_BEGIN();
105 File__Tags_Helper::Accept("MpcSv8");
106
107 File__Tags_Helper::Stream_Prepare(Stream_Audio);
108 Fill(Stream_Audio, 0, Audio_Format, "Musepack SV8");
109 Fill(Stream_Audio, 0, Audio_Codec, "SV8");
110 FILLING_END();
111}

127//---------------------------------------------------------------------------
128void File_MpcSv8::Header_Parse()
129{
130 //Parsing
131 int64u Size;
132 int16u Key;
133 Get_C2 (Key, "Key");
134 Get_VS (Size, "Size");
135
136 //Filling
137 Header_Fill_Code(Key, Ztring().From_CC4(Key<<16)); //Quick filling for CC2 with text
138 Header_Fill_Size(Key==Elements::AP?Element_Offset:Size); //We don't need the data of audio packet, and we will stop here...
139}

145//---------------------------------------------------------------------------
146void File_MpcSv8::Data_Parse()
147{
148 #define CASE_INFO(_NAME, _DETAIL) \
149 case Elements::_NAME : Element_Info1(_DETAIL); _NAME(); break;
150
151 //Parsing
152 switch (Element_Code)
153 {
154 CASE_INFO(AP, "Audio Packet");
155 CASE_INFO(CT, "Chapter-Tag");
156 CASE_INFO(EI, "Encoder Info");
157 CASE_INFO(RG, "Replay Gain");
158 CASE_INFO(SE, "Stream End");
159 CASE_INFO(SH, "Stream Header");
160 CASE_INFO(SO, "Seek Table Offset");
161 CASE_INFO(ST, "Seek Table");
162 default : Skip_XX(Element_Size, "Data");
163 }
164}

177//---------------------------------------------------------------------------
178void File_MpcSv8::EI()
179{
180 //Parsing
181 int8u Quality, Version1, Version2, Version3;
182 bool PNS;
183 BS_Begin();
184 Get_S1 (7, Quality, "Quality");
185 Get_SB ( PNS, "PNS");
186 BS_End();
187 Get_B1 (Version1, "Major version");
188 Get_B1 (Version2, "Minor version");
189 Get_B1 (Version3, "Build");
190}
191
192//---------------------------------------------------------------------------
193void File_MpcSv8::RG()
194{
195 //Parsing
196 int16u TitleGain, AlbumGain;
197 Skip_B1( "Version");
198 Get_L2 (TitleGain, "Title gain"); Param_Info3(((float32)((int16s)TitleGain))/1000, " dB", 2);
199 Skip_L2( "Title peak");
200 Get_L2 (AlbumGain, "Album gain"); Param_Info3(((float32)((int16s)TitleGain))/1000, " dB", 2);
201 Skip_L2( "Album peak");
202}
203
204//---------------------------------------------------------------------------
205void File_MpcSv8::SH()
206{
207 //Parsing
208 int64u SampleCount;
209 int8u Version, SampleFrequency, ChannelCount;
210 bool MidSideStereo;
211 Skip_B4( "CRC32");
212 Get_B1 (Version, "Version");
213 Get_VS (SampleCount, "Sample count");
214 Skip_VS( "Beginning silence");
215 BS_Begin();
216 Get_S1 (3, SampleFrequency, "Sample frequency"); Param_Info1(Mpc_SampleFreq[SampleFrequency]);
217 Skip_S1(5, "Max used bands");
218 Get_S1 (4, ChannelCount, "Channel count");
219 Get_SB ( MidSideStereo, "Mid side stereo used");
220 Skip_S1(3, "Audio block frames");
221 BS_End();
222
223 //Filling
224 FILLING_BEGIN();
225 Fill(Stream_Audio, 0, Audio_SamplingRate, Mpc_SampleFreq[SampleFrequency]);
226 if (SampleCount)
227 {
228 Fill(Stream_Audio, 0, Audio_SamplingCount, SampleCount);
229 Fill(Stream_Audio, 0, Audio_Duration, SampleCount*1000/Mpc_SampleFreq[SampleFrequency]);
230 Fill(Stream_Audio, 0, Audio_BitRate, File_Size*8*Mpc_SampleFreq[SampleFrequency]/SampleCount); //Should be more precise...
231 }
232 Fill(Stream_Audio, 0, Audio_BitDepth, 16); //MPC support only 16 bits
233 FILLING_END();
234}
