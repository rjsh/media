105//***************************************************************************
106// Streams management

108
109//---------------------------------------------------------------------------
110void File_Pcm_M2ts::Streams_Fill()
111{
112 Stream_Prepare(Stream_Audio);
113 Fill(Stream_Audio, 0, Audio_Format, "PCM");
114 Fill(Stream_Audio, 0, Audio_Codec, "PCM");
115 Fill(Stream_Audio, 0, Audio_Codec_Family, "PCM");
116 Fill(Stream_Audio, 0, Audio_MuxingMode, "Blu-ray");
117 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "CBR");
118
119 int8u Channels=Pcm_M2TS_channel_assignment[channel_assignment];
120 if (Channels)
121 {
122 if (Pcm_M2TS_sampling_frequency[sampling_frequency])
123 Fill(Stream_Audio, 0, Audio_SamplingRate, Pcm_M2TS_sampling_frequency[sampling_frequency]);
124 if (Pcm_M2TS_bits_per_sample[bits_per_sample])
125 Fill(Stream_Audio, 0, Audio_BitDepth, Pcm_M2TS_bits_per_sample[bits_per_sample]);
126 Fill(Stream_Audio, 0, Audio_Channel_s_, Channels);
127 Fill(Stream_Audio, 0, Audio_ChannelPositions, Pcm_VOB_ChannelsPositions(channel_assignment));
128 Fill(Stream_Audio, 0, Audio_ChannelPositions_String2, Pcm_VOB_ChannelsPositions2(channel_assignment));
129 if (Pcm_M2TS_sampling_frequency[sampling_frequency] && Pcm_M2TS_bits_per_sample[bits_per_sample])
130 {
131 if (Channels%2)
132 Fill(Stream_Audio, 0, Audio_BitRate_Encoded, Pcm_M2TS_sampling_frequency[sampling_frequency]*(Channels+1)*Pcm_M2TS_bits_per_sample[bits_per_sample]); //Always by pair
133 Fill(Stream_Audio, 0, Audio_BitRate, Pcm_M2TS_sampling_frequency[sampling_frequency]*Channels*Pcm_M2TS_bits_per_sample[bits_per_sample]);
134 }
135 }
136
137 Fill(Stream_Audio, 0, Audio_Format_Settings, "Big");
138 Fill(Stream_Audio, 0, Audio_Format_Settings_Endianness, "Big");
139 Fill(Stream_Audio, 0, Audio_Codec_Settings, "Big");
140 Fill(Stream_Audio, 0, Audio_Codec_Settings_Endianness, "Big");
141 Fill(Stream_Audio, 0, Audio_Format_Settings, "Signed");
142 Fill(Stream_Audio, 0, Audio_Format_Settings_Sign, "Signed");
143 Fill(Stream_Audio, 0, Audio_Codec_Settings, "Signed");
144 Fill(Stream_Audio, 0, Audio_Codec_Settings_Sign, "Signed");
145}

147//***************************************************************************

151//---------------------------------------------------------------------------
152void File_Pcm_M2ts::Read_Buffer_Continue()
153{
154 if (Buffer_Size==0)
155 return;
156
157 //Parsing
158 int16u audio_data_payload_size;
159 Get_B2 ( audio_data_payload_size, "audio_data_payload_size");
160 BS_Begin();
161 Get_S1 (4, channel_assignment, "channel_assignment"); Param_Info2(Pcm_M2TS_channel_assignment[channel_assignment], " channel(s)");
162 Get_S1 (4, sampling_frequency, "sampling_frequency"); Param_Info2(Pcm_M2TS_sampling_frequency[sampling_frequency], " Hz");
163 Get_S1 (2, bits_per_sample, "bits_per_sample"); Param_Info2(Pcm_M2TS_bits_per_sample[bits_per_sample], " bits");
164 Skip_SB( "start_flag");
165 Skip_S1(5, "reserved");
166 BS_End();
167 Skip_XX(audio_data_payload_size, "audio_data_payload");
168
169 FILLING_BEGIN_PRECISE();
170 if (!Status[IsAccepted])
171 {
172 Accept();
173 Finish();
174 }
175 FILLING_END();
176}
