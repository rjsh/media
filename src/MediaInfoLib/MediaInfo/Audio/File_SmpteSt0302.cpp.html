70void File_SmpteSt0302::Streams_Accept()
71{
72 // SMPTE ST 337
73 {
74 File_SmpteSt0337* SmpteSt0337=new File_SmpteSt0337();
75 SmpteSt0337->Container_Bits=(4+bits_per_sample)*4;
76 SmpteSt0337->Endianness='L';
77 SmpteSt0337->Aligned=true;
78 #if MEDIAINFO_DEMUX
79 if (Config->Demux_Unpacketize_Get())
80 {
81 Demux_Level=4; //Intermediate
82 SmpteSt0337->Demux_Level=2; //Container
83 SmpteSt0337->Demux_UnpacketizeContainer=true;
84 }
85 #endif //MEDIAINFO_DEMUX
86 Parsers.push_back(SmpteSt0337);
87 }
88
89 // Raw PCM
90 {
91 File_Pcm* Pcm=new File_Pcm();
92 Pcm->Codec.From_Local("SMPTE ST 302");
93 Pcm->BitDepth=(4+bits_per_sample)*4;
94 Pcm->Channels=(1+number_channels)*2;
95 Pcm->SamplingRate=48000;
96 Pcm->Endianness='L';
97 #if MEDIAINFO_DEMUX
98 if (Config->Demux_Unpacketize_Get())
99 {
100 Demux_Level=4; //Intermediate
101 Pcm->Demux_Level=2; //Container
102 Pcm->Demux_UnpacketizeContainer=true;
103 }
104 #endif //MEDIAINFO_DEMUX
105 Parsers.push_back(Pcm);
106 }
107
108 // Init
109 for (size_t Pos=0; Pos<Parsers.size(); Pos++)
110 Open_Buffer_Init(Parsers[Pos]);
111
112 //Time stamps
113 Frequency_b=48000;
114}

116//---------------------------------------------------------------------------
117void File_SmpteSt0302::Streams_Fill()
118{
119 if (Parsers.size()==1 && Parsers[0]->Status[IsAccepted])
120 {
121 Fill(Parsers[0]);
122 Merge(*Parsers[0]);
123 }
124
125 for (size_t Pos=0; Pos<Count_Get(Stream_Audio); Pos++)
126 if (Retrieve(Stream_Audio, Pos, Audio_MuxingMode).empty()) //TODO: put "SMPTE ST 302" in this field, the current name is there only for legacy
127 Fill(Stream_Audio, Pos, Audio_MuxingMode, "SMPTE ST 302");
128 else
129 Fill(Stream_Audio, Pos, Audio_MuxingMode, __T("SMPTE ST 302 / ")+Retrieve(Stream_Audio, Pos, Audio_MuxingMode), true);
130
131 if (Count_Get(Stream_Audio)==1)
132 {
133 if (Retrieve(Stream_Audio, 0, Audio_BitRate).empty())
134 Fill(Stream_Audio, 0, Audio_BitRate, (4+bits_per_sample)*(1+number_channels)*8*48000);
135 if (Retrieve(Stream_Audio, 0, Audio_Format)==__T("PCM"))
136 {
137 Fill(Stream_Audio, 0, Audio_Codec, "SMPTE ST 302", Unlimited, true, true);
138 Fill(Stream_Audio, 0, Audio_Codec_String, "SMPTE ST 302", Unlimited, true, true);
139 Clear(Stream_Audio, 0, Audio_Codec_Family);
140 }
141 }
142
143 Fill(Stream_Audio, 0, Audio_BitRate_Encoded, (5+bits_per_sample)*(1+number_channels)*8*48000);
144 for (size_t Pos=1; Pos<Count_Get(Stream_Audio); Pos++)
145 Fill(Stream_Audio, Pos, Audio_BitRate_Encoded, 0);
146}

153static inline int8u Reverse8(int n)
154{
155 // Input: bit order is 76543210
156 //Output: bit order is 45670123
157 n=((n>>1)&0x55) | ((n<<1) & 0xaa);
158 n=((n>>2)&0x33) | ((n<<2) & 0xcc);
159 n=((n>>4)&0x0f) | ((n<<4) & 0xf0);
160 return (int8u)n;
161}

163//---------------------------------------------------------------------------
164void File_SmpteSt0302::Read_Buffer_Continue()
165{
166 //Parsing
167 Get_B2 (audio_packet_size, "audio_packet_size");
168 BS_Begin();
169 Get_S1 (2, number_channels, "number_channels"); Param_Info2((1+number_channels)*2, " channels");
170 Info_S1(8, channel_identification, "channel_identification");
171 Get_S1 (2, bits_per_sample, "bits_per_sample"); Param_Info2((4+bits_per_sample)*4, " bits");
172 Info_S1(4, alignment_bits, "alignment_bits");
173 BS_End();
174
175 //Enough data
176 if (Element_Size<4+(int64u)audio_packet_size)
177 {
178 Element_Offset=0;
179 Element_WaitForMoreData();
180 return;
181 }
182
183 //Cohenrancy test
184 if (Element_Size!=4+(int64u)audio_packet_size || bits_per_sample==3 || audio_packet_size%((1+number_channels)*(5+bits_per_sample)))
185 {
186 Trusted_IsNot("Wrong size");
187 Skip_XX(Element_Size-4, "Problem?");
188 return;
189 }
190
191 if (!Status[IsAccepted])
192 Accept("SMPTE ST 302");
193
194 //Decyphering
195 size_t PcmSize=0;
196 float64 Ratio=0;
197 switch (bits_per_sample)
198 {
199 case 0 : PcmSize=audio_packet_size*4/5; Ratio=4.0/5.0; break;
200 case 1 : PcmSize=audio_packet_size*5/6; Ratio=5.0/6.0; break;
201 case 2 : PcmSize=audio_packet_size*6/7; Ratio=6.0/7.0; break;
202 case 3 : Reject(); return;
203 default: ;
204 }
205
206 int8u* Info=new int8u[PcmSize];
207 size_t Info_Offset=0;
208
209 while (Element_Offset<Element_Size)
210 {
211 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
212
213 switch (bits_per_sample)
214 {
215 case 0 : //16 bits
216 // Source: L1L0 L3L2 R0XX R2R1 XXR3
217 // Dest : 16LE / L1L0 L3L2 R1R0 R3R2
218 Info[Info_Offset+0]= Reverse8(Buffer[Buffer_Pos+0]);
219 Info[Info_Offset+1]= Reverse8(Buffer[Buffer_Pos+1]);
220 Info[Info_Offset+2]=(Reverse8(Buffer[Buffer_Pos+3])<<4 ) | (Reverse8(Buffer[Buffer_Pos+2])>>4 );
221 Info[Info_Offset+3]=(Reverse8(Buffer[Buffer_Pos+4])<<4 ) | (Reverse8(Buffer[Buffer_Pos+3])>>4 );
222
223 Info_Offset+=4;
224 Element_Offset+=5;
225 break;
226
227 case 1 : //20 bits
228 // Source: L1L0 L3L2 XXL4 R1R0 R3R2 XXR4
229 // Dest : 20LE / L1L0 L3L2 R0L4 R2R1 R4R3
230 Info[Info_Offset+0]= Reverse8(Buffer[Buffer_Pos+0]) ;
231 Info[Info_Offset+1]= Reverse8(Buffer[Buffer_Pos+1]) ;
232 Info[Info_Offset+2]=(Reverse8(Buffer[Buffer_Pos+3])<<4 ) | (Reverse8(Buffer[Buffer_Pos+2])&0x0F);
233 Info[Info_Offset+3]=(Reverse8(Buffer[Buffer_Pos+4])<<4 ) | (Reverse8(Buffer[Buffer_Pos+3])>>4 );
234 Info[Info_Offset+4]=(Reverse8(Buffer[Buffer_Pos+5])<<4 ) | (Reverse8(Buffer[Buffer_Pos+4])>>4 );
235
236 Info_Offset+=5;
237 Element_Offset+=6;
238 break;
239
240 case 2 : //24 bits
241 // Source: L1L0 L3L2 L5L4 R0XX R2R1 R4R3 XXR5
242 // Dest : 16LE / L1L0 L3L2 L5L4 R1R0 R3R2 R5R4
243 Info[Info_Offset+0] = Reverse8(Buffer[Buffer_Pos+0]) ;
244 Info[Info_Offset+1] = Reverse8(Buffer[Buffer_Pos+1]) ;
245 Info[Info_Offset+2] = Reverse8(Buffer[Buffer_Pos+2]) ;
246 Info[Info_Offset+3] =(Reverse8(Buffer[Buffer_Pos+4])<<4 ) | (Reverse8(Buffer[Buffer_Pos+3])>>4 );
247 Info[Info_Offset+4] =(Reverse8(Buffer[Buffer_Pos+5])<<4 ) | (Reverse8(Buffer[Buffer_Pos+4])>>4 );
248 Info[Info_Offset+5] =(Reverse8(Buffer[Buffer_Pos+6])<<4 ) | (Reverse8(Buffer[Buffer_Pos+5])>>4 );
249
250 Info_Offset+=6;
251 Element_Offset+=7;
252 break;
253 default : ;
254 }
255 }
256 Element_Offset=4;
257
258 FrameInfo.PTS=FrameInfo.DTS;
259 FrameInfo.DUR=((int64u)audio_packet_size)*1000000000/((1+number_channels)*(5+bits_per_sample)*48000);
260
261 #if MEDIAINFO_DEMUX
262 Demux_random_access=true;
263 Demux(Info, Info_Offset, ContentType_MainStream);
264 #endif //MEDIAINFO_DEMUX
265
266 //Parsers
267 for (size_t Pos=0; Pos<Parsers.size(); Pos++)
268 {
269 Parsers[Pos]->FrameInfo=FrameInfo;
270 Open_Buffer_Continue(Parsers[Pos], Info, Info_Offset, true, Ratio);
271
272 if (Parsers.size()>1 && Parsers[Pos]->Status[IsAccepted])
273 {
274 for (size_t Pos2=0; Pos2<Pos; Pos2++)
275 delete Parsers[Pos2]; //Parsers[Pos2]=NULL;
276 for (size_t Pos2=Pos+1; Pos2<Parsers.size(); Pos2++)
277 delete Parsers[Pos2]; //Parsers[Pos2]=NULL;
278 Parsers.resize(Pos+1);
279 Parsers.erase(Parsers.begin(), Parsers.begin()+Parsers.size()-1);
280 }
281 }
282 Element_Offset=Element_Size;
283
284 delete[] Info;
285
286 //Filling
287 TS_Add(audio_packet_size/((1+number_channels)*(5+bits_per_sample)));
288 if (Parsers.size()>1 && Frame_Count>=2)
289 {
290 for (size_t Pos=0; Pos<Parsers.size()-1; Pos++)
291 delete Parsers[Pos]; //Parsers[Pos]=NULL;
292 Parsers.erase(Parsers.begin(), Parsers.begin()+Parsers.size()-1);
293 }
294
295 if (!Status[IsFilled] && Parsers.size()==1 && Parsers[0]->Status[IsFinished])
296 {
297 //Filling
298 Finish("SMPTE ST 302");
299 }
300}
