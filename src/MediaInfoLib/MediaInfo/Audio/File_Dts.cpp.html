162std::string DTS_HD_SpeakerActivityMask (int16u SpeakerActivityMask, bool AddCs=false, bool AddLrsRrs=false)
163{
164 std::string Text;
165 if ((SpeakerActivityMask&0x0003)==0x0003)
166 Text+="Front: L C R";
167 else
168 {
169 if (SpeakerActivityMask&0x0001)
170 Text+="Front: C";
171 if (SpeakerActivityMask&0x0002)
172 Text+="Front: L R";
173 }
174
175 if (SpeakerActivityMask&0x0004)
176 Text+=", Side: L R";
177
178 if (SpeakerActivityMask&0x0010 || AddCs)
179 Text+=", Back: C";
180
181 if ((SpeakerActivityMask&0x00A0)==0x00A0)
182 Text+=", High: L C R";
183 else
184 {
185 if (SpeakerActivityMask&0x0020 || AddLrsRrs)
186 Text+=", High: L R";
187 if (SpeakerActivityMask&0x0080)
188 Text+=", High: C";
189 }
190
191 if (SpeakerActivityMask&0x0800)
192 Text+=", Side: L R";
193
194 if (SpeakerActivityMask&0x0040)
195 Text+=", Back: L R";
196
197 if (SpeakerActivityMask&0x0100)
198 Text+=", TopCtrSrrd";
199 if (SpeakerActivityMask&0x0200)
200 Text+=", Ctr: L R";
201 if (SpeakerActivityMask&0x0400)
202 Text+=", Wide: L R";
203 if (SpeakerActivityMask&0x2000)
204 Text+=", HiSide: L R";
205
206 if ((SpeakerActivityMask&0xC000)==0x0C000)
207 Text+=", HiRear: L C R";
208 else
209 {
210 if (SpeakerActivityMask&0x4000)
211 Text+=", HiRear: C";
212 if (SpeakerActivityMask&0x8000)
213 Text+=", HiRear: L R";
214 }
215
216 if (SpeakerActivityMask&0x0008)
217 Text+=", LFE";
218 if (SpeakerActivityMask&0x1000)
219 Text+=", LFE2";
220
221 return Text;
222}

224//---------------------------------------------------------------------------
225std::string DTS_HD_SpeakerActivityMask2 (int16u SpeakerActivityMask, bool AddCs=false, bool AddLrsRrs=false)
226{
227 std::string Text;
228 if ((SpeakerActivityMask&0x0003)==0x0003)
229 Text+="3";
230 else
231 {
232 if (SpeakerActivityMask&0x0001)
233 Text+="1";
234 else if (SpeakerActivityMask&0x0002)
235 Text+="2";
236 else
237 Text+="0";
238 }
239
240 if (SpeakerActivityMask&0x0004)
241 Text+="/2";
242 else if ((SpeakerActivityMask&0x0840)==0x0000)
243 Text+="/0";
244
245 if (SpeakerActivityMask&0x0010 || AddCs)
246 Text+="/1";
247 else if ((SpeakerActivityMask&0x0840)==0x0000)
248 Text+="/0";
249
250 if ((SpeakerActivityMask&0x00A0)==0x00A0)
251 Text+=".3";
252 else
253 {
254 if (SpeakerActivityMask&0x0020 || AddLrsRrs)
255 Text+=".2";
256 if (SpeakerActivityMask&0x0080)
257 Text+=".2";
258 }
259
260 if (SpeakerActivityMask&0x0800)
261 Text+="/2";
262 if (SpeakerActivityMask&0x0040)
263 Text+="/2";
264
265 if (SpeakerActivityMask&0x0100)
266 Text+=".1";
267 if (SpeakerActivityMask&0x0200)
268 Text+=".2";

270 Text+=".2";
271 if (SpeakerActivityMask&0x2000)

282 }

401 if (Presence[presence_Core_XXCh] && i<Core_Core_Channels+Core_XXCh_nuNumChSetsInXXCh)
402 {

412 if (Presence[presence_Core_XCh] && i<Core_Core_Channels+Core_XCh_AMODE)

414 //AddCs=true;
415
416 //i=Core_Core_Channels+Core_XCh_AMODE;

466{

485 else if (Core_Core_AMODE == 9 && (Presence[presence_Core_XCh] && Core_XCh_AMODE == 2))
486 {
487 Data[ChannelPositions].push_back(Ztring(__T("Front: L C R, Side: L R, Back: L R"))+(Core_Core_LFF?__T(", LFE"):__T("")));

493 Ztring Debug=__T("Debug, Core_Core_AMODE=")+Ztring::ToZtring(Core_Core_AMODE)+(Presence[presence_Core_XCh]?(__T(", Core_XCh_AMODE=")+Ztring::ToZtring(Core_XCh_AMODE)):__T(""))+__T(", Core_Core_LFF=")+Ztring::ToZtring(Core_Core_LFF);

500//---------------------------------------------------------------------------
501void File_Dts::Streams_Fill_Core()
502{
503 Ztring Core_BitRate;
504 if (bit_rate<25)
505 Core_BitRate=Ztring::ToZtring(BitRate_Get(), 0);
506 else if (bit_rate==29)
507 Core_BitRate=__T("Open");
508 else
509 Core_BitRate=__T("Unknown");
510
511 if (Core_Core_AMODE<16)
512 {
513 Data[Channels].push_back(Ztring::ToZtring(DTS_Channels[Core_Core_AMODE]+(Core_Core_LFF?1:0)));
514 Data[ChannelPositions].push_back(Ztring().From_Local(DTS_ChannelPositions[Core_Core_AMODE])+(Core_Core_LFF?__T(", LFE"):__T("")));
515 Data[ChannelPositions2].push_back(Ztring().From_Local(DTS_ChannelPositions2[Core_Core_AMODE])+(Core_Core_LFF?__T(".1"):__T(".0")));
516 Data[ChannelLayout].push_back(Ztring().From_Local(DTS_ChannelLayout[Core_Core_AMODE])+(Core_Core_LFF?__T(" LFE"):__T("")));
517 }
518 else
519 {
520 Data[Channels].push_back(__T("User Defined"));
521 Data[ChannelPositions].push_back(Ztring(__T("User Defined"))+(Core_Core_LFF?__T(", LFE"):__T("")));
522 Data[ChannelPositions2].push_back(Ztring(__T("User Defined"))+(Core_Core_LFF?__T(".1"):__T(".0")));
523 Data[ChannelLayout].push_back(Ztring(__T("User Defined"))+(Core_Core_LFF?__T(" LFE"):__T("")));
524 }
525 Data[BitDepth].push_back(Ztring::ToZtring(DTS_Resolution[bits_per_sample]));
526 Data[SamplingRate].push_back(Ztring::ToZtring(DTS_SamplingRate[sample_frequency]));
527 Data[BitRate].push_back(Core_BitRate);
528 Data[BitRate_Mode].push_back(__T("CBR"));
529 Data[Compression_Mode].push_back(__T("Lossy"));
530}

532//---------------------------------------------------------------------------
533void File_Dts::Streams_Fill()
534{
535 Fill(Stream_General, 0, General_Format, "DTS");
536
537 Stream_Prepare(Stream_Audio);
538 Fill(Stream_Audio, 0, Audio_Format, "DTS");
539
540 // DTS:X
541 if (Presence[presence_Extended_XLL] && Extension_XLL_X_Yes && !Extension_XLL_X_No)
542 Presence.set(presence_Extended_XLL_X);
543 if (Presence[presence_Extended_XLL_X])
544 {
545 Data[Profiles].push_back(__T("X"));
546
547 Data[Channels].push_back(__T("Object Based"));
548 Data[ChannelPositions].push_back(__T("Object Based"));
549 Data[ChannelPositions2].push_back(__T("Object Based"));
550 Data[ChannelLayout].push_back(__T("Object Based"));
551 Data[BitDepth].push_back(Ztring());
552 Data[SamplingRate].push_back(Ztring());
553 Data[BitRate].push_back(__T("Unknown"));
554 Data[BitRate_Mode].push_back(__T("VBR"));
555 Data[Compression_Mode].push_back(Ztring());
556 }
557
558 // DTS-HD MA
559 if (Presence[presence_Extended_XLL])
560 {
561 Data[Profiles].push_back(__T("MA"));
562
563 Streams_Fill_Extension();
564 Data[BitRate].pop_back();
565 Data[BitRate_Mode].pop_back();
566 Data[Compression_Mode].pop_back();
567
568 Data[BitRate].push_back(__T("Unknown"));
569 Data[BitRate_Mode].push_back(__T("VBR"));
570 Data[Compression_Mode].push_back(__T("Lossless"));
571 }
572
573 // DTS Express
574 if (Presence[presence_Extended_LBR])
575 {
576 Data[Profiles].push_back(__T("Express"));
577
578 Streams_Fill_Extension();
579 }
580
581 // DTS-HD HRA
582 if (Presence[presence_Extended_XBR]
583 || Presence[presence_Extended_X96]
584 || Presence[presence_Extended_XXCh]
585 || Presence[presence_Core_XXCh])
586 {
587 Data[Profiles].push_back(__T("HRA"));
588
589 if (Presence[presence_Extended_XBR]
590 || Presence[presence_Extended_X96]
591 || Presence[presence_Extended_XXCh])
592 Streams_Fill_Extension();
593 else
594 {
595 Streams_Fill_Core_ES();
596
597 if (Presence[presence_Extended_X96])
598 {
599 Data[SamplingRate].pop_back();
600
601 Data[SamplingRate].push_back(Ztring::ToZtring(96000));
602 }
603 }
604 }
605
606 // DTS-HD 96/24
607 else
608 if (Presence[presence_Core_X96])
609 {
610 Data[Profiles].push_back(__T("96/24"));
611
612 if (Presence[presence_Extended_X96])
613 Streams_Fill_Extension();
614 else
615 {
616 Streams_Fill_Core_ES();
617 Data[SamplingRate].pop_back();
618
619 Data[SamplingRate].push_back(Ztring::ToZtring(96000));
620 }
621 }
622
623 // DTS-ES
624 if (ES || Presence[presence_Core_XCh])
625 {
626 if (!ES)
627 Data[Profiles].push_back(__T("ES Discrete without ES Matrix"));
628 else if (Presence[presence_Core_XCh])
629 Data[Profiles].push_back(__T("ES Discrete"));
630 else
631 Data[Profiles].push_back(__T("ES Matrix"));
632 Streams_Fill_Core_ES();
633 }
634
635 // Core
636 if (Presence[presence_Core_Core])
637 {
638 Data[Profiles].push_back(__T("Core"));
639 Streams_Fill_Core();
640 }
641
642 // Cleanup up
643 for (size_t Pos=0; Pos<10; ++Pos)
644 {
645 Data[Pos].Separator_Set(0, __T(" / "));
646 if (Data[Pos].size()>1)
647 {
648 bool IsNotSame=false;
649 for (size_t i=1; i<Data[Pos].size(); ++i)
650 if (Data[Pos][i]!=Data[Pos][0])
651 IsNotSame=true;
652 if (!IsNotSame)
653 Data[Pos].resize(1);
654 }
655 }
656 if (Data[Profiles].size()==1 && Data[Profiles][0]==__T("Core"))
657 Data[Profiles].clear(); //Core is the default one
658
659 // Filling
660 Fill(Stream_Audio, 0, Audio_Format_Profile, Data[Profiles].Read());
661 Fill(Stream_Audio, 0, Audio_Codec, (Data[Profiles].Find(__T("MA"))!=string::npos || Data[Profiles].Find(__T("HRA"))!=string::npos)?"DTS-HD":"DTS");
662 Fill(Stream_General, 0, General_Format_Profile, Retrieve(Stream_Audio, 0, Audio_Format_Profile));
663 Fill(Stream_Audio, 0, Audio_Channel_s_, Data[Channels].Read());
664 Fill(Stream_Audio, 0, Audio_ChannelPositions, Data[ChannelPositions].Read());
665 Fill(Stream_Audio, 0, Audio_ChannelPositions_String2, Data[ChannelPositions2].Read());
666 Fill(Stream_Audio, 0, Audio_ChannelLayout, Data[ChannelLayout].Read());
667 Fill(Stream_Audio, 0, Audio_BitDepth, Data[BitDepth].Read());
668 Fill(Stream_Audio, 0, Audio_SamplingRate, Data[SamplingRate].Read());
669 Fill(Stream_Audio, 0, Audio_BitRate, Data[BitRate].Read());
670 Fill(Stream_Audio, 0, Audio_BitRate_Mode, Data[BitRate_Mode].Read());
671 Fill(Stream_General, 0, General_OverallBitRate_Mode, Retrieve(Stream_Audio, 0, Audio_BitRate_Mode));
672 Fill(Stream_Audio, 0, Audio_Compression_Mode, Data[Compression_Mode].Read(), true);
673
674 // Cleanup up
675 for (size_t Pos=0; Pos<10; ++Pos)
676 Data[Pos].clear();
677
678 //Samples per frame (to be done after SamplingRate filling because it is used during Fill())
679 if (Presence[presence_Core_Core])
680 Fill(Stream_Audio, 0, Audio_SamplesPerFrame, Number_Of_PCM_Sample_Blocks*32); // Samples per Core frames. TODO: samples per HD frames (interesting?)
681}

721//***************************************************************************
722// Buffer - Synchro
723//***************************************************************************
724
725//---------------------------------------------------------------------------
726bool File_Dts::Synchronize()
727{
728 //Synchronizing
729 while (Buffer_Offset+6<=Buffer_Size)
730 {
731 if (!FrameSynchPoint_Test())
732 return false; //Need more data
733 if (Synched)
734 break;
735 Buffer_Offset++;
736 }
737
738 //Parsing last bytes if needed
739 if (Buffer_Offset+6>Buffer_Size)
740 {
741 if (Buffer_Offset+5==Buffer_Size)
742 {
743 int64u Value=CC5(Buffer+Buffer_Offset);
744 if ((Value&0xFFFFFFFFFCLL)!=0x7FFE8001FCLL //16 bits and big endian Core
745 && (Value&0xFFFFFFFF00LL)!=0xFE7F018000LL //16 bits and little endian Core
746 && (Value&0xFFFFFFFFF7LL)!=0x1FFFE80007LL //14 bits and big endian Core
747 && (Value&0xFFFFFFFFF0LL)!=0xFF1F00E8F0LL //14 bits and little endian Core
748 && (Value&0xFFFFFFFF00LL)!=0x6458202500LL) //16 bits and big endian HD
749 Buffer_Offset++;
750 }
751 if (Buffer_Offset+4==Buffer_Size)
752 {
753 int32u Value=CC4(Buffer+Buffer_Offset);
754 if (Value!=0x7FFE8001 //16 bits and big endian Core
755 && Value!=0xFE7F0180 //16 bits and little endian Core
756 && Value!=0x1FFFE800 //14 bits and big endian Core
757 && Value!=0xFF1F00E8 //14 bits and little endian Core
758 && Value!=0x64582025) //16 bits and big endian HD
759 Buffer_Offset++;
760 }
761 if (Buffer_Offset+3==Buffer_Size)
762 {
763 int32u Value=CC3(Buffer+Buffer_Offset);
764 if (Value!=0x7FFE80 //16 bits and big endian Core
765 && Value!=0xFE7F01 //16 bits and little endian Core
766 && Value!=0x1FFFE8 //14 bits and big endian Core
767 && Value!=0xFF1F00 //14 bits and little endian Core
768 && Value!=0x645820) //16 bits and big endian HD
769 Buffer_Offset++;
770 }
771 if (Buffer_Offset+2==Buffer_Size)
772 {
773 int16u Value=CC2(Buffer+Buffer_Offset);
774 if (Value!=0x7FFE //16 bits and big endian Core
775 && Value!=0xFE7F //16 bits and little endian Core
776 && Value!=0x1FFF //14 bits and big endian Core
777 && Value!=0xFF1F //14 bits and little endian Core
778 && Value!=0x6458) //16 bits and big endian HD
779 Buffer_Offset++;
780 }
781 if (Buffer_Offset+1==Buffer_Size)
782 {
783 int8u Value=CC1(Buffer+Buffer_Offset);
784 if (Value!=0x7F //16 bits and big endian Core
785 && Value!=0xFE //16 bits and little endian Core
786 && Value!=0x1F //14 bits and big endian Core
787 && Value!=0xFF //14 bits and little endian Core
788 && Value!=0x64) //16 bits and big endian HD
789 Buffer_Offset++;
790 }
791 return false;
792 }
793
794 //Synched
795 return true;
796}

1280//---------------------------------------------------------------------------
1281void File_Dts::HD()
1282{
1283 //Parsing
1284 while (Element_Offset<Element_Size)
1285
1286 {
1287 //Looking for size
1288 int64u Next=Element_Offset+4;
1289 while (Next+4<=Element_Size)
1290 {
1291 int32u CC=CC4(Buffer+Buffer_Offset+(size_t)Next);
1292 if (CC==0x0A801921
1293 || CC==0x1D95F262
1294 || CC==0x41A29547
1295 || CC==0x47004A03
1296 || CC==0x5A5A5A5A
1297 || CC==0x655E315E)
1298 break;
1299 Next++;
1300 }
1301 if (Next+4>Element_Size)
1302 Next=Element_Size;
1303
1304 Element_Begin0();
1305 int32u Magic;
1306 Get_B4 (Magic, "Magic");
1307 switch (Magic)
1308 {
1309 case 0x0A801921 : HD_XSA(Next-Element_Offset); break;
1310 case 0x1D95F262 : HD_X96k(Next-Element_Offset); break;
1311 case 0x41A29547 : HD_XLL(Next-Element_Offset); break;
1312 case 0x47004A03 : HD_XXCh(Next-Element_Offset); break;
1313 case 0x5A5A5A5A : HD_XCh(Next-Element_Offset); break;
1314 case 0x655E315E : HD_XBR(Next-Element_Offset); break;
1315 default :
1316 //Magic value is unknown
1317 if (Profile.empty())
1318 Profile="HD";
1319 Skip_XX(Next-Element_Offset, "Data");
1320 }
1321 Element_End0();
1322 }
1323
1324 //Filling
1325 FILLING_BEGIN();
1326 if (Count_Get(Stream_Audio)==0 && Frame_Count>=Frame_Count_Valid)
1327 {
1328 Accept("DTS");
1329 Fill("DTS");
1330
1331 //No more need data

1508//---------------------------------------------------------------------------
1509bool File_Dts::FrameSynchPoint_Test()
1510{
1511 if (Save_Buffer)
1512 return true; //Test already made by Synchronize()
1513
1514 int64u Value=CC6(Buffer+Buffer_Offset);
1515 if ((Value&0xFFFFFFFFFC00LL)==0x7FFE8001FC00LL //16 bits and big endian Core
1516 || (Value&0xFFFFFFFF00FCLL)==0xFE7F018000FCLL //16 bits and little endian Core
1517 || (Value&0xFFFFFFFFF7F0LL)==0x1FFFE80007F0LL //14 bits and big endian Core
1518 || (Value&0xFFFFFFFFF0F7LL)==0xFF1F00E8F007LL //14 bits and little endian Core
1519 || (Value&0xFFFFFFFF0000LL)==0x645820250000LL) //16 bits and big endian HD
1520 {
1521 //Configuration - 14 bits or Little Endian
1522 switch (Buffer[Buffer_Offset])
1523 {
1524 default : Word=true; BigEndian=true; break; //16 bits and big endian bitstream
1525 case 0xFE : Word=true; BigEndian=false; break; //16 bits and little endian bitstream
1526 case 0x1F : Word=false; BigEndian=true; break; //14 bits and big endian bitstream
1527 case 0xFF : Word=false; BigEndian=false; break; //14 bits and little endian bitstream
1528 }
1529
1530 int32u Size;
1531 if (Buffer[Buffer_Offset]==0x64) //HD
1532 {
1533 if (Buffer_Offset+10>Buffer_Size)
1534 return false; //Need more data
1535
1536 bool isBlownUpHeader=(Buffer[Buffer_Offset+5]&0x20)?true:false;
1537 if (isBlownUpHeader)
1538 Size=((Buffer[Buffer_Offset+6]&0x01)<<19)
1539 | ( Buffer[Buffer_Offset+7] <<11)
1540 | ( Buffer[Buffer_Offset+8] << 3)
1541 | ( Buffer[Buffer_Offset+9] >> 5);
1542 else
1543 Size=((Buffer[Buffer_Offset+6]&0x1F)<<11)
1544 | ( Buffer[Buffer_Offset+7] << 3)
1545 | ( Buffer[Buffer_Offset+8] >> 5);
1546 Size++;
1547 Original_Size=Size;
1548 }
1549 else
1550 {
1551 if (Word)
1552 {
1553 if (BigEndian)
1554 Size=((Buffer[Buffer_Offset+5]&0x03)<<12)
1555 | ( Buffer[Buffer_Offset+6] << 4)
1556 | ( Buffer[Buffer_Offset+7] >> 4);
1557 else
1558 Size=((Buffer[Buffer_Offset+4]&0x03)<<12)
1559 | ( Buffer[Buffer_Offset+7] << 4)
1560 | ( Buffer[Buffer_Offset+6] >> 4);
1561 Size++;
1562 Original_Size=Size;
1563 }
1564 else
1565 {
1566 if (BigEndian)
1567 Size=((Buffer[Buffer_Offset+6]&0x03)<<12)
1568 | ( Buffer[Buffer_Offset+7] << 4)
1569 | ((Buffer[Buffer_Offset+8]&0x3C)>> 2);
1570 else
1571 Size=((Buffer[Buffer_Offset+7]&0x03)<<12)
1572 | ( Buffer[Buffer_Offset+6] << 4)
1573 | ((Buffer[Buffer_Offset+9]&0x3C)>> 2);
1574 Size++;
1575 Original_Size=Size*16/14;
1576 }
1577 }
1578 if (Buffer_Offset+Size>Buffer_Size)
1579 return false; //Need more data
1580
1581 if (!Word || !BigEndian)
1582 {
1583 Save_Buffer=Buffer;
1584 Save_Buffer_Offset=Buffer_Offset;
1585 Save_Buffer_Size=Buffer_Size;
1586
1587 //Preparing new buffer
1588 if (Word)
1589 {
1590 int8u* Buffer_Little=new int8u[Size];
1591 for (size_t Pos=0; Pos+1<Size; Pos+=2)
1592 {
1593 Buffer_Little[Pos+1]=Save_Buffer[Buffer_Offset+Pos ];
1594 Buffer_Little[Pos ]=Save_Buffer[Buffer_Offset+Pos+1];
1595 }
1596 Save_Buffer=Buffer_Little;
1597 Save_Buffer_Offset=0;
1598 Save_Buffer_Size=Size;
1599 }
1600 else
1601 {
1602 int8u* Dest=new int8u[Size];
1603 for (size_t Pos=0; Pos+8<=Original_Size; Pos+=8)
1604 {
1605 int64u Value;
1606 if (BigEndian)
1607 Value =(((int64u)( BigEndian2int16u(Buffer+Buffer_Offset+Pos )&0x3FFF))<<42)
1608 | (((int64u)( BigEndian2int16u(Buffer+Buffer_Offset+Pos+2)&0x3FFF))<<28)
1609 | (((int64u)( BigEndian2int16u(Buffer+Buffer_Offset+Pos+4)&0x3FFF))<<14)
1610 | (((int64u)( BigEndian2int16u(Buffer+Buffer_Offset+Pos+6)&0x3FFF)) );
1611 else
1612 Value =(((int64u)(LittleEndian2int16u(Buffer+Buffer_Offset+Pos )&0x3FFF))<<42)
1613 | (((int64u)(LittleEndian2int16u(Buffer+Buffer_Offset+Pos+2)&0x3FFF))<<28)
1614 | (((int64u)(LittleEndian2int16u(Buffer+Buffer_Offset+Pos+4)&0x3FFF))<<14)
1615 | (((int64u)(LittleEndian2int16u(Buffer+Buffer_Offset+Pos+6)&0x3FFF)) );
1616 int56u2BigEndian(Dest+Pos*14/16, Value);
1617 }
1618 Save_Buffer=Dest;
1619 Save_Buffer_Offset=0;
1620 Save_Buffer_Size=Size;
1621 }
1622 }
1623
1624 Synched=true;
1625 }
1626 else
1627 Synched=false;
1628
1629 return true;
1630}
