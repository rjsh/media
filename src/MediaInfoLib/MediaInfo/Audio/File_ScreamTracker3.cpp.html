35bool File_ScreamTracker3::FileHeader_Begin()
36{
37 //Element_Size
38 if (Buffer_Size<44)
39 return false; //Must wait for more data
40
41 if (CC1(Buffer+28)!=0x1A || CC4(Buffer+44)!=0x5343524D) //"SCRM"
42 {
43 Reject("Scream Tracker 3");
44 return false;
45 }
46
47 //All should be OK...
48 return true;
49}

51//***************************************************************************
52// Buffer - Global
53//***************************************************************************
54
55//---------------------------------------------------------------------------
56void File_ScreamTracker3::Read_Buffer_Continue()
57{
58 //Parsing
59 Ztring SongName;
60 int16u OrdNum, InsNum, PatNum, Flags, Special;
61 int8u SoftwareVersionMajor, SoftwareVersionMinor, IS, TS;
62 Get_Local(28, SongName, "Song name");
63 Skip_L1( "0x1A");
64 Skip_L1( "Type");
65 Skip_L1( "Unknown");
66 Skip_L1( "Unknown");
67 Get_L2 (OrdNum, "Orders count");
68 Get_L2 (InsNum, "Instruments count");
69 Get_L2 (PatNum, "Paterns count");
70 Get_L2 (Flags, "Flags");
71 Skip_Flags(Flags, 0, "st2vibrato");
72 Skip_Flags(Flags, 1, "st2tempo");
73 Skip_Flags(Flags, 2, "amigaslides");
74 Skip_Flags(Flags, 3, "0vol optimizations");
75 Skip_Flags(Flags, 4, "amiga limits");
76 Skip_Flags(Flags, 5, "enable filter/sfx with sb");
77 Skip_Flags(Flags, 6, "st3.00 volumeslides");
78 Skip_Flags(Flags, 7, "pecial custom data in file");
79 Get_L1 (SoftwareVersionMajor, "Cwt/v (Major)");
80 Get_L1 (SoftwareVersionMinor, "Cwt/v (Minor)");
81 Skip_L2( "File format information");
82 Skip_B4( "Signature");
83 Skip_L1( "global volume");
84 Get_L1 (IS, "Initial Speed");
85 Get_L1 (TS, "Initial Temp");
86 Skip_L1( "master volume");
87 Skip_L1( "ultra click removal");
88 Skip_L1( "Default channel pan positions are present");
89 Skip_L1( "Unknown");
90 Skip_L1( "Unknown");
91 Skip_L1( "Unknown");
92 Skip_L1( "Unknown");
93 Skip_L1( "Unknown");
94 Skip_L1( "Unknown");
95 Skip_L1( "Unknown");
96 Skip_L1( "Unknown");
97 Get_L2 (Special, "Special");
98 Skip_XX(32, "Channel settings");
99 Skip_XX(OrdNum, "Orders");
100 Skip_XX(InsNum*2, "Instruments");
101 Skip_XX(PatNum*2, "Patterns");
102
103 FILLING_BEGIN();
104 Accept("Scream Tracker 3");
105
106 Fill(Stream_General, 0, General_Format, "Scream Tracker 3");
107 Fill(Stream_General, 0, General_Track, SongName);
108 if ((SoftwareVersionMajor&0xF0)==0x10)
109 Fill(Stream_General, 0, General_Encoded_Application, Ztring(__T("Scream Tracker ")+Ztring::ToZtring(SoftwareVersionMajor)+__T(".")+Ztring::ToZtring(SoftwareVersionMinor/16)+Ztring::ToZtring(SoftwareVersionMinor%16)));
110 Fill(Stream_General, 0, "BPM", TS);
111
112 Stream_Prepare(Stream_Audio);
113
114 Finish("Scream Tracker 3");
115 FILLING_END();
116
117}
