47const char* Smpte_St0331_ChannelsPositions(int8u number_channels)
48{
49 switch (number_channels)
50 {
51 case 2 : return "Front: L R"; //2 channels
52 case 4 : return "Front: L C R, LFE"; //4 channels
53 case 6 : return "Front: L C R, Side: L R, LFE"; //6 channels
54 case 8 : return "Front: L C R, Side: L R, Back: L R, LFE"; //8 channels
55 default : return "";
56 }
57}
58
59//---------------------------------------------------------------------------
60const char* Smpte_St0331_ChannelsPositions2(int8u number_channels)
61{
62 switch (number_channels)
63 {
64 case 2 : return "2/0/0.0"; //2 channels
65 case 4 : return "3/0/0.1"; //4 channels
66 case 6 : return "3/2/0.1"; //6 channels
67 case 8 : return "3/2/2.1"; //8 channels
68 default : return "";
69 }
70}

98//---------------------------------------------------------------------------
99void File_SmpteSt0331::Streams_Fill()
100{
101 int8u Channels_Count=0;
102 for (int8u Pos=0; Pos<8; Pos++)
103 if (Channels_valid&(1<<Pos))
104 Channels_Count++;
105
106 Stream_Prepare(Stream_Audio);
107 Fill(Stream_Audio, 0, Audio_Format, "PCM");
108 Fill(Stream_Audio, 0, Audio_Codec, "PCM");
109 Fill(Stream_Audio, 0, Audio_SamplingRate, 48000);
110 if (QuantizationBits)
111 Fill(Stream_Audio, 0, Audio_BitRate, Channels_Count*QuantizationBits*48000);
112 Fill(Stream_Audio, 0, Audio_BitRate_Encoded, 8*32*48000);
113 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "CBR");
114
115 Fill(Stream_Audio, 0, Audio_Format_Settings_Endianness, "Little");
116 Fill(Stream_Audio, 0, Audio_Channel_s_, Channels_Count);
117 Fill(Stream_Audio, 0, Audio_ChannelPositions, Smpte_St0331_ChannelsPositions(Channels_Count));
118 Fill(Stream_Audio, 0, Audio_ChannelPositions_String2, Smpte_St0331_ChannelsPositions2(Channels_Count));
119 if (QuantizationBits)
120 Fill(Stream_Audio, 0, Audio_BitDepth, QuantizationBits);
121}

123//***************************************************************************
124// Buffer - Global
125//***************************************************************************
126
127//---------------------------------------------------------------------------
128void File_SmpteSt0331::Read_Buffer_Continue()
129{
130 if (!Status[IsAccepted])
131 Accept("SMPTE ST 331");
132
133 //SMPTE 331M
134 BS_Begin();
135 Skip_SB( "FVUCP Valid Flag");
136 Skip_S1(4, "Reserved");
137 Skip_S1(3, "5-sequence count");
138 BS_End();
139 Skip_L2( "Audio Sample Count");
140 Get_B1 (Channels_valid, "Channels valid");
141
142 if (QuantizationBits && Element_Offset<Element_Size)
143 {
144 int8u* Info = new int8u[(size_t)((Element_Size - Element_Offset)*((QuantizationBits == 16) ? 2 : 3) / 4)];
145 size_t Info_Offset=0;
146
147 while (Element_Offset+8*4<=Element_Size)
148 {
149 for (int8u Pos=0; Pos<8; Pos++)
150 {
151 if (Channels_valid&(1<<Pos))
152 {
153 size_t Buffer_Pos=Buffer_Offset+(size_t)Element_Offset;
154
155 if (QuantizationBits==16)
156 {
157 Info[Info_Offset+0] = (Buffer[Buffer_Pos+1]>>4) | ((Buffer[Buffer_Pos+2]<<4)&0xF0 );
158 Info[Info_Offset+1] = (Buffer[Buffer_Pos+2]>>4) | ((Buffer[Buffer_Pos+3]<<4)&0xF0 );
159 }
160 else
161 {
162 Info[Info_Offset+0] = (Buffer[Buffer_Pos+0]>>4) | ((Buffer[Buffer_Pos+1]<<4)&0xF0 );
163 Info[Info_Offset+1] = (Buffer[Buffer_Pos+1]>>4) | ((Buffer[Buffer_Pos+2]<<4)&0xF0 );
164 Info[Info_Offset+2] = (Buffer[Buffer_Pos+2]>>4) | ((Buffer[Buffer_Pos+3]<<4)&0xF0 );
165 }
166
167 Info_Offset+=QuantizationBits==16?2:3;
168 }
169 Element_Offset+=4;
170 }
171 }
172 Element_Offset=4;
173
174 #if MEDIAINFO_DEMUX
175 OriginalBuffer_Size=(size_t)Element_Size;
176 OriginalBuffer=(int8u*)(Buffer+Buffer_Offset);
177
178 FrameInfo.PTS=FrameInfo.DTS;
179 FrameInfo.DUR=(Element_Size-4)*1000000000/48000/32; // 48 kHz, 4 bytes per sample
180 Demux_random_access=true;
181 Element_Code=(int64u)-1;
182 Element_Offset=0;
183 Demux(Info, Info_Offset, ContentType_MainStream);
184 Element_Offset=4;
185
186 OriginalBuffer_Size=0;
187 OriginalBuffer=NULL;
188 #endif //MEDIAINFO_DEMUX
189
190 delete[] Info;
191 }
192
193 Skip_XX(Element_Size-4, "Data");
194
195 Frame_Count_InThisBlock++;
196 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
197 Frame_Count_NotParsedIncluded++;
198 #if MEDIAINFO_DEMUX
199 if (FrameInfo.DTS!=(int64u)-1 && FrameInfo.DUR!=(int64u)-1)
200 {
201 FrameInfo.DTS+=FrameInfo.DUR;
202 FrameInfo.PTS=FrameInfo.DTS;
203 }
204 #endif //MEDIAINFO_DEMUX
205
206 FILLING_BEGIN();
207 if (!Status[IsAccepted])
208 {
209 Accept("SMPTE ST 331");
210
211 int8u Channels=0;
212 for (int8u Pos=0; Pos<8; Pos++)
213 {
214 if (Channels_valid&(1<<Pos))
215 Channels++;
216 Element_Offset+=4;
217 }
218
219 Stream_Prepare(Stream_Audio);
220 Fill(Stream_Audio, 0, Audio_Format, "PCM");
221 Fill(Stream_Audio, 0, Audio_Channel_s_, Channels);
222 }
223 FILLING_END();
224}
