81//***************************************************************************
82// Buffer - File header
83//***************************************************************************
84
85//---------------------------------------------------------------------------
86bool File_Tak::FileHeader_Begin()
87{
88 if (!File__Tags_Helper::FileHeader_Begin())
89 return false;
90
91 //Synchro
92 if (Buffer_Offset+4>Buffer_Size)
93 return false;
94 if (CC4(Buffer+Buffer_Offset)!=0x7442614B) //"tBaK"
95 {
96 File__Tags_Helper::Reject("TAK");
97 return false;
98 }
99
100 return true;
101}

114// Buffer - Per element

117//---------------------------------------------------------------------------
118void File_Tak::Header_Parse()
119{
120 //Parsing
121 int32u block_length;
122 int8u block_type;
123 Get_L1 (block_type, "Block Type");
124 Get_L3 (block_length, "Block Length");
125
126 //Filling
127 Header_Fill_Code(block_type, Ztring().From_CC1(block_type));
128 Header_Fill_Size(Element_Offset+block_length);
129}

131//---------------------------------------------------------------------------
132void File_Tak::Data_Parse()
133{
134 #define CASE_INFO(_NAME) \
135 case Elements::_NAME : Element_Info1(#_NAME); _NAME(); break;
136
137 //Parsing
138 switch (Element_Code)
139 {
140 CASE_INFO(ENDOFMETADATA);
141 CASE_INFO(STREAMINFO);
142 CASE_INFO(SEEKTABLE);
143 CASE_INFO(WAVEMETADATA);
144 CASE_INFO(ENCODERINFO);
145 CASE_INFO(PADDING);
146 default : Skip_XX(Element_Size, "Data");
147 }
148
149 Element_Offset=Element_Size;
150}

157void File_Tak::ENDOFMETADATA()
158{
159 //Filling
160 Fill(Stream_General, 0, General_StreamSize, 0); //File_Offset+Buffer_Offset+Element_Size);

162 File__Tags_Helper::Finish("TAK");

165//---------------------------------------------------------------------------
166void File_Tak::STREAMINFO()
167{
168 //Parsing
169 int32u num_samples_hi, samplerate;
170 int8u num_samples_lo, framesizecode, samplesize;
171 bool channels;
172
173 Skip_L1 ( "unknown");
174 BS_Begin();
175 Get_S1 ( 2, num_samples_lo, "num_samples (lo)");
176 Get_S1 ( 3, framesizecode, "framesizecode");
177 Skip_S1( 2, "unknown");
178 BS_End();
179 Get_L4 (num_samples_hi, "num_samples (hi)"); Param_Info2((((int64u)num_samples_hi)<<2 | num_samples_lo), " samples");
180 Get_L3 (samplerate, "samplerate"); Param_Info2((samplerate/16)+6000, " Hz");
181 BS_Begin();
182 Skip_S1( 4, "unknown");
183 Get_SB ( channels, "channels"); Param_Info1(channels?"Stereo":"Mono");
184 Get_S1 ( 2, samplesize, "samplesize"); Param_Info1(Tak_samplesize[samplesize]);
185 Skip_SB( "unknown");
186 BS_End();
187 Skip_L3( "crc");
188
189 FILLING_BEGIN();
190 //Coherency
191 if (samplerate==0)
192 return;
193
194 //Computing
195 int64u Samples=((int64u)num_samples_hi)<<2 | num_samples_lo;
196 int32u SamplingRate=(samplerate/16)+6000;
197
198 //Filling
199 File__Tags_Helper::Accept("TAK");
200
201 File__Tags_Helper::Stream_Prepare(Stream_Audio);
202 Fill(Stream_Audio, 0, Audio_Format, "TAK");
203 Fill(Stream_Audio, 0, Audio_Codec, "TAK");
204 Fill(Stream_Audio, 0, Audio_SamplingRate, SamplingRate);
205 Fill(Stream_Audio, 0, Audio_Channel_s_, channels?2:1);
206 if (Tak_samplesize[samplesize])
207 Fill(Stream_Audio, 0, Audio_BitDepth, Tak_samplesize[samplesize]);
208 Fill(Stream_Audio, 0, Audio_Duration, Samples*1000/SamplingRate);
209 FILLING_END();
210}

212//---------------------------------------------------------------------------
213void File_Tak::SEEKTABLE()
214{
215 //Parsing
216 int16u num_seekpoints;
217 Get_L2 (num_seekpoints, "num_seekpoints");
218 Skip_L1 ( "unknown");
219 Skip_L1 ( "seek interval");
220 Element_Begin1("seekpoints");
221 for (int16u Pos=0; Pos<num_seekpoints; Pos++)
222 Skip_L5 ( "seekpoint");
223 Element_End0();
224 Skip_L3( "crc");
225}

227//---------------------------------------------------------------------------
228void File_Tak::WAVEMETADATA()
229{
230 //Parsing
231 int32u HeaderLength, FooterLength;
232 Get_L3 (HeaderLength, "HeaderLength");
233 Get_L3 (FooterLength, "FooterLength");
234 #if defined(MEDIAINFO_RIFF_YES)
235 //Creating the parser
236 File_Riff MI;
237 Open_Buffer_Init(&MI);
238
239 //Parsing
240 Open_Buffer_Continue(&MI, HeaderLength);
241 Element_Offset+=HeaderLength;
242
243 //Filling
244 //Finish(&MI);
245 //Merge(MI, StreamKind_Last, 0, StreamPos_Last);
246
247 //The RIFF header is for PCM
248 //Clear(Stream_Audio, StreamPos_Last, Audio_ID);
249 //Fill(Stream_Audio, StreamPos_Last, Audio_Format, "TAK", Unlimited, true, true);
250 //Fill(Stream_Audio, StreamPos_Last, Audio_Codec, "TAK", Unlimited, true, true);
251 //Clear(Stream_Audio, StreamPos_Last, Audio_CodecID);
252 //Clear(Stream_Audio, StreamPos_Last, Audio_CodecID_Hint);
253 //Clear(Stream_Audio, StreamPos_Last, Audio_CodecID_Url);
254 //Clear(Stream_Audio, StreamPos_Last, Audio_BitRate);
255 //Clear(Stream_Audio, StreamPos_Last, Audio_BitRate_Mode);
256 //Clear(Stream_Audio, StreamPos_Last, Audio_Codec_CC);
257 #else
258 Skip_XX(HeaderLength, "Wave header");
259 #endif
260 if (FooterLength)
261 Skip_XX(FooterLength, "Wave footer");
262 Skip_L3( "crc");
263}

265//---------------------------------------------------------------------------
266void File_Tak::ENCODERINFO()
267{
268 //Parsing
269 int8u Revision, Minor, Major, Preset_hi, Preset_lo;
270 Get_L1 (Revision, "Revision");
271 Get_L1 (Minor, "Minor");
272 Get_L1 (Major, "Major");
273 BS_Begin();
274 Get_S1 (4, Preset_hi, "Preset (hi)");
275 Get_S1 (4, Preset_lo, "Preset (lo)");
276 BS_End();
277
278 FILLING_BEGIN();
279 Ztring Version=Ztring::ToZtring(Major)+__T('.')+Ztring::ToZtring(Minor)+__T('.')+Ztring::ToZtring(Revision);
280 Ztring Preset=__T("-p")+Ztring::ToZtring(Preset_lo);
281 switch (Preset_hi)
282 {
283 case 0x00 : break;
284 case 0x01 : Preset+=__T('e'); break;
285 case 0x02 : Preset+=__T('m'); break;
286 default : Preset+=__T('-')+Ztring::ToZtring(Preset_hi, 16); //Unknown
287 }
288
289 Fill(Stream_Audio, 0, Audio_Encoded_Library, "TAK");
290 Fill(Stream_Audio, 0, Audio_Encoded_Library_String, __T("TAK ")+Version);
291 Fill(Stream_Audio, 0, Audio_Encoded_Library_Name, "TAK");
292 Fill(Stream_Audio, 0, Audio_Encoded_Library_Version, Version);
293 Fill(Stream_Audio, 0, Audio_Encoded_Library_Settings, Preset);
294 FILLING_END();
295}
