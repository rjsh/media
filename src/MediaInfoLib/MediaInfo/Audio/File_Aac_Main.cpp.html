234int8u Aac_AudioSpecificConfig_sampling_frequency_index(const int64s sampling_frequency)

236 if (sampling_frequency>=92017) return 0;
237 if (sampling_frequency>=75132) return 1;
238 if (sampling_frequency>=55426) return 2;
239 if (sampling_frequency>=46009) return 3;
240 if (sampling_frequency>=37566) return 4;

246 if (sampling_frequency>=9391) return 10;

250//---------------------------------------------------------------------------
251void File_Aac::AudioSpecificConfig (size_t End)
252{
253 //Parsing
254 bool sbrData=false, sbrPresentFlag=false, psData=false, psPresentFlag=false;
255 Element_Begin1("AudioSpecificConfig");
256 GetAudioObjectType(audioObjectType, "audioObjectType");
257 Get_S1 (4, sampling_frequency_index, "samplingFrequencyIndex"); Param_Info1(Aac_sampling_frequency[sampling_frequency_index]);
258 if (sampling_frequency_index==0xF)
259 {
260 int32u samplingFrequency;
261 Get_S3 (24, samplingFrequency, "samplingFrequency");
262 Frequency_b=samplingFrequency;
263 sampling_frequency_index=Aac_AudioSpecificConfig_sampling_frequency_index(Frequency_b);
264 }
265 else
266 Frequency_b=Aac_sampling_frequency[sampling_frequency_index];
267 Get_S1 (4, channelConfiguration, "channelConfiguration"); Param_Info1(Aac_ChannelConfiguration[channelConfiguration]);
268 if (audioObjectType==5 || audioObjectType==29)
269 {
270 extensionAudioObjectType=5;
271 sbrPresentFlag=true;
272 if (audioObjectType==29)
273 psPresentFlag=true;
274 Get_S1 (4, extension_sampling_frequency_index, "extensionSamplingFrequencyIndex"); Param_Info1(Aac_sampling_frequency[extension_sampling_frequency_index]);
275 if (extension_sampling_frequency_index==0xF)
276 {
277 Get_S3 (24, extension_sampling_frequency, "extensionSamplingFrequency");
278 sampling_frequency_index=Aac_AudioSpecificConfig_sampling_frequency_index(extension_sampling_frequency);
279 }
280 else
281 extension_sampling_frequency=Aac_sampling_frequency[extension_sampling_frequency_index];
282 GetAudioObjectType(audioObjectType, "audioObjectType");
283 if (audioObjectType==22) //BSAC
284 Skip_S1(4, "extensionChannelConfiguration");
285 }
286 else
287 extensionAudioObjectType=0x00;
288
289 switch (audioObjectType)
290 {
291 case 1:
292 case 2:
293 case 3:
294 case 4:
295 case 6:
296 case 7:
297 case 17:
298 case 19:
299 case 20:
300 case 21:
301 case 22:
302 case 23:
303 GASpecificConfig();
304 break;
305 case 8:
306 CelpSpecificConfig();
307 break;
308 case 9:
309 HvxcSpecificConfig();
310 break;
311 case 12:
312 TTSSpecificConfig();
313 break;
314 //~ case 13:
315 //~ case 14:
316 //~ case 15:
317 //~ case 16:
318 //~ StructuredAudioSpecificConfig();
319 //~ break;
320 case 24:
321 ErrorResilientCelpSpecificConfig();
322 break;
323 case 25:
324 ErrorResilientHvxcSpecificConfig();
325 break;
326 case 26:
327 case 27:
328 ParametricSpecificConfig();
329 break;
330 case 28:
331 SSCSpecificConfig();
332 break;
333 //~ case 30:
334 //~ Skip_S1(1, "sacPayloadEmbedding");
335 //~ SpatialSpecificConfig(); //ISO/IEC 23003-1
336 //~ break;
337 case 32:
338 case 33:
339 case 34:
340 MPEG_1_2_SpecificConfig();
341 break;
342 case 35:
343 DSTSpecificConfig();
344 break;
345 case 36:
346 Skip_S1(5, "fillBits");
347 ALSSpecificConfig();
348 break;
349 case 37:
350 case 38:
351 SLSSpecificConfig();
352 break;
353 case 39:
354 ELDSpecificConfig();
355 break;
356 //~ case 40:
357 //~ case 41:
358 //~ SymbolicMusicSpecificConfig(); //ISO/IEC 14496-23
359 //~ break;
360 default:
361 Element_Begin1("not implemented part");
362 Skip_BS(Data_BS_Remain()-((End==(size_t)-1)?0:End), "(Not implemented)");
363 Element_End0();
364 FILLING_BEGIN()
365 if (Mode==File_Aac::Mode_ADIF || Mode==File_Aac::Mode_ADTS)
366 File__Tags_Helper::Finish();
367 else if (Mode==Mode_AudioSpecificConfig)
368 File__Analyze::Finish();
369 Frame_Count=(size_t)-1; //Forcing not to parse following data anymore (if ParseSpeed==1)
370 FILLING_END()
371 return;
372 }
373
374 switch (audioObjectType)
375 {
376 case 17:
377 case 19:
378 case 20:
379 case 21:
380 case 22:
381 case 23:
382 case 24:
383 case 25:
384 case 26:
385 case 27:
386 case 39:
387 int8u epConfig;
388 Get_S1(2,epConfig, "epConfig");
389 if ( epConfig == 2 || epConfig == 3 )
390 ErrorProtectionSpecificConfig();
391 if ( epConfig == 3 )
392 {
393 bool directMapping;
394 Get_SB(directMapping, "directMapping");
395 if ( ! directMapping )
396 {
397 Element_Begin1("not implemented part");
398 Skip_BS(Data_BS_Remain()-((End==(size_t)-1)?0:End), "(Not implemented)");
399 Element_End0();
400 if (Mode==File_Aac::Mode_ADIF || Mode==File_Aac::Mode_ADTS)
401 File__Tags_Helper::Finish();
402 else
403 {
404 if (Mode==Mode_LATM)
405 File__Analyze::Accept();
406 File__Analyze::Finish();
407 }
408 Frame_Count=(size_t)-1; //Forcing not to parse following data anymore (if ParseSpeed==1)
409 return;
410 }
411 }
412 default : ;
413 }
414
415 if (extensionAudioObjectType!=5 && End!=(size_t)-1 && Data_BS_Remain()>=End+16)
416 {
417 int16u syncExtensionType;
418 Get_S2(11,syncExtensionType, "syncExtensionType");
419 if (syncExtensionType == 0x2b7)
420 {
421 sbrData=true;
422 GetAudioObjectType(extensionAudioObjectType, "extensionAudioObjectType");
423 if (extensionAudioObjectType==5 )
424 {
425 Get_SB(sbrPresentFlag, "sbrPresentFlag");
426 if (sbrPresentFlag)
427 {
428 Get_S1 (4, extension_sampling_frequency_index, "extensionSamplingFrequencyIndex"); Param_Info1(Aac_sampling_frequency[extension_sampling_frequency_index]);
429 if (extension_sampling_frequency_index==0xF)
430 {
431 Get_S3 (24, extension_sampling_frequency, "extensionSamplingFrequency");
432 extension_sampling_frequency_index=Aac_AudioSpecificConfig_sampling_frequency_index(extension_sampling_frequency);
433 }
434 else
435 extension_sampling_frequency=Aac_sampling_frequency[extension_sampling_frequency_index];
436 if (End!=(size_t)-1 && Data_BS_Remain()>=End+12)
437 {
438 int16u syncExtensionType;
439 Get_S2(11,syncExtensionType, "syncExtensionType");
440 if (syncExtensionType == 0x548)
441 {
442 psData=true;
443 Get_SB (psPresentFlag, "psPresentFlag");
444 }
445 }
446 }
447 }
448 if ( extensionAudioObjectType == 22 )
449 {
450 Get_SB(sbrPresentFlag, "sbrPresentFlag");
451 if (sbrPresentFlag)
452 {
453 Get_S1 (4, extension_sampling_frequency_index, "extensionSamplingFrequencyIndex"); Param_Info1(Aac_sampling_frequency[extension_sampling_frequency_index]);
454 if (extension_sampling_frequency_index==0xF)
455 {
456 Get_S3 (24, extension_sampling_frequency, "extensionSamplingFrequency");
457 extension_sampling_frequency_index=Aac_AudioSpecificConfig_sampling_frequency_index(extension_sampling_frequency);
458 }
459 else
460 extension_sampling_frequency=Aac_sampling_frequency[extension_sampling_frequency_index];
461 }
462 Skip_S1(4, "extensionChannelConfiguration");
463 }
464 }
465 }
466 Element_End0();
467 if (Data_BS_Remain()>End)
468 {
469 int8u LastByte=0xFF;
470 if (Data_BS_Remain()-End<8)
471 Peek_S1((int8u)(Data_BS_Remain()-End), LastByte);
472 Skip_BS(Data_BS_Remain()-End, LastByte?"Unknown":"Padding");
473 }
474
475 FILLING_BEGIN();
476 AudioSpecificConfig_OutOfBand (Frequency_b, audioObjectType, sbrData, psData, sbrPresentFlag, psPresentFlag);
477 FILLING_END();
478}

480//---------------------------------------------------------------------------
481void File_Aac::AudioSpecificConfig_OutOfBand (int64s sampling_frequency_, int8u audioObjectType_, bool sbrData, bool psData, bool sbrPresentFlag, bool psPresentFlag)
482{
483 if (!Frequency_b && sampling_frequency_) //Only if not yet set
484 {
485 Frequency_b=sampling_frequency_;
486 sampling_frequency_index=Aac_AudioSpecificConfig_sampling_frequency_index((int32u)Frequency_b);
487 }
488
489 if (audioObjectType_==(int8u)-1)
490 {
491 if (audioObjectType==(int8u)-1)
492 return; //All data is not yet available
493 if (Infos.find("Format_Settings_SBR")!=Infos.end())
494 {
495 sbrData=true;
496 sbrPresentFlag=Infos["Format_Settings_SBR"].find(__T("Yes"))!=string::npos;
497 }
498 else
499 {
500 sbrData=false;
501 sbrPresentFlag=false;
502 }
503 if (Infos.find("Format_Settings_PS")!=Infos.end())
504 {
505 psData=true;
506 psPresentFlag=Infos["Format_Settings_PS"].find(__T("Yes"))!=string::npos;
507 }
508 else
509 {
510 psData=false;
511 psPresentFlag=false;
512 }
513 }
514 else
515 audioObjectType=audioObjectType_;
516
517 if (Frequency_b)
518 Infos["SamplingRate"].From_Number(Frequency_b);
519 Infos["Format"].From_Local(Aac_Format(audioObjectType));
520 Infos["Format_Profile"].From_Local(Aac_Format_Profile(audioObjectType));
521 Infos["Codec"].From_Local(Aac_audioObjectType(audioObjectType));
522 if (channelConfiguration && channelConfiguration<8)
523 {
524 Infos["Channel(s)"].From_Number(Aac_Channels[channelConfiguration]);
525 Infos["ChannelPositions"].From_Local(Aac_ChannelConfiguration[channelConfiguration]);
526 Infos["ChannelPositions/String2"].From_Local(Aac_ChannelConfiguration2[channelConfiguration]);
527 Infos["ChannelLayout"].From_Local(Aac_ChannelLayout[channelConfiguration]);
528 }
529
530 if (sbrPresentFlag || !Infos["Format_Settings_SBR"].empty())
531 {
532 Infos["Format_Profile"]=__T("HE-AAC");
533 Ztring SamplingRate_Previous=Infos["SamplingRate"];
534 int32u SamplingRate=(extension_sampling_frequency_index==(int8u)-1)?(((int32u)Frequency_b)*2):extension_sampling_frequency;
535 if (SamplingRate)
536 {
537 Infos["SamplingRate"].From_Number(SamplingRate, 10);
538 if (MediaInfoLib::Config.LegacyStreamDisplay_Get())
539 {
540 Infos["Format_Profile"]+=__T(" / LC");
541 Infos["SamplingRate"]+=__T(" / ")+SamplingRate_Previous;
542 }
543 }
544 Infos["Format_Settings_SBR"]=__T("Yes (Implicit)");
545 Infos["Codec"]=Ztring().From_Local(Aac_audioObjectType(audioObjectType))+__T("-SBR");
546 }
547 else if (sbrData)
548 Infos["Format_Settings_SBR"]=__T("No (Explicit)");
549
550 if (psPresentFlag || !Infos["Format_Settings_PS"].empty())
551 {
552 Infos["Format_Profile"]=__T("HE-AACv2");
553 Ztring Channels=Infos["Channel(s)"];
554 Ztring ChannelPositions=Infos["ChannelPositions"];
555 Ztring SamplingRate_Previous=Infos["SamplingRate"];
556 Infos["Channel(s)"]=__T("2");
557 Infos["ChannelPositions"]=__T("Front: L R");
558 if (MediaInfoLib::Config.LegacyStreamDisplay_Get())
559 {
560 Infos["Format_Profile"]+=__T(" / HE-AAC / LC");
561 Infos["Channel(s)"]+=__T(" / ")+Channels+__T(" / ")+Channels;
562 Infos["ChannelPositions"]+=__T(" / ")+ChannelPositions+__T(" / ")+ChannelPositions;
563 int32u SamplingRate=(extension_sampling_frequency_index==(int8u)-1)?(((int32u)Frequency_b)*2):extension_sampling_frequency;
564 if (SamplingRate)
565 Infos["SamplingRate"]=Ztring().From_Number(SamplingRate, 10)+__T(" / ")+SamplingRate_Previous;
566 }
567 Infos["Format_Settings_PS"]=__T("Yes (Implicit)");
568 if (StreamPos_Last!=(size_t)-1)
569 {
570 Ztring Codec=Retrieve(Stream_Audio, StreamPos_Last, Audio_Codec);
571 Infos["Codec"]=Ztring().From_Local(Aac_audioObjectType(audioObjectType))+__T("-SBR-PS");
572 }
573 }
574 else if (psData)
575 Infos["Format_Settings_PS"]=__T("No (Explicit)");
576}

596//---------------------------------------------------------------------------
597void File_Aac::AudioMuxElement()
598{
599 Element_Begin1("AudioMuxElement");
600 if (muxConfigPresent)
601 {
602 bool useSameStreamMux;
603 Get_SB (useSameStreamMux, "useSameStreamMux");
604 if (!useSameStreamMux)
605 StreamMuxConfig();
606 }
607 if (sampling_frequency_index==(int8u)-1) //No previous config
608 {
609 CanFill=false;
610 Skip_BS(Data_BS_Remain(), "(Waiting for configuration)");
611 return;
612 }
613 if (audioMuxVersionA==0)
614 {
615 for (int8u i=0; i<=numSubFrames; i++)
616 {
617 PayloadLengthInfo();
618 PayloadMux();
619 }
620 if (otherDataPresent)
621 {
622 for(int32u i=0; i<otherDataLenBits; i++)
623 Skip_SB( "otherDataBit");
624 }
625 }
626 else
627 {
628 Element_Begin1("(not implemented)");
629 Skip_BS(Data_BS_Remain(), "(not implemented)");
630 Element_End0();
631 }
632 if (Data_BS_Remain()%8)
633 Skip_S1(Data_BS_Remain()%8, "byte_alignment");
634 Element_End0();
635}

637//---------------------------------------------------------------------------
638void File_Aac::StreamMuxConfig()
639{
640 Element_Begin1("StreamMuxConfig");
641
642 bool audioMuxVersion;
643 Get_SB (audioMuxVersion, "audioMuxVersion");
644 if (audioMuxVersion)
645 Get_SB (audioMuxVersionA, "audioMuxVersionA");
646 else
647 audioMuxVersionA=false;
648
649 if (!audioMuxVersionA)
650 {
651 if (audioMuxVersion==1)
652 {
653 Element_Begin1("(not implemented)");
654 Skip_BS(Data_BS_Remain(), "(not implemented)");
655 Element_End0();
656 //taraBufferFullness=LatmGetValue();
657 }
658
659 int8u streamCnt = 0;
660 bool useSameConfig;
661
662 Get_SB (allStreamsSameTimeFraming, "allStreamsSameTimeFraming");
663 Get_S1 (6, numSubFrames, "numSubFrames");
664 Get_S1 (4, numProgram, "numProgram");
665
666 for (int8u prog=0; prog<=numProgram; prog++)
667 {
668 Get_S1(3,numLayer, "numLayer");
669 for (int8u lay=0; lay<=numLayer; lay++)
670 {
671 progSIndx[streamCnt]=prog;
672 laySIndx[streamCnt]=lay;
673 streamID[prog][lay]=streamCnt++;
674 if (prog==0 && lay==0)
675 useSameConfig=0;
676 else
677 Get_SB(useSameConfig, "useSameConfig");
678
679 if (!useSameConfig)
680 {
681 if (audioMuxVersion==0)
682 AudioSpecificConfig();
683 else
684 {
685 int ascLen=LatmGetValue();
686 AudioSpecificConfig(Data_BS_Remain()-ascLen);
687 }
688 }
689
690 Get_S1(3,frameLengthType[streamID[prog][lay]], "frameLengthType[streamID[prog][lay]]");
691 switch(frameLengthType[streamID[prog][lay]])
692 {
693 case 0 :
694 Skip_S1(8, "latmBufferFullness[streamID[prog][lay]]");
695 if(!allStreamsSameTimeFraming)
696 {
697 Element_Begin1("(not implemented)");
698 Skip_BS(Data_BS_Remain(), "(not implemented)");
699 Element_End0();
700 }
701 //~ if ((!allStreamsSameTimeFraming) &&
702 //~ (AudioObjectType[lay] == 6 || AudioObjectType[lay] == 20) &&
703 //~ (AudioObjectType[lay-1] == 8 || AudioObjectType[lay-1] == 24))
704 //~ {
705 //~ Skip_S1(6, "coreFrameOffset");
706 //~ }
707 break;
708 case 1 :
709 Get_S2(9,frameLength[streamID[prog][lay]],"frameLength[streamID[prog][lay]]");
710 break;
711 case 3 :
712 case 4 :
713 case 5 :
714 Skip_S1(6, "CELPframeLengthTableIndex[streamID[prog][lay]]");
715 break;
716 case 6 :
717 case 7 :
718 Skip_S1(1, "HVXCframeLengthTableIndex[streamID[prog][lay]]");
719 break;
720 default :
721 Element_Begin1("(not implemented)");
722 Skip_BS(Data_BS_Remain(), "(not implemented)");
723 Element_End0();
724 }
725 }
726 }
727 bool crcCheckPresent;
728 Get_SB (otherDataPresent, "otherDataPresent");
729 if (otherDataPresent)
730 {
731 if (audioMuxVersion==1)
732 otherDataLenBits=LatmGetValue();
733 else
734 {
735 otherDataLenBits=0;
736 bool otherDataLenEsc;
737 do
738 {
739 int8u otherDataLenTmp;
740 otherDataLenBits<<=8;
741 Get_SB( otherDataLenEsc, "otherDataLenEsc");
742 Get_S1(8, otherDataLenTmp, "otherDataLenTmp");
743 otherDataLenBits+=otherDataLenTmp;
744 }
745 while (otherDataLenEsc);
746 }
747 }
748 Get_SB(crcCheckPresent, "crcCheckPresent");
749 if(crcCheckPresent)
750 Skip_S1(8, "crcCheckSum");
751 }
752 else
753 {
754 Element_Begin1("(not implemented)");
755 Skip_BS(Data_BS_Remain(), "(not implemented)");
756 Element_End0();
757 }
758 Element_End0();
759
760 FILLING_BEGIN();
761 CanFill=true;
762 FILLING_END();
763}

765//---------------------------------------------------------------------------
766int32u File_Aac::LatmGetValue()
767{
768 Element_Begin1("LatmGetValue");
769 int8u valueTmp, bytesForValue;
770 Get_S1(2, bytesForValue, "bytesForValue");
771 int32u value=0;
772 for (int8u i=0; i<=bytesForValue; i++)
773 {
774 value<<=8;

832 }

834 || frameLengthType[streamID[prog][lay]]==5
835 || frameLengthType[streamID[prog][lay]]==7)
836 {
837 Skip_S1(2, "MuxSlotLengthCoded[streamID[prog][lay]]");

839 }
840 }

845void File_Aac::PayloadMux()
846{
847 Element_Begin1("PayloadMux");
848 if (allStreamsSameTimeFraming)
849 {
850 for (int8u prog=0; prog<=numProgram; prog++)
851 for (int8u lay=0; lay<=numLayer; lay++)
852 {
853 switch(frameLengthType[streamID[prog][lay]])
854 {
855 case 0 :
856 if (CA_system_ID_MustSkipSlices)

862 else
863 raw_data_block();
864 break;
865 case 1 :
866 Skip_BS(8 * (frameLength[streamID[prog][lay]] + 20),"payload[streamID[prog][lay]]");
867 break;
868 default:
869 Element_Begin1("(not implemented)");
870 Skip_BS(Data_BS_Remain(), "(not implemented)");
871 Element_End0();
872 }
873 }
874 }
875 else
876 {
877 for (int8u chunkCnt=0; chunkCnt<=numChunk; chunkCnt++)
878 {
879 int8u prog=progCIndx[chunkCnt];
880 int8u lay=layCIndx[chunkCnt];
881 switch(frameLengthType[streamID[prog][lay]])
882 {
883 case 0 :
884 raw_data_block(); //Skip_BS(MuxSlotLengthBytes[streamID[prog][lay]], "payload[streamID[prog][lay]]");
885 break;
886 case 1 :
887 Skip_BS(8*(frameLength[streamID[prog][lay]]+20), "payload[streamID[prog][lay]]");
888 break;
889 default:
890 Element_Begin1("not implemented");
891 Element_End0();
892 }
893 }
894 }
895 Element_End0();

904{
905 Element_Begin1("ErrorProtectionSpecificConfig");
906 int8u number_of_predefined_set,number_of_concatenated_frame,interleave_type;
907 Get_S1(8,number_of_predefined_set, "number_of_predefined_set");
908 Get_S1(2,interleave_type, "interleave_type");
909 Skip_S1(3, "bit_stuffing");
910 Get_S1(3,number_of_concatenated_frame, "number_of_concatenated_frame");
911 for (int8u i = 0; i < number_of_predefined_set; i++ )
912 {
913 int8u number_of_class;
914 Get_S1(6,number_of_class, "number_of_class[i]");
915 for (int8u j = 0; j < number_of_class; j++)
916 {
917 bool length_escape,fec_type,rate_escape,crclen_escape;
918 Get_SB(length_escape, "length_escape[i][j]");
919 Get_SB(rate_escape, "rate_escape[i][j]");
920 Get_SB(crclen_escape, "crclen_escape[i][j]");
921 if (number_of_concatenated_frame != 1)
922 Skip_SB( "concatenate_flag[i][j]");
923 Get_SB(fec_type, "fec_type[i][j]");
924 if(!fec_type)
925 Skip_SB( "termination_switch[i][j]");
926 if (interleave_type == 2)
927 Skip_S1(2, "interleave_switch[i][j]");
928 Skip_SB( "class_optional");
929 if (length_escape)
930 {
931 /* ESC */
932 Skip_S1(4, "number_of_bits_for_length[i][j]");
933 }
934 else
935 {
936 Skip_S2(16, "class_length[i][j]");
937 }
938 if (!rate_escape)
939 { /* not ESC */
940 if(fec_type)
941 {
942 Skip_S1(7, "class_rate[i][j]");
943 }
944 else
945 {
946 Skip_S1(5, "class_rate[i][j]");
947 }
948 }
949 if (!crclen_escape)
950 {
951 /* not ESC */
952 Skip_S1(5, "class_crclen[i][j]");
953 }
954 }
955 bool class_reordered_output;
956 Get_SB(class_reordered_output, "class_reordered_output");
957 if ( class_reordered_output )
958 {
959 for (int j = 0; j < number_of_class; j++ )
960 {
961 Skip_S1(6, "class_output_order[i][j]");
962 }
963 }
964 }
965 bool header_protection;
966 Get_SB(header_protection, "header_protection");
967 if (header_protection)
968 {
969 Skip_S1(5, "header_rate");
970 Skip_S1(5, "header_crclen");
971 }
972 Element_End0();
973}

986 Skip_C4( "adif_id");
987 BS_Begin();
988 TEST_SB_SKIP( "copyright_id_present");
989 Skip_S4(32, "copyright_id");

991 Skip_S4( 8, "copyright_id");
992 TEST_SB_END();

1002 BS_End();
1003
1004 FILLING_BEGIN();
1005 Fill(Stream_General, 0, General_Format, "ADIF", Unlimited, true, true);
1006 Fill(Stream_General, 0, General_HeaderSize, Element_Size);
1007 Fill(Stream_General, 0, General_OverallBitRate_Mode, bitstream_type?"VBR":"CBR");

1015 Infos[bitstream_type?"BitRate_Maximum":"BitRate"].From_Number(bitrate);

1020 FILLING_END();
