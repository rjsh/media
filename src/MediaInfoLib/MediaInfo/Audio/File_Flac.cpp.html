59//***************************************************************************
60// Buffer - File header
61//***************************************************************************
62
63//---------------------------------------------------------------------------
64bool File_Flac::FileHeader_Begin()
65{
66 if (!File__Tags_Helper::FileHeader_Begin())
67 return false;
68
69 //Element_Size
70 if (Buffer_Size<Buffer_Offset+4+(VorbisHeader?9:0))
71 return false; //Must wait for more data
72
73 if (CC4(Buffer+Buffer_Offset+(VorbisHeader?9:0))!=0x664C6143) //"fLaC"
74 {
75 File__Tags_Helper::Finish("Flac");
76 return false;
77 }
78
79 //All should be OK...
80 return true;
81}

83//---------------------------------------------------------------------------
84void File_Flac::FileHeader_Parse()
85{
86 //Parsing
87 if (VorbisHeader)
88 {
89 Skip_B1( "Signature");
90 Skip_Local(4, "Signature");
91 Skip_B1( "Major version");
92 Skip_B1( "Minor version");
93 Skip_B2( "Number of header");
94 }
95 Skip_C4( "Signature");
96}

102//---------------------------------------------------------------------------
103void File_Flac::Header_Parse()
104{
105 //Parsing
106 int32u Length;
107 int8u BLOCK_TYPE;
108 BS_Begin();
109 Get_SB ( Last_metadata_block, "Last-metadata-block");
110 Get_S1 (7, BLOCK_TYPE, "BLOCK_TYPE");
111 BS_End();
112 Get_B3 (Length, "Length");
113
114 //Filling
115 Header_Fill_Code(BLOCK_TYPE, Ztring().From_CC1(BLOCK_TYPE));
116 Header_Fill_Size(Element_Offset+Length);
117}

119//---------------------------------------------------------------------------
120void File_Flac::Data_Parse()
121{
122 #define CASE_INFO(_NAME) \
123 case Flac::_NAME : Element_Info1(#_NAME); _NAME(); break;
124
125 //Parsing
126 switch ((int16u)Element_Code)
127 {
128 CASE_INFO(STREAMINFO);
129 CASE_INFO(PADDING);
130 CASE_INFO(APPLICATION);
131 CASE_INFO(SEEKTABLE);
132 CASE_INFO(VORBIS_COMMENT);
133 CASE_INFO(CUESHEET);
134 CASE_INFO(PICTURE);
135 default : Skip_XX(Element_Size, "Data");
136 }
137
138 if (Last_metadata_block)
139 {
140 if (!IsSub)
141 Fill(Stream_Audio, 0, Audio_StreamSize, File_Size-(File_Offset+Buffer_Offset+Element_Size));
142
143 if (Retrieve(Stream_Audio, 0, Audio_ChannelPositions).empty() && Retrieve(Stream_Audio, 0, Audio_ChannelPositions_String2).empty())
144 {
145 int32u t = 0;
146 int32s c = Retrieve(Stream_Audio, 0, Audio_Channel_s_).To_int32s();
147 if (c==1) t=4;
148 else if (c==2) t=3;
149 else if (c==3) t=7;
150 else if (c==4) t=1539;
151 else if (c==5) t=1543;
152 else if (c==6) t=1551;
153 else if (c==7) t=1807;
154 else if (c==8) t=1599;
155 if (t) {
156 Fill(Stream_Audio, 0, Audio_ChannelPositions, ExtensibleWave_ChannelMask(t));
157 Fill(Stream_Audio, 0, Audio_ChannelPositions_String2, ExtensibleWave_ChannelMask2(t));
158 }
159 }
160
161 //No more need data
162 File__Tags_Helper::Finish("Flac");
163 }

170//---------------------------------------------------------------------------
171void File_Flac::STREAMINFO()
172{
173 //Parsing
174 int64u Samples;
175 int32u FrameSize_Min, FrameSize_Max, SampleRate;
176 int8u Channels, BitPerSample;
177 Skip_B2( "BlockSize_Min"); //The minimum block size (in samples) used in the stream.
178 Skip_B2( "BlockSize_Max"); //The maximum block size (in samples) used in the stream. (Minimum blocksize == maximum blocksize) implies a fixed-blocksize stream.
179 Get_B3 ( FrameSize_Min, "FrameSize_Min"); //The minimum frame size (in bytes) used in the stream. May be 0 to imply the value is not known.
180 Get_B3 ( FrameSize_Max, "FrameSize_Max"); //The maximum frame size (in bytes) used in the stream. May be 0 to imply the value is not known.
181 BS_Begin();
182 Get_S3 (20, SampleRate, "SampleRate"); //Sample rate in Hz. Though 20 bits are available, the maximum sample rate is limited by the structure of frame headers to 1048570Hz. Also, a value of 0 is invalid.
183 Get_S1 ( 3, Channels, "Channels"); Param_Info2(Channels+1, " channels"); //(number of channels)-1. FLAC supports from 1 to 8 channels
184 Get_S1 ( 5, BitPerSample, "BitPerSample"); Param_Info2(BitPerSample+1, " bits"); //(bits per sample)-1. FLAC supports from 4 to 32 bits per sample. Currently the reference encoder and decoders only support up to 24 bits per sample.
185 Get_S5 (36, Samples, "Samples");
186 BS_End();
187 Skip_B16( "MD5 signature of the unencoded audio data");
188
189 FILLING_BEGIN();
190 if (SampleRate==0)
191 return;
192 File__Tags_Helper::Accept("FLAC");
193
194 File__Tags_Helper::Streams_Fill();
195
196 File__Tags_Helper::Stream_Prepare(Stream_Audio);
197 Fill(Stream_Audio, 0, Audio_Format, "FLAC");
198 Fill(Stream_Audio, 0, Audio_Codec, "FLAC");
199 if (FrameSize_Min==FrameSize_Max && FrameSize_Min!=0 ) // 0 means it is unknown
200 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "CBR");
201 else
202 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "VBR");
203 Fill(Stream_Audio, 0, Audio_SamplingRate, SampleRate);
204 Fill(Stream_Audio, 0, Audio_Channel_s_, Channels+1);
205 Fill(Stream_Audio, 0, Audio_BitDepth, BitPerSample+1);
206 if (!IsSub)
207 Fill(Stream_Audio, 0, Audio_Duration, Samples*1000/SampleRate);
208 FILLING_END();
209}

220//---------------------------------------------------------------------------
221void File_Flac::VORBIS_COMMENT()
222{
223 //Parsing
224 #if defined(MEDIAINFO_VORBISCOM_YES)
225 File_VorbisCom VorbisCom;
226 VorbisCom.StreamKind_Specific=Stream_Audio;
227 Open_Buffer_Init(&VorbisCom);
228 Open_Buffer_Continue(&VorbisCom);
229 File__Analyze::Finish(&VorbisCom);
230
231 //Specific case: bit depth
232 if (!VorbisCom.Retrieve(Stream_Audio, 0, Audio_BitDepth).empty() && VorbisCom.Retrieve(Stream_Audio, 0, Audio_BitDepth).To_int64u()<Retrieve(Stream_Audio, 0, Audio_BitDepth).To_int64u())
233 {
234 //Bit depth information from tags is the real value, the one from Flac is the count of bits stored
235 Fill(Stream_Audio, 0, Audio_BitDepth_Stored, Retrieve(Stream_Audio, 0, Audio_BitDepth));
236 Fill(Stream_Audio, 0, Audio_BitDepth, VorbisCom.Retrieve(Stream_Audio, 0, Audio_BitDepth), true);
237 VorbisCom.Clear(Stream_Audio, 0, Audio_BitDepth);
238 }
239
240 Merge(VorbisCom, Stream_General, 0, 0);
241 Merge(VorbisCom, Stream_Audio, 0, 0);
242 Merge(VorbisCom, Stream_Menu, 0, 0);
243 #else
244 Skip_XX(Element_Offset, "Data");
245 #endif
246}

248//---------------------------------------------------------------------------
249void File_Flac::PICTURE()
250{
251 //Parsing
252 int32u PictureType, MimeType_Size, Description_Size, Data_Size;
253 Ztring MimeType, Description;
254 Get_B4 (PictureType, "Picture type"); Element_Info1(Id3v2_PictureType((int8u)PictureType));
255 Get_B4 (MimeType_Size, "MIME type size");
256 Get_Local(MimeType_Size, MimeType, "MIME type");
257 Get_B4 (Description_Size, "Description size");
258 Get_UTF8(Description_Size, Description, "Description");
259 Skip_B4( "Width");
260 Skip_B4( "Height");
261 Skip_B4( "Color depth");
262 Skip_B4( "Number of colors used");
263 Get_B4 (Data_Size, "Data size");
264 if (Element_Offset+Data_Size>Element_Size)
265 return; //There is a problem
266 std::string Data_Raw((const char*)(Buffer+(size_t)(Buffer_Offset+Element_Offset)), Data_Size);
267 std::string Data_Base64(Base64::encode(Data_Raw));
268 Skip_XX(Element_Size-Element_Offset, "Data");
269
270 //Filling
271 Fill(Stream_General, 0, General_Cover, "Yes");
272 Fill(Stream_General, 0, General_Cover_Description, Description);
273 Fill(Stream_General, 0, General_Cover_Type, Id3v2_PictureType((int8u)PictureType));
274 Fill(Stream_General, 0, General_Cover_Mime, MimeType);
275 Fill(Stream_General, 0, General_Cover_Data, Data_Base64);
276}
