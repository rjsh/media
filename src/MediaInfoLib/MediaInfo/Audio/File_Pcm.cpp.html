38const char* Smpte_St0302_ChannelsPositions(int8u Channels)
39{
40 switch (Channels)
41 {
42 case 2 : return "Front: L R"; //2 channels
43 case 4 : return "Front: L C R, LFE"; //4 channels
44 case 6 : return "Front: L C R, Side: L R, LFE"; //6 channels
45 case 8 : return "Front: L C R, Side: L R, Back: L R, LFE"; //8 channels
46 default : return "";
47 }
48}
49
50//---------------------------------------------------------------------------
51const char* Smpte_St0302_ChannelsPositions2(int8u Channels)
52{
53 switch (Channels)
54 {
55 case 2 : return "2/0/0.0"; //2 channels
56 case 4 : return "3/0/0.1"; //4 channels
57 case 6 : return "3/2/0.1"; //6 channels
58 case 8 : return "3/2/2.1"; //8 channels
59 default : return "";
60 }
61}

96void File_Pcm::Streams_Fill()
97{
98 if (Count_Get(Stream_Audio)==0)
99 {
100 Stream_Prepare(Stream_Audio);
101 Fill(Stream_Audio, 0, Audio_Format, "PCM");
102 Fill(Stream_Audio, 0, Audio_Codec, "PCM");
103 }
104
105 //Filling
106 Ztring Firm, ITU;
107 if (Codec==__T("EVOB")) {Firm=__T(""); Endianness='B'; Sign='S';} //PCM Signed 16 bits Big Endian, Interleavement is for 2 samples*2 channels L0-1/L0-0/R0-1/R0-0/L1-1/L1-0/R1-1/R1-0/L0-2/R0-2/L1-2/R1-2, http://wiki.multimedia.cx/index.php?title=PCM
108 else if (Codec==__T("VOB")) {Firm=__T(""); Endianness='B'; Sign='S';} //PCM Signed 16 bits Big Endian, Interleavement is for 2 samples*2 channels L0-1/L0-0/R0-1/R0-0/L1-1/L1-0/R1-1/R1-0/L0-2/R0-2/L1-2/R1-2, http://wiki.multimedia.cx/index.php?title=PCM
109 else if (Codec==__T("M2TS")) {Firm=__T(""); Endianness='B'; Sign='S';} //PCM Signed Big Endian
110 else if (Codec==__T("A_PCM/INT/BIG")) {Firm=__T(""); Endianness='B';}
111 else if (Codec==__T("A_PCM/INT/LITTLE")) {Firm=__T(""); Endianness='L';}
112 else if (Codec==__T("A_PCM/INT/FLOAT")) {Firm=__T(""); Endianness='B'; Sign='F';}
113 else if (Codec==__T("fl32")) { if (!Endianness) Endianness='B'; if (!Sign) Sign='F'; BitDepth=32;}
114 else if (Codec==__T("fl64")) { if (!Endianness) Endianness='B'; if (!Sign) Sign='F'; BitDepth=64;}
115 else if (Codec==__T("in24")) { if (!Endianness) Endianness='B'; if (!Sign) Sign='U'; BitDepth=24;}
116 else if (Codec==__T("in32")) { if (!Endianness) Endianness='B'; if (!Sign) Sign='U'; BitDepth=32;}
117 else if (Codec==__T("raw ")) { if (!Endianness) Endianness='L'; Sign='U';}
118 else if (Codec==__T("twos")) { Endianness='B'; Sign='S';}
119 else if (Codec==__T("sowt")) { Endianness='L'; Sign='S';}
120 else if (Codec==__T("lpcm")) { if (!Endianness) Endianness='B'; if (!Sign) Sign='S';}
121 else if (Codec==__T("SWF ADPCM")) {Firm=__T("SWF");}
122 else if (Codec==__T("1")) { if (BitDepth)
123 {
124 if (BitDepth>8)
125 { Endianness='L'; Sign='S';}
126 else
127 { Sign='U';}
128 }
129 }
130 else if (Codec==__T("2")) {Firm=__T("Microsoft");}
131 else if (Codec==__T("3")) { Endianness='F';}
132 else if (Codec==__T("10")) {Firm=__T("OKI");}
133 else if (Codec==__T("11")) {Firm=__T("Intel");}
134 else if (Codec==__T("12")) {Firm=__T("Mediaspace");}
135 else if (Codec==__T("13")) {Firm=__T("Sierra");}
136 else if (Codec==__T("14")) {Firm=__T("Antex");}
137 else if (Codec==__T("17")) {Firm=__T("Dialogic");}
138 else if (Codec==__T("18")) {Firm=__T("Mediavision");}
139 else if (Codec==__T("20")) {Firm=__T("Yamaha");}
140 else if (Codec==__T("33")) {Firm=__T("Antex");}
141 else if (Codec==__T("36")) {Firm=__T("DSP Solution");}
142 else if (Codec==__T("38")) {Firm=__T("Natural MicroSystems");}
143 else if (Codec==__T("39")) {Firm=__T("Crystal Semiconductor");}
144 else if (Codec==__T("3B")) {Firm=__T("Rockwell");}
145 else if (Codec==__T("40")) {Firm=__T("Antex Electronics");}
146 else if (Codec==__T("42")) {Firm=__T("IBM");}
147 else if (Codec==__T("45")) {Firm=__T("Microsoft"); ITU=__T("G.726");}
148 else if (Codec==__T("64")) {Firm=__T("Apicom"); ITU=__T("G.726");}
149 else if (Codec==__T("65")) {Firm=__T("Apicom"); ITU=__T("G.722");}
150 else if (Codec==__T("85")) {Firm=__T("DataFusion Systems"); ITU=__T("G.726");}
151 else if (Codec==__T("8B")) {Firm=__T("Infocom"); ITU=__T("G.721");}
152 else if (Codec==__T("97")) {Firm=__T("ZyXEL");}
153 else if (Codec==__T("100")) {Firm=__T("Rhetorex");}
154 else if (Codec==__T("125")) {Firm=__T("Sanyo");}
155 else if (Codec==__T("140")) {Firm=__T("Dictaphone"); ITU=__T("G.726");}
156 else if (Codec==__T("170")) {Firm=__T("Unisys");}
157 else if (Codec==__T("175")) {Firm=__T("SyCom"); ITU=__T("G.726");}
158 else if (Codec==__T("178")) {Firm=__T("Knownledge");}
159 else if (Codec==__T("200")) {Firm=__T("Creative");}
160 else if (Codec==__T("210")) {Firm=__T("Uher");}
161 else if (Codec==__T("285")) {Firm=__T("Norcom Voice Systems");}
162 else if (Codec==__T("1001")) {Firm=__T("Olivetti");}
163 else if (Codec==__T("1C03")) {Firm=__T("Lucent"); ITU=__T("G.723");}
164 else if (Codec==__T("1C0C")) {Firm=__T("Lucent"); ITU=__T("G.723");}
165 else if (Codec==__T("4243")) {ITU=__T("G.726");}
166 else if (Codec==__T("A105")) {ITU=__T("G.726");}
167 else if (Codec==__T("A107")) {ITU=__T("G.726");}
168
169 //Format
170 Fill(Stream_Audio, 0, Audio_Codec_String, "PCM");
171 Fill(Stream_Audio, 0, Audio_Codec_Family, "PCM");
172 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "CBR");
173
174 //SamplingRate
175 if (SamplingRate)
176 Fill(Stream_Audio, 0, Audio_SamplingRate, SamplingRate);
177
178 //Firm
179 Fill(Stream_Audio, 0, Audio_Format_Settings, Firm);
180 Fill(Stream_Audio, 0, Audio_Format_Settings_Firm, Firm);
181 Fill(Stream_Audio, 0, Audio_Codec_Settings, Firm);
182 Fill(Stream_Audio, 0, Audio_Codec_Settings_Firm, Firm);
183
184 //Endianess
185 const char* Value;
186 switch (Endianness)
187 {
188 case 'B': Value="Big"; break;
189 case 'L': Value="Little"; break;
190 default : Value="";
191 }
192 Fill(Stream_Audio, 0, Audio_Format_Settings, Value);
193 Fill(Stream_Audio, 0, Audio_Format_Settings_Endianness, Value);
194 Fill(Stream_Audio, 0, Audio_Codec_Settings, Value);
195 Fill(Stream_Audio, 0, Audio_Codec_Settings_Endianness, Value);
196
197 //Sign
198 switch (Sign)
199 {
200 case 'S': Value="Signed"; break;
201 case 'U': Value="Unsigned"; break;
202 default : Value="";
203 }
204 Fill(Stream_Audio, 0, Audio_Format_Settings, Value);
205 Fill(Stream_Audio, 0, Audio_Format_Settings_Sign, Value);
206 Fill(Stream_Audio, 0, Audio_Codec_Settings, Value);
207 Fill(Stream_Audio, 0, Audio_Codec_Settings_Sign, Value);
208
209 //ITU
210 Fill(Stream_Audio, 0, Audio_Format_Settings, ITU);
211 Fill(Stream_Audio, 0, Audio_Format_Settings_ITU, ITU);
212 Fill(Stream_Audio, 0, Audio_Codec_Settings, ITU);
213 Fill(Stream_Audio, 0, Audio_Codec_Settings_ITU, ITU);
214
215 //BitDepth
216 if (BitDepth_Significant)
217 {
218 Fill(Stream_Audio, 0, Audio_BitDepth, BitDepth_Significant);
219 Fill(Stream_Audio, 0, Audio_BitDepth_Stored, BitDepth);
220 }
221 else if (BitDepth)
222 Fill(Stream_Audio, 0, Audio_BitDepth, BitDepth);
223
224 //Channels
225 if (Channels)
226 Fill(Stream_Audio, 0, Audio_Channel_s_, Channels);
227
228 //Bit rate
229 if (SamplingRate && BitDepth && Channels)
230 Fill(Stream_Audio, 0, Audio_BitRate, SamplingRate*BitDepth*Channels);
231
232 //ChannelsPositions
233 if (Codec==__T("SMPTE ST 337"))
234 {
235 Fill(Stream_Audio, 0, Audio_ChannelPositions, Smpte_St0302_ChannelsPositions(Channels));
236 Fill(Stream_Audio, 0, Audio_ChannelPositions_String2, Smpte_St0302_ChannelsPositions2(Channels));
237 }
238}

253void File_Pcm::Read_Buffer_Continue()
254{
255 //Testing if we get enough data
256 if (SamplingRate && BitDepth && Channels)
257 {
258 int64u BitRate=SamplingRate*BitDepth*Channels;
259 int64u ByteRate=BitRate/8;
260 if (Buffer_Size>=ByteRate/4) // 1/4 of second is enough for detection
261 Frame_Count_Valid=2;
262 }
263
264 #if MEDIAINFO_DEMUX
265 if (Demux_UnpacketizeContainer && !Status[IsAccepted])
266 {
267 Frame_Count_Valid_Demux++;
268 if (Frame_Count_Valid_Demux<Frame_Count_Valid)
269 Element_WaitForMoreData();
270 }
271 #endif //MEDIAINFO_DEMUX
272}

294//---------------------------------------------------------------------------
295void File_Pcm::Header_Parse()
296{
297 //Filling
298 Header_Fill_Code(0, "Block");
299 if (BitDepth && Channels)
300 {
301 int64u Size=(Element_Size/(BitDepth*Channels/8))*(BitDepth*Channels/8); //A complete sample
302 if (Element_Size && Size==0)
303 {
304 Element_WaitForMoreData();
305 return;
306 }
307 Header_Fill_Size(Size);
308 }
309 else
310 Header_Fill_Size(Element_Size); // Unknown sample size
311}
312
313//---------------------------------------------------------------------------
314void File_Pcm::Data_Parse()
315{
316 #if MEDIAINFO_DEMUX
317 FrameInfo.PTS=FrameInfo.DTS;
318 if (Frame_Count_Valid_Demux)
319 {
320 if (FrameInfo.DUR!=(int64u)-1)
321 FrameInfo.DUR*=Frame_Count_Valid_Demux;
322 if (Frame_Count_NotParsedIncluded!=(int64u)-1 && Frame_Count_NotParsedIncluded>=Frame_Count_Valid_Demux)
323 Frame_Count_NotParsedIncluded-=Frame_Count_Valid_Demux-1;
324 }
325 Demux_random_access=true;
326 Element_Code=(int64u)-1;
327
328 if (BitDepth==20 && Endianness=='L' && Config->Demux_PCM_20bitTo16bit_Get())
329 {
330 size_t Info_Offset=(size_t)Element_Size;
331 const int8u* Info=Buffer+Buffer_Offset;
332 size_t Info2_Size=Info_Offset*4/5;
333 int8u* Info2=new int8u[Info2_Size];
334 size_t Info2_Pos=0;
335 size_t Info_Pos=0;
336
337 //Removing bits 3-0 (Little endian)
338 // Dest : 20LE / L1L0 L3L2 R0L4 R2R1 R4R3
339 // Source: L2L1 L4L3 R2R1 R4R2
340 while (Info_Pos+5<=Info_Offset)
341 {
342 Info2[Info2_Pos ] =(Info[Info_Pos+1]<<4 ) | (Info[Info_Pos+0]>>4 );
343 Info2[Info2_Pos+1] =(Info[Info_Pos+2]<<4 ) | (Info[Info_Pos+1]>>4 );
344 Info2[Info2_Pos+2] = Info[Info_Pos+3] ;
345 Info2[Info2_Pos+3] = Info[Info_Pos+4] ;
346
347 Info2_Pos+=4;
348 Info_Pos+=5;
349 }
350
351 Demux(Info2, Info2_Pos, ContentType_MainStream);
352
353 delete[] Info2;
354 }
355 else if (BitDepth==20 && Endianness=='L' && Config->Demux_PCM_20bitTo24bit_Get())
356 {
357 size_t Info_Offset=(size_t)Element_Size;
358 const int8u* Info=Buffer+Buffer_Offset;
359 size_t Info2_Size=Info_Offset*6/5;
360 int8u* Info2=new int8u[Info2_Size];
361 size_t Info2_Pos=0;
362 size_t Info_Pos=0;
363
364 //Padding bits 3-0 (Little endian)
365 // Dest : 20LE / L1L0 L3L2 R0L4 R2R1 R4R3
366 // Source: L0XX L2L1 L4L3 R0XX R2R1 R4R2
367 while (Info_Pos+5<=Info_Offset)
368 {
369 Info2[Info2_Pos ] = Info[Info_Pos+0]<<4 ;
370 Info2[Info2_Pos+1] =(Info[Info_Pos+1]<<4 ) | (Info[Info_Pos+0]>>4 );
371 Info2[Info2_Pos+2] =(Info[Info_Pos+2]<<4 ) | (Info[Info_Pos+1]>>4 );
372 Info2[Info2_Pos+3] = Info[Info_Pos+2]&0xF0 ;
373 Info2[Info2_Pos+4] = Info[Info_Pos+3] ;
374 Info2[Info2_Pos+5] = Info[Info_Pos+4] ;
375
376 Info2_Pos+=6;
377 Info_Pos+=5;
378 }
379
380 Demux(Info2, Info2_Pos, ContentType_MainStream);
381
382 delete[] Info2;
383 }
384 else
385 {
386 Demux(Buffer+Buffer_Offset, (size_t)Element_Size, ContentType_MainStream);
387 }
388 #endif //MEDIAINFO_DEMUX
389
390 //Parsing
391 Skip_XX(Element_Size, "Data"); //It is impossible to detect... Default is no detection, only filling
392
393 #if MEDIAINFO_DEMUX
394 if (Frame_Count_Valid_Demux)
395 {
396 Frame_Count+=Frame_Count_Valid_Demux-1;
397 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
398 Frame_Count_NotParsedIncluded+=Frame_Count_Valid_Demux-1;
399 FrameInfo.DUR/=Frame_Count_Valid_Demux;
400 if (FrameInfo.DTS!=(int64u)-1)
401 FrameInfo.DTS+=FrameInfo.DUR*Frame_Count;
402 Frame_Count_Valid_Demux=0;
403 }
404 #endif //MEDIAINFO_DEMUX
405 Frame_Count++;
406 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
407 Frame_Count_NotParsedIncluded++;
408 if (FrameInfo.DTS!=(int64u)-1 && FrameInfo.DUR!=(int64u)-1)
409 {
410 if (BitDepth && Channels && SamplingRate)
411 FrameInfo.DTS+=Element_Size*1000000000*8/BitDepth/Channels/SamplingRate;
412 else
413 FrameInfo.DTS+=FrameInfo.DUR;
414 FrameInfo.PTS=FrameInfo.DTS;
415 }
416 if ((!Status[IsAccepted] && Frame_Count>=Frame_Count_Valid) || File_Offset+Buffer_Size>=File_Size)
417 {
418 Accept();
419 Fill();
420 }
421}
