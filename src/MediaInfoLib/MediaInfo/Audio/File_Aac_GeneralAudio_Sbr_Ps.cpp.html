34void File_Aac::ps_data(size_t End)
35{
36 FILLING_BEGIN();
37 if (Infos["Format_Settings_PS"].empty())
38 {
39 Infos["Format_Profile"]=__T("HE-AACv2");
40 Ztring Channels=Infos["Channel(s)"];
41 Ztring ChannelPositions=Infos["ChannelPositions"];
42 Ztring SamplingRate=Infos["SamplingRate"];
43 Infos["Channel(s)"]=__T("2");
44 Infos["ChannelPositions"]=__T("Front: L R");
45 if (MediaInfoLib::Config.LegacyStreamDisplay_Get())
46 {
47 Infos["Format_Profile"]+=__T(" / HE-AAC / LC");
48 Infos["Channel(s)"]+=__T(" / ")+Channels+__T(" / ")+Channels;
49 Infos["ChannelPositions"]+=__T(" / ")+ChannelPositions+__T(" / ")+ChannelPositions;
50 Infos["SamplingRate"]=Ztring().From_Number((extension_sampling_frequency_index==(int8u)-1)?(Frequency_b*2):extension_sampling_frequency, 10)+__T(" / ")+SamplingRate;
51 }
52 Infos["Format_Settings_PS"]=__T("Yes (Implicit)");
53 Ztring Codec=Retrieve(Stream_Audio, StreamPos_Last, Audio_Codec);
54 Infos["Codec"]=Ztring().From_Local(Aac_audioObjectType(audioObjectType))+__T("-SBR-PS");
55 }
56 FILLING_END();
57
58 //Parsing
59 Element_Begin1("ps_data");
60 bool enable_ps_header;
61 Get_SB(enable_ps_header, "enable_ps_header");
62 if (enable_ps_header)
63 {
64 //Init
65 delete ps; ps=new ps_handler();
66
67 Get_SB(ps->enable_iid, "enable_iid");
68 if (ps->enable_iid)
69 {
70 Get_S1 (3, ps->iid_mode, "iid_mode");
71 }
72 Get_SB(ps->enable_icc, "enable_icc");
73 if (ps->enable_icc)
74 {
75 Get_S1 (3, ps->icc_mode, "icc_mode");
76 }
77 Get_SB(ps->enable_ext, "enable_ext");
78 }
79
80 if (ps==NULL)
81 {
82 if (Data_BS_Remain()>End)
83 Skip_BS(Data_BS_Remain()-End, "(Waiting for header)");
84 Element_End0();
85 return;
86 }
87
88 //PS not yet parsed
89 if (Data_BS_Remain()>End)
90 Skip_BS(Data_BS_Remain()-End, "Data");
91 Element_End0();
92}
