63//---------------------------------------------------------------------------
64void File_Celt::Data_Parse()
65{
66 //Parsing
67 if (Identification_Done)
68 Comment();
69 else
70 Identification();
71}

73//***************************************************************************
74// Elements
75//***************************************************************************
76
77//---------------------------------------------------------------------------
78void File_Celt::Identification()
79{
80 Element_Name("Identification");
81
82 //Parsing
83 Ztring celt_version;
84 int32u Celt_version_id, sample_rate, nb_channels;
85 Skip_Local(8, "celt_codec_id");
86 Get_Local(20, celt_version, "celt_version");
87 Get_L4 (Celt_version_id, "celt_version_id");
88 Skip_L4( "header_size");
89 Get_L4 (sample_rate, "rate");
90 Get_L4 (nb_channels, "nb_channels");
91 Skip_L4( "frame_size");
92 Skip_L4( "overlap");
93 Skip_L4( "bytes_per_packet");
94 Skip_L4( "extra_headers");
95
96 //Filling
97 FILLING_BEGIN();
98 Accept("CELT");
99
100 Stream_Prepare(Stream_Audio);
101 Fill(Stream_Audio, 0, Audio_Format, "CELT");
102 Fill(Stream_Audio, 0, Audio_Codec, "CELT");
103
104 if (!celt_version.empty())
105 {
106 //Fill(Stream_Audio, 0, Audio_Encoded_Library, celt_version); //Need more info about hte different possibilities, in the meanwhile trusting more the comment part
107 Fill(Stream_Audio, 0, Audio_SamplingRate, sample_rate);
108 Fill(Stream_Audio, 0, Audio_Channel_s_, nb_channels);
109 }
110
111 FILLING_END();
112
113 //Filling
114 Identification_Done=true;
115}
116
117//---------------------------------------------------------------------------
118void File_Celt::Comment()
119{
120 Element_Name("Comment?");
121
122 while (Element_Offset<Element_Size)
123 {
124 Ztring value;
125 int32u size;
126 Get_L4(size, "size");
127 if (size)
128 Get_Local(size, value, "value");
129
130 //Filling
131 if (value.find(__T("CELT "))!=std::string::npos)
132 {
133 Ztring Version=value.SubString(__T("CELT "), __T(" ("));
134 Fill(Stream_Audio, 0, Audio_Encoded_Library, __T("CELT ")+Version);
135 Fill(Stream_Audio, 0, Audio_Encoded_Library_Name, __T("CELT"));
136 Fill(Stream_Audio, 0, Audio_Encoded_Library_Version, Version);
137 }
138 else if (!value.empty())
139 Fill(Stream_Audio, 0, "Comment", value);
140 }
141
142 Finish("CELT");
143}
