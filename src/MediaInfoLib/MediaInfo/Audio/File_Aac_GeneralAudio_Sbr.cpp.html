41//---------------------------------------------------------------------------
42// Master frequency band table
43// k0 = lower frequency boundary
44int8u Aac_k0_Compute(int8u bs_start_freq, int8u extension_sampling_frequency_index)
45{
46 return Aac_k0_startMin[extension_sampling_frequency_index]+Aac_k0_offset[extension_sampling_frequency_index][bs_start_freq];
47}
48
49//---------------------------------------------------------------------------
50// Master frequency band table
51// k2 = upper frequency boundary
52int8u Aac_k2_Compute(int8u bs_stop_freq, int8u extension_sampling_frequency_index, int8u k0)
53{
54 switch (bs_stop_freq)
55 {
56 case 14 : return (int8u)min(64, 2*k0);
57 case 15 : return (int8u)min(64, 3*k0);
58 default : ;
59 }
60
61 return (int8u)min(64, Aac_k2_stopMin[extension_sampling_frequency_index]+Aac_k2_offset[extension_sampling_frequency_index][bs_stop_freq]);
62}

64//---------------------------------------------------------------------------
65//Helper
66int8u Aac_bands_Compute(bool warp, int8u bands, int8u a0, int8u a1)
67{
68 float div=(float)log(2.0);
69 if (warp)
70 div*=(float)1.3;
71
72 return (int8u)(bands*log((float)a1/(float)a0)/div+0.5);
73}

75//---------------------------------------------------------------------------
76// Master frequency band table
77// Computing for bs_freq_scale = 0
78bool Aac_f_master_Compute_0(int8u &num_env_bands_Master, int8u* f_Master, sbr_handler *sbr, int8u k0, int8u k2)
79{
80 int8u dk, numBands;
81 if (sbr->bs_alter_scale)
82 {
83 dk=1;
84 numBands=(((k2-k0+2)>>2)<<1);
85 }
86 else
87 {
88 dk=2;
89 numBands=(((k2-k0)>>1)<<1);
90 }
91
92 int8u k2Achieved=k0+numBands*dk;
93 int8s k2Diff=k2-k2Achieved;
94 int8s vDk[64] = { 0 };
95 for (int8u k=0; k<numBands; k++)
96 vDk[k]=dk;
97
98 if (k2Diff)
99 {
100 int8s incr;
101 int8u k;
102 if (k2Diff>0)
103 {
104 incr=-1;
105 k=numBands-1;
106 }
107 else
108 {
109 incr=1;
110 k=0;
111 }
112
113 while (k2Diff)
114 {
115 if (k >= 64)
116 break;
117 vDk[k]-=incr;
118 k+=incr;
119 k2Diff+=incr;
120 }
121 }
122
123 f_Master[0]=k0;
124 for (int8u k=1; k<=numBands; k++)
125 f_Master[k]=f_Master[k-1]+vDk[k-1];
126
127 num_env_bands_Master=numBands;
128
129 return true;
130}

132//---------------------------------------------------------------------------
133// Master frequency band table
134// Computing for bs_freq_scale != 0
135int int8u_cmp(const void *a, const void *b)
136{
137 return ((int8u)(*(int8u*)a - *(int8u*)b));
138}
139bool Aac_f_master_Compute(int8u &num_env_bands_Master, int8u* f_Master, sbr_handler *sbr, int8u k0, int8u k2)
140{
141 int8u temp1[]={6, 5, 4 };
142 int8u bands=temp1[sbr->bs_freq_scale-1];
143
144 int8u twoRegions, k1;
145 if ((float)k2/(float)k0>2.2449)
146 {
147 twoRegions=1;
148 k1=2*k0;
149 }
150 else
151 {
152 twoRegions=0;
153 k1=k2;
154 }
155
156 int8u numBands0=2*Aac_bands_Compute(false, bands, k0, k1);
157 if (numBands0 == 0 || numBands0 >= 64)
158 return false;
159
160 int8u vDk0[64];
161 int8u vk0[64];
162 float Power=pow((float)k1/(float)k0, (float)1/(float)numBands0);
163 float Power2=(float)k0;
164 int8s Temp1=(int8s)(Power2+0.5);
165 for (int8u k=0; k<=numBands0-1; k++)
166 {
167 int8s Temp0=Temp1;
168 Power2*=Power;
169 Temp1=(int8s)(Power2+0.5);
170 vDk0[k]=Temp1-Temp0;
171 }
172 qsort(vDk0, numBands0, sizeof(int8u), int8u_cmp);
173 vk0[0]=k0;
174 for (int8u k=1; k<=numBands0; k++)
175 {
176 if (vDk0[k-1]==0)
177 return false;
178 vk0[k]=vk0[k-1]+vDk0[k-1];
179 }
180
181 if (!twoRegions)
182 {
183 for (int8u k=0; k<=numBands0; k++)
184 f_Master[k]=vk0[k];
185 num_env_bands_Master=numBands0;
186 return true;
187 }
188
189 //With twoRegions
190 int8u numBands1;
191 int8u vDk1[64] = { 0 };
192 int8u vk1[64];
193 numBands1=2*Aac_bands_Compute(true, bands, k1, k2);
194 if (numBands1 == 0 || numBands0 + numBands1 >= 64)
195 return false;
196
197 Power=(float)pow((float)k2/(float)k1, (float)1/(float)numBands1);
198 Power2=(float)k1;
199 Temp1=(int8s)(Power2+0.5);
200 for (int8u k=0; k<=numBands1-1; k++)
201 {
202 int8s Temp0=Temp1;
203 Power2*=Power;
204 Temp1=(int8s)(Power2+0.5);
205 vDk1[k]=Temp1-Temp0;
206 }
207
208 if (vDk1[0]<vDk0[numBands0-1])
209 {
210 qsort(vDk1, numBands1+1, sizeof(int8u), int8u_cmp);
211 int8u change=vDk0[numBands0-1]-vDk1[0];
212 vDk1[0]=vDk0[numBands0-1];
213 vDk1[numBands1 - 1] = vDk1[numBands1 - 1] - change;
214 }
215
216 qsort(vDk1, numBands1, sizeof(int8u), int8u_cmp);
217 vk1[0]=k1;
218 for (int8u k=1; k<=numBands1; k++)
219 {
220 if (vDk1[k-1]==0)
221 return false;
222 vk1[k]=vk1[k-1]+vDk1[k-1];
223 }
224
225 num_env_bands_Master=numBands0+numBands1;
226 for (int8u k=0; k<=numBands0; k++)
227 f_Master[k]=vk0[k];
228 for (int8u k=numBands0+1; k <=num_env_bands_Master; k++)
229 f_Master[k]=vk1[k-numBands0];
230
231 return true;
232}

234//---------------------------------------------------------------------------
235// Derived frequency border tables
236bool Aac_bands_Compute(int8u &num_env_bands_Master, int8u* f_Master, sbr_handler *sbr, int8u k2)
237{
238 sbr->num_env_bands[1]=num_env_bands_Master-sbr->bs_xover_band;
239 sbr->num_env_bands[0]=(sbr->num_env_bands[1]>>1)+(sbr->num_env_bands[1]-((sbr->num_env_bands[1]>>1)<<1));
240
241 if (f_Master[sbr->bs_xover_band]>32)
242 return false;
243
244 if (sbr->bs_noise_bands==0)
245 sbr->num_noise_bands=1;
246 else
247 {
248 sbr->num_noise_bands=Aac_bands_Compute(false, sbr->bs_noise_bands, f_Master[sbr->bs_xover_band], k2);
249 if (sbr->num_noise_bands<1 || sbr->num_noise_bands>5)
250 return false;
251 }
252
253 return true;
254}

256//---------------------------------------------------------------------------
257bool Aac_Sbr_Compute(sbr_handler *sbr, int8u extension_sampling_frequency_index)
258{
259 if (extension_sampling_frequency_index>=9)
260 return 0; //Not supported
261 int8u k0=Aac_k0_Compute(sbr->bs_start_freq, extension_sampling_frequency_index);
262 int8u k2=Aac_k2_Compute(sbr->bs_stop_freq, extension_sampling_frequency_index, k0);
263 if (k2<=k0) return false;
264 switch (extension_sampling_frequency_index)
265 {
266 case 0 :
267 case 1 :
268 case 2 :
269 case 3 : if ((k2-k0)>32) return false; break;
270 case 4 : if ((k2-k0)>35) return false; break;
271 case 5 :
272 case 6 :
273 case 7 :
274 case 8 : if ((k2-k0)>48) return false; break;
275 default : ;
276 }
277
278 int8u num_env_bands_Master;
279 int8u f_Master[64];
280 if (sbr->bs_freq_scale==0)
281 {
282 if (!Aac_f_master_Compute_0(num_env_bands_Master, f_Master, sbr, k0, k2))
283 return false;
284 }
285 else
286 {
287 if (!Aac_f_master_Compute(num_env_bands_Master, f_Master, sbr, k0, k2))
288 return false;
289 }
290 if (num_env_bands_Master<=sbr->bs_xover_band)
291 return false;
292 if (!Aac_bands_Compute(num_env_bands_Master, f_Master, sbr, k2))
293 return false;
294
295 return true;
296}

299int16u File_Aac::sbr_huff_dec(sbr_huffman Table, const char* Name)
300{
301 int8u bit;
302 int16s index = 0;
303
304 Element_Begin1(Name);
305 while (index>=0)
306 {
307 Get_S1(1, bit, "bit");
308 index=Table[index][bit];
309 }
310 Element_End0();
311
312 return index+64;
313}

315//---------------------------------------------------------------------------
316void File_Aac::sbr_extension_data(size_t End, int8u id_aac, bool crc_flag)
317{
318 FILLING_BEGIN();
319 if (Infos["Format_Settings_SBR"].empty())
320 {
321 Infos["Format_Profile"]=__T("HE-AAC");
322 Ztring SamplingRate=Infos["SamplingRate"];
323 if (SamplingRate.empty())
324 SamplingRate.From_Number(Frequency_b);
325 Infos["SamplingRate"].From_Number((extension_sampling_frequency_index==(int8u)-1)?(Frequency_b*2):extension_sampling_frequency, 10);
326 if (MediaInfoLib::Config.LegacyStreamDisplay_Get())
327 {
328 Infos["Format_Profile"]+=__T(" / LC");
329 Infos["SamplingRate"]+=__T(" / ")+SamplingRate;
330 }
331 Infos["Format_Settings_SBR"]=__T("Yes (Implicit)");
332 Infos["Codec"]=Ztring().From_Local(Aac_audioObjectType(audioObjectType))+__T("-SBR");
333
334 if (Frame_Count_Valid<32)
335 Frame_Count_Valid=32; //We need to find the SBR header
336 }
337 FILLING_END();
338
339 Element_Begin1("sbr_extension_data");
340 bool bs_header_flag;
341 if (crc_flag)
342 Skip_S2(10, "bs_sbr_crc_bits");
343 //~ if (sbr_layer != SBR_STEREO_ENHANCE)
344 //~ {
345 Get_SB(bs_header_flag, "bs_header_flag");
346 if (bs_header_flag)
347 {
348 if (extension_sampling_frequency_index==(int8u)-1)
349 {
350 extension_sampling_frequency=(int32u)(Frequency_b*2);
351 extension_sampling_frequency_index=Aac_AudioSpecificConfig_sampling_frequency_index(extension_sampling_frequency);
352 }
353
354 delete sbr;
355 sbr=new sbr_handler;
356
357 sbr_header();
358
359 if (!Aac_Sbr_Compute(sbr, extension_sampling_frequency_index))
360 {
361 delete sbr; sbr=NULL;
362 }
363 }
364 //~ }
365
366 //Parsing
367 if (sbr) //only if a header is found
368 {
369 sbr->bs_amp_res[0]=sbr->bs_amp_res_FromHeader; //Set up with header data
370 sbr->bs_amp_res[1]=sbr->bs_amp_res_FromHeader; //Set up with header data
371 sbr_data(id_aac);
372
373 FILLING_BEGIN();
374 if (MediaInfoLib::Config.ParseSpeed_Get()<0.3)
375 {
376 Frame_Count_Valid=Frame_Count+1;
377 if (Frame_Count<8)
378 Frame_Count_Valid+=8-Frame_Count;
379 }
380 FILLING_END();
381 }
382 if (Data_BS_Remain()>End)
383 Skip_BS(Data_BS_Remain()-End, "bs_fill_bits");
384 Element_End0();
385}

387//---------------------------------------------------------------------------
388void File_Aac::sbr_header()
389{
390 Element_Begin1("sbr_header");
391 Get_S1 (1, sbr->bs_amp_res_FromHeader, "bs_amp_res");
392 Get_S1 (4, sbr->bs_start_freq, "bs_start_freq");
393 Get_S1 (4, sbr->bs_stop_freq, "bs_stop_freq");
394 Get_S1 (3, sbr->bs_xover_band, "bs_xover_band");
395 Skip_S1(2, "bs_reserved");
396 bool bs_header_extra_1, bs_header_extra_2;
397 Get_SB (bs_header_extra_1, "bs_header_extra_1");
398 Get_SB (bs_header_extra_2, "bs_header_extra_2");
399 if(bs_header_extra_1)
400 {
401 Get_S1 (2, sbr->bs_freq_scale, "bs_freq_scale");
402 Get_S1 (1, sbr->bs_alter_scale, "bs_alter_scale");
403 Get_S1 (2, sbr->bs_noise_bands, "bs_noise_bands");
404 }
405 else
406 {
407 sbr->bs_freq_scale=2;
408 sbr->bs_alter_scale=1;
409 sbr->bs_noise_bands=2;
410 }
411 if(bs_header_extra_2)
412 {
413 Skip_S1(2, "bs_limiter_bands");
414 Skip_S1(2, "bs_limiter_gains");
415 Skip_SB( "bs_interpol_freq");
416 Skip_SB( "bs_smoothing_mode");
417 }
418
419 Element_End0();
420}

435//---------------------------------------------------------------------------
436void File_Aac::sbr_single_channel_element()
437{
438 Element_Begin1("sbr_single_channel_element");
439 bool bs_data_extra, bs_add_harmonic_flag, bs_extended_data;
440 int8u bs_extension_size, bs_esc_count, bs_extension_id;
441 Get_SB (bs_data_extra, "bs_data_extra");
442 if (bs_data_extra)
443 {
444 Skip_S1(4, "bs_reserved");
445 }
446 sbr_grid(0);
447 sbr_dtdf(0);
448 sbr_invf(0);
449 sbr_envelope(0, 0);
450 sbr_noise(0, 0);
451 Get_SB (bs_add_harmonic_flag, "bs_add_harmonic_flag[0]");
452 if (bs_add_harmonic_flag)
453 sbr_sinusoidal_coding(0);
454
455 Get_SB (bs_extended_data, "bs_extended_data[0]");
456 if (bs_extended_data) {
457 Get_S1 (4,bs_extension_size, "bs_extension_size");
458 size_t cnt=bs_extension_size;
459 if (cnt==15)
460 {
461 Get_S1 (8, bs_esc_count, "bs_esc_count");
462 cnt+=bs_esc_count;
463 }
464
465 if (Data_BS_Remain()>=8*cnt)
466 {
467 size_t End=Data_BS_Remain()-8*cnt;
468 while (Data_BS_Remain()>End+7)
469 {
470 Get_S1 (2,bs_extension_id, "bs_extension_id");
471 switch (bs_extension_id)
472 {
473 case 2 : ps_data(End); break; //EXTENSION_ID_PS
474 default: ;
475 }
476 }
477 if (End<Data_BS_Remain())
478 Skip_BS(Data_BS_Remain()-End, "bs_fill_bits");
479 }
480 else
481 Skip_BS(Data_BS_Remain(), "(Error)");
482
483 }
484 Element_End0();
485}

487//---------------------------------------------------------------------------
488void File_Aac::sbr_grid(bool ch)
489{
490 Element_Begin1("sbr_grid");
491 int8u bs_frame_class, bs_num_rel_0, bs_num_rel_1, tmp;
492 int ptr_bits;
493 Get_S1(2, bs_frame_class, "bs_frame_class");
494 switch (bs_frame_class)
495 {
496 case 0 : //FIXFIX
497 Get_S1 (2, tmp, "tmp");
498 sbr->bs_num_env[ch]=(int8u)pow(2.0, tmp);
499 if (sbr->bs_num_env[ch]==1)
500 sbr->bs_amp_res[ch]=0;
501 Get_SB ( sbr->bs_freq_res[ch][0], "bs_freq_res[ch][0]");
502 for (int8u env=1; env<sbr->bs_num_env[ch]; env++)
503 sbr->bs_freq_res[ch][env]=sbr->bs_freq_res[ch][0];
504 break;
505 case 1 : //FIXVAR
506 Skip_S1(2, "bs_var_bord_1[ch]");
507 Get_S1 (2, bs_num_rel_1, "bs_num_rel_1[ch]");
508 sbr->bs_num_env[ch]=bs_num_rel_1+1;
509 for (int8u rel=0; rel<sbr->bs_num_env[ch]-1; rel++) {
510 Skip_S1(2, "tmp");
511 }
512 ptr_bits=(int8u)ceil(log((double)sbr->bs_num_env[ch]+1)/log((double)2));
513 Skip_BS(ptr_bits, "bs_pointer[ch]");
514 Element_Begin1("bs_freq_res[ch]");
515 for (int8u env=0; env<sbr->bs_num_env[ch]; env++)
516 Get_SB (sbr->bs_freq_res[ch][sbr->bs_num_env[ch]-1-env], "bs_freq_res[ch][bs_num_env[ch]-1-env]");
517 Element_End0();
518 break;
519 case 2 : //VARFIX
520 Skip_S1(2, "bs_var_bord_0[ch]");
521 Get_S1 (2,bs_num_rel_0, "bs_num_rel_0[ch]");
522 sbr->bs_num_env[ch] = bs_num_rel_0 + 1;
523 for (int8u rel=0; rel<sbr->bs_num_env[ch]-1; rel++)
524 Skip_S1(2, "tmp");
525 ptr_bits=(int8u)ceil(log((double)sbr->bs_num_env[ch]+1)/log((double)2));
526 Skip_BS(ptr_bits, "bs_pointer[ch]");
527 Element_Begin1("bs_freq_res[ch]");
528 for (int8u env = 0; env < sbr->bs_num_env[ch]; env++)
529 Get_SB (sbr->bs_freq_res[ch][env], "bs_freq_res[ch][env]");
530 Element_End0();
531 break;
532 case 3 : //VARVAR
533 Skip_S1(2, "bs_var_bord_0[ch]");
534 Skip_S1(2, "bs_var_bord_1[ch]");
535 Get_S1 (2,bs_num_rel_0, "bs_num_rel_0[ch]");
536 Get_S1 (2,bs_num_rel_1, "bs_num_rel_1[ch]");
537 sbr->bs_num_env[ch] = bs_num_rel_0 + bs_num_rel_1 + 1;
538 for (int8u rel=0; rel<bs_num_rel_0; rel++)
539 Skip_S1(2, "tmp");
540 for (int8u rel=0; rel<bs_num_rel_1; rel++)
541 Skip_S1(2, "tmp");
542 ptr_bits=(int8u)ceil(log((double)(sbr->bs_num_env[ch]+1))/log((double)2));
543 Skip_BS(ptr_bits, "bs_pointer[ch]");
544 Element_Begin1("bs_freq_res[ch]");
545 for (int8u env=0; env<sbr->bs_num_env[ch]; env++)
546 Get_SB (sbr->bs_freq_res[ch][env], "bs_freq_res[ch][env]");
547 Element_End0();
548 break;
549 }
550 if (sbr->bs_num_env[ch]>1)
551 sbr->bs_num_noise[ch]=2;
552 else
553 sbr->bs_num_noise[ch]=1;
554 Element_End0();
555}

557//---------------------------------------------------------------------------
558void File_Aac::sbr_channel_pair_element()
559{
560 Element_Begin1("sbr_channel_pair_element");
561 bool bs_data_extra,bs_coupling,bs_add_harmonic_flag,bs_extended_data;
562 int8u bs_extension_size,bs_esc_count,bs_extension_id;
563 Get_SB (bs_data_extra, "bs_data_extra");
564 if (bs_data_extra) {
565 Skip_S1(4, "bs_reserved");
566 Skip_S1(4, "bs_reserved");
567 }
568
569 Get_SB (bs_coupling, "bs_coupling");
570 sbr_grid(0);
571 if (bs_coupling)
572 {
573 //Coupling
574 sbr->bs_num_env [1]=sbr->bs_num_env [0];
575 sbr->bs_num_noise [1]=sbr->bs_num_noise [0];
576 for (int8u env=0; env<sbr->bs_num_env[0]; env++)
577 sbr->bs_freq_res[1][env]=sbr->bs_freq_res[0][env];
578 }
579 else
580 sbr_grid(1);
581 sbr_dtdf(0);
582 sbr_dtdf(1);
583 sbr_invf(0);
584 if (!bs_coupling)
585 sbr_invf(1);
586 sbr_envelope(0, bs_coupling);
587 if (bs_coupling)
588 {
589 sbr_noise(0, bs_coupling);
590 sbr_envelope(1, bs_coupling);
591 }
592 else
593 {
594 sbr_envelope(1, bs_coupling);
595 sbr_noise(0, bs_coupling);
596 }
597 sbr_noise(1, bs_coupling);
598 Get_SB (bs_add_harmonic_flag, "bs_add_harmonic_flag[0]");
599 if (bs_add_harmonic_flag)
600 sbr_sinusoidal_coding(0);
601 Get_SB (bs_add_harmonic_flag, "bs_add_harmonic_flag[1]");
602 if (bs_add_harmonic_flag)
603 sbr_sinusoidal_coding(1);
604
605 Get_SB (bs_extended_data, "bs_extended_data");
606 if (bs_extended_data) {
607 Get_S1(4,bs_extension_size, "bs_extension_size");
608 size_t cnt = bs_extension_size;
609 if (cnt == 15)
610 {
611 Get_S1(8,bs_esc_count, "bs_esc_count");
612 cnt += bs_esc_count;
613 }
614 if (Data_BS_Remain()>=8*cnt)
615 {
616 size_t End=Data_BS_Remain()-8*cnt;
617 while (Data_BS_Remain()>End+7)
618 {
619 Get_S1 (2,bs_extension_id, "bs_extension_id");
620 switch (bs_extension_id)
621 {
622 case 2 : ps_data(End); break; //EXTENSION_ID_PS
623 default: ;
624 }
625 }
626 if (End<Data_BS_Remain())
627 Skip_BS(Data_BS_Remain()-End, "bs_fill_bits");
628 }
629 else
630 Skip_BS(Data_BS_Remain(), "(Error)");
631
632 }
633 Element_End0();
634}

636//---------------------------------------------------------------------------
637void File_Aac::sbr_dtdf(bool ch)
638{
639 Element_Begin1("sbr_dtdf");
640 for (int env=0; env<sbr->bs_num_env[ch]; env++)
641 Get_S1 (1, sbr->bs_df_env[ch][env], "bs_df_env[ch][env]");

658{
659 sbr_huffman t_huff, f_huff;
660 Element_Begin1("sbr_envelope");
661 if (bs_coupling && ch)
662 {
663 if (sbr->bs_amp_res[ch])
664 {
665 t_huff = t_huffman_env_bal_3_0dB;
666 f_huff = f_huffman_env_bal_3_0dB;
667 }
668 else
669 {
670 t_huff = t_huffman_env_bal_1_5dB;
671 f_huff = f_huffman_env_bal_1_5dB;
672 }
673 }
674 else
675 {
676 if (sbr->bs_amp_res[ch])
677 {
678 t_huff = t_huffman_env_3_0dB;
679 f_huff = f_huffman_env_3_0dB;
680 }
681 else
682 {
683 t_huff = t_huffman_env_1_5dB;
684 f_huff = f_huffman_env_1_5dB;
685 }
686 }
687
688 for (int8u env=0; env<sbr->bs_num_env[ch]; env++)
689 {
690 if (sbr->bs_df_env[ch][env] == 0)
691 {
692 if (bs_coupling && ch)
693 Skip_S1(sbr->bs_amp_res[ch]?5:6, "bs_env_start_value_balance");
694 else
695 Skip_S1(sbr->bs_amp_res[ch]?6:7, "bs_env_start_value_level");
696 for (int8u band = 1; band < sbr->num_env_bands[sbr->bs_freq_res[ch][env]]; band++)
697 sbr_huff_dec(f_huff, "bs_data_env[ch][env][band]");
698 }
699 else
700 {
701 for (int8u band = 0; band < sbr->num_env_bands[sbr->bs_freq_res[ch][env]]; band++)
702 sbr_huff_dec(t_huff, "bs_data_env[ch][env][band]");
703 }
704 }
705 Element_End0();
706}
