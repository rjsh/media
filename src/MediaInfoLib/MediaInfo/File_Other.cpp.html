37void File_Other::Read_Buffer_Continue()
38{
39 //Integrity
40 if (Buffer_Size<16)
41 {
42 Element_WaitForMoreData();
43 return;
44 }
45
46 Ztring Format;
47 if (Buffer[0]==0xEA
48 && Buffer[1]==0x22
49 && Buffer[2]<=0x03)
50 {
51 Accept();
52
53 Stream_Prepare(Stream_Text);
54 Fill(Stream_Text, 0, Text_Format, "Cheetah");
55
56 Finish();
57 return;
58 }
59 if (Buffer[0]==0x4C
60 && Buffer[1]==0x61
61 && Buffer[2]==0x6D
62 && Buffer[3]==0x62
63 && Buffer[4]==0x64
64 && Buffer[5]==0x61)
65 {
66 Accept();
67
68 Stream_Prepare(Stream_Text);
69 Fill(Stream_Text, 0, Text_Format, "Lambda");
70
71 Finish();
72 return;
73 }
74 if (CC4(Buffer)==0xC5C6CBC3) {Format=__T("RISC OS Chunk data");}
75 else if (CC4(Buffer)==0x110000EF) {Format=__T("RISC OS AIF executable");}
76 else if (CC4(Buffer)==CC4("Draw")) {Format=__T("RISC OS Draw");}
77 else if (CC4(Buffer)==CC4("FONT")) {Format=__T("RISC OS Font");}
78 else if (CC8(Buffer)==CC8("Maestro\r")) {Format=__T("RISC OS music file");}
79 else if (CC4(Buffer)==CC4("FC14")) {Format=__T("Amiga Future Composer");}
80 else if (CC4(Buffer)==CC4("SMOD")) {Format=__T("Amiga Future Composer");}
81 else if (CC4(Buffer)==CC4("AON4")) {Format=__T("Amiga Art Of Noise");}
82 else if (CC8(Buffer+1)==CC8("MUGICIAN")) {Format=__T("Amiga Mugician");}
83 else if (Buffer_Size>=66 && CC8(Buffer+58)==CC8("SIDMON I")) {Format=__T("Amiga Sidmon");}
84 else if (CC8(Buffer)==CC8("Synth4.0")) {Format=__T("Amiga Synthesis");}
85 else if (CC4(Buffer)==CC4("ARP.")) {Format=__T("Amiga Holy Noise");}
86 else if (CC4(Buffer)==CC4("BeEp")) {Format=__T("Amiga JamCracker");}
87 else if (CC4(Buffer)==CC4("COSO")) {Format=__T("Amiga Hippel-COSO");}
88 else if (CC3(Buffer)==CC3("LSX")) {Format=__T("Amiga LZX");}
89 else if (CC4(Buffer)==CC4("MOVI")) {Format=__T("Silicon Graphics movie");}
90 else if (CC4(Buffer+10)==CC4("Vivo")) {Format=__T("Vivo");}
91 else if (CC4(Buffer+1)==CC4("VRML")) {Format=__T("VRML");}
92 else if (CC5(Buffer)==CC5("HVQM4")) {Format=__T("GameCube Movie");}
93 else if (CC8(Buffer)==CC8("KW-DIRAC"))
94 {
95 Accept("Dirac");
96
97 Stream_Prepare(Stream_Video);
98 Fill(Stream_Video, 0, Video_Format, "Dirac");
99
100 Finish("Dirac");
101 return;
102 }
103 else if (CC5(Buffer)==CC5("ustar")) {Format=__T("Tar archive");}
104 //TODO: all archive magic numbers
105 else if (CC4(Buffer+1)==CC4("MSCB")) {Format=__T("MS Cabinet");}
106 else if (CC4(Buffer)==CC4(".snd")) {Format=__T("SUN Audio");}
107 else if (CC4(Buffer)==0x2E736400) {Format=__T("DEC Audio");}
108 else if (CC4(Buffer)==CC4("MThd")) {Format=__T("MIDI");}
109 else if (CC4(Buffer)==CC4("CTMF")) {Format=__T("CMF");}
110 else if (CC3(Buffer)==CC3("SBI")) {Format=__T("SoundBlaster");}
111 else if (CC4(Buffer)==CC4("EMOD")) {Format=__T("Ext. MOD");}
112 //TODO: Other Sound magic numbers
113 else if (CC7(Buffer)==CC7("BLENDER")) {Format=__T("Blender");}
114 else if (CC4(Buffer)==CC4("AC10")) {Format=__T("AutoCAD"); ;}
115 else if (CC2(Buffer)==0x1F9D) {Format=__T("Compress");}
116 else if (CC2(Buffer)==0x1F8B) {Format=__T("GZip");}
117 else if (CC2(Buffer)==0x1F1E) {Format=__T("Huffman");}
118 else if (CC3(Buffer)==CC3("BZh")) {Format=__T("BZip2");}
119 else if (CC2(Buffer)==CC2("BZ")) {Format=__T("BZip1");}
120 else if (CC3(Buffer)==CC3("NES")) {Format=__T("NES ROM");}
121 else if (Buffer_Size>=0x108 && CC4(Buffer+0x104)==0xCEED6666) {Format=__T("GameBoy");}
122 else if (Buffer_Size>=0x104 && CC4(Buffer+0x100)==CC4("SEGA")) {Format=__T("MegaDrive");}
123 else if (Buffer_Size>=0x284 && CC4(Buffer+0x280)==CC4("EAGN")) {Format=__T("SupeMegaDrive");}
124 else if (Buffer_Size>=0x284 && CC4(Buffer+0x280)==CC4("EAMG")) {Format=__T("SupeMegaDrive");}
125 else if (CC4(Buffer)==0x21068028) {Format=__T("Dreamcast");}
126 else if (CC4(Buffer)==CC4("LCDi")) {Format=__T("Dreamcast");}
127 else if (CC4(Buffer)==0x37804012) {Format=__T("Nintendo64");}
128 else if (CC8(Buffer)==CC8("PS-X EXE")) {Format=__T("Playstation");}
129 else if (CC4(Buffer)==CC4("LCDi")) {Format=__T("Dreamcast");}
130 else if (CC4(Buffer)==CC4("XBEH")) {Format=__T("X-Box");}
131 else if (CC4(Buffer)==CC4("XIP0")) {Format=__T("X-Box");}
132 else if (CC4(Buffer)==CC4("XTF0")) {Format=__T("X-Box");}
133 else if (CC2(Buffer)==0x8008) {Format=__T("Lynx");}
134 else if (CC7(Buffer)==CC7("\x01ZZZZZ\x01")) {Format=__T("");}
135 else if (CC6(Buffer)==CC6("1\x0A\x0D""00:"))
136 {
137 Accept("SRT");
138
139 Stream_Prepare(Stream_Text);
140 Fill(Stream_Text, 0, Text_Format, "SRT");
141
142 Finish("SRT");
143 return;
144 }
145 else if ((CC1(Buffer+0)==CC1("[") && CC1(Buffer+2)==CC1("S") && CC1(Buffer+22)==CC1("o") && CC1(Buffer+24)==CC1("]")) //Unicode Text is : "[Script Info]
146 || (CC1(Buffer+2)==CC1("[") && CC1(Buffer+4)==CC1("S") && CC1(Buffer+24)==CC1("o") && CC1(Buffer+26)==CC1("]")))
147 {
148 Accept("SSA");
149
150 Stream_Prepare(Stream_Text);
151 Fill(Stream_Text, 0, Text_Format, "SSA");
152
153 Finish("SSA");
154 return;
155 }
156 else if (CC4(Buffer)==CC4("RIFF") && CC4(Buffer+8)==CC4("AMV ")) {Format=__T("AMV");}
157 else if (CC4(Buffer)==CC4("RIFF") && CC4(Buffer+8)==CC4("WEBP"))
158 {
159 Accept("WEBP");
160
161 Stream_Prepare(Stream_Image);
162 Fill(Stream_Image, 0, Image_Format, "WebP");
163
164 Finish("WEBP");
165 return;
166 }
167 else if (CC4(Buffer)==0x414D5697) {Format=__T("MTV");}
168 else if (CC6(Buffer)==CC6("Z\0W\0F\0"))
169 {
170 Accept("ZWF");
171
172 Stream_Prepare(Stream_Audio);
173 Fill(Stream_Audio, 0, Audio_Format, "ZWF");
174
175 Finish("ZWF");
176 return;
177 }
178 else if (CC4(Buffer)==0x616A6B67) //"ajkg"
179 {
180 Accept("Shorten");
181
182 Fill(Stream_General, 0, General_Format_Version, CC1(Buffer+4));
183
184 Stream_Prepare(Stream_Audio);
185 Fill(Stream_Audio, 0, Audio_Format, "Shorten");
186
187 Finish("Shorten");
188 return;
189 }
190 else if (CC4(Buffer)==0x504C5646) {Format=__T("PlayLater Video");}
191 else if (CC4(Buffer)==CC4("")) {Format=__T("");}
192
193 if (Format.empty())
194 {
195 Reject();
196 return;
197 }
198
199 Accept();
200
201 Element_Offset=File_Size-(File_Offset+Buffer_Offset);
202 Fill(Stream_General, 0, General_Format, Format);
203
204 Finish();
205}
206
207} //NameSpace
