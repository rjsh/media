791//---------------------------------------------------------------------------
792void File__MultipleParsing::Read_Buffer_Continue()
793{
794 //Parsing
795 for (size_t Pos=0; Pos<Parser.size(); Pos++)
796 {
797 //Parsing
798 Parser[Pos]->Open_Buffer_Continue(Buffer+Buffer_Offset, (size_t)Element_Size);
799 if (File_Offset+Buffer_Size==File_Size)
800 Parser[Pos]->Open_Buffer_Finalize();
801
802 //Testing if the parser failed
803 if (Parser[Pos]->Status[IsFinished] && !Parser[Pos]->Status[IsAccepted])
804 {
805 delete Parser[Pos];
806 Parser.erase(Parser.begin()+Pos);
807 Pos--; //for the next position
808
809 if (Parser.empty())
810 {
811 File__Analyze* Temp=new File_Unknown(); Parser.push_back(Temp);
812 Read_Buffer_Init();
813 }
814 }
815 else
816 {
817 //If Parser is found, erasing all the other parsers
818 if (Parser.size()>1 && Parser[Pos]->Status[IsAccepted])
819 {
820 File__Analyze* Temp=Parser[Pos];
821 for (size_t To_Delete_Pos=0; To_Delete_Pos<Parser.size(); To_Delete_Pos++)
822 if (To_Delete_Pos!=Pos)
823 delete Parser[To_Delete_Pos]; //Parser[Pos]=NULL
824 Parser.clear();
825 Parser.push_back(Temp);
826 Pos=0;
827 }
828
829 if (Parser.size()==1)
830 {
831 //Status
832 if (!Status[IsAccepted] && Parser[Pos]->Status[IsAccepted])
833 Status[IsAccepted]=true;
834 if (!Status[IsFilled] && Parser[Pos]->Status[IsFilled])
835 Status[IsFilled]=true;
836 if (!Status[IsUpdated] && Parser[Pos]->Status[IsUpdated])
837 Status[IsUpdated]=true;
838 if (!Status[IsFinished] && Parser[Pos]->Status[IsFinished])
839 Status[IsFinished]=true;
840
841 //Positionning if requested
842 if (Parser[0]->File_GoTo!=(int64u)-1)
843 File_GoTo=Parser[0]->File_GoTo;
844 }
845 }
846 }
