48#if MEDIAINFO_READTHREAD
49void Reader_File_Thread::Entry()
50{
51 ReadSize_Max=Base->Buffer_Max>>3;
52
53 for (;;)
54 {
55 Base->CS.Enter();
56 if (Base->Buffer_Begin==Base->Buffer_Max)
57 {
58 Base->IsLooping=false;
59 Base->Buffer_End=Base->Buffer_End2;
60 Base->Buffer_End2=0;
61 Base->Buffer_Begin=0;
62 }
63
64 size_t ToRead;
65 size_t Buffer_ToReadOffset;
66 if (Base->IsLooping)
67 {
68 ToRead=Base->Buffer_Begin-Base->Buffer_End2;
69 Buffer_ToReadOffset=Base->Buffer_End2;
70 }
71 else
72 {
73 ToRead=Base->Buffer_Max-Base->Buffer_End;
74 Buffer_ToReadOffset=Base->Buffer_End;
75 }
76 Base->CS.Leave();
77
78 if (ToRead)
79 {
80 if (ToRead>ReadSize_Max)
81 ToRead=ReadSize_Max;
82 size_t BytesRead=Base->F.Read(Base->Buffer+Buffer_ToReadOffset, ToRead);
83 if (!BytesRead)
84 break;
85
86 Base->CS.Enter();
87 if (Base->IsLooping)
88 {
89 Base->Buffer_End2+=BytesRead;
90 }
91 else
92 {
93 Base->Buffer_End+=BytesRead;
94 if (Base->Buffer_End==Base->Buffer_Max)
95 {
96 Base->IsLooping=true;
97 }
98 }
99 Base->CS.Leave();
100
101 #ifdef WINDOWS
102 SetEvent(Base->Condition_WaitingForMoreData);
103 #endif //WINDOWS
104 }
105 #ifdef WINDOWS
106 else
107 WaitForSingleObject(Base->Condition_WaitingForMorePlace, INFINITE);
108 #endif //WINDOWS
109
110 if (IsTerminating())
111 break;
112 Yield();
113 }
114
115 #ifdef WINDOWS
116 SetEvent(Base->Condition_WaitingForMoreData); //Sending the last event in case the main threading is waiting for more data
117 #endif //WINDOWS
118}
119#endif //MEDIAINFO_READTHREAD

148size_t Reader_File::Format_Test(MediaInfo_Internal* MI, String File_Name)
149{
150 //std::cout<<Ztring(File_Name).To_Local().c_str()<<std::endl;
151 #if MEDIAINFO_EVENTS
152 {
153 string File_Name_Local=Ztring(File_Name).To_Local();
154 wstring File_Name_Unicode=Ztring(File_Name).To_Unicode();
155 struct MediaInfo_Event_General_Start_0 Event;
156 memset(&Event, 0xFF, sizeof(struct MediaInfo_Event_Generic));
157 Event.EventCode=MediaInfo_EventCode_Create(MediaInfo_Parser_None, MediaInfo_Event_General_Start, 0);
158 Event.EventSize=sizeof(struct MediaInfo_Event_General_Start_0);
159 Event.StreamIDs_Size=0;
160 Event.Stream_Size=File::Size_Get(File_Name);
161 Event.FileName=File_Name_Local.c_str();
162 Event.FileName_Unicode=File_Name_Unicode.c_str();
163 MI->Config.Event_Send(NULL, (const int8u*)&Event, sizeof(MediaInfo_Event_General_Start_0));
164 }
165 #endif //MEDIAINFO_EVENTS
166
167 //With Parser MultipleParsing
168 /*
169 MI->Open_Buffer_Init((int64u)-1, File_Name);
170 if (Format_Test_PerParser(MI, File_Name))
171 return 1;
172 return 0; //There is a problem
173 */
174
175 //Get the Extension
176 Ztring Extension=FileName::Extension_Get(File_Name);
177 Extension.MakeLowerCase();
178
179 //Search the theorical format from extension
180 InfoMap &FormatList=MediaInfoLib::Config.Format_Get();
181 InfoMap::iterator Format=FormatList.end();
182 if (!MI->Config.File_ForceParser_Get().empty())
183 Format=FormatList.find(MI->Config.File_ForceParser_Get());
184 if (Format==FormatList.end())
185 {
186 Format=FormatList.begin();
187 while (Format!=FormatList.end())
188 {
189 const Ztring &Extensions=FormatList.Get(Format->first, InfoFormat_Extensions);
190 if (Extensions.find(Extension)!=Error)
191 {
192 if(Extension.size()==Extensions.size())
193 break; //Only one extenion in the list
194 if(Extensions.find(Extension+__T(" "))!=Error
195 || Extensions.find(__T(" ")+Extension)!=Error)
196 break;
197 }
198 ++Format;
199 }
200 }
201 if (Format!=FormatList.end())
202 {
203 const Ztring &Parser=Format->second(InfoFormat_Parser);
204 if (MI->SelectFromExtension(Parser))
205 {
206 //Test the theorical format
207 if (Format_Test_PerParser(MI, File_Name)>0)
208 return 1;
209 }
210 }
211
212 size_t ToReturn=MI->ListFormats(File_Name);
213 return ToReturn;
214}

216//---------------------------------------------------------------------------
217size_t Reader_File::Format_Test_PerParser(MediaInfo_Internal* MI, const String &File_Name)
218{
219 //Init
220 MI_Internal=MI;
221 #if MEDIAINFO_READTHREAD
222 ThreadInstance=NULL;
223 Buffer_End2=0; //Is also used for counting bytes before activating the thread
224 #endif //MEDIAINFO_READTHREAD
225
226 //Opening the file
227 F.Open(File_Name);
228 if (!F.Opened_Get())
229 return 0;
230
231 //Info
232 Status=0;
233 MI->Config.File_Size=F.Size_Get();
234 MI->Config.File_Current_Offset=0;
235 MI->Config.File_Current_Size=MI->Config.File_Size;
236 MI->Config.File_Sizes.clear();
237 MI->Config.File_Sizes.push_back(MI->Config.File_Size);
238 if (MI->Config.File_Names.size()>1)
239 {
240 #if MEDIAINFO_ADVANCED
241 if (MI->Config.File_IgnoreSequenceFileSize_Get())
242 {
243 MI->Config.File_Size=(int64u)-1;
244 }
245 else
246 #endif //MEDIAINFO_ADVANCED
247 {
248 for (size_t Pos=1; Pos<MI->Config.File_Names.size(); Pos++)
249 {
250 int64u Size=File::Size_Get(MI->Config.File_Names[Pos]);
251 MI->Config.File_Sizes.push_back(Size);
252 MI->Config.File_Size+=Size;
253 }
254 }
255 }
256
257 //Partial file handling
258 Ztring Config_Partial_Begin=MI->Config.File_Partial_Begin_Get();
259 if (!Config_Partial_Begin.empty() && Config_Partial_Begin[0]>=__T('0') && Config_Partial_Begin[0]<=__T('9'))
260 {
261 if (Config_Partial_Begin.find(__T('%'))==Config_Partial_Begin.size()-1)
262 Partial_Begin=float64_int64s(MI->Config.File_Size*Config_Partial_Begin.To_float64()/100);
263 else
264 Partial_Begin=Config_Partial_Begin.To_int64u();
265 if (Partial_Begin)
266 F.GoTo(Partial_Begin);
267 }
268 else
269 Partial_Begin=0;
270 Ztring Config_Partial_End=MI->Config.File_Partial_End_Get();
271 if (!Config_Partial_End.empty() && Config_Partial_End[0]>=__T('0') && Config_Partial_End[0]<=__T('9'))
272 {
273 if (Config_Partial_End.find(__T('%'))==Config_Partial_End.size()-1)
274 Partial_End=float64_int64s(MI->Config.File_Size*Config_Partial_End.To_float64()/100);
275 else
276 Partial_End=Config_Partial_End.To_int64u();
277 }
278 else
279 Partial_End=(int64u)-1;
280 if (Partial_Begin>MI->Config.File_Size)
281 Partial_Begin=0; //Wrong value
282 if (Partial_Begin>Partial_End)
283 Partial_Begin=0; //Wrong value
284
285 //Parser
286 MI->Open_Buffer_Init((Partial_End<=MI->Config.File_Size?Partial_End:MI->Config.File_Size)-Partial_Begin, File_Name);
287
288 //Buffer
289 MI->Option(__T("File_Buffer_Size_Hint_Pointer"), Ztring::ToZtring((size_t)(&MI->Config.File_Buffer_Size_ToRead)));
290 MI->Config.File_Buffer_Repeat_IsSupported=true;
291
292 //Test the format with buffer
293 return Format_Test_PerParser_Continue(MI);
294}

296//---------------------------------------------------------------------------
297size_t Reader_File::Format_Test_PerParser_Continue (MediaInfo_Internal* MI)
298{
299 if (MI == NULL)
300 return 0;
301
302 bool StopAfterFilled=MI->Config.File_StopAfterFilled_Get();
303 bool ShouldContinue=true;
304 if (MI->Info)
305 Status=MI->Info->Status;
306
307 //Previous data
308 if (MI->Config.File_Buffer_Repeat)
309 {
310 MI->Config.File_Buffer_Repeat=false;
311 #if MEDIAINFO_DEMUX
312 MI->Config.Demux_EventWasSent=false;
313 #endif //MEDIAINFO_DEMUX
314
315 Status=MI->Open_Buffer_Continue(MI->Config.File_Buffer, MI->Config.File_Buffer_Size);
316
317 #if MEDIAINFO_READTHREAD
318 if (ThreadInstance && !MI->Config.File_Buffer_Repeat)
319 {
320 CS.Enter();
321 Buffer_Begin+=MI->Config.File_Buffer_Size;
322 #ifdef WINDOWS
323 if (Buffer_Begin==Buffer_Max)
324 {
325 CS.Leave();
326 SetEvent(Condition_WaitingForMorePlace);
327 }
328 else
329 #endif //WINDOWS
330 CS.Leave();
331 }
332 #endif //MEDIAINFO_READTHREAD
333
334 #if MEDIAINFO_DEMUX
335 //Demux
336 if (MI->Config.Demux_EventWasSent)
337 return 2; //Must return immediately
338 #endif //MEDIAINFO_DEMUX
339
340 //Threading
341 if (MI->IsTerminating())
342 return 1; //Termination is requested
343
344 if (Status[File__Analyze::IsFinished] || (StopAfterFilled && Status[File__Analyze::IsFilled]))
345 ShouldContinue=false;
346 }
347
348 #if MEDIAINFO_DEMUX
349 //PerPacket
350 if (ShouldContinue && MI->Config.Demux_EventWasSent)
351 {
352 MI->Config.Demux_EventWasSent=false;
353
354 Status=MI->Open_Buffer_Continue(NULL, 0);
355
356 //Demux
357 if (MI->Config.Demux_EventWasSent)
358 return 2; //Must return immediately
359
360 //Threading
361 if (MI->IsTerminating())
362 return 1; //Termination is requested
363
364 if (Status[File__Analyze::IsFinished] || (StopAfterFilled && Status[File__Analyze::IsFilled]))
365 ShouldContinue=false;
366 }
367 #endif //MEDIAINFO_DEMUX
368
369 if (ShouldContinue)
370 {
371 //Test the format with buffer
372 while (!(Status[File__Analyze::IsFinished] || (StopAfterFilled && Status[File__Analyze::IsFilled])))
373 {
374 //Seek (if needed)
375 if (MI->Open_Buffer_Continue_GoTo_Get()!=(int64u)-1)
376 {
377 #ifdef MEDIAINFO_DEBUG
378 std::cout<<std::hex<<Reader_File_Offset<<" - "<<Reader_File_Offset+Reader_File_BytesRead<<" : "<<std::dec<<Reader_File_BytesRead<<" bytes"<<std::endl;
379 Reader_File_Offset=MI->Open_Buffer_Continue_GoTo_Get();
380 Reader_File_BytesRead=0;
381 Reader_File_Count++;
382 #endif //MEDIAINFO_DEBUG
383
384 #if MEDIAINFO_READTHREAD
385 if (ThreadInstance)
386 {
387 ThreadInstance->RequestTerminate();
388 SetEvent(Condition_WaitingForMorePlace);
389 while (!ThreadInstance->IsExited())
390 Sleep(0);
391 #ifdef WINDOWS
392 CloseHandle(Condition_WaitingForMorePlace);
393 CloseHandle(Condition_WaitingForMoreData);
394 #endif //WINDOWS
395 delete ThreadInstance; ThreadInstance=NULL;
396
397 MI->Config.File_Buffer=NULL;
398 MI->Config.File_Buffer_Size=0;
399 MI->Config.File_Buffer_Size_Max=0;
400 Buffer_Max=0;
401 delete[] Buffer; Buffer=NULL;
402 Buffer_Begin=0;
403 Buffer_End=0;
404 Buffer_End2=0;
405 IsLooping=false;
406 }
407 if (Buffer_End2!=(size_t)-1)
408 Buffer_End2=0;
409 #endif //MEDIAINFO_READTHREAD
410
411 int64u GoTo=Partial_Begin+MI->Open_Buffer_Continue_GoTo_Get();
412 MI->Config.File_Current_Offset=0;
413 int64u Buffer_NoJump_Temp=Buffer_NoJump;
414 if (MI->Config.File_Names.size()>1)
415 {
416 size_t Pos;
417 #if MEDIAINFO_SEEK
418 if (MI->Config.File_GoTo_IsFrameOffset)
419 {
420 Pos=(size_t)MI->Open_Buffer_Continue_GoTo_Get(); //File_GoTo is the frame offset in that case
421 MI->Info->File_GoTo=(int64u)-1;
422 MI->Config.File_GoTo_IsFrameOffset=false;
423 GoTo=0;
424 }
425 else
426 #endif //MEDIAINFO_SEEK
427 {
428 for (Pos=0; Pos<MI->Config.File_Names.size(); Pos++)
429 {
430 if (Pos==MI->Config.File_Sizes.size())
431 MI->Config.File_Sizes.push_back(F.Size_Get());
432 else if (MI->Config.File_Sizes[Pos]==(int64u)-1)
433 MI->Config.File_Sizes[Pos]=F.Size_Get();
434
435 if (Pos>=MI->Config.File_Sizes.size() || MI->Config.File_Sizes[Pos]==(int64u)-1)
436 break;
437
438 if (GoTo<MI->Config.File_Sizes[Pos])
439 break;
440
441 GoTo-=MI->Config.File_Sizes[Pos];
442 MI->Config.File_Current_Offset+=MI->Config.File_Sizes[Pos];
443 }
444 if (Pos>=MI->Config.File_Sizes.size())
445 break;
446 }
447 if (Pos!=MI->Config.File_Names_Pos-1)
448 {
449 F.Close();
450 F.Open(MI->Config.File_Names[Pos]);
451 if (Pos>=MI->Config.File_Sizes.size())
452 {
453 MI->Config.File_Sizes.resize(Pos, (int64u)-1);
454 MI->Config.File_Sizes.push_back(F.Size_Get());
455 }
456 MI->Config.File_Names_Pos=Pos+1;
457 MI->Config.File_Current_Size=MI->Config.File_Current_Offset+F.Size_Get();
458 Buffer_NoJump_Temp=0;
459 }
460 }
461
462 if (GoTo>=F.Size_Get())
463 break; //Seek requested, but on a file bigger in theory than what is in the real file, we can't do this
464 if (!(GoTo>F.Position_Get() && GoTo<F.Position_Get()+Buffer_NoJump_Temp)) //No smal jumps
465 {
466 if (!F.GoTo(GoTo))
467 break; //File is not seekable
468
469 MI->Open_Buffer_Init((int64u)-1, MI->Config.File_Current_Offset+F.Position_Get()-Partial_Begin);
470 }
471 }
472
473 #if MEDIAINFO_READTHREAD
474 if (ThreadInstance==NULL && Buffer_End2!=(size_t)-1 && Buffer_End2>=16*1024*1024)
475 {
476 if (!MI->Config.File_IsGrowing && MI->Config.File_Names.size()==1)
477 {
478 delete[] MI->Config.File_Buffer; MI->Config.File_Buffer=NULL;
479 MI->Config.File_Buffer_Size_Max=0;
480 Buffer_Max=MI->Config.File_Buffer_Read_Size_Get();
481 Buffer=new int8u[Buffer_Max];
482 Buffer_Begin=0;
483 Buffer_End=0;
484 Buffer_End2=0;
485 IsLooping=false;
486 #ifdef WINDOWS
487 Condition_WaitingForMorePlace=CreateEvent(NULL, FALSE, FALSE, NULL);
488 Condition_WaitingForMoreData=CreateEvent(NULL, FALSE, FALSE, NULL);
489 #endif //WINDOWS
490 ThreadInstance=new Reader_File_Thread();
491 ThreadInstance->Base=this;
492 ThreadInstance->Run();
493 }
494 else
495 Buffer_End2=(size_t)-1;
496 }
497 #endif //MEDIAINFO_READTHREAD
498
499 //Handling of hints
500 if (MI->Config.File_Buffer_Size_ToRead==0)
501 break; //Problem while config
502 if (
503 #if MEDIAINFO_READTHREAD
504 ThreadInstance==NULL &&
505 #endif //MEDIAINFO_READTHREAD
506 MI->Config.File_Buffer_Size_ToRead>MI->Config.File_Buffer_Size_Max)
507 {
508 delete[] MI->Config.File_Buffer;
509 if (MI->Config.File_Buffer_Size_Max==0)
510 MI->Config.File_Buffer_Size_Max=1;
511 while (MI->Config.File_Buffer_Size_ToRead>MI->Config.File_Buffer_Size_Max)
512 MI->Config.File_Buffer_Size_Max*=2;
513 if (MI->Config.File_Buffer_Size_Max>=64*1024*1024)
514 MI->Config.File_Buffer_Size_Max=64*1024*1024; //limitation of the buffer in order to avoid to big memory usage
515 MI->Config.File_Buffer=new int8u[MI->Config.File_Buffer_Size_Max];
516 }
517
518 //Testing multiple file per stream
519 if (
520 #if MEDIAINFO_READTHREAD
521 ThreadInstance==NULL &&
522 #endif //MEDIAINFO_READTHREAD
523 F.Position_Get()>=F.Size_Get())
524 {
525 #if MEDIAINFO_ADVANCED2
526 MI->Open_Buffer_SegmentChange();
527 #endif //MEDIAINFO_ADVANCED2
528 if (MI->Config.File_Names_Pos && MI->Config.File_Names_Pos<MI->Config.File_Names.size())
529 {
530 MI->Config.File_Current_Offset+=MI->Config.File_Names_Pos<=MI->Config.File_Sizes.size()?MI->Config.File_Sizes[MI->Config.File_Names_Pos-1]:F.Size_Get();
531 F.Close();
532 #if MEDIAINFO_EVENTS
533 MI->Config.Event_SubFile_Start(MI->Config.File_Names[MI->Config.File_Names_Pos]);
534 #endif //MEDIAINFO_EVENTS
535 F.Open(MI->Config.File_Names[MI->Config.File_Names_Pos]);
536 while (!F.Opened_Get())
537 {
538 if (MI->Config.File_Names_Pos+1<MI->Config.File_Names.size())
539 {
540 MI->Config.File_Names_Pos++;
541 F.Open(MI->Config.File_Names[MI->Config.File_Names_Pos]);
542 }
543 }
544 if (MI->Config.File_Names_Pos>=MI->Config.File_Sizes.size())
545 {
546 MI->Config.File_Sizes.resize(MI->Config.File_Names_Pos, 0);
547 MI->Config.File_Sizes.push_back(F.Size_Get());
548 }
549 MI->Config.File_Names_Pos++;
550 MI->Config.File_Current_Size+=F.Size_Get();
551 }
552 }
553
554 #if MEDIAINFO_READTHREAD
555 if (ThreadInstance)
556 {
557 CS.Enter();
558 #ifdef WINDOWS
559 if (Buffer_End2+Buffer_End-Buffer_Begin<Buffer_Max/8*7)
560 {
561 CS.Leave();
562 SetEvent(Condition_WaitingForMorePlace);
563 CS.Enter();
564 }
565 #endif //WINDOWS
566
567 for (;;)
568 {
569 MI->Config.File_Buffer_Size=Buffer_End-Buffer_Begin;
570
571 if (MI->Config.File_Buffer_Size)
572 break;
573
574 if (!ThreadInstance->IsExited())
575 {
576 CS.Leave();
577 #ifdef WINDOWS
578 WaitForSingleObject(Condition_WaitingForMoreData, INFINITE);
579 #else //WINDOWS
580 Sleep(0);
581 #endif //WINDOWS
582 CS.Enter();
583 }
584 else
585 {
586 if (IsLooping)
587 {
588 IsLooping=false;
589 Buffer_End=Buffer_End2;
590 Buffer_End2=0;
591 Buffer_Begin=0;
592 }
593
594 MI->Config.File_Buffer_Size=Buffer_End-Buffer_Begin;
595 break;
596 }
597 }
598 MI->Config.File_Buffer=Buffer+Buffer_Begin;
599 CS.Leave();
600 if (MI->Config.File_Buffer_Size>MI->Config.File_Buffer_Size_ToRead)
601 MI->Config.File_Buffer_Size=MI->Config.File_Buffer_Size_ToRead;
602 }
603 else
604 #endif //MEDIAINFO_READTHREAD
605 {
606 MI->Config.File_Buffer_Size=F.Read(MI->Config.File_Buffer, (F.Position_Get()+MI->Config.File_Buffer_Size_ToRead<(Partial_End<=MI->Config.File_Size?Partial_End:MI->Config.File_Size))?MI->Config.File_Buffer_Size_ToRead:((size_t)((Partial_End<=MI->Config.File_Size?Partial_End:MI->Config.File_Size)-F.Position_Get())));
607 #if MEDIAINFO_READTHREAD
608 if (ThreadInstance==NULL && Buffer_End2!=(size_t)-1)
609 Buffer_End2+=MI->Config.File_Buffer_Size;
610 #endif //MEDIAINFO_READTHREAD
611 }
612
613 /* High CPU usage
614 #if MEDIAINFO_EVENTS
615 if (MI->Config.File_Buffer_Size)
616 {
617 struct MediaInfo_Event_Global_BytesRead_0 Event;
618 memset(&Event, 0xFF, sizeof(struct MediaInfo_Event_Generic));
619 Event.EventCode=MediaInfo_EventCode_Create(MediaInfo_Parser_None, MediaInfo_Event_Global_BytesRead, 0);
620 Event.EventSize=sizeof(struct MediaInfo_Event_Global_BytesRead_0);
621 Event.StreamIDs_Size=0;
622 Event.StreamOffset=F.Position_Get()-MI->Config.File_Buffer_Size;
623 Event.Content_Size=MI->Config.File_Buffer_Size;
624 Event.Content=MI->Config.File_Buffer;
625 MI->Config.Event_Send(NULL, (const int8u*)&Event, sizeof(MediaInfo_Event_Global_BytesRead_0));
626 }
627 #endif //MEDIAINFO_EVENTS
628 */
629
630 //Testing growing files
631 int64u Growing_Temp=(int64u)-1;
632 if (MI->Config.ParseSpeed>=1.0 && !MI->Config.File_IsGrowing && MI->Config.File_Current_Offset+F.Position_Get()>=MI->Config.File_Size)
633 {
634 if (MI->Config.File_Names.size()==1)
635 {
636 Growing_Temp=F.Size_Get();
637 if (MI->Config.File_Size!=Growing_Temp)
638 MI->Config.File_IsGrowing=true;
639 }
640 else if (MI->Config.File_TestContinuousFileNames_Get())
641 {
642 Growing_Temp=MI->Config.File_Names.size();
643 MI->TestContinuousFileNames();
644 if (MI->Config.File_Names.size()!=Growing_Temp)
645 MI->Config.File_IsGrowing=true;
646 }
647 }
648 if (MI->Config.File_IsNotGrowingAnymore)
649 {
650 MI->Config.File_Current_Size=MI->Config.File_Size=F.Size_Get();
651 MI->Open_Buffer_Init(MI->Config.File_Size, F.Position_Get()-MI->Config.File_Buffer_Size);
652 MI->Config.File_IsGrowing=false;
653 MI->Config.File_IsNotGrowingAnymore=false;
654 }
655 if (MI->Config.File_IsGrowing && (Growing_Temp!=(int64u)-1 || MI->Config.File_Current_Offset+F.Position_Get()>=MI->Config.File_Size))
656 {
657 for (size_t CountOfSeconds=0; CountOfSeconds<(size_t)MI->Config.File_GrowingFile_Delay_Get(); CountOfSeconds++)
658 {
659 int64u LastFile_Size_Old=MI->Config.File_Sizes[MI->Config.File_Sizes.size()-1];
660 size_t Files_Count_Old=MI->Config.File_Names.size();
661 MI->TestContinuousFileNames();
662 int64u LastFile_Size_New=F.Size_Get();
663 size_t Files_Count_New=MI->Config.File_Names.size();
664
665 if (LastFile_Size_New!=LastFile_Size_Old || Files_Count_New!=Files_Count_Old)
666 {
667 if (MI->Config.File_Names.size()==1) //if more than 1 file, file size config is already done in TestContinuousFileNames()
668 {
669 MI->Config.File_Current_Size=MI->Config.File_Size=LastFile_Size_New;
670 MI->Open_Buffer_Init(MI->Config.File_Size, MI->Config.File_Current_Offset+F.Position_Get()-MI->Config.File_Buffer_Size);
671 }
672 break;
673 }
674
675 #ifdef WINDOWS
676 Sleep(1000);
677 #endif //WINDOWS
678 }
679 }
680
681 #ifdef MEDIAINFO_DEBUG
682 Reader_File_BytesRead_Total+=MI->Config.File_Buffer_Size;
683 Reader_File_BytesRead+=MI->Config.File_Buffer_Size;
684 #endif //MEDIAINFO_DEBUG
685
686 //Parser
687 Status=MI->Open_Buffer_Continue(MI->Config.File_Buffer, MI->Config.File_Buffer_Size);
688
689 #if MEDIAINFO_READTHREAD
690 if (ThreadInstance && !MI->Config.File_Buffer_Repeat)
691 {
692 CS.Enter();
693 Buffer_Begin+=MI->Config.File_Buffer_Size;
694 #ifdef WINDOWS
695 if (Buffer_Begin==Buffer_Max)
696 {
697 CS.Leave();
698 SetEvent(Condition_WaitingForMorePlace);
699 }
700 else
701 #endif //WINDOWS
702 CS.Leave();
703 }
704 #endif //MEDIAINFO_READTHREAD
705
706 if (MI->Config.File_Buffer_Size==0)
707 {
708 #if MEDIAINFO_EVENTS
709 MediaInfoLib::Config.Log_Send(0xC0, 0xFF, 0xF0F00101, "File read error");
710 #endif //MEDIAINFO_EVENTS
711 break;
712 }
713
714 #if MEDIAINFO_DEMUX
715 if (MI->Config.Demux_EventWasSent)
716 return 2; //Must return immediately
717 #endif //MEDIAINFO_DEMUX
718
719 //Threading
720 if (MI->IsTerminating())
721 break; //Termination is requested
722 }
723 }
724
725 //Deleting buffer
726 #if MEDIAINFO_READTHREAD
727 if (ThreadInstance)
728 {
729 ThreadInstance->RequestTerminate();
730 SetEvent(Condition_WaitingForMorePlace);
731 while (!ThreadInstance->IsExited())
732 Sleep(0);
733 #ifdef WINDOWS
734 CloseHandle(Condition_WaitingForMorePlace);
735 CloseHandle(Condition_WaitingForMoreData);
736 #endif //WINDOWS
737 delete ThreadInstance; ThreadInstance=NULL;
738
739 MI->Config.File_Buffer=NULL;
740 MI->Config.File_Buffer_Size=0;
741 MI->Config.File_Buffer_Size_Max=0;
742 Buffer_Max=0;
743 delete[] Buffer; Buffer=NULL;
744 Buffer_Begin=0;
745 Buffer_End=0;
746 Buffer_End2=0;
747 IsLooping=false;
748 }
749 else
750 #endif //MEDIAINFO_READTHREAD
751 {
752 delete[] MI->Config.File_Buffer; MI->Config.File_Buffer=NULL;
753 MI->Config.File_Buffer_Size_Max=0;
754 }
755
756 #ifdef MEDIAINFO_DEBUG
757 std::cout<<std::hex<<Reader_File_Offset<<" - "<<Reader_File_Offset+Reader_File_BytesRead<<" : "<<std::dec<<Reader_File_BytesRead<<" bytes"<<std::endl;
758 std::cout<<"Total: "<<std::dec<<Reader_File_BytesRead_Total<<" bytes in "<<Reader_File_Count<<" blocks"<<std::endl;
759 #endif //MEDIAINFO_DEBUG
760
761 if (!MI->Config.File_KeepInfo_Get())
762 {
763 //File
764 F.Close();
765 }
766
767 //Is this file detected?
768 if (!Status[File__Analyze::IsAccepted])
769 return 0;
770
771 MI->Open_Buffer_Finalize();
772
773 #if MEDIAINFO_DEMUX
774 if (MI->Config.Demux_EventWasSent)
775 return 2; //Must return immediately
776 #endif //MEDIAINFO_DEMUX
777
778 return 1;
779}
