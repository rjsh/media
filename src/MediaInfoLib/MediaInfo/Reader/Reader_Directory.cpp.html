33size_t Reader_Directory::Format_Test(MediaInfo_Internal* MI, String File_Name)
34{
35 #ifdef MEDIAINFO_BDMV_YES
36 if (Dir::Exists(File_Name) && File_Name.find(Ztring(1, PathSeparator)+__T("BDMV"))+5==File_Name.size())
37 return Bdmv_Format_Test(MI, File_Name);
38 #endif //MEDIAINFO_BDMV_YES
39
40 #ifdef MEDIAINFO_P2_YES
41 if (Dir::Exists(File_Name) && File_Name.rfind(Ztring(1, PathSeparator)+__T("CONTENT"))+8==File_Name.size())
42 return P2_Format_Test(MI, File_Name);
43 #endif //MEDIAINFO_P2_YES
44
45 #ifdef MEDIAINFO_XDCAM_YES
46 if (Dir::Exists(File_Name) && File_Name.rfind(Ztring(1, PathSeparator)+__T("XDCAM"))+5==File_Name.size())
47 return Xdcam_Format_Test(MI, File_Name);
48 #endif //MEDIAINFO_XDCAM_YES
49
50 return 0;
51}

74#ifdef MEDIAINFO_BDMV_YES
75int Reader_Directory::Bdmv_Format_Test(MediaInfo_Internal* MI, const String &File_Name)
76{
77 if (!MI->SelectFromExtension(__T("Bdmv")))
78 return 0;
79
80 MI->Open_Buffer_Init(0, File_Name);
81
82 MI->Open_Buffer_Continue(NULL, 0);
83
84 MI->Open_Buffer_Finalize();
85
86 return 1;
87}
88#endif //MEDIAINFO_BDMV_YES

90//---------------------------------------------------------------------------
91#ifdef MEDIAINFO_BDMV_YES
92void Reader_Directory::Bdmv_Directory_Cleanup(ZtringList &List)
93{
94 //if there is a BDMV folder, this is blu-ray
95 Ztring ToSearch=Ztring(1, PathSeparator)+__T("BDMV")+PathSeparator+__T("index.bdmv"); //"\BDMV\index.bdmv"
96 for (size_t File_Pos=0; File_Pos<List.size(); File_Pos++)
97 {
98 size_t BDMV_Pos=List[File_Pos].find(ToSearch);
99 if (BDMV_Pos!=string::npos && BDMV_Pos!=0 && BDMV_Pos+16==List[File_Pos].size())
100 {
101 //This is a BDMV index, parsing the directory only if index and movie objects are BOTH present
102 ToSearch=List[File_Pos];
103 ToSearch.resize(ToSearch.size()-10);
104 ToSearch+=__T("MovieObject.bdmv"); //"%CompletePath%\BDMV\MovieObject.bdmv"
105 if (List.Find(ToSearch)!=string::npos)
106 {
107 //We want the folder instead of the files
108 List[File_Pos].resize(List[File_Pos].size()-11); //only %CompletePath%\BDMV
109 ToSearch=List[File_Pos];
110
111 for (size_t Pos=0; Pos<List.size(); Pos++)
112 {
113 if (List[Pos].find(ToSearch)==0 && List[Pos]!=ToSearch) //Remove all subdirs of ToSearch but not ToSearch
114 {
115 //Removing the file in the blu-ray directory
116 List.erase(List.begin()+Pos);
117 Pos--;
118 }
119 }
120 }
121 }
122 }
123}
124#endif //MEDIAINFO_BDMV_YES

132int Reader_Directory::P2_Format_Test(MediaInfo_Internal* MI, const String &File_Name)
133{
134 if (!MI->SelectFromExtension(__T("P2_Clip")))
135 return 0;

137 MI->Open(File_Name+__T("CLIP")+PathSeparator+__T("0013MM.XML"));
138
139 return 1;

143//---------------------------------------------------------------------------
144#ifdef MEDIAINFO_P2_YES
145void Reader_Directory::P2_Directory_Cleanup(ZtringList &List)
146{
147 //if there is a CONTENTS/CLIP folder, this is P2
148 Ztring ToSearch=Ztring(1, PathSeparator)+__T("CONTENTS")+PathSeparator+__T("CLIP")+PathSeparator; //"/CONTENTS/CLIP/"
149 for (size_t File_Pos=0; File_Pos<List.size(); File_Pos++)
150 {
151 size_t P2_Pos=List[File_Pos].find(ToSearch);
152 if (P2_Pos!=string::npos && P2_Pos!=0 && P2_Pos+1+8+1+4+1+10==List[File_Pos].size())
153 {
154 //This is a P2 CLIP
155 Ztring Path_Begin=List[File_Pos];
156 Path_Begin.resize(Path_Begin.size()-(1+8+1+4+1+10));
157 Path_Begin+=Ztring(1, PathSeparator);
158 bool HasChanged=false;
159 for (size_t Pos=0; Pos<List.size(); Pos++)
160 {
161 if (List[Pos].find(Path_Begin)==0 && List[Pos].find(Path_Begin+__T("CONTENTS")+PathSeparator+__T("CLIP")+PathSeparator)==string::npos) //Remove all subdirs of Path_Begin but not with CLIP
162 {
163 //Removing the file in the P2 directory
164 List.erase(List.begin()+Pos);
165 Pos--;
166 HasChanged=true;
167 }
168 }
169 if (HasChanged)
170 File_Pos=0;
171 }
172 }
173}
174#endif //MEDIAINFO_P2_YES

194#ifdef MEDIAINFO_XDCAM_YES
195void Reader_Directory::Xdcam_Directory_Cleanup(ZtringList &List)
196{
197 //if there is a XDCAM/Clip folder, this is Xdcam
198 Ztring ToSearch=Ztring(1, PathSeparator)+__T("Clip")+PathSeparator; //"/Clip/"
199 for (size_t File_Pos=0; File_Pos<List.size(); File_Pos++)
200 {
201 size_t Xdcam_Pos=List[File_Pos].find(ToSearch);
202 FileName Xdcam_FileName(List[File_Pos]);
203 if (Xdcam_Pos!=string::npos && Xdcam_Pos!=0 && (Xdcam_FileName.Extension_Get()==__T("XML")) && Xdcam_Pos+1+4+1+Xdcam_FileName.Name_Get().size()+4==List[File_Pos].size())
204 {
205 //This is a XDCAM CLIP
206 Ztring Path_Begin=List[File_Pos];
207 Path_Begin.resize(Path_Begin.size()-(1+4+1+12));
208 Path_Begin+=Ztring(1, PathSeparator);
209 if (Dir::Exists(Path_Begin+__T("Edit")) && Dir::Exists(Path_Begin+__T("General")) && Dir::Exists(Path_Begin+__T("Sub")))
210 {
211 bool HasChanged=false;
212 for (size_t Pos=0; Pos<List.size(); Pos++)
213 {
214 if (List[Pos].find(Path_Begin)==0 && (List[Pos].find(Path_Begin+__T("Clip")+PathSeparator)==string::npos || (List[Pos].find(__T(".XML"))!=List[Pos].size()-4))) //Remove all subdirs of Path_Begin but not with the right XML
215 {
216 //Removing the file in the XDCAM directory
217 List.erase(List.begin()+Pos);
218 Pos--;
219 HasChanged=true;
220 }
221 }
222 if (HasChanged)
223 File_Pos=0;
224 }
225 }
226 }
227}
228#endif //MEDIAINFO_XDCAM_YES
