59size_t Reader_libmms::Format_Test(MediaInfo_Internal* MI, String File_Name)
60{
61 mmsx_t* Handle;
62
63 //Opening the file
64 #if MEDIAINFO_LIBMMS_DESCRIBE_SUPPORT
65 if (MI->Config.File_Mmsh_Describe_Only_Get())
66 {
67 // Use MMSH & Send a DESCRIBE request
68 mmsh_t* MmshHandle;
69
70 MmshHandle=mmsh_describe_request(0, 0, Ztring(File_Name).To_Local().c_str());
71 if (MmshHandle==NULL)
72 return 0;
73
74 Handle=mmsx_set_mmsh_handle(MmshHandle);
75 if (Handle==NULL)
76 {
77 mmsh_close(MmshHandle);
78 return 0;
79 }
80 }
81 else
82 #endif //MEDIAINFO_LIBMMS_DESCRIBE_SUPPORT
83 {
84 // Use MMS or MMSH (Send a DESCRIBE & PLAY request)
85 Handle=mmsx_connect(0, 0, Ztring(File_Name).To_Local().c_str(), (int)-1);
86 if (Handle==NULL)
87 return 0;
88 }
89
90 //Init
91 size_t Buffer_Size_Max;
92 uint32_t Length;
93 if (!MI->Config.File_Mmsh_Describe_Only_Get())
94 {
95 //Buffer
96 Buffer_Size_Max=Buffer_NormalSize;
97
98 //MediaInfo init
99 mms_off_t Offset=mmsx_seek(0, Handle, 0, SEEK_SET);
100 uint32_t Length=mmsx_get_length(Handle);
101 MI->Open_Buffer_Init(Length, File_Name);
102 }
103 else
104 {
105 //Buffer
106 Buffer_Size_Max=mmsx_get_asf_header_len(Handle);
107
108 //MediaInfo init
109 Length=(uint32_t)-1;
110 MI->Open_Buffer_Init((int64u)-1, File_Name);
111 }
112 int8u* Buffer=new int8u[Buffer_Size_Max];
113
114 //Test the format with buffer
115 bool StopAfterFilled=MI->Config.File_StopAfterFilled_Get();
116 std::bitset<32> Status;
117 do
118 {
119 //Seek (if needed)
120 if (MI->Open_Buffer_Continue_GoTo_Get()!=(int64u)-1)
121 {
122 if (MI->Open_Buffer_Continue_GoTo_Get()>=Length)
123 break; //Seek requested, but on a file bigger in theory than what is in the real file, we can't do this
124 if (mmsx_seek(0, Handle, mms_off_t(MI->Open_Buffer_Continue_GoTo_Get()), SEEK_SET)!=MI->Open_Buffer_Continue_GoTo_Get())
125 break; //File is not seekable
126
127 MI->Open_Buffer_Init((int64u)-1, MI->Open_Buffer_Continue_GoTo_Get());
128 }
129
130 //Buffering
131 size_t Buffer_Size;
132 if (!MI->Config.File_Mmsh_Describe_Only_Get())
133 Buffer_Size=mmsx_read(0, Handle, (char*)Buffer, (int)Buffer_Size_Max);
134 else
135 Buffer_Size=mmsx_peek_header(Handle, (char*)Buffer, (int)Buffer_Size_Max);
136
137 //Parser
138 Status=MI->Open_Buffer_Continue(Buffer, Buffer_Size);
139 if (Buffer_Size==0 || MI->Config.File_Mmsh_Describe_Only_Get())
140 break;
141 }
142 while (!(Status[File__Analyze::IsFinished] || (StopAfterFilled && Status[File__Analyze::IsFilled])));
143
144 //File
145 mmsx_close(Handle);
146
147 //Buffer
148 delete[] Buffer; //Buffer=NULL;
149
150 //Is this file detected?
151 if (!Status[File__Analyze::IsAccepted])
152 return 0;
153
154 MI->Open_Buffer_Finalize();
155
156 return 1;
157}
