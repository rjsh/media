35//Integrity test
36#define INTEGRITY(TOVALIDATE, ERRORTEXT, OFFSET) \
37 if (!(TOVALIDATE)) \
38 { \
39 Trusted_IsNot(ERRORTEXT); \
40 return; \
41 } \
42
43#define INTEGRITY_INT(TOVALIDATE, ERRORTEXT, OFFSET) \
44 if (!(TOVALIDATE)) \
45 { \
46 Trusted_IsNot(ERRORTEXT); \
47 Info=0; \
48 return; \
49 } \
50
51#define INTEGRITY_SIZE_ATLEAST(_BYTES) \
52 if (Element_Offset+_BYTES>Element_Size) \
53 { \
54 Trusted_IsNot("Size is wrong"); \
55 return; \
56 } \
57
58#define INTEGRITY_SIZE_ATLEAST_STRING(_BYTES) \
59 if (Element_Offset+_BYTES>Element_Size) \
60 { \
61 Trusted_IsNot("Size is wrong"); \
62 Info.clear(); \
63 return; \
64 } \
65
66#define INTEGRITY_SIZE_ATLEAST_INT(_BYTES) \
67 if (Element_Offset+_BYTES>Element_Size) \
68 { \
69 Trusted_IsNot("Size is wrong"); \
70 Info=0; \
71 return; \
72 } \
73
74#define INTEGRITY_SIZE_ATLEAST_BUFFER() \
75 if (BS->Remain()==0) \
76 { \
77 Trusted_IsNot("Size is wrong"); \
78 Info=0; \
79 return; \
80 } \

86//---------------------------------------------------------------------------
87void File__Analyze::BS_Begin()
88{
89 size_t BS_Size;
90 if (Element_Offset>=Element_Size)
91 BS_Size=0;
92 else if (Buffer_Offset+Element_Size<=Buffer_Size)
93 BS_Size=(size_t)(Element_Size-Element_Offset);
94 else if (Buffer_Offset+Element_Offset<=Buffer_Size)
95 BS_Size=Buffer_Size-(size_t)(Buffer_Offset+Element_Offset);
96 else
97 BS_Size=0;
98
99 BS->Attach(Buffer+Buffer_Offset+(size_t)Element_Offset, BS_Size);
100}
101
102//---------------------------------------------------------------------------
103void File__Analyze::BS_Begin_LE()
104{
105 size_t BS_Size;
106 if (Buffer_Offset+Element_Size<=Buffer_Size)
107 BS_Size=(size_t)(Element_Size-Element_Offset);
108 else if (Buffer_Offset+Element_Offset<=Buffer_Size)
109 BS_Size=Buffer_Size-(size_t)(Buffer_Offset+Element_Offset);
110 else
111 BS_Size=0;
112
113 BT->Attach(Buffer+Buffer_Offset+(size_t)Element_Offset, BS_Size);
114}

121 BS->Attach(NULL, 0);

222// Big Endian - float 16 bits
223// TODO: remove it when Linux version of ZenLib is updated
224float32 BigEndian2float16corrected(const char* Liste)
225{
226 //sign 1 bit
227 //exponent 5 bit
228 //significand 10 bit
229
230 //Retrieving data
231 int16u Integer=BigEndian2int16u(Liste);
232
233 //Retrieving elements
234 bool Sign =(Integer&0x8000)?true:false;
235 int32u Exponent=(Integer>>10)&0xFF;
236 int32u Mantissa= Integer&0x03FF;
237
238 //Some computing
239 if (Exponent==0 || Exponent==0xFF)
240 return 0; //These are denormalised numbers, NANs, and other horrible things
241 Exponent-=0x0F; //Bias
242 float64 Answer=(((float64)Mantissa)/8388608+1.0)*std::pow((float64)2, (int)Exponent); //(1+Mantissa) * 2^Exponent
243 if (Sign)
244 Answer=-Answer;
245
246 return (float32)Answer;
247}
248inline float32 BigEndian2float16corrected (const int8u* List) {return BigEndian2float16corrected ((const char*)List);}

287void File__Analyze::Get_BFP4(int8u Bits, float32 &Info, const char* Name)
288{
289 INTEGRITY_SIZE_ATLEAST_INT(4);
290 BS_Begin();
291 int32s Integer=(int32s)BS->Get4(Bits);
292 int32u Fraction=BS->Get4(32-Bits);
293 BS_End();
294 Element_Offset-=4; //Because of BS_End()
295 if (Integer>=(1<<Bits)/2)
296 Integer-=1<<Bits;
297 Info=Integer+((float32)Fraction)/(1<<(32-Bits));
298 if (Trace_Activated) Param(Name, Info);
299 Element_Offset+=4;

1010 BS_Begin();
1011 while (Size_Mark==0 && BS->Remain() && Size<=8)
1012 {
1013 Size++;
1014 Peek_BS(Size, Size_Mark);

1017 //Integrity

1033 Info=0;

1064 break;
1065 case 5 : {

1430 CountOfBits+=Vlc[Info].bit_increment;
1431 break;
1432 case 1 :
1433 Value<<=1;

1437 case 0 : ;
1438 }
1439
1440 if (Value==Vlc[Info].value)

1442 if (Trace_Activated)
1443 {
1444 Ztring ToDisplay=Ztring::ToZtring(Value, 2);
1445 ToDisplay.insert(0, CountOfBits-ToDisplay.size(), __T('0'));
1446 ToDisplay+=__T(" (")+Ztring::ToZtring(CountOfBits)+__T(" bits)");
1447 Param(Name, ToDisplay);
1448 }
1449 return;
1450 }

1452 }

1455//---------------------------------------------------------------------------
1456void File__Analyze::Get_VL_Prepare(vlc_fast &Vlc)
1457{
1458 Vlc.Array=new int8u[((size_t)1)<<Vlc.Size];
1459 Vlc.BitsToSkip=new int8u[((size_t)1)<<Vlc.Size];
1460 memset(Vlc.Array, 0xFF, ((size_t)1)<<Vlc.Size);
1461 int8u Increment=0;
1462 int8u Pos=0;
1463 for(; ; Pos++)
1464 {
1465 if (Vlc.Vlc[Pos].bit_increment==255)
1466 break;
1467 Increment+=Vlc.Vlc[Pos].bit_increment;
1468 size_t Value=((size_t)Vlc.Vlc[Pos].value)<<(Vlc.Size-Increment);
1469 size_t ToFill_Size=((size_t)1)<<(Vlc.Size-Increment);
1470 for (size_t ToFill_Pos=0; ToFill_Pos<ToFill_Size; ToFill_Pos++)
1471 {
1472 Vlc.Array[Value+ToFill_Pos]=Pos;
1473 Vlc.BitsToSkip[Value+ToFill_Pos]=Increment;
1474 }
1475 }
1476 for (size_t Pos2=0; Pos2<(((size_t)1)<<Vlc.Size); Pos2++)
1477 if (Vlc.Array[Pos2]==(int8u)-1)
1478 {
1479 Vlc.Array[Pos2]=Pos;
1480 Vlc.BitsToSkip[Pos2]=(int8u)-1;
1481 }
1482}
