1683Ztring MediaInfo_Config_MediaInfo::File_Duplicate_Set (const Ztring &Value_In)
1684{
1685 //Preparing for File__Duplicate...
1686 CS.Enter();
1687 File__Duplicate_List.push_back(Value_In);
1688
1689 //Handling Memory index
1690 Ztring ToReturn;
1691 ZtringList List=Value_In;
1692 for (size_t Pos=0; Pos<List.size(); Pos++)
1693 {
1694 //Form= "(-)Data", if "-" the value will be removed
1695 Ztring &Value=List[Pos];
1696 bool ToRemove=false;
1697 if (Value.find(__T('-'))==0)
1698 {
1699 Value.erase(Value.begin());
1700 ToRemove=true;
1701 }
1702
1703 //Testing if this is information about a target
1704 if (List[Pos].find(__T("memory:"))==0 || List[Pos].find(__T("file:"))==0)
1705 {
1706 //Searching if already exist
1707 size_t Memory_Pos=File__Duplicate_Memory_Indexes.Find(List[Pos]);
1708 if (!ToRemove && Memory_Pos==Error)
1709 {
1710 //Does not exist yet (and adding is wanted)
1711 Memory_Pos=File__Duplicate_Memory_Indexes.Find(__T(""));
1712 if (Memory_Pos!=Error)
1713 File__Duplicate_Memory_Indexes[Memory_Pos]=List[Pos]; //A free place is found
1714 else
1715 {
1716 //Adding the place at the end
1717 Memory_Pos=File__Duplicate_Memory_Indexes.size();
1718 File__Duplicate_Memory_Indexes.push_back(List[Pos]);
1719 }
1720 }
1721 else if (ToRemove)
1722 {
1723 //Exists yet but Removal is wanted
1724 File__Duplicate_Memory_Indexes[Memory_Pos].clear();
1725 Memory_Pos=(size_t)-1;
1726 }
1727
1728 ToReturn+=__T(";")+Ztring().From_Number(Memory_Pos);
1729 }
1730 }
1731 if (!ToReturn.empty())
1732 ToReturn.erase(ToReturn.begin()); //Remove first ";"
1733
1734 CS.Leave();
1735 File_IsSeekable_Set(false); //If duplication, we can not seek anymore
1736
1737 return ToReturn;
1738}
