160//---------------------------------------------------------------------------
161bool File_Elf::FileHeader_Begin()
162{
163 //Element_Size
164 if (Buffer_Size<4)
165 return false; //Must wait for more data
166
167 if (Buffer[0]!=0x7F //".ELF"
168 || Buffer[1]!=0x45
169 || Buffer[2]!=0x4C
170 || Buffer[3]!=0x46)
171 {
172 Reject("ELF");
173 return false;
174 }
175
176 //All should be OK...
177 return true;
178}

180//***************************************************************************
181// Buffer - Global
182//***************************************************************************
183
184//---------------------------------------------------------------------------
185void File_Elf::Read_Buffer_Continue()
186{
187 //Parsing
188 int32u version4=(int32u)-1;
189 int16u type=(int16u)-1, machine=(int16u)-1;
190 int8u classs, data, version1, osabi, abiversion;
191 Skip_C4( "magic");
192 Get_L1 (classs, "class");
193 Get_L1 (data, "data");
194 Get_L1 (version1, "version");
195 Get_L1 (osabi, "osabi"); Param_Info1(Elf_osabi(osabi));
196 Get_L1 (abiversion, "abiversion");
197 Skip_XX(7, "reserved");
198 if (data==1) //LE
199 {
200 Get_L2 (type, "type"); Param_Info1(Elf_type(type));
201 Get_L2 (machine, "machine"); Param_Info1(Elf_machine(machine));
202 Get_L4 (version4, "version");
203 }
204 if (data==2) //BE
205 {
206 Get_B2 (type, "type"); Param_Info1(Elf_type(type));
207 Get_B2 (machine, "machine"); Param_Info1(Elf_machine(machine));
208 Get_B4 (version4, "version");
209 }
210 Skip_XX(Element_Size-Element_Offset, "Data");
211
212 FILLING_BEGIN();
213 if (version4!=(int32u)-1 && version1!=version4)
214 {
215 Reject("ELF");
216 return;
217 }
218
219 Accept("ELF");
220
221 Fill(Stream_General, 0, General_Format, "ELF");
222 if (type!=(int16u)-1)
223 Fill(Stream_General, 0, General_Format_Profile, Elf_type(type));
224 if (machine!=(int16u)-1)
225 Fill(Stream_General, 0, General_Format_Profile, Elf_machine(machine));
226
227 //No need of more
228 Finish("ELF");
229 FILLING_END();
230}
