34//---------------------------------------------------------------------------
35bool File_Ace::FileHeader_Begin()
36{
37 //Element_Size
38 if (Buffer_Size<7)
39 return false; //Must wait for more data
40
41 if (Buffer[0]!=0x2A //"**ACE**"
42 || Buffer[1]!=0x2A
43 || Buffer[2]!=0x41
44 || Buffer[3]!=0x43
45 || Buffer[4]!=0x45
46 || Buffer[5]!=0x2A
47 || Buffer[6]!=0x2A)
48 {
49 Reject("Ace");
50 return false;
51 }
52
53 //All should be OK...
54 return true;
55}

61//---------------------------------------------------------------------------
62void File_Ace::Read_Buffer_Continue()
63{
64 //Parsing
65 Skip_B7( "Magic");
66 Skip_XX(File_Size-7, "Data");
67
68 FILLING_BEGIN();
69 Accept("Ace");
70
71 Fill(Stream_General, 0, General_Format, "ACE");
72
73 Finish("Ace");
74 FILLING_END();
75}
