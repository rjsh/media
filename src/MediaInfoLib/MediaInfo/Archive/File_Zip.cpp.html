93bool File_Zip::FileHeader_Begin()
94{
95 //Element_Size
96 if (Buffer_Size<4)
97 return false; //Must wait for more data
98
99 if (Buffer[0]!=0x50 //"PK.."
100 || Buffer[1]!=0x4B
101 || Buffer[2]!=0x03
102 || Buffer[3]!=0x04)
103 {
104 Reject("ZIP");
105 return false;
106 }
107
108 //This is OK, ZIP detected
109 Accept();
110 Fill(Stream_General, 0, General_Format, "ZIP");
111
112 //Init
113 signature=0x00000000;
114 local_file_Step=0;
115 end_of_central_directory_IsParsed=false;
116
117 //Jumping to the end of the file minus end_of_central_directory size (we hope there is no comment)
118 GoTo(File_Size-22);
119
120 return true;
121}

127//---------------------------------------------------------------------------
128void File_Zip::Read_Buffer_Continue()
129{
130 for (;;)
131 {
132 //Probing
133 if (signature==0x00000000) //If not already tested (else an element is being parsed)
134 {
135 if (Element_Offset+4>Element_Size) //signature size
136 return; //Not enough data
137 signature=LittleEndian2int32u(Buffer+(size_t)Element_Offset);
138 }
139
140 //Parsing
141 switch (signature)
142 {
143 case 0x04034b50 : if (!local_file())
144 return; //Not enough data
145 break;
146 case 0x02014b50 : if (!central_directory())
147 return; //Not enough data
148 break;
149 case 0x05054b50 : if (!digital_signature())
150 return; //Not enough data
151 break;
152 case 0x06054b50 : if (!end_of_central_directory())
153 return; //Not enough data
154 break;
155 case 0x08064b50 : if(!archive_extra_data_record())
156 return;
157 break;
158 case 0x06064b50 : if(!Zip64_end_of_central_directory_record())
159 return;
160 break;
161 case 0x07064b50 : if(!Zip64_end_of_central_directory_locator())
162 return;
163 break;
164 default: Finish(); return; //Reject(); return; //Unknown value //Décommenter quand toutes les signatures sont implémentées
165 }
166
167 //Cleanup
168 signature=0x00000000; //Reset, must probe again the signature (next element)
169 }
170}

176bool File_Zip::local_file()
177{
178 switch(local_file_Step)
179 {
180 case 0 :
181 if (!local_file_header())
182 return false;
183 local_file_Step=1; //local_file_header parsed
184 break;
185 case 1 :
186 local_file_Step=2; //file_data is always parsed
187 if (!file_data())
188 return false;
189 break;
190 case 2:
191 if (!data_descriptor())
192 return false;
193 local_file_Step=0; //data_descriptor is parsed, back to begin
194 break;
195 default: ; //Should never happen
196 }
197
198 return true;
199}

201bool File_Zip::archive_extra_data_record()
202{
203 if (Element_Offset+8>Element_Size) //archive_extra_data_record
204 return false; //Not enough data
205
206 //Retrieving complete archive_extra_data_record size
207 int32u extra_field_length=LittleEndian2int32u(Buffer+(size_t)Element_Offset+4);
208
209 //Parsing
210 Element_Begin1("archive_extra_data_record");
211 Skip_C4("Archive extra data signature");
212 Skip_L4("extra field length");
213 Skip_XX(extra_field_length,"extra_field_data");
214 Element_End0();
215
216 return true;
217}

237bool File_Zip::local_file_header()
238{
239 if (Element_Offset+30>Element_Size) //local_file_header up to extra_field_length included
240 return false; //Not enough data
241
242 //Retrieving complete local_file_header size
243 int16u file_name_length=LittleEndian2int16u(Buffer+(size_t)Element_Offset+26);
244 int16u extra_field_length=LittleEndian2int16u(Buffer+(size_t)Element_Offset+28);
245 if (Element_Offset+30+file_name_length+extra_field_length>Element_Size) //local_file_header all included
246 return false; //Not enough data
247
248 //Parsing
249 Element_Begin1("local_file_header");
250 int16u general_purpose_bit_flag,compression_method;
251 bool efs;
252 Skip_C4("Local file header signature");
253 Skip_L2("Version needed to extract");
254 Get_L2 (general_purpose_bit_flag,"general purpose bit flag");
255 Skip_Flags(general_purpose_bit_flag, 0, "encrypted file");
256 Skip_Flags(general_purpose_bit_flag, 1, "8K sliding dictionary");
257 Skip_Flags(general_purpose_bit_flag, 2, "3 Shannon-Fano trees");
258 Get_Flags (general_purpose_bit_flag, 3, data_descriptor_set, "data descriptor");
259 Skip_Flags(general_purpose_bit_flag, 4, "Reserved for use with method 8");
260 Skip_Flags(general_purpose_bit_flag, 4, "file is compressed patched data");
261 Skip_Flags(general_purpose_bit_flag, 4, "Strong encryption");
262 //~ Skip_Flags(general_purpose_bit_flag, 4, "Currently unused");
263 //~ Skip_Flags(general_purpose_bit_flag, 4, "Currently unused");
264 //~ Skip_Flags(general_purpose_bit_flag, 4, "Currently unused");
265 //~ Skip_Flags(general_purpose_bit_flag, 4, "Currently unused");
266 Get_Flags (general_purpose_bit_flag, 11, efs, "Language encoding flag (EFS)");
267 //~ Skip_Flags(general_purpose_bit_flag, 4, "Reserved by PKWARE for enhanced compression");
268 //~ Skip_Flags(general_purpose_bit_flag, 4, "");
269 //~ Skip_Flags(general_purpose_bit_flag, 4, "Reserved by PKWARE");
270 //~ Skip_Flags(general_purpose_bit_flag, 4, "Reserved by PKWARE");
271 Get_L2 (compression_method,"compression method");
272 Param_Info1C((compression_method<20), Zip_compression_method[compression_method]);
273 Param_Info1C((compression_method==97||compression_method==98), Zip_compression_method[compression_method-97+20]);
274 Skip_L2("last mod file time");
275 Skip_L2("last mod file date");
276 Skip_L4("crc-32");
277 Get_L4(compressed_size,"compressed size");
278 Skip_L4("uncompressed size");
279 Get_L2(file_name_length,"file name lenth");
280 Get_L2(extra_field_length,"extra field length");
281 if(efs) {
282 Skip_UTF8(file_name_length,"file name");
283 Skip_UTF8(extra_field_length,"extra field");
284 } else {
285 Skip_Local(file_name_length,"file name");
286 Skip_Local(extra_field_length,"extra field");
287 }
288 Element_End0();
289
290 FILLING_BEGIN();
291 Accept("Zip");
292 Fill(Stream_General, 0, General_Format, "ZIP");
293 FILLING_END();
294 return true;
295}

297bool File_Zip::file_data()
298{
299 Element_Begin1("file_data");
300 Skip_XX(compressed_size,"File_data");
301 Element_End0();
302
303 if (Element_Offset>Element_Size)
304 {
305 GoTo(File_Offset+Element_Offset);
306 return false;
307 }
308 return true;

328bool File_Zip::central_directory()
329{
330 if (Element_Offset+46>Element_Size) //central_directory up to relative offset of local header included
331 return false; //Not enough data
332
333 //Retrieving complete local_file_header size
334 int16u file_name_length=LittleEndian2int16u(Buffer+(size_t)Element_Offset+28);
335 int16u extra_field_length=LittleEndian2int16u(Buffer+(size_t)Element_Offset+30);
336 int16u file_comment_length=LittleEndian2int16u(Buffer+(size_t)Element_Offset+32);
337 if (Element_Offset+46+file_name_length+extra_field_length+file_comment_length>Element_Size) //central_directory_structure all included
338 return false; //Not enough data
339
340 int16u general_purpose_bit_flag;
341 bool efs;
342 int16u version_made_by,compression_method;
343
344 //Parsing
345 Element_Begin1("Central directory");
346 Skip_C4("central file header signature");
347 Get_L2 (version_made_by,"version made by");Param_Info1((version_made_by>>8)>20?"unused":Zip_made_by[version_made_by>>8]);
348 Skip_L2("version needed to extract");
349 Get_L2 (general_purpose_bit_flag,"general purpose bit flag");
350 Skip_Flags(general_purpose_bit_flag, 0, "encrypted file");
351 Skip_Flags(general_purpose_bit_flag, 1, "8K sliding dictionary");
352 Skip_Flags(general_purpose_bit_flag, 2, "3 Shannon-Fano trees");
353 Skip_Flags(general_purpose_bit_flag, 3, "data descriptor");
354 Skip_Flags(general_purpose_bit_flag, 4, "Reserved for use with method 8");
355 Skip_Flags(general_purpose_bit_flag, 4, "file is compressed patched data");
356 Skip_Flags(general_purpose_bit_flag, 4, "Strong encryption");
357 //~ Skip_Flags(general_purpose_bit_flag, 4, "Currently unused");
358 //~ Skip_Flags(general_purpose_bit_flag, 4, "Currently unused");
359 //~ Skip_Flags(general_purpose_bit_flag, 4, "Currently unused");
360 //~ Skip_Flags(general_purpose_bit_flag, 4, "Currently unused");
361 Get_Flags (general_purpose_bit_flag, 11, efs, "Language encoding flag (EFS)");
362 //~ Skip_Flags(general_purpose_bit_flag, 4, "Reserved by PKWARE for enhanced compression");
363 //~ Skip_Flags(general_purpose_bit_flag, 4, "");
364 //~ Skip_Flags(general_purpose_bit_flag, 4, "Reserved by PKWARE");
365 //~ Skip_Flags(general_purpose_bit_flag, 4, "Reserved by PKWARE");
366 Get_L2 (compression_method,"compression method");
367 Param_Info1C((compression_method<20), Zip_compression_method[compression_method]);
368 Param_Info1C((compression_method==97||compression_method==98), Zip_compression_method[compression_method-97+20]);
369 Skip_L2("last mod file time");
370 Skip_L2("last mod file date");
371 Skip_L4("crc-32");
372 Skip_L4("compressed size");
373 Skip_L4("uncompressed size");
374 Skip_L2("file name length");
375 Skip_L2("extra field length");
376 Skip_L2("file comment length");
377 Skip_L2("disk number start");
378 Skip_L2("internal file attributes");
379 Skip_L4("external file attributes");
380 Skip_L4("relative offset of local header");
381 if(efs) {
382 Skip_UTF8(file_name_length,"file name");
383 Skip_UTF8(extra_field_length,"extra field");
384 Skip_UTF8(file_comment_length,"file comment");
385 } else {
386 Skip_Local(file_name_length,"file name");
387 Skip_Local(extra_field_length,"extra field");
388 Skip_Local(file_comment_length,"file comment");
389 }
390 Element_End0();
391
392 return true;
393}

395bool File_Zip::end_of_central_directory()
396{
397 if (Element_Offset+22>Element_Size) //end_of_central_directory up to relative offset of .ZIP file comment length included
398 return false; //Not enough data
399
400 //Retrieving complete local_file_header size
401 int16u zip_comment_length=LittleEndian2int16u(Buffer+(size_t)Element_Offset+20);
402 if (Element_Offset+22+zip_comment_length>Element_Size) //end_of_central_directory all included
403 return false; //Not enough data
404
405 //Parsing
406 int32u offset;
407 Element_Begin1("End of central directory");
408 Skip_C4( "end of central dir signature");
409 Skip_L2( "number of this disk");
410 Skip_L2( "number of the disk");// with the start of the central directory
411 Skip_L2( "total number of entries on this disk");// in the central directory
412 Skip_L2( "total number of entries");// in the central directory
413 Skip_L4( "size of the central directory");
414 Get_L4 (offset, "offset of start of central directory");// with respect to the starting disk number
415 Skip_L2( "zip file comment length");
416 Skip_XX(zip_comment_length, "zip file comment");
417 Element_End0();
418
419 //Going to first central directory (once)
420 if (!end_of_central_directory_IsParsed)
421 {
422 end_of_central_directory_IsParsed=true;
423 GoTo(offset);
424 }
425 return true;
426}

428bool File_Zip::Zip64_end_of_central_directory_record()
429{
430 if (Element_Offset+12>Element_Size) //Zip64_end_of_central_directory_record
431 return false; //Not enough data
432
433 //Retrieving complete Zip64_end_of_central_directory_record size
434 int64u size_of_Zip64_end_of_central_directory_record=LittleEndian2int64u(Buffer+(size_t)Element_Offset+4);
435 if (Element_Offset+12+size_of_Zip64_end_of_central_directory_record>Element_Size) //end_of_central_directory all included
436 return false; //Not enough data
437
438 //Parsing
439 //~ int32u offset;
440 int16u version_made_by;
441 Element_Begin1("Zip64 End of central directory record");
442 Skip_C4( "Zip64 end of central dir signature");
443 Skip_L8( "size of zip64 end of central directory record");
444 Get_L2 (version_made_by, "version made by");
445 Param_Info1((version_made_by>>8)>20?"unused":Zip_made_by[version_made_by>>8]);
446 Skip_L2( "version needed to extract");
447 Skip_L4( "number of this disk");
448 Skip_L4( "number of the disk");// with the start of the central directory
449 Skip_L8( "total number of entries on this disk");// in the central directory
450 Skip_L8( "total number of entries");// in the central directory
451 Skip_L8( "size of the central directory");
452 Skip_L8( "offset of start of central directory"); // with respect to the starting disk number
453 Skip_XX(size_of_Zip64_end_of_central_directory_record-44, "zip64 extensible data sector");
454 Element_End0();
455
456 return true;
457}
