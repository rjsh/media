30//***************************************************************************
31// File header
32//***************************************************************************
33
34//---------------------------------------------------------------------------
35bool File_Iso9660::FileHeader_Begin()
36{
37 //Element_Size
38 if (Buffer_Size<65536)
39 return false; //Must wait for more data
40
41 if (CC6(Buffer+32768)!=0x014344303031LL) //"Iso9660...."
42 {
43 Reject("ISO 9660");
44 return false;
45 }
46
47 //All should be OK...
48 Accept("ISO 9660");
49 return true;
50}
51
52//***************************************************************************
53// Buffer - Global
54//***************************************************************************
55
56//---------------------------------------------------------------------------
57void File_Iso9660::Read_Buffer_Continue()
58{
59 Skip_XX(32768, "System Area");
60 Primary_Volume_Descriptor();
61
62 FILLING_BEGIN();
63 Fill(Stream_General, 0, General_Format, "ISO 9660");
64
65 Finish("ISO 9660");
66 FILLING_END();
67}
68
69//***************************************************************************
70// Elements
71//***************************************************************************
72
73//---------------------------------------------------------------------------
74void File_Iso9660::Primary_Volume_Descriptor()
75{
76 int32u Volume_Space_Size;
77 Element_Begin1("Volume_Descriptor");
78 Skip_B1("Volume Descriptor Type");
79 Skip_Local(5,"Standard Identifier");
80 Skip_B1("Volume Descriptor Version");
81 Skip_B1("Unused field");
82 Skip_Local(32,"System Identifier");
83 Skip_Local(32,"Volume Identifier");
84 Skip_XX(8,"Unused field");
85 Get_D4(Volume_Space_Size, "Volume Space Size"); Param_Info2(Volume_Space_Size*(int64u)2048, " bytes");
86 Skip_XX(32,"Unused field");
87 Skip_B4("Volume Set Size");
88 Skip_B4("Volume Sequence Number");
89 Skip_B4("Logical Block Size");
90 Skip_B8("Path Table Size");
91 Skip_B4("Location of Occurrence of Type L Path Table");
92 Skip_B4("Location of Optional Occurrence of Type L Path Table");
93 Skip_B4("Location of Occurrence of Type M Path Table");
94 Skip_B4("Location of Optional Occurrence of Type M Path Table");
95 Skip_Local(34,"Directory Record for Root Directory");
96 Skip_Local(128,"Volume Set Identifier");
97 Skip_Local(128,"Publisher Identifier");
98 Skip_Local(128,"Data Preparer Identifier");
99 Skip_Local(128,"Application Identifier");
100 Skip_Local(37,"Copyright File Identifier");
101 Skip_Local(37,"Abstract File Identifier");
102 Skip_Local(37,"Bibliographic File Identifier");
103 Skip_XX(17,"Volume Creation Date and Time");
104 Element_End0();
105}
