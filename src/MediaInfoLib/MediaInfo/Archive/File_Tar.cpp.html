35void File_Tar::Read_Buffer_Continue()
36{
37 if (File_Size<257)
38 {
39 Reject();
40 return;
41 }
42 if (Buffer_Size<257)
43 return; //Wait for more data
44
45 //Parsing
46 Ztring ChecksumO;
47 Skip_Local(100, "File name");
48 Skip_Local( 8, "File mode");
49 Skip_Local( 8, "Owner's numeric user ID");
50 Skip_Local( 12, "Group's numeric user ID");
51 Skip_Local( 12, "File size in bytes");
52 Skip_Local( 8, "Last modification time in numeric Unix time format");
53 Get_Local ( 8, ChecksumO, "Checksum for header block");
54 Skip_B1( "Link indicator (file type)");
55 Skip_Local(100, "Name of linked file");
56 Skip_XX(File_Size-257, "Data");
57
58 FILLING_BEGIN();
59 //Handling Checksum
60 int32u Checksum=ChecksumO.To_int32u(8);
61 int32u ChecksumU=0;
62 int32u ChecksumS=0;
63 for (size_t Pos=0; Pos<257; Pos++)
64 {
65 if (Pos==148)
66 {
67 ChecksumU+=32*8; //8 spaces
68 ChecksumS+=32*8; //8 spaces
69 Pos+=7; //Skiping Checksum
70 }
71 ChecksumU+=(int8u)Buffer[Pos];
72 ChecksumS+=(int8s)Buffer[Pos];
73 }
74
75 if (ChecksumU!=Checksum && ChecksumS!=Checksum)
76 {
77 Reject("Tar");
78 return;
79 }
80
81 //Filling
82 Accept("Tar");
83
84 Fill(Stream_General, 0, General_Format, "Tar");
85
86 Reject("Tar");
87 FILLING_END();
88}
