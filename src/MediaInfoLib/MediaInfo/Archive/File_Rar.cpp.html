59//---------------------------------------------------------------------------
60const char* Rar_HEADER_TYPE(int8u HEADER_TYPE)
61{
62 switch(HEADER_TYPE)
63 {
64 case 0x72 : return "marker block";
65 case 0x73 : return "archive header";
66 case 0x74 : return "file header";
67 case 0x75 : return "old style comment header";
68 case 0x76 : return "old style authenticity information";
69 case 0x77 : return "old style subblock";
70 case 0x78 : return "old style recovery record";
71 case 0x79 : return "old style authenticity informatio";
72 case 0x7A : return "subblock";
73 case 0x7B : return "end of file";
74 default : return "";
75 }
76};
77
78//---------------------------------------------------------------------------
79Ztring Rar_version_number(int8u byte)
80{
81 //Version number is encoded as 10 * Major version + minor version.
82 return Ztring::ToZtring(byte/10)+Ztring(".")+Ztring::ToZtring(byte%10);
83}

101//---------------------------------------------------------------------------
102bool File_Rar::FileHeader_Begin()
103{
104 //Element_Size
105 if (Buffer_Size<7)
106 return false; //Must wait for more data
107
108 if (Buffer[0]!=0x52 //"Rar!"
109 || Buffer[1]!=0x61
110 || Buffer[2]!=0x72
111 || Buffer[3]!=0x21
112 || Buffer[4]!=0x1A
113 || Buffer[5]!=0x07
114 || Buffer[6]!=0x00)
115 {
116 Reject("RAR");
117 return false;
118 }
119
120 state=0;
121
122 //All should be OK...
123 return true;
124}

130//---------------------------------------------------------------------------
131bool File_Rar::Header_Begin()
132{
133 if (Element_Offset+7>Element_Size)
134 return false; //Not enough data for header size
135 int16u HEAD_SIZE=LittleEndian2int16u(Buffer+Buffer_Offset+(size_t)Element_Offset+5);
136
137 if (Element_Offset+HEAD_SIZE>Element_Size)
138 return false; //Not enough data for header
139
140 return true;
141}

143//---------------------------------------------------------------------------
144void File_Rar::Header_Parse()
145{
146 //Config
147 HIGH_PACK_SIZE=0;
148 PACK_SIZE=0;
149
150 //Parsing
151 int16u HEAD_SIZE;
152 Skip_L2( "HEAD_CRC"); //CRC of total block or block part
153 Get_L1 (HEAD_TYPE, "HEAD_TYPE"); //Block type
154 Get_L2 (HEAD_FLAGS, "HEAD_FLAGS");
155 Header_Parse_Flags();
156 Get_L2 (HEAD_SIZE, "HEAD_SIZE");
157 Header_Parse_Content();
158 Skip_XX(HEAD_SIZE-(size_t)Element_Offset, "REST OF HEADER");
159
160 //Filling
161 Header_Fill_Size(HEAD_SIZE+HIGH_PACK_SIZE*0x100000000LL+PACK_SIZE);
162 Header_Fill_Code(HEAD_TYPE, Rar_HEADER_TYPE(HEAD_TYPE));
163}

167{

172 default : Header_Parse_Flags_XX();
173 }
174}

176//---------------------------------------------------------------------------
177// archive header
178void File_Rar::Header_Parse_Flags_73()
179{
180 Skip_Flags(HEAD_FLAGS, 1, "Volume attribute (archive volume)");
181 Skip_Flags(HEAD_FLAGS, 2, "Archive comment present");
182 Skip_Flags(HEAD_FLAGS, 3, "Archive lock attribute");
183 Skip_Flags(HEAD_FLAGS, 4, "Solid attribute (solid archive)");
184 Skip_Flags(HEAD_FLAGS, 5, "New volume naming scheme"); // (\'volname.partN.rar\')
185 Skip_Flags(HEAD_FLAGS, 6, "Authenticity information present");
186 Skip_Flags(HEAD_FLAGS, 7, "Recovery record present");
187 Skip_Flags(HEAD_FLAGS, 8, "Block headers are encrypted");
188 Skip_Flags(HEAD_FLAGS, 9, "First volume (set only by RAR 3.0 and later)");
189 Get_Flags (HEAD_FLAGS, 15, add_size, "ADD_SIZE present");
190}

192//---------------------------------------------------------------------------
193// file header
194void File_Rar::Header_Parse_Flags_74()
195{
196 Skip_Flags(HEAD_FLAGS,0, "file continued from previous volume");
197 Skip_Flags(HEAD_FLAGS,1, "file continued in next volume");
198 Skip_Flags(HEAD_FLAGS,2, "file encrypted with password");
199 Skip_Flags(HEAD_FLAGS,3, "file comment present");
200 Skip_Flags(HEAD_FLAGS,4, "information from previous files is used"); // (solid flag)
201 // bits 7 6 5 (for RAR 2.0 and later)
202 // 0 0 0 - dictionary size 64 KB
203 // 0 0 1 - dictionary size 128 KB
204 // 0 1 0 - dictionary size 256 KB
205 // 0 1 1 - dictionary size 512 KB
206 // 1 0 0 - dictionary size 1024 KB
207 // 1 0 1 - dictionary size 2048 KB
208 // 1 1 0 - dictionary size 4096 KB
209 // 1 1 1 - file is directory
210 Get_Flags (HEAD_FLAGS, 8, high_fields, "HIGH_PACK_SIZE and HIGH_UNP_SIZE fields");
211 Get_Flags (HEAD_FLAGS, 9, usual_or_utf8, "FILE_NAME contains usual and encoded unicode");
212 Get_Flags (HEAD_FLAGS, 10, salt, "SALT present");
213 Skip_Flags(HEAD_FLAGS, 11, "Version flag.");
214 Get_Flags (HEAD_FLAGS, 12, exttime, "Extended time field present");
215 Get_Flags (HEAD_FLAGS, 15, add_size, "ADD_SIZE present");
216}

236//---------------------------------------------------------------------------

238void File_Rar::Header_Parse_Content_73()
239{
240 Skip_L2( "RESERVED_1");
241 Skip_L4( "RESERVED_2");
242}

244//---------------------------------------------------------------------------
245// file header
246void File_Rar::Header_Parse_Content_74()
247{
248 int16u name_size;
249 int8u HOST_OS, METHOD, UNP_VER;
250 Get_L4 (PACK_SIZE, "PACK_SIZE"); //Compressed file size
251 Skip_L4( "UNP_SIZE"); //Uncompressed file size
252 Get_L1 (HOST_OS, "HOST_OS"); Param_Info1((HOST_OS<6?Rar_host_os[HOST_OS]:"Unknown"));
253 Skip_L4( "FILE_CRC");
254 Skip_L4( "FTIME"); //Date and time in standard MS DOS format
255 Get_L1 (UNP_VER, "UNP_VER"); Param_Info1(Rar_version_number(UNP_VER)); //RAR version needed to extract file
256 Get_L1 (METHOD, "METHOD"); Param_Info1(((METHOD>=0x30)&&(METHOD<0x36)?Rar_packing_method[METHOD-0x30]:"Unknown"));
257 Get_L2 (name_size, "NAME_SIZE"); //File name size
258 Skip_L4( "ATTR"); //File attributes
259 if(high_fields)
260 {
261 Get_L4 (HIGH_PACK_SIZE, "HIGH_PACK_SIZE"); //High 4 bytes of 64 bit value of compressed file size.
262 Skip_L4( "HIGH_UNP_SIZE"); //High 4 bytes of 64 bit value of uncompressed file size.
263 }
264 else
265 HIGH_PACK_SIZE=0;
266 if (usual_or_utf8)
267 {
268 //Must test the content before reading, looking fore zero byte
269 if (Element_Offset+name_size>Element_Size)
270 {
271 Skip_XX(Element_Size-Element_Offset, "Error");
272 return;
273 }
274 int64u ZeroPos=0;
275 while (ZeroPos<name_size)
276 {
277 if (Buffer[Buffer_Offset+(size_t)(Element_Offset+ZeroPos)]==0)
278 break; //Found
279 ZeroPos++;
280 }
281
282 if (ZeroPos==name_size)
283 Skip_UTF8(name_size, "FILE_NAME");
284 else
285 {
286 Skip_Local(ZeroPos, "FILE_NAME"); //Up to ZeroPos
287 Skip_L1( "Zero");
288 Skip_UTF16L(name_size-(ZeroPos+1), "FILE_NAME"); //Spec is not precise, "Unicode" without encoding format (character size, endianess), because RAR is from Windows, we hope this is the format from Windows (UTF-16 Little Endian)
289 }
290 }
291 else
292 Skip_Local(name_size, "FILE_NAME");
293
294 if (salt)
295 Skip_L8( "SALT");
296 //if(exttime)
297 //Skip_XX("EXT_TIME"); //Which size?
298}
