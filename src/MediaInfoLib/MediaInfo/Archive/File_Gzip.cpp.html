34//---------------------------------------------------------------------------
35bool File_Gzip::FileHeader_Begin()
36{
37 //Element_Size
38 if (Buffer_Size<2)
39 return false; //Must wait for more data
40
41 if (Buffer[0]!=0x1F
42 || Buffer[1]!=0x8B)
43 {
44 Reject("Gzip");
45 return false;
46 }
47
48 //All should be OK...
49 return true;
50}
51
52//***************************************************************************
53// Buffer - Global
54//***************************************************************************
55
56//---------------------------------------------------------------------------
57void File_Gzip::Read_Buffer_Continue()
58{
59 //Parsing
60 int8u CM;
61 Skip_B2( "IDentification");
62 Get_B1 (CM, "Compression Method");
63 Skip_B1( "FLaGs");
64 Skip_B4( "Modified TIME");
65 Skip_XX(File_Size-10, "Data");
66
67 FILLING_BEGIN();
68 //Filling
69 Accept("Gzip");
70
71 Fill(Stream_General, 0, General_Format, "GZip");
72 Fill(Stream_General, 0, General_Format_Profile, "deflate");
73
74 Finish("Gzip");
75 FILLING_END();
76}
