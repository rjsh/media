35bool File_Bzip2::FileHeader_Begin()
36{
37 //Element_Size
38 if (Buffer_Size<2)
39 return false; //Must wait for more data
40
41 if (Buffer[0]!=0x42 //"BZ"
42 || Buffer[1]!=0x5A)
43 {
44 Reject("Bzip2");
45 return false;
46 }
47
48 //All should be OK...
49 return true;
50}
51

53// Buffer - Global
54//***************************************************************************
55
56//---------------------------------------------------------------------------
57void File_Bzip2::Read_Buffer_Continue()
58{
59 //Parsing
60 Skip_B7( "Magic");
61 Skip_XX(File_Size-2, "Data");
62
63 FILLING_BEGIN();
64 Accept("Bzip2");
65
66 Fill(Stream_General, 0, General_Format, "BZip2");
67
68 Finish("Bzip2");
69 FILLING_END();
70}
