76#if MEDIAINFO_TRACE
77void File__Base::Init (MediaInfo_Config_MediaInfo * Config_, string* Details_, std::vector<std::vector<ZtringList> > * Stream_, std::vector<std::vector<ZtringListList> > * Stream_More_)
78#else //MEDIAINFO_TRACE
79void File__Base::Init (MediaInfo_Config_MediaInfo * Config_, std::vector<std::vector<ZtringList> > * Stream_, std::vector<std::vector<ZtringListList> > * Stream_More_)
80#endif //MEDIAINFO_TRACE
81{
82 if (Config)
83 return; //Already done
84
85 if (Stream_)
86 {
87 Stream=Stream_;
88 Stream_More=Stream_More_;
89 Stream_MustBeDeleted=false;
90 }
91 else
92 {
93 Stream=new std::vector<std::vector<ZtringList> >;
94 Stream->resize(Stream_Max);
95 Stream_More=new std::vector<std::vector<ZtringListList> >;
96 Stream_More->resize(Stream_Max);
97 Stream_MustBeDeleted=true;
98 }
99
100 Config=Config_;
101 #if MEDIAINFO_TRACE
102 Details=Details_;
103 #endif //MEDIAINFO_TRACE
104}

110//---------------------------------------------------------------------------
111size_t File__Base::Count_Get (stream_t StreamKind, size_t Pos) const
112{
113 //Integrity
114 if (StreamKind>=Stream_Max)
115 return 0;
116
117 //Count of streams
118 if (!Stream)
119 return 0;
120 if (Pos==Error)
121 return (*Stream)[StreamKind].size();
122
123 //Integrity
124 if (Pos>=(*Stream)[StreamKind].size())
125 return 0;
126
127 //Count of piece of information in a stream
128 return MediaInfoLib::Config.Info_Get(StreamKind).size()+(*Stream_More)[StreamKind][Pos].size();
129}
130
131//---------------------------------------------------------------------------
132const Ztring &File__Base::Get (stream_t StreamKind, size_t StreamNumber, size_t Parameter, info_t KindOfInfo)
133{
134 //Check integrity
135 if (StreamKind>=Stream_Max || StreamNumber>=(*Stream)[StreamKind].size() || Parameter>=MediaInfoLib::Config.Info_Get(StreamKind).size()+(StreamNumber>=(*Stream_More)[StreamKind].size()?0:(*Stream_More)[StreamKind][StreamNumber].size()) || KindOfInfo>=Info_Max)
136 return MediaInfoLib::Config.EmptyString_Get(); //Parameter is unknown
137
138 else if (Parameter<MediaInfoLib::Config.Info_Get(StreamKind).size())
139 {
140 //Optimization : KindOfInfo>Info_Text is in static lists
141 if (KindOfInfo!=Info_Text)
142 return MediaInfoLib::Config.Info_Get(StreamKind)[Parameter][KindOfInfo]; //look for static information only
143 else if (Parameter<(*Stream)[StreamKind][StreamNumber].size())
144 return (*Stream)[StreamKind][StreamNumber][Parameter];
145 else
146 return MediaInfoLib::Config.EmptyString_Get(); //This parameter is known, but not filled
147 }
148 else
149 {
150 if (((size_t)(Parameter-MediaInfoLib::Config.Info_Get(StreamKind).size()))<(*Stream_More)[StreamKind][StreamNumber].size() && ((size_t)KindOfInfo)<(*Stream_More)[StreamKind][StreamNumber][Parameter-MediaInfoLib::Config.Info_Get(StreamKind).size()].size())
151 return (*Stream_More)[StreamKind][StreamNumber][Parameter-MediaInfoLib::Config.Info_Get(StreamKind).size()][KindOfInfo];
152 else
153 return MediaInfoLib::Config.EmptyString_Get(); //Not filled
154 }
155}
156
157//---------------------------------------------------------------------------
158const Ztring &File__Base::Get (stream_t StreamKind, size_t StreamPos, const Ztring &Parameter, info_t KindOfInfo, info_t KindOfSearch)
159{
160 size_t ParameterI=0;
161
162 //Check integrity
163 if (StreamKind>=Stream_Max || StreamPos>=(*Stream)[StreamKind].size() || KindOfInfo>=Info_Max)
164 return MediaInfoLib::Config.EmptyString_Get();
165 if ((ParameterI=MediaInfoLib::Config.Info_Get(StreamKind).Find(Parameter, KindOfSearch))==Error)
166 {
167 ParameterI=(*Stream_More)[StreamKind][StreamPos].Find(Parameter, KindOfSearch);
168 if (ParameterI==Error)
169 return MediaInfoLib::Config.EmptyString_Get(); //Parameter is unknown
170 return (*Stream_More)[StreamKind][StreamPos][ParameterI](KindOfInfo);
171 }
172
173 return Get(StreamKind, StreamPos, ParameterI, KindOfInfo);
174}

176//---------------------------------------------------------------------------
177int File__Base::Set (stream_t StreamKind, size_t StreamNumber, size_t Parameter, const Ztring &ToSet, const Ztring &OldValue)
178{
179 //Integrity
180 if (Count_Get(StreamKind)<=StreamNumber)
181 return 0;
182
183 return Set(StreamKind, StreamNumber, Get(StreamKind, StreamNumber, Parameter, Info_Name), ToSet, OldValue);
184}
185
186//---------------------------------------------------------------------------
187int File__Base::Set (stream_t StreamKind, size_t StreamNumber, const Ztring &Parameter, const Ztring &ToSet, const Ztring &OldValue)
188{
189 //Integrity
190 if (Count_Get(StreamKind)<=StreamNumber)
191 return 0;
192
193 return Write(StreamKind, StreamNumber, Parameter, ToSet, OldValue);
194}
