44//---------------------------------------------------------------------------
45void File__Analyze::Streams_Finish_Global()
46{
47 if (IsSub)
48 return;
49
50 #if MEDIAINFO_ADVANCED
51 //Default frame rate
52 if (Count_Get(Stream_Video)==1 && Retrieve(Stream_Video, 0, Video_FrameRate).empty() && Config->File_DefaultFrameRate_Get())
53 Fill(Stream_Video, 0, Video_FrameRate, Config->File_DefaultFrameRate_Get());
54 #endif //MEDIAINFO_ADVANCED
55
56 //Video Frame count
57 if (Count_Get(Stream_Video)==1 && Count_Get(Stream_Audio)==0 && Retrieve(Stream_Video, 0, Video_FrameCount).empty())
58 {
59 if (Frame_Count_NotParsedIncluded!=(int64u)-1 && File_Offset+Buffer_Size==File_Size)
60 Fill(Stream_Video, 0, Video_FrameCount, Frame_Count_NotParsedIncluded);
61 else if (Config->File_Names.size()>1)
62 Fill(Stream_Video, 0, Video_FrameCount, Config->File_Names.size());
63 #if MEDIAINFO_IBIUSAGE
64 else
65 {
66 //External IBI
67 std::string IbiFile=Config->Ibi_Get();
68 if (!IbiFile.empty())
69 {
70 if (IbiStream)
71 IbiStream->Infos.clear(); //TODO: support IBI data from different inputs
72 else
73 IbiStream=new ibi::stream;
74
75 File_Ibi MI;
76 Open_Buffer_Init(&MI, IbiFile.size());
77 MI.Ibi=new ibi;
78 MI.Open_Buffer_Continue((const int8u*)IbiFile.c_str(), IbiFile.size());
79 if (!MI.Ibi->Streams.empty())
80 (*IbiStream)=(*MI.Ibi->Streams.begin()->second);
81 }
82
83 if (IbiStream && !IbiStream->Infos.empty() && IbiStream->Infos[IbiStream->Infos.size()-1].IsContinuous && IbiStream->Infos[IbiStream->Infos.size()-1].FrameNumber!=(int64u)-1)
84 Fill(Stream_Video, 0, Video_FrameCount, IbiStream->Infos[IbiStream->Infos.size()-1].FrameNumber);
85 }
86 #endif //MEDIAINFO_IBIUSAGE
87 }
88
89 Streams_Finish_StreamOnly();
90 Streams_Finish_StreamOnly();
91 Streams_Finish_InterStreams();
92 Streams_Finish_StreamOnly();
93 Streams_Finish_InterStreams();
94 Streams_Finish_StreamOnly();
95 Streams_Finish_InterStreams();
96 Streams_Finish_StreamOnly();
97
98 if (!IsSub && !Config->File_IsReferenced_Get() && MediaInfoLib::Config.ReadByHuman_Get())
99 Streams_Finish_HumanReadable();
100}

103void File__Analyze::TestContinuousFileNames(size_t CountOfFiles, Ztring FileExtension, bool SkipComputeDelay)
104{
105 if (IsSub || !Config->File_TestContinuousFileNames_Get())
106 return;
107
108 size_t Pos=Config->File_Names.size();
109 if (!Pos)
110 return;
111
112 //Trying to detect continuous file names (e.g. video stream as an image or HLS)
113 size_t Pos_Base = (size_t)-1;
114 bool AlreadyPresent=Config->File_Names.size()==1?true:false;
115 FileName FileToTest(Config->File_Names.Read(Config->File_Names.size()-1));
116 #ifdef WIN32
117 FileToTest.FindAndReplace(__T("/"), __T("\\"), 0, Ztring_Recursive); // "/" is sometimes used on Windows and it is considered as valid
118 #endif //WIN32
119 Ztring FileToTest_Name=FileToTest.Name_Get();
120 Ztring FileToTest_Name_After=FileToTest_Name;
121 size_t FileNameToTest_End=FileToTest_Name.size();
122 while (FileNameToTest_End && !(FileToTest_Name[FileNameToTest_End-1]>=__T('0') && FileToTest_Name[FileNameToTest_End-1]<=__T('9')))
123 FileNameToTest_End--;
124 size_t FileNameToTest_Pos=FileNameToTest_End;
125 while (FileNameToTest_Pos && FileToTest_Name[FileNameToTest_Pos-1]>=__T('0') && FileToTest_Name[FileNameToTest_Pos-1]<=__T('9'))
126 FileNameToTest_Pos--;
127 if (FileNameToTest_Pos!=FileToTest_Name.size() && FileNameToTest_Pos!=FileNameToTest_End)
128 {
129 size_t Numbers_Size=FileNameToTest_End-FileNameToTest_Pos;
130 int64u Pos=Ztring(FileToTest_Name.substr(FileNameToTest_Pos)).To_int64u();
131 FileToTest_Name.resize(FileNameToTest_Pos);
132 FileToTest_Name_After.erase(0, FileToTest_Name.size()+Numbers_Size);
133
134 /*
135 for (;;)
136 {
137 Pos++;
138 Ztring Pos_Ztring; Pos_Ztring.From_Number(Pos);
139 if (Numbers_Size>Pos_Ztring.size())
140 Pos_Ztring.insert(0, Numbers_Size-Pos_Ztring.size(), __T('0'));
141 Ztring Next=FileToTest.Path_Get()+PathSeparator+FileToTest_Name+Pos_Ztring+__T('.')+(FileExtension.empty()?FileToTest.Extension_Get():FileExtension);
142 if (!File::Exists(Next))
143 break;
144 Config->File_Names.push_back(Next);
145 }
146 */
147
148 //Detecting with a smarter algo (but missing frames are not detected)
149 Ztring FileToTest_Name_Begin=FileToTest.Path_Get()+PathSeparator+FileToTest_Name;
150 Ztring FileToTest_Name_End=FileToTest_Name_After+__T('.')+(FileExtension.empty()?FileToTest.Extension_Get():FileExtension);
151 Pos_Base = (size_t)Pos;
152 size_t Pos_Add_Max = 1;
153 #if MEDIAINFO_ADVANCED
154 bool File_IgnoreSequenceFileSize=Config->File_IgnoreSequenceFilesCount_Get(); //TODO: double check if it is expected
155
156 size_t SequenceFileSkipFrames=Config->File_SequenceFilesSkipFrames_Get();
157 if (SequenceFileSkipFrames)
158 {
159 for (;;)
160 {
161 size_t Pos_Add_Max_Old=Pos_Add_Max;
162 for (size_t TempPos=Pos_Add_Max; TempPos<=Pos_Add_Max+SequenceFileSkipFrames; TempPos++)
163 {
164 Ztring Pos_Ztring; Pos_Ztring.From_Number(Pos_Base+TempPos);
165 if (Numbers_Size>Pos_Ztring.size())
166 Pos_Ztring.insert(0, Numbers_Size-Pos_Ztring.size(), __T('0'));
167 Ztring Next=FileToTest_Name_Begin+Pos_Ztring+FileToTest_Name_End;
168 if (File::Exists(Next))
169 {
170 Pos_Add_Max=TempPos+1;
171 break;
172 }
173 }
174 if (Pos_Add_Max==Pos_Add_Max_Old)
175 break;
176 }
177 }
178 else
179 {
180 #endif //MEDIAINFO_ADVANCED
181 for (;;)
182 {
183 Ztring Pos_Ztring; Pos_Ztring.From_Number(Pos_Base+Pos_Add_Max);
184 if (Numbers_Size>Pos_Ztring.size())
185 Pos_Ztring.insert(0, Numbers_Size-Pos_Ztring.size(), __T('0'));
186 Ztring Next=FileToTest_Name_Begin+Pos_Ztring+FileToTest_Name_End;
187 if (!File::Exists(Next))
188 break;
189 Pos_Add_Max<<=1;
190 #if MEDIAINFO_ADVANCED
191 if (File_IgnoreSequenceFileSize && Pos_Add_Max>=CountOfFiles)
192 break;
193 #endif //MEDIAINFO_ADVANCED
194 }
195 size_t Pos_Add_Min = Pos_Add_Max >> 1;
196 while (Pos_Add_Min+1<Pos_Add_Max)
197 {
198 size_t Pos_Add_Middle = Pos_Add_Min + ((Pos_Add_Max - Pos_Add_Min) >> 1);
199 Ztring Pos_Ztring; Pos_Ztring.From_Number(Pos_Base+Pos_Add_Middle);
200 if (Numbers_Size>Pos_Ztring.size())
201 Pos_Ztring.insert(0, Numbers_Size-Pos_Ztring.size(), __T('0'));
202 Ztring Next=FileToTest_Name_Begin+Pos_Ztring+FileToTest_Name_End;
203 if (File::Exists(Next))
204 Pos_Add_Min=Pos_Add_Middle;
205 else
206 Pos_Add_Max=Pos_Add_Middle;
207 }
208
209 #if MEDIAINFO_ADVANCED
210 } //SequenceFileSkipFrames
211 #endif //MEDIAINFO_ADVANCED
212
213 size_t Pos_Max = Pos_Base + Pos_Add_Max;
214 Config->File_Names.reserve(Pos_Add_Max);
215 for (Pos=Pos_Base+1; Pos<Pos_Max; ++Pos)
216 {
217 Ztring Pos_Ztring; Pos_Ztring.From_Number(Pos);
218 if (Numbers_Size>Pos_Ztring.size())
219 Pos_Ztring.insert(0, Numbers_Size-Pos_Ztring.size(), __T('0'));
220 Config->File_Names.push_back(FileToTest_Name_Begin+Pos_Ztring+FileToTest_Name_End);
221 }
222
223 if (!Config->File_IsReferenced_Get() && Config->File_Names.size()<CountOfFiles && AlreadyPresent)
224 Config->File_Names.resize(1); //Removing files, wrong detection
225 }
226
227 if (Config->File_Names.size()==Pos)
228 return;
229
230 Config->File_IsImageSequence=true;
231 Frame_Count_NotParsedIncluded=Pos_Base;
232 #if MEDIAINFO_DEMUX
233 float64 Demux_Rate=Config->Demux_Rate_Get();
234 if (!Demux_Rate)
235 Demux_Rate=24;
236 if (!SkipComputeDelay)
237 Fill(Stream_Video, 0, Video_Delay, float64_int64s(Frame_Count_NotParsedIncluded*1000/Demux_Rate));
238 #endif //MEDIAINFO_DEMUX
239
240 #if MEDIAINFO_ADVANCED
241 if (!Config->File_IgnoreSequenceFileSize_Get() || Config->File_Names.size()<=1)
242 #endif //MEDIAINFO_ADVANCED
243 {
244 for (; Pos<Config->File_Names.size(); Pos++)
245 {
246 int64u Size=File::Size_Get(Config->File_Names[Pos]);
247 Config->File_Sizes.push_back(Size);
248 Config->File_Size+=Size;
249 }
250 }
251 #if MEDIAINFO_ADVANCED
252 else
253 {
254 Config->File_Size=(int64u)-1;
255 File_Size=(int64u)-1;
256 Clear(Stream_General, 0, General_FileSize);
257 }
258 #endif //MEDIAINFO_ADVANCED
259
260 File_Size=Config->File_Size;
261 Element[0].Next=File_Size;
262 #if MEDIAINFO_ADVANCED
263 if (!Config->File_IgnoreSequenceFileSize_Get() || Config->File_Names.size()<=1)
264 #endif //MEDIAINFO_ADVANCED
265 Fill (Stream_General, 0, General_FileSize, File_Size, 10, true);
266 #if MEDIAINFO_ADVANCED
267 if (!Config->File_IgnoreSequenceFilesCount_Get())
268 #endif //MEDIAINFO_ADVANCED
269 {
270 Fill (Stream_General, 0, General_CompleteName_Last, Config->File_Names[Config->File_Names.size()-1], true);
271 Fill (Stream_General, 0, General_FolderName_Last, FileName::Path_Get(Config->File_Names[Config->File_Names.size()-1]), true);
272 Fill (Stream_General, 0, General_FileName_Last, FileName::Name_Get(Config->File_Names[Config->File_Names.size()-1]), true);
273 Fill (Stream_General, 0, General_FileExtension_Last, FileName::Extension_Get(Config->File_Names[Config->File_Names.size()-1]), true);
274 }
275
276 #if MEDIAINFO_ADVANCED
277 if (Config->File_Source_List_Get())
278 {
279 Ztring SourcePath=FileName::Path_Get(Retrieve(Stream_General, 0, General_CompleteName));
280 size_t SourcePath_Size=SourcePath.size()+1; //Path size + path separator size
281 for (size_t Pos=0; Pos<Config->File_Names.size(); Pos++)
282 {
283 Ztring Temp=Config->File_Names[Pos];
284 Temp.erase(0, SourcePath_Size);
285 Fill(Stream_General, 0, "Source_List", Temp);
286 }
287 (*Stream_More)[Stream_General][0](Ztring().From_Local("Source_List"), Info_Options)=__T("N NT");
288 }
289 #endif //MEDIAINFO_ADVANCED
290}

310//---------------------------------------------------------------------------
311void File__Analyze::Streams_Finish_StreamOnly(stream_t StreamKind, size_t Pos)
312{
313 //BitRate from Duration and StreamSize
314 if (StreamKind!=Stream_General && StreamKind!=Stream_Other && StreamKind!=Stream_Menu && Retrieve(StreamKind, Pos, "BitRate").empty() && !Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_StreamSize)).empty() && !Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_Duration)).empty())
315 {
316 float64 Duration=0;
317 if (StreamKind==Stream_Video && !Retrieve(Stream_Video, Pos, Video_FrameCount).empty() && !Retrieve(Stream_Video, Pos, Video_FrameRate).empty())
318 {
319 int64u FrameCount=Retrieve(Stream_Video, Pos, Video_FrameCount).To_int64u();
320 float64 FrameRate=Retrieve(Stream_Video, Pos, Video_FrameRate).To_float64();
321 if (FrameCount && FrameRate)
322 Duration=FrameCount*1000/FrameRate; //More precise (example: 1 frame at 29.97 fps)
323 }
324 if (Duration==0)
325 Duration=Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_Duration)).To_float64();
326 int64u StreamSize=Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_StreamSize)).To_int64u();
327 if (Duration>0 && StreamSize>0)
328 Fill(StreamKind, Pos, "BitRate", StreamSize*8*1000/Duration, 0);
329 }
330
331 //BitRate_Encoded from Duration and StreamSize_Encoded
332 if (StreamKind!=Stream_General && StreamKind!=Stream_Other && StreamKind!=Stream_Menu && Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_BitRate_Encoded)).empty() && !Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_StreamSize_Encoded)).empty() && !Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_Duration)).empty())
333 {
334 float64 Duration=0;
335 if (StreamKind==Stream_Video && !Retrieve(Stream_Video, Pos, Video_FrameCount).empty() && !Retrieve(Stream_Video, Pos, Video_FrameRate).empty())
336 {
337 int64u FrameCount=Retrieve(Stream_Video, Pos, Video_FrameCount).To_int64u();
338 float64 FrameRate=Retrieve(Stream_Video, Pos, Video_FrameRate).To_float64();
339 if (FrameCount && FrameRate)
340 Duration=FrameCount*1000/FrameRate; //More precise (example: 1 frame at 29.97 fps)
341 }
342 if (Duration==0)
343 Duration=Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_Duration)).To_float64();
344 int64u StreamSize_Encoded=Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_StreamSize_Encoded)).To_int64u();
345 if (Duration>0 && StreamSize_Encoded>0)
346 Fill(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_BitRate_Encoded), StreamSize_Encoded*8*1000/Duration, 0);
347 }
348
349 //Duration from Bitrate and StreamSize
350 if (StreamKind!=Stream_Other && Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_Duration)).empty() && !Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_StreamSize)).empty() && !Retrieve(StreamKind, Pos, "BitRate").empty() && Count_Get(Stream_Video)+Count_Get(Stream_Audio)>1) //If only one stream, duration will be copied later, useful for exact bitrate calculation
351 {
352 int64u BitRate=Retrieve(StreamKind, Pos, "BitRate").To_int64u();
353 int64u StreamSize=Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_StreamSize)).To_int64u();
354 if (BitRate>0 && StreamSize>0)
355 Fill(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_Duration), StreamSize*8*1000/BitRate);
356 }
357
358 //StreamSize from BitRate and Duration
359 if (StreamKind!=Stream_Other && Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_StreamSize)).empty() && !Retrieve(StreamKind, Pos, "BitRate").empty() && !Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_Duration)).empty() && Retrieve(StreamKind, Pos, "BitRate").find(__T(" / "))==std::string::npos) //If not done the first time or by other routine
360 {
361 int64u BitRate=Retrieve(StreamKind, Pos, "BitRate").To_int64u();
362 int64u Duration=Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_Duration)).To_int64u();
363 if (BitRate>0 && Duration>0)
364 Fill(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_StreamSize), BitRate*Duration/8/1000);
365 }
366
367 //Bit rate and maximum bit rate
368 if (!Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_BitRate)).empty() && Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_BitRate))==Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_BitRate_Maximum)))
369 {
370 Clear(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_BitRate_Maximum));
371 if (Retrieve(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_BitRate_Mode)).empty())
372 Fill(StreamKind, Pos, Fill_Parameter(StreamKind, Generic_BitRate_Mode), "CBR");
373 }
374}

381//---------------------------------------------------------------------------
382void File__Analyze::Streams_Finish_StreamOnly_Video(size_t Pos)
383{
384 //Frame count
385 if (Retrieve(Stream_Video, Pos, Video_FrameCount).empty() && Frame_Count_NotParsedIncluded!=(int64u)-1 && File_Offset+Buffer_Size==File_Size)
386 {
387 if (Count_Get(Stream_Video)==1 && Count_Get(Stream_Audio)==0)
388 Fill(Stream_Video, 0, Video_FrameCount, Frame_Count_NotParsedIncluded);
389 }
390
391 //FrameCount from Duration and FrameRate
392 if (Retrieve(Stream_Video, Pos, Video_FrameCount).empty())
393 {
394 int64s Duration=Retrieve(Stream_Video, Pos, Video_Duration).To_int64s();
395 if (Duration==0)
396 Duration=Retrieve(Stream_General, 0, General_Duration).To_int64s();
397 float64 FrameRate=Retrieve(Stream_Video, Pos, Video_FrameRate).To_float64();
398 if (Duration && FrameRate)
399 Fill(Stream_Video, Pos, Video_FrameCount, Duration*FrameRate/1000, 0);
400 }
401
402 //Duration from FrameCount and FrameRate
403 if (Retrieve(Stream_Video, Pos, Video_Duration).empty())
404 {
405 int64u FrameCount=Retrieve(Stream_Video, Pos, Video_FrameCount).To_int64u();
406 float64 FrameRate=Retrieve(Stream_Video, Pos, "FrameRate").To_float64();
407 if (FrameCount && FrameRate)
408 Fill(Stream_Video, Pos, Video_Duration, FrameCount/FrameRate*1000, 0);
409 }
410
411 //FrameRate from FrameCount and Duration
412 if (Retrieve(Stream_Video, Pos, Video_FrameRate).empty())
413 {
414 int64u FrameCount=Retrieve(Stream_Video, Pos, Video_FrameCount).To_int64u();
415 float64 Duration=Retrieve(Stream_Video, Pos, Video_Duration).To_float64()/1000;
416 if (FrameCount && Duration)
417 Fill(Stream_Video, Pos, Video_FrameRate, FrameCount/Duration, 3);
418 }
419
420 //Pixel Aspect Ratio forced to 1.000 if none
421 if (Retrieve(Stream_Video, Pos, Video_PixelAspectRatio).empty())
422 Fill(Stream_Video, Pos, Video_PixelAspectRatio, 1.000);
423
424 //Standard
425 if (Retrieve(Stream_Video, Pos, Video_Standard).empty() && (Retrieve(Stream_Video, Pos, Video_Width)==__T("720") || Retrieve(Stream_Video, Pos, Video_Width)==__T("704")))
426 {
427 if (Retrieve(Stream_Video, Pos, Video_Height)==__T("576") && Retrieve(Stream_Video, Pos, Video_FrameRate)==__T("25.000"))
428 Fill(Stream_Video, Pos, Video_Standard, "PAL");
429 else if ((Retrieve(Stream_Video, Pos, Video_Height)==__T("486") || Retrieve(Stream_Video, Pos, Video_Height)==__T("480")) && Retrieve(Stream_Video, Pos, Video_FrameRate)==__T("29.970"))
430 Fill(Stream_Video, Pos, Video_Standard, "NTSC");
431 }
432 if (Retrieve(Stream_Video, Pos, Video_Standard).empty() && Retrieve(Stream_Video, Pos, Video_Width)==__T("352"))
433 {
434 if ((Retrieve(Stream_Video, Pos, Video_Height)==__T("576") || Retrieve(Stream_Video, Pos, Video_Height)==__T("288")) && Retrieve(Stream_Video, Pos, Video_FrameRate)==__T("25.000"))
435 Fill(Stream_Video, Pos, Video_Standard, "PAL");
436 else if ((Retrieve(Stream_Video, Pos, Video_Height)==__T("486") || Retrieve(Stream_Video, Pos, Video_Height)==__T("480") || Retrieve(Stream_Video, Pos, Video_Height)==__T("243") || Retrieve(Stream_Video, Pos, Video_Height)==__T("240")) && Retrieve(Stream_Video, Pos, Video_FrameRate)==__T("29.970"))
437 Fill(Stream_Video, Pos, Video_Standard, "NTSC");
438 }
439
440 //Known ScanTypes
441 if (Retrieve(Stream_Video, Pos, Video_ScanType).empty()
442 && (Retrieve(Stream_Video, Pos, Video_Format)==__T("RED")
443 || Retrieve(Stream_Video, Pos, Video_Format)==__T("CineForm")
444 || Retrieve(Stream_Video, Pos, Video_Format)==__T("DPX")
445 || Retrieve(Stream_Video, Pos, Video_Format)==__T("EXR")))
446 Fill(Stream_Video, Pos, Video_ScanType, "Progressive");
447
448 //Useless chroma subsampling
449 if (Retrieve(Stream_Video, Pos, Video_ColorSpace)==__T("RGB")
450 && Retrieve(Stream_Video, Pos, Video_ChromaSubsampling)==__T("4:4:4"))
451 Clear(Stream_Video, Pos, Video_ChromaSubsampling);
452
453 //Chroma subsampling position
454 if (Retrieve(Stream_Video, Pos, Video_ChromaSubsampling_String).empty() && !Retrieve(Stream_Video, Pos, Video_ChromaSubsampling).empty())
455 {
456 if (Retrieve(Stream_Video, Pos, Video_ChromaSubsampling_Position).empty())
457 Fill(Stream_Video, Pos, Video_ChromaSubsampling_String, Retrieve(Stream_Video, Pos, Video_ChromaSubsampling));
458 else
459 Fill(Stream_Video, Pos, Video_ChromaSubsampling_String, Retrieve(Stream_Video, Pos, Video_ChromaSubsampling)+__T(" (")+ Retrieve(Stream_Video, Pos, Video_ChromaSubsampling_Position)+__T(')'));
460 }
461
462 //Commercial name
463 #if defined(MEDIAINFO_VC3_YES)
464 if (Retrieve(Stream_Video, Pos, Video_Format_Commercial_IfAny).empty() && Retrieve(Stream_Video, Pos, Video_Format)==__T("VC-3"))
465 {
466 //http://www.avid.com/static/resources/US/documents/dnxhd.pdf
467 int64u Height=Retrieve(Stream_Video, Pos, Video_Height).To_int64u();
468 int64u BitRate=float64_int64s(Retrieve(Stream_Video, Pos, Video_BitRate).To_float64()/1000000);
469 int64u FrameRate=float64_int64s(Retrieve(Stream_Video, Pos, Video_FrameRate).To_float64());
470 int64u BitRate_Final=0;
471 if (Height>=900 && Height<=1300)
472 {
473 if (FrameRate==60)
474 {
475 if (BitRate>=420 && BitRate<440) //440
476 BitRate_Final=440;
477 if (BitRate>=271 && BitRate<311) //291
478 BitRate_Final=290;
479 if (BitRate>=80 && BitRate<100) //90
480 BitRate_Final=90;
481 }
482 if (FrameRate==50)
483 {
484 if (BitRate>=347 && BitRate<387) //367
485 BitRate_Final=365;
486 if (BitRate>=222 && BitRate<262) //242
487 BitRate_Final=240;
488 if (BitRate>=65 && BitRate<85) //75
489 BitRate_Final=75;
490 }
491 if (FrameRate==30)
492 {
493 if (BitRate>=420 && BitRate<440) //440
494 BitRate_Final=440;
495 if (BitRate>=200 && BitRate<240) //220
496 BitRate_Final=220;
497 if (BitRate>=130 && BitRate<160) //145
498 BitRate_Final=145;
499 if (BitRate>=90 && BitRate<110) //100
