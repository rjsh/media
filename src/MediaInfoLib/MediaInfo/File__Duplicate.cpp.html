66}
67
68bool File__Duplicate::File__Duplicate_HasChanged ()
69{
70 //Retrieving general configuration
71 while (Config->File_Duplicate_Get_AlwaysNeeded(Config_File_Duplicate_Get_AlwaysNeeded_Count))
72 {
73 if (File__Duplicate_Set(Config->File_Duplicate_Get(Config_File_Duplicate_Get_AlwaysNeeded_Count)))
74 File__Duplicate_HasChanged_=true;
75 Config_File_Duplicate_Get_AlwaysNeeded_Count++;
76 }
77
78 bool File__Duplicate_HasChanged_Temp=File__Duplicate_HasChanged_;
79 File__Duplicate_HasChanged_=false;
80 return File__Duplicate_HasChanged_Temp;
81}
