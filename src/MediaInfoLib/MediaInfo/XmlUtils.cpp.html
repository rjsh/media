29// Finds the value of the named attribute within the given element or its
30// closest ancestor
31static const char *AncestorAttrVal(const XMLElement* Elem, const char* AttrName)
32{
33 if (!Elem)
34 return NULL;
35 if (const XMLAttribute* Attr=Elem->FindAttribute(AttrName))
36 return Attr->Value();
37 const XMLElement* ParentElem=NULL;
38 if (const XMLNode* ParentNode=Elem->Parent())
39 ParentElem=ParentNode->ToElement();
40 return AncestorAttrVal(ParentElem, AttrName);
41}

48const char *LocalName(const XMLElement* Elem)
49{
50 const char* Name=Elem->Name();
51 const char* Colon=strrchr(Name, ':');
52 return Colon ? Colon+1 : Name;
53}
54
55//---------------------------------------------------------------------------
56
57const char *LocalName(const XMLElement* Elem, const char* &Ns /* out */)
58{
59 const char* Name=Elem->Name();
60 if (const char* Colon=strrchr(Name, ':'))
61 {
62 // Search element and ancestors for given namespace prefix
63 Ns=AncestorAttrVal(Elem, ("xmlns:"+std::string(Name, Colon)).c_str());
64 // NB no default namespace if given prefix not found, will return NULL
65 return Colon+1;
66 }
67 else
68 {
69 Ns=AncestorAttrVal(Elem, "xmlns");
70 if (!Ns)
71 Ns = ""; // default namespace
72 return Name;
73 }
74}
75
76//---------------------------------------------------------------------------
77
78bool MatchQName(const XMLElement* Elem, const char* Name, const char* NameSpace)
79{
80 if (strcmp(LocalName(Elem), Name))
81 return false; // exit cheaply if local name doesn't match
82 const char *ElemNs;
83 LocalName(Elem, ElemNs); // look up namespace
84 return ElemNs && // ElemNs can be null if Elem uses a bad namespace prefix
85 !strcmp(ElemNs, NameSpace); // compare namespaces
86}
