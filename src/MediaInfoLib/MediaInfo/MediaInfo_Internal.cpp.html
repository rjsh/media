325size_t MediaInfo_Internal::Open(const String &File_Name_)
326{
327 Close();
328
329 //External IBI
330 #if MEDIAINFO_IBIUSAGE
331 if (Config.Ibi_UseIbiInfoIfAvailable_Get())
332 {
333 std::string IbiFile=Config.Ibi_Get();
334 if (!IbiFile.empty())
335 {
336 Info=new File_Ibi();
337 Open_Buffer_Init(IbiFile.size(), File_Name_);
338 Open_Buffer_Continue((const int8u*)IbiFile.c_str(), IbiFile.size());
339 Open_Buffer_Finalize();
340
341 if (!Get(Stream_General, 0, __T("Format")).empty() && Get(Stream_General, 0, __T("Format"))!=__T("Ibi"))
342 return 1;
343
344 //Nothing interesting
345 Close();
346 }
347 }
348 #endif //MEDIAINFO_IBIUSAGE
349
350 CS.Enter();
351 MEDIAINFO_DEBUG_CONFIG_TEXT(Debug+=__T("Open, File=");Debug+=Ztring(File_Name_).c_str();)
352 Config.File_Names.clear();
353 if (Config.File_FileNameFormat_Get()==__T("CSV"))
354 {
355 Config.File_Names.Separator_Set(0, __T(","));
356 Config.File_Names.Write(File_Name_);
357 }
358 else if (!File_Name_.empty())
359 Config.File_Names.push_back(File_Name_);
360 if (Config.File_Names.empty())
361 {
362 CS.Leave();
363 return 0;
364 }
365 Config.File_Names_Pos=1;
366 CS.Leave();
367
368 //Parsing
369 if (BlockMethod==1)
370 {
371 if (!IsInThread) //If already created, the routine will read the new files
372 {
373 Run();
374 IsInThread=true;
375 }
376 return 0;
377 }
378 else
379 {
380 Entry(); //Normal parsing
381 return Count_Get(Stream_General);
382 }
383}

926 #endif //MEDIAINFO_TRACE

1202// Information
1203//***************************************************************************
1204
1205//---------------------------------------------------------------------------
1206String MediaInfo_Internal::Option (const String &Option, const String &Value)
1207{
1208 CriticalSectionLocker CSL(CS);
1209 MEDIAINFO_DEBUG_CONFIG_TEXT(Debug+=__T("Option, Option=");Debug+=Ztring(Option);Debug+=__T(", Value=");Debug+=Ztring(Value);)
1210 Ztring OptionLower=Option; OptionLower.MakeLowerCase();
1211 if (Option.empty())
1212 return __T("");
1213 else if (OptionLower==__T("language_update"))
1214 {
1215 if (!Info || Info->Get(Stream_General, 0, __T("CompleteName"))==__T(""))
1216 return __T("");
1217
1218 ZtringListList Language=Value.c_str();
1219 MediaInfoLib::Config.Language_Set(Language);
1220
1221 return __T("");
1222 }
1223 else if (OptionLower==__T("create_dummy"))
1224 {
1225 CreateDummy (Value);
1226 delete Info; Info=NULL;
1227 return __T("");
1228 }
1229 else if (OptionLower==__T("thread"))
1230 {
1231 BlockMethod=1;
1232 return __T("");
1233 }
1234 else if (Option==__T("info_capacities"))
1235 {
1236 return __T("Option removed");
1237 }
1238 #if MEDIAINFO_TRACE
1239 else if (OptionLower.find(__T("file_details_clear"))==0)
1240 {
1241 if (Info)
1242 Info->Details_Clear();
1243
1244 return __T("");
1245 }
1246 #endif //MEDIAINFO_TRACE
1247 else if (OptionLower.find(__T("file_seek"))==0)
1248 {
1249 #if MEDIAINFO_SEEK
1250 if (Reader==NULL && Info==NULL)
1251 return __T("Error: Reader pointer is empty");
1252
1253 size_t Method=(size_t)-1;
1254 int64u SeekValue=(int64u)-1;
1255 int64u ID=(int64u)-1;
1256
1257 ZtringList List; List.Separator_Set(0, __T(","));
1258 List.Write(Value);
1259 for (size_t Pos=0; Pos<List.size(); Pos++)
1260 {
1261 if (!List[Pos].empty() && List[Pos].find(__T('%'))==List[Pos].size()-1)
1262 {
1263 Method=1;
1264 SeekValue=(int64u)(Ztring(List[Pos]).To_float32()*100);
1265 }
1266 else if (!List[Pos].empty() && List[Pos].find_first_not_of(__T("0123456789"))==string::npos)
1267 {
1268 Method=0;
1269 SeekValue=Ztring(List[Pos]).To_int64u();
1270 }
1271 else if (!List[Pos].empty() && List[Pos].find(__T("Frame="))!=string::npos)
1272 {
1273 Method=3;
1274 Ztring FrameNumberZ=List[Pos].substr(List[Pos].find(__T("Frame="))+6, string::npos);
1275 SeekValue=FrameNumberZ.To_int64u();
1276 }
1277 else if (!List[Pos].empty() && List[Pos].find(__T(":"))!=string::npos)
1278 {
1279 Method=2;
1280 Ztring ValueZ=List[Pos];
1281 SeekValue=0;
1282 size_t Value_Pos=ValueZ.find(__T(":"));
1283 if (Value_Pos==string::npos)
1284 Value_Pos=ValueZ.size();
1285 SeekValue+=Ztring(ValueZ.substr(0, Value_Pos)).To_int64u()*60*60*1000*1000*1000;
1286 ValueZ.erase(0, Value_Pos+1);
1287 Value_Pos=ValueZ.find(__T(":"));
1288 if (Value_Pos==string::npos)
1289 Value_Pos=ValueZ.size();
1290 SeekValue+=Ztring(ValueZ.substr(0, Value_Pos)).To_int64u()*60*1000*1000*1000;
1291 ValueZ.erase(0, Value_Pos+1);
1292 Value_Pos=ValueZ.find(__T("."));
1293 if (Value_Pos==string::npos)
1294 Value_Pos=ValueZ.size();
1295 SeekValue+=Ztring(ValueZ.substr(0, Value_Pos)).To_int64u()*1000*1000*1000;
1296 ValueZ.erase(0, Value_Pos+1);
1297 if (!ValueZ.empty())
1298 SeekValue+=Ztring(ValueZ).To_int64u()*1000*1000*1000/(int64u)pow(10.0, (int)ValueZ.size());
1299 }
1300 else if (!List[Pos].empty() && List[Pos].find(__T("ID="))!=string::npos)
1301 {
1302 Ztring IDZ=List[Pos].substr(List[Pos].find(__T("ID="))+3, string::npos);
1303 ID=IDZ.To_int64u();
1304 }
1305 }
1306
1307 CS.Leave();
1308 size_t Result;
1309 if (Reader)
1310 Result=Reader->Format_Test_PerParser_Seek(this, Method, SeekValue, ID);
1311 else
1312 Result=Open_Buffer_Seek(Method, SeekValue, ID);
1313 CS.Enter();
1314 switch (Result)
1315 {
1316 case 1 : return __T("");
1317 case 2 : return __T("Invalid value");
1318 #if MEDIAINFO_IBIUSAGE
1319 case 3 : return __T("Feature not supported / IBI file not provided");
1320 case 4 : return __T("Problem during IBI file parsing");
1321 #endif //MEDIAINFO_IBIUSAGE
1322 case 5 : return __T("Invalid ID");
1323 case 6 : return __T("Internal error");
1324 #if !MEDIAINFO_IBIUSAGE
1325 case (size_t)-2 : return __T("Feature not supported / IBI support disabled due to compilation options");
1326 #endif //MEDIAINFO_IBIUSAGE
1327 case (size_t)-1 : return __T("Feature not supported");
1328 default : return __T("Unknown error");
1329 }
1330 #else //MEDIAINFO_SEEK
1331 return __T("Seek manager is disabled due to compilation options");
1332 #endif //MEDIAINFO_SEEK
1333 }
1334 #if MEDIAINFO_TRACE
1335 if (OptionLower.find(__T("file_details_stringpointer")) == 0 && MediaInfoLib::Config.Inform_Get()!=__T("MAXML") && (MediaInfoLib::Config.Trace_Level_Get() || MediaInfoLib::Config.Inform_Get()==__T("Details")) && !Details.empty())
1336 {
1337 return Ztring::ToZtring((int64u)Details.data())+__T(':')+Ztring::ToZtring((int64u)Details.size());
1338 }
1339 #endif //MEDIAINFO_TRACE
1340 else if (OptionLower.find(__T("file_"))==0)
1341 {
1342 Ztring ToReturn2=Config.Option(Option, Value);
1343 if (Info)
1344 Info->Option_Manage();
1345
1346 MEDIAINFO_DEBUG_OUTPUT_INIT(ToReturn2, Debug+=__T("Option, will return ");Debug+=ToReturn;)
1347 }
1348 else
1349 EXECUTE_STRING(MediaInfoLib::Config.Option(Option, Value), Debug+=__T("Option, will return ");Debug+=ToReturn;)
1350}
