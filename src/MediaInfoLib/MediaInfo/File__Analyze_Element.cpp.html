36element_details::Element_Node_Data& element_details::Element_Node_Data::operator=(const Element_Node_Data& v)
37{
38 if (this == &v)
39 return *this;
40
41 clear();
42
43 type = v.type;
44 switch (type)
45 {
46 case ELEMENT_NODE_STR:
47 {
48 size_t len = strlen(v.val.Str);
49 val.Str = new char[len + 1];
50 std::memcpy(val.Str, v.val.Str, len);
51 val.Str[len] = '\0';
52 break;
53 }
54 case element_details::Element_Node_Data::ELEMENT_NODE_FLOAT80:
55 val.f80 = new float80;
56 *val.f80 = *v.val.f80;
57 break;
58 case ELEMENT_NODE_INT128U:
59 val.i128u = new int128u;
60 *val.i128u = *v.val.i128u;
61 break;
62 default:
63 val = v.val;
64 break;
65 }
66
67 format_out = v.format_out;
68 Option = v.Option;
69
70 return *this;
71}

291//---------------------------------------------------------------------------
292void element_details::Element_Node_Data::get_hexa_from_deci_limited_by_bits(std::string& val, int8u bits, int8u default_bits)
293{
294 if (bits == (int8u)-1)
295 bits = default_bits;
296
297 std::string zeros;
298 bool full = (bits % 4 == 0);
299 int nb_zero = (bits / 4) + (full?0:1);
300 nb_zero -= val.size();
301 if (nb_zero > 0)
302 zeros.resize(nb_zero, '0');
303 val = zeros + val;
304}

306//---------------------------------------------------------------------------
307static size_t Xml_Name_Escape_MustEscape(const char* Name, size_t Size)
308{
309 // Case it is empty
310 if (!Size)
311 return (size_t)-1;
312
313 // Case first char is a digit
314 if ((Name[0] >= '0' && Name[0] <= '9'))
315 return 0;
316
317 // Cheking all chars
318 for (size_t Pos = 0; Pos < Size; Pos++)
319 {
320 const char C = Name[Pos];
321 switch (C)
322 {
323 case ' ':
324 case '/':
325 case '(':
326 case ')':
327 case '*':
328 case ',':
329 case ':':
330 case '@':
331 return Pos;
332 default:
333 if (!(C >= 'A' && C <= 'Z')
334 && !(C >= 'a' && C <= 'z')
335 && !(C >= '0' && C <= '9')
336 && !(C == '_'))
337 return Pos;
338 }
339 }
340
341 return (size_t)-1;
342}

344//---------------------------------------------------------------------------
345static inline size_t Xml_Name_Escape_MustEscape(const std::string &Name)
346{
347 return Xml_Name_Escape_MustEscape(Name.c_str(), Name.size());
348}

350//---------------------------------------------------------------------------
351static void Xml_Name_Escape(const char* Name, size_t Size, std::string& ToReturn, size_t Pos)
352{
353 ToReturn = Name;
354
355 // Case first char is a digit
356 if (Pos == 0 && Name[0] >= '0' && Name[0] <= '9')
357 ToReturn.insert(0, 1, '_');
358
359 while (Pos < Size)
360 {
361 const char C = ToReturn[Pos];
362
363 switch (C)
364 {
365 case ' ':
366 case '/':
367 case '(':
368 case ')':
369 case '*':
370 case ',':
371 case ':':
372 case '@':
373 ToReturn[Pos] = '_';
374 Pos++;
375 break;
376 default:
377 if (!(C >= 'A' && C <= 'Z')
378 && !(C >= 'a' && C <= 'z')
379 && !(C >= '0' && C <= '9')
380 && !(C == '_'))
381 {
382 ToReturn.erase(Pos, 1);
383 Size--;
384 }
385 else
386 Pos++;
387 }
388 }
389
390 if (ToReturn.empty())
391 ToReturn = "Unknown";
392}

396{
397 Xml_Name_Escape(Name.c_str(), Name.size(), ToReturn, Pos);
398}

400//---------------------------------------------------------------------------
401static size_t Xml_Content_Escape_MustEscape(const char* Content, size_t Size)
402{
403 // Cheking all chars
404 for (size_t Pos = 0; Pos < Size; Pos++)
405 {
406 const char C = Content[Pos];
407 switch (C)
408 {
409 case '\"':
410 case '&':
411 case '\'':
412 case '<':
413 case '>':
414 return Pos;
415 default:
416 if (C<0x20)
417 return Pos;
418 }
419 }
420
421 return (size_t)-1;
422}
423
424//---------------------------------------------------------------------------
425static inline size_t Xml_Content_Escape_MustEscape(const std::string &Content)
426{
427 return Xml_Content_Escape_MustEscape(Content.c_str(), Content.size());
428}

430//---------------------------------------------------------------------------
431static void Xml_Content_Escape(const char* Content, size_t Size, std::string& ToReturn, size_t Pos)
432{
433 ToReturn = Content;
434
435 for (; Pos<Size; Pos++)
436 {
437 const char C = ToReturn[Pos];
438 switch (C)
439 {
440 case '\"':
441 ToReturn[Pos] = '&';
442 ToReturn.insert(Pos + 1, "quot;");
443 Pos += 5;
444 Size += 5;
445 break;
446 case '&':
447 ToReturn[Pos] = '&';
448 ToReturn.insert(Pos + 1, "amp;");
449 Pos += 4;
450 Size += 4;
451 break;
452 case '\'':
453 ToReturn[Pos] = '&';
454 ToReturn.insert(Pos + 1, "apos;");
455 Pos += 5;
456 Size += 5;
457 break;
458 case '<':
459 ToReturn[Pos] = '&';
460 ToReturn.insert(Pos + 1, "lt;");
461 Pos += 3;
462 Size += 3;
463 break;
464 case '>':
465 ToReturn[Pos] = '&';
466 ToReturn.insert(Pos + 1, "gt;");
467 Pos += 3;
468 Size += 3;
469 break;
470 case '\r':
471 case '\n':
472 break;
473 default:
474 if (C<0x20)
475 {
476 ToReturn = Base64::encode(Content);
477 Pos = ToReturn.size(); //End
478 }
479 }
480 }
481}

485{
486 Xml_Content_Escape(Content.c_str(), Content.size(), ToReturn, Pos);
487}

505//---------------------------------------------------------------------------
506std::ostream& operator<<(std::ostream& os, const element_details::Element_Node_Data& v)
507{
508 switch (v.type)
509 {
510 case element_details::Element_Node_Data::ELEMENT_NODE_CHAR8:
511 {
512 size_t MustEscape = Xml_Content_Escape_MustEscape(v.val.Chars, v.Option);
513 if (MustEscape != (size_t)-1)
514 {
515 std::string str;
516 Xml_Content_Escape(v.val.Chars, v.Option, str, MustEscape);
517 os << str;
518 }
519 else
520 {
521 for (int8u i = 0; i < v.Option; ++i)
522 os.rdbuf()->sputc(v.val.Chars[i]);
523 }
524 break;
525 }
526 case element_details::Element_Node_Data::ELEMENT_NODE_STR:
527 {
528 size_t MustEscape = Xml_Content_Escape_MustEscape(v.val.Chars, v.Option);
529 if (MustEscape != (size_t)-1)
530 {
531 std::string str;
532 Xml_Content_Escape(v.val.Str, str, MustEscape);
533 os << str;
534 }
535 else
536 os << v.val.Str;
537 break;
538 }
539 case element_details::Element_Node_Data::ELEMENT_NODE_BOOL:
540 if (v.val.b)
541 os << "Yes";
542 else
543 os << "No";
544 break;
545 case element_details::Element_Node_Data::ELEMENT_NODE_INT8U:
546 VALUE_INTEGER_FORMATED(v.val.i8u, unsigned, 8); //use unsigned for stringstream
547 break;
548 case element_details::Element_Node_Data::ELEMENT_NODE_INT8S:
549 VALUE_INTEGER_FORMATED(v.val.i8s, int, 8); //use int for stringstream
550 break;
551
552 case element_details::Element_Node_Data::ELEMENT_NODE_INT16U:
553 VALUE_INTEGER_FORMATED(v.val.i16u, unsigned, 16); //use unsigned for stringstream
554 break;
555 case element_details::Element_Node_Data::ELEMENT_NODE_INT16S:
556 VALUE_INTEGER_FORMATED(v.val.i16s, int, 16); //use int for stringstream
557 break;
558
559 case element_details::Element_Node_Data::ELEMENT_NODE_INT32U:
560 VALUE_INTEGER_FORMATED(v.val.i32u, int32u, 32); //use unsigned for stringstream
561 break;
562 case element_details::Element_Node_Data::ELEMENT_NODE_INT32S:
563 VALUE_INTEGER_FORMATED(v.val.i32s, int32s, 32); //use int for stringstream
564 break;
565
566 case element_details::Element_Node_Data::ELEMENT_NODE_INT64U:
567 VALUE_INTEGER_FORMATED(v.val.i64u, int64u, 64);
568 break;
569 case element_details::Element_Node_Data::ELEMENT_NODE_INT64S:
570 VALUE_INTEGER_FORMATED(v.val.i64s, int64s, 64); //use int for stringstream
571 break;
572
573 case element_details::Element_Node_Data::ELEMENT_NODE_INT128U:
574 os << Ztring::ToZtring(*v.val.i128u).To_UTF8();
575 if (v.format_out == element_details::Element_Node_Data::Format_Tree)
576 {
577 std::string str = Ztring::ToZtring(*v.val.i128u, 16).To_UTF8();
578 element_details::Element_Node_Data::get_hexa_from_deci_limited_by_bits(str, v.Option, 128);
579 os << " (0x" << str << ")";
580 }
581 break;
582 case element_details::Element_Node_Data::ELEMENT_NODE_FLOAT32:
583 os << Ztring::ToZtring(v.val.f32, v.Option == (int8u)-1 ? 3 : v.Option).To_UTF8();
584 break;
585 case element_details::Element_Node_Data::ELEMENT_NODE_FLOAT64:
586 os << Ztring::ToZtring(v.val.f64, v.Option == (int8u)-1 ? 3 : v.Option).To_UTF8();
587 break;
588 case element_details::Element_Node_Data::ELEMENT_NODE_FLOAT80:
589 os << Ztring::ToZtring(*v.val.f80, v.Option == (int8u)-1 ? 3 : v.Option).To_UTF8();
590 break;
591 }
592 return os;
593}

690{

692 Size = 0;

694 {

700 for (size_t i = 0; i < Children.size(); ++i)
701 delete Children[i];
702 Children.clear();
703 if (OwnChildren && Infos.size())

705 delete Infos[i];
706 Infos.clear();
707 Current_Child = -1;
708 NoShow = false;
709 OwnChildren = true;
710 IsCat = false;
711}

713//---------------------------------------------------------------------------
714int element_details::Element_Node::Print_Micro_Xml(std::ostringstream& ss, size_t level)
715{
716 if (IsCat || !Name)
717 goto print_children;
718
719 if (Value.empty())
720 ss << "<b";
721 else
722 ss << "<d";
723
724 {
725 size_t MustEscape = Xml_Content_Escape_MustEscape(Name);
726 if (MustEscape != (size_t)-1)
727 {
728 std::string str;
729 Xml_Content_Escape(Name, str, MustEscape);
730 ss << " o=\"" << Pos << "\" n=\"" << str << "\"";
731 }
732 else
733 ss << " o=\"" << Pos << "\" n=\"" << Name << "\"";
734 }
735
736 for (size_t i = 0, info_nb = 0; i < Infos.size(); ++i)
737 {
738 Element_Node_Info* Info = Infos[i];
739
740 if (Info->Measure && !std::strcmp(Info->Measure, "Parser"))
741 {
742 if (!(Info->data == string()))
743 ss << " parser=\"" << Info->data << "\"";
744 continue;
745 }
746 else
747 info_nb++;
748
749 ss << " i";
750 if (info_nb > 1)
751 ss << info_nb;
752 ss << "=\"" << Infos[i] << "\"";
753 }
754
755 if (!Value.empty())
756 {
757 Value.Set_Output_Format(Element_Node_Data::Format_Xml);
758 ss << ">" << Value << "</d>";
759 }
760 else
761 ss << " s=\"" << Size << "\">";
762
763 level += 4;
764print_children:
765 for (size_t i = 0; i < Children.size(); ++i)
766 Children[i]->Print_Micro_Xml(ss, level);
767
768 if (!IsCat && Name)
769 {
770 //end tag
771 if (Value.empty())
772 {
773 //block
774 ss << "</b>";
775 }
776 }
777
778 return 0;
779}

781//---------------------------------------------------------------------------
782int element_details::Element_Node::Print_Xml(std::ostringstream& ss, size_t level)
783{
784 std::string spaces;
785 bool Modified = false;
786
787 if (IsCat || !Name)
788 goto print_children;
789
790 spaces.resize(level, ' ');
791 ss << spaces;
792
793 if (Value.empty())
794 ss << "<block";
795 else
796 ss << "<data";
797
798 {
799 size_t MustEscape = Xml_Content_Escape_MustEscape(Name);
800 if (MustEscape != (size_t)-1)
801 {
802 std::string str;
803 Xml_Content_Escape(Name, str, MustEscape);
804 ss << " offset=\"" << Pos << "\" name=\"" << str << "\"";
805 }
806 else
807 ss << " offset=\"" << Pos << "\" name=\"" << Name << "\"";
808 }
809
810 for (size_t i = 0, info_nb = 0; i < Infos.size(); ++i)
811 {
812 Element_Node_Info* Info = Infos[i];
813
814 if (Info->Measure && !std::strcmp(Info->Measure, "Parser"))
815 {
816 if (!(Info->data == string()))
817 ss << " parser=\"" << Info->data << "\"";
818 continue;
819 }
820 else
821 info_nb++;
822
823 ss << " info";
824 if (info_nb > 1)
825 ss << info_nb;
826 ss << "=\"" << Infos[i] << "\"";
827 }
828
829 if (!Value.empty())
830 {
831 Value.Set_Output_Format(Element_Node_Data::Format_Xml);
832 ss << ">" << Value << "</data>";
833 }
834 else
835 ss << " size=\"" << Size << "\">";
836
837 ss << "\n";
838
839 level += 4;
840print_children:
841 for (size_t i = 0; i < Children.size(); ++i)
842 Children[i]->Print_Xml(ss, level);
843
844 if (!IsCat && Name)
845 {
846 //end tag
847 if (Value.empty())
848 {
849 //block
850 ss << spaces << "</block>" << "\n";
851 }
852 }
853
854 return 0;
855}

857//---------------------------------------------------------------------------
858int element_details::Element_Node::Print_Tree_Cat(std::ostringstream& ss, size_t level)
859{
860 std::ostringstream offset;
861 offset << std::setfill('0') << std::setw(8) << std::hex << std::uppercase << Pos << std::nouppercase << std::dec;
862
863 std::string spaces;
864 spaces.resize(level, ' ');
865
866 std::string ToShow;
867 ToShow += "--- ";
868 ToShow += Name;
869 ToShow += " ---";
870
871 std::string minuses;
872 minuses.resize(ToShow.size(), '-');
873
874 ss << offset.str() << spaces << minuses << "\n";
875 ss << offset.str() << spaces << ToShow << "\n";
876 ss << offset.str() << spaces << minuses << "\n";
877 return 0;
878}

880//---------------------------------------------------------------------------
881int element_details::Element_Node::Print_Tree(std::ostringstream& ss, size_t level)
882{
883 std::string spaces;
884
885 if (IsCat)
886 return Print_Tree_Cat(ss, level);
887 else if (!Name)
888 goto print_children;
889
890 ss << std::setfill('0') << std::setw(8) << std::hex << std::uppercase << Pos << std::nouppercase << std::dec;
891 spaces.resize(level, ' ');
892 ss << spaces;
893 ss << Name;
894
895 spaces.clear();
896
897#define NB_SPACES 40
898 if (!Value.empty())
899 {
900 ss << ":";
901 int nb_free = NB_SPACES - level - (Name ? 0 : strlen(Name)); // 40 - len(Name) - len(spaces)
902 spaces.resize(nb_free > 0 ? nb_free : 1, ' ');
903 Value.Set_Output_Format(Element_Node_Data::Format_Tree);
904 ss << spaces << Value;
905 spaces.clear();
906 }
907#undef NB_SPACES
908
909 for (size_t i = 0; i < Infos.size(); ++i)
910 {
911 Element_Node_Info* Info = Infos[i];
912
913 if (Info->Measure && !std::strcmp(Info->Measure, "Parser"))
914 {
915 if (!(Info->data == string()))
916 ss << " - Parser=" << Info->data;
917 continue;
918 }
919
920 ss << " - " << Info;
921 }
922
923 if (Value.empty())
924 ss << " (" << Size << " bytes)";
925
926 ss << "\n";
927
928 level += 1;
929print_children:
930 for (size_t i = 0; i < Children.size(); ++i)
931 Children[i]->Print_Tree(ss, level);
932
933 return 0;
934}
