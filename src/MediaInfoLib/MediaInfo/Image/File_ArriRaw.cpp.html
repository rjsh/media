53//***************************************************************************
54// Streams management
55//***************************************************************************
56
57//---------------------------------------------------------------------------
58void File_ArriRaw::Streams_Accept()
59{
60 Fill(Stream_General, 0, General_Format, "Arri Raw");
61
62 if (!IsSub)
63 {
64 TestContinuousFileNames();
65
66 Stream_Prepare((Config->File_Names.size()>1 || Config->File_IsReferenced_Get())?Stream_Video:Stream_Image);
67 Fill(StreamKind_Last, StreamPos_Last, "StreamSize", File_Size);
68 if (StreamKind_Last==Stream_Video)
69 Fill(Stream_Video, StreamPos_Last, Video_FrameCount, Config->File_Names.size());
70 }
71 else
72 Stream_Prepare(StreamKind_Last);
73
74 //Configuration
75 Frame_Count_NotParsedIncluded=0;
76}
77
78//***************************************************************************
79// Header
80//***************************************************************************
81
82//---------------------------------------------------------------------------
83bool File_ArriRaw::FileHeader_Begin()
84{
85 // Minimum buffer size
86 if (Buffer_Size<8)
87 return false; // Must wait for more data
88
89 // Testing
90 if (Buffer[0]!=0x41 // "ARRI.4Vx"
91 || Buffer[1]!=0x52
92 || Buffer[2]!=0x52
93 || Buffer[3]!=0x49
94 || Buffer[4]!=0x12
95 || Buffer[5]!=0x34
96 || Buffer[6]!=0x56
97 || Buffer[7]!=0x78)
98 {
99 Reject("Arri Raw");
100 return false;
101 }
102
103 Accept();
104
105 //All should be OK...
106 return true;
107}

116 Read_Buffer_Unsynched_OneFramePerFile();

118
119//---------------------------------------------------------------------------
120void File_ArriRaw::Read_Buffer_Continue()
121{
122 //Parsing
123 Skip_C4( "Signature");
124 Skip_C1( "Signature");
125 Skip_C3( "Signature");
126 Skip_XX(File_Size-8, "Data");
127
128 FILLING_BEGIN();
129 Frame_Count++;
130 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
131 Frame_Count_NotParsedIncluded++;
132 if (!Status[IsFilled])
133 {
134 Fill();
135 if (Config->ParseSpeed<1.0)
136 Finish();
137 }
138 FILLING_END();
139}
