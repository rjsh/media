65//---------------------------------------------------------------------------
66void File_Exr::Streams_Accept()
67{
68 Fill(Stream_General, 0, General_Format, "EXR");
69
70 if (!IsSub)
71 {
72 TestContinuousFileNames();
73
74 Stream_Prepare((Config->File_Names.size()>1 || Config->File_IsReferenced_Get())?Stream_Video:Stream_Image);
75 if (File_Size!=(int64u)-1)
76 Fill(StreamKind_Last, StreamPos_Last, Fill_Parameter(StreamKind_Last, Generic_StreamSize), File_Size);
77 if (StreamKind_Last==Stream_Video)
78 Fill(Stream_Video, StreamPos_Last, Video_FrameCount, Config->File_Names.size());
79 }
80 else
81 Stream_Prepare(Stream_Image);
82
83 //Configuration
84 Buffer_MaximumSize=64*1024*1024; //Some big frames are possible (e.g YUV 4:2:2 10 bits 1080p)
85}

91//---------------------------------------------------------------------------
92bool File_Exr::FileHeader_Begin()
93{
94 //Element_Size
95 if (Buffer_Size<4)
96 return false; //Must wait for more data
97
98 if (CC4(Buffer)!=0x762F3101) //"v/1"+1
99 {
100 Reject();
101 return false;
102 }
103
104 //All should be OK...
105 Accept();
106
107 return true;
108}

110//***************************************************************************
111// Buffer - Per element
112//***************************************************************************
113
114//---------------------------------------------------------------------------
115bool File_Exr::Header_Begin()
116{
117 //Header
118 if (Buffer_Offset+4>Buffer_Size)
119 return false;
120 if (CC4(Buffer+Buffer_Offset)==0x762F3101) //"v/1"+1
121 return Buffer_Offset+12<=Buffer_Size;
122
123 //Name
124 name_End=0;
125 while (Buffer_Offset+name_End<Buffer_Size)
126 {
127 if (Buffer[Buffer_Offset+name_End]=='\0')
128 break;
129 if (name_End==31)
130 break;
131 name_End++;
132 }
133 if (Buffer_Offset+name_End>=Buffer_Size)
134 return false;
135 if (name_End>=31)
136 {
137 Reject();
138 return false;
139 }
140 if (name_End==0)
141 return true;
142
143 //Type
144 type_End=0;
145 while (Buffer_Offset+name_End+1+type_End<Buffer_Size)
146 {
147 if (Buffer[Buffer_Offset+name_End+1+type_End]=='\0')
148 break;
149 if (type_End==31)
150 break;
151 type_End++;
152 }
153
154 if (Buffer_Offset+name_End+1+type_End>=Buffer_Size)
155 return false;
156 if (type_End>=31)
157 {
158 Reject();
159 return false;
160 }
161
162 if (Buffer_Offset+name_End+1+type_End+1+4>=Buffer_Size)
163 return false;
164
165 return true;
166}

168//---------------------------------------------------------------------------
169void File_Exr::Header_Parse()
170{
171 //Header
172 if (CC4(Buffer+Buffer_Offset)==0x762F3101) //"v/1"+1
173 {
174 //Filling
175 Header_Fill_Code(0, "File header");
176 Header_Fill_Size(12);
177 return;
178 }
179
180 //Image data
181 if (name_End==0)
182 {
183 //Filling
184 Header_Fill_Code(0, "Image data");
185 Header_Fill_Size(ImageData_End-(File_Offset+Buffer_Offset));
186 return;
187 }
188
189 int32u size;
190 Get_String(name_End, name, "name");
191 Element_Offset++; //Null byte
192 Get_String(type_End, type, "type");
193 Element_Offset++; //Null byte
194 Get_L4 (size, "size");
195
196 //Filling
197 Header_Fill_Code(0, Ztring().From_Local(name.c_str()));
198 Header_Fill_Size(name_End+1+type_End+1+4+size);
199}

201//---------------------------------------------------------------------------
202void File_Exr::Data_Parse()
203{
204 if (CC4(Buffer+Buffer_Offset)==0x762F3101) //"v/1"+1 //Header
205 Header();
206 else if (name_End==0)
207 ImageData();
208 else if (name=="comments" && type=="string")
209 comments();
210 else if (name=="compression" && type=="compression" && Element_Size==1)
211 compression();
212 else if (name=="dataWindow" && type=="box2i" && Element_Size==16)
213 dataWindow();
214 else if (name=="displayWindow" && type=="box2i" && Element_Size==16)
215 displayWindow();
216 else if (name=="pixelAspectRatio" && type=="float" && Element_Size==4)
217 pixelAspectRatio();
218 else
219 Skip_XX(Element_Size, "value");
220}

222//---------------------------------------------------------------------------
223void File_Exr::Header()
224{
225 //Parsing
226 int32u Flags;
227 int8u Version;
228 Skip_L4( "Magic number");
229 Get_L1 (Version, "Version field");
230 Get_L3 (Flags, "Flags");
231
232 //Filling
233 if (Frame_Count==0)
234 {
235 Fill(Stream_General, 0, General_Format_Version, __T("Version ")+Ztring::ToZtring(Version));
236 Fill(StreamKind_Last, 0, "Format", "EXR");
237 Fill(StreamKind_Last, 0, "Format_Version", __T("Version ")+Ztring::ToZtring(Version));
238 Fill(StreamKind_Last, 0, "Format_Profile", (Flags&0x02)?"Tile":"Line");
239 }
240 Frame_Count++;
241 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
242 Frame_Count_NotParsedIncluded++;
243 ImageData_End=Config->File_Current_Size;
244}

246//---------------------------------------------------------------------------
247void File_Exr::ImageData()
248{
249 Skip_XX(Element_Size, "data");
250
251 if (!Status[IsFilled])
252 Fill();
253 if (Config->ParseSpeed<1.0)
254 Finish();
255}

257//---------------------------------------------------------------------------
258void File_Exr::comments ()
259{
260 //Parsing
261 Ztring value;
262 Get_Local(Element_Size, value, "value");
263
264 //Filling
265 if (Frame_Count==1)
266 Fill(StreamKind_Last, 0, General_Comment, value);
267}

269//---------------------------------------------------------------------------
270void File_Exr::compression ()
271{
272 //Parsing
273 int8u value;
274 Get_L1 (value, "value");
275
276 //Filling
277 std::string Compression;
278 switch (value)
279 {
280 case 0x00 : Compression="raw"; break;
281 case 0x01 : Compression="RLZ"; break;
282 case 0x02 : Compression="ZIPS"; break;
283 case 0x03 : Compression="ZIP"; break;
284 case 0x04 : Compression="PIZ"; break;
285 case 0x05 : Compression="PXR24"; break;
286 case 0x06 : Compression="B44"; break;
287 case 0x07 : Compression="B44A"; break;
288 default : ;
289 }
290
291 if (Frame_Count==1)
292 Fill(StreamKind_Last, 0, "Format_Compression", Compression.c_str());
293}

295//---------------------------------------------------------------------------
296void File_Exr::dataWindow ()
297{
298 //Parsing
299 int32u xMin, yMin, xMax, yMax;
300 Get_L4 (xMin, "xMin");
301 Get_L4 (yMin, "yMin");
302 Get_L4 (xMax, "xMax");
303 Get_L4 (yMax, "yMax");
304}
305
306//---------------------------------------------------------------------------
307void File_Exr::displayWindow ()
308{
309 //Parsing
310 int32u xMin, yMin, xMax, yMax;
311 Get_L4 (xMin, "xMin");
312 Get_L4 (yMin, "yMin");
313 Get_L4 (xMax, "xMax");
314 Get_L4 (yMax, "yMax");
315
316 //Filling
317 if (Frame_Count==1)
318 {
319 Fill(StreamKind_Last, 0, "Width", xMax-xMin+1);
320 Fill(StreamKind_Last, 0, "Height", yMax-yMin+1);
321 }
322}

324//---------------------------------------------------------------------------
325void File_Exr::pixelAspectRatio ()
326{
327 //Parsing

329 Get_LF4(value, "value");
330
331 //Filling
332 if (Frame_Count==1)
333 Fill(StreamKind_Last, 0, "PixelAspectRatio", value?value:1, 3);
