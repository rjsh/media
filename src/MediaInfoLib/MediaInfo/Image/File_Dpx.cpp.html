307string DPX_DateTime2Iso(const string &FromDpx)
308{
309 if (FromDpx.size()<20
310 || FromDpx[ 0]<'0' || FromDpx[ 0]>'9'
311 || FromDpx[ 1]<'0' || FromDpx[ 1]>'9'
312 || FromDpx[ 2]<'0' || FromDpx[ 2]>'9'
313 || FromDpx[ 3]<'0' || FromDpx[ 3]>'9'
314 || FromDpx[ 4] != ':'
315 || FromDpx[ 5]<'0' || FromDpx[ 5]>'9'
316 || FromDpx[ 6]<'0' || FromDpx[ 6]>'9'
317 || FromDpx[ 7] != ':'
318 || FromDpx[ 8]<'0' || FromDpx[ 8]>'9'
319 || FromDpx[ 9]<'0' || FromDpx[ 9]>'9'
320 || FromDpx[10] != ':'
321 || FromDpx[11]<'0' || FromDpx[11]>'9'
322 || FromDpx[12]<'0' || FromDpx[12]>'9'
323 || FromDpx[13] != ':'
324 || FromDpx[14]<'0' || FromDpx[14]>'9'
325 || FromDpx[15]<'0' || FromDpx[15]>'9'
326 || FromDpx[16] != ':'
327 || FromDpx[17]<'0' || FromDpx[17]>'9'
328 || FromDpx[18]<'0' || FromDpx[18]>'9'
329 )
330 return FromDpx; //Invalid, returning the value in the file
331
332 // Date/Time
333 string ToReturn(FromDpx.substr(0, 19));
334 ToReturn[ 4]='-';
335 ToReturn[ 7]='-';
336 ToReturn[10]='T'; // Note: should be 'T' (ISO 8601), TODO: change to 'T' during next API change
337
338 // Time zone
339 if (FromDpx.size()>20)
340 {
341 size_t Offset;
342 if (FromDpx[19]==':') // Note: Cineon and DPX v1 header date/time format is with ":LTZ", DPX v2 date/time format is "LTZ", but both are atually seen in any version. Always testing both.
343 Offset=20;
344 else
345 Offset=19;
346
347 size_t Max=FromDpx.find('\0');
348 if (Max==(size_t)-1)
349 Max=FromDpx.size();
350
351 ToReturn+=FromDpx.substr(Offset, Max-Offset);
352
353 if (ToReturn.size()>22)
354 ToReturn.insert(ToReturn.begin(), ':'); //Hours/Minutes offset sepearator added
355 else if (ToReturn.size()==22 && (ToReturn[19]=='+' || ToReturn[19]=='-'))
356 ToReturn+=":00"; // Minutes offset, default
357 }
358
359 return ToReturn;
360}

398//***************************************************************************
399// Buffer
400//***************************************************************************
401
402//---------------------------------------------------------------------------
403bool File_Dpx::FileHeader_Begin()
404{
405 //Element_Size
406 if (Buffer_Size<4)
407 return false; //Must wait for more data
408
409 int32u Magic=CC4(Buffer);
410 switch (Magic)
411 {
412 case 0x802A5FD7 : // (Cineon Big)
413 case 0xD75F2A80 : // (Cineon Little)
414 case 0x53445058 : //"SPDX" (Dpx Big)
415 case 0x58504453 : //"XDPS" (Dpx Little)
416 break;
417 default :
418 Reject();
419 return false;
420 }
421
422 //Generic Section size
423 if (Buffer_Size<28)
424 return false; //Must wait for more data
425 Sizes_Pos=Pos_GenericSection;
426 switch (Magic)
427 {
428 case 0x802A5FD7 : // (Cineon Big)
429 case 0xD75F2A80 : // (Cineon Little)
430 IsDpx=false;
431 break;
432 case 0x58504453 : //"XDPS" (Dpx Little)
433 case 0x53445058 : //"SPDX" (Dpx Big)
434 IsDpx=true;
435 break;
436 default : ;
437 }
438 switch (Magic)
439 {
440 case 0xD75F2A80 : // (Cineon Little)
441 case 0x58504453 : //"XDPS" (Dpx Little)
442 LittleEndian=true;
443 break;
444 case 0x802A5FD7 : // (Cineon Big)
445 case 0x53445058 : //"SPDX" (Dpx Big)
446 LittleEndian=false;
447 break;
448 default : ;
449 }
450
451 //All should be OK...
452 Accept();
453
454 return true;
455}

461//---------------------------------------------------------------------------
462void File_Dpx::Header_Parse()
463{
464 //Filling
465 Header_Fill_Code(Sizes_Pos); //We use Sizes_Pos as the unique key
466 if (Sizes.empty())
467 {
468 if (Element_Size<28)
469 {
470 Element_WaitForMoreData();
471 return;
472 }
473 int32u Size=LittleEndian?LittleEndian2int32u(Buffer+Buffer_Offset+24):BigEndian2int32u(Buffer+Buffer_Offset+24);
474 if (Size==(int32u)-1)
475 Size=LittleEndian?LittleEndian2int32u(Buffer+Buffer_Offset+4):BigEndian2int32u(Buffer+Buffer_Offset+4);
476 Header_Fill_Size(Size);
477 }
478 else
479 Header_Fill_Size(Sizes[Sizes_Pos]);
480}
481
482//---------------------------------------------------------------------------
483void File_Dpx::Data_Parse()
484{
485 if (!IsDpx) // Is Cineon
486 {
487 switch (Element_Code)
488 {
489 case Pos_GenericSection : GenericSectionHeader_Cineon(); break;
490 case Pos_IndustrySpecific : IndustrySpecificHeader_Cineon(); break;
491 case Pos_UserDefined : UserDefinedHeader_Cineon(); break;
492 case Pos_Padding : Padding(); break;
493 case Pos_ImageData : ImageData(); break;
494 default : ;
495 }
496 }
497 else
498 {
499 switch (Element_Code)
500 {
501 case Pos_GenericSection : GenericSectionHeader_Dpx(); break;
502 case Pos_IndustrySpecific : IndustrySpecificHeader_Dpx(); break;
503 case Pos_UserDefined : UserDefinedHeader_Dpx(); break;
504 case Pos_Padding : Padding(); break;
505 case Pos_ImageData : ImageData(); break;
506 default : ;
507 }
508 }
509
510 do
511 Sizes_Pos++; //We go automaticly to the next block
512 while (Sizes_Pos<Sizes.size() && Sizes[Sizes_Pos]==0);
513 if (Sizes_Pos>=Sizes.size())
514 {
515 Sizes.clear();
516 Sizes_Pos=0;
517
518 if (!Status[IsFilled])
519 Fill();
520 if (File_Offset+Buffer_Offset+Element_Size<Config->File_Current_Size)
521 GoTo(Config->File_Current_Size);
522 }
523}

529//---------------------------------------------------------------------------
530void File_Dpx::GenericSectionHeader_Cineon()
531{
532 Element_Name("Generic section header");
533
534 //Parsing
535 Element_Begin1("File information");
536 string CreationDate, CreationTime;
537 string Version;
538 int32u Size_Header, Size_Total, Size_Generic, Size_Industry, Size_User;
539 Skip_B4( "Magic number");
540 Get_X4 (Size_Header, "Offset to image data");
541 Get_X4 (Size_Generic, "Generic section header length");
542 Get_X4 (Size_Industry, "Industry specific header length");
543 Get_X4 (Size_User, "User-defined header length");
544 Get_X4 (Size_Total, "Total image file size");
545 Get_String(8, Version, "Version number of header format");
546 Skip_UTF8 (100, "FileName");
547 Get_String (12, CreationDate, "Creation Date");
548 Get_String (12, CreationTime, "Creation Time");
549 Skip_XX(36, "Reserved for future use");
550 Element_End0();
551
552 Element_Begin1("Image information");
553 int8u ImageElements;
554 Info_B1(ImageOrientation, "Image orientation"); Param_Info1(DPX_Orientation[ImageOrientation>8?8:ImageOrientation]);
555 Get_B1 (ImageElements, "Number of image elements");
556 Skip_B2( "Unused");
557 if (ImageElements>8)
558 ImageElements=8;
559 for(int8u ImageElement=0; ImageElement<ImageElements; ImageElement++)
560 GenericSectionHeader_Cineon_ImageElement();
561 if (ImageElements!=8)
562 Skip_XX((8-ImageElements)*28, "Padding");
563 Skip_BF4( "White point - x");
564 Skip_BF4( "White point - y");
565 Skip_BF4( "Red primary chromaticity - x");
566 Skip_BF4( "Red primary chromaticity - u");
567 Skip_BF4( "Green primary chromaticity - x");
568 Skip_BF4( "Green primary chromaticity - y");
569 Skip_BF4( "Blue primary chromaticity - x");
570 Skip_BF4( "Blue primary chromaticity - y");
571 Skip_UTF8(200, "Label text");
572 Skip_XX(28, "Reserved for future use");
573 Element_End0();
574
575 Element_Begin1("Image Data Format Information");
576 Skip_B1( "Data interleave");
577 Skip_B1( "Packing");
578 Skip_B1( "Data signed or unsigned");
579 Skip_B1( "Image sense");
580 Skip_B4( "End of line padding");
581 Skip_B4( "End of channel padding");
582 Skip_XX(20, "Reserved for future use");
583
584 Element_Begin1("Image Origination Information");
585 Skip_B4( "X offset");
586 Skip_B4( "Y offset");
587 Skip_UTF8 (100, "FileName");
588 Get_String (12, CreationDate, "Creation Date");
589 Get_String (12, CreationTime, "Creation Time");
590 Skip_UTF8(64, "Input device");
591 Skip_UTF8(32, "Input device model number");
592 Skip_UTF8(32, "Input device serial number");
593 Skip_BF4( "X input device pitch");
594 Skip_BF4( "Y input device pitch");
595 Skip_BF4( "Image gamma of capture device");
596 Skip_XX(40, "Reserved for future use");
597 Element_End0();
598
599 FILLING_BEGIN();
600 //Coherency tests
601 if (File_Offset+Buffer_Offset+Size_Total>=Config->File_Current_Size)
602 Size_Total=(int32u)(Config->File_Current_Size-(File_Offset+Buffer_Offset)); //The total size is bigger than the real size
603 if (Size_Generic+Size_Industry+Size_User>Size_Header || Size_Header>Size_Total)
604 {
605 Reject();
606 return;
607 }
608
609 //Filling sizes
610 Sizes.push_back(Size_Header);
611 Sizes.push_back(Size_Industry);
612 Sizes.push_back(Size_User);
613 Sizes.push_back(Size_Header-(Size_Generic+Size_Industry+Size_User)); //Size of padding
614 Sizes.push_back(Size_Total-Size_Header); //Size of image
615
616 //Filling meta
617 if (Frame_Count==0)
618 {
619 Fill(Stream_General, 0, General_Encoded_Date, DPX_DateTime2Iso(CreationDate+':'+CreationTime));
620 Fill(StreamKind_Last, StreamPos_Last, "Encoded_Date", DPX_DateTime2Iso(CreationDate+':'+CreationTime));
621 Fill(StreamKind_Last, StreamPos_Last, "Format", "Cineom");
622 if (Version.size()>2 && Version[0]=='V' && Version[1]>='0' && Version[2]<='9')
623 Version.insert(1, "ersion ");
624 Fill(StreamKind_Last, StreamPos_Last, "Format_Version", Version);
625 Fill(Stream_General, 0, General_Format_Version, Version);
626 }
627 FILLING_END();
628}

724 FILLING_BEGIN();
725 //Coherency tests
726 if (File_Offset+Buffer_Offset+Size_Total!=Config->File_Current_Size)
727 Size_Total=(int32u)(Config->File_Current_Size-(File_Offset+Buffer_Offset)); //The total size is bigger than the real size
728 if (Size_Generic==(int32u)-1)
729 Size_Generic=(int32u)Element_Size;
730 if (Size_Industry==(int32u)-1)
731 Size_Industry=0;
732 if (Size_User==(int32u)-1)
733 Size_User=0;
734 if (Size_Generic+Size_Industry+Size_User>Size_Header || Size_Header>Size_Total)
735 {
736 Reject();
737 return;
738 }
739
740 //Filling sizes
741 Sizes.push_back(Size_Header);
742 Sizes.push_back(Size_Industry);
743 Sizes.push_back(Size_User);
744 Sizes.push_back(Size_Header-(Size_Generic+Size_Industry+Size_User)); //Size of padding
745 Sizes.push_back(Size_Total-Size_Header); //Size of image
746
747 //Filling meta
748 if (Frame_Count==0)
749 {
750 Fill(Stream_General, 0, General_Encoded_Date, DPX_DateTime2Iso(CreationDate));
751 Fill(StreamKind_Last, StreamPos_Last, "Encoded_Date", DPX_DateTime2Iso(CreationDate));
752 Fill(Stream_General, 0, General_Encoded_Library, Creator);
753 Fill(StreamKind_Last, StreamPos_Last, "Encoded_Library", Creator);
754 Fill(Stream_General, 0, "Project", Project); //ToDo: map to a MediaInfo field (which one?)
755 Fill(Stream_General, 0, General_Copyright, Copyright);
756 Fill(StreamKind_Last, StreamPos_Last, "Format", "DPX");
757 if (Version.size()>2 && Version[0]=='V' && Version[1]>='0' && Version[2]<='9')
758 Version.insert(1, "ersion ");
759 Fill(StreamKind_Last, StreamPos_Last, "Format_Version", Version);
760 Fill(Stream_General, 0, General_Format_Version, Version);
761
762 Fill(StreamKind_Last, StreamPos_Last, "Width", Width);
763 Fill(StreamKind_Last, StreamPos_Last, "Height", Height);
764 if (PAR_V && PAR_H!=(int32u)-1 && PAR_V!=(int32u)-1)
765 Fill(StreamKind_Last, StreamPos_Last, "PixelAspectRatio", ((float)PAR_H)/PAR_V);
766 else
767 Fill(StreamKind_Last, StreamPos_Last, "PixelAspectRatio", (float)1, 3);
768 }
769 FILLING_END();
770}
