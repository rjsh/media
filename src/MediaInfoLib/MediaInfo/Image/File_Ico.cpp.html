40//***************************************************************************
41// Streams management
42//***************************************************************************
43
44//---------------------------------------------------------------------------
45void File_Ico::Streams_Fill()
46{
47 Fill(Stream_General, 0, General_Format, Type==1?"ICO":"CUR");
48
49 for (size_t Pos=0; Pos<Streams.size(); Pos++)
50 {
51 Stream_Prepare(Stream_Image);
52 Fill(Stream_Image, StreamPos_Last, Image_Width, Streams[Pos].Width?Streams[Pos].Width:256);
53 Fill(Stream_Image, StreamPos_Last, Image_Height, Streams[Pos].Height?Streams[Pos].Height:256);
54 if (Type==1)
55 Fill(Stream_Image, StreamPos_Last, Image_BitDepth, Streams[Pos].BitsPerPixel);
56 Fill(Stream_Image, StreamPos_Last, Image_StreamSize, Streams[Pos].Size);
57 }
58}

60//***************************************************************************
61// Buffer - File header
62//***************************************************************************
63
64//---------------------------------------------------------------------------
65bool File_Ico::FileHeader_Begin()
66{
67 //Element_Size
68 if (Buffer_Size<4)
69 return false; //Must wait for more data
70
71 if (CC2(Buffer) || (LittleEndian2int16u(Buffer+2)!=1 && LittleEndian2int16u(Buffer+2)!=2))
72 {
73 Reject("ICO");
74 return false;
75 }
76
77 //All should be OK...
78 return true;
79}

101//---------------------------------------------------------------------------
102void File_Ico::Data_Parse()
103{
104 //Parsing
105 int32u Size, Offset;
106 int16u BitsPerPixel;
107 int8u Width, Height;
108 Get_L1 (Width, "Width");
109 Get_L1 (Height, "Height");
110 Skip_L1( "Colour count");
111 Skip_L1( "Reserved");
112 Skip_L2( Type==1?"Colour planes":"X hotspot");
113 Get_L2 (BitsPerPixel, Type==1?"Bits per pixel":"Y hotspot");
114 Get_L4 (Size, "Size of the bitmap data");
115 Get_L4 (Offset, "Offset of the bitmap data");
116
117 FILLING_BEGIN_PRECISE();
118 stream Stream;
119 Stream.Width=Width;
120 Stream.Height=Height;
121 Stream.BitsPerPixel=BitsPerPixel;
122 Stream.Size=Size;
123 Stream.Offset=Offset;
124 Streams.push_back(Stream);
125
126 IcoDataSize+=Size;
127 if (Offset>File_Size || File_Offset+Buffer_Offset+Element_Size+IcoDataSize>File_Size)
128 Reject("ICO");
129 Count--;
130 if (Count==0)
131 {
132 if (File_Offset+Buffer_Offset+Element_Size+IcoDataSize!=File_Size)
133 Reject("ICO");
134 else
135 {
136 Accept("ICO");
137 Finish("ICO");
138 }
139 }
140 FILLING_END();
141}
