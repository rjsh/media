39const char* Bpg_ColorSpace(int8u ColorSpace)
40{
41 switch (ColorSpace)
42 {
43 case 0:
44 case 3:
45 case 4: return "YUV";
46 case 1: return "RGB";
47 case 2: return "YCgCo";
48 default: return "";
49 }
50};
51
52//---------------------------------------------------------------------------
53const char* Bpg_colour_primaries(int8u ColorSpace)
54{
55 switch (ColorSpace)
56 {
57 case 0: return "BT.601";
58 case 3: return "BT.701";
59 case 4: return "BT.2020";
60 default: return "";
61 }
62};
63
64//---------------------------------------------------------------------------
65const char* Bpg_Pixel_format(int8u PixelFormat)
66{
67 switch (PixelFormat)
68 {
69 case 0 : return "Grayscale";
70 case 1 :
71 case 4 : return "4:2:0";
72 case 2 :
73 case 5 : return "4:2:2";
74 case 3 : return "4:4:4";
75 default: return "";
76 }
77
78};

85bool File_Bpg::FileHeader_Begin()
86{
87 //Element_Size
88 if (Buffer_Size<4)
89 return false; //Must wait for more data
90
91 if (CC4(Buffer) != 0x425047FB) //"BPG"
92 {
93 Reject("BPG");
94 return false;
95 }
96
97 //All should be OK...
98 return true;
99}

101//***************************************************************************
102// Buffer - Global
103//***************************************************************************
104
105//---------------------------------------------------------------------------
106void File_Bpg::Read_Buffer_Continue()
107{
108 //Parsing
109 Ztring Version;
110 int64u Width, Height;
111 int8u pixelformat, BitsDepth, ColorSpace;
112 bool Alpha1_flag, Alpha2_flag, ReservedZeros, limited_range_flag, ExtensionPresentFlag;
113 Element_Begin1("File header");
114 Skip_C4( "Magic"); // File Magic
115 BS_Begin();
116 Get_S1 (3, pixelformat, "pixel_format"); Param_Info1(Bpg_Pixel_format(pixelformat));
117 Get_SB (Alpha1_flag, "Alpha1 Present Flag");
118 Get_S1 (4, BitsDepth, "bit_depth_minus_8");
119
120 Get_S1(4, ColorSpace, "color_space"); Param_Info1(Bpg_ColorSpace(ColorSpace)); Param_Info1(Bpg_colour_primaries(ColorSpace));
121 Get_SB (ExtensionPresentFlag, "Extension Present Flag");
122 Get_SB (Alpha2_flag, "Alpha2 Present Flag");
123 Get_SB (limited_range_flag, "limited_range_flag");
124 Get_SB (ReservedZeros, "Reserved");
125 BS_End();
126
127 Get_VS(Width, "Picture Width");
128 Get_VS(Height, "Picture Height");
129
130
131 Element_End0();
132
133 FILLING_BEGIN();
134 Accept("BPG");
135
136 Stream_Prepare(Stream_Image);
137 Fill(Stream_Image, 0, Image_Width, Width);
138 Fill(Stream_Image, 0, Image_Height, Height);
139 Fill(Stream_Image, 0, Image_Format, __T("BPG"));
140 Fill(Stream_Image, 0, Image_ChromaSubsampling, Bpg_Pixel_format(pixelformat));
141 Fill(Stream_Image, 0, Image_ColorSpace, Bpg_ColorSpace(ColorSpace));
142 Fill(Stream_Image, 0, Image_colour_primaries, Bpg_colour_primaries(ColorSpace));
143 Fill(Stream_Image, 0, Image_BitDepth, BitsDepth + 8);
144 Fill(Stream_Image, 0, Image_Codec, __T("BPG"));
145 FILLING_END();
146
147 Finish("BPG");
148}
