44const char* Bmp_CompressionMethod(int32u CompressionMethod)
45{
46 switch(CompressionMethod)
47 {
48 case 0 : return "RGB";
49 case 1 : return "RLE";
50 case 2 : return "RLE";
51 case 3 : return "Bit field";
52 case 4 : return "JPEG";
53 case 5 : return "PNG";
54 default: return "";
55 }
56}

63bool File_Bmp::FileHeader_Begin()
64{
65 //Element_Size
66 if (Buffer_Size<2)
67 return false; //Must wait for more data
68
69 if (CC2(Buffer)!=0x424D) //"BM"
70 {
71 Reject("BMP");
72 return false;
73 }
74
75 //All should be OK...
76 return true;
77}

79//***************************************************************************
80// Buffer - Global
81//***************************************************************************
82
83//---------------------------------------------------------------------------
84void File_Bmp::Read_Buffer_Continue()
85{
86 //Parsing
87 int32u Size, DIB_Size, Offset;
88 Element_Begin1("File header");
89 Skip_C2( "Magic");
90 Get_L4 (Size, "Size");
91 Skip_L2( "Reserved");
92 Skip_L2( "Reserved");
93 Get_L4 (Offset, "Offset of data");
94 Element_End0();
95
96 FILLING_BEGIN();
97 if (Size!=(int32u)-1 && Size!=File_Size)
98 {
99 Reject("BMP");
100 return;
101 }
102
103 Accept("BMP");
104
105 Fill(Stream_General, 0, General_Format, "Bitmap");
106
107 Stream_Prepare(Stream_Image);
108 FILLING_END();
109
110 Element_Begin1("DIB header");
111 Peek_L4 (DIB_Size);
112 switch (DIB_Size)
113 {
114 case 12 : BitmapCoreHeader(1); break;
115 case 40 : BitmapInfoHeader(1); break;
116 case 52 : BitmapInfoHeader(2); break;
117 case 56 : BitmapInfoHeader(3); break;
118 case 64 : BitmapCoreHeader(2); break;
119 case 108 : BitmapInfoHeader(4); break;
120 case 124 : BitmapInfoHeader(5); break;
121 default : if (DIB_Size>124)
122 {
123 BitmapInfoHeader((int8u)-1); //Future versions of BitmapInfoHeader (OS/2 is abandonned)
124 Skip_XX(14+124-Element_Offset, "Unknown");
125 }
126 }
127 Element_End0();
128
129 if (Element_Offset<Offset)
130 Skip_XX(Offset-Element_Offset, "Other header data");
131 Skip_XX(File_Size-Offset, "Image data");
132
133 //No need of more
134 Finish("BMP");
135}

141void File_Bmp::BitmapCoreHeader(int8u Version)
142{
143 #if MEDIAINFO_TRACE
144 switch (Version)
145 {
146 case 1 : Element_Info1("OS/2 1.x BITMAPCOREHEADER"); break;
147 case 2 : Element_Info1("OS/2 2.x BITMAPCOREHEADER"); break;
148 default: Element_Info1("OS/2 ? BITMAPCOREHEADER");
149 }
150 #endif //MEDIAINFO_TRACE
151
152 //Parsing
153 int16u Width, Height, BitsPerPixel;
154 Skip_L4( "Size");
155 Get_L2 (Width, "Width");
156 Get_L2 (Height, "Height");
157 Skip_L2( "Color planes");
158 Get_L2 (BitsPerPixel, "Bits per pixel");
159
160 FILLING_BEGIN();
161 if (BitsPerPixel<8)
162 BitsPerPixel=8; //It is a palette
163
164 Fill(Stream_Image, 0, Image_Width, Width);
165 Fill(Stream_Image, 0, Image_Height, Height);
166 Fill(Stream_Image, 0, Image_BitDepth, BitsPerPixel);
167 Fill(Stream_Image, 0, Image_ColorSpace, "RGB");
168 FILLING_END();
169
170 if (Version>1) //V2 additional fields for information only
171 {
172 Skip_L4( "Compression");
173 Skip_L4( "ImageDataSize");
174 Skip_L4( "XResolution");
175 Skip_L4( "YResolution");
176 Skip_L4( "ColorsUsed");
177 Skip_L4( "ColorsImportant");
178 Skip_L2( "Units");
179 Skip_L2( "Reserved");
180 Skip_L2( "Recording");
181 Skip_L2( "Rendering");
182 Skip_L4( "Size1");
183 Skip_L4( "Size2");
184 Skip_L4( "ColorEncoding");
185 Skip_L4( "Identifier");
186 }

189void File_Bmp::BitmapInfoHeader(int8u Version)
190{
191 #if MEDIAINFO_TRACE
192 switch (Version)
193 {
194 case 1 : Element_Info1("BITMAPINFOHEADER"); break;
195 case 2 : Element_Info1("BITMAPV2INFOHEADER"); break;
196 case 3 : Element_Info1("BITMAPV3INFOHEADER"); break;
197 case 4 : Element_Info1("BITMAPV4HEADER"); break;
198 case 5 : Element_Info1("BITMAPV5HEADER"); break;
199 default: Element_Info1("BITMAPV?HEADER");
200 }
201 #endif //MEDIAINFO_TRACE
202
203 //Parsing
204 int32u Width, Height, CompressionMethod;
205 int16u BitsPerPixel;
206 Skip_L4( "Size");
207 Get_L4 (Width, "Width");
208 Get_L4 (Height, "Height");
209 Skip_L2( "Color planes");
210 Get_L2 (BitsPerPixel, "Bits per pixel");
211 Get_L4 (CompressionMethod, "Compression method"); Param_Info1(Bmp_CompressionMethod(CompressionMethod));
212 Skip_L4( "Image size");
213 Skip_L4( "Horizontal resolution");
214 Skip_L4( "Vertical resolution");
215 Skip_L4( "Number of colors in the color palette");
216 Skip_L4( "Number of important colors used");
217
218 FILLING_BEGIN();
219 if (BitsPerPixel<8)
220 BitsPerPixel=8; //It is a palette
221
222 Fill(Stream_Image, 0, Image_Width, Width);
223 Fill(Stream_Image, 0, Image_Height, Height);
224 Fill(Stream_Image, 0, Image_BitDepth, BitsPerPixel);
225 Fill(Stream_Image, 0, Image_Format, Bmp_CompressionMethod(CompressionMethod));
226 Fill(Stream_Image, 0, Image_Codec, Bmp_CompressionMethod(CompressionMethod));
227 Fill(Stream_Image, 0, Image_ColorSpace, "RGB");
228 FILLING_END();
229
230 if (Version>1)
231 {
232 Skip_L4( "Red Channel bit mask");
233 Skip_L4( "Green Channel bit mask");
234 Skip_L4( "Blue Channel bit mask");
235 if (Version>2)
236 {
237 Skip_L4( "Alpha Channel bit mask");
238 if (Version>3)
239 {
240 Skip_L4( "Color Space endpoints");
241 Skip_L4( "Color Space endpoints");
242 Skip_L4( "Color Space endpoints");
243 Skip_L4( "Color Space endpoints");
244 Skip_L4( "Color Space endpoints");
245 Skip_L4( "Color Space endpoints");
246 Skip_L4( "Color Space endpoints");
247 Skip_L4( "Red Gamma");
248 Skip_L4( "Green Gamma");
249 Skip_L4( "Blue Gamma");
250 if (Version>4)
251 {
252 Skip_L4( "Intent");
253 Skip_L4( "ProfileData");
254 Skip_L4( "ProfileSize");
255 Skip_L4( "Reserved");
256 }
257 }
258 }
259 }
