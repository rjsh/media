43const char* Pcx_VersionInfo(int16u Version)
44{
45 switch(Version)
46 {
47 case 0 : return "Paintbrush v2.5";
48 case 2 : return "Paintbrush v2.8 with palette information";
49 case 3 : return "Paintbrush v2.8 without palette information";
50 case 4 : return "Paintbrush/Windows";
51 case 5 : return "Paintbrush v3.0+";
52 default: return "";
53 }
54}

60//---------------------------------------------------------------------------
61bool File_Pcx::FileHeader_Begin()
62{
63 //Element_Size
64 if (Buffer_Size<130) //Size of the header
65 return false; //Must wait for more data
66
67 if (Buffer[0]!=0x0A
68 || Buffer[1]>0x05
69 || Buffer[2]!=0x01
70 || !(Buffer[3]==1 || Buffer[3]==4 || Buffer[3]==8 || Buffer[3]==24))
71 {
72 Reject("PCX");
73 return false;
74 }
75
76 //All should be OK...
77 return true;
78}

80//***************************************************************************
81// File Header Information
82//***************************************************************************
83
84//---------------------------------------------------------------------------
85void File_Pcx::Read_Buffer_Continue()
86{
87 //Parsing
88 int16u XMin, YMin, XMax, YMax, HorDPI, VertDPI, BytesPerLine, PaletteType, HScrSize, VScrSize;
89 int8u Manufacturer, Version, EncodingScheme, BitsPerPixel, ColorPlanes;
90
91 Get_L1 (Manufacturer, "Manufacturer");
92 Get_L1 (Version, "Version"); // 0,2,3,4,5
93 Get_L1 (EncodingScheme, "EncodingScheme"); // RLE=1
94 Get_L1 (BitsPerPixel, "Bits Per Pixel"); // 1,4,8,24
95 Get_L2 (XMin, "Left margin of image");
96 Get_L2 (YMin, "Upper margin of image");
97 Get_L2 (XMax, "Right margin of image");
98 Get_L2 (YMax, "Lower margin of image");
99 Get_L2 (HorDPI, "Horizontal Resolution");
100 Get_L2 (VertDPI, "Vertical Resolution");
101 Skip_XX(48, "Palette");
102 Skip_L1( "Reserved");
103 Get_L1 (ColorPlanes, "ColorPlanes");
104 Get_L2 (BytesPerLine, "BytesPerLine");
105 Get_L2 (PaletteType, "PaletteType");
106 Get_L2 (HScrSize, "Horizontal Screen Size");
107 Get_L2 (VScrSize, "Vertical Screen Size");
108 Skip_XX(56, "Filler");
109
110
111 FILLING_BEGIN();
112 //Integrity tests
113 if (XMax<=XMin
114 || YMax<=YMin
115 || BytesPerLine<XMax-XMin)
116 {
117 Reject("PCX");
118 return;
119 }
120
121 Accept("PCX");
122 Stream_Prepare(Stream_Image);
123 Fill(Stream_Image, 0, Image_Format, "PCX");
124 Fill(Stream_Image, 0, Image_Format_Version, Pcx_VersionInfo(Version));
125 Fill(Stream_Image, 0, Image_Width, XMax-XMin);
126 Fill(Stream_Image, 0, Image_Height, YMax-YMin);
127 Fill(Stream_Image, 0, Image_BitDepth, BitsPerPixel);
128 Fill(Stream_Image, 0, "DPI", Ztring::ToZtring(VertDPI) + __T(" x ") + Ztring::ToZtring(HorDPI));
129 Finish("PCX");
130 FILLING_END();
131}
