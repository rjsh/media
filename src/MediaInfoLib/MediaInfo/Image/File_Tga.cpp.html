44const char* Tga_Image_Type_Compression(int8u Image_Type)
45{
46 switch (Image_Type)
47 {
48 case 1 : return "Color-mapped";
49 case 2 :
50 case 3 : return "Raw";
51 case 9 : return "Color-mapped + RLE";
52 case 10 :
53 case 11 : return "RLE";
54 case 32 :
55 case 33 : return "Huffman";
56 default : return "";
57 }
58}

62{
63 switch (Image_Type)
64 {
65 case 1 :
66 case 2 :
67 case 9 :
68 case 10 :
69 case 32 :
70 case 33 : return "RGB";
71 case 3 :
72 case 11 : return "Y";
73 default : return "";
74 }
75}

93//---------------------------------------------------------------------------
94void File_Tga::Streams_Fill()
95{
96 Fill(Stream_General, 0, General_Format, "TGA");
97 Fill(Stream_General, 0, General_Format_Version, __T("Version ")+Ztring::ToZtring(Version));
98 Fill(Stream_General, 0, General_Title, Image_ID);
99
100 Stream_Prepare(Stream_Image);
101 Fill(Stream_Image, 0, Image_Format, Tga_Image_Type_Compression(Image_Type));
102 Fill(Stream_Image, 0, Image_ColorSpace, Tga_Image_Type_ColorSpace(Image_Type));
103 Fill(Stream_Image, 0, Image_CodecID, Image_Type);
104 Fill(Stream_Image, 0, Image_Width, Image_Width_);
105 Fill(Stream_Image, 0, Image_Height, Image_Height_);
106 Fill(Stream_Image, 0, Image_BitDepth, Pixel_Depth);
107}

109//***************************************************************************
110// Buffer - File header
111//***************************************************************************
112
113//---------------------------------------------------------------------------
114bool File_Tga::FileHeader_Begin()
115{
116 //Synchro
117 if (18>Buffer_Size)
118 return false;
119 if (Buffer[2]==0x00
120 || Buffer[16]>32) //bit depth
121 {
122 Reject();
123 return false;
124 }
125 if (Buffer_Size<File_Size)
126 return false; //Must wait for more data
127
128 //All should be OK...
129 return true;
130}

132//***************************************************************************
133// Buffer - Global
134//***************************************************************************
135
136//---------------------------------------------------------------------------
137void File_Tga::Read_Buffer_Continue()
138{
139 //Parsing
140 Tga_File_Header();
141 Image_Color_Map_Data();
142 Tga_File_Footer();
143
144 FILLING_BEGIN();
145 //Coherency in case of no magic value
146 if (Version==1)
147 {
148 switch (Image_Type)
149 {
150 case 1 : //Color mapped images
151 case 9 : //Color mapped images, RLE
152 if (Color_Map_Type!=1)
153 {
154 Reject();
155 return;
156 }
157 break;
158 case 2 : //True-color images
159 case 10 : //True-color images, RLE
160 if (Color_Map_Type)
161 {
162 Reject();
163 return;
164 }
165 break;
166 case 3 : //Black and White
167 case 11 : //Black and White, RLE
168 if (Color_Map_Type)
169 {
170 Reject();
171 return;
172 }
173 break;
174 default : Reject();
175 return;
176 }
177 switch (Color_Map_Type)
178 {
179 case 0 :
180 if (First_Entry_Index || Color_map_Length || Color_map_Entry_Size)
181 {
182 Reject();
183 return;
184 }
185 break;
186 case 1 :
187 switch (Color_map_Entry_Size)
188 {
189 case 15 :
190 case 16 :
191 case 24 :
192 case 32 :
193 break;
194 default : Reject();
195 return;
196 ;
197 }
198 break;
199 default : Reject();
200 return;
201 }
202 switch (Pixel_Depth)
203 {
204 case 8 :
205 case 16 :
206 case 24 :
207 case 32 :
208 break;
209 default : Reject();
210 return;
211 ;
212 }
213 }
214
215 Accept();
216 Fill();
217 Finish();
218 FILLING_END();
219}

225//---------------------------------------------------------------------------
226void File_Tga::Tga_File_Header()
227{
228 //Parsing
229 Element_Begin1("Tga File Header");
230 Get_L1 (ID_Length, "ID Length");
231 Get_L1 (Color_Map_Type, "Color Map Type");
232 Get_L1 (Image_Type, "Image Type"); Param_Info1(Tga_Image_Type_Compression(Image_Type));
233 Element_End0();
234 Element_Begin1("Color Map Specification");
235 Get_L2 (First_Entry_Index, "First Entry Index");
236 Get_L2 (Color_map_Length, "Color map Length");
237 Get_L1 (Color_map_Entry_Size, "Color map Entry Size");
238 Element_End0();
239 Element_Begin1("Image Specification");
240 Skip_L2( "X-origin of Image");
241 Skip_L2( "Y-origin of Image");
242 Get_L2 (Image_Width_, "Image Width");
243 Get_L2 (Image_Height_, "Image Height");
244 Get_L1 (Pixel_Depth, "Pixel Depth");
245 Get_L1 (Image_Descriptor, "Image Descriptor");
246 Element_End0();
247}
248
249//---------------------------------------------------------------------------
250void File_Tga::Image_Color_Map_Data()
251{
252 Element_Begin1("Image/Color Map Data");
253 Get_Local(ID_Length, Image_ID, "Image ID");
254 if (Color_Map_Type==1)
255 {
256 int64u EntrySizeInBits=8;
257 if (Color_map_Entry_Size<24)
258 EntrySizeInBits=Color_map_Entry_Size/3;
259
260 Skip_XX(Color_map_Length*EntrySizeInBits/8, "Color Map Data");
261
262 }
263 if (Element_Offset+26<Element_Size
264 && Buffer[Buffer_Size- 18]==0x54
265 && Buffer[Buffer_Size- 17]==0x52
266 && Buffer[Buffer_Size- 16]==0x55
267 && Buffer[Buffer_Size- 15]==0x45
268 && Buffer[Buffer_Size- 14]==0x56
269 && Buffer[Buffer_Size- 13]==0x49
270 && Buffer[Buffer_Size- 12]==0x53
271 && Buffer[Buffer_Size- 11]==0x49
272 && Buffer[Buffer_Size- 10]==0x4F
273 && Buffer[Buffer_Size- 9]==0x4E
274 && Buffer[Buffer_Size- 8]==0x2D
275 && Buffer[Buffer_Size- 7]==0x58
276 && Buffer[Buffer_Size- 6]==0x46
277 && Buffer[Buffer_Size- 5]==0x49
278 && Buffer[Buffer_Size- 4]==0x4C
279 && Buffer[Buffer_Size- 3]==0x45
280 && Buffer[Buffer_Size- 2]==0x2E
281 && Buffer[Buffer_Size- 1]==0x00)
282 Version=2;
283 else
284 Version=1;
285 Skip_XX(Element_Size-Element_Offset-(Version==2?26:0), "Image Data");
286 Element_End0();
287}
288
289//---------------------------------------------------------------------------
290void File_Tga::Tga_File_Footer()
291{
292 if (Version==1)
293 return; //No footer
294
295 Element_Begin1("Image/color Map Data");
296 Skip_L4( "Extension Area Offset");
297 Skip_L4( "Developer Directory Offset");
298 Skip_Local(16, "Signature");
299 Skip_Local( 1, "Reserved Character");
300 Skip_L1( "Binary Zero String Terminator");
301 Element_End0();
302}
