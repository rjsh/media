44const char* Png_Colour_type(int8u Colour_type)
45{
46 switch (Colour_type)
47 {
48 case 0 : return "Greyscale";
49 case 2 : return "Truecolour";
50 case 3 : return "Indexed-colour";
51 case 4 : return "Greyscale with alpha";
52 case 6 : return "Truecolour with alpha";
53 default: return "";
54 }

92void File_Png::Streams_Accept()
93{
94 if (!IsSub)
95 {
96 TestContinuousFileNames();
97
98 Stream_Prepare((Config->File_Names.size()>1 || Config->File_IsReferenced_Get())?Stream_Video:Stream_Image);
99 if (File_Size!=(int64u)-1)
100 Fill(StreamKind_Last, StreamPos_Last, Fill_Parameter(StreamKind_Last, Generic_StreamSize), File_Size);
101 if (StreamKind_Last==Stream_Video)
102 Fill(Stream_Video, StreamPos_Last, Video_FrameCount, Config->File_Names.size());
103 }
104 else
105 Stream_Prepare(StreamKind_Last);
106}

108//***************************************************************************
109// Header
110//***************************************************************************
111
112//---------------------------------------------------------------------------
113bool File_Png::FileHeader_Begin()
114{
115 //Element_Size
116 if (Buffer_Size<8)
117 return false; //Must wait for more data
118
119 if (CC4(Buffer+4)!=0x0D0A1A0A) //Byte order
120 {
121 Reject("PNG");
122 return false;
123 }
124
125 switch (CC4(Buffer)) //Signature
126 {
127 case 0x89504E47 :
128 Accept("PNG");
129
130 Fill(Stream_General, 0, General_Format, "PNG");
131 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_Format), "PNG");
132 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_Codec), "PNG");
133
134 break;
135
136 case 0x8A4E4E47 :
137 Accept("PNG");
138
139 Fill(Stream_General, 0, General_Format, "MNG");
140 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_Format), "MNG");
141 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_Codec), "MNG");
142
143 Finish("PNG");
144 break;
145
146 case 0x8B4A4E47 :
147 Accept("PNG");
148
149 Fill(Stream_General, 0, General_Format, "JNG");
150 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_Format), "JNG");
151 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_Codec), "JNG");
152
153 Finish("PNG");
154 break;
155
156 default:
157 Reject("PNG");
158 }
159
160 //All should be OK...
161 return true;
162}

176//***************************************************************************
177// Buffer - Per element
178//***************************************************************************
179
180//---------------------------------------------------------------------------
181void File_Png::Header_Parse()
182{
183 if (!Signature_Parsed)
184 {
185 //Filling
186 Header_Fill_Size(8);
187 Header_Fill_Code(0, "File header");
188
189 return;
190 }
191
192 //Parsing
193 int32u Length, Chunk_Type;
194 Get_B4 (Length, "Length");
195 Get_C4 (Chunk_Type, "Chunk Type");
196
197 //Filling
198 Header_Fill_Size(12+Length); //+4 for CRC
199 Header_Fill_Code(Chunk_Type, Ztring().From_CC4(Chunk_Type));
200}
201
202//---------------------------------------------------------------------------
203void File_Png::Data_Parse()
204{
205 if (!Signature_Parsed)
206 {
207 Signature();
208 return;
209 }
210
211 Element_Size-=4; //For CRC
212
213 #define CASE_INFO(_NAME, _DETAIL) \
214 case Elements::_NAME : Element_Info1(_DETAIL); _NAME(); break;
215
216 //Parsing
217 switch (Element_Code)
218 {
219 CASE_INFO(IDAT, "Image data");
220 CASE_INFO(IEND, "Image trailer");
221 CASE_INFO(IHDR, "Image header");
222 CASE_INFO(PLTE, "Palette table");
223 default : Skip_XX(Element_Size, "Unknown");
224 }
225
226 Element_Size+=4; //For CRC
227 Skip_B4( "CRC");
228}

235void File_Png::Signature()
236{

239 Skip_B4( "ByteOrder");

243 Frame_Count_NotParsedIncluded++;
244 Signature_Parsed=true;

253//---------------------------------------------------------------------------
254void File_Png::IHDR()
255{
256 //Parsing
257 int32u Width, Height;
258 int8u Bit_depth, Colour_type, Compression_method, Interlace_method;
259 Get_B4 (Width, "Width");
260 Get_B4 (Height, "Height");
261 Get_B1 (Bit_depth, "Bit depth");
262 Get_B1 (Colour_type, "Colour type"); Param_Info1(Png_Colour_type(Colour_type));
263 Get_B1 (Compression_method, "Compression method");
264 Skip_B1( "Filter method");
265 Get_B1 (Interlace_method, "Interlace method");
266
267 FILLING_BEGIN_PRECISE();
268 if (!Status[IsFilled])
269 {
270 Fill(StreamKind_Last, 0, "Width", Width);
271 Fill(StreamKind_Last, 0, "Height", Height);
272 int8u Resolution;
273 switch (Colour_type)
274 {
275 case 0 : Resolution=Bit_depth; break;
276 case 2 : Resolution=Bit_depth*3; break;
277 case 3 : Resolution=Bit_depth; break;
278 case 4 : Resolution=Bit_depth*2; break;
279 case 6 : Resolution=Bit_depth*4; break;
280 default: Resolution=0;
281 }
282 if (Resolution)
283 Fill(StreamKind_Last, 0, "BitDepth", Resolution);
284 switch (Compression_method)
285 {
286 case 0 :
287 Fill(StreamKind_Last, 0, "Format_Compression", "LZ77");
288 break;
289 default: ;
290 }
291 switch (Interlace_method)
292 {
293 case 0 :
294 break;
295 case 1 :
296 break;
297 default: ;
298 }
299
300 Fill();
301 }
302
303 if (Config->ParseSpeed<1.0)
304 Finish("PNG"); //No need of more
305 FILLING_END();
306}
