42//---------------------------------------------------------------------------
43bool File_Gif::FileHeader_Begin()
44{
45 //Element_Size
46 if (Buffer_Size<3)
47 return false; //Must wait for more data
48
49 if (CC3(Buffer)!=0x474946) //"GIF"
50 {
51 Reject("GIF");
52 return false;
53 }
54
55 //All should be OK...
56 return true;
57}

59//***************************************************************************
60// Buffer - Global
61//***************************************************************************
62
63//---------------------------------------------------------------------------
64void File_Gif::Read_Buffer_Continue()
65{
66 //Parsing
67 Ztring Version;
68 int16u Width, Height;
69 int8u BackgroundColorIndex, PixelAspectRatio, Resolution, GCT_Size;
70 bool GCT_Flag, Sort;
71 Skip_Local(3, "Header");
72 Get_Local (3, Version, "Version");
73 Get_L2 (Width, "Logical Screen Width");
74 Get_L2 (Height, "Logical Screen Height");
75 BS_Begin();
76 Get_SB ( GCT_Flag, "Global Color Table Flag");
77 Get_S1 (3, Resolution, "Color Resolution");
78 Get_SB ( Sort, "Sort Flag to Global Color Table");
79 Get_S1 (3, GCT_Size, "Size of Global Color Table"); Param_Info1(Ztring::ToZtring((int16u)pow(2.0, 1+GCT_Size)));
80 BS_End();
81 Get_L1 (BackgroundColorIndex, "Background Color Index");
82 Get_L1 (PixelAspectRatio, "Pixel Aspect Ratio");
83 if (GCT_Flag)
84 Skip_XX((int16u)pow(2.0, 1+GCT_Size)*3, "Global Color Table");
85 Element_End0();
86
87 FILLING_BEGIN();
88 Accept("GIF");
89
90 Stream_Prepare(Stream_Image);
91 Fill(Stream_Image, 0, Image_Width, Width);
92 Fill(Stream_Image, 0, Image_Height, Height);
93 Fill(Stream_Image, 0, Image_Format, __T("GIF"));
94 Fill(Stream_Image, 0, Image_Format_Profile, Version);
95 Fill(Stream_Image, 0, Image_Codec, __T("GIF")+Version);
96 if (PixelAspectRatio)
97 Fill(Stream_Image, 0, Image_PixelAspectRatio, (((float)PixelAspectRatio)+15)/64);
98
99 Finish("GIF");
100 FILLING_END();
101}
