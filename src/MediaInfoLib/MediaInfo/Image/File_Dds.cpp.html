57//---------------------------------------------------------------------------
58void File_Dds::Streams_Accept()
59{
60 Fill(Stream_General, 0, General_Format, "DDS");
61
62 if (!IsSub)
63 {
64 TestContinuousFileNames();
65
66 Stream_Prepare((Config->File_Names.size()>1 || Config->File_IsReferenced_Get())?Stream_Video:Stream_Image);
67 Fill(StreamKind_Last, StreamPos_Last, "StreamSize", File_Size);
68 if (StreamKind_Last==Stream_Video)
69 Fill(Stream_Video, StreamPos_Last, Video_FrameCount, Config->File_Names.size());
70 if (pfFlags&0x4) //DDPF_FOURCC
71 CodecID_Fill(Ztring().From_CC4(FourCC), StreamKind_Last, StreamPos_Last, InfoCodecID_Format_Riff, Stream_Video);
72 if (Flags&0x2) //DDSD_HEIGHT
73 Fill(StreamKind_Last, 0, "Height", Height);
74 if (Flags&0x4) //DDSD_WIDTH
75 Fill(StreamKind_Last, 0, "Width", Width);
76 if (Flags&0x800000) //DDSD_DEPTH
77 Fill(StreamKind_Last, 0, "BitDepth", Depth);
78 }
79 else
80 Stream_Prepare(StreamKind_Last);
81}

83//***************************************************************************
84// Header
85//***************************************************************************
86
87//---------------------------------------------------------------------------
88bool File_Dds::FileHeader_Begin()
89{
90 // Minimum buffer size
91 if (Buffer_Size<8)
92 return false; // Must wait for more data
93
94 // Testing
95 if (Buffer[0]!=0x44 // "DDS "
96 || Buffer[1]!=0x44
97 || Buffer[2]!=0x53
98 || Buffer[3]!=0x20
99 || LittleEndian2int32u(Buffer+4)<124)
100 {
101 Reject();
102 return false;
103 }
104
105 //All should be OK...
106 return true;
107}

109//---------------------------------------------------------------------------
110void File_Dds::FileHeader_Parse()
111{
112 //Parsing
113 int32u Size;
114 Skip_C4( "Magic");
115 Get_L4 (Size, "Size");
116 Get_L4 (Flags, "Flags");
117 Get_L4 (Height, "Height");
118 Get_L4 (Width, "Width");
119 Skip_L4( "PitchOrLinearSize");
120 Skip_L4( "Depth");
121 Skip_L4( "MipMapCount");
122 Skip_XX(4*11, "Reserved1");
123 Element_Begin1("Pixel format");
124 int32u pf_Size;
125 Get_L4 (pf_Size, "Size");
126 if (pf_Size>=32)
127 {
128 Get_L4 (pfFlags, "Flags");
129 Get_C4 (FourCC, "FourCC");
130 Skip_L4( "RGBBitCount");
131 Skip_L4( "RBitMask");
132 Skip_L4( "GBitMask");
133 Skip_L4( "BBitMask");
134 Skip_L4( "ABitMask");
135 if (pf_Size>32)
136 Skip_XX(Size-32, "(Data)");
137 }
138 else if (pf_Size>4)
139 Skip_XX(pf_Size-4, "");
140 Element_End0();
141 Skip_L4( "Caps");
142 Skip_L4( "Caps2");
143 Skip_L4( "Caps3");
144 Skip_L4( "Caps4");
145 Skip_L4( "Reserved2");
146 if (Size>124)
147 Skip_XX(Size-124, "(Data)");
148}

160//---------------------------------------------------------------------------
161void File_Dds::Read_Buffer_Continue()
162{
163 Skip_XX(File_Size-(File_Offset+Buffer_Offset), "Data");
164
165 FILLING_BEGIN();
166 Frame_Count++;
167 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
168 Frame_Count_NotParsedIncluded++;
169 if (!Status[IsAccepted])
170 {
171 Accept();
172 Fill();
173 if (Config->ParseSpeed<1.0)
174 Finish();
175 }
176 FILLING_END();
177}
