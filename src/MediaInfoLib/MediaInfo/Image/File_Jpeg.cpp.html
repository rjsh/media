146//---------------------------------------------------------------------------
147string Jpeg2000_Rsiz(int16u Rsiz)
148{
149 switch (Rsiz)
150 {
151 case 0x0000: return "No restrictions";
152 case 0x0001: return "Profile-0";
153 case 0x0002: return "Profile-1";
154 case 0x0003: return "D-Cinema 2k";
155 case 0x0004: return "D-Cinema 4k";
156 case 0x0005: return "D-Cinema 2k Scalable";
157 case 0x0006: return "D-Cinema 4k Scalable";
158 case 0x0007: return "Long-term storage";
159 case 0x0101: return "BCS@L1"; //Broadcast Contribution Single Tile
160 case 0x0102: return "BCS@L2"; //Broadcast Contribution Single Tile
161 case 0x0103: return "BCS@L3"; //Broadcast Contribution Single Tile
162 case 0x0104: return "BCS@L4"; //Broadcast Contribution Single Tile
163 case 0x0105: return "BCS@L5"; //Broadcast Contribution Single Tile
164 case 0x0205: return "BCM@L5"; //Broadcast Contribution Multi-tile
165 case 0x0306: return "BCMR@L6"; //Broadcast Contribution Multi-tile Reversible
166 case 0x0307: return "BCMR@L7"; //Broadcast Contribution Multi-tile Reversible
167 default: return Ztring::ToZtring(Rsiz, 16).To_UTF8();
168 }

233bool File_Jpeg::FileHeader_Begin()
234{
235 //Element_Size
236 if (Buffer_Size<3)
237 return false; //Must wait for more data
238
239 if (Buffer[2]!=0xFF
240 || (CC2(Buffer)!=Elements::SOI
241 && CC2(Buffer)!=Elements::SOC))
242 {
243 Reject("JPEG");
244 return false;
245 }
246
247 //All should be OK...
248 return true;
249}

253//***************************************************************************
254
255//---------------------------------------------------------------------------
256bool File_Jpeg::Synchronize()
257{
258 //Synchronizing
259 while(Buffer_Offset+2<=Buffer_Size && (Buffer[Buffer_Offset ]!=0xFF
260 || Buffer[Buffer_Offset+1]==0x00))
261 Buffer_Offset++;
262
263 if (Buffer_Offset+1==Buffer_Size && Buffer[Buffer_Offset ]!=0xFF)
264 Buffer_Offset++;
265
266 if (Buffer_Offset+2>Buffer_Size)
267 return false;
268
269 //Synched is OK
270 Synched=true;
271 return true;
272}

307//---------------------------------------------------------------------------
308#if MEDIAINFO_DEMUX
309bool File_Jpeg::Demux_UnpacketizeContainer_Test()
310{
311 if (!IsSub)
312 {
313 if (!Status[IsAccepted])
314 Accept();
315 if (Config->File_Names.size()>1)
316 return Demux_UnpacketizeContainer_Test_OneFramePerFile();
317 }
318
319 if (Interlaced && Buffer_Offset==0)
320 {
321 bool StartIsFound=false;
322 while (Demux_Offset+2<=Buffer_Size)
323 {
324 int16u code=BigEndian2int16u(Buffer+Demux_Offset);
325 Demux_Offset+=2;
326 switch (code)
327 {
328 case Elements::SOD : //JPEG-2000 start
329 StartIsFound=true;
330 case Elements::TEM :
331 case Elements::RST0 :
332 case Elements::RST1 :
333 case Elements::RST2 :
334 case Elements::RST3 :
335 case Elements::RST4 :
336 case Elements::RST5 :
337 case Elements::RST6 :
338 case Elements::RST7 :
339 case Elements::SOC :
340 case Elements::SOI :
341 case Elements::EOI :
342 break;
343 default :
344 if (Demux_Offset+2>Buffer_Size)
345 break;
346 {
347 int16u size=BigEndian2int16u(Buffer+Demux_Offset);
348 if (Demux_Offset+2+size>Buffer_Size)
349 break;
350 Demux_Offset+=size;
351 if (code==Elements::SOS) //JPEG start
352 StartIsFound=true;
353 }
354 }
355 if (StartIsFound)
356 break;
357 }
358
359 while (Demux_Offset+2<=Buffer_Size)
360 {
361 while (Demux_Offset<Buffer_Size && Buffer[Demux_Offset]!=0xFF)
362 Demux_Offset++;
363 if (Demux_Offset+2<=Buffer_Size && Buffer[Demux_Offset+1]==0xD9) //EOI (JPEG 2000)
364 break;
365 Demux_Offset++;
366 }
367 if (Demux_Offset+2<=Buffer_Size)
368 Demux_Offset+=2;
369 }
370 else
371 Demux_Offset=Buffer_Size;
372
373 if (Interlaced)
374 {
375 if (Field_Count==0 && FrameRate && Demux_Offset!=Buffer_Size)
376 FrameRate*=2; //Now field rate
377 if (FrameRate)
378 FrameInfo.DUR=float64_int64s(1000000000/FrameRate); //Actually, field or frame rate
379 }
380
381 Demux_UnpacketizeContainer_Demux();
382
383 if (Interlaced)
384 {
385 if (FrameInfo.DTS!=(int64u)-1 && FrameInfo.DUR!=(int64u)-1)
386 FrameInfo.DTS+=FrameInfo.DUR;
387 }
388
389 return true;
390}
391#endif //MEDIAINFO_DEMUX

436 if (Interlaced)
437 Field_Count+=2;
438 Frame_Count++;
439 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
440 Frame_Count_NotParsedIncluded++;
441 }

449//---------------------------------------------------------------------------
450void File_Jpeg::Header_Parse()
451{
452 if (SOS_SOD_Parsed)
453 {
454 Header_Fill_Code(0, "Data");
455 if (!Header_Parser_Fill_Size())
456 {
457 Element_WaitForMoreData();
458 return;
459 }
460 return;
461 }
462
463 //Parsing
464 int16u code, size;
465 Get_B2 (code, "Marker");
466 switch (code)
467 {
468 case Elements::TEM :
469 case Elements::RST0 :
470 case Elements::RST1 :
471 case Elements::RST2 :
472 case Elements::RST3 :
473 case Elements::RST4 :
474 case Elements::RST5 :
475 case Elements::RST6 :
476 case Elements::RST7 :
477 case Elements::SOC :
478 case Elements::SOD :
479 case Elements::SOI :
480 case Elements::EOI :
481 size=0; break;
482 default : Get_B2 (size, "Fl - Frame header length");
483 }
484
485 //Filling
486 Header_Fill_Code(code, Ztring().From_CC2(code));
487 Header_Fill_Size(2+size);
488}
489
490//---------------------------------------------------------------------------
491bool File_Jpeg::Header_Parser_Fill_Size()
492{
493 //Look for next Sync word
494 if (Buffer_Offset_Temp==0) //Buffer_Offset_Temp is not 0 if Header_Parse_Fill_Size() has already parsed first frames
495 Buffer_Offset_Temp=Buffer_Offset;
496
497 #if MEDIAINFO_DEMUX
498 if (Buffer_TotalBytes+2<Demux_TotalBytes)
499 Buffer_Offset_Temp=(size_t)(Demux_TotalBytes-(Buffer_TotalBytes+2));
500 #endif //MEDIAINFO_DEMUX
501
502 while (Buffer_Offset_Temp+2<=Buffer_Size)
503 {
504 while (Buffer_Offset_Temp<Buffer_Size && Buffer[Buffer_Offset_Temp]!=0xFF)
505 Buffer_Offset_Temp++;
506 if (Buffer_Offset_Temp+2<=Buffer_Size && Buffer[Buffer_Offset_Temp+1]==0xD9) //EOI
507 break;
508 Buffer_Offset_Temp++;
509 }
510
511 //Must wait more data?
512 if (Buffer_Offset_Temp+2>Buffer_Size)
513 {
514 if (/*FrameIsAlwaysComplete ||*/ File_Offset+Buffer_Size>=File_Size)
515 Buffer_Offset_Temp=Buffer_Size; //We are sure that the next bytes are a start
516 else
517 return false;
518 }
519
520 //OK, we continue
521 Header_Fill_Size(Buffer_Offset_Temp-Buffer_Offset);
522 Buffer_Offset_Temp=0;
523 return true;
524}

632//---------------------------------------------------------------------------
633void File_Jpeg::SIZ()
634{
635 //Parsing
636 vector<float> SamplingFactors;
637 vector<int8u> BitDepths;
638 int8u SamplingFactors_Max=0;
639 int32u Xsiz, Ysiz;
640 int16u Rsiz, Count;
641 Get_B2 (Rsiz, "Rsiz - Capability of the codestream");
642 Get_B4 (Xsiz, "Xsiz - Image size X");
643 Get_B4 (Ysiz, "Ysiz - Image size Y");
644 Skip_B4( "XOsiz - Image offset X");
645 Skip_B4( "YOsiz - Image offset Y");
646 Skip_B4( "tileW - Size of tile W");
647 Skip_B4( "tileH - Size of tile H");
648 Skip_B4( "XTOsiz - Upper-left tile offset X");
649 Skip_B4( "YTOsiz - Upper-left tile offset Y");
650 Get_B2 (Count, "Components and initialize related arrays");
651 for (int16u Pos=0; Pos<Count; Pos++)
652 {
653 Element_Begin1("Initialize related array");
654 int8u BitDepth, compSubsX, compSubsY;
655 BS_Begin();
656 Skip_SB( "Signed");
657 Get_S1 (7, BitDepth, "BitDepth"); Param_Info1(1+BitDepth); Element_Info1(1+BitDepth);
658 BS_End();
659 Get_B1 ( compSubsX, "compSubsX"); Element_Info1(compSubsX);
660 Get_B1 ( compSubsY, "compSubsY"); Element_Info1(compSubsY);
661 Element_End0();
662
663 //Filling list of HiVi
664 if (compSubsX)
665 {
666 SamplingFactors.push_back(((float)compSubsY)/compSubsX);
667 if (((float)compSubsY)/compSubsX>SamplingFactors_Max)
668 SamplingFactors_Max=(int8u)((float)compSubsY)/compSubsX;
669 }
670
671 if (BitDepths.empty() || BitDepth!=BitDepths[0])
672 BitDepths.push_back(BitDepth);
673 }
674
675 FILLING_BEGIN_PRECISE();
676 if (Frame_Count==0 && Field_Count==0)
677 {
678 Accept("JPEG 2000");
679 Fill("JPEG 2000");
680
681 if (Count_Get(StreamKind_Last)==0)
682 Stream_Prepare(StreamKind_Last);
683 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_Format), "JPEG 2000");
684 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_Codec), "JPEG 2000");
685 Fill(StreamKind_Last, 0, "Format_Profile", Jpeg2000_Rsiz(Rsiz));
686 if (StreamKind_Last==Stream_Image)
687 Fill(Stream_Image, 0, Image_Codec_String, "JPEG 2000", Unlimited, true, true); //To Avoid automatic filling
688 Fill(StreamKind_Last, 0, StreamKind_Last==Stream_Image?(size_t)Image_Width:(size_t)Video_Width, Xsiz);
689 Fill(StreamKind_Last, 0, StreamKind_Last==Stream_Image?(size_t)Image_Height:(size_t)Video_Height, Ysiz*(Interlaced?2:1)); //If image is from interlaced content, must multiply height by 2
690
691 if (BitDepths.size()==1)
692 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_BitDepth), 1+BitDepths[0]);

695 if (SamplingFactors_Max)

701 }
702 while (SamplingFactors.size()<3)

705 for (size_t Pos=0; Pos<SamplingFactors.size(); Pos++)

707 if (!ChromaSubsampling.empty())
708 {
709 ChromaSubsampling.resize(ChromaSubsampling.size()-1);
710 Fill(StreamKind_Last, 0, "ChromaSubsampling", ChromaSubsampling);
711
712 //Not for sure
713 if (ChromaSubsampling==__T("4:4:4") && (Retrieve(StreamKind_Last, 0, "Format_Profile")==__T("D-Cinema 2k") || Retrieve(StreamKind_Last, 0, "Format_Profile")==__T("D-Cinema 4k")))
714 Fill(StreamKind_Last, 0, "ColorSpace", "XYZ");
715 else if (!IsSub)
716 {
717 if (ChromaSubsampling==__T("4:2:0") || ChromaSubsampling==__T("4:2:2"))
718 Fill(StreamKind_Last, 0, "ColorSpace", "YUV");
719 else if (ChromaSubsampling==__T("4:4:4"))
720 Fill(StreamKind_Last, 0, "ColorSpace", "RGB");

811//---------------------------------------------------------------------------
812void File_Jpeg::SOF_()
813{
814 //Parsing
815 vector<Jpeg_samplingfactor> SamplingFactors;
816 int16u Height, Width;
817 int8u Resolution, Count;
818 Get_B1 (Resolution, "P - Sample precision");
819 Get_B2 (Height, "Y - Number of lines");
820 Get_B2 (Width, "X - Number of samples per line");
821 Get_B1 (Count, "Nf - Number of image components in frame");
822 for (int8u Pos=0; Pos<Count; Pos++)
823 {
824 Jpeg_samplingfactor SamplingFactor;
825 Element_Begin1("Component");
826 Get_B1 ( SamplingFactor.Ci, "Ci - Component identifier"); if (SamplingFactor.Ci>Count) Element_Info1(Ztring().append(1, (Char)SamplingFactor.Ci)); else Element_Info1(SamplingFactor.Ci);
827 BS_Begin();
828 Get_S1 (4, SamplingFactor.Hi, "Hi - Horizontal sampling factor"); Element_Info1(SamplingFactor.Hi);
829 Get_S1 (4, SamplingFactor.Vi, "Vi - Vertical sampling factor"); Element_Info1(SamplingFactor.Vi);
830 BS_End();
831 Skip_B1( "Tqi - Quantization table destination selector");
832 Element_End0();
833
834 //Filling list of HiVi
835 SamplingFactors.push_back(SamplingFactor);
836 }
837
838 FILLING_BEGIN_PRECISE();
839 if (Frame_Count==0 && Field_Count==0)
840 {
841 Accept("JPEG");
842 Fill("JPEG");
843
844 if (Count_Get(StreamKind_Last)==0)
845 Stream_Prepare(StreamKind_Last);
846 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_Format), "JPEG");
847 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_Codec), "JPEG");
848 if (StreamKind_Last==Stream_Image)
849 Fill(Stream_Image, 0, Image_Codec_String, "JPEG", Unlimited, true, true); //To Avoid automatic filling
850 if (StreamKind_Last==Stream_Video)
851 Fill(Stream_Video, 0, Video_InternetMediaType, "video/JPEG", Unlimited, true, true);
852 Fill(StreamKind_Last, 0, Fill_Parameter(StreamKind_Last, Generic_BitDepth), Resolution);
853 Fill(StreamKind_Last, 0, "Height", Height*(Interlaced?2:1));
854 Fill(StreamKind_Last, 0, "Width", Width);
855
856 //ColorSpace from http://docs.oracle.com/javase/1.4.2/docs/api/javax/imageio/metadata/doc-files/jpeg_metadata.html
857 switch (APPE_Adobe0_transform)
858 {
859 case 0x01 :
860 if (Count==3)
861 Fill(StreamKind_Last, 0, "ColorSpace", "YUV");
862 break;
863 case 0x02 :
864 if (Count==4)
865 Fill(StreamKind_Last, 0, "ColorSpace", "YCCB");
866 break;
867 default :
868 {
869 int8u Ci[256];
870 memset(Ci, 0, 256);
871 for (int8u Pos=0; Pos<Count; Pos++)
872 Ci[SamplingFactors[Pos].Ci]++;
873
874 switch (Count)
875 {
876 case 1 : Fill(StreamKind_Last, 0, "ColorSpace", "Y"); break;
877 case 2 : Fill(StreamKind_Last, 0, "ColorSpace", "YA"); break;
878 case 3 :
879 if (!APP0_JFIF_Parsed && Ci['R']==1 && Ci['G']==1 && Ci['B']==1) //RGB
880 Fill(StreamKind_Last, 0, "ColorSpace", "RGB");
881 else if ((Ci['Y']==1 && ((Ci['C']==1 && Ci['c']==1) //YCc
882 || Ci['C']==2)) //YCC
883 || APP0_JFIF_Parsed //APP0 JFIF header present so YCC
884 || APPE_Adobe0_transform==0 //transform set to YCC
885 || (SamplingFactors[0].Ci==0 && SamplingFactors[1].Ci==1 && SamplingFactors[2].Ci==2) //012
886 || (SamplingFactors[0].Ci==1 && SamplingFactors[1].Ci==2 && SamplingFactors[2].Ci==3)) //123
887 Fill(StreamKind_Last, 0, "ColorSpace", "YUV");
888 break;
889 case 4 :
890 if (!APP0_JFIF_Parsed && Ci['R']==1 && Ci['G']==1 && Ci['B']==1 && Ci['A']==1) //RGBA
891 Fill(StreamKind_Last, 0, "ColorSpace", "RGBA");
892 else if ((Ci['Y']==1 && Ci['A']==1 && ((Ci['C']==1 && Ci['c']==1) //YCcA
893 || Ci['C']==2)) //YCCA
894 || APP0_JFIF_Parsed //APP0 JFIF header present so YCCA
895 || (SamplingFactors[0].Ci==0 && SamplingFactors[1].Ci==1 && SamplingFactors[2].Ci==2 && SamplingFactors[3].Ci==3) //0123
896 || (SamplingFactors[0].Ci==1 && SamplingFactors[1].Ci==2 && SamplingFactors[2].Ci==3 && SamplingFactors[3].Ci==4)) //1234
897 Fill(StreamKind_Last, 0, "ColorSpace", "YUVA");
898 else if (APPE_Adobe0_transform==0) //transform set to CMYK
899 Fill(StreamKind_Last, 0, "ColorSpace", "YCCB");
900 break;
901 default: ;
902 }
903 }
904 }
905
906 //Chroma subsampling
907 if ((SamplingFactors.size()==3 || SamplingFactors.size()==4) && SamplingFactors[1].Hi==1 && SamplingFactors[2].Hi==1 && SamplingFactors[1].Vi==1 && SamplingFactors[2].Vi==1)
908 {
909 string ChromaSubsampling;
910 switch (SamplingFactors[0].Hi)
911 {
912 case 1 :
913 switch (SamplingFactors[0].Vi)
914 {
915 case 1 : ChromaSubsampling="4:4:4"; break;
916 default: ;
917 }
918 break;
919 case 2 :
920 switch (SamplingFactors[0].Vi)
921 {
922 case 1 : ChromaSubsampling="4:2:2"; break;
923 case 2 : ChromaSubsampling="4:2:0"; break;
924 default: ;
925 }
926 break;
927 case 4 :
928 switch (SamplingFactors[0].Vi)
929 {
930 case 1 : ChromaSubsampling="4:1:1"; break;
931 case 2 : ChromaSubsampling="4:1:0"; break;
932 default: ;
933 }
934 break;
935 default: ;
936 }
937 if (!ChromaSubsampling.empty())
938 {
939 if (SamplingFactors.size()==4)
940 {
941 if (ChromaSubsampling=="4:4:4" && SamplingFactors[3].Hi==1 && SamplingFactors[3].Vi==1)
942 ChromaSubsampling+=":4";
943 else
944 ChromaSubsampling+=":?";
945 }
946 Fill(StreamKind_Last, 0, "ChromaSubsampling", ChromaSubsampling);
947 }
948 }
949 }
950 FILLING_END();
951}
