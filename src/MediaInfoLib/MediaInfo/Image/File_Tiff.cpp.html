205//---------------------------------------------------------------------------
206bool File_Tiff::FileHeader_Begin()
207{
208 //Element_Size
209 /* Minimum header for a tiff file is 8 byte */
210 if (Buffer_Size<8)
211 return false; //Must wait for more data
212 if (CC4(Buffer)==0x49492A00)
213 LittleEndian = true;
214 else if (CC4(Buffer)==0x4D4D002A)
215 LittleEndian = false;
216 else
217 {
218 Reject("TIFF");
219 return false;
220 }
221
222 //All should be OK...
223 Accept("TIFF");
224 Fill(Stream_General, 0, General_Format, "TIFF");
225 return true;
226}

228//---------------------------------------------------------------------------
229void File_Tiff::FileHeader_Parse()
230{
231 //The only IFD that is known at forehand is the first one, it's offset is placed byte 4-7 in the file.
232 int32u IFDOffset;
233 Skip_B4( "Magic");
234 Get_X4 (IFDOffset, "IFDOffset");
235
236 FILLING_BEGIN();
237 //Initial IFD
238 GoTo(IFDOffset, "TIFF");
239 FILLING_END();
240}

246//---------------------------------------------------------------------------
247void File_Tiff::Header_Parse()
248{
249 //Handling remaining IFD data
250 if (!IfdItems.empty())
251 {
252 if (File_Offset+Buffer_Offset!=IfdItems.begin()->first)
253 IfdItems.clear(); //There was a problem during the seek, trashing remaining positions from last IFD
254 else
255 {
256 Header_Fill_Code(IfdItems.begin()->second.Tag, Tiff_Tag_Name(IfdItems.begin()->second.Tag));
257 Header_Fill_Size(Tiff_Type_Size(IfdItems.begin()->second.Type)*IfdItems.begin()->second.Count);
258 return;
259 }
260 }
261
262 /* A tiff images consist in principle of two types of blocks, IFD's and data blocks */
263 /* Each datablock, which could be a image, tiles, transperancy filter is described by one IFD. */
264 /* These IFD's can be placed at any offset in the file and are linked in a chain fashion way. */
265 /* where one IFD points out where the next IFD is placed */
266 /* */
267 /* A creator of a tiff file must describe the "main image" in the first IFD, this means that a */
268 /* reader, such this one, only need to read the first IFD in order to get the bitdepth, resolution etc. */
269 /* of the main image. */
270
271 /* Read one IFD and print out the result */
272
273 /* Scan the tiff file for the IFD's (Image File Directory) */
274 /* As long as the IFD offset to the next IFD in the file is not 0 */
275
276 /* Get number of directories for this IFD */
277 int16u NrOfDirectories;
278 Get_X2 (NrOfDirectories, "NrOfDirectories");
279
280 //Filling
281 Header_Fill_Code(0xFFFFFFFF, "IFD"); //OxFFFFFFFF can not be a Tag, so using it as a magic value
282 Header_Fill_Size(2+12*((int64u)NrOfDirectories)+4); //2 for header + 12 per directory + 4 for next IFD offset
283}

285//---------------------------------------------------------------------------
286void File_Tiff::Data_Parse()
287{
288 int32u IFDOffset=0;
289 if (IfdItems.empty())
290 {
291 //Default values
292 Infos.clear();
293 Infos[Tiff_Tag::BitsPerSample]=__T("1");
294
295 //Parsing new IFD
296 while (Element_Offset+8+4<Element_Size)
297 Read_Directory();
298 Get_X4 (IFDOffset, "IFDOffset");
299 }
300 else
301 {
302 //Handling remaining IFD data from a previous IFD
303 GetValueOffsetu(IfdItems.begin()->second); //Parsing the IFD item
304 IfdItems.erase(IfdItems.begin()->first); //Removing IFD item from the list of IFD items to parse
305 }
306
307 //Some items are not inside the directory, jumping to the offset
308 if (!IfdItems.empty())
309 GoTo(IfdItems.begin()->first, "TIFF");
310 else
311 {
312 //This IFD is finished, filling data then going to next IFD
313 Data_Parse_Fill();
314 if (IFDOffset)
315 GoTo(IFDOffset, "TIFF");
316 else
317 {
318 Finish(); //No more IFDs
319 GoToFromEnd(0);
320 }
321 }
322}

324//---------------------------------------------------------------------------
325void File_Tiff::Data_Parse_Fill()
326{
327 Stream_Prepare(Stream_Image);
328
329 infos::iterator Info;
330
331 //Width
332 Info=Infos.find(Tiff_Tag::ImageWidth);
333 if (Info!=Infos.end())
334 Fill(Stream_Image, StreamPos_Last, Image_Width, Info->second.Read());
335
336 //Height
337 Info=Infos.find(Tiff_Tag::ImageLength);
338 if (Info!=Infos.end())
339 Fill(Stream_Image, StreamPos_Last, Image_Height, Info->second.Read());
340
341 //BitsPerSample
342 Info=Infos.find(Tiff_Tag::BitsPerSample);
343 if (Info!=Infos.end())
344 {
345 if (Info->second.size()>1)
346 {
347 bool IsOk=true;
348 for (size_t Pos=1; Pos<Info->second.size(); ++Pos)
349 if (Info->second[Pos]!=Info->second[0])
350 IsOk=false;
351 if (IsOk)
352 Info->second.resize(1); //They are all same, we display 1 piece of information
353 }
354
355 Fill(Stream_Image, StreamPos_Last, Image_BitDepth, Info->second.Read());
356 }
357
358 //Compression
359 Info=Infos.find(Tiff_Tag::Compression);
360 if (Info!=Infos.end())
361 {
362 int32u Value=Info->second.Read().To_int32u();
363 Fill(Stream_Image, StreamPos_Last, Image_Format, Tiff_Compression(Value));
364 Fill(Stream_Image, StreamPos_Last, Image_Codec, Tiff_Compression(Value));
365 Fill(Stream_Image, StreamPos_Last, Image_Compression_Mode, Tiff_Compression_Mode(Value));
366 }
367
368 //PhotometricInterpretation
369 Info=Infos.find(Tiff_Tag::PhotometricInterpretation);
370 if (Info!=Infos.end())
371 {
372 int32u Value=Info->second.Read().To_int32u();
373 Fill(Stream_Image, StreamPos_Last, Image_ColorSpace, Tiff_PhotometricInterpretation_ColorSpace(Value));
374 //Note: should we differeniate between raw RGB and palette (also RGB actually...)
375 }
376
377 //ImageDescription
378 Info=Infos.find(Tiff_Tag::ImageDescription);
379 if (Info!=Infos.end())
380 Fill(Stream_Image, StreamPos_Last, Image_Title, Info->second.Read());
381
382 //ExtraSamples
383 Info=Infos.find(Tiff_Tag::ExtraSamples);
384 if (Info!=Infos.end())
385 {
386 Ztring ColorSpace=Retrieve(Stream_Image, StreamPos_Last, Image_ColorSpace);
387 ColorSpace+=Ztring().From_Local(Tiff_ExtraSamples_ColorSpace(Info->second.Read().To_int32u()));
388 Fill(Stream_Image, StreamPos_Last, Image_ColorSpace, ColorSpace, true);
389 }
390}

396//---------------------------------------------------------------------------
397void File_Tiff::Read_Directory()
398{
399 /* Each directory consist of 4 fields */
400 /* Get information for this directory */
401 Element_Begin0();
402 ifditem IfdItem;
403 Get_X2 (IfdItem.Tag, "Tag"); Param_Info1(Tiff_Tag_Name(IfdItem.Tag));
404 Get_X2 (IfdItem.Type, "Type"); Param_Info1(Tiff_Type_Name(IfdItem.Type));
405 Get_X4 (IfdItem.Count, "Count");
406 Element_Name(Tiff_Tag_Name(IfdItem.Tag));
407
408 if (Tiff_Type_Size(IfdItem.Type)*IfdItem.Count<=4)
409 {
410 GetValueOffsetu(IfdItem);
411
412 /* Padding up, skip dummy bytes */
413 if (Tiff_Type_Size(IfdItem.Type)==0)
414 {
415 if (Element_Offset+4<Element_Size)
416 Skip_XX(Element_Size-(Element_Offset+4), "Unknown");
417 }
418 else if (Tiff_Type_Size(IfdItem.Type)*IfdItem.Count<4)
419 Skip_XX(Tiff_Type_Size(IfdItem.Type)*IfdItem.Count, "Padding");
420 }
421 else
422 {
423 int32u IFDOffset;
424 Get_X4 (IFDOffset, "IFDOffset");
425 IfdItems[IFDOffset]=IfdItem;
426 }
427 Element_End0();
428}

452//---------------------------------------------------------------------------
453void File_Tiff::GetValueOffsetu(ifditem &IfdItem)
454{
455 ZtringList &Info=Infos[IfdItem.Tag]; Info.clear(); Info.Separator_Set(0, __T(" / "));
456 const char* Name=Tiff_Tag_Name(IfdItem.Tag);
457
458 if (IfdItem.Count>=10)
459 {
460 //Too many data, we don't currently need it and we skip it
461 Skip_XX(Element_Size-(Element_Offset+4), Name);
462 Info.clear();
463 return;
464 }
465
466 switch (IfdItem.Type)
467 {
468 case 1: /* 8-bit unsigned integer. */
469 for (int16u Pos=0; Pos<IfdItem.Count; Pos++)
470 {
471 int8u Ret8;
472 #if MEDIAINFO_TRACE
473 if (LittleEndian)
474 Get_L1 (Ret8, Name);
475 else
476 Get_B1 (Ret8, Name);
477 Element_Info1(Ztring::ToZtring(Ret8));
478 #else //MEDIAINFO_TRACE
479 if (Element_Offset+1>Element_Size)
480 {
481 Trusted_IsNot();
482 break;
483 }
484 if (LittleEndian)
485 Ret8=LittleEndian2int8u(Buffer+Buffer_Offset+(size_t)Element_Offset);
486 else
487 Ret8=BigEndian2int8u(Buffer+Buffer_Offset+(size_t)Element_Offset);
488 Element_Offset++;
489 #endif //MEDIAINFO_TRACE
490 Info.push_back(Ztring::ToZtring(Ret8));
491 }
492 break;
493
494 case 3: /* 16-bit (2-byte) unsigned integer. */
495 for (int16u Pos=0; Pos<IfdItem.Count; Pos++)
496 {
497 int16u Ret16;
498 #if MEDIAINFO_TRACE
499 if (LittleEndian)
500 Get_L2 (Ret16, Name);
501 else
502 Get_B2 (Ret16, Name);
503 switch (IfdItem.Tag)
504 {
505 case Tiff_Tag::Compression : Element_Info1(Tiff_Compression(Ret16)); break;
506 case Tiff_Tag::PhotometricInterpretation : Element_Info1(Tiff_PhotometricInterpretation(Ret16)); break;
507 default : Element_Info1(Ztring::ToZtring(Ret16));
508 }
509 #else //MEDIAINFO_TRACE
510 if (Element_Offset+2>Element_Size)
511 {
512 Trusted_IsNot();
513 break;
514 }
515 if (LittleEndian)
516 Ret16=LittleEndian2int16u(Buffer+Buffer_Offset+(size_t)Element_Offset);
517 else
518 Ret16=BigEndian2int16u(Buffer+Buffer_Offset+(size_t)Element_Offset);
519 Element_Offset+=2;
520 #endif //MEDIAINFO_TRACE
521 Info.push_back(Ztring::ToZtring(Ret16));
522 }
523 break;
524
525 case 4: /* 32-bit (4-byte) unsigned integer */
526 for (int16u Pos=0; Pos<IfdItem.Count; Pos++)
527 {
528 int32u Ret32;
529 #if MEDIAINFO_TRACE
530 if (LittleEndian)
531 Get_L4 (Ret32, Name);
532 else
533 Get_B4 (Ret32, Name);
534 Element_Info1(Ztring::ToZtring(Ret32));
535 #else //MEDIAINFO_TRACE
536 if (Element_Offset+4>Element_Size)
537 {
538 Trusted_IsNot();
539 break;
540 }
541 if (LittleEndian)
542 Ret32=LittleEndian2int32u(Buffer+Buffer_Offset+(size_t)Element_Offset);
543 else
544 Ret32=BigEndian2int32u(Buffer+Buffer_Offset+(size_t)Element_Offset);
545 Element_Offset+=4;
546 #endif //MEDIAINFO_TRACE
547 Info.push_back(Ztring::ToZtring(Ret32));
548 }
549 break;
550
551 default: //Unknown
552 {
553 if (LittleEndian)
554 Skip_L4( Name);
555 else
556 Skip_B4( Name);
557 Info.clear(); //We actually do not know the type
558 }
559 }
