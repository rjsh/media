113 struct Element_Node_Info
114 {
115 template<typename T>
116 Element_Node_Info(T parameter, const char* _Measure=NULL, int8u Option=(int8u)-1)
117 {
118 data.set_Option(Option);
119 data = parameter;
120 if (_Measure)
121 {
122 size_t len = strlen(_Measure);
123 Measure = new char[len + 1];
124 std::memcpy(Measure, _Measure, len);
125 Measure[len] = '\0';
126 }
127 else
128 Measure = NULL;
129 }
130
131 ~Element_Node_Info()
132 {
133 delete[] Measure;
134 }
135
136 friend std::ostream& operator<<(std::ostream& os, element_details::Element_Node_Info* v);
137
138 Element_Node_Data data;
139 char* Measure;
140
141 Element_Node_Info& operator=(const Element_Node_Info&);
142
143 private:
144 Element_Node_Info(const Element_Node_Info&);
145 };
