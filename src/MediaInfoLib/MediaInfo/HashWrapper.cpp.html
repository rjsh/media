62void HashWrapper::Init (const HashFunctions &Functions)
63{
64 m = new void*[HashFunction_Max];
65
66 #if MEDIAINFO_MD5
67 if (Functions[MD5])
68 {
69 ((void**)m)[MD5]=new struct MD5Context;
70 MD5Init((struct MD5Context*)((void**)m)[MD5]);
71 }
72 else
73 ((void**)m)[MD5]=NULL;
74 #endif //MEDIAINFO_MD5
75
76 #if MEDIAINFO_SHA1
77 if (Functions[SHA1])
78 {
79 ((void**)m)[SHA1]=new sha1_ctx;
80 sha1_begin((sha1_ctx*)((void**)m)[SHA1]);
81 }
82 else
83 ((void**)m)[SHA1]=NULL;
84 #endif //MEDIAINFO_SHA1
85
86 #if MEDIAINFO_SHA2
87 if (Functions[SHA224])
88 {
89 ((void**)m)[SHA224]=new sha224_ctx;
90 sha224_begin((sha224_ctx*)((void**)m)[SHA224]);
91 }
92 else
93 ((void**)m)[SHA224]=NULL;
94 if (Functions[SHA256])
95 {
96 ((void**)m)[SHA256]=new sha256_ctx;
97 sha256_begin((sha256_ctx*)((void**)m)[SHA256]);
98 }
99 else
100 ((void**)m)[SHA256]=NULL;
101 if (Functions[SHA384])
102 {
103 ((void**)m)[SHA384]=new sha384_ctx;
104 sha384_begin((sha384_ctx*)((void**)m)[SHA384]);
105 }
106 else
107 ((void**)m)[SHA384]=NULL;
108 if (Functions[SHA512])
109 {
110 ((void**)m)[SHA512]=new sha512_ctx;
111 sha512_begin((sha512_ctx*)((void**)m)[SHA512]);
112 }
113 else
114 ((void**)m)[SHA512]=NULL;
115 #endif //MEDIAINFO_SHA2
116}
117
118HashWrapper::~HashWrapper ()
119{
120 #if MEDIAINFO_MD5
121 delete (struct MD5Context*)((void**)m)[MD5];
122 #endif //MEDIAINFO_MD5
123
124 #if MEDIAINFO_SHA1
125 delete (sha1_ctx*)((void**)m)[SHA1];
126 #endif //MEDIAINFO_SHA1
127
128 #if MEDIAINFO_SHA2
129 delete (sha224_ctx*)((void**)m)[SHA224];
130 delete (sha256_ctx*)((void**)m)[SHA256];
131 delete (sha384_ctx*)((void**)m)[SHA384];
132 delete (sha512_ctx*)((void**)m)[SHA512];
133 #endif //MEDIAINFO_SHA2
134
135 delete[] m;
136}
137
138void HashWrapper::Update (const int8u* Buffer, const size_t Buffer_Size)
139{
140 #if MEDIAINFO_MD5
141 if (((void**)m)[MD5])
142 MD5Update((struct MD5Context*)((void**)m)[MD5], Buffer, (unsigned int)Buffer_Size);
143 #endif //MEDIAINFO_MD5
144
145 #if MEDIAINFO_SHA1
146 if (((void**)m)[SHA1])
147 sha1_hash(Buffer, (unsigned long)Buffer_Size, (sha1_ctx*)((void**)m)[SHA1]);
148 #endif //MEDIAINFO_SHA1
149
150 #if MEDIAINFO_SHA2
151 if (((void**)m)[SHA224])
152 sha224_hash(Buffer, (unsigned long)Buffer_Size, (sha224_ctx*)((void**)m)[SHA224]);
153 if (((void**)m)[SHA256])
154 sha256_hash(Buffer, (unsigned long)Buffer_Size, (sha256_ctx*)((void**)m)[SHA256]);
155 if (((void**)m)[SHA384])
156 sha384_hash(Buffer, (unsigned long)Buffer_Size, (sha384_ctx*)((void**)m)[SHA384]);
157 if (((void**)m)[SHA512])
158 sha512_hash(Buffer, (unsigned long)Buffer_Size, (sha512_ctx*)((void**)m)[SHA512]);
159 #endif //MEDIAINFO_SHA2
160}

162string HashWrapper::Generate (const HashFunction Function)
163{
164 #if MEDIAINFO_MD5
165 if (Function==MD5 && ((void**)m)[MD5])
166 {
167 unsigned char Digest[16];
168 MD5Final(Digest, (struct MD5Context*)((void**)m)[MD5]);
169 return Hex2String(Digest, 16);
170 }
171 #endif //MEDIAINFO_MD5
172
173 #if MEDIAINFO_SHA1
174 if (Function==SHA1 && ((void**)m)[SHA1])
175 {
176 unsigned char Digest[20];
177 sha1_end(Digest, (sha1_ctx*)((void**)m)[SHA1]);
178 return Hex2String(Digest, 20);
179 }
180 #endif //MEDIAINFO_SHA1
181
182 #if MEDIAINFO_SHA2
183 if (Function==SHA224 && ((void**)m)[SHA224])
184 {
185 unsigned char Digest[28];
186 sha224_end(Digest, (sha224_ctx*)((void**)m)[SHA224]);
187 return Hex2String(Digest, 28);
188 }
189 if (Function==SHA256 && ((void**)m)[SHA256])
190 {
191 unsigned char Digest[32];
192 sha256_end(Digest, (sha256_ctx*)((void**)m)[SHA256]);
193 return Hex2String(Digest, 32);
194 }
195 if (Function==SHA384 && ((void**)m)[SHA384])
196 {
197 unsigned char Digest[48];
198 sha384_end(Digest, (sha384_ctx*)((void**)m)[SHA384]);
199 return Hex2String(Digest, 48);
200 }
201 if (Function==SHA512 && ((void**)m)[SHA512])
202 {
203 unsigned char Digest[64];
204 sha512_end(Digest, (sha512_ctx*)((void**)m)[SHA512]);
205 return Hex2String(Digest, 64);
206 }
207 #endif //MEDIAINFO_SHA2
208
209 return string();
210}

212string HashWrapper::Name (const HashFunction Function)
213{
214 #if MEDIAINFO_MD5
215 if (Function==MD5)
216 return "MD5";
217 #endif //MEDIAINFO_MD5
218
219 #if MEDIAINFO_SHA1
220 if (Function==SHA1)
221 return "SHA-1";
222 #endif //MEDIAINFO_SHA1
223
224 #if MEDIAINFO_SHA2
225 if (Function==SHA224)
226 return "SHA-224";
227 if (Function==SHA256)
228 return "SHA-256";
229 if (Function==SHA384)
230 return "SHA-384";
231 if (Function==SHA512)
232 return "SHA-512";
233 #endif //MEDIAINFO_SHA2
234
235 return string();
236}

238string HashWrapper::Hex2String(const int8u* Digest, const size_t Digest_Size)
239{
240 string DigestS;
241 DigestS.resize(Digest_Size*2);
242 for (size_t i=0, j=0; i<Digest_Size; ++i)
243 {
244 DigestS[j++]=HashWrapper_Hex[Digest[i] >> 4];
245 DigestS[j++]=HashWrapper_Hex[Digest[i] & 0xF];
246 }
247 return DigestS;
248}
