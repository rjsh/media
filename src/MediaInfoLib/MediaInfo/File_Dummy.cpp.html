35void File_Dummy::FileHeader_Parse()
36{
37 File_Name=__T("D:\\Example\\"); File_Name+=KindOfDummy;
38 if (KindOfDummy==__T("Album"))
39 {
40 Fill_Dummy_General();
41 Fill_Dummy_Audio();
42 }
43 else if (KindOfDummy==__T("Comic"))
44 {
45 Fill_Dummy_General();
46 Fill_Dummy_Video();
47 }
48 else if (KindOfDummy==__T("Movie"))
49 {
50 Fill_Dummy_General();
51 Fill_Dummy_Video();
52 Fill_Dummy_Video();
53 Fill_Dummy_Audio();
54 Fill_Dummy_Audio();
55 Fill_Dummy_Audio();
56 Fill_Dummy_Audio();
57 Fill_Dummy_Text();
58 Fill_Dummy_Text();
59 Fill_Dummy_Text();
60 Fill_Dummy_Text();
61 Fill_Dummy_Chapters();
62 Fill_Dummy_Chapters();
63 }
64 else
65 {
66 File_Name=__T("D:\\WhatIsIt.mkv");
67 Fill(Stream_General, 0, General_Domain, "Starwars saga");
68 Fill(Stream_General, 0, General_Movie, "Starwars 4");
69 Fill(Stream_General, 0, General_Movie_More, "A new hope");
70 Fill(Stream_General, 0, General_Director, "Georges Lucas");
71 Fill(Stream_General, 0, General_Released_Date, "1977");
72 Fill(Stream_General, 0, General_FileSize, "734000000");
73 Fill(Stream_General, 0, General_Format, "Matroska");
74 Fill(Stream_General, 0, General_Format_Url, "http://MediaArea.net/MediaInfo");
75 Stream_Prepare(Stream_Video);
76 Fill(Stream_Video, 0, Video_Codec, "XVID");
77 Fill(Stream_Video, 0, Video_Codec_Url, "http://MediaArea.net/MediaInfo");
78 Fill(Stream_Video, 0, Video_BitRate, "800000");
79 Fill(Stream_Video, 0, Video_Width, "720");
80 Fill(Stream_Video, 0, Video_Height, "320");
81 Fill(Stream_Video, 0, Video_FrameRate, "24.976");
82 Fill(Stream_Video, 0, Video_BitDepth, "8");
83 Fill(Stream_Video, 0, Video_Language, "en");
84 Stream_Prepare(Stream_Audio);
85 Fill(Stream_Audio, 0, Audio_Format, "AC-3");
86 Fill(Stream_Audio, 0, Audio_Codec, "AC3");
87 Fill(Stream_Audio, 0, Audio_Codec_Url, "http://MediaArea.net/MediaInfo");
88 Fill(Stream_Audio, 0, Audio_BitRate, "384000");
89 Fill(Stream_Audio, 0, Audio_BitRate_Mode, "CBR");
90 Fill(Stream_Audio, 0, Audio_Channel_s_, "6");
91 Fill(Stream_Audio, 0, Audio_SamplingRate, "48000");
92 Fill(Stream_Audio, 0, Audio_Language, "en");
93 Stream_Prepare(Stream_Text);
94 Fill(Stream_Text, 0, Text_Codec, "SSA");
95 Fill(Stream_Text, 0, Text_Codec_Url, "http://MediaArea.net/MediaInfo");
96 Fill(Stream_Text, 0, Text_Language, "en");
97 Fill(Stream_Text, 0, Text_Language_More, "Forced");
98 Stream_Prepare(Stream_Other);
99 Fill(Stream_Other, 0, Chapters_Total, "16");
100 Fill(Stream_Other, 0, Chapters_Language, "en");
101 }
102
103 Accept();
104}
