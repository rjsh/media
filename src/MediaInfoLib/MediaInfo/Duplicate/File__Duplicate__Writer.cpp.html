62//***************************************************************************
63// Configure
64//***************************************************************************
65
66//---------------------------------------------------------------------------
67void File__Duplicate__Writer::Configure (const Ztring &Target)
68{
69 //Form: "memory://pointer:size" <--Memory block is specified by user
70 //WARNING: pointer must be in ***DECIMAL*** format.
71 //Example: "memory://123456789:1316"
72 if (Target.find(__T("memory://"))==0 && Target.find(__T(":"), 9)!=std::string::npos)
73 {
74 size_t SemiColumn_Pos=Target.find(__T(":"), 9);
75 Ztring Address=Target.substr(9, SemiColumn_Pos-9);
76 Ztring Size=Target.substr(SemiColumn_Pos+1);
77 Method=method_buffer;
78 Buffer=(int8u*)Address.To_int64u();
79 Buffer_Size_Max=(size_t)Size.To_int64u();
80 }
81
82 //Form: "file://filename" or <--the exported filename is specified by user
83 else if (Target.find(__T("file://"))==0)
84 {
85 Method=method_filename;
86 File_Name=Target.substr(7, std::string::npos);
87 }
88}
89
90//***************************************************************************
91// Write
92//***************************************************************************
93
94void File__Duplicate__Writer::Write (const int8u* ToAdd, size_t ToAdd_Size)
95{
96 //Integrity
97 if (ToAdd==NULL || ToAdd_Size==0)
98 return;
99
100 //Writing
101 switch (Method)
102 {
103 //File based
104 case method_filename :
105 //Open the file if not yet done
106 if (File_Pointer==NULL)
107 {
108 File_Pointer=new File;
109 ((File*)File_Pointer)->Open(File_Name, File::Access_Write_Append);
110 }
111 //Write to the file
112 ((File*)File_Pointer)->Write(ToAdd, ToAdd_Size);
113 break;
114
115 //Memory based
116 case method_buffer :
117 //Testing if enough place
118 if (Buffer_Size+ToAdd_Size>Buffer_Size_Max)
119 {
120 Buffer_Size=0; //Buffer underrun, cleaning the buffer
121 if (ToAdd_Size>Buffer_Size_Max)
122 ToAdd_Size=0; //Buffer is too small, writing nothing
123 }
124
125 //Copying buffer
126 std::memcpy(Buffer+Buffer_Size, ToAdd, ToAdd_Size);
127 Buffer_Size+=ToAdd_Size;
128 break;
129 default : ;
130 }
131}

138size_t File__Duplicate__Writer::Output_Buffer_Get ()
139{
140 size_t Output_Buffer_Size=Buffer_Size;

142 return Output_Buffer_Size;
143}

145//---------------------------------------------------------------------------
146size_t File__Duplicate__Writer::Output_Buffer_Get (unsigned char** Output_Buffer)
147{
148 if (Output_Buffer)
149 *Output_Buffer=Buffer;
150 size_t Output_Buffer_Size=Buffer_Size;
151 Buffer_Size=0; //Can be erased later...
152 return Output_Buffer_Size;
153}
