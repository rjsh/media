41//---------------------------------------------------------------------------
42// CRC_32_Table
43// A CRC is computed like this:
44// Init: int32u CRC_32 = 0xFFFFFFFF;
45// for each data byte do
46// CRC_32=(CRC_32<<8) ^ CRC_32_Table[(CRC_32>>24)^(data_byte)];
47extern int32u Psi_CRC_32_Table[256];

69bool File__Duplicate_MpegTs::Configure (const Ztring &Value, bool ToRemove)
70{
71 //Form: "program_number"
72 if (Value.find(__T("program_number="))==0)
73 {
74 int16u program_number=Ztring(Value.substr(15, std::string::npos)).To_int16u();
75 if (ToRemove)
76 {
77 if (Wanted_program_numbers.find(program_number)!=Wanted_program_numbers.end())
78 Wanted_program_numbers.erase(program_number);
79 else if (Remove_program_numbers.find(program_number)==Remove_program_numbers.end())
80 Remove_program_numbers.insert(program_number);
81 }
82 else
83 {
84 if (Remove_program_numbers.find(program_number)!=Remove_program_numbers.end())
85 Remove_program_numbers.erase(program_number);
86 if (Wanted_program_numbers.find(program_number)==Wanted_program_numbers.end())
87 Wanted_program_numbers.insert(program_number);
88 }
89 if (!PAT.empty())
90 PAT.begin()->second.ConfigurationHasChanged=true;
91 }
92 //Form: "program_map_PID"
93 else if (Value.find(__T("program_map_PID="))==0)
94 {
95 int16u program_map_PID=Ztring(Value.substr(16, std::string::npos)).To_int16u();
96 if (ToRemove)
97 {
98 if (Wanted_program_map_PIDs.find(program_map_PID)!=Wanted_program_map_PIDs.end())
99 Wanted_program_map_PIDs.erase(program_map_PID);
100 else if (Remove_program_map_PIDs.find(program_map_PID)==Remove_program_map_PIDs.end())
101 Remove_program_map_PIDs.insert(program_map_PID);
102 }
103 else
104 {
105 if (Remove_program_map_PIDs.find(program_map_PID)!=Remove_program_map_PIDs.end())
106 Remove_program_map_PIDs.erase(program_map_PID);
107 if (Wanted_program_map_PIDs.find(program_map_PID)==Wanted_program_map_PIDs.end())
108 Wanted_program_map_PIDs.insert(program_map_PID);
109 }
110 if (PMT.find(program_map_PID)!=PMT.end())
111 PMT[program_map_PID].ConfigurationHasChanged=true;
112 }
113 //Form: "elementary_PID"
114 else if (Value.find(__T("elementary_PID="))==0)
115 {
116 int16u elementary_PID=Ztring(Value.substr(15, std::string::npos)).To_int16u();
117 if (ToRemove)
118 {
119 if (Wanted_elementary_PIDs.find(elementary_PID)!=Wanted_elementary_PIDs.end())
120 Wanted_elementary_PIDs.erase(elementary_PID);
121 else if (Remove_elementary_PIDs.find(elementary_PID)==Remove_elementary_PIDs.end())
122 Remove_elementary_PIDs.insert(elementary_PID);
123 }
124 else
125 {
126 if (Remove_elementary_PIDs.find(elementary_PID)!=Remove_elementary_PIDs.end())
127 Remove_elementary_PIDs.erase(elementary_PID);
128 if (Wanted_elementary_PIDs.find(elementary_PID)==Wanted_elementary_PIDs.end())
129 Wanted_elementary_PIDs.insert(elementary_PID);
130 }
131 if (PMT.find(elementary_PIDs_program_map_PIDs[elementary_PID])!=PMT.end())
132 PMT[elementary_PIDs_program_map_PIDs[elementary_PID]].ConfigurationHasChanged=true;
133 }
134 //Old
135 else
136 {
137 int16u program_number=Ztring(Value).To_int16u();
138 if (ToRemove)
139 {
140 if (Wanted_program_numbers.find(program_number)!=Wanted_program_numbers.end())
141 Wanted_program_numbers.erase(program_number);
142 else if (Remove_program_numbers.find(program_number)==Remove_program_numbers.end())
143 Remove_program_numbers.insert(program_number);
144 }
145 else
146 {
147 if (Remove_program_numbers.find(program_number)!=Remove_program_numbers.end())
148 Remove_program_numbers.erase(program_number);
149 if (Wanted_program_numbers.find(program_number)==Wanted_program_numbers.end())
150 Wanted_program_numbers.insert(program_number);
151 }
152 if (!PAT.empty())
153 PAT.begin()->second.ConfigurationHasChanged=true;
154 }
155
156 //Can be disabled?
157 if (Wanted_program_numbers.empty()
158 && Wanted_program_map_PIDs.empty()
159 && Wanted_elementary_PIDs.empty()
160 && Remove_program_numbers.empty()
161 && Remove_program_map_PIDs.empty()
162 && Remove_elementary_PIDs.empty())
163 return true; //It can be erased
164 else
165 return false; //We always need it
166}

172bool File__Duplicate_MpegTs::Write (int16u PID, const int8u* ToAdd, size_t ToAdd_Size)
173{
174 if (elementary_PIDs[PID])
175 {
176 Writer.Write(ToAdd, ToAdd_Size);
177 return false;
178 }
179 else if (program_map_PIDs[PID])
180 return Manage_PMT(ToAdd, ToAdd_Size);
181 else if (PID==0x0000)
182 return Manage_PAT(ToAdd, ToAdd_Size);
183 else
184 return false;
185}
186
187bool File__Duplicate_MpegTs::Manage_PAT (const int8u* ToAdd, size_t ToAdd_Size)
188{
189 if (!Parsing_Begin(ToAdd, ToAdd_Size, PAT))
190 return false;
191
192 //Programs
193 program_map_PIDs.clear();
194 program_map_PIDs.resize(0x2000, 0);
195 elementary_PIDs.clear();
196 elementary_PIDs.resize(0x2000, 0);
197 while (FromTS.Offset+4<=FromTS.End)
198 {
199 //For each program
200 int16u program_number =CC2(FromTS.Buffer+FromTS.Offset+0);
201 int16u program_map_PID=CC2(FromTS.Buffer+FromTS.Offset+2)&0x1FFF;
202 if (Wanted_program_numbers.find(program_number) !=Wanted_program_numbers.end()
203 || Wanted_program_map_PIDs.find(program_map_PID)!=Wanted_program_map_PIDs.end())
204 {
205 //Integrating it
206 program_map_PIDs[program_map_PID]=1;
207 std::memcpy(PAT[StreamID].Buffer+PAT[StreamID].Offset, FromTS.Buffer+FromTS.Offset, 4);
208 PAT[StreamID].Offset+=4;
209 PMT[program_number].ConfigurationHasChanged=true;
210 }
211 FromTS.Offset+=4;
212 }
213
214 Parsing_End(PAT);
215
216 //Reseting
217 std::vector<int16u> StreamID_List;
218 for (std::map<int16u, buffer>::iterator PAT_=PAT.begin(); PAT_!=PAT.end(); ++PAT_)
219 if (PAT_->first!=StreamID)
220 StreamID_List.push_back(PAT_->first);
221 for (size_t Pos=0; Pos<StreamID_List.size(); Pos++)
222 PAT[StreamID_List[Pos]].FromTS_version_number_Last=0xFF;
223
224 return true;
225}
226
227bool File__Duplicate_MpegTs::Manage_PMT (const int8u* ToAdd, size_t ToAdd_Size)
228{
229 if (!Parsing_Begin(ToAdd, ToAdd_Size, PMT))
230 return false;
231
232 //Testing program_number
233 if (Wanted_program_numbers.find(StreamID)==Wanted_program_numbers.end()
234 && Wanted_program_map_PIDs.find(elementary_PIDs_program_map_PIDs[StreamID]) == Wanted_program_map_PIDs.end())
235 {
236 delete[] PMT[StreamID].Buffer; PMT[StreamID].Buffer=NULL;
237 return false;
238 }
239
240 //program_info_length
241 int16u program_info_length=CC2(FromTS.Buffer+FromTS.Offset+2)&0x0FFF;
242 std::memcpy(PMT[StreamID].Buffer+PMT[StreamID].Offset, FromTS.Buffer+FromTS.Offset, 4+program_info_length);
243 FromTS.Offset+=4+program_info_length;
244 PMT[StreamID].Offset+=4+program_info_length;
245
246 //elementary_PIDs
247 while (FromTS.Offset+5<=FromTS.End)
248 {
249 //For each elementary_PID
250 int16u elementary_PID=CC2(FromTS.Buffer+FromTS.Offset+1)&0x1FFF;
251 int16u ES_info_length=CC2(FromTS.Buffer+FromTS.Offset+3)&0x0FFF;
252 if (Wanted_elementary_PIDs.empty() || Wanted_elementary_PIDs.find(elementary_PID)!=Wanted_elementary_PIDs.end())
253 {
254 //Integrating it
255 elementary_PIDs[elementary_PID]=1;
256 elementary_PIDs_program_map_PIDs[elementary_PID]=StreamID;
257 std::memcpy(PMT[StreamID].Buffer+PMT[StreamID].Offset, FromTS.Buffer+FromTS.Offset, 5+ES_info_length);
258 PMT[StreamID].Offset+=5+ES_info_length;
259 }
260 else
261 elementary_PIDs[elementary_PID]=0;
262 FromTS.Offset+=5+ES_info_length;
263 }
264
265 Parsing_End(PMT);
266 return true;
267}

273bool File__Duplicate_MpegTs::Parsing_Begin (const int8u* ToAdd, size_t ToAdd_Size, std::map<int16u, buffer> &ToModify_)
274{
275 //Managing big chunks
276 int16u PID=((ToAdd[1]&0x1F)<<8)|ToAdd[2]; //BigEndian2int16u(ToAdd+1)&0x1FFF;
277 if (ToAdd[1]&0x40) //payload_unit_start_indicator
278 {
279 FromTS.Buffer=ToAdd;
280 FromTS.Size=ToAdd_Size;
281 FromTS.Offset=0;
282 }
283 else
284 {
285 if (BigBuffers.find(PID)==BigBuffers.end())
286 return false; //Start is missing
287 if (ToAdd_Size<4 || BigBuffers[PID].Buffer_Size+ToAdd_Size-4>BigBuffers[PID].Buffer_Size_Max)
288 return false; //Problem
289 std::memcpy(BigBuffers[PID].Buffer+BigBuffers[PID].Buffer_Size, ToAdd+4, ToAdd_Size-4);
290 BigBuffers[PID].Buffer_Size+=ToAdd_Size-4;
291
292 FromTS.Buffer=BigBuffers[PID].Buffer;
293 FromTS.Size=BigBuffers[PID].Buffer_Size;
294 FromTS.Offset=0;
295 }
296
297 //adaptation_field_length
298 int8u adaptation_field_length=0;
299 if (CC1(FromTS.Buffer+3)&0x20) //adaptation_field_control (adaptation) == true
300 adaptation_field_length=1+CC1(FromTS.Buffer+4);
301
302 //pointer_field
303 FromTS.Offset+=4+adaptation_field_length;
304 int8u pointer_field=CC1(FromTS.Buffer+FromTS.Offset);
305
306 //table_id
307 FromTS.Offset+=1+pointer_field;
308 int8u table_id=FromTS.Buffer[FromTS.Offset];
309 if (table_id!=0x00 && table_id!=0x02) //Currently only PAT and PMT are handled
310 return false;
311
312 //section_length
313 FromTS.Offset++;
314 if (FromTS.Offset+2>FromTS.Size)
315 return false;
316 FromTS.Begin=FromTS.Offset-1;
317 int16u section_length=CC2(FromTS.Buffer+FromTS.Offset)&0x0FFF;
318 FromTS.End=4+adaptation_field_length+section_length;
319
320 //Positionning just after section_length
321 FromTS.Offset+=2;
322
323 //Retrieving StreamID
324 StreamID=CC2(FromTS.Buffer+FromTS.Offset);
325 buffer &ToModify=ToModify_[StreamID];
326
327 //version_number
328 int8u FromTS_version_number=(CC1(FromTS.Buffer+FromTS.Offset+2)>>1)&0x1F;
329 if (ToModify.version_number==0xFF && FromTS.End<=FromTS.Size-4) //Only if we have enough data
330 ToModify.version_number=FromTS_version_number;
331 if (ToModify.continuity_counter==0xFF && FromTS.End<=FromTS.Size-4) //Only if we have enough data
332 ToModify.continuity_counter=FromTS.Buffer[3]&0xF;
333 if (FromTS_version_number!=ToModify.FromTS_version_number_Last || ToModify.ConfigurationHasChanged)
334 {
335 if (FromTS.End<=FromTS.Size-4) //Only if we have enough data
336 {
337 ToModify.version_number++;
338 if (ToModify.version_number>0x1F)
339 ToModify.version_number=0;
340 ToModify.FromTS_version_number_Last=FromTS_version_number;
341 ToModify.ConfigurationHasChanged=false;
342 }
343 }
344 else
345 {
346 if (ToModify.Buffer==NULL)
347 return false;
348
349 //This is the same as before --> Copying the last version, except continuity_counter (incremented)
350 for (size_t Pos=0; Pos<ToModify.Size; Pos+=188)
351 {
352 ToModify.continuity_counter++;
353 if (ToModify.continuity_counter>0x0F)
354 ToModify.continuity_counter=0x00;
355 ToModify.Buffer[Pos+3]&=0xF0;
356 ToModify.Buffer[Pos+3]|=ToModify.continuity_counter;
357 }
358
359 //Managing big chunks
360 if (BigBuffers.find(PID)!=BigBuffers.end())
361 BigBuffers.erase(BigBuffers.find(PID));
362
363 Writer.Write(ToModify.Buffer, ToModify.Size);
364 return false;
365 }
366
367 //Test if we have enough data
368 if (FromTS.End>FromTS.Size-4)
369 {
370 //Waiting for more data
371 if (BigBuffers[PID].Buffer==NULL)
372 {
373 //Saving the data (not done at the beginning)
374 BigBuffers[PID].Buffer_Size=FromTS.Size;
375 BigBuffers[PID].Buffer_Size_Max=FromTS.End+188;
376 BigBuffers[PID].Buffer=new int8u[BigBuffers[PID].Buffer_Size_Max];
377 std::memcpy(BigBuffers[PID].Buffer, ToAdd, ToAdd_Size);
378 }
379 return false;
380 }
381
382 //Verifying CRC
383 int32u CRC_32=0xFFFFFFFF;
384 for (int32u CRC_32_Offset=(int32u)FromTS.Begin; CRC_32_Offset<FromTS.End+4; CRC_32_Offset++) //After syncword
385 CRC_32=(CRC_32<<8) ^ Psi_CRC_32_Table[(CRC_32>>24)^(FromTS.Buffer[CRC_32_Offset])];
386 if (CRC_32)
387 return false; //Problem
388
389 //Copying
390 if (ToModify.Buffer!=NULL && ToModify.Size<FromTS.Size)
391 {delete[] ToModify.Buffer; ToModify.Buffer=NULL;}
392 if (ToModify.Buffer==NULL)
393 ToModify.Buffer=new int8u[FromTS.Size+(FromTS.Size/188)*4];
394 std::memcpy(ToModify.Buffer, FromTS.Buffer, FromTS.Begin+8); //Only up to last_section_number included
395 ToModify.Offset=FromTS.Offset;
396 ToModify.Begin=FromTS.Begin;
397 ToModify.End=FromTS.End;
398 ToModify.Size=FromTS.Size;
399
400 //Changing version_number
401 int8u ToReplace=FromTS.Buffer[FromTS.Offset+2];
402 ToReplace&=0xC1; //11000001, for removing old version_number
403 ToReplace|=ToModify.version_number<<1; //merging, with 1 bit offset
404 ToModify.Buffer[ToModify.Offset+2]=ToReplace;
405
406 //Positionning after last_section_number
407 ToModify.Offset+=5;
408 FromTS.Offset+=5;
409
410 return true;
411}
412
413void File__Duplicate_MpegTs::Parsing_End (std::map<int16u, buffer> &ToModify_)
414{
415 buffer &ToModify=ToModify_[StreamID];
416
417 ToModify.End=ToModify.Offset;
418 if (ToModify.End+4>ToModify.Size)
419 return; //There was an error somewhere!
420
421 //section_length
422 int8u ToReplace=CC1(ToModify.Buffer+ToModify.Begin+1)&0xF0; //before section_length
423 int16u section_length=(int16u)(ToModify.End-ToModify.Begin+1); //+4 for CRC, -3 for header size
424 ToReplace|=section_length>>8;
425 ToModify.Buffer[ToModify.Begin+1+0]=ToReplace;
426 ToModify.Buffer[ToModify.Begin+1+1]=(int8u)(section_length&0xFF);
427
428 //CRC32
429 int32u CRC_32=0xFFFFFFFF;
430 for (size_t Buffer_CRC_Pos=ToModify.Begin; Buffer_CRC_Pos<ToModify.End; Buffer_CRC_Pos++)
431 CRC_32=(CRC_32<<8) ^ Psi_CRC_32_Table[(CRC_32>>24)^(ToModify.Buffer[Buffer_CRC_Pos])];
432
433 ToModify.Buffer[ToModify.Offset+0]=(CRC_32>>24)&0xFF;
434 ToModify.Buffer[ToModify.Offset+1]=(CRC_32>>16)&0xFF;
435 ToModify.Buffer[ToModify.Offset+2]=(CRC_32>> 8)&0xFF;
436 ToModify.Buffer[ToModify.Offset+3]= CRC_32 &0xFF;
437
438 //Managing big chunks
439 for (size_t Pos=188; Pos<ToModify.Size; Pos+=188)
440 {
441 std::memmove(ToModify.Buffer+Pos+4, ToModify.Buffer+Pos, ToModify.Size-Pos);
442 std::memcpy(ToModify.Buffer+Pos, ToModify.Buffer, 4);
443 ToModify.Buffer[Pos+1]&=0xBF; //Removing payload_unit_start_indicator
444 ToModify.Offset+=4;
445 ToModify.Size+=4;
446
447 ToModify.continuity_counter++;
448 if (ToModify.continuity_counter>0x0F)
449 ToModify.continuity_counter=0x00;
450 ToModify.Buffer[Pos+3]&=0xF0;
451 ToModify.Buffer[Pos+3]|=ToModify.continuity_counter;
452 }
453 while (ToModify.Size-(ToModify.Offset+4)>188)
454 ToModify.Size-=188;
455
456 //Padding
457 for (size_t Buffer_CRC_Pos=ToModify.End+4; Buffer_CRC_Pos<ToModify.Size; Buffer_CRC_Pos++)
458 ToModify.Buffer[Buffer_CRC_Pos]=0xFF;
459
460 Writer.Write(ToModify.Buffer, ToModify.Size);
461
462 //Managing big chunks
463 int16u PID=((ToModify.Buffer[1]&0x1F)<<8)|ToModify.Buffer[2]; //BigEndian2int16u(ToAdd+1)&0x1FFF;
464 if (BigBuffers.find(PID)!=BigBuffers.end())
465 BigBuffers.erase(BigBuffers.find(PID));
466}
