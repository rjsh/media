53Ztring Xml_Name_Escape_0_7_78 (const Ztring &Name)
54{
55 Ztring ToReturn(Name);
56
57 if (ToReturn.operator()(0)>='0' && ToReturn.operator()(0)<='9')
58 ToReturn.insert(0, 1, __T('_'));
59 ToReturn.FindAndReplace(__T(" "), __T("_"), 0, Ztring_Recursive);
60 ToReturn.FindAndReplace(__T("/"), __T("_"), 0, Ztring_Recursive);
61 ToReturn.FindAndReplace(__T("("), Ztring(), 0, Ztring_Recursive);
62 ToReturn.FindAndReplace(__T(")"), Ztring(), 0, Ztring_Recursive);
63 ToReturn.FindAndReplace(__T("*"), __T("_"), 0, Ztring_Recursive);
64 ToReturn.FindAndReplace(__T(","), __T("_"), 0, Ztring_Recursive);
65 ToReturn.FindAndReplace(__T(":"), __T("_"), 0, Ztring_Recursive);
66 ToReturn.FindAndReplace(__T("@"), __T("_"), 0, Ztring_Recursive);
67 ToReturn.FindAndReplace(__T("."), __T("_"), 0, Ztring_Recursive);
68 size_t ToReturn_Pos=0;
69 while (ToReturn_Pos<ToReturn.size())
70 {
71 if (!(ToReturn[ToReturn_Pos]>=__T('A') && ToReturn[ToReturn_Pos]<=__T('Z'))
72 && !(ToReturn[ToReturn_Pos]>=__T('a') && ToReturn[ToReturn_Pos]<=__T('z'))
73 && !(ToReturn[ToReturn_Pos]>=__T('0') && ToReturn[ToReturn_Pos]<=__T('9'))
74 && !(ToReturn[ToReturn_Pos]==__T('_')))
75 ToReturn.erase(ToReturn_Pos, 1);
76 else
77 ToReturn_Pos++;
78 }
79 if (ToReturn.empty())
80 ToReturn="Unknown";
81
82 return ToReturn;
83}
84#endif //defined(MEDIAINFO_XML_YES)

416 #endif //defined(MEDIAINFO_XML_YES)

421 #endif //defined(MEDIAINFO_TEXT_YES) && (defined(MEDIAINFO_HTML_YES) || defined(MEDIAINFO_XML_YES) || defined(MEDIAINFO_CSV_YES))
422 size_t Size=Count_Get(StreamKind, StreamPos);
423 bool IsExtra=false;
424 for (size_t Champ_Pos=0; Champ_Pos<Size; Champ_Pos++)
425 {
426 //Pour chaque champ
427 //Ztring A=Get((stream_t)4, 2, 0, Info_Measure_Text); // TODO Bug sinon? voir Ztring
428 Ztring A=Get((stream_t)StreamKind, StreamPos, Champ_Pos, Info_Measure_Text); // TODO Bug sinon? voir Ztring
429 bool Shouldshow=false;
430 #if defined(MEDIAINFO_XML_YES)
431 if (XML_0_7_78)
432 {
433 if (Champ_Pos>=Stream[StreamKind][StreamPos].size())
434 Shouldshow=true;
435 else
436 {
437 Ztring Options=Get((stream_t)StreamKind, StreamPos, Champ_Pos, Info_Options);
438 if (InfoOption_ShowInXml<Options.size() && Options[InfoOption_ShowInXml]==__T('Y'))
439 Shouldshow=true;
440 }
441 }
442 else
443 #endif //defined(MEDIAINFO_XML_YES
444 if ((MediaInfoLib::Config.Complete_Get() || Get((stream_t)StreamKind, StreamPos, Champ_Pos, Info_Options)[InfoOption_ShowInInform]==__T('Y')))
445 Shouldshow=true;
446 if (Shouldshow && !Get((stream_t)StreamKind, StreamPos, Champ_Pos, Info_Text).empty())
447 {
448 #if defined(MEDIAINFO_XML_YES)
449 //Extra
450 if (XML_0_7_78 && !IsExtra && Champ_Pos>=Stream[StreamKind][StreamPos].size())
451 {
452 Retour+=__T("<extra>\n");
453 IsExtra=true;
454 }
455 #endif //defined(MEDIAINFO_XML_YES

457 Ztring Nom=Get((stream_t)StreamKind, StreamPos, Champ_Pos, Info_Name_Text);
458 #if defined(MEDIAINFO_XML_YES)
459 if (Nom.empty() || XML_0_7_78)
460 #else
461 if (Nom.empty())
462 #endif //defined(MEDIAINFO_XML_YES
463
464 Nom=Get((stream_t)StreamKind, StreamPos, Champ_Pos, Info_Name); //Texte n'existe pas
465 #if defined(MEDIAINFO_TEXT_YES) && (defined(MEDIAINFO_HTML_YES) || defined(MEDIAINFO_XML_YES) || defined(MEDIAINFO_CSV_YES))
466 if (Text)
467 #endif //defined(MEDIAINFO_TEXT_YES) && (defined(MEDIAINFO_HTML_YES) || defined(MEDIAINFO_XML_YES) || defined(MEDIAINFO_CSV_YES))
468 {
469 int8u Nom_Size=MediaInfoLib::Config.Language_Get(__T(" Config_Text_ColumnSize")).To_int8u();
470 if (Nom_Size==0)
471 Nom_Size=32; //Default
472 Nom.resize(Nom_Size, ' ');
473 }
474 Ztring Valeur=Get((stream_t)StreamKind, StreamPos, Champ_Pos, Info_Text);
475
476 //Handling values with \r\n inside
477 #if defined(MEDIAINFO_TEXT_YES) && (defined(MEDIAINFO_HTML_YES) || defined(MEDIAINFO_XML_YES) || defined(MEDIAINFO_CSV_YES))
478 if (Text)
479 #endif //defined(MEDIAINFO_TEXT_YES) && (defined(MEDIAINFO_HTML_YES) || defined(MEDIAINFO_XML_YES) || defined(MEDIAINFO_CSV_YES))
480 {
