50Ztring MediaInfo_Config_PerPackage::Option (const String &Option, const String &Value)
51{
52 String Option_Lower(Option);
53 size_t Egal_Pos=Option_Lower.find(__T('='));
54 if (Egal_Pos==string::npos)
55 Egal_Pos=Option_Lower.size();
56 transform(Option_Lower.begin(), Option_Lower.begin()+Egal_Pos, Option_Lower.begin(), (int(*)(int))tolower); //(int(*)(int)) is a patch for unix
57
58 if (Option_Lower==__T("file_event_callbackfunction"))
59 {
60 #if MEDIAINFO_EVENTS
61 return Event_CallBackFunction_Set(Value);
62 #else //MEDIAINFO_EVENTS
63 return __T("Event manager is disabled due to compilation options");
64 #endif //MEDIAINFO_EVENTS
65 }
66 else
67 return __T("Option not known");
68}

84//---------------------------------------------------------------------------
85#if MEDIAINFO_EVENTS
86Ztring MediaInfo_Config_PerPackage::Event_CallBackFunction_Set (const Ztring &Value)
87{
88 ZtringList List=Value;
89
90 CriticalSectionLocker CSL(CS);
91
92 if (List.empty())
93 {
94 Event_CallBackFunction=(MediaInfo_Event_CallBackFunction*)NULL;
95 Event_UserHandler=NULL;
96 }
97 else
98 for (size_t Pos=0; Pos<List.size(); Pos++)
99 {
100 if (List[Pos].find(__T("CallBack=memory://"))==0)
101 Event_CallBackFunction=(MediaInfo_Event_CallBackFunction*)Ztring(List[Pos].substr(18, std::string::npos)).To_int64u();
102 else if (List[Pos].find(__T("UserHandle=memory://"))==0)
103 Event_UserHandler=(void*)Ztring(List[Pos].substr(20, std::string::npos)).To_int64u();
104 else if (List[Pos].find(__T("UserHandler=memory://"))==0)
105 Event_UserHandler=(void*)Ztring(List[Pos].substr(21, std::string::npos)).To_int64u();
106 else
107 return("Problem during Event_CallBackFunction value parsing");
108 }
109
110 return Ztring();
111}
112#endif //MEDIAINFO_EVENTS
113
114//---------------------------------------------------------------------------
115#if MEDIAINFO_EVENTS
116Ztring MediaInfo_Config_PerPackage::Event_CallBackFunction_Get ()
117{
118 CriticalSectionLocker CSL(CS);
119
120 return __T("CallBack=memory://")+Ztring::ToZtring((size_t)Event_CallBackFunction)+__T(";UserHandler=memory://")+Ztring::ToZtring((size_t)Event_UserHandler);
121}
122#endif //MEDIAINFO_EVENTS
