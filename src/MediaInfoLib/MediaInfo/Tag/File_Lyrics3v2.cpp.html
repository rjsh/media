60//***************************************************************************
61// Buffer - File header
62//***************************************************************************
63
64//---------------------------------------------------------------------------
65void File_Lyrics3v2::FileHeader_Parse()
66{
67 if (TotalSize==(int64u)-1)
68 TotalSize=Buffer_Size;
69
70 //Parsing
71 Skip_Local(11, "Signature");
72
73 FILLING_BEGIN();
74 Accept("Lyrics3v2");
75
76 TotalSize-=11;
77 FILLING_END();
78}
79
80//***************************************************************************
81// Buffer - Per element
82//***************************************************************************
83
84//---------------------------------------------------------------------------
85void File_Lyrics3v2::Header_Parse()
86{
87 if (TotalSize<=15) //First 10 is minimum size of a tag, Second 10 is ID3v2 header size
88 {
89 //Place for footer
90 Header_Fill_Code((int64u)-1, "File Footer");
91 Header_Fill_Size(TotalSize);
92 return;
93 }
94
95 //Parsing
96 Ztring SizeT;
97 int64u Size;
98 int32u Field;
99 Get_C3 (Field, "Field");
100 Get_Local(5, SizeT, "Size");
101 Size=8+SizeT.To_int64u();
102
103 //Filling
104 if (Size+15>TotalSize)
105 Size=TotalSize-15;
106 Header_Fill_Code(Field, Ztring().From_CC3(Field));
107 Header_Fill_Size(Size);
108 TotalSize-=Size;
109}
110
111//---------------------------------------------------------------------------
112void File_Lyrics3v2::Data_Parse()
113{
114 #define CASE_INFO(_NAME, _DETAIL) \
115 case Elements::_NAME : Element_Info1(_DETAIL); _NAME(); break;
116
117 //Parsing
118 switch (Element_Code)
119 {
120 CASE_INFO(AUT, "Lyrics Author Name");
121 CASE_INFO(CRC, "CRC");
122 CASE_INFO(EAL, "Extended Album name");
123 CASE_INFO(EAR, "Extended Artist name");
124 CASE_INFO(ETT, "Extended Track Title");
125 CASE_INFO(IMG, "Image location");
126 CASE_INFO(IND, "Indications field");
127 CASE_INFO(INF, "Additional information");
128 CASE_INFO(LYR, "Lyrics");
129 case (int64u)-1 : Footer(); break;
130 default : Skip_XX(Element_Size, "Data");
131 }
132}

138//---------------------------------------------------------------------------
139void File_Lyrics3v2::Footer()
140{
141 //Parsing
142 Skip_Local(6, "Size");
143 Skip_Local(9, "Signature");
144
145 Finish("Lyrics3v2");
146}

181//---------------------------------------------------------------------------
182void File_Lyrics3v2::IND()
183{
184 //Parsing
185 if (Element_Size>=1)
186 Skip_Local(1, "lyrics present");
187 if (Element_Size>=2)
188 Skip_Local(1, "timestamp in lyrics");
189 if (Element_Size>=3)
190 Skip_Local(1, "inhibits tracks for random selection");
191 while (Element_Offset<Element_Size)
192 Skip_Local(1, "unknown");
193}

204}

206//---------------------------------------------------------------------------

212 //Filling
213 Stream_Prepare(Stream_Text);
214 Fill(Stream_Text, 0, Text_Codec, "Lyrics3v2");
215}
