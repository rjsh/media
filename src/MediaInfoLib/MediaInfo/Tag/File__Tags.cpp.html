84//---------------------------------------------------------------------------
85void File__Tags_Helper::Streams_Fill()
86{
87 for (size_t Pos=0; Pos<Parser_Streams_Fill.size(); Pos++)
88 {
89 if (Parser_Streams_Fill[Pos] && Parser_Streams_Fill[Pos]->Status[File__Analyze::IsAccepted])
90 {
91 #ifndef MEDIAINFO_ID3V2_YES
92 const bool Priority=false;
93 #else
94 bool Priority=Parser_Streams_Fill_Priority[Pos];
95 #endif
96
97 Parser_Streams_Fill[Pos]->Read_Buffer_Finalize();
98 Base->Merge(*(Parser_Streams_Fill[Pos]), Stream_General, 0, 0, Priority);
99 Base->Merge(*(Parser_Streams_Fill[Pos]), Stream_Audio , 0, 0, Priority);
100 }
101 delete Parser_Streams_Fill[Pos]; //Parser_Streams_Fill[Pos]=NULL;
102 }
103 Parser_Streams_Fill.clear();
104}
105
106//---------------------------------------------------------------------------
107void File__Tags_Helper::Streams_Finish()
108{
109 if (!Base->IsSub)
110 {
111 Base->Fill(Stream_General, 0, General_StreamSize, TagsSize+Base->Retrieve(Stream_General, 0, General_StreamSize).To_int64u(), 10, true);
112 if (Base->Retrieve(Stream_Audio, 0, Audio_StreamSize).empty())
113 Base->Fill(Stream_Audio, 0, Audio_StreamSize, Base->File_Size-TagsSize);
114 }

135//---------------------------------------------------------------------------
136bool File__Tags_Helper::Synchronize(bool &Tag_Found, size_t Synchro_Offset)
137{
138 //Buffer size
139 if (Parser)
140 {
141 Synched_Test();
142 if (Parser)
143 return false;
144 }
145
146 if (SearchingForEndTags)
147 {
148 GoTo(0, "Tags detected");
149 return false;
150 }
151
152 if (!Synchro_Offset)
153 {
154 if (!Synched_Test()) //Handling begin/intermediate Id3v2
155 return false;
156 }
157
158 //ID
159 if (Base->Buffer_Offset+Synchro_Offset+8>Base->Buffer_Size)
160 return false;
161 int32u ID3=CC3(Base->Buffer+Base->Buffer_Offset+Synchro_Offset);
162 int64u ID8=CC8(Base->Buffer+Base->Buffer_Offset+Synchro_Offset);
163 if (ID3==0x494433 //"ID3"
164 || ID8==0x4150455441474558LL //"APETAGEX"
165 || ID8==0x4C59524943534245LL //"LYRICSBE"
166 || ID3==0x544147) //"TAG" / "TAG+"
167 Tag_Found=true;
168 else
169 Tag_Found=false;
170
171 return true;
172}

174//---------------------------------------------------------------------------
175bool File__Tags_Helper::Synched_Test()
176{
177 if (SearchingForEndTags)
178 return true;
179
180 for (;;)
181 {
182 #ifdef MEDIAINFO_ID3V2_YES
183 bool Priority=false;
184 #else
185 const bool Priority=false;
186 #endif
187
188 if (!Parser)
189 {
190 //Must have enough buffer for having header
191 if (Base->Buffer_Offset+8>Base->Buffer_Size)
192 return Base->IsSub; //If IsSub, we consider this is a complete block
193
194 //Quick test of synchro
195 int32u ID=CC3(Base->Buffer+Base->Buffer_Offset);
196 int32u ID4=CC4(Base->Buffer+Base->Buffer_Offset);
197 int64u ID8=CC8(Base->Buffer+Base->Buffer_Offset);
198 if (ID==0x494433 || ID==0x656133) //"ID3" and "ea3" (found in OpenMG)
199 {
200 if (Base->Buffer_Offset+10>Base->Buffer_Size)
201 return false;
202 int32u Size=BigEndian2int32u(Base->Buffer+Base->Buffer_Offset+6);
203 Parser_Buffer_Size=(((Size>>0)&0x7F)
204 | ((Size>>1)&0x3F80)
205 | ((Size>>2)&0x1FC000)
206 | ((Size>>3)&0x0FE00000))
207 +10;
208 if (Base->Buffer_Offset+Parser_Buffer_Size>Base->Buffer_Size)
209 return false;
210 if (!TagSizeIsFinal)
211 File_BeginTagSize+=Parser_Buffer_Size;
212 if (Base->File_Offset_FirstSynched==(int64u)-1)
213 Base->Buffer_TotalBytes_FirstSynched_Max+=Parser_Buffer_Size;
214 #ifdef MEDIAINFO_ID3V2_YES
215 Parser=new File_Id3v2;
216 Priority=true;
217 #else
218 Parser=new File_Unknown;
219 #endif
220 Base->Element_Begin1("Id3v2");
221 }
222 else if (ID8==0x4150455441474558LL) //"APETAGEX"
223 {
224 if (Base->Buffer_Offset+16>Base->Buffer_Size)
225 return false;
226 Parser_Buffer_Size=LittleEndian2int32u(Base->Buffer+Base->Buffer_Offset+12);
227 if (LittleEndian2int32u(Base->Buffer+Base->Buffer_Offset+8))
228 Parser_Buffer_Size+=32;
229 if (Base->Buffer_Offset+Parser_Buffer_Size>Base->Buffer_Size)
230 return false;
231 if (!TagSizeIsFinal)
232 File_BeginTagSize+=Parser_Buffer_Size;
233 if (Base->File_Offset_FirstSynched==(int64u)-1)
234 Base->Buffer_TotalBytes_FirstSynched_Max+=Parser_Buffer_Size;
235 #ifdef MEDIAINFO_APETAG_YES
236 Parser=new File_ApeTag;
237 #else
238 Parser=new File_Unknown;
239 #endif
240 Base->Element_Begin1("ApeTag");
241 }
242 else if (ID8==0x4C59524943534245LL) //"LYRICSBE"
243 {
244 if (Base->Buffer_Offset+16>Base->Buffer_Size)
245 return false;
246 //Searching for "LYRICS200"
247 std::string Buf((const char*)(Base->Buffer+Base->Buffer_Offset), Base->Buffer_Size-Base->Buffer_Offset);
248 size_t Pos2=Buf.find("LYRICS200");
249 size_t Pos1=std::string::npos;
250 if (Pos2==std::string::npos)
251 Pos1=Buf.find("LYRICSEND");
252 if (Pos2!=std::string::npos)
253 Parser_Buffer_Size=Pos2+9;
254 else if (Pos1!=std::string::npos)
255 Parser_Buffer_Size=Pos1+9;
256 else
257 Parser_Buffer_Size=(size_t)(Base->File_Size-(Base->File_Offset+Base->Buffer_Offset));
258 if (Base->Buffer_Offset+Parser_Buffer_Size>Base->Buffer_Size)
259 return false;
260 if (!TagSizeIsFinal)
261 File_BeginTagSize+=Parser_Buffer_Size;
262 if (Base->File_Offset_FirstSynched==(int64u)-1)
263 Base->Buffer_TotalBytes_FirstSynched_Max+=Parser_Buffer_Size;
264 if (Pos2!=std::string::npos)
265 {
266 #ifdef MEDIAINFO_LYRICS3V2_YES
267 Parser=new File_Lyrics3v2;
268 ((File_Lyrics3v2*)Parser)->TotalSize=Parser_Buffer_Size;
269 #else
270 Parser=new File_Unknown;
271 #endif
272 Base->Element_Begin1("Lyrics2");
273 }
274 else if (Pos1!=std::string::npos)
275 {
276 #ifdef MEDIAINFO_LYRICS3_YES
277 Parser=new File_Lyrics3;
278 ((File_Lyrics3*)Parser)->TotalSize=Parser_Buffer_Size;
279 #else
280 Parser=new File_Unknown;
281 #endif
282 Base->Element_Begin1("Lyrics");
283 }
284 else
285 {
286 Parser=new File_Unknown;
287 Base->Element_Begin1("Problem");
288 }
289 }
290 else if (ID4==0x5441472B) //"TAG+"
291 {
292 #ifdef MEDIAINFO_ID3_YES
293 Parser=new File_Id3;
294 #else
295 Parser=new File_Unknown;
296 #endif
297 Parser_Buffer_Size=227+128;
298 Base->Element_Begin1("Id3+");
299 }
300 else if (ID==0x544147) //"TAG"
301 {
302 #ifdef MEDIAINFO_ID3_YES
303 Parser=new File_Id3;
304 #else
305 Parser=new File_Unknown;
306 #endif
307 Parser_Buffer_Size=128;
308 if (!TagSizeIsFinal)
309 File_BeginTagSize+=Parser_Buffer_Size;
310 Base->Element_Begin1("Id3");
311 }
312 else if (Base->File_Offset+Base->Buffer_Offset==Lyrics3_Offset)
313 {
314 #ifdef MEDIAINFO_LYRICS3_YES
315 Parser=new File_Lyrics3;
316 ((File_Lyrics3*)Parser)->TotalSize=Lyrics3_Size;
317 #else
318 Parser=new File__Analyze;
319 #endif
320 Parser_Buffer_Size=(size_t)Lyrics3_Size;
321 if (!TagSizeIsFinal)
322 File_BeginTagSize+=Parser_Buffer_Size;
323 Base->Element_Begin1("Lyrics3");
324 }
325 else if (Base->File_Offset+Base->Buffer_Offset==Lyrics3v2_Offset)
326 {
327 #ifdef MEDIAINFO_LYRICS3V2_YES
328 Parser=new File_Lyrics3v2;
329 ((File_Lyrics3v2*)Parser)->TotalSize=Lyrics3v2_Size;
330 #else
331 Parser=new File_Unknown;
332 #endif
333 Parser_Buffer_Size=(size_t)Lyrics3v2_Size;
334 if (!TagSizeIsFinal)
335 File_BeginTagSize+=Parser_Buffer_Size;
336 Base->Element_Begin1("Lyrics3v2");
337 }
338 else if (Base->File_Offset+Base->Buffer_Offset==ApeTag_Offset)
339 {
340 #ifdef MEDIAINFO_APETAG_YES
341 Parser=new File_ApeTag;
342 #else
343 Parser=new File_Unknown;
344 #endif
345 if (!TagSizeIsFinal)
346 File_BeginTagSize+=Parser_Buffer_Size;
347 Parser_Buffer_Size=(size_t)ApeTag_Size;
348 Base->Element_Begin1("ApeTag");
349 }
350 else
351 break;
352 Base->Open_Buffer_Init(Parser);
353 }
354
355 //We continue
356 if (Parser)
357 {
358 if (!Parser_Buffer_Size)
359 {
360 delete Parser; Parser=NULL;
361 return false; //Probably stuck in a loop, exit
362 }
363 int64u Size_ToParse=(Parser_Buffer_Size<Base->Buffer_Size-Base->Buffer_Offset)?Parser_Buffer_Size:(Base->Buffer_Size-Base->Buffer_Offset);
364 Base->Open_Buffer_Continue(Parser, Base->Buffer+Base->Buffer_Offset, (size_t)Size_ToParse);
365 Base->Buffer_Offset+=(size_t)Size_ToParse;
366 TagsSize+=Size_ToParse;
367 Parser_Buffer_Size-=(size_t)Size_ToParse;
368 if (Parser->Status[File__Analyze::IsFinished] || Parser_Buffer_Size==0)
369 {
370 if (Base->Status[File__Analyze::IsAccepted] && Parser->Count_Get(Stream_General)>0)
371 {
372 if (!Base->Status[File__Analyze::IsFilled])
373 Base->Fill();
374 Parser->Read_Buffer_Finalize();
375 Base->Merge(*Parser, Stream_General, 0, 0, Priority);
376 Base->Merge(*Parser, Stream_Audio , 0, 0, Priority);
377 delete Parser; Parser=NULL;
378 }
379 else
380 {
381 Parser_Streams_Fill.push_back(Parser);
382 #ifdef MEDIAINFO_ID3V2_YES
383 Parser_Streams_Fill_Priority.push_back(Priority);
384 #endif
385 Parser=NULL;
386 }
387 if (Parser_Buffer_Size)
388 Base->Skip_XX(Parser_Buffer_Size, "Data continued");
389 Base->Element_Show();
390 Base->Element_End0();
391 }
392 else
393 break;
394 }
395 }
396
397 return true;
398}

416//---------------------------------------------------------------------------
417void File__Tags_Helper::GoTo (int64u GoTo, const char* ParserName)
418{
419 if (Base->IsSub)
420 {
421 Base->GoTo(GoTo, ParserName);
422 return;
423 }
424
425 //Normal Data_GoTo;
426
427 //Configuring
428 if (!SearchingForEndTags)
429 {
430 if (GoTo==(int64u)-1)
431 JumpTo_WantedByParser=Base->File_Offset+Base->Buffer_Offset;
432 else
433 JumpTo_WantedByParser=GoTo;
434 SearchingForEndTags=true;
435 }
436
437 //Trying to parse tags
438 while (!TagSizeIsFinal && DetectBeginOfEndTags_Test());
439
440 //If a jump is requested
441 if (!TagSizeIsFinal)
442 return;
443
444 //Positionning (if finnished)
445 if (TagSizeIsFinal)
446 {
447 if (JumpTo_WantedByParser!=(int64u)-1)
448 {
449 if (JumpTo_WantedByParser<Id3v1_Size+Lyrics3_Size+Lyrics3v2_Size+ApeTag_Size)
450 Base->GoTo(JumpTo_WantedByParser, ParserName);
451 else
452 Base->GoTo(JumpTo_WantedByParser-Id3v1_Size-Lyrics3_Size-Lyrics3v2_Size-ApeTag_Size, ParserName);
453 }
454 SearchingForEndTags=false;
455 }
456 else
457 Base->GoTo(GoTo, ParserName);
458}

460//---------------------------------------------------------------------------
461void File__Tags_Helper::GoToFromEnd (int64u GoToFromEnd, const char* ParserName)
462{
463 if (GoToFromEnd>Base->File_Size)
464 {
465 if (ParserName)
466 {
467 bool MustElementBegin=Base->Element_Level?true:false;
468 if (Base->Element_Level>0)
469 Base->Element_End0(); //Element
470 Base->Info(std::string(ParserName)+", wants to go to somewhere, but not valid");
471 if (MustElementBegin)
472 Base->Element_Level++;
473 }
474 Finish(ParserName);
475 return;
476 }
477
478 GoTo(Base->File_Size-GoToFromEnd, ParserName);
479}

504 }

509//---------------------------------------------------------------------------
510bool File__Tags_Helper::DetectBeginOfEndTags_Test()
511{
512 TagSizeIsFinal=true;
513
514 //Id3v1
515 if (Id3v1_Size==0 && File_EndTagSize==0 && Base->File_Size>=128) //Only one, at the end, larger than 128 bytes
516 {
517 if (Base->File_Offset>Base->File_Size-128-32) //Must be at least at the end less 128 bytes
518 {
519 Base->GoTo(Base->File_Size-128-32, "Tags detection"); //32 to be able to quickly see another tag system
520 TagSizeIsFinal=false;
521 return false;
522 }
523
524 if (Base->File_Offset+Base->Buffer_Size<Base->File_Size) //Must be at least at the end less 128 bytes of tags
525 {
526 if (Base->File_Offset<Base->File_Size-128-32)
527 Base->GoTo(Base->File_Size-128-32, "Tags detection");
528 TagSizeIsFinal=false;
529 return false;
530 }
531
532 if (CC3(Base->Buffer+((size_t)((Base->File_Size-128)-Base->File_Offset)))==CC3("TAG"))
533 {
534 Id3v1_Size=128;
535 File_EndTagSize+=128;
536 Id3v1_Offset=Base->File_Size-File_EndTagSize;
537 TagSizeIsFinal=false;
538 }
539 }
540
541 //Lyrics3
542 if (Lyrics3_Size==0 && Base->File_Size>=File_EndTagSize+9) //Only one, larger than File_EndTagSize+15 bytes
543 {
544 if (Base->File_Offset>Base->File_Size-File_EndTagSize-9) //Must be at the end less 15 bytes
545 {
546 Base->GoTo(Base->File_Size-File_EndTagSize-9, "Tags");
547 TagSizeIsFinal=false;
548 return false;
549 }
550
551 if (Base->File_Offset+Base->Buffer_Size<Base->File_Size-File_EndTagSize) //Must be at the end less File_EndTagSize+9 bytes plus 9 bytes of tags
552 {
553 if (Base->File_Offset!=Base->File_Size-File_EndTagSize)
554 Base->GoTo(Base->File_Size-File_EndTagSize, "Tags");
555 TagSizeIsFinal=false;
556 return false;
557 }
558
559 if (CC8(Base->Buffer+((size_t)((Base->File_Size-File_EndTagSize-9)-Base->File_Offset)))==CC8("LYRICSEN"))
560 {
561 //Must find the beginning, 5100 bytes before
562 if (Base->File_Offset>Base->File_Size-File_EndTagSize-5100) //Must be at the end less 15 bytes
563 {
564 Base->GoTo(Base->File_Size-File_EndTagSize-5100, "Tags");
565 TagSizeIsFinal=false;
566 return false;
567 }
568
569 if (Base->File_Offset+Base->Buffer_Size<Base->File_Size-File_EndTagSize) //Must be at the end less File_EndTagSize
570 {
571 if (Base->File_Offset!=Base->File_Size-File_EndTagSize)
572 Base->GoTo(Base->File_Size-File_EndTagSize, "Tags");
573 TagSizeIsFinal=false;
574 return false;
575 }
576
577 //Searching for "LYRICSBEGIN"
578 std::string Buf((const char*)(Base->Buffer+Base->Buffer_Offset), Base->Buffer_Size-Base->Buffer_Offset);
579 size_t Pos=Buf.find("LYRICSBEGIN");
580 if (Pos!=std::string::npos)
581 {
582 Lyrics3_Size=Base->File_Size-File_EndTagSize-(Base->File_Offset+Base->Buffer_Offset+Pos);
583 File_EndTagSize+=Lyrics3_Size;
584 Lyrics3_Offset=Base->File_Size-File_EndTagSize;
585 TagSizeIsFinal=false;
586 }
587 }
588 }
589
590 //Lyrics3v2
591 if (Lyrics3v2_Size==0 && Base->File_Size>=File_EndTagSize+15) //Only one, larger than File_EndTagSize+15 bytes
592 {
593 if (Base->File_Offset>Base->File_Size-File_EndTagSize-15) //Must be at the end less 15 bytes
594 {
595 Base->GoTo(Base->File_Size-File_EndTagSize-15, "Tags");
596 TagSizeIsFinal=false;
597 return false;
598 }
599
600 if (Base->File_Offset+Base->Buffer_Size<Base->File_Size-File_EndTagSize) //Must be at the end less File_EndTagSize
601 {
602 if (Base->File_Offset!=Base->File_Size-File_EndTagSize)
603 Base->GoTo(Base->File_Size-File_EndTagSize, "Tags");
604 TagSizeIsFinal=false;
605 return false;
606 }
607
608 if (CC8(Base->Buffer+((size_t)((Base->File_Size-File_EndTagSize-15+6)-Base->File_Offset)))==CC8("LYRICS20"))
609 {
610 Lyrics3v2_Size=Ztring((const char*)(Base->Buffer+((size_t)((Base->File_Size-File_EndTagSize-15)-Base->File_Offset))), 6).To_int32u()+15;
611 File_EndTagSize+=Lyrics3v2_Size;
612 Lyrics3v2_Offset=Base->File_Size-File_EndTagSize;
613 TagSizeIsFinal=false;
614 }
615 }
616
617 //ApeTag
618 if (ApeTag_Size==0 && Base->File_Size>=File_EndTagSize+32) //Only one, larger than File_EndTagSize+32 bytes
619 {
620 if (Base->File_Offset>Base->File_Size-File_EndTagSize-32) //Must be at the end less 32 bytes
621 {
622 Base->GoTo(Base->File_Size-File_EndTagSize-32, "Tags");
623 TagSizeIsFinal=false;
624 return false;
625 }
626
627 if (Base->File_Offset+Base->Buffer_Size<Base->File_Size-File_EndTagSize) //Must be at the end less File_EndTagSize+15 bytes plus 15 bytes of tags
628 {
629 if (Base->File_Offset!=Base->File_Size-File_EndTagSize-32)
630 Base->GoTo(Base->File_Size-File_EndTagSize-32, "Tags");
631 TagSizeIsFinal=false;
632 return false;
633 }
634
635 if (CC8(Base->Buffer+((size_t)((Base->File_Size-File_EndTagSize-32)-Base->File_Offset)))==CC8("APETAGEX"))
636 {
637 ApeTag_Size=LittleEndian2int32u(Base->Buffer+((size_t)((Base->File_Size-File_EndTagSize-32+12)-Base->File_Offset)));
638 if (LittleEndian2int32u(Base->Buffer+((size_t)((Base->File_Size-File_EndTagSize-32+8)-Base->File_Offset)))==2000)
639 ApeTag_Size+=32;
640 File_EndTagSize+=ApeTag_Size;
641 ApeTag_Offset=Base->File_Size-File_EndTagSize;
642 TagSizeIsFinal=false;
643 }
644 }
645
646 //Finished
647 return true;
648}
