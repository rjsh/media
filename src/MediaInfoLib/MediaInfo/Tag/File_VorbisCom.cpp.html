57//---------------------------------------------------------------------------
58void File_VorbisCom::Streams_Fill()
59{
60 if (!Performers.empty())
61 {
62 Artists.Separator_Set(0, __T(" / "));
63 Fill(StreamKind_Common, 0, "Performer", Performers.Read());
64 }
65
66 if (!Artists.empty() && Artists!=Performers)
67 {
68 Artists.Separator_Set(0, __T(" / "));
69 Fill(StreamKind_Common, 0, Performers.empty()?"Performer":"Composer", Artists.Read());
70 }
71
72 if (!Accompaniments.empty() && Accompaniments!=Artists && Accompaniments!=Performers)
73 {
74 Artists.Separator_Set(0, __T(" / "));
75 Fill(StreamKind_Common, 0, "Accompaniment", Accompaniments.Read());
76 }
77
78 if (!AlbumArtists.empty())
79 {
80 AlbumArtists.Separator_Set(0, __T(" / "));
81 Fill(StreamKind_Common, 0, (Performers==Artists || Performers.empty())?"Album/Performer":"Album/Composer", AlbumArtists.Read());
82 }
83}

89//---------------------------------------------------------------------------
90void File_VorbisCom::FileHeader_Parse()
91{
92 //Parsing
93 Ztring vendor_string;
94 int32u vendor_length;
95 Get_L4 (vendor_length, "vendor_length");
96 Get_Local(vendor_length, vendor_string, "vendor_string");
97 Get_L4 (user_comment_list_length, "user_comment_list_length");
98
99 FILLING_BEGIN();
100 Accept("VorbisCom");
101
102 if (Count_Get(Stream_General)==0)
103 Stream_Prepare(Stream_General);
104 if (StreamKind_Specific!=Stream_General)
105 Stream_Prepare(StreamKind_Specific);
106 if (StreamKind_Multiple!=Stream_General && StreamKind_Multiple!=StreamKind_Specific)
107 Stream_Prepare(StreamKind_Multiple);
108
109 //vendor_string
110 if (StreamKind_Specific!=Stream_Audio && vendor_string.find(__T("Xiph.Org libVorbis"))==0)
111 vendor_string.clear(); //string was set "by default"
112 Ztring Library_Name, Library_Version, Library_Date;
113 Ztring vendor_string_Without=vendor_string; vendor_string_Without.FindAndReplace(__T(";"), __T(""), 0, Ztring_Recursive);
114 Library_Version=MediaInfoLib::Config.Library_Get(InfoLibrary_Format_VorbisCom, vendor_string_Without, InfoLibrary_Version);
115 Library_Date=MediaInfoLib::Config.Library_Get(InfoLibrary_Format_VorbisCom, vendor_string_Without, InfoLibrary_Date);
116 if (Library_Version.empty())
117 {
118 if (vendor_string.find(__T(" I "))!=std::string::npos)
119 {
120 Library_Name=vendor_string.SubString(__T(""), __T(" I "));
121 Library_Date=vendor_string.SubString(__T(" I "), __T(""));
122 if (Library_Date.size()>9)
123 {
124 Library_Version=Library_Date.substr(9, std::string::npos);
125 if (Library_Version.find(__T("("))==std::string::npos)
126 {
127 Library_Version.FindAndReplace(__T(" "), __T("."), 0, Ztring_Recursive);
128 Library_Date.resize(8);
129 }
130 }
131 }
132 else if (vendor_string.size()>9 && Ztring(vendor_string.substr(vendor_string.size()-8, std::string::npos)).To_int32u()>20000000)
133 {
134 Library_Name=vendor_string.substr(0, vendor_string.size()-9);
135 Library_Date=vendor_string.substr(vendor_string.size()-8, std::string::npos);
136 if (!Library_Name.empty())
137 {
138 std::string::size_type Pos=Library_Name.rfind(__T(' '));
139 if (Pos<Library_Name.size()-2 && Library_Name[Pos+1]>=__T('0') && Library_Name[Pos+1]<=__T('9'))
140 {
141 Library_Version=Library_Name.substr(Pos+1, std::string::npos);
142 Library_Name.resize(Pos);
143 }
144 }
145 }
146 else if (vendor_string.find(__T("aoTuV "))!=std::string::npos)
147 {
148 Library_Name=__T("aoTuV");
149 Library_Version=vendor_string.SubString(__T("aoTuV "), __T("["));
150 Library_Date=vendor_string.SubString(__T("["), __T("]"));
151 }
152 else if (vendor_string.find(__T("Lancer "))!=std::string::npos)
153 {
154 Library_Name=__T("Lancer");
155 Library_Date=vendor_string.SubString(__T("["), __T("]"));
156 }
157 if (Library_Version.empty())
158 Library_Version=Library_Date;
159 if (Library_Date.size()==8)
160 {
161 Library_Date.insert(6, 1, __T('-'));
162 Library_Date.insert(4, 1, __T('-'));
163 Library_Date.insert(0, __T("UTC "));
164 }
165 }
166 if (vendor_string.find(__T("libFLAC"))!=std::string::npos) Library_Name="libFLAC";
167 if (vendor_string.find(__T("libVorbis I"))!=std::string::npos) Library_Name="libVorbis";
168 if (vendor_string.find(__T("libTheora I"))!=std::string::npos) Library_Name="libTheora";
169 if (vendor_string.find(__T("AO; aoTuV"))==0) Library_Name="aoTuV";
170 if (vendor_string.find(__T("BS; Lancer"))==0) Library_Name="Lancer";
171
172 Fill(StreamKind_Specific, 0, "Encoded_Library", vendor_string);
173 Fill(StreamKind_Specific, 0, "Encoded_Library_Name", Library_Name);
174 Fill(StreamKind_Specific, 0, "Encoded_Library_Version", Library_Version);
175 Fill(StreamKind_Specific, 0, "Encoded_Library_Date", Library_Date);
176 FILLING_END();
177}

194//---------------------------------------------------------------------------
195void File_VorbisCom::Data_Parse()
196{
197 user_comment_list_length--;
198
199 //Parsing
200 Ztring comment;
201 Get_UTF8(Element_Size, comment, "comment");
202 if (Element_Size && comment.empty())
203 {
204 Element_Offset=0; //Retry
205 Get_Local(Element_Size, comment, "comment");
206 }
207 Element_Name(comment);
208
209 FILLING_BEGIN_PRECISE();
210 Ztring Key=comment.SubString(__T(""), __T("="));
211 Key.MakeUpperCase();
212 Ztring Value=comment.SubString(__T("="), __T(""));
213
214 if (Key==__T("ADDED_TIMESTAMP")) Fill(StreamKind_Common, 0, "Added_Date", Ztring().Date_From_Milliseconds_1601(Value.To_int64u()/1000));
215 else if (Key==__T("ALBUM ARTIST")) AlbumArtists.push_back(Value);
216 else if (Key==__T("ALBUM")) Fill(StreamKind_Common, 0, "Album", Value);
217 else if (Key==__T("ALBUM_COMMENT")) Fill(StreamKind_Common, 0, "Comment", Value);
218 else if (Key==__T("ALBUMARTIST")) AlbumArtists.push_back(Value);
219 else if (Key==__T("ARTIST")) Artists.push_back(Value);
220 else if (Key==__T("AUTHOR")) Fill(StreamKind_Common, 0, "WrittenBy", Value);
221 else if (Key==__T("BUYCDURL")) {}
222 else if (Key==__T("CLASS")) Fill(StreamKind_Common, 0, "ContentType", Value);
223 else if (Key==__T("COMPOSER")) Fill(StreamKind_Common, 0, "Composer", Value);
224 else if (Key==__T("COMMENT")) Fill(StreamKind_Common, 0, "Comment", Value);
225 else if (Key==__T("COMMENTS")) Fill(StreamKind_Common, 0, "Comment", Value);
226 else if (Key==__T("CONDUCTOR")) Fill(StreamKind_Common, 0, "Conductor", Value);
227 else if (Key==__T("CONTACT")) Fill(StreamKind_Common, 0, "Publisher", Value);
228 else if (Key==__T("COPYRIGHT")) Fill(StreamKind_Common, 0, "Copyright", Value);
229 else if (Key==__T("DATE")) Fill(StreamKind_Common, 0, "Recorded_Date", Value, true);
230 else if (Key==__T("DESCRIPTION")) Fill(StreamKind_Common, 0, "Description", Value);
231 else if (Key==__T("DISC")) Fill(StreamKind_Common, 0, "Part", Value, true);
232 else if (Key==__T("DISCID")) {}
233 else if (Key==__T("DISCNUMBER")) Fill(StreamKind_Common, 0, "Part", Value, true);
234 else if (Key==__T("DISCTOTAL")) Fill(StreamKind_Common, 0, "Part/Position_Total", Value);
235 else if (Key==__T("ENCODEDBY")) Fill(StreamKind_Common, 0, "EncodedBy", Value);
236 else if (Key==__T("ENCODED-BY")) Fill(StreamKind_Common, 0, "EncodedBy", Value);
237 else if (Key==__T("ENCODER")) Fill(StreamKind_Common, 0, "Encoded_Application", Value);
238 else if (Key==__T("ENCODED_USING")) Fill(StreamKind_Common, 0, "Encoded_Application", Value);
239 else if (Key==__T("ENCODER_URL")) Fill(StreamKind_Common, 0, "Encoded_Application/Url", Value);
240 else if (Key==__T("ENSEMBLE")) Accompaniments.push_back(Value);
241 else if (Key==__T("GENRE")) Fill(StreamKind_Common, 0, "Genre", Value);
242 else if (Key==__T("FIRST_PLAYED_TIMESTAMP")) Fill(StreamKind_Common, 0, "Played_First_Date", Ztring().Date_From_Milliseconds_1601(Value.To_int64u()/10000));
243 else if (Key==__T("ISRC")) Fill(StreamKind_Multiple, 0, "ISRC", Value);
244 else if (Key==__T("LABEL")) Fill(StreamKind_Common, 0, "Label", Value);
245 else if (Key==__T("LANGUAGE")) {if (Value.find(__T("Director"))==0) Fill(StreamKind_Specific, 0, "Language_More", Value); else if (!Value.SubString(__T("["), __T("]")).empty()) Fill(StreamKind_Specific, 0, "Language", Value.SubString(__T("["), __T("]"))); else Fill(StreamKind_Specific, 0, "Language", Value);}
246 else if (Key==__T("LAST_PLAYED_TIMESTAMP")) Fill(StreamKind_Multiple, 0, "Played_Last_Date", Ztring().Date_From_Milliseconds_1601(Value.To_int64u()/10000));
247 else if (Key==__T("LICENCE")) Fill(StreamKind_Common, 0, "TermsOfUse", Value);
248 else if (Key==__T("LICENSE")) Fill(StreamKind_Common, 0, "TermsOfUse", Value);
249 else if (Key==__T("LYRICS")) Fill(StreamKind_Common, 0, "Lyrics", Value);
250 else if (Key==__T("LWING_GAIN")) Fill(StreamKind_Multiple, 0, "ReplayGain_Gain", Value.To_float64(), 2);
251 else if (Key==__T("LOCATION")) Fill(StreamKind_Common, 0, "Recorded/Location", Value);
252 else if (Key==__T("MUSICBRAINZ_ALBUMID")) {}
253 else if (Key==__T("MUSICBRAINZ_ALBUMARTISTID")) {}
254 else if (Key==__T("MUSICBRAINZ_ARTISTID")) {}
255 else if (Key==__T("MUSICBRAINZ_TRACKID")) {}
256 else if (Key==__T("MUSICBRAINZ_SORTNAME")) Fill(StreamKind_Common, 0, "Performer/Sort", Value);
257 else if (Key==__T("MUSICBRAINZ_DISCID")) {}
258 else if (Key==__T("ORGANIZATION")) Fill(StreamKind_Common, 0, "Producer", Value);
259 else if (Key==__T("PERFORMER")) Performers.push_back(Value);
260 else if (Key==__T("PLAY_COUNT")) Fill(StreamKind_Multiple, 0, "Played_Count", Value.To_int64u());
261 else if (Key==__T("RATING")) Fill(StreamKind_Multiple, 0, "Rating", Value);
262 else if (Key==__T("REPLAYGAIN_ALBUM_GAIN")) Fill(StreamKind_Common, 0, "Album_ReplayGain_Gain", Value.To_float64(), 2);
263 else if (Key==__T("REPLAYGAIN_ALBUM_PEAK")) Fill(StreamKind_Common, 0, "Album_ReplayGain_Peak", Value.To_float64(), 6);
264 else if (Key==__T("REPLAYGAIN_REFERENCE_LOUDNESS")) {}
265 else if (Key==__T("REPLAYGAIN_TRACK_GAIN")) Fill(StreamKind_Specific, 0, "ReplayGain_Gain", Value.To_float64(), 2);
266 else if (Key==__T("REPLAYGAIN_TRACK_PEAK")) Fill(StreamKind_Specific, 0, "ReplayGain_Peak", Value.To_float64(), 6);
267 else if (Key==__T("TITLE")) Fill(StreamKind_Common, 0, "Title", Value);
268 else if (Key==__T("TOTALTRACKS")) Fill(StreamKind_Common, 0, "Track/Position_Total", Value);
269 else if (Key==__T("TOTALDISCS")) Fill(StreamKind_Common, 0, "Part/Position_Total", Value);
270 else if (Key==__T("TRACK_COMMENT")) Fill(StreamKind_Multiple, 0, "Comment", Value);
271 else if (Key==__T("TRACKNUMBER")) Fill(StreamKind_Multiple, 0, "Track/Position", Value);
272 else if (Key==__T("TRACKTOTAL")) Fill(StreamKind_Multiple, 0, "Track/Position_Total", Value);
273 else if (Key==__T("VERSION")) Fill(StreamKind_Common, 0, "Track_More", Value);
274 else if (Key==__T("BPM")) Fill(StreamKind_Common, 0, "BPM", Value);
275 else if (Key==__T("WAVEFORMATEXTENSIBLE_CHANNEL_MASK"))
276 {
277 //This is an hexadecimal value
278 if (Value.size()>2 && Value[0]==__T('0') && (Value[1]==__T('x') || Value[1]==__T('X')))
279 {
280 int16u ValueI=0;
281 for (size_t Pos=2; Pos<Value.size(); Pos++)
282 {
283 ValueI*=16;
284 if (Value[Pos]>=__T('0') && Value[Pos]<=__T('9'))
285 ValueI+=Value[Pos]-__T('0');
286 else if (Value[Pos]>=__T('A') && Value[Pos]<=__T('F'))
287 ValueI+=10+Value[Pos]-__T('A');
288 else if (Value[Pos]>=__T('a') && Value[Pos]<=__T('f'))
289 ValueI+=10+Value[Pos]-__T('a');
290 else
291 break;
292 }
293 Fill(Stream_Audio, 0, Audio_ChannelPositions, ExtensibleWave_ChannelMask(ValueI));
294 Fill(Stream_Audio, 0, Audio_ChannelPositions_String2, ExtensibleWave_ChannelMask2(ValueI));
295 }
296 }
297 else if (Key==__T("VALID_BITS"))
298 {
299 if (Value.To_int64u())
300 {
301 // Not related to bit depth, only a detected value, see http://forum.doom9.org/showthread.php?t=125966&page=680 // Fill(Stream_Audio, 0, Audio_BitDepth, Value);
302 Fill(Stream_Audio, 0, Audio_BitDepth_Detected, Value);
303 }
304 }
305 else if (Key==__T("HDCD")) { if (Value != __T("0")) { Fill(Stream_Audio, 0, "HDCD", "Yes"); } }
306 else if (Key==__T("YEAR")) {if (Value!=Retrieve(StreamKind_Common, 0, "Recorded_Date")) Fill(StreamKind_Common, 0, "Recorded_Date", Value);}
307 else if (Key.find(__T("COVERART"))==0)
308 {
309 if (Key==__T("COVERARTCOUNT"))
310 ;
311 else if (Key.find(__T("COVERARTMIME"))==0)
312 Fill(Stream_General, 0, General_Cover_Mime, Value);
313 else if (Key.find(__T("COVERARTFILELINK"))==0)
314 Fill(Stream_General, 0, General_Cover_Data, __T("file://")+Value);
315 else if (Key.find(__T("COVERARTTYPE"))==0)
316 Fill(Stream_General, 0, General_Cover_Type, Id3v2_PictureType(Value.To_int8u()));
317 }
318 else if (Key.find(__T("CHAPTER"))==0)
319 {
320 if (Count_Get(Stream_Menu)==0)
321 {
322 Stream_Prepare(Stream_Menu);
323 Fill(Stream_Menu, StreamPos_Last, Menu_Chapters_Pos_Begin, Count_Get(Stream_Menu, StreamPos_Last), 10, true);
324 }
325 if (Key.find(__T("NAME"))==Error)
326 {
327 Chapter_Pos=Key.SubString(__T("CHAPTER"), __T(""));
328 Chapter_Time=Value;
329 }
330 else
331 {
332 Value.FindAndReplace(__T("\n"), __T(""), Count_Get(Stream_Text)-1); //Some chapters names have extra characters, not needed
333 Value.FindAndReplace(__T("\r"), __T(""), Count_Get(Stream_Text)-1); //Some chapters names have extra characters, not needed
334 Value.FindAndReplace(__T(" "), __T(""), Count_Get(Stream_Text)-1); //Some chapters names have extra characters, not needed
335 Fill(Stream_Menu, 0, Chapter_Time.To_UTF8().c_str(), Value);
336 }
337 Fill(Stream_Menu, StreamPos_Last, Menu_Chapters_Pos_End, Count_Get(Stream_Menu, StreamPos_Last), 10, true);
338 }
339 else Fill(Stream_General, 0, comment.SubString(__T(""), __T("=")).To_Local().c_str(), Value);
340 FILLING_END();
341
342 if (user_comment_list_length==0)
343 Finish("VorbisCom");
344}
