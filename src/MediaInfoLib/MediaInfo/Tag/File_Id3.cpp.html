35void File_Id3::Read_Buffer_Continue()
36{
37 //Buffer size
38 if (Buffer_Size<128)
39 return;
40
41 int32u Magic;
42 Peek_B4(Magic);
43 Ztring TitleAddition;
44 Ztring ArtistAddition;
45 Ztring AlbumAddition;
46 Ztring GenreAddition;
47 if (Magic==0x5441472B)
48 {
49 if (Buffer_Size<227+128)
50 return;
51
52 Skip_C4 ( "ID");
53 Get_Local (60, TitleAddition, "Title");
54 Get_Local (60, ArtistAddition, "Artist");
55 Get_Local (60, AlbumAddition, "Album");
56 Skip_B1 ( "Speed");
57 Get_Local (30, GenreAddition, "Genre");
58 Skip_Local(6, "Start time"); //mmm:ss
59 Skip_Local(6, "End time"); //mmm:ss
60
61 TitleAddition.TrimRight();
62 ArtistAddition.TrimRight();
63 AlbumAddition.TrimRight();
64 GenreAddition.TrimRight();
65 }
66
67 //Parsing
68 Ztring Title, Artist, Album, Year, Comment;
69 int8u Track=0, Genre;
70 Skip_C3 ( "ID");
71 Get_Local (30, Title, "Title");
72 Get_Local (30, Artist, "Artist");
73 Get_Local (30, Album, "Album");
74 Get_Local ( 4, Year, "Year");
75 Get_Local (30, Comment, "Comment");
76 if (Comment.size()<29) //Id3v1.1 specifications : Track number addition
77 {
78 Element_Offset-=2;
79 int8u Zero;
80 Peek_B1(Zero);
81 if (Zero==0)
82 {
83 Skip_B1( "Zero");
84 Get_B1 (Track, "Track");
85 }
86 else
87 Element_Offset+=2;
88 }
89 Get_B1 (Genre, "Genre");
90
91 FILLING_BEGIN();
92 if (TitleAddition.empty())
93 Title.TrimRight();
94 if (ArtistAddition.empty())
95 Artist.TrimRight();
96 if (AlbumAddition.empty())
97 Album.TrimRight();
98 Year.TrimRight();
99 Comment.TrimRight();
100
101 Accept("Id3");
102
103 Stream_Prepare(Stream_General);
104 Fill(Stream_General, 0, General_Album, Album+AlbumAddition);
105 Fill(Stream_General, 0, General_Track, Title+TitleAddition);
106 Fill(Stream_General, 0, General_Performer, Artist+ArtistAddition);
107 if (Comment.find(__T("ExactAudioCopy"))==0)
108 Fill(Stream_General, 0, General_Encoded_Application, Comment);
109 else
110 Fill(Stream_General, 0, General_Comment, Comment);
111 Fill(Stream_General, 0, General_Recorded_Date, Year);
112 if (GenreAddition.empty())
113 Fill(Stream_General, 0, General_Genre, GenreAddition);
114 if (Genre && Genre!=(int8u)-1)
115 Fill(Stream_General, 0, General_Genre, Genre);
116 if (Track)
117 Fill(Stream_General, 0, General_Track_Position, Track);
118
119 Finish("Id3");
120 FILLING_END();
121}
