39bool File_Xmp::FileHeader_Begin()
40{
41 XMLDocument document;
42 if (!FileHeader_Begin_XML(document))
43 return false;
44
45 std::string NameSpace;
46 XMLElement* XmpMeta=document.FirstChildElement("xmpmeta");
47 if (XmpMeta==NULL)
48 {
49 NameSpace="x:";
50 XmpMeta=document.FirstChildElement((NameSpace+"xmpmeta").c_str());
51 }
52 if (!XmpMeta)
53 {
54 Reject("XMP");
55 return false;
56 }
57
58 XMLElement* Rdf=XmpMeta->FirstChildElement("RDF");
59 if (Rdf==NULL)
60 {
61 NameSpace="rdf:";
62 Rdf=XmpMeta->FirstChildElement((NameSpace+"RDF").c_str());
63 }
64 if (!Rdf)
65 {
66 Reject("XMP");
67 return false;
68 }
69
70 Accept("XMP");
71
72 for (XMLElement* Rdf_Item=Rdf->FirstChildElement(); Rdf_Item; Rdf_Item=Rdf_Item->NextSiblingElement())
73 {
74 //RDF item
75 if (!strcmp(Rdf_Item->Value(), (NameSpace+"Description").c_str()))
76 {
77 const char* Attribute;
78
79 Attribute=Rdf_Item->Attribute("xmlns:pdfaid");
80 if (Attribute)
81 {
82 string Profile;
83
84 if (strcmp(Attribute, "http://www.aiim.org/pdfa/ns/id/"))
85 Profile=Attribute;
86 else
87 {
88 Profile+="A";
89
90 Attribute=Rdf_Item->Attribute("pdfaid:part");
91 if (Attribute)
92 {
93 Profile+='-';
94 Profile+=Attribute;
95
96 Attribute=Rdf_Item->Attribute("pdfaid:conformance");
97 if (Attribute)
98 {
99 string Conformance(Attribute);
100 if (Conformance.size()==1 && Conformance[0]>='A' && Conformance[0]<='Z')
101 Conformance[0]+=0x20; // From "A" to "a"
102 Profile+=Conformance;
103 }
104 }
105 }
106
107 Fill(Stream_General, 0, General_Format_Profile, Profile);
108 }
109 }
110 }
111
112 Finish();
113 return true;
114}
