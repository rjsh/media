37//---------------------------------------------------------------------------
38void File_ApeTag::FileHeader_Parse()
39{
40 //Parsing
41 int64u Signature;
42 Peek_B8(Signature);
43 if (Signature==0x4150455441474558LL) //"APETAGEX"
44 HeaderFooter(); //v2
45
46 FILLING_BEGIN();
47 Accept("ApeTag");
48
49 Stream_Prepare(Stream_General);
50
51 Stream_Prepare(Stream_Audio);
52 FILLING_END();
53}

55//***************************************************************************
56// Buffer - Per element
57//***************************************************************************
58
59//---------------------------------------------------------------------------
60bool File_ApeTag::Header_Begin()
61{
62 if (Buffer_Size<0x20)
63 return false; //At least 32 bytes are needed for footer
64
65 return true;
66}
67
68//---------------------------------------------------------------------------
69void File_ApeTag::Header_Parse()
70{
71 //Testing if begin or end of tags
72 if (CC8(Buffer+Buffer_Offset)==0x4150455441474558LL) //"APETAGEX"
73 {
74 //Filling
75 Header_Fill_Code((int64u)-1, "File Footer");
76 Header_Fill_Size(0x20);
77 return;
78 }
79
80 //Parsing
81 Ztring Value;
82 int32u Flags, Length;
83 Get_L4 (Length, "Length");
84 Get_L4 (Flags, "Flags");
85 Skip_Flags(Flags, 0, "Read Only");
86 Skip_Flags(Flags, 1, "Binary");
87 Skip_Flags(Flags, 2, "Locator of external stored information");
88 Skip_Flags(Flags, 29, "Is the header");
89 Skip_Flags(Flags, 30, "Contains a footer");
90 Skip_Flags(Flags, 31, "Contains a header");
91 size_t Pos=(size_t)Element_Offset;
92 for (; Pos<Element_Size; Pos++)
93 if (Buffer[Buffer_Offset+Pos]==0x00)
94 break;
95 if (Pos==Element_Size)
96 {
97 Element_WaitForMoreData();
98 return;
99 }
100 Get_String(Pos-Element_Offset, Key, "Key");
101 Skip_L1( "0x00");
102
103 //Filling
104 Header_Fill_Code(0, Key.c_str());
105 Header_Fill_Size(Element_Offset+Length);
106}

112//---------------------------------------------------------------------------
113void File_ApeTag::HeaderFooter()
114{
115 //Parsing
116 int32u Flags;
117 Skip_C8( "Preamble");
118 Skip_L4( "Version");
119 Skip_L4( "Size");
120 Skip_L4( "Count");
121 Get_L4 (Flags, "Flags");
122 Skip_Flags(Flags, 0, "Read Only");
123 Skip_Flags(Flags, 1, "Binary");
124 Skip_Flags(Flags, 2, "Locator of external stored information");
125 Skip_Flags(Flags, 29, "Is the header");
126 Skip_Flags(Flags, 30, "Contains a footer");
127 Skip_Flags(Flags, 31, "Contains a header");
128 Skip_L8( "Reserved");
129}
130
131//---------------------------------------------------------------------------
132void File_ApeTag::Data_Parse()
133{
134 //If footer
135 if (Element_Code==(int64u)-1)
136 {
137 HeaderFooter();
138 Finish("ApeTag");
139 return;
140 }
141
142 //Parsing
143 Ztring Value;
144 Get_UTF8(Element_Size, Value, "Value"); Element_Info1(Value);
145
146 //Filling
147 transform(Key.begin(), Key.end(), Key.begin(), (int(*)(int))toupper); //(int(*)(int)) is a patch for unix
148 if (Key=="ALBUM") Fill(Stream_General, 0, General_Album, Value);
149 else if (Key=="ARTIST") Fill(Stream_General, 0, General_Performer, Value);
150 else if (Key=="AUTHOR") Fill(Stream_General, 0, General_WrittenBy, Value);
151 else if (Key=="BAND") Fill(Stream_General, 0, General_Performer, Value);
152 else if (Key=="COMMENT") Fill(Stream_General, 0, General_Comment, Value);
153 else if (Key=="COMMENTS") Fill(Stream_General, 0, General_Comment, Value);
154 else if (Key=="COMPOSER") Fill(Stream_General, 0, General_Composer, Value);
155 else if (Key=="CONTENTGROUP") Fill(Stream_General, 0, General_Genre, Value);
156 else if (Key=="COPYRIGHT") Fill(Stream_General, 0, General_Copyright, Value);
157 else if (Key=="DISK")
158 {
159 if (Value.find(__T("/"))!=Error)
160 {
161 Fill(Stream_General, 0, General_Part_Position_Total, Value.SubString(__T("/"), __T("")));
162 Fill(Stream_General, 0, General_Part_Position, Value.SubString(__T(""), __T("/")));
163 }
164 else
165 Fill(Stream_General, 0, General_Track_Position, Value);
166 }
167 else if (Key=="ENCODEDBY") Fill(Stream_General, 0, General_EncodedBy, Value);
168 else if (Key=="GENRE") Fill(Stream_General, 0, General_Genre, Value);
169 else if (Key=="ORIGARTIST") Fill(Stream_General, 0, General_Original_Performer, Value);
170 else if (Key=="TITLE") Fill(Stream_General, 0, General_Title, Value);
171 else if (Key=="TRACK")
172 {
173 if (Value.find(__T("/"))!=Error)
174 {
175 Fill(Stream_General, 0, General_Track_Position_Total, Value.SubString(__T("/"), __T("")));
176 Fill(Stream_General, 0, General_Track_Position, Value.SubString(__T(""), __T("/")));
177 }
178 else
179 Fill(Stream_General, 0, General_Track_Position, Value);
180 }
181 else if (Key=="UNSYNCEDLYRICS") Fill(Stream_General, 0, General_Lyrics, Value);
182 else if (Key=="WWW") Fill(Stream_General, 0, General_Title_Url, Value);
183 else if (Key=="YEAR") Fill(Stream_General, 0, General_Recorded_Date, Value);
184 else if (Key=="CONTENT GROUP DESCRIPTION") Fill(Stream_General, 0, General_Title, Value);
185 else if (Key=="ORIGINAL ALBUM/MOVIE/SHOW TITLE") Fill(Stream_General, 0, General_Original_Album, Value);
186 else if (Key=="ORIGINAL ARTIST(S)/PERFORMER(S)") Fill(Stream_General, 0, General_Original_Performer, Value);
187 else if (Key=="MP3GAIN_MINMAX") Fill(Stream_Audio, 0, "MP3Gain, Min/Max", Value);
188 else if (Key=="MP3GAIN_UNDO") Fill(Stream_Audio, 0, "MP3Gain, Undo", Value);
189 else if (Key=="REPLAYGAIN_TRACK_GAIN") Fill(Stream_Audio, 0, Audio_ReplayGain_Gain, Value.To_float64(), 2, true);
190 else if (Key=="REPLAYGAIN_TRACK_PEAK") Fill(Stream_Audio, 0, Audio_ReplayGain_Peak, Value.To_float64(), 6, true);
191 else Fill(Stream_General, 0, Key.c_str(), Value);
192}
