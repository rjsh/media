311bool File_Id3v2::Static_Synchronize_Tags(const int8u* Buffer, size_t Buffer_Offset, size_t Buffer_Size, bool &Tag_Found)
312{
313 //Buffer size
314 if (Buffer_Offset+3>Buffer_Size)
315 return false;
316
317 //ID
318 if ((Buffer[Buffer_Offset ]==0x49 // "ID3"
319 && Buffer[Buffer_Offset+1]==0x44
320 && Buffer[Buffer_Offset+2]==0x33)
321 || (Buffer[Buffer_Offset ]==0x65 // "ea3", found in OpenMG
322 && Buffer[Buffer_Offset+1]==0x61
323 && Buffer[Buffer_Offset+2]==0x33))
324 Tag_Found=true;
325 else
326 Tag_Found=false;
327
328 //Continue
329 return true;
330}

374//---------------------------------------------------------------------------
375void File_Id3v2::FileHeader_Parse()
376{
377 //Parsing
378 int32u Size;
379 int8u Flags;
380 bool ExtendedHeader;
381 Skip_C3( "identifier");
382 Get_B1 (Id3v2_Version, "version_major");
383 Skip_B1( "version_revision");
384 Get_B1 (Flags, "flags");
385 Get_Flags (Flags, 7, Unsynchronisation_Global, "Unsynchronisation");
386 Get_Flags (Flags, 6, ExtendedHeader, "Extended header");
387 Skip_Flags(Flags, 5, "Experimental indicator");
388 Get_B4 (Size, "Size");
389 Id3v2_Size=((Size>>0)&0x7F)
390 | ((Size>>1)&0x3F80)
391 | ((Size>>2)&0x1FC000)
392 | ((Size>>3)&0x0FE00000);
393 Param_Info1(Id3v2_Size);
394 if (ExtendedHeader)
395 {
396 Element_Begin1("Extended header");
397 int32u Size_Extended;
398 Get_B4 (Size_Extended, "Size");
399 Skip_XX(Size_Extended, "Extended header");
400 Element_End0();
401 }
402
403 FILLING_BEGIN();
404 //Versions
405 switch (Id3v2_Version)
406 {
407 case 2 : break;
408 case 3 : break;
409 case 4 : break;
410 default :
411 Skip_XX(Id3v2_Size, "Data");
412 return;
413 }
414
415 Accept("Id3v2");
416 Stream_Prepare(Stream_General);
417
418 Stream_Prepare(Stream_Audio);
419 FILLING_END();
420}

426//---------------------------------------------------------------------------
427void File_Id3v2::Header_Parse()
428{
429 Unsynchronisation_Frame=false;
430 DataLengthIndicator=false;
431
432 if (Id3v2_Size<10) //first 10 is minimum size of a tag, Second 10 is ID3v2 header size
433 {
434 //Not enough place for a tag, must be padding
435 Header_Fill_Code((int64u)-1, "Padding");
436 Header_Fill_Size(Id3v2_Size);
437 return;
438 }
439
440 if (Buffer_Offset+10>Buffer_Size)
441 {
442 Element_WaitForMoreData();
443 return; //Not enough buffer
444 }
445
446 //Testing padding
447 int32u Frame_ID, Size;
448 Frame_ID=CC1(Buffer+Buffer_Offset);
449 if (Frame_ID==0x00)
450 {
451 //This is the padding
452 Header_Fill_Code(0xFFFFFFFF, "Padding");
453 Header_Fill_Size(Id3v2_Size);
454 return;
455 }
456
457 //Parsing
458 if (Id3v2_Version==2)
459 {
460 Get_C3 (Frame_ID, "Frame ID");
461 Get_B3 (Size, "Size");
462 }
463 else
464 {
465 int16u Flags;
466 Get_C4 (Frame_ID, "Frame ID");
467 if (!(Frame_ID&0xFF))
468 Frame_ID>>=8;
469 Get_B4 (Size, "Size");
470 if (Id3v2_Version!=3)
471 {
472 Size=((Size>>0)&0x7F)
473 | ((Size>>1)&0x3F80)
474 | ((Size>>2)&0x1FC000)
475 | ((Size>>3)&0x0FE00000);
476 Param_Info2(Size, " bytes");
477 }
478 Get_B2 (Flags, "Flags");
479 if (Id3v2_Version==3)
480 {
481 Skip_Flags(Flags, 15, "Tag alter preservation");
482 Skip_Flags(Flags, 14, "File alter preservation");
483 Skip_Flags(Flags, 13, "Read only");
484 Skip_Flags(Flags, 7, "Compression");
485 Skip_Flags(Flags, 6, "Encryption");
486 Skip_Flags(Flags, 5, "Grouping identity");
487 }
488 if (Id3v2_Version==4)
489 {
490 Skip_Flags(Flags, 14, "Tag alter preservation");
491 Skip_Flags(Flags, 13, "File alter preservation");
492 Skip_Flags(Flags, 12, "Read only");
493 Skip_Flags(Flags, 6, "Grouping identity");
494 Skip_Flags(Flags, 3, "Compression");
495 Skip_Flags(Flags, 2, "Encryption");
496 Get_Flags (Flags, 1, Unsynchronisation_Frame, "Unsynchronisation");
497 Get_Flags (Flags, 0, DataLengthIndicator, "Data length indicator");
498 }
499 }
500
501 //Hanlding Unsynchronisation
502 if (Unsynchronisation_Global || Unsynchronisation_Frame)
503 {
504 if (Buffer_Offset+(size_t)Element_Offset+Size>Buffer_Size)
505 {
506 Element_WaitForMoreData();
507 return;
508 }
509 for (size_t Element_Offset_Unsynch=0; Element_Offset_Unsynch+2<Element_Offset+Size; Element_Offset_Unsynch++)
510 if (CC2(Buffer+Buffer_Offset+Element_Offset_Unsynch)==0xFF00)
511 {
512 Size++;
513 if (Buffer_Offset+(size_t)Element_Offset+Size>Buffer_Size)
514 {
515 Element_WaitForMoreData();
516 return;
517 }
518 }
519 }
520
521 //Filling
522 Ztring ToShow;
523 if (Id3v2_Version==2)
524 ToShow.From_CC3(Frame_ID);
525 else
526 ToShow.From_CC4(Frame_ID);
527 Header_Fill_Code(Frame_ID, ToShow);
528 Header_Fill_Size(Element_Offset+Size);
529}

532void File_Id3v2::Data_Parse()
533{
534 Id3v2_Size-=Header_Size+Element_Size;
535
536 int32u DataLength=(int32u)-1;
537 if (DataLengthIndicator)
538 {
539 Get_B4 (DataLength, "Data length");
540 DataLength=((DataLength>>0)&0x7F)
541 | ((DataLength>>1)&0x3F80)
542 | ((DataLength>>2)&0x1FC000)
543 | ((DataLength>>3)&0x0FE00000);
544 Param_Info2(DataLength, " bytes");
545 }
546
547 //Unsynchronisation
548 int8u* Buffer_Unsynch=NULL;
549 const int8u* Save_Buffer=Buffer;
550 int64u Save_File_Offset=File_Offset;
551 size_t Save_Buffer_Offset=Buffer_Offset;
552 int64u Save_Element_Size=Element_Size;
553 int64u Element_Offset_Unsynch=Element_Offset;
554 std::vector<size_t> Unsynch_List;
555 if (Unsynchronisation_Global || Unsynchronisation_Frame)
556 {
557 while (Element_Offset_Unsynch+2<Element_Size)
558 {
559 if (CC2(Buffer+Buffer_Offset+(size_t)Element_Offset_Unsynch)==0xFF00)
560 Unsynch_List.push_back((size_t)(Element_Offset_Unsynch+1));
561 Element_Offset_Unsynch++;
562 }
563 if (DataLength!=(int32u)-1 && 4+DataLength!=Element_Size-Unsynch_List.size())
564 {
565 Skip_XX(Element_Size-Element_Offset, "Size coherency issue");
566 return;
567 }
568 if (!Unsynch_List.empty())
569 {
570 //We must change the buffer for keeping out
571 File_Offset=Save_File_Offset+Buffer_Offset;
572 Element_Size=Save_Element_Size-Unsynch_List.size();
573 Buffer_Offset=0;
574 Buffer_Unsynch=new int8u[(size_t)Element_Size];
575 for (size_t Pos=0; Pos<=Unsynch_List.size(); Pos++)
576 {
577 size_t Pos0=(Pos==Unsynch_List.size())?(size_t)Save_Element_Size:(Unsynch_List[Pos]);
578 size_t Pos1=(Pos==0)?0:(Unsynch_List[Pos-1]+1);
579 size_t Buffer_Unsynch_Begin=Pos1-Pos;
580 size_t Save_Buffer_Begin =Pos1;
581 size_t Size= Pos0-Pos1;
582 std::memcpy(Buffer_Unsynch+Buffer_Unsynch_Begin, Save_Buffer+Save_Buffer_Offset+Save_Buffer_Begin, Size);
583 }

585 }
