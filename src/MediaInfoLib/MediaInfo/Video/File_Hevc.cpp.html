217//---------------------------------------------------------------------------
218void File_Hevc::Streams_Fill(std::vector<seq_parameter_set_struct*>::iterator seq_parameter_set_Item)
219{
220 int32u Width = (*seq_parameter_set_Item)->pic_width_in_luma_samples;
221 int32u Height= (*seq_parameter_set_Item)->pic_height_in_luma_samples;
222 int32u CropUnitX=Hevc_SubWidthC [(*seq_parameter_set_Item)->ChromaArrayType()];
223 int32u CropUnitY=Hevc_SubHeightC[(*seq_parameter_set_Item)->ChromaArrayType()];
224 Width -=((*seq_parameter_set_Item)->conf_win_left_offset+(*seq_parameter_set_Item)->conf_win_right_offset)*CropUnitX;
225 Height-=((*seq_parameter_set_Item)->conf_win_top_offset +(*seq_parameter_set_Item)->conf_win_bottom_offset)*CropUnitY;
226
227 Ztring Profile;
228 if ((*seq_parameter_set_Item)->profile_space==0)
229 {
230 if ((*seq_parameter_set_Item)->profile_idc)
231 Profile=Ztring().From_Local(Hevc_profile_idc((*seq_parameter_set_Item)->profile_idc));
232 if ((*seq_parameter_set_Item)->level_idc)
233 {
234 if ((*seq_parameter_set_Item)->profile_idc)
235 Profile+=__T('@');
236 Profile+=__T('L')+Ztring().From_Number(((float)(*seq_parameter_set_Item)->level_idc)/30, ((*seq_parameter_set_Item)->level_idc%10)?1:0);
237 Profile+=__T('@');
238 Profile+=Ztring().From_Local(Hevc_tier_flag((*seq_parameter_set_Item)->tier_flag));
239 }
240 }
241 Fill(Stream_Video, 0, Video_Format_Profile, Profile);
242 Fill(Stream_Video, 0, Video_Codec_Profile, Profile);
243 Fill(Stream_Video, StreamPos_Last, Video_Width, Width);
244 Fill(Stream_Video, StreamPos_Last, Video_Height, Height);
245 if ((*seq_parameter_set_Item)->conf_win_left_offset || (*seq_parameter_set_Item)->conf_win_right_offset)
246 Fill(Stream_Video, StreamPos_Last, Video_Stored_Width, (*seq_parameter_set_Item)->pic_width_in_luma_samples);
247 if ((*seq_parameter_set_Item)->conf_win_top_offset || (*seq_parameter_set_Item)->conf_win_bottom_offset)
248 Fill(Stream_Video, StreamPos_Last, Video_Stored_Height, (*seq_parameter_set_Item)->pic_height_in_luma_samples);
249
250 Fill(Stream_Video, 0, Video_ColorSpace, "YUV");
251 Fill(Stream_Video, 0, Video_Colorimetry, Hevc_chroma_format_idc((*seq_parameter_set_Item)->chroma_format_idc));
252 if ((*seq_parameter_set_Item)->bit_depth_luma_minus8==(*seq_parameter_set_Item)->bit_depth_chroma_minus8)
253 Fill(Stream_Video, 0, Video_BitDepth, (*seq_parameter_set_Item)->bit_depth_luma_minus8+8);
254
255 if ((*seq_parameter_set_Item)->vui_parameters)
256 {
257 if ((*seq_parameter_set_Item)->vui_parameters->timing_info_present_flag)
258 {
259 if ((*seq_parameter_set_Item)->vui_parameters->time_scale && (*seq_parameter_set_Item)->vui_parameters->num_units_in_tick)
260 Fill(Stream_Video, StreamPos_Last, Video_FrameRate, (float64)(*seq_parameter_set_Item)->vui_parameters->time_scale / (*seq_parameter_set_Item)->vui_parameters->num_units_in_tick);
261 }
262
263 if ((*seq_parameter_set_Item)->vui_parameters->aspect_ratio_info_present_flag)
264 {
265 float64 PixelAspectRatio = 1;
266 if ((*seq_parameter_set_Item)->vui_parameters->aspect_ratio_idc<Avc_PixelAspectRatio_Size)
267 PixelAspectRatio = Avc_PixelAspectRatio[(*seq_parameter_set_Item)->vui_parameters->aspect_ratio_idc];
268 else if ((*seq_parameter_set_Item)->vui_parameters->aspect_ratio_idc == 0xFF && (*seq_parameter_set_Item)->vui_parameters->sar_height)
269 PixelAspectRatio = ((float64) (*seq_parameter_set_Item)->vui_parameters->sar_width) / (*seq_parameter_set_Item)->vui_parameters->sar_height;
270
271 Fill(Stream_Video, 0, Video_PixelAspectRatio, PixelAspectRatio, 3, true);
272 Fill(Stream_Video, 0, Video_DisplayAspectRatio, Width*PixelAspectRatio/Height, 3, true); //More precise
273 }
274
275 //Colour description
276 if ((*seq_parameter_set_Item)->vui_parameters->video_signal_type_present_flag)
277 {
278 Fill(Stream_Video, 0, Video_Standard, Avc_video_format[(*seq_parameter_set_Item)->vui_parameters->video_format]);
279 Fill(Stream_Video, 0, Video_colour_range, Avc_video_full_range[(*seq_parameter_set_Item)->vui_parameters->video_full_range_flag]);
280 if ((*seq_parameter_set_Item)->vui_parameters->colour_description_present_flag)
281 {
282 Fill(Stream_Video, 0, Video_colour_description_present, "Yes");
283 Fill(Stream_Video, 0, Video_colour_primaries, Mpegv_colour_primaries((*seq_parameter_set_Item)->vui_parameters->colour_primaries));
284 Fill(Stream_Video, 0, Video_transfer_characteristics, Mpegv_transfer_characteristics((*seq_parameter_set_Item)->vui_parameters->transfer_characteristics));
285 Fill(Stream_Video, 0, Video_matrix_coefficients, Mpegv_matrix_coefficients((*seq_parameter_set_Item)->vui_parameters->matrix_coefficients));
286 }
287 }
288 }

317//***************************************************************************
318// Buffer - Synchro
319//***************************************************************************
320
321//---------------------------------------------------------------------------
322bool File_Hevc::Synchronize()
323{
324 //Synchronizing
325 size_t Buffer_Offset_Min=Buffer_Offset;
326 while(Buffer_Offset+4<=Buffer_Size && (Buffer[Buffer_Offset ]!=0x00
327 || Buffer[Buffer_Offset+1]!=0x00
328 || Buffer[Buffer_Offset+2]!=0x01))
329 {
330 Buffer_Offset+=2;
331 while(Buffer_Offset<Buffer_Size && Buffer[Buffer_Offset]!=0x00)
332 Buffer_Offset+=2;
333 if (Buffer_Offset>=Buffer_Size || Buffer[Buffer_Offset-1]==0x00)
334 Buffer_Offset--;
335 }
336 if (Buffer_Offset>Buffer_Offset_Min && Buffer[Buffer_Offset-1]==0x00)
337 Buffer_Offset--;
338
339 //Parsing last bytes if needed
340 if (Buffer_Offset+4==Buffer_Size && (Buffer[Buffer_Offset ]!=0x00
341 || Buffer[Buffer_Offset+1]!=0x00
342 || Buffer[Buffer_Offset+2]!=0x00
343 || Buffer[Buffer_Offset+3]!=0x01))
344 Buffer_Offset++;
345 if (Buffer_Offset+3==Buffer_Size && (Buffer[Buffer_Offset ]!=0x00
346 || Buffer[Buffer_Offset+1]!=0x00
347 || Buffer[Buffer_Offset+2]!=0x01))
348 Buffer_Offset++;
349 if (Buffer_Offset+2==Buffer_Size && (Buffer[Buffer_Offset ]!=0x00
350 || Buffer[Buffer_Offset+1]!=0x00))
351 Buffer_Offset++;
352 if (Buffer_Offset+1==Buffer_Size && Buffer[Buffer_Offset ]!=0x00)
353 Buffer_Offset++;
354
355 if (Buffer_Offset+4>Buffer_Size)
356 return false;
357
358 if (File_Offset==0 && Buffer_Offset==0 && (Buffer[3]==0xE0 || Buffer[3]==0xFE))
359 {
360 //It is from MPEG-PS
361 Reject();
362 return false;
363 }
364
365 //Synched is OK
366 Synched=true;
367 return true;
368}

398//---------------------------------------------------------------------------
399#if MEDIAINFO_DEMUX
400bool File_Hevc::Demux_UnpacketizeContainer_Test()
401{
402 const int8u* Buffer_Temp=NULL;
403 size_t Buffer_Temp_Size=0;
404 bool RandomAccess=true; //Default, in case of problem
405
406 if ((MustParse_VPS_SPS_PPS || SizedBlocks) && Demux_Transcode_Iso14496_15_to_AnnexB)
407 {
408 if (MustParse_VPS_SPS_PPS)
409 return true; //Wait for SPS and PPS
410
411 //Random access check
412 RandomAccess=false;
413
414 //Computing final size
415 size_t TranscodedBuffer_Size=0;
416 while (Buffer_Offset+lengthSizeMinusOne+1+1<=Buffer_Size)
417 {
418 size_t Size;
419 switch (lengthSizeMinusOne)
420 {
421 case 0: Size=Buffer[Buffer_Offset];
422 TranscodedBuffer_Size+=2;
423 break;
424 case 1: Size=BigEndian2int16u(Buffer+Buffer_Offset);
425 TranscodedBuffer_Size++;
426 break;
427 case 2: Size=BigEndian2int24u(Buffer+Buffer_Offset);
428 break;
429 case 3: Size=BigEndian2int32u(Buffer+Buffer_Offset);
430 TranscodedBuffer_Size--;
431 break;
432 default: return true; //Problem
433 }
434 Size+=lengthSizeMinusOne+1;
435
436 //Coherency checking
437 if (Size==0 || Buffer_Offset+Size>Buffer_Size || (Buffer_Offset+Size!=Buffer_Size && Buffer_Offset+Size+lengthSizeMinusOne+1>Buffer_Size))
438 Size=Buffer_Size-Buffer_Offset;
439
440 //Random access check
441 if (!RandomAccess && Buffer_Offset+lengthSizeMinusOne+1<Buffer_Size && (Buffer[Buffer_Offset+lengthSizeMinusOne+1]&0x40)==0) //Is a slice
442 {
443 Element_Offset=lengthSizeMinusOne+1+2;
444 Element_Size=Size;
445
446 BS_Begin();
447 Get_SB ( first_slice_segment_in_pic_flag, "first_slice_segment_in_pic_flag");
448 if (first_slice_segment_in_pic_flag)
449 {
450 Element_Code=(Buffer[Buffer_Offset+lengthSizeMinusOne+1]&0x3E)>>1; //nal_unit_type
451 RapPicFlag=Element_Code>=16 && Element_Code<=23;
452 if (RapPicFlag)
453 Skip_SB( "no_output_of_prior_pics_flag");
454 Get_UE ( slice_pic_parameter_set_id, "slice_pic_parameter_set_id");
455 int8u num_extra_slice_header_bits=(int8u)-1;
456 std::vector<pic_parameter_set_struct*>::iterator pic_parameter_set_Item;
457 if (MustParse_VPS_SPS_PPS_FromFlv)
458 num_extra_slice_header_bits=0; // We bet it is old, so without num_extra_slice_header_bits
459 else if (!(slice_pic_parameter_set_id>=pic_parameter_sets.size() || (*(pic_parameter_set_Item=pic_parameter_sets.begin()+slice_pic_parameter_set_id))==NULL))
460 {
461 num_extra_slice_header_bits=(*pic_parameter_set_Item)->num_extra_slice_header_bits;
462 }
463 if (num_extra_slice_header_bits!=(int8u)-1)
464 {
465 int32u slice_type;
466 Skip_S1(num_extra_slice_header_bits, "slice_reserved_flags");
467 Get_UE (slice_type, "slice_type");
468
469 switch (slice_type)
470 {
471 case 2 :
472 case 7 :
473 RandomAccess=true;
474 }
475 }
476 }
477 BS_End();
478 }
479
480 TranscodedBuffer_Size+=Size;
481 Buffer_Offset+=Size;
482 }
483 Buffer_Offset=0;
484
485 //Adding VPS/SPS/PPS sizes
486 if (RandomAccess)
487 {
488 for (video_parameter_set_structs::iterator Data_Item=video_parameter_sets.begin(); Data_Item!=video_parameter_sets.end(); ++Data_Item)
489 TranscodedBuffer_Size+=(*Data_Item)->AnnexB_Buffer_Size;
490 for (seq_parameter_set_structs::iterator Data_Item=seq_parameter_sets.begin(); Data_Item!=seq_parameter_sets.end(); ++Data_Item)
491 TranscodedBuffer_Size+=(*Data_Item)->AnnexB_Buffer_Size;
492 for (pic_parameter_set_structs::iterator Data_Item=pic_parameter_sets.begin(); Data_Item!=pic_parameter_sets.end(); ++Data_Item)
493 TranscodedBuffer_Size+=(*Data_Item)->AnnexB_Buffer_Size;
494 }
495
496 //Copying
497 int8u* TranscodedBuffer=new int8u[TranscodedBuffer_Size+100];
498 size_t TranscodedBuffer_Pos=0;
499 if (RandomAccess)
500 {
501 for (video_parameter_set_structs::iterator Data_Item=video_parameter_sets.begin(); Data_Item!=video_parameter_sets.end(); ++Data_Item)
502 {
503 std::memcpy(TranscodedBuffer+TranscodedBuffer_Pos, (*Data_Item)->AnnexB_Buffer, (*Data_Item)->AnnexB_Buffer_Size);
504 TranscodedBuffer_Pos+=(*Data_Item)->AnnexB_Buffer_Size;
505 }
506 for (seq_parameter_set_structs::iterator Data_Item=seq_parameter_sets.begin(); Data_Item!=seq_parameter_sets.end(); ++Data_Item)
507 {
508 std::memcpy(TranscodedBuffer+TranscodedBuffer_Pos, (*Data_Item)->AnnexB_Buffer, (*Data_Item)->AnnexB_Buffer_Size);
509 TranscodedBuffer_Pos+=(*Data_Item)->AnnexB_Buffer_Size;
510 }
511 for (pic_parameter_set_structs::iterator Data_Item=pic_parameter_sets.begin(); Data_Item!=pic_parameter_sets.end(); ++Data_Item)
512 {
513 std::memcpy(TranscodedBuffer+TranscodedBuffer_Pos, (*Data_Item)->AnnexB_Buffer, (*Data_Item)->AnnexB_Buffer_Size);
514 TranscodedBuffer_Pos+=(*Data_Item)->AnnexB_Buffer_Size;
515 }
516 }
517 while (Buffer_Offset<Buffer_Size)
518 {
519 //Sync layer
520 TranscodedBuffer[TranscodedBuffer_Pos]=0x00;
521 TranscodedBuffer_Pos++;
522 TranscodedBuffer[TranscodedBuffer_Pos]=0x00;
523 TranscodedBuffer_Pos++;
524 TranscodedBuffer[TranscodedBuffer_Pos]=0x01;
525 TranscodedBuffer_Pos++;
526
527 //Block
528 size_t Size;
529 switch (lengthSizeMinusOne)
530 {
531 case 0: Size=Buffer[Buffer_Offset];
532 Buffer_Offset++;
533 break;
534 case 1: Size=BigEndian2int16u(Buffer+Buffer_Offset);
535 Buffer_Offset+=2;
536 break;
537 case 2: Size=BigEndian2int24u(Buffer+Buffer_Offset);
538 Buffer_Offset+=3;
539 break;
540 case 3: Size=BigEndian2int32u(Buffer+Buffer_Offset);
541 Buffer_Offset+=4;
542 break;
543 default: //Problem
544 delete [] TranscodedBuffer;
545 return false;
546 }
547
548 //Coherency checking
549 if (Size==0 || Buffer_Offset+Size>Buffer_Size || (Buffer_Offset+Size!=Buffer_Size && Buffer_Offset+Size+lengthSizeMinusOne+1>Buffer_Size))
550 Size=Buffer_Size-Buffer_Offset;
551
552 std::memcpy(TranscodedBuffer+TranscodedBuffer_Pos, Buffer+Buffer_Offset, Size);
553 TranscodedBuffer_Pos+=Size;
554 Buffer_Offset+=Size;
555 }
556 Buffer_Offset=0;
557
558 Buffer_Temp=Buffer;
559 Buffer=TranscodedBuffer;
560 Buffer_Temp_Size=Buffer_Size;
561 Buffer_Size=TranscodedBuffer_Size;
562 Demux_Offset=Buffer_Size;
563 }
564 else
565 {
566 bool zero_byte=Buffer[Buffer_Offset+2]==0x00;
567 if (!(((Buffer[Buffer_Offset+(zero_byte?4:3)]&0x40)==0 && (Buffer[Buffer_Offset+(zero_byte?6:5)]&0x80)!=0x80)
568 || (Buffer[Buffer_Offset+(zero_byte?4:3)]&0x7E)==(38<<1)))
569 {
570 if (Demux_Offset==0)
571 {
572 Demux_Offset=Buffer_Offset;
573 Demux_IntermediateItemFound=false;
574 }
575 while (Demux_Offset+6<=Buffer_Size)
576 {
577 //Synchronizing
578 while(Demux_Offset+6<=Buffer_Size && (Buffer[Demux_Offset ]!=0x00
579 || Buffer[Demux_Offset+1]!=0x00
580 || Buffer[Demux_Offset+2]!=0x01))
581 {
582 Demux_Offset+=2;
583 while(Demux_Offset<Buffer_Size && Buffer[Buffer_Offset]!=0x00)
584 Demux_Offset+=2;
585 if (Demux_Offset>=Buffer_Size || Buffer[Demux_Offset-1]==0x00)
586 Demux_Offset--;
587 }
588
589 if (Demux_Offset+6>Buffer_Size)
590 {
591 if (File_Offset+Buffer_Size==File_Size)
592 Demux_Offset=Buffer_Size;
593 break;
594 }
595
596 zero_byte=Buffer[Demux_Offset+2]==0x00;
597 int8u nal_unit_type=Buffer[Demux_Offset+(zero_byte?4:3)]>>1;
598 bool Next;
599 switch (nal_unit_type)
600 {
601 case 0 :
602 case 1 :
603 case 2 :
604 case 3 :
605 case 4 :
606 case 5 :
607 case 6 :
608 case 7 :
609 case 8 :
610 case 9 :
611 case 16 :
612 case 17 :
613 case 18 :
614 case 19 :
615 case 20 :
616 case 21 :
617 if (Demux_IntermediateItemFound)
618 {
619 if (Buffer[Demux_Offset+(zero_byte?6:5)]&0x80)
620 Next=true;
621 else
622 Next=false;
623 }
624 else
625 {
626 Next=false;
627 Demux_IntermediateItemFound=true;
628 }
629 break;
630 case 32 :
631 case 33 :
632 case 34 :
633 case 35 :
634 if (Demux_IntermediateItemFound)
635 Next=true;
636 else
637 Next=false;
638 break;
639 default : Next=false;
640 }
641
642 if (Next)
643 {
644 Demux_IntermediateItemFound=false;
645 break;
646 }
647
648 Demux_Offset++;
649 }
650

847//---------------------------------------------------------------------------
848bool File_Hevc::Header_Parser_Fill_Size()
849{
850 //Look for next Sync word
851 if (Buffer_Offset_Temp==0) //Buffer_Offset_Temp is not 0 if Header_Parse_Fill_Size() has already parsed first frames
852 Buffer_Offset_Temp=Buffer_Offset+4;
853 while (Buffer_Offset_Temp+5<=Buffer_Size
854 && CC3(Buffer+Buffer_Offset_Temp)!=0x000001)
855 {
856 Buffer_Offset_Temp+=2;
857 while(Buffer_Offset_Temp<Buffer_Size && Buffer[Buffer_Offset_Temp]!=0x00)
858 Buffer_Offset_Temp+=2;
859 if (Buffer_Offset_Temp>=Buffer_Size || Buffer[Buffer_Offset_Temp-1]==0x00)
860 Buffer_Offset_Temp--;
861 }
862
863 //Must wait more data?
864 if (Buffer_Offset_Temp+5>Buffer_Size)
865 {
866 if (FrameIsAlwaysComplete || File_Offset+Buffer_Size>=File_Size)
867 Buffer_Offset_Temp=Buffer_Size; //We are sure that the next bytes are a start
868 else
869 return false;
870 }
871
872 if (Buffer[Buffer_Offset_Temp-1]==0x00)
873 Buffer_Offset_Temp--;
874
875 //OK, we continue
876 Header_Fill_Size(Buffer_Offset_Temp-Buffer_Offset);
877 Buffer_Offset_Temp=0;
878 return true;
879}
