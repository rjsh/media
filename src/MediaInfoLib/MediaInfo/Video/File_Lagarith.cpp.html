53//***************************************************************************
54// Buffer - Global
55//***************************************************************************
56
57//---------------------------------------------------------------------------
58void File_Lagarith::Read_Buffer_Continue()
59{
60 //Parsing
61 int8u version;
62 Get_L1 (version, "version");
63 Skip_XX(Element_Size-Element_Offset, "data");
64
65 FILLING_BEGIN();
66 Accept();
67 Fill();
68 switch (version)
69 {
70 case 0x02 :
71 case 0x04 : Fill(Stream_Video, 0, Video_ColorSpace, "RGB"); Fill(Stream_Video, 0, Video_BitDepth, 8); break;
72 case 0x03 : Fill(Stream_Video, 0, Video_ColorSpace, "YUV"); Fill(Stream_Video, 0, Video_ChromaSubsampling, "4:2:2"); Fill(Stream_Video, 0, Video_BitDepth, 8); break;
73 case 0x05 : Fill(Stream_Video, 0, Video_ColorSpace, "Y"); break;
74 case 0x06 :
75 case 0x07 : Fill(Stream_Video, 0, Video_ColorSpace, "RGB"); break;
76 case 0x08 :
77 case 0x09 : Fill(Stream_Video, 0, Video_ColorSpace, "RGBA"); break;
78 case 0x0A :
79 case 0x0B : Fill(Stream_Video, 0, Video_ColorSpace, "YUV"); Fill(Stream_Video, 0, Video_ChromaSubsampling, "4:2:0"); Fill(Stream_Video, 0, Video_BitDepth, 8); break;
80 default : ;
81 }
82
83 FILLING_END();
84 Finish();
85}
