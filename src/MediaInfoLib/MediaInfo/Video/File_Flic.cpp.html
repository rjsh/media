41void File_Flic::FileHeader_Parse()
42{
43 //Parsing
44 int32u DelayBetweenFrames;
45 int16u Type, Frames, Width, Height, BitsPerPixel, AspectX=0, AspectY=0;
46 Skip_L4( "Size of FLIC including this header");
47 Get_L2 (Type, "File type");
48 Get_L2 (Frames, "Number of frames in first segment");
49 Get_L2 (Width, "Width");
50 Get_L2 (Height, "Height");
51 Get_L2 (BitsPerPixel, "Bits per pixel");
52 Skip_L2( "Flags");
53 Get_L4 (DelayBetweenFrames, "Delay between frames");
54 if (Type!=0xAF11)
55 {
56 Skip_L2( "Reserved");
57 Skip_L4( "Date of Creation)");
58 Skip_L4( "Serial number or compiler id");
59 Skip_L4( "Date of FLIC update");
60 Skip_L4( "Serial number");
61 Get_L2 (AspectX, "Width of square rectangle");
62 Get_L2 (AspectY, "Height of square rectangle");
63 }
64 else
65 Skip_XX(22, "Reserved");
66 Skip_L2( "EGI: flags for specific EGI extensions");
67 Skip_L2( "EGI: key-image frequency");
68 Skip_L2( "EGI: total number of frames (segments)");
69 Skip_L4( "EGI: maximum chunk size (uncompressed)");
70 Skip_L2( "EGI: max. number of regions in a CHK_REGION chunk");
71 Skip_L2( "EGI: number of transparent levels");
72 if (Type!=0xAF11)
73 {
74 Skip_XX(24, "Reserved");
75 Skip_L4( "Offset to frame 1");
76 Skip_L4( "Offset to frame 2");
77 Skip_XX(40, "Reserved");
78 }
79 else
80 Skip_XX(72, "Reserved");
81
82 //Filling
83 FILLING_BEGIN();
84 switch (Type)
85 {
86 case 0xAF11 :
87 case 0xAF12 :
88 case 0xAF30 :
89 case 0xAF31 :
90 case 0xAF44 :
91 break;
92 default :
93 Reject("FLIC");
94 return;
95 }
96
97 //Filling
98 Accept("FLIC");
99
100 Fill(Stream_General, 0, General_Format, "FLIC");
101
102 Stream_Prepare(Stream_Video);
103 if (Type==0xAF11)
104 {
105 Fill(Stream_Video, 0, Video_Format, "FLI");
106 Fill(Stream_Video, 0, Video_Codec, "FLI");
107 if (DelayBetweenFrames)
108 {
109 Fill(Stream_Video, StreamPos_Last, Video_FrameRate, 1000.0/(DelayBetweenFrames*70)); //multiple of 1/70 per frame
110 Fill(Stream_Video, 0, Video_Duration, Frames*DelayBetweenFrames*70);
111 }
112 }
113 else
114 {
115 Fill(Stream_Video, 0, Video_Format, "FLC");
116 Fill(Stream_Video, 0, Video_Codec, "FLC");
117 if (DelayBetweenFrames)
118 {
119 Fill(Stream_Video, StreamPos_Last, Video_FrameRate, 1000.0/DelayBetweenFrames); //ms per frame
120 Fill(Stream_Video, 0, Video_Duration, Frames*DelayBetweenFrames);
121 }
122 if (AspectY>0)
123 Fill(Stream_Video, StreamPos_Last, Video_DisplayAspectRatio, AspectX/AspectY, 3, true);
124 }
125 Fill(Stream_Video, 0, Video_FrameCount, Frames);
126 Fill(Stream_Video, StreamPos_Last, Video_Width, Width);
127 Fill(Stream_Video, StreamPos_Last, Video_Height, Height);
128 Fill(Stream_Video, 0, Video_BitDepth, (BitsPerPixel%3)?BitsPerPixel:(BitsPerPixel/3), 10, true); //If not a multiple of 3, the total resolution is filled
129 //No more need data
130 Finish("FLIC");
131 FILLING_END();
132}
