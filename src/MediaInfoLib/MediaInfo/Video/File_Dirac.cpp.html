440 Fill(Stream_Video, 0, Video_Colorimetry, Dirac_chroma_format(chroma_format));
441 Fill(Stream_Video, 0, Video_ScanType, Dirac_source_sampling(source_sampling));
442 Fill(Stream_Video, 0, Video_Interlacement, Dirac_source_sampling_Codec(source_sampling));
443}

457//---------------------------------------------------------------------------
458bool File_Dirac::Synchronize()
459{
460 //Synchronizing
461 while(Buffer_Offset+4<=Buffer_Size && (Buffer[Buffer_Offset ]!=0x42
462 || Buffer[Buffer_Offset+1]!=0x42
463 || Buffer[Buffer_Offset+2]!=0x43
464 || Buffer[Buffer_Offset+3]!=0x44)) //"BBCD"
465 {
466 Buffer_Offset+=2;
467 while(Buffer_Offset<Buffer_Size && Buffer[Buffer_Offset]!=0x42)
468 Buffer_Offset+=2;
469 if (Buffer_Offset>=Buffer_Size || Buffer[Buffer_Offset-1]==0x42)
470 Buffer_Offset--;
471 }
472
473 //Parsing last bytes if needed
474 if (Buffer_Offset+4>Buffer_Size)
475 {
476 if (Buffer_Offset+3==Buffer_Size && CC3(Buffer+Buffer_Offset)!=0x424243) //"BBC"
477 Buffer_Offset++;
478 if (Buffer_Offset+2==Buffer_Size && CC2(Buffer+Buffer_Offset)!=0x4242) //"BB"
479 Buffer_Offset++;
480 if (Buffer_Offset+1==Buffer_Size && CC1(Buffer+Buffer_Offset)!=0x42) //"B"
481 Buffer_Offset++;
482 return false;
483 }
484
485 //Synched is OK
486 return true;
487}

489//---------------------------------------------------------------------------
490bool File_Dirac::Synched_Test()
491{
492 //Must have enough buffer for having header
493 if (Buffer_Offset+4>Buffer_Size)
494 return false;
495
496 //Quick test of synchro
497 if (CC4(Buffer+Buffer_Offset)!=0x42424344) //"BBCD"
498 Synched=false;
499
500 //Quick search
501 if (Synched && !Header_Parser_QuickSearch())
502 return false;
503
504 //We continue
505 return true;
506}

525//---------------------------------------------------------------------------
526void File_Dirac::Header_Parse()
527{
528 //Parsing
529 int32u Next_Parse_Offset, Previous_Parse_Offset;
530 int8u Parse_Code;
531 Skip_C4( "Parse Info Prefix");
532 Get_B1 (Parse_Code, "Parse Code");
533 Get_B4 (Next_Parse_Offset, "Next Parse Offset");
534 Get_B4 (Previous_Parse_Offset, "Previous Parse Offset");
535
536 //Filling
537 Header_Fill_Code(Parse_Code, Ztring().From_CC1(Parse_Code));
538 Header_Fill_Size((Parse_Code==0x10 && Next_Parse_Offset==0)?13:Next_Parse_Offset); //Speacial case if this is the End Of Sequence
539}

541//---------------------------------------------------------------------------
542bool File_Dirac::Header_Parser_QuickSearch()
543{
544 while ( Buffer_Offset+5<=Buffer_Size
545 && Buffer[Buffer_Offset ]==0x42
546 && Buffer[Buffer_Offset+1]==0x42
547 && Buffer[Buffer_Offset+2]==0x43
548 && Buffer[Buffer_Offset+3]==0x44) //"BBCD"
549 {
550 //Getting start_code
551 int8u start_code=CC1(Buffer+Buffer_Offset+4);
552
553 //Searching start
554 if (Streams[start_code].Searching_Payload)
555 return true;
556
557 //Getting size
558 Buffer_Offset+=BigEndian2int32u(Buffer+Buffer_Offset+5);
559 }
560
561 if (Buffer_Offset+4==Buffer_Size)
562 return false; //Sync is OK, but start_code is not available
563 if (Buffer_Offset+5<=Buffer_Size)
564 Trusted_IsNot("Dirac, Synchronisation lost");
565 Synched=false;
566 return Synchronize();
567}

569//---------------------------------------------------------------------------
570void File_Dirac::Data_Parse()
571{
572 //Parsing
573 switch (Element_Code)
574 {
575 case 0x00 : Sequence_header(); break;
576 case 0x10 : End_of_Sequence(); break;
577 case 0x20 : Auxiliary_data(); break;
578 case 0x30 : Padding_data(); break;
579 case 0x0C : Intra_Reference_Picture(); break;
580 case 0x08 : Intra_Non_Reference_Picture(); break;
581 case 0x4C : Intra_Reference_Picture_No(); break;
582 case 0x48 : Intra_Non_Reference_Picture_No(); break;
583 case 0x0D : Inter_Reference_Picture_1(); break;
584 case 0x0E : Inter_Reference_Picture_2(); break;
585 case 0x09 : Inter_Non_Reference_Picture_1(); break;
586 case 0x0A : Inter_Non_Reference_Picture_2(); break;
587 case 0xCC : Reference_Picture_Low(); break;
588 case 0xC8 : Intra_Non_Reference_Picture_Low(); break;
589 default : Reserved();
590 }
591}

597//---------------------------------------------------------------------------
598// Packet "00"
599void File_Dirac::Sequence_header()
600{
601 Element_Name("Sequence header");
602
603 //Parsing
604 int32u version_major, version_minor, profile, level, base_video_format;
605 BS_Begin();
606 Get_UI(version_major, "version major");
607 Get_UI(version_minor, "version minor");
608 Get_UI(profile, "profile");
609 Get_UI(level, "level");
610
611 if (version_major<=2)
612 {
613 Get_UI(base_video_format, "base video format"); //Param_Info1(Dirac_base_video_format(base_video_format));
614 Dirac_base_video_format(base_video_format, frame_width, frame_height, chroma_format, source_sampling,
615 clean_width, clean_height, clean_left_offset, clean_top_offset,
616 frame_rate, pixel_aspect_ratio);
617 TEST_SB_SKIP( "custom dimensions flag");
618 Get_UI (frame_width, "frame width");
619 Get_UI (frame_height, "frame height");
620 TEST_SB_END();
621 TEST_SB_SKIP( "custom chroma format flag");
622 Get_UI (chroma_format, "chroma format"); Param_Info1(Dirac_chroma_format(chroma_format));
623 TEST_SB_END();
624 TEST_SB_SKIP( "custom scan format flag");
625 Get_UI (source_sampling, "source sampling"); Param_Info1(Dirac_source_sampling(source_sampling));
626 TEST_SB_END();
627 TEST_SB_SKIP( "frame rate flag");
628 int32u frame_rate_index;
629 Get_UI (frame_rate_index, "index"); Param_Info1(Dirac_frame_rate(frame_rate_index));
630 if (frame_rate_index==0)
631 {
632 int32u frame_rate_numer, frame_rate_denom;
633 Get_UI (frame_rate_numer, "frame rate numer");
634 Get_UI (frame_rate_denom, "frame rate denom");
635 frame_rate=((float32)frame_rate_numer)/((float32)frame_rate_denom);
636 }
637 else
638 frame_rate=Dirac_frame_rate(frame_rate_index);
639 TEST_SB_END();
640 TEST_SB_SKIP( "pixel aspect ratio flag");
641 int32u pixel_aspect_ratio_index;
642 Get_UI (pixel_aspect_ratio_index, "index"); Param_Info1(Dirac_pixel_aspect_ratio(pixel_aspect_ratio_index));
643 if (pixel_aspect_ratio_index==0)
644 {
645 int32u pixel_aspect_ratio_numer, pixel_aspect_ratio_denom;
646 Get_UI (pixel_aspect_ratio_numer, "pixel aspect ratio numer");
647 Get_UI (pixel_aspect_ratio_denom, "pixel aspect ratio denom");
648 pixel_aspect_ratio=((float32)pixel_aspect_ratio_numer)/((float32)pixel_aspect_ratio_denom);
649 }
650 else
651 pixel_aspect_ratio=Dirac_pixel_aspect_ratio(pixel_aspect_ratio_index);
652 TEST_SB_END();
653 TESTELSE_SB_SKIP( "custom clean area flag");
654 Get_UI (clean_width, "clean width");
655 Get_UI (clean_height, "clean height");
656 Get_UI (clean_left_offset, "clean left offset");
657 Get_UI (clean_top_offset, "clean top offset");
658 TESTELSE_SB_ELSE( "custom clean area flag");
659 clean_width=frame_width;
660 clean_height=frame_height;
661 TESTELSE_SB_END();
662 TEST_SB_SKIP( "custom signal range flag");
663 int32u custom_signal_range_index;
664 Get_UI(custom_signal_range_index, "index");
665 if (custom_signal_range_index==0)
666 {
667 Skip_UI( "luma offset");
668 Skip_UI( "luma excursion");
669 Skip_UI( "chroma offset");
670 Skip_UI( "chroma excursion");
671 }
672 TEST_SB_END();
673 TEST_SB_SKIP( "custom colour spec flag");
674 int32u custom_colour_spec_index;
675 Get_UI(custom_colour_spec_index, "index");
676 if (custom_colour_spec_index==0)
677 {
678 TEST_SB_SKIP( "custom colour primaries flag");
679 Skip_UI( "custom colour primaries index");
680 TEST_SB_END();
681 TEST_SB_SKIP( "colour matrix flag");
682 Skip_UI( "colour matrix index");
683 TEST_SB_END();
684 TEST_SB_SKIP( "custom transfer function flag");
685 Skip_UI( "custom transfer function index");
686 TEST_SB_END();
687 }
688 TEST_SB_END();
689 Info_UI(picture_coding_mode, "picture coding mode"); Param_Info1(Dirac_picture_coding_mode(picture_coding_mode));
690 }
691 else
692 {
693 Skip_XX(Element_Size-Element_Offset, "Unknown");
694 }
695
696 FILLING_BEGIN();
697 //Autorisation of other streams
698 Streams[0x10].Searching_Payload=true; //End_of_Sequence
699 Streams[0x20].Searching_Payload=true; //Auxiliary_data
700 Streams[0x30].Searching_Payload=true; //Padding_data
701 Streams[0x0C].Searching_Payload=true; //Intra_Reference_Picture
702 Streams[0x08].Searching_Payload=true; //Intra_Non_Reference_Picture
703 Streams[0x4C].Searching_Payload=true; //Intra_Reference_Picture_No
704 Streams[0x48].Searching_Payload=true; //Intra_Non_Reference_Picture_No
705 Streams[0x0D].Searching_Payload=true; //Inter_Reference_Picture_1
706 Streams[0x0E].Searching_Payload=true; //Inter_Reference_Picture_2

857//---------------------------------------------------------------------------
858void File_Dirac::picture()
859{
860 //Parsing
861 Skip_XX(Element_Size, "Data");
862
863 FILLING_BEGIN();
864 //Counting
865 if (File_Offset+Buffer_Offset+Element_Size==File_Size)
866 Frame_Count_Valid=Frame_Count; //Finalize frames in case of there are less than Frame_Count_Valid frames
867
868 //Name
869 Element_Info1(Ztring::ToZtring(Frame_Count));
870
871 //Filling only if not already done
872 Frame_Count++;
873 Frame_Count_InThisBlock++;
874 if (Frame_Count>=Frame_Count_Valid && Count_Get(Stream_Video)==0)
875 {
876 NextCode_Clear();
877 Accept("Dirac");
878 Finish("Dirac");
879 }
880 FILLING_END();
881}
