39//---------------------------------------------------------------------------
40const char* ProRes_chrominance_factor(int8u chrominance_factor)
41{
42 switch (chrominance_factor)
43 {
44 case 0x02 : return "4:2:2";
45 case 0x03 : return "4:4:4";
46 default : return "";
47 }
48}
49
50//---------------------------------------------------------------------------
51const char* ProRes_frame_type_ScanType(int8u frame_type)
52{
53 switch (frame_type)
54 {
55 case 0x00 : return "Progressive";
56 case 0x01 :
57 case 0x02 : return "Interlaced";
58 default : return "";
59 }
60}
61
62//---------------------------------------------------------------------------
63const char* ProRes_frame_type_ScanOrder(int8u frame_type)
64{
65 switch (frame_type)
66 {
67 case 0x01 : return "TFF";
68 case 0x02 : return "BFF";
69 default : return "";
70 }
71}
72
73//---------------------------------------------------------------------------
74Ztring ProRes_creatorID(int32u creatorID)
75{
76 switch (creatorID)
77 {
78 case 0x61706C30 : return __T("Apple"); //apl0
79 case 0x61727269 : return __T("Arnold & Richter Cine Technik"); //arri
80 case 0x616A6130 : return __T("AJA Kona Hardware"); //aja0
81 default : return Ztring().From_CC4(creatorID);
82 }
83}

108//***************************************************************************
109// Buffer - Global
110//***************************************************************************
111
112//---------------------------------------------------------------------------
113void File_ProRes::Read_Buffer_Continue()
114{
115 //Parsing
116 int32u Name, creatorID;
117 int16u hdrSize, version, frameWidth, frameHeight;
118 int8u chrominance_factor, frame_type, primaries, transf_func, colorMatrix;
119 bool IsOk=true, luma, chroma;
120 Element_Begin1("Header");
121 Skip_B4( "Size");
122 Get_C4 (Name, "Name");
123 Element_End();
124 Element_Begin1("Frame header");
125 Get_B2 (hdrSize, "hdrSize");
126 Get_B2 (version, "version");
127 Get_C4 (creatorID, "creatorID");
128 Get_B2 (frameWidth, "frameWidth");
129 Get_B2 (frameHeight, "frameHeight");
130 BS_Begin();
131 Get_S1 (2, chrominance_factor, "chrominance factor"); Param_Info1(ProRes_chrominance_factor(chrominance_factor));
132 Skip_S1(2, "reserved");
133 Get_S1 (2, frame_type, "frame type"); Param_Info1(ProRes_frame_type_ScanType(frame_type)); Param_Info1(ProRes_frame_type_ScanOrder(frame_type));
134 Skip_S1(2, "reserved");
135 BS_End();
136 Skip_B1( "reserved");
137 Get_B1 (primaries, "primaries"); Param_Info1(Mpegv_colour_primaries(primaries));
138 Get_B1 (transf_func, "transf_func"); Param_Info1(Mpegv_transfer_characteristics(transf_func));
139 Get_B1 (colorMatrix, "colorMatrix"); Param_Info1(Mpegv_matrix_coefficients(colorMatrix));
140 BS_Begin();
141 Skip_S1(4, "src_pix_fmt");
142 Skip_S1(4, "alpha_info");
143 BS_End();
144 Skip_B1( "reserved");
145 BS_Begin();
146 Skip_S1(6, "reserved");
147 Get_SB (luma, "custom luma quant matrix present");
148 Get_SB (chroma, "custom chroma quant matrix present");
149 BS_End();
150 if (luma)
151 Skip_XX(64, "QMatLuma");
152 if (chroma)
153 Skip_XX(64, "QMatChroma");
154 Element_End();
155 if (Element_Offset!=8+hdrSize)
156 IsOk=false;
157 for (int8u PictureNumber=0; PictureNumber<(frame_type?2:1); PictureNumber++)
158 {
159 Element_Begin1("Picture layout");
160 int16u total_slices;
161 vector<int16u> slices_size;
162 Element_Begin1("Picture header");
163 int64u pic_hdr_End, pic_data_End;
164 int32u pic_data_size;
165 int8u pic_hdr_size;
166 Get_B1 (pic_hdr_size, "pic_hdr_size");
167 if (pic_hdr_size<64)
168 {
169 Trusted_IsNot("pic_hdr_size");
170 Element_End();
171 Element_End();
172 return;
173 }
174 pic_hdr_End=Element_Offset+pic_hdr_size/8-((pic_hdr_size%8)?0:1);
175 Get_B4 (pic_data_size, "pic_data_size");
176 if (pic_data_size<8)
177 {
178 Trusted_IsNot("pic_data_size");
179 Element_End();
180 Element_End();
181 return;
182 }
183 pic_data_End=Element_Offset+pic_data_size-5;
184 Get_B2 (total_slices, "total_slices");
185 BS_Begin();
186 Skip_S1(4, "slice_width_factor");
187 Skip_S1(4, "slice_height_factor");
188 BS_End();
189 if (Element_Offset<pic_hdr_End)
190 Skip_XX(pic_hdr_End-Element_Offset, "Unknown");
191 Element_End();
192 Element_Begin1("Slice index table");
193 for (int16u Pos=0; Pos<total_slices; Pos++)
194 {
195 int16u slice_size;
196 Get_B2 (slice_size, "slice_size");
197 slices_size.push_back(slice_size);
198 }
199 Element_End();
200 for (int16u Pos=0; Pos<slices_size.size(); Pos++)
201 {
202 Skip_XX(slices_size[Pos], "slice data");
203 }
204 if (Element_Offset<pic_data_End)
205 Skip_XX(pic_data_End-Element_Offset, "Unknown");
206 Element_End();
207 }
208 bool IsZeroes=true;
209 for (size_t Pos=(size_t)Element_Offset; Pos<(size_t)Element_Size; Pos++)
210 if (Buffer[Buffer_Offset+Pos])
211 {
212 IsZeroes=false;
213 break;
214 }
215 Skip_XX(Element_Size-Element_Offset, IsZeroes?"Zeroes":"Unknown");
216
217 FILLING_BEGIN();
218 if (IsOk && Name==0x69637066 && !Status[IsAccepted]) //icpf
219 {
220 Accept();
221 Fill();
222
223 Fill(Stream_Video, 0, Video_Format_Version, __T("Version ")+Ztring::ToZtring(version));
224 Fill(Stream_Video, 0, Video_Width, frameWidth);
225 Fill(Stream_Video, 0, Video_Height, frameHeight);
226 Fill(Stream_Video, 0, Video_Encoded_Library, ProRes_creatorID(creatorID));
227 Fill(Stream_Video, 0, Video_ChromaSubsampling, ProRes_chrominance_factor(chrominance_factor));
228 Fill(Stream_Video, 0, Video_ScanType, ProRes_frame_type_ScanType(frame_type));
229 Fill(Stream_Video, 0, Video_ScanOrder, ProRes_frame_type_ScanOrder(frame_type));
230 Fill(Stream_Video, 0, Video_colour_description_present, "Yes");
231 Fill(Stream_Video, 0, Video_colour_primaries, Mpegv_colour_primaries(primaries));
232 Fill(Stream_Video, 0, Video_transfer_characteristics, Mpegv_transfer_characteristics(transf_func));
233 Fill(Stream_Video, 0, Video_matrix_coefficients, Mpegv_matrix_coefficients(colorMatrix));
234
235 Finish();
236 }
237 FILLING_END();
238}
