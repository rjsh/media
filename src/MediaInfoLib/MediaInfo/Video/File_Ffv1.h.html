101 class Context
102 {
103 public:
104 static const int32u N0; // Determined threshold to divide N and B
105 static const int32s Cmax; // Storage limitation, C never upper than 127
106 static const int32s Cmin; // Storage limitation, C never below than -128
107 int32s N; // Count where the context was encountred
108 int32s B; // Accumulated sum of corrected prediction residuals
109 int32s A; // Accumulated sum of the absolute corrected prediction residuals (St + Nt)
110 int32s C; // Correction value
111
112 Context() :
113 N(1),
114 B(0),
115 A(4),
116 C(0)
117 {}
118
119 inline void update_correlation_value_and_shift()
120 {
121 // Step 11: Resets
122 if (N == N0)
123 {
124 // divide by 2, if >= 0 : ROUND, otherwise, CEIL
125 N >>= 1;
126 A >>= 1;
127 B >>= 1;
128 }
129 ++N; // context meets
130
131 // Step 12: Bias computation procedure
132 // Keep B in (-N;0]
133 if (B <= -N)
134 {
135 if (C > Cmin)
136 --C;
137
138 B += N;
139 if (B <= -N)
140 B = -N + 1;
141 }
142 else if (B > 0)
143 {
144 if (C < Cmax)
145 ++C;
146
147 if (B > N)
148 B = 0;
149 else
150 B -= N;
151 }
152 }
153 inline int compute_k()
154 {
155 int k = 0;
156 while ((N << k) < A)
157 ++k;
158 return k;
159 }
160 };
161
162 typedef Context* ContextPtr;
163
164 ContextPtr contexts[MAX_PLANES];
165 states_context_plane plane_states[MAX_QUANT_TABLES];
166 size_t plane_states_maxsizes[MAX_QUANT_TABLES];
167
168 // HELPER
169 void contexts_init(int32u plane_count, int32u quant_table_index[MAX_PLANES], int32u context_count[MAX_QUANT_TABLES]);
170 void contexts_clean();
171};
