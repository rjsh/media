60//***************************************************************************
61// Set
62//***************************************************************************
63
64//---------------------------------------------------------------------------
65#if MEDIAINFO_DUPLICATE
66bool File_Avc::File__Duplicate_Set (const Ztring &Value)
67{
68 ZtringList List(Value);
69
70 //Searching Target
71 bool IsForUs=false;
72 std::vector<ZtringList::iterator> Targets_ToAdd;
73 std::vector<ZtringList::iterator> Targets_ToRemove;
74 std::vector<ZtringList::iterator> Orders_ToAdd;
75 std::vector<ZtringList::iterator> Orders_ToRemove;
76 for (ZtringList::iterator Current=List.begin(); Current<List.end(); ++Current)
77 {
78 //Detecting if we want to remove
79 bool ToRemove=false;
80 if (Current->find(__T('-'))==0)
81 {
82 ToRemove=true;
83 Current->erase(Current->begin());
84 }
85
86 //Managing targets
87 if (Current->find(__T("file:"))==0
88 || Current->find(__T("memory:"))==0)
89 (ToRemove?Targets_ToRemove:Targets_ToAdd).push_back(Current);
90 //Parser name
91 else if (Current->find(__T("parser=Avc"))==0)
92 IsForUs=true;
93 //Managing orders
94 else
95 (ToRemove?Orders_ToRemove:Orders_ToAdd).push_back(Current);
96 }
97
98 //For us?
99 if (!IsForUs)
100 return false;
101
102 //Configuration of initial values
103 frame_num_Old=(int32u)-1;
104 Duplicate_Buffer_Size=0;
105 SPS_PPS_AlreadyDone=false;
106 FLV=false;
107
108 //For each target to add
109 for (std::vector<ZtringList::iterator>::iterator Target=Targets_ToAdd.begin(); Target<Targets_ToAdd.end(); ++Target)
110 Writer.Configure(**Target);
111
112 //For each order to add
113 for (std::vector<ZtringList::iterator>::iterator Order=Orders_ToAdd.begin(); Order<Orders_ToAdd.end(); ++Order)
114 if ((**Order)==__T("format=Flv"))
115 FLV=true;
116
117 return true;
118}
119#endif //MEDIAINFO_DUPLICATE

125#if MEDIAINFO_DUPLICATE
126void File_Avc::File__Duplicate_Write (int64u Element_Code, int32u frame_num)
127{
128 const int8u* ToAdd=Buffer+Buffer_Offset-(size_t)Header_Size+3;
129 size_t ToAdd_Size=(size_t)(Element_Size+Header_Size-3);
130
131 if (!SPS_PPS_AlreadyDone)
132 {
133 if (Element_Code==7)
134 {
135 std::memcpy(Duplicate_Buffer, ToAdd, ToAdd_Size);
136 Duplicate_Buffer_Size=ToAdd_Size;
137
138 }
139 else if (Element_Code==8)
140 {
141
142 // Form:
143 // 8 bytes : PTS
144 // 8 bytes : DTS
145 // 8 bytes : Size (without header)
146 // 1 byte : Type (0=Frame, 1=Header);
147 // 7 bytes : Reserved
148 size_t Extra;
149 if (FLV)
150 Extra=1; //FLV
151 else
152 Extra=0; //MPEG-4
153 int8u Header[32];
154 int64u2BigEndian(Header+ 0, FrameInfo.PTS);
155 int64u2BigEndian(Header+ 8, FrameInfo.DTS);
156 int64u2BigEndian(Header+16, 5+Extra+2+Duplicate_Buffer_Size+1+2+ToAdd_Size); //5+Extra for SPS_SQS header, 2 for SPS size, 1 for PPS count, 2 for PPS size
157 Header[24]=1;
158 int56u2BigEndian(Header+25, 0);
159 Writer.Write(Header, 32);
160
161 //SPS_PPS
162 int8u* SPS_SQS=new int8u[5+Extra];
163 if (Extra==1)
164 {
165 SPS_SQS[0]=0x01; //Profile FLV
166 SPS_SQS[1]=(!seq_parameter_sets.empty() && seq_parameter_sets[0])?seq_parameter_sets[0]->profile_idc:0x00; //Compatible Profile. TODO: Handling more than 1 seq_parameter_set
167 SPS_SQS[2]=0x00; //Reserved
168 }
169 else
170 {
171 SPS_SQS[0]=(!seq_parameter_sets.empty() && seq_parameter_sets[0])?seq_parameter_sets[0]->profile_idc:0x00; //Profile MPEG-4. TODO: Handling more than 1 seq_parameter_set
172 SPS_SQS[1]=0x00; //Compatible Profile
173 }
174 SPS_SQS[2+Extra]=(!seq_parameter_sets.empty() && seq_parameter_sets[0])?seq_parameter_sets[0]->level_idc:0x00; //Level. TODO: Handling more than 1 seq_parameter_set
175 SPS_SQS[3+Extra]=0xFF; //Reserved + Size of NALU length minus 1
176 SPS_SQS[4+Extra]=0xE1; //Reserved + seq_parameter_set count
177 Writer.Write(SPS_SQS, 5+Extra);
178
179 //NALU
180 int8u NALU[2];
181 NALU[0]=((Duplicate_Buffer_Size)>> 8)&0xFF;
182 NALU[1]=((Duplicate_Buffer_Size)>> 0)&0xFF;
183 Writer.Write(NALU, 2);
184
185 //SPS
186 Writer.Write(Duplicate_Buffer, Duplicate_Buffer_Size);
187 Duplicate_Buffer_Size=0;
188
189 //PPS count
190 SPS_SQS[0]=0x01; //pic_parameter_set count
191 Writer.Write(SPS_SQS, 1);
192 delete[] SPS_SQS;
193
194 //NALU
195 NALU[0]=((ToAdd_Size)>> 8)&0xFF;
196 NALU[1]=((ToAdd_Size)>> 0)&0xFF;
197 Writer.Write(NALU, 2);
198
199 //PPS
200 Writer.Write(ToAdd, ToAdd_Size);
201
202 SPS_PPS_AlreadyDone=true;
203 }
204 }
205 else if (frame_num!=(int32u)-1)
206 {
207 if (frame_num!=frame_num_Old && frame_num_Old!=(int32u)-1 && frame_num!=(int32u)-1)
208 {
209 // Form:
210 // 8 bytes : PTS
211 // 8 bytes : DTS
212 // 8 bytes : Size (without header)
213 // 1 byte : Type (0=Frame, 1=Header);
214 // 7 bytes : Reserved
215 int8u Header[32];
216 int64u2BigEndian(Header+ 0, FrameInfo.PTS);
217 int64u2BigEndian(Header+ 8, FrameInfo.DTS);
218 int64u2BigEndian(Header+16, Duplicate_Buffer_Size);
219 Header[24]=0;
220 int56u2BigEndian(Header+25, 0);
221 Writer.Write(Header, 32);
222
223 Writer.Write(Duplicate_Buffer, Duplicate_Buffer_Size);
224 Duplicate_Buffer_Size=0;
225 }
226
227
228 //NALU
229 int32u2BigEndian(Duplicate_Buffer+Duplicate_Buffer_Size, (int32u)ToAdd_Size); //4 bytes for NALU header
230 Duplicate_Buffer_Size+=4;
231
232 //Frame (partial)
233 std::memcpy(Duplicate_Buffer+Duplicate_Buffer_Size, ToAdd, ToAdd_Size);
234 Duplicate_Buffer_Size+=ToAdd_Size;
235 frame_num_Old=frame_num;
236 }
237}
238#endif //MEDIAINFO_DUPLICATE
