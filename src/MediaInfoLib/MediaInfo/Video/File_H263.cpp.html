184//---------------------------------------------------------------------------
185bool File_H263::Synchronize()
186{
187 //Synchronizing
188 while(Buffer_Offset+3<=Buffer_Size && (Buffer[Buffer_Offset ]!=0x00
189 || Buffer[Buffer_Offset+1]!=0x00
190 || (Buffer[Buffer_Offset+2]&0xFC)!=0x80))
191 {
192 Buffer_Offset+=2;
193 while(Buffer_Offset<Buffer_Size && Buffer[Buffer_Offset]!=0x00)
194 Buffer_Offset+=2;
195 if ((Buffer_Offset<Buffer_Size && Buffer[Buffer_Offset-1]==0x00) || Buffer_Offset>=Buffer_Size)
196 Buffer_Offset--;
197 }
198
199 //Parsing last bytes if needed
200 if (Buffer_Offset+3==Buffer_Size && (Buffer[Buffer_Offset ]!=0x00
201 || Buffer[Buffer_Offset+1]!=0x00
202 || (Buffer[Buffer_Offset+2]&0xFC)!=0x80))
203 Buffer_Offset++;
204 if (Buffer_Offset+2==Buffer_Size && (Buffer[Buffer_Offset ]!=0x00
205 || Buffer[Buffer_Offset+1]!=0x00))
206 Buffer_Offset++;
207 if (Buffer_Offset+1==Buffer_Size && Buffer[Buffer_Offset ]!=0x00)
208 Buffer_Offset++;
209
210 if (Buffer_Offset+3>Buffer_Size)
211 return false;
212
213 //Synched is OK
214 Synched=true;
215 return true;
216}

218//---------------------------------------------------------------------------
219bool File_H263::Synched_Test()
220{
221 //Must have enough buffer for having header
222 if (Buffer_Offset+4>Buffer_Size)
223 return false;
224
225 //Quick test of synchro
226 if (Buffer[Buffer_Offset ]!=0x00
227 || Buffer[Buffer_Offset+1]!=0x00
228 || (Buffer[Buffer_Offset+2]&0xFC)!=0x80)
229 {
230 Synched=false;
231 return true;
232 }
233
234 //We continue
235 return true;
236}

261//---------------------------------------------------------------------------
262void File_H263::Header_Parse()
263{
264 Header_Fill_Code(0x00, "Frame");
265 Header_Parser_Fill_Size();
266}
267
268//---------------------------------------------------------------------------
269bool File_H263::Header_Parser_Fill_Size()
270{
271 //Look for next Sync word
272 if (Buffer_Offset_Temp==0) //Buffer_Offset_Temp is not 0 if Header_Parse_Fill_Size() has already parsed first frames
273 Buffer_Offset_Temp=Buffer_Offset+3;
274 while (Buffer_Offset_Temp+3<=Buffer_Size
275 && (Buffer[Buffer_Offset_Temp ]!=0x00
276 || Buffer[Buffer_Offset_Temp+1]!=0x00
277 || (Buffer[Buffer_Offset_Temp+2]&0xFC)!=0x80))
278 {
279 Buffer_Offset_Temp+=2;
280 while(Buffer_Offset_Temp<Buffer_Size && Buffer[Buffer_Offset_Temp]!=0x00)
281 Buffer_Offset_Temp+=2;
282 if (Buffer_Offset_Temp>=Buffer_Size || Buffer[Buffer_Offset_Temp-1]==0x00)
283 Buffer_Offset_Temp--;
284 }
285
286 //Must wait more data?
287 if (Buffer_Offset_Temp+3>Buffer_Size)
288 {
289 if (FrameIsAlwaysComplete || File_Offset+Buffer_Size==File_Size)
290 Buffer_Offset_Temp=Buffer_Size; //We are sure that the next bytes are a start
291 else
292 return false;
293 }
294
295 //OK, we continue
296 Header_Fill_Size(Buffer_Offset_Temp-Buffer_Offset);
297 Buffer_Offset_Temp=0;
298 return true;
299
300}

302//---------------------------------------------------------------------------
303void File_H263::Data_Parse()
304{
305 //Parsing
306 int8u Temporal_Reference_Temp;
307 BS_Begin();
308 Skip_S3(22, "Picture Start Code (PSC)");
309 Get_S1 ( 8, Temporal_Reference_Temp, "Temporal Reference (TR)");
310 if (!Temporal_Reference_IsValid)
311 {
312 Temporal_Reference=Temporal_Reference_Temp;
313 Temporal_Reference_IsValid=true;
314 }
315 else
316 Temporal_Reference++;
317 if (Temporal_Reference_Temp!=Temporal_Reference)
318 {
319 Trusted_IsNot("Out of Order");
320 Open_Buffer_Unsynch();
321 return;
322 }
323 Element_Begin1("Type Information (PTYPE)");
324 Mark_1();
325 Mark_0();
326 Skip_SB( "Split screen indicator");
327 Skip_SB( "Document camera indicator");
328 Skip_SB( "Full Picture Freeze Release");
329 Get_S1 (3, Source_Format, "Source Format"); Param_Info1(H263_Source_Format[Source_Format]);
330 if (Source_Format!=7)
331 {
332 Skip_SB( "Picture Coding Type");
333 Skip_SB( "Unrestricted Motion Vector mode");
334 Skip_SB( "Syntax-based Arithmetic Coding mode");
335 Skip_SB( "Advanced Prediction mode");
336 Skip_SB( "PB-frames mode");
337 }
338 Element_End0();
339 if (Source_Format==7) // Extended PTYPE
340 {
341 Element_Begin1("Plus PTYPE (PLUSPTYPE)");
342 int8u Ufep, PixelAspectRatioCode=0, Width=0, Height=0;
343 Get_S1 ( 3, Ufep, "Update Full Extended PTYPE (UFEP)");
344 switch (Ufep)
345 {
346 case 0 :
347 break;
348 case 1 :
349 Element_Begin1("Optional Part of PLUSPTYPE (OPPTYPE)");
350 Get_S1 (3, Source_Format, "Source Format"); Param_Info1(H263_Source_Format[Source_Format]);
351 Skip_SB( "Custom PCF");
352 Skip_SB( "Unrestricted Motion Vector (UMV) mode");
353 Skip_SB( "Syntax-based Arithmetic Coding (SAC) mode");
354 Skip_SB( "Advanced Prediction (AP) mode");
355 Skip_SB( "Advanced INTRA Coding (AIC) mode");
356 Skip_SB( "Deblocking Filter (DF) mode");
357 Skip_SB( "Slice Structured (SS) mode");
358 Skip_SB( "Reference Picture Selection (RPS) mode");
359 Skip_SB( "Independent Segment Decoding (ISD) mode");
360 Skip_SB( "Alternative INTER VLC (AIV) mode");
361 Skip_SB( "Modified Quantization (MQ) mode");
362 Mark_1();
363 Mark_0();
364 Mark_0();
365 Mark_0();
366 Element_End0();
367 break;
368 default :
369 BS_End();
370 Skip_XX(Element_Size-Element_Offset, "Unknown");
371 return; //TODO: frame count...
372 }
373 Element_Begin1("mandatory part of PLUSPTYPE when PLUSPTYPE present (MPPTYPE)");
374 Skip_S1(3, "Picture Type Code");
375 Skip_SB( "Reference Picture Resampling (RPR) mode");
376 Skip_SB( "Reduced-Resolution Update (RRU) mode");
377 Skip_SB( "Rounding Type (RTYPE)");
378 Mark_0();
379 Mark_0();
380 Mark_1();
381 Element_End0();
382 Element_End0();
383 Skip_SB( "CPM");
384 Skip_S1(2, "PSBI");
385 Element_Begin1("Custom Picture Format (CPFMT)");
386 Get_S1 (4, PixelAspectRatioCode, "Pixel Aspect Ratio Code");
387 Get_S1 (4, Width, "Picture Width Indication");
388 Width++; Width<<=2; Param_Info2(Width, " pixels");
389 Mark_1();
390 Get_S1 (4, Height, "Picture Height Indication");
391 Height<<=2; Param_Info2(Height, " pixels");
392 Element_End0();
393 if (PixelAspectRatioCode==0xF)
394 {
395 Element_Begin1("Extended Pixel Aspect Ratio (EPAR)");
396 Get_S1 (8, PAR_W, "PAR Width");
397 Get_S1 (8, PAR_H, "PAR Height");
398 Element_End0();
399 }
400 else
401 {
402 PAR_W=H263_PAR_W[PixelAspectRatioCode];
403 PAR_H=H263_PAR_H[PixelAspectRatioCode];
404 }
405 }
406 BS_End();
407 Skip_XX(Element_Size-Element_Offset, "Other data");
408
409 FILLING_BEGIN();
410 Element_Info1(Frame_Count);
411 Frame_Count++;
412
413 //Filling
414 if (!Status[IsFilled] && Frame_Count>=Frame_Count_Valid)
415 {
416 Accept("H.263");
417 Finish("H.263");
418 }
419 FILLING_END();
420}
