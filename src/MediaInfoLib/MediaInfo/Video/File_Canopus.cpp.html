54//***************************************************************************
55// Buffer - Global
56//***************************************************************************
57
58//---------------------------------------------------------------------------
59void File_Canopus::Read_Buffer_Continue()
60{
61 //Parsing
62 int32u PAR_X=0, PAR_Y=0, FieldOrder=(int32u)-1;
63 while (Element_Offset<Element_Size)
64 {
65 Element_Begin0();
66 int32u FourCC;
67 Get_C4 (FourCC, "FourCC");
68 switch (FourCC)
69 {
70 case 0x494E464F : // "INFO"
71 {
72 Element_Name("Information");
73 int32u Info_Size;
74 Get_L4 (Info_Size, "Size");
75 int64u Info_End=Element_Offset+Info_Size;
76 if (Info_Size<16 || Info_End>Element_Size)
77 {
78 Skip_XX(Element_Size-Element_Offset, "Problem");
79 Element_End0();
80 return;
81 }
82 Skip_L4( "Unknown");
83 Skip_L4( "Unknown");
84 Get_L4 (PAR_X, "PAR_X");
85 Get_L4 (PAR_Y, "PAR_Y");
86 while (Element_Offset<Info_End)
87 {
88 Element_Begin0();
89 Get_C4 (FourCC, "FourCC");
90 switch (FourCC)
91 {
92 case 0x4649454C : // "FIEL"
93 {
94 Element_Name("Field information?");
95 int32u FIEL_Size;
96 Get_L4 (FIEL_Size, "Size");
97 int64u FIEL_End=Element_Offset+FIEL_Size;
98 if (FIEL_End>Info_End)
99 {
100 Skip_XX(Info_End-Element_Offset, "Problem");
101 break;
102 }
103 if (Element_Offset<FIEL_End)
104 Get_L4(FieldOrder, "Field order");
105 while (Element_Offset<FIEL_End)
106 Skip_L4( "Unknown");
107 }
108 break;
109 case 0x52445254 : // "RDRT"
110 {
111 Element_Name("RDRT?");
112 int32u RDRT_Size;
113 Get_L4 (RDRT_Size, "Size");
114 int64u RDRT_End=Element_Offset+RDRT_Size;
115 if (RDRT_End>Info_End)
116 {
117 Skip_XX(Info_End-Element_Offset, "Problem");
118 break;
119 }
120 while (Element_Offset<RDRT_End)
121 Skip_L4( "Unknown");
122 }
123 break;
124 default: Element_Name("Unknown");
125 Skip_XX(Info_End-Element_Offset, "Unknown");
126 }
127 Element_End0();
128 }
129 }
130 break;
131 case 0x55564307 : // "UVC" 7
132 Element_Name("Data?");
133 Skip_XX(Element_Size-Element_Offset, "Unknown");
134 break;
135 default: Element_Name("Unknown");
136 Skip_XX(Element_Size-Element_Offset, "Unknown");
137 }
138 Element_End0();
139 }
140
141 FILLING_BEGIN();
142 if (!Status[IsAccepted])
143 {
144 Accept();
145 Fill();
146
147 //Info
148 if (PAR_X && PAR_Y)
149 Fill(Stream_Video, 0, Video_PixelAspectRatio, ((float32)PAR_X)/PAR_Y, 3);
150 switch (FieldOrder)
151 {
152 case 0 :
153 Fill(Stream_Video, 0, Video_ScanType, "Interlaced");
154 Fill(Stream_Video, 0, Video_ScanOrder, "TFF");
155 break;
156 case 1 :
157 Fill(Stream_Video, 0, Video_ScanType, "Interlaced");
158 Fill(Stream_Video, 0, Video_ScanOrder, "BFF");
159 break;
160 case 2 :
161 Fill(Stream_Video, 0, Video_ScanType, "Progressive");
162 break;
163 default : ;
164 }
165
166 if (Config->ParseSpeed<1.0)
167 Finish();
168 }
169 FILLING_END();
170}
