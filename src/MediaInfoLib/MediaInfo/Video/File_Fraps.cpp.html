54// Buffer - Global
55//***************************************************************************
56
57//---------------------------------------------------------------------------
58void File_Fraps::Read_Buffer_Continue()
59{
60 //Parsing
61 int8u version, flags;
62 Get_L1 (version, "version");
63 Skip_L2( "unknown");
64 Get_L1 (flags, "flags");
65 if (flags&0x40)
66 Skip_L4( "unknown");
67 switch (version)
68 {
69 case 0x00 :
70 Version0(); break;
71 case 0x01 :
72 Version1(); break;
73 case 0x02 :
74 case 0x04 :
75 Version2(); break;
76 default : Skip_XX(Element_Size-Element_Offset, "data");
77 }
78
79 Finish();
80}
81
82//---------------------------------------------------------------------------
83void File_Fraps::Version0()
84{
85 //Parsing
86 Skip_XX(Element_Size-Element_Offset, "data");
87
88 FILLING_BEGIN();
89 Accept();
90 Fill();
91 Fill(Stream_Video, 0, Video_ColorSpace, "YUV");
92 Fill(Stream_Video, 0, Video_ChromaSubsampling, "4:2:0");
93 Fill(Stream_Video, 0, Video_BitDepth, 8);
94 FILLING_END();
95}
96
97//---------------------------------------------------------------------------
98void File_Fraps::Version1()
99{
100 //Parsing
101 Skip_XX(Element_Size-Element_Offset, "data");
102
103 FILLING_BEGIN();
104 Accept();
105 Fill();
106 Fill(Stream_Video, 0, Video_ColorSpace, "RGB");
107 Fill(Stream_Video, 0, Video_BitDepth, 8);
108 FILLING_END();
109}
110
111//---------------------------------------------------------------------------
112void File_Fraps::Version2()
113{
114 //Parsing
115 if (Element_Size>8) //Else this is a repeat frame
116 {
117 Skip_C4( "FPSx");
118 Skip_L4( "offset to the Y plane (minus 8)");
119 Skip_L4( "offset to the U plane (minus 8)");
120 Skip_L4( "offset to the V plane (minus 8)");
121 Skip_XX(Element_Size-Element_Offset, "data");
122 }
123
124 FILLING_BEGIN();
125 Accept();
126 Fill();
127 Fill(Stream_Video, 0, Video_ColorSpace, "YUV");
128 Fill(Stream_Video, 0, Video_BitDepth, 8);
129 FILLING_END();
130}
