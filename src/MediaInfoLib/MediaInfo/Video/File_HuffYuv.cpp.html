45const char* HuffYUV_ColorSpace (int16u BitCount)
46{
47 switch (BitCount&0xFFF8)
48 {
49 case 8 :
50 case 16 : return "YUV";
51 case 24 : return "RGB";
52 case 32 : return "RGBA";
53 default : return "";
54 }
55}
56
57const char* HuffYUV_ChromaSubsampling (int16u BitCount)
58{
59 switch (BitCount&0xFFF8)
60 {
61 case 8 : return "4:2:0";
62 case 16 : return "4:2:2";
63 default : return "";
64 }
65}
66
67const string HuffYUV_ColorSpace(bool rgb, bool chroma, bool alpha)
68{
69 string ToReturn;
70
71 if (rgb)
72 {
73 ToReturn="RGB";
74 }
75 else
76 {
77 ToReturn=chroma?"YUV":"Y";
78 }
79
80 if (alpha)
81 ToReturn+='A';
82
83 return ToReturn;
84}
85
86const char* HuffYUV_method (int16u BitCount, int8u method)
87{
88 switch (BitCount&0x0007)
89 {
90 case 0 :
91 switch (method)
92 {
93 case 0xFE : return "old";
94 case 0 : return "left";
95 case 1 : return "gradient";
96 case 2 : return "median";
97 case 64 : return "left with decorrelation";
98 case 65 : return "gradient with decorrelation";
99 default : return "";
100 }
101 case 1 : return "left";
102 case 2 : return "left with decorrelation";
103 case 3 :
104 switch (BitCount&0xFFF8)
105 {
106 case 16 : return "gradient";
107 case 24 :
108 case 32 : return "gradient with decorrelation";
109 default : return "";
110 }
111 case 4 : return "median";
112 default : return "";
113 }
114}

148//***************************************************************************
149// Buffer - Global
150//***************************************************************************
151
152//---------------------------------------------------------------------------
153void File_HuffYuv::Read_Buffer_Continue()
154{
155 Accept();
156
157 if (IsOutOfBandData)
158 {
159 FrameHeader();
160 }
161 else if (Frame_Count==0)
162 Fill(Stream_Video, 0, Video_Format_Version, "Version 1");
163
164 Skip_XX(Element_Size-Element_Offset, "Other data");
165
166 Frame_Count++;
167 Finish();
168}

170//***************************************************************************
171// Elements
172//***************************************************************************
173
174//---------------------------------------------------------------------------
175void File_HuffYuv::FrameHeader()
176{
177 //Version
178 if (Element_Size<4)
179 {
180 Reject();
181 return;
182 }
183 int8u Version;
184 if (Buffer[3]==0)
185 {
186 Version=Element_Size>4?2:1;
187 }
188 else
189 Version=3;
190
191 //Parsing
192 int8u bpp_override=0, chroma_v_shift=0, chroma_h_shift=0, interlace;
193 bool alpha=false, chroma=false, rgb=false;
194 Element_Begin1("method");
195 BS_Begin();
196 Skip_SB( "unknown");
197 Skip_SB( "decorrelate");
198 Skip_S1(6, "predictor");
199 Element_End0();
200 if (Version<=2)
201 {
202 Get_S1 (8, bpp_override, "bpp_override");
203 }
204 else
205 {
206 Get_S1 (4, bpp_override, "bit_depth"); Param_Info2(bpp_override+1, "bits");
207 Get_S1 (2, chroma_v_shift, "chroma_v_shift");
208 Get_S1 (2, chroma_h_shift, "chroma_h_shift");
209 }
210 Skip_SB( "unknown");
211 Skip_SB( "context");
212 Get_S1 (2, interlace, "interlace");
213 if (Version<=2)
214 {
215 Skip_S1(4, "unknown");
216 Skip_S1(8, "zero");
217 }
218 else
219 {
220 Skip_SB( "unknown");
221 Get_SB ( alpha, "alpha");
222 Get_SB ( rgb, "rgb");
223 if (rgb)
224 Skip_SB( "unused");
225 else
226 Get_SB ( chroma, "chroma");
227 Skip_S1(7, "unused");
228 Skip_SB( "version 3+ indicator");
229 }
230 BS_End();
231
232 if (Frame_Count==0)
233 {
234 if (Version==2)
235 {
236 //BiCount;
237 if (bpp_override)
238 BitCount=bpp_override;
239 Fill(Stream_Video, 0, Video_BitDepth, 8);
240 }
241 else
242 {
243 Fill(Stream_Video, 0, Video_BitDepth, bpp_override+1);
244 }
245
246 Fill(Stream_Video, 0, Video_Format_Version, __T("Version ")+Ztring::ToZtring(Version));
247
248 if (Version==2)
249 {
250 Fill(Stream_Video, 0, Video_ColorSpace, HuffYUV_ColorSpace(BitCount));
251 Fill(Stream_Video, 0, Video_ChromaSubsampling, HuffYUV_ChromaSubsampling(BitCount));
252 }
253 else
254 {
255 Fill(Stream_Video, 0, Video_ColorSpace, HuffYUV_ColorSpace(rgb, chroma, alpha));
256 string ChromaSubsampling;
257 if (chroma)
258 {
259 switch (chroma_h_shift)
260 {
261 case 0 :
262 switch (chroma_v_shift)
263 {
264 case 0 : ChromaSubsampling="4:4:4"; break;
265 default: ;
266 }
267 break;
268 case 1 :
269 switch (chroma_v_shift)
270 {
271 case 0 : ChromaSubsampling="4:2:2"; break;
272 case 1 : ChromaSubsampling="4:2:0"; break;
273 default: ;
274 }
275 break;
276 case 2 :
277 switch (chroma_v_shift)
278 {
279 case 0 : ChromaSubsampling="4:1:1"; break;
280 case 1 : ChromaSubsampling="4:1:0"; break;
281 case 2 : ChromaSubsampling="4:1:0 (4x4)"; break;
282 default: ;
283 }
284 break;
285 default: ;
286 }
287 }
288 if (!ChromaSubsampling.empty() && alpha)
289 ChromaSubsampling+=":4";
290 Fill(Stream_Video, 0, Video_ChromaSubsampling, ChromaSubsampling);
291 }
292
293 switch (interlace)
294 {
295 case 0 :
296 if (Version<=2 && Height)
297 Fill(Stream_Video, 0, Video_ScanType, Height>288?"Interlaced":"Progressive");
298 break;
299 case 1 :
300 Fill(Stream_Video, 0, Video_ScanType, "Interlaced");
301 break;
302 case 2 :
303 Fill(Stream_Video, 0, Video_ScanType, "Progressive");
304 break;
305 default:;
306 }
307 }
308}
