132//---------------------------------------------------------------------------
133File_AfdBarData::File_AfdBarData()
134:File__Analyze()
135{
136 //Configuration
137 PTS_DTS_Needed=true;
138
139 //In
140 Format=Format_Unknown;
141 aspect_ratio_FromContainer=(int8u)-1;
142}

144//***************************************************************************
145// Streams management
146//***************************************************************************
147
148//---------------------------------------------------------------------------
149void File_AfdBarData::Streams_Fill()
150{
151 //Filling
152 Stream_Prepare(Stream_Video);
153 if (active_format!=(int8u)-1)
154 {
155 Fill(Stream_Video, 0, Video_ActiveFormatDescription, Stream.active_format);
156 if (aspect_ratio==(int8u)-1)
157 aspect_ratio=aspect_ratio_FromContainer;
158 if (aspect_ratio!=(int8u)-1)
159 {
160 Fill(Stream_Video, 0, Video_ActiveFormatDescription_String, Stream.aspect_ratio?AfdBarData_active_format_16_9[Stream.active_format]:AfdBarData_active_format_4_3[Stream.active_format]);
161 switch (Format)
162 {
163 case Format_A53_4_DTG1 : Fill(Stream_Video, 0, Video_ActiveFormatDescription_MuxingMode, "A/53"); break;
164 case Format_S2016_3 : Fill(Stream_Video, 0, Video_ActiveFormatDescription_MuxingMode, "SMPTE ST 2016-3"); break;
165 default :
166 Skip_XX(Element_Size, "Unknown");
167 return;
168 }
169 }
170 }
171}

173//***************************************************************************
174// Buffer - Global
175//***************************************************************************
176
177//---------------------------------------------------------------------------
178void File_AfdBarData::Read_Buffer_Continue()
179{
180 //Default
181 line_number_end_of_top_bar=(int16u)-1;
182 line_number_start_of_bottom_bar=(int16u)-1;
183 pixel_number_end_of_left_bar=(int16u)-1;
184 pixel_number_start_of_right_bar=(int16u)-1;
185 active_format=(int8u)-1;
186 aspect_ratio=(int8u)-1;
187
188 //Parsing
189 switch (Format)
190 {
191 case Format_A53_4_DTG1 :
192 afd_data();
193 break;
194 case Format_A53_4_GA94_06 :
195 bar_data();
196 break;
197 case Format_S2016_3 :
198 afd_data();
199 Skip_B1( "Reserved");
200 Skip_B1( "Reserved");
201 bar_data();
202 break;
203 default :
204 Skip_XX(Element_Size, "Unknown");
205 return;
206 }
207
208 FILLING_BEGIN();
209 //Filling
210 Stream.line_number_end_of_top_bar=line_number_end_of_top_bar;
211 Stream.line_number_start_of_bottom_bar=line_number_start_of_bottom_bar;
212 Stream.pixel_number_end_of_left_bar=pixel_number_end_of_left_bar;
213 Stream.pixel_number_start_of_right_bar=pixel_number_start_of_right_bar;
214 Stream.active_format=active_format;
215 Stream.aspect_ratio=aspect_ratio;
216
217 if (!Status[IsAccepted])
218 {
219 Accept("AfdBarData");
220 Fill("AfdBarData");
221 }
222 if (MediaInfoLib::Config.ParseSpeed_Get()<1)
223 Finish("AfdBarData");
224 FILLING_END();
225}

231//---------------------------------------------------------------------------
232void File_AfdBarData::afd_data()
233{
234 //Parsing
235 Element_Begin1("Active Format Description");
236 BS_Begin();
237 if (Format==Format_S2016_3)
238 {
239 Mark_0_NoTrustError();
240 Get_S1 (4, active_format, "active_format"); Param_Info1(AfdBarData_active_format[active_format]);
241 Get_S1 (1, aspect_ratio, "aspect_ratio"); Param_Info1(AfdBarData_aspect_ratio[aspect_ratio]);
242 Mark_0_NoTrustError();
243 Mark_0_NoTrustError();
244 }
245 else
246 {
247 bool active_format_flag;
248 Mark_0();
249 Get_SB (active_format_flag, "active_format_flag");
250 Mark_0_NoTrustError();
251 Mark_0_NoTrustError();
252 Mark_0_NoTrustError();
253 Mark_0_NoTrustError();
254 Mark_0_NoTrustError();
255 Mark_1_NoTrustError();
256 if (active_format_flag)
257 {
258 Mark_1_NoTrustError();
259 Mark_1_NoTrustError();
260 Mark_1_NoTrustError();
261 Mark_1_NoTrustError();
262 Get_S1 (4, active_format, "active_format"); Param_Info1(AfdBarData_active_format[active_format]);
263 }
264 }
265 BS_End();
266 Element_End0();
267}
268
269//---------------------------------------------------------------------------
270void File_AfdBarData::bar_data()
271{
272 //Parsing
273 Element_Begin1("bar_data");
274 bool top_bar_flag, bottom_bar_flag, left_bar_flag, right_bar_flag;
275 BS_Begin();
276 Get_SB (top_bar_flag, "top_bar_flag");
277 Get_SB (bottom_bar_flag, "bottom_bar_flag");
278 Get_SB (left_bar_flag, "left_bar_flag");
279 Get_SB (right_bar_flag, "right_bar_flag");
280 if (Format==Format_S2016_3)
281 {
282 Mark_0_NoTrustError();
283 Mark_0_NoTrustError();
284 Mark_0_NoTrustError();
285 Mark_0_NoTrustError();
286 }
287 else
288 {
289 Mark_1_NoTrustError();
290 Mark_1_NoTrustError();
291 Mark_1_NoTrustError();
292 Mark_1_NoTrustError();
293 }
294 if (top_bar_flag)
295 {
296 Mark_1();
297 Mark_1();
298 Get_S2 (14, line_number_end_of_top_bar, "line_number_end_of_top_bar");
299 }
300 if (bottom_bar_flag)
301 {
302 Mark_1();
303 Mark_1();
304 Get_S2 (14, line_number_start_of_bottom_bar, "line_number_start_of_bottom_bar");
305 }
306 if (left_bar_flag)
307 {
308 Mark_1();
309 Mark_1();
310 Get_S2 (14, pixel_number_end_of_left_bar, "pixel_number_end_of_left_bar");
311 }
312 if (right_bar_flag)
313 {
314 Mark_1();
315 Mark_1();
316 Get_S2 (14, pixel_number_start_of_right_bar, "pixel_number_start_of_right_bar");
317 }
318 if (!top_bar_flag && !bottom_bar_flag && !left_bar_flag && !right_bar_flag)
319 {
320 Mark_1_NoTrustError();
321 Mark_1_NoTrustError();
322 Skip_S2(14, "reserved");
323 Mark_1_NoTrustError();
324 Mark_1_NoTrustError();
325 Skip_S2(14, "reserved");
326 }
327 BS_End();
328 Element_End0();
329
330 if (Format==Format_A53_4_DTG1)
331 {
332 BS_Begin();
333 Mark_1();
334 Mark_1();
335 Mark_1();
336 Mark_1();
337 Mark_1();
338 Mark_1();
339 Mark_1();
340 Mark_1();
341 BS_End();
342
343 if (Element_Size-Element_Offset)
344 Skip_XX(Element_Size-Element_Offset, "additional_bar_data");
345 }
346}
