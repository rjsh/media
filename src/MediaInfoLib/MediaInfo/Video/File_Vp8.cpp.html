92//---------------------------------------------------------------------------
93void File_Vp8::Read_Buffer_Continue()
94{
95 Accept();
96
97 BS_Begin_LE(); //VP8 bitstream is Little Endian
98 bool frame_type;
99 Get_TB ( frame_type, "frame type");
100 Skip_T1( 3, "version number");
101 Skip_TB( "show_frame flag");
102 Skip_T4(19, "size of the first data partition");
103 BS_End();
104
105 if (!frame_type) //I-Frame
106 {
107 Skip_B3( "0x9D012A");
108 Skip_L2( "Width");
109 Skip_L2( "Height");
110 }
111 Skip_XX(Element_Size-Element_Offset, "Other data");
112
113 Frame_Count++;
114 if (Frame_Count>=Frame_Count_Valid)
115 Finish();
116}
