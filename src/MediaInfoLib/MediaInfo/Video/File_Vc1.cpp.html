269//---------------------------------------------------------------------------
270void File_Vc1::Streams_Fill()
271{
272 //Calculating - PixelAspectRatio
273 float32 PixelAspectRatio;
274 if (AspectRatio!=0x0F)
275 PixelAspectRatio=Vc1_PixelAspectRatio[AspectRatio];
276 else if (AspectRatioY)
277 PixelAspectRatio=((float)AspectRatioX)/((float)AspectRatioY);
278 else
279 PixelAspectRatio=1; //Unknown
280
281 Ztring Profile;
282 if (profile!=(int8u)-1)
283 Profile=Vc1_Profile[profile];
284 if (profile==3 && level!=(int8u)-1)
285 Profile+=__T("@L")+Ztring::ToZtring(level);
286 Fill(Stream_Video, 0, Video_Format_Profile, Profile);
287 Fill(Stream_Video, 0, Video_Codec_Profile, Profile);
288 Fill(Stream_Video, 0, Video_ColorSpace, "YUV");
289 Fill(Stream_Video, 0, Video_Colorimetry, Vc1_ColorimetryFormat[colordiff_format]);
290 if (coded_width && coded_height)
291 {
292 Fill(Stream_Video, StreamPos_Last, Video_Width, (coded_width+1)*2);
293 Fill(Stream_Video, StreamPos_Last, Video_Height, (coded_height+1)*2);
294 }
295 if (PixelAspectRatio!=0)
296 Fill(Stream_Video, 0, Video_PixelAspectRatio, PixelAspectRatio, 3, true);
297 if (FrameRate!=0)
298 Fill(Stream_Video, StreamPos_Last, Video_FrameRate, FrameRate);
299
300 //Interlacement
301 if (!interlace || (PictureFormat_Count[1]==0 && PictureFormat_Count[2]==0)) //No interlaced frame/field
302 {
303 Fill(Stream_Video, 0, Video_ScanType, "Progressive");
304 Fill(Stream_Video, 0, Video_Interlacement, "PPF");
305 }
306 else if (PictureFormat_Count[0]>0) //Interlaced and non interlaced frames/fields
307 {
308 Fill(Stream_Video, 0, Video_ScanType, "Mixed");
309 Fill(Stream_Video, 0, Video_Interlacement, "Mixed");
310 }
311 else
312 {
313 Fill(Stream_Video, 0, Video_ScanType, "Interlaced");
314 Fill(Stream_Video, 0, Video_Interlacement, "Interlaced");
315 }
316 if (Frame_Count>0 && interlace)
317 Fill(Stream_Video, 0, Video_ScanOrder, Interlaced_Bottom?"BFF":"TFF");
318 std::string TempRef;
319 for (std::map<int16u, temporalreference>::iterator Temp=TemporalReference.begin(); Temp!=TemporalReference.end(); ++Temp)
320 {
321 TempRef+=Temp->second.top_field_first?"T":"B";
322 TempRef+=Temp->second.repeat_first_field?"3":"2";
323 }
324 if (TempRef.find('3')!=std::string::npos) //A pulldown maybe is detected
325 {
326 if (TempRef.find("T2T3B2B3T2T3B2B3")!=std::string::npos
327 || TempRef.find("B2B3T2T3B2B3T2T3")!=std::string::npos)
328 {
329 Fill(Stream_Video, 0, Video_ScanOrder, "2:3 Pulldown", Unlimited, true, true);
330 Fill(Stream_Video, 0, Video_FrameRate, FrameRate*24/30, 3, true); //Real framerate
331 Fill(Stream_Video, 0, Video_ScanType, "Progressive", Unlimited, true, true);
332 Fill(Stream_Video, 0, Video_Interlacement, "PPF", Unlimited, true, true);
333 }
334 if (TempRef.find("T2T2T2T2T2T2T2T2T2T2T2T3B2B2B2B2B2B2B2B2B2B2B2B3")!=std::string::npos
335 || TempRef.find("B2B2B2B2B2B2B2B2B2B2B2B3T2T2T2T2T2T2T2T2T2T2T2T3")!=std::string::npos)
336 {
337 Fill(Stream_Video, 0, Video_ScanOrder, "2:2:2:2:2:2:2:2:2:2:2:3 Pulldown", Unlimited, true, true);
338 Fill(Stream_Video, 0, Video_FrameRate, FrameRate*24/25, 3, true); //Real framerate
339 Fill(Stream_Video, 0, Video_ScanType, "Progressive", Unlimited, true, true);
340 Fill(Stream_Video, 0, Video_Interlacement, "PPF", Unlimited, true, true);
341 }
342 }
343
344 //Buffer
345 for (size_t Pos=0; Pos<hrd_buffers.size(); Pos++)
346 Fill(Stream_Video, 0, Video_BufferSize, hrd_buffers[Pos]);
347}

349//---------------------------------------------------------------------------
350void File_Vc1::Streams_Finish()
351{
352 if (PTS_End>PTS_Begin)
353 Fill(Stream_Video, 0, Video_Duration, float64_int64s(((float64)(PTS_End-PTS_Begin))/1000000));
354
355 #if MEDIAINFO_IBIUSAGE
356 int64u Numerator=0, Denominator=0;
357 if (framerate_present)
358 {
359 if (framerate_form)
360 {
361 Numerator=framerateexp+1;
362 Denominator=32;
363 }
364 else if (Vc1_FrameRate_dr(frameratecode_dr))
365 {
366 Numerator=(int64u)Vc1_FrameRate_enr(frameratecode_enr);
367 Denominator=(int64u)Vc1_FrameRate_dr(frameratecode_dr);
368 }
369 }
370 if (Numerator)
371 Ibi_Stream_Finish(Numerator, Denominator);
372 #endif //MEDIAINFO_IBIUSAGE
373}

400//---------------------------------------------------------------------------
401bool File_Vc1::Synched_Test()
402{
403 //Must have enough buffer for having header
404 if (Buffer_Offset+4>Buffer_Size)
405 return false;
406
407 //Quick test of synchro
408 if (Buffer[Buffer_Offset ]!=0x00
409 || Buffer[Buffer_Offset+1]!=0x00
410 || Buffer[Buffer_Offset+2]!=0x01)
411 {
412 Synched=false;
413 return true;
414 }
415
416 //Quick search
417 if (!Header_Parser_QuickSearch())
418 return false;
419
420 #if MEDIAINFO_IBIUSAGE
421 bool RandomAccess=Buffer[Buffer_Offset+3]==0x0F; //SequenceHeader
422 if (RandomAccess)
423 Ibi_Add();
424 #endif //MEDIAINFO_IBIUSAGE
425
426 //We continue
427 return true;
428}

482//---------------------------------------------------------------------------
483#if MEDIAINFO_DEMUX
484bool File_Vc1::Demux_UnpacketizeContainer_Test()
485{
486 if ((Demux_IntermediateItemFound && Buffer[Buffer_Offset+3]==0x0D) || Buffer[Buffer_Offset+3]==0x0F)
487 {
488 if (Demux_Offset==0)
489 {
490 Demux_Offset=Buffer_Offset;
491 Demux_IntermediateItemFound=false;
492 }
493 while (Demux_Offset+4<=Buffer_Size)
494 {
495 //Synchronizing
496 while(Demux_Offset+3<=Buffer_Size && (Buffer[Demux_Offset ]!=0x00
497 || Buffer[Demux_Offset+1]!=0x00
498 || Buffer[Demux_Offset+2]!=0x01))
499 {
500 Demux_Offset+=2;
501 while(Demux_Offset<Buffer_Size && Buffer[Buffer_Offset]!=0x00)
502 Demux_Offset+=2;
503 if (Demux_Offset>=Buffer_Size || Buffer[Demux_Offset-1]==0x00)
504 Demux_Offset--;
505 }
506
507 if (Demux_Offset+4<=Buffer_Size)
508 {
509 if (Demux_IntermediateItemFound)
510 {
511 bool MustBreak;
512 switch (Buffer[Demux_Offset+3])
513 {
514 case 0x0D :
515 case 0x0F :
516 MustBreak=true; break;
517 default :
518 Demux_Offset+=3;
519 MustBreak=false;
520 }
521 if (MustBreak)
522 break; //while() loop
523 }
524 else
525 {
526 if (Buffer[Demux_Offset+3]==0x0D)
527 Demux_IntermediateItemFound=true;
528 }
529 }
530 Demux_Offset++;
531 }
532
533 if (Demux_Offset+4>Buffer_Size && File_Offset+Buffer_Size!=File_Size)
534 return false; //No complete frame
535
536 if (!Status[IsAccepted])
537 {
538 Accept("VC-1");
539 if (Config->Demux_EventWasSent)
540 return false;
541 }
542
543 //Demux
544 #if MEDIAINFO_DEMUX
545 if (InitData_Buffer_Size && Buffer[Buffer_Offset+3]==0x0F) //First SequenceHeader
546 {
547 //Searching begin of frame (after SequenceHeader/EntryPointHeader)
548 size_t Header_End=4;
549 for (; Header_End<Demux_Offset; Header_End++)
550 if (Buffer[Header_End ]==0x00
551 && Buffer[Header_End+1]==0x00
552 && Buffer[Header_End+2]==0x01
553 && Buffer[Header_End+2]==0x0D)
554 break;
555
556 switch (Config->Demux_InitData_Get())
557 {
558 case 0 : //In demux event
559 break; //Will be done in the first demux event
560 case 1 : //In field
561 {
562 std::string Data_Raw((const char*)(Buffer+Buffer_Offset), (size_t)(Header_End-Buffer_Offset));
563 std::string Data_Base64(Base64::encode(Data_Raw));
564 Fill(Stream_Video, StreamPos_Last, "Demux_InitBytes", Data_Base64);
565 (*Stream_More)[Stream_Video][StreamPos_Last](Ztring().From_Local("Demux_InitBytes"), Info_Options)=__T("N NT");
566 }
567 break;
568 default : ;
569 }
570
571 delete[] InitData_Buffer; InitData_Buffer=NULL;
572 InitData_Buffer_Size=0;
573 }
574 #endif //MEDIAINFO_DEMUX
575
576 Demux_UnpacketizeContainer_Demux(Buffer[Buffer_Offset+3]==0x0F);
577 }
578
579 return true;
580}
581#endif //MEDIAINFO_DEMUX

600//---------------------------------------------------------------------------
601void File_Vc1::Header_Parse()
602{
603 //Specific
604 if (From_WMV3 || Only_0D)
605 {
606 Header_Fill_Size(Buffer_Size);
607 Header_Fill_Code(From_WMV3?0x0F:0x0D, Ztring().From_CC1(From_WMV3?0x0F:0x0D));
608 return;
609 }
610
611 //Parsing
612 Skip_B3( "synchro");
613 Get_B1 (start_code, "start_code");
614 if (!Header_Parser_Fill_Size())
615 {
616 Element_WaitForMoreData();
617 return;
618 }
619
620 //Filling
621 Header_Fill_Code(start_code, Ztring().From_CC1(start_code));
622}

624//---------------------------------------------------------------------------
625void File_Vc1::Data_Parse()
626{
627 //Parse
628 switch (Element_Code)
629 {
630 case 0x0A: EndOfSequence(); break;
631 case 0x0B: Slice(); break;
632 case 0x0C: Field(); break;
633 case 0x0D: FrameHeader(); break;
634 case 0x0E: EntryPointHeader(); break;
635 case 0x0F: SequenceHeader(); break;
636 case 0x1B: UserDefinedSlice(); break;
637 case 0x1C: UserDefinedField(); break;
638 case 0x1D: UserDefinedFrameHeader(); break;
639 case 0x1E: UserDefinedEntryPointHeader(); break;
640 case 0x1F: UserDefinedSequenceHeader(); break;
641 default:
642 Trusted_IsNot("Unattended element!");
643 }
644}
645
646//---------------------------------------------------------------------------
647bool File_Vc1::Header_Parser_Fill_Size()
648{
649 //Look for next Sync word
650 if (Buffer_Offset_Temp==0) //Buffer_Offset_Temp is not 0 if Header_Parse_Fill_Size() has already parsed first frames
651 Buffer_Offset_Temp=Buffer_Offset+4;
652 while (Buffer_Offset_Temp+4<=Buffer_Size
653 && CC3(Buffer+Buffer_Offset_Temp)!=0x000001)
654 {
655 Buffer_Offset_Temp+=2;
656 while(Buffer_Offset_Temp<Buffer_Size && Buffer[Buffer_Offset_Temp]!=0x00)
657 Buffer_Offset_Temp+=2;
658 if (Buffer_Offset_Temp>=Buffer_Size || Buffer[Buffer_Offset_Temp-1]==0x00)
659 Buffer_Offset_Temp--;
660
661 if (start_code==0x0D) //FrameHeader, we need only few bytes
662 {
663 if (Buffer_Offset_Temp-Buffer_Offset>20)
664 {
665 //OK, we continue, we have enough for a slice
666 Header_Fill_Size(16);
667 Buffer_Offset_Temp=0;
668 return true;
669 }
670 }
671 }
672
673 //Must wait more data?
674 if (Buffer_Offset_Temp+4>Buffer_Size)
675 {
676 if (FrameIsAlwaysComplete || File_Offset+Buffer_Size==File_Size)
677 Buffer_Offset_Temp=Buffer_Size; //We are sure that the next bytes are a start
678 else
679 return false;
680 }
681
682 //OK, we continue
683 Header_Fill_Size(Buffer_Offset_Temp-Buffer_Offset);
684 Buffer_Offset_Temp=0;
685 return true;
686}
687
688//---------------------------------------------------------------------------
689bool File_Vc1::Header_Parser_QuickSearch()
690{
691 while ( Buffer_Offset+4<=Buffer_Size
692 && Buffer[Buffer_Offset ]==0x00
693 && Buffer[Buffer_Offset+1]==0x00
694 && Buffer[Buffer_Offset+2]==0x01)
695 {
696 //Getting start_code
697 int8u start_code=CC1(Buffer+Buffer_Offset+3);
698
699 //Searching start
700 if (Streams[start_code].Searching_Payload)
701 return true;
702
703 //Synchronizing
704 Buffer_Offset+=4;
705 Synched=false;
706 if (!Synchronize())
707 {
708 UnSynched_IsNotJunk=true;
709 return false;
710 }
711
712 if (Buffer_Offset+4>Buffer_Size)
713 {
714 UnSynched_IsNotJunk=true;
715 return false;
716 }
717 }
718
719 if (Buffer_Offset+3==Buffer_Size)
720 return false; //Sync is OK, but start_code is not available
721 Trusted_IsNot("VC-1, Synchronisation lost");
722 return Synchronize();
723}

750//---------------------------------------------------------------------------
751// Packet "0D"
752void File_Vc1::FrameHeader()
753{
754 //Name
755 Element_Name("FrameHeader");
756 Element_Info1(Ztring(__T("Frame ")+Ztring::ToZtring(Frame_Count)));
757 if (FrameRate)
758 {
759 Element_Info1C((FrameInfo.PTS!=(int64u)-1), __T("PTS ")+Ztring().Duration_From_Milliseconds(float64_int64s(((float64)FrameInfo.PTS)/1000000+Frame_Count_InThisBlock*1000/FrameRate)));
760 Element_Info1C((FrameInfo.DTS!=(int64u)-1), __T("DTS ")+Ztring().Duration_From_Milliseconds(float64_int64s(((float64)FrameInfo.DTS)/1000000)));
761 }
762
763 //Counting
764 if (File_Offset+Buffer_Offset+Element_Size==File_Size)
765 Frame_Count_Valid=Frame_Count; //Finish frames in case of there are less than Frame_Count_Valid frames
766 Frame_Count++;
767 Frame_Count_InThisBlock++;
768 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
769 Frame_Count_NotParsedIncluded++;
770
771 //Parsing
772 BS_Begin();
773 int8u ptype=(int8u)-1;
774 if (profile==3) //Advanced
775 {
776 int8u PictureFormat=0; //Default=Progressive frame
777 if (interlace)
778 {
779 bool fcm_1;
780 Get_SB ( fcm_1, "fcm_1");
781 if (fcm_1)
782 {
783 bool fcm_2;
784 Get_SB ( fcm_2, "fcm_2");
785 PictureFormat=fcm_2?2:1; //Interlaced Field : Interlaced Frame
786 }
787 }
788 Param_Info1(Vc1_PictureFormat[PictureFormat]);
789 PictureFormat_Count[PictureFormat]++;
790
791 if (PictureFormat==2) //Interlaced Field
792 {
793 int8u ptype_;
794 Get_S1 ( 3, ptype_, "ptype");
795 if (ptype_<5)
796 {
797 Param_Info1(Vc1_Type[Vc1_FieldTypeTable[ptype_][0]]); Element_Info1(Vc1_Type[Vc1_FieldTypeTable[ptype_][0]]); //First field
798 Param_Info1(Vc1_Type[Vc1_FieldTypeTable[ptype_][1]]); Element_Info1(Vc1_Type[Vc1_FieldTypeTable[ptype_][1]]); //Second field
799 ptype=Vc1_FieldTypeTable[ptype_][0]; //Saving the ptype from the first field
800 }
801 else
802 {
803 Trusted_IsNot("ptype is out of range");
804 ptype=0; //Error
805 }
806 }
807 else
808 {
809 size_t ptype_;
810 Get_VL (Vc1_ptype, ptype_, "ptype"); if (ptype_<5) {Param_Info1(Vc1_Type[Vc1_ptype[ptype_].mapped_to3]); Element_Info1(Vc1_Type[Vc1_ptype[ptype_].mapped_to3]);}
811 ptype=(int8u)Vc1_ptype[ptype_].mapped_to3;
812 }
813 if (RefFramesCount<2 && (ptype==0 || ptype==1))
814 RefFramesCount++;
815 if (FrameInfo.DTS!=(int64u)-1)
816 {
817 if (framerate_present)
818 FrameInfo.DTS+=float64_int64s(((float64)1000000000)/FrameRate);
819 }
820 if (FrameInfo.PTS!=(int64u)-1)
821 {
822 if (PTS_Begin==(int64u)-1 && ptype==0) //IFrame
823 PTS_Begin=FrameInfo.PTS;
824 if ((ptype==0 || ptype==1) && Frame_Count_InThisBlock<=1) //IFrame or PFrame
825 PTS_End=FrameInfo.PTS;
826 if ((ptype==0 || ptype==1) || (Frame_Count_InThisBlock>=2 && RefFramesCount>=2)) //IFrame or PFrame or more than 2 RefFrame for BFrames
827 {
828 if (framerate_present)
829 PTS_End+=float64_int64s(((float64)1000000000)/FrameRate);
830 }
831 }
832
833 if (ptype!=4) //!=Skipping
834 {
835 if (tfcntrflag)
836 {
837 Skip_S1( 8, "tfcntr - frame counter");
838 }
839 }
840
841 if (interlace && !psf)
842 {
843 bool tff=true, rff=false;
844 if (pulldown)
845 {
846 Get_SB (tff, "tff - top field first");
847 Get_SB (rff, "rff - repeat first field");
848 if (tff)
849 Interlaced_Top++;
850 else
851 Interlaced_Bottom++;
852
853 if (TemporalReference.size()<30)
854 {
855 if (ptype!=2 && ptype!=3 //if not B and BI-frame
856 && !TemporalReference_Waiting.empty()) //We must have 2 I or P pictures to be sure not having B picture later
857 {
858 //We have 2 I or P pictures
859 for (size_t Pos=1; Pos<TemporalReference_Waiting.size(); Pos++) //All B frames (not the first frame, which is I or P)
860 {
861 TemporalReference_Offset++;
862 TemporalReference[TemporalReference_Offset]=TemporalReference_Waiting[Pos];
863 }
864 TemporalReference_Offset++;
865 TemporalReference[TemporalReference_Offset]=TemporalReference_Waiting[0];
866 TemporalReference_Waiting.clear();
867 }
868
869 //We must wait for having another I or P picture
870 temporalreference Temp;
871 Temp.top_field_first=tff;
872 Temp.repeat_first_field=rff;
873 TemporalReference_Waiting.push_back(Temp);
874 }
875 }
876 }
877 else
878 {
879 int8u rptfrm=0;
880 if (pulldown)
881 {
882 Get_S1 ( 2, rptfrm, "rptfrm - repeate frame");
883 }
884 }
885
886 /*
887 if (panscan_flag)
888 {
889 //TODO
890 }
891
892 if (ptype!=4) //!=Skipping
893 {
894 bool rndctrl;
895 Get_SB( rndctrl, "rndctrl - rounding control");
896 if (rndctrl && (ptype==0 || ptype==3)) //I or BI type
897 Trusted_IsNot("Should not be true!");
898
899 if (interlace)
900 Skip_SB( "uvsamp - uv sampling mode");
901
902 if (finterpflag && PictureFormat==0) //Progressive frame
903 Skip_SB( "interrpfrm");
904
905 if (PictureFormat!=1) //!=Interlaced frame
906 {
907 if (ptype==2 //Type B
908 || (ptype==3 && PictureFormat==2)) //Type BI and Interlaced field
909 Skip_VL(Vc1_bfraction, "bfraction");
910 }
911 }
912 */
913 }
914 BS_End();
915 if (Element_Size-Element_Offset)
916 Skip_XX(Element_Size-Element_Offset, "Data");
917
918 FILLING_BEGIN();
919 //NextCode
920 NextCode_Test();
921 NextCode_Clear();
922 NextCode_Add(0x0D);
923 NextCode_Add(0x0F);
924
925 //Autorisation of other streams
926 Streams[0x0D].Searching_Payload=true;
927 Streams[0x0F].Searching_Payload=true;
928
929 //Filling only if not already done
930 if (!Status[IsFilled] && Frame_Count>=Frame_Count_Valid)
931 {
932 Fill("VC-1");
933
934 if (!IsSub && MediaInfoLib::Config.ParseSpeed_Get()<1)
935 Finish("VC-1");
936 }
937
938 #if MEDIAINFO_EVENTS
939 {
940 EVENT_BEGIN (Video, SliceInfo, 0)
941 Event.FieldPosition=Field_Count;
942 Event.SlicePosition=0;
943 switch (ptype)
944 {
945 case 0 :
946 Event.SliceType=0; break;
947 case 1 :
948 Event.SliceType=1; break;
949 case 2 :
950 case 3 :
951 Event.SliceType=2; break;
952 case 4 :
953 Event.SliceType=3; break;
954 default:
955 Event.SliceType=(int8u)-1;
956 }
957 Event.Flags=0;
958 EVENT_END ()
959 }
960 #endif //MEDIAINFO_EVENTS
961 FILLING_END();
962
963 Synched=false; //We do not have the complete FrameHeader
964}

1056 Element_Name("SequenceHeader");
1057
1058 //Parsing
1059 BS_Begin();
1060 Get_S1 ( 2, profile, "profile"); Param_Info1(Vc1_Profile[profile]);
1061 if (profile==0 || profile==1) //Simple or Main
1062 {
1063 Skip_S1( 2, "res_sm");
1064 Skip_S1( 3, "frmrtq_postproc");
1065 Skip_S1( 5, "bitrtq_postproc");
1066 Skip_SB( "loopfilter");
1067 Skip_SB( "res_x8");
1068 Skip_SB( "multires");
1069 Skip_SB( "res_fasttx");
1070 Skip_SB( "fastuvmc");
1071 Skip_SB( "extended_mv");
1072 Skip_S1( 2, "dquant");
1073 Skip_SB( "vtransform");
1074 Skip_SB( "res_transtab");
1075 Skip_SB( "overlap");
1076 Skip_SB( "syncmarker");
1077 Skip_SB( "rangered");
1078 Skip_S1( 2, "maxbframes");
1079 Skip_S1( 2, "quantizer");
1080 Skip_SB( "finterpflag");
1081 Skip_SB( "res_rtm_flag");
1082 }
1083 else if (profile==3) //Advanced
1084 {
1085 Get_S1 ( 3, level, "level");
1086 Get_S1 ( 2, colordiff_format, "colordiff_format"); Param_Info1(Vc1_ColorimetryFormat[colordiff_format]);
1087 Skip_S1( 3, "frmrtq_postproc");
1088 Skip_S1( 5, "bitrtq_postproc");
1089 Skip_SB( "postprocflag");
1090 Get_S2 (12, coded_width, "max_coded_width"); Param_Info2((coded_width+1)*2, " pixels");
1091 Get_S2 (12, coded_height, "max_coded_height"); Param_Info2((coded_height+1)*2, " pixels");
1092 Get_SB ( pulldown, "pulldown");
1093 Get_SB ( interlace, "interlace");
1094 Get_SB ( tfcntrflag, "tfcntrflag - frame counter");
1095 Get_SB ( finterpflag, "finterpflag");
1096 Skip_SB( "reserved");
1097 Get_SB ( psf, "psf - progressive segmented frame");
1098 TEST_SB_SKIP( "display_ext");
1099 Info_S2(14, display_x, "display_horiz_size"); Param_Info2(display_x+1, " pixels");
1100 Info_S2(14, display_y, "display_vert_size"); Param_Info2(display_y+1, " pixels");
1101 TEST_SB_SKIP( "aspectratio_flag");
1102 Get_S1 ( 4, AspectRatio, "aspect_ratio"); Param_Info1(Vc1_PixelAspectRatio[AspectRatio]);
1103 if (AspectRatio==0x0F)
1104 {
1105 Get_S1 ( 8, AspectRatioX, "aspect_horiz_size");
1106 Get_S1 ( 8, AspectRatioY, "aspect_vert_size");
1107 }
1108 TEST_SB_END();
1109 TEST_SB_GET(framerate_present, "framerate_flag");
1110 TESTELSE_SB_GET(framerate_form, "framerateind");
1111 Get_S2 (16, framerateexp, "framerateexp"); Param_Info3((float32)((framerateexp+1)/32.0), " fps", 3);
1112 TESTELSE_SB_ELSE( "framerateind");
1113 Get_S1 ( 8, frameratecode_enr, "frameratenr"); Param_Info1(Vc1_FrameRate_enr(frameratecode_enr));
1114 Get_S1 ( 4, frameratecode_dr, "frameratedr"); Param_Info1(Vc1_FrameRate_dr(frameratecode_dr));
1115 TESTELSE_SB_END();
1116 TEST_SB_END();
1117 TEST_SB_SKIP( "color_format_flag");
1118 Skip_S1( 8, "color_prim");
1119 Skip_S1( 8, "transfer_char");
1120 Skip_S1( 8, "matrix_coef");
1121 TEST_SB_END();
1122 TEST_SB_END();
1123 TEST_SB_GET (hrd_param_flag, "hrd_param_flag");
1124 int8u buffer_size_exponent;
1125 Get_S1 ( 5, hrd_num_leaky_buckets, "hrd_num_leaky_buckets");
1126 Skip_S1( 4, "bitrate_exponent");
1127 Get_S1 ( 4, buffer_size_exponent, "buffer_size_exponent");
1128 hrd_buffers.clear();
1129 for (int8u Pos=0; Pos<hrd_num_leaky_buckets; Pos++)
1130 {
1131 Element_Begin1("leaky_bucket");
1132 int16u hrd_buffer;
1133 Skip_S2(16, "hrd_rate");
1134 Get_S2(16, hrd_buffer, "hrd_buffer");
1135 int32u hrd_buffer_value=(int32u)((hrd_buffer+1)*pow(2.0, 1+buffer_size_exponent)); Param_Info2(hrd_buffer_value, " bytes");
1136 Element_End0();
1137
1138 //Filling
1139 hrd_buffers.push_back(hrd_buffer_value);
1140 }
1141 TEST_SB_END();
1142 }
1143 else //forbidden
1144 {
1145 Element_DoNotTrust("Forbidden value");
1146 }
1147 Mark_1();
1148 BS_End();
1149
1150 FILLING_BEGIN();
1151 //NextCode
1152 NextCode_Clear();
1153 NextCode_Add(0x0D);
1154 NextCode_Add(0x0E);
1155
1156 //Autorisation of other streams
1157 Streams[0x0D].Searching_Payload=true;
1158 Streams[0x0E].Searching_Payload=true;
1159
1160 //Frame rate
1161 if (framerate_present)
1162 {
1163 if (framerate_form)
1164 FrameRate=((float64)(framerateexp+1))/(float32)64;
1165 else if (Vc1_FrameRate_dr(frameratecode_dr))
1166 FrameRate=((float64)Vc1_FrameRate_enr(frameratecode_enr))/Vc1_FrameRate_dr(frameratecode_dr);
1167 }
1168
1169 if (From_WMV3)
1170 {
1171 if (!Status[IsAccepted])
1172 Accept("VC-1");
1173 Finish("VC-1");
1174 }
1175
1176 #if MEDIAINFO_DEMUX
1177 if (InitData_Buffer_Size)
1178 {
1179 InitData_Buffer_Size=(size_t)(Header_Size+Element_Size);
1180 InitData_Buffer=new int8u[InitData_Buffer_Size];
1181 std::memcpy(InitData_Buffer, Buffer+Buffer_Offset-(size_t)Header_Size, (size_t)(Header_Size+Element_Size));
1182 }
1183 #endif //MEDIAINFO_DEMUX
1184 FILLING_END();
1185}
