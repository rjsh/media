61 RangeCoder(const int8u* Buffer, size_t Buffer_Size, const state_transitions default_state_transition);

63 void AssignStateTransitions(const state_transitions new_state_transition);
64 void ResizeBuffer(size_t Buffer_Size); //Adapt the buffer limit

67 void ForceUnderrun();
68
69 bool get_rac(int8u* States);
70 int32u get_symbol_u(int8u* States);
71 int32s get_symbol_s(int8u* States);
72
73 int32u Current;
74 int32u Mask;
75 state_transitions zero_state;

84//---------------------------------------------------------------------------
85RangeCoder::RangeCoder (const int8u* Buffer, size_t Buffer_Size, const state_transitions default_state_transition)
86{
87 //Assign buffer
88 Buffer_Beg=Buffer;
89 Buffer_Cur=Buffer;
90 Buffer_End=Buffer+Buffer_Size;
91
92 //Init
93 if (Buffer_Size)
94 Current=*Buffer_Cur;
95 Mask=0xFF;
96 Buffer_Cur++;
97
98 AssignStateTransitions(default_state_transition);
99}

101void RangeCoder::AssignStateTransitions (const state_transitions new_state_transition)
102{
103 //Assign StateTransitions
104 std::memcpy (one_state, new_state_transition, state_transitions_size);
105 zero_state[0]=0;
106 for (size_t i=1; i<state_transitions_size; i++)
107 zero_state[i]=-one_state[state_transitions_size-i];
108}

116//---------------------------------------------------------------------------
117size_t RangeCoder::BytesUsed()
118{
119 return Buffer_Cur-Buffer_Beg-(Mask<0x100?0:1);

122//---------------------------------------------------------------------------
123bool RangeCoder::Underrun()
124{
125 return (Buffer_Cur-(Mask<0x100?0:1)>Buffer_End)?true:false;

128//---------------------------------------------------------------------------
129void RangeCoder::ForceUnderrun()
130{
131 Mask=0;
132 Buffer_Cur=Buffer_End+1;

135//---------------------------------------------------------------------------
136bool RangeCoder::get_rac(int8u* States)
137{
138 // Next byte
139 if (Mask<0x100)
140 {
141 Current<<=8;
142
143 // If less, consume the next byte
144 // If equal, last byte assumed to be 0x00
145 // If more, underrun, we return 0
146 if (Buffer_Cur<Buffer_End)
147 {
148 Current|=*Buffer_Cur;
149 }
150 else if (Buffer_Cur>Buffer_End)
151 {
152 return false;
153 }
154
155 Buffer_Cur++;
156 Mask<<=8;
157 }
158
159 //Range Coder boolean value computing
160 int32u Mask2=(Mask*(*States))>>8;
161 Mask-=Mask2;
162 if (Current<Mask)
163 {
164 *States=zero_state[*States];
165 return false;
166 }
167 Current-=Mask;
168 Mask=Mask2;
169 *States=one_state[*States];
170 return true;
171}

173//---------------------------------------------------------------------------
174int32u RangeCoder::get_symbol_u(int8u* States)
175{
176 if (get_rac(States))
177 return 0;
178
179 int e = 0;
180 while (get_rac(States + 1 + min(e, 9))) // 1..10
181 {
182 e++;
183 if (e > 31)
184 {
185 ForceUnderrun(); // stream is buggy or unsupported, we disable it completely and we indicate that it is NOK
186 return 0;
187 }
188 }
189
190 int32u a = 1;
191 int i = e - 1;
192 while (i >= 0)
193 {
194 a <<= 1;
195 if (get_rac(States + 22 + min(i, 9))) // 22..31
196 ++a;
197 i--;
198 }
199
200 return a;
201}
202
203//---------------------------------------------------------------------------
204int32s RangeCoder::get_symbol_s(int8u* States)
205{
206 if (get_rac(States))
207 return 0;
208
209 int e = 0;
210 while (get_rac(States + 1 + min(e, 9))) // 1..10
211 {
212 e++;
213 if (e > 31)
214 {
215 ForceUnderrun(); // stream is buggy or unsupported, we disable it completely and we indicate that it is NOK
216 return 0;
217 }
218 }
219
220 int32s a = 1;
221 int i = e - 1;
222 while (i >= 0)
223 {
224 a <<= 1;
225 if (get_rac(States + 22 + min(i, 9))) // 22..31
226 ++a;
227 i--;
228 }
229
230 if (get_rac(States + 11 + min(e, 10))) // 11..21
231 return -a;
232 else
233 return a;
234}

591//---------------------------------------------------------------------------
592void File_Ffv1::Read_Buffer_OutOfBand()
593{
594 ConfigurationRecordIsPresent=true;
595
596 int32u CRC_32=CRC_Compute((size_t)Element_Size);
597
598 if (Buffer_Size < 4 || CRC_32)
599 {
600 Reject();
601 return;
602 }
603
604 if (!RC)
605 RC = new RangeCoder(Buffer, Buffer_Size-4, Ffv1_default_state_transition);
606
607 FrameHeader();
608 Element_Offset+=RC->BytesUsed();
609 if (Element_Offset<Element_Size)
610 Skip_XX(Element_Size-Element_Offset, "Reserved");
611 Skip_B4( "CRC-32");
612
613 delete RC; RC=NULL;
614}

616//---------------------------------------------------------------------------
617void File_Ffv1::Skip_Frame()
618{
619 Skip_XX(Element_Size-Element_Offset, "Other data");
620
621 Frame_Count++;
622
623 delete RC;
624 RC = NULL;
625
626 Fill();
627 if (Config->ParseSpeed<1.0)
628 Finish();
629}
630
631//---------------------------------------------------------------------------
632void File_Ffv1::Read_Buffer_Continue()
633{
634 if (!Status[IsAccepted])
635 Accept();
636
637 if (!RC)
638 RC = new RangeCoder(Buffer, Buffer_Size, Ffv1_default_state_transition);
639
640 states States;
641 memset(States, 128, states_size);
642
643 Get_RB (States, keyframe, "keyframe");
644
645 if (keyframe && !ConfigurationRecordIsPresent)
646 {
647 #if MEDIAINFO_TRACE
648 bool Trace_Activated_Save=Trace_Activated;
649 if (Trace_Activated && Frame_Count)
650 Trace_Activated=false; // Trace is relatively huge, temporarary deactivating it. TODO: an option for it
651 #endif //MEDIAINFO_TRACE
652
653 FrameHeader();
654
655 #if MEDIAINFO_TRACE
656 Trace_Activated=Trace_Activated_Save; // Trace is too huge, reactivating it.
657 #endif //MEDIAINFO_TRACE
658 }
659 else if (!KeyFramePassed)
660 {
661 // If stream does not start with a key frame
662 Skip_Frame();
663 return;
664 }
665
666 int32u tail = (version >= 3) ? 3 : 0;
667 tail += error_correction == 1 ? 5 : 0;
668
669 int64u Slices_BufferPos=Element_Size;
670 vector<int32u> Slices_BufferSizes;
671 if (version < 2)
672 Slices_BufferSizes.push_back(Element_Size);
673 else if (version == 2)
674 {
675 Skip_Frame();
676 return;
677 }
678 else
679 {
680 while (Slices_BufferPos)
681 {
682 int32u Size = BigEndian2int24u(Buffer + Buffer_Offset + (size_t)Slices_BufferPos - tail);
683 Size += tail;
684
685 if (Slices_BufferPos < Size)
686 Slices_BufferPos = Size;
687 Slices_BufferPos-=Size;
688
689 Slices_BufferSizes.insert(Slices_BufferSizes.begin(), Size);
690 }
691 }
692
693 Element_Offset=0;
694 for (size_t Pos = 0; Pos < Slices_BufferSizes.size(); Pos++)
695 {
696 Element_Begin1("Slice");
697 int64u Element_Size_Save=Element_Size;
698 Element_Size=Element_Offset+Slices_BufferSizes[Pos]-tail;
699 int32u crc_left=0;
700
701 if (error_correction == 1)
702 crc_left=CRC_Compute(Slices_BufferSizes[Pos]);
703
704 if (Pos)
705 {
706 delete RC; RC = new RangeCoder(Buffer+Buffer_Offset+(size_t)Element_Offset, Slices_BufferSizes[Pos]-tail, state_transitions_table);
707 }
708 else
709 {
710 RC->ResizeBuffer(Slices_BufferSizes[0]-tail);
711 RC->AssignStateTransitions(state_transitions_table);
712 }
713
714#if MEDIAINFO_TRACE
715 if (!Frame_Count || Trace_Activated) // Parse slice only if trace feature is activated
716 {
717 int64u Start=Element_Offset;
718
719 if (!slice(States))
720 {
721 int64u SliceRealSize=Element_Offset-Start;
722 Element_Offset=Start;
723 Skip_XX(SliceRealSize, "slice_data");
724 if (Trusted_Get() && !RC->Underrun() && Element_Offset==Element_Size)
725 Param_Info1("OK");
726 else
727 Param_Info1("NOK");
728 }
729 }
730#endif //MEDIAINFO_TRACE
731
732 if (Element_Offset<Element_Size)
733 Skip_XX(Element_Size-Element_Offset, "Other data");
734 Element_Size=Element_Size_Save;
735 if (version > 2)
736 Skip_B3( "slice_size");
737 if (error_correction == 1)
738 {
739 Skip_B1( "error_status");
740 Skip_B4( "crc_parity");
741 if (!crc_left)
742 Param_Info1("OK");
743 else
744 Param_Info1("NOK");
745 }
746 Element_End0();
747 }
748
749 FILLING_BEGIN();
750 if (Frame_Count==0)
751 {
752 Fill(Stream_Video, 0, Video_ScanType, Ffv1_picture_structure_ScanType(picture_structure));
753 Fill(Stream_Video, 0, Video_ScanOrder, Ffv1_picture_structure_ScanOrder(picture_structure));
754 if (sample_aspect_ratio_num && sample_aspect_ratio_den)
755 Fill(Stream_Video, 0, Video_PixelAspectRatio, sample_aspect_ratio_num/sample_aspect_ratio_den);
756 }
757
758 Frame_Count++;
759 FILLING_END();
760
761 delete RC;
762 RC = NULL;
763
764 Fill();
765 if (Config->ParseSpeed<1.0)
766 Finish();
767}

769//***************************************************************************
770// Elements
771//***************************************************************************
772
773//---------------------------------------------------------------------------
774void File_Ffv1::FrameHeader()
775{
776 //Parsing
777 states States;
778 memset(States, 128, states_size);
779 int32u coder_type, colorspace_type, bits_per_raw_sample=8, num_h_slices_minus1=0, num_v_slices_minus1=0, intra;
780
781 KeyFramePassed = true;
782 micro_version = 0;
783 Get_RU (States, version, "version");
784 if (( ConfigurationRecordIsPresent && version<=1)
785 || (!ConfigurationRecordIsPresent && version> 1))
786 {
787 Trusted_IsNot("Invalid version in global header");
788 return;
789 }
790 else if (version == 2)
791 {
792 FILLING_BEGIN();
793 if (Frame_Count==0)
794 {
795 Accept();
796
797 Ztring Version=__T("Version ")+Ztring::ToZtring(version);
798 Fill(Stream_Video, 0, Video_Format_Version, Version);
799 }
800 FILLING_END();
801 return;
802 }
803
804 if (version>2)
805 Get_RU (States, micro_version, "micro_version");
806 Get_RU (States, coder_type, "coder_type");
807 this->coder_type = coder_type;
808 if (coder_type == 2) //Range coder with custom state transition table
809 {
810 Element_Begin1("state_transition_deltas");
811 for (size_t i = 1; i < state_transitions_size; i++)
812 {
813 int32s state_transition_delta;
814 Get_RS (States, state_transition_delta, "state_transition_delta");
815 state_transitions_table[i]=state_transition_delta+RC->one_state[i];
816 Param_Info1(state_transitions_table[i]);
817 }
818 Element_End0();
819 }
820 Get_RU (States, colorspace_type, "colorspace_type");
821 this->colorspace_type = colorspace_type;
822 if (version)
823 {
824 Get_RU (States, bits_per_raw_sample, "bits_per_raw_sample");
825 if (bits_per_raw_sample==0)
826 bits_per_raw_sample=8; //I don't know the reason, 8-bit is coded 0 and 10-bit coded 10 (not 2?).
827 }
828 this->bits_per_sample = bits_per_raw_sample;
829 Get_RB (States, chroma_planes, "chroma_planes");
830 Get_RU (States, chroma_h_shift, "log2(h_chroma_subsample)");
831 Get_RU (States, chroma_v_shift, "log2(v_chroma_subsample)");
832 Get_RB (States, alpha_plane, "alpha_plane");
833 if (version>1)
834 {
835 Get_RU (States, num_h_slices_minus1, "num_h_slices_minus1");
836 Get_RU (States, num_v_slices_minus1, "num_v_slices_minus1");
837 num_h_slices = num_h_slices_minus1 + 1;
838 num_v_slices = num_v_slices_minus1 + 1;
839 Get_RU (States, quant_table_count, "quant_table_count");
840 }
841 else
842 quant_table_count = 2 + alpha_plane;
843
844 if (!slices)
845 {
846 size_t nb_slices = (num_h_slices_minus1 + 1) * (num_v_slices_minus1 + 1);
847 slices = new Slice[nb_slices];
848 current_slice = &slices[0];
849 }
850
851 if (version > 1)
852 for (size_t i = 0; i < quant_table_count; i++)
853 read_quant_tables(i);
854 else
855 {
856 current_slice->x = 0;
857 current_slice->y = 0;
858 current_slice->w = Width;
859 current_slice->h = Height;
860 read_quant_tables(0);
861 quant_table_index[0] = 0;
862 for (size_t i = 1; i < quant_table_count; i++)
863 {
864 quant_table_index[i] = 0;
865 context_count[i] = context_count[0];
866 }
867 }
868 memset(quant_tables+quant_table_count, 0x00, (MAX_QUANT_TABLES-quant_table_count)*MAX_CONTEXT_INPUTS*256*sizeof(pixel_t));
869
870 for (size_t i = 0; i < quant_table_count; i++)
871 {
872 Element_Begin1("initial_state");
873 bool present = false;
874 if (version > 1)
875 Get_RB (States, present, "present");
876
877 if (coder_type && context_count[i]>plane_states_maxsizes[i])
878 {
879 states_context_plane plane_state_old = plane_states[i];
880 plane_states[i] = new int8u*[context_count[i]];
881 if (plane_state_old)
882 {
883 memcpy(plane_states[i], plane_state_old, plane_states_maxsizes[i] * sizeof(int8u*));
884 delete[] plane_state_old;
885 }
886 for (size_t j = plane_states_maxsizes[i]; j < context_count[i]; j++)
887 plane_states[i][j] = new int8u[states_size];
888 plane_states_maxsizes[i] = context_count[i];
889 }
890
891 for (size_t j = 0; j < context_count[i]; j++)
892 {
893 if (present)
894 {
895 Element_Begin1("initial_state");
896 for (size_t k = 0; k < states_size; k++)
897 {
898 int32s value;
899 Get_RS (States, value, "value");
900 if (coder_type)
901 plane_states[i][j][k] = value;
902 }
903 Element_End0();
904 }
905 else if (coder_type)
906 {
907 for (size_t k = 0; k < states_size; k++)
908 plane_states[i][j][k] = 128;
909 }
910 }
911 Element_End0();
912 }
913
914 if (version > 2)
915 {
916 Get_RU (States, error_correction, "ec");
917 if (micro_version)
918 Get_RU (States, intra, "intra");
919 }
920
921 //Marking handling of 16-bit overflow computing
922 is_overflow_16bit=(colorspace_type==0 && bits_per_raw_sample==16 && (coder_type==1 || coder_type==2))?true:false; //TODO: check in FFmpeg the version when the stream is fixed. Note: only with YUV colorspace
923
924 FILLING_BEGIN();
925 if (Frame_Count==0)
926 {
927 Accept();
928
929 Ztring Version=__T("Version ")+Ztring::ToZtring(version);
930 if (version>2)
931 {
932 Version+=__T('.');
933 Version+=Ztring::ToZtring(micro_version);
934 }
935 Fill(Stream_Video, 0, "coder_type", Ffv1_coder_type(coder_type));
936 Fill(Stream_Video, 0, Video_Format_Version, Version);
937 Fill(Stream_Video, 0, Video_BitDepth, bits_per_raw_sample);
938 if (version > 1)
939 {
940 Fill(Stream_Video, 0, "MaxSlicesCount", (num_h_slices_minus1+1)*(num_v_slices_minus1+1));
941 }
942 if (version > 2)
943 {
944 if (error_correction)
945 Fill(Stream_Video, 0, "ErrorDetectionType", "Per slice");
946 if (micro_version && intra)
947 Fill(Stream_Video, 0, Video_Format_Settings_GOP, "N=1");
948 }
949 Fill(Stream_Video, 0, Video_ColorSpace, Ffv1_colorspace_type(colorspace_type, chroma_planes, alpha_plane));
950 if (colorspace_type==0 && chroma_planes)
951 {
952 string ChromaSubsampling;
953 switch (chroma_h_shift)
954 {
955 case 0 :
956 switch (chroma_v_shift)
957 {
958 case 0 : ChromaSubsampling="4:4:4"; break;
959 default: ;
960 }
961 break;
962 case 1 :
963 switch (chroma_v_shift)
964 {
965 case 0 : ChromaSubsampling="4:2:2"; break;
966 case 1 : ChromaSubsampling="4:2:0"; break;
967 default: ;
968 }
969 break;
970 case 2 :
971 switch (chroma_v_shift)
972 {
973 case 0 : ChromaSubsampling="4:1:1"; break;
974 case 1 : ChromaSubsampling="4:1:0"; break;
975 case 2 : ChromaSubsampling="4:1:0 (4x4)"; break;
976 default: ;
977 }
978 break;
979 default: ;
980 }
981 if (!ChromaSubsampling.empty() && alpha_plane)
982 ChromaSubsampling+=":4";
983 Fill(Stream_Video, 0, Video_ChromaSubsampling, ChromaSubsampling);
984 }
985 }
986 FILLING_END();
987}

989//---------------------------------------------------------------------------
990int File_Ffv1::slice(states &States)
991{
992 if (version>2)
993 if (slice_header(States) < 0)
994 return -1;
995
996 #if MEDIAINFO_TRACE
997 bool Trace_Activated_Save=Trace_Activated;
998 if (Trace_Activated)
999 Trace_Activated=false; // Trace is too huge, deactivating it during pixel decoding
1000 #endif //MEDIAINFO_TRACE
1001
1002 if (!coder_type)
1003 {
1004 if ((version == 3 && micro_version > 1) || version > 3)
1005 {
1006 states States;
1007 memset(States, 129, states_size);
1008 Skip_RC(States, "?");
1009 }
1010 if ((version > 2 || (!current_slice->x && !current_slice->y)))
1011 Element_Offset+=RC->BytesUsed(); // Computing how many bytes where consumed by the range coder
1012 else
1013 Element_Offset=0;
1014 BS_Begin();
1015 }
1016
1017 if (keyframe)
1018 {
1019 int8u plane_count=1+(alpha_plane?1:0);
1020 if (version < 4 || chroma_planes) // Warning: chroma is considered as 1 plane
1021 plane_count+=1;
1022 if (!coder_type)
1023 current_slice->contexts_init(plane_count, quant_table_index, context_count);
1024 else
1025 copy_plane_states_to_slice(plane_count);
1026 }
1027 current_slice->sample_buffer_new((current_slice->w + 6) * 3 * MAX_PLANES);
1028
1029 if (colorspace_type == 0)
1030 {
1031 // YCbCr
1032 plane(0); // Y
1033 if (chroma_planes)
1034 {
1035 int32u w = current_slice->w;
1036 int32u h = current_slice->h;
1037
1038 current_slice->w = w >> chroma_h_shift;
1039 if (w & ((1 << chroma_h_shift) - 1))
1040 current_slice->w++; //Is ceil
1041 current_slice->h = h >> chroma_v_shift;
1042 if (h & ((1 << chroma_v_shift) - 1))
1043 current_slice->h++; //Is ceil
1044 plane(1); // Cb
1045 plane(1); // Cr
1046 current_slice->w = w;
1047 current_slice->h = h;
1048 }
1049 if (alpha_plane)
1050 plane(2); // Alpha
1051 }
1052 else if (colorspace_type == 1)
1053 rgb();
1054
1055 if (!coder_type && ((version == 3 && micro_version > 1) || version > 3))
1056 BS_End();
1057
1058 if (coder_type && version > 2)
1059 {
1060 int8u s = 129;
1061 RC->get_rac(&s);
1062 Element_Offset+=RC->BytesUsed();
1063 }
1064
1065 #if MEDIAINFO_TRACE
1066 Trace_Activated=Trace_Activated_Save; // Trace is too huge, reactivating after during pixel decoding
1067 #endif //MEDIAINFO_TRACE
1068
1069 return 0;
1070}

1072//---------------------------------------------------------------------------
1073int File_Ffv1::slice_header(states &States)
1074{
1075 Element_Begin1("SliceHeader");
1076
1077 memset(States, 128, states_size);
1078
1079 int32u slice_x, slice_y, slice_width_minus1, slice_height_minus1;
1080 Get_RU (States, slice_x, "slice_x");
1081 if (slice_x >= num_v_slices)
1082 {
1083 Param_Info1("NOK");
1084 Element_End0();
1085 return -1;
1086 }
1087
1088 Get_RU (States, slice_y, "slice_y");
1089 if (slice_y >= num_h_slices)
1090 {
1091 Param_Info1("NOK");
1092 Element_End0();
1093 return -1;
1094 }
1095
1096 Get_RU (States, slice_width_minus1, "slice_width_minus1");
1097 int32u slice_x2 = slice_x + slice_width_minus1 + 1; //right boundary
1098 if (slice_x2 > num_h_slices)
1099 {
1100 Param_Info1("NOK");
1101 Element_End0();
1102 return -1;
1103 }
1104
1105 Get_RU (States, slice_height_minus1, "slice_height_minus1");
1106 int32u slice_y2 = slice_y + slice_height_minus1 + 1; //bottom boundary
1107 if (slice_y2 > num_v_slices)
1108 {
1109 Param_Info1("NOK");
1110 Element_End0();
1111 return -1;
1112 }
1113
1114 current_slice = &slices[slice_x + slice_y * num_h_slices];
1115
1116 //Computing boundaries, being careful about how are computed boundaries when there is not an integral number for Width / num_h_slices or Height / num_v_slices (the last slice has more pixels)
1117 current_slice->x = slice_x * Width / num_h_slices;
1118 current_slice->y = slice_y * Height / num_v_slices;
1119 current_slice->w = slice_x2 * Width / num_h_slices - current_slice->x;
1120 current_slice->h = slice_y2 * Height / num_v_slices - current_slice->y;
1121
1122
1123 int8u plane_count=1+(alpha_plane?1:0);
1124 if (version < 4 || chroma_planes) // Warning: chroma is considered as 1 plane
1125 plane_count += 1;
1126 for (int8u i = 0; i < plane_count; i++)
1127 Get_RU (States, quant_table_index[i], "quant_table_index");
1128 Get_RU (States, picture_structure, "picture_structure");
1129 Get_RU (States, sample_aspect_ratio_num, "sample_aspect_ratio num");
1130 Get_RU (States, sample_aspect_ratio_den, "sample_aspect_ratio den");
1131 if (version > 3)
1132 {
1133 //TODO
1134 }
1135
1136 RC->AssignStateTransitions(state_transitions_table);
1137
1138 Element_End0();
1139 return 0;
1140}

1143void File_Ffv1::plane(int32u pos)
1144{
1145 #if MEDIAINFO_TRACE_FFV1CONTENT
1146 Element_Begin1("Plane");
1147 #endif //MEDIAINFO_TRACE_FFV1CONTENT
1148
1149 if (bits_per_sample <= 8)
1150 bits_max = 8;
1151 else
1152 bits_max = bits_per_sample;
1153 bits_mask1 = ((1 << bits_max) - 1);
1154 bits_mask2 = 1 << (bits_max - 1);
1155 bits_mask3 = bits_mask2 - 1;
1156
1157 pixel_t *sample[2];
1158 sample[0] = current_slice->sample_buffer + 3;
1159 sample[1] = sample[0] + current_slice->w + 6;
1160
1161 memset(current_slice->sample_buffer, 0, 2 * (current_slice->w + 6) * sizeof(*current_slice->sample_buffer));
1162
1163 current_slice->run_index = 0;
1164
1165 for (size_t y = 0; y < current_slice->h; y++)
1166 {
1167 #if MEDIAINFO_TRACE_FFV1CONTENT
1168 Element_Begin1("Line");
1169 Element_Info1(y);
1170 #endif //MEDIAINFO_TRACE_FFV1CONTENT
1171
1172 swap(sample[0], sample[1]);
1173
1174 sample[1][-1] = sample[0][0];
1175 sample[0][current_slice->w] = sample[0][current_slice->w - 1];
1176
1177 line(pos, sample);
1178
1179 #if MEDIAINFO_TRACE_FFV1CONTENT
1180 Element_End0();
1181 #endif //MEDIAINFO_TRACE_FFV1CONTENT
1182 }
1183
1184 #if MEDIAINFO_TRACE_FFV1CONTENT
1185 Element_End0();
1186 #endif //MEDIAINFO_TRACE_FFV1CONTENT

1242static inline int32s get_median_number(int32s one, int32s two, int32s three)
1243{
1244 if (one > two)
1245 {
1246 // one > two > three
1247 if (two > three)
1248 return two;
1249
1250 // three > one > two
1251 if (three > one)
1252 return one;
1253 // one > three > two
1254 return three;
1255 }
1256
1257 // three > two > one
1258 if (three > two)
1259 return two;
1260
1261 // two > one && two > three
1262
1263 // two > three > one
1264 if (three > one)
1265 return three;
1266 return one;
1267}

1269//---------------------------------------------------------------------------
1270static inline int32s predict(pixel_t *current, pixel_t *current_top, bool is_overflow_16bit)
1271{
1272 int32s LeftTop, Top, Left;
1273 if (is_overflow_16bit)
1274 {
1275 LeftTop = (int16s)current_top[-1];
1276 Top = (int16s)current_top[0];
1277 Left = (int16s)current[-1];
1278 }
1279 else
1280 {
1281 LeftTop = current_top[-1];
1282 Top = current_top[0];
1283 Left = current[-1];
1284 }
1285
1286 return get_median_number(Left, Left + Top - LeftTop, Top);
1287}

1289//---------------------------------------------------------------------------
1290static inline int get_context_3(pixel_t quant_table[MAX_CONTEXT_INPUTS][256], pixel_t *src, pixel_t *last)
1291{
1292 const int LT = last[-1];
1293 const int T = last[0];
1294 const int RT = last[1];
1295 const int L = src[-1];
1296
1297 return quant_table[0][(L - LT) & 0xFF]
1298 + quant_table[1][(LT - T) & 0xFF]
1299 + quant_table[2][(T - RT) & 0xFF];
1300}
1301static inline int get_context_5(pixel_t quant_table[MAX_CONTEXT_INPUTS][256], pixel_t *src, pixel_t *last)
1302{
1303 const int LT = last[-1];
1304 const int T = last[0];
1305 const int RT = last[1];
1306 const int L = src[-1];
1307 const int TT = src[0];
1308 const int LL = src[-2];
1309 return quant_table[0][(L - LT) & 0xFF]
1310 + quant_table[1][(LT - T) & 0xFF]
1311 + quant_table[2][(T - RT) & 0xFF]
1312 + quant_table[3][(LL - L) & 0xFF]
1313 + quant_table[4][(TT - T) & 0xFF];
1314}

1327int32s File_Ffv1::pixel_GR(int32s context)
1328{
1329#if MEDIAINFO_TRACE_FFV1CONTENT
1330 int32s u;
1331
1332 // New symbol, go to "run mode"
1333 if (context == 0 && current_slice->run_mode == RUN_MODE_STOP)
1334 current_slice->run_mode = RUN_MODE_PROCESSING;
1335

1338 {
1339 u = get_symbol_with_bias_correlation(&Context_GR[context]);
1340 #if MEDIAINFO_TRACE
1341 Param("symbol", u);
1342 #endif //MEDIAINFO_TRACE
1343 return u;
1344 }
1345
1346 if (current_slice->run_segment_length == 0 && current_slice->run_mode == RUN_MODE_PROCESSING) // Same symbol length
1347 {
1348 bool hits;
1349 Get_SB (hits, "hits/miss");
1350 //if (bsf.GetB()) // "hits"
1351 if (hits) // "hits"
1352 {
1353 current_slice->run_segment_length = run[current_slice->run_index];
1354 if (x + current_slice->run_segment_length <= current_slice->w) //Do not go further as the end of line
1355 ++current_slice->run_index;
1356 }
1357 else // "miss"
1358 {
1359 //current_slice->run_segment_length = bsf.Get4(log2_run[current_slice->run_index]);

1363 if (current_slice->run_index)
1364 --current_slice->run_index;
1365 current_slice->run_mode = RUN_MODE_INTERRUPTED;
1366 }
1367 }
1368
1369 current_slice->run_segment_length--;
1370 if (current_slice->run_segment_length < 0) // we passed the length of same symbol, time to get the new symbol
1371 {
1372 u = get_symbol_with_bias_correlation(&Context_GR[context]);
1373 #if MEDIAINFO_TRACE
1374 Param("symbol", u);
1375 #endif //MEDIAINFO_TRACE
1376 if (u >= 0) // GR(u - 1, ...)
1377 u++;
1378
1379 // Time for the new symbol length run
1380 current_slice->run_mode_init();
1381 current_slice->run_segment_length = 0;
1382 } else // same symbol as previous pixel, no difference, waiting
1383 u = 0;
1384 return u;
1385#else //MEDIAINFO_TRACE_FFV1CONTENT
1386 if (current_slice->run_mode == RUN_MODE_STOP)
1387 {
1388 if (context)
1389 return get_symbol_with_bias_correlation(&Context_GR[context]); // If not running, get the symbol
1390
1391 current_slice->run_mode = RUN_MODE_PROCESSING; // New symbol, go to "run mode"
1392 }
1393
1394 if (current_slice->run_segment_length == 0 && current_slice->run_mode == RUN_MODE_PROCESSING) // Same symbol length
1395 {
1396 if (BS->GetB()) // "hits"
1397 {
1398 current_slice->run_segment_length = run[current_slice->run_index];
1399 if (x + current_slice->run_segment_length <= current_slice->w) //Do not go further as the end of line
1400 ++current_slice->run_index;
1401 if (--current_slice->run_segment_length >= 0)
1402 return 0;
1403 }
1404 else // "miss"
1405 {
1406 current_slice->run_mode = RUN_MODE_INTERRUPTED;
1407
1408 if (current_slice->run_index)
1409 {
1410 int8u count = log2_run[current_slice->run_index--];
1411 if (count)
1412 {
1413 current_slice->run_segment_length = ((int32s)BS->Get4(count)) - 1;
1414 if (current_slice->run_segment_length >= 0)
1415 return 0;
1416 }
1417 else
1418 current_slice->run_segment_length = -1;
1419 }
1420 else
1421 current_slice->run_segment_length = -1;
1422 }
1423 }
1424 else if (--current_slice->run_segment_length >= 0)
1425 return 0;
1426
1427 // Time for the new symbol length run
1428 current_slice->run_mode_init();
1429
1430 int32s u = get_symbol_with_bias_correlation(&Context_GR[context]);
1431 if (u >= 0) // GR(u - 1, ...)
1432 u++;
1433 return u;
1434#endif //MEDIAINFO_TRACE_FFV1CONTENT
1435}

1437//---------------------------------------------------------------------------
1438void File_Ffv1::line(int pos, pixel_t *sample[2])
1439{
1440 // TODO: slice_coding_mode (version 4)
1441
1442 quant_table_struct& quant_table = quant_tables[quant_table_index[pos]];
1443 bool Is5 = quant_table[3][127] ? true : false;
1444 pixel_t* s0c = sample[0];
1445 pixel_t* s0e = s0c + current_slice->w;
1446 pixel_t* s1c = sample[1];
1447
1448 if (coder_type)
1449 {
1450 Context_RC = current_slice->plane_states[pos];
1451
1452 while (s0c<s0e)
1453 {
1454 int32s context = Is5 ? get_context_5(quant_table, s1c, s0c) : get_context_3(quant_table, s1c, s0c);
1455
1456 int32s Value = predict(s1c, s0c, is_overflow_16bit);
1457 #if MEDIAINFO_TRACE_FFV1CONTENT
1458 if (context >= 0)
1459 Value += pixel_RC(context);
1460 else
1461 Value -= pixel_RC(-context);
1462 #else //MEDIAINFO_TRACE_FFV1CONTENT
1463 if (context >= 0)
1464 Value += RC->get_symbol_s(Context_RC[context]);
1465 else
1466 Value -= RC->get_symbol_s(Context_RC[-context]);
1467 #endif //MEDIAINFO_TRACE_FFV1CONTENT;
1468 *s1c = Value & bits_mask1;
1469
1470 s0c++;
1471 s1c++;
1472 }
1473 }
1474 else
1475 {
1476 current_slice->run_mode_init();
1477
1478 Context_GR = current_slice->contexts[pos];
1479 x = 0;
1480
1481 while (s0c < s0e)
1482 {
1483 int32s context = Is5 ? get_context_5(quant_table, s1c, s0c) : get_context_3(quant_table, s1c, s0c);
1484
1485 *s1c = (predict(s1c, s0c, is_overflow_16bit) + (context >= 0 ? pixel_GR(context) : -pixel_GR(-context))) & bits_mask1;
1486
1487 s0c++;
1488 s1c++;
1489 x++;
1490 }
1491 }
1492}

1494//---------------------------------------------------------------------------
1495void File_Ffv1::read_quant_tables(int i)
1496{
1497 Element_Begin1("quant_table");
1498
1499 int32u scale = 1;
1500
1501 for (int j = 0; j < 5; j++)
1502 {
1503 read_quant_table(i, j, scale);
1504 scale *= 2 * len_count[i][j] - 1;
1505 if (scale > 32768U)
1506 {
1507 //TODO Error
1508 context_count[i] = (scale + 1) / 2;
1509 Element_End0();
1510 return;
1511 }
1512 }
1513 context_count[i] = (scale + 1) / 2;
1514
1515 Element_End0();
1516}

1518//---------------------------------------------------------------------------
1519void File_Ffv1::read_quant_table(int i, int j, size_t scale)
1520{
1521 Element_Begin1("per context");
1522
1523 int8u States[states_size];
1524 memset(States, 128, sizeof(States));
1525
1526 int v = 0;
1527 for (int k=0; k < 128;)
1528 {
1529 int32u len_minus1;
1530 Get_RU (States, len_minus1, "len_minus1");
1531
1532 if (k+len_minus1 >= 128)
1533 {
1534 Element_End0();
1535 return;
1536 }
1537
1538 for (size_t a=0; a<=len_minus1; a++)
1539 {
1540 quant_tables[i][j][k] = scale * v;
1541 k++;
1542 }
1543
1544 v++;
1545 }
1546
1547 for (int k = 1; k < 128; k++)
1548 quant_tables[i][j][256 - k] = -quant_tables[i][j][k];
1549 quant_tables[i][j][128] = -quant_tables[i][j][127];
1550
1551 len_count[i][j]=v;
1552
1553 Element_End0();
1554}

1630//---------------------------------------------------------------------------
1631int32s File_Ffv1::get_symbol_with_bias_correlation(Slice::ContextPtr context)
1632{
1633 int k = 0;
1634 // Step 8: compute the Golomb parameter k
1635 for (k = 0; (context->N << k) < context->A; ++k);
1636
1637 // Step 10: Decode Golomb code (using limitation PREFIX_MAX == 12)
1638 int32s code = golomb_rice_decode(k);
1639
1640 // Step 9: Mapping
1641 int32s M = 2 * context->B + context->N;
1642 code = code ^ (M >> 31);
1643
1644 // Step 11
1645 context->B += code;
1646 context->A += code >= 0 ? code : -code;
1647
1648 code += context->C;
1649
1650 context->update_correlation_value_and_shift();
1651
1652 // Step 7 (TODO better way)
1653 bool neg = code & bits_mask2; // check if the number is negative
1654 code = code & bits_mask3; // Keep only the n bits
1655 if (neg)
1656 code = - 1 - (~code & bits_mask3); // 0xFFFFFFFF - positive value on n bits
1657
1658 return code;
1659}
