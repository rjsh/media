88// Buffer - File header
89//***************************************************************************

91//---------------------------------------------------------------------------
92bool File_Y4m::FileHeader_Begin()
93{
94 if (Buffer_Size<10)
95 return false;
96
97 if (!(Buffer[0]==0x59
98 && Buffer[1]==0x55
99 && Buffer[2]==0x56
100 && Buffer[3]==0x34
101 && Buffer[4]==0x4D
102 && Buffer[5]==0x50
103 && Buffer[6]==0x45
104 && Buffer[7]==0x47
105 && Buffer[8]==0x32
106 && Buffer[9]==0x20))
107 {
108 Reject();
109 return false;
110 }
111
112 for (; HeaderEnd<Buffer_Size; HeaderEnd++)
113 {
114 if (Buffer[HeaderEnd]==0x0A)
115 {
116 Accept();
117 return true;
118 }
119 }
120
121 return false;
122}

124//---------------------------------------------------------------------------
125void File_Y4m::FileHeader_Parse()
126{
127 //Parsing
128 Ztring Header;
129 Get_Local(HeaderEnd, Header, "Data");
130
131 ZtringList List; List.Separator_Set(0, " ");
132 List.Write(Header);
133 int64u Width=0, Height=0, Multiplier=0, Divisor=1;
134 float64 FrameRate=0;
135 for (size_t Pos=1; Pos<List.size(); Pos++)
136 {
137 if (!List[Pos].empty())
138 {
139 switch (List[Pos][0])
140 {
141 case 'A' : //Pixel aspect ratio
142 {
143 ZtringList Value; Value.Separator_Set(0, ":");
144 Value.Write(List[Pos].substr(1));
145 if (Value.size()==2)
146 {
147 float64 x=Value[0].To_float64();
148 float64 y=Value[1].To_float64();
149 if (x && y)
150 Fill(Stream_Video, 0, Video_PixelAspectRatio, x/y, 3);
151 }
152 }
153 break;
154 case 'C' : //Color space
155 if (List[Pos]==__T("C420jpeg") || List[Pos]==__T("C420paldv") || List[Pos]==__T("C420"))
156 {
157 Fill(Stream_Video, 0, Video_ChromaSubsampling, "4:2:0");
158 Multiplier=3;
159 Divisor=2;
160 }
161 if (List[Pos]==__T("C422"))
162 {
163 Fill(Stream_Video, 0, Video_ChromaSubsampling, "4:2:2");
164 Multiplier=2;
165 }
166 if (List[Pos]==__T("C444"))
167 {
168 Fill(Stream_Video, 0, Video_ChromaSubsampling, "4:4:4");
169 Multiplier=3;
170 }
171 break;
172 case 'F' : //Frame rate
173 {
174 ZtringList Value; Value.Separator_Set(0, ":");
175 Value.Write(List[Pos].substr(1));
176 if (Value.size()==2)
177 {
178 float64 N=Value[0].To_float64();
179 float64 D=Value[1].To_float64();
180 if (N && D)
181 {
182 FrameRate=N/D;
183 Fill(Stream_Video, 0, Video_FrameRate, FrameRate, 3);
184 }
185 }
186 }
187 break;
188 case 'H' : //Height
189 {
190 Ztring Value=List[Pos].substr(1);
191 Height=Value.To_int64u();
192 Fill(Stream_Video, 0, Video_Height, Height); break;
193 }
194 break;
195 case 'I' : //Interlacing
196 if (List[Pos].size()==2)
197 switch (List[Pos][1])
198 {
199 case 'p' : Fill(Stream_Video, 0, Video_ScanType, "Progressive"); break;
200 case 't' : Fill(Stream_Video, 0, Video_ScanType, "Progressive"); Fill(Stream_Video, 0, Video_ScanOrder, "TFF"); break;
201 case 'b' : Fill(Stream_Video, 0, Video_ScanType, "Progressive"); Fill(Stream_Video, 0, Video_ScanOrder, "BFF"); break;
202 case 'm' : Fill(Stream_Video, 0, Video_ScanType, "Mixed"); break;
203 default : ;
204 }
205 break;
206 case 'W' : //Width
207 {
208 Ztring Value=List[Pos].substr(1);
209 Width=Value.To_int64u();
210 Fill(Stream_Video, 0, Video_Width, Width); break;
211 }
212 break;
213 default : ;
214 }
215 }
216 }
217
218 //Duration (we expect no meta per frame)
219 if (Width && Height && Multiplier)
220 {
221 int64u Frame_ByteSize=6+Width*Height*Multiplier/Divisor; //5 for "FRAME\0"
222 Fill(Stream_Video, 0, Video_FrameCount, File_Size/Frame_ByteSize);
223 if (FrameRate)
224 Fill(Stream_Video, 0, Video_BitRate, Width*Height*Multiplier/Divisor*8*FrameRate);
225 }
226
227 Finish();
228}
