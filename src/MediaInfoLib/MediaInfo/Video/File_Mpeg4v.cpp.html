272bool File_Mpeg4v::Synched_Test()
273{
274 //Must have enough buffer for having header
275 if (Buffer_Offset+4>Buffer_Size)
276 return false;
277
278 //Quick test of synchro
279 if (Buffer[Buffer_Offset ]!=0x00
280 || Buffer[Buffer_Offset+1]!=0x00
281 || Buffer[Buffer_Offset+2]!=0x01)
282 {
283 Synched=false;
284 return true;
285 }
286
287 //Quick search
288 if (!Header_Parser_QuickSearch())
289 return false;
290
291 #if MEDIAINFO_IBIUSAGE
292 bool RandomAccess=Buffer[Buffer_Offset+3]==0xB0; //SequenceHeader
293 if (RandomAccess)
294 Ibi_Add();
295 #endif //MEDIAINFO_IBIUSAGE
296
297 //We continue
298 return true;
299}

404void File_Mpeg4v::Streams_Fill()
405{
406 //Filling
407 Stream_Prepare(Stream_Video);
408 Fill(Stream_Video, 0, Video_Format, "MPEG-4 Visual");
409 Fill(Stream_Video, 0, Video_Codec, "MPEG-4V");
410
411 if (profile_and_level_indication>0)
412 {
413 Fill(Stream_Video, 0, Video_Format_Profile, Mpeg4v_Profile_Level(profile_and_level_indication));
414 Fill(Stream_Video, 0, Video_Codec_Profile, Mpeg4v_Profile_Level(profile_and_level_indication));
415 }
416
417 if (frame_rate_code!=(int8u)-1)
418 Fill(Stream_Video, StreamPos_Last, Video_FrameRate, Mpegv_frame_rate[frame_rate_code]);
419 if (bit_rate!=(int32u)-1)
420 Fill(Stream_Video, StreamPos_Last, Video_BitRate_Nominal, bit_rate*400);
421 if (vbv_buffer_size!=(int32u)-1)
422 Fill(Stream_Video, StreamPos_Last, Video_BufferSize, vbv_buffer_size*2048);
423 if (fixed_vop_time_increment && vop_time_increment_resolution)
424 Fill(Stream_Video, StreamPos_Last, Video_FrameRate, ((float)vop_time_increment_resolution)/fixed_vop_time_increment);
425 if (object_layer_height)
426 {
427 Fill(Stream_Video, StreamPos_Last, Video_Width, object_layer_width);
428 Fill(Stream_Video, StreamPos_Last, Video_Height, object_layer_height);
429 float32 PixelAspectRatio_Value=1.0;
430 if (aspect_ratio_info==0x01) PixelAspectRatio_Value=(float32)1;
431 if (aspect_ratio_info==0x02) PixelAspectRatio_Value=(float32)12/(float32)11;
432 else if (aspect_ratio_info==0x03) PixelAspectRatio_Value=(float32)10/(float32)11;
433 else if (aspect_ratio_info==0x04) PixelAspectRatio_Value=(float32)16/(float32)11;
434 else if (aspect_ratio_info==0x05) PixelAspectRatio_Value=(float32)40/(float32)33;
435 else if (aspect_ratio_info==0x0F && par_height) PixelAspectRatio_Value=((float32)par_width)/par_height;
436 Fill(Stream_Video, 0, Video_PixelAspectRatio, PixelAspectRatio_Value, 3, true);
437 Fill(Stream_Video, StreamPos_Last, Video_DisplayAspectRatio, ((float)object_layer_width)/object_layer_height*PixelAspectRatio_Value, 3, true);
438 }
439 Fill(Stream_Video, 0, Video_ColorSpace, rgb_components?"RGB":"YUV");
440 Fill(Stream_Video, 0, Video_BitDepth, bits_per_pixel);
441 if (chroma_format<4)
442 Fill(Stream_Video, 0, Video_Colorimetry, Mpeg4v_Colorimetry[chroma_format]);
443 if (colour_description)
444 {
445 Fill(Stream_Video, 0, Video_colour_description_present, "Yes");
446 Fill(Stream_Video, 0, Video_colour_primaries, Mpegv_colour_primaries(colour_primaries));
447 Fill(Stream_Video, 0, Video_transfer_characteristics, Mpegv_transfer_characteristics(transfer_characteristics));
448 Fill(Stream_Video, 0, Video_matrix_coefficients, Mpegv_matrix_coefficients(matrix_coefficients));
449 }
450 if (low_delay)
451 {
452 Fill(Stream_Video, 0, Video_Format_Settings_BVOP, "No");
453 Fill(Stream_Video, 0, Video_Codec_Settings_BVOP, "No");
454 }
455 else
456 {
457 Fill(Stream_Video, 0, Video_Format_Settings, "BVOP");
458 Fill(Stream_Video, 0, Video_Format_Settings_BVOP, "Yes");
459 Fill(Stream_Video, 0, Video_Codec_Settings, "BVOP");
460 Fill(Stream_Video, 0, Video_Codec_Settings_BVOP, "Yes");
461 }
462 if (no_of_sprite_warping_points)
463 {
464 Fill(Stream_Video, 0, Video_Format_Settings, Ztring(__T("GMC"))+Ztring::ToZtring(no_of_sprite_warping_points));
465 Fill(Stream_Video, 0, Video_Format_Settings_GMC, no_of_sprite_warping_points);
466 Fill(Stream_Video, 0, Video_Codec_Settings, Ztring(__T("GMC"))+Ztring::ToZtring(no_of_sprite_warping_points));
467 Fill(Stream_Video, 0, Video_Codec_Settings_GMC, no_of_sprite_warping_points);
468 }
469 else
470 {
471 Fill(Stream_Video, 0, Video_Format_Settings_GMC, 0);
472 Fill(Stream_Video, 0, Video_Codec_Settings_GMC, 0);
473 }
474 if (quarter_sample)
475 {
476 Fill(Stream_Video, 0, Video_Format_Settings, "QPel");
477 Fill(Stream_Video, 0, Video_Format_Settings_QPel, "Yes");
478 Fill(Stream_Video, 0, Video_Codec_Settings, "QPel");
479 Fill(Stream_Video, 0, Video_Codec_Settings_QPel, "Yes");
480 }
481 else
482 {
483 Fill(Stream_Video, 0, Video_Format_Settings_QPel, "No");
484 Fill(Stream_Video, 0, Video_Codec_Settings_QPel, "No");
485 }
486 if (!quant_type)
487 {
488 Fill(Stream_Video, 0, Video_Format_Settings_Matrix, "Default (H.263)");
489 Fill(Stream_Video, 0, Video_Codec_Settings_Matrix, "Default (H.263)");
490 }
491 else if (load_intra_quant_mat_grayscale || load_nonintra_quant_mat_grayscale)
492 {
493 Fill(Stream_Video, 0, Video_Format_Settings, "Custom Matrix (Gray)");
494 Fill(Stream_Video, 0, Video_Format_Settings_Matrix, "Custom (Gray)");
495 Fill(Stream_Video, 0, Video_Codec_Settings, "Custom Matrix (Gray)");
496 Fill(Stream_Video, 0, Video_Codec_Settings_Matrix, "Custom (Gray)");
497 }
498 else if (load_intra_quant_mat || load_nonintra_quant_mat)
499 {
500 Fill(Stream_Video, 0, Video_Format_Settings, "Custom Matrix");
501 Fill(Stream_Video, 0, Video_Format_Settings_Matrix, "Custom");
502 Fill(Stream_Video, 0, Video_Format_Settings_Matrix_Data, Matrix_intra);
503 Fill(Stream_Video, 0, Video_Format_Settings_Matrix_Data, Matrix_nonintra);
504 Fill(Stream_Video, 0, Video_Codec_Settings, "Custom Matrix");
505 Fill(Stream_Video, 0, Video_Codec_Settings_Matrix, "Custom");
506 }
507 else
508 {
509 Fill(Stream_Video, 0, Video_Format_Settings_Matrix, "Default (MPEG)");
510 Fill(Stream_Video, 0, Video_Codec_Settings_Matrix, "Default (MPEG)");
511 }
512 if (interlaced)
513 {
514 Fill(Stream_Video, 0, Video_ScanType, "Interlaced");
515 if ((Interlaced_Top && Interlaced_Bottom) || (!Interlaced_Top && !Interlaced_Bottom))
516 Fill(Stream_Video, 0, Video_Interlacement, "Interlaced");
517 else
518 {
519 Fill(Stream_Video, 0, Video_ScanOrder, Interlaced_Top?"TFF":"BFF");
520 Fill(Stream_Video, 0, Video_Interlacement, Interlaced_Top?"TFF":"BFF");
521 }
522 }
523 else
524 {
525 Fill(Stream_Video, 0, Video_ScanType, "Progressive");
526 Fill(Stream_Video, 0, Video_Interlacement, "PPF");
527 }
528 if (!Library.empty())
529 {
530 Fill(Stream_Video, 0, Video_Encoded_Library, Library);
531 Fill(Stream_Video, 0, Video_Encoded_Library_Name, Library_Name);
532 Fill(Stream_Video, 0, Video_Encoded_Library_Version, Library_Version);
533 Fill(Stream_Video, 0, Video_Encoded_Library_Date, Library_Date);
534 }
535 for (size_t Pos=0; Pos<user_data_start_SNC_Data.size(); Pos++)
536 {
537 if (user_data_start_SNC_Data[Pos][0]==__T("CamTim"))
538 Fill(Stream_General, 0, General_Recorded_Date, Ztring().Date_From_String(user_data_start_SNC_Data[Pos][1].To_UTF8().c_str()));
539 if (user_data_start_SNC_Data[Pos][0]==__T("FrmRate"))
540 Fill(Stream_Video, 0, Video_FrameRate, user_data_start_SNC_Data[Pos][1].To_float32(), 3);
541 if (user_data_start_SNC_Data[Pos][0]==__T("TimStamp"))
542 {
543 Fill(Stream_Video, 0, Video_Delay, user_data_start_SNC_Data[Pos][1].To_int64u());
544 Fill(Stream_Video, 0, Video_Delay_Source, "Stream");
545 }
546 if (user_data_start_SNC_Data[Pos][0]==__T("CamPos") && user_data_start_SNC_Data[Pos][1].size()==16)
547 {
548 Fill(Stream_Video, 0, "Pan / Tilt / Zoom / Status", Ztring(user_data_start_SNC_Data[Pos][1].substr( 3, 4)).To_int8u(16));
549 Fill(Stream_Video, 0, "Pan / Tilt / Zoom / Status", Ztring(user_data_start_SNC_Data[Pos][1].substr( 7, 4)).To_int8u(16));
550 Fill(Stream_Video, 0, "Pan / Tilt / Zoom / Status", Ztring(user_data_start_SNC_Data[Pos][1].substr(11, 4)).To_int8u(16));
551 if (user_data_start_SNC_Data[Pos][1][15]==__T('M'))
552 Fill(Stream_Video, 0, "Pan / Tilt / Zoom / Status", __T("Move"));
553 else if (user_data_start_SNC_Data[Pos][1][15]==__T('S'))
554 Fill(Stream_Video, 0, "Pan / Tilt / Zoom / Status", __T("Stop"));
555 else
556 Fill(Stream_Video, 0, "Pan / Tilt / Zoom / Status", user_data_start_SNC_Data[Pos][1][15]);
557 }
558 if (user_data_start_SNC_Data[Pos][0]==__T("AlmEvent") && user_data_start_SNC_Data[Pos][1].size()==16)
559 Fill(Stream_Video, 0, "Alarm event", user_data_start_SNC_Data[Pos][1]);
560 }
561 if (video_object_layer_start_IsParsed && shape!=2 && !complexity_estimation_disable)
562 {
563 Fill(Stream_Video, 0, "data_partitioned", data_partitioned?"Yes":"No");
564 (*Stream_More)[Stream_Video][0](Ztring().From_Local("data_partitioned"), Info_Options)=__T("N NT");
565 if (data_partitioned)
566 {
567 Fill(Stream_Video, 0, "reversible_vlc", reversible_vlc?"Yes":"No");
568 (*Stream_More)[Stream_Video][0](Ztring().From_Local("reversible_vlc"), Info_Options)=__T("N NT");
569 }
570 }
571
572 //BVOPs
573 if (BVOP_Count_Max)
574 {
575 Ztring Format_Settings=Retrieve(Stream_Video, 0, Video_Format_Settings);
576 Format_Settings.FindAndReplace(__T("BVOP"), __T("BVOP")+Ztring::ToZtring(BVOP_Count_Max));
577 Fill(Stream_Video, 0, Video_Format_Settings, Format_Settings, true);
578 Fill(Stream_Video, 0, Video_Format_Settings_BVOP, BVOP_Count_Max, 10, true);
579 }
580
581 //Packed Bitstream
582 if (Frame_Count_InThisBlock_Max==2)
583 {
584 Fill(Stream_Video, 0, Video_MuxingMode, MediaInfoLib::Config.Language_Get("MuxingMode_PackedBitstream"));
585 Fill(Stream_Video, 0, Video_Codec_Settings, "Packed Bitstream");
586 Fill(Stream_Video, 0, Video_Codec_Settings_PacketBitStream, "Yes");
587 }
588 else
589 {
590 Fill(Stream_Video, 0, Video_Codec_Settings_PacketBitStream, "No");
591 }
592}

594//---------------------------------------------------------------------------
595void File_Mpeg4v::Streams_Finish()
596{
597 //Duration
598 if (!IsSub && Time_End_Seconds!=(int32u)-1 && Time_Begin_Seconds!=(int32u)-1)
599 {

723//---------------------------------------------------------------------------
724bool File_Mpeg4v::Header_Parser_Fill_Size()
725{
726 //Look for next Sync word
727 if (Buffer_Offset_Temp==0) //Buffer_Offset_Temp is not 0 if Header_Parse_Fill_Size() has already parsed first frames
728 Buffer_Offset_Temp=Buffer_Offset+4;
729 while (Buffer_Offset_Temp+4<=Buffer_Size
730 && CC3(Buffer+Buffer_Offset_Temp)!=0x000001)
731 {
732 Buffer_Offset_Temp+=2;
733 while(Buffer_Offset_Temp<Buffer_Size && Buffer[Buffer_Offset_Temp]!=0x00)
734 Buffer_Offset_Temp+=2;
735 if (Buffer_Offset_Temp>=Buffer_Size || Buffer[Buffer_Offset_Temp-1]==0x00)
736 Buffer_Offset_Temp--;
737 }
738
739 //Must wait more data?
740 if (Buffer_Offset_Temp+4>Buffer_Size)
741 {
742 if (FrameIsAlwaysComplete || File_Offset+Buffer_Size==File_Size)
743 Buffer_Offset_Temp=Buffer_Size; //We are sure that the next bytes are a start
744 else
745 return false;
746 }
747
748 //OK, we continue
749 Header_Fill_Size(Buffer_Offset_Temp-Buffer_Offset);
750 Buffer_Offset_Temp=0;
751 return true;
752}

754//---------------------------------------------------------------------------
755bool File_Mpeg4v::Header_Parser_QuickSearch()
756{
757 while ( Buffer_Offset+4<=Buffer_Size
758 && Buffer[Buffer_Offset ]==0x00
759 && Buffer[Buffer_Offset+1]==0x00
760 && Buffer[Buffer_Offset+2]==0x01)
761 {
762 //Getting start_code
763 int8u start_code=CC1(Buffer+Buffer_Offset+3);
764
765 //Searching start
766 if (Streams[start_code].Searching_Payload)
767 return true;
768
769 //Synchronizing
770 Buffer_Offset+=4;
771 Synched=false;
772 if (!Synchronize())
773 {
774 UnSynched_IsNotJunk=true;
775 return false;
776 }
777
778 if (Buffer_Offset+4>Buffer_Size)
779 {
780 UnSynched_IsNotJunk=true;
781 return false;
782 }
783 }
784
785 if (Buffer_Offset+3==Buffer_Size)
786 return false; //Sync is OK, but start_code is not available
787 Trusted_IsNot("MPEG-4 Visual, Synchronisation lost");
788 return Synchronize();
789}
