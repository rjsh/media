22
23class TimeCode
24{
25public:
26 //constructor/Destructor
27 TimeCode ();
28 TimeCode (int8u Hours, int8u Minutes, int8u Seconds, int8u Frames, int8u FramesPerSecond, bool DropFrame, bool MustUseSecondField=false, bool IsSecondField=false);
29 TimeCode (int64s Frames, int8u FramesPerSecond, bool DropFrame, bool MustUseSecondField=false, bool IsSecondField_=false);
30
31 //Operators
32 TimeCode &operator ++()
33 {
34 PlusOne();
35 return *this;
36 }
37 TimeCode operator ++(int)
38 {
39 PlusOne();
40 return *this;
41 }
42 TimeCode &operator --()
43 {
44 MinusOne();
45 return *this;
46 }
47 TimeCode operator --(int)
48 {
49 MinusOne();
50 return *this;
51 }
52 bool operator== (const TimeCode &tc) const
53 {
54 return Hours ==tc.Hours
55 && Minutes ==tc.Minutes
56 && Seconds ==tc.Seconds
57 && Frames ==tc.Frames
58 && FramesPerSecond ==tc.FramesPerSecond
59 && DropFrame ==tc.DropFrame
60 && MustUseSecondField ==tc.MustUseSecondField
61 && IsSecondField ==tc.IsSecondField;
62 }
63 bool operator!= (const TimeCode &tc) const
64 {
65 return Hours !=tc.Hours
66 || Minutes !=tc.Minutes
67 || Seconds !=tc.Seconds
68 || Frames !=tc.Frames
69 || FramesPerSecond !=tc.FramesPerSecond
70 || DropFrame !=tc.DropFrame
71 || MustUseSecondField !=tc.MustUseSecondField
72 || IsSecondField !=tc.IsSecondField;
73 }
74
75 //Helpers
76 bool IsValid() const
77 {
78 return FramesPerSecond?true:false;
79 }
80 void PlusOne();
81 void MinusOne();
82 string ToString();
83 int64s ToFrames();
84 int64s ToMilliseconds();
85
86public:
87 int8u Hours;
88 int8u Minutes;
89 int8u Seconds;
90 int8u Frames;
91 int8u FramesPerSecond;
92 bool DropFrame;
93 bool MustUseSecondField;
94 bool IsSecondField;
95 bool IsNegative;
96};
