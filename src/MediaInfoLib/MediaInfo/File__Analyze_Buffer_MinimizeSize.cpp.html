38 { \
39 Trusted_IsNot(); \
40 return; \
41 } \

47 Info=false; \
48 return; \
49 } \

56 return; \

94//---------------------------------------------------------------------------
95void File__Analyze::BS_Begin()
96{
97 size_t BS_Size;
98 if (Element_Offset>=Element_Size)
99 BS_Size=0;
100 else if (Buffer_Offset+Element_Size<=Buffer_Size)
101 BS_Size=(size_t)(Element_Size-Element_Offset);
102 else if (Buffer_Offset+Element_Offset<=Buffer_Size)
103 BS_Size=Buffer_Size-(size_t)(Buffer_Offset+Element_Offset);
104 else
105 BS_Size=0;
106
107 BS->Attach(Buffer+Buffer_Offset+(size_t)Element_Offset, BS_Size);
108}
109
110//---------------------------------------------------------------------------
111void File__Analyze::BS_Begin_LE()
112{
113 size_t BS_Size;
114 if (Buffer_Offset+Element_Size<=Buffer_Size)
115 BS_Size=(size_t)(Element_Size-Element_Offset);
116 else if (Buffer_Offset+Element_Offset<=Buffer_Size)
117 BS_Size=Buffer_Size-(size_t)(Buffer_Offset+Element_Offset);
118 else
119 BS_Size=0;
120
121 BT->Attach(Buffer+Buffer_Offset+(size_t)Element_Offset, BS_Size);
122}

127 BS->Byte_Align();
128 Element_Offset+=BS->Offset_Get();
129 BS->Attach(NULL, 0);

136 Element_Offset+=BT->Offset_Get();
137 BT->Attach(NULL, 0);
138}

218//---------------------------------------------------------------------------
219// Big Endian - float 16 bits
220// TODO: remove it when Linux version of ZenLib is updated
221float32 BigEndian2float16corrected(const char* Liste)
222{
223 //sign 1 bit
224 //exponent 5 bit
225 //significand 10 bit
226
227 //Retrieving data
228 int16u Integer=BigEndian2int16u(Liste);
229
230 //Retrieving elements
231 bool Sign =(Integer&0x8000)?true:false;
232 int32u Exponent=(Integer>>10)&0xFF;
233 int32u Mantissa= Integer&0x03FF;
234
235 //Some computing
236 if (Exponent==0 || Exponent==0xFF)
237 return 0; //These are denormalised numbers, NANs, and other horrible things
238 Exponent-=0x0F; //Bias
239 float64 Answer=(((float64)Mantissa)/8388608+1.0)*std::pow((float64)2, (int)Exponent); //(1+Mantissa) * 2^Exponent
240 if (Sign)
241 Answer=-Answer;
242
243 return (float32)Answer;
244}
245inline float32 BigEndian2float16corrected (const int8u* List) {return BigEndian2float16corrected ((const char*)List);}

279//---------------------------------------------------------------------------
280void File__Analyze::Get_BFP4_(int8u Bits, float32 &Info)
281{
282 INTEGRITY_SIZE_ATLEAST_INT(4);
283 BS_Begin();
284 int32s Integer=(int32s)BS->Get4(Bits);
285 int32u Fraction=BS->Get4(32-Bits);
286 BS_End();
287 if (Integer>=(1<<Bits)/2)
288 Integer-=1<<Bits;
289 Info=Integer+((float32)Fraction)/(1<<(32-Bits));
290}

668//***************************************************************************
669// EBML
670//***************************************************************************
671
672//---------------------------------------------------------------------------
673void File__Analyze::Get_EB(int64u &Info)
674{
675 //Element size
676 INTEGRITY_SIZE_ATLEAST_INT(1);
677 if (Buffer[Buffer_Offset+Element_Offset]==0xFF)
678 {
679 Info=File_Size-(File_Offset+Buffer_Offset+Element_Offset);
680 Element_Offset++;
681 return;
682 }
683 int8u Size=0;
684 int8u Size_Mark=0;
685 BS_Begin();
686 while (Size_Mark==0 && BS->Remain() && Size<=8)
687 {
688 Size++;
689 Peek_S1(Size, Size_Mark);
690 }
691
692 //Integrity
693 if (!BS->Remain() || Size>8)
694 {
695 if (Size>8)
696 {
697 //Element[Element_Level].IsComplete=true; //If it is in a header
698 Trusted_IsNot("EBML integer parsing error");
699 }
700 Info=0;
701 return;
702 }
703 BS_End();
704 if (File_Offset+Buffer_Offset+Element_Offset>=Element[Element_Level].Next)
705 {
706 //Element[Element_Level].IsComplete=true; //If it is in a header
707 Trusted_IsNot("Not enough place to have an EBML");
708 Info=0;
709 return; //Not enough space
710 }
711 INTEGRITY_SIZE_ATLEAST_INT(Size);
712
713 //Element Name
714 switch (Size)
715 {
716 case 1 : {
717 int8u Element_Name;
718 Peek_B1 (Element_Name);
719 Info=Element_Name&0x7F; //Keep out first bit
720 }
721 break;
722 case 2 : {
723 int16u Element_Name;
724 Peek_B2(Element_Name);
725 Info=Element_Name&0x3FFF; //Keep out first bits
726 }
727 break;
728 case 3 : {
729 int32u Element_Name;
730 Peek_B3(Element_Name);
731 Info=Element_Name&0x1FFFFF; //Keep out first bits
732 }
733 break;
734 case 4 : {
735 int32u Element_Name;
736 Peek_B4(Element_Name);
737 Info=Element_Name&0x0FFFFFFF; //Keep out first bits
738 }
739 break;
740 case 5 : {
741 int64u Element_Name;
742 Peek_B5(Element_Name);
743 Info=Element_Name&0x07FFFFFFFFLL; //Keep out first bits
744 }
745 break;
746 case 6 : {
747 int64u Element_Name;
748 Peek_B6(Element_Name);

769//---------------------------------------------------------------------------
770void File__Analyze::Get_ES(int64s &Info)
771{
772 //Element size
773 INTEGRITY_SIZE_ATLEAST_INT(1);
774 int8u Size=0;
775 int8u Size_Mark=0;
776 BS_Begin();
777 while (Size_Mark==0 && BS->Remain() && Size<=8)
778 {
779 Size++;
780 Peek_S1(Size, Size_Mark);

783 //Integrity
784 if (!BS->Remain() || Size>8)

837 case 6 : {
838 int64u Element_Name;
839 Peek_B6(Element_Name);
840 Info=(Element_Name&0x03FFFFFFFFFFLL)-0x01FFFFFFFFFFLL; //Keep out first bits and sign
841 }
842 break;

844 int64u Element_Name;
845 Peek_B7(Element_Name);

847 }

1095//---------------------------------------------------------------------------
1096void File__Analyze::Get_VL_Prepare(vlc_fast &Vlc)
1097{
1098 Vlc.Array=new int8u[((size_t)1)<<Vlc.Size];
1099 Vlc.BitsToSkip=new int8u[((size_t)1)<<Vlc.Size];
1100 memset(Vlc.Array, 0xFF, ((size_t)1)<<Vlc.Size);
1101 int8u Increment=0;
1102 int8u Pos=0;
1103 for(; ; Pos++)
1104 {
1105 if (Vlc.Vlc[Pos].bit_increment==255)
1106 break;
1107 Increment+=Vlc.Vlc[Pos].bit_increment;
1108 size_t Value = ((size_t)Vlc.Vlc[Pos].value) << (Vlc.Size - Increment);
1109 size_t ToFill_Size=((size_t)1)<<(Vlc.Size-Increment);
1110 for (size_t ToFill_Pos=0; ToFill_Pos<ToFill_Size; ToFill_Pos++)
1111 {
1112 Vlc.Array[Value+ToFill_Pos]=Pos;
1113 Vlc.BitsToSkip[Value+ToFill_Pos]=Increment;
1114 }
1115 }
1116 for (size_t Pos2=0; Pos2<(((size_t)1)<<Vlc.Size); Pos2++)
1117 if (Vlc.Array[Pos2]==(int8u)-1)
1118 {
1119 Vlc.Array[Pos2]=Pos;
1120 Vlc.BitsToSkip[Pos2]=(int8u)-1;
1121 }
1122}
