31TimeCode::TimeCode ()
32: Hours((int8u)-1),
33 Minutes((int8u)-1),
34 Seconds((int8u)-1),
35 Frames((int8u)-1),
36 FramesPerSecond(0),
37 DropFrame(false),
38 MustUseSecondField(false),
39 IsSecondField(false),
40 IsNegative(false)
41{
42}

44//---------------------------------------------------------------------------
45TimeCode::TimeCode (int8u Hours_, int8u Minutes_, int8u Seconds_, int8u Frames_, int8u FramesPerSecond_, bool DropFrame_, bool MustUseSecondField_, bool IsSecondField_)
46: Hours(Hours_),
47 Minutes(Minutes_),
48 Seconds(Seconds_),
49 Frames(Frames_),
50 FramesPerSecond(FramesPerSecond_),
51 DropFrame(DropFrame_),
52 MustUseSecondField(MustUseSecondField_),
53 IsSecondField(IsSecondField_),
54 IsNegative(false)
55{
56}

58//---------------------------------------------------------------------------
59TimeCode::TimeCode (int64s Frames_, int8u FramesPerSecond_, bool DropFrame_, bool MustUseSecondField_, bool IsSecondField_)
60: FramesPerSecond(FramesPerSecond_),
61 DropFrame(DropFrame_),
62 MustUseSecondField(MustUseSecondField_),
63 IsSecondField(IsSecondField_)
64{
65 if (!FramesPerSecond_)
66 {
67 Frames = 0;
68 Seconds = 0;
69 Minutes = 0;
70 Hours = 0;
71 IsNegative = true; //Forcing a weird display
72 return;
73 }
74
75 if (Frames_<0)
76 {
77 IsNegative=true;
78 Frames_=-Frames_;
79 }
80 else
81 IsNegative=false;
82
83 int8u Dropped=0;
84 if (DropFrame_)
85 {
86 Dropped=2;
87 if (FramesPerSecond_>30)
88 Dropped+=2;
89 if (FramesPerSecond_>60)
90 Dropped+=2;
91 if (FramesPerSecond_>90)
92 Dropped+=2;
93 if (FramesPerSecond_>120)
94 Dropped+=2;
95 }
96
97 int64u Minutes_Tens = Frames_/(600*FramesPerSecond_-Dropped*9); //Count of 10 minutes
98 int64u Minutes_Units = (Frames_-Minutes_Tens*(600*FramesPerSecond_-Dropped*9))/(60*FramesPerSecond_-Dropped);
99
100 Frames_ += 9*Dropped*Minutes_Tens+Dropped*Minutes_Units;
101 if (Minutes_Units && ((Frames_/FramesPerSecond_)%60)==0 && (Frames_%FramesPerSecond_)<Dropped) // If Minutes_Tens is not 0 (drop) but count of remaining seconds is 0 and count of remaining frames is less than 2, 1 additional drop was actually counted, removing it
102 Frames_-=Dropped;
103
104 Frames = Frames_ % FramesPerSecond_;
105 Seconds = (Frames_ / FramesPerSecond_) % 60;
106 Minutes = ((Frames_ / FramesPerSecond_) / 60) % 60;
107 Hours = (((Frames_ / FramesPerSecond_) / 60) / 60) % 24;
108}

114//---------------------------------------------------------------------------
115void TimeCode::PlusOne()
116{
117 //TODO: negative values
118
119 if (FramesPerSecond==0)
120 return;
121 if (MustUseSecondField)
122 {
123 if (IsSecondField)
124 {
125 Frames++;
126 IsSecondField=false;
127 }
128 else
129 IsSecondField=true;
130 }
131 else
132 Frames++;
133 if (Frames>=FramesPerSecond)
134 {
135 Seconds++;
136 Frames=0;
137 if (Seconds>=60)
138 {
139 Seconds=0;
140 Minutes++;
141
142 if (DropFrame && Minutes%10)
143 Frames=2; //frames 0 and 1 are dropped for every minutes except 00 10 20 30 40 50
144
145 if (Minutes>=60)
146 {
147 Minutes=0;
148 Hours++;
149 if (Hours>=24)
150 {
151 Hours=0;
152 }
153 }
154 }
155 }
156}

158//---------------------------------------------------------------------------
159void TimeCode::MinusOne()
160{
161 //TODO: negative values
162
163 if (FramesPerSecond==0)
164 return;
165 if (MustUseSecondField && IsSecondField)
166 IsSecondField=false;
167 else
168 {
169 if (Frames==0 || (DropFrame && Minutes%10 && Frames<=2))
170 {
171 Frames=FramesPerSecond;
172 if (Seconds==0)
173 {
174 Seconds=60;
175 if (Minutes==0)
176 {
177 Minutes=60;
178 if (Hours==0)
179 Hours=24;
180 Hours--;
181 }
182 Minutes--;
183 }
184 Seconds--;
185 }
186 Frames--;
187
188 if (MustUseSecondField)
189 IsSecondField=true;
190 }
191}

193//---------------------------------------------------------------------------
194string TimeCode::ToString()
195{
196 if (!FramesPerSecond)
197 return string();
198
199 string TC;
200 if (IsNegative)
201 TC+='-';
202 TC+=('0'+Hours/10);
203 TC+=('0'+Hours%10);
204 TC+=':';
205 TC+=('0'+Minutes/10);
206 TC+=('0'+Minutes%10);
207 TC+=':';
208 TC+=('0'+Seconds/10);
209 TC+=('0'+Seconds%10);
210 TC+=DropFrame?';':':';
211 TC+=('0'+(Frames*(MustUseSecondField?2:1)+(IsSecondField?1:0))/10);
212 TC+=('0'+(Frames*(MustUseSecondField?2:1)+(IsSecondField?1:0))%10);
213
214 return TC;
215}

217//---------------------------------------------------------------------------
218int64s TimeCode::ToFrames()
219{
220 if (!FramesPerSecond)
221 return 0;
222
223 int64s TC=(int64s(Hours) *3600
224 + int64s(Minutes) * 60
225 + int64s(Seconds) )*int64s(FramesPerSecond)
226 + int64s(Frames);
227
228 if (DropFrame)
229 {
230 TC-= int64s(Hours) *108
231 + (int64s(Minutes)/10)*18
232 + (int64s(Minutes)%10)*2;
233 }
234
235 TC*=(MustUseSecondField?2:1);
236 TC+=(IsSecondField?1:0);
237
238 return IsNegative?-TC:TC;
239}
240
241//---------------------------------------------------------------------------
242int64s TimeCode::ToMilliseconds()
243{
244 if (!FramesPerSecond)
245 return 0;
246
247 int64s MS=(int64s(Hours) *3600
248 + int64s(Minutes) * 60
249 + int64s(Seconds) )*1000
250 + float64_int64s((float64(Frames*(MustUseSecondField?2:1)+(IsSecondField?1:0)))*1000/(FramesPerSecond*(MustUseSecondField?2:1)));
251
252 return IsNegative?-MS:MS;
253}
