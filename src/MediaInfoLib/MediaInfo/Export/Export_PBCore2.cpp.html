48Ztring PBCore2_MediaType(MediaInfo_Internal &MI)
49{
50 if (MI.Count_Get(Stream_Video))
51 return __T("Moving Image");
52 else if (MI.Count_Get(Stream_Audio))
53 return __T("Sound");
54 else if (MI.Count_Get(Stream_Image))
55 return __T("Static Image");
56 else if (MI.Count_Get(Stream_Text))
57 return __T("Text");
58 else
59 return Ztring();
60}

81void PBCore2_Transform(Ztring &ToReturn, MediaInfo_Internal &MI, stream_t StreamKind, size_t StreamPos)
82{
83 //Menu: only if TimeCode
84 if (StreamKind==Stream_Menu && MI.Get(Stream_Menu, StreamPos, Menu_Format)!=__T("TimeCode"))
85 return;
86
87 //essenceTrackType
88 Ztring essenceTrackType;
89 switch (StreamKind)
90 {
91 case Stream_Video:
92 essenceTrackType=__T("Video");
93 break;
94 case Stream_Audio:
95 essenceTrackType=__T("Audio");
96 break;
97 case Stream_Image:
98 essenceTrackType=__T("Image");
99 break;
100 case Stream_Text:
101 {
102 Ztring Format=MI.Get(Stream_Text, StreamPos, Text_Format);
103 if (Format==__T("EIA-608") || Format==__T("EIA-708"))
104 essenceTrackType=__T("CC");
105 else
106 essenceTrackType=__T("Text");
107 }
108 break;
109 case Stream_Menu:
110 if (MI.Get(Stream_Menu, StreamPos, Menu_Format)==__T("TimeCode"))

112 essenceTrackType=__T("TimeCode");
113 break;
114 }
115 else
116 return; //Not supported
117 default: return; //Not supported

123 ToReturn+=essenceTrackType;
124 ToReturn+=__T("</essenceTrackType>\n");

143 ToReturn+=__T("</essenceTrackIdentifier>\n");
144 }

226 ToReturn+=__T("</essenceTrackSamplingRate>\n");
227 }

252 ToReturn+=__T("</essenceTrackAspectRatio>\n");
253 }

255 //essenceTrackDuration

258 ToReturn+=__T("\t\t<essenceTrackDuration>");
259 ToReturn+=MI.Get(StreamKind, StreamPos, __T("Duration_String3"));

261 }

265 {
266 ToReturn+=__T("\t\t<essenceTrackLanguage>");
267 ToReturn+=MediaInfoLib::Config.Iso639_2_Get(MI.Get(StreamKind, StreamPos, __T("Language")));
268 ToReturn+=__T("</essenceTrackLanguage>\n");

274 MI.Get(StreamKind, StreamPos, Pos, Info_Name).find(__T("String"))==std::string::npos &&

281 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Codec") &&

283 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Codec_Settings") &&
284 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Codec_Settings_CABAC") &&
285 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Codec_Settings_RefFrames") &&
286 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Codec/CC") &&
287 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Codec/Family") &&
288 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Codec/Info") &&
289 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Codec/Url") &&
290 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("CodecID") &&
291 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("CodecID/Info") &&
292 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("CodecID/Url") &&
293 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Codec_Settings_Floor") &&
294 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Colorimetry") &&
295 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Count") &&
296 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("DisplayAspectRatio") &&
297 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Duration") &&
298 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Encoded_Date") &&
299 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Encoded_Library") &&
300 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Format") &&
301 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Format/Info") &&
302 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Format/Url") &&
303 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Format_Commercial") &&
304 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Format_Profile") &&
305 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Format_Version") &&
306 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("FrameRate") &&
307 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("FrameRate_Mode") &&
308 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Height") &&
309 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("ID") &&
310 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("InternetMediaType") &&
311 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Language") &&
312 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Resolution") &&
313 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("SamplingRate") &&
314 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Standard") &&
315 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("StreamCount") &&
316 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("StreamKind") &&
317 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("StreamKindID") &&
318 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("StreamKindPos") &&
319 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("StreamOrder") &&
320 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("StreamSize_Proportion") &&
321 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Tagged_Date") &&
322 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("UniqueID") &&
323 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Video_DisplayAspectRatio") &&
324 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Video_FrameRate") &&
325 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Video_FrameRate_Mode") &&
326 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Video_Height") &&
327 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Video_Standard") &&
328 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Video_Width") &&
329 MI.Get(StreamKind, StreamPos, Pos, Info_Name)!=__T("Width")
330 )
331 {
332 ToReturn+=__T("\t\t<essenceTrackAnnotation");
333 ToReturn+=__T(" annotationType=\"");
334 ToReturn+=MI.Get(StreamKind, StreamPos, Pos, Info_Name);
335 ToReturn+=__T("\">");
336 ToReturn+=MI.Get(StreamKind, StreamPos, Pos);
337 ToReturn+=__T("</essenceTrackAnnotation>\n");
338 }
339 ToReturn+=__T("\t</instantiationEssenceTrack>\n");
340}

342//---------------------------------------------------------------------------
343Ztring Export_PBCore2::Transform(MediaInfo_Internal &MI)
344{
345 //Current date/time is ISO format
346 time_t Time=time(NULL);
347 Ztring TimeS; TimeS.Date_From_Seconds_1970((int32u)Time);
348 TimeS.FindAndReplace(__T("UTC "), __T(""));
349 TimeS.FindAndReplace(__T(" "), __T("T"));
350 TimeS+=__T('Z');
351
352 Ztring ToReturn;
353 ToReturn+=__T("<?xml version=\"1.0\" encoding=\"UTF-8\"?>\n");
354 ToReturn+=__T("<pbcoreInstantiationDocument xsi:schemaLocation=\"http://www.pbcore.org/PBCore/PBCoreNamespace.html http://pbcore.org/xsd/pbcore-2.0.xsd\" xmlns:xsi=\"http://www.w3.org/2001/XMLSchema-instance\" xmlns=\"http://www.pbcore.org/PBCore/PBCoreNamespace.html\">\n");
355 ToReturn+=__T("<!-- Generated at ")+TimeS+__T(" by ")+MediaInfoLib::Config.Info_Version_Get()+__T(" -->\n");
356
357 //instantiationIdentifier
358 ToReturn+=__T("\t<instantiationIdentifier source=\"File Name\">");
359 ToReturn+=MI.Get(Stream_General, 0, General_FileName);
360 if (!MI.Get(Stream_General, 0, General_FileExtension).empty())
361 {
362 ToReturn+=__T(".");
363 ToReturn+=MI.Get(Stream_General, 0, General_FileExtension);
364 }
365 ToReturn+=__T("</instantiationIdentifier>\n");
366
367 // need to figure out how to get to non-internally-declared-values
368 //if (!MI.Get(Stream_General, 0, General_Media/UUID).empty())
369 //{
370 // ToReturn+=__T("\t<instantiationIdentifier source=\"Media UUID\">")+MI.Get(Stream_General, 0, General_Media/UUID)+__T("</instantiationIdentifier>\n");
371 //}
372
373 //instantiationDates
374 //dateIssued
375 if (!MI.Get(Stream_General, 0, General_Recorded_Date).empty())
376 {
377 Ztring dateIssued=MI.Get(Stream_General, 0, General_Recorded_Date);
378 dateIssued.FindAndReplace(__T("UTC"), __T(""));
379 dateIssued.FindAndReplace(__T(" "), __T("T"));
380 dateIssued+=__T('Z');
381 ToReturn+=__T("\t<instantiationDate dateType=\"issued\">");
382 ToReturn+=dateIssued+__T("</instantiationDate>\n");
383 }
384
385 //dateFileModified
386 if (!MI.Get(Stream_General, 0, General_File_Modified_Date).empty())
387 {
388 Ztring dateModified=MI.Get(Stream_General, 0, General_File_Modified_Date);
389 dateModified.FindAndReplace(__T("UTC "), __T(""));
390 dateModified.FindAndReplace(__T(" "), __T("T"));
391 dateModified+=__T('Z');
392 ToReturn+=__T("\t<instantiationDate dateType=\"file modification\">");
393 ToReturn+=dateModified+__T("</instantiationDate>\n");
394 }
395
396 //dateEncoder
397 if (!MI.Get(Stream_General, 0, General_Encoded_Date).empty())
398 {
399 Ztring dateEncoded=MI.Get(Stream_General, 0, General_Encoded_Date);
400 dateEncoded.FindAndReplace(__T("UTC "), __T(""));
401 dateEncoded.FindAndReplace(__T(" "), __T("T"));
402 dateEncoded+=__T('Z');
403 ToReturn+=__T("\t<instantiationDate dateType=\"encoded\">");
404 ToReturn+=dateEncoded+__T("</instantiationDate>\n");
405 }
406
407 //dateTagged
408 if (!MI.Get(Stream_General, 0, General_Tagged_Date).empty())
409 {
410 Ztring dateTagged=MI.Get(Stream_General, 0, General_Tagged_Date);
411 dateTagged.FindAndReplace(__T("UTC "), __T(""));
412 dateTagged.FindAndReplace(__T(" "), __T("T"));
413 dateTagged+=__T('Z');
414 ToReturn+=__T("\t<instantiationDate dateType=\"tagged\">");
415 ToReturn+=dateTagged+__T("</instantiationDate>\n");
416 }
417
418 //formatDigital
419 if (!MI.Get(Stream_General, 0, General_InternetMediaType).empty())

421 ToReturn+=__T("\t<instantiationDigital>");

438 ToReturn+=__T("</instantiationDigital>\n");
439 }

441 //formatLocation
442 ToReturn+=__T("\t<instantiationLocation>");
443 ToReturn+=MI.Get(Stream_General, 0, General_CompleteName);
444 ToReturn+=__T("</instantiationLocation>\n");
445
446 //formatMediaType
447 if (!PBCore2_MediaType(MI).empty())
448 {
449 ToReturn+=__T("\t<instantiationMediaType>");
450 ToReturn+=PBCore2_MediaType(MI);
451 ToReturn+=__T("</instantiationMediaType>\n");
452 }
453
454 //formatFileSize
455 if (!MI.Get(Stream_General, 0, General_FileSize).empty())
456 {
457 ToReturn+=__T("\t<instantiationFileSize");
458 ToReturn+=__T(" unitsOfMeasure=\"bytes\"");
459 ToReturn+=__T(">");
460 ToReturn+=MI.Get(Stream_General, 0, General_FileSize);
461 ToReturn+=__T("</instantiationFileSize>\n");

463
464 //formatTimeStart
465 if (!MI.Get(Stream_Video, 0, Video_Delay_Original_String3).empty())
466 {
467 ToReturn+=__T("\t<instantiationTimeStart>");
468 ToReturn+=MI.Get(Stream_Video, 0, Video_Delay_Original_String3);
469 ToReturn+=__T("</instantiationTimeStart>\n");
470 }
471 else if (!MI.Get(Stream_Video, 0, Video_Delay_String3).empty())
472 {
473 ToReturn+=__T("\t<instantiationTimeStart>");
474 ToReturn+=MI.Get(Stream_Video, 0, Video_Delay_String3);
475 ToReturn+=__T("</instantiationTimeStart>\n");
476 }
477
478 //formatDuration
479 if (!MI.Get(Stream_General, 0, General_Duration_String3).empty())
480 {
481 ToReturn+=__T("\t<instantiationDuration>");
482 ToReturn+=MI.Get(Stream_General, 0, General_Duration_String3);
483 ToReturn+=__T("</instantiationDuration>\n");
484 }
485
486 //formatDataRate
487 if (!MI.Get(Stream_General, 0, General_OverallBitRate).empty())
488 {
489 ToReturn+=__T("\t<instantiationDataRate");
490 ToReturn+=__T(" unitsOfMeasure=\"bits/second\"");
491 if (!MI.Get(Stream_General, 0, General_OverallBitRate_Mode).empty())
492 {
493 ToReturn+=__T(" annotation=\"");
494 ToReturn+=MI.Get(Stream_General, 0, General_OverallBitRate_Mode);
495 ToReturn+=__T("\"");
496 }
497 ToReturn+=__T(">");
498 ToReturn+=MI.Get(Stream_General, 0, General_OverallBitRate);
499 ToReturn+=__T("</instantiationDataRate>\n");
500 }
501
502 //formatTracks
503 ToReturn+=__T("\t<instantiationTracks>")+Ztring::ToZtring(MI.Count_Get(Stream_Video)+MI.Count_Get(Stream_Audio)+MI.Count_Get(Stream_Image)+MI.Count_Get(Stream_Text))+__T("</instantiationTracks>\n");
504
505 //Streams
506 for (size_t StreamKind=Stream_General+1; StreamKind<Stream_Max; StreamKind++)
507 for (size_t StreamPos=0; StreamPos<MI.Count_Get((stream_t)StreamKind); StreamPos++)
508 PBCore2_Transform(ToReturn, MI, (stream_t)StreamKind, StreamPos);
509
510 //instantiationAnnotations
511 for (size_t Pos=0; Pos<MI.Count_Get(Stream_General, 0); Pos++)
512 if (
513 MI.Get(Stream_General, 0, Pos, Info_Name).find(__T("String"))==std::string::npos &&
514 !MI.Get(Stream_General, 0, Pos).empty() &&
515 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Count") &&
516 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("StreamCount") &&
517 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("StreamKind") &&
518 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("StreamKindID") &&
519 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("UniqueID") &&
520 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Format/Url") &&
521 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Format_Commercial") &&
522 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Codec") &&
523 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Codec/Url") &&
524 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("CodecID") &&
525 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("CodecID/Url") &&
526 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Encoded_Date") &&
527 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Tagged_Date") &&
528 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Codec/Url") &&
529 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Duration") &&
530 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("StreamSize_Proportion") &&
531 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("VideoCount") &&
532 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("AudioCount") &&
533 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("TextCount") &&
534 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("MenuCount") &&
535 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Video_Format_List") &&
536 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Video_Format_WithHint_List") &&
537 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Video_Codec_List") &&
538 MI.Get(Stream_General, 0, Pos, Info_Name)!=__T("Video_Language_List") &&
