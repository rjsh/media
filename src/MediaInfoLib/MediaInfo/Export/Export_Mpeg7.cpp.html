42Ztring Mpeg7_XML_Encode(const Ztring& Data)
43{
44 Ztring Result;
45 wstring::size_type Pos;
46 for (Pos = 0; Pos<Data.size(); Pos++)
47 {
48 switch (Data[Pos])
49 {
50 case __T('"'): Result += __T("&quot;"); break;
51 case __T('&'): Result += __T("&amp;"); break;
52 case __T('\''): Result += __T("&apos;"); break;
53 case __T('<'): Result += __T("&lt;"); break;
54 case __T('>'): Result += __T("&lg;"); break;
55 default: Result += Data[Pos];
56 }
57 }
58 return Result;
59}
