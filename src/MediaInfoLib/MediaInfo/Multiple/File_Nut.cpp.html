54//---------------------------------------------------------------------------
55void File_Nut::FileHeader_Parse()
56{
57 //Parsing
58 Element_Begin1("Nut header");
59 std::string file_id_string;
60 int8u file_id_string_zero;
61 Get_String(24, file_id_string, "file_id_string");
62 Get_B1 (file_id_string_zero, "file_id_string zero");
63 Element_End0();
64
65 FILLING_BEGIN();
66 //Integrity
67 if (file_id_string!="nut/multimedia container" || file_id_string_zero)
68 {
69 Reject("Nut");
70 return;
71 }
72
73 //Filling
74 Accept("Nut");
75
76 Fill(Stream_General, 0, General_Format, "Nut");
77 FILLING_END();
78}

84//---------------------------------------------------------------------------
85void File_Nut::Header_Parse()
86{
87 //Parsing
88 int8u N;
89 Peek_B1(N);
90 if (N==0x4E) //'N'
91 {
92 //Header
93 int64u startcode, forward_ptr;
94 Get_B8(startcode, "startcode");
95 Get_VS(forward_ptr, "forward_ptr");
96 if (forward_ptr>4096)
97 Skip_B4( "header_checksum");
98
99 Header_Fill_Code(startcode, Ztring().From_Number(startcode, 16)); //Quick filling for CC8 with text
100 Header_Fill_Size(Element_Offset+forward_ptr); //4 for cheksum
101
102 }
103 else
104 {
105 //Frame
106 Header_Fill_Code(0, "Frame");
107 Header_Fill_Size(0);
108 Finish();
109 }
110}

112//---------------------------------------------------------------------------
113void File_Nut::Data_Parse()
114{
115 #define ELEMENT_CASE(_NAME) \
116 case Elements::_NAME : _NAME(); break;
117
118 //Parsing
119 if (Element_Size < 4)
120 {
121 Skip_XX(Element_Size, "Unknown");
122 return;
123 }
124 Element_Size-=4;
125 #ifndef __BORLANDC__
126 switch (Element_Code)
127 #else //__BORLANDC__
128 switch (Element_Code&0xFFFFFFFF) //Borland does not like int64u for const?
129 #endif //__BORLANDC__
130 {
131 ELEMENT_CASE(main);
132 ELEMENT_CASE(stream);
133 ELEMENT_CASE(syncpoint);
134 ELEMENT_CASE(index);
135 ELEMENT_CASE(info);
136 default : Skip_XX(Element_Size, "Data");
137 }
138
139 Element_Size+=4;
140 if (Element_Offset+4!=Element_Size)
141 Skip_XX(Element_Size - 4 - Element_Offset, "Unknown");
142 Skip_B4( "cheksum");
143}

149//---------------------------------------------------------------------------
150void File_Nut::main()
151{
152 Element_Name("main");
153
154 //Parsing
155 int64u time_base_count;
156 Skip_VS( "version");
157 Skip_VS( "stream_count");
158 Skip_VS( "max_distance");
159 Get_VS (time_base_count, "time_base_count");
160 for(int64u i=0; i<time_base_count; i++)
161 {
162 Skip_VS( "time_base_num");
163 Skip_VS( "time_base_denom");
164 //time_base[i]= time_base_num/time_base_denom
165 }
166 int64u tmp_mul=1, tmp_stream=0;
167 int64s tmp_pts=0;
168 for(int16u i=0; i<256;)
169 {
170 int64u tmp_fields, tmp_size, tmp_res, count;
171 Skip_VS( "tmp_flag");
172 Get_VS (tmp_fields, "tmp_fields");
173 if(tmp_fields>0)
174 Skip_VS( "tmp_pts"); //TODO: signed
175 if(tmp_fields>1)
176 Skip_VS( "tmp_mul");
177 if(tmp_fields>2)
178 Skip_VS( "tmp_stream");
179 if(tmp_fields>3)
180 Get_VS (tmp_size, "tmp_size");
181 else
182 tmp_size=0;
183 if(tmp_fields>4)
184 Get_VS (tmp_res, "tmp_res");
185 else
186 tmp_res=0;
187 if(tmp_fields>5)
188 Get_VS(count, "count");
189 else
190 count=tmp_mul-tmp_size;
191 for(int64u j=6; j<tmp_fields; j++)
192 Skip_VS( "tmp_reserved[i]");
193
194 for(int64u j=0; j<count && i<256; j++, i++)
195 {
196 if (i == 'N')
197 {
198 //flags[i]= FLAG_INVALID;
199 j--;
200 continue;
201 }
202 //flags[i]= tmp_flag;
203 //stream_id[i]= tmp_stream;
204 //data_size_mul[i]= tmp_mul;
205 //data_size_lsb[i]= tmp_size + j;
206 //pts_delta[i]= tmp_pts;
207 //reserved_count[i]= tmp_res;
208 }
209 }
210}

212//---------------------------------------------------------------------------
213void File_Nut::stream()
214{
215 Element_Name("stream");
216
217 //Parsing
218 int64u stream_class, fourcc_length, codec_specific_data_length;
219 Skip_VS( "stream_id");
220 Get_VS (stream_class, "stream_class");
221 Get_VS (fourcc_length, "fourcc length");
222 switch (fourcc_length)
223 {
224 case 2 : Skip_C2( "fourcc"); break;
225 case 4 : Skip_C4( "fourcc"); break;
226 default: Skip_XX(fourcc_length, "fourcc");
227 }
228 Skip_VS( "time_base_id");
229 Skip_VS( "msb_pts_shift");
230 Skip_VS( "max_pts_distance");
231 Skip_VS( "decode_delay");
232 Skip_VS( "stream_flags");
233 Get_VS (codec_specific_data_length, "codec_specific_data length");
234 Skip_XX(codec_specific_data_length, "codec_specific_data");
235 switch (stream_class)
236 {
237 case 0 : //video
238 {
239 Skip_VS( "width");
240 Skip_VS( "height");
241 Skip_VS( "sample_width");
242 Skip_VS( "sample_height");
243 Skip_VS( "colorspace_type");
244 }
245 break;
246 case 1 : //audio
247 {
248 Skip_VS( "samplerate_num");
249 Skip_VS( "samplerate_denom");
250 Skip_VS( "channel_count");
251 }
252 break;
253 case 2 : //subtitles
254 {
255 }
256 break;
257 case 3 : //userdata
258 {
259 }
260 break;
261 default: ;
262 }
263
264 if (Element_Offset!=Element_Size)
265 Skip_XX(Element_Size - Element_Offset, "Data");
266}
