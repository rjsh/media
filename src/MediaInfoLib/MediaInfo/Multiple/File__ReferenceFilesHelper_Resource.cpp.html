83void resource::UpdateFileName(const Ztring& OldFileName, const Ztring& NewFileName)
84{
85 size_t FileNames_Size=FileNames.size();
86 for (size_t Pos=0; Pos<FileNames_Size; Pos++)
87 if (FileNames[Pos]==OldFileName)
88 FileNames[Pos]=NewFileName;
89}
