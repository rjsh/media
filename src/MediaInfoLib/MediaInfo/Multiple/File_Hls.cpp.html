90//***************************************************************************
91// Buffer - File header
92//***************************************************************************
93
94//---------------------------------------------------------------------------
95bool File_Hls::FileHeader_Begin()
96{
97 //Element_Size
98 if (File_Size>1024*1024 || File_Size<10)
99 {
100 Reject("HLS");
101 return false; //HLS files are not big
102 }
103
104 if (Buffer_Size<File_Size)
105 return false; //Wait for complete file
106
107 Ztring Document; Document.From_UTF8((char*)Buffer, Buffer_Size);
108 ZtringList Lines;
109 size_t LinesSeparator_Pos=Document.find_first_of(__T("\r\n"));
110 if (LinesSeparator_Pos>File_Size-1)
111 {
112 Reject("HLS");
113 return false;
114 }
115 Ztring LinesSeparator;
116 if (Document[LinesSeparator_Pos]==__T('\r') && LinesSeparator_Pos+1<Document.size() && Document[LinesSeparator_Pos+1]==__T('\n'))
117 LinesSeparator=__T("\r\n");
118 else if (Document[LinesSeparator_Pos]==__T('\r'))
119 LinesSeparator=__T("\r");
120 else if (Document[LinesSeparator_Pos]==__T('\n'))
121 LinesSeparator=__T("\n");
122 else
123 {
124 Reject("HLS");
125 return false;
126 }
127 Lines.Separator_Set(0, LinesSeparator);
128 Lines.Write(Document);
129
130 if (Lines(0)!=__T("#EXTM3U"))
131 {
132 Reject("HLS");
133 return false;
134 }
135
136 Accept("HLS");
137 Fill(Stream_General, 0, General_Format, "HLS");
138
139 ReferenceFiles=new File__ReferenceFilesHelper(this, Config);
140 if (!IsSub)
141 ReferenceFiles->ContainerHasNoId=true;
142
143 sequence* Sequence=new sequence;
144
145 bool IsGroup=false;
146 Ztring PreviousFile;
147 for (size_t Line=0; Line<Lines.size(); Line++)
148 {
149 if (!Lines[Line].empty())
150 {
151 if (Lines[Line].find(__T("#EXT-X-KEY:"))==0)
152 {
153 ZtringListList List;
154 List.Separator_Set(0, __T(","));
155 List.Separator_Set(1, __T("="));
156 List.Write(Lines[Line].substr(11, string::npos));
157 for (size_t Pos=0; Pos<List.size(); ++Pos)
158 {
159 if (List[Pos](0)==__T("METHOD"))
160 {
161 if (List[Pos](1).find(__T("AES-128"))==0)
162 {
163 Fill(Stream_General, 0, General_Encryption_Format, "AES");
164 Fill(Stream_General, 0, General_Encryption_Length, "128");
165 Fill(Stream_General, 0, General_Encryption_Method, "Segment");
166 Fill(Stream_General, 0, General_Encryption_Mode, "CBC");
167 Fill(Stream_General, 0, General_Encryption_Padding, "PKCS7");
168 Fill(Stream_General, 0, General_Encryption_InitializationVector, "Sequence number");
169
170 #if MEDIAINFO_AES
171 //Trying to get the key from FileName.FileExt.key
172 if (Config->Encryption_Key_Get().empty())
173 {
174 File KeyFile;
175 if (KeyFile.Open(File_Name+__T(".key")))
176 {
177 if (KeyFile.Size_Get()==16)
178 {
179 int8u Key[16];
180 if (KeyFile.Read(Key, 16)==16)
181 Config->Encryption_Key_Set(Key, 16);
182 }
183 else
184 Fill(Stream_General, 0, "Encryption_Key_Problem", KeyFile.Size_Get());
185 }
186 }
187 #endif
188 }
189 Fill(Stream_General, 0, General_Encryption, List[Pos](1));
190 }
191 }
192 }
193 else if (Lines[Line].find(__T("#EXT-X-STREAM-INF:"))==0)
194 {
195 IsGroup=true;
196 }
197 else if (Lines[Line][0]==__T('#'))
198 ;
199 else
200 {
201 if (IsGroup)
202 {
203 Sequence->AddFileName(Lines[Line]);
204 Sequence->StreamID=ReferenceFiles->Sequences_Size()+1;
205 ReferenceFiles->AddSequence(Sequence);
206 IsGroup=false;
207 Sequence=new sequence();
208 #if MEDIAINFO_EVENTS
209 ParserIDs[0]=MediaInfo_Parser_HlsIndex;
210 StreamIDs_Width[0]=sizeof(size_t);
211 #endif //MEDIAINFO_EVENTS
212 }
213 else if (Lines[Line]!=PreviousFile)
214 {
215 PreviousFile=Lines[Line];
216 Sequence->AddFileName(Lines[Line]);
217 }
218 }
219 }
220 }
221
222 if (!Sequence->FileNames.empty())
223 {
224 ReferenceFiles->AddSequence(Sequence);
225 Fill(Stream_General, 0, General_Format_Profile, "Media");
226 }
227 else
228 {
229 Fill(Stream_General, 0, General_Format_Profile, "Master");
230 }
231
232 Element_Offset=File_Size;
233
234 //All should be OK...
235 Config->File_TestContinuousFileNames_Set(false);
236 return true;
237}
