36void File_DvDif::Read_Buffer_Continue()
37{
38 if (!Analyze_Activated)
39 {
40 if (Config->File_DvDif_Analysis_Get())
41 Analyze_Activated=true;
42 else
43 return;
44 }
45
46 //Errors stats
47 while (Buffer_Offset+80<=Buffer_Size)
48 {
49 if ((Buffer[Buffer_Offset]&0xE0)==0x00 //SCT=0 (Header)
50 && !(Buffer[Buffer_Offset ]==0x00
51 && Buffer[Buffer_Offset+1]==0x00
52 && Buffer[Buffer_Offset+2]==0x00))
53 {
54 Frame_AtLeast1DIF=true;
55
56 if (!DSF_IsValid)
57 {
58 DSF=(Buffer[Buffer_Offset+3]&0x80)?true:false;
59 DSF_IsValid=true;
60 Dseq_Old=DSF?11:9;
61 }
62 }
63
64 //Quick search depends of SCT
65 switch(Buffer[Buffer_Offset]&0xE0)
66 {
67 case 0x20 : //SCT=1 (Subcode)
68 {
69 Frame_AtLeast1DIF=true;
70
71 for (size_t Pos=3*8; Pos<40; Pos+=2*8)
72 {
73 int8u PackType=Buffer[Buffer_Offset+3+Pos+3];
74 //dv_timecode
75 if (PackType==0x13) //Pack type=0x13 (dv_timecode)
76 {
77 bool DropFrame =( Buffer[Buffer_Offset+3+Pos+3+1]&0x40)?true:false;
78 int8u Frames =((Buffer[Buffer_Offset+3+Pos+3+1]&0x30)>>4)*10
79 + ((Buffer[Buffer_Offset+3+Pos+3+1]&0x0F) ) ;
80 int8u Seconds =((Buffer[Buffer_Offset+3+Pos+3+2]&0x70)>>4)*10
81 + ((Buffer[Buffer_Offset+3+Pos+3+2]&0x0F)) ;
82 int8u Minutes =((Buffer[Buffer_Offset+3+Pos+3+3]&0x70)>>4)*10
83 + ((Buffer[Buffer_Offset+3+Pos+3+3]&0x0F) ) ;
84 int8u Hours =((Buffer[Buffer_Offset+3+Pos+3+4]&0x30)>>4)*10
85 + ((Buffer[Buffer_Offset+3+Pos+3+4]&0x0F) ) ;
86
87 if (Frames ==0x00
88 && Seconds==0x00
89 && Minutes==0x00
90 && Hours ==0x00
91 && Buffer[Buffer_Offset+3+Pos+3+1]==0x00
92 && Buffer[Buffer_Offset+3+Pos+3+2]==0x00
93 && Buffer[Buffer_Offset+3+Pos+3+3]==0x00
94 && Buffer[Buffer_Offset+3+Pos+3+4]==0x00
95 )
96 {
97 Frames =45;
98 Seconds=85;
99 Minutes=85;
100 Hours =45;
101 }
102 if (Frames !=45
103 && Seconds!=85
104 && Minutes!=85
105 && Hours !=45) //If not disabled
106 {
107 if (Speed_TimeCode_Current.IsValid
108 && (Speed_TimeCode_Current.Time.DropFrame !=DropFrame
109 || Speed_TimeCode_Current.Time.Frames !=Frames
110 || Speed_TimeCode_Current.Time.Seconds !=Seconds
111 || Speed_TimeCode_Current.Time.Minutes !=Minutes
112 || Speed_TimeCode_Current.Time.Hours !=Hours))
113 {
114 Speed_TimeCode_Current.MultipleValues=true; //There are 2+ different values
115 }
116 else if (!Speed_TimeCode_Current.IsValid && !Speed_TimeCode_Current.MultipleValues)
117 {
118 Speed_TimeCode_Current.Time.DropFrame=DropFrame;
119 Speed_TimeCode_Current.Time.Frames =Frames;
120 Speed_TimeCode_Current.Time.Seconds =Seconds;
121 Speed_TimeCode_Current.Time.Minutes =Minutes;
122 Speed_TimeCode_Current.Time.Hours =Hours;
123 Speed_TimeCode_Current.IsValid =true;
124 }
125 }
126 }
127
128 //video_recdate
129 if (PackType==0x62) //Pack type=0x62 (video_rectime)
130 {
131 int8u Days =((Buffer[Buffer_Offset+3+Pos+2]&0x30)>>4)*10
132 + ((Buffer[Buffer_Offset+3+Pos+2]&0x0F) ) ;
133 int8u Months =((Buffer[Buffer_Offset+3+Pos+3]&0x10)>>4)*10
134 + ((Buffer[Buffer_Offset+3+Pos+3]&0x0F) ) ;
135 int8u Years =((Buffer[Buffer_Offset+3+Pos+4]&0xF0)>>4)*10
136 + ((Buffer[Buffer_Offset+3+Pos+4]&0x0F) ) ;
137 if (Months<=12
138 && Days <=31)
139 {
140 if (Speed_RecDate_Current.IsValid
141 && Speed_RecDate_Current.Days !=Days
142 && Speed_RecDate_Current.Months !=Months
143 && Speed_RecDate_Current.Years !=Years)
144 {
145 Speed_RecDate_Current.MultipleValues=true; //There are 2+ different values
146 }
147 else if (!Speed_RecTime_Current.MultipleValues)
148 {
149 Speed_RecDate_Current.Days =Days;
150 Speed_RecDate_Current.Months =Months;
151 Speed_RecDate_Current.Years =Years;
152 Speed_RecDate_Current.IsValid =true;
153 }
154 }
155 }
156
157 //video_rectime
158 if (PackType==0x63) //Pack type=0x63 (video_rectime)
159 {
160 int8u Frames =((Buffer[Buffer_Offset+3+Pos+1]&0x30)>>4)*10
161 + ((Buffer[Buffer_Offset+3+Pos+1]&0x0F) ) ;
162 int8u Seconds =((Buffer[Buffer_Offset+3+Pos+2]&0x70)>>4)*10
163 + ((Buffer[Buffer_Offset+3+Pos+2]&0x0F)) ;
164 int8u Minutes =((Buffer[Buffer_Offset+3+Pos+3]&0x70)>>4)*10
165 + ((Buffer[Buffer_Offset+3+Pos+3]&0x0F) ) ;
166 int8u Hours =((Buffer[Buffer_Offset+3+Pos+4]&0x30)>>4)*10
167 + ((Buffer[Buffer_Offset+3+Pos+4]&0x0F) ) ;
168 if (Seconds<61
169 && Minutes<60
170 && Hours <24) //If not disabled
171 {
172 if (Speed_RecTime_Current.IsValid
173 && Speed_RecTime_Current.Time.Frames !=Frames
174 && Speed_RecTime_Current.Time.Seconds !=Seconds
175 && Speed_RecTime_Current.Time.Minutes !=Minutes
176 && Speed_RecTime_Current.Time.Hours !=Hours)
177 {
178 Speed_RecTime_Current.MultipleValues=true; //There are 2+ different values
179 }
180 else if (!Speed_RecTime_Current.MultipleValues)
181 {
182 Speed_RecTime_Current.Time.Frames =Frames;
183 Speed_RecTime_Current.Time.Seconds =Seconds;
184 Speed_RecTime_Current.Time.Minutes =Minutes;
185 Speed_RecTime_Current.Time.Hours =Hours;
186 Speed_RecTime_Current.IsValid =true;
187 }
188 }
189 }
190 }
191 }
192 break;
193
194 case 0x40 : //SCT=2 (VAUX)
195 {
196 Frame_AtLeast1DIF=true;
197
198 for (size_t Pos=0; Pos<15*5; Pos+=5)
199 {
200 int8u PackType=Buffer[Buffer_Offset+3+Pos];
201 //video_source
202 if (PackType==0x60 && !System_IsValid) //Pack type=0x60 (video_source)
203 {
204 System=(Buffer[Buffer_Offset+3+Pos+3]&0x20)==0x20?true:false;
205 System_IsValid=true;
206 video_source_Detected=true;
207 }
208
209 //video_recdate
210 if (PackType==0x62) //Pack type=0x62 (video_rectime)
211 {
212 int8u Days =((Buffer[Buffer_Offset+3+Pos+2]&0x30)>>4)*10
213 + ((Buffer[Buffer_Offset+3+Pos+2]&0x0F) ) ;
214 int8u Months =((Buffer[Buffer_Offset+3+Pos+3]&0x10)>>4)*10
215 + ((Buffer[Buffer_Offset+3+Pos+3]&0x0F) ) ;
216 int8u Years =((Buffer[Buffer_Offset+3+Pos+4]&0xF0)>>4)*10
217 + ((Buffer[Buffer_Offset+3+Pos+4]&0x0F) ) ;
218 if (Months<=12
219 && Days <=31)
220 {
221 if (Speed_RecDate_Current.IsValid
222 && Speed_RecDate_Current.Days !=Days
223 && Speed_RecDate_Current.Months !=Months
224 && Speed_RecDate_Current.Years !=Years)
225 {
226 Speed_RecDate_Current.MultipleValues=true; //There are 2+ different values
227 }
228 else if (!Speed_RecTime_Current.MultipleValues)
229 {
230 Speed_RecDate_Current.Days =Days;
231 Speed_RecDate_Current.Months =Months;
232 Speed_RecDate_Current.Years =Years;
233 Speed_RecDate_Current.IsValid =true;
234 }
235 }
236 }
237
238 //video_rectime
239 if (PackType==0x63) //Pack type=0x63 (video_rectime)
240 {
241 int8u Frames =((Buffer[Buffer_Offset+3+Pos+1]&0x30)>>4)*10
242 + ((Buffer[Buffer_Offset+3+Pos+1]&0x0F) ) ;
243 int8u Seconds =((Buffer[Buffer_Offset+3+Pos+2]&0x70)>>4)*10
244 + ((Buffer[Buffer_Offset+3+Pos+2]&0x0F)) ;
245 int8u Minutes =((Buffer[Buffer_Offset+3+Pos+3]&0x70)>>4)*10
246 + ((Buffer[Buffer_Offset+3+Pos+3]&0x0F) ) ;
247 int8u Hours =((Buffer[Buffer_Offset+3+Pos+4]&0x30)>>4)*10
248 + ((Buffer[Buffer_Offset+3+Pos+4]&0x0F) ) ;
249 if (Seconds!=85
250 && Minutes!=85
251 && Hours !=45) //If not disabled
252 {
253 if (Speed_RecTime_Current.IsValid
254 && Speed_RecTime_Current.Time.Frames !=Frames
255 && Speed_RecTime_Current.Time.Seconds !=Seconds
256 && Speed_RecTime_Current.Time.Minutes !=Minutes
257 && Speed_RecTime_Current.Time.Hours !=Hours)
258 {
259 Speed_RecTime_Current.MultipleValues=true; //There are 2+ different values
260 }
261 else if (!Speed_RecTime_Current.MultipleValues)
262 {
263 Speed_RecTime_Current.Time.Frames =Frames;
264 Speed_RecTime_Current.Time.Seconds =Seconds;
265 Speed_RecTime_Current.Time.Minutes =Minutes;
266 Speed_RecTime_Current.Time.Hours =Hours;
267 Speed_RecTime_Current.IsValid =true;
268 }
269 }
270 }
271 }
272 }
273 break;
274
275 case 0x60 : //SCT=3 (Audio)
276 {
277 //audio_source
278 if (Buffer[Buffer_Offset+3+0]==0x50) //audio_source
279 {
280 QU_FSC =(Buffer[Buffer_Offset+1 ]&0x08)?true:false; //FSC
281 QU_System =(Buffer[Buffer_Offset+3+3]&0x20)?true:false; //50/60
282
283 int8u AUDIO_MODE=Buffer[Buffer_Offset+3+2]&0x0F;
284 QU =Buffer[Buffer_Offset+3+4]&0x07;
285
286 size_t Channel=(QU_FSC?2:0)+((Buffer[Buffer_Offset+1]>>4)>=(QU_System?6:5)?1:0); //If Dseq>=5 or 6
287 if (audio_source_IsPresent.empty())
288 audio_source_IsPresent.resize(8);
289 audio_source_IsPresent[Channel]=true;
290
291 if (AUDIO_MODE==0x0F)
292 {
293 if (Audio_Invalids.empty())
294 {
295 Audio_Invalids.resize(8);
296 for (size_t Pos=0; Pos<8; Pos++)
297 Audio_Invalids[Pos].resize(16);
298 }
299 Audio_Invalids[Channel][Buffer[Buffer_Offset+1]>>4]+=9;
300 }
301 }
302
303 //audio_source_control
304 if (Buffer[Buffer_Offset+3+0]==0x51) //audio_source_control
305 {
306 REC_ST =(Buffer[Buffer_Offset+3+2]&0x80)?true:false;
307 REC_END=(Buffer[Buffer_Offset+3+2]&0x40)?true:false;
308 REC_IsValid=true;
309 }
310
311 //audio_recdate
312 if (Buffer[Buffer_Offset+3+0]==0x52) //Pack type=0x52 (audio_rectime)
313 {
314 int8u Days =((Buffer[Buffer_Offset+3+2]&0x30)>>4)*10
315 + ((Buffer[Buffer_Offset+3+2]&0x0F) ) ;
316 int8u Months =((Buffer[Buffer_Offset+3+3]&0x10)>>4)*10
317 + ((Buffer[Buffer_Offset+3+3]&0x0F) ) ;
318 int8u Years =((Buffer[Buffer_Offset+3+4]&0xF0)>>4)*10
319 + ((Buffer[Buffer_Offset+3+4]&0x0F) ) ;
320 if (Months<=12
321 && Days <=31)
322 {
323 if (Speed_RecDate_Current.IsValid
324 && Speed_RecDate_Current.Days !=Days
325 && Speed_RecDate_Current.Months !=Months
326 && Speed_RecDate_Current.Years !=Years)
327 {
328 Speed_RecDate_Current.MultipleValues=true; //There are 2+ different values
329 }
330 else if (!Speed_RecTime_Current.MultipleValues)
331 {
332 Speed_RecDate_Current.Days =Days;
333 Speed_RecDate_Current.Months =Months;
334 Speed_RecDate_Current.Years =Years;
335 Speed_RecDate_Current.IsValid =true;
336 }
337 }
338 }
339
340 //audio_rectime
341 if (Buffer[Buffer_Offset+3+0]==0x53) //Pack type=0x53 (audio_rectime)
342 {
343 int8u Frames =((Buffer[Buffer_Offset+3+1]&0x30)>>4)*10
344 + ((Buffer[Buffer_Offset+3+1]&0x0F) ) ;
345 int8u Seconds =((Buffer[Buffer_Offset+3+2]&0x70)>>4)*10
346 + ((Buffer[Buffer_Offset+3+2]&0x0F)) ;
347 int8u Minutes =((Buffer[Buffer_Offset+3+3]&0x70)>>4)*10
348 + ((Buffer[Buffer_Offset+3+3]&0x0F) ) ;
349 int8u Hours =((Buffer[Buffer_Offset+3+4]&0x30)>>4)*10
350 + ((Buffer[Buffer_Offset+3+4]&0x0F) ) ;
351 if (Seconds!=85
352 && Minutes!=85
353 && Hours !=45) //If not disabled
354 {
355 if (Speed_RecTime_Current.IsValid
356 && Speed_RecTime_Current.Time.Frames !=Frames
357 && Speed_RecTime_Current.Time.Seconds !=Seconds
358 && Speed_RecTime_Current.Time.Minutes !=Minutes
359 && Speed_RecTime_Current.Time.Hours !=Hours)
360 {
361 Speed_RecTime_Current.MultipleValues=true; //There are 2+ different values
362 }
363 else if (!Speed_RecTime_Current.MultipleValues)
364 {
365 Speed_RecTime_Current.Time.Frames =Frames;
366 Speed_RecTime_Current.Time.Seconds =Seconds;
367 Speed_RecTime_Current.Time.Minutes =Minutes;
368 Speed_RecTime_Current.Time.Hours =Hours;
369 Speed_RecTime_Current.IsValid =true;
370 }
371 }
372 }
373
374 //Audio errors
375 if (Buffer[Buffer_Offset+8]==0x80)
376 {
377 if ((QU==0 && Buffer[Buffer_Offset+ 9]==0x00) //16-bit 0x8000
378 || (QU==1 && Buffer[Buffer_Offset+10]==0x00) //12-bit 0x800
379 || (QU==(int8u)-1 && ((Buffer[Buffer_Offset+ 9]==0x00 && Buffer[Buffer_Offset+10]==0x80 && Buffer[Buffer_Offset+11]==0x00)
380 || (Buffer[Buffer_Offset+ 9]==0x80 && Buffer[Buffer_Offset+10]==0x00)))) //In case of QU is not already detected
381 {
382 if (Audio_Errors.empty())
383 Audio_Errors.resize(16);
384 Audio_Errors[Buffer[Buffer_Offset+1]>>4]++;
385 }
386 }
387 }
388 break;
389
390 case 0x80 : //SCT=4 (Video)
391 {
392 //Speed_Arb_Current
393 int8u Value=Buffer[Buffer_Offset+0]&0x0F;
394 Speed_Arb_Current.Value_Counters[Value]++;
395 if (Value==0xF)
396 {
397 if (!Speed_Arb_Current.IsValid)
398 {
399 Speed_Arb_Current.Value =0xF;
400 Speed_Arb_Current.IsValid=true;
401 }
402 }
403 else
404 {
405 if (Speed_Arb_Current.IsValid
406 && Speed_Arb_Current.Value!=0xF
407 && Speed_Arb_Current.Value!=Value)
408 Speed_Arb_Current.MultipleValues=true; //There are 2+ different values
409 else if (!Speed_Arb_Current.MultipleValues)
410 {
411 Speed_Arb_Current.Value =Value;
412 Speed_Arb_Current.IsValid=true;
413 }
414 }
415
416 //STA
417 if (Buffer[Buffer_Offset+3]&0xF0)
418 {
419 if (video_source_Detected)
420 {
421 if (Video_STA_Errors.empty())
422 Video_STA_Errors.resize(16);
423 Video_STA_Errors[Buffer[Buffer_Offset+3]>>4]++;
424 }
425 }
426 }
427 break;
428 }
429
430 //Errors stats
431 if (Frame_AtLeast1DIF &&
432 (Buffer[Buffer_Offset ]&0xE0)==0x00 //SCT=0 (Header)
433 && (Buffer[Buffer_Offset+1]&0xF0)==0x00 //Dseq=0
434 && (Buffer[Buffer_Offset+1]&0x08)==0x00) //FSC=0
435 Errors_Stats_Update();
436
437 //Coherency test
438 if (Buffer[Buffer_Offset ]==0x00
439 && Buffer[Buffer_Offset+1]==0x00
440 && Buffer[Buffer_Offset+2]==0x00)
441 Speed_Contains_NULL++;
442
443 Buffer_Offset+=80;
444 }
445
446 if (!Status[IsAccepted])
447 File__Analyze::Buffer_Offset=0;
448 Config->State_Set(((float)File_Offset)/File_Size);
449}

451void File_DvDif::Errors_Stats_Update()
452{
453 if (!Analyze_Activated)
454 {
455 if (Config->File_DvDif_Analysis_Get())
456 Analyze_Activated=true;
457 else
458 return;
459 }
460
461 Ztring Errors_Stats_Line;
462 if (Speed_FrameCount) //We must have at least one complete frame
463 {
464 bool Errors_AreDetected=false;
465 bool Infos_AreDetected=false;
466 bool Arb_AreDetected=false;
467
468 #if MEDIAINFO_EVENTS
469 //Demux
470 struct MediaInfo_Event_DvDif_Analysis_Frame_0 Event;
471 Event.EventCode=MediaInfo_EventCode_Create(MediaInfo_Parser_DvDif, MediaInfo_Event_DvDif_Analysis_Frame, 0);
472 Event.TimeCode=0;
473 Event.RecordedDateTime1=0;
474 Event.RecordedDateTime2=0;
475 Event.Arb=0;
476 Event.Verbosity=0;
477 Event.Errors=NULL;
478 #endif //MEDIAINFO_EVENTS
479
480 //Framerate computing
481 float64 FrameRate=29.970;
482 if (System_IsValid)
483 FrameRate=System?25.000:29.970;
484 else if (DSF_IsValid)
485 FrameRate=DSF?25.000:29.970;
486 else
487 FrameRate=29.970;
488 if (FrameRate==29.970 && Speed_TimeCode_Current.IsValid && !Speed_TimeCode_Current.Time.DropFrame)
489 FrameRate=30.000;
490
491 //Frame number
492 Ztring Frame_Number_Padded=Ztring::ToZtring(Speed_FrameCount-1);
493 if (Frame_Number_Padded.size()<8)
494 Frame_Number_Padded.insert(0, 8-Frame_Number_Padded.size(), __T(' '));
495 Errors_Stats_Line+=Frame_Number_Padded;
496 Errors_Stats_Line+=__T('\t');
497
498 //Time Offset
499 float64 Time_Offset=(Speed_FrameCount-1)*1000/FrameRate;
500 Errors_Stats_Line+=Ztring().Duration_From_Milliseconds((int64u)Time_Offset);
501 Errors_Stats_Line+=__T('\t');
502
503 //Timecode
504 if (Speed_TimeCode_Current.IsValid)
505 {
506 Speed_TimeCodeZ_Last=Speed_TimeCodeZ_Current;
507 Speed_TimeCodeZ_Current.clear();
508 Speed_TimeCodeZ_Current.reserve(12);
509 Speed_TimeCodeZ_Current.reserve(11);
510 Speed_TimeCodeZ_Current+=__T('0')+Speed_TimeCode_Current.Time.Hours /10;
511 Speed_TimeCodeZ_Current+=__T('0')+Speed_TimeCode_Current.Time.Hours %10;
512 Speed_TimeCodeZ_Current+=__T(':');
513 Speed_TimeCodeZ_Current+=__T('0')+Speed_TimeCode_Current.Time.Minutes/10;
514 Speed_TimeCodeZ_Current+=__T('0')+Speed_TimeCode_Current.Time.Minutes%10;
515 Speed_TimeCodeZ_Current+=__T(':');
516 Speed_TimeCodeZ_Current+=__T('0')+Speed_TimeCode_Current.Time.Seconds/10;
517 Speed_TimeCodeZ_Current+=__T('0')+Speed_TimeCode_Current.Time.Seconds%10;
518 Speed_TimeCodeZ_Current+=(!DSF && Speed_TimeCode_Current.Time.DropFrame)?__T(';'):__T(':');
519 Speed_TimeCodeZ_Current+=__T('0')+Speed_TimeCode_Current.Time.Frames /10;
520 Speed_TimeCodeZ_Current+=__T('0')+Speed_TimeCode_Current.Time.Frames %10;
521 Errors_Stats_Line+=Speed_TimeCodeZ_Current;
522 if (Speed_TimeCodeZ.empty())
523 {
524 Speed_TimeCodeZ.resize(1);
525 Speed_TimeCodeZ[0].First.FramePos=Speed_FrameCount;
526 Speed_TimeCodeZ[0].First.TimeCode=Speed_TimeCodeZ_Current;
527 }
528 if (Speed_TimeStampsZ.empty())
529 {
530 Speed_TimeStampsZ.resize(1);
531 Speed_TimeStampsZ[0].First.FramePos=Speed_FrameCount;
532 Speed_TimeStampsZ[0].First.Time+=Ztring().Duration_From_Milliseconds((int64u)Time_Offset);
533 }
534 if (Speed_TimeStampsZ[0].First.FramePos==Speed_FrameCount)
535 Speed_TimeStampsZ[0].First.TimeCode=Speed_TimeCodeZ_Current;
536 #if MEDIAINFO_EVENTS
537 int32u Seconds=Speed_TimeCode_Current.Time.Hours *60*60
538 + Speed_TimeCode_Current.Time.Minutes *60
539 + Speed_TimeCode_Current.Time.Seconds ;
540 Event.TimeCode|=Seconds<<8;
541 Event.TimeCode|=(!DSF && Speed_TimeCode_Current.Time.DropFrame)<<7;
542 Event.TimeCode|=Speed_TimeCode_Current.Time.Frames;
543 #endif //MEDIAINFO_EVENTS
544 }
545 else
546 {
547 Errors_Stats_Line+=__T("XX:XX:XX:XX");
548 #if MEDIAINFO_EVENTS
549 Event.TimeCode|=0x7FFFF<<8;
550 //Event.TimeCode|=Speed_TimeCode_Current.Time.DropFrame<<7;
551 Event.TimeCode|=0x3F;
552 #endif //MEDIAINFO_EVENTS
553 }
554 Errors_Stats_Line+=__T('\t');
555
556 //Timecode order coherency
557 if (!Speed_TimeCode_IsValid && Speed_TimeCode_Current.IsValid
558 && (Speed_TimeCode_Current.Time.Hours!=0
559 || Speed_TimeCode_Current.Time.Minutes!=0
560 || Speed_TimeCode_Current.Time.Seconds!=0
561 || Speed_TimeCode_Current.Time.Frames!=0))
562 Speed_TimeCode_IsValid=true;
563 bool TimeCode_Disrupted=false;
564 if (Speed_TimeCode_IsValid && Speed_TimeCode_Current.IsValid && Speed_TimeCode_Last.IsValid
565 && Speed_TimeCode_Current.Time.Frames ==Speed_TimeCode_Last.Time.Frames
566 && Speed_TimeCode_Current.Time.Seconds==Speed_TimeCode_Last.Time.Seconds
567 && Speed_TimeCode_Current.Time.Minutes==Speed_TimeCode_Last.Time.Minutes
568 && Speed_TimeCode_Current.Time.Hours ==Speed_TimeCode_Last.Time.Hours)
569 {
570 Errors_Stats_Line+=__T('R');
571 #if MEDIAINFO_EVENTS
572 Event.TimeCode|=1<<31;
573 #endif //MEDIAINFO_EVENTS
574 if (Speed_TimeCode_Current.Time.Hours
575 || Speed_TimeCode_Current.Time.Seconds
576 || Speed_TimeCode_Current.Time.Minutes)
577 Errors_AreDetected=true;
578 }
579 else if (Speed_TimeCode_IsValid && Speed_TimeCode_Current.IsValid && Speed_TimeCode_Current_Theory.IsValid
580 && ( Speed_TimeCode_Current.Time.Frames !=Speed_TimeCode_Current_Theory.Time.Frames
581 || Speed_TimeCode_Current.Time.Seconds!=Speed_TimeCode_Current_Theory.Time.Seconds
582 || Speed_TimeCode_Current.Time.Minutes!=Speed_TimeCode_Current_Theory.Time.Minutes
583 || Speed_TimeCode_Current.Time.Hours !=Speed_TimeCode_Current_Theory.Time.Hours))
584 {
585 size_t Speed_TimeCodeZ_Pos=Speed_TimeCodeZ.size();
586 Speed_TimeCodeZ.resize(Speed_TimeCodeZ_Pos+1);
587 Speed_TimeCodeZ[Speed_TimeCodeZ_Pos].First.FramePos=Speed_FrameCount-1;
588 Speed_TimeCodeZ[Speed_TimeCodeZ_Pos].First.TimeCode=Speed_TimeCodeZ_Current;
589 Speed_TimeCodeZ[Speed_TimeCodeZ_Pos-1].Last.FramePos=Speed_FrameCount-1;
590 Speed_TimeCodeZ[Speed_TimeCodeZ_Pos-1].Last.TimeCode=Speed_TimeCodeZ_Last;
591
