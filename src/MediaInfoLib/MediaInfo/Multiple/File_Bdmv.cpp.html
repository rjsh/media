454//---------------------------------------------------------------------------
455bool File_Bdmv::FileHeader_Begin()
456{
457 size_t BDMV_Pos=File_Name.find(Ztring(1, PathSeparator)+__T("BDMV"));
458 if (BDMV_Pos!=string::npos && BDMV_Pos+5==File_Name.size()) //Blu-ray directory
459 return true;
460
461 //Element_Size
462 if (Buffer_Size<4)
463 return false; //Must wait for more data
464
465 switch (CC4(Buffer))
466 {
467 case Elements::CLPI :
468 case Elements::INDX :
469 case Elements::MOBJ :
470 case Elements::MPLS :
471 break;
472 default : Reject("Blu-ray");
473 return false;
474 }
475
476 //Init
477 Mpls_PlayList_IsParsed=false;
478
479 //All should be OK...
480 return true;
481}

487//---------------------------------------------------------------------------
488void File_Bdmv::Read_Buffer_Continue()
489{
490 size_t BDMV_Pos=File_Name.find(Ztring(1, PathSeparator)+__T("BDMV"));
491 if (BDMV_Pos!=string::npos && BDMV_Pos+5==File_Name.size()) //Blu-ray directory
492 {
493 BDMV();
494 return;
495 }
496
497 if (Buffer_Size<File_Size)
498 {
499 Element_WaitForMoreData();
500 return; //Wait for more data
501 }
502
503 //Parsing
504 int32u type_indicator;
505 int16u version_numberH;
506 Element_Begin1("Header");
507 Get_C4 (type_indicator, "type_indicator");
508 Data_Accept("Blu-ray");
509 Get_C2 (version_numberH, "version_number (High)");
510 Skip_C2( "version_number (Low)");
511 Element_End0();
512
513 FILLING_BEGIN();
514 Accept("BDMV");
515 switch (type_indicator)
516 {
517 case Elements::CLPI : Fill(Stream_General, 0, General_Format, "Blu-ray Clip info"); break;
518 case Elements::INDX : Fill(Stream_General, 0, General_Format, "Blu-ray Index"); break;
519 case Elements::MOBJ : Fill(Stream_General, 0, General_Format, "Blu-ray Movie object"); break;
520 case Elements::MPLS : Fill(Stream_General, 0, General_Format, "Blu-ray Playlist"); break;
521 default : ;
522 }
523 FILLING_END();
524
525 if (version_numberH==0x3031 || version_numberH==0x3032) //Version 1 or 2
526 {
527 Element_Begin1("Offsets");
528 Types[0x28]=0; //First object
529 for (size_t Start_Adress_Pos=1; Start_Adress_Pos<9; Start_Adress_Pos++)
530 {
531 int32u Start_Adress;
532 Get_B4 (Start_Adress, Bdmv_Type(type_indicator, Start_Adress_Pos));
533 Types[Start_Adress]=Start_Adress_Pos;
534 }
535 Element_End0();
536
537 for (std::map<int32u, size_t>::iterator Type=Types.begin(); Type!=Types.end(); ++Type)
538 {
539 if (Type->first>=Element_Offset) //If valid
540 {
541 if (Type->first>Element_Offset)
542 Skip_XX(Type->first-Element_Offset, "unknown");
543
544 Element_Begin1(Bdmv_Type(type_indicator, Type->second));
545 int32u length;
546 Get_B4 (length, "length");
547 int64u End=Element_Offset+length;
548 switch (type_indicator)
549 {
550 case Elements::CLPI :
551 switch(Type->second)
552 {
553 case 2 : Clpi_ProgramInfo(); break;
554 case 5 : Clpi_ExtensionData(); break;
555 default: ;
556 }
557 break;
558 case Elements::INDX :
559 switch(Type->second)
560 {
561 case 0 : Indx_AppInfoBDMV(); break;
562 case 1 : Indx_Indexes(); break;
563 case 2 : Indx_ExtensionData(); break;
564 default: ;
565 }
566 break;
567 case Elements::MOBJ :
568 switch(Type->second)
569 {
570 case 0 : Mobj_MovieObjects(); break;
571 case 1 : Mobj_ExtensionData(); break;
572 default: ;
573 }
574 break;
575 case Elements::MPLS :
576 switch(Type->second)
577 {
578 case 0 : Mpls_AppInfoPlayList(); break;
579 case 1 : Mpls_PlayList(); break;
580 case 2 : Mpls_PlayListMarks (); break;
581 case 3 : Mpls_ExtensionData (); break;
582 default: ;
583 }
584 break;
585 default : ;
586 }
587 if (End>Element_Offset)
588 Skip_XX(End-Element_Offset, "Unknown");
589 Element_End0();
590 }
591 }
592
593 if (Element_Size>Element_Offset)
594 Skip_XX(Element_Size-Element_Offset, "Unknown");
595 }
596 else
597 Skip_XX(Element_Size-Element_Offset, "Unknown");
598}

604//---------------------------------------------------------------------------
605void File_Bdmv::BDMV()
606{
607 Accept("BDMV");
608
609 //Searching the longest playlist
610 ZtringList List=Dir::GetAllFileNames(File_Name+PathSeparator+__T("PLAYLIST")+PathSeparator+__T("*.mpls"), Dir::Include_Files);
611 std::vector<MediaInfo_Internal*> MIs;
612 MIs.resize(List.size());
613 size_t MaxDuration_Pos=(size_t)-1;
614 int64u MaxDuration=0;
615 if (Config->File_Bdmv_ParseTargetedFile_Get())
616 {
617 for (size_t Pos=0; Pos<MIs.size(); Pos++)
618 {
619 MIs[Pos]=new MediaInfo_Internal();
620 MIs[Pos]->Option(__T("File_Bdmv_ParseTargetedFile"), __T("0"));
621 MIs[Pos]->Option(__T("File_IsReferenced"), __T("1"));
622 MIs[Pos]->Open(List[Pos]);
623 int64u Duration=Ztring(MIs[Pos]->Get(Stream_General, 0, General_Duration)).To_int64u();
624 if (Duration>MaxDuration)
625 {
626 MaxDuration=Duration;
627 MaxDuration_Pos=Pos;
628 }
629 }
630 }
631
632 if (MaxDuration_Pos!=(size_t)-1)
633 {
634 //Merging
635 MediaInfo_Internal MI;
636 MI.Option(__T("File_IsReferenced"), __T("1"));
637 MI.Open(List[MaxDuration_Pos]); //Open it again for having the M2TS part
638 Merge(MI);
639
640 Clear(Stream_General, 0, General_Format);
641 Clear(Stream_General, 0, General_Format_String);
642 Clear(Stream_General, 0, General_Format_Extensions);
643 Clear(Stream_General, 0, General_Format_Info);
644 Clear(Stream_General, 0, General_Codec);
645 Clear(Stream_General, 0, General_Codec_String);
646 Clear(Stream_General, 0, General_Codec_Extensions);
647 Clear(Stream_General, 0, General_FileSize);
648 Clear(Stream_Video, 0, Video_ScanType_String);
649 Clear(Stream_Video, 0, Video_Bits__Pixel_Frame_);
650 }
651
652 for (size_t Pos=0; Pos<MIs.size(); Pos++)
653 delete MIs[Pos]; //MIs[Pos]=NULL;
654 MIs.clear();
655
656 //Detecting some directories
657 if (Dir::Exists(File_Name+PathSeparator+__T("BDSVM"))
658 || Dir::Exists(File_Name+PathSeparator+__T("SLYVM"))
659 || Dir::Exists(File_Name+PathSeparator+__T("ANYVM")))
660 Fill(Stream_General, 0, General_Format_Profile, "BD+");
661 if (Dir::Exists(File_Name+PathSeparator+__T("BDJO")) && !Dir::GetAllFileNames(File_Name+PathSeparator+__T("BDJO")).empty())
662 Fill(Stream_General, 0, General_Format_Profile, "BD-Java");
663
664 //Filling
665 File_Name.resize(File_Name.size()-5); //Removing "/BDMV"
666 Fill(Stream_General, 0, General_Format, "Blu-ray movie", Unlimited, true, true);
667 Fill(Stream_General, 0, General_CompleteName, File_Name, true);
668 Fill(Stream_General, 0, General_FolderName, FileName::Path_Get(File_Name), true);
669 if (FileName::Extension_Get(File_Name).empty())
670 Fill(Stream_General, 0, General_FileName, FileName::Name_Get(File_Name), true);
671 else
672 Fill(Stream_General, 0, General_FileName, FileName::Name_Get(File_Name)+__T('.')+FileName::Extension_Get(File_Name), true);
673 File_Name.clear();
674
675 Finish("BDMV");
676}

678//---------------------------------------------------------------------------
679void File_Bdmv::Clpi_ProgramInfo()
680{
681 //Retrieving data from the M2TS file
682 std::map<int16u, stream_t> PIDs_StreamKind;
683 std::map<int16u, size_t> PIDs_StreamPos;
684 if (Config->File_Bdmv_ParseTargetedFile_Get() && File_Name.size()>10+1+7)
685 {
686 Ztring file=File_Name.substr(File_Name.size()-10, 5);
687 Ztring M2TS_File=File_Name;
688 M2TS_File.resize(M2TS_File.size()-(10+1+7));
689 M2TS_File+=__T("STREAM");
690 M2TS_File+=PathSeparator;
691 M2TS_File+=file;
692 M2TS_File+=__T(".m2ts");
693
694 MediaInfo_Internal MI;
695 MI.Option(__T("File_Bdmv_ParseTargetedFile"), __T("0"));
696 MI.Option(__T("File_IsReferenced"), __T("1"));
697 if (MI.Open(M2TS_File))
698 {
699 Merge(MI);
700 for (size_t StreamKind=Stream_General+1; StreamKind<Stream_Max; StreamKind++)
701 for (size_t StreamPos=0; StreamPos<Count_Get((stream_t)StreamKind); StreamPos++)
702 Fill((stream_t)StreamKind, StreamPos, "Source", file+__T(".m2ts"));
703 }
704
705 //Retrieving PID mapping
706 for (size_t StreamKind=(size_t)Stream_General+1; StreamKind<(size_t)Stream_Max; StreamKind++)
707 for (size_t StreamPos=0; StreamPos<Count_Get((stream_t)StreamKind); StreamPos++)
708 {
709 int16u PID=Retrieve((stream_t)StreamKind, StreamPos, General_ID).To_int16u();
710 PIDs_StreamKind[PID]=(stream_t)StreamKind;
711 PIDs_StreamPos[PID]=StreamPos;
712 }
713 }
714
715 //Parsing
716 int8u number_of_program_sequences;
717 Skip_B1( "Unknown");
718 Get_B1 (number_of_program_sequences, "number_of_program_sequences");
719 for (int8u program_sequence=0; program_sequence<number_of_program_sequences; program_sequence++)
720 {
721 int8u number_of_streams_in_ps;
722 Skip_B4( "Unknown");
723 Skip_B2( "program_map_PID");
724 Get_B1 (number_of_streams_in_ps, "number_of_streams_in_ps");
725 Skip_B1( "Unknown");
726 for (int16u Pos=0; Pos<number_of_streams_in_ps; Pos++)
727 {
728 Element_Begin1("Stream");
729 int16u stream_PID;
730 int8u Stream_Length;
731 Get_B2 (stream_PID, "stream_PID");
732 Get_B1 (Stream_Length, "Length");
733 int64u Stream_End=Element_Offset+Stream_Length;
734 StreamKind_Last=Stream_Max;
735 std::map<int16u, stream_t>::iterator PID_StreamKind=PIDs_StreamKind.find(stream_PID);
736 if (PID_StreamKind!=PIDs_StreamKind.end())
737 {
738 StreamKind_Last=PID_StreamKind->second;
739 StreamPos_Last=PIDs_StreamPos.find(stream_PID)->second;
740 }
741 Get_B1 (stream_type, "Stream type"); Param_Info1(Clpi_Format(stream_type)); Element_Info1(Clpi_Format(stream_type));
742 switch (Clpi_Type(stream_type))
743 {
744 case Stream_Video : StreamCodingInfo_Video(); break;
745 case Stream_Audio : StreamCodingInfo_Audio(); break;
746 case Stream_Text : StreamCodingInfo_Text() ; break;
747 default : ;
748 }
749
750 if (Stream_End-Element_Offset)
751 Skip_XX(Stream_End-Element_Offset, "Unknown");
752 Element_End0();
753
754 FILLING_BEGIN();
755 if (StreamKind_Last!=Stream_Max)
756 {
757 Fill(StreamKind_Last, StreamPos_Last, General_ID, stream_PID, 10, true);
758 Fill(StreamKind_Last, StreamPos_Last, General_ID_String, Bdmv_Decimal_Hexa(stream_PID), true);
759 }
760 FILLING_END();
761 }
762 }
763}

765struct entry
766{
767 int16u ID1;
768 int16u ID2;
769 int32u Length;
770};
771typedef std::map<int32u, entry> entries; //Key is the start address
772
773//---------------------------------------------------------------------------
774void File_Bdmv::Clpi_ExtensionData()
775{
776 entries Entries; //Key is the start address
777
778 int32u Base_Pos=(int32u)Element_Offset-4;
779
780 int8u number_of_ext_data_entries;
781 Skip_B4( "Unknown");
782 Skip_B3( "Unknown");
783 Element_Begin1("Offsets");
784 Get_B1 (number_of_ext_data_entries, "number_of_ext_data_entries");
785 for (size_t Start_Adress_Pos=0; Start_Adress_Pos<number_of_ext_data_entries; Start_Adress_Pos++)
786 {
787 int32u Start_Adress, Length;
788 int16u ID1, ID2;
789 Get_B2 (ID1, "ID1");
790 Get_B2 (ID2, "ID2");
791 Get_B4 (Start_Adress, "Start_Adress");
792 Get_B4 (Length, "Length");
793 Entries[Base_Pos+Start_Adress].ID1=ID1;
794 Entries[Base_Pos+Start_Adress].ID2=ID2;
795 Entries[Base_Pos+Start_Adress].Length=Length;
796 }
797 Element_End0();
798
799 for (entries::iterator Entry=Entries.begin(); Entry!=Entries.end(); ++Entry)
800 {
801 if (Entry->first>=Element_Offset) //If valid
802 {
803 if (Entry->first>Element_Offset)
804 Skip_XX(Entry->first-Element_Offset, "unknown");
805
806 Element_Begin1("Entry");
807 int32u length;
808 Get_B4 (length, "length");
809 int64u End=Element_Offset+length;
810 switch (Entry->second.ID1)
811 {
812 case 0x0002 :
813 switch(Entry->second.ID2)
814 {
815 case 0x0005 : Clpi_ProgramInfo(); break;
816 default: ;
817 }
818 break;
819 default : ;
820 }
821 if (End>Element_Offset)
822 Skip_XX(End-Element_Offset, "Unknown");
823 Element_End0();
824 }
825 }
826
827 if (Element_Size>Element_Offset)
828 Skip_XX(Element_Size-Element_Offset, "Unknown");
829}
