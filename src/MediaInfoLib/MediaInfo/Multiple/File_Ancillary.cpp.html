56const char* Ancillary_DataID(int8u DataID, int8u SecondaryDataID)
57{
58 // TODO: check http://www.itu.int/dms_pubrec/itu-r/rec/bt/R-REC-BT.1364-2-201003-I!!PDF-E.pdf
59 switch (DataID)
60 {
61 case 0x00 : return "Undefined";
62 case 0x08 :
63 switch (SecondaryDataID)
64 {
65 case 0x0C : return "MPEG-2 Recoding Information"; //SMPTE ST 353
66 default : return "(Reserved for 8-bit applications)";
67 }
68 break;
69 case 0x40 :
70 switch (SecondaryDataID)
71 {
72 case 0x01 : //SMPTE ST 305
73 case 0x02 : return "SDTI"; //SMPTE ST 348
74 case 0x04 : //SMPTE ST 427
75 case 0x05 : //SMPTE ST 427
76 case 0x06 : return "Link Encryption Key"; //SMPTE ST 427
77 default : return "(Internationally registered)";
78 }
79 break;
80 case 0x41 :
81 switch (SecondaryDataID)
82 {
83 case 0x01 : return "Payload identifier"; //SMPTE ST 352
84 case 0x05 : return "Bar Data"; //SMPTE ST 2016
85 case 0x06 : return "Pan-Scan Information"; //SMPTE ST 2016
86 case 0x07 : return "ANSI/SCTE 104 Messages"; //SMPTE ST 2010
87 case 0x08 : return "DVB/SCTE VBI Data"; //SMPTE ST 2031
88 default : return "(Internationally registered)";
89 }
90 break;
91 case 0x43 :
92 switch (SecondaryDataID)
93 {
94 case 0x02 : return "SDP"; //OP-47 SDP, also RDD 8
95 case 0x03 : return "Multipacket"; //OP-47 Multipacket, also RDD 8
96 case 0x05 : return "Acquisition Metadata"; //RDD 18
97 default : return "(Internationally registered)";
98 }
99 break;
100 case 0x44 :
101 switch (SecondaryDataID)
102 {
103 case 0x44 : return "ISAN or UMID"; //SMPTE RP 223
104 default : return "(Internationally registered)";
105 }
106 break;
107 case 0x45 :
108 //SMPTE 2020-1-2008
109 switch (SecondaryDataID)
110 {
111 case 0x01 : return "Audio Metadata - No association";
112 case 0x02 : return "Audio Metadata - Channels 1/2";
113 case 0x03 : return "Audio Metadata - Channels 3/4";
114 case 0x04 : return "Audio Metadata - Channels 5/6";
115 case 0x05 : return "Audio Metadata - Channels 7/8";
116 case 0x06 : return "Audio Metadata - Channels 9/10";
117 case 0x07 : return "Audio Metadata - Channels 11/12";
118 case 0x08 : return "Audio Metadata - Channels 13/14";
119 case 0x09 : return "Audio Metadata - Channels 15/16";
120 default : return "(Internationally registered)";
121 }
122 break;
123 case 0x46 :
124 switch (SecondaryDataID)
125 {
126 case 0x01 : return "Two-Frame Marker"; //SMPTE RP 2051
127 default : return "(Internationally registered)";
128 }
129 break;
130 case 0x50 :
131 switch (SecondaryDataID)
132 {
133 case 0x01 : return "WSS"; //RDD 8
134 default : return "(Reserved)";
135 }
136 break;
137 case 0x51 :
138 switch (SecondaryDataID)
139 {
140 case 0x01 : return "Film Transfer and Video Production Information"; //RP 215
141 default : return "(Reserved)";
142 }
143 break;
144 case 0x5F :
145 switch (SecondaryDataID&0xF0)
146 {
147 case 0xD0 : return "ARIB STD B37"; //ARIB STD B37
148 default : return "(Reserved)";
149 }
150 break;
151 case 0x60 :
152 switch (SecondaryDataID)
153 {
154 case 0x60 : return "ATC"; //SMPTE RP 188 / SMPTE ST 12-2
155 default : return "(Internationally registered)";
156 }
157 break;
158 case 0x61 :
159 switch (SecondaryDataID)
160 {
161 case 0x01 : return "CDP"; //SMPTE 334
162 case 0x02 : return "CEA-608"; //SMPTE 334
163 default : return "(Internationally registered)";
164 }
165 break;
166 case 0x62 :
167 switch (SecondaryDataID)
168 {
169 case 0x01 : return "Program description"; //SMPTE 334
170 case 0x02 : return "Data broadcast"; //SMPTE 334
171 case 0x03 : return "VBI data"; //SMPTE 334
172 default : return "(Internationally registered)";
173 }
174 break;
175 case 0x64 :
176 switch (SecondaryDataID)
177 {
178 case 0x64 : return "LTC"; //SMPTE RP 196
179 case 0x6F : return "VITC"; //SMPTE RP 196
180 default : return "(Internationally registered)";
181 }
182 break;
183 case 0x80 : return "Marked for deletion";
184 case 0x84 : return "Data end marker";
185 case 0x88 : return "Data start marker";
186 default :
187 if (DataID<=0x03)
188 return "(Reserved)";
189 else if (DataID<=0x0F)
190 return "(Reserved for 8-bit applications)";
191 else if (DataID<=0x3F)
192 return "(Reserved)";
193 else if (DataID<=0x4F)
194 return "(Internationally registered)";
195 else if (DataID<=0x5F)
196 return "(Reserved)";
197 else if (DataID<=0x7F)
198 return "(Internationally registered)";
199 else if (DataID<=0x83)
200 return "(Reserved)";
201 else if (DataID<=0x87)
202 return "(Reserved)";
203 else if (DataID<=0x9F)
204 return "(Reserved)";
205 else if (DataID<=0xBF)
206 return "(Internationally registered)";
207 else if (DataID<=0xCF)
208 return "User application";
209 else
210 return "(Internationally registered)";
211 }
212}

277{
278 Clear();
279 Stream_Prepare(Stream_General);
280 Fill(Stream_General, 0, General_Format, "Ancillary");
281
282 #if defined(MEDIAINFO_CDP_YES)
283 if (Cdp_Parser && !Cdp_Parser->Status[IsFinished] && Cdp_Parser->Status[IsAccepted])
284 {
285 size_t StreamPos_Base=Count_Get(Stream_Text);
286 Finish(Cdp_Parser);
287 for (size_t StreamPos=0; StreamPos<Cdp_Parser->Count_Get(Stream_Text); StreamPos++)
288 {
289 Merge(*Cdp_Parser, Stream_Text, StreamPos, StreamPos_Base+StreamPos);
290 Ztring MuxingMode=Cdp_Parser->Retrieve(Stream_Text, StreamPos, "MuxingMode");
291 Fill(Stream_Text, StreamPos_Last, "MuxingMode", __T("Ancillary data / ")+MuxingMode, true);
292 }
293
294 Ztring LawRating=Cdp_Parser->Retrieve(Stream_General, 0, General_LawRating);
295 if (!LawRating.empty())
296 Fill(Stream_General, 0, General_LawRating, LawRating, true);
297 Ztring Title=Cdp_Parser->Retrieve(Stream_General, 0, General_Title);
298 if (!Title.empty() && Retrieve(Stream_General, 0, General_Title).empty())
299 Fill(Stream_General, 0, General_Title, Title);
300 }
301 #endif //defined(MEDIAINFO_CDP_YES)
302
303 #if defined(MEDIAINFO_ARIBSTDB24B37_YES)
304 if (AribStdB34B37_Parser && !AribStdB34B37_Parser->Status[IsFinished] && AribStdB34B37_Parser->Status[IsAccepted])
305 {
306 size_t StreamPos_Base=Count_Get(Stream_Text);
307 Finish(AribStdB34B37_Parser);
308 for (size_t StreamPos=0; StreamPos<AribStdB34B37_Parser->Count_Get(Stream_Text); StreamPos++)
309 {
310 Merge(*AribStdB34B37_Parser, Stream_Text, StreamPos, StreamPos_Base+StreamPos);
311 Ztring MuxingMode=AribStdB34B37_Parser->Retrieve(Stream_Text, StreamPos, "MuxingMode");
312 Fill(Stream_Text,StreamPos_Last, "MuxingMode", __T("Ancillary data / ")+MuxingMode, true);
313 }
314 }
315 #endif //defined(MEDIAINFO_ARIBSTDB24B37_YES)
316
317 #if defined(MEDIAINFO_SDP_YES)
318 if (Sdp_Parser && !Sdp_Parser->Status[IsFinished] && Sdp_Parser->Status[IsAccepted])
319 {
320 size_t StreamPos_Base=Count_Get(Stream_Text);
321 Finish(Sdp_Parser);
322 for (size_t StreamPos=0; StreamPos<Sdp_Parser->Count_Get(Stream_Text); StreamPos++)
323 {
324 Merge(*Sdp_Parser, Stream_Text, StreamPos, StreamPos_Base+StreamPos);
325 Ztring MuxingMode=Sdp_Parser->Retrieve(Stream_General, 0, General_Format);
326 Fill(Stream_Text, StreamPos_Last, "MuxingMode", __T("Ancillary data / OP-47 / ")+MuxingMode, true);
327 }
328 }
329 #endif //defined(MEDIAINFO_SDP_YES)
330
331 #if defined(MEDIAINFO_MXF_YES)
332 if (Rdd18_Parser && !Rdd18_Parser->Status[IsFinished] && Rdd18_Parser->Status[IsAccepted])
333 {
334 size_t StreamPos_Base=Count_Get(Stream_Other);
335 Finish(Rdd18_Parser);
336 for (size_t StreamPos=0; StreamPos<Rdd18_Parser->Count_Get(Stream_Other); StreamPos++)
337 {
338 Merge(*Rdd18_Parser, Stream_Other, StreamPos, StreamPos_Base+StreamPos);
339 Fill(Stream_Other, StreamPos_Last, Other_Format, "Acquisition Metadata", Unlimited, true, true);
340 Fill(Stream_Other, StreamPos_Last, Other_MuxingMode, "Ancillary data / RDD 18");
341 }
342 }
343 #endif //defined(MEDIAINFO_MXF_YES)
344
345 //Unsupported streams
346 for (DataID = 0; DataID<Unknown.size(); DataID++)
347 for (SecondaryDataID = 0; SecondaryDataID<Unknown[DataID].size(); SecondaryDataID++)
348 for (perid::iterator Stream = Unknown[DataID][SecondaryDataID].begin(); Stream!=Unknown[DataID][SecondaryDataID].end(); ++Stream)
349 {
350 Stream_Prepare(Stream->second.StreamKind);
351 for (std::map<string, Ztring>::iterator Info=Stream->second.Infos.begin(); Info!=Stream->second.Infos.end(); ++Info)
352 Fill(Stream->second.StreamKind, StreamPos_Last, Info->first.c_str(), Info->second);
353 }

360//---------------------------------------------------------------------------
361bool File_Ancillary::Synchronize()
362{
363 //Synchronizing
364 while (Buffer_Offset+6<=Buffer_Size && ( Buffer[Buffer_Offset ]!=0x00
365 || Buffer[Buffer_Offset+ 1]!=0xFF
366 || Buffer[Buffer_Offset+ 2]!=0xFF))
367 Buffer_Offset++;
368
369 //Parsing last bytes if needed
370 if (Buffer_Offset+6>Buffer_Size)
371 {
372 if (Buffer_Offset+5==Buffer_Size && CC3(Buffer+Buffer_Offset)!=0x00FFFF)
373 Buffer_Offset++;
374 if (Buffer_Offset+4==Buffer_Size && CC3(Buffer+Buffer_Offset)!=0x00FFFF)
375 Buffer_Offset++;
376 if (Buffer_Offset+3==Buffer_Size && CC3(Buffer+Buffer_Offset)!=0x00FFFF)
377 Buffer_Offset++;
378 if (Buffer_Offset+2==Buffer_Size && CC2(Buffer+Buffer_Offset)!=0x00FF)
379 Buffer_Offset++;
380 if (Buffer_Offset+1==Buffer_Size && CC1(Buffer+Buffer_Offset)!=0x00)
381 Buffer_Offset++;
382 return false;
383 }
384
385 if (!Status[IsAccepted])
386 {
387 Accept();
388 }
389
390 //Synched is OK
391 return true;
392}
393
394//---------------------------------------------------------------------------
395bool File_Ancillary::Synched_Test()
396{
397 //Must have enough buffer for having header
398 if (Buffer_Offset+6>Buffer_Size)
399 return false;
400
401 //Quick test of synchro
402 if (CC3(Buffer+Buffer_Offset)!=0x00FFFF)
403 {
404 Synched=false;
405 if (IsSub)
406 Buffer_Offset=Buffer_Size; // We don't trust the rest of the stream, we never saw real data when sync is lost and there are lot of false-positives if we sync only on 0x00FFFF
407 }
408
409 //We continue
410 return true;
411}

443 #endif //defined(MEDIAINFO_AFDBARDATA_YES)
444
445 return;
446 }
447
448 if (!Status[IsAccepted] && !MustSynchronize)
449 Accept();

497//---------------------------------------------------------------------------
498void File_Ancillary::Header_Parse()
499{
500 //Parsing
501 if (MustSynchronize)
502 {
503 if (WithTenBit)
504 {
505 Skip_L2( "Ancillary data flag");
506 Skip_L2( "Ancillary data flag");
507 Skip_L2( "Ancillary data flag");
508 }
509 else
510 {
511 Skip_L1( "Ancillary data flag");
512 Skip_L1( "Ancillary data flag");
513 Skip_L1( "Ancillary data flag");
514 }
515 }
516 Get_L1 (DataID, "Data ID");
517 if (WithTenBit)
518 Skip_L1( "Parity+Unused"); //even:1, odd:2
519 Get_L1 (SecondaryDataID, "Secondary Data ID"); Param_Info1(Ancillary_DataID(DataID, SecondaryDataID));
520 if (WithTenBit)
521 Skip_L1( "Parity+Unused"); //even:1, odd:2
522 Get_L1 (DataCount, "Data count");
523 if (WithTenBit)
524 Skip_L1( "Parity+Unused"); //even:1, odd:2
525
526 //Test (in some container formats, Cheksum is present sometimes)
527 bool WithChecksum_Temp=WithChecksum;
528 if (!MustSynchronize && !WithChecksum && (size_t)((3+DataCount+1)*(WithTenBit?2:1))==Buffer_Size)
529 WithChecksum_Temp=true;
530
531 //Filling
532 Header_Fill_Code((((int16u)DataID)<<8)|SecondaryDataID, Ztring().From_CC1(DataID)+__T('-')+Ztring().From_CC1(SecondaryDataID));
533 Header_Fill_Size(((MustSynchronize?3:0)+3+DataCount+(WithChecksum_Temp?1:0))*(WithTenBit?2:1));
534}

536//---------------------------------------------------------------------------
537void File_Ancillary::Data_Parse()
538{
539 Element_Info1(Ancillary_DataID(DataID, SecondaryDataID));
540
541 //Buffer
542 int8u* Payload=new int8u[DataCount];
543 Element_Begin1("Raw data");
544 for(int8u Pos=0; Pos<DataCount; Pos++)
545 {
546 Get_L1 (Payload[Pos], "Data");
547 if (WithTenBit)
548 Skip_L1( "Parity+Unused"); //even:1, odd:2
549 }
550
551 //Parsing
552 if (WithChecksum)
553 Skip_L1( "Checksum");
554 if (WithTenBit)
555 Skip_L1( "Parity+Unused"); //even:1, odd:2
556 Element_End0();
557
558 FILLING_BEGIN();
559 switch (DataID)
560 {
561 case 0x08 :
562 switch (SecondaryDataID)
563 {
564 case 0x0C : // (from SMPTE ST 353)
565 if (TestAndPrepare())
566 {
567 Unknown[DataID][SecondaryDataID][string()].Infos["Format"]="MPEG-2 Recoding Information";
568 Unknown[DataID][SecondaryDataID][string()].Infos["MuxingMode"]="Ancillary data / SMPTE ST 353";
569 }
570 break;
571 default :
572 if (TestAndPrepare())
573 {
574 Unknown[DataID][SecondaryDataID][string()].Infos["Format"]=Ztring().From_CC1(DataID)+__T('-')+Ztring().From_CC1(SecondaryDataID);
575 }
576 }
577 break;
578 case 0x40 :
579 switch (SecondaryDataID)
580 {
581 case 0x01 : // (from SMPTE ST 305)
582 if (TestAndPrepare())
583 {
584 Unknown[DataID][SecondaryDataID][string()].Infos["Format"]="SDTI";
585 Unknown[DataID][SecondaryDataID][string()].Infos["MuxingMode"]="Ancillary data / SMPTE ST 305";
586 }
587 break;
588 case 0x02 : // (from SMPTE ST 348)
589 if (TestAndPrepare())
590 {
591 Unknown[DataID][SecondaryDataID][string()].Infos["Format"]="SDTI";
592 Unknown[DataID][SecondaryDataID][string()].Infos["MuxingMode"]="Ancillary data / SMPTE ST 348";
593 }
594 break;
595 case 0x04 :
596 case 0x05 :
597 case 0x06 : // (from SMPTE ST 427)
598 if (TestAndPrepare())
599 {
600 Unknown[DataID][SecondaryDataID][string()].Infos["Format"]="Link Encryption Key";
601 Unknown[DataID][SecondaryDataID][string()].Infos["MuxingMode"]="Ancillary data / SMPTE ST 427";
602 }
603 break;
604 default :
605 if (TestAndPrepare())
606 {
607 Unknown[DataID][SecondaryDataID][string()].Infos["Format"]=Ztring().From_CC1(DataID)+__T('-')+Ztring().From_CC1(SecondaryDataID);
608 }
609 }
610 break;
611 case 0x41 :
612 switch (SecondaryDataID)
613 {
614 case 0x01 : //SMPTE ST 352
615 if (TestAndPrepare())
616 {
617 Unknown[DataID][SecondaryDataID][string()].Infos["Format"]="Payload identifier";
618 Unknown[DataID][SecondaryDataID][string()].Infos["MuxingMode"]="Ancillary data / SMPTE ST 352";
619 }
620 break;
621 case 0x05 : //Bar Data (from SMPTE 2016-3), saving data for future use
622 #if defined(MEDIAINFO_AFDBARDATA_YES)
623 {
624 buffered_data* AfdBarData=new buffered_data;
625 AfdBarData->Data=new int8u[(size_t)DataCount];
626 std::memcpy(AfdBarData->Data, Payload, (size_t)DataCount);
627 AfdBarData->Size=(size_t)DataCount;
628 AfdBarData_Data.push_back(AfdBarData);
629 }
630 #endif //MEDIAINFO_AFDBARDATA_YES
631 break;
632 case 0x06 : //SMPTE ST 2016
633 if (TestAndPrepare())
634 {
635 Unknown[DataID][SecondaryDataID][string()].Infos["Format"]="Pan-Scan Information";
636 Unknown[DataID][SecondaryDataID][string()].Infos["MuxingMode"]="Ancillary data / SMPTE ST 2016";
637 }
638 break;
639 case 0x07 : //SMPTE ST 2010
640 if (TestAndPrepare())
641 {
642 Unknown[DataID][SecondaryDataID][string()].Infos["Format"]="ANSI/SCTE 104 Messages";
643 Unknown[DataID][SecondaryDataID][string()].Infos["MuxingMode"]="Ancillary data / SMPTE ST 2010";
644 }
645 break;
646 case 0x08 : //SMPTE ST 2031
647 if (TestAndPrepare())
648 {
649 Unknown[DataID][SecondaryDataID][string()].Infos["Format"]="DVB/SCTE VBI Data";
650 Unknown[DataID][SecondaryDataID][string()].Infos["MuxingMode"]="Ancillary data / SMPTE ST 2031";
651 }
652 break;
653 default :

989bool File_Ancillary::TestAndPrepare(const string* Unique)
990{
991 if (DataID>=Unknown.size())
992 Unknown.resize(DataID+1);
993 int8u RealSecondaryDataID=(DataID<0x80?SecondaryDataID:0);
994 if (RealSecondaryDataID>=Unknown[DataID].size())
995 Unknown[DataID].resize(RealSecondaryDataID+1);
996 if (Unique)
997 {
998 perid::iterator Item = Unknown[DataID][RealSecondaryDataID].find(*Unique);
999 if (Item!=Unknown[DataID][RealSecondaryDataID].end())
1000 return false;
1001 }
1002 else
1003 {
1004 if (!Unknown[DataID][RealSecondaryDataID].empty())
1005 return false;
1006 }
1007
1008 return true;
1009}
