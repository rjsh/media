63//---------------------------------------------------------------------------
64void ibi::stream::Add (const info &Info)
65{
66 if (!IsSynchronized)
67 {
68 //Searching the right insertion point
69 for (Infos_Pos=0; Infos_Pos<Infos.size(); Infos_Pos++)
70 if (Info.StreamOffset<Infos[Infos_Pos].StreamOffset)
71 break;
72 }
73
74 //Testing if new data is same as current insertion point
75 if (Infos_Pos && Infos[Infos_Pos-1].FrameNumber==Info.FrameNumber && Info.FrameNumber!=(int64u)-1)
76 {
77 IsSynchronized=true;
78 return;
79 }
80 if (Infos_Pos && Info.Dts==Infos[Infos_Pos-1].Dts && Info.Dts!=(int64u)-1)
81 {
82 //Duplicate, updating it (in case of new frame count)
83 if (Infos_Pos && Info.FrameNumber!=(int64u)-1 && Info.FrameNumber<Infos[Infos_Pos-1].FrameNumber)
84 {
85 Infos[Infos_Pos-1].FrameNumber=Info.FrameNumber;
86 Infos[Infos_Pos-1].Dts=Info.Dts;
87 }
88
89 IsSynchronized=true;
90 return;
91 }
92 if (Infos_Pos && Info.StreamOffset==Infos[Infos_Pos-1].StreamOffset)
93 {
94 //Duplicate, updating it (in case of new frame count)
95 if (Infos_Pos && Info.FrameNumber!=(int64u)-1 && Info.FrameNumber<Infos[Infos_Pos-1].FrameNumber)
96 {
97 Infos[Infos_Pos-1].FrameNumber=Info.FrameNumber;
98 Infos[Infos_Pos-1].Dts=Info.Dts;
99 }
100
101 IsSynchronized=true;
102 return;
103 }
104
105 //Previous item
106 if (IsSynchronized && Infos_Pos)
107 {
108 Infos[Infos_Pos-1].IsContinuous=true;
109 IsContinuous=true;
110
111 IsModified=true;
112 }
113 IsSynchronized=true;
114
115 //Testing if new data is same as next insertion point
116 if (Infos_Pos<Infos.size() && Infos[Infos_Pos].FrameNumber==Info.FrameNumber && Info.FrameNumber!=(int64u)-1)
117 {
118 Infos_Pos++;
119
120 IsSynchronized=true;
121 return;
122 }
123 if (Infos_Pos<Infos.size() && Info.Dts==Infos[Infos_Pos].Dts && Info.Dts!=(int64u)-1)
124 {
125 Infos_Pos++;
126
127 IsSynchronized=true;
128 return;
129 }
130 if (Infos_Pos<Infos.size() && Info.StreamOffset==Infos[Infos_Pos].StreamOffset)
131 {
132 //Duplicate, updating it (in case of new frame count)
133 if (IsSynchronized && Info.FrameNumber!=(int64u)-1)
134 Infos[Infos_Pos].FrameNumber=Info.FrameNumber;
135
136 Infos_Pos++;
137
138 IsSynchronized=true;
139 return;
140 }
141
142 Infos.insert(Infos.begin()+Infos_Pos, Info);
143 Infos_Pos++;
144
145 IsModified=true;
146}

148//---------------------------------------------------------------------------
149void ibi::stream::Unsynch ()
150{
151 Infos_Pos=0;
152 IsModified=false;
153 IsContinuous=false;
154 IsSynchronized=false;
155}

161size_t int64u2Ebml(int8u* List, int64u Value)
162{
163 //1 byte
164 if (Value<=126) //2^7-2
165 {
166 if (List)
167 {
168 List[0]=0x80|((int8u)Value);
169 }
170 return 1;
171 }
172
173 //2 bytes
174 if (Value<=16382) //2^14-2
175 {
176 if (List)
177 {
178 List[0]=0x40|((int8u)(Value>>8));
179 List[1]=(int8u)Value;
180 }
181 return 2;
182 }
183
184 //3 bytes
185 if (Value<=2097150) //2^21-2
186 {
187 if (List)
188 {
189 List[0]=0x20|((int8u)(Value>>16));
190 List[1]=(int8u)(Value>>8);
191 List[2]=(int8u)Value;
192 }
193 return 3;
194 }
195
196 //4 bytes
197 if (Value<=268435454) //2^28-2
198 {
199 if (List)
200 {
201 List[0]=0x10|((int8u)(Value>>24));
202 List[1]=(int8u)(Value>>16);
203 List[2]=(int8u)(Value>>8);
204 List[3]=(int8u)Value;
205 }
206 return 4;
207 }
208
209 //5 bytes
210 if (Value<=34359738366LL) //2^35-2
211 {
212 if (List)
213 {
214 List[0]=0x08|((int8u)(Value>>32));
215 List[1]=(int8u)(Value>>24);
216 List[2]=(int8u)(Value>>16);
217 List[3]=(int8u)(Value>>8);
218 List[4]=(int8u)Value;
219 }
220 return 5;
221 }
222
223 //6 bytes
224 if (Value<=4398046511102LL) //2^42-2
225 {
226 if (List)
227 {
228 List[0]=0x04|((int8u)(Value>>40));
229 List[1]=(int8u)(Value>>32);
230 List[2]=(int8u)(Value>>24);
231 List[3]=(int8u)(Value>>16);
232 List[4]=(int8u)(Value>>8);
233 List[5]=(int8u)Value;
234 }
235 return 6;
236 }
237
238 //7 bytes
239 if (Value<=562949953421310LL) //2^49-2
240 {
241 if (List)
242 {
243 List[0]=0x02|((int8u)(Value>>48));
244 List[1]=(int8u)(Value>>40);
245 List[2]=(int8u)(Value>>32);
246 List[3]=(int8u)(Value>>24);
247 List[4]=(int8u)(Value>>16);
248 List[5]=(int8u)(Value>>8);
249 List[6]=(int8u)Value;
250 }
251 return 7;
252 }
253
254 //8 bytes
255 if (Value<=72057594037927934LL) //2^56-2
256 {
257 if (List)
258 {
259 List[0]=0x01;
260 List[1]=(int8u)(Value>>56);
261 List[2]=(int8u)(Value>>40);
262 List[3]=(int8u)(Value>>32);
263 List[4]=(int8u)(Value>>24);
264 List[5]=(int8u)(Value>>16);
265 List[6]=(int8u)(Value>>8);
266 List[7]=(int8u)Value;
267 }
268 return 8;
269 }
270
271 if (List)
272 List[0]=0xFF;
273 return 1;
274}

276size_t EbmlBlock(int8u* List, size_t List_MaxSize, int64u Code, int8u* Content, size_t Content_Size)
277{
278 if (Content_Size==0)
279 return 0;
280
281 size_t Code_EbmlSize=int64u2Ebml(NULL, Code);
282 size_t Content_EbmlSize=int64u2Ebml(NULL, Content_Size);
283
284 if (List && Code_EbmlSize+Content_EbmlSize+Content_Size>List_MaxSize)
285 return 0;
286
287 if (List)
288 {
289 List+=int64u2Ebml(List, Code);
290 List+=int64u2Ebml(List, Content_Size);
291 std::memcpy(List, Content, Content_Size);
292 //List+=Content_Size; //Content
293 }
294
295 return Code_EbmlSize+Content_EbmlSize+Content_Size;
296}

319//---------------------------------------------------------------------------
320void File_Ibi_Creation::Set(const ibi &Ibi)
321{
322 //Source information
323 if (!Ibi.FileName.empty())
324 {
325 int64s CurrentDate=(int64s)time(NULL)*1000000000LL; //From seconds to nanoseconds
326 CurrentDate-=978307200000000000LL; //Count of nanoseconds between January 1, 1970 (time_t base) and January 1, 2001 (EBML base)
327
328 int64s LastModifiedDate=0;
329 bool LastModifiedDate_IsValid=false;
330 int64s FileSize=0;
331 bool FileSize_IsValid=false;
332
333 #if defined WINDOWS
334 HANDLE File_Handle;
335 #ifdef UNICODE
336 File_Handle=CreateFileW(Ibi.FileName.c_str(), 0, FILE_SHARE_READ|FILE_SHARE_WRITE, NULL, OPEN_EXISTING, 0, NULL);
337 #else
338 File_Handle=CreateFile(Ibi.FileName.To_Local().c_str(), 0, FILE_SHARE_READ|FILE_SHARE_WRITE, NULL, OPEN_EXISTING, 0, NULL);
339 #endif //UNICODE
340 if (File_Handle!=INVALID_HANDLE_VALUE)
341 {
342 FILETIME TimeFT;
343 if (GetFileTime(File_Handle, NULL, NULL, &TimeFT))
344 {
345 LastModifiedDate=(0x100000000LL*TimeFT.dwHighDateTime+TimeFT.dwLowDateTime)*100; //From 100-nanoseconds to nanoseconds
346 LastModifiedDate-=12622780800000000LL; //Count of nanoseconds between January 1, 1601 (Windows base) and January 1, 2001 (EBML base)
347 LastModifiedDate_IsValid=true;
348 }
349
350 DWORD High; DWORD Low=GetFileSize(File_Handle, &High);
351 if (Low!=INVALID_FILE_SIZE)
352 {
353 FileSize=0x100000000ULL*High+Low;
354 FileSize_IsValid=true;
355 }
356
357 CloseHandle(File_Handle);
358 }
359 #endif //WINDOWS
360
361 size_t BlockSizeWithoutHeader=1+1+8; //Index creation date + Source file size
362 if (LastModifiedDate_IsValid)
363 BlockSizeWithoutHeader+=1+1+8; //Source file modification date
364 if (FileSize_IsValid)
365 BlockSizeWithoutHeader+=1+1+8; //Source file size
366
367 buffer* Buffer=new buffer;
368 Buffer->Content=new int8u[1+int64u2Ebml(NULL, 1+int64u2Ebml(NULL, BlockSizeWithoutHeader))+BlockSizeWithoutHeader];
369 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 0x05); //Source information
370 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, BlockSizeWithoutHeader); //Size
371 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 0x01); //Index creation date
372 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 8); //Size
373 int64u2BigEndian(Buffer->Content+Buffer->Size, (int64u)CurrentDate); Buffer->Size+=8; //Content
374 if (LastModifiedDate_IsValid)
375 {
376 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 0x02); //Source file modification date
377 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 8); //Size
378 int64u2BigEndian(Buffer->Content+Buffer->Size, (int64u)LastModifiedDate); Buffer->Size+=8; //Content
379 }
380 if (FileSize_IsValid)
381 {
382 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 0x03); //Source file size
383 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 8); //Size
384 int64u2BigEndian(Buffer->Content+Buffer->Size, (int64u)CurrentDate); Buffer->Size+=8; //Content
385 }
386 Buffers.push_back(Buffer);
387 }
388
389 //Writing application
390 {
391 string Version=Ztring(MediaInfo_Version).SubString(__T(" - v"), Ztring()).To_UTF8();
392 buffer* Buffer=new buffer;
393 Buffer->Content=new int8u[1+int64u2Ebml(NULL, 1+1+9+1+int64u2Ebml(NULL, Version.size())+Version.size())+1+1+9+1+int64u2Ebml(NULL, Version.size())+Version.size()];
394 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 0x03); //Writing application
395 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 1+1+9+1+1+Version.size()); //Size
396 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 0x01); //Writing application name
397 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 9); //Size
398 std::memcpy(Buffer->Content+Buffer->Size, "MediaInfo", 9); Buffer->Size+=9; //Content
399 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 0x02); //Writing application version
400 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, Version.size()); //Size
401 std::memcpy(Buffer->Content+Buffer->Size, Version.c_str(), Version.size()); Buffer->Size+=Version.size(); //Content
402 Buffers.push_back(Buffer);
403 }
404
405 //InformData
406 if (!Ibi.Inform_Data.empty())
407 {
408 string Content(Ibi.Inform_Data.To_UTF8());
409 buffer* Buffer=new buffer;
410 Buffer->Content=new int8u[1+int64u2Ebml(NULL, Content.size())+Content.size()];
411 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 0x04); //InformData
412 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, Content.size()); //Size
413 std::memcpy(Buffer->Content+Buffer->Size, Content.c_str(), Content.size()); Buffer->Size+=Content.size(); //Content
414 Buffers.push_back(Buffer);
415 }
416
417 //Streams
418 for (ibi::streams::const_iterator IbiStream_Temp=Ibi.Streams.begin(); IbiStream_Temp!=Ibi.Streams.end(); ++IbiStream_Temp)
419 Add(IbiStream_Temp->first, *IbiStream_Temp->second);
420}

422//---------------------------------------------------------------------------
423void File_Ibi_Creation::Add(int64u ID, const ibi::stream &Stream)
424{
425 //Useful?
426 if (Stream.Infos.empty())
427 return;
428
429 //Header
430 int8u* IbiHeader;
431 size_t IbiHeader_Offset=0;
432 if (ID!=(int64u)-1)
433 {
434 IbiHeader=new int8u[int64u2Ebml(NULL, ID)];
435 IbiHeader_Offset+=int64u2Ebml(IbiHeader, ID);
436 }
437 else
438 IbiHeader=NULL;
439
440 //Init - Byte Offset
441 int8u* IbiByteOffset=new int8u[8*Stream.Infos.size()];
442 size_t IbiByteOffset_Offset=0;
443
444 //Init - Frame Number
445 int8u* IbiFrameNumber=new int8u[8*Stream.Infos.size()];
446 size_t IbiFrameNumber_Offset=0;
447
448 //Init - DTS
449 int8u* IbiDts=new int8u[16+8*Stream.Infos.size()];
450 size_t IbiDts_Offset=0;
451 IbiDts_Offset+=int64u2Ebml(IbiDts+IbiDts_Offset, Stream.DtsFrequencyNumerator);
452 IbiDts_Offset+=int64u2Ebml(IbiDts+IbiDts_Offset, Stream.DtsFrequencyDenominator);
453
454 //Per item
455 for (size_t Pos=0; Pos<Stream.Infos.size(); Pos++)
456 {
457 IbiByteOffset_Offset+=int64u2Ebml(IbiByteOffset+IbiByteOffset_Offset, Stream.Infos[Pos].StreamOffset-(Pos?Stream.Infos[Pos-1].StreamOffset:0));
458 IbiFrameNumber_Offset+=int64u2Ebml(IbiFrameNumber+IbiFrameNumber_Offset, Stream.Infos[Pos].FrameNumber-(Pos?Stream.Infos[Pos-1].FrameNumber:0));
459 IbiDts_Offset+=int64u2Ebml(IbiDts+IbiDts_Offset, Stream.Infos[Pos].Dts-(Pos?Stream.Infos[Pos-1].Dts:0));
460 }
461
462 //Sizes
463 size_t IbiHeader_EbmlSize=EbmlBlock(NULL, 0, 0x01, NULL, IbiHeader_Offset);
464 size_t IbiByteOffset_EbmlSize=EbmlBlock(NULL, 0, 0x02, NULL, IbiByteOffset_Offset);
465 size_t IbiFrameNumber_EbmlSize=EbmlBlock(NULL, 0, 0x03, NULL, IbiFrameNumber_Offset);
466 size_t IbiDts_EbmlSize=EbmlBlock(NULL, 0, 0x04, NULL, IbiDts_Offset);
467
468 //Buffer
469 buffer* Buffer=new buffer;
470 Buffer->Content=new int8u[1+int64u2Ebml(NULL, IbiHeader_EbmlSize+IbiByteOffset_EbmlSize+IbiFrameNumber_EbmlSize+IbiDts_EbmlSize)+IbiHeader_EbmlSize+IbiByteOffset_EbmlSize+IbiFrameNumber_EbmlSize+IbiDts_EbmlSize];
471 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, 0x01); //Stream
472 Buffer->Size+=int64u2Ebml(Buffer->Content+Buffer->Size, IbiHeader_EbmlSize+IbiByteOffset_EbmlSize+IbiFrameNumber_EbmlSize+IbiDts_EbmlSize); //Size
473 Buffer->Size+=EbmlBlock(Buffer->Content+Buffer->Size, IbiHeader_EbmlSize, 0x01, IbiHeader, IbiHeader_Offset); //IbiHeader
474 Buffer->Size+=EbmlBlock(Buffer->Content+Buffer->Size, IbiByteOffset_EbmlSize, 0x02, IbiByteOffset, IbiByteOffset_Offset); //ByteOffset
475 Buffer->Size+=EbmlBlock(Buffer->Content+Buffer->Size, IbiFrameNumber_EbmlSize, 0x03, IbiFrameNumber, IbiFrameNumber_Offset); //FrameNumber
476 Buffer->Size+=EbmlBlock(Buffer->Content+Buffer->Size, IbiDts_EbmlSize, 0x04, IbiDts, IbiDts_Offset); //Dts
477
478 Buffers.push_back(Buffer);
479
480 //Finish
481 delete[] IbiByteOffset; //IbiByteOffset=NULL;
482 delete[] IbiFrameNumber; //IbiFrameNumber=NULL;
483 delete[] IbiDts; //IbiDts=NULL;
484}

486Ztring File_Ibi_Creation::Finish()
487{
488 //Test
489 if (Buffers.empty())
490 return Ztring();
491
492 //Size computing
493 size_t Size=4+1+2+1+15+2+1+1; //Header size
494 for (size_t Pos=0; Pos<Buffers.size(); Pos++)
495 Size+=Buffers[Pos]->Size;
496
497 //Buffer creation
498 int8u* Main=new int8u[Size];
499 size_t Main_Offset=0;
500
501 //Header
502 size_t Header_Offset=4+1+2+1+15+2+1+1; //Size (Code + Size + Content, twice)
503 Main_Offset+=int64u2Ebml(Main+Main_Offset, 0x0A45DFA3); //EBML
504 Main_Offset+=int64u2Ebml(Main+Main_Offset, Header_Offset-(4+1)); //Size (Complete header size minus header header size)
505 Main_Offset+=int64u2Ebml(Main+Main_Offset, 0x0282); //DocType
506 Main_Offset+=int64u2Ebml(Main+Main_Offset, 15); //Size
507 std::memcpy(Main+Main_Offset, "MediaInfo Index", 15); Main_Offset+=15; //Content
508 Main_Offset+=int64u2Ebml(Main+Main_Offset, 0x0285); //DocTypeReadVersion
509 Main_Offset+=int64u2Ebml(Main+Main_Offset, 1); //Size
510 Main[Main_Offset]=0x01; Main_Offset+=1; //Content
511
512 //for each stream
513 for (size_t Pos=0; Pos<Buffers.size(); Pos++)
514 {
515 std::memcpy(Main+Main_Offset, Buffers[Pos]->Content, Buffers[Pos]->Size);
516 Main_Offset+=Buffers[Pos]->Size;
517 }
518
519 //Compressed
520 if (Header_Offset < Main_Offset)
521 {
522 buffer Buffer;
523 size_t UncompressedSize = Main_Offset - Header_Offset;
524 int8u* Compressed = new int8u[UncompressedSize];
525 unsigned long CompressedSize = (unsigned long)Main_Offset;
526 if (compress2(Compressed, &CompressedSize, Main + Header_Offset, (unsigned long)UncompressedSize, Z_BEST_COMPRESSION) == Z_OK && CompressedSize < UncompressedSize)
527 {
528 Main_Offset = Header_Offset; //Removing uncompressed content
529 Main_Offset += int64u2Ebml(Main + Main_Offset, 0x02); //Compressed index
530 Main_Offset += int64u2Ebml(Main + Main_Offset, int64u2Ebml(NULL, UncompressedSize) + CompressedSize); //Size
531 Main_Offset += int64u2Ebml(Main + Main_Offset, UncompressedSize); //Uncompressed size
532
533 //Filling
534 Buffer.Size = Main_Offset + CompressedSize;
535 Buffer.Content = new int8u[Buffer.Size];
536 std::memcpy(Buffer.Content, Main, Main_Offset); //File header + compressed data header
537 std::memcpy(Buffer.Content + Main_Offset, Compressed, CompressedSize); //Compressed data
538 }
539 else
540 {
541 //Filling
542 Buffer.Size = Size;
543 Buffer.Content = new int8u[Buffer.Size];
544 std::memcpy(Buffer.Content, Main, Size);
545 }
546
547 std::string Data_Raw((const char*)Buffer.Content, Buffer.Size);
548 std::string Data_Base64(Base64::encode(Data_Raw));
549
550 delete[] Main; //Main=NULL;
551
552 return Ztring().From_UTF8(Data_Base64);
553 }
554 else
555 {
556 delete[] Main; //Main=NULL;
557
558 return Ztring();
559 }
560}
