445//---------------------------------------------------------------------------
446void File_Flv::Streams_Fill()
447{
448 //Coherency
449 if (Count_Get(Stream_Video) && Count_Get(Stream_Audio) && !Retrieve(Stream_Video, 0, Video_BitRate).empty() && Retrieve(Stream_Audio, 0, Audio_BitRate).empty())
450 {
451 Fill(Stream_General, 0, General_OverallBitRate, Retrieve(Stream_Video, 0, Video_BitRate));
452 Clear(Stream_Video, 0, Video_BitRate);
453 }
454
455 //Trying to detect VFR
456 std::vector<int64u> video_stream_FrameRate_Between;
457 for (size_t Pos=1; Pos<video_stream_FrameRate.size(); Pos++)
458 video_stream_FrameRate_Between.push_back(video_stream_FrameRate[Pos]-video_stream_FrameRate[Pos-1]);
459 std::sort(video_stream_FrameRate_Between.begin(), video_stream_FrameRate_Between.end());
460 if (!video_stream_FrameRate_Between.empty())
461 {
462 if (video_stream_FrameRate_Between[0]*0.9<video_stream_FrameRate_Between[video_stream_FrameRate_Between.size()-1]
463 && video_stream_FrameRate_Between[0]*1.1>video_stream_FrameRate_Between[video_stream_FrameRate_Between.size()-1])
464 {
465 float Time;
466 if (video_stream_FrameRate.size()>30)
467 Time=((float)(video_stream_FrameRate[30]-video_stream_FrameRate[0]))/30; //30 frames for handling 30 fps rounding problems
468 else
469 Time=((float)(video_stream_FrameRate[video_stream_FrameRate.size()-1]-video_stream_FrameRate[0]))/(video_stream_FrameRate.size()-1); //30 frames for handling 30 fps rounding problems
470 if (Time)
471 {
472 Fill(Stream_Video, 0, Video_FrameRate, 1000/Time);
473 Fill(Stream_Video, 0, Video_FrameRate_Mode, "CFR");
474 }
475 }
476 else
477 Fill(Stream_Video, 0, Video_FrameRate_Mode, "VFR");
478 }
479
480 //Parsers
481 if (Stream[Stream_Video].Parser!=NULL)
482 {
483 Fill(Stream[Stream_Video].Parser);
484 Merge(*Stream[Stream_Video].Parser, Stream_Video, 0, 0);
485 }
486 if (Stream[Stream_Audio].Parser!=NULL)
487 {
488 Fill(Stream[Stream_Audio].Parser);
489 Merge(*Stream[Stream_Audio].Parser, Stream_Audio, 0, 0);
490
491 //Special case: AAC
492 if (Retrieve(Stream_Audio, 0, Audio_Format)==__T("AAC")
493 || Retrieve(Stream_Audio, 0, Audio_Format)==__T("MPEG Audio")
494 || Retrieve(Stream_Audio, 0, Audio_Format)==__T("Vorbis"))
495 Clear(Stream_Audio, 0, Audio_BitDepth); //Resolution is not valid for AAC / MPEG Audio / Vorbis
496 }
497
498 //Delay
499 if (Stream[Stream_Video].Delay!=(int32u)-1)
500 {
501 Fill(Stream_Video, 0, Video_Delay, Stream[Stream_Video].Delay+Retrieve(Stream_Video, 0, Video_Delay).To_int32u(), 10, true);
502 Fill(Stream_Video, 0, Video_Delay_Source, "Container");
503 }
504 if (Stream[Stream_Audio].Delay!=(int32u)-1)
505 {
506 Fill(Stream_Audio, 0, Audio_Delay, Stream[Stream_Audio].Delay+Retrieve(Stream_Audio, 0, Audio_Delay).To_int32u(), 10, true);
507 Fill(Stream_Audio, 0, Audio_Delay_Source, "Container");
508 }

570//---------------------------------------------------------------------------
571void File_Flv::Streams_Finish_PerStream(stream_t StreamKind)
572{
573 if (Stream[StreamKind].TimeStamp!=(int32u)-1)
574 {
575 //Calculating the last timestamp (last block included)
576 if (!Stream[StreamKind].Durations.empty())
577 {
578 int64u Durations_Total=0;
579 for (size_t Pos=0; Pos<Stream[StreamKind].Durations.size(); Pos++)
580 Durations_Total+=Stream[StreamKind].Durations[Pos];
581 int32u Duration_Average=float32_int32s(((float32)Durations_Total)/Stream[StreamKind].Durations.size());
582 Stream[StreamKind].TimeStamp+=Duration_Average;
583 }
584
585 Fill(StreamKind, 0, "Duration", Stream[StreamKind].TimeStamp, 10, true);
586 }

594bool File_Flv::FileHeader_Begin()
595{
596 //Synchro
597 if (3>Buffer_Size)
598 return false;
599 if (Buffer[0]!=0x46 //"FLV"
600 || Buffer[1]!=0x4C
601 || Buffer[2]!=0x56)
602 {
603 Reject();
604 return false;
605 }
606 if (9>Buffer_Size)
607 return false;
608
609 return true;

612//---------------------------------------------------------------------------
613void File_Flv::FileHeader_Parse()
614{
615 //Parsing
616 Element_Begin1("FLV header");
617 int32u Size;
618 int8u Version, Flags;
619 Skip_String(3, "Signature");
620 Get_B1 (Version, "Version");
621 Get_B1 (Flags, "Flags");
622 Get_Flags (Flags, 0, video_stream_Count, "Video");
623 Get_Flags (Flags, 2, audio_stream_Count, "Audio");
624 Get_B4 (Size, "Size");
625 if (Size>9)
626 Skip_XX(Size-9, "Unknown");
627 Element_End0();
628
629 FILLING_BEGIN();
630 //Integrity
631 if (Version==0 || Size<9)
632 {
633 Reject();
634 return;
635 }
636
637 //Filling
638 Accept();
639
640 Fill(Stream_General, 0, General_Format, "Flash Video");
641 if (!video_stream_Count && !audio_stream_Count)
642 {
643 //TODO: quick and awful hack for a file having both bools unset, should detect directly the streams
644 video_stream_Count=true;
645 audio_stream_Count=true;
646 }
647 if (video_stream_Count)
648 {
649 Stream_Prepare(Stream_Video);
650 #if MEDIAINFO_DEMUX
651 if (Config->Demux_ForceIds_Get())
652 Fill(Stream_Video, 0, Video_ID, 9);
653 #endif //MEDIAINFO_DEMUX
654 video_stream_FrameRate_Detected=false;
655 }
656 else
657 video_stream_FrameRate_Detected=true;
658 if (audio_stream_Count)
659 {
660 Stream_Prepare(Stream_Audio);
661 #if MEDIAINFO_DEMUX
662 if (Config->Demux_ForceIds_Get())
663 Fill(Stream_Audio, 0, Audio_ID, 8);
664 #endif //MEDIAINFO_DEMUX
665 }
666
667 if (Version>1)
668 {
669 Finish();
670 return; //Version more than 1 is not supported
671 }
672 FILLING_ELSE()
673 Reject();
674 FILLING_END();
675}

681//---------------------------------------------------------------------------
682bool File_Flv::Synchronize()
683{
684 if (File_Offset+Buffer_Offset+4==File_Size)
685 return true; // Used by seek from end
686
687 //Synchronizing
688 while (Buffer_Offset+15<=Buffer_Size)
689 {
690 int32u BodyLength=BigEndian2int24u(Buffer+Buffer_Offset+5);
691 if ((Buffer[Buffer_Offset ]
692 || Buffer[Buffer_Offset+1]
693 || Buffer[Buffer_Offset+2]
694 || Buffer[Buffer_Offset+3]>=11)
695 && File_Offset+Buffer_Offset+15+BodyLength==File_Size)
696 break; //Last block
697 if (File_Offset+Buffer_Offset+15+BodyLength<File_Size)
698 {
699 if (Buffer_Offset+15+BodyLength+15>Buffer_Size)
700 return false; //Need more data
701
702 if ((Buffer[Buffer_Offset ]
703 || Buffer[Buffer_Offset+1]
704 || Buffer[Buffer_Offset+2]
705 || Buffer[Buffer_Offset+3]>=11)
706 && (BigEndian2int32u(Buffer+Buffer_Offset+15+BodyLength)==11+BodyLength // PreviousTagSize
707 || BigEndian2int32u(Buffer+Buffer_Offset+15+BodyLength)==BodyLength)) // PreviousTagSize without 11, found in some buggy files
708 {
709 PreviousTagSize_Add11=(BigEndian2int32u(Buffer+Buffer_Offset+15+BodyLength)==BodyLength)?0:11;
710 break;
711 }
712 }
713
714 Buffer_Offset++;
715 }
716
717 //Parsing last bytes if needed
718 if (Buffer_Offset+15>Buffer_Size)
719 return false;
720
721 //Synched
722 return true;
723}

725//---------------------------------------------------------------------------
726bool File_Flv::Synched_Test()
727{
728 if (File_Offset+Buffer_Offset+4==File_Size)
729 return true; // Used by seek from end
730
731 //Must have enough buffer for having header
732 if (Buffer_Offset+15>Buffer_Size)
733 return false;
734
735 //Quick test of synchro
736 if (Buffer[Buffer_Offset ]==0
737 && Buffer[Buffer_Offset+1]==0
738 && Buffer[Buffer_Offset+2]==0
739 && Buffer[Buffer_Offset+3]<PreviousTagSize_Add11
740 && File_Offset+Buffer_Offset>9)
741 {
742 if (Searching_Duration)
743 {
744 //Error during parsing, stopping
745 Finish();
746 Searching_Duration=false;
747 GoTo(File_Size);
748 return true;
749 }
750
751 Synched=false;
752 return true;
753 }
754
755 //We continue
756 return true;
757}

777//***************************************************************************
778// Buffer - Per element
779//***************************************************************************
780
781//---------------------------------------------------------------------------
782void File_Flv::Header_Parse()
783{
784 if (Searching_Duration && File_Offset+Buffer_Offset==File_Size-4)
785 {
786 Get_B4 (PreviousTagSize, "PreviousTagSize");
787
788 //Filling
789 Header_Fill_Code((int64u)-1, "End Of File");
790 Header_Fill_Size(4);
791 return;
792 }
793
794 //Parsing
795 int32u BodyLength;
796 int8u Type;
797 Get_B4 (PreviousTagSize, "PreviousTagSize");
798 if (File_Offset+Buffer_Offset+4<File_Size)
799 {
800 int32u Timestamp_Base;
801 int8u Timestamp_Extended;
802 Get_B1 (Type, "Type"); //Param_Info1(Type<19?Flv_Type[Type]:__T("Unknown"));
803 Get_B3 (BodyLength, "BodyLength");
804 Get_B3 (Timestamp_Base, "Timestamp_Base"); //in ms
805 Get_B1 (Timestamp_Extended, "Timestamp_Extended"); //TimeStamp = Timestamp_Extended*0x01000000+Timestamp_Base
806 Skip_B3( "StreamID");
807
808 // For audio, check if it's just an audio config.
809 bool Skip_Timestamps=false;
810 if (Type==0x08)
811 {
812 int16u Format_Info;
813 Peek_B2(Format_Info);
814 int8u Format=(Format_Info>>12)&0x0F;
815 if (Format==10 && (Format_Info&0xFF)==0) // AAC sequence header
816 Skip_Timestamps=true;
817 }
818
819 //Filling
820 if ((Type==0x08 && !Skip_Timestamps) || Type==0x09)
821 {
822 Time=(((int32u)Timestamp_Extended)<<24)|Timestamp_Base;
823 stream_t StreamKind=(Type==0x08)?Stream_Audio:Stream_Video;
824 if (Stream[StreamKind].Delay==(int32u)-1)
825 Stream[StreamKind].Delay=Time;
826 else if (Stream[StreamKind].TimeStamp!=(int32u)-1 && Time>Stream[StreamKind].TimeStamp)
827 Stream[StreamKind].Durations.push_back(Time-Stream[StreamKind].TimeStamp);
828 if (!Searching_Duration || Stream[StreamKind].TimeStamp==(int32u)-1)
829 Stream[StreamKind].TimeStamp=Time;
830 }
831
832 if (Type==0)
833 Trusted_IsNot("Wrong type");
834 }
835 else
836 {
837 Type=0;
838 BodyLength=0;
839 }
840
841 //Filling
842 Header_Fill_Code(Type, Ztring().From_Number(Type, 16));
843 Header_Fill_Size(Element_Offset+BodyLength);
844}

846//---------------------------------------------------------------------------
847void File_Flv::Data_Parse()
848{
849 switch (Element_Code)
850 {
851 case 0x00 : Element_Name("End Of File"); break;
852 case 0x08 : audio(); break;
853 case 0x09 : video(); break;
854 case 0x12 : meta(); break;
855 case 0xFA : Rm(); break;
856 case (int64u)-1 : //When searching the last frame
857 if (8+PreviousTagSize>File_Size)
858 {
859 Searching_Duration=false;
860 Open_Buffer_Unsynch(); //There is a problem, trying to sync
861 PreviousTagSize=1024*1024;
862 }
863 GoTo(File_Size-PreviousTagSize-8, "FLV");
864 return;
865 default : if (Searching_Duration)
866 {
867 Finish(); //This is surely a bad en of file, don't try anymore
868 return;
869 }
870
871 }

873 if (Searching_Duration)

876 && (Count_Get(Stream_Audio)==0 || Stream[Stream_Audio].TimeStamp!=(int32u)-1))
877 || (File_Size>1024*1024*2 && File_Offset+Buffer_Offset-Header_Size-PreviousTagSize-4<File_Size-1024*1024))
878 && Config->ParseSpeed<1)

885 GoToFromEnd(Header_Size+Element_Size+1024*1024);

887 }

890 }

894
895 //Trying to find the last frame for duration
896 Read_Buffer_Unsynched(); //This is not synched yet, so we call directly this method instead of Open_Buffer_Unsynched
