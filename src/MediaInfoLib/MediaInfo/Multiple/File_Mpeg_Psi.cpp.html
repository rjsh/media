609// Init: int32u CRC_32 = 0xFFFFFFFF;
610// for each data byte do
611// CRC_32=(CRC_32<<8) ^ CRC_32_Table[(CRC_32>>24)^(data_byte)];

613{
614 0x00000000, 0x04C11DB7, 0x09823B6E, 0x0D4326D9,

785//---------------------------------------------------------------------------
786void File_Mpeg_Psi::Header_Parse()
787{
788 //From Program stream
789 if (!From_TS)
790 {
791 //Filling
792 table_id=0xFF; //Make it invalid
793 section_syntax_indicator=false;
794 Header_Fill_Code((int64u)-1, "program_stream_map"); //(int64u)-1 for precising "out of scope"
795 Header_Fill_Size(Element_Size-4);
796 return;
797 }
798
799 //Parsing
800 int16u section_length;
801 Get_B1 (table_id, "table_id");
802 BS_Begin();
803 Get_SB ( section_syntax_indicator, "section_syntax_indicator");
804 Skip_SB( "private_indicator");
805 Skip_S1( 2, "reserved");
806 Get_S2 (12, section_length, "section_length");
807 BS_End();
808
809 //Size
810 if ((size_t)section_length<Element_Offset+(section_syntax_indicator?4:0)) //We must have 4 more byte for CRC
811 {
812 Reject("PSI"); //Error, we exit
813 return;
814 }
815 if (Element_Size<Element_Offset+section_length)
816 {
817 Element_WaitForMoreData();
818 return;
819 }
820 //Element[Element_Level-1].IsComplete=true;
821
822 //CRC32
823 if (table_id<=0x06 && !section_syntax_indicator)
824 {
825 Trusted_IsNot("CRC error");
826 CRC_32=0xffffffff;
827 Reject();
828 return;
829 }
830 if (section_syntax_indicator || table_id==0xC1)
831 {
832 CRC_32=0xffffffff;
833 const int8u* CRC_32_Buffer=Buffer+Buffer_Offset+(size_t)Element_Offset-3; //table_id position
834
835 while(CRC_32_Buffer<Buffer+Buffer_Offset+(size_t)Element_Offset+section_length) //from table_id to the end, CRC_32 included
836 {
837 CRC_32=(CRC_32<<8) ^ Psi_CRC_32_Table[(CRC_32>>24)^(*CRC_32_Buffer)];
838 CRC_32_Buffer++;
839 }
840
841 if (CRC_32)
842 {
843 Trusted_IsNot("CRC error");
844 Reject();
845 return;
846 }
847 }
848
849 //Filling
850 Header_Fill_Code(table_id, Ztring().From_Number(table_id, 16));
851 Header_Fill_Size(3+section_length);
852}
