90//***************************************************************************
91// Buffer - File header
92//***************************************************************************
93
94//---------------------------------------------------------------------------
95bool File_P2_Clip::FileHeader_Begin()
96{
97 XMLDocument document;
98 if (!FileHeader_Begin_XML(document))
99 return false;
100
101 {
102 XMLElement* Root=document.FirstChildElement("P2Main");
103 if (Root)
104 {
105 Accept("P2_Clip");
106 Fill(Stream_General, 0, General_Format, "P2 Clip");
107
108 ReferenceFiles=new File__ReferenceFilesHelper(this, Config);
109
110 XMLElement* ClipContent=Root->FirstChildElement("ClipContent");
111 if (ClipContent)
112 {
113 XMLElement* ChildElement;
114
115 //ID
116 ChildElement=ClipContent->FirstChildElement("GlobalClipID");
117 if (ChildElement)
118 Fill(Stream_General, 0, General_UniqueID, ChildElement->GetText());
119
120 //Duration
121 Ztring Duration, EditUnit;
122 ChildElement=ClipContent->FirstChildElement("Duration");
123 if (ChildElement)
124 Duration=ChildElement->GetText();
125 ChildElement=ClipContent->FirstChildElement("EditUnit");
126 if (ChildElement)
127 EditUnit=ChildElement->GetText();
128 int64u Duration_Frames=Duration.To_int64u();
129 int64u EditUnit_Numerator=EditUnit.SubString(Ztring(), __T("/")).To_int64u();
130 int64u EditUnit_Denominator=EditUnit.SubString(__T("/"), Ztring()).To_int64u();
131 if (Duration_Frames && EditUnit_Numerator && EditUnit_Denominator)
132 Fill(Stream_General, 0, General_Duration, ((float32)Duration_Frames)*1000*EditUnit_Numerator/EditUnit_Denominator, 0);
133
134 //EssenceList
135 XMLElement* EssenceList=ClipContent->FirstChildElement("EssenceList");
136 if (EssenceList)
137 {
138 XMLElement* Track=EssenceList->FirstChildElement();
139 #if defined(MEDIAINFO_MXF_YES)
140 size_t Audio_Count=0;
141 #endif
142 while (Track)
143 {
144 string Field=Track->Value();
145 if (Field=="Video")
146 {
147 sequence* Sequence=new sequence;
148 float64 FrameRate=0;
149
150 //FrameRate
151 ChildElement=Track->FirstChildElement("FrameRate");
152 if (ChildElement)
153 {
154 Ztring FrameRateS=Ztring(ChildElement->GetText());
155 if (FrameRateS.find(__T("23.97"))==0)
156 FrameRate=((float64)24)*1000/1001;
157 else if (FrameRateS.find(__T("29.97"))==0)
158 FrameRate=((float64)30)*1000/1001;
159 else if (FrameRateS.find(__T("59.94"))==0)
160 FrameRate=((float64)60)*1000/1001;
161 else
162 FrameRate=FrameRateS.To_float64();
163 if (FrameRateS.find('i')!=string::npos)
164 FrameRate/=2;
165 Sequence->FrameRate_Set(FrameRate);
166 }
167
168 //StartTimecode
169 ChildElement=Track->FirstChildElement("StartTimecode");
170 if (ChildElement)
171 {
172 string Text=ChildElement->GetText();
173 if (Text.size()==11)
174 {
175 int64u ToFill=(Text[0]-'0')*10*60*60*1000
176 + (Text[1]-'0') *60*60*1000
177 + (Text[3]-'0') *10*60*1000
178 + (Text[4]-'0') *60*1000
179 + (Text[6]-'0') *10*1000
180 + (Text[7]-'0') *1000;
181 if (FrameRate)
182 ToFill+=float64_int64s(((Text[9]-'0')*10+(Text[10]-'0'))*1000/FrameRate);
183 //Fill(Stream_Video, StreamPos_Last, Video_Delay, ToFill);
184 //Fill(Stream_Video, StreamPos_Last, Video_Delay_Source, "P2 Clip");
185 }
186 }
187
188 #if defined(MEDIAINFO_MXF_YES)
189 if (File_Name.size()>10+1+4+1
190 && File_Name[File_Name.size()-10-1]==PathSeparator
191 && (File_Name[File_Name.size()-10-2]&(~0x20))==__T('P')
192 && (File_Name[File_Name.size()-10-3]&(~0x20))==__T('I')
193 && (File_Name[File_Name.size()-10-4]&(~0x20))==__T('L')
194 && (File_Name[File_Name.size()-10-5]&(~0x20))==__T('C')
195 && File_Name[File_Name.size()-10-6]==PathSeparator)
196 {
197 Ztring file=File_Name.substr(File_Name.size()-10, 6);
198 Ztring MXF_File=__T("..");
199 MXF_File+=PathSeparator;
200
201 Ztring Path=File_Name.substr(0, File_Name.size()-10-6);
202 ZtringList Dirs=Dir::GetAllFileNames(Path, Dir::Include_Dirs);
203 bool Exists=false;
204 for (size_t Pos=0; Pos<Dirs.size(); Pos++)
205 if (Dirs[Pos].size()>6
206 && (Dirs[Pos][Dirs[Pos].size()-1]&(~0x20))==__T('O')
207 && (Dirs[Pos][Dirs[Pos].size()-2]&(~0x20))==__T('E')
208 && (Dirs[Pos][Dirs[Pos].size()-3]&(~0x20))==__T('D')
209 && (Dirs[Pos][Dirs[Pos].size()-4]&(~0x20))==__T('I')
210 && (Dirs[Pos][Dirs[Pos].size()-5]&(~0x20))==__T('V')
211 && Dirs[Pos][Dirs[Pos].size()-6]==PathSeparator)
212 {
213 MXF_File+=Dirs[Pos].substr(Dirs[Pos].size()-5, 5);
214 Exists=true;
215 break;
216 }
217 if (!Exists)
218 MXF_File+=__T("AUDIO");
219
220 MXF_File+=PathSeparator;
221 MXF_File+=file;
222
223 Dirs=Dir::GetAllFileNames(File_Name.substr(0, File_Name.size()-10-1)+PathSeparator+MXF_File+__T(".MXF"), Dir::Include_Files);
224 Exists=false;
225 for (size_t Pos=0; Pos<Dirs.size(); Pos++)
226 if (Dirs[Pos].size()>4
227 && (Dirs[Pos][Dirs[Pos].size()-1]&(~0x20))==__T('F')
228 && (Dirs[Pos][Dirs[Pos].size()-2]&(~0x20))==__T('X')
229 && (Dirs[Pos][Dirs[Pos].size()-3]&(~0x20))==__T('M')
230 && Dirs[Pos][Dirs[Pos].size()-4]==__T('.'))
231 {
232 MXF_File+=Dirs[Pos].substr(Dirs[Pos].size()-4, 4);
233 Exists=true;
234 break;
235 }
236 if (!Exists)
237 MXF_File+=__T(".MXF");
238
239 Sequence->AddFileName(MXF_File);
240 Sequence->StreamKind=Stream_Video;
241 Sequence->StreamID=ReferenceFiles->Sequences_Size()+1;
242 ReferenceFiles->AddSequence(Sequence);
243 }
244 #endif //defined(MEDIAINFO_MXF_YES)
245 }
246 else if (Field=="Audio")
247 {
248 #if defined(MEDIAINFO_MXF_YES)
249 if (File_Name.size()>10+1+4+1
250 && File_Name[File_Name.size()-10-1]==PathSeparator
251 && (File_Name[File_Name.size()-10-2]&(~0x20))==__T('P')
252 && (File_Name[File_Name.size()-10-3]&(~0x20))==__T('I')
253 && (File_Name[File_Name.size()-10-4]&(~0x20))==__T('L')
254 && (File_Name[File_Name.size()-10-5]&(~0x20))==__T('C')
255 && File_Name[File_Name.size()-10-6]==PathSeparator)
256 {
257 Ztring file=File_Name.substr(File_Name.size()-10, 6);
258 Ztring MXF_File=__T("..");
259 MXF_File+=PathSeparator;
260
261 Ztring Path=File_Name.substr(0, File_Name.size()-10-6);
262 ZtringList Dirs=Dir::GetAllFileNames(Path, Dir::Include_Dirs);
263 bool Exists=false;
264 for (size_t Pos=0; Pos<Dirs.size(); Pos++)
265 if (Dirs[Pos].size()>6
266 && (Dirs[Pos][Dirs[Pos].size()-1]&(~0x20))==__T('O')
267 && (Dirs[Pos][Dirs[Pos].size()-2]&(~0x20))==__T('I')
268 && (Dirs[Pos][Dirs[Pos].size()-3]&(~0x20))==__T('D')
269 && (Dirs[Pos][Dirs[Pos].size()-4]&(~0x20))==__T('U')
270 && (Dirs[Pos][Dirs[Pos].size()-5]&(~0x20))==__T('A')
271 && Dirs[Pos][Dirs[Pos].size()-6]==PathSeparator)
272 {
273 MXF_File+=Dirs[Pos].substr(Dirs[Pos].size()-5, 5);
274 Exists=true;
275 break;
276 }
277 if (!Exists)
278 MXF_File+=__T("AUDIO");
279
280 MXF_File+=PathSeparator;
281 MXF_File+=file;
282 Ztring Pos=Ztring::ToZtring(Audio_Count);
283 if (Pos.size()<2)
284 Pos.insert(0, 1, __T('0'));
285 MXF_File+=Pos;
286
287 Dirs=Dir::GetAllFileNames(File_Name.substr(0, File_Name.size()-10-1)+PathSeparator+MXF_File+__T(".MXF"), Dir::Include_Files);
288 Exists=false;
289 for (size_t Pos=0; Pos<Dirs.size(); Pos++)
290 if (Dirs[Pos].size()>4
291 && (Dirs[Pos][Dirs[Pos].size()-1]&(~0x20))==__T('F')
292 && (Dirs[Pos][Dirs[Pos].size()-2]&(~0x20))==__T('X')
293 && (Dirs[Pos][Dirs[Pos].size()-3]&(~0x20))==__T('M')
294 && Dirs[Pos][Dirs[Pos].size()-4]==__T('.'))
295 {
296 MXF_File+=Dirs[Pos].substr(Dirs[Pos].size()-4, 4);
297 Exists=true;
298 break;
299 }
300 if (!Exists)
301 MXF_File+=__T(".MXF");
302
303 sequence* Sequence=new sequence;
304 Sequence->AddFileName(MXF_File);
305 Sequence->StreamKind=Stream_Audio;
306 Sequence->StreamID=ReferenceFiles->Sequences_Size()+1;
307 ReferenceFiles->AddSequence(Sequence);
308
309 Audio_Count++;
310 }
311 #endif //defined(MEDIAINFO_MXF_YES)
312 }
313
314 Track=Track->NextSiblingElement();
315 }
316 }
317
318 //ClipMetadata
319 XMLElement* ClipMetadata=ClipContent->FirstChildElement("ClipMetadata");
320 if (ClipMetadata)
321 {
322 XMLElement* Access=ClipMetadata->FirstChildElement("Access");
323 if (Access)
324 {
325 //CreationDate
326 ChildElement=Access->FirstChildElement("CreationDate");
327 if (ChildElement)
328 {
329 Ztring Content=ChildElement->GetText();
330 if (Content.size()>=11 && Content[10]==__T('T'))
331 Content[10]=__T(' ');
332 if (Content.find(__T("+00:00"))!=string::npos)
333 {
334 Content.resize(10+1+8);
335 Content.insert(0, __T("UTC "));
336 }
337 Fill(Stream_General, 0, General_Recorded_Date, Content);
338 }
339
340 //CreationDate
341 ChildElement=Access->FirstChildElement("LastUpdateDate");
342 if (ChildElement)
343 {
344 Ztring Content=ChildElement->GetText();
345 if (Content.size()>=11 && Content[10]==__T('T'))
346 Content[10]=__T(' ');
347 if (Content.find(__T("+00:00"))!=string::npos)
348 {
349 Content.resize(10+1+8);
350 Content.insert(0, __T("UTC "));
351 }
352 Fill(Stream_General, 0, General_Tagged_Date, Content);
353 }
354 }
355
356 XMLElement* Device=ClipMetadata->FirstChildElement("Device");
357 if (Device)
358 {
359 //Manufacturer+ModelName
360 XMLElement* Manufacturer=Device->FirstChildElement("Manufacturer");
361 XMLElement* ModelName=Device->FirstChildElement("ModelName");
362 if (Manufacturer && ModelName)
363 Fill(Stream_General, 0, General_Encoded_Application, string(Manufacturer->GetText())+" "+ModelName->GetText());
364 }
365
366 XMLElement* Shoot=ClipMetadata->FirstChildElement("Shoot");
367 if (Shoot)
368 {
369 //StartDate
370 ChildElement=Shoot->FirstChildElement("StartDate");
371 if (ChildElement)
372 {
373 Ztring Content=ChildElement->GetText();
374 if (Content.size()>=11 && Content[10]==__T('T'))
375 Content[10]=__T(' ');
376 if (Content.find(__T("+00:00"))!=string::npos)
377 {
378 Content.resize(10+1+8);
379 Content.insert(0, __T("UTC "));
380 }
381 Fill(Stream_General, 0, General_Duration_Start, Content);
382 }
383
384 //EndDate
385 ChildElement=Shoot->FirstChildElement("EndDate");
386 if (ChildElement)
387 {
388 Ztring Content=ChildElement->GetText();
389 if (Content.size()>=11 && Content[10]==__T('T'))
390 Content[10]=__T(' ');
391 if (Content.find(__T("+00:00"))!=string::npos)
392 {
393 Content.resize(10+1+8);
394 Content.insert(0, __T("UTC "));
395 }
396 Fill(Stream_General, 0, General_Duration_End, Content);
397 }
398
399 //Location
400 XMLElement* Location=Shoot->FirstChildElement("Location");
401 if (Location)
402 {
403 //Longitude+Latitude
404 XMLElement* Longitude=Location->FirstChildElement("Longitude");
405 XMLElement* Latitude=Location->FirstChildElement("Latitude");
406 if (Longitude && Latitude)
407 Fill(Stream_General, 0, General_Recorded_Location, string(Latitude->GetText())+", "+Longitude->GetText());
408 }
409 }
410
411 XMLElement* Scenario=ClipMetadata->FirstChildElement("Scenario");
412 if (Scenario)
413 {
414 //ProgramName
415 ChildElement=Scenario->FirstChildElement("ProgramName");
416 if (ChildElement)
417 Fill(Stream_General, 0, General_Title, ChildElement->GetText());
418
419 //SceneNo.
420 ChildElement=Scenario->FirstChildElement("SceneNo.");
421 if (ChildElement)
422 Fill(Stream_General, 0, "Scene Number", ChildElement->GetText());
423
424 //TakeNo.
425 ChildElement=Scenario->FirstChildElement("TakeNo.");
426 if (ChildElement)
427 Fill(Stream_General, 0, "Take Number", ChildElement->GetText());
428 }
429
430 XMLElement* News=ClipMetadata->FirstChildElement("News");
431 if (News)
432 {
433 //Reporter
434 ChildElement=News->FirstChildElement("Reporter");
435 if (ChildElement)
436 Fill(Stream_General, 0, "Reporter", ChildElement->GetText());
437
438 //Purpose
439 ChildElement=News->FirstChildElement("Purpose");
440 if (ChildElement)
441 Fill(Stream_General, 0, "Purpose", ChildElement->GetText());
442
443 //Object
444 ChildElement=News->FirstChildElement("Object");
445 if (ChildElement)
446 Fill(Stream_General, 0, "Object", ChildElement->GetText());
447 }
448 }
449 }
450 }
451 else
452 {
453 Reject("P2_Clip");
454 return false;
455 }
456 }
457
458 Element_Offset=File_Size;
459
460 //All should be OK...
461 return true;
462}
