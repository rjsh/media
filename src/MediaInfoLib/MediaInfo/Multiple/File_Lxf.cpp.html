217//---------------------------------------------------------------------------
218void File_Lxf::Streams_Fill_PerStream(File__Analyze* Parser, stream_t Container_StreamKind, size_t Parser_Pos, int8u Format)
219{
220 if (Parser==NULL)
221 return;
222
223 Fill(Parser);
224 if (Parser->Count_Get(Stream_Audio) && Config->File_Audio_MergeMonoStreams_Get() && Parser->Retrieve(Stream_Audio, 0, Audio_Format)==__T("PCM"))
225 {
226 if (Count_Get(Stream_Audio)==0)
227 {
228 Merge(*Parser);
229 Fill(Stream_Audio, 0, Audio_Channel_s_, Audio_Sizes.size(), 10, true);
230 int64u BitRate=Retrieve(Stream_Audio, 0, Audio_BitRate).To_int64u();
231 Fill(Stream_Audio, 0, Audio_BitRate, BitRate*Audio_Sizes.size(), 10, true);
232 #if MEDIAINFO_DEMUX
233 if (Config->Demux_ForceIds_Get())
234 {
235 for (size_t Audio_Pos=0; Audio_Pos<Audio_Sizes.size(); Audio_Pos++)
236 Fill(StreamKind_Last, StreamPos_Last, General_ID, 0x200+Audio_Pos);
237 }
238 #endif //MEDIAINFO_DEMUX
239 }
240 }
241 else
242 {
243 Merge(*Parser);
244
245 Ztring LawRating=Parser->Retrieve(Stream_General, 0, General_LawRating);
246 if (!LawRating.empty())
247 Fill(Stream_General, 0, General_LawRating, LawRating, true);
248 Ztring Title=Parser->Retrieve(Stream_General, 0, General_Title);
249 if (!Title.empty() && Retrieve(Stream_General, 0, General_Title).empty())
250 Fill(Stream_General, 0, General_Title, Title);
251
252 #if MEDIAINFO_DEMUX
253 if (Config->Demux_ForceIds_Get())
254 for (size_t StreamKind=Stream_General+1; StreamKind<Stream_Max; StreamKind++)
255 for (size_t StreamPos=0; StreamPos<Parser->Count_Get((stream_t)StreamKind); StreamPos++)
256 {
257 Ztring ID;
258 if (Parser->Count_Get(Stream_Audio) && Parser->Retrieve(Stream_Audio, 0, Audio_MuxingMode)==__T("SMPTE ST 337") && Parser_Pos%2)
259 ID+=Ztring::ToZtring(0x100*Container_StreamKind+Parser_Pos-1)+__T(" / ");
260 ID+=Ztring::ToZtring(0x100*Container_StreamKind+Parser_Pos);
261 if (!Parser->Retrieve((stream_t)StreamKind, StreamPos, General_ID).empty())
262 ID+=__T('-')+Parser->Retrieve((stream_t)StreamKind, StreamPos, General_ID);
263 Fill((stream_t)StreamKind, Count_Get((stream_t)StreamKind)-Parser->Count_Get((stream_t)StreamKind)+StreamPos, General_ID, ID, true);
264 }
265 #endif //MEDIAINFO_DEMUX
266 }
267 if (Format!=(int8u)-1)
268 Fill(Container_StreamKind, Container_StreamKind==Stream_Video?0:Parser_Pos, Fill_Parameter(Container_StreamKind, Generic_CodecID), Format);
269 if (Container_StreamKind==Stream_Video)
270 for (size_t Pos=Count_Get(Stream_Audio)-Parser->Count_Get(Stream_Audio); Pos<Count_Get(Stream_Audio); Pos++)
271 Fill(Stream_Audio, Pos, Audio_MuxingMode, Parser->Retrieve(Stream_General, 0, General_Format));
272}

274//---------------------------------------------------------------------------
275void File_Lxf::Streams_Finish()
276{
277 if (Videos.size()>1 && Videos[1].Parsers.size()==1) //TODO: better handling of fill/finish for Ancillary data
278 {
279 Finish(Videos[1].Parsers[0]);
280 Streams_Fill_PerStream(Videos[1].Parsers[0], Stream_Video, 1);
281 }
282 if (Videos.size()>2 && Videos[2].Parsers.size()==1)
283 {
284 Finish(Videos[2].Parsers[0]);
285 Merge(*Videos[2].Parsers[0], Stream_Video, 0, 0);
286
287 Ztring LawRating=Videos[2].Parsers[0]->Retrieve(Stream_General, 0, General_LawRating);
288 if (!LawRating.empty())
289 Fill(Stream_General, 0, General_LawRating, LawRating, true);
290 Ztring Title=Videos[2].Parsers[0]->Retrieve(Stream_General, 0, General_Title);
291 if (!Title.empty() && Retrieve(Stream_General, 0, General_Title).empty())
292 Fill(Stream_General, 0, General_Title, Title);
293 }
294
295 if (Audios_Header.TimeStamp_End!=(int64u)-1 && Audios_Header.TimeStamp_Begin!=(int64u)-1 && Audios_Header.Duration_First!=(int64u)-1)
296 {
297 int64u Duration=float64_int64s(((float64)(Audios_Header.TimeStamp_End-Audios_Header.TimeStamp_Begin))/TimeStamp_Rate*1000);
298 int64u FrameCount=float64_int64s(((float64)(Audios_Header.TimeStamp_End-Audios_Header.TimeStamp_Begin))/Audios_Header.Duration_First);
299 for (size_t Pos=0; Pos<Count_Get(Stream_Audio); Pos++)
300 {
301 if (Retrieve(Stream_Audio, Pos, Audio_Duration).empty())
302 Fill(Stream_Audio, Pos, Audio_Duration, Duration);
303 if (Retrieve(Stream_Audio, Pos, Audio_FrameCount).empty())
304 Fill(Stream_Audio, Pos, Audio_FrameCount, FrameCount);
305 }
306 Info_General_StreamSize+=FrameCount*0x48;
307 }
308 if (Videos_Header.TimeStamp_End!=(int64u)-1 && Videos_Header.TimeStamp_Begin!=(int64u)-1)
309 {
310 int64u Duration=float64_int64s(((float64)(Videos_Header.TimeStamp_End-Videos_Header.TimeStamp_Begin))/TimeStamp_Rate*1000);
311 int64u FrameCount=float64_int64s(((float64)(Videos_Header.TimeStamp_End-Videos_Header.TimeStamp_Begin))/Videos_Header.Duration);
312 if (Duration)
313 for (size_t Pos=0; Pos<Count_Get(Stream_Video); Pos++)
314 Fill(Stream_Video, Pos, Video_Duration, Duration, 10, true);
315 Info_General_StreamSize+=FrameCount*0x48;
316
317 if (Count_Get(Stream_Video)==1 && Retrieve(Stream_Video, 0, Video_BitRate).empty())
318 {
319 for (size_t Pos=0; Pos<Videos.size(); Pos++)
320 if (Videos[Pos].BytesPerFrame!=(int64u)-1)
321 Info_General_StreamSize+=Videos[Pos].BytesPerFrame*FrameCount;
322 for (size_t Pos=0; Pos<Audios.size(); Pos++)
323 if (Audios[Pos].BytesPerFrame!=(int64u)-1)
324 Info_General_StreamSize+=Audios[Pos].BytesPerFrame*Retrieve(Stream_Audio, Pos, Audio_FrameCount).To_int64u();
325 Fill(Stream_General, 0, General_StreamSize, Info_General_StreamSize);
326 if (Info_General_StreamSize<File_Size)
327 Fill(Stream_Video, 0, Video_StreamSize, File_Size-Info_General_StreamSize);
328 }
329 }

336//---------------------------------------------------------------------------
337bool File_Lxf::Synchronize()
338{
339 //Synchronizing
340 while (Buffer_Offset+20<=Buffer_Size && ( Buffer[Buffer_Offset ]!=0x4C
341 || Buffer[Buffer_Offset+ 1]!=0x45
342 || Buffer[Buffer_Offset+ 2]!=0x49
343 || Buffer[Buffer_Offset+ 3]!=0x54
344 || Buffer[Buffer_Offset+ 4]!=0x43
345 || Buffer[Buffer_Offset+ 5]!=0x48
346 || Buffer[Buffer_Offset+ 6]!=0x00
347 || Buffer[Buffer_Offset+ 7]!=0x00
348 ))
349 {
350 Buffer_Offset+=6+2;
351 while (Buffer_Offset<Buffer_Size && Buffer[Buffer_Offset]!=0x00)
352 Buffer_Offset+=2;
353 if (Buffer_Offset>=Buffer_Size || Buffer[Buffer_Offset-1]==0x00)
354 Buffer_Offset--;
355 Buffer_Offset-=6;
356 }
357
358 //Parsing last bytes if needed
359 if (Buffer_Offset+20>Buffer_Size)
360 {
361 while (Buffer_Offset+8>Buffer_Size)
362 if (Buffer_Offset+8==Buffer_Size && CC8(Buffer+Buffer_Offset)!=0x4C45495443480000LL)
363 Buffer_Offset++;
364 else
365 break;
366 if (Buffer_Offset+7==Buffer_Size && CC7(Buffer+Buffer_Offset)!=0x4C454954434800LL)
367 Buffer_Offset++;
368 if (Buffer_Offset+6==Buffer_Size && CC6(Buffer+Buffer_Offset)!=0x4C4549544348LL)
369 Buffer_Offset++;
370 if (Buffer_Offset+5==Buffer_Size && CC5(Buffer+Buffer_Offset)!=0x4C45495443LL)
371 Buffer_Offset++;
372 if (Buffer_Offset+4==Buffer_Size && CC4(Buffer+Buffer_Offset)!=0x4C454954)
373 Buffer_Offset++;
374 if (Buffer_Offset+3==Buffer_Size && CC3(Buffer+Buffer_Offset)!=0x4C4549)
375 Buffer_Offset++;
376 if (Buffer_Offset+2==Buffer_Size && CC2(Buffer+Buffer_Offset)!=0x4C45)
377 Buffer_Offset++;
378 if (Buffer_Offset+1==Buffer_Size && CC1(Buffer+Buffer_Offset)!=0x4C)
379 Buffer_Offset++;
380 return false;
381 }
382
383 if (!Status[IsAccepted])
384 {
385 Accept();
386
387 Fill(Stream_General, 0, General_Format, "LXF");
388
389 File_Buffer_Size_Hint_Pointer=Config->File_Buffer_Size_Hint_Pointer_Get();
390 }
391
392 #if MEDIAINFO_SEEK
393 //TimeStamp
394 if (SeekRequest!=(int64u)-1)
395 {
396 if (TimeOffsets.find(File_Offset+Buffer_Offset)==TimeOffsets.end()) //Not already saved
397 {
398 if (Buffer_Offset+0x48>=Buffer_Size)
399 return false;
400 int32u Type =LittleEndian2int32u(Buffer+Buffer_Offset+16);
401 if (Type==0) //Video
402 {
403 //Filling with the new frame
404 Version=LittleEndian2int32u(Buffer+Buffer_Offset+8);
405 int64u TimeStamp, Duration;
406 switch (Version)
407 {
408 case 0 : TimeStamp =LittleEndian2int32u(Buffer+Buffer_Offset+24);
409 Duration =LittleEndian2int32u(Buffer+Buffer_Offset+28);
410 break;
411 case 1 : TimeStamp =LittleEndian2int64u(Buffer+Buffer_Offset+24);
412 Duration =LittleEndian2int64u(Buffer+Buffer_Offset+32);
413 break;
414 default: TimeStamp=Duration=0;
415 }
416 int8u PictureType=(LittleEndian2int8u (Buffer+Buffer_Offset+42)&0xC0)>>6;
417 TimeOffsets[File_Offset+Buffer_Offset]=stream_header(TimeStamp, TimeStamp+Duration, Duration, PictureType);
418 SeekRequest_Divider=2;
419 }
420 }
421 if (Read_Buffer_Seek(2, (int64u)-1, (int64u)-1))
422 return false;
423 }
424 #endif //MEDIAINFO_SEEK
425
426 //Synched is OK
427 return true;
428}

488#if MEDIAINFO_SEEK
489size_t File_Lxf::Read_Buffer_Seek (size_t Method, int64u Value, int64u)
490{
491 //Init
492 if (!Duration_Detected)
493 {
494 MediaInfo_Internal MI;
495 MI.Option(__T("File_KeepInfo"), __T("1"));
496 Ztring ParseSpeed_Save=MI.Option(__T("ParseSpeed_Get"), __T(""));
497 Ztring Demux_Save=MI.Option(__T("Demux_Get"), __T(""));
498 MI.Option(__T("ParseSpeed"), __T("0"));
499 MI.Option(__T("Demux"), Ztring());
500 size_t MiOpenResult=MI.Open(File_Name);
501 MI.Option(__T("ParseSpeed"), ParseSpeed_Save); //This is a global value, need to reset it. TODO: local value
502 MI.Option(__T("Demux"), Demux_Save); //This is a global value, need to reset it. TODO: local value
503 if (!MiOpenResult || MI.Get(Stream_General, 0, General_Format)!=__T("LXF"))
504 return 0;
505 for (time_offsets::iterator TimeOffset=((File_Lxf*)MI.Info)->TimeOffsets.begin(); TimeOffset!=((File_Lxf*)MI.Info)->TimeOffsets.end(); ++TimeOffset)
506 TimeOffsets[TimeOffset->first]=TimeOffset->second;
507 int64u Duration=float64_int64s(Ztring(MI.Get(Stream_General, 0, __T("Duration"))).To_float64()*TimeStamp_Rate/1000);
508 TimeOffsets[File_Size]=stream_header(Duration, Duration, 0, (int8u)-1);
509 SeekRequest_Divider=2;
510 Duration_Detected=true;
511 }
512
513 //Parsing
514 switch (Method)
515 {
516 case 0 : Open_Buffer_Unsynch(); GoTo(Value); return 1;
517 case 1 : Open_Buffer_Unsynch(); GoTo(File_Size*Value/10000); return 1;
518 case 3 : //Frame
519 {
520 if (FrameRate==0 && Videos_Header.TimeStamp_End!=(int64u)-1 && Videos_Header.TimeStamp_End-Videos_Header.TimeStamp_Begin!=0)
521 FrameRate=TimeStamp_Rate/(Videos_Header.TimeStamp_End-Videos_Header.TimeStamp_Begin);
522 if (FrameRate==0)
523 return (size_t)-1; //Not supported
524 float64 TimeStamp=((float64)Value)/FrameRate;
525 Value=float64_int64s(TimeStamp*1000000000); // In nanoseconds
526 }
527 case 2 : //Timestamp
528 {
529 if (Value!=(int64u)-1)
530 {
531 Value=float64_int64s((float64)Value*TimeStamp_Rate/1000000000); //Convert in LXF unit
532 time_offsets::iterator End=TimeOffsets.end();
533 --End;
534 if (Value>=End->second.TimeStamp_End)
535 return 2; //Higher than total size
536 SeekRequest=Value;
537 }
538
539 //Looking if we already have the timestamp
540 int64u SeekRequest_Mini=SeekRequest; if (SeekRequest_Mini>1000000) SeekRequest_Mini-=float64_int64s(TimeStamp_Rate/1000); //-1ms
541 int64u SeekRequest_Maxi=SeekRequest+float64_int64s(TimeStamp_Rate/1000); //+1ms
542 for (time_offsets::iterator TimeOffset=TimeOffsets.begin(); TimeOffset!=TimeOffsets.end(); ++TimeOffset)
543 {
544 if (TimeOffset->second.TimeStamp_Begin<=SeekRequest_Maxi && TimeOffset->second.TimeStamp_End>=SeekRequest_Mini) //If it is found in a frame we know
545 {
546 //Looking for the corresponding I-Frame
547 while (TimeOffset->second.PictureType&0x2 && TimeOffset!=TimeOffsets.begin()) //Not an I-Frame (and not fisrt frame)
548 {
549 time_offsets::iterator Previous=TimeOffset;
550 --Previous;
551 if (Previous->second.TimeStamp_End!=TimeOffset->second.TimeStamp_Begin) //Testing if the previous frame is not known.
552 {
553 SeekRequest=TimeOffset->second.TimeStamp_Begin-(float64_int64s(TimeStamp_Rate/1000)+1); //1ms+1, so we are sure to not synch on the current frame again
554 Open_Buffer_Unsynch();
555 GoTo((Previous->first+TimeOffset->first)/2);
556 return 1; //Looking for previous frame
557
558 }
559 TimeOffset=Previous;
560 }
561
562 //We got the right I-Frame
563 if (Value==0)
564 {
565 for (size_t Pos=0; Pos<Videos.size(); Pos++)
566 for (size_t Pos2=0; Pos2<Videos[Pos].Parsers.size(); Pos2++)
567 Videos[Pos].Parsers[Pos2]->Unsynch_Frame_Count=0;
568 }
569 Open_Buffer_Unsynch();
570 GoTo(TimeOffset->first);
571 SeekRequest=(int64u)-1;
572 return 1;
573 }
574
575 if (TimeOffset->second.TimeStamp_Begin>SeekRequest_Maxi) //Testing if too far
576 {
577 time_offsets::iterator Previous=TimeOffset; --Previous;
578 int64u ReferenceOffset;
579 if (File_Offset+Buffer_Offset==TimeOffset->first && TimeOffset->second.TimeStamp_Begin>SeekRequest) //If current frame is already too far
580 ReferenceOffset=File_Offset+Buffer_Offset;
581 else
582 ReferenceOffset=TimeOffset->first;
583 if (SeekRequest_Divider==0)
584 {
585 SeekRequest=Previous->second.TimeStamp_Begin-(float64_int64s(TimeStamp_Rate/1000)+1); //1ms+1, so we are sure to not synch on the current frame again
586 ReferenceOffset=Previous->first;
587 --Previous;
588 SeekRequest_Divider=2;
589 }
590 Open_Buffer_Unsynch();
591 GoTo(Previous->first+(ReferenceOffset-Previous->first)/SeekRequest_Divider);
592 SeekRequest_Divider*=2;
593 return 1;
594 }
595 }
596 }
597 return 0;
598 default : return (size_t)-1;
599 }
600}

632 {

635 Header_Fill_Size(Audio_Sizes[Audio_Sizes_Pos]);

640 //Parsing
641 int64u BlockSize=0, TimeStamp=0, Duration=0;
642 int32u HeaderSize, Type;
643 int32u Size; //Video and Audio
644 int8u VideoFormat, GOP_M, PictureType; //Video
645 int8u Channels_Count=0; //Audio
646 Skip_C8( "Signature");
647 Get_L4 (Version, "Version"); //0=start and duration are in field, 1=in 27 MHz values
648 Get_L4 (HeaderSize, "Header size");
649 if (Element_Size<HeaderSize)
650 {
651 Element_WaitForMoreData();
652 return;
653 }
654 if (Version>1)
655 {
656 //Filling
657 Header_Fill_Code(0, "Unknown");
658 Header_Fill_Size(HeaderSize);
659 Synched=false;
660 return;
661 }
662 Get_L4 (Type, "Type");
663 Skip_L4( "Stream ID");
664 switch (Version)
665 {
666 case 0 :
667 {
668 int32u TimeStamp4, Duration4;
669 Get_L4 (TimeStamp4, "TimeStamp");
670 TimeStamp=TimeStamp4;
671 Param_Info3(((float64)TimeStamp4)/TimeStamp_Rate, " s", 3);
672 FrameInfo.DTS=FrameInfo.PTS=float64_int64s(((float64)TimeStamp)*1000000000/TimeStamp_Rate);
673 Get_L4 (Duration4, "Duration");
674 Duration=Duration4;
675 Param_Info3(((float64)Duration)/TimeStamp_Rate, " s", 3);
676 FrameInfo.DUR=float64_int64s(((float64)Duration)*1000000000/TimeStamp_Rate);
677 }
678 break;
679 case 1 :
680 Get_L8 (TimeStamp, "TimeStamp"); Param_Info3(((float64)TimeStamp)/720000, " s", 3); FrameInfo.DTS=FrameInfo.PTS=float64_int64s(((float64)TimeStamp)*1000000/720);
681 Get_L8 (Duration, "Duration"); Param_Info3(((float64)Duration)/720000, " s", 3); FrameInfo.DUR=float64_int64s(((float64)Duration)*1000000/720);
682 break;
683 default: ;
684 }
685 switch(Type)
686 {
687 case 0 : //Video
688 {
689 Video_Sizes.resize(3);
690 BS_Begin_LE();
691 Get_T1 (4, VideoFormat, "Format"); Param_Info1(Lxf_Format_Video[VideoFormat]);
692 Skip_T1(7, "GOP (N)");
693 Get_T1 (3, GOP_M, "GOP (M)");
694 Info_T1(8, BitRate, "Bit rate"); Param_Info2((BitRate*(BitRate>60?10:(BitRate>50?5:1))-(BitRate>60?500:(BitRate>50?200:0)))*1000000, " bps");
695 Get_T1 (2, PictureType, "Picture type"); Param_Info1(Lxf_PictureType[PictureType]);
696 BS_End_LE();
697 Skip_L1( "Reserved");
698 Get_L4(Size, "Video data size");
699 Skip_L4( "Zero");
700 if (!Video_Sizes.empty())
701 Video_Sizes[2]=Size;
702 BlockSize+=Size;
703 Get_L4(Size, "VBI data size");
704 if (!Video_Sizes.empty())
705 Video_Sizes[1]=Size;
706 BlockSize+=Size;
707 Skip_L4( "Zero");
708 Get_L4(Size, "Meta data size");
709 if (!Video_Sizes.empty())
710 Video_Sizes[0]=Size;
711 BlockSize+=Size;
712 }
713 break;
714 case 1 : //Audio
715 {
716 if (Version==0)
717 {
718 Skip_L4( "First Active Field");
719 Skip_L4( "Total fields in packet");
720 }
721 BS_Begin_LE();
722 Get_T1 ( 6, SampleSize, "Sample size");
723 Skip_T1( 6, "Sample precision");
724 Skip_T1(20, "Reserved");
725 BS_End_LE();
726 Element_Begin1("Tracks mask");
727 BS_Begin_LE();
728 for (size_t Pos=0; Pos<32; Pos++)
729 {
730 bool Channel;
731 Get_TB(Channel, "Channel");
732 if (Channel)
733 Channels_Count++;
734 }
735 BS_End_LE();
736 Element_End0();
737 Get_L4(Size, "Track size");
738 Skip_L4( "Zero");
739 if (Version>=1)
740 {
741 Skip_L4( "Zero");
742 Skip_L4( "Zero");
743 }
744 Audio_Sizes.resize(Channels_Count);
745 for (size_t Pos=0; Pos<Audio_Sizes.size(); Pos++)
746 Audio_Sizes[Pos]=Size;
747 BlockSize=Size*Channels_Count;
748 }
749 break;
750 case 2 : //Header
751 {
752 Header_Sizes.resize(2);
753 int32u SegmentFormat, Size;
754
755 Get_L4 (SegmentFormat, "Segment format");
756 Get_L4 (Size, "Data size");
757 Header_Sizes[0]=Size;
758 BlockSize+=Size;
759 if (SegmentFormat)
760 {
761 Get_L4 (Size, "Extended fields size");
762 Header_Sizes[1]=Size;
763 BlockSize+=Size;
764 }
765 Skip_L4( "Zero");
766 Skip_L4( "Zero");
767 Skip_L4( "Zero");
768 }
769 break;
770 default : BlockSize=0;
771 }
772 Skip_L4( "Checksum");
773 if (Version)
774 Skip_L4( "Zero");
775
776 if (Element_Offset<HeaderSize)
777 Skip_XX(Header_Size-Element_Offset, "Unknown");
778
779 //Filling
780 Header_Fill_Code(Type, Ztring::ToZtring(Type));
781 Header_Fill_Size(HeaderSize+BlockSize);
782
783 FILLING_BEGIN();
784 if (Buffer_Offset+Element_Offset+BlockSize>Buffer_Size)
785 {
786 //Hints
787 if (File_Buffer_Size_Hint_Pointer)
788 {
789 size_t Buffer_Size_Target=(size_t)(Buffer_Offset+0x48+BlockSize+0x48); //+0x48 for next packet header
790 if ((*File_Buffer_Size_Hint_Pointer)<Buffer_Size_Target)
791 (*File_Buffer_Size_Hint_Pointer)=Buffer_Size_Target;
792 }
793 }
794
795 switch(Type)
796 {
797 case 0 : //Video
798 {
799 if (Videos_Header.TimeStamp_Begin==(int64u)-1)
800 Videos_Header.TimeStamp_Begin=TimeStamp;
801 Videos_Header.TimeStamp_End=TimeStamp+Duration;
802 Videos_Header.Duration=Duration;
803 if (TimeStamp==LastAudio_TimeOffset.TimeStamp_Begin)
804 TimeOffsets[LastAudio_BufferOffset]=stream_header(TimeStamp, TimeStamp+Duration, Duration, PictureType);
805 else
806 TimeOffsets[File_Offset+Buffer_Offset]=stream_header(TimeStamp, TimeStamp+Duration, Duration, PictureType);
807 int64u PTS_Computing=TimeStamp;
808 #if MEDIAINFO_DEMUX
809 switch (PictureType)
810 {
811 case 2 :
812 case 3 : Demux_random_access=false; break; //P-Frame, B-Frame
813 default: Demux_random_access=true ; //I-Frame
814 }
815 #endif //MEDIAINFO_DEMUX
816 if (2>Videos.size())
817 Videos.resize(2+1);
818 if (!Video_Sizes.empty())
819 Videos[2].Format=VideoFormat;
820 if (GOP_M>1) //With B-frames
821 {
822 switch (PictureType)
823 {
824 case 2 : PTS_Computing+=GOP_M*Duration; break; //P-Frame
825 case 3 : break; //B-Frame
826 default: PTS_Computing+=Duration; //I-Frame
827 }
828 }
829 FrameInfo.PTS=float64_int64s(((float64)PTS_Computing)*1000000000/TimeStamp_Rate);
830 }
831 break;
832 case 1 : //Audio
833 {
834 if (Audios_Header.TimeStamp_Begin==(int64u)-1)
835 Audios_Header.TimeStamp_Begin=TimeStamp;
836 Audios_Header.TimeStamp_End=TimeStamp+Duration;
837 Audios_Header.Duration=Duration;
838 if (Audios_Header.Duration_First==(int64u)-1 && Duration)
839 Audios_Header.Duration_First=Duration;
840 LastAudio_BufferOffset=File_Offset+Buffer_Offset;
841 LastAudio_TimeOffset=stream_header(TimeStamp, TimeStamp+Duration, Duration, (int8u)-1);
842 #if MEDIAINFO_DEMUX
843 Demux_random_access=true;
844 #endif //MEDIAINFO_DEMUX
845 }
846 break;
847 default : BlockSize=0;
848 }
849 FILLING_END();
850}

854{

857 case 0 : Video(); break;
858 case 1 : Audio(); break;
859 case 2 : Header(); break;

865 else
866 Skip_XX(Element_Size, "Unknown");

881 GoToFromEnd((File_Offset+Buffer_Offset)*2*6/Frame_Count);
882 Open_Buffer_Unsynch();
883 }
884 }
885 }
886 }
887 FILLING_END();
