70const Ztring &File_Ibi::Get (stream_t /*StreamKind*/, size_t /*StreamNumber*/, const Ztring &Parameter, info_t /*KindOfInfo*/, info_t /*KindOfSearch*/)
71{
72 ibi::streams::iterator IbiStream_Temp=Ibi->Streams.begin(); //TODO: management of multiple streams
73 if (IbiStream_Temp!=Ibi->Streams.end() && !IbiStream_Temp->second->Infos.empty())
74 {
75 int64u FrameNumber=Parameter.To_int64u();
76
77 for (size_t Pos=0; Pos<IbiStream_Temp->second->Infos.size()-1; Pos++)
78 if (IbiStream_Temp->second->Infos[Pos].FrameNumber==FrameNumber || IbiStream_Temp->second->Infos[Pos+1].FrameNumber>FrameNumber)
79 {
80 Get_Temp=__T("StreamOffset=")+Ztring::ToZtring(IbiStream_Temp->second->Infos[Pos].StreamOffset)
81 + __T(" / FrameNumber=")+Ztring::ToZtring(IbiStream_Temp->second->Infos[Pos].FrameNumber)
82 + __T(" / Dts=")+Ztring::ToZtring(IbiStream_Temp->second->Infos[Pos].Dts);
83 return Get_Temp;
84 }
85 }
86
87 //Can not be found
88 Get_Temp.clear();
89 return Get_Temp;
90}
91#endif //MEDIAINFO_IBIUSAGE

130//---------------------------------------------------------------------------
131void File_Ibi::Header_Parse()
132{
133 //Test of zero padding
134 int8u Null;
135 Peek_B1(Null);
136 if (Null==0x00)
137 {
138 if (Buffer_Offset_Temp==0)
139 Buffer_Offset_Temp=Buffer_Offset+1;
140
141 while (Buffer_Offset_Temp<Buffer_Size)
142 {
143 if (Buffer[Buffer_Offset_Temp])
144 break;
145 Buffer_Offset_Temp++;
146 }
147 if (Buffer_Offset_Temp>=Buffer_Size)
148 {
149 Element_WaitForMoreData();
150 return;
151 }
152
153 Header_Fill_Code((int32u)-1); //Should be (int64u)-1 but Borland C++ does not like this
154 Header_Fill_Size(Buffer_Offset_Temp-Buffer_Offset);
155 Buffer_Offset_Temp=0;
156
157 return;
158 }
159
160 //Parsing
161 int64u Name, Size;
162 Get_EB (Name, "Name");
163 Get_EB (Size, "Size");
164
165 //Filling
166 Header_Fill_Code(Name, Ztring().From_Number(Name, 16));
167 Header_Fill_Size(Element_Offset+Size);
168}

205//---------------------------------------------------------------------------
206void File_Ibi::Data_Parse()
207{
208 #define LIS2(_ATOM, _NAME) \
209 case Elements::_ATOM : \
210 if (Level==Element_Level) \
211 { \
212 Element_Name(_NAME); \
213 _ATOM(); \
214 Element_ThisIsAList(); \
215 } \
216
217 #define ATO2(_ATOM, _NAME) \
218 case Elements::_ATOM : \
219 if (Level==Element_Level) \
220 { \
221 if (Element_IsComplete_Get()) \
222 { \
223 Element_Name(_NAME); \
224 _ATOM(); \
225 } \
226 else \
227 { \
228 Element_WaitForMoreData(); \
229 return; \
230 } \
231 } \
232 break; \
233
234 #define ATOM_END_MK \
235 ATOM(Zero) \
236 ATOM(CRC32) \
237 ATOM(Void) \
238 ATOM_END
239
240 //Parsing
241 DATA_BEGIN
242 LIST(Ebml)
243 ATOM_BEGIN
244 ATOM(Ebml_Version)
245 ATOM(Ebml_ReadVersion)
246 ATOM(Ebml_MaxIDLength)
247 ATOM(Ebml_MaxSizeLength)
248 ATOM(Ebml_DocType)
249 ATOM(Ebml_DocTypeVersion)
250 ATOM(Ebml_DocTypeReadVersion)
251 ATOM_END_MK
252 LIST(Stream)
253 ATOM_BEGIN
254 ATOM(Stream_Header)
255 ATOM(Stream_ByteOffset)
256 ATOM(Stream_FrameNumber)
257 ATOM(Stream_Dts)
258 ATOM_END_MK
259 ATOM(CompressedIndex)
260 LIST(WritingApplication)
261 ATOM_BEGIN
262 ATOM(WritingApplication_Name)
263 ATOM(WritingApplication_Version)
264 ATOM_END_MK
265 LIST(SourceInfo)
266 ATOM_BEGIN
267 ATOM(SourceInfo_IndexCreationDate)
268 ATOM(SourceInfo_SourceModificationDate)
269 ATOM(SourceInfo_SourceSize)
270 ATOM_END_MK
271 ATOM(InformData)
272 DATA_END_DEFAULT
273}

556void File_Ibi::CompressedIndex()
557{
558 if (!Status[IsAccepted])
559 {
560 Reject("Ibi");
561 return;
562 }
563
564 Element_Name("Compressed Index");
565 int64u UncompressedSize;
566 Get_EB (UncompressedSize, "Uncompressed size");
567
568 //Sizes
569 unsigned long Source_Size=(unsigned long)(Element_Size-Element_Offset);
570 unsigned long Dest_Size=(unsigned long)UncompressedSize;
571
572 //Uncompressing
573 int8u* Dest;
574 try
575 {
576 Dest=new int8u[Dest_Size];
577 }
578 catch (...)
579 {
580 //Memory error
581 Reject();
582 return;
583 }
584 if (uncompress((Bytef*)Dest, &Dest_Size, (const Bytef*)Buffer+Buffer_Offset+(size_t)Element_Offset, Source_Size)<0)
585 {
586 Skip_XX(Element_Size-Element_Offset, "Problem during the decompression");
587 delete[] Dest; //Dest=NULL;
588 return;
589 }
590
591 //Exiting this element
592 Skip_XX(Element_Size-Element_Offset, "Will be parsed");
593
594 //Configuring buffer
595 const int8u* Buffer_Sav=Buffer;
596 size_t Buffer_Size_Sav=Buffer_Size;
597 int8u* Buffer_Temp_Sav=Buffer_Temp;
598 size_t Buffer_Temp_Size_Sav=Buffer_Temp_Size;
599 size_t Buffer_Offset_Sav=Buffer_Offset;
600 size_t Buffer_Offset_Temp_Sav=Buffer_Offset_Temp;
601 Buffer=NULL;
602 Buffer_Size=0;
603 Buffer_Temp=NULL;
604 Buffer_Temp_Size=0;
605 Buffer_Offset=0;
606 Buffer_Offset_Temp=0;
607
608 //Configuring level
609 std::vector<int64u> Element_Sizes_Sav;
610 size_t Element_Level_Sav=Element_Level;
611 while(Element_Level)
612 {
613 Element_Sizes_Sav.push_back(Element_TotalSize_Get());
614 Element_End0();
615 }
616
617 //Configuring file size
618 int64u File_Size_Sav=File_Size;
619 if (File_Size<File_Offset+Buffer_Offset+Element_Offset+Dest_Size)
620 File_Size=File_Offset+Buffer_Offset+Element_Offset+Dest_Size;
621 Element_Level++;
622 Header_Fill_Size(File_Size);
623 Element_Level--;
624
625 //Parsing
626 Buffer=Dest;
627 Buffer_Size=Dest_Size;
628 while (Open_Buffer_Continue_Loop());
629 delete[] Dest; //Dest=NULL;
630
631 //Resetting file size
632 File_Size=File_Size_Sav;
633 while(Element_Level)
634 Element_End0();
635 Element_Level++;
636 Header_Fill_Size(File_Size);
637 Element_Level--;
638
639 //Configuring level
640 while(Element_Level<Element_Level_Sav)
641 {
642 Element_Begin0();
643 Element_Begin0();
644 Header_Fill_Size(Element_Sizes_Sav[0]);
645 Element_End0();
646 }
647
648 //Resetting buffer
649 Buffer=Buffer_Sav;
650 Buffer_Size=Buffer_Size_Sav;
651 Buffer_Temp=Buffer_Temp_Sav;
652 Buffer_Temp_Size=Buffer_Temp_Size_Sav;
653 Buffer_Offset=Buffer_Offset_Sav;
654 Buffer_Offset_Temp=Buffer_Offset_Temp_Sav;
655}

714//---------------------------------------------------------------------------
715void File_Ibi::InformData()
716{
717 Element_Name("InformData");
718
719 //Parsing
720 Ztring InformData_FromFile;
721 Get_UTF8 (Element_Size, InformData_FromFile, "Data");
722
723 //Filling
724 #if MEDIAINFO_IBIUSAGE
725 if (Config->Ibi_UseIbiInfoIfAvailable_Get())
726 {
727 ZtringListList Fields(InformData_FromFile);
728 for (size_t Pos=0; Pos<Fields.size(); Pos++)
729 {
730 if (Pos==0 || Fields[Pos].size()<2)
731 {
732 if (Pos)
733 Pos++;
734 if (Pos>Fields.size() || Fields[Pos].size()<1)
735 break; //End or problem
736
737 if (Fields[Pos][0]==__T("General"))
738 ; //Nothing to do
739 else if (Fields[Pos][0]==__T("Video"))
740 Stream_Prepare(Stream_Video);
741 else if (Fields[Pos][0]==__T("Audio"))
742 Stream_Prepare(Stream_Audio);
743 else if (Fields[Pos][0]==__T("Text"))
744 Stream_Prepare(Stream_Text);
745 else if (Fields[Pos][0]==__T("Other"))
746 Stream_Prepare(Stream_Other);
747 else if (Fields[Pos][0]==__T("Image"))
748 Stream_Prepare(Stream_Image);
749 else if (Fields[Pos][0]==__T("Menu"))
750 Stream_Prepare(Stream_Menu);
751 else
752 break; //Problem
753 Pos++;
754 }
755
756 Fill(StreamKind_Last, StreamPos_Last, Fields[Pos][0].To_UTF8().c_str(), Fields[Pos][1], true);
757 if (Info_Options<Fields[Pos].size())
758 (*Stream_More)[StreamKind_Last][StreamPos_Last](Fields[Pos][0].To_UTF8().c_str(), Info_Options)=Fields[Pos][Info_Options];
759 }
760 }
761 #endif //MEDIAINFO_IBIUSAGE
762}
