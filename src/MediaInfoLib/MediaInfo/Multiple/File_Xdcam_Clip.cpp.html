42bool File_Xdcam_Clip::FileHeader_Begin()
43{
44 XMLDocument document;
45 if (!FileHeader_Begin_XML(document))
46 return false;
47
48 {
49 XMLElement* Root=document.FirstChildElement("NonRealTimeMeta");
50 if (Root)
51 {
52 Accept("Xdcam_Clip");
53 Fill(Stream_General, 0, General_Format, "XDCAM Clip");
54
55 XMLElement* Element;
56
57 //CreationDate
58 Element=Root->FirstChildElement("CreationDate");
59 if (Element)
60 Fill(Stream_General, 0, General_Recorded_Date, Element->Attribute("value"));
61
62 //LastUpdate
63 Element=Root->FirstChildElement("LastUpdate");
64 if (Element)
65 Fill(Stream_General, 0, General_Tagged_Date, Element->Attribute("value"));
66
67 //Duration
68 Ztring Duration, EditUnit;
69 Element=Root->FirstChildElement("Duration");
70 if (Element)
71 Duration=Element->Attribute("value");
72 Element=Root->FirstChildElement("LtcChangeTable");
73 if (Element)
74 EditUnit=Element->Attribute("tcFps");
75 int64u Duration_Frames=Duration.To_int64u();
76 int64u EditUnit_Denominator=EditUnit.To_int64u();
77 if (Duration_Frames && EditUnit_Denominator)
78 Fill(Stream_General, 0, General_Duration, ((float32)Duration_Frames)*1000/EditUnit_Denominator, 0);
79
80 int64u File_Size_Total=File_Size;
81
82 #if defined(MEDIAINFO_MXF_YES)
83 if (File_Name.size()>12
84 && File_Name[File_Name.size()-7]==__T('M')
85 && File_Name[File_Name.size()-6]==__T('0')
86 && File_Name[File_Name.size()-5]==__T('1')
87 && File_Name[File_Name.size()-4]==__T('.')
88 && File_Name[File_Name.size()-3]==__T('X')
89 && File_Name[File_Name.size()-2]==__T('M')
90 && File_Name[File_Name.size()-1]==__T('L'))
91 {
92 Ztring file=File_Name.substr(File_Name.size()-12, 5);
93 Ztring MXF_File=File_Name;
94 MXF_File.resize(MXF_File.size()-12);
95 MXF_File+=file;
96 if (File::Exists(MXF_File+__T(".MXF")))
97 MXF_File+=__T(".MXF");
98 if (File::Exists(MXF_File+__T(".MP4")))
99 MXF_File+=__T(".MP4");
100
101 //int8u ReadByHuman=Ztring(MediaInfo::Option_Static(__T("ReadByHuman_Get"))).To_int8u();
102 //MediaInfo::Option_Static(__T("ReadByHuman"), __T("0"));
103 MediaInfo_Internal MI;
104 MI.Option(__T("File_IsReferenced"), __T("1"));
105 if (MI.Open(MXF_File))
106 {
107 //MediaInfo::Option_Static(__T("ReadByHuman"), ReadByHuman?__T("1"):__T("0"));
108 Merge(MI);
109 Fill(Stream_Video, StreamPos_Last, "Source", MXF_File);
110 File_Size_Total+=Ztring(MI.Get(Stream_General, 0, General_FileSize)).To_int64u();
111
112 //Commercial names
113 Fill(Stream_General, 0, General_Format_Commercial_IfAny, MI.Get(Stream_General, 0, General_Format_Commercial_IfAny));
114 Ztring Format_Commercial=MI.Get(Stream_General, 0, General_Format_Commercial_IfAny);
115 if (!Format_Commercial.empty())
116 {
117 Format_Commercial.FindAndReplace(__T("XDCAM "), Ztring());
118 Fill(Stream_General, 0, General_Format_Commercial, __T("XDCAM Clip ")+Format_Commercial, true);
119 }
120 }
121 //else
122 // MediaInfo::Option_Static(__T("ReadByHuman"), ReadByHuman?__T("1"):__T("0"));
123 }
124 #endif //defined(MEDIAINFO_MXF_YES)
125
126 //Device
127 Element=Root->FirstChildElement("Device");
128 if (Element)
129 {
130 const char* manufacturer=Element->Attribute("manufacturer");
131 if (manufacturer)
132 {
133 string manufacturer_modelName(manufacturer);
134 const char* modelName=Element->Attribute("modelName");
135 if (modelName)
136 {
137 manufacturer_modelName+=' ';
138 manufacturer_modelName+=modelName;
139 }
140 Fill(Stream_General, 0, General_Encoded_Application, manufacturer_modelName, true, true);
141 }
142 }
143
144 if (File_Size_Total!=File_Size)
145 Fill(Stream_General, 0, General_FileSize, File_Size_Total, 10, true);
146 }
147 else
148 {
149 Reject("Xdcam_Clip");
150 return false;
151 }
152 }
153
154 Element_Offset=File_Size;
155
156 //All should be OK...
157 return true;
158}
159
160} //NameSpace
