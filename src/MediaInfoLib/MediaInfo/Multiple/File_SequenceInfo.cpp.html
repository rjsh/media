89//***************************************************************************
90// Buffer - File header
91//***************************************************************************
92
93//---------------------------------------------------------------------------
94bool File_SequenceInfo::FileHeader_Begin()
95{
96 XMLDocument document;
97 if (!FileHeader_Begin_XML(document))
98 return false;
99
100 {
101 XMLElement* Root=document.FirstChildElement("SEQUENCEINFO");
102 if (Root)
103 {
104 Accept("SequenceInfo");
105 Fill(Stream_General, 0, General_Format, "SequenceInfo");
106
107 ReferenceFiles=new File__ReferenceFilesHelper(this, Config);
108
109 sequence* Sequence=new sequence;
110 Sequence->StreamKind=Stream_Video;
111
112 FileName FN(File_Name);
113 Ztring Base=FN.Path_Get();
114 size_t Pos=Base.rfind(PathSeparator);
115 if (Pos!=string::npos)
116 {
117 Ztring ToAdd=Base.substr(Pos, string::npos);
118 Ztring DirectoryBase=Base;
119 DirectoryBase+=ToAdd;
120 DirectoryBase+=__T('_');
121
122 size_t DirNumberCount=1;
123 Ztring Directory=DirectoryBase;
124 for (; DirNumberCount<9; DirNumberCount++)
125 {
126 Directory+=__T('0');
127 if (Dir::Exists(Directory))
128 break;
129 }
130
131 if (DirNumberCount<9)
132 {
133 int32u DirNumber=0;
134 do
135 {
136 Ztring Number=Ztring::ToZtring(DirNumber);
137 if (Number.size()<DirNumberCount)
138 Number.insert(0, DirNumberCount-Number.size(), __T('0'));
139
140 Directory=DirectoryBase;
141 Directory+=Number;
142 if (!Dir::Exists(Directory))
143 break;
144
145 Ztring FileBase=Directory;
146 FileBase+=ToAdd;
147 FileBase+=__T('_');
148 FileBase+=__T('.');
149
150 size_t FileNumberCount=1;
151 Ztring FullFile=FileBase;
152 Ztring Extension;
153 for (; FileNumberCount<10; FileNumberCount++)
154 {
155 FullFile.insert(FullFile.begin()+FullFile.size()-Extension.size()-1, __T('0'));
156 if (Extension.empty())
157 {
158 ZtringList List=Dir::GetAllFileNames(FullFile+__T('*'));
159 if (List.size()>=2)
160 {
161 FileNumberCount=(size_t)-1; //Problem, which one to choose?
162 break;
163 }
164 else if (List.size()==1)
165 {
166 FileName Temp(List[0]);
167 Extension=Temp.Extension_Get();
168 FileBase+=Extension;
169 FullFile=FileBase;
170 break;
171 }
172 }
173 else if (File::Exists(FullFile))
174 break;
175 }
176 bool FromZero=true;
177 if (FileNumberCount>=9)
178 {
179 //Trying with consecutive file numbers betweens dirs
180 Number=Ztring::ToZtring(Sequence->FileNames.size());
181 FullFile=FileBase;
182 FullFile.insert(FullFile.size()-Extension.size()-1, Number);
183 FileNumberCount=Number.size();
184 if (!File::Exists(FullFile))
185 {
186 FileNumberCount++;
187 for (; FileNumberCount<10; FileNumberCount++)
188 {
189 FullFile.insert(FullFile.begin()+FullFile.size()-Extension.size()-Number.size()-1, __T('0'));
190 if (File::Exists(FullFile))
191 {
192 FromZero=false;
193 break;
194 }
195 }
196 }
197 else
198 FromZero=false;
199 }
200
201 if (FileNumberCount<9)
202 {
203 size_t FileNumber=FromZero?0:Sequence->FileNames.size();
204 do
205 {
206 Number=Ztring::ToZtring(FileNumber);
207 if (Number.size()<FileNumberCount)
208 Number.insert(0, FileNumberCount-Number.size(), __T('0'));
209
210 FullFile=FileBase;
211 FullFile.insert(FullFile.size()-Extension.size()-1, Number);
212 if (!File::Exists(FullFile))
213 break;
214
215 Sequence->AddFileName(FullFile);
216
217 FileNumber++;
218 }
219 while (FileNumber<1000000000);
220 }
221
222 DirNumber++;
223 }
224 while (DirNumber<1000000000);
225
226 ReferenceFiles->AddSequence(Sequence);
227 }
228 }
229 }
230 else
231 {
232 Reject("SequenceInfo");
233 return false;
234 }
235 }
236
237 Element_Offset=File_Size;
238
239 //All should be OK...
240 return true;
241}
