239//---------------------------------------------------------------------------
240void File_Wm::Header_FileProperties()
241{
242 Element_Name("File Properties");
243
244 //Parsing
245 int64u CreationDate, PlayDuration, SendDuration, Preroll;
246 int32u Flags, MaximumBitRate;
247 Skip_GUID( "File ID");
248 Skip_L8( "File Size");
249 Get_L8 (CreationDate, "Creation Date"); Param_Info1(Ztring().Date_From_Milliseconds_1601(CreationDate/10000));
250 Skip_L8( "Data Packets Count");
251 Get_L8 (PlayDuration, "Play Duration"); Param_Info_From_Milliseconds(PlayDuration/10000);
252 Get_L8 (SendDuration, "Send Duration"); Param_Info_From_Milliseconds(SendDuration/10000);
253 Get_L8 (Preroll, "Preroll"); Param_Info_From_Milliseconds(Preroll);
254 Get_L4 (Flags, "Flags");
255 Skip_Flags(Flags, 0, "Broadcast");
256 Skip_Flags(Flags, 1, "Seekable");
257 Skip_Flags(Flags, 2, "Use Packet Template");
258 Skip_Flags(Flags, 3, "Live");
259 Skip_Flags(Flags, 4, "Recordable");
260 Skip_Flags(Flags, 5, "Unknown Data Size");
261 Skip_L4( "Minimum Data Packet Size");
262 Get_L4 (MaximumDataPacketSize, "Maximum Data Packet Size");
263 Get_L4 (MaximumBitRate, "Maximum Bitrate");
264
265 //Filling
266 if (MaximumBitRate)
267 Fill(Stream_General, 0, General_OverallBitRate_Maximum, MaximumBitRate);
268 Fill(Stream_General, 0, General_Encoded_Date, Ztring().Date_From_Milliseconds_1601(CreationDate/10000));
269 if (PlayDuration/1000>Preroll)
270 Fill(Stream_General, 0, General_Duration, PlayDuration/10000-Preroll);
271 FileProperties_Preroll=(int32u)(Preroll);
272}

274//---------------------------------------------------------------------------
275void File_Wm::Header_StreamProperties ()
276{
277 Element_Name("Stream Properties");
278
279 //Parsing
280 int128u StreamType;
281 int32u StreamTypeLength, ErrorCorrectionTypeLength;
282 Get_GUID(StreamType, "StreamType"); Param_Info1(Wm_StreamType(StreamType)); Element_Info1(Wm_StreamType(StreamType));
283 Skip_GUID( "Error Correction Type");
284 Skip_L8( "Time Offset");
285 Get_L4 (StreamTypeLength, "Type-Specific Data Length");
286 Get_L4 (ErrorCorrectionTypeLength, "Error Correction Data Length");
287 Get_L2 (Stream_Number, "Stream Number");
288 if (Stream_Number&0x8000)
289 {
290 Param_Info1("Encrypted Content");
291 Stream[Stream_Number&0x007F].Info["Encryption"]=__T("Encrypted");
292 }
293 Stream_Number&=0x007F; //Only 7bits
294 Element_Info1(Stream_Number);
295 Skip_L4( "Reserved");
296 switch (StreamType.hi)
297 {
298 case Elements::Header_StreamProperties_Audio : Element_Begin0(); //size is StreamTypeLength
299 Header_StreamProperties_Audio();
300 Element_End0(); break;
301 case Elements::Header_StreamProperties_Video : Element_Begin0(); //size is StreamTypeLength
302 Header_StreamProperties_Video();
303 Element_End0(); break;
304 case Elements::Header_StreamProperties_JFIF : Element_Begin0(); //size is StreamTypeLength
305 Header_StreamProperties_JFIF();
306 Element_End0(); break;
307 case Elements::Header_StreamProperties_DegradableJPEG : Element_Begin0(); //size is StreamTypeLength
308 Header_StreamProperties_DegradableJPEG();
309 Element_End0(); break;
310 case Elements::Header_StreamProperties_FileTransfer :
311 case Elements::Header_StreamProperties_Binary : Element_Begin0(); //size is StreamTypeLength
312 Header_StreamProperties_Binary();
313 StreamKind_Last=Stream_Max; StreamPos_Last=(size_t)-1;
314 Element_End0(); break;
315 default : if (StreamTypeLength>0)
316 Skip_XX(StreamTypeLength, "Type-Specific Data");
317 StreamKind_Last=Stream_Max; StreamPos_Last=(size_t)-1;
318 }
319 if (ErrorCorrectionTypeLength)
320 Skip_XX(ErrorCorrectionTypeLength, "Error Correction Data");
321
322 //Filling
323 Stream[Stream_Number].StreamKind=StreamKind_Last;
324 Stream[Stream_Number].StreamPos=StreamPos_Last;
325 Stream[Stream_Number].Info["ID"].From_Number(Stream_Number);
326 Stream[Stream_Number].Info["StreamOrder"].From_Number(Header_StreamProperties_StreamOrder);
327 Header_StreamProperties_StreamOrder++;
328}

330//---------------------------------------------------------------------------
331void File_Wm::Header_StreamProperties_Audio ()
332{
333 Element_Name("Audio");
334
335 //Parsing
336 int32u SamplingRate, BytesPerSec;
337 int16u CodecID, Channels, Data_Size, Resolution;
338 Get_L2 (CodecID, "Codec ID");
339 Get_L2 (Channels, "Number of Channels");
340 Get_L4 (SamplingRate, "Samples Per Second");
341 Get_L4 (BytesPerSec, "Average Number of Bytes Per Second");
342 Skip_L2( "Block Alignment");
343 Get_L2 (Resolution, "Bits / Sample");
344 Get_L2 (Data_Size, "Codec Specific Data Size");
345
346 //Filling
347 Stream_Prepare(Stream_Audio);
348 Stream[Stream_Number].IsCreated=true;
349 Ztring Codec; Codec.From_Number(CodecID, 16);
350 Codec.MakeUpperCase();
351 CodecID_Fill(Codec, Stream_Audio, StreamPos_Last, InfoCodecID_Format_Riff);
352 Fill(Stream_Audio, StreamPos_Last, Audio_Codec, Codec); //May be replaced by codec parser
353 Fill(Stream_Audio, StreamPos_Last, Audio_Codec_CC, Codec);
354 Fill(Stream_Audio, StreamPos_Last, Audio_Channel_s_, Channels);
355 Fill(Stream_Audio, StreamPos_Last, Audio_SamplingRate, SamplingRate);
356 Fill(Stream_Audio, StreamPos_Last, Audio_BitRate, BytesPerSec*8);
357 Fill(Stream_Audio, StreamPos_Last, Audio_BitDepth, Resolution);
358
359 FILLING_BEGIN();
360 //Creating the parser
361 if (0);
362 #if defined(MEDIAINFO_MPEGA_YES)
363 else if (MediaInfoLib::Config.CodecID_Get(Stream_Audio, InfoCodecID_Format_Riff, Ztring::ToZtring(CodecID, 16))==__T("MPEG Audio"))
364 {
365 Stream[Stream_Number].Parser=new File_Mpega;
366 ((File_Mpega*)Stream[Stream_Number].Parser)->Frame_Count_Valid=8;
367 Stream[Stream_Number].Parser->ShouldContinueParsing=true;
368 }
369 #endif
370 Open_Buffer_Init(Stream[Stream_Number].Parser);
371 FILLING_END();
372
373 //Parsing
374 if (Data_Size>0)
375 {
376 Element_Begin1("Codec Specific Data");
377 switch (CodecID)
378 {
379 case 0x0161 :
380 case 0x0162 :
381 case 0x0163 : Header_StreamProperties_Audio_WMA(); break;
382 case 0x7A21 :
383 case 0x7A22 : Header_StreamProperties_Audio_AMR(); break;
384 default : Skip_XX(Data_Size, "Unknown");
385 }
386 Element_End0();
387 }
388}

437//---------------------------------------------------------------------------
438void File_Wm::Header_StreamProperties_Video ()
439{
440 Element_Name("Video");
441
442 //Parsing
443 int32u Width, Height, Compression;
444 int16u Data_Size, Resolution;
445 Get_L4 (Width, "Width");
446 Get_L4 (Height, "Height");
447 Skip_L1( "Flags");
448 Get_L2 (Data_Size, "Format Data Size");
449 Skip_L4( "Size");
450 Get_L4 (Width, "Width");
451 Get_L4 (Height, "Height");
452 Skip_L2( "Planes");
453 Get_L2 (Resolution, "BitCount");
454 Get_C4 (Compression, "Compression");
455 Skip_L4( "SizeImage");
456 Skip_L4( "XPelsPerMeter");
457 Skip_L4( "YPelsPerMeter");
458 Skip_L4( "ClrUsed");
459 Skip_L4( "ClrImportant");
460
461 //Filling
462 Stream_Prepare(Stream_Video);
463 Stream[Stream_Number].IsCreated=true;
464 CodecID_Fill(Ztring().From_CC4(Compression), Stream_Video, StreamPos_Last, InfoCodecID_Format_Riff);
465 Fill(Stream_Video, StreamPos_Last, Video_Codec, Ztring().From_CC4(Compression)); //May be replaced by codec parser
466 Fill(Stream_Video, StreamPos_Last, Video_Codec_CC, Ztring().From_CC4(Compression));
467 Fill(Stream_Video, StreamPos_Last, Video_Width, Width);
468 Fill(Stream_Video, StreamPos_Last, Video_Height, Height);
469 if (Resolution>0)
470 Fill(Stream_Video, StreamPos_Last, Video_BitDepth, (Resolution%3)?Resolution:(Resolution/3)); //If not a multiple of 3, the total resolution is filled
471 if (Compression==CC4("DVR "))
472 IsDvrMs=true;
473
474 //From Content description (we imagine that data is for all video streams...)
475 if (Header_ExtendedContentDescription_AspectRatioX && Header_ExtendedContentDescription_AspectRatioY)
476 {
477 if (Header_ExtendedContentDescription_AspectRatioX==16 && Header_ExtendedContentDescription_AspectRatioY==9)
478 Fill(Stream_Video, StreamPos_Last, Video_DisplayAspectRatio, ((float32)16)/9, 3);
479 else if (Header_ExtendedContentDescription_AspectRatioX==4 && Header_ExtendedContentDescription_AspectRatioY==3)
480 Fill(Stream_Video, StreamPos_Last, Video_DisplayAspectRatio, ((float32)4)/3, 3);
481 else
482 Fill(Stream_Video, StreamPos_Last, Video_PixelAspectRatio, ((float32)Header_ExtendedContentDescription_AspectRatioX)/Header_ExtendedContentDescription_AspectRatioY, 3, true);
483 }
484
485 //Creating the parser
486 if (0);
487 #if defined(MEDIAINFO_VC1_YES)
488 else if (MediaInfoLib::Config.CodecID_Get(Stream_Video, InfoCodecID_Format_Riff, Ztring().From_CC4(Compression), InfoCodecID_Format)==__T("VC-1"))
489 {
490 Stream[Stream_Number].Parser=new File_Vc1;
491 if (Compression==CC4("WMV3"))
492 {
493 ((File_Vc1*)Stream[Stream_Number].Parser)->From_WMV3=true;
494 ((File_Vc1*)Stream[Stream_Number].Parser)->MustSynchronize=false;
495 }
496 ((File_Vc1*)Stream[Stream_Number].Parser)->FrameIsAlwaysComplete=true; //Warning: this is not always the case, see data parsing
497 Open_Buffer_Init(Stream[Stream_Number].Parser);
498 if (Data_Size>40)
499 {
500
501 //Demux
502 #if MEDIAINFO_DEMUX
503 switch (Config->Demux_InitData_Get())
504 {
505 case 0 : //In demux event
506 Element_Code=Stream_Number;
507 Demux_Level=2; //Container
508 Demux(Buffer+(size_t)Element_Offset, (size_t)(Data_Size-40), ContentType_Header);
509 break;
510 case 1 : //In field
511 {
512 std::string Data_Raw((const char*)(Buffer+(size_t)Element_Offset), (size_t)(Data_Size-40));
513 std::string Data_Base64(Base64::encode(Data_Raw));
514 Fill(Stream_Video, StreamPos_Last, "Demux_InitBytes", Data_Base64);
515 (*Stream_More)[Stream_Video][StreamPos_Last](Ztring().From_Local("Demux_InitBytes"), Info_Options)=__T("N NT");
516 }
517 break;
518 default : ;
519 }
520 #endif //MEDIAINFO_DEMUX
521
522 Open_Buffer_Continue(Stream[Stream_Number].Parser, (size_t)(Data_Size-40));
523 if (Stream[Stream_Number].Parser->Status[IsFinished])
524 {
525 Finish(Stream[Stream_Number].Parser);
526 Merge(*Stream[Stream_Number].Parser, Stream_Video, 0, StreamPos_Last);
527 delete Stream[Stream_Number].Parser; Stream[Stream_Number].Parser=NULL;
528 }
529 else
530 {
531 ((File_Vc1*)Stream[Stream_Number].Parser)->Only_0D=true;
532 ((File_Vc1*)Stream[Stream_Number].Parser)->MustSynchronize=false;
533 }
534 }
535 }
536 #endif
537 #if defined(MEDIAINFO_MPEGV_YES)
538 else if (MediaInfoLib::Config.Codec_Get(Ztring().From_CC4(Compression), InfoCodec_KindofCodec).find(__T("MPEG-2"))==0)
539 {
540 Stream[Stream_Number].Parser=new File_Mpegv;
541 ((File_Mpegv*)Stream[Stream_Number].Parser)->Frame_Count_Valid=30; //For searching Pulldown
542 Open_Buffer_Init(Stream[Stream_Number].Parser);
543 }
544 #endif
545 else if (Data_Size>40) //TODO: see "The Mummy_e"
546 Skip_XX(Data_Size-40, "Codec Specific Data");
547}

1371void File_Wm::Data_Packet()
1372{
1373 //Counting
1374 Packet_Count++;
1375 Element_Info1(Packet_Count);
1376 size_t Element_Show_Count=0;
1377
1378 //Parsing
1379 int32u PacketLength=0, SizeOfMediaObject=0;
1380 int8u Flags, ErrorCorrectionData_Length, ErrorCorrectionLengthType, SequenceType, PaddingLengthType, PacketLengthType;
1381 bool ErrorCorrectionPresent;
1382 Element_Begin1("Error Correction");
1383 Get_L1 (Flags, "Flags");
1384 Get_FlagsM(Flags&0x0F, ErrorCorrectionData_Length, "Error Correction Data Length"); //4 lowest bits
1385 Skip_Flags(Flags, 4, "Opaque Data Present");
1386 Get_FlagsM((Flags>>5)&0x03, ErrorCorrectionLengthType, "Error Correction Length Type"); //bits 6 and 7
1387 Get_Flags (Flags, 7, ErrorCorrectionPresent, "Error Correction Present");
1388 if (ErrorCorrectionPresent && ErrorCorrectionLengthType==0 && ErrorCorrectionData_Length==2)
1389 {
1390 int8u TypeNumber;
1391 Get_L1 (TypeNumber, "Type/Number");
1392 Skip_FlagsM((TypeNumber>>4)&0x0F, "Type");
1393 Skip_FlagsM( TypeNumber &0x0F, "Number");
1394 Skip_L1( "Cycle");
1395 }
1396 Element_End0();
1397
1398 Element_Begin1("Payload Parsing Information");
1399 Get_L1 (Flags, "Length Type Flags");
1400 Get_Flags (Flags, 0, MultiplePayloadsPresent, "Multiple Payloads Present");
1401 Get_FlagsM((Flags>>1)&0x3, SequenceType, "Sequence Type");
1402 Get_FlagsM((Flags>>3)&0x3, PaddingLengthType, "Padding Length Type");
1403 Get_FlagsM((Flags>>5)&0x3, PacketLengthType, "Packet Length Type");
1404 Skip_Flags(Flags, 7, "Error Correction Present");
1405 Get_L1 (Flags, "Property Flags");
1406 Get_FlagsM( Flags &0x3, ReplicatedDataLengthType, "Replicated Data Length Type");
1407 Get_FlagsM((Flags>>2)&0x3, OffsetIntoMediaObjectLengthType, "Offset Into Media Object Length Type");
1408 Get_FlagsM((Flags>>4)&0x3, MediaObjectNumberLengthType, "Media Object Number Length Type");
1409 Get_FlagsM((Flags>>6)&0x3, StreamNumberLengthType, "Stream Number Length Type");
1410 switch (PacketLengthType)
1411 {
1412 case 1 : {int8u Data; Get_L1(Data, "Packet Length"); PacketLength=Data;} break;
1413 case 2 : {int16u Data; Get_L2(Data, "Packet Length"); PacketLength=Data;} break;
1414 case 3 : Get_L4(PacketLength, "Packet Length"); break;
1415 default: ;
1416 }
1417 switch (SequenceType)
1418 {
1419 case 1 : Skip_L1( "Sequence"); break;
1420 case 2 : Skip_L2( "Sequence"); break;
1421 case 3 : Skip_L4( "Sequence"); break;
1422 default: ;
1423 }
1424 switch (PaddingLengthType)
1425 {
1426 case 1 : {int8u Data; Get_L1(Data, "Padding Length"); Data_Parse_Padding=Data;} break;
1427 case 2 : {int16u Data; Get_L2(Data, "Padding Length"); Data_Parse_Padding=Data;} break;
1428 case 3 : Get_L4(Data_Parse_Padding, "Padding Length"); break;
1429 default: Data_Parse_Padding=0;
1430 }
1431 Skip_L4( "Send Time");
1432 Skip_L2( "Duration");
1433 Element_End0();
1434
1435 if (MultiplePayloadsPresent)
1436 {
1437 //Parsing
1438 Element_Begin1("Multiple Payloads additional flags");
1439 int8u AdditionalFlags;
1440 Get_L1 (AdditionalFlags, "Flags");
1441 Get_FlagsM( AdditionalFlags &0x3F, NumberPayloads, "Number of Payloads"); //6 bits
1442 Get_FlagsM((AdditionalFlags>>6)&0x03, PayloadLengthType, "Payload Length Type"); //bits 6 and 7
1443 Element_End0();
1444 }
1445 else
1446 {
1447 SizeOfMediaObject=(int32u)(Element_Size-Element_Offset-Data_Parse_Padding);
1448 NumberPayloads=1;
1449 }
1450
1451 for (NumberPayloads_Pos=0; NumberPayloads_Pos<NumberPayloads; NumberPayloads_Pos++)
1452 {
1453 Element_Begin1("Payload");
1454 int32u ReplicatedDataLength=0, PayloadLength=0;
1455 int8u StreamNumber;
1456 Get_L1 (StreamNumber, "Stream Number");
1457 Stream_Number=StreamNumber&0x7F; //For KeyFrame
1458 Element_Info1(Stream_Number);
1459 switch (MediaObjectNumberLengthType)
1460 {
1461 case 1 : Skip_L1( "Media Object Number"); break;
1462 case 2 : Skip_L2( "Media Object Number"); break;
1463 case 3 : Skip_L4( "Media Object Number"); break;
1464 default: Trusted_IsNot("Media Object Number"); return; //Problem
1465 }
1466 switch (OffsetIntoMediaObjectLengthType)
1467 {
1468 case 1 : Skip_L1( "Offset Into Media Object"); break;
1469 case 2 : Skip_L2( "Offset Into Media Object"); break;
1470 case 3 : Skip_L4( "Offset Into Media Object"); break;
1471 default: Trusted_IsNot("Offset Into Media Object"); return; //Problem
1472 }
1473 switch (ReplicatedDataLengthType)
1474 {
1475 case 1 : {int8u Data; Get_L1(Data, "Replicated Data Length"); ReplicatedDataLength=Data;} break;
1476 case 2 : {int16u Data; Get_L2(Data, "Replicated Data Length"); ReplicatedDataLength=Data;} break;
1477 case 3 : Get_L4(ReplicatedDataLength, "Replicated Data Length"); break;
1478 default: Trusted_IsNot("Replicated Data Length"); return; //Problem
1479 }
1480 if (ReplicatedDataLengthType!=0 && ReplicatedDataLength>0)
1481 {
1482 if (ReplicatedDataLength>=8)
1483 {
1484 int32u PresentationTime;
1485 Get_L4 (SizeOfMediaObject, "Size Of Media Object");
1486 Get_L4 (PresentationTime, "Presentation Time");
1487 if (ReplicatedDataLength>8)
1488 Data_Packet_ReplicatedData(ReplicatedDataLength-8);
1489
1490 //Presentation time delta
1491 std::map<int16u, stream>::iterator Strea=Stream.find(Stream_Number);
1492 if (Strea!=Stream.end() && Strea->second.StreamKind==Stream_Video)
1493 {
1494 Strea->second.PresentationTimes.insert(PresentationTime);
1495 }
1496 }
1497 else if (ReplicatedDataLength==1)
1498 {
1499 Skip_L1( "Presentation Time Delta");
1500 //TODO
1501 }
1502 else
1503 {
1504 Skip_XX(ReplicatedDataLength, "Replicated Data");
1505 }
1506 }
1507
1508 if (MultiplePayloadsPresent)
1509 {
1510 switch (PayloadLengthType)
1511 {
1512 case 1 : {int8u Data; Get_L1(Data, "Payload Length"); PayloadLength=Data;} break;
1513 case 2 : {int16u Data; Get_L2(Data, "Payload Length"); PayloadLength=Data;} break;
1514 case 3 : Get_L4(PayloadLength, "Payload Length"); break;
1515 default: Trusted_IsNot("Payload Length"); return; //Problem
1516 }
1517 }
1518 else if (Element_Size-Element_Offset>Data_Parse_Padding)
1519 PayloadLength=(int32u)(Element_Size-(Element_Offset+Data_Parse_Padding));
1520 else
1521 {
1522 Trusted_IsNot("Padding size problem");
1523 }
1524 if (Element_Offset+PayloadLength+Data_Parse_Padding>Element_Size)
1525 {
1526 Trusted_IsNot("Payload Length problem");
1527 }
1528 else
1529 {
1530 //Demux
1531 Element_Code=Stream_Number;
1532 Demux(Buffer+(size_t)Element_Offset, (size_t)PayloadLength, ContentType_MainStream);
1533
1534 //Analyzing
1535 if (Stream[Stream_Number].Parser && Stream[Stream_Number].SearchingPayload)
1536 {
1537 //Handling of spanned on multiple chunks
1538 #if defined(MEDIAINFO_VC1_YES)
1539 bool FrameIsAlwaysComplete=true;
1540 #endif
1541 if (PayloadLength!=SizeOfMediaObject)
1542 {
1543 if (SizeOfMediaObject_BytesAlreadyParsed==0)
1544 SizeOfMediaObject_BytesAlreadyParsed=SizeOfMediaObject-PayloadLength;
1545 else
1546 SizeOfMediaObject_BytesAlreadyParsed-=PayloadLength;
1547 if (SizeOfMediaObject_BytesAlreadyParsed==0)
1548 Element_Show_Count++;
1549 #if defined(MEDIAINFO_VC1_YES)
1550 else
1551 FrameIsAlwaysComplete=false;
1552 #endif
1553 }
1554 else
1555 Element_Show_Count++;
1556
1557 //Codec specific
1558 #if defined(MEDIAINFO_VC1_YES)
1559 if (Retrieve(Stream[Stream_Number].StreamKind, Stream[Stream_Number].StreamPos, Fill_Parameter(Stream[Stream_Number].StreamKind, Generic_Format))==__T("VC-1"))
1560 ((File_Vc1*)Stream[Stream_Number].Parser)->FrameIsAlwaysComplete=FrameIsAlwaysComplete;
1561 #endif
1562
1563 Open_Buffer_Continue(Stream[Stream_Number].Parser, (size_t)PayloadLength);
1564 if (Stream[Stream_Number].Parser->Status[IsFinished]
1565 || (Stream[Stream_Number].PresentationTimes.size()>=300 && MediaInfoLib::Config.ParseSpeed_Get()<1))
1566 {
1567 Stream[Stream_Number].Parser->Open_Buffer_Unsynch();
1568 Stream[Stream_Number].SearchingPayload=false;
1569 Streams_Count--;
1570 }
1571
1572 Element_Show();
1573 }
1574 else
1575 {
1576 Skip_XX(PayloadLength, "Data");
1577 if (Stream[Stream_Number].SearchingPayload
1578 && (Stream[Stream_Number].StreamKind==Stream_Video && Stream[Stream_Number].PresentationTimes.size()>=300))
1579 {
1580 Stream[Stream_Number].SearchingPayload=false;
1581 Streams_Count--;
1582 }
1583 }
1584 }
1585 Element_End0();
1586 }
1587
1588 if (Data_Parse_Padding)
1589 Skip_XX(Data_Parse_Padding, "Padding");
1590
1591 //Jumping if needed
1592 if (Streams_Count==0 || (Packet_Count>=1000 && MediaInfoLib::Config.ParseSpeed_Get()<1))
1593 {
1594 Info("Data, Jumping to end of chunk");
1595 GoTo(Data_AfterTheDataChunk, "Windows Media");
1596 }
1597
1598 if (Element_Show_Count>0)
1599 Element_Show();
1600}
