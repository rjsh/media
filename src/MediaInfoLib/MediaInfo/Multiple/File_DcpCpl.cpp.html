97static bool IsSmpteSt2067_2(const char* Ns)
98{
99 return Ns &&
100 (!strcmp(Ns, "http://www.smpte-ra.org/schemas/2067-2/2013") ||
101 !strcmp(Ns, "http://www.smpte-ra.org/schemas/2067-2/XXXX")); //Some muxers use XXXX instead of year
102
103}
104
105//---------------------------------------------------------------------------
106
107static bool IsSmpteSt2067_3(const char* Ns)
108{
109 return Ns &&
110 (!strcmp(Ns, "http://www.smpte-ra.org/schemas/2067-3/2013") ||
111 !strcmp(Ns, "http://www.smpte-ra.org/schemas/2067-3/XXXX")); //Some muxers use XXXX instead of year
112
113}

115//***************************************************************************
116// Buffer - File header
117//***************************************************************************
118
119//---------------------------------------------------------------------------
120bool File_DcpCpl::FileHeader_Begin()
121{
122 XMLDocument document;
123 if (!FileHeader_Begin_XML(document))
124 return false;
125
126 XMLElement* Root=document.FirstChildElement();
127 const char *NameSpace;
128 if (!Root || strcmp(LocalName(Root, NameSpace), "CompositionPlaylist"))
129 {
130 Reject("DcpCpl");
131 return false;
132 }
133
134 bool IsDcp=false, IsImf=false;
135 if (!strcmp(NameSpace, "http://www.digicine.com/PROTO-ASDCP-CPL-20040511#") ||
136 !strcmp(NameSpace, "http://www.smpte-ra.org/schemas/429-7/2006/CPL"))
137 {
138 IsDcp=true;
139 }
140 else if (IsSmpteSt2067_3(NameSpace))
141 {
142 IsImf=true;
143 }
144 else
145 {
146 Reject("DcpCpl");
147 return false;
148 }
149
150 Accept("DcpCpl");
151 Fill(Stream_General, 0, General_Format, IsDcp?"DCP CPL":"IMF CPL");
152 Config->File_ID_OnlyRoot_Set(false);
153
154 ReferenceFiles=new File__ReferenceFilesHelper(this, Config);
155
156 //Parsing main elements
157 for (XMLElement* CompositionPlaylist_Item=Root->FirstChildElement(); CompositionPlaylist_Item; CompositionPlaylist_Item=CompositionPlaylist_Item->NextSiblingElement())
158 {
159 //CompositionTimecode
160 if (IsImf && MatchQName(CompositionPlaylist_Item, "CompositionTimecode", NameSpace))
161 {
162 sequence* Sequence=new sequence;
163 Sequence->StreamKind=Stream_Other;
164 Sequence->Infos["Type"]=__T("Time code");
165 Sequence->Infos["Format"]=__T("CPL TC");
166 Sequence->Infos["TimeCode_Striped"]=__T("Yes");
167 bool IsDropFrame=false;
168
169 for (XMLElement* CompositionTimecode_Item=CompositionPlaylist_Item->FirstChildElement(); CompositionTimecode_Item; CompositionTimecode_Item=CompositionTimecode_Item->NextSiblingElement())
170 {
171 const char* Text=CompositionTimecode_Item->GetText();
172 if (!Text)
173 continue;
174 const char *CtItemNs, *CtItemName = LocalName(CompositionTimecode_Item, CtItemNs);
175 if (!CtItemNs || strcmp(CtItemNs, NameSpace))
176 continue; // item has wrong namespace
177
178 //TimecodeDropFrame
179 if (!strcmp(CtItemName, "TimecodeDropFrame"))
180 {
181 if (strcmp(Text, "") && strcmp(Text, "0"))
182 IsDropFrame=true;
183 }
184
185 //TimecodeRate
186 if (!strcmp(CtItemName, "TimecodeRate"))
187 Sequence->Infos["FrameRate"].From_UTF8(Text);
188
189 //TimecodeStartAddress
190 if (!strcmp(CtItemName, "TimecodeStartAddress"))
191 Sequence->Infos["TimeCode_FirstFrame"].From_UTF8(Text);
192 }
193
194 //Adaptation
195 if (IsDropFrame)
196 {
197 std::map<string, Ztring>::iterator Info=Sequence->Infos.find("TimeCode_FirstFrame");
198 if (Info!=Sequence->Infos.end() && Info->second.size()>=11 && Info->second[8]!=__T(';'))
199 Info->second[8]=__T(';');
200 }
201
202 Sequence->StreamID=ReferenceFiles->Sequences_Size()+1;
203 ReferenceFiles->AddSequence(Sequence);
204
205 Stream_Prepare(Stream_Other);
206 Fill(Stream_Other, StreamPos_Last, Other_ID, Sequence->StreamID);
207 for (std::map<string, Ztring>::iterator Info=Sequence->Infos.begin(); Info!=Sequence->Infos.end(); ++Info)
208 Fill(Stream_Other, StreamPos_Last, Info->first.c_str(), Info->second);
209 }
210
211 //ReelList / SegmentList
212 if (MatchQName(CompositionPlaylist_Item, IsDcp?"ReelList":"SegmentList", NameSpace))
213 {
214 for (XMLElement* ReelList_Item=CompositionPlaylist_Item->FirstChildElement(); ReelList_Item; ReelList_Item=ReelList_Item->NextSiblingElement())
215 {
216 //Reel
217 if (MatchQName(ReelList_Item, IsDcp?"Reel":"Segment", NameSpace))
218 {
219 for (XMLElement* Reel_Item=ReelList_Item->FirstChildElement(); Reel_Item; Reel_Item=Reel_Item->NextSiblingElement())
220 {
221 //AssetList
222 if (MatchQName(Reel_Item, IsDcp?"AssetList":"SequenceList", NameSpace))
223 {
224 for (XMLElement* AssetList_Item=Reel_Item->FirstChildElement(); AssetList_Item; AssetList_Item=AssetList_Item->NextSiblingElement())
225 {
226 const char *AlItemNs, *AlItemName = LocalName(AssetList_Item, AlItemNs);
227 if (!AlItemNs)
228 continue;
229 //File
230 //if ((IsDcp && (!strcmp(AssetList_Item->Value(), "MainPicture") || !strcmp(AssetList_Item->Value(), "MainSound")))
231 // || (IsImf && (!strcmp(AssetList_Item->Value(), "cc:MainImageSequence") || !strcmp(AssetList_Item->Value(), "cc:MainImage"))))
232 if (strcmp(AlItemName, "MarkerSequence")) //Ignoring MarkerSequence for the moment. TODO: check what to do with MarkerSequence
233 {
234 sequence* Sequence=new sequence;
235 Ztring Asset_Id;
236
237 if (IsDcp && !strcmp(NameSpace, AlItemNs))
238 {
239 if (!strcmp(AlItemName, "MainPicture"))
240 Sequence->StreamKind=Stream_Video;
241 else if (!strcmp(AlItemName, "MainSound"))
242 Sequence->StreamKind=Stream_Audio;
243 }
244 else if (IsImf && IsSmpteSt2067_2(AlItemNs))
245 {
246 if (!strcmp(AlItemName, "MainImageSequence"))
247 Sequence->StreamKind=Stream_Video;
248 else if (!strcmp(AlItemName, "MainAudioSequence"))
249 Sequence->StreamKind=Stream_Audio;
250 }
251
252 for (XMLElement* File_Item=AssetList_Item->FirstChildElement(); File_Item; File_Item=File_Item->NextSiblingElement())
253 {
254 //Id
255 if (MatchQName(File_Item, "Id", NameSpace) && Asset_Id.empty())
256 Asset_Id.From_UTF8(File_Item->GetText());
257
258 //ResourceList
259 if (IsImf && MatchQName(File_Item, "ResourceList", NameSpace))
260 {
261 for (XMLElement* ResourceList_Item=File_Item->FirstChildElement(); ResourceList_Item; ResourceList_Item=ResourceList_Item->NextSiblingElement())
262 {
263 //Resource
264 if (MatchQName(ResourceList_Item, "Resource", NameSpace))
265 {
266 Ztring Resource_Id;
267
268 resource* Resource=new resource;
269 for (XMLElement* Resource_Item=ResourceList_Item->FirstChildElement(); Resource_Item; Resource_Item=Resource_Item->NextSiblingElement())
270 {
271 const char* ResText=Resource_Item->GetText();
272 if (!ResText)
273 continue;
274 const char *ResItemNs, *ResItemName = LocalName(Resource_Item, ResItemNs);
275 if (!ResItemNs || strcmp(ResItemNs, NameSpace))
276 continue; // item has wrong namespace
277
278 //EditRate
279 if (!strcmp(ResItemName, "EditRate"))
280 {
281 Resource->EditRate=atof(ResText);
282 const char* EditRate2=strchr(ResText, ' ');
283 if (EditRate2!=NULL)
284 {
285 float64 EditRate2f=atof(EditRate2);
286 if (EditRate2f)
287 Resource->EditRate/=EditRate2f;
288 }
289 }
290
291 //EntryPoint
292 if (!strcmp(ResItemName, "EntryPoint"))
293 {
294 Resource->IgnoreEditsBefore=atoi(ResText);
295 if (Resource->IgnoreEditsAfter!=(int64u)-1)
296 Resource->IgnoreEditsAfter+=Resource->IgnoreEditsBefore;
297 }
298
299 //Id
300 if (!strcmp(ResItemName, "Id") && Resource_Id.empty())
301 Resource_Id.From_UTF8(ResText);
302
303 //SourceDuration
304 if (!strcmp(ResItemName, "SourceDuration"))
305 Resource->IgnoreEditsAfter=Resource->IgnoreEditsBefore+atoi(ResText);
306
307 //TrackFileId
308 if (!strcmp(ResItemName, "TrackFileId"))
309 Resource->FileNames.push_back(Ztring().From_UTF8(ResText));
310 }
311
312 if (Resource->FileNames.empty())
313 Resource->FileNames.push_back(Resource_Id);
314 Sequence->AddResource(Resource);
315 }
316 }
317 }
318 }
319
320 if (Sequence->Resources.empty())
321 {
322 resource* Resource=new resource;
323 Resource->FileNames.push_back(Asset_Id);
324 Sequence->AddResource(Resource);
325 }
326 Sequence->StreamID=ReferenceFiles->Sequences_Size()+1;
327 ReferenceFiles->AddSequence(Sequence);
328 }
329 }
330 }
331 }
332 }
333 }
334 }
335 }
336
337 Element_Offset=File_Size;
338
339 //Getting files names
340 FileName Directory(File_Name);
341 Ztring DirPath = Directory.Path_Get();
342 if (!DirPath.empty())
343 DirPath += PathSeparator;
344 Ztring Assetmap_FileName=DirPath+__T("ASSETMAP.xml");
345 bool IsOk=false;
346 if (File::Exists(Assetmap_FileName))
347 IsOk=true;
348 else
349 {
350 Assetmap_FileName.resize(Assetmap_FileName.size()-4); //Old fashion, without ".xml"
351 if (File::Exists(Assetmap_FileName))
352 IsOk=true;
353 }
354 if (IsOk)
355 {
356 MediaInfo_Internal MI;
357 MI.Option(__T("File_KeepInfo"), __T("1"));
358 Ztring ParseSpeed_Save=MI.Option(__T("ParseSpeed_Get"), __T(""));
359 Ztring Demux_Save=MI.Option(__T("Demux_Get"), __T(""));
360 MI.Option(__T("ParseSpeed"), __T("0"));
361 MI.Option(__T("Demux"), Ztring());
362 MI.Option(__T("File_IsReferenced"), __T("1"));
363 size_t MiOpenResult=MI.Open(Assetmap_FileName);
364 MI.Option(__T("ParseSpeed"), ParseSpeed_Save); //This is a global value, need to reset it. TODO: local value
365 MI.Option(__T("Demux"), Demux_Save); //This is a global value, need to reset it. TODO: local value
366 if (MiOpenResult
367 && (MI.Get(Stream_General, 0, General_Format)==__T("DCP AM")
368 || MI.Get(Stream_General, 0, General_Format)==__T("IMF AM")))
369 {
370 MergeFromAm(((File_DcpAm*)MI.Info)->Streams);
371 }
372 }
373
374 ReferenceFiles->FilesForStorage=true;
375
376 //All should be OK...
377 return true;
378}

385void File_DcpCpl::MergeFromAm (File_DcpPkl::streams &StreamsToMerge)
386{
387 for (File_DcpPkl::streams::iterator StreamToMerge=StreamsToMerge.begin(); StreamToMerge!=StreamsToMerge.end(); ++StreamToMerge)
388 if (!StreamToMerge->ChunkList.empty()) // Note: ChunkLists with more than 1 file are not yet supported)
389 ReferenceFiles->UpdateFileName(Ztring().From_UTF8(StreamToMerge->Id), Ztring().From_UTF8(StreamToMerge->ChunkList[0].Path));
390}
