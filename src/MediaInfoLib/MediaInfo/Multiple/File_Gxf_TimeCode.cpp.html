105void File_Gxf_TimeCode::Streams_Fill()
106{
107 Stream_Prepare(Stream_Video);
108 Fill(Stream_Video, 0, Video_Delay, TimeCode_FirstFrame_ms);
109 if (TimeCode_FirstFrame.size()==11)
110 Fill(Stream_Video, StreamPos_Last, Video_Delay_DropFrame, TimeCode_FirstFrame[8]==';'?"Yes":"No");
111 Fill(Stream_Video, 0, Video_Delay_Source, "Container");
112 Fill(Stream_Video, 0, Video_TimeCode_FirstFrame, TimeCode_FirstFrame.c_str());
113 //Fill(Stream_Video, 0, Video_TimeCode_Source, "Time code track");
114
115 Stream_Prepare(Stream_Audio);
116 Fill(Stream_Audio, 0, Audio_Delay, TimeCode_FirstFrame_ms);
117 if (TimeCode_FirstFrame.size()==11)
118 Fill(Stream_Audio, StreamPos_Last, Audio_Delay_DropFrame, TimeCode_FirstFrame[8]==';'?"Yes":"No");
119 Fill(Stream_Audio, 0, Audio_Delay_Source, "Container");
120 Fill(Stream_Audio, 0, Video_TimeCode_FirstFrame, TimeCode_FirstFrame.c_str());
121 //Fill(Stream_Audio, 0, Video_TimeCode_Source, "Time code track");
122}

124//***************************************************************************
125// Buffer - Global
126//***************************************************************************
127
128//---------------------------------------------------------------------------
129void File_Gxf_TimeCode::Read_Buffer_Continue()
130{
131 int8u Validity[504];
132
133 if (!IsAtc)
134 {
135 if (Element_Size!=4096)
136 {
137 Skip_XX(Element_Size, "Data");
138 return;
139 }
140
141 //Reading bitmap first (validity of first byte is at the end)
142 Element_Offset=504*8;
143 Element_Begin1("Validity");
144 BS_Begin_LE(); //is Little Endian
145 for (size_t Pos=0; Pos<504; Pos++)
146 {
147 bool Validity_Bit;
148 Get_TB (Validity_Bit, "Bit");
149 Validity[Pos]=Validity_Bit?1:0;
150 }
151 BS_End_LE();
152 Skip_B1( "Pad");
153 Element_End0();
154 }
155
156 //Parsing
157 Element_Offset=0;
158 for (size_t Pos=0; Pos<(IsAtc?(size_t)1:(size_t)504); Pos++)
159 {
160 if (IsAtc || Validity[Pos])
161 {
162 Element_Begin1("TimeCode");
163 int8u Frames_Units, Frames_Tens, Seconds_Units, Seconds_Tens, Minutes_Units, Minutes_Tens, Hours_Units, Hours_Tens;
164 bool DropFrame;
165 int8u DBB1=0, DBB2=0;
166 if (IsAtc)
167 {
168 bool Temp;
169 BS_Begin();
170
171 Get_S1 (4, Frames_Units, "Frames (Units)");
172 Get_SB ( Temp, "DBB1_0"); if (Temp) DBB1|=(1<<0);
173 Skip_S1(3, "Zero");
174
175 Skip_S1(4, "BG1");
176 Get_SB ( Temp, "DBB1_1"); if (Temp) DBB1|=(1<<1);
177 Skip_S1(3, "Zero");
178
179 Skip_SB( "CF - Color fame");
180 Get_SB ( DropFrame, "DP - Drop frame");
181 Get_S1 (2, Frames_Tens, "Frames (Tens)");
182 Get_SB ( Temp, "DBB1_2"); if (Temp) DBB1|=(1<<2);
183 Skip_S1(3, "Zero");
184
185 Skip_S1(4, "BG2");
186 Get_SB ( Temp, "DBB1_3"); if (Temp) DBB1|=(1<<3);
187 Skip_S1(3, "Zero");
188
189 Get_S1 (4, Seconds_Units, "Seconds (Units)");
190 Get_SB ( Temp, "DBB1_4"); if (Temp) DBB1|=(1<<4);
191 Skip_S1(3, "Zero");
192
193 Skip_S1(4, "BG3");
194 Get_SB ( Temp, "DBB1_5"); if (Temp) DBB1|=(1<<5);
195 Skip_S1(3, "Zero");
196
197 Skip_SB( "FM - Frame Mark");
198 Get_S1 (3, Seconds_Tens, "Seconds (Tens)");
199 Get_SB ( Temp, "DBB1_6"); if (Temp) DBB1|=(1<<6);
200 Skip_S1(3, "Zero");
201
202 Skip_S1(4, "BG4");
203 Get_SB ( Temp, "DBB1_7"); if (Temp) DBB1|=(1<<7);
204 Skip_S1(3, "Zero");
205
206 Get_S1 (4, Minutes_Units, "Minutes (Units)");
207 Get_SB ( Temp, "DBB2_0"); if (Temp) DBB2|=(1<<0);
208 Skip_S1(3, "Zero");
209
210 Skip_S1(4, "BG5");
211 Get_SB ( Temp, "DBB2_1"); if (Temp) DBB2|=(1<<1);
212 Skip_S1(3, "Zero");
213
214 Skip_SB( "BGF0");
215 Get_S1 (3, Minutes_Tens, "Minutes (Tens)");
216 Get_SB ( Temp, "DBB2_2"); if (Temp) DBB2|=(1<<2);
217 Skip_S1(3, "Zero");
218
219 Skip_S1(4, "BG6");
220 Get_SB ( Temp, "DBB2_3"); if (Temp) DBB2|=(1<<3);
221 Skip_S1(3, "Zero");
222
223 Get_S1 (4, Hours_Units, "Hours (Units)");
224 Get_SB ( Temp, "DBB2_4"); if (Temp) DBB2|=(1<<4);
225 Skip_S1(3, "Zero");
226
227 Skip_S1(4, "BG7");
228 Get_SB ( Temp, "DBB2_5"); if (Temp) DBB2|=(1<<5);
229 Skip_S1(3, "Zero");
230
231 Skip_SB( "BGF2");
232 Skip_SB( "BGF1");
233 Get_S1 (2, Hours_Tens, "Hours (Tens)");
234 Get_SB ( Temp, "DBB2_6"); if (Temp) DBB2|=(1<<6);
235 Skip_S1(3, "Zero");
236
237 Skip_S1(4, "BG8");
238 Get_SB ( Temp, "DBB2_7"); if (Temp) DBB2|=(1<<7);
239 Skip_S1(3, "Zero");
240
241 BS_End();
242 }
243 else
244 {
245 BS_Begin();
246
247 Skip_S1(4, "BG1");
248 Get_S1 (4, Frames_Units, "Frames (Units)");
249
250 Skip_S1(4, "BG2");
251 Skip_SB( "CF - Color fame");
252 Get_SB ( DropFrame, "DP - Drop frame");
253 Get_S1 (2, Frames_Tens, "Frames (Tens)");
254
255 Skip_S1(4, "BG3");
256 Get_S1 (4, Seconds_Units, "Seconds (Units)");
257
258 Skip_S1(4, "BG4");
259 Skip_SB( "FM - Frame Mark");
260 Get_S1 (3, Seconds_Tens, "Seconds (Tens)");
261
262 Skip_S1(4, "BG5");
263 Get_S1 (4, Minutes_Units, "Minutes (Units)");
264
265 Skip_S1(4, "BG6");
266 Skip_SB( "BGF0");
267 Get_S1 (3, Minutes_Tens, "Minutes (Tens)");
268
269 Skip_S1(4, "BG7");
270 Get_S1 (4, Hours_Units, "Hours (Units)");
271
272 Skip_S1(4, "BG8");
273 Skip_SB( "BGF2");
274 Skip_SB( "BGF1");
275 Get_S1 (2, Hours_Tens, "Hours (Tens)");
276
277 BS_End();
278 }
279
280 int64u TimeCode_Ms=(int64u)(Hours_Tens *10*60*60*1000
281 + Hours_Units *60*60*1000
282 + Minutes_Tens *10*60*1000
283 + Minutes_Units *60*1000
284 + Seconds_Tens *10*1000
285 + Seconds_Units *1000
286 + (Gxf_FrameRate(FrameRate_Code)==0?0:((Frames_Tens*10+Frames_Units)*1000/float64_int32s(Gxf_FrameRate(FrameRate_Code)/(Gxf_FrameRate(FrameRate_Code)>30?2:1)))));
287
288 if (TimeCode_FirstFrame.empty())
289 {
290 TimeCode_FirstFrame+=('0'+Hours_Tens);
291 TimeCode_FirstFrame+=('0'+Hours_Units);
292 TimeCode_FirstFrame+=':';
293 TimeCode_FirstFrame+=('0'+Minutes_Tens);
294 TimeCode_FirstFrame+=('0'+Minutes_Units);
295 TimeCode_FirstFrame+=':';
296 TimeCode_FirstFrame+=('0'+Seconds_Tens);
297 TimeCode_FirstFrame+=('0'+Seconds_Units);
298 TimeCode_FirstFrame+=DropFrame?';':':';
299 TimeCode_FirstFrame+=('0'+Frames_Tens);
300 TimeCode_FirstFrame+=('0'+Frames_Units);
301 }
302
303 #if MEDIAINFO_TRACE
304 string TimeCode;
305 TimeCode+=('0'+Hours_Tens);
306 TimeCode+=('0'+Hours_Units);
307 TimeCode+=':';
308 TimeCode+=('0'+Minutes_Tens);
309 TimeCode+=('0'+Minutes_Units);
310 TimeCode+=':';
311 TimeCode+=('0'+Seconds_Tens);
312 TimeCode+=('0'+Seconds_Units);
313 TimeCode+=DropFrame?';':':';
314 TimeCode+=('0'+Frames_Tens);
315 TimeCode+=('0'+Frames_Units);
316 Element_Info1(TimeCode.c_str());
317 #endif //MEDIAINFO_TRACE
318 if (IsAtc)
319 {
320 Settings=Atc_PayloadType(DBB1);
321 Element_Info1(__T("PayloadType=")+Ztring().From_UTF8(Settings.c_str()));
322 Element_Info1(__T("VitcLineSelect=")+Ztring::ToZtring(DBB2&0x1F));
323 }
324 Element_End0();
325
326 FILLING_BEGIN();
327 if (TimeCode_FirstFrame_ms==(int64u)-1)
328 TimeCode_FirstFrame_ms=TimeCode_Ms;
329 FILLING_END();
330 }
331 else
332 Skip_XX(8, "Junk");
333 }
334
335 //bitmap, already parsed
336 Element_Offset+=64;
337
338 FILLING_BEGIN();
339 if (!Status[IsFilled] && TimeCode_FirstFrame_ms!=(int64u)-1)
340 {
341 Accept();
342 Fill();
343
344 if (MediaInfoLib::Config.ParseSpeed_Get()<1)
345 Finish();
346 }
347
348 FILLING_END();
349}
