153//---------------------------------------------------------------------------
154void File_Riff::Streams_Finish ()
155{
156 //Ancillary specific
157 #if defined(MEDIAINFO_ANCILLARY_YES)
158 if (Ancillary && (*Ancillary))
159 {
160 Clear();
161 Stream_Prepare(Stream_General);
162 Finish(*Ancillary);
163 Merge(**Ancillary);
164 Ztring LawRating=(*Ancillary)->Retrieve(Stream_General, 0, General_LawRating);
165 if (!LawRating.empty())
166 Fill(Stream_General, 0, General_LawRating, LawRating, true);
167 Ztring Title=(*Ancillary)->Retrieve(Stream_General, 0, General_Title);
168 if (!Title.empty() && Retrieve(Stream_General, 0, General_Title).empty())
169 Fill(Stream_General, 0, General_Title, Title);
170 return;
171 }
172 #endif //defined(MEDIAINFO_ANCILLARY_YES)
173
174 //Global
175 if (IsRIFF64)
176 Fill(Stream_General, 0, General_Format_Profile, "RF64");
177
178 //Time codes
179 TimeCode_Fill(__T("ISMP"), INFO_ISMP);
180 TimeCode_Fill(__T("Adobe tc_A"), Tdat_tc_A);
181 TimeCode_Fill(__T("Adobe tc_O"), Tdat_tc_O);
182
183 //For each stream
184 std::map<int32u, stream>::iterator Temp=Stream.begin();
185 while (Temp!=Stream.end())
186 {
187 //Preparing
188 StreamKind_Last=Temp->second.StreamKind;
189 StreamPos_Last=Temp->second.StreamPos;
190
191 //StreamSize
192 if (Temp->second.StreamSize>0)
193 Fill(StreamKind_Last, StreamPos_Last, Fill_Parameter(StreamKind_Last, Generic_StreamSize), Temp->second.StreamSize);
194
195 //When there are few frames, difficult to detect PCM
196 if (Temp->second.IsPcm && !Temp->second.Parsers.empty() && !Temp->second.Parsers[0]->Status[IsAccepted])
197 {
198 for (size_t Pos=0; Pos<Temp->second.Parsers.size()-1; Pos++)
199 delete Temp->second.Parsers[Pos];
200 Temp->second.Parsers.erase(Temp->second.Parsers.begin(), Temp->second.Parsers.begin()+Temp->second.Parsers.size()-1);
201 Temp->second.Parsers[0]->Accept();
202 }
203
204 //PAR
205 if (PAR && StreamKind_Last==Stream_Video)
206 Fill(Stream_Video, StreamPos_Last, Video_PixelAspectRatio, PAR);
207
208 Ztring ID;
209 if (Temp->first!=(int32u)-1)
210 ID.From_Number(((Temp->first>>24)-'0')*10+(((Temp->first>>16)&0xFF)-'0'));
211
212 //Parser specific
213 if (Temp->second.Parsers.size()==1)
214 {
215 //Finalizing and Merging (except Video codec and 120 fps hack)
216 Temp->second.Parsers[0]->ShouldContinueParsing=false;
217
218 //Hack - Before
219 Ztring StreamSize, Codec_Temp;
220 if (StreamKind_Last==Stream_Video)
221 Codec_Temp=Retrieve(Stream_Video, StreamPos_Last, Video_Codec); //We want to keep the 4CC of AVI
222 StreamSize=Retrieve(StreamKind_Last, StreamPos_Last, Fill_Parameter(StreamKind_Last, Generic_StreamSize)); //We want to keep the 4CC of AVI
223
224 //Merging
225 if (Config->ParseSpeed<=1.0)
226 {
227 Fill(Temp->second.Parsers[0]);
228 Temp->second.Parsers[0]->Open_Buffer_Unsynch();
229 }
230 Finish(Temp->second.Parsers[0]);
231
232 if (!Temp->second.Parsers.empty() && Temp->second.Parsers[0]->Count_Get(StreamKind_Last))
233 {
234 //Special case: Compressed audio hidden in PCM
235 if (StreamKind_Last==Stream_Audio
236 && Temp->second.Compression==1
237 && Retrieve(Stream_General, 0, General_Format)==__T("Wave")
238 && Temp->second.Parsers[0]->Get(Stream_Audio, 0, Audio_Format)!=__T("PCM")) //Some DTS or SMPTE ST 337 streams are coded "1"
239 Clear(Stream_Audio, 0, Audio_Channel_s_);
240
241 size_t StreamPos_Base=StreamPos_Last;
242 for (size_t Pos=0; Pos<Temp->second.Parsers[0]->Count_Get(StreamKind_Last); Pos++)
243 {
244 Ztring Temp_ID=ID;
245 Ztring Temp_ID_String=ID;
246 Merge(*Temp->second.Parsers[0], StreamKind_Last, Pos, StreamPos_Base+Pos);
247 if (!Retrieve(StreamKind_Last, StreamPos_Last, General_ID).empty())
248 {
249 if (!Temp_ID.empty())
250 {
251 Temp_ID+=__T('-');
252 Temp_ID_String+=__T('-');
253 }
254 Temp_ID+=Retrieve(StreamKind_Last, StreamPos_Last, General_ID);
255 Temp_ID_String+=Retrieve(StreamKind_Last, StreamPos_Last, General_ID);
256 }
257 Fill(StreamKind_Last, StreamPos_Last, General_ID, Temp_ID, true);
258 Fill(StreamKind_Last, StreamPos_Last, General_StreamOrder, Temp_ID_String, true);
259
260
261 //Special case - MPEG Video + Captions
262 if (StreamKind_Last==Stream_Video && Temp->second.Parsers[0]->Count_Get(Stream_Text))
263 {
264 //Video and Text are together
265 size_t Parser_Text_Count=Temp->second.Parsers[0]->Count_Get(Stream_Text);
266 for (size_t Parser_Text_Pos=0; Parser_Text_Pos<Parser_Text_Count; Parser_Text_Pos++)
267 {
268 size_t StreamPos_Video=StreamPos_Last;
269 Fill_Flush();
270 Stream_Prepare(Stream_Text);
271 Temp->second.Parsers[0]->Finish();
272 Merge(*Temp->second.Parsers[0], Stream_Text, Parser_Text_Pos, StreamPos_Last);
273 Fill(Stream_Text, StreamPos_Last, Text_Duration, Retrieve(Stream_Video, StreamPos_Video, Video_Duration));
274 Ztring ID=Retrieve(Stream_Text, StreamPos_Last, Text_ID);
275 Fill(Stream_Text, StreamPos_Last, Text_ID, Retrieve(Stream_Video, Count_Get(Stream_Video)-1, Video_ID)+__T("-")+ID, true);
276 Fill(Stream_Text, StreamPos_Last, Text_ID_String, Retrieve(Stream_Video, Count_Get(Stream_Video)-1, Video_ID_String)+__T("-")+ID, true);
277 Fill(Stream_Text, StreamPos_Last, Text_Title, Retrieve(Stream_Video, Count_Get(Stream_Video)-1, Video_Title), true);
278 Fill(Stream_Text, StreamPos_Last, "MuxingMode_MoreInfo", __T("Muxed in Video #")+Ztring().From_Number(Count_Get(Stream_Video)), true);
279 }
280
281 StreamKind_Last=Stream_Video;
282 StreamPos_Last=Count_Get(Stream_Video)-1;
283 }
284 }
285 }
286 else
287 {
288 Fill(StreamKind_Last, StreamPos_Last, General_ID, ID, true);
289 Fill(StreamKind_Last, StreamPos_Last, General_StreamOrder, ID, true);
290 }
291
292 //Hacks - After
293 Fill(StreamKind_Last, StreamPos_Last, Fill_Parameter(StreamKind_Last, Generic_StreamSize), StreamSize, true);
294 if (StreamKind_Last==Stream_Video)
295 {
296 if (!Codec_Temp.empty())
297 Fill(Stream_Video, StreamPos_Last, Video_Codec, Codec_Temp, true);
298
299 //120 fps hack
300 const Ztring &FrameRate=Retrieve(Stream_Video, StreamPos_Last, Video_FrameRate);
301 if (FrameRate.To_int32u()==120)
302 {
303 float32 FrameRateF=FrameRate.To_float32();
304 Fill(Stream_Video, StreamPos_Last, Video_FrameRate_Minimum, FrameRateF/5, 3, true);
305 Fill(Stream_Video, StreamPos_Last, Video_FrameRate_Maximum, FrameRateF/4, 3, true);
306 Fill(Stream_Video, StreamPos_Last, Video_FrameRate_Mode, "VFR");
307 }
308 }
309
310 //Hack - SMPTE ST 337, RIFF channels count is wrong
311 if (StreamKind_Last==Stream_Audio && Retrieve(Stream_Audio, StreamPos_Last, Audio_MuxingMode)==__T("SMPTE ST 337") && Temp->second.Parsers[0]->Get(Stream_Audio, 0, Audio_Channel_s_).empty())
312 Clear(Stream_Audio, StreamPos_Last, Audio_Channel_s_);
313
314 //Alignment
315 if (StreamKind_Last==Stream_Audio && Count_Get(Stream_Video)>0) //Only if this is not a WAV file
316 {
317 Fill(Stream_Audio, StreamPos_Last, Audio_Alignment, Temp->second.ChunksAreComplete?"Aligned":"Split");
318 Fill(Stream_Audio, StreamPos_Last, Audio_Alignment_String, MediaInfoLib::Config.Language_Get(Temp->second.ChunksAreComplete?__T("Alignment_Aligned"):__T("Alignment_Split")));
319 }
320
321 //Delay
322 if (StreamKind_Last==Stream_Audio && Count_Get(Stream_Video)==1 && Temp->second.Rate!=0 && Temp->second.Parsers[0]->Status[IsAccepted])
323 {
324 float Delay=0;
325 bool Delay_IsValid=false;
326
327 if (Temp->second.Parsers[0]->Buffer_TotalBytes_FirstSynched==0)
328 {
329 Delay=0;
330 Delay_IsValid=true;
331 }
332 else if (Temp->second.Rate!=0)
333 {
334 Delay=((float)Temp->second.Parsers[0]->Buffer_TotalBytes_FirstSynched)*1000/Temp->second.Rate;
335 Delay_IsValid=true;
336 }
337 else if (Temp->second.Parsers[0]->Retrieve(Stream_Audio, 0, Audio_BitRate).To_int64u()!=0)
338 {
339 Delay=((float)Temp->second.Parsers[0]->Buffer_TotalBytes_FirstSynched)*1000/Temp->second.Parsers[0]->Retrieve(Stream_Audio, 0, Audio_BitRate).To_int64u();
340 Delay_IsValid=true;
341 }
342 else if (Temp->second.Parsers[0]->Retrieve(Stream_Audio, 0, Audio_BitRate_Nominal).To_int64u()!=0)
343 {
344 Delay=((float)Temp->second.Parsers[0]->Buffer_TotalBytes_FirstSynched)*1000/Temp->second.Parsers[0]->Retrieve(Stream_Audio, 0, Audio_BitRate_Nominal).To_int64u();
345 Delay_IsValid=true;
346 }
347
348 if (Delay_IsValid)
349 {
350 Delay+=((float)Temp->second.Start)*1000/Temp->second.Rate;
351 Fill(Stream_Audio, StreamPos_Last, Audio_Delay, Delay, 0, true);
352 Fill(Stream_Audio, StreamPos_Last, Audio_Delay_Source, "Stream", Unlimited, true, true);
353 for (size_t StreamPos=0; StreamPos<Count_Get(Stream_Video); StreamPos++)
354 Fill(Stream_Video, StreamPos, Video_Delay, 0, 10, true);
355 }
356 }
357
358 //Special case: AAC
359 if (StreamKind_Last==Stream_Audio
360 && (Retrieve(Stream_Audio, StreamPos_Last, Audio_Format)==__T("AAC")
361 || Retrieve(Stream_Audio, StreamPos_Last, Audio_Format)==__T("MPEG Audio")
362 || Retrieve(Stream_Audio, StreamPos_Last, Audio_Format)==__T("Vorbis")))
363 Clear(Stream_Audio, StreamPos_Last, Audio_BitDepth); //Resolution is not valid for AAC / MPEG Audio / Vorbis
364
365 //Format specific
366 #if defined(MEDIAINFO_DVDIF_YES)
367 if (StreamKind_Last==Stream_Video && (MediaInfoLib::Config.Codec_Get(Ztring().From_CC4(Temp->second.Compression), InfoCodec_KindofCodec).find(__T("DV"))==0
368 || Retrieve(Stream_Video, StreamPos_Last, Video_Format)==__T("DV")
369 || Retrieve(Stream_Video, StreamPos_Last, Video_Codec)==__T("DV")))
370 {
371 if (Retrieve(Stream_General, 0, General_Recorded_Date).empty())
372 Fill(Stream_General, 0, General_Recorded_Date, Temp->second.Parsers[0]->Retrieve(Stream_General, 0, General_Recorded_Date));
373
374 //Video and Audio are together
375 size_t Audio_Count=Temp->second.Parsers[0]->Count_Get(Stream_Audio);
376 for (size_t Audio_Pos=0; Audio_Pos<Audio_Count; Audio_Pos++)
377 {
378 Fill_Flush();
379 Stream_Prepare(Stream_Audio);
380 size_t Pos=Count_Get(Stream_Audio)-1;
381 Merge(*Temp->second.Parsers[0], Stream_Audio, Audio_Pos, StreamPos_Last);
382 Fill(Stream_Audio, Pos, Audio_MuxingMode, "DV");
383 Fill(Stream_Audio, Pos, Audio_Duration, Retrieve(Stream_Video, Temp->second.StreamPos, Video_Duration));
384 Fill(Stream_Audio, Pos, "MuxingMode_MoreInfo", __T("Muxed in Video #")+Ztring().From_Number(Temp->second.StreamPos+1));
385 Fill(Stream_Audio, Pos, Audio_StreamSize_Encoded, 0); //Included in the DV stream size
386 Ztring ID=Retrieve(Stream_Audio, Pos, Audio_ID);
387 Fill(Stream_Audio, Pos, Audio_ID, Retrieve(Stream_Video, Temp->second.StreamPos, Video_ID)+__T("-")+ID, true);
388 }
389
390 StreamKind_Last=Stream_Video;
391 StreamPos_Last=Temp->second.StreamPos;
392 }
393 #endif
394 }
395 else if (StreamKind_Last!=Stream_General)
396 Fill(StreamKind_Last, StreamPos_Last, General_ID, ID);
397
398 //Duration
399 if (Temp->second.PacketCount>0)
400 {
401 if (StreamKind_Last==Stream_Video) // && Retrieve(Stream_Video, StreamPos_Last, Video_Duration).empty())
402 {
403 //Duration in case it is missing from header (malformed header...)
404 if (Temp->second.indx_Duration && Temp->second.Rate)
405 Fill(Stream_Video, StreamPos_Last, Video_Duration, ((float64)Temp->second.indx_Duration)*1000*Temp->second.Scale/Temp->second.Rate, 0, true);
406 else
407 Fill(Stream_Video, StreamPos_Last, Video_FrameCount, Temp->second.PacketCount, 10, true);
408 }
409 if (StreamKind_Last==Stream_Audio)
410 {
411 //Duration in case it is missing from header (malformed header...)
412 int64u SamplingCount=0;
413 #if defined(MEDIAINFO_MPEGA_YES)
414 if (Retrieve(Stream_Audio, StreamPos_Last, Audio_Format)==__T("MPEG Audio"))
415 {
416 if (Temp->second.Parsers[0] && Temp->second.PacketPos==((File_Mpega*)Temp->second.Parsers[0])->Frame_Count_Valid) //Only for stream with one frame per chunk
417 {
418 Ztring Version=Retrieve(Stream_Audio, StreamPos_Last, Audio_Format_Version);
419 Ztring Layer=Retrieve(Stream_Audio, StreamPos_Last, Audio_Format_Profile);
420 if (Version==__T("Version 1") && Layer==__T("Layer 1"))
421 SamplingCount = (int64u)Temp->second.PacketCount * 384; //MPEG-1 Layer 1
422 else if ((Version==__T("Version 2") || Version==__T("Version 2.5")) && Layer==__T("Layer 1"))
423 SamplingCount = (int64u)Temp->second.PacketCount * 192; //MPEG-2 or 2.5 Layer 1
424 else if ((Version==__T("Version 2") || Version==__T("Version 2.5")) && Layer==__T("Layer 3"))
425 SamplingCount = (int64u)Temp->second.PacketCount * 576; //MPEG-2 or 2.5 Layer 3
426 else
427 SamplingCount = (int64u)Temp->second.PacketCount * 1152;
428 }
429 }
430 #endif
431 if (Retrieve(Stream_Audio, StreamPos_Last, Audio_Format)==__T("PCM"))
432 {
433 int64u Resolution=Retrieve(Stream_Audio, StreamPos_Last, Audio_BitDepth).To_int64u();
434 int64u Channels=Retrieve(Stream_Audio, StreamPos_Last, Audio_Channel_s_).To_int64u();
435 if (Resolution>0 && Channels>0)
436 SamplingCount=Temp->second.StreamSize*8/Resolution/Channels;
437 }
438 if (Retrieve(Stream_Audio, StreamPos_Last, Audio_Format)==__T("ADPCM"))
439 {
440 int64u Resolution=Retrieve(Stream_Audio, StreamPos_Last, Audio_BitDepth).To_int64u();
441 int64u Channels=Retrieve(Stream_Audio, StreamPos_Last, Audio_Channel_s_).To_int64u();
442 if (Resolution>0 && Channels>0)
443 SamplingCount=(int64u)(Temp->second.StreamSize*8/Resolution/Channels*0.98); //0.98 is not precise!
444
445 //ADPCM estimation is not precise, if container sampling count is around our value, using it rather than the estimation
446 float32 SamplingRate=Retrieve(Stream_Audio, StreamPos_Last, Audio_SamplingRate).To_float32();
447 if (SamplingRate>0
448 && SamplingCount*1000/SamplingRate<((float32)avih_TotalFrame)/avih_FrameRate*1000*1.10
449 && SamplingCount*1000/SamplingRate>((float32)avih_TotalFrame)/avih_FrameRate*1000*0.10)
450 SamplingCount=0; //Value disabled
451 }
452 //One AC-3 frame is 32 ms
453 //One DTS frame is 21 ms
454
455 float32 SamplingRate=Retrieve(Stream_Audio, StreamPos_Last, Audio_SamplingRate).To_float32();
456 if (SamplingCount>0 && SamplingRate>0)
457 Fill(Stream_Audio, StreamPos_Last, Audio_Duration, SamplingCount*1000/SamplingRate, 0, true);
458 else if (Temp->second.indx_Duration && Temp->second.Rate)
459 Fill(Stream_Audio, StreamPos_Last, Audio_Duration, ((float64)Temp->second.indx_Duration)*1000*Temp->second.Scale/Temp->second.Rate, 0, true);
460 else if (Temp->second.Rate && Temp->second.Scale!=1) //Note: some files with Scale==1 are buggy
461 Fill(Stream_Audio, StreamPos_Last, Audio_Duration, ((float64)Temp->second.Length)*1000*Temp->second.Scale/Temp->second.Rate, 0, true);
462
463 //Interleave
464 if (Stream[0x30300000].PacketCount && Temp->second.PacketCount)
465 {
466 Fill(Stream_Audio, StreamPos_Last, "Interleave_VideoFrames", (float)Stream[0x30300000].PacketCount/Temp->second.PacketCount, 2);
467 if (Retrieve(Stream_Video, 0, Video_FrameRate).To_float32())
468 {
469 Fill(Stream_Audio, StreamPos_Last, "Interleave_Duration", (float)Stream[0x30300000].PacketCount/Temp->second.PacketCount*1000/Retrieve(Stream_Video, 0, Video_FrameRate).To_float32(), 0);
470 Ztring Interleave_Duration_String;
471 Interleave_Duration_String+=Retrieve(Stream_Audio, StreamPos_Last, "Interleave_Duration");
472 Interleave_Duration_String+=__T(" ");
473 Interleave_Duration_String+=MediaInfoLib::Config.Language_Get(__T("ms"));
474 if (!Retrieve(Stream_Audio, StreamPos_Last, "Interleave_VideoFrames").empty())
475 {
476 Interleave_Duration_String+=__T(" (");
477 Interleave_Duration_String+=MediaInfoLib::Config.Language_Get(Retrieve(Stream_Audio, StreamPos_Last, "Interleave_VideoFrames"), __T(" video frames"));
478 Interleave_Duration_String+=__T(")");
479 }
480 Fill(Stream_Audio, StreamPos_Last, "Interleave_Duration/String", Interleave_Duration_String);
481 }
482 int64u Audio_FirstBytes=0;
483 for (std::map<int64u, stream_structure>::iterator Stream_Structure_Temp=Stream_Structure.begin(); Stream_Structure_Temp!=Stream_Structure.end(); ++Stream_Structure_Temp)
484 {
485 if (Stream_Structure_Temp->second.Name==0x30300000)
486 break;
487 if (Stream_Structure_Temp->second.Name==Temp->first)
488 Audio_FirstBytes+=Stream_Structure_Temp->second.Size;
489 }
490 if (Audio_FirstBytes && Temp->second.AvgBytesPerSec)
491 {
492 Fill(Stream_Audio, StreamPos_Last, "Interleave_Preload", Audio_FirstBytes*1000/Temp->second.AvgBytesPerSec);
493 Fill(Stream_Audio, StreamPos_Last, "Interleave_Preload/String", Retrieve(Stream_Audio, StreamPos_Last, "Interleave_Preload")+__T(" ")+MediaInfoLib::Config.Language_Get(__T("ms")));
494 }
495 }
496 }
497
498 //Source duration
499 if (Temp->second.PacketCount && Temp->second.Length!=Temp->second.PacketCount)
500 {
501 if (StreamKind_Last==Stream_Video && Temp->second.Rate)
502 Fill(Stream_Video, StreamPos_Last, "Source_Duration", ((float64)Temp->second.PacketCount)*1000*Temp->second.Scale/Temp->second.Rate, 0);
503 if (StreamKind_Last==Stream_Audio && Temp->second.Rate)
504 {
505 float64 Duration_Source=((float64)Temp->second.StreamSize)*1000/Temp->second.AvgBytesPerSec;
506 float64 Duration_Header=Retrieve(Stream_Audio, StreamPos_Last, Audio_Duration).To_float64();
507 float64 Difference=Duration_Source-Duration_Header;
508 if (Temp->second.Scale!=1 && float64_int64s(Duration_Header/Duration_Source)==Temp->second.Scale)
509 Fill(Stream_Audio, StreamPos_Last, Audio_Duration, Duration_Source, 0, true); //Found 1 stream with Scale not being right
510 else if (Difference<-2 || Difference>2) //+/- 2 ms
511 Fill(Stream_Audio, StreamPos_Last, "Source_Duration", Duration_Source, 0);
512 }
513 }
514 }
515
516 ++Temp;
517 }
518
519 //Some work on the first video stream
520 if (Count_Get(Stream_Video))
521 {
522 //ODML
523 if (dmlh_TotalFrame!=0 && Retrieve(Stream_Video, 0, Video_Duration).empty())
524 for (size_t StreamPos=0; StreamPos<Count_Get(Stream_Video); StreamPos++)
525 Fill(Stream_Video, StreamPos, Video_FrameCount, dmlh_TotalFrame, 10, true);
526 }
527
528 //Rec
529 if (rec__Present)
530 Fill(Stream_General, 0, General_Format_Settings, "rec");
531
532 //Interleaved
533 if (Interleaved0_1 && Interleaved0_10 && Interleaved1_1 && Interleaved1_10)
534 Fill(Stream_General, 0, General_Interleaved, ((Interleaved0_1<Interleaved1_1 && Interleaved0_10>Interleaved1_1)
535 || (Interleaved1_1<Interleaved0_1 && Interleaved1_10>Interleaved0_1))?"Yes":"No");
536
537 //MD5
538 size_t Pos=0;
539 for (size_t StreamKind=Stream_General+1; StreamKind<Stream_Max; StreamKind++)
540 for (size_t StreamPos=0; StreamPos<Count_Get((stream_t)StreamKind); StreamPos++)
541 if (Pos<MD5s.size())
542 {
543 Fill((stream_t)StreamKind, StreamPos, "MD5", MD5s[Pos]);
544 Pos++;
545 }
546
547 //Commercial names
548 if (Count_Get(Stream_Video)==1)
549 {
550 Streams_Finish_StreamOnly();
551 if (!Retrieve(Stream_Video, 0, Video_Format_Commercial_IfAny).empty())
552 {
553 Fill(Stream_General, 0, General_Format_Commercial_IfAny, Retrieve(Stream_Video, 0, Video_Format_Commercial_IfAny));
554 Fill(Stream_General, 0, General_Format_Commercial, __T("AVI ")+Retrieve(Stream_Video, 0, Video_Format_Commercial_IfAny));
555 }
556 else if (Retrieve(Stream_Video, 0, Video_Format)==__T("DV"))
557 {
558 Fill(Stream_General, 0, General_Format_Commercial_IfAny, "DV");
559 Fill(Stream_General, 0, General_Format_Commercial, "AVI DV");
560 }
561 else if (Retrieve(Stream_Video, 0, Video_Format)==__T("MPEG Video") && Retrieve(Stream_Video, 0, Video_Format_Settings_GOP)==__T("N=1") && Retrieve(Stream_Video, 0, Video_Colorimetry)==__T("4:2:2") && Retrieve(Stream_Video, 0, Video_BitRate)==__T("30000000"))
562 {
563 Fill(Stream_General, 0, General_Format_Commercial_IfAny, "MPEG IMX 30");
564 Fill(Stream_Video, 0, Video_Format_Commercial_IfAny, "MPEG IMX 30");
565 }
566 else if (Retrieve(Stream_Video, 0, Video_Format)==__T("MPEG Video") && Retrieve(Stream_Video, 0, Video_Format_Settings_GOP)==__T("N=1") && Retrieve(Stream_Video, 0, Video_Colorimetry)==__T("4:2:2") && Retrieve(Stream_Video, 0, Video_BitRate)==__T("40000000"))
567 {
568 Fill(Stream_General, 0, General_Format_Commercial_IfAny, "MPEG IMX 40");
569 Fill(Stream_Video, 0, Video_Format_Commercial_IfAny, "MPEG IMX 40");
570 }
571 else if (Retrieve(Stream_Video, 0, Video_Format)==__T("MPEG Video") && Retrieve(Stream_Video, 0, Video_Format_Settings_GOP)==__T("N=1") && Retrieve(Stream_Video, 0, Video_Colorimetry)==__T("4:2:2") && Retrieve(Stream_Video, 0, Video_BitRate)==__T("50000000"))
572 {
573 Fill(Stream_General, 0, General_Format_Commercial_IfAny, "MPEG IMX 50");
574 Fill(Stream_Video, 0, Video_Format_Commercial_IfAny, "MPEG IMX 50");
575 }

598size_t File_Riff::Read_Buffer_Seek (size_t Method, int64u Value, int64u /*ID*/)
599{
600 //Only Wave and AIFF
601 switch (Kind)
602 {
603 case Kind_Wave :
604 case Kind_Aiff :
605 break;
606 default : return (size_t)-1;
607 }
608
609 //Parsing
610 switch (Method)
611 {
612 case 0 :
613 if (Value<Buffer_DataToParse_Begin)
614 Value=Buffer_DataToParse_Begin;
615 if (Value>Buffer_DataToParse_End)
616 Value=Buffer_DataToParse_End;
617 GoTo(Value);
618 Open_Buffer_Unsynch();
619 return 1;
620 case 1 :
621 GoTo(Buffer_DataToParse_Begin+(Buffer_DataToParse_End-Buffer_DataToParse_Begin)*Value/10000);
622 Open_Buffer_Unsynch();
623 return 1;
624 case 2 : //Timestamp
625 {
626 if (AvgBytesPerSec==0)
627 return (size_t)-1;
628
629 float64 ValueF=(float64)Value;
630 ValueF/=1000000000; //Value is in ns
631 ValueF*=AvgBytesPerSec;
632 GoTo(Buffer_DataToParse_Begin+float64_int64s(ValueF));
633 return 1;
634 }
635 case 3 : //FrameNumber
636 {
637 if (AvgBytesPerSec==0 || Demux_Rate==0 || BlockAlign==0)
638 return (size_t)-1;
639
640 float64 BytesPerFrame=AvgBytesPerSec/Demux_Rate;
641 int64u StreamOffset=(int64u)(Value*BytesPerFrame);
642 StreamOffset/=BlockAlign;
643 StreamOffset*=BlockAlign;
644
645 GoTo(Buffer_DataToParse_Begin+StreamOffset);
646 return 1;
647 }
648 default : return (size_t)-1; //Not supported
649 }
650}

691//---------------------------------------------------------------------------
692bool File_Riff::Header_Begin()
693{
694 while (File_Offset+Buffer_Offset<Buffer_DataToParse_End)
695 {
696 #if MEDIAINFO_DEMUX
697 if (AvgBytesPerSec && Demux_Rate && BlockAlign)
698 {
699 float64 BytesPerFrame=((float64)AvgBytesPerSec)/Demux_Rate;
700 Frame_Count_NotParsedIncluded=float64_int64s(((float64)(File_Offset+Buffer_Offset-Buffer_DataToParse_Begin))/BytesPerFrame);
701 Element_Size=float64_int64s(BytesPerFrame*(Frame_Count_NotParsedIncluded+1));
702 Element_Size/=BlockAlign;
703 Element_Size*=BlockAlign;
704 Element_Size-=File_Offset+Buffer_Offset-Buffer_DataToParse_Begin;
705 FrameInfo.PTS=FrameInfo.DTS=float64_int64s(((float64)Frame_Count_NotParsedIncluded)*1000000000/Demux_Rate);
706 while (Element_Size && File_Offset+Buffer_Offset+Element_Size>Buffer_DataToParse_End)
707 Element_Size-=BlockAlign;
708 if (Element_Size==0)
709 Element_Size=BlockAlign;
710 if (Buffer_Offset+Element_Size>Buffer_Size)
711 return false;
712 }
713 else
714 #endif //MEDIAINFO_DEMUX
715 if (File_Offset+Buffer_Size<=Buffer_DataToParse_End)
716 Element_Size=Buffer_Size; //All the buffer is used
717 else
718 {
719 Element_Size=Buffer_DataToParse_End-(File_Offset+Buffer_Offset);
720 Buffer_DataToParse_End=0;
721 }
722
723 if (Buffer_Offset+(size_t)Element_Size>Buffer_Size)
724 return false;
725
726 // Fake header
727 Element_Begin1("...Continued"); //TODO: better method
728 Element_ThisIsAList();
729 Element_Begin1("...Continued");
730 Element_ThisIsAList();
731 if (Buffer_DataToParse_End)
732 Header_Fill_Size(Buffer_DataToParse_End-(File_Offset+Buffer_Offset));
733 else
734 Header_Fill_Size(Element_Size);
735 Element_End();
736
737 switch (Kind)
738 {
739 case Kind_Wave : WAVE_data_Continue(); break;
740 case Kind_Aiff : AIFF_SSND_Continue(); break;
741 case Kind_Rmp3 : RMP3_data_Continue(); break;
742 default : AVI__movi_xxxx();
743 }
744
745 if (Config->ParseSpeed<1.0 && File_Offset+Buffer_Offset+Element_Offset-Buffer_DataToParse_Begin>=256*1024)
746 {
747 File_GoTo=Buffer_DataToParse_End;
748 Buffer_Offset=Buffer_Size;
749 Element_Size=0;
750 }
751 else
752 {
753 Buffer_Offset+=(size_t)Element_Size;
754 Element_Size-=Element_Offset;
755 }
756 Element_Offset=0;
757 Element_End0();
758
759 if (Status[IsFinished] || (File_GoTo!=(int64u)-1 && (File_GoTo<=Buffer_DataToParse_Begin || File_GoTo>=Buffer_DataToParse_End)))
760 {
761 Buffer_DataToParse_Begin=(int64u)-1;
762 Buffer_DataToParse_End=0;
763 return false;
764 }
765
766 if (Buffer_Offset>=Buffer_Size)
767 return false;
768
769 #if MEDIAINFO_DEMUX
770 if (Config->Demux_EventWasSent)
771 return false;
772 #endif //MEDIAINFO_DEMUX
773 }
774
775 return true;
776}

778//---------------------------------------------------------------------------
779void File_Riff::Header_Parse()
780{
781 //Special case : W3DI tags (unknown format!) are at the end of the file
782 if (Element_Level==2 && File_Offset+Buffer_Size==File_Size && Buffer_Size>8)
783 {
784 if (CC4(Buffer+Buffer_Size-4)==Elements::W3DI)
785 {
786 int32u Size=LittleEndian2int32u(Buffer+Buffer_Size-8);
787 if (Size>8 && Size<=Buffer_Size && Buffer_Offset+Size==Buffer_Size)
788 {
789 //Filling
790 Header_Fill_Code(Elements::W3DI, "W3DI");
791 Header_Fill_Size(Size);
792 return;
793 }
794 }
795 }
796
797 //Special case : SMV file detected
798 if (SMV_BlockSize)
799 {
800 //Filling
801 Header_Fill_Code(Elements::SMV0_xxxx, "SMV Block");
802 Header_Fill_Size(SMV_BlockSize);
803 return;
804 }
805
806 //Parsing
807 int32u Size, Name;
808 Get_C4 (Name, "Name");
809 if (Name==Elements::SMV0)
810 {
811 //SMV specific
812 //Filling
813 Header_Fill_Code(Elements::SMV0, "SMV header");
814 Header_Fill_Size(51);
815 return;
816 }
817 if (Name==Elements::riff)
818 IsWave64=true;
819 if (IsWave64)
820 {
821 //Wave64 specific
822 int64u Size_Complete;
823 Skip_XX(12, "Name (GUID)");
824 Get_L8 (Size_Complete, "Size");
825
826 //Alignment
827 if (Name!=Elements::riff && Size_Complete%8)
828 {
829 Alignement_ExtraByte=Size_Complete%8;
830 Size_Complete+=Alignement_ExtraByte; //Always 8-byte aligned
831 }
832 else
833 Alignement_ExtraByte=0;
834
835 //Top level chunks
836 if (Name==Elements::riff)
837 {
838 Get_C4 (Name, "Real Name");
839 Skip_XX(12, "Real Name (GUID)");
840 }
841
842 //Special case: we don't need the full data
843 if (Name==Elements::WAVE_data)
844 {
845 Buffer_DataToParse_Begin=File_Offset+Buffer_Offset;
846 if (Size_Complete)
847 Buffer_DataToParse_End=File_Offset+Buffer_Offset+Size_Complete;
848 else
849 Buffer_DataToParse_End=File_Size; //Found one file with 0 as size of data part
850 Size_Complete=Element_Offset;
851 }
852
853 //Filling
854 Header_Fill_Code(Name, Ztring().From_CC4(Name));
855 Header_Fill_Size(Size_Complete);
856 return;
857 }
858 if (Name==Elements::FORM
859 || Name==Elements::MThd)
860 IsBigEndian=true; //Swap from Little to Big Endian for "FORM" files (AIFF...)
861 if (IsBigEndian)
862 Get_B4 (Size, "Size");
863 else
864 {
865 Get_L4 (Size, "Size");
866
867 //Testing malformed (not word aligned)
868 if (!IsNotWordAligned_Tested && Size%2)
869 {
870 if (File_Offset+Buffer_Offset+8+Size==File_Size)
871 IsNotWordAligned=true;
872 else if (!File_Name.empty())
873 {
874 File F(File_Name);
875 F.GoTo(File_Offset+Buffer_Offset+8+Size);
876 int8u Temp;
877 if (F.Read(&Temp, 1))
878 {
879 if (!((Temp<'A' || Temp>'z') && Temp!=' '))
880 IsNotWordAligned=true;
881 }
882 }
883 IsNotWordAligned_Tested=true;
884 }
885 }
886
887 //RF64
888 int64u Size_Complete=Size;
889 if (Size==0 && Name==Elements::RIFF)
890 Size_Complete=File_Size-8;
891 else if (Size==0xFFFFFFFF)
892 {
893 if (Element_Size<0x1C)
894 {
895 Element_WaitForMoreData();
896 return;
897 }
898 if (Name==Elements::RF64 && CC4(Buffer+Buffer_Offset+0x0C)==Elements::WAVE_ds64)
899 {
900 Size_Complete=LittleEndian2int64u(Buffer+Buffer_Offset+0x14);
901 Param_Info1(Size_Complete);
902 }
903 else if (Name==Elements::WAVE_data)
904 {
905 Size_Complete=WAVE_data_Size;
906 Param_Info1(Size_Complete);
907 }
908 }
909
910 //Coherency
911 if (Stream_Structure_Temp!=Stream_Structure.end() && Stream_Structure_Temp->second.Size==0)
912 {
913 Name=(int32u)-1;
914 Size_Complete=0; //Hack in some indexes with Size==0 (why?), ignoring content of header
915 }
916 if (File_Offset+Buffer_Offset+8+Size_Complete>File_Size)
917 Size_Complete=File_Size-(File_Offset+Buffer_Offset+8);
918
919 //Alignment
920 if (Size_Complete%2 && !IsNotWordAligned)
921 {
922 Size_Complete++; //Always 2-byte aligned
923 Alignement_ExtraByte=1;
924 }
925 else
926 Alignement_ExtraByte=0;
927
928 //Top level chunks
929 if (Name==Elements::LIST
930 || Name==Elements::RIFF
931 || Name==Elements::RF64
932 || Name==Elements::ON2_
933 || Name==Elements::FORM)
934 {
935 if (Name==Elements::RF64)
936 IsRIFF64=true;
937 Get_C4 (Name, "Real Name");
938 }
939
940 //Integrity
941 if (Name==0x00000000)
942 {
943 //Filling
944 Header_Fill_Code(0, "Junk");
945 Header_Fill_Size(File_Size-(File_Offset+Buffer_Offset));
946 Alignement_ExtraByte=0;
947 return;
948 }
949
950 //Specific
951 if (Name==Elements::ON2f)
952 Name=Elements::AVI_;
953
954 //Tests
955 if (Element_Level==2 && Name==Elements::WAVE && !IsRIFF64 && File_Size>0xFFFFFFFF)
956 IsWaveBroken=true; //Non standard big files detection
957 if (IsWaveBroken && (Name==Elements::WAVE || Name==Elements::WAVE_data))
958 Size_Complete=File_Size-(File_Offset+Buffer_Offset+8); //Non standard big files detection
959 if (movi_Size && Size_Complete>movi_Size/2 && 8+Size_Complete>1024*1024 && !((Name&0xFFFF0000)==0x69780000 || (Name&0x0000FFFF)==0x00006978) && Element_Level==(rec__Present?(size_t)5:(size_t)4) && Buffer_Offset+8+Size_Complete>Buffer_Size)
960 {
961 Buffer_DataToParse_End=File_Offset+Buffer_Offset+8+Size_Complete;
962 Size_Complete=Buffer_Size-(Buffer_Offset+8);
963 }
964 if ((Name==Elements::WAVE_data || Name==Elements::AIFF_SSND))
965 {
966 Buffer_DataToParse_Begin=File_Offset+Buffer_Offset+8;
967 if (Size_Complete)
968 Buffer_DataToParse_End=File_Offset+Buffer_Offset+8+Size_Complete;
969 else
970 Buffer_DataToParse_End=File_Size; //Found one file with 0 as size of data part
971 Size_Complete=0;
972 }
973
974 //Filling
975 Header_Fill_Code(Name, Ztring().From_CC4(Name));
976 Header_Fill_Size(Size_Complete+8);
977}

979//---------------------------------------------------------------------------
980bool File_Riff::BookMark_Needed()
981{
982 if (!movi_Size || SecondPass)
983 return false;
984
985 //Go to the first usefull chunk
986 if (stream_Count==0 && Stream_Structure.empty())
987 return false; //No need
988
989 Stream_Structure_Temp=Stream_Structure.begin();
990 if (!Stream_Structure.empty())

992 #if MEDIAINFO_HASH

995 GoTo(0);
996 Hash_ParseUpTo=Stream_Structure_Temp->first;
997 }
998 else
999 #endif //MEDIAINFO_HASH
1000 GoTo(Stream_Structure_Temp->first);
1001 }
1002 NeedOldIndex=false;
1003 SecondPass=true;
1004 Index_Pos.clear(); //We didn't succeed to find theses indexes :(
1005 return true;
