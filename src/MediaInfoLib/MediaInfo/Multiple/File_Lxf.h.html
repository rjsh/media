78 {

91 typedef std::vector<stream> streams;
92 streams Videos;
93 streams Audios;
94 struct stream_header
95 {
96 int64u TimeStamp_Begin;
97 int64u TimeStamp_End;
98 int64u Duration;
99 int64u Duration_First;
100 int8u PictureType;
101
102 stream_header()
103 {
104 TimeStamp_Begin=(int64u)-1;
105 TimeStamp_End=(int64u)-1;
106 Duration=(int64u)-1;
107 Duration_First=(int64u)-1;
108 PictureType=(int8u)-1;
109 }
110 stream_header(int64u TimeStamp_Begin_, int64u TimeStamp_End_, int64u Duration_, int8u PictureType_)
111 {
112 TimeStamp_Begin=TimeStamp_Begin_;
113 TimeStamp_End=TimeStamp_End_;
114 Duration=Duration_;
115 Duration_First = (int64u)-1;
116 PictureType = PictureType_;
117 }
118 };
