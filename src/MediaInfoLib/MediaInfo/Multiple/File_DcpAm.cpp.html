72//---------------------------------------------------------------------------
73void File_DcpAm::Streams_Finish()
74{
75 if (ReferenceFiles==NULL)
76 return;
77
78 ReferenceFiles->ParseReferences();
79
80 // Detection of IMF CPL
81 bool IsImf=false;
82 for (size_t StreamKind=Stream_General+1; StreamKind<Stream_Max; StreamKind++)
83 for (size_t StreamPos=0; StreamPos<Count_Get((stream_t)StreamKind); StreamPos++)
84 if (Retrieve((stream_t)StreamKind, StreamPos, "MuxingMode").find(__T("IMF CPL"))==0)
85 IsImf=true;
86 if (IsImf)
87 {
88 Fill(Stream_General, 0, General_Format, "IMF AM", Unlimited, true, true);
89 Clear(Stream_General, 0, General_Format_Version);
90 }
91}

108//***************************************************************************
109// Buffer - File header
110//***************************************************************************
111
112//---------------------------------------------------------------------------
113bool File_DcpAm::FileHeader_Begin()
114{
115 static const char *InteropNs="http://www.digicine.com/PROTO-ASDCP-AM-20040311#";
116 static const char *SmpteNs="http://www.smpte-ra.org/schemas/429-9/2007/AM";
117 XMLDocument document;
118 if (!FileHeader_Begin_XML(document))
119 return false;
120
121 XMLElement* AssetMap=document.FirstChildElement();
122 const char *NameSpace;
123 if (!AssetMap || strcmp(LocalName(AssetMap, NameSpace), "AssetMap") || !NameSpace)
124 {
125 Reject("DcpAm");
126 return false;
127 }
128 std::string FmtVersion;
129 if (!strcmp(NameSpace, InteropNs))
130 {
131 FmtVersion="Interop";
132 }
133 else if (!strcmp(NameSpace, SmpteNs))
134 {
135 FmtVersion="SMPTE";
136 }
137 else
138 {
139 Reject("DcpAm");
140 return false;
141 }
142
143 Accept("DcpAm");
144 Fill(Stream_General, 0, General_Format, "DCP AM");
145 Fill(Stream_General, 0, General_Format_Version, FmtVersion);
146 Config->File_ID_OnlyRoot_Set(false);
147
148 //Parsing main elements
149 for (XMLElement* AssetMap_Item=AssetMap->FirstChildElement(); AssetMap_Item; AssetMap_Item=AssetMap_Item->NextSiblingElement())
150 {
151 const char *AmItemNs, *AmItemName = LocalName(AssetMap_Item, AmItemNs);
152 if (!AmItemNs || strcmp(AmItemNs, NameSpace))
153 continue; // item has wrong namespace
154
155 //AssetList
156 if (!strcmp(AmItemName, "AssetList"))
157 {
158 for (XMLElement* AssetList_Item=AssetMap_Item->FirstChildElement(); AssetList_Item; AssetList_Item=AssetList_Item->NextSiblingElement())
159 {
160 //Asset
161 if (MatchQName(AssetList_Item, "Asset", NameSpace))
162 {
163 File_DcpPkl::stream Stream;
164
165 for (XMLElement* Asset_Item=AssetList_Item->FirstChildElement(); Asset_Item; Asset_Item=Asset_Item->NextSiblingElement())
166 {
167 //ChunkList
168 if (MatchQName(Asset_Item, "ChunkList", NameSpace))
169 {
170 for (XMLElement* ChunkList_Item=Asset_Item->FirstChildElement(); ChunkList_Item; ChunkList_Item=ChunkList_Item->NextSiblingElement())
171 {
172 //Chunk
173 if (MatchQName(ChunkList_Item, "Chunk", NameSpace))
174 {
175 File_DcpPkl::stream::chunk Chunk;
176
177 for (XMLElement* Chunk_Item=ChunkList_Item->FirstChildElement(); Chunk_Item; Chunk_Item=Chunk_Item->NextSiblingElement())
178 {
179 //Path
180 if (MatchQName(Chunk_Item, "Path", NameSpace))
181 if (const char* Text=Chunk_Item->GetText())
182 Chunk.Path=Text;
183 }
184
185 Stream.ChunkList.push_back(Chunk);
186 }
187 }
188 }
189
190 //Id
191 if (MatchQName(Asset_Item, "Id", NameSpace))
192 if (const char* Text=Asset_Item->GetText())
193 Stream.Id=Text;
194
195 //PackingList
196 if (MatchQName(Asset_Item, "PackingList", NameSpace) &&
197 Asset_Item->GetText() && !strcmp(Asset_Item->GetText(), "true"))
198 {
199 PKL_Pos=Streams.size();
200 Stream.StreamKind=(stream_t)(Stream_Max+2); // Means PKL
201 }
202 }
203
204 Streams.push_back(Stream);
205 }
206 }
207 }
208
209 //Creator
210 if (!strcmp(AmItemName, "Creator"))
211 Fill(Stream_General, 0, General_Encoded_Library, AssetMap_Item->GetText());
212
213 //IssueDate
214 if (!strcmp(AmItemName, "IssueDate"))
215 Fill(Stream_General, 0, General_Encoded_Date, AssetMap_Item->GetText());
216
217 //Issuer
218 if (!strcmp(AmItemName, "Issuer"))
219 Fill(Stream_General, 0, General_EncodedBy, AssetMap_Item->GetText());
220 }
221 Element_Offset=File_Size;
222
223 //Merging with PKL
224 if (PKL_Pos<Streams.size() && Streams[PKL_Pos].ChunkList.size()==1)
225 {
226 FileName Directory(File_Name);
227 Ztring PKL_FileName; PKL_FileName.From_UTF8(Streams[PKL_Pos].ChunkList[0].Path);
228 if (PKL_FileName.find(__T("file://"))==0 && PKL_FileName.find(__T("file:///"))==string::npos)
229 PKL_FileName.erase(0, 7); //TODO: better handling of relative and absolute file naes
230 MediaInfo_Internal MI;
231 MI.Option(__T("File_KeepInfo"), __T("1"));
232 Ztring ParseSpeed_Save=MI.Option(__T("ParseSpeed_Get"), __T(""));
233 Ztring Demux_Save=MI.Option(__T("Demux_Get"), __T(""));
234 MI.Option(__T("ParseSpeed"), __T("0"));
235 MI.Option(__T("Demux"), Ztring());
236 MI.Option(__T("File_IsReferenced"), __T("1"));
237 Ztring DirPath = Directory.Path_Get();
238 if (!DirPath.empty())
239 DirPath += PathSeparator;
240 size_t MiOpenResult=MI.Open(DirPath+PKL_FileName);
241 MI.Option(__T("ParseSpeed"), ParseSpeed_Save); //This is a global value, need to reset it. TODO: local value
242 MI.Option(__T("Demux"), Demux_Save); //This is a global value, need to reset it. TODO: local value
243 if (MiOpenResult
244 && (MI.Get(Stream_General, 0, General_Format)==__T("DCP PKL")
245 || MI.Get(Stream_General, 0, General_Format)==__T("IMF PKL")))
246 {
247 MergeFromPkl(((File_DcpPkl*)MI.Info)->Streams);
248
249 for (size_t Pos=0; Pos<MI.Count_Get(Stream_Other); ++Pos)
250 {
251 Stream_Prepare(Stream_Other);
252 Merge(*MI.Info, Stream_Other, Pos, StreamPos_Last);
253 }
254 }
255 }
256
257 //Creating the playlist
258 if (!Config->File_IsReferenced_Get())
259 {
260 ReferenceFiles=new File__ReferenceFilesHelper(this, Config);
261
262 for (File_DcpPkl::streams::iterator Stream=Streams.begin(); Stream!=Streams.end(); ++Stream)
263 if (Stream->StreamKind==(stream_t)(Stream_Max+1) && Stream->ChunkList.size()==1) // Means CPL
264 {
265 sequence* Sequence=new sequence;
266 Sequence->FileNames.push_back(Ztring().From_UTF8(Stream->ChunkList[0].Path));
267
268 Sequence->StreamID=ReferenceFiles->Sequences_Size()+1;
269 ReferenceFiles->AddSequence(Sequence);
270 }
271
272 ReferenceFiles->FilesForStorage=true;
273 }
274
275 //All should be OK...
276 return true;
277}

283//---------------------------------------------------------------------------
284void File_DcpAm::MergeFromPkl (File_DcpPkl::streams &StreamsToMerge)
285{
286 for (File_DcpPkl::streams::iterator Stream=Streams.begin(); Stream!=Streams.end(); ++Stream)
287 {
288 for (File_DcpPkl::streams::iterator StreamToMerge=StreamsToMerge.begin(); StreamToMerge!=StreamsToMerge.end(); ++StreamToMerge)
289 if (StreamToMerge->Id==Stream->Id)
290 {
291 if (Stream->StreamKind==Stream_Max)
292 Stream->StreamKind=StreamToMerge->StreamKind;
293 if (Stream->OriginalFileName.empty())
294 Stream->OriginalFileName=StreamToMerge->OriginalFileName;
295 if (Stream->Type.empty())
296 Stream->Type=StreamToMerge->Type;
297 if (Stream->AnnotationText.empty())
298 Stream->AnnotationText=StreamToMerge->AnnotationText;
299 }
300 }
301}
