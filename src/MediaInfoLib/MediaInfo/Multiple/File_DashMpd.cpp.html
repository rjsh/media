48void DashMpd_Transform (Ztring &Value, std::map<Ztring, Ztring> &Attributes)
49{
50 size_t Pos1=0;
51 for (;;)
52 {
53 Pos1=Value.find(__T('$'), Pos1);
54 if (Pos1==string::npos)
55 break;
56 size_t Pos2=Value.find(__T('$'), Pos1+1);
57 if (Pos2==string::npos)
58 break;
59 Ztring Name=Value.substr(Pos1+1, Pos2-Pos1-1);
60 if (Name.empty())
61 Value.erase(Pos1, 1);
62 else
63 {
64 if (Name==__T("RepresentationID"))
65 Name=__T("id");
66 if (Name==__T("Bandwidth"))
67 Name=__T("bandwidth");
68 std::map<Ztring, Ztring>::iterator Attribute_It=Attributes.find(Name);
69 if (Attribute_It!=Attributes.end())
70 {
71 Value.erase(Pos1, Pos2-Pos1+1);
72 Value.insert(Pos1, Attribute_It->second);
73 }
74 else
75 Pos1+=2+Name.size();
76 }
77 }
78}

80//---------------------------------------------------------------------------
81stream_t DashMpd_mimeType_StreamKind (const char* mimeType)
82{
83 Ztring StreamKind; StreamKind.From_UTF8(mimeType);
84 if (StreamKind.find(__T("video"))==0)
85 return Stream_Video;
86 else if (StreamKind.find(__T("audio"))==0)
87 return Stream_Audio;
88 else if (StreamKind.find(__T("application/ttml+xml"))==0)
89 return Stream_Text;
90 else
91 return Stream_Other;
92}

94//---------------------------------------------------------------------------
95Ztring DashMpd_codecid_CodecID (const char* codecid)
96{
97 Ztring CodecID;
98
99 Ztring Codecs; Codecs.From_UTF8(codecid);
100 size_t DotPos=Codecs.find(__T('.'));
101 if (DotPos==4 && Codecs.substr(0, DotPos).find(__T("mp4"))==0)
102 DotPos=Codecs.find(__T('.'), 5);
103 if (DotPos==string::npos)
104 {
105 CodecID=Codecs;
106 }
107 else
108 {
109 CodecID=Codecs.substr(0, DotPos);
110 //TODO per format, rfc 6381 //Sequence->Infos["Format_Profile"]=;
111 }
112 CodecID.FindAndReplace(__T("0x"), Ztring(), 0, Ztring_Recursive);
113
114 return CodecID;
115}

118struct template_generic
119{
120 sequence* Sequence;
121 Ztring SourceDir;
122 Ztring BaseURL;
123 Ztring initialization;
124 Ztring media;
125 int64u duration;
126 int64u startNumber;
127 int64u duration_Max;
128 int64u startNumber_Max;
129 struct segmenttimeline
130 {
131 int64u t; //start time
132 int64u d; //duration per segment
133 int64u r; //repeat count
134
135 segmenttimeline()
136 {
137 t=1;
138 d=1;
139 r=0;
140 }
141 };
142 std::vector<segmenttimeline> SegmentTimeLines;
143 std::map<Ztring, Ztring> Attributes_ForMedia;
144
145 template_generic(const Ztring &BaseURL=Ztring(), const Ztring &SourceDir=Ztring())
146 {
147 Sequence=new sequence;
148 template_generic::BaseURL=BaseURL;
149 template_generic::SourceDir=SourceDir;
150 duration=1;
151 startNumber=1;
152 duration_Max=0;
153 startNumber_Max=0;
154 }
155
156 template_generic(const template_generic &ToCopy)
157 {
158 Sequence=new sequence;
159 *Sequence=*ToCopy.Sequence;
160 template_generic::BaseURL=ToCopy.BaseURL;
161 template_generic::SourceDir=ToCopy.SourceDir;
162 initialization=ToCopy.initialization;
163 media=ToCopy.media;
164 duration=ToCopy.duration;
165 startNumber=ToCopy.startNumber;
166 duration_Max=ToCopy.duration_Max;
167 startNumber_Max=ToCopy.startNumber_Max;
168 }
169
170 void AdaptationSet_Attributes_Parse (XMLElement* Item);
171 void SegmentTemplate_Attributes_Parse (XMLElement* Item);
172 void SegmentTimeline_Attributes_Parse (XMLElement* Item);
173 void Representation_Attributes_Parse (XMLElement* Item);
174
175 void Decode ();
176};

178void template_generic::AdaptationSet_Attributes_Parse (XMLElement* Item)
179{
180 const char* Attribute;
181
182 //Attributes - mineType
183 Attribute=Item->Attribute("mimeType");
184 if (Attribute)
185 Sequence->StreamKind=DashMpd_mimeType_StreamKind(Attribute);
186
187 //Attributes - codecs
188 Attribute=Item->Attribute("codecs");
189 if (Attribute)
190 Sequence->Infos["CodecID"]=DashMpd_codecid_CodecID(Attribute);
191
192 //Attributes - lang
193 Attribute=Item->Attribute("lang");
194 if (Attribute)
195 Sequence->Infos["Language"].From_UTF8(Attribute);
196}

198void template_generic::SegmentTemplate_Attributes_Parse (XMLElement* Item)
199{
200 const char* Attribute;
201
202 //Attributes - initialization
203 Attribute=Item->Attribute("initialization");
204 if (Attribute)
205 {
206 initialization.From_UTF8(Attribute);
207 }
208
209 //Attributes - media
210 Attribute=Item->Attribute("media");
211 if (Attribute)
212 {
213 media.From_UTF8(Attribute);
214 }
215
216 //Attributes - duration
217 Attribute=Item->Attribute("duration");
218 if (Attribute)
219 {
220 duration=Ztring().From_UTF8(Attribute).To_int64u();
221 }
222
223 //Attributes - startNumber
224 Attribute=Item->Attribute("startNumber");
225 if (Attribute)
226 {
227 startNumber=Ztring().From_UTF8(Attribute).To_int64u();
228 }
229}
230
231void template_generic::SegmentTimeline_Attributes_Parse (XMLElement* Item)
232{
233 const char* Attribute;
234 segmenttimeline SegmentTimeLine;
235
236 //Attributes - t (start time)
237 Attribute=Item->Attribute("t");
238 if (Attribute)
239 {
240 SegmentTimeLine.t=Ztring().From_UTF8(Attribute).To_int64u();
241 }
242 else
243 SegmentTimeLine.t=startNumber;
244
245 //Attributes - d (duration per segment)
246 Attribute=Item->Attribute("d");
247 if (Attribute)
248 {
249 SegmentTimeLine.d=Ztring().From_UTF8(Attribute).To_int64u();
250 }
251 else
252 SegmentTimeLine.d=duration;
253
254 //Attributes - r (repeat count)
255 Attribute=Item->Attribute("r");
256 if (Attribute)
257 {
258 SegmentTimeLine.r=Ztring().From_UTF8(Attribute).To_int64u();
259 }
260
261 SegmentTimeLines.push_back(SegmentTimeLine);
262 duration_Max+=SegmentTimeLine.d*(SegmentTimeLine.r+1);
263 startNumber_Max+=SegmentTimeLine.r+1;
264}

266void template_generic::Representation_Attributes_Parse (XMLElement* Item)
267{
268 const char* Attribute;
269
270 //Attributes - id
271 Attribute=Item->Attribute("id");
272 if (Attribute)
273 {
274 Sequence->StreamID=Ztring().From_UTF8(Attribute).To_int64u(16);
275 }
276
277 //Attributes - bandwidth
278 Attribute=Item->Attribute("bandwidth");
279 if (Attribute)
280 {
281 Sequence->Infos["BitRate"].From_UTF8(Attribute);
282 }
283
284 //Attributes - frame size
285 Attribute=Item->Attribute("width");
286 if (Attribute)
287 {
288 Sequence->Infos["Width"].From_UTF8(Attribute);
289 }
290 Attribute=Item->Attribute("height");
291 if (Attribute)
292 {
293 Sequence->Infos["Height"].From_UTF8(Attribute);
294 }
295
296 //Attributes - mineType
297 Attribute=Item->Attribute("mimeType");
298 if (Attribute)
299 Sequence->StreamKind=DashMpd_mimeType_StreamKind(Attribute);
300
301 //Attributes - codecs
302 Attribute=Item->Attribute("codecs");
303 if (Attribute)
304 Sequence->Infos["CodecID"]=DashMpd_codecid_CodecID(Attribute);
305
306 //Attributes - lang
307 Attribute=Item->Attribute("lang");
308 if (Attribute)
309 Sequence->Infos["Language"].From_UTF8(Attribute);
310
311 //Attributes - Saving all attributes
312 for (const XMLAttribute* Attribute_Item=Item->FirstAttribute(); Attribute_Item; Attribute_Item=Attribute_Item->Next())
313 {
314 Ztring Name; Name.From_UTF8(Attribute_Item->Name());
315 Ztring Value; Value.From_UTF8(Attribute_Item->Value());
316 Attributes_ForMedia[Name]=Value;
317 }
318}

320//---------------------------------------------------------------------------
321void template_generic::Decode()
322{
323 //initialization - URL decoding, template adaptation and add it
324 if (!initialization.empty())
325 {
326 DashMpd_Transform(initialization, Attributes_ForMedia);
327 Sequence->AddFileName(BaseURL+initialization);
328 }
329
330 //media - URL decoding, template adaptation and add it
331 if (!media.empty())
332 {
333 DashMpd_Transform(media, Attributes_ForMedia);
334 size_t Index_Pos=media.find(__T("$Index"));
335 size_t Index_StringSize=5;
336 if (Index_Pos==string::npos)
337 {
338 Index_Pos=media.find(__T("$Number"));
339 Index_StringSize++;
340 }
341 int8u Index_Size=1;
342 if (Index_Pos!=string::npos)
343 {
344 size_t Index_Pos_End=media.find(__T('$'), Index_Pos+1+Index_StringSize);
345 if (Index_Pos_End!=string::npos && Index_Pos+1+Index_StringSize+2<Index_Pos_End && media[Index_Pos+1+Index_StringSize]=='%' && media[Index_Pos+1+Index_StringSize+1]=='0')
346 {
347 Index_Size=Ztring(media.substr(Index_Pos+1+Index_StringSize+2, Index_Pos_End-(Index_Pos+1+Index_StringSize+2))).To_int8u();
348 }
349 else if (Index_Pos_End==string::npos || Index_Pos+1+Index_StringSize!=Index_Pos_End)
350 Index_Pos=string::npos;
351 }
352 size_t Time_Pos=media.find(__T("$Time$"));
353 if (Index_Pos!=string::npos || Time_Pos!=string::npos)
354 {
355 Ztring Media_Name(media);
356 if (Index_Pos!=string::npos)
357 {
358 Media_Name.erase(Index_Pos, 1+Index_StringSize+1);
359 if (Time_Pos!=string::npos && Time_Pos>Index_Pos)
360 Time_Pos-=1+Index_StringSize+1;
361 }
362 if (Time_Pos!=string::npos)
363 {
364 Media_Name.erase(Time_Pos, 6);
365 if (Index_Pos!=string::npos && Index_Pos>Time_Pos)
366 Index_Pos-=6;
367 }
368 if (SegmentTimeLines.empty())
369 {
370 int64u Index_Pos_Temp=startNumber;
371 for (;;)
372 {
373 Ztring Media_Name_Temp(Media_Name);
374 Ztring Index; Index.From_Number(Index_Pos_Temp);
375 if (Index.size()<Index_Size)
376 Index.insert(0, Index_Size-Index.size(), __T('0'));
377 if (Index_Pos!=(size_t)-1)
378 Media_Name_Temp.insert(Index_Pos, Index);
379 else
380 Media_Name_Temp.insert(Time_Pos, Index);
381
382 Ztring File_Name;
383 Ztring File_Name_With_Path;
384 if (!SourceDir.empty())
385 File_Name_With_Path+=SourceDir+PathSeparator;
386 File_Name+=BaseURL+Media_Name_Temp;
387 File_Name_With_Path+=BaseURL+Media_Name_Temp;
388 if (!File::Exists(File_Name_With_Path))
389 break;
390 Sequence->AddFileName(File_Name);
391 Index_Pos_Temp++;
392 }
393 }
394 else
395 {
396 int64u SegmentTimeLines_duration=0;
397 int64u SegmentTimeLines_startNumber=startNumber;
398 for (size_t SegmentTimeLines_Pos=0; SegmentTimeLines_Pos<SegmentTimeLines.size(); SegmentTimeLines_Pos++)
399 {
400 for (int64u Pos=0; Pos<=SegmentTimeLines[SegmentTimeLines_Pos].r; Pos++)
401 {
402 Ztring Media_Name_Temp(Media_Name);
403 size_t Time_Pos_Temp=Time_Pos;
404 if (Index_Pos!=string::npos)
405 {
406 Ztring Index; Index.From_Number(SegmentTimeLines_startNumber);
407 if (Index.size()<Index_Size)
408 Index.insert(0, Index_Size-Index.size(), __T('0'));
409 Media_Name_Temp.insert(Index_Pos, Index);
410 if (Time_Pos!=string::npos && Time_Pos>Index_Pos)
411 Time_Pos_Temp+=Index.size();
412 }
413 if (Time_Pos_Temp!=string::npos)
414 {
415 Ztring Time; Time.From_Number(SegmentTimeLines_duration);
416 Media_Name_Temp.insert(Time_Pos_Temp, Time);
417 }
418
419 Sequence->AddFileName(BaseURL+Media_Name_Temp);
420 SegmentTimeLines_duration+=SegmentTimeLines[SegmentTimeLines_Pos].d;
421 SegmentTimeLines_startNumber++;
422 }
423 }
424 }
425 }
426 else
427 Sequence->AddFileName(BaseURL+media);
428 }

482//***************************************************************************
483// Buffer - File header
484//***************************************************************************
485
486//---------------------------------------------------------------------------
487bool File_DashMpd::FileHeader_Begin()
488{
489 XMLDocument document;
490 if (!FileHeader_Begin_XML(document))
491 return false;
492
493 {
494 XMLElement* Root=document.FirstChildElement("MPD");
495 if (Root)
496 {
497 const char* Attribute=Root->Attribute("xmlns");
498 if (Attribute==NULL
499 || (Ztring().From_UTF8(Attribute)!=__T("urn:mpeg:DASH:schema:MPD:2011")
500 && Ztring().From_UTF8(Attribute)!=__T("urn:mpeg:dash:schema:mpd:2011") //Some muxers use lower case version
501 && Ztring().From_UTF8(Attribute)!=__T("urn:3GPP:ns:PSS:AdaptiveHTTPStreamingMPD:2009")))
502 {
503 Reject("DashMpd");
504 return false;
505 }
506
507 Accept("DashMpd");
508 Fill(Stream_General, 0, General_Format, "DASH MPD");
509 Config->File_ID_OnlyRoot_Set(false);
510
511 ReferenceFiles=new File__ReferenceFilesHelper(this, Config);
512
513 //Parsing main elements
514 Ztring BaseURL;
515
516 for (XMLElement* Root_Item=Root->FirstChildElement(); Root_Item; Root_Item=Root_Item->NextSiblingElement())
517 {
518 //Common information
519 if (string(Root_Item->Value())=="BaseURL")
520 {
521 if (BaseURL.empty()) //Using the first one
522 BaseURL=Root_Item->GetText();
523 }
524
525 //Period
526 if (string(Root_Item->Value())=="Period")
527 {
528 for (XMLElement* Period_Item=Root_Item->FirstChildElement(); Period_Item; Period_Item=Period_Item->NextSiblingElement())
529 {
530 //AdaptationSet (=a stream)
531 if (string(Period_Item->Value())=="AdaptationSet")
532 {
533 template_generic Template_Generic(BaseURL, FileName(File_Name).Path_Get());
534
535 Template_Generic.AdaptationSet_Attributes_Parse(Period_Item);
536
537 //Sub
538 for (XMLElement* AdaptationSet_Item=Period_Item->FirstChildElement(); AdaptationSet_Item; AdaptationSet_Item=AdaptationSet_Item->NextSiblingElement())
539 {
540 //SegmentTemplate
541 if (string(AdaptationSet_Item->Value())=="SegmentTemplate")
542 {
543 Template_Generic.SegmentTemplate_Attributes_Parse(AdaptationSet_Item);
544
545 //Sub
546 for (XMLElement* SegmentTemplate_Item=AdaptationSet_Item->FirstChildElement(); SegmentTemplate_Item; SegmentTemplate_Item=SegmentTemplate_Item->NextSiblingElement())
547 {
548 //SegmentTimeline
549 if (string(SegmentTemplate_Item->Value())=="SegmentTimeline")
550 {
551 //Sub
552 for (XMLElement* SegmentTimeline_Item=SegmentTemplate_Item->FirstChildElement(); SegmentTimeline_Item; SegmentTimeline_Item=SegmentTimeline_Item->NextSiblingElement())
553 {
554 //SegmentTimeline
555 if (string(SegmentTimeline_Item->Value())=="S")
556 {
557 Template_Generic.SegmentTimeline_Attributes_Parse(SegmentTimeline_Item);
558 }
559 }
560 }
561 }
562 }
563
564 //Representation
565 if (string(AdaptationSet_Item->Value())=="Representation")
566 {
567 template_generic Template_Generic_PerRepresentation(Template_Generic);
568
569 Template_Generic_PerRepresentation.Representation_Attributes_Parse(AdaptationSet_Item);
570
571 //Sub
572 for (XMLElement* Representation_Item=AdaptationSet_Item->FirstChildElement(); Representation_Item; Representation_Item=Representation_Item->NextSiblingElement())
573 {
574 //BaseURL
575 if (string(Representation_Item->Value())=="BaseURL")
576 {
577 Template_Generic_PerRepresentation.Sequence->AddFileName(BaseURL+Ztring().From_UTF8(Representation_Item->GetText()));
578 }
579
580 //SegmentTemplate
581 if (string(Representation_Item->Value())=="SegmentTemplate")
582 {
583 Template_Generic_PerRepresentation.SegmentTemplate_Attributes_Parse(Representation_Item);
584 }
585
586 //SegmentBase
587 if (string(Representation_Item->Value())=="SegmentBase")
588 {
589 //Sub
590 for (XMLElement* SegmentBase_Item=Representation_Item->FirstChildElement(); SegmentBase_Item; SegmentBase_Item=SegmentBase_Item->NextSiblingElement())
591 {
592 //Initialization
593 if (string(SegmentBase_Item->Value())=="Initialization")
594 {
595 Attribute=SegmentBase_Item->Attribute("sourceURL");
596 if (Attribute)
597 Template_Generic_PerRepresentation.Sequence->AddFileName(BaseURL+Ztring().From_UTF8(Attribute), 0);
598 }
599 }
600 }
601
602 //SegmentList
603 if (string(Representation_Item->Value())=="SegmentList")
604 {
605 //Sub
606 for (XMLElement* SegmentBase_Item=Representation_Item->FirstChildElement(); SegmentBase_Item; SegmentBase_Item=SegmentBase_Item->NextSiblingElement())
607 {
608 //Initialization
609 if (string(SegmentBase_Item->Value())=="SegmentURL")
610 {
611 bool IsSupported=true;
612 Attribute=SegmentBase_Item->Attribute("mediaRange");
613 if (Attribute)
614 {
615 size_t Length=strlen(Attribute);
616 if (Length<2
617 || Attribute[0]!='0'
618 || Attribute[1]!='-')
619 IsSupported=false; //Currently, we do not support ranges
620 }
621
622 Attribute=SegmentBase_Item->Attribute("media");
623 if (Attribute && IsSupported)
624 Template_Generic_PerRepresentation.Sequence->AddFileName(BaseURL+Ztring().From_UTF8(Attribute));
625 }
626 }
627 }
628 }
629
630 Template_Generic_PerRepresentation.Decode();
631 ReferenceFiles->AddSequence(Template_Generic_PerRepresentation.Sequence);
632 }
633 }
634 }
635
636 //Representation (=a stream)
637 if (string(Period_Item->Value())=="Representation")
638 {
639 sequence* Sequence=new sequence;
640 int64u duration=1;
641
642 //Attributes - mineType
643 Attribute=Period_Item->Attribute("mimeType");
644 if (Attribute)
645 Sequence->StreamKind=DashMpd_mimeType_StreamKind(Attribute);
646
647 //Attributes - codecs
648 Attribute=Period_Item->Attribute("codecs");
649 if (Attribute)
650 Sequence->Infos["CodecID"]=DashMpd_codecid_CodecID(Attribute);
651
652 //Attributes - lang
653 Attribute=Period_Item->Attribute("lang");
654 if (Attribute)
655 Sequence->Infos["Language"].From_UTF8(Attribute);
656
657 //Sub
658 for (XMLElement* AdaptationSet_Item=Period_Item->FirstChildElement(); AdaptationSet_Item; AdaptationSet_Item=AdaptationSet_Item->NextSiblingElement())
659 {
660 //SegmentInfo
661 if (string(AdaptationSet_Item->Value())=="SegmentInfo")
662 {
663 //Attributes - duration
664 Attribute=AdaptationSet_Item->Attribute("duration");
665 if (Attribute)
666 {
667 duration=Ztring().From_UTF8(Attribute).To_int64u();
668 }
669
670 //Sub
671 for (XMLElement* SegmentInfo_Item=AdaptationSet_Item->FirstChildElement(); SegmentInfo_Item; SegmentInfo_Item=SegmentInfo_Item->NextSiblingElement())
672 {
673 //InitialisationSegmentURL
674 if (string(SegmentInfo_Item->Value())=="InitialisationSegmentURL")
675 {
676 Attribute=SegmentInfo_Item->Attribute("sourceURL");
677 if (Attribute)
678 Sequence->AddFileName(BaseURL+Ztring().From_UTF8(Attribute), 0);
679 }
680
681 //Url
682 if (string(SegmentInfo_Item->Value())=="Url")
683 {
684 Attribute=SegmentInfo_Item->Attribute("sourceURL");
685 if (Attribute)
686 Sequence->AddFileName(BaseURL+Ztring().From_UTF8(Attribute));
687 }
688 }
689
690 ReferenceFiles->AddSequence(Sequence);
691 }
692 }
693 }
694 }
695 }
696 }
697 }
698 else
699 {
700 Reject("DashMpd");
701 return false;
702 }
703 }
704
705 Element_Offset=File_Size;
706
707 //All should be OK...
708 return true;
709}
