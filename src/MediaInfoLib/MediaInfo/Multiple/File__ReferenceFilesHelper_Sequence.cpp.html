88//---------------------------------------------------------------------------
89void sequence::AddResource(resource* NewResource, size_t Pos)
90{
91 NewResource->Sequence=Common;
92
93 #if MEDIAINFO_DEMUX
94 if (Resources.empty())
95 {
96 NewResource->Demux_Offset_DTS=0;
97 }
98 #endif //MEDIAINFO_DEMUX
99
100 if (Pos>=Resources.size())
101 Resources.push_back(NewResource);
102 else
103 Resources.insert(Resources.begin()+Pos, NewResource);
104}
105
106//---------------------------------------------------------------------------
107void sequence::UpdateFileName(const Ztring& OldFileName, const Ztring& NewFileName)
108{
109 size_t Resources_Size=Resources.size();
110 for (size_t Resources_Pos=0; Resources_Pos<Resources_Size; ++Resources_Pos)
111 {
112 resource* Resource=Resources[Resources_Pos];
113
114 Resource->UpdateFileName(OldFileName, NewFileName);
115
116 for (size_t Pos=0; Pos<Resource->FileNames.size(); Pos++)
117 if (Resource->FileNames[Pos]==NewFileName)
118 Infos["UniqueID"]=OldFileName;
119 }
120}
