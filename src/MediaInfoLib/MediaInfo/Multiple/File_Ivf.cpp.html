43//---------------------------------------------------------------------------
44bool File_Ivf::FileHeader_Begin()
45{
46 //Synchro
47 if (4>Buffer_Size)
48 return false;
49 if (Buffer[0]!=0x44 //"DKIF"
50 || Buffer[1]!=0x4B
51 || Buffer[2]!=0x49
52 || Buffer[3]!=0x46)
53 {
54 Reject();
55 return false;
56 }
57 if (6>Buffer_Size)
58 return false;
59
60 return true;
61}
62
63//---------------------------------------------------------------------------
64void File_Ivf::FileHeader_Parse()
65{
66 //Parsing
67 int32u frame_rate_num, frame_rate_den, frame_count, fourcc;
68 int16u version, header_size, width, height;
69
70 Skip_C4 ( "Signature");
71 Get_L2 (version, "Version");
72 if (version==0)
73 {
74 Get_L2 (header_size, "Header Size");
75 if (header_size>=32)
76 {
77 Get_C4 (fourcc, "Fourcc");
78 Get_L2 (width, "Width");
79 Get_L2 (height, "Height");
80 Get_L4 (frame_rate_num, "FrameRate Numerator");
81 Get_L4 (frame_rate_den, "FrameRate Denominator");
82 Get_L4 (frame_count, "Frame Count");
83 Skip_L4( "Unused");
84 if (header_size-32)
85 Skip_XX(header_size-32, "Unknown");
86 }
87 else
88 {
89 fourcc=0x00000000;
90 width=0;
91 height=0;
92 frame_rate_num=0;
93 frame_rate_den=0;
94 frame_count=0;
95 }
96 }
97 else
98 {
99 header_size=0;
100 fourcc=0x00000000;
101 width=0;
102 height=0;
103 frame_rate_num=0;
104 frame_rate_den=0;
105 frame_count=0;
106 }
107
108 FILLING_BEGIN();
109 Accept("IVF");
110
111 Fill(Stream_General, 0, General_Format, "IVF");
112
113 if (version==0 && header_size>=32)
114 {
115 Stream_Prepare(Stream_Video);
116 CodecID_Fill(Ztring().From_CC4(fourcc), Stream_Video, 0, InfoCodecID_Format_Riff);
117 Fill(Stream_Video, 0, Video_FrameRate, (float)frame_rate_num / frame_rate_den);
118 Fill(Stream_Video, 0, Video_FrameCount, frame_count);
119 Fill(Stream_Video, 0, Video_Width, width);
120 Fill(Stream_Video, 0, Video_Height, height);
121 Fill(Stream_Video, 0, Video_StreamSize, File_Size-header_size-12*frame_count); //Overhead is 12 byte per frame
122 }
123
124 //No more need data
125 Finish("IVF");
126 FILLING_END();
127}
