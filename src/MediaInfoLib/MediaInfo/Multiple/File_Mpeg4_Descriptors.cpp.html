321//---------------------------------------------------------------------------
322void File_Mpeg4_Descriptors::Header_Parse()
323{
324 //Parsing
325 size_t Size=0;
326 int8u type, Size_ToAdd;
327 Get_B1(type, "type");
328 if (type==0)
329 {
330 Header_Fill_Code(0x00, "Padding");
331 Header_Fill_Size(1);
332 return;
333 }
334 do
335 {
336 Get_B1(Size_ToAdd, "size");
337 Size=(Size<<7) | (Size_ToAdd&0x7F);
338 }
339 while (Size_ToAdd&0x80);
340
341 //Filling
342 Header_Fill_Code(type, Ztring().From_CC1(type));
343 if (Element_Offset+Size>=Element_Size)
344 Size=(size_t)(Element_Size-Element_Offset); //Found one file with too big size but content is OK, cutting the block
345 Header_Fill_Size(Element_Offset+Size);
346}

418//---------------------------------------------------------------------------
419void File_Mpeg4_Descriptors::Descriptor_01()
420{
421 //Parsing
422 bool URL_Flag;
423 BS_Begin();
424 Skip_S2(10, "ObjectDescriptorID");
425 Get_SB ( URL_Flag, "URL_Flag");
426 Skip_SB( "includeInlineProfileLevelFlag");
427 Skip_S1( 4, "reserved");
428 BS_End();
429 if (URL_Flag)
430 {
431 int8u URLlength;
432 Get_B1 (URLlength, "URLlength");
433 Skip_UTF8(URLlength, "URLstring");
434 }
435 if (Element_Code==0x02 || Element_Code==0x10)
436 {
437 Info_B1(ODProfileLevel, "ODProfileLevelIndication"); Param_Info1(Mpeg4_Descriptors_ODProfileLevelIndication(ODProfileLevel));
438 Info_B1(SceneProfileLevel, "sceneProfileLevelIndication"); Param_Info1(Mpeg4_Descriptors_SceneProfileLevelIndication(SceneProfileLevel));
439 Info_B1(AudioProfileLevel, "audioProfileLevelIndication"); Param_Info1(Mpeg4_Descriptors_AudioProfileLevelIndication(AudioProfileLevel));
440 Info_B1(VisualProfileLevel, "visualProfileLevelIndication"); Param_Info1(Mpeg4v_Profile_Level(VisualProfileLevel));
441 Info_B1(GraphicsProfileLevel, "graphicsProfileLevelIndication"); Param_Info1(Mpeg4_Descriptors_GraphicsProfileLevelIndication(GraphicsProfileLevel));
442 }
443
444 FILLING_BEGIN();
445 Element_ThisIsAList();
446 FILLING_END();
447}

449//---------------------------------------------------------------------------
450void File_Mpeg4_Descriptors::Descriptor_03()
451{
452 //Parsing
453 bool streamDependenceFlag, URL_Flag, OCRstreamFlag;
454 Get_B2 (ES_ID, "ES_ID");
455 BS_Begin();
456 Get_SB ( streamDependenceFlag, "streamDependenceFlag");
457 Get_SB ( URL_Flag, "URL_Flag");
458 Get_SB ( OCRstreamFlag, "OCRstreamFlag");
459 Skip_S1(5, "streamPriority");
460 BS_End();
461 if (streamDependenceFlag)
462 Skip_B2( "dependsOn_ES_ID");
463 if (URL_Flag)
464 {
465 int8u URLlength;
466 Get_B1 (URLlength, "URLlength");
467 Skip_UTF8(URLlength, "URLstring");
468 }
469 if (OCRstreamFlag)
470 Skip_B2( "OCR_ES_Id");
471
472 FILLING_BEGIN();
473 Element_ThisIsAList();
474 FILLING_END();
475}

477//---------------------------------------------------------------------------
478void File_Mpeg4_Descriptors::Descriptor_04()
479{
480 //Parsing
481 int32u bufferSizeDB, MaxBitrate, AvgBitrate;
482 int8u streamType;
483 Get_B1 (ObjectTypeId, "objectTypeIndication"); Param_Info1(Mpeg4_Descriptors_ObjectTypeIndication(ObjectTypeId));
484 BS_Begin();
485 Get_S1 (6, streamType, "streamType"); Param_Info1(Mpeg4_Descriptors_StreamType(streamType));
486 Skip_SB( "upStream");
487 Skip_SB( "reserved");
488 BS_End();
489 Get_B3 (bufferSizeDB, "bufferSizeDB");
490 Get_B4 (MaxBitrate, "maxBitrate");
491 Get_B4 (AvgBitrate, "avgBitrate");
492
493 FILLING_BEGIN();
494 if (KindOfStream==Stream_Max)
495 switch (ObjectTypeId)
496 {
497 case 0x20 :
498 case 0x21 :
499 case 0x60 :
500 case 0x61 :
501 case 0x62 :
502 case 0x63 :
503 case 0x64 :
504 case 0x65 :
505 case 0x6A :
506 case 0x6C :
507 case 0x6D :
508 case 0x6E :
509 case 0xA3 :
510 case 0xA4 :
511 KindOfStream=Stream_Video; break;
512 case 0x40 :
513 case 0x66 :
514 case 0x67 :
515 case 0x68 :
516 case 0x69 :
517 case 0x6B :
518 case 0xA0 :
519 case 0xA1 :
520 case 0xA5 :
521 case 0xA6 :
522 case 0xA9 :
523 case 0xAA :
524 case 0xAB :
525 case 0xAC :
526 case 0xD1 :
527 case 0xD3 :
528 case 0xD4 :
529 case 0xE1 :
530 KindOfStream=Stream_Audio; break;
531 case 0x08 :
532 KindOfStream=Stream_Text; break;
533 default: ;
534 }
535 if (Count_Get(KindOfStream)==0)
536 Stream_Prepare(KindOfStream);
537 switch (ObjectTypeId)
538 {
539 case 0x01 : Fill(StreamKind_Last, StreamPos_Last, Fill_Parameter(StreamKind_Last, Generic_Format), "System", Error, false, true); break;
540 case 0x02 : Fill(StreamKind_Last, StreamPos_Last, Fill_Parameter(StreamKind_Last, Generic_Format), "System Core", Error, false, true); break;
541 //case 0x03 Interaction Stream
542 //case 0x05 AFX
543 //case 0x06 Font Data
544 //case 0x07 Synthesized Texture Stream
545 case 0x08 : Fill(Stream_Text , StreamPos_Last, Text_Format, "Streaming Text", Error, false, true); break;
546 case 0x20 : Fill(Stream_Video , StreamPos_Last, Video_Format, "MPEG-4 Visual", Error, false, true); break;
547 case 0x21 : Fill(Stream_Video , StreamPos_Last, Video_Format, "AVC", Error, false, true); break;
548 //case 0x22 Parameter Sets for AVC
549 case 0x40 : Fill(Stream_Audio , StreamPos_Last, Audio_Format, "AAC", Error, false, true); break; //MPEG-4 AAC
550 case 0x60 : Fill(Stream_Video , StreamPos_Last, Video_Format, "MPEG Video", Error, false, true); Fill(Stream_Video, StreamPos_Last, Video_Format_Profile, "Simple" , Error, false, true); Fill(Stream_Video, StreamPos_Last, Video_Format_Version, "Version 2", Error, false, true); break; //MPEG-2V Simple
551 case 0x61 : Fill(Stream_Video , StreamPos_Last, Video_Format, "MPEG Video", Error, false, true); Fill(Stream_Video, StreamPos_Last, Video_Format_Profile, "Main" , Error, false, true); Fill(Stream_Video, StreamPos_Last, Video_Format_Version, "Version 2", Error, false, true); break; //MPEG-2V Main
552 case 0x62 : Fill(Stream_Video , StreamPos_Last, Video_Format, "MPEG Video", Error, false, true); Fill(Stream_Video, StreamPos_Last, Video_Format_Profile, "SNR" , Error, false, true); Fill(Stream_Video, StreamPos_Last, Video_Format_Version, "Version 2", Error, false, true); break; //MPEG-2V SNR
553 case 0x63 : Fill(Stream_Video , StreamPos_Last, Video_Format, "MPEG Video", Error, false, true); Fill(Stream_Video, StreamPos_Last, Video_Format_Profile, "Spatial", Error, false, true); Fill(Stream_Video, StreamPos_Last, Video_Format_Version, "Version 2", Error, false, true); break; //MPEG-2V Spatial
554 case 0x64 : Fill(Stream_Video , StreamPos_Last, Video_Format, "MPEG Video", Error, false, true); Fill(Stream_Video, StreamPos_Last, Video_Format_Profile, "High" , Error, false, true); Fill(Stream_Video, StreamPos_Last, Video_Format_Version, "Version 2", Error, false, true); break; //MPEG-2V High
555 case 0x65 : Fill(Stream_Video , StreamPos_Last, Video_Format, "MPEG Video", Error, false, true); Fill(Stream_Video, StreamPos_Last, Video_Format_Profile, "4:2:2" , Error, false, true); Fill(Stream_Video, StreamPos_Last, Video_Format_Version, "Version 2", Error, false, true); break; //MPEG-2V 4:2:2
556 case 0x66 : Fill(Stream_Audio , StreamPos_Last, Audio_Format, "AAC", Error, false, true); Fill(Stream_Audio, StreamPos_Last, Audio_Format_Profile, "Main", Error, false, true); break; //MPEG-2 AAC Main
557 case 0x67 : Fill(Stream_Audio , StreamPos_Last, Audio_Format, "AAC", Error, false, true); Fill(Stream_Audio, StreamPos_Last, Audio_Format_Profile, "LC", Error, false, true); break; //MPEG-2 AAC LC
558 case 0x68 : Fill(Stream_Audio , StreamPos_Last, Audio_Format, "AAC", Error, false, true); Fill(Stream_Audio, StreamPos_Last, Audio_Format_Profile, "SSR", Error, false, true); break; //MPEG-2 AAC SSR
559 case 0x69 : Fill(Stream_Audio , StreamPos_Last, Audio_Format, "MPEG Audio", Error, false, true); Fill(Stream_Audio, StreamPos_Last, Audio_Format_Version, "Version 2", Error, false, true); Fill(Stream_Audio, StreamPos_Last, Audio_Format_Profile, "Layer 3", Error, false, true); break;

561 case 0x6B : Fill(Stream_Audio , StreamPos_Last, Audio_Format, "MPEG Audio", Error, false, true); Fill(Stream_Audio, StreamPos_Last, Audio_Format_Version, "Version 1", Error, false, true); break;
562 case 0x6C : Fill(Stream_Video , StreamPos_Last, Video_Format, "JPEG", Error, false, true); break;
563 case 0x6D : Fill(Stream_Video , StreamPos_Last, Video_Format, "PNG", Error, false, true); break;
564 case 0x6E : Fill(Stream_Video , StreamPos_Last, Video_Format, "MPEG Video", Error, false, true); break;

752{

815 Parser->Fill(KindOfStream, PosOfStream, "Demux_InitBytes", Data_Base64);
816 if (PosOfStream<(*Parser->Stream_More)[KindOfStream].size())
817 (*Parser->Stream_More)[KindOfStream][PosOfStream](Ztring().From_Local("Demux_InitBytes"), Info_Options)=__T("N NT");
818 }
819 break;

821 }

828 case 0x61 :
