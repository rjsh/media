71//---------------------------------------------------------------------------
72bool File_Rm::FileHeader_Begin()
73{
74 if (IsSub)
75 return true;
76
77 if (4>Buffer_Size)
78 return false;
79 if (Buffer[0]!=0x2E //".RMF"
80 || Buffer[1]!=0x52
81 || Buffer[2]!=0x4D
82 || Buffer[3]!=0x46)
83 {
84 Reject();
85 return false;
86 }
87
88 return true;
89}

95//---------------------------------------------------------------------------
96void File_Rm::Header_Parse()
97{
98 //Specific case
99 if (FromMKV_StreamType!=Stream_Max)
100 {
101 //Filling
102 Header_Fill_Code(0, __T("Real Media Header"));
103 Header_Fill_Size(Element_Size);
104 return;
105 }
106
107 //Parsing
108 int32u Name, Size;
109 Get_C4 (Name, "Name");
110
111 if (Name==Elements::RMMD)
112 {
113 Size=8; //Name + Size of the Metadata section (after Version) in bytes.
114 }
115 else if (Name==Elements::RJMD)
116 {
117 Skip_B4( "Version");
118 Get_B4 (Size, "Size");
119 Size+=8; //Name + Version + Size of the section (after Version) in bytes.
120 if (Element_Size>=12)
121 Element_Offset-=8; //Is valid, so we must keep Version and size in the stream
122 }
123 else if (Name==Elements::RMJE)
124 {
125 Size=12;
126 }
127 else if ((Name&0xFFFFFF00)==Elements::TAG)
128 {
129 Name=Elements::TAG;
130 Element_Offset-=4;
131 Size=0;
132 }
133 else
134 {
135 Get_B4 (Size, "Size");
136 }
137
138 //Filling
139 Header_Fill_Code(Name, Ztring().From_CC4(Name));
140 Header_Fill_Size(Size);
141}

143//---------------------------------------------------------------------------
144void File_Rm::Data_Parse()
145{
146 //Specific case
147 if (FromMKV_StreamType!=Stream_Max)
148 {
149 switch (FromMKV_StreamType)
150 {
151 case Stream_Video : MDPR_realvideo(); break;
152 case Stream_Audio : MDPR_realaudio(); break;
153 default : ;
154 }
155
156 Finish("RealMedia");
157 return;
158 }
159
160 //Parsing
161 DATA_BEGIN
162 ATOM( RMF)
163 ATOM(CONT)
164 LIST_SKIP(DATA)
165 ATOM(INDX)
166 ATOM(MDPR)
167 ATOM(PROP)
168 ATOM(RJMD)
169 ATOM(RMJE)
170 ATOM(RMMD)
171 ATOM(TAG)
172 DATA_END
173}

193//---------------------------------------------------------------------------
194void File_Rm::RMF()
195{
196 NAME_VERSION("Real Media Format");
197 INTEGRITY_VERSION(1);
198
199 //Parsing
200 if (Element_Size==4)
201 Skip_B2( "file_version"); //The version of the RealMedia file.
202 else
203 Skip_B4( "file_version"); //The version of the RealMedia file.
204 Skip_B4( "num_headers"); //The number of headers in the header section that follow the RealMedia File Header.
205
206 //Filling
207 Accept("RealMedia");
208 Fill(Stream_General, 0, General_Format, "RealMedia");
209}

211//---------------------------------------------------------------------------
212void File_Rm::CONT()
213{
214 NAME_VERSION("Content");
215 INTEGRITY_VERSION(0);
216
217 //Parsing
218 Ztring title, author, copyright, comment;
219 int16u title_len, author_len, copyright_len, comment_len;
220 Get_B2 (title_len, "title_len"); //The length of the title data in bytes.

222 Get_B2 (author_len, "author_len"); //The length of the author data in bytes.
223 Get_Local(author_len, author, "author"); //An array of ASCII characters that represents the author information for the RealMedia file.
224 Get_B2 (copyright_len, "copyright_len"); //The length of the copyright data in bytes.
225 Get_Local(copyright_len, copyright, "copyright"); //An array of ASCII characters that represents the copyright information for the RealMedia file.
226 Get_B2 (comment_len, "comment_len"); //The length of the comment data in bytes.
227 Get_Local(comment_len, comment, "comment"); //An array of ASCII characters that represents the comment information for the RealMedia file.

229 //Filling
230 Fill(Stream_General, 0, General_Title, title);
231 Fill(Stream_General, 0, General_Performer, author);
232 Fill(Stream_General, 0, General_Copyright, copyright);
233 Fill(Stream_General, 0, General_Comment, comment);

294void File_Rm::INDX()
295{
296 NAME_VERSION("INDeX");
297
298 //Parsing
299 int32u num_indices;
300 Get_B4 (num_indices, "num_indices"); //Number of index records in the index chunk.
301 Skip_B2( "stream_number"); //The stream number for which the index records in this index chunk are associated.
302 Skip_B4( "next_index_header"); //Offset from start of file to the next index chunk. This member enables RealMedia file format readers to find all the index chunks quickly. A value of zero for this member indicates there are no more index headers in this file.
303 for (int32u Pos=0; Pos<num_indices; Pos++)
304 {
305 Element_Begin1("index");
306 Get_B2 (Version, "object_version");
307 INTEGRITY_VERSION(0);
308 Element_Info1("Media_Packet_Header");
309 Skip_B4( "timestamp"); //The time stamp (in milliseconds) associated with this record.
310 Skip_B4( "offset"); //The offset from the start of the file at which this packet can be found.
311 Skip_B4( "packet_count_for_this_packet"); //The packet number of the packet for this record. This is the same number of packets that would have been seen had the file been played from the beginning to this point.
312 Element_End0();
313 }

316//---------------------------------------------------------------------------
317void File_Rm::MDPR()
318{
319 NAME_VERSION("MeDia PRoperties");
320 INTEGRITY_VERSION(0);
321
322 //Parsing
323 Ztring stream_name;
324 std::string mime_type;
325 int32u avg_bit_rate, start_time, duration, type_specific_len;
326 int16u stream_number;
327 int8u stream_name_size, mime_type_size;
328 Get_B2 (stream_number, "stream_number"); //Unique value that identifies a physical stream
329 Skip_B4( "max_bit_rate"); //The maximum bit rate required to deliver this stream over a network.
330 Get_B4 (avg_bit_rate, "avg_bit_rate"); //The average bit rate required to deliver this stream over a network.
331 Skip_B4( "max_packet_size"); //The largest packet size (in bytes) in the stream of media data.
332 Skip_B4( "avg_packet_size"); //The average packet size (in bytes) in the stream of media data.
333 Get_B4 (start_time, "start_time"); //The time offset in milliseconds to add to the time stamp of each packet in a physical stream.
334 Skip_B4( "preroll"); //The time offset in milliseconds to subtract from the time stamp of each packet in a physical stream.
335 Get_B4 (duration, "duration"); //The duration of the stream in milliseconds.
336 Get_B1 (stream_name_size, "stream_name_size"); //The length of the following stream_name member in bytes.
337 Get_Local(stream_name_size, stream_name, "stream_name"); //A nonunique alias or name for the stream.
338 Get_B1 (mime_type_size, "mime_type_size"); //The length of the following mime_type field in bytes.
339 Get_String(mime_type_size, mime_type, "mime_type"); //A nonunique MIME style type/subtype string for data associated with the stream.
340 Get_B4 (type_specific_len, "type_specific_len"); //The length of the following type_specific_data in bytes
341
342 //Parsing TypeSpecific
343 Element_Info1(mime_type.c_str());
344 MDPR_IsStream=true;
345 if (mime_type=="audio/x-pn-multirate-realaudio")
346 MDPR_IsStream=false; //What do we with this?
347 else if (mime_type=="audio/X-MP3-draft-00")
348 {
349 Stream_Prepare(Stream_Audio);
350 CodecID_Fill(Ztring(mime_type.c_str()), Stream_Audio, StreamPos_Last, InfoCodecID_Format_Real);
351 Fill(Stream_Audio, StreamPos_Last, Audio_Codec, "MPEG1AL3");
352 }
353 else if (mime_type=="audio/x-pn-realaudio")
354 MDPR_realaudio();
355 else if (mime_type=="audio/x-pn-realaudio-encrypted")
356 {
357 MDPR_realaudio();
358 Fill(Stream_Audio, StreamPos_Last, Audio_Encryption, "Y");
359 }
360 else if (mime_type=="audio/x-ralf-mpeg4")
361 {
362 Stream_Prepare(Stream_Audio);
363 CodecID_Fill(Ztring(mime_type.c_str()), Stream_Audio, StreamPos_Last, InfoCodecID_Format_Real);
364 Fill(Stream_Audio, StreamPos_Last, Audio_Codec, "ralf");
365 }
366 else if (mime_type=="audio/x-ralf-mpeg4-generic")
367 {
368 Stream_Prepare(Stream_Audio);
369 CodecID_Fill(Ztring(mime_type.c_str()), Stream_Audio, StreamPos_Last, InfoCodecID_Format_Real);
370 Fill(Stream_Audio, StreamPos_Last, Audio_Codec, "ralf");
371 }
372 else if (mime_type.find("audio/")==0)
373 Stream_Prepare(Stream_Audio);
374 else if (mime_type=="video/text")
375 Stream_Prepare(Stream_Text);
376 else if (mime_type=="video/x-pn-multirate-realvideo")
377 MDPR_IsStream=false; //What do we with this?
378 else if (mime_type=="video/x-pn-realvideo")
379 MDPR_realvideo();
380 else if (mime_type=="video/x-pn-realvideo-encrypted")
381 {
382 MDPR_realvideo();
383 Fill(Stream_Video, StreamPos_Last, Video_Encryption, "Y");
384 }
385 else if (mime_type.find("video/")==0)
386 Stream_Prepare(Stream_Video);
387 else if (mime_type=="logical-audio/x-pn-multirate-realaudio")
388 MDPR_IsStream=false; //What do we with this?
389 else if (mime_type.find("logical-audio/")==0)
390 MDPR_IsStream=false; //What do we with this?
391 else if (mime_type=="logical-fileinfo")
392 MDPR_fileinfo();
393 //else if (mime_type=="logical-video/x-pn-multirate-realvideo")
394 // MDPR_IsStream=false; //What do we with this?
395 //else if (mime_type.find("logical-video/")==0)
396 // MDPR_IsStream=false; //What do we with this?
397 else
398 MDPR_IsStream=false;
399
400 //Filling
401 FILLING_BEGIN();
402 if (MDPR_IsStream)
403 {
404 Fill(StreamKind_Last, StreamPos_Last, General_ID, stream_number);
405 Fill(StreamKind_Last, StreamPos_Last, Fill_Parameter(StreamKind_Last, Generic_BitRate), avg_bit_rate, 10, true);
406 //Fill(StreamKind_Last, StreamPos_Last, Fill_Parameter(StreamKind_Last, Generic_Delay), start_time);
407 //Fill(StreamKind_Last, StreamPos_Last, Fill_Parameter(StreamKind_Last, Generic_Delay_Source), "Container");
408 Fill(StreamKind_Last, StreamPos_Last, Fill_Parameter(StreamKind_Last, Generic_Duration), duration);
409 }
410 FILLING_END();
411}

413//---------------------------------------------------------------------------
414void File_Rm::MDPR_realvideo()
415{
416 //Parsing
417 int32u Codec;
418 int16u Width, Height, FrameRate;
419 Skip_B4( "Size");
420 Skip_C4( "FCC");
421 Get_C4 (Codec, "Compression");
422 Get_B2 (Width, "Width");
423 Get_B2 (Height, "Height");
424 Skip_B2( "bpp"); //Do not use it
425 Skip_B4( "Unknown");
426 Get_B2 (FrameRate, "fps");
427 Skip_B2( "Unknown");
428 Skip_C4( "Type1");
429 Skip_C4( "Type2");
430
431 //Filling
432 if (!Status[IsAccepted])
433 Accept("RealMedia"); //Is subs
434
435 Stream_Prepare(Stream_Video);
436 if (FromMKV_StreamType==Stream_Max) //Using the one from the container
437 CodecID_Fill(Ztring().From_CC4(Codec), Stream_Video, StreamPos_Last, InfoCodecID_Format_Real);
438 Fill(Stream_Video, StreamPos_Last, Video_Codec, Ztring().From_CC4(Codec));
439 Fill(Stream_Video, StreamPos_Last, Video_Width, Width); //Width
440 Fill(Stream_Video, StreamPos_Last, Video_Height, Height); //Height
441 switch (FrameRate)
442 {
443 case 23 : Fill(Stream_Video, StreamPos_Last, Video_FrameRate, ((float)24)*1000/1001); break; //Hard-coded frame rate?
444 case 29 : Fill(Stream_Video, StreamPos_Last, Video_FrameRate, ((float)30)*1000/1001); break; //Hard-coded frame rate?
445 default : Fill(Stream_Video, StreamPos_Last, Video_FrameRate, (float)FrameRate);
446 }
447}

449//---------------------------------------------------------------------------
450void File_Rm::MDPR_realaudio()
451{
452 //Parsing
453 Ztring FourCC3="lpcJ"; //description of this codec : http://focus.ti.com/lit/an/spra136/spra136.pdf http://en.wikipedia.org/wiki/VSELP
454 Ztring FourCC4;
455 int32u FourCC5=0, BytesPerMinute=0;
456 int16u Version, Samplerate=8000, Samplesize=16, Channels=0;
457 Skip_C4( "Header signature");
458 Get_B2 (Version, "Version");
459 INTEGRITY_VERSION(5);
460 if (Version==3)
461 {
462 Ztring title, author, copyright, comment;
463 int32u length;
464 int8u title_len, author_len, copyright_len, comment_len;
465 Skip_B2( "Header size"); //Header size, after this tag.
466 Get_B2 (Channels, "Channels");
467 Skip_B4( "Uknown");
468 Skip_B4( "Uknown");
469 Skip_B4( "Data size");
470 Get_B1 (title_len, "title_len"); //The length of the title data in bytes.
471 Get_Local(title_len, title, "title"); //An array of ASCII characters that represents the title information for the RealMedia file.
472 Get_B1 (author_len, "author_len"); //The length of the author data in bytes.
473 Get_Local(author_len, author, "author"); //An array of ASCII characters that represents the author information for the RealMedia file.
474 Get_B1 (copyright_len, "copyright_len"); //The length of the copyright data in bytes.
475 Get_Local(copyright_len, copyright, "copyright"); //An array of ASCII characters that represents the copyright information for the RealMedia file.
476 Get_B1 (comment_len, "comment_len"); //The length of the comment data in bytes.
477 Get_Local(comment_len, comment, "comment"); //An array of ASCII characters that represents the comment information for the RealMedia file.
478 if (Element_Offset<Element_Size) //Optional
479 {
480 Skip_B1( "Uknown");
481 Get_B4 (length, "Fourcc string length");
482 Get_Local(length, FourCC3, "Fourcc string");
483 }
484
485 //Filling
486 Fill(Stream_General, 0, General_Duration, title);
487 Fill(Stream_General, 0, General_Performer, author);
488 Fill(Stream_General, 0, General_Copyright, copyright);
489 Fill(Stream_General, 0, General_Comment, comment);
490 }
491 if (Version==4 || Version==5)
492 {
493 Skip_B2( "Unused");
494 Skip_C4( "ra signature");
495 Skip_B4( "AudioFileSize");
496 Skip_B2( "Version2");
497 Skip_B4( "Header size");
498 Skip_B2( "Codec flavor");
499 Skip_B4( "Coded frame size");
500 Skip_B4( "AudioBytes");
501 Get_B4 (BytesPerMinute, "BytesPerMinute");
502 Skip_B4( "Unknown");
503 Skip_B2( "Sub packet h");
504 Skip_B2( "Frame size");
505 Skip_B2( "Subpacket size");
506 Skip_B2( "Unknown");
507 }
508 if (Version==5)
509 {
510 Skip_B2( "Unknown");
511 Skip_B2( "Unknown");
512 Skip_B2( "Unknown");
513 }
514 if (Version==4 || Version==5)
515 {
516 Get_B2 (Samplerate, "Samplerate");
517 Skip_B2( "Unknown");
518 Get_B2 (Samplesize, "Samplesize");
519 Get_B2 (Channels, "Channels");
520 }
521 if (Version==4)
522 {
523 int8u length;
524 Get_B1 (length, "Interleaver ID string lengt");
525 Skip_Local(length, "Interleaver ID string");
526 Get_B1 (length, "FourCC string lengt");
527 Get_Local(length, FourCC4, "FourCC string");
528 }
529 if (Version==5)
530 {
531 Skip_C4( "Interleaver ID");
532 Get_C4 (FourCC5, "FourCC");
533 }
534 if (Version==4 || Version==5)
535 {
536 Skip_B1( "Unknown");
537 Skip_B1( "Unknown");
538 Skip_B1( "Unknown");
539 }
540 if (Version==5)
541 {
542 Skip_B1( "Unknown");
543 }
544 if (Version==4 || Version==5)
545 {
546 int32u length;
547 Get_B4 (length, "Codec extradata length");
548 Skip_XX(length, "Codec extradata");
549 }
550
551 //Filling
552 if (!Status[IsAccepted])
553 Accept("RealMedia"); //Is subs
554
555 Stream_Prepare(Stream_Audio);
556 if (Version==3)
557 {
558 if (FromMKV_StreamType==Stream_Max) //Using the one from the container
559 CodecID_Fill(FourCC3, Stream_Audio, StreamPos_Last, InfoCodecID_Format_Real);
560 Fill(Stream_Audio, StreamPos_Last, Audio_Codec, FourCC3);
561 }
562 if (Version==4)
563 {
564 if (FromMKV_StreamType==Stream_Max) //Using the one from the container
565 CodecID_Fill(FourCC4, Stream_Audio, StreamPos_Last, InfoCodecID_Format_Real);
566 Fill(Stream_Audio, StreamPos_Last, Audio_Codec, FourCC4);
567 }
568 if (Version==5)
569 {
570 if (FromMKV_StreamType==Stream_Max) //Using the one from the container
571 CodecID_Fill(Ztring().From_CC4(FourCC5), Stream_Audio, StreamPos_Last, InfoCodecID_Format_Real);
572 Fill(Stream_Audio, StreamPos_Last, Audio_Codec, Ztring().From_CC4(FourCC5));
573 }
574 Fill(Stream_Audio, StreamPos_Last, Audio_SamplingRate, Samplerate);
575 Fill(Stream_Audio, StreamPos_Last, Audio_BitDepth, Samplesize);
576 Fill(Stream_Audio, StreamPos_Last, Audio_Channel_s_, Channels);
577 if (BytesPerMinute)
578 Fill(Stream_Audio, StreamPos_Last, Audio_BitRate, BytesPerMinute*8/60, 10, true);
579}

581//---------------------------------------------------------------------------
582void File_Rm::MDPR_fileinfo()
583{
584 MDPR_IsStream=false;
585
586 //Parsing
587 int16u Version, num_physical_streams, num_rules, num_properties;
588 Skip_B4( "size");
589 Get_B2 (Version, "object_version");
590 INTEGRITY_VERSION(0);
591 Get_B2 (num_physical_streams, "num_physical_streams"); //The number of physical streams that make up this logical stream.
592 for (int16u Pos=0; Pos<num_physical_streams; Pos++)
593 {
594 Skip_B2( "physical_stream_numbers"); //The list of physical stream numbers that comprise this logical stream.
595 Skip_B4( "data_offsets"); //The list of data offsets indicating the start of the data section for each physical stream.
596 }
597 Get_B2 (num_rules, "num_rules"); //The number of ASM rules for the logical stream. Each physical stream in the logical stream has at least one ASM rule associated with it or it will never get played. The mapping of ASM rule numbers to physical stream numbers is stored in a list immediately following this member. These physical stream numbers refer to the stream_number field found in the Media Properties Object for each physical stream belonging to this logical stream.
598 for (int16u Pos=0; Pos<num_physical_streams; Pos++)
599 Skip_B2( "rule_to_physical_stream_number_map"); //The list of physical stream numbers that map to each rule. Each entry in the map corresponds to a 0-based rule number. The value in each entry is set to the physical stream number for the rule.
600 Get_B2 (num_properties, "num_properties"); //The number of NameValueProperty structures contained in this structure. These name/value structures can be used to identify properties of this logical stream (for example, language).
601
602 //Parsing
603 for (int16u Pos=0; Pos<num_properties; Pos++)
604 {
605 Element_Begin1("property");
606 std::string name;
607 int32u size, type;
608 int16u value_length;
609 int8u name_length;
610 Peek_B4(size);
611 Skip_B4( "size");
612 Skip_B2( "object_version");
613 Get_B1 (name_length, "name_length"); //The length of the name data.
614 Get_String(name_length, name, "name"); //The name string data.
615 Get_B4 (type, "type"); //The type of the value data.
616 Get_B2 (value_length, "value_length"); //value_length
617 switch (type)
618 {
619 case 0 : //Unsigned integer
620 Skip_B4( "value_data"); break; //unsigned integer
621 case 2 : //String
622 Skip_Local(value_length, "value_data"); break; //string
623 default : Skip_XX(value_length, "unknown");
624 }
625 Element_End0();
626 }
627}

629//---------------------------------------------------------------------------
630void File_Rm::PROP()
631{
632 NAME_VERSION("PROPerties");
633 INTEGRITY_VERSION(0);
634
635 //Parsing
636 int32u avg_bit_rate, duration;
637 int16u flags;
638 Skip_B4( "max_bit_rate"); //The maximum bit rate required to deliver this file over a network.

671//---------------------------------------------------------------------------
672void File_Rm::RJMD_property(std::string Name)
673{
674 //Element_Name("Property");
675
676 //Parsing
677 Ztring value;
678 std::string name;
679 int32u type, flags, num_subproperties, name_length, value_length;
680 Element_Begin1("MetadataProperty");
681 Skip_B4( "size");
682 Get_B4 (type, "type");
683 Get_B4 (flags, "flags");
684 Skip_Flags(flags, 0, "readonly"); //Read only, cannot be modified.
685 Skip_Flags(flags, 1, "private"); //Private, do not expose to users.
686 Skip_Flags(flags, 2, "type_dexcriptor"); //Type descriptor used to further define type of value.
687 Skip_B4( "value_offset"); //The offset to the value_length , relative to the beginning of the MetadataProperty structure.
688 Skip_B4( "subproperties_offset"); //The offset to the subproperties_list , relative to the beginning of the MetadataProperty structure.
689 Get_B4 (num_subproperties, "num_subproperties"); //The number of subproperties for this MetadataProperty structure.
690 Get_B4 (name_length, "name_length"); //The length of the name data, including the null-terminator.
691 Get_String(name_length, name, "name"); //The name of the property (string data).
692 Get_B4 (value_length, "value_length"); //The length of the value data.
693 switch(type)
694 {
695 case 0x00 : //Nothing
696 Skip_XX(value_length, "Junk");
697 break;
698 case 0x01 : //String (text).
699 Get_Local(value_length, value, "value"); //String.
700 break;
701 case 0x02 : //Separated list of strings, separator specified as sub-property/type descriptor.
702 Get_Local(value_length, value, "value"); //String.
703 break;
704 case 0x03 : //Boolean flageither 1 byte or 4 bytes, check size value.
705 switch(value_length)
706 {
707 case 1 : {
708 int8u valueI;
709 Get_L1(valueI, "value"); //1-byte boolean.
710 value.From_Number(valueI);
711 }
712 break;
713 case 4 : {
714 int32u valueI;
715 Get_L4(valueI, "value"); //4-byte boolean.
716 value.From_Number(valueI);
717 }
718 break;
719 default: Skip_XX(value_length, "Unknown");
720 }
721 break;
722 case 0x04 : //Four-byte integer.
723 {
724 int32u valueI;
725 Get_L4(valueI, "value");
726 value.From_Number(valueI);
727 }
728 break;
729 case 0x05 : //Byte stream.
730 Skip_XX(value_length, "Byte stream");
731 break;
732 case 0x06 : //String (URL).
733 Get_Local(value_length, value, "value"); //String.
734 break;
735 case 0x07 : //String representation of the date in the form: YYYYmmDDHHMMSS (m = month, M = minutes).
736 Get_Local(value_length, value, "value"); //String.
737 break;
738 case 0x08 : //String (file name)
739 Get_Local(value_length, value, "value"); //String.
740 break;
741 case 0x09 : //This property has subproperties, but its own value is empty.
742 Skip_XX(value_length, "junk");
743 break;
744 case 0x0A : //Large buffer of data, use sub-properties/type descriptors to identify mime-type.
745 Skip_XX(value_length, "data");
746 break;
747 default : //Unknown
748 Skip_XX(value_length, "unknown");
749 break;
750 }
751
752 //Filling
753 if (!Name.empty())
754 Name+='/';
755 Name+=name;
756 if (Name!="Track/Comments/DataSize"
757 && Name!="Track/Comments/MimeType"
758 )
759 Fill(Stream_General, 0, Name.c_str(), value);
760
761 //Parsing
762 for (int32u Pos=0; Pos<num_subproperties; Pos++)
763 {
764 Element_Begin1("PropListEntry");
765 Skip_B4( "offset"); //The offset for this indexed sub-property, relative to the beginning of the containing MetadataProperty.
766 Skip_B4( "num_props_for_name"); //The number of sub-properties that share the same name. For example, a lyrics property could have multiple versions as differentiated by the language sub-property type descriptor.
767 Element_End0();
768 }
769 for (int32u Pos=0; Pos<num_subproperties; Pos++)
770 {
771 RJMD_property(Name);
772 }
773
774 Element_End0();
775}
