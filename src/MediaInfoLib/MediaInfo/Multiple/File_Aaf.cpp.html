41const char* AAf_tagSTGTY (int8u tagSTGTY)
42{
43 switch (tagSTGTY)
44 {
45 case 0 : return "unknown";
46 case 1 : return "storage";
47 case 2 : return "stream";
48 case 3 : return "ILockBytes";
49 case 4 : return "IPropertyStorage";
50 case 5 : return "root";
51 default: return "";
52 }
53}

56{
57 switch (tagDECOLOR)
58 {
59 case 0 : return "red";
60 case 1 : return "black";
61 default: return "";
62 }
63}

123//---------------------------------------------------------------------------
124bool File_Aaf::FileHeader_Begin()
125{
126 if (File_Size<0x100)
127 {
128 Reject("Aaf");
129 return false;
130 }
131
132 //Element_Size
133 if (Buffer_Size<0x18)
134 return false; //Must wait for more data
135
136 if (Buffer[ 0x0]!=0xD0
137 || Buffer[ 0x1]!=0xCF
138 || Buffer[ 0x2]!=0x11
139 || Buffer[ 0x3]!=0xE0
140 || Buffer[ 0x4]!=0xA1
141 || Buffer[ 0x5]!=0xB1
142 || Buffer[ 0x6]!=0x1A
143 || Buffer[ 0x7]!=0xE1
144 || Buffer[ 0x8]!=0x41
145 || Buffer[ 0x9]!=0x41
146 || Buffer[ 0xA]!=0x46
147 || Buffer[ 0xB]!=0x42
148 || Buffer[ 0xC]!=0x0D
149 || Buffer[ 0xD]!=0x00
150 || Buffer[ 0xE]!=0x4F
151 || Buffer[ 0xF]!=0x4D
152 || Buffer[0x10]!=0x06
153 || Buffer[0x11]!=0x0E
154 || Buffer[0x12]!=0x2B
155 || Buffer[0x13]!=0x34
156 || Buffer[0x14]!=0x01
157 || Buffer[0x15]!=0x01
158 || Buffer[0x16]!=0x01
159 || Buffer[0x17]!=0xFF)
160 {
161 Reject("Aaf");
162 return false;
163 }
164
165 //Element_Size
166 if (Buffer_Size<File_Size)
167 return false; //Must wait for more data
168
169 //Accept the file
170 Accept("Aaf");
171 Fill(Stream_General, 0, General_Format, "AAF");
172
173 Step=Step_None;
174 ReferenceFiles=new File__ReferenceFilesHelper(this, Config);
175
176 //All should be OK...
177 return true;
178}

180//---------------------------------------------------------------------------
181void File_Aaf::Read_Buffer_Continue()
182{
183 if (File_Offset || Buffer_Offset)
184 return;
185
186 //Parsing
187 Element_Begin1("Header");
188 int32u csectFat;
189 int16u DllVersion, ByteOrder;
190 Skip_B8( "abSig");
191 Skip_B16( "clsid");
192 Skip_L2( "MinorVersion");
193 Get_L2 (DllVersion, "DllVersion");
194 Get_L2 (ByteOrder, "ByteOrder");
195 Get_L2 (SectorShift, "SectorShift");
196 Get_L2 (MiniSectorShift, "MiniSectorShift");
197 Skip_L2( "Reserved");
198 Skip_L4( "Reserved");
199 Skip_L4( "csectDir");
200 Get_L4 (csectFat, "csectFat");
201 Get_L4 (sectDirStart, "sectDirStart");
202 Skip_L4( "signature");
203 Get_L4 (MiniSectorCutoff, "MiniSectorCutoff");
204 Get_L4 (sectMiniFatStart, "sectMiniFatStart");
205 Skip_L4( "csectMiniFat");
206 Skip_L4( "sectDifStart");
207 Skip_L4( "sectDif");
208 Element_Begin1("sectFat");
209 for (int16u Pos=0; Pos<(csectFat>109?109:csectFat); Pos++)
210 {
211 int32u sectFat;
212 Get_L4 (sectFat, "sectFat");
213 sectsFat.push_back(sectFat);
214 }
215 if (csectFat<109)
216 Skip_XX((109-csectFat)*4, "unused sectsFat");
217 Element_End();
218 Element_End();
219
220 FILLING_BEGIN();
221 Fill("Aaf");
222 Step=Step_Fat;
223 sectsFat_Pos=0;
224 if (sectsFat.empty())
225 {
226 Finish();
227 }
228 else
229 GoTo((1+sectsFat[0])*(1<<SectorShift));
230 FILLING_END();
231}

237//---------------------------------------------------------------------------
238void File_Aaf::Header_Parse()
239{
240 switch (Step)
241 {
242 case Step_Fat : Header_Fill_Code(0, "FAT");
243 Header_Fill_Size(((int64u)1) << SectorShift);
244 break;
245 case Step_MiniFat : Header_Fill_Code(0, "MiniFAT");
246 Header_Fill_Size(((int64u)1) << SectorShift);
247 break;
248 case Step_Directory : Header_Fill_Code(0, "Directory");
249 Header_Fill_Size(((int64u)1) << SectorShift);
250 break;
251 case Step_Stream : Header_Fill_Code(0, "Stream");
252 Header_Fill_Size(((int64u)1) << (Streams[0]->Size<MiniSectorCutoff ? MiniSectorShift : SectorShift));
253 break;
254 default : ;
255 }
256}

258//---------------------------------------------------------------------------
259void File_Aaf::Data_Parse()
260{
261 //Parsing
262 switch (Step)
263 {
264 case Step_Fat : Fat(); return;
265 case Step_MiniFat : MiniFat(); break;
266 case Step_Directory : Directory(); break;
267 case Step_Stream : StreamElement(); return;
268 default : Skip_XX(Element_Size, "Unknown");
269 }
270
271 size_t Pointers_Pos=(size_t)((File_Offset+Buffer_Offset)>>SectorShift)-1;
272 if (Pointers_Pos<Pointers.size())
273 {
274 if (Pointers[Pointers_Pos]<0xFFFFFFF0)
275 GoTo((1+Pointers[Pointers_Pos])*(1<<SectorShift));
276 else if (Step==Step_MiniFat)
277 {
278 Step=Step_Directory;
279 Directory_Pos=0;
280 GoTo((1+sectDirStart)*(1<<SectorShift));
281 }
282 else if (Step==Step_Directory)
283 {
284 Step=Step_Stream;
285 if (Streams.empty())
286 Finish();
287 else
288 {
289 Streams_Pos=0;
290 Streams_Pos2=0;
291 GoTo(Streams[0]->StreamOffsets[0]);
292 }
293 }
294 else
295 Finish();
296 }
297 else
298 Finish();
299}

301//---------------------------------------------------------------------------
302void File_Aaf::Fat()
303{
304 //Parsing
305 while (Element_Offset<Element_Size)
306 {
307 int32u Pointer;
308 Get_L4 (Pointer, "Pointer"); Param_Info1(Ztring::ToZtring(Pointers.size()));
309 //Pointer=LittleEndian2int32u(Buffer+Buffer_Offset+(size_t)Element_Offset);
310 //Element_Offset+=4;
311 Pointers.push_back(Pointer);
312 }
313
314 //Next FAT sector or next step
315 sectsFat_Pos++;
316 if (sectsFat_Pos<sectsFat.size())
317 GoTo((1+sectsFat[sectsFat_Pos])*(1<<SectorShift));
318 else
319 {
320 Step=Step_MiniFat;
321 GoTo((1+sectMiniFatStart)*(1<<SectorShift));
322 }
323}

347//---------------------------------------------------------------------------
348void File_Aaf::Directory_Entry()
349{
350 //Parsing
351 Element_Begin1("Directory entry");
352 Element_Info1(Directory_Pos);
353 Ztring ab;
354 int64u Size;
355 int32u SectStart;
356 int8u mse;
357 Get_UTF16L(64, ab, "ab"); Element_Info1(ab);
358 Skip_L2( "cb");
359 Get_L1 (mse, "mse"); Element_Info1(AAf_tagSTGTY(mse));
360 Info_L1(flags, "flags"); Element_Info1(AAf_tagDECOLOR(flags));
361 Skip_L4( "LeftSib SID");
362 Skip_L4( "RightSib SID");
363 Skip_L4( mse==2?"0":"Child SID"); //Zero if stream
364 Skip_L16( mse==2?"0":"clsId");//Zero if stream
365 Skip_L4( "UserFlags");
366 Info_L8(CreationTime, mse==2?"0":"Create time"); if (mse!=2) {Param_Info1(CreationTime?Ztring().Date_From_Milliseconds_1601(CreationTime/10000):Ztring());} //Zero if stream
367 Info_L8(ModificationTime, mse==2?"0":"Modify time"); if (mse!=2) {Param_Info1(CreationTime?Ztring().Date_From_Milliseconds_1601(ModificationTime/10000):Ztring());} //Zero if stream
368 Get_L4 (SectStart, mse==1?"0":"SectStart"); //Zero if storage
369 if (SectorShift<=9)
370 {
371 int32u Size32;
372 Get_L4 (Size32, mse==1?"0":"Size"); //Zero if storage
373 Skip_L4( "PropType");
374 Size=Size32;
375
376 }
377 else
378 {
379 Get_L8 (Size, mse==1?"0":"Size"); //Zero if storage
380 }
381
382 if (mse==5 && Size) //If root
383 {
384 //Building sectMiniFats_FatPointers
385 int32u Pointers_Pos=SectStart;
386 while (Pointers_Pos<Pointers.size())
387 {
388 Param_Info1(Ztring::ToZtring(Pointers_Pos<<SectorShift));
389 sectsMiniStream.push_back(Pointers_Pos);
390 Pointers_Pos=Pointers[Pointers_Pos];
391 }
392 }
393 else if (mse==2 && Size) //If stream
394 {
395 Param_Info1("StreamOffset");
396 stream* Stream = new stream(
397 ab,
398 Directory_Pos,
399 Size
400 );
401 if (Size<MiniSectorCutoff) //MiniFAT
402 {
403 int32u Pointers_Pos=SectStart;
404 while (Pointers_Pos<MiniPointers.size())
405 {
406 int32u SectPos=Pointers_Pos>>(SectorShift-MiniSectorShift);
407 int32u MiniSectPos=Pointers_Pos&((((size_t)1)<<(SectorShift-MiniSectorShift))-1);
408 Stream->StreamOffsets.push_back(((1+sectsMiniStream[SectPos])<<SectorShift)+(MiniSectPos<<MiniSectorShift));
409 Param_Info1(Ztring::ToZtring(((1+sectsMiniStream[SectPos])<<SectorShift)+(MiniSectPos<<MiniSectorShift)));
410 Pointers_Pos=MiniPointers[Pointers_Pos];
411 }
412 }
413 else //FAT
414 {
415 int32u Pointers_Pos=SectStart;
416 while (Pointers_Pos<Pointers.size())
417 {
418 Stream->StreamOffsets.push_back((1+Pointers_Pos)<<SectorShift);
419 Param_Info1(Ztring::ToZtring((1+Pointers_Pos)<<SectorShift));
420 Pointers_Pos=Pointers[Pointers_Pos];
421 }
422 }
423 Streams.push_back(Stream);
424 }
425 Element_End0();
426
427 Directory_Pos++;
428}

430//---------------------------------------------------------------------------
431void File_Aaf::StreamElement()
432{
433 if (Streams_Pos>=Streams.size() || Streams[Streams_Pos]->Size>=0x1000000) //TODO: more serious test about size
434 return; //Incoherancy
435
436 //Saving data
437 if (Streams[Streams_Pos]->StreamOffsets.size()!=1)
438 {
439 Skip_XX(Element_Size, "Stream data");
440 int16u Shift=(Streams[Streams_Pos]->Size<MiniSectorCutoff?MiniSectorShift:SectorShift);
441 if (Streams[Streams_Pos]->Buffer==NULL)
442 Streams[Streams_Pos]->Buffer=new int8u[(size_t)((1+(Streams[Streams_Pos]->Size>>Shift))<<Shift)];
443 memcpy(Streams[Streams_Pos]->Buffer+Streams_Pos2*(((int64u)1)<<Shift), Buffer+Buffer_Offset, (size_t)Element_Size);
444 }
445
446 //Next Element
447 Streams_Pos2++;
448 if (Streams_Pos2>=Streams[Streams_Pos]->StreamOffsets.size())
449 {
450 Element_Offset=0;
451 StreamElement_Parse();
452 Streams_Pos++;
453 Streams_Pos2=0;
454 }
455 if (Streams_Pos<Streams.size())
456 GoTo(Streams[Streams_Pos]->StreamOffsets[Streams_Pos2]);
457 else
458 Finish();
459}

461//---------------------------------------------------------------------------
462void File_Aaf::StreamElement_Parse()
463{
464 //Searching emulation_prevention_three_byte
465 const int8u* Save_Buffer=Buffer;
466 int64u Save_File_Offset=File_Offset;
467 size_t Save_Buffer_Offset=Buffer_Offset;
468 int64u Save_Element_Size=Element_Size;
469
470 if (Streams[Streams_Pos]->Buffer)
471 {
472 //We must change the buffer for keeping out
473 Element_Size=Streams[Streams_Pos]->Size;
474 File_Offset=Streams[Streams_Pos]->StreamOffsets[0];
475 Buffer_Offset=0;
476 Buffer=Streams[Streams_Pos]->Buffer;
477 }
478
479 //Parsing
480 Element_Info1(Streams[Streams_Pos]->Directory_Pos);
481 Element_Info1(Streams[Streams_Pos]->Name);
482 int16u Count;
483 Skip_L2( "0x204C?");
484 Get_L2 (Count, "Count");
485 vector<int16u> Sizes;
486 vector<int16u> Keys;
487 for (int16u Pos=0; Pos<Count; Pos++)
488 {
489 int16u Key, Size;
490 Get_L2 (Key, "Key");
491 Skip_L2( "Flags?");
492 Get_L2 (Size, "Size");
493 Sizes.push_back(Size);
494 Keys.push_back(Key);
495 }
496
497 #define ELEMENT(_ELEMENT, _NAME) \
498
499 for (int16u Pos=0; Pos<Count; Pos++)
500 {
501 Element_Begin0();
502 xxxSize=Sizes[Pos];
503 switch (Keys[Pos])
504 {
505 case 0x0001 : Element_Name("MetaDictionary"); MetaDictionary(); break;
506 case 0x0002 : Element_Name("Header"); Header(); break;
507 case 0x0003 : Element_Name("ClassDefinitions"); ClassDefinitions(); break;
508 case 0x0004 : Element_Name("TypeDefinitions"); TypeDefinitions(); break;
509 case 0x0005 : Element_Name("Identification"); Identification(); break;
510 case 0x0006 : Element_Name("Name"); Name(); break;
511 case 0x0007 : Element_Name("MetaDefinition"); MetaDefinition(); break;
512 case 0x0008 : Element_Name("ParentClass"); ParentClass(); break;
513 case 0x0009 : Element_Name("Properties"); Properties(); break;
514 case 0x000A : Element_Name("IsConcrete"); IsConcrete(); break;
515 case 0x000B : Element_Name("Type"); Type(); break;
516 case 0x000C : Element_Name("IsOptional"); IsOptional(); break;
517 case 0x000D : Element_Name("LocalIdentification"); LocalIdentification(); break;
518 case 0x000E : Element_Name("IsUniqueIdentifier"); IsUniqueIdentifier(); break;
519 case 0x000F : Element_Name("Size"); Size(); break;
520 case 0x3D02 : Element_Name("Locked"); Locked(); break;
521 case 0x4001 : Element_Name("NetworkLocator"); NetworkLocator(); break;
522 default : Skip_XX(xxxSize, "Unknown");
523 }
524 Element_End0();
525 }
526
527 //
528 if (Streams[Streams_Pos]->Buffer)
529 {
530 //We must change the buffer for keeping out
531 Element_Size=Save_Element_Size;
532 File_Offset=Save_File_Offset;
533 Buffer_Offset=Save_Buffer_Offset;
534 delete[] Buffer; Buffer=Save_Buffer;
535 Element_Offset=Element_Size;
536 }
537}
