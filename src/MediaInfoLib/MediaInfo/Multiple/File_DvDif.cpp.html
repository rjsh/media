306//---------------------------------------------------------------------------
307void File_DvDif::Streams_Fill()
308{
309 Fill(Stream_General, 0, General_Format, "DV", Unlimited, true, true);
310
311 Stream_Prepare(Stream_Video);
312 Fill(Stream_Video, 0, Video_Format, "DV");
313 Fill(Stream_Video, 0, Video_Codec, "DV");
314 Fill(Stream_Video, 0, Video_Standard, system?"PAL":"NTSC");
315 Fill(Stream_Video, 0, Video_BitDepth, 8);
316 bool IsHd=false;
317 float FrameRate_Multiplicator=1;
318 switch (video_source_stype)
319 {
320 case 0x00 :
321 case 0x04 :
322 Fill(Stream_Video, 0, Video_Width, 720);
323 Fill(Stream_Video, 0, Video_Height, system?576:480);
324 break;
325 case 0x14 :
326 case 0x15 :
327 Fill(Stream_Video, 0, Video_Width, system?1440:1280);
328 Fill(Stream_Video, 0, Video_Height, video_source_stype==0x14?1080:1035);
329 IsHd=true;
330 break;
331 case 0x18 :
332 Fill(Stream_Video, 0, Video_Width, 960);
333 Fill(Stream_Video, 0, Video_Height, 720);
334 FrameRate_Multiplicator=2;
335 IsHd=true;
336 break;
337 default : ;
338 }
339 Fill(Stream_Video, 0, Video_FrameRate, (system?25.000:29.970)*FrameRate_Multiplicator);
340 Fill(Stream_Video, 0, Video_FrameRate_Mode, "CFR");
341 if (video_sourcecontrol_IsParsed)
342 {
343 if (FSC_WasSet && FSP_WasNotSet)
344 {
345 switch (video_source_stype)
346 {
347 case 0x14 :
348 case 0x15 :
349 Fill(Stream_Video, 0, Video_ScanType, "Interlaced");
350 if (FieldOrder_FF)
351 Fill(Stream_Video, 0, Video_ScanOrder, FieldOrder_FS?"TFF":"BFF");
352 else
353 Fill(Stream_Video, 0, Video_ScanOrder, FieldOrder_FS?"Top field only":"Bottom field only");
354 Fill(Stream_Video, 0, Video_Interlacement, "Interlaced");
355 break;
356 case 0x18 :
357 Fill(Stream_Video, 0, Video_ScanType, "Progressive");
358 Fill(Stream_Video, 0, Video_Interlacement, "Progressive");
359 break;
360 default : ;
361 }
362 }
363 else if (Interlaced)
364 {
365 Fill(Stream_Video, 0, Video_ScanType, "Interlaced");
366 if (FieldOrder_FF)
367 Fill(Stream_Video, 0, Video_ScanOrder, FieldOrder_FS?"BFF":"TFF");
368 else
369 Fill(Stream_Video, 0, Video_ScanOrder, FieldOrder_FS?"Bottom field only":"Top field only");
370 Fill(Stream_Video, 0, Video_Interlacement, "Interlaced");
371 }
372 else
373 {
374 Fill(Stream_Video, 0, Video_ScanType, Interlaced?"Interlaced":"Progressive");
375 Fill(Stream_Video, 0, Video_Interlacement, Interlaced?"Interlaced":"PFF");
376 }
377 switch (aspect)
378 {
379 case 0 :
380 case 4 : Fill(Stream_Video, 0, Video_DisplayAspectRatio, 4.0/3.0, 3, true); break;
381 case 1 :
382 case 2 : Fill(Stream_Video, 0, Video_DisplayAspectRatio, 16.0/9.0, 3, true); break;
383 case 7 : switch (ssyb_AP3)
384 {
385 case 0 : Fill(Stream_Video, 0, Video_DisplayAspectRatio, 16.0/9.0, 3, true); break;
386 case 7 : Fill(Stream_Video, 0, Video_DisplayAspectRatio, 4.0/3.0, 3, true); break;
387 default: ; //No indication of aspect ratio?
388 }
389 default: ;
390 }
391 }
392
393 if (!FSC_WasSet) //Original DV 25 Mbps
394 {
395 if (system==false) //NTSC
396 {
397 switch (video_source_stype)
398 {
399 case 0 : Fill(Stream_Video, 0, Video_Colorimetry, "4:1:1"); break; //NTSC 25 Mbps
400 default : ;
401 }
402 }
403 else //PAL
404 {
405 switch (video_source_stype)
406 {
407 case 0 : if (APT==0)
408 Fill(Stream_Video, 0, Video_Colorimetry, "4:2:0"); //PAL 25 Mbps (IEC 61834)
409 else
410 Fill(Stream_Video, 0, Video_Colorimetry, "4:1:1"); //PAL 25 Mbps (SMPTE 314M)
411 break;
412 default : ;
413 }
414 }
415 }
416 else //DV 50 Mbps and 100 Mbps
417 Fill(Stream_Video, 0, Video_Colorimetry, "4:2:2");
418
419 if (FrameSize_Theory && !IsHd)
420 {
421 float64 OverallBitRate=FrameSize_Theory*(DSF?((float64)25.000):((float64)30000/1001))*8;
422 if (FSC_WasSet)
423 {
424 if (FSP_WasNotSet)
425 OverallBitRate*=4; //DV100
426 else
427 OverallBitRate*=2; //DV50
428 }
429 if (OverallBitRate)
430 {
431 if (IsSub)
432 Fill(Stream_Video, 0, Video_BitRate_Encoded, OverallBitRate, 0);
433 else
434 Fill(Stream_General, 0, General_OverallBitRate, OverallBitRate, 0);
435 Fill(Stream_Video, 0, (FSC_WasSet && FSP_WasNotSet)?Video_BitRate_Maximum:Video_BitRate, OverallBitRate*134/150*76/80, 0); //134 Video DIF from 150 DIF, 76 bytes from 80 byte DIF
436 }
437 }
438
439 if (!Config->File_DvDif_DisableAudioIfIsInContainer_Get())
440 for (size_t Pos=0; Pos<Streams_Audio.size(); Pos++)
441 {
442 Stream_Prepare(Stream_Audio);
443 for (std::map<std::string, Ztring>::iterator Info=Streams_Audio[Pos]->Infos.begin(); Info!=Streams_Audio[Pos]->Infos.end(); ++Info)
444 Fill(Stream_Audio, StreamPos_Last, Info->first.c_str(), Info->second, true);
445 Fill(Stream_Audio, StreamPos_Last, Audio_BitRate_Encoded, 0);
446 }
447
448 if (Stream_BitRateFromContainer && Retrieve(Stream_Video, 0, Video_BitRate).empty())
449 {
450 if (Stream_BitRateFromContainer>=28800000*0.98 && Stream_BitRateFromContainer<=28800000*1.02)
451 {
452 Fill(Stream_Video, 0, Video_BitRate, ((float64)28800000)*134/150*76/80, 0);
453 Fill(Stream_Video, 0, Video_BitRate_Encoded, 28800000);
454 }
455 if (Stream_BitRateFromContainer>=57600000*0.98 && Stream_BitRateFromContainer<=57600000*1.02)
456 {
457 Fill(Stream_Video, 0, Video_BitRate, ((float64)57600000)*134/150*76/80, 0);
458 Fill(Stream_Video, 0, Video_BitRate_Encoded, 57600000);
459 }
460 if (Stream_BitRateFromContainer>=115200000*0.98 && Stream_BitRateFromContainer<=115200000*1.02)
461 {
462 Fill(Stream_Video, 0, Video_BitRate, ((float64)115200000)*134/150*76/80, 0);
463 Fill(Stream_Video, 0, Video_BitRate_Encoded, 115200000);
464 }
465 }
466
467 //Library settings
468 Fill(Stream_Video, 0, Video_Encoded_Library_Settings, Encoded_Library_Settings);
469
470 //Profile
471 if (FSC_WasSet || IsHd)
472 {
473 if (FSP_WasNotSet || IsHd)
474 {
475 Fill(Stream_General, 0, General_Format_Commercial_IfAny, "DVCPRO HD");
476 Fill(Stream_Video, 0, Video_Format_Commercial_IfAny, "DVCPRO HD");
477 Clear(Stream_Video, 0, Video_BitDepth); //MXF files say that DVCPRO HD streams are 8 or 10 bits, exact?
478 }
479 else
480 {
481 Fill(Stream_General, 0, General_Format_Commercial_IfAny, "DVCPRO 50");
482 Fill(Stream_Video, 0, Video_Format_Commercial_IfAny, "DVCPRO 50");
483 Fill(Stream_Video, 0, Video_BitRate_Mode, "CBR");
484 }
485 }
486 else if (audio_locked || (Retrieve(Stream_Video, 0, Video_Standard)==__T("PAL") && Retrieve(Stream_Video, 0, Video_Colorimetry)==__T("4:1:1")))
487 {
488 Fill(Stream_General, 0, General_Format_Commercial_IfAny, "DVCPRO");
489 Fill(Stream_Video, 0, Video_Format_Commercial_IfAny, "DVCPRO");
490 Fill(Stream_Video, 0, Video_BitRate_Mode, "CBR");
491 }
492 else
493 Fill(Stream_Video, 0, Video_BitRate_Mode, "CBR");
494
495 //Delay
496 if (TimeCode_FirstFrame_ms!=(int64u)-1)
497 {
498 Fill(Stream_Video, 0, Video_Delay, TimeCode_FirstFrame_ms);
499 if (TimeCode_FirstFrame.size()==11)
500 Fill(Stream_Video, 0, Video_Delay_DropFrame, TimeCode_FirstFrame[8]==';'?"Yes":"No");
501 Fill(Stream_Video, 0, Video_Delay_Source, "Stream");
502 Fill(Stream_Video, 0, Video_TimeCode_FirstFrame, TimeCode_FirstFrame.c_str());
503 Fill(Stream_Video, 0, Video_TimeCode_Source, "Subcode time code");
504 for (size_t Pos=0; Pos<Count_Get(Stream_Audio); Pos++)
505 {
506 Fill(Stream_Audio, Pos, Audio_Delay, TimeCode_FirstFrame_ms);
507 Fill(Stream_Audio, Pos, Audio_Delay_Source, "Stream");
508 }
509 }
510
511 #if defined(MEDIAINFO_EIA608_YES)
512 for (size_t Pos=0; Pos<CC_Parsers.size(); Pos++)
513 if (CC_Parsers[Pos] && CC_Parsers[Pos]->Status[IsAccepted])
514 {
515 Finish(CC_Parsers[Pos]);
516 for (size_t Pos2=0; Pos2<CC_Parsers[Pos]->Count_Get(Stream_Text); Pos2++)
517 {
518 Stream_Prepare(Stream_Text);
519 Merge(*CC_Parsers[Pos], Stream_Text, Pos2, StreamPos_Last);
520 Fill(Stream_Text, StreamPos_Last, Text_ID, CC_Parsers[Pos]->Retrieve(Stream_Text, Pos2, Text_ID), true);
521 }
522 }
523 #endif //defined(MEDIAINFO_EIA608_YES)
524}

560bool File_DvDif::FileHeader_Begin()

562 //Must have enough buffer for having header
563 if (Buffer_Size<8)

567 if (CC4(Buffer)==0x52494646 //RIFF
568 || CC4(Buffer+4)==0x66747970 //ftyp
569 || CC4(Buffer+4)==0x66726565 //free
570 || CC4(Buffer+4)==0x6D646174 //mdat
571 || CC4(Buffer+4)==0x6D6F6F76 //moov
572 || CC4(Buffer+4)==0x736B6970 //skip
573 || CC4(Buffer+4)==0x77696465 //wide
574 || CC4(Buffer)==0x060E2B34) //MXF begin
575 {
576 Finish();
577 return false;
578 }
579
580 //All should be OK...
581 return true;
582}

586//***************************************************************************
587
588//---------------------------------------------------------------------------
589bool File_DvDif::Synchronize()
590{
591 if (AuxToAnalyze)
592 {
593 Accept();
594 return true;
595 }
596
597 while (Buffer_Offset+8*80<=Buffer_Size //8 blocks
598 && !((Buffer[Buffer_Offset+0*80]&0xE0)==0x00 && (Buffer[Buffer_Offset+0*80+1]&0xF0)==0x00 && Buffer[Buffer_Offset+0*80+2]==0x00 //Header 0
599 && (Buffer[Buffer_Offset+1*80]&0xE0)==0x20 && (Buffer[Buffer_Offset+1*80+1]&0xF0)==0x00 && Buffer[Buffer_Offset+1*80+2]==0x00 //Subcode 0
600 && (Buffer[Buffer_Offset+2*80]&0xE0)==0x20 && (Buffer[Buffer_Offset+2*80+1]&0xF0)==0x00 && Buffer[Buffer_Offset+2*80+2]==0x01 //Subcode 1
601 && (Buffer[Buffer_Offset+3*80]&0xE0)==0x40 && (Buffer[Buffer_Offset+3*80+1]&0xF0)==0x00 && Buffer[Buffer_Offset+3*80+2]==0x00 //VAUX 0
602 && (Buffer[Buffer_Offset+4*80]&0xE0)==0x40 && (Buffer[Buffer_Offset+4*80+1]&0xF0)==0x00 && Buffer[Buffer_Offset+4*80+2]==0x01 //VAUX 1
603 && (Buffer[Buffer_Offset+5*80]&0xE0)==0x40 && (Buffer[Buffer_Offset+5*80+1]&0xF0)==0x00 && Buffer[Buffer_Offset+5*80+2]==0x02 //VAUX 2
604 && (Buffer[Buffer_Offset+6*80]&0xE0)==0x60 && (Buffer[Buffer_Offset+6*80+1]&0xF0)==0x00 && Buffer[Buffer_Offset+6*80+2]==0x00 //Audio 0
605 && (Buffer[Buffer_Offset+7*80]&0xE0)==0x80 && (Buffer[Buffer_Offset+7*80+1]&0xF0)==0x00 && Buffer[Buffer_Offset+7*80+2]==0x00)) //Video 0
606 Buffer_Offset++;
607
608 if (Buffer_Offset+8*80>Buffer_Size)
609 return false;
610
611 if (!Status[IsAccepted])
612 Accept();
613
614 return true;
615}

617//---------------------------------------------------------------------------
618bool File_DvDif::Synched_Test()
619{
620 if (AuxToAnalyze)
621 return true;
622
623 //Must have enough buffer for having header
624 if (Buffer_Offset+80>Buffer_Size)
625 return false;
626
627 //NULL blocks
628 if (Buffer[Buffer_Offset ]==0x00
629 && Buffer[Buffer_Offset+1]==0x00
630 && Buffer[Buffer_Offset+2]==0x00)
631 return true;
632
633 SCT =(Buffer[Buffer_Offset ]&0xE0)>>5;
634 DBN = Buffer[Buffer_Offset+2];
635
636 //DIF Sequence Numbers
637 if (DSF_IsValid)
638 {
639 if (Dseq_Old!=Dseq)
640 {
641 if (Dseq==0
642 && !(!DSF && Dseq_Old==9)
643 && !( DSF && Dseq_Old==11))
644 {
645 if (!Status[IsAccepted])
646 Trusted_IsNot("Wrong order");
647 else
648 Synched_Test_Reset();
649 }
650 Dseq_Old=Dseq;
651 }
652 }
653
654 //DIF Block Numbers
655 if (SCT!=(int8u)-1)
656 {
657 int8u Number=DBN_Olds[SCT]+1;
658 switch (SCT)
659 {
660 case 0 : //Header
661 if (SCT_Old!=4
662 || DBN!=0)
663 {
664 if (!Status[IsAccepted])
665 Trusted_IsNot("Wrong order");
666 else
667 Synched_Test_Reset();
668 }
669 break;
670 case 1 : //Subcode
671 if (!((DBN==0 && SCT_Old==0) || (DBN!=0 && SCT_Old==1))
672 || (Number!=DBN && !(Number==2 && DBN==0)))
673 {
674 if (!Status[IsAccepted])
675 Trusted_IsNot("Wrong order");
676 else
677 Synched_Test_Reset();
678 }
679 break;
680 case 2 : //VAUX
681 if (!((DBN==0 && SCT_Old==1) || (DBN!=0 && SCT_Old==2))
682 || (Number!=DBN && !(Number==3 && DBN==0)))
683 {
684 if (!Status[IsAccepted])
685 Trusted_IsNot("Wrong order");
686 else
687 Synched_Test_Reset();
688 }
689 break;
690 case 3 : //Audio
691 if (!((DBN==0 && SCT_Old==2) || (DBN!=0 && SCT_Old==4))
692 || (Number!=DBN && !(Number==9 && DBN==0)))
693 {
694 if (!Status[IsAccepted])
695 Trusted_IsNot("Wrong order");
696 else
697 Synched_Test_Reset();
698 }
699 break;
700 case 4 : //Video
701 if (!(SCT_Old==3 || SCT_Old==4)
702 || (Number!=DBN && !(Number==135 && DBN==0)))
703 {
704 if (!Status[IsAccepted])
705 Trusted_IsNot("Wrong order");
706 else
707 Synched_Test_Reset();
708 }
709 break;
710 default: ;
711 }
712
713 if (SCT!=(int8u)-1)
714 {
715 SCT_Old=SCT;
716 DBN_Olds[SCT]=DBN;
717 }
718 }
719
720 //We continue
721 return true;
722}

724//---------------------------------------------------------------------------
725void File_DvDif::Synched_Test_Reset()
726{
727 //Temp
728 SCT=(int8u)-1;
729 SCT_Old=4; //Video
730 DBN_Olds[0]=0;
731 DBN_Olds[1]=1; //SubCode
732 DBN_Olds[2]=2; //Vaux
733 DBN_Olds[3]=8; //Audio
