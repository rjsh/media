79//***************************************************************************
80// Streams management
81//***************************************************************************
82
83//---------------------------------------------------------------------------
84void File_Wm::Streams_Finish()
85{
86 //Encryption management
87 /*const Ztring& Encryption=Retrieve(Stream_General, 0, General_Encryption);
88 if (!Encryption.empty())
89 {
90 for (size_t StreamKind=Stream_General+1; StreamKind<Stream_Max; StreamKind++)
91 for (size_t Pos=0; Pos<(*Stream[StreamKind]).size(); Pos++)
92 Fill ((stream_t)StreamKind, 0, "Encryption", Encryption);
93 }
94
95 Fill("BitRate", CurrentBitRate[StreamNumber]);
96 */
97
98 std::map<int16u, stream>::iterator Temp=Stream.begin();
99 while (Temp!=Stream.end())
100 {
101 for (std::map<std::string, ZenLib::Ztring>::iterator Info_Temp=Temp->second.Info.begin(); Info_Temp!=Temp->second.Info.end(); ++Info_Temp)
102 Fill(Temp->second.StreamKind, Temp->second.StreamPos, Info_Temp->first.c_str(), Info_Temp->second, true);
103
104 //Codec Info
105 for (size_t Pos=0; Pos<CodecInfos.size(); Pos++)
106 {
107 if ((CodecInfos[Pos].Type==1 && Temp->second.StreamKind==Stream_Video)
108 || (CodecInfos[Pos].Type==2 && Temp->second.StreamKind==Stream_Audio))
109 {
110 Fill(Temp->second.StreamKind, Temp->second.StreamPos, "CodecID_Description", CodecInfos[Pos].Info, true);
111 Fill(Temp->second.StreamKind, Temp->second.StreamPos, "Codec_Description", CodecInfos[Pos].Info, true);
112 }
113 }
114
115 if (Temp->second.StreamKind==Stream_Video)
116 {
117 //Some tests about the frame rate
118 int32u PresentationTime_Previous=(int32u)-1;
119 size_t TimeDiffs_Sum=0;
120 std::map<int32u, size_t> TimeDiffs;
121 for (std::set<int32u>::iterator PresentationTime=Temp->second.PresentationTimes.begin(); PresentationTime!=Temp->second.PresentationTimes.end(); ++PresentationTime)
122 {
123 if (PresentationTime_Previous!=(int32u)-1)
124 TimeDiffs[*PresentationTime-PresentationTime_Previous]++;
125 PresentationTime_Previous=*PresentationTime;
126 }
127 for (std::map<int32u, size_t>::iterator TimeDiff=TimeDiffs.begin(); TimeDiff!=TimeDiffs.end();)
128 {
129 if (TimeDiff->second<=2)
130 TimeDiffs.erase(TimeDiff++);
131 else
132 {
133 TimeDiffs_Sum+=TimeDiff->second;
134 ++TimeDiff;
135 }
136 }
137
138 if (TimeDiffs.empty()
139 || (TimeDiffs.size()==1 && TimeDiffs_Sum<16)
140 || (TimeDiffs.size()==2 && TimeDiffs_Sum<32)
141 || TimeDiffs.begin()->first==1)
142 {
143 if (Temp->second.AverageTimePerFrame>0)
144 Fill(Stream_Video, Temp->second.StreamPos, Video_FrameRate, ((float)10000000)/(Temp->second.AverageTimePerFrame*(Temp->second.Parser && Temp->second.Parser->Retrieve(Stream_Video, 0, Video_ScanType)==__T("Interlaced")?2:1)), 3, true);
145 }
146 else if (TimeDiffs.size()==1)
147 {
148 Fill(Stream_Video, Temp->second.StreamPos, Video_FrameRate, 1000/((float64)TimeDiffs.begin()->first), 3, true);
149 if (Temp->second.AverageTimePerFrame>0)
150 Fill(Stream_Video, Temp->second.StreamPos, Video_FrameRate_Nominal, ((float)10000000)/(Temp->second.AverageTimePerFrame*(Temp->second.Parser && Temp->second.Parser->Retrieve(Stream_Video, 0, Video_ScanType)==__T("Interlaced")?2:1)), 3, true);
151 }
152 else if (TimeDiffs.size()==2)
153 {
154 std::map<int32u, size_t>::iterator PresentationTime_Delta_Most=TimeDiffs.begin();
155 float64 PresentationTime_Deltas_1_Value=(float64)PresentationTime_Delta_Most->first;
156 float64 PresentationTime_Deltas_1_Count=(float64)PresentationTime_Delta_Most->second;
157 ++PresentationTime_Delta_Most;
158 float64 PresentationTime_Deltas_2_Value=(float64)PresentationTime_Delta_Most->first;
159 float64 PresentationTime_Deltas_2_Count=(float64)PresentationTime_Delta_Most->second;
160 float64 FrameRate_Real=1000/(((PresentationTime_Deltas_1_Value*PresentationTime_Deltas_1_Count)+(PresentationTime_Deltas_2_Value*PresentationTime_Deltas_2_Count))/(PresentationTime_Deltas_1_Count+PresentationTime_Deltas_2_Count));
161 Fill(Temp->second.StreamKind, Temp->second.StreamPos, Video_FrameRate, FrameRate_Real, 3, true);
162 if (Temp->second.AverageTimePerFrame>0)
163 Fill(Stream_Video, Temp->second.StreamPos, Video_FrameRate_Nominal, ((float)10000000)/(Temp->second.AverageTimePerFrame*(Temp->second.Parser && Temp->second.Parser->Retrieve(Stream_Video, 0, Video_ScanType)==__T("Interlaced")?2:1)), 3, true);
164 }
165 else
166 {
167 Fill(Stream_Video, Temp->second.StreamPos, Video_FrameRate_Mode, "VFR");
168 if (Temp->second.AverageTimePerFrame>0)
169 Fill(Stream_Video, Temp->second.StreamPos, Video_FrameRate_Nominal, ((float)10000000)/(Temp->second.AverageTimePerFrame*(Temp->second.Parser && Temp->second.Parser->Retrieve(Stream_Video, 0, Video_ScanType)==__T("Interlaced")?2:1)), 3, true);
170 }
171 }
172 if (Temp->second.AverageBitRate>0)
173 Fill(Temp->second.StreamKind, Temp->second.StreamPos, "BitRate", Temp->second.AverageBitRate, 10, true);
174 if (Temp->second.LanguageID!=(int16u)-1 && Temp->second.LanguageID<(int16u)Languages.size())
175 Fill(Temp->second.StreamKind, Temp->second.StreamPos, "Language", Languages[Temp->second.LanguageID]);
176 else if (!Language_ForAll.empty())
177 Fill(Temp->second.StreamKind, Temp->second.StreamPos, "Language", Language_ForAll);
178 if (Temp->second.Parser)
179 {
180 if (Temp->second.StreamKind==Stream_Max)
181 if (Temp->second.Parser->Count_Get(Stream_Audio))
182 {
183 Stream_Prepare(Stream_Audio);
184 Temp->second.StreamKind=StreamKind_Last;
185 Temp->second.StreamPos=StreamPos_Last;
186 }
187 Ztring Format_Profile;
188 if (Temp->second.StreamKind==Stream_Video)
189 Format_Profile=Retrieve(Stream_Video, Temp->second.StreamPos, Video_Format_Profile);
190 Finish(Temp->second.Parser);
191 if (Temp->second.Parser->Get(Stream_Video, 0, Video_Format)==__T("MPEG Video"))
192 {
193 //Width/Height are junk
194 Clear(Stream_Video, Temp->second.StreamPos, Video_Width);
195 Clear(Stream_Video, Temp->second.StreamPos, Video_Height);
196 Clear(Stream_Video, Temp->second.StreamPos, Video_PixelAspectRatio);
197 Clear(Stream_Video, Temp->second.StreamPos, Video_DisplayAspectRatio);
198 }
199
200 //Delay (in case of MPEG-PS)
201 if (Temp->second.TimeCode_First!=(int64u)-1)
202 {
203 Fill(Temp->second.StreamKind, Temp->second.StreamPos, Fill_Parameter(Temp->second.StreamKind, Generic_Delay), Temp->second.TimeCode_First, 10);
204 Fill(Temp->second.StreamKind, Temp->second.StreamPos, Fill_Parameter(Temp->second.StreamKind, Generic_Delay_Source), "Container");
205 }
206
207
208 Merge(*Temp->second.Parser, Temp->second.StreamKind, 0, Temp->second.StreamPos);
209 if (!Format_Profile.empty() && Format_Profile.find(Retrieve(Stream_Video, Temp->second.StreamPos, Video_Format_Profile))==0)
210 Fill(Stream_Video, Temp->second.StreamPos, Video_Format_Profile, Format_Profile, true);
211 }
212
213 ++Temp;
214 }
215
216 if (Count_Get(Stream_Video)==0 && Count_Get(Stream_Image)==0)
217 Fill(Stream_General, 0, General_InternetMediaType, "audio/x-ms-wma", Unlimited, true, true);
218
219 //Purge what is not needed anymore
220 if (!File_Name.empty()) //Only if this is not a buffer, with buffer we can have more data
221 Stream.clear();
222}

228//---------------------------------------------------------------------------
229void File_Wm::Header_Parse()
230{
231 if (!MustUseAlternativeParser)
232 {
233 //Parsing
234 int128u Name;
235 int64u Size;
236 Get_GUID(Name, "Name");
237 Get_L8 (Size, "Size");
238
239 //Filling
240 Header_Fill_Code(Name.hi, Ztring().From_GUID(Name));
241 Header_Fill_Size(Size);
242 }
243 else
244 {
245 Header_Fill_Code(0, "Packet");
246 Header_Fill_Size(MaximumDataPacketSize);
247 }
248}
