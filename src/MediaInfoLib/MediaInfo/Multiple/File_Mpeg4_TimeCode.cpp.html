36File_Mpeg4_TimeCode::File_Mpeg4_TimeCode()
37:File__Analyze()
38{
39 //Out
40 Pos=(int32u)-1;
41
42 NumberOfFrames=0;
43 DropFrame=false;
44 NegativeTimes=false;
45}

47//***************************************************************************
48// Streams management
49//***************************************************************************
50
51//---------------------------------------------------------------------------
52void File_Mpeg4_TimeCode::Streams_Fill()
53{
54 if (Pos!=(int32u)-1 && NumberOfFrames)
55 {
56 int64s Pos_Temp=Pos;
57 float64 FrameRate_WithDF=NumberOfFrames;
58 if (DropFrame)
59 {
60 float64 FramesPerHour_NDF=FrameRate_WithDF*60*60;
61 FrameRate_WithDF*=(FramesPerHour_NDF-108)/FramesPerHour_NDF;
62 }
63
64 Fill(Stream_General, 0, "Delay", Pos_Temp*1000/FrameRate_WithDF, 0);
65
66 TimeCode TC(Pos_Temp, NumberOfFrames, DropFrame);
67 Stream_Prepare(Stream_Other);
68 Fill(Stream_Other, StreamPos_Last, Other_Type, "Time code");
69 Fill(Stream_Other, StreamPos_Last, Other_TimeCode_FirstFrame, TC.ToString().c_str());
70 if (Frame_Count==1)
71 Fill(Stream_Other, StreamPos_Last, Other_TimeCode_Striped, "Yes");
72 }
73}
74
75//***************************************************************************
76// Buffer - Global
77//***************************************************************************
78
79//---------------------------------------------------------------------------
80void File_Mpeg4_TimeCode::Read_Buffer_Continue()
81{
82 //Parsing
83 int32u Position=0;
84 while (Element_Offset<Element_Size)
85 {
86 Get_B4 (Position, "Position");
87 if (Pos==(int32u)-1) //First time code
88 {
89 Pos=Position;
90 if (NegativeTimes)
91 Pos=(int32s)Position;
92 }
93 }
94
95 FILLING_BEGIN();
96 Frame_Count+=Element_Size/4;
97
98 if (!Status[IsAccepted])
99 {
100 Accept("TimeCode");
101 Fill("TimeCode");
102 }
103 FILLING_END();
104}
105
