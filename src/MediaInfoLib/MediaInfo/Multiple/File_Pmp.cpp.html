41const char* Pmp_video_format(int32u video_format)
42{
43 switch (video_format)
44 {
45 case 0 : return "MPEG-4 Visual";
46 case 1 : return "AVC";
47 default : return "";
48 }
49}
50
51//---------------------------------------------------------------------------
52const char* Pmp_audio_format(int32u audio_format)
53{
54 switch (audio_format)
55 {
56 case 0 : return "MPEG Audio";
57 case 1 : return "AAC";
58 default : return "";
59 }
60}

62//***************************************************************************
63// Buffer - File header
64//***************************************************************************
65
66//---------------------------------------------------------------------------
67bool File_Pmp::FileHeader_Begin()
68{
69 //Synchro
70 if (4>Buffer_Size)
71 return false;
72 if (Buffer[0]!=0x70 //"pmpm"
73 || Buffer[1]!=0x6D
74 || Buffer[2]!=0x70
75 || Buffer[3]!=0x6D)
76 {
77 Reject();
78 return false;
79 }
80
81 return true;
82}
83
84//---------------------------------------------------------------------------
85void File_Pmp::FileHeader_Parse()
86{
87 //Parsing
88 int32u version, video_format=0, nb_frames=0, video_width=0, video_height=0, time_base_num=0, time_base_den=0, audio_format=0, sample_rate=0, channels=0;
89
90 Skip_C4( "Signature");
91 Get_L4 (version, "Version");
92 if (version==1)
93 {
94 Get_L4 (video_format, "video_format");
95 Get_L4 (nb_frames, "number of frames");
96 Get_L4 (video_width, "video_width");
97 Get_L4 (video_height, "video_height");
98 Get_L4 (time_base_num, "time_base_num");
99 Get_L4 (time_base_den, "time_base_den");
100 Skip_L4( "number of audio streams");
101 Get_L4 (audio_format, "audio_format");
102 Get_L4 (channels, "channels");
103 Skip_L4( "unknown");
104 Get_L4 (sample_rate, "sample_rate");
105 }
106
107 FILLING_BEGIN();
108 Accept("PMP");
109
110 Fill(Stream_General, 0, General_Format, "PMP");
111
112 if (version==1)
113 {
114 Stream_Prepare(Stream_Video);
115 Fill(Stream_Video, 0, Video_Format, Pmp_video_format(video_format));
116 Fill(Stream_Video, 0, Video_FrameCount, nb_frames);
117 Fill(Stream_Video, 0, Video_Width, video_width);
118 Fill(Stream_Video, 0, Video_Height, video_height);
119 Fill(Stream_Video, 0, Video_FrameRate, (float)time_base_den / 100);
120 Stream_Prepare(Stream_Audio);
121 Fill(Stream_Audio, 0, Audio_Format, Pmp_audio_format(audio_format));
122 Fill(Stream_Audio, 0, Audio_Channel_s_, channels);
123 Fill(Stream_Audio, 0, Audio_SamplingRate, sample_rate);
124 }
125
126 //No more need data
127 Finish("PMP");
128 FILLING_END();
129}
