91//***************************************************************************
92// Buffer - File header
93//***************************************************************************
94
95//---------------------------------------------------------------------------
96bool File_Dxw::FileHeader_Begin()
97{
98 XMLDocument document;
99 if (!FileHeader_Begin_XML(document))
100 return false;
101
102 {
103 XMLElement* Root=document.FirstChildElement("indexFile");
104 if (Root)
105 {
106 const char* Attribute=Root->Attribute("xmlns");
107 if (Attribute==NULL || Ztring().From_UTF8(Attribute)!=__T("urn:digimetrics-xml-wrapper"))
108 {
109 Reject("DXW");
110 return false;
111 }
112
113 Accept("DXW");
114 Fill(Stream_General, 0, General_Format, "DXW");
115
116 ReferenceFiles=new File__ReferenceFilesHelper(this, Config);
117
118 XMLElement* Track=Root->FirstChildElement();
119 while (Track)
120 {
121 if (string(Track->Value())=="clip")
122 {
123 sequence* Sequence=new sequence;
124
125 Attribute=Track->Attribute("file");
126 if (Attribute)
127 {
128 Sequence->AddFileName(Ztring().From_UTF8(Attribute));
129
130 Attribute=Track->Attribute("type");
131 if (Attribute)
132 {
133 Ztring StreamKind; StreamKind.From_UTF8(Attribute);
134 if (StreamKind==__T("video"))
135 Sequence->StreamKind=Stream_Video;
136 if (StreamKind==__T("audio"))
137 Sequence->StreamKind=Stream_Audio;
138 if (StreamKind==__T("data"))
139 Sequence->StreamKind=Stream_Text; //Not sure this is a right mapping, but this is only used when file is missing
140 }
141
142 Attribute=Track->Attribute("source");
143 if (Attribute)
144 {
145 Ztring StreamKind; StreamKind.From_UTF8(Attribute);
146 if (StreamKind==__T("main"))
147 Sequence->IsMain=true;
148 }
149
150 Sequence->StreamID=ReferenceFiles->Sequences_Size()+1;
151 }
152
153 Attribute=Track->Attribute("framerate");
154 if (Attribute)
155 {
156 Sequence->FrameRate_Set(Ztring().From_UTF8(Attribute).To_float64());
157
158 Attribute=Track->Attribute("type");
159 if (Attribute)
160 {
161 Ztring StreamKind; StreamKind.From_UTF8(Attribute);
162 if (StreamKind==__T("video"))
163 Sequence->StreamKind=Stream_Video;
164 if (StreamKind==__T("audio"))
165 Sequence->StreamKind=Stream_Audio;
166 if (StreamKind==__T("data"))
167 Sequence->StreamKind=Stream_Text; //Not sure this is a right mapping, but this is only used when file is missing
168 }
169
170 XMLElement* Frame=Track->FirstChildElement();
171 while (Frame)
172 {
173 if (string(Frame->Value())=="frame")
174 {
175 Attribute=Frame->Attribute("file");
176 if (Attribute)
177 Sequence->AddFileName(Ztring().From_UTF8(Attribute));
178 }
179
180 Frame=Frame->NextSiblingElement();
181 }
182 }
183
184 Sequence->StreamID=ReferenceFiles->Sequences_Size()+1;
185 ReferenceFiles->AddSequence(Sequence);
186 }
187
188 Track=Track->NextSiblingElement();
189 }
190 }
191 else
192 {
193 Reject("DXW");
194 return false;
195 }
196 }
197
198 Element_Offset=File_Size;
199
200 //All should be OK...
201 return true;
202}
