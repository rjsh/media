98//***************************************************************************
99// Buffer - File header
100//***************************************************************************
101
102//---------------------------------------------------------------------------
103bool File_Ism::FileHeader_Begin()
104{
105 XMLDocument document;
106 if (!FileHeader_Begin_XML(document))
107 return false;
108
109 {
110 XMLElement* Root=document.FirstChildElement("smil");
111 if (Root)
112 {
113 ReferenceFiles=new File__ReferenceFilesHelper(this, Config);
114
115 std::set<Ztring> FileNames;
116
117 XMLElement* Body=Root->FirstChildElement();
118 while (Body)
119 {
120 if (string(Body->Value())=="body")
121 {
122 XMLElement* Switch=Body->FirstChildElement();
123 while (Switch)
124 {
125 if (string(Switch->Value())=="switch")
126 {
127 Accept("ISM");
128
129 XMLElement* Stream=Switch->FirstChildElement();
130 while (Stream)
131 {
132 string Value(Stream->Value());
133 if (Value=="video" || Value=="videostream" || Value=="audio" || Value=="audiostream" || Value=="text" || Value=="textstream")
134 {
135 sequence* Sequence=new sequence;
136
137 if (Value=="video" || Value=="videostream")
138 Sequence->StreamKind=Stream_Video;
139 if (Value=="audio" || Value=="audiostream")
140 Sequence->StreamKind=Stream_Audio;
141 if (Value=="text" || Value=="textstream" )
142 Sequence->StreamKind=Stream_Text;
143
144 const char* Attribute=Stream->Attribute("src");
145 if (Attribute)
146 Sequence->AddFileName(Ztring().From_UTF8(Attribute));
147
148 XMLElement* Param=Stream->FirstChildElement();
149 while (Param)
150 {
151 if (string(Param->Value())=="param")
152 {
153 Attribute=Param->Attribute("name");
154 if (Attribute && Ztring().From_UTF8(Attribute)==__T("trackID"))
155 {
156 Attribute=Param->Attribute("value");
157 if (Attribute)
158 Sequence->StreamID=Ztring().From_UTF8(Attribute).To_int64u();
159 }
160 }
161 Param=Param->NextSiblingElement();
162 }
163
164 if (!Sequence->FileNames.empty() && !Sequence->FileNames[0].empty() && FileNames.find(Sequence->FileNames[0])==FileNames.end())
165 {
166 ReferenceFiles->AddSequence(Sequence);
167 FileNames.insert(Sequence->FileNames[0]);
168 }
169 }
170
171 Stream=Stream->NextSiblingElement();
172 }
173 }
174
175 Switch=Switch->NextSiblingElement();
176 }
177 }
178
179 Body=Body->NextSiblingElement();
180 }
181 }
182 else
183 {
184 Reject("ISM");
185 return false;
186 }
187 }
188
189 Element_Offset=File_Size;
190
191 //All should be OK...
192 return true;
193}
