66//---------------------------------------------------------------------------
67bool File_Dpg::FileHeader_Begin()
68{
69 //Element_Size
70 if (Buffer_Size<0x14)
71 return false; //Must wait for more data
72
73 if ( CC4(Buffer )!=0x44504730 //"DPG0"
74 || LittleEndian2int32u(Buffer+0x10)!=0) //Zero
75 {
76 Reject("DPG");
77 return false;
78 }
79
80 //All should be OK...
81 return true;
82}

84//---------------------------------------------------------------------------
85void File_Dpg::FileHeader_Parse()
86{
87 //Parsing
88 int32u FrameCount, FrameRate, SamplingRate;
89 Skip_C4( "Signature");
90 Get_L4 (FrameCount, "Frame count");
91 Get_L4 (FrameRate, "Frame rate"); Param_Info2(FrameRate/0x100, " fps");
92 Get_L4 (SamplingRate, "Sampling rate");
93 Skip_L4( "0x00000000");
94 Get_L4 (Audio_Offset, "Audio Offset");
95 Get_L4 (Audio_Size, "Audio Size");
96 Get_L4 (Video_Offset, "Video Offset");
97 Get_L4 (Video_Size, "Video Size");
98
99 FILLING_BEGIN();
100 Accept("DPG");
101
102 Fill(Stream_General, 0, General_Format, "DPG");
103
104 Stream_Prepare(Stream_Video);
105 Fill(Stream_Video, 0, Video_FrameRate, (float)(FrameRate/0x100), 3);
106 Fill(Stream_Video, 0, Video_FrameCount, FrameCount);
107 Fill(Stream_Video, 0, Video_StreamSize, Video_Size);
108
109 Stream_Prepare(Stream_Audio);
110 Fill(Stream_Audio, 0, Audio_SamplingRate, SamplingRate);
111 Fill(Stream_Audio, 0, Audio_StreamSize, Audio_Size);
112
113 //Positionning
114 #if defined(MEDIAINFO_MPEGA_YES)
115 Parser=new File_Mpega();
116 Open_Buffer_Init(Parser);
117 GoTo(Audio_Offset, "DPG");
118 #elif defined(MEDIAINFO_MPEGV_YES)
119 Audio_Size=0;
120 Parser=new File_Mpegv();
121 Open_Buffer_Init(Parser);
122 GoTo(Video_Offset, "DPG");
123 #else
124 Finish("DPG");
125 #endif
126 FILLING_END();
127}

140//---------------------------------------------------------------------------
141void File_Dpg::Read_Buffer_Continue()
142{
143 if (!Parser)
144 return; //Not ready
145
146 #if defined(MEDIAINFO_MPEGA_YES) || defined(MEDIAINFO_MPEGV_YES)
147 if (Audio_Size)
148 {
149 #if defined(MEDIAINFO_MPEGA_YES)
150 Open_Buffer_Continue(Parser, (size_t)((File_Offset+Buffer_Size<Audio_Offset+Audio_Size)?Buffer_Size:(Audio_Offset+Audio_Size-File_Offset)));
151 if (Parser->Status[IsAccepted])
152 {
153 Parser->Open_Buffer_Unsynch();
154 Finish(Parser);
155 Merge(*Parser, Stream_Audio, 0, 0);
156 #if defined(MEDIAINFO_MPEGV_YES)
157 Audio_Size=0;
158 Data_GoTo(Video_Offset, "DPG");
159 delete Parser; Parser=new File_Mpegv();
160 Open_Buffer_Init(Parser);
161 #else
162 Finish("DPG");
163 #endif
164 }
165 #endif
166 }
167 else
168 {
169 #if defined(MEDIAINFO_MPEGV_YES)
170 Open_Buffer_Continue(Parser, (size_t)((File_Offset+Buffer_Size<Video_Offset+Video_Size)?Buffer_Size:(Video_Offset+Video_Size-File_Offset)));
171 if (Parser->Status[IsAccepted])
172 {
173 //Merging
174 Parser->Open_Buffer_Unsynch();
175 Finish(Parser);
176 Merge(*Parser, Stream_Video, 0, 0);
177
178 Finish("DPG");
179 }
180 #endif
181 }
182 #endif //defined(MEDIAINFO_MPEGA_YES) || defined(MEDIAINFO_MPEGV_YES)
183
184 //Positioning
185 Buffer_Offset=Buffer_Size; //We have already parsed this data
186}
