41private :
42 //Streams management
43 void Streams_Finish();

45 //Buffer - Synchro

54 void Read_Buffer_AfterParsing();

56 //Buffer - Per element
57 bool Header_Begin();
58 void Header_Parse();
59 void Data_Parse();

61 //Packets
62 void map();
63 void media();

72 int32u Material_Fields_First;
73 int32u Material_Fields_Last;
74 int32u Material_File_Size;
75 int32u Material_Fields_FieldsPerFrame;
76 int8u Parsers_Count;
77 int8u AncillaryData_StreamID;
78 struct tc
79 {
80 int64u Milliseconds;
81 string String;
82
83 tc()
84 {
85 Milliseconds=(int64u)-1;
86 }
87 };
88 std::map<int8u, tc> TimeCodes; //Key is StreamID
89 bool Material_Fields_First_IsValid;
90 bool Material_Fields_Last_IsValid;

93 //Temp - Stream
94 struct stream
95 {
96 std::vector<File__Analyze*> Parsers;
97 int64u FirstFrameDuration; //In case of audio, indicates the duration of the first frame
98 stream_t StreamKind;
99 size_t StreamPos;
100 int32u TimeStamp_Start;
101 int32u TimeStamp_End;
102 int32u FrameRate_Code;
103 int32u LinesPerFrame_Code;
104 int32u FieldsPerFrame_Code;
105 int8u MediaType;
106 int8u TrackID;
107 bool Searching_Payload;
108 bool Searching_TimeStamp_Start;
109 bool Searching_TimeStamp_End;
110 bool IsChannelGrouping;
111 bool DisplayInfo; //In case of channel grouping, info is about the complete (2*half) stream, so second stream info must not be used
112 Ztring MediaName;
113 std::map<std::string, Ztring> Infos;
114 #if MEDIAINFO_DEMUX
115 bool Demux_EventWasSent;
116 #endif //MEDIAINFO_DEMUX
117
118 stream()
119 {
120 FirstFrameDuration=0;
121 StreamKind=Stream_Max;
122 StreamPos=(size_t)-1;
123 TimeStamp_Start = (int32u)-1;
124 TimeStamp_End = (int32u)-1;
125 Searching_Payload = false;
126 Searching_TimeStamp_Start=false;
127 Searching_TimeStamp_End=false;
128 FrameRate_Code=(int32u)-1;
129 LinesPerFrame_Code=(int32u)-1;
130 FieldsPerFrame_Code=(int32u)-1;
131 MediaType=(int8u)-1;
132 TrackID=(int8u)-1;
133 IsChannelGrouping=false;
134 DisplayInfo=true;
135 #if MEDIAINFO_DEMUX
136 Demux_EventWasSent=false;
137 #endif //MEDIAINFO_DEMUX
138 }
139 ~stream()
140 {
141 for (size_t Pos=0; Pos<Parsers.size(); Pos++)
142 delete Parsers[Pos];
143 }
144 };
145 std::vector<stream> Streams;
146 File__Analyze* UMF_File;
147 int64u SizeToAnalyze; //Total size of a chunk to analyse, it may be changed by the parser
148 int64u IsParsingMiddle_MaxOffset;
149 int8u Audio_Count;
150 int8u TrackNumber;
151
152 //File__Analyze helpers
153 void Streams_Finish_PerStream(size_t StreamID, stream &Temp);
154 void Detect_EOF();
155 File__Analyze* ChooseParser_ChannelGrouping(int8u TrackID);
156 void TryToFinish();
157
158 #if MEDIAINFO_DEMUX
159 bool Demux_HeaderParsed;
160 #endif //MEDIAINFO_DEMUX
161
162 #if MEDIAINFO_SEEK
163 int32u Flt_FieldPerEntry;
164 std::vector<int32u> Flt_Offsets; //In 1024-byte
165 struct seek
166 {
167 int64u FrameNumber;
168 int32u StreamOffset; //In 1024-byte
169 };
170 std::vector<seek> Seeks;
171 bool IFrame_IsParsed;
172 #endif //MEDIAINFO_SEEK
173};
