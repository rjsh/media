62//---------------------------------------------------------------------------
63void File_Ogg::Streams_Fill()
64{
65 std::map<int64u, stream>::iterator Stream_Temp=Stream.begin();
66 while (Stream_Temp!=Stream.end())
67 {
68 //Filling
69 if (Stream_Temp->second.Parser)
70 {
71 Stream_Temp->second.Parser->Fill();
72 Merge(*Stream_Temp->second.Parser);
73 Merge(*Stream_Temp->second.Parser, Stream_General, 0, 0);
74 Stream_Temp->second.StreamKind=((File_Ogg_SubElement*)Stream_Temp->second.Parser)->StreamKind;
75 Stream_Temp->second.StreamPos=Count_Get(Stream_Temp->second.StreamKind)-1;
76 if (!SizedBlocks && !XiphLacing)
77 Stream_Temp->second.absolute_granule_position_Resolution=((File_Ogg_SubElement*)Stream_Temp->second.Parser)->absolute_granule_position_Resolution;
78 if (Stream_Temp->second.StreamKind==Stream_Audio && Stream_Temp->second.absolute_granule_position_Resolution==0)
79 Stream_Temp->second.absolute_granule_position_Resolution=Retrieve(Stream_Audio, Stream_Temp->second.StreamPos, Audio_SamplingRate).To_int64u();
80 if (!IsSub && Stream_Temp->second.absolute_granule_position && Stream_Temp->second.absolute_granule_position_Resolution)
81 {
82 if (Stream_Temp->second.StreamKind==Stream_Audio)
83 Fill(Stream_Temp->second.StreamKind, Stream_Temp->second.StreamPos, Fill_Parameter(Stream_Temp->second.StreamKind, Generic_Duration), float64_int64s(((float64)(Stream_Temp->second.absolute_granule_position))*1000/Stream_Temp->second.absolute_granule_position_Resolution), 10, true);
84 }
85 if (!IsSub)
86 {
87 if (Stream_Temp->second.StreamKind==Stream_Max)
88 {
89 Stream_Temp->second.StreamKind=Stream_General;
90 Stream_Temp->second.StreamPos=0;
91 }
92 Fill(Stream_Temp->second.StreamKind, Stream_Temp->second.StreamPos, General_ID, Stream_Temp->first);
93 Fill(Stream_Temp->second.StreamKind, Stream_Temp->second.StreamPos, General_ID_String, Ztring::ToZtring(Stream_Temp->first)+__T(" (0x")+Ztring::ToZtring(Stream_Temp->first, 16)+__T(')'), true);
94 }
95 }
96 ++Stream_Temp;
97 }
98
99 Fill(Stream_General, 0, General_Format, "Ogg", Unlimited, true, true);
100 if (Count_Get(Stream_Video)==0 && Count_Get(Stream_Image)==0)
101 Fill(Stream_General, 0, General_InternetMediaType, "audio/ogg", Unlimited, true, true);
102}

104//---------------------------------------------------------------------------
105void File_Ogg::Streams_Finish()
106{
107 std::map<int64u, stream>::iterator Stream_Temp=Stream.begin();
108 while (Stream_Temp!=Stream.end())
109 {
110 //Filling
111 if (Stream_Temp->second.Parser)
112 {
113 Finish(Stream_Temp->second.Parser);
114 Merge(*Stream_Temp->second.Parser, Stream_Temp->second.StreamKind, 0, Stream_Temp->second.StreamPos);
115 Merge(*Stream_Temp->second.Parser, Stream_General, 0, 0);
116 }
117 ++Stream_Temp;
118 }
119
120 //No more need
121 if (!File_Name.empty()) //Only if this is not a buffer, with buffer we can have more data
122 Stream.clear();
123}

129//---------------------------------------------------------------------------
130bool File_Ogg::FileHeader_Begin()
131{
132 //Must have enough buffer for having header
133 if (Buffer_Size<4)
134 return false; //Must wait for more data
135
136 //False positives detection: Detect AVI files, or the parser can synchronize with OggS stream in a AVI chunk
137 if (CC4(Buffer)==0x52494646) //"RIFF"
138 {
139 Finish("OGG");
140 return false;
141 }
142
143 //All should be OK...
144 return true;
145}

147//***************************************************************************
148// Buffer - Synchro
149//***************************************************************************
150
151//---------------------------------------------------------------------------
152bool File_Ogg::Synchronize()
153{
154 //Synchronizing
155 while (Buffer_Offset+4<=Buffer_Size)
156 {
157 while(Buffer_Offset+4<=Buffer_Size && (Buffer[Buffer_Offset ]!=0x4F
158 || Buffer[Buffer_Offset+1]!=0x67
159 || Buffer[Buffer_Offset+2]!=0x67
160 || Buffer[Buffer_Offset+3]!=0x53)) //"OggS"
161 {
162 Buffer_Offset+=1+2;
163 while(Buffer_Offset<Buffer_Size && Buffer[Buffer_Offset]!=0x67)
164 Buffer_Offset+=2;
165 if (Buffer_Offset>=Buffer_Size || Buffer[Buffer_Offset-1]==0x67)
166 Buffer_Offset--;
167 Buffer_Offset--;
168 }
169
170 if (Buffer_Offset+4<=Buffer_Size) //Testing if size is coherant
171 {
172 //Retrieving some info
173 if (Buffer_Offset+27>Buffer_Size)
174 return false; //Need more data
175 int8u page_segments=CC1(Buffer+Buffer_Offset+26);
176 if (Buffer_Offset+27+page_segments>Buffer_Size)
177 return false; //Need more data
178 size_t Size=0;
179 for (int8u Pos=0; Pos<page_segments; Pos++)
180 Size+=CC1(Buffer+Buffer_Offset+27+Pos);
181
182 //Testing
183 if (Buffer_Offset+27+page_segments+Size+4>Buffer_Size)
184 return false; //Need more data
185 if (CC4(Buffer+Buffer_Offset+27+page_segments+Size)!=0x4F676753) //"OggS"
186 Buffer_Offset++;
187 else
188 break;
189 }
190 }
191
192 //Parsing last bytes if needed
193 if (Buffer_Offset+4>Buffer_Size)
194 {
195 if (Buffer_Offset+3==Buffer_Size && CC3(Buffer+Buffer_Offset)!=0x4F6767) //"Ogg"
196 Buffer_Offset++;
197 if (Buffer_Offset+2==Buffer_Size && CC2(Buffer+Buffer_Offset)!=0x4F67) //"Og"
198 Buffer_Offset++;
199 if (Buffer_Offset+1==Buffer_Size && CC1(Buffer+Buffer_Offset)!=0x4F) //"O"
200 Buffer_Offset++;
201 return false;
202 }
203
204 //Synched is OK
205 return true;
206}

227//---------------------------------------------------------------------------
228void File_Ogg::Header_Parse()
229{
230 //Specific case
231 if (SizedBlocks)
232 {
233 int16u Size;
234 Get_B2 (Size, "Size");
235
236 Chunk_Sizes.clear();
237 Chunk_Sizes.push_back(Size);
238 Header_Fill_Size(2+Size);
239 Header_Fill_Code(0, Ztring::ToZtring(0, 16));
240 return;
241 }
242 if (XiphLacing)
243 {
244 if (Chunk_Sizes.empty())
245 {
246 int8u CountMinus1;
247 Get_B1 (CountMinus1, "Number of frames minus one");
248
249 int64u UsedSize=0;
250 for (size_t Pos=0; Pos<CountMinus1; Pos++)
251 {
252 int32u Size=0;
253 int8u Size8;
254 do
255 {
256 Get_B1 (Size8, "Size");
257 Size+=Size8;
258 }
259 while (Size8==0xFF);
260 Param_Info1(Size);
261 Chunk_Sizes.push_back(Size);
262 UsedSize+=Size;
263 }
264 Chunk_Sizes.push_back((size_t)(Element_Size-UsedSize-1));
265 }
266
267 Header_Fill_Size(Element_Size);
268 Header_Fill_Code(0, Ztring::ToZtring(0, 16));
269 return;
270 }
271
272 //Parsing
273 int64u absolute_granule_position;
274 int32u stream_serial_number, page_sequence_no;
275 int16u total_page_size;
276 int8u stream_structure_version, flags, page_segments, packet_lacing_value;
277 Skip_C4( "capture_pattern");
278 Get_L1 (stream_structure_version, "stream_structure_version");
279 Get_L1 (flags, "header_type_flag");
280 Get_Flags (flags, 0, continued, "continued packet");
281 Skip_Flags(flags, 1, "first page of logical bitstream (bos)");
282 Get_Flags (flags, 2, eos, "last page of logical bitstream (eos)");
283 Get_L8 (absolute_granule_position, "absolute granule position");
284 Get_L4 (stream_serial_number, "stream serial number");
285 Get_L4 (page_sequence_no, "page sequence no");
286 Skip_L4( "page checksum");
287 Get_L1 (page_segments, "page_segments");
288 total_page_size=0;
289 Chunk_Sizes.clear();
290 Chunk_Sizes.push_back(0);
291 for (int8u Pos=0; Pos<page_segments; Pos++)
292 {
293 Get_L1 (packet_lacing_value, "packet lacing value");
294 total_page_size+=packet_lacing_value;
295 Chunk_Sizes[Chunk_Sizes.size()-1]+=packet_lacing_value;
296 if (packet_lacing_value!=0xFF)
297 {
298 Chunk_Sizes.push_back(0);
299 Chunk_Sizes_Finished=true;
300 }
301 else
302 Chunk_Sizes_Finished=false;
303 }
304 if (Chunk_Sizes_Finished)
305 Chunk_Sizes.resize(Chunk_Sizes.size()-1); //Keep out the last value
306
307 //Filling
308 Header_Fill_Size(27+page_segments+total_page_size);
309 Header_Fill_Code(stream_serial_number, Ztring::ToZtring(stream_serial_number, 16));
310 Stream[stream_serial_number].absolute_granule_position=absolute_granule_position;
311}

313//---------------------------------------------------------------------------
314void File_Ogg::Data_Parse()
315{
316 //Counting
317 Frame_Count++;
318
319 //If first chunk of a stream
320 stream& Stream_Item=Stream[Element_Code]; //[+] FlylinkDC++ Team
321 if (Stream_Item.Parser==NULL)
322 {
323 if (Parsing_End)
324 return; //Maybe multitracks concatained, not supported
325 Stream_Item.Parser=new File_Ogg_SubElement;
326 Open_Buffer_Init(Stream_Item.Parser);
327 ((File_Ogg_SubElement*)Stream_Item.Parser)->InAnotherContainer=IsSub;
328 StreamsToDo++;
329 }
330 ((File_Ogg_SubElement*)Stream_Item.Parser)->MultipleStreams=Stream.size()>1; //has no sens for the first init, must check allways
331
332 //Parsing
333 File_Ogg_SubElement* Parser=(File_Ogg_SubElement*)Stream_Item.Parser;
334 if (Stream_Item.SearchingPayload)
335 //For each chunk
336 for (size_t Chunk_Sizes_Pos=0; Chunk_Sizes_Pos<Chunk_Sizes.size(); Chunk_Sizes_Pos++)
337 {
338 //Info
339 if (!continued)
340 Peek_L1(packet_type); //Only for information
341 Element_Info1(Ztring::ToZtring(packet_type, 16));
342 Element_Info1C((continued), "Continue");
343
344 //Parsing
345 if (continued || Parser->File_Offset!=Parser->File_Size)
346 {
347 int64u Size=Chunk_Sizes[Chunk_Sizes_Pos];
348 if (Element_Offset+Size>Element_Size)
349 Size=Element_Size-Element_Offset; // Shcunk size is bigger than content size, buggy file
350 Open_Buffer_Continue(Parser, Buffer+Buffer_Offset+(size_t)Element_Offset, Size);
351 }
352 if (Chunk_Sizes_Pos<Chunk_Sizes.size()-1
353 || (Chunk_Sizes_Pos==Chunk_Sizes.size()-1 && Chunk_Sizes_Finished))
354 {
355 Open_Buffer_Continue(Parser, Buffer+Buffer_Offset, 0); //Purge old datas
356 }
357
358 Element_Offset+=Chunk_Sizes[Chunk_Sizes_Pos];
359 continued=false; //If there is another chunk, this can not be a continued chunk
360 if (Parser->File_GoTo!=(int64u)-1)
361 Chunk_Sizes_Pos=Chunk_Sizes.size();
362
363 if (!Status[IsAccepted] && Parser->Status[IsAccepted])
364 Accept("OGG");
365 if (Parser->Status[IsFinished] || (Element_Offset==Element_Size && eos))
366 {
367 StreamsToDo--;
368 Stream_Item.SearchingPayload=false;
369 break;
370 }
371 }
372 else
373 Skip_XX(Element_Size, "Data");
374
375 //End of stream
376 if (!Parsing_End &&
377 (StreamsToDo==0 || File_Offset+Buffer_Offset+Element_Offset>256*1024))
378 {
379 if (IsSub)
380 Finish("OGG");
381 else
382 GoToFromEnd(256*1024, "OGG");
383 std::map<int64u, stream>::iterator Stream_Temp=Stream.begin();
384 if (File_GoTo!=(int64u)-1)
385 while (Stream_Temp!=Stream.end())
386 {
387 Stream_Temp->second.absolute_granule_position=0;
388 ++Stream_Temp;
389 }
390 Parsing_End=true;
391 }
392
393 Element_Show();
394}
