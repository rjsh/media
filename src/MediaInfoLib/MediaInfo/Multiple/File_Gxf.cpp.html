73 case 0x47 :

75 case 0x49 :

77 case 0x4B : return "Reserved";

289//---------------------------------------------------------------------------
290void File_Gxf::Streams_Finish()
291{
292 //Merging audio if Title are same
293 for (size_t StreamID=0; StreamID<Streams.size(); StreamID++)
294 {
295 if (Gxf_MediaTypes_StreamKind(Streams[StreamID].MediaType)==Stream_Video)
296 {
297 Ztring Title=Streams[StreamID].MediaName;
298 size_t Title_Extension_Offset=Title.find(__T(".M0"));
299 if (Title_Extension_Offset==std::string::npos || Title_Extension_Offset!=Title.size()-3)
300 Title_Extension_Offset=Title.find(__T(".H0"));
301 if (Title_Extension_Offset!=std::string::npos && Title_Extension_Offset==Title.size()-3)
302 {
303 Title.resize(Title.size()-3);
304 Streams[StreamID].MediaName=Title;
305 }
306 }
307 if (Gxf_MediaTypes_StreamKind(Streams[StreamID].MediaType)==Stream_Audio && Config->File_Audio_MergeMonoStreams_Get())
308 {
309 Ztring Title=Streams[StreamID].MediaName;
310 size_t Title_Extension_Offset=Title.find(__T(".A0"));
311 if (Title_Extension_Offset!=std::string::npos && Title_Extension_Offset==Title.size()-3)
312 {
313 Title.resize(Title.size()-3);
314 for (size_t StreamID2=StreamID+1; StreamID2<Streams.size(); StreamID2++)
315 {
316 if (Streams[StreamID2].MediaName==Title+__T(".A")+Ztring::ToZtring(StreamID2-StreamID))
317 {
318 Streams[StreamID].MediaName=Title;
319 if (Streams[StreamID].Parsers.size()==1 && Streams[StreamID2].Parsers.size()==1)
320 {
321 int32u Channels=Streams[StreamID].Parsers[0]->Retrieve(Stream_Audio, 0, Audio_Channel_s_).To_int32u()+Streams[StreamID2].Parsers[0]->Retrieve(Stream_Audio, 0, Audio_Channel_s_).To_int32u();
322 Streams[StreamID].Parsers[0]->Fill(Stream_Audio, 0, Audio_Channel_s_, Channels, 10, true);
323 int32u BitRate=Streams[StreamID].Parsers[0]->Retrieve(Stream_Audio, 0, Audio_BitRate).To_int32u()+Streams[StreamID2].Parsers[0]->Retrieve(Stream_Audio, 0, Audio_BitRate).To_int32u();
324 Streams[StreamID].Parsers[0]->Fill(Stream_Audio, 0, Audio_BitRate, BitRate, 10, true);
325 }
326 Streams[StreamID2].MediaType=(int8u)-1;
327 }
328 }
329 }
330 }
331 }
332
333 //For each Streams
334 for (size_t StreamID=0; StreamID<Streams.size(); StreamID++)
335 Streams_Finish_PerStream(StreamID, Streams[StreamID]);
336
337 //Global
338 if (Material_Fields_First_IsValid && Material_Fields_Last_IsValid && Material_Fields_Last-Material_Fields_First)
339 {
340 int64u FrameCount=(Material_Fields_Last+1-Material_Fields_First)/Material_Fields_FieldsPerFrame;
341 Fill(Stream_Video, 0, Video_FrameCount, FrameCount);
342 if (Gxf_FrameRate(Streams[0x00].FrameRate_Code))
343 Fill(Stream_Video, 0, Video_Duration, ((float64)FrameCount)/Gxf_FrameRate(Streams[0x00].FrameRate_Code)*1000, 0); //In milliseconds
344
345 //We trust more the MPEG Video bitrate thant the rest
346 //TODO: Chech why there is incohenrency (mainly about Material File size info in the sample)
347 if (Retrieve(Stream_Video, 0, Video_Format)==__T("MPEG Video"))
348 Fill(Stream_Video, 0, Video_BitRate, Retrieve(Stream_Video, 0, Video_BitRate_Nominal));
349 }
350 if (Material_File_Size_IsValid)
351 {
352 //Fill(Stream_General, 0, General_OverallBitRate, ((int64u)Material_File_Size)*1024*8/???);
353 }
354
355 //Time code tracks
356 for (std::map<int8u, tc>::iterator TimeCode=TimeCodes.begin(); TimeCode!=TimeCodes.end(); ++TimeCode)
357 {
358 int64u TimeCode_FirstFrame_ms=((File_Gxf_TimeCode*)Streams[TimeCode->first].Parsers[0])->TimeCode_FirstFrame_ms;
359 string TimeCode_FirstFrame=((File_Gxf_TimeCode*)Streams[TimeCode->first].Parsers[0])->TimeCode_FirstFrame;
360 bool TimeCode_FirstFrame_Striped=false;
361 if (TimeCode_FirstFrame_ms==(int64u)-1)
362 {
363 TimeCode_FirstFrame_ms=TimeCode->second.Milliseconds;
364 TimeCode_FirstFrame=TimeCode->second.String;
365 TimeCode_FirstFrame_Striped=true;
366 }
367 if (TimeCode_FirstFrame_ms!=(int64u)-1)
368 {
369 Stream_Prepare(Stream_Other);
370 Fill(Stream_Other, StreamPos_Last, Other_CodecID, Streams[TimeCode->first].MediaType);
371 Fill(Stream_Other, StreamPos_Last, Other_ID, TimeCode->first);
372 Fill(Stream_Other, StreamPos_Last, Other_Type, "Time code");
373 Fill(Stream_Other, StreamPos_Last, Other_Format, "SMPTE TC");
374 Fill(Stream_Other, StreamPos_Last, Other_TimeCode_FirstFrame, TimeCode_FirstFrame.c_str());
375 if (TimeCode_FirstFrame_Striped)
376 Fill(Stream_Other, StreamPos_Last, Other_TimeCode_Striped, "Yes");
377 if (TimeCode->first<Streams.size())
378 Fill(Stream_Other, StreamPos_Last, Other_Title, Streams[TimeCode->first].MediaName);
379 }
380 }
381}

383//---------------------------------------------------------------------------
384void File_Gxf::Streams_Finish_PerStream(size_t StreamID, stream &Temp)
385{
386 if (Temp.MediaType==(int8u)-1)
387 return;
388
389 //By the parser
390 if (Temp.Parsers.size()==1 && Temp.Parsers[0]->Status[IsAccepted])
391 {
392 StreamKind_Last=Stream_Max;
393 StreamPos_Last=(size_t)-1;
394 if (Config->ParseSpeed<=1.0)
395 {
396 Fill(Temp.Parsers[0]);
397 Temp.Parsers[0]->Open_Buffer_Unsynch();
398 }
399 Finish(Temp.Parsers[0]);
400
401 //Video
402 bool IsTimeCode=false;
403 for (std::map<int8u, tc>::iterator TimeCode=TimeCodes.begin(); TimeCode!=TimeCodes.end(); ++TimeCode)
404 if (StreamID==TimeCode->first)
405 IsTimeCode=true;
406 if (!IsTimeCode && Temp.DisplayInfo)
407 {
408 if (Temp.Parsers[0]->Count_Get(Stream_Video))
409 {
410 Stream_Prepare(Stream_Video);
411
412 if (TimeCodes.empty())
413 {
414 Fill(Stream_Video, StreamPos_Last, Video_Delay, ((float64)(Material_Fields_First/Material_Fields_FieldsPerFrame))/Gxf_FrameRate(Streams[0x00].FrameRate_Code)*1000, 0);
415 Fill(Stream_Video, StreamPos_Last, Video_Delay_Source, "Container");
416 }
417 else
418 for (std::map<int8u, tc>::iterator TimeCode=TimeCodes.begin(); TimeCode!=TimeCodes.end(); ++TimeCode)
419 {
420 int64u TimeCode_FirstFrame_ms=((File_Gxf_TimeCode*)Streams[TimeCode->first].Parsers[0])->TimeCode_FirstFrame_ms;
421 string TimeCode_FirstFrame=((File_Gxf_TimeCode*)Streams[TimeCode->first].Parsers[0])->TimeCode_FirstFrame;
422 if (TimeCode_FirstFrame_ms==(int64u)-1)
423 {
424 TimeCode_FirstFrame_ms=TimeCode->second.Milliseconds;
425 TimeCode_FirstFrame=TimeCode->second.String;
426 }
427 if (TimeCode_FirstFrame_ms!=(int64u)-1)
428 {
429 Fill(Stream_Video, StreamPos_Last, Video_Delay, TimeCode_FirstFrame_ms, 0);
430 if (TimeCode_FirstFrame.size()==11)
431 Fill(Stream_Video, StreamPos_Last, Video_Delay_DropFrame, TimeCode_FirstFrame[8]==';'?"Yes":"No");
432 Fill(Stream_Video, StreamPos_Last, Video_Delay_Source, "Container");
433
434 //Fill(Stream_Video, StreamPos_Last, Video_TimeCode_FirstFrame, TimeCode_FirstFrame.c_str());
435 //Fill(Stream_Video, StreamPos_Last, Video_TimeCode_Source, "Time code track");
436 }
437 }
438
439 Merge(*Temp.Parsers[0], Stream_Video, 0, StreamPos_Last);
440 Fill(Stream_Video, StreamPos_Last, Video_CodecID, Temp.MediaType);
441
442 Fill(Stream_Video, StreamPos_Last, Video_ID, StreamID, 10, true);
443 Fill(Stream_Video, StreamPos_Last, "Title", Temp.MediaName);
444
445 Ztring LawRating=Temp.Parsers[0]->Retrieve(Stream_General, 0, General_LawRating);
446 if (!LawRating.empty())
447 Fill(Stream_General, 0, General_LawRating, LawRating, true);
448 Ztring Title=Temp.Parsers[0]->Retrieve(Stream_General, 0, General_Title);
449 if (!Title.empty() && Retrieve(Stream_General, 0, General_Title).empty())
450 Fill(Stream_General, 0, General_Title, Title);
451
452 //Special cases
453 if (Temp.Parsers[0]->Count_Get(Stream_Text))
454 {
455 //Video and Text are together
456 size_t Parser_Text_Count=Temp.Parsers[0]->Count_Get(Stream_Text);
457 for (size_t Parser_Text_Pos=0; Parser_Text_Pos<Parser_Text_Count; Parser_Text_Pos++)
458 {
459 Stream_Prepare(Stream_Text);
460 Merge(*Temp.Parsers[0], Stream_Text, Parser_Text_Pos, StreamPos_Last);
461 Ztring ID=Retrieve(Stream_Text, StreamPos_Last, Text_ID);
462 Fill(Stream_Text, StreamPos_Last, Text_ID, Ztring::ToZtring(AncillaryData_StreamID)+__T("-")+ID, true);
463 Fill(Stream_Text, StreamPos_Last, Text_ID_String, Ztring::ToZtring(AncillaryData_StreamID)+__T("-")+ID, true);
464 Fill(Stream_Text, StreamPos_Last, Text_Delay, Retrieve(Stream_Video, Count_Get(Stream_Video)-1, Video_Delay), true);
465 Fill(Stream_Text, StreamPos_Last, Text_Delay_Source, Retrieve(Stream_Video, Count_Get(Stream_Video)-1, Video_Delay_Source), true);
466 Fill(Stream_Text, StreamPos_Last, Text_Delay_Original, Retrieve(Stream_Video, Count_Get(Stream_Video)-1, Video_Delay_Original), true);
467 Fill(Stream_Text, StreamPos_Last, Text_Delay_Original_Source, Retrieve(Stream_Video, Count_Get(Stream_Video)-1, Video_Delay_Original_Source), true);
468 }
469
470 StreamKind_Last=Stream_Video;
471 StreamPos_Last=Count_Get(Stream_Video)-1;
472 }
473 }
474
475 //Audio
476 for (size_t Pos=0; Pos<Temp.Parsers[0]->Count_Get(Stream_Audio); Pos++)
477 {
478 Stream_Prepare(Stream_Audio);
479 Fill(Stream_Audio, StreamPos_Last, Audio_CodecID, Temp.MediaType);
480
481 if (TimeCodes.empty())
482 {
483 Fill(Stream_Audio, StreamPos_Last, Audio_Delay, ((float64)(Material_Fields_First/Material_Fields_FieldsPerFrame))/Gxf_FrameRate(Streams[0x00].FrameRate_Code)*1000, 0);
484 Fill(Stream_Audio, StreamPos_Last, Audio_Delay_Source, "Container");
485 }
486 else
487 for (std::map<int8u, tc>::iterator TimeCode=TimeCodes.begin(); TimeCode!=TimeCodes.end(); ++TimeCode)
488 {
489 int64u TimeCode_FirstFrame_ms=((File_Gxf_TimeCode*)Streams[TimeCode->first].Parsers[0])->TimeCode_FirstFrame_ms;
490 string TimeCode_FirstFrame=((File_Gxf_TimeCode*)Streams[TimeCode->first].Parsers[0])->TimeCode_FirstFrame;
491 if (TimeCode_FirstFrame_ms==(int64u)-1)
492 {
493 TimeCode_FirstFrame_ms=TimeCode->second.Milliseconds;
494 TimeCode_FirstFrame=TimeCode->second.String;
495 }
496 if (TimeCode_FirstFrame_ms!=(int64u)-1)
497 {
498 Fill(Stream_Audio, StreamPos_Last, Audio_Delay, TimeCode_FirstFrame_ms, 0);
499 if (TimeCode_FirstFrame.size()==11)
500 Fill(Stream_Audio, StreamPos_Last, Audio_Delay_DropFrame, TimeCode_FirstFrame[8]==';'?"Yes":"No");
501 Fill(Stream_Audio, StreamPos_Last, Audio_Delay_Source, "Container");
502
503 //Fill(Stream_Audio, StreamPos_Last, Audio_TimeCode_FirstFrame, TimeCode_FirstFrame.c_str());
504 //Fill(Stream_Audio, StreamPos_Last, Audio_TimeCode_Source, "Time code track");
505 }
506 }
507
508 Merge(*Temp.Parsers[0], Stream_Audio, Pos, StreamPos_Last, false);
509
510 Ztring ID;
511 if (Temp.IsChannelGrouping)
512 ID=Ztring::ToZtring(StreamID-1)+__T(" / "); //First half of the channel grouping
513 ID+=Ztring::ToZtring(StreamID); //Second half of the channel grouping or standalone
514 Ztring ID_String=ID;
515 if (!Retrieve(Stream_Audio, StreamPos_Last, Audio_ID).empty())
516 {
517 ID+=__T('-')+Retrieve(Stream_Audio, StreamPos_Last, Audio_ID);
518 ID_String+=__T('-')+Retrieve(Stream_Audio, StreamPos_Last, Audio_ID_String);
519 }
520 Fill(Stream_Audio, StreamPos_Last, Audio_ID, ID, true);
521 Fill(Stream_Audio, StreamPos_Last, Audio_ID_String, ID_String, true);
522 if (Temp.IsChannelGrouping)
523 Fill(Stream_Audio, StreamPos_Last, "Title", Streams[StreamID-1].MediaName); //First half of the channel grouping
524 Fill(StreamKind_Last, StreamPos_Last, "Title", Temp.MediaName); //Second half of the channel grouping or standalone
525
526 for (std::map<std::string, Ztring>::iterator Info=Temp.Infos.begin(); Info!=Temp.Infos.end(); ++Info)
527 if (Info->first=="BitRate" && Temp.Parsers[0]->Count_Get(Stream_Audio)>1)
528 Fill(Stream_Audio, StreamPos_Last, Audio_BitRate_Encoded, Pos?Ztring(__T("0")):Info->second); // In case of more than 1 audio sub-stream Encoded bit rate is the bit rate of all streams + overhead
529 else if (Retrieve(Stream_Audio, StreamPos_Last, Info->first.c_str()).empty())
530 Fill(Stream_Audio, StreamPos_Last, Info->first.c_str(), Info->second);
531 }
532
533 //Text
534 if (Temp.Parsers[0]->Count_Get(Stream_Text))
535 {
536 size_t Parser_Text_Count=Temp.Parsers[0]->Count_Get(Stream_Text);
537 for (size_t Parser_Text_Pos=0; Parser_Text_Pos<Parser_Text_Count; Parser_Text_Pos++)
538 {
539 Stream_Prepare(Stream_Text);
540 Merge(*Temp.Parsers[0], Stream_Text, Parser_Text_Pos, StreamPos_Last);
541 Ztring ID=Retrieve(Stream_Text, StreamPos_Last, Text_ID);
542 Fill(Stream_Text, StreamPos_Last, Text_ID, Ztring::ToZtring(AncillaryData_StreamID)+__T("-")+ID, true);
543 Fill(Stream_Text, StreamPos_Last, Text_ID_String, Ztring::ToZtring(AncillaryData_StreamID)+__T("-")+ID, true);
544 Fill(Stream_Text, StreamPos_Last, Text_Delay, Retrieve(Stream_Video, Count_Get(Stream_Video)-1, Video_Delay), true);
545 Fill(Stream_Text, StreamPos_Last, Text_Delay_Source, Retrieve(Stream_Video, Count_Get(Stream_Video)-1, Video_Delay_Source), true);
546 Fill(Stream_Text, StreamPos_Last, Text_Delay_Original, Retrieve(Stream_Video, Count_Get(Stream_Video)-1, Video_Delay_Original), true);
547 Fill(Stream_Text, StreamPos_Last, Text_Delay_Original_Source, Retrieve(Stream_Video, Count_Get(Stream_Video)-1, Video_Delay_Original_Source), true);
548 Fill(Stream_Text, StreamPos_Last, "Title", Temp.MediaName);
549 }
550
551 Ztring LawRating=Temp.Parsers[0]->Retrieve(Stream_General, 0, General_LawRating);
552 if (!LawRating.empty())
553 Fill(Stream_General, 0, General_LawRating, LawRating, true);
554 Ztring Title=Temp.Parsers[0]->Retrieve(Stream_General, 0, General_Title);
555 if (!Title.empty() && Retrieve(Stream_General, 0, General_Title).empty())
556 Fill(Stream_General, 0, General_Title, Title);
557
558 StreamKind_Last=Stream_Max;
559 StreamPos_Last=(size_t)-1;
560 }
561
562 //Other
563 if (Temp.Parsers[0]->Count_Get(Stream_Other))
564 {
565 size_t Parser_Other_Count=Temp.Parsers[0]->Count_Get(Stream_Other);
566 for (size_t Parser_Other_Pos=0; Parser_Other_Pos<Parser_Other_Count; Parser_Other_Pos++)
567 {
568 Stream_Prepare(Stream_Other);

594//---------------------------------------------------------------------------
595bool File_Gxf::Synchronize()
596{
597 //Synchronizing
598 while (Buffer_Offset+16<=Buffer_Size)
599 {
600 while (Buffer_Offset+16<=Buffer_Size && (Buffer[Buffer_Offset ]!=0x00
601 || Buffer[Buffer_Offset+1]!=0x00
602 || Buffer[Buffer_Offset+2]!=0x00
603 || Buffer[Buffer_Offset+3]!=0x00
604 || Buffer[Buffer_Offset+4]!=0x01
605 || Buffer[Buffer_Offset+14]!=0xE1
606 || Buffer[Buffer_Offset+15]!=0xE2))
607 {
608 Buffer_Offset+=4;
609 while (Buffer_Offset<Buffer_Size && Buffer[Buffer_Offset]!=0x00)
610 Buffer_Offset+=4;
611 for (int8u Pos=0; Pos<3; Pos++)
612 if (Buffer_Offset>=Buffer_Size || Buffer[Buffer_Offset-1]==0x00)
613 Buffer_Offset--;
614 }
615
616 if (Buffer_Offset+16<=Buffer_Size) //Testing if size is coherant
617 {
618 //Retrieving some info
619 int32u Size=BigEndian2int32u(Buffer+Buffer_Offset+6);
620
621 //Testing
622 if (Buffer_Offset+Size+16>Buffer_Size)
623 return false; //Need more data
624 if (Buffer[Buffer_Offset+Size ]!=0x00
625 || Buffer[Buffer_Offset+Size+1]!=0x00
626 || Buffer[Buffer_Offset+Size+2]!=0x00
627 || Buffer[Buffer_Offset+Size+3]!=0x00
628 || Buffer[Buffer_Offset+Size+4]!=0x01
629 || Buffer[Buffer_Offset+Size+14]!=0xE1
630 || Buffer[Buffer_Offset+Size+15]!=0xE2)
631 Buffer_Offset++;
632 else
633 break;
634 }
635 }
636
637 //Parsing last bytes if needed
638 if (Buffer_Offset+16>Buffer_Size)
639 {
640 return false;
641 }
642
643 if (!Status[IsAccepted])
644 {
645 Accept("GXF");
646 Fill(Stream_General, 0, General_Format, "GXF");
647 Streams.resize(0x40);
648 }
649
650 //Synched is OK
651 return true;
652}

657 //Must have enough buffer for having header

662 if (CC5(Buffer+Buffer_Offset )!=0x0000000001

1274 if (TrackID==TimeCode->first)
1275 for (size_t Pos=0; Pos<Streams[TrackID].Parsers.size(); Pos++)
1276 ((File_Gxf_TimeCode*)Streams[TrackID].Parsers[Pos])->FieldsPerFrame_Code=Streams[0x00].FieldsPerFrame_Code;
1277 }
1278 else
1279 Skip_XX(DataLength, "Unknown");
1280 break;
1281 default : Skip_XX(DataLength, "Unknown");
1282 }
1283 Element_End0();
1284 }
1285 Element_End0();
1286
1287 //Test on TimeCode
1288 if (TimeCode_Parsed && !Invalid)
1289 {
1290 std::map<int8u, tc>::iterator TimeCode=TimeCodes.find(TrackID);
1291 if (TimeCode==TimeCodes.end() || TimeCode->second.Milliseconds==(int64u)-1)
1292 {
1293 float64 FrameRate=Gxf_FrameRate(Streams[TrackID].FrameRate_Code);
1294 TimeCodes[TrackID].Milliseconds=Hours *60*60*1000
1295 +Minutes *60*1000
1296 +Seconds *1000;
1297 MediaInfoLib::TimeCode TC;
1298 TC.Hours=Hours;
1299 TC.Minutes=Minutes;
1300 TC.Seconds=Seconds;
1301 TC.Frames=Fields/2;
1302 TC.DropFrame=DropFrame;
1303 TimeCodes[TrackID].String=TC.ToString();
1304
1305 if (!FrameRate)
1306 {
1307 //Time code frame rate is missing, using the video frame rate
1308 for (size_t Pos=0; Pos<Streams.size(); Pos++)
1309 if (Streams[Pos].FrameRate_Code!=(int32u)-1)
1310 {
1311 FrameRate=Gxf_FrameRate(Streams[Pos].FrameRate_Code);
1312 break;
1313 }
1314 }
1315 if (FrameRate)
1316 float64_int64s(Fields*1000/(FrameRate*2));
1317 }
1318 }
1319 }
1320 Element_End0();
1321 if (Element_Offset<Element_Size)
1322 Skip_XX(Element_Size-Element_Offset, "Padding");
1323
1324 //Filling missing frame rates for PCM
1325 for (size_t TrackID=0; TrackID<Streams.size(); TrackID++)
1326 {
1327 if (Gxf_FrameRate(Streams[TrackID].FrameRate_Code)==0)
1328 {
1329 Streams[TrackID].FrameRate_Code=Stream_Video_FrameRate_Code;
1330 Streams[TrackID].FieldsPerFrame_Code=Stream_Video_FieldsPerFrame_Code;
1331 }
1332 if (Material_Fields_First_IsValid && Gxf_MediaTypes_StreamKind(Streams[TrackID].MediaType)==Stream_Audio) //In case of offset, MediaFieldNumber-Material_Fields_First is not well rounded
1333 {
1334 float64 Temp=((float64)Material_Fields_First/Streams[TrackID].FieldsPerFrame_Code)/Gxf_FrameRate(Streams[TrackID].FrameRate_Code);
1335 Temp*=48000; //TODO: find where this piece of info is available
1336 Temp/=32768;
1337 Temp-=(int64u)(Temp);
1338 if (Temp)
1339 {
1340 Temp=((float64)32768)/48000*(1-Temp); //Duration of the first frame not counted
1341 Streams[TrackID].FirstFrameDuration=float64_int64s(Temp*1000000000);
1342 }
1343 }
1344 }

1391 IFrame_IsParsed=true;

1398 Element_Code=TrackNumber;
1399 int64u TimeCode_First=0;
1400 if (!TimeCodes.empty())

1402 TimeCode_First=((File_Gxf_TimeCode*)Streams[TimeCodes.begin()->first].Parsers[0])->TimeCode_FirstFrame_ms;
