77//***************************************************************************
78// Buffer - File header
79//***************************************************************************
80
81//---------------------------------------------------------------------------
82bool File_Ptx::FileHeader_Begin()
83{
84 if (File_Size<0x100)
85 {
86 Reject("Ptx");
87 return false;
88 }
89
90 //Element_Size
91 if (11>Buffer_Size)
92 return false; //Must wait for more data
93
94 if (Buffer[ 0x0]!=0x03
95 || Buffer[ 0x1]!=0x30
96 || Buffer[ 0x2]!=0x30
97 || Buffer[ 0x3]!=0x31
98 || Buffer[ 0x4]!=0x30
99 || Buffer[ 0x5]!=0x31
100 || Buffer[ 0x6]!=0x31
101 || Buffer[ 0x7]!=0x31
102 || Buffer[ 0x8]!=0x31
103 || Buffer[ 0x9]!=0x30
104 || Buffer[ 0xA]!=0x30
105 || Buffer[ 0xB]!=0x31
106 || Buffer[ 0xC]!=0x30
107 || Buffer[ 0xD]!=0x31
108 || Buffer[ 0xE]!=0x30
109 || Buffer[ 0xF]!=0x31
110 || Buffer[0x10]!=0x31)
111 {
112 Reject("Ptx");
113 return false;
114 }
115
116 //Element_Size
117 if (Buffer_Size<File_Size)
118 return false; //Must wait for more data
119
120 ReferenceFiles=new File__ReferenceFilesHelper(this, Config);
121
122 //All should be OK...
123 return true;
124}

221 Element_End();

233 Element_End();

236 Skip_L4( "0x00000000");
237 Skip_L4( "0x00000000 or 0x0000002A");
238 Skip_L4( "Unknown [7]");
239 Skip_L4( "Unknown [8]");
240 Element_Begin1("Names list 2");
241 Get_L4 (Names_Count, "Names count");
242 if (4*Names_Count>Element_Size)
243 {
244 Element_End();
245 Reject();
246 return;
247 }

249 {
250 Element_Begin1("Name");
251 Get_L4 (Name_Length, "Name length");

254 Info_UTF8(Name_Length, Name, "Name"); Element_Name(Name);
255 }
256 Element_End();
257 }
258 Element_End();

260 Skip_L4( "0x0000002A [1]");
261 Skip_L4( "Unknown [9]");
262 Skip_L4( "Unknown [10]");
263 Skip_L4( "0x00000000");

265 Skip_L1( "0x01");

292 }
293 Skip_L2( "0x2519");
294 Skip_L2( "0x0001");
295 Skip_L4( "0x00000000 or B5112287");
296 Skip_L4( "0x00000000 or 4037F9DC");
297 Skip_L4( "0x00000001 [1]");
298 Skip_L2( "0x0003");
299 Element_End();
300 Get_L4 (Audio_Count, "Audio count");
301 if (111*Audio_Count>Element_Size)
302 {
303 Reject();
304 return;
305 }
306 Element_Begin1("Audio tracks list 1");
307 for (int16u Pos=0; Pos<Audio_Count; Pos++)
308 {
309 Element_Begin1("Name");
310 Skip_L2( "0x0000 [New]");
311 Get_L4 (Name_Length, "(Same 1/2/3) Name length");
312 Info_UTF8(Name_Length, Name, "(Same 1/2/3) Name");
313 Skip_L2( "(Same 1/2/3) 0x0000 ");

319 Element_Info1(Name);

321
322 if (Name==__T("Lf")) //Exception? Typo?
323 Name=__T("Lfe");
324 Name.MakeLowerCase();

326 }
327 Element_End();
328 Element_Begin1("Audio tracks list 2");
329 for (int16u Pos=0; Pos<Audio_Count; Pos++)
330 {
331 Element_Begin1("Name");
332 int32u Size;
333 Skip_L3( "(Same 2/3) 0x00025A [1]");
334 Get_L4 (Size, "(Same 2/3) Size");
335 Skip_L4( "(Same 2/3) 0x0000251A");
336 Get_L4 (Name_Length, "(Same 1/2/3) Name length");
337 Info_UTF8(Name_Length, Name, "(Same 1/2/3) Name");
338 Skip_L2( "(Same 1/2/3) 0x0000 ");
339 Skip_L4( "(Same 1/2/3) 0x00000000");
340 Skip_L4( "(Same 1/2/3) 0x0000002A");
341 Skip_L4( "(Same 1/2/3) Unknown");
342 Skip_L4( "(Same 1/2/3) Unknown");
343 Info_L3(Number, "(Same 1/2/3) Ordered number"); Element_Info1(Number);
344 Skip_L2( "(Same 2/3) 0x0000");
345 Element_Info1(Name);
346 if (Name_Length+31!=Size)
347 {
348 Element_End();
349 Element_End();
350 Reject();
351 return;
352 }
353 Element_End();
354 }
355 Element_End();
356 Get_L4 (Audio_Count, "Audio count");
357 if (4*Audio_Count>Element_Size)
358 {
359 Reject();
360 return;
361 }
362 Element_Begin1("Audio tracks list 3");
363 for (int16u Pos=0; Pos<Audio_Count; Pos++)
364 {
365 Element_Begin1("Name");
366 int32u Size;
367 Skip_L3( "(Same 2/3) 0x00025A [2]");
368 Get_L4 (Size, "(Same 2/3) Size");
369 if (Size>0x10000)
370 {
371 Element_End();
372 Element_End();
373 Reject();
374 return;
375 }
376 Skip_L4( "(Same 2/3) 0x0000251A");
377 Get_L4 (Name_Length, "(Same 1/2/3) Name length");
378 Info_UTF8(Name_Length, Name, "(Same 1/2/3) Name");
379 Skip_L2( "(Same 1/2/3) 0x0000 ");
380 Skip_L4( "(Same 1/2/3) 0x00000000");
381 Skip_L4( "(Same 1/2/3) 0x0000002A");
382 Skip_L4( "(Same 1/2/3) Unknown");
383 Skip_L4( "(Same 1/2/3) Unknown");
384 Info_L3(Number, "(Same 1/2/3) Ordered number"); Element_Info1(Number);
385 Skip_L2( "(Same 2/3) 0x0000");
386 Element_Info1(Name);
387 if (Name_Length+31!=Size)
388 {
389 Element_End();
390 Element_End();
391 Reject();
392 return;
393 }
394 Element_End();
395 }
396 Element_End();
397 Skip_L2( "0x0000 [4]");
398 Skip_L2( "0x0018");
399 Skip_L4( "0x00000001 [2]");
400 Skip_L2( "0x0018");
401 Skip_L4( "0x00000001 [2]");
402 Skip_L2( "0x0001 [3]");
403 Skip_L3( "0x00095A");
404 Get_L4 (Opaque2_Length, "Opaque2 length");
405 Skip_XX(Opaque2_Length, "Opaque2");
406 Skip_L1( "0x5A [2]");
407 Skip_L2( "0x0003 (10.0) or 0x0004 (10.2+)");
408 Get_L4 (Opaque2_Length, "Opaque3 length"); //0x0012 (10.0) or 0x0016 (10.2+)
409 if (Opaque2_Length<0x12)
410 Skip_XX(Opaque2_Length, "Opaque3");
411 else
412 {
413 Skip_L4( "Opaque3 - 0x06002026");
414 Skip_L4( "Opaque3 - 0x00000000 [1]");
415 Skip_L2( "Opaque3 - 0x0000");
416 Skip_L4( "Opaque3 - Unknown [1]");
417 Skip_L4( "Opaque3 - Unknown [2]");
418 if (Opaque2_Length<0x16)
419 {
420 if (Opaque2_Length-0x12)
421 Skip_XX(Opaque2_Length-0x12, "Opaque3 - Unknown [3]");
422 }
423 else
424 {
425 Skip_L4( "Opaque3 - 0x00000000 [2]");
426 if (Opaque2_Length>0x16)
427 Skip_XX(Opaque2_Length-0x16, "Opaque3 - Unknown [4]");
428 }
429 }
430 Skip_L3( "0x00025A [3]");
431 Get_L4 (Opaque2_Length, "0x00000015 (Opaque4 length?) or something else");
432 if (Opaque2_Length==0x00000015)
433 {
434 Skip_L4( "0x075A2032");
435 Skip_L4( "0x00000C00");
436 Skip_L4( "0x01204200");
437 Skip_L4( "0x00000000 or 0x01000000");
438 Skip_L4( "Unknown [13]");
439 Skip_L1( "0x00 [3]");
440 Skip_L3( "0x00025A [4]");
441 Skip_L4( "Unknown [14]");
442 }
443 Skip_L4( "Unknown [15]");
444 Skip_L4( "0x015A0000");
445 Skip_L4( "Unknown [16]");
446 Skip_L4( "Unknown [17]");
447 Skip_L4( "0x01000000");
448 Get_L4 (FileName_Count, "File name count");
449 if (13*FileName_Count>Element_Size)
450 {
451 Reject();
452 return;
453 }
454 Get_L4 (Directory_Length, "Directory length");
455 Get_UTF8(Directory_Length, Directory, "Directory");
456 Skip_L4( "0x00000000 [11]");
457 Element_Begin1("File names");
458 vector<int8u> Roles;
459 vector<Ztring> FileNames;
460 vector<Ztring> FileNamesLowerCase;

462 for (int32u Pos=0; Pos<FileName_Count; Pos++)

464 Ztring FileName;
465 int32u FileName_Length, Purpose;
466 int8u Role; //
467 Element_Begin1("File names");
468 Get_L1 (Role, "role? (0x02 for WAV files)");

470 Get_L4 (FileName_Length, "File Name length");

479 Purposes.push_back(Purpose);
480 }
481 Element_End();
482 Skip_XX(Element_Size-Element_Offset, "Unknown");

485 Accept("Ptx"); //Could be Ptf (former format but not supported, so we don't care currently)
486 Fill("Ptx");

491
492 // Role==2 + Purpose==EWAV + listed
493 if (Names.size()>1 || FileNames.size()==1)
494 {
495 size_t Pos_Offset=0;
496 for (int32u Pos=0; Pos<FileName_Count; Pos++)
