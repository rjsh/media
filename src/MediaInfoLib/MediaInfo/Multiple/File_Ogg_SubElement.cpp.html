158//---------------------------------------------------------------------------
159void File_Ogg_SubElement::Streams_Fill()
160{
161 if (Parser==NULL)
162 return;
163
164 Fill(Parser);
165 if (Parser->Count_Get(Stream_Video))
166 {
167 //Hack - Before
168 Ztring Codec_Temp=Retrieve(Stream_Video, 0, Video_Codec); //We want to keep the 4CC of AVI
169
170 Merge(*Parser, Stream_Video, 0, 0);
171
172 //Hacks - After
173 if (!Codec_Temp.empty())
174 Fill(Stream_Video, StreamPos_Last, Video_Codec, Codec_Temp, true);
175 }
176 if (Parser->Count_Get(Stream_Audio))
177 {
178 //Hack - Before
179 Ztring Codec_Temp=Retrieve(Stream_Audio, 0, Audio_Codec); //We want to keep the 2CC of AVI
180
181 Merge(*Parser, Stream_Audio, 0, 0);
182
183 //Hacks - After
184 if (!Codec_Temp.empty())
185 Fill(Stream_Audio, StreamPos_Last, Audio_Codec, Codec_Temp, true);
186 }
187 Merge(*Parser, Stream_Text, 0, 0);
188 Merge(*Parser, Stream_Image, 0, 0);
189}

238//---------------------------------------------------------------------------
239bool File_Ogg_SubElement::Header_Begin()
240{
241 //Already parsed (there is only one pass)
242 if (Buffer_Offset!=0)
243 return false;
244
245 //We are waiting for the end of the stream, signaled by a empty buffer adding
246 if (Buffer_Size!=OldSize)
247 {
248 OldSize=Buffer_Size;
249 return false;
250 }
251 else
252 {
253 OldSize=0;
254 return true;
255 }
256}

258//---------------------------------------------------------------------------
259void File_Ogg_SubElement::Header_Parse()
260{
261 //Parsing
262 int8u Type;
263 if (Identified && WithType)
264 {
265 bool lenbytes0, lenbytes1, lenbytes2;
266 Get_L1 (Type, "Type");
267 Skip_Flags(Type, 0, "Indicates data packet");
268 Get_Flags (Type, 1, lenbytes2, "Bit 2 of lenbytes");
269 Skip_Flags(Type, 2, "unused");
270 Skip_Flags(Type, 3, "Keyframe");
271 Skip_Flags(Type, 4, "unused");
272 Skip_Flags(Type, 5, "unused");
273 Get_Flags (Type, 6, lenbytes0, "Bit 0 of lenbytes");
274 Get_Flags (Type, 7, lenbytes1, "Bit 1 of lenbytes");
275 if ((Type&0x01)==0) //TODO : find a better algo
276 {
277 if (lenbytes2)
278 {
279 if (lenbytes1)
280 {
281 if (lenbytes0)
282 Skip_L7( "SamplesCount");
283 else
284 Skip_L6( "SamplesCount");
285 }
286 else
287 {
288 if (lenbytes0)
289 Skip_L5( "SamplesCount");
290 else
291 Skip_L4( "SamplesCount");
292 }
293 }
294 else
295 {
296 if (lenbytes1)
297 {
298 if (lenbytes0)
299 Skip_L3 ( "SamplesCount");
300 else
301 Skip_L2 ( "SamplesCount");
302 }
303 else
304 {
305 if (lenbytes0)
306 Skip_L1 ( "SamplesCount");
307 }
308 }
309 }
310 //Filling
311 Header_Fill_Code(Type, Ztring::ToZtring(Type, 16));
312 }
313 else
314 //Filling
315 Header_Fill_Code(0, "Identification");
316
317 //Filling
318 Header_Fill_Size(Element_Size);
319}

321//---------------------------------------------------------------------------
322void File_Ogg_SubElement::Data_Parse()
323{
324 //Parsing
325 if (!Identified)
326 Identification();
327 else if (!WithType)
328 Default();
329 else
330 switch (Element_Code&0x7F)
331 {
332 case 0x01 :
333 case 0x03 : Comment(); break;
334 case 0x00 :
335 case 0x02 :
336 case 0x05 :
337 case 0x08 : Default(); break;
338 default : Skip_XX(Element_Size, "Unknown");
339 Finish("OggSubElement");
340 }
341}

343//---------------------------------------------------------------------------
344void File_Ogg_SubElement::Identification()
345{
346 Element_Name("Identification");
347
348 //Parsing
349 int64u ID_Identification;
350 if (Element_Size==4)
351 {
352 int32u ID_Identification_32;
353 Peek_B4(ID_Identification_32);
354 ID_Identification=((int64u)ID_Identification_32)<<32;
355 }
356 else
357 Peek_B8(ID_Identification);
358
359 //Filling
360 #undef ELEMENT_CASE
361 #ifdef __BORLANDC__ //Borland converts int64u to int32u
362 #define ELEMENT_CASE(_NAME) \
363 else if (ID_Identification>>(64-8*Elements::Identifier_##_NAME##3)==(((int64u)Elements::Identifier_##_NAME##1)*0x100000000LL+Elements::Identifier_##_NAME##2)) Identification_##_NAME();
364
365 #else //__BORLANDC__
366 #define ELEMENT_CASE(_NAME) \
367 else if (ID_Identification>>(64-8*Elements::Identifier_##_NAME##3)==Elements::Identifier_##_NAME) Identification_##_NAME();
368
369 #endif //__BORLANDC__
370
371 if (0) ;
372 ELEMENT_CASE(CELT)
373 ELEMENT_CASE(CMML)
374 ELEMENT_CASE(BBCD)
375 ELEMENT_CASE(FLAC)
376 ELEMENT_CASE(JNG)
377 ELEMENT_CASE(kate)
378 ELEMENT_CASE(KW_DIRAC)
379 ELEMENT_CASE(OggMIDI)
380 ELEMENT_CASE(MNG)
381 ELEMENT_CASE(OpusHead)
382 ELEMENT_CASE(PCM)
383 ELEMENT_CASE(PNG)
384 ELEMENT_CASE(Speex)
385 ELEMENT_CASE(theora)
386 ELEMENT_CASE(vorbis)
387 ELEMENT_CASE(YUV4MPEG)
388 ELEMENT_CASE(video)
389 ELEMENT_CASE(audio)
390 ELEMENT_CASE(text)
391 ELEMENT_CASE(fLaC)
392 ELEMENT_CASE(fishead)
393 ELEMENT_CASE(fisbone)
394 else
395 {
396 Skip_XX(Element_Size, "Unknown");
397 Accept("OggSubElement");
398 Finish("OggSubElement");
399 return;
400 }
401 Open_Buffer_Init(Parser);
402
403 //Parsing
404 Default();
405
406 //Filling
407 StreamKind=StreamKind_Last;
408 if (0) ;
409 ELEMENT_CASE(fishead)
410 ELEMENT_CASE(fisbone)
411 else
412 Identified=true;
413 Accept("OggSubElement");
414 Element_Show();
415}

628//---------------------------------------------------------------------------
629void File_Ogg_SubElement::Identification_video()
630{
631 Element_Info1("Video");
632
633 //Parsing
634 int64u TimeUnit;
635 int32u fccHandler, Width, Height;
636 Skip_B1 ( "Signature");
637 Skip_Local(6, "Signature");
638 Skip_L2( "Reserved");
639 Get_C4 (fccHandler, "fccHandler");
640 Skip_L4( "SizeOfStructure");
641 Get_L8 (TimeUnit, "TimeUnit"); //10000000/TimeUnit is stream tick rate in ticks/sec
642 Skip_L4( "SamplesPerUnit");
643 Skip_L8( "DefaultLengh"); //in media time
644 Skip_L4( "BufferSize");
645 Skip_L2( "BitsPerSample");
646 Skip_L2( "Reserved");
647 Get_L4 (Width, "Width");
648 Get_L4 (Height, "Height");
649 if (Element_Offset<Element_Size)
650 Skip_XX(Element_Size-Element_Offset, "Unknown");
651
652 //Filling
653 Stream_Prepare(Stream_Video);
654 CodecID_Fill(Ztring().From_CC4(fccHandler), Stream_Video, StreamPos_Last, InfoCodecID_Format_Riff);
655 Fill(Stream_Video, StreamPos_Last, Video_Codec, Ztring().From_CC4(fccHandler));
656 Fill(Stream_Video, StreamPos_Last, Video_FrameRate, (float)10000000/(float)TimeUnit, 3);
657 Fill(Stream_Video, StreamPos_Last, Video_Width, Width);
658 Fill(Stream_Video, StreamPos_Last, Video_Height, Height);
659
660 //Creating the parser
661 if (0);
662 #if defined(MEDIAINFO_MPEG4V_YES)
663 else if (MediaInfoLib::Config.CodecID_Get(Stream_Video, InfoCodecID_Format_Riff, Ztring().From_CC4(fccHandler))==__T("MPEG-4 Visual"))
664 {
665 Parser=new File_Mpeg4v;
666 ((File_Mpeg4v*)Parser)->FrameIsAlwaysComplete=true;
667 }
668 #endif
669}

671//---------------------------------------------------------------------------
672void File_Ogg_SubElement::Identification_audio()
673{
674 Element_Info1("Audio");
675
676 //Parsing
677 int64u TimeUnit, SamplesPerUnit;
678 int32u fccHandler, AvgBytesPerSec;
679 int16u Channels;
680 Skip_B1 ( "Signature");
681 Skip_Local(6, "Signature");
682 Skip_L2( "Reserved");
683 Get_C4 (fccHandler, "fccHandler");
684 Skip_L4( "SizeOfStructure");
685 Get_L8 (TimeUnit, "TimeUnit"); //10000000/TimeUnit is stream tick rate in ticks/sec
686 Get_L8 (SamplesPerUnit, "SamplesPerUnit");
687 Skip_L4( "DefaultLengh"); //in media time
688 Skip_L4( "BufferSize");
689 Skip_L2( "BitsPerSample");
690 Skip_L2( "Reserved");
691 Get_L2 (Channels, "Channels");
692 Skip_L2( "BlockAlign");
693 Get_L4 (AvgBytesPerSec, "AvgBytesPerSec");
694 if (Element_Offset<Element_Size)
695 Skip_XX(Element_Size-Element_Offset, "Unknown");
696
697 //Filling
698 Stream_Prepare(Stream_Audio);
699 Ztring Codec; Codec.From_CC4(fccHandler);
700 Codec.TrimLeft(__T('0'));
701 CodecID_Fill(Codec, Stream_Audio, StreamPos_Last, InfoCodecID_Format_Riff);
702 Fill(Stream_Audio, StreamPos_Last, Audio_Codec, Codec);
703 if (AvgBytesPerSec<0x80000000) //This is a signed value, and negative values are not OK
704 Fill(Stream_Audio, StreamPos_Last, Audio_BitRate, AvgBytesPerSec*8);
705 Fill(Stream_Audio, StreamPos_Last, Audio_Channel_s_, Channels==5?6:Channels); //5 channels are 5.1
706 Fill(Stream_Audio, StreamPos_Last, Audio_SamplingRate, SamplesPerUnit);
707 absolute_granule_position_Resolution=SamplesPerUnit;
708
709 //Creating the parser
710 if (0);
711 #if defined(MEDIAINFO_MPEGA_YES)
712 else if (MediaInfoLib::Config.Codec_Get(Codec, InfoCodec_KindofCodec).find(__T("MPEG-"))==0)
713 {
714 Parser=new File_Mpega;
715 }
716 #endif
717 #if defined(MEDIAINFO_AC3_YES)
718 else if (fccHandler==0x32303030)
719 {
720 Parser=new File_Ac3;
721 ((File_Ac3*)Parser)->Frame_Count_Valid=2;
722 }
723 #endif
724}

726//---------------------------------------------------------------------------

728{
729 Element_Info1("Text");
730
731 //Parsing

813//---------------------------------------------------------------------------
814void File_Ogg_SubElement::Comment()
815{
816 //Integrity
817 if (Element_Size<8)
818 return;
819
820 //Parsing
821 int64u ID_Identification;
822 Peek_B8(ID_Identification);
823
824 #undef ELEMENT_CASE
825 #ifdef __BORLANDC__ //Borland converts int64u to int32u
826 #define ELEMENT_CASE(_NAME) \
827 else if (ID_Identification>>(64-8*Elements::Identifier_##_NAME##3)==(((int64u)Elements::Identifier_##_NAME##1)*0x100000000LL|Elements::Identifier_##_NAME##2))
828
829 #else //__BORLANDC__
830 #define ELEMENT_CASE(_NAME) \
831 else if (ID_Identification>>(64-8*Elements::Identifier_##_NAME##3)==Elements::Identifier_##_NAME)
832
833 #endif //__BORLANDC__
834
835 int32u ID_Identification_Size;
836 if (0) ;
837 ELEMENT_CASE(OpusTags) ID_Identification_Size=8;
838 else if (WithType)
839 {
840 if (0) ;
841 ELEMENT_CASE(kateTags) ID_Identification_Size=8;
842 else
843 ID_Identification_Size=6; //Default
844 }
845 else
846 return; //Not a comment
847
848 Element_Name("Comment");
849
850 Skip_Local(ID_Identification_Size, "ID");
851
852 //Preparing
853 File_VorbisCom Vorbis;
854 Vorbis.StreamKind_Specific=StreamKind;
855 Vorbis.StreamKind_Multiple=MultipleStreams?StreamKind:Stream_General;
856 Vorbis.StreamKind_Common=InAnotherContainer?StreamKind:Stream_General;
857 Open_Buffer_Init(&Vorbis);
858
859 //Parsing
860 Open_Buffer_Continue(&Vorbis);
861
862 //Filling
863 Finish(&Vorbis);
864 Merge(Vorbis, Stream_General, 0, 0);
865 Merge(Vorbis, StreamKind, 0, 0);
866 Merge(Vorbis, Stream_Menu, 0, 0);
867
868 //Testing if we must continue
869 if (Identified && (Parser==NULL || Parser->Status[IsFinished]))
870 Finish("OggSubElement");
871}

877
