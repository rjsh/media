58//---------------------------------------------------------------------------
59void File_MpegTs::Option_Manage()
60{
61 if (Complete_Stream && !Complete_Stream->Streams.empty())
62 {
63 #if MEDIAINFO_FILTER
64 //File_Filter configuration
65 if (Config->File_Filter_HasChanged())
66 {
67 bool Searching_Payload_Start=!Config->File_Filter_Get();
68 for (int32u Pos=0x01; Pos<0x10; Pos++)
69 Complete_Stream->Streams[Pos]->Searching_Payload_Start_Set(Searching_Payload_Start); //base PID depends of File_Filter configuration
70 Complete_Stream->Streams[0x0000]->Searching_Payload_Start_Set(true); //program_map
71 }
72 #endif //MEDIAINFO_FILTER
73
74 #if MEDIAINFO_DUPLICATE
75 //File__Duplicate configuration
76 if (File__Duplicate_HasChanged())
77 {
78 for (size_t Pos=0x0000; Pos<0x2000; Pos++)
79 Complete_Stream->Streams[Pos]->ShouldDuplicate=false;
80 Complete_Stream->Streams[0x0000]->ShouldDuplicate=true;
81
82 //For each program
83 for (complete_stream::transport_stream::programs::iterator Program=Complete_Stream->Transport_Streams[Complete_Stream->transport_stream_id].Programs.begin(); Program!=Complete_Stream->Transport_Streams[Complete_Stream->transport_stream_id].Programs.end(); ++Program)
84 {
85 //Do we want this program?
86 bool Wanted=false;
87 for (std::map<const String, File__Duplicate_MpegTs*>::iterator Duplicate=Complete_Stream->Duplicates.begin(); Duplicate!=Complete_Stream->Duplicates.end(); ++Duplicate)
88 {
89 if (Duplicate->second->Wanted_program_numbers.find(Program->first)!=Duplicate->second->Wanted_program_numbers.end())
90 Wanted=true;
91 if (Duplicate->second->Wanted_program_map_PIDs.find(Program->second.pid)!=Duplicate->second->Wanted_program_map_PIDs.end())
92 Wanted=true;
93 }
94
95 //Enabling it if wanted
96 if (Wanted)
97 {
98 Complete_Stream->Streams[Program->second.pid]->ShouldDuplicate=true;
99 for (size_t Pos=0; Pos<Program->second.elementary_PIDs.size(); Pos++)
100 Complete_Stream->Streams[Program->second.elementary_PIDs[Pos]]->ShouldDuplicate=true;
101 }
102 }
103 }
104 #endif //MEDIAINFO_DUPLICATE
105 }
106}

113#if MEDIAINFO_DUPLICATE
114bool File_MpegTs::File__Duplicate_Set (const Ztring &Value)
115{
116 //Form: "Code;Target" <--Generic
117 //Form: "program_number" or <--clear it
118 //Form: "program_number;file" or <--the exported filename is filename.program_number
119 //Form: "program_number;file://filename" or <--the exported filename is specified by user
120 //Form: "program_number;memory" or <--This will be a MediaInfo memory block
121 //Form: "program_number;memory://pointer:size" <--Memory block is specified by user
122 //WARNING: program_number & pointer must be in ***DECIMAL*** format.
123 //Example: "451;memory://123456789:1316"
124 ZtringList List(Value);
125
126 //Backward compatibility
127 bool Orders_ToRemove_Global=false;
128
129 //Searching Target
130 bool IsForUs=true; //True by default for backward compatibility
131 std::vector<ZtringList::iterator> Targets_ToAdd;
132 std::vector<ZtringList::iterator> Targets_ToRemove;
133 std::vector<ZtringList::iterator> Orders_ToAdd;
134 std::vector<ZtringList::iterator> Orders_ToRemove;
135 for (ZtringList::iterator Current=List.begin(); Current<List.end(); ++Current)
136 {
137 //Detecting if we want to remove
138 bool ToRemove=false;
139 if (Current->find(__T('-'))==0)
140 {
141 ToRemove=true;
142 Current->erase(Current->begin());
143 }
144
145 //Managing targets
146 if (Current->find(__T("file:"))==0
147 || Current->find(__T("memory:"))==0)
148 (ToRemove?Targets_ToRemove:Targets_ToAdd).push_back(Current);
149 //Parser name
150 else if (Current->find(__T("parser="))==0)
151 {
152 if (*Current==__T("parser=MpegTs"))
153 IsForUs=true;
154 else
155 IsForUs=false; //Backward compatibility with missing parser name
156 }
157 //Backward compatibility with "0"
158 else if (*Current==__T("0"))
159 Orders_ToRemove_Global=true;
160 //Managing orders
161 else
162 (ToRemove?Orders_ToRemove:Orders_ToAdd).push_back(Current);
163 }
164
165 //For us?
166 if (!IsForUs)
167 return false;
168
169 //Backward compatibility
170 if (Orders_ToRemove_Global) //with "0"
171 {
172 for (std::vector<ZtringList::iterator>::iterator Order=Orders_ToAdd.begin(); Order<Orders_ToAdd.end(); ++Order)
173 Orders_ToRemove.push_back(*Order);
174 Orders_ToAdd.clear();
175 }
176
177 //For each target to add
178 for (std::vector<ZtringList::iterator>::iterator Target=Targets_ToAdd.begin(); Target<Targets_ToAdd.end(); ++Target)
179 {
180 //Adding the target if it does not exist yet
181 if (Complete_Stream->Duplicates.find(**Target)==Complete_Stream->Duplicates.end())
182 {
183 Complete_Stream->Duplicates[**Target]=new File__Duplicate_MpegTs(**Target);
184 size_t Pos=Config->File__Duplicate_Memory_Indexes_Get(**Target);
185 if (Pos!=Error)
186 {
187 if (Pos>=Complete_Stream->Duplicates_Speed.size())
188 Complete_Stream->Duplicates_Speed.resize(Pos+1);
189 Complete_Stream->Duplicates_Speed[Pos]=Complete_Stream->Duplicates[**Target];
190 }
191 }
192
193 //For each order to add
194 for (std::vector<ZtringList::iterator>::iterator Order=Orders_ToAdd.begin(); Order<Orders_ToAdd.end(); ++Order)
195 Complete_Stream->Duplicates[**Target]->Configure(**Order, false);
196
197 //For each order to remove
198 for (std::vector<ZtringList::iterator>::iterator Order=Orders_ToRemove.begin(); Order<Orders_ToRemove.end(); ++Order)
199 Complete_Stream->Duplicates[**Target]->Configure(**Order, true);
200 }
201
202 //For each target to remove
203 for (std::vector<ZtringList::iterator>::iterator Target=Targets_ToRemove.begin(); Target<Targets_ToRemove.end(); ++Target)
204 {
205 std::map<const String, File__Duplicate_MpegTs*>::iterator Pointer=Complete_Stream->Duplicates.find(**Target);
206 if (Pointer!=Complete_Stream->Duplicates.end())
207 {
208 //Duplicates_Speed
209 for (std::vector<File__Duplicate_MpegTs*>::iterator Duplicate=Complete_Stream->Duplicates_Speed.begin(); Duplicate<Complete_Stream->Duplicates_Speed.end(); ++Duplicate)
210 if (*Duplicate==Pointer->second)
211 *Duplicate=NULL;
212
213 //Duplicates_Speed_FromPID
214 for (std::vector<std::vector<File__Duplicate_MpegTs*> >::iterator Duplicate_FromPID=Complete_Stream->Duplicates_Speed_FromPID.begin(); Duplicate_FromPID<Complete_Stream->Duplicates_Speed_FromPID.end(); ++Duplicate_FromPID)
215 for (std::vector<File__Duplicate_MpegTs*>::iterator Duplicate=Duplicate_FromPID->begin(); Duplicate<Duplicate_FromPID->end(); ++Duplicate)
216 if (*Duplicate==Pointer->second)
217 *Duplicate=NULL;
218
219 //Duplicate
220 Complete_Stream->Duplicates.erase(**Target);
221 }
222 }
223
224 //Informing the status has changed
225 Complete_Stream->File__Duplicate_HasChanged_=true;
226 if (Complete_Stream->Duplicates_Speed_FromPID.empty())
227 Complete_Stream->Duplicates_Speed_FromPID.resize(0x2000);
228
229 Complete_Stream->Duplicates_Speed_FromPID[0x00]=Complete_Stream->Duplicates_Speed;
230
231 return true;
232}
233#endif //MEDIAINFO_DUPLICATE

239#if MEDIAINFO_DUPLICATE
240void File_MpegTs::File__Duplicate_Write ()
241{
242 const int8u* ToAdd=Buffer+Buffer_Offset-(size_t)Header_Size;
243 size_t ToAdd_Size=(size_t)(Element_Size+Header_Size);
244
245 std::vector<File__Duplicate_MpegTs*> &Dup_FromPID=Complete_Stream->Duplicates_Speed_FromPID[pid];
246 size_t Duplicates_Speed_FromPID_Size=Complete_Stream->Duplicates_Speed_FromPID[pid].size();
247 bool ToUpdate=false;
248 for (size_t Pos=0; Pos<Duplicates_Speed_FromPID_Size; Pos++)
249 if (Dup_FromPID[Pos] && Dup_FromPID[Pos]->Write(pid, ToAdd, ToAdd_Size))
250 ToUpdate=true;
251 if (ToUpdate)
252 {
253 Complete_Stream->Duplicates_Speed_FromPID.clear();
254 Complete_Stream->Duplicates_Speed_FromPID.resize(0x2000);
255 Complete_Stream->Duplicates_Speed_FromPID[0x0000]=Complete_Stream->Duplicates_Speed;
256 size_t Duplicates_Speed_Size=Complete_Stream->Duplicates_Speed.size();
257 for (size_t Pos=0; Pos<Duplicates_Speed_Size; Pos++)
258 {
259 File__Duplicate_MpegTs* Dup=Complete_Stream->Duplicates_Speed[Pos];
260 size_t program_map_PIDs_Size=Complete_Stream->Duplicates_Speed[Pos]->program_map_PIDs.size();
261 for (size_t program_map_PIDs_Pos=0; program_map_PIDs_Pos<program_map_PIDs_Size; program_map_PIDs_Pos++)
262 if (Dup->program_map_PIDs[program_map_PIDs_Pos])
263 {
264 bool AlreadyPresent=false;
265 for (size_t Duplicates_Speed_FromPID_Pos=0; Duplicates_Speed_FromPID_Pos<Complete_Stream->Duplicates_Speed_FromPID[program_map_PIDs_Pos].size(); Duplicates_Speed_FromPID_Pos++)
266 if (Complete_Stream->Duplicates_Speed_FromPID[program_map_PIDs_Pos][Duplicates_Speed_FromPID_Pos]==Dup)
267 AlreadyPresent=true;
268 if (!AlreadyPresent)
269 Complete_Stream->Duplicates_Speed_FromPID[program_map_PIDs_Pos].push_back(Dup);
270 }
271 size_t elementary_PIDs_Size=Complete_Stream->Duplicates_Speed[Pos]->program_map_PIDs.size();
272 for (size_t elementary_PIDs_Pos=0; elementary_PIDs_Pos<elementary_PIDs_Size; elementary_PIDs_Pos++)
273 if (Dup->elementary_PIDs[elementary_PIDs_Pos])
274 {
275 bool AlreadyPresent=false;
276 for (size_t Duplicates_Speed_FromPID_Pos=0; Duplicates_Speed_FromPID_Pos<Complete_Stream->Duplicates_Speed_FromPID[elementary_PIDs_Pos].size(); Duplicates_Speed_FromPID_Pos++)
277 if (Complete_Stream->Duplicates_Speed_FromPID[elementary_PIDs_Pos][Duplicates_Speed_FromPID_Pos]==Dup)
278 AlreadyPresent=true;
279 if (!AlreadyPresent)
280 Complete_Stream->Duplicates_Speed_FromPID[elementary_PIDs_Pos].push_back(Dup);
281 }
282 }
283 }
284}
