94//---------------------------------------------------------------------------
95void File_Eia608::Streams_Fill()
96{
97 if (Config->File_Eia608_DisplayEmptyStream_Get() && Streams.size()<2)
98 Streams.resize(2);
99
100 if (!HasContent && ServiceDescriptors && ServiceDescriptors->ServiceDescriptors608.find(cc_type)!=ServiceDescriptors->ServiceDescriptors608.end())
101 {
102 TextMode=0;
103 DataChannelMode=0;
104 Special_14(0x20); //CC1/CC3 fake RCL - Resume Caption Loading
105 }
106
107 for (size_t Pos=0; Pos<Streams.size(); Pos++)
108 if (Streams[Pos] || (Pos<2 && Config->File_Eia608_DisplayEmptyStream_Get()))
109 {
110 Stream_Prepare(Stream_Text);
111 Fill(Stream_Text, StreamPos_Last, Text_Format, "EIA-608");
112 Fill(Stream_Text, StreamPos_Last, Text_StreamSize, 0);
113 Fill(Stream_Text, StreamPos_Last, Text_BitRate_Mode, "CBR");
114 if (cc_type!=(int8u)-1)
115 {
116 string ID=Pos<2?"CC":"T";
117 ID+='1'+(cc_type*2)+(Pos%2);
118 Fill(Stream_Text, StreamPos_Last, Text_ID, ID);
119 Fill(Stream_Text, StreamPos_Last, "CaptionServiceName", ID);
120 (*Stream_More)[StreamKind_Last][StreamPos_Last](Ztring().From_Local("CaptionServiceName"), Info_Options)=__T("N NT");
121 }
122 if (Config->ParseSpeed>=1.0)
123 {
124 Fill(Stream_Text, StreamPos_Last, "CaptionServiceContent_IsPresent", DataDetected[Pos+1]?"Yes":"No", Unlimited, true, true); //1 bit per service, starting at 1
125 (*Stream_More)[Stream_Text][StreamPos_Last](Ztring().From_Local("CaptionServiceContent_IsPresent"), Info_Options)=__T("N NT");
126 }
127 if (ServiceDescriptors)
128 {
129 servicedescriptors608::iterator ServiceDescriptor=ServiceDescriptors->ServiceDescriptors608.find(cc_type);
130 if (ServiceDescriptor!=ServiceDescriptors->ServiceDescriptors608.end())
131 {
132 if (Pos==0 && Retrieve(Stream_Text, StreamPos_Last, Text_Language).empty()) //Only CC1/CC3
133 Fill(Stream_Text, StreamPos_Last, Text_Language, ServiceDescriptor->second.language, true);
134 Fill(Stream_Text, StreamPos_Last, "CaptionServiceDescriptor_IsPresent", "Yes", Unlimited, true, true);
135 (*Stream_More)[Stream_Text][StreamPos_Last](Ztring().From_Local("CaptionServiceDescriptor_IsPresent"), Info_Options)=__T("N NT");
136 }
137 else //ServiceDescriptors pointer is for the support by the transport layer of the info
138 {
139 Fill(Stream_Text, StreamPos_Last, "CaptionServiceDescriptor_IsPresent", "No", Unlimited, true, true);
140 (*Stream_More)[Stream_Text][StreamPos_Last](Ztring().From_Local("CaptionServiceDescriptor_IsPresent"), Info_Options)=__T("N NT");
141 }
142 }
143 }

156void File_Eia608::Read_Buffer_Unsynched()
157{
158 for (size_t StreamPos=0; StreamPos<Streams.size(); StreamPos++)
159 if (Streams[StreamPos])
160 {
161 for (size_t Pos_Y=0; Pos_Y<Streams[StreamPos]->CC_Displayed.size(); Pos_Y++)
162 {
163 for (size_t Pos_X=0; Pos_X<Streams[StreamPos]->CC_Displayed[Pos_Y].size(); Pos_X++)
164 if (Streams[StreamPos])
165 {
166 Streams[StreamPos]->CC_Displayed[Pos_Y][Pos_X].Value=L' ';
167 Streams[StreamPos]->CC_Displayed[Pos_Y][Pos_X].Attribute=0;
168 if (StreamPos<2)
169 {
170 Streams[StreamPos]->CC_NonDisplayed[Pos_Y][Pos_X].Value=L' ';
171 Streams[StreamPos]->CC_NonDisplayed[Pos_Y][Pos_X].Attribute=0;
172 }
173 }
174 }
175 Streams[StreamPos]->Synched=false;
176 }
177
178 XDS_Data.clear();
179 XDS_Level=(size_t)-1;
180
181 #if MEDIAINFO_EVENTS
182 TextMode=true; DataChannelMode=true;
183 HasChanged();
184 TextMode=true; DataChannelMode=false;
185 HasChanged();
186 TextMode=false; DataChannelMode=true;
187 HasChanged();
188 TextMode=false; DataChannelMode=false;
189 HasChanged();
190 #endif //MEDIAINFO_EVENTS
191}

232//---------------------------------------------------------------------------
233void File_Eia608::Read_Buffer_Continue()
234{
235 FrameInfo.PTS=FrameInfo.DTS;
236
237 if (!Status[IsAccepted])
238 Accept("EIA-608");
239
240 while (Element_Offset+1<Element_Size)
241 {
242 int8u cc_data_1, cc_data_2;
243 Get_B1 (cc_data_1, "cc_data");
244 Get_B1 (cc_data_2, "cc_data");
245
246 //Removing checksume
247 cc_data_1&=0x7F;
248 cc_data_2&=0x7F;
249
250 //Test if non-printing chars (0x10-0x1F) are repeated (CEA-608-E section D.2)
251 if (cc_data_1_Old)
252 {
253 if (cc_data_1_Old==cc_data_1 && cc_data_2_Old==cc_data_2)
254 {
255 //This is duplicate
256 cc_data_1_Old=0x00;
257 cc_data_2_Old=0x00;
258 return; //Nothing to do
259 }
260 else if (cc_type==0) // Field 1 only
261 {
262 //They should be duplicated, there is a problem
263 }
264 cc_data_1_Old=0x00;
265 cc_data_2_Old=0x00;
266 }
267
268 if ((cc_data_1 && cc_data_1<0x10) || (XDS_Level!=(size_t)-1 && cc_data_1>=0x20)) //XDS
269 {
270 XDS(cc_data_1, cc_data_2);
271 }
272 else if (cc_data_1>=0x20) //Basic characters
273 {
274 size_t StreamPos=TextMode*2+DataChannelMode;
275 if (StreamPos>=Streams.size() || Streams[StreamPos]==NULL || !Streams[StreamPos]->Synched)
276 return; //Not synched
277
278 Standard(cc_data_1);
279 if ((cc_data_2&0x7F)>=0x20)
280 Standard(cc_data_2);
281 }
282 else if (cc_data_1) //Special
283 Special(cc_data_1, cc_data_2);
284 }
285}

291//---------------------------------------------------------------------------
292void File_Eia608::XDS(int8u cc_data_1, int8u cc_data_2)
293{
294 if (cc_data_1 && cc_data_1<0x10 && cc_data_1%2==0)
295 {
296 // Continue
297 cc_data_1--;
298 for (XDS_Level=0; XDS_Level<XDS_Data.size(); XDS_Level++)
299 if (XDS_Data[XDS_Level].size()>=2 && XDS_Data[XDS_Level][0]==cc_data_1 && XDS_Data[XDS_Level][1]==cc_data_2)
300 break;
301 if (XDS_Level>=XDS_Data.size())
302 XDS_Level=(size_t)-1; // There is a problem
303
304 return;
305 }
306 else if (cc_data_1 && cc_data_1<0x0F)
307 {
308 // Start
309 for (XDS_Level=0; XDS_Level<XDS_Data.size(); XDS_Level++)
310 if (XDS_Data[XDS_Level].size()>=2 && XDS_Data[XDS_Level][0]==cc_data_1 && XDS_Data[XDS_Level][1]==cc_data_2)
311 break;
312 if (XDS_Level>=XDS_Data.size())
313 {
314 XDS_Level=XDS_Data.size();
315 XDS_Data.resize(XDS_Level+1);
316 }
317 else
318 XDS_Data[XDS_Level].clear(); // There is a problem, erasing the previous item
319 }
320
321 if (XDS_Level==(size_t)-1)
322 return; //There is a problem
323
324 XDS_Data[XDS_Level].push_back(cc_data_1);
325 XDS_Data[XDS_Level].push_back(cc_data_2);
326 if (cc_data_1==0x0F)
327 XDS();
328 if (XDS_Level!=(size_t)-1 && XDS_Data[XDS_Level].size()>=36)
329 XDS_Data[XDS_Level].clear(); // Clear, this is a security
330 TextMode=0; // This is CC
331}

333//---------------------------------------------------------------------------
334void File_Eia608::XDS()
335{
336 if (XDS_Data[XDS_Level].size()<4)
337 {
338 XDS_Data.erase(XDS_Data.begin()+XDS_Level);
339 XDS_Level=(size_t)-1;
340 return; //There is a problem
341 }
342
343 switch (XDS_Data[XDS_Level][0])
344 {
345 case 0x01 : XDS_Current(); break;
346 case 0x05 : XDS_Channel(); break;
347 case 0x09 : XDS_PublicService(); break;
348 default : ;
349 }
350
351 XDS_Data.erase(XDS_Data.begin()+XDS_Level);
352 XDS_Level=(size_t)-1;
353
354 DataDetected[5]=true; //bit 5=XDS
355}

369//---------------------------------------------------------------------------
370void File_Eia608::XDS_Current_ContentAdvisory()
371{
372 if (XDS_Data[XDS_Level].size()!=6)
373 {
374 return; //There is a problem
375 }
376
377 Clear(Stream_General, 0, General_LawRating);
378
379 int8u a1a0=(XDS_Data[XDS_Level][2]>>3)&0x3;
380 const char* ContentAdvisory=NULL;
381 string ContentDescriptors;
382 switch (a1a0)
383 {
384 case 0:
385 case 2:
386 switch (XDS_Data[XDS_Level][2]&0x7) //r2r1r0
387 {
388 case 0 : ContentAdvisory="N/A"; break;
389 case 1 : ContentAdvisory="G"; break;
390 case 2 : ContentAdvisory="PG"; break;
391 case 3 : ContentAdvisory="PG-13"; break;
392 case 4 : ContentAdvisory="R"; break;
393 case 5 : ContentAdvisory="NC-17"; break;
394 case 6 : ContentAdvisory="C"; break;
395 default: ;
396 }
397 break;
398 case 1:
399 switch (XDS_Data[XDS_Level][3]&0x7) //g2g1g0
400 {
401 case 0 : ContentAdvisory="None"; break;
402 case 1 : ContentAdvisory="TV-Y"; break;
403 case 2 : ContentAdvisory="TV-Y7"; break;
404 case 3 : ContentAdvisory="TV-G"; break;
405 case 4 : ContentAdvisory="TV-PG"; break;
406 case 5 : ContentAdvisory="TV-14"; break;
407 case 6 : ContentAdvisory="TV-MA"; break;
408 case 7 : ContentAdvisory="None"; break;
409 default: ;
410 }
411 if (XDS_Data[XDS_Level][2]&0x20) //Suggestive dialogue
412 ContentDescriptors+='D';
413 if (XDS_Data[XDS_Level][3]&0x8) //Coarse language
414 ContentDescriptors+='L';
415 if (XDS_Data[XDS_Level][3]&0x10) //Sexual content
416 ContentDescriptors+='S';
417 if (XDS_Data[XDS_Level][3]&0x20) //Violence
418 {
419 if ((XDS_Data[XDS_Level][3]&0x7)==2) //"TV-Y7" --> Fantasy Violence
420 ContentDescriptors+="FV";
421 else
422 ContentDescriptors+='V';
423 }
424 break;
425 case 3:
426 if (XDS_Data[XDS_Level][3]&0x8) //a3
427 {
428 ContentAdvisory="(Reserved)";
429 }
430 else
431 {
432 if (XDS_Data[XDS_Level][2]&0x20) //a2
433 switch (XDS_Data[XDS_Level][3]&0x7) //g2g1g0
434 {
435 case 0 : ContentAdvisory="E"; break;
436 case 1 : ContentAdvisory="G"; break;
437 case 2 : ContentAdvisory="8+"; break;
438 case 3 : ContentAdvisory="13+"; break;
439 case 4 : ContentAdvisory="16+"; break;
440 case 5 : ContentAdvisory="18+"; break;
441 default: ;
442 }
443 else
444 switch (XDS_Data[XDS_Level][3]&0x7) //g2g1g0
445 {
446 case 0 : ContentAdvisory="E"; break;
447 case 1 : ContentAdvisory="C"; break;
448 case 2 : ContentAdvisory="C8+"; break;
449 case 3 : ContentAdvisory="G"; break;
450 case 4 : ContentAdvisory="PG"; break;
451 case 5 : ContentAdvisory="14+"; break;
452 case 6 : ContentAdvisory="18+"; break;
453 default: ;
454 }
455 }
456 break;
457 default: ;
458 }
459
460 if (ContentAdvisory)
461 {
462 string ContentAdvisory_String=ContentAdvisory;
463 if (!ContentDescriptors.empty())
464 ContentAdvisory_String+=" ("+ContentDescriptors+')';
465 Fill(Stream_General, 0, General_LawRating, ContentAdvisory_String.c_str());
466 }
467}

531//---------------------------------------------------------------------------
532void File_Eia608::Special(int8u cc_data_1, int8u cc_data_2)
533{
534 //Data channel check
535 DataChannelMode=(cc_data_1&0x08)!=0; //bit3 is the Data Channel number
536
537 //Field check
538 if (cc_type==(int8u)-1)
539 {
540 if ((cc_data_1==0x14 || cc_data_1==0x1C) && (cc_data_2&0xF0)==0x20)
541 cc_type=0;
542 if ((cc_data_1==0x15 || cc_data_1==0x1D) && (cc_data_2&0xF0)==0x20)
543 cc_type=1;
544 }
545
546 cc_data_1&=0xF7;
547 if (cc_data_1==0x15 && (cc_data_2&0xF0)==0x20)
548 cc_data_1=0x14;
549
550 if (cc_data_1>=0x10 && cc_data_1<=0x17 && cc_data_2>=0x40)
551 {
552 PreambleAddressCode(cc_data_1, cc_data_2);
553 }
554 else
555 {
556 switch (cc_data_1)
557 {
558 case 0x10 : Special_10(cc_data_2); break;
559 case 0x11 : Special_11(cc_data_2); break;
560 case 0x12 : Special_12(cc_data_2); break;
561 case 0x13 : Special_13(cc_data_2); break;
562 case 0x14 : Special_14(cc_data_2); break;
563 case 0x17 : Special_17(cc_data_2); break;
564 default : Illegal(cc_data_1, cc_data_2);
565 }
566 }
567
568 //Saving data, for repetition of the code
569 cc_data_1_Old=cc_data_1;
570 cc_data_2_Old=cc_data_2;
571}

573//---------------------------------------------------------------------------
574void File_Eia608::PreambleAddressCode(int8u cc_data_1, int8u cc_data_2)
575{
576 //CEA-608-E, Section F.1.1.5
577
578 size_t StreamPos=TextMode*2+DataChannelMode;
579 if (StreamPos>=Streams.size() || Streams[StreamPos]==NULL || !Streams[StreamPos]->Synched)
580 return; //Not synched
581 Streams[StreamPos]->x=0; //I am not sure of this, specifications are not precise
582
583 //Horizontal position
584 if (!TextMode)
585 {
586 Streams[StreamPos]->y=Eia608_PAC_Row[cc_data_1&0x07]+((cc_data_2&0x20)?1:0);
587 if (Streams[StreamPos]->y>=Eia608_Rows)
588 {
589 Streams[StreamPos]->y=Eia608_Rows-1;
590 }
591 }
592
593 //Attributes (except Underline)
594 if (cc_data_2&0x10) //0x5x and 0x7x
595 {
596 Streams[StreamPos]->x=(((size_t)cc_data_2)&0x0E)<<1;
597 Streams[StreamPos]->Attribute_Current=Attribute_Color_White;
598 }
599 else if ((cc_data_2&0x0E)==0x0E) //0x4E, 0x4F, 0x6E, 0x6F
600 {
601 Streams[StreamPos]->Attribute_Current=Attribute_Color_White|Attribute_Italic;
602 }
603 else //0x40-0x4D, 0x60-0x6D
604 Streams[StreamPos]->Attribute_Current=(cc_data_2&0x0E)>>1;
605
606 //Underline
607 if (cc_data_2&0x01)
608 Streams[StreamPos]->Attribute_Current|=Attribute_Underline;
609}

637//---------------------------------------------------------------------------
638void File_Eia608::Special_11(int8u cc_data_2)
639{
640 size_t StreamPos=TextMode*2+DataChannelMode;
641 if (StreamPos>=Streams.size() || Streams[StreamPos]==NULL || !Streams[StreamPos]->Synched)
642 return; //Not synched
643
644 switch (cc_data_2)
645 {
646 //CEA-608-E, Section F.1.1.3
647 case 0x20 : //White
648 case 0x21 : //White Underline
649 case 0x22 : //
650 case 0x23 : //
651 case 0x24 : //
652 case 0x25 : //
653 case 0x26 : //
654 case 0x27 : //
655 case 0x28 : //
656 case 0x29 : //
657 case 0x2A : //
658 case 0x2B : //
659 case 0x2C : //
660 case 0x2D : //
661 case 0x2E : //
662 case 0x2F : //
663 //Color or Italic
664 if ((cc_data_2&0xFE)==0x2E) //Italic
665 Streams[StreamPos]->Attribute_Current|=Attribute_Italic;
666 else //Other attributes
667 Streams[StreamPos]->Attribute_Current=(cc_data_2&0x0F)>>1;
668
669 //Underline
670 if (cc_data_2&0x01)
671 Streams[StreamPos]->Attribute_Current|=Attribute_Underline;
672
673 break;
674 //CEA-608-E, Section F.1.1.1
675 case 0x30 : Character_Fill(L'\u2122'); break; //Registered mark symbol
676 case 0x31 : Character_Fill(L'\u00B0'); break; //Degree sign
677 case 0x32 : Character_Fill(L'\u00BD'); break; //1/2
678 case 0x33 : Character_Fill(L'\u00BF'); break; //interogation mark inverted
679 case 0x34 : Character_Fill(L'\u00A9'); break; //Trademark symbol
680 case 0x35 : Character_Fill(L'\u00A2'); break; //Cents sign
681 case 0x36 : Character_Fill(L'\u00A3'); break; //Pounds Sterling sign
682 case 0x37 : Character_Fill(L'\u266A'); break; //Music note
683 case 0x38 : Character_Fill(L'\u00E0'); break; //a grave

793//---------------------------------------------------------------------------
794void File_Eia608::Special_14(int8u cc_data_2)
795{
796 size_t StreamPos=TextMode*2+DataChannelMode;
797
798 switch (cc_data_2)
799 {
800 case 0x20 : //RCL - Resume Caption Loading
801 case 0x25 : //RU2 - Roll-Up Captions2 Rows
802 case 0x26 : //RU3 - Roll-Up Captions3 Rows
803 case 0x27 : //RU4 - Roll-Up Captions4 Rows
804 case 0x29 : //RDC - Resume Direct Captioning
805 case 0x2A : //TR - Text Restart
806 case 0x2B : //RTD - Resume Text Display
807 case 0x2C : //EDM - Erase Displayed Memory
808 TextMode=(cc_data_2&0xFE)==0x2A;
809 XDS_Level=(size_t)-1; // No more XDS
810 StreamPos=TextMode*2+DataChannelMode;
811
812 //Alloc
813 if (StreamPos>=Streams.size())
814 Streams.resize(StreamPos+1);
815 if (Streams[StreamPos]==NULL)
816 {
817 Streams[StreamPos]=new stream();
818 Streams[StreamPos]->CC_Displayed.resize(Eia608_Rows);
819 for (size_t Pos=0; Pos<Streams[StreamPos]->CC_Displayed.size(); Pos++)
820 Streams[StreamPos]->CC_Displayed[Pos].resize(Eia608_Columns);
821 if (StreamPos<2) //CC only, not in Text
822 {
823 Streams[StreamPos]->CC_NonDisplayed.resize(Eia608_Rows);
824 for (size_t Pos=0; Pos<Streams[StreamPos]->CC_NonDisplayed.size(); Pos++)
825 Streams[StreamPos]->CC_NonDisplayed[Pos].resize(Eia608_Columns);
826 }
827 }
828 Streams[StreamPos]->Synched=true;
829
830 break;
831 case 0x2F : //EOC - end of Caption
832 TextMode=false;
833 StreamPos=TextMode*2+DataChannelMode;
834 default: ;
835 }
836
837 if (StreamPos>=Streams.size() || Streams[StreamPos]==NULL || !Streams[StreamPos]->Synched)
838 return; //Not synched
839
840 switch (cc_data_2)
841 {
842 case 0x20 : TextMode=false;
843 Streams[StreamPos]->InBack=true;
844 break; //RCL - Resume Caption Loading (Select pop-on style)
845 case 0x21 : if (Streams[StreamPos]->x)
846 Streams[StreamPos]->x--;
847 (Streams[StreamPos]->InBack?Streams[StreamPos]->CC_NonDisplayed:Streams[StreamPos]->CC_Displayed)[Streams[StreamPos]->y][Streams[StreamPos]->x].Value=L' '; //Clear the character
848 if (!Streams[StreamPos]->InBack)
849 HasChanged();
850 break; //BS - Backspace
851 case 0x22 : Special_14(0x2D); //Found 1 file with AOF and non CR
852 break; //AOF - Alarm Off
853 case 0x23 : break; //AON - Alarm On
854 case 0x24 : for (size_t Pos=Streams[StreamPos]->x; Pos<Eia608_Columns; Pos++)
855 (Streams[StreamPos]->InBack?Streams[StreamPos]->CC_NonDisplayed:Streams[StreamPos]->CC_Displayed)[Streams[StreamPos]->y][Pos].Value=L' '; //Clear up to the end of line
856 if (!Streams[StreamPos]->InBack)
857 HasChanged();
858 break; //DER - Delete to End of Row
859 case 0x25 : //RU2 - Roll-Up Captions2 Rows
860 case 0x26 : //RU3 - Roll-Up Captions3 Rows
861 case 0x27 : //RU4 - Roll-Up Captions4 Rows
862 Streams[StreamPos]->RollUpLines=cc_data_2-0x25+2;
863 Streams[StreamPos]->InBack=false;
864 break; //RUx - Roll-Up Captionsx Rows
865 case 0x28 : break; //FON - Flash On
866 case 0x29 : Streams[StreamPos]->InBack=false;
867 break; //RDC - Resume Direct Captioning (paint-on style)
868 case 0x2A : TextMode=true;
869 Streams[StreamPos]->RollUpLines=Eia608_Rows; //Roll up all the lines
870 Streams[StreamPos]->y=Eia608_Rows-1; //Base is the bottom line
871 Streams[StreamPos]->Attribute_Current=0; //Reset all attributes
872 Special_14(0x2D); //Next line
873 break; //TR - Text Restart (clear Text, but not boxes)
874 case 0x2B : TextMode=true;
875 break; //RTD - Resume Text Display
876 case 0x2C :
877 if (StreamPos<Streams.size() && Streams[StreamPos])
878 {
879 bool HasChanged_=false;
880 for (size_t Pos_Y=0; Pos_Y<Eia608_Rows; Pos_Y++)
881 for (size_t Pos_X=0; Pos_X<Eia608_Columns; Pos_X++)
882 if (Streams[StreamPos]->CC_Displayed[Pos_Y][Pos_X].Value!=L' ')
883 {
884 Streams[StreamPos]->CC_Displayed[Pos_Y][Pos_X].Value=L' ';
885 Streams[StreamPos]->CC_Displayed[Pos_Y][Pos_X].Attribute=0;
886 HasChanged_=true;
887 }
888 if (HasChanged_)
889 HasChanged();
890 }
891 break; //EDM - Erase Displayed Memory
892 case 0x2D : for (size_t Pos=1; Pos<Streams[StreamPos]->RollUpLines; Pos++)
893 {
894 if (Streams[StreamPos]->y>=Streams[StreamPos]->RollUpLines-Pos && Streams[StreamPos]->y-Streams[StreamPos]->RollUpLines+Pos+1<Eia608_Rows)
895 Streams[StreamPos]->CC_Displayed[Streams[StreamPos]->y-Streams[StreamPos]->RollUpLines+Pos]=Streams[StreamPos]->CC_Displayed[Streams[StreamPos]->y-Streams[StreamPos]->RollUpLines+Pos+1];
896 }
897 for (size_t Pos_X=0; Pos_X<Eia608_Columns; Pos_X++)
898 {
899 Streams[StreamPos]->CC_Displayed[Streams[StreamPos]->y][Pos_X].Value=L' ';
900 Streams[StreamPos]->CC_Displayed[Streams[StreamPos]->y][Pos_X].Attribute=0;
901 }
902 if (!Streams[StreamPos]->InBack)
903 HasChanged();
904 Streams[StreamPos]->x=0;
905 break; //CR - Carriage Return
906 case 0x2E : for (size_t Pos_Y=0; Pos_Y<Streams[StreamPos]->CC_NonDisplayed.size(); Pos_Y++)
907 for (size_t Pos_X=0; Pos_X<Streams[StreamPos]->CC_NonDisplayed[Pos_Y].size(); Pos_X++)
908 {
909 Streams[StreamPos]->CC_NonDisplayed[Pos_Y][Pos_X].Value=L' ';
910 Streams[StreamPos]->CC_NonDisplayed[Pos_Y][Pos_X].Attribute=0;
911 }
912 break; //ENM - Erase Non-Displayed Memory
913 case 0x2F : Streams[StreamPos]->CC_Displayed.swap(Streams[StreamPos]->CC_NonDisplayed);
914 HasChanged();
915 Streams[StreamPos]->Synched=false;
916 break; //EOC - End of Caption
917 default : Illegal(0x14, cc_data_2);
918 }
