36int64u Scc_str2timecode(const char* Value)
37{
38 size_t Length=strlen(Value);
39 if (Length==11
40 && Value[0]>='0' && Value[0]<='9'
41 && Value[1]>='0' && Value[1]<='9'
42 && Value[2]==':'
43 && Value[3]>='0' && Value[3]<='9'
44 && Value[4]>='0' && Value[4]<='9'
45 && Value[5]==':'
46 && Value[6]>='0' && Value[6]<='9'
47 && Value[7]>='0' && Value[7]<='9'
48 && (Value[8]==':' || Value[8]==';')
49 && Value[9]>='0' && Value[9]<='9'
50 && Value[10]>='0' && Value[10]<='9')
51 {
52 int64u ToReturn=(int64u)(Value[0]-'0')*10*60*60*1000000000
53 +(int64u)(Value[1]-'0') *60*60*1000000000
54 +(int64u)(Value[3]-'0') *10*60*1000000000
55 +(int64u)(Value[4]-'0') *60*1000000000
56 +(int64u)(Value[6]-'0') *10*1000000000
57 +(int64u)(Value[7]-'0') *1000000000
58 +(int64u)(Value[9]-'0') * 333333333
59 +(int64u)(Value[10]-'0') * 33333333;
60
61 return ToReturn;
62 }
63 else
64 return (int64u)-1;
65}

132void File_Scc::Streams_Finish()
133{
134 if (Parser && Parser->Status[IsAccepted])
135 {
136 Finish(Parser);
137 for (size_t Pos2=0; Pos2<Parser->Count_Get(Stream_Text); Pos2++)
138 {
139 Stream_Prepare(Stream_Text);
140 Merge(*Parser, Stream_Text, StreamPos_Last, Pos2);
141 Fill(Stream_Text, StreamPos_Last, Text_ID, Parser->Retrieve(Stream_Text, Pos2, Text_ID), true);
142 }
143 }

147// Buffer - File header

150//---------------------------------------------------------------------------
151bool File_Scc::FileHeader_Begin()
152{
153 //Element_Size
154 if (File_Size<22)
155 {
156 Reject("N19");
157 return false;
158 }
159 if (Buffer_Size<22)
160 return false; //Must wait for more data
161
162 if (Buffer[ 0]!=0x53
163 || Buffer[ 1]!=0x63
164 || Buffer[ 2]!=0x65
165 || Buffer[ 3]!=0x6E
166 || Buffer[ 4]!=0x61
167 || Buffer[ 5]!=0x72
168 || Buffer[ 6]!=0x69
169 || Buffer[ 7]!=0x73
170 || Buffer[ 8]!=0x74
171 || Buffer[ 9]!=0x5F
172 || Buffer[10]!=0x53
173 || Buffer[11]!=0x43
174 || Buffer[12]!=0x43
175 || Buffer[13]!=0x20
176 || Buffer[14]!=0x56
177 || Buffer[15]!=0x31
178 || Buffer[16]!=0x2E
179 || Buffer[17]!=0x30
180 )
181 {
182 Reject("SCC");
183 return false;
184 }
185
186 //Element_Size
187 if (Buffer_Size<File_Size)
188 return false; //Must wait for more data
189
190 //All should be OK...
191 return true;
192}

194//---------------------------------------------------------------------------
195void File_Scc::FileHeader_Parse()
196{
197 Skip_String(18, "Magic");
198 while (Element_Offset<Buffer_Size)
199 {
200 if (Buffer[(size_t)Element_Offset]!=0x0D && Buffer[(size_t)Element_Offset]!=0x0A)
201 break;
202 Element_Offset++;
203 }
204
205 Accept();
206 Fill(Stream_General, 0, General_Format, "SCC");
207
208 //Init
209 Parser=new File_Eia608();
210 Open_Buffer_Init(Parser);
211}

213//***************************************************************************
214// Buffer - Per element
215//***************************************************************************
216
217//---------------------------------------------------------------------------
218void File_Scc::Header_Parse()
219{
220 size_t End=Buffer_Offset;
221
222 //Content
223 while (End<Buffer_Size)
224 {
225 if (Buffer[End]==0x0D || Buffer[End]==0x0A)
226 break;
227 End++;
228 }
229
230 //EOL
231 while (End<Buffer_Size)
232 {
233 if (Buffer[End]!=0x0D && Buffer[End]!=0x0A)
234 break;
235 End++;
236 }
237
238 //Filling
239 Header_Fill_Size(End-Buffer_Offset);
240 Header_Fill_Code(0, __T("Block"));
241}
242
243//---------------------------------------------------------------------------
244void File_Scc::Data_Parse()
245{
246 while (Element_Offset<Element_Size && (Buffer[Buffer_Offset+(size_t)Element_Offset]==0x0D || Buffer[Buffer_Offset+(size_t)Element_Offset]==0x0A))
247 Element_Offset++;
248 if (Element_Offset==Element_Size)
249 return;
250
251 //Parsing
252 string TimeStamp;
253 Get_String(11, TimeStamp, "TimeStamp");
254 Parser->FrameInfo.DTS=Scc_str2timecode(TimeStamp.c_str());
255 while (Element_Offset+5<=Element_Size)
256 {
257 int8u Buffer_Temp[2];
258 Buffer_Temp[0]=(Buffer[Buffer_Offset+(size_t)Element_Offset+1]-(Buffer[Buffer_Offset+(size_t)Element_Offset+1]>='a'?('a'-10):'0'))<<4
259 | (Buffer[Buffer_Offset+(size_t)Element_Offset+2]-(Buffer[Buffer_Offset+(size_t)Element_Offset+2]>='a'?('a'-10):'0'));
260 Buffer_Temp[1]=(Buffer[Buffer_Offset+(size_t)Element_Offset+3]-(Buffer[Buffer_Offset+(size_t)Element_Offset+3]>='a'?('a'-10):'0'))<<4
261 | (Buffer[Buffer_Offset+(size_t)Element_Offset+4]-(Buffer[Buffer_Offset+(size_t)Element_Offset+4]>='a'?('a'-10):'0'));
262 Open_Buffer_Continue(Parser, Buffer_Temp, 2);
263 Element_Offset+=5;
264 if (Parser->FrameInfo.DTS!=(int64u)-1)
265 Parser->FrameInfo.DTS+=33333333;
266 }
267}
