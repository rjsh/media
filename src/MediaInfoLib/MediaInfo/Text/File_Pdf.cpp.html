62bool File_Pdf::FileHeader_Begin()
63{
64 //Synchro
65 if (5>Buffer_Size)
66 return false;
67 if (Buffer[0]!=0x25 //"%PDF-"
68 || Buffer[1]!=0x50
69 || Buffer[2]!=0x44
70 || Buffer[3]!=0x46
71 || Buffer[4]!=0x2D)
72 {
73 Reject();
74 return false;
75 }
76
77 Accept();
78
79 //Temp
80 Catalog_Level=0;
81 Offsets_Max=0;
82
83 //All should be OK...
84 return true;
85}

87//---------------------------------------------------------------------------
88void File_Pdf::FileHeader_Parse()
89{
90 string PdfHeader;
91 Get_String(SizeOfLine(), PdfHeader, "Header");
92 for (;;)
93 {
94 int64u CommentSize=SizeOfLine();
95 if (Buffer_Offset+Element_Offset>=Buffer_Size)
96 {
97 Element_WaitForMoreData();
98 return;
99 }
100 if (Buffer[Buffer_Offset+Element_Offset]!='%')
101 break;
102 Skip_String(CommentSize, "Comment");
103 }
104
105 //Filling
106 Fill(Stream_General, 0, General_Format_Version, PdfHeader.substr(5));
107
108 GoToFromEnd(9+2+10+2+5+2); // "startxref" + EOL + 10max digits + EOL + "%%EOF" + EOL
109 State=State_Parsing_startxref;
110}

122 case State_Parsing_startxref : eof(); startxref(); break;
123 case State_Parsing_object : break; //Using elements
124 default : Finish();
125 }

147//---------------------------------------------------------------------------
148void File_Pdf::Header_Parse()
149{
150 offsets::iterator Offsets_Next=upper_bound(Offsets.begin(), Offsets.end(), (int32u)(File_Offset+Buffer_Offset));
151 if (Offsets_Next!=Offsets.end() && *Offsets_Next>File_Offset+Buffer_Size)
152 {
153 Element_WaitForMoreData();
154 return;
155 }
156
157 int64u Size;
158 //if (Offsets_Current==Offsets.end())
159 // Size=Offsets_Max-(File_Offset+Buffer_Offset);
160 //else
161 // Size=Offsets_Current->first-(File_Offset+Buffer_Offset);
162 if (Offsets_Next==Offsets.end())
163 Size=Offsets_Max-(File_Offset+Buffer_Offset);
164 else
165 Size=*Offsets_Next-(File_Offset+Buffer_Offset);
166
167 Header_Fill_Size(Size);

170//---------------------------------------------------------------------------
171void File_Pdf::Data_Parse()
172{
173 Element_Name("Object");
174
175 string Line;
176 Get_String(SizeOfLine(), Line, "Header");
177
178 size_t Space_Pos=Line.find(' ');
179 int32u ObjectNumber=Ztring().From_UTF8(Line.substr(0, Space_Pos)).To_int32u();
180 Element_Info1(ObjectNumber);
181 objects::iterator Object=Objects.find(ObjectNumber);
182 if (Object==Objects.end())
183 Skip_XX(Element_Size-Element_Offset, "Data");
184 else
185 switch(Object->second.Type)
186 {
187 case Type_Root : Object_Root(); break;
188 case Type_Info : Object_Info(); break;
189 case Type_Metadata : Object_Metadata(); break;
190 default : Skip_XX(Element_Size-Element_Offset, "Data");
191 }
192
193 for (;;)
194 {
195 Objects_Current->second.BottomPos++;
196 if (Objects_Current->second.BottomPos>=Objects_Current->second.Bottoms.size())
197 {
198 if (Objects_Current->first==(int32u)-1)
199 {
200 //No more to parse
201 Objects_Current=Objects.end();
202 Objects.clear();
203 Finish();
204 break;
205 }
206
207 Objects_Current=Objects.find(Objects_Current->second.TopObject);
208 continue;
209 }
210
211 Objects_Current=Objects.find(Objects_Current->second.Bottoms[Objects_Current->second.BottomPos]);
212 GoTo(Objects_Current->second.Offset);
213 break;
214 }
215}

221//---------------------------------------------------------------------------
222void File_Pdf::xref()
223{
224 //Parsing
225 Element_Begin1("Cross-Reference Table");
226
227 Element_Begin1("Cross-Reference Section");
228
229 string FirstLine;
230 Skip_String(SizeOfLine(), "Object name");
231 Element_Begin1("Cross-Reference SubSection");
232 Get_String(SizeOfLine(), FirstLine, "Header");
233 size_t FirstLine_Space=FirstLine.find(' ');
234 int32u Base=atoi((const char*)FirstLine.c_str());
235 int32u Count=0;
236 if (FirstLine_Space!=string::npos)
237 Count=atoi((const char*)FirstLine.c_str()+FirstLine_Space+1);
238
239 if (0x10000+20*Count>Buffer_Size && File_Offset+Buffer_Size<File_Size)
240 {
241 // We wait for more data
242 Buffer_Offset=0;
243 Element_Offset=0;
244 Element_DoNotShow();
245 Element_End0();
246 Element_End0();
247 Element_End0();
248 Element_WaitForMoreData();
249 return;
250 }
251
252 while (Element_Offset<Element_Size && (Buffer[Buffer_Offset+(size_t)Element_Offset]=='\r' || Buffer[Buffer_Offset+(size_t)Element_Offset]=='\n'))
253 Element_Offset++;
254 const int8u* Buffer_Temp=Buffer+Buffer_Offset+(size_t)Element_Offset+17;
255 for (int32u Pos=0; Pos<Count; ++Pos)
256 {
257 if (*Buffer_Temp=='n')
258 {
259 int32u Offset=(int32u)atoi((const char*)(Buffer_Temp-17));
260 Objects[Base+Pos].Offset=Offset;
261 //Offsets[Offset]=Base+Pos;
262 Offsets.push_back(Offset);
263 }
264 Buffer_Temp+=20;
265 if (Pos>100)
266 Element_Offset+=20;
267 else
268 {
269 Skip_String(18, "Entry"); Param_Info1(Base+Pos);
270 Element_Offset+=2; //Skipping spaces at end and line return
271 }
272 }
273 Element_End0();
274
275 Element_End0();
276
277 Element_End0();
278
279 if (File_Offset+Buffer_Offset>Offsets_Max)
280 Offsets_Max=(int32u)(File_Offset+Buffer_Offset);
281}

283//---------------------------------------------------------------------------
284void File_Pdf::trailer()
285{
286 Element_Begin1("Trailer");
287
288 //Parsing
289 int32u Prev=(int32u)-1;
290 string Key;
291 Ztring Value;
292 Skip_String(SizeOfLine(), "Object name");
293 while (Element_Offset<Element_Size)
294 {
295 if (Get_Next(Key, Value))
296 {
297 for (;;)
298 {
299 Get_Next(Key, Value);
300 if (Key.empty())
301 break;
302 else if (Key=="Root")
303 {
304 int32u ObjectNumber=Value.To_int32u();
305 Objects[ObjectNumber].Type=Type_Root;
306 Objects[ObjectNumber].TopObject=(int32u)-1;
307 Objects[(int32u)-1].Bottoms.push_back(ObjectNumber);
308 Param_Info1(__T("Document Catalog is at offset 0x"+Ztring().From_Number(Objects[ObjectNumber].Offset, 16)));
309 }
310 else if (Key=="Info")
311 {
312 int32u ObjectNumber=Value.To_int32u();
313 Objects[ObjectNumber].Type=Type_Info;
314 Objects[ObjectNumber].TopObject=(int32u)-1;
315 Objects[(int32u)-1].Bottoms.push_back(ObjectNumber);
316 Param_Info1(__T("Info is at offset 0x"+Ztring().From_Number(Objects[ObjectNumber].Offset, 16)));
317 }
318 else if (Key=="Prev")
319 {
320 Prev=Value.To_int32u();
321 Param_Info1(__T("Previous Cross-Reference Table is at offset 0x"+Ztring().From_Number(Prev, 16)));
322 }
323 }
324
325 continue;
326 }
327
328 if (Key.empty())
329 break;
330 }
331
332 Element_End0();
333
334 //Previous Cross-Reference Table
335 if (Prev!=(int32u)-1)
336 {
337 GoTo(Prev);
338 return;
339 }
340
341 objects::iterator Object_Top=Objects.find((int32u)-1);
342 if (Offsets.empty() || Object_Top==Objects.end())
343 {
344 Finish();
345 return;
346 }
347
348 sort(Offsets.begin(), Offsets.end());
349
350 //Offsets_Current=Offsets.end(); //No more used for the moment
351 Objects[(int32u)-1].BottomPos=0;
352 Objects_Current=Objects.find(Object_Top->second.Bottoms[0]);
353 GoTo(Objects_Current->second.Offset);
354 State=State_Parsing_object;
355}

357//---------------------------------------------------------------------------
358void File_Pdf::startxref()
359{
360 //We need to find the exact begin
361 Buffer_Offset=Buffer_Size-1;
362 while (Buffer_Offset && (Buffer[Buffer_Offset]=='\r' || Buffer[Buffer_Offset]=='\n'))
363 Buffer_Offset--;
364 Buffer_Offset-=5; // "%%EOF"
365 while (Buffer_Offset && (Buffer[Buffer_Offset]=='\r' || Buffer[Buffer_Offset]=='\n'))
366 Buffer_Offset--;
367 while (Buffer_Offset && Buffer[Buffer_Offset]>='0' && Buffer[Buffer_Offset]<='9') // Value
368 Buffer_Offset--;
369 while (Buffer_Offset && (Buffer[Buffer_Offset]=='\r' || Buffer[Buffer_Offset]=='\n'))
370 Buffer_Offset--;
371
372 Buffer_Offset-=8;
373
374 //Parsing
375 Element_Begin1("Cross-Reference Table Offset");
376
377 string xrefOffsetS;
378 Skip_String(SizeOfLine(), "Object name");
379 Get_String (SizeOfLine(), xrefOffsetS, "xref Offset");
380 while (Buffer_Offset<Buffer_Size && (Buffer[Buffer_Offset]=='\r' || Buffer[Buffer_Offset]=='\n'))
381 ++Buffer_Offset;
382 int32u xref_Offset=atoi(xrefOffsetS.c_str());
383
384 Element_End0();
385
386 //Going to xref
387 if (xref_Offset>Offsets_Max)
388 Offsets_Max=xref_Offset;
389 GoTo (xref_Offset);
390 State=State_Parsing_xref;
391}

393//---------------------------------------------------------------------------
394void File_Pdf::eof()
395{
396 if (File_Size!=(int64u)-1 && File_Offset+Buffer_Size<File_Size)
397 {
398 Element_WaitForMoreData();
399 return;
400 }
401
402 //We need to find the exact begin
403 Buffer_Offset=Buffer_Size-1;
404 while (Buffer_Offset && (Buffer[Buffer_Offset]=='\r' || Buffer[Buffer_Offset]=='\n'))
405 Buffer_Offset--;
406
407 Buffer_Offset-=5;
408
409 //Parsing
410 Element_Begin1("End Of File");
411
412 Skip_String(SizeOfLine(), "Object name");
413
414 Element_End0();
415}

479//---------------------------------------------------------------------------
480void File_Pdf::Object_Metadata()
481{
482 Element_Info1("Metadata");
483
484 //Parsing
485 string Key;
486 Ztring Value;
487 int32u Length=0;
488 while (Element_Offset<Element_Size)
489 {
490 if (Get_Next(Key, Value))
491 {
492 for (;;)
493 {
494 Get_Next(Key, Value);
495 if (Key.empty())
496 break;
497 else if (Key=="Length")
498 {
499 Length=Value.To_int32u();
500 }
501 }
502
503 continue;
504 }
505
506 if (Key.empty())
507 break;
508 else if (Key=="stream")
509 {
510 //Removig end of lines
511 if (Element_Offset<Element_Size && Buffer[Buffer_Offset+(size_t)Element_Offset]=='\r')
512 Element_Offset++;
513 if (Element_Offset<Element_Size && Buffer[Buffer_Offset+(size_t)Element_Offset]=='\n')
514 Element_Offset++;
515
516 File_Xmp MI;
517 Open_Buffer_Init(&MI, Length);
518 Open_Buffer_Continue(&MI, Buffer+Buffer_Offset+(size_t)Element_Offset, Length);
519 Skip_XX(Length, "Stream, Data");
520 Open_Buffer_Finalize(&MI);
521 Merge(MI, Stream_General, 0, 0);
522 }
523 }

531int64u File_Pdf::SizeOfLine()
532{
533 //while (Element_Offset<Element_Size && (Buffer[Buffer_Offset+(size_t)Element_Offset]=='\r' || Buffer[Buffer_Offset+(size_t)Element_Offset]=='\n' || Buffer[Buffer_Offset+(size_t)Element_Offset]=='<' || Buffer[Buffer_Offset+(size_t)Element_Offset]=='>'))
534 while (Element_Offset<Element_Size && (Buffer[Buffer_Offset+(size_t)Element_Offset]=='\r' || Buffer[Buffer_Offset+(size_t)Element_Offset]=='\n' || Buffer[Buffer_Offset+(size_t)Element_Offset]==' '))
535 Element_Offset++;
536 size_t End=Buffer_Offset+(size_t)Element_Offset;
537 while (End<Buffer_Size && Buffer[End]!='\r' && Buffer[End]!='\n' && !(End+1<Buffer_Size && Buffer[End]=='<' && Buffer[End+1]=='<') && !(End+1<Buffer_Size && Buffer[End]=='>' && Buffer[End+1]=='>'))
538 End++;
539 return End-(Buffer_Offset+(size_t)Element_Offset);
540}

542//---------------------------------------------------------------------------
543bool File_Pdf::Get_Next(string &Key, Ztring &Value)
544{
545 Key.clear();
546 Value.clear();
547
548 string Line;
549
550 //Removig end of lines
551 while (Element_Offset<Element_Size && (Buffer[Buffer_Offset+(size_t)Element_Offset]=='\r' || Buffer[Buffer_Offset+(size_t)Element_Offset]=='\n' || Buffer[Buffer_Offset+(size_t)Element_Offset]==' '))
552 Element_Offset++;
553
554 //End
555 if (Element_Offset>=Element_Size)
556 return true;
557
558 //Testing Catalog
559 Peek_String (2, Line);
560 if (Line=="<<")
561 {
562 Element_Offset+=2;
563 Catalog_Level++;
564 return true;
565 }
566 else if (Line==">>")
567 {
568 Element_Offset+=2;
569 Catalog_Level--;
570 return true;
571 }
572
573 //Getting a complete line
574 Peek_String (SizeOfLine(), Line);
575
576 //Testing Catalog
577 size_t Catalog_End=Line.find(">>");
578 if (Catalog_End!=String::npos)
579 Line.resize(Catalog_End);
580
581 //Testing stream
582 if (Line=="stream")
583 {
584 Skip_String(Line.size(), "Stream, Header");
585 Key=Line;
586 return false;
587 }
588 if (Line=="endstream")
589 {
590 Skip_String(Line.size(), "Stream, Footer");
591 Key=Line;
592 return false;
593 }
594
595 //Testing object
596 if (Line=="endobj")
597 {
598 Skip_String(Line.size(), "Footer");
599 Key=Line;
600 return false;
601 }
602
603 //Base
604 int64u Line_Base=Element_Offset;
605
606 //Testing next key
607 size_t Line_End=0;
608 size_t Line_Begin=Line_End;
609
610 // Key-Value
611 if (Line_Begin<Line.size() && Line[Line_Begin]=='/')
612 {
613 Line_End= Line_Begin+1;
614 size_t HasParenthesis=0;
615 size_t HasBracket=0;
616 size_t HasSpace=0;
617 size_t HasValue=0;
618 for (;;)
619 {
620 if (Line_End==Line.size())
621 break;
622
623 if (!HasParenthesis && !HasBracket && HasValue && Line[Line_End]=='<' && Line_End+1<Line.size() && Line[Line_End+1]=='<')
624 break;
625 if (!HasParenthesis && !HasBracket && HasValue && Line[Line_End]=='/')
626 break;
627 else if (!HasValue && Line[Line_End]=='/')
628 ++HasValue;
629 else if (!HasValue && HasSpace)
630 ++HasValue;
631
632 if (Line[Line_End]==' ')
633 ++HasSpace;
634
635 if (Line[Line_End]=='(')
636 ++HasParenthesis;
637 if (HasParenthesis && Line[Line_End]==')')
638 --HasParenthesis;
639
640 if (Line[Line_End]=='[')
641 ++HasBracket;
642 if (HasBracket && Line[Line_End]==']')
643 --HasBracket;
644
645 ++Line_End;
646 }
647 while(Line_End && Line[Line_End-1]==' ')
648 Line_End--; //Removing trailing spaces
649
650 Element_Offset=Line_Base+Line_Begin;
651 string KeyValue;
652 Get_String(Line_End-Line_Begin, KeyValue, "Key-Value");
653
654 size_t Key_Max=KeyValue.find_first_of(" (");
655 if (Key_Max==string::npos)
656 Key_Max=KeyValue.size();
657 Key=KeyValue.substr(1, Key_Max-1);
658 size_t Value_Min=Key_Max;
659 while (Value_Min<KeyValue.size() && KeyValue[Value_Min]==' ')
660 ++Value_Min;
661 if (Value_Min<KeyValue.size() && KeyValue[Value_Min]=='(')
662 {
663 ++Value_Min;
664 size_t Value_Max=KeyValue.find(')', Value_Min);
665 if (Value_Max!=string::npos)
666 {
667 //TODO
668 Value.From_UTF8(KeyValue.c_str()+Value_Min, Value_Max-Value_Min);
669 }
670 else
671 Value.From_UTF8(KeyValue.c_str()+Value_Min);
672 }
673 else
674 Value.From_UTF8(KeyValue.c_str()+Value_Min);
675 return false;
676 }
677
678 return false;
679}
