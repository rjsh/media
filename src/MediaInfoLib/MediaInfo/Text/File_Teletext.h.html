65 //Streams
66 struct stream
67 {
68 vector<std::wstring> CC_Displayed_Values;
69
70 stream()
71 {
72 CC_Displayed_Values.resize(26);
73 for (size_t PosY=0; PosY<26; ++PosY)
74 CC_Displayed_Values[PosY].resize(40, L' ');
75 }
76
77 void Clear()
78 {
79 for (size_t PosY=0; PosY<26; ++PosY)
80 for (size_t PosX=0; PosX<40; ++PosX)
81 CC_Displayed_Values[PosY][PosX]=L' ';
82 }
83
84 };

86 streams Streams;
