42int64u Ttml_str2timecode(const char* Value)
43{
44 size_t Length=strlen(Value);
45 if (Length>=8
46 && Value[0]>='0' && Value[0]<='9'
47 && Value[1]>='0' && Value[1]<='9'
48 && Value[2]==':'
49 && Value[3]>='0' && Value[3]<='9'
50 && Value[4]>='0' && Value[4]<='9'
51 && Value[5]==':'
52 && Value[6]>='0' && Value[6]<='9'
53 && Value[7]>='0' && Value[7]<='9')
54 {
55 int64u ToReturn=(int64u)(Value[0]-'0')*10*60*60*1000000000
56 +(int64u)(Value[1]-'0') *60*60*1000000000
57 +(int64u)(Value[3]-'0') *10*60*1000000000
58 +(int64u)(Value[4]-'0') *60*1000000000
59 +(int64u)(Value[6]-'0') *10*1000000000
60 +(int64u)(Value[7]-'0') *1000000000;
61 if (Length>=9 && (Value[8]=='.' || Value[8]==','))
62 {
63 if (Length>9+9)
64 Length=9+9; //Nanoseconds max
65 const char* Value_End=Value+Length;
66 Value+=9;
67 int64u Multiplier=100000000;
68 while (Value<Value_End)
69 {
70 ToReturn+=(int64u)(*Value-'0')*Multiplier;
71 Multiplier/=10;
72 Value++;
73 }
74 }
75
76 return ToReturn;
77 }
78 else if (Length>=2
79 && Value[Length-1]=='s')
80 {
81 return (int64u)(atof(Value)*1000000000);
82 }
83 else
84 return (int64u)-1;
85}

149void File_Ttml::Read_Buffer_Continue()
150{
151 tinyxml2::XMLDocument document;
152
153 if (!FileHeader_Begin_XML(document))
154 return;
155
156 XMLElement* Root=document.FirstChildElement("tt");
157 if (!Root)
158 {
159 Reject();
160 return;
161 }
162
163 if (!Status[IsAccepted])
164 {
165 Accept();
166
167 #if MEDIAINFO_EVENTS
168 MuxingMode=(int8u)-1;
169 if (StreamIDs_Size>=2 && ParserIDs[StreamIDs_Size-2]==MediaInfo_Parser_Mpeg4)
170 MuxingMode=11; //MPEG-4
171 if (StreamIDs_Size>2 && ParserIDs[StreamIDs_Size-2]==MediaInfo_Parser_Mxf) //Only if referenced MXF
172 MuxingMode=13; //MXF
173 #endif //MEDIAINFO_EVENTS
174
175 #if MEDIAINFO_DEMUX && MEDIAINFO_NEXTPACKET
176 if (Config->NextPacket_Get() && Config->Event_CallBackFunction_IsSet())
177 return; // Waiting for NextPacket
178 #endif //MEDIAINFO_DEMUX && MEDIAINFO_NEXTPACKET
179 }
180
181 tinyxml2::XMLElement* div=NULL;
182 #if MEDIAINFO_EVENTS
183 tinyxml2::XMLElement* p=NULL;
184 #endif //MEDIAINFO_EVENTS
185 for (XMLElement* tt_element=Root->FirstChildElement(); tt_element; tt_element=tt_element->NextSiblingElement())
186 {
187 //body
188 if (!strcmp(tt_element->Value(), "body"))
189 {
190 for (XMLElement* body_element=tt_element->FirstChildElement(); body_element; body_element=body_element->NextSiblingElement())
191 {
192 //div
193 if (!strcmp(body_element->Value(), "div"))
194 {
195 for (XMLElement* div_element=body_element->FirstChildElement(); div_element; div_element=div_element->NextSiblingElement())
196 {
197 //p
198 if (!strcmp(div_element->Value(), "p"))
199 {
200 div=body_element;
201 #if MEDIAINFO_EVENTS
202 p=div_element;
203 #endif //MEDIAINFO_EVENTS
204 break;
205 }
206 }
207
208 if (div)
209 break;
210 }
211 }
212
213 if (div)
214 break;
215 }
216 }
217
218 #if MEDIAINFO_DEMUX
219 Demux(Buffer, Buffer_Size, ContentType_MainStream);
220 #endif //MEDIAINFO_DEMUX
221
222 // Output
223 #if MEDIAINFO_EVENTS
224 for (; p; p=p->NextSiblingElement())
225 {
226 //p
227 if (!strcmp(p->Value(), "p"))
228 {
229 const char* Attribute;
230
231 int64u DTS_Begin=(int64u)-1;
232 Attribute=p->Attribute("begin");
233 if (Attribute)
234 DTS_Begin=Ttml_str2timecode(Attribute);
235 int64u DTS_End=(int64u)-1;
236 Attribute=p->Attribute("end");
237 if (Attribute)
238 DTS_End=Ttml_str2timecode(Attribute);
239 string ContentUtf8;
240 XMLPrinter printer;
241 p->Accept(&printer);
242 ContentUtf8+=printer.CStr();
243 while (!ContentUtf8.empty() && (ContentUtf8[ContentUtf8.size()-1]=='\r' || ContentUtf8[ContentUtf8.size()-1]=='\n'))
244 ContentUtf8.resize(ContentUtf8.size()-1);
245 Ztring Content; if (p->FirstChild()) Content.From_UTF8(p->FirstChild()->Value());
246 Frame_Count++;
247 }
248 }
249 #endif //MEDIAINFO_EVENTS
250
251 Buffer_Offset=Buffer_Size;
252}
