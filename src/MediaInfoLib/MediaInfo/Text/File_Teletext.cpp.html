85//---------------------------------------------------------------------------
86void File_Teletext::Streams_Finish()
87{
88 //Filling when it is from MpegPs
89 //TODO: filter subtitles and non subtitles, some files have normal teletext in subtitles block.
90 if (Parser)
91 {
92 for (size_t StreamKind=Stream_General+1; StreamKind<Stream_Max; StreamKind++)
93 for (size_t StreamPos=0; StreamPos<Parser->Count_Get((stream_t)StreamKind); StreamPos++)
94 {
95 Stream_Prepare((stream_t)StreamKind);
96 Merge(*Parser, (stream_t)StreamKind, StreamPos_Last, StreamPos);
97 }
98
99 return;
100 }
101
102 if (Teletexts && !Teletexts->empty())
103 for (std::map<int16u, teletext>::iterator Teletext=Teletexts->begin(); Teletext!=Teletexts->end(); ++Teletext)
104 {
105 std::map<std::string, Ztring>::iterator Info_Format=Teletext->second.Infos.find("Format");
106 Stream_Prepare((Info_Format!=Teletext->second.Infos.end() && Info_Format->second==__T("Teletext"))?Stream_Other:Stream_Text);
107 Fill(StreamKind_Last, StreamPos_Last, General_ID, Ztring::ToZtring(Teletext->first), true);
108 Fill(StreamKind_Last, StreamPos_Last, General_ID_String, Ztring::ToZtring(Teletext->first), true);
109
110 for (std::map<std::string, ZenLib::Ztring>::iterator Info=Teletext->second.Infos.begin(); Info!=Teletext->second.Infos.end(); ++Info)
111 {
112 if (Retrieve(StreamKind_Last, StreamPos_Last, Info->first.c_str()).empty())
113 Fill(StreamKind_Last, StreamPos_Last, Info->first.c_str(), Info->second);
114 }
115 }
116 else
117 for (streams::iterator Stream = Streams.begin(); Stream != Streams.end(); ++Stream)
118 {
119 Stream_Prepare(IsSubtitle?Stream_Text:Stream_Other);
120 Fill(StreamKind_Last, StreamPos_Last, General_ID, Ztring::ToZtring(Stream->first, 16));
121 Fill(StreamKind_Last, StreamPos_Last, "Format", IsSubtitle?"Teletext Subtitle":"Teletext");
122 }
123}

129//---------------------------------------------------------------------------
130bool File_Teletext::Synchronize()
131{
132 //Synchronizing
133 while (Buffer_Offset+3<=Buffer_Size)
134 {
135 while (Buffer_Offset+3<=Buffer_Size)
136 {
137 if (Buffer[Buffer_Offset ]==0x55
138 && Buffer[Buffer_Offset+1]==0x55
139 && Buffer[Buffer_Offset+2]==0x27)
140 break; //while()
141
142 Buffer_Offset++;
143 }
144
145 if (Buffer_Offset+3<=Buffer_Size) //Testing if size is coherant
146 {
147 if (Buffer_Offset+45==Buffer_Size)
148 break;
149
150 if (Buffer_Offset+45+3>Buffer_Size)
151 return false; //Wait for more data
152
153 if (Buffer[Buffer_Offset ]==0x55
154 && Buffer[Buffer_Offset+1]==0x55
155 && Buffer[Buffer_Offset+2]==0x27)
156 break; //while()
157
158 Buffer_Offset++;
159 }
160 }
161
162 //Must have enough buffer for having header
163 if (Buffer_Offset+3>=Buffer_Size)
164 return false;
165
166 //Synched is OK
167 if (!Status[IsAccepted])
168 {
169 //For the moment, we accept only if the file is in sync, the test is not strict enough
170 if (Buffer_Offset)
171 {
172 Reject();
173 return false;
174 }
175
176 Accept();
177 }
178 return true;
179}

210//---------------------------------------------------------------------------
211void File_Teletext::Read_Buffer_Unsynched()
212{
213 for (streams::iterator Stream=Streams.begin(); Stream!=Streams.end(); ++Stream)
214 {
215 Stream_HasChanged=0;
216 for (size_t PosY=0; PosY<26; ++PosY)
217 for (size_t PosX=0; PosX<40; ++PosX)
218 if (Stream->second.CC_Displayed_Values[PosY][PosX]!=L' ')
219 {
220 Stream->second.CC_Displayed_Values[PosY][PosX]=L' ';
221 Stream_HasChanged=Stream->first;
222 }
223
224 if (Stream_HasChanged)
225 {
226 HasChanged();
227 Stream_HasChanged=0;
228 }
229 }
230
231 //Unsynching when it is from MpegPs
232 if (Parser)
233 Parser->Open_Buffer_Unsynch();
234}

236static inline int8u ReverseBits(int8u c)
237{
238 // Input: bit order is 76543210
239 //Output: bit order is 01234567
240 c = (c & 0x0F) << 4 | (c & 0xF0) >> 4;
241 c = (c & 0x33) << 2 | (c & 0xCC) >> 2;
242 c = (c & 0x55) << 1 | (c & 0xAA) >> 1;
243 return c;
244}

246//---------------------------------------------------------------------------
247void File_Teletext::Read_Buffer_Continue()
248{
249 #if defined(MEDIAINFO_MPEGPS_YES)
250 if (FromMpegPs)
251 {
252 if (!Status[IsAccepted])
253 {
254 Accept();
255 MustSynchronize=false;
256 }
257
258 Skip_B1( "data_identifier");
259 while (Element_Offset<Element_Size)
260 {
261 int8u data_unit_id, data_unit_length;
262 Get_B1 (data_unit_id, "data_unit_id");
263 Get_B1 (data_unit_length, "data_unit_length");
264 if (data_unit_length)
265 {
266 Skip_B1( "field/line");
267 if (data_unit_id==0x03 && data_unit_length==0x2C)
268 {
269 int8u Data[43];
270 for (int8u Pos=0; Pos<43; ++Pos)
271 Data[Pos]=ReverseBits(Buffer[Buffer_Offset+(size_t)Element_Offset+Pos]);
272
273 if (Parser==NULL)
274 {
275 Parser=new File_Teletext();
276 Parser->MustSynchronize=false;
277 Parser->IsSubtitle=data_unit_id==0x03;
278 Parser->Teletexts=Teletexts;
279 Open_Buffer_Init(Parser);
280 Parser->Accept(); //Force to be accepted because there is no other synchronization layer (MustSynchronize set to false)
281 }
282 Element_Code=data_unit_id;
283 int8u Temp[2];
284 Temp[0]=0x55;
285 Temp[1]=0x55;
286 Demux(Temp, 2, ContentType_MainStream);
287 Demux(Data, 43, ContentType_MainStream);
288 Parser->FrameInfo=FrameInfo;
289 Open_Buffer_Continue(Parser, Data, 43);
290 Element_Offset+=43;
291 }
292 else
293 Skip_XX(data_unit_length-1, "Data");
294 }
295 }
296 }
297 #endif
298}

304//---------------------------------------------------------------------------
305void File_Teletext::Header_Parse()
306{
307 //Parsing
308 if (MustSynchronize)
309 Skip_B2( "Clock run-in");
310 Skip_B1( "Framing code");
311
312 //Magazine and Packet Number (for all packets)
313 X=0, Y=0;
314 bool P1, D1, P2, D2, P3, D3, P4, D4;
315 bool B;
316 BS_Begin_LE();
317 Element_Begin1("Magazine (X or M)");
318 Get_TB (P1, "Hamming 8/4");
319 Get_TB (D1, "Magazine 0");
320 if (D1)
321 X|=1<<0;
322 Get_TB (P2, "Hamming 8/4");
323 Get_TB (D2, "Magazine 1");
324 if (D2)
325 X|=1<<1;
326 Get_TB (P3, "Hamming 8/4");
327 Get_TB (D3, "Magazine 2");
328 if (D3)
329 X|=1<<2;
330 Element_Info1(X);
331 Element_End0();
332 Element_Begin1("Packet Number (Y)");
333 Get_TB (P4, "Hamming 8/4");
334 Get_TB (D4, "Packet Number 0");
335 if (D4)
336 Y|=1<<0;
337 /*
338 {
339 //Hamming 8/4
340 bool A=P1^D1^D3^D4;
341 bool B=D1^P2^D2^D4;
342 bool C=D1^D2^P3^D3;
343 bool D=P1^D1^P2^D2^P3^D3^P4^D4;
344 if (A && B && C && D)
345 {
346 }
347 else
348 {
349 }
350 }
351 */
352 Skip_TB( "Hamming 8/4");
353 Get_TB (B, "Packet Number 1");
354 if (B)
355 Y|=1<<1;
356 Skip_TB( "Hamming 8/4");
357 Get_TB (B, "Packet Number 2");
358 if (B)
359 Y|=1<<2;
360 Skip_TB( "Hamming 8/4");
361 Get_TB (B, "Packet Number 3");
362 if (B)
363 Y|=1<<3;
364 Skip_TB( "Hamming 8/4");
365 Get_TB (B, "Packet Number 4");
366 if (B)
367 Y|=1<<4;
368 if (X==0)
369 X=8; // A packet with a magazine value of 0 is referred to as belonging to magazine 8
370 Element_Info1(Y);
371 Element_End0();
372
373 //Page header
374 if (Y==0)
375 {
376 C.reset();
377 CharacterSubset=0;
378
379 Element_Begin1("Page header");
380 int8u PU=0, PT=0;
381 bool B;
382 Element_Begin1("Page Units");
383 Skip_TB( "Hamming 8/4");
384 Get_TB (B, "Page Units 0");
385 if (B)
386 PU|=1<<0;
387 Skip_TB( "Hamming 8/4");
388 Get_TB (B, "Page Units 1");
389 if (B)
390 PU|=1<<1;
391 Skip_TB( "Hamming 8/4");
392 Get_TB (B, "Page Units 2");
393 if (B)
394 PU|=1<<2;
395 Skip_TB( "Hamming 8/4");
396 Get_TB (B, "Page Units 3");
397 if (B)
398 PU|=1<<3;
399 Element_Info1(PU);
400 Element_End0();
401 Element_Begin1("Page Tens");
402 Skip_TB( "Hamming 8/4");
403 Get_TB (B, "Page Tens 0");
404 if (B)
405 PT|=1<<0;
406 Skip_TB( "Hamming 8/4");
407 Get_TB (B, "Page Tens 1");
408 if (B)
409 PT|=1<<1;
410 Skip_TB( "Hamming 8/4");
411 Get_TB (B, "Page Tens 2");
412 if (B)
413 PT|=1<<2;
414 Skip_TB( "Hamming 8/4");
415 Get_TB (B, "Page Tens 3");
416 if (B)
417 PT|=1<<3;
418 Element_Info1(PT);
419 Element_End0();
420 PageNumber=(PT<<4)|PU;
421 Element_Info1(Ztring::ToZtring(PageNumber, 16));
422
423 int8u S1=0, S2=0, S3=0, S4=0;
424 Element_Begin1("Page sub-code 1");
425 Skip_TB( "Hamming 8/4");
426 Get_TB (B, "S1 0");
427 if (B)
428 S1|=1<<0;
429 Skip_TB( "Hamming 8/4");
430 Get_TB (B, "S1 1");
431 if (B)
432 S1|=1<<1;
433 Skip_TB( "Hamming 8/4");
434 Get_TB (B, "S1 2");
435 if (B)
436 S1|=1<<2;
437 Skip_TB( "Hamming 8/4");
438 Get_TB (B, "S1 3");
439 if (B)
440 S1|=1<<3;
441 Element_Info1(S1);
442 Element_End0();
443 Element_Begin1("Page sub-code 2");
444 Skip_TB( "Hamming 8/4");
445 Get_TB (B, "S2 0");
446 if (B)
447 S2|=1<<0;
448 Skip_TB( "Hamming 8/4");
449 Get_TB (B, "S2 1");
450 if (B)
451 S2|=1<<1;
452 Skip_TB( "Hamming 8/4");
453 Get_TB (B, "S2 2");
454 if (B)
455 S2|=1<<2;
456 Skip_TB( "Hamming 8/4");
457 Get_TB (B, "C4 - Erase Page");
458 if (B)
459 C[4]=true;
460 Element_Info1(S2);
461 Element_End0();
462 Element_Begin1("Page sub-code 3");
463 Skip_TB( "Hamming 8/4");
464 Get_TB (B, "S3 0");
465 if (B)
466 S3|=1<<0;
467 Skip_TB( "Hamming 8/4");
468 Get_TB (B, "S3 1");
469 if (B)
470 S3|=1<<1;
471 Skip_TB( "Hamming 8/4");
472 Get_TB (B, "S3 2");
473 if (B)
474 S3|=1<<2;
475 Skip_TB( "Hamming 8/4");
476 Get_TB (B, "S3 3");
477 if (B)
478 S3|=1<<3;
479 Element_Info1(S3);
480 Element_End0();
481 Element_Begin1("Page sub-code 4");
482 Skip_TB( "Hamming 8/4");
483 Get_TB (B, "S4 0");
484 if (B)
485 S4|=1<<0;
486 Skip_TB( "Hamming 8/4");
487 Get_TB (B, "S4 1");
488 if (B)
489 S4|=1<<1;
490 Skip_TB( "Hamming 8/4");
491 Get_TB (B, "C5 - Newsflash");
492 if (B)
493 C[5]=true;
494 #if MEDIAINFO_TRACE
495 if (B)
496 Element_Info1("Newsflash");
497 #endif //MEDIAINFO_TRACE
498 Skip_TB( "Hamming 8/4");
499 Get_TB (B, "C6 - Subtitle");
500 if (B)
501 C[6]=true;
502 Element_Info1(S4);
503 Element_End0();
504 Element_Begin1("Control bits");
505 Skip_TB( "Hamming 8/4");
506 Get_TB (B, "C7 - Suppress Header");
507 if (B)
508 C[7]=true;
509 Skip_TB( "Hamming 8/4");
510 Get_TB (B, "C8 - Update Indicator");
511 if (B)
512 C[8]=true;
513 Skip_TB( "Hamming 8/4");
514 Get_TB (B, "C9 - Interrupted Sequence");
515 if (B)
516 C[9]=true;
517 Skip_TB( "Hamming 8/4");
518 Get_TB (B, "C10 - Inhibit Display");
519 if (B)
520 C[10]=true;
521 Skip_TB( "Hamming 8/4");
522 Get_TB (B, "C11 - Magazine Serial");
523 if (B)
524 C[11]=true;
525 Skip_TB( "Hamming 8/4");
526 Get_TB (B, "C12 - Character Subset");
527 if (B)
528 CharacterSubset |= (1<<2);
529 Skip_TB( "Hamming 8/4");
530 Get_TB (B, "C13 - Character Subset");
531 if (B)
532 CharacterSubset |= (1<<1);
533 Skip_TB( "Hamming 8/4");
534 Get_TB (B, "C14 - Character Subset");
535 if (B)
536 CharacterSubset |= (1<<0);
537 Element_End0();
538
539 SubCode=(S4<<12)|(S3<<8)|(S2<<4)|S1;
540
541 Element_End0();
542 }
543 BS_End_LE();
544
545 #if MEDIAINFO_TRACE
546 if (C[4])
547 Element_Info1("Erase Page");
548 if (C[5])
549 Element_Info1("Newsflash");
550 if (C[6])
551 Element_Info1("Subtitle");
552 if (C[7])
553 Element_Info1("Suppress Header");
554 if (C[8])
555 Element_Info1("Update Indicator");
556 if (C[9])
557 Element_Info1("Interrupted Sequence");
558 if (C[10])
559 Element_Info1("Inhibit Display");
560 if (C[11])
561 Element_Info1("Magazine Serial");
562 Element_Info1(Ztring::ToZtring((X<<8)|PageNumber, 16)+__T(':')+Ztring().From_CC2(SubCode));
563 Element_Info1(Y);
564 #endif // MEDIAINFO_TRACE
565
566 Header_Fill_Size(45-(MustSynchronize?0:2));
567
568 if (Y==0)
569 {
570 if (Stream_HasChanged)
571 {
572 HasChanged();
573 Stream_HasChanged=0;
574 }
575
576 if (C[4] && PageNumber!=0xFF)
577 {
578 stream &Stream=Streams[(X<<8)|PageNumber];
579 for (size_t PosY=0; PosY<26; ++PosY)
580 for (size_t PosX=0; PosX<40; ++PosX)
581 if (Stream.CC_Displayed_Values[PosY][PosX]!=L' ')
582 {
583 Stream.CC_Displayed_Values[PosY][PosX]=L' ';
584 Stream_HasChanged=(X<<8)|PageNumber;
585 }
586 }
587 }

590//---------------------------------------------------------------------------
591void File_Teletext::Data_Parse()
592{
593 if (PageNumber==0xFF)
594 {
595 Skip_XX(Y?40:32, "Junk");
596 }
597 else if (Y>=26)
598 {
599 Skip_XX(40, "Special commands");
600 }
601 else
602 {
603 Element_Begin1("Data bytes");
604 stream &Stream=Streams[(X<<8)|PageNumber];
605 size_t PosX=Y?0:8;
606 for (; PosX<40; ++PosX)
607 {
608 int8u byte;
609 Get_B1(byte, "Byte");
610 byte&=0x7F;
611 if (byte<0x20)
612 byte=0x20;
613 Param_Info1(Ztring().From_Local((const char*)&byte, 1));
614 if (byte!=Stream.CC_Displayed_Values[Y][PosX] && (!C[7] || Y)) // C[7] is "Suppress Header", to be tested when Y==0
615 {
616 Char Uni;
617 switch (CharacterSubset)
618 {
619 case 0x00: //English
620 switch(byte)
621 {
622 case 0x23: Uni = __T('\xA3'); break;
623 case 0x24: Uni = __T('$'); break;
624 case 0x40: Uni = __T('@'); break;
625 case 0x5B: Uni = __T('\x2190'); break;
626 case 0x5C: Uni = __T('\xBD'); break;
627 case 0x5D: Uni = __T('\x2192'); break;
628 case 0x5E: Uni = __T('\x2191'); break;
629 case 0x5F: Uni = __T('#'); break;
630 case 0x60: Uni = __T('-'); break;
631 case 0x7B: Uni = __T('\xBC'); break;
632 case 0x7C: Uni = __T('|'); break;
633 case 0x7D: Uni = __T('\xBE'); break;
634 case 0x7E: Uni = __T('\xF7'); break;
635 default: Uni=byte;
636 }
637 break;
638 case 0x04: //French
639 switch(byte)
640 {
641 case 0x23: Uni = __T('\xE9'); break;
642 case 0x24: Uni = __T('\xEF'); break;
643 case 0x40: Uni = __T('\xE0'); break;
644 case 0x5B: Uni = __T('\xEB'); break;
645 case 0x5C: Uni = __T('\xEA'); break;
646 case 0x5D: Uni = __T('\xF9'); break;
647 case 0x5E: Uni = __T('\xEE'); break;
648 case 0x5F: Uni = __T('#'); break;
649 case 0x60: Uni = __T('\xE8'); break;
650 case 0x7B: Uni = __T('\xE2'); break;
651 case 0x7C: Uni = __T('\xF4'); break;
652 case 0x7D: Uni = __T('\xFB'); break;
653 case 0x7E: Uni = __T('\xE7'); break;
654 default: Uni=byte;
655 }
656 break;
657 case 0x05: //Portuguese/Spanish
658 switch(byte)
659 {
660 case 0x23: Uni = __T('\xE7'); break;
661 case 0x24: Uni = __T('$'); break;
662 case 0x40: Uni = __T('\xA1'); break;
663 case 0x5B: Uni = __T('\xE1'); break;
664 case 0x5C: Uni = __T('\xE9'); break;
665 case 0x5D: Uni = __T('\xED'); break;
666 case 0x5E: Uni = __T('\xF3'); break;
667 case 0x5F: Uni = __T('\xFA'); break;
668 case 0x60: Uni = __T('\xBF'); break;
669 case 0x7B: Uni = __T('\xFC'); break;
670 case 0x7C: Uni = __T('\xF1'); break;
671 case 0x7D: Uni = __T('\xE8'); break;
672 case 0x7E: Uni = __T('\xE0'); break;
673 default: Uni=byte;
674 }
675 break;
676 default: Uni=byte;
677 }
678
679 Stream.CC_Displayed_Values[Y][PosX]=Uni;
680 Stream_HasChanged=(X<<8)|PageNumber;
681 }
682 }
683 Element_End0();
684 }
685
686 #if MEDIAINFO_TRACE
687 if (PageNumber==0xFF)
688 {
689 Element_Name("Skip");
690 }
691 else
692 {
693 Element_Name(Ztring::ToZtring((X<<8)|PageNumber, 16)+__T(':')+Ztring().From_CC2(SubCode));
694 Element_Info1(Y);
695 if (Y<26)
696 {
697 Element_Info1(Ztring().From_Unicode(Streams[(X<<8)|PageNumber].CC_Displayed_Values[Y]));
698 if (Y==0)
699 {
700 if (C[4])
701 Element_Info1("Erase Page");
702 if (C[5])
703 Element_Info1("Newsflash");
704 if (C[6])
705 Element_Info1("Subtitle");
706 if (C[7])
707 Element_Info1("Suppress Header");
708 if (C[8])
709 Element_Info1("Update Indicator");
710 if (C[9])
711 Element_Info1("Interrupted Sequence");
712 if (C[10])
713 Element_Info1("Inhibit Display");
714 if (C[11])
715 Element_Info1("Magazine Serial");
716 }
717 }
718 }
719 #endif //MEDIAINFO_TRACE
