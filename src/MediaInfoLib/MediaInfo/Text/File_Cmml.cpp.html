42void File_Cmml::Header_Parse()
43{
44 //Filling
45 Header_Fill_Code(0, "Cmml");
46 Header_Fill_Size(Element_Size);
47}
48
49//---------------------------------------------------------------------------
50void File_Cmml::Data_Parse()
51{
52 //Parsing
53 if (Status[IsAccepted])
54 Configuration();
55 else
56 Identification();
57}
58
59//***************************************************************************
60// Elements
61//***************************************************************************
62
63//---------------------------------------------------------------------------
64void File_Cmml::Identification()
65{
66 Element_Name("Identification");
67
68 //Parsing
69 int16u VersionMajor, VersionMinor;
70 Skip_Local(8, "Signature");
71 Get_L2 (VersionMajor, "version major");
72 Get_L2 (VersionMinor, "version minor");
73 Skip_L8( "granule rate numerator");
74 Skip_L8( "granule rate denominator");
75 Skip_L1( "granule shift");
76
77 FILLING_BEGIN();
78 Accept("CMML");
79
80 Stream_Prepare(Stream_Text);
81 Fill(Stream_Text, 0, Text_Format, "CMML");
82 Fill(Stream_Text, 0, Text_Codec, "CMML");
83 FILLING_END();
84}
85
86//---------------------------------------------------------------------------
87void File_Cmml::Configuration()
88{
89 Element_Name("Configuration");
90
91 //Parsing
92 Ztring Data;
93 Get_UTF8(Element_Size, Data, "Data");
94
95 FILLING_BEGIN();
96 Ztring Value;
97 Value=Data.SubString(__T("<head>"), __T("</head>"));
98 if (!Value.empty())
99 Fill(Stream_Text, 0, Text_Title, Value.SubString(__T("<title>"), __T("</title>")));
100 if (Data.find(__T("<clip"))!=string::npos)
101 Finish("CMML");
102 FILLING_END();
103}
