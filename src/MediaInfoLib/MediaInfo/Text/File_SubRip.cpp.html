39int64u SubRip_str2timecode(const char* Value)
40{
41 size_t Length=strlen(Value);
42 if (Length>=8
43 && Value[0]>='0' && Value[0]<='9'
44 && Value[1]>='0' && Value[1]<='9'
45 && Value[2]==':'
46 && Value[3]>='0' && Value[3]<='9'
47 && Value[4]>='0' && Value[4]<='9'
48 && Value[5]==':'
49 && Value[6]>='0' && Value[6]<='9'
50 && Value[7]>='0' && Value[7]<='9')
51 {
52 int64u ToReturn=(int64u)(Value[0]-'0')*10*60*60*1000000000
53 +(int64u)(Value[1]-'0') *60*60*1000000000
54 +(int64u)(Value[3]-'0') *10*60*1000000000
55 +(int64u)(Value[4]-'0') *60*1000000000
56 +(int64u)(Value[6]-'0') *10*1000000000
57 +(int64u)(Value[7]-'0') *1000000000;
58 if (Length>=9 && (Value[8]=='.' || Value[8]==','))
59 {
60 if (Length>9+9)
61 Length=9+9; //Nanoseconds max
62 const char* Value_End=Value+Length;
63 Value+=9;
64 int64u Multiplier=100000000;
65 while (Value<Value_End)
66 {
67 ToReturn+=(int64u)(*Value-'0')*Multiplier;
68 Multiplier/=10;
69 Value++;
70 }
71 }
72
73 return ToReturn;
74 }
75 else if (Length>=2
76 && Value[Length-1]=='s')
77 {
78 return (int64u)(atof(Value)*1000000000);
79 }
80 else
81 return (int64u)-1;
82}

111bool File_SubRip::FileHeader_Begin()
112{
113 if (!IsSub && (Buffer_Size<File_Size && Buffer_Size<65536))
114 {
115 Element_WaitForMoreData();
116 return false;
117 }
118
119 ZtringListList List;
120 List.Separator_Set(0, __T("\n\n"));
121 List.Separator_Set(1, __T("\n"));
122
123 if (Buffer[0]==0xEF
124 && Buffer[1]==0xBB
125 && Buffer[2]==0xBF)
126 HasBOM=true;
127 bool IsLocal=false;
128 Ztring Temp;
129 Temp.From_UTF8((const char*)Buffer+(HasBOM?3:0), (Buffer_Size>65536?65536:Buffer_Size)-(HasBOM?3:0));
130 if (Temp.empty())
131 {
132 Temp.From_Local((const char*)Buffer+(HasBOM?3:0), (Buffer_Size>65536?65536:Buffer_Size)-(HasBOM?3:0)); // Trying from local code page
133 IsLocal=true;
134 }
135 Temp.FindAndReplace(__T("\r\n"), __T("\n"), 0, Ztring_Recursive);
136 Temp.FindAndReplace(__T("\r"), __T("\n"), 0, Ztring_Recursive);
137 List.Write(Temp);
138
139 if (List(0, 0)==__T("WEBVTT FILE") || List(0, 0)==__T("WEBVTT"))
140 IsVTT=true;
141
142 if (!IsVTT)
143 {
144 size_t IsOk=0;
145 size_t IsNok=0;
146 for (size_t Pos=0; Pos<List.size(); Pos++)
147 {
148 if (List(Pos, 0).To_int64u()==Pos+1)
149 IsOk++;
150 else
151 IsNok++;
152
153 if (List(Pos, 1).size()>22 && List(Pos, 1)[2]==__T(':') && List(Pos, 1)[5]==__T(':') && List(Pos, 1).find(__T(" --> "))!=string::npos)
154 IsOk++;
155 else
156 IsNok++;
157 }
158
159 if (!IsOk || IsNok>IsOk/2)
160 {
161 Reject();
162 return true;
163 }
164 }
165
166 if (!IsSub && File_Size!=(int64u)-1 && Buffer_Size!=File_Size)
167 {
168 Element_WaitForMoreData();
169 return false;
170 }
171
172 if (!Status[IsAccepted])
173 {
174 Accept();
175 Fill(Stream_General, 0, General_Format, IsVTT?"WebVTT":"SubRip");
176 Stream_Prepare(Stream_Text);
177 Fill(Stream_Text, 0, "Format", IsVTT?"WebVTT":"SubRip");
178 Fill(Stream_Text, 0, "Codec", IsVTT?"WebVTT":"SubRip");
179 }
180
181 if (IsLocal)
182 Temp.From_Local((const char*)Buffer+(HasBOM?3:0), Buffer_Size-(HasBOM?3:0));
183 else
184 Temp.From_UTF8((const char*)Buffer+(HasBOM?3:0), Buffer_Size-(HasBOM?3:0));
185 Temp.FindAndReplace(__T("\r\n"), __T("\n"), 0, Ztring_Recursive);
186 Temp.FindAndReplace(__T("\r"), __T("\n"), 0, Ztring_Recursive);
187 List.Write(Temp);
188
189 #if MEDIAINFO_DEMUX
190 size_t Pos=0;
191 for (;;)
192 {
193 if (Pos>=List.size())
194 break;
195
196 if (List[Pos].size()>=3 || (IsVTT && List[Pos].size()>=2))
197 {
198 Ztring PTS_Begin_String=List[Pos][IsVTT?0:1].SubString(Ztring(), __T(" --> "));
199 Ztring PTS_End_String=List[Pos][IsVTT?0:1].SubString(__T(" --> "), Ztring());
200 if (IsVTT)
201 {
202 size_t Extra_Pos=PTS_End_String.find(__T(' '));
203 if (Extra_Pos!=string::npos)
204 PTS_End_String.resize(Extra_Pos); //Discarding positioning
205 }
206 item Item;
207 Item.PTS_Begin=SubRip_str2timecode(PTS_Begin_String.To_UTF8().c_str());
208 Item.PTS_End=SubRip_str2timecode(PTS_End_String.To_UTF8().c_str());
209 for (size_t Pos2=IsVTT?1:2; Pos2<List[Pos].size(); Pos2++)
210 {
211 List[Pos][Pos2].Trim();
212 Item.Content+=List[Pos][Pos2];
213 if (Pos2+1<List[Pos].size())
214 Item.Content+=EOL;
215 }
216 Items.push_back(Item);
217 }
218
219 Pos++;
220 }
221 #endif //MEDIAINFO_DEMUX
222
223 return true;
224}
