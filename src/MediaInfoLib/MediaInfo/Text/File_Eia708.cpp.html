68//---------------------------------------------------------------------------
69void File_Eia708::Streams_Fill()
70{
71 if (Config->File_Eia708_DisplayEmptyStream_Get() && Streams.size()<2)
72 Streams.resize(2);
73
74 if (ServiceDescriptors)
75 {
76 for (servicedescriptors708::iterator ServiceDescriptor=ServiceDescriptors->ServiceDescriptors708.begin(); ServiceDescriptor!=ServiceDescriptors->ServiceDescriptors708.end(); ++ServiceDescriptor)
77 {
78 service_number=ServiceDescriptor->first;
79 block_size=0;
80 Service();
81 }
82 }
83
84 for (size_t Pos=0; Pos<Streams.size(); Pos++)
85 if (Streams[Pos] || (Pos && Pos<2 && Config->File_Eia708_DisplayEmptyStream_Get()))
86 {
87 Stream_Prepare(Stream_Text);
88 Fill(Stream_Text, StreamPos_Last, Text_ID, Pos);
89 Fill(Stream_Text, StreamPos_Last, "CaptionServiceName", Pos);
90 (*Stream_More)[StreamKind_Last][StreamPos_Last](Ztring().From_Local("CaptionServiceName"), Info_Options)=__T("N NT");
91 Fill(Stream_Text, StreamPos_Last, Text_Format, "EIA-708");
92 Fill(Stream_Text, StreamPos_Last, Text_StreamSize, 0);
93 Fill(Stream_Text, StreamPos_Last, Text_BitRate_Mode, "CBR");
94 if (Config->ParseSpeed>=1.0)
95 {
96 Fill(Stream_Text, StreamPos_Last, "CaptionServiceContent_IsPresent", (DataDetected&((int64u)1)<<Pos)?"Yes":"No", Unlimited, true, true); //1 bit per service
97 (*Stream_More)[Stream_Text][StreamPos_Last](Ztring().From_Local("CaptionServiceContent_IsPresent"), Info_Options)=__T("N NT");
98 }
99 if (ServiceDescriptors)
100 {
101 servicedescriptors708::iterator ServiceDescriptor=ServiceDescriptors->ServiceDescriptors708.find((int8u)Pos);
102 if (ServiceDescriptor!=ServiceDescriptors->ServiceDescriptors708.end())
103 {
104 Fill(Stream_Text, StreamPos_Last, Text_Language, ServiceDescriptor->second.language, true);
105 Fill(Stream_Text, StreamPos_Last, "CaptionServiceDescriptor_IsPresent", "Yes", Unlimited, true, true);
106 (*Stream_More)[Stream_Text][StreamPos_Last](Ztring().From_Local("CaptionServiceDescriptor_IsPresent"), Info_Options)=__T("N NT");
107 }
108 else //ServiceDescriptors pointer is for the support by the transport layer of the info
109 {
110 Fill(Stream_Text, StreamPos_Last, "CaptionServiceDescriptor_IsPresent", "No", Unlimited, true, true);
111 (*Stream_More)[Stream_Text][StreamPos_Last](Ztring().From_Local("CaptionServiceDescriptor_IsPresent"), Info_Options)=__T("N NT");
112 }
113 }
114 }
115}

164//---------------------------------------------------------------------------
165void File_Eia708::Read_Buffer_Unsynched()
166{
167 for (service_number=1; service_number<Streams.size(); service_number++)
168 if (Streams[service_number])
169 {
170 //Per window
171 for (size_t WindowID=0; WindowID<Streams[service_number]->Windows.size(); WindowID++)
172 {
173 window* Window=Streams[service_number]->Windows[WindowID];
174 if (Window)
175 for (size_t Pos_Y=0; Pos_Y<Window->Minimal.CC.size(); Pos_Y++)
176 {
177 for (size_t Pos_X=0; Pos_X<Window->Minimal.CC[Pos_Y].size(); Pos_X++)
178 {
179 Window->Minimal.CC[Pos_Y][Pos_X].Value=L' ';
180 Window->Minimal.CC[Pos_Y][Pos_X].Attribute=0;
181 }
182 }
183 }
184
185 //Global display
186 for (size_t Pos_Y=0; Pos_Y<Streams[service_number]->Minimal.CC.size(); Pos_Y++)
187 {
188 for (size_t Pos_X=0; Pos_X<Streams[service_number]->Minimal.CC[Pos_Y].size(); Pos_X++)
189 {
190 Streams[service_number]->Minimal.CC[Pos_Y][Pos_X].Value=L' ';
191 Streams[service_number]->Minimal.CC[Pos_Y][Pos_X].Attribute=0;
192 }
193 }
194 }
195
196 #if MEDIAINFO_EVENTS
197 for (service_number=1; service_number<Streams.size(); service_number++)
198 if (Streams[service_number])
199 HasChanged();
200 #endif //MEDIAINFO_EVENTS
201}

207//---------------------------------------------------------------------------
208void File_Eia708::Header_Parse()
209{
210 //Parsing
211 int8u packet_size, sequence_number;
212 BS_Begin();
213 Get_S1(2, sequence_number, "sequence_number");
214 Get_S1(6, packet_size, "packet_size_code");
215 BS_End();
216
217 Header_Fill_Code(0, __T("DTVCC packet"));
218 Header_Fill_Size(packet_size==0?128:(packet_size*2));
219}
220
221//---------------------------------------------------------------------------
222void File_Eia708::Data_Parse()
223{
224 //Parsing
225 while (Element_Offset<Element_Size)
226 {
227 BS_Begin();
228 Get_S1(3, service_number, "service_number");
229 Get_S1(5, block_size, "block_size");
230 if (service_number==7)
231 {
232 Mark_0();
233 Mark_0();
234 Get_S1(6, service_number, "extended_service_number");
235 }
236 BS_End();
237
238 if (service_number)
239 {
240 Element_Begin1("Service Block Packet");
241 Service();
242 Element_End0();
243 }
244 }
245}

256 for (int8u Pos_Y=0; Pos_Y<15; Pos_Y++)
257 Streams[service_number]->Minimal.CC[Pos_Y].resize((size_t)(24*AspectRatio));
258 Streams[service_number]->Windows.resize(8);

261 for (int8u Pos=0; Pos<block_size; Pos++)
262 {

321 //CEA-708-D, Section 7.1.10 (C2)
322 if (cc_data_2>=0x08 && cc_data_2<0x10)

324 Skip_B1( "EXT1 1 byte");
325 Pos++;
326 }
327 else if (cc_data_2>=0x10 && cc_data_2<0x18)
