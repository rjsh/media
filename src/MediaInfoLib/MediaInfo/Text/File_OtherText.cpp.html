34void File_OtherText::Read_Buffer_Continue()
35{
36 if (Buffer_Size<0x200)
37 {
38 Element_WaitForMoreData();
39 return;
40 }
41
42 Element_Offset=File_Size-(File_Offset+Buffer_Offset);
43
44 Ztring Format, FormatMore, Codec;
45 Ztring File;
46 ZtringList Lines;
47
48 //Feed File and Lines
49 File.From_UTF8((const char*)Buffer, Buffer_Size>65536?65536:Buffer_Size);
50 if (File.empty())
51 File.From_Local((const char*)Buffer, Buffer_Size>65536?65536:Buffer_Size); // Trying from local code page
52 if (File.size()<0x100)
53 {
54 File.From_Unicode((wchar_t*)Buffer, 0, Buffer_Size/sizeof(wchar_t)); //Unicode with BOM
55 //TODO: Order of bytes (big or Little endian)
56 if (File.size()<0x100)
57 {
58 Reject("Other text");
59 return;
60 }
61 }
62 if (File.size()>0x1000)
63 File.resize(0x1000); //Do not work on too big
64 File.FindAndReplace(__T("\r\n"), __T("\n"), 0, Ztring_Recursive);
65 File.FindAndReplace(__T("\r"), __T("\n"), 0, Ztring_Recursive);
66 Lines.Separator_Set(0, __T("\n"));
67 Lines.Write(File);
68 Lines.resize(0x20);
69
70 if (Lines[0]==__T("[Script Info]")
71 && (Lines.Find(__T("ScriptType: v4.00"))!=Error || Lines.Find(__T("Script Type: V4.00"))!=Error)
72 && Lines.Find(__T("[V4 Styles]"))!=Error
73 )
74 {
75 Format=__T("SSA");
76 FormatMore=__T("SubStation Alpha");
77 Codec=__T("SSA");
78 }
79 else if (Lines[0]==__T("[Script Info]")
80 && (Lines.Find(__T("ScriptType: v4.00+"))!=Error || Lines.Find(__T("Script Type: V4.00+"))!=Error)
81 && Lines.Find(__T("[V4+ Styles]"))!=Error
82 )
83 {
84 Format=__T("ASS");
85 FormatMore=__T("Advanced SubStation Alpha");
86 Codec=__T("ASS");
87 }
88 else if (Lines[0].size()>24
89 && Lines[0][ 0]==__T('0') && Lines[0][ 1]==__T('0')
90 && Lines[0][ 2]==__T(':') && Lines[0][ 5]==__T(':') && Lines[0][ 8]==__T(':')
91 && Lines[0][11]==__T(' ')
92 && Lines[0][12]==__T('0') && Lines[0][13]==__T('0')
93 && Lines[0][14]==__T(':') && Lines[0][17]==__T(':') && Lines[0][20]==__T(':')
94 && Lines[0][23]==__T(' ')
95 )
96 {
97 Format=__T("Adobe encore DVD");
98 Codec=__T("Adobe");
99 }
100 else if (Lines[0].size()==11
101 && Lines[0][0]==__T('-') && Lines[0][1]==__T('-') && Lines[0][2]==__T('>') && Lines[0][3]==__T('>') && Lines[0][4]==__T(' ')
102 && Lines[0][5]==__T('0')
103 && Lines[1].empty()!=true
104 )
105 {
106 Format=__T("AQTitle");
107 Codec=__T("AQTitle");
108 }
109 else if (Lines[0].size()>28
110 && Lines[0][ 0]==__T('0') && Lines[0][ 1]==__T('0')
111 && Lines[0][ 2]==__T(':') && Lines[0][ 5]==__T(':') && Lines[0][ 8]==__T(':')
112 && Lines[0][11]==__T(' ') && Lines[0][12]==__T(',') && Lines[0][13]==__T(' ')
113 && Lines[0][14]==__T('0') && Lines[0][15]==__T('0')
114 && Lines[0][16]==__T(':') && Lines[0][19]==__T(':') && Lines[0][22]==__T(':')
115 && Lines[0][25]==__T(' ') && Lines[0][16]==__T(',') && Lines[0][27]==__T(' ')
116 )
117 {
118 Format=__T("Captions 32");
119 Codec=__T("Caption 32");
120 }
121 else if (Lines[0].size()==23
122 && Lines[0]==__T("*Timecode type: PAL/EBU")
123 && Lines[1].empty()
124 && Lines[2].size()==23
125 && Lines[2][ 0]==__T('0') && Lines[2][ 1]==__T('0')
126 && Lines[2][ 2]==__T(':') && Lines[2][ 5]==__T(':') && Lines[2][ 8]==__T(':')
127 && Lines[2][11]==__T(' ')
128 && Lines[2][12]==__T('0') && Lines[2][13]==__T('0')
129 && Lines[2][14]==__T(':') && Lines[2][17]==__T(':') && Lines[2][20]==__T(':')
130 && Lines[2].size()>0
131 )
132 {
133 Format=__T("Captions Inc");
134 Codec=__T("Captions inc");
135 }
136 else if (Lines[0].size()>1
137 && Lines[0][0]==__T('*')
138 && Lines.Find(__T("** Caption Number 1"))!=Error
139 )
140 {
141 Format=__T("Cheeta");
142 }
143 else if (Lines[0].size()>10
144 && Lines[0][0]==__T('~') && Lines[0][1]==__T('C') && Lines[0][2]==__T('P') && Lines[0][3]==__T('C') && Lines[0][9]==__T('~')
145 && Lines[1][ 0]==__T('0') && Lines[1][ 1]==__T('0')
146 && Lines[1][ 2]==__T(':') && Lines[1][ 5]==__T(':') && Lines[1][ 8]==__T(':')
147 )
148 {
149 Format=__T("CPC Captioning");
150 Codec=__T("CPC Captioning");
151 }
152 else if (Lines[0].find(__T("<SAMI>"))==0)
153 {
154 Format=__T("SAMI");
155 }
156 else
157 return;
158
159 if (Format.empty())
160 return;
161
162 Accept("Other text");
163
164 if (!IsSub)
165 {
166 Fill(Stream_General, 0, General_Format, Format);
167 Fill(Stream_General, 0, General_Format_Info, FormatMore, true);
168 }
169
170 Stream_Prepare(Stream_Text);
171 Fill(Stream_Text, 0, Text_Format, Format);
172 Fill(Stream_Text, 0, Text_Codec, Codec);
173
174 //No more need data
175 Element_Begin1(Format);
176 Element_End0();
177 Finish("Other text");
178}
