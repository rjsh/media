395//---------------------------------------------------------------------------
396void File_AribStdB24B37::Read_Buffer_Continue()
397{
398 if (Buffer_Size==0)
399 return;
400
401 if (IsAncillaryData)
402 {
403 if (!Status[IsAccepted])
404 Accept();
405 int8u DataIdentifier;
406 BS_Begin();
407 Skip_SB( "Error correction");
408 Skip_SB( "Undefined");
409 Skip_SB( "Undefined");
410 Skip_SB( "Undefined");
411 Skip_S1(4, "Continuity Index");
412 Skip_S1(8, "Undefined");
413 Skip_SB( "Undefined");
414 Skip_SB( "Start packet flag");
415 Skip_SB( "End packet flag");
416 Skip_SB( "Send mode");
417 Info_S1(4, Caption_conversion_type, "Format identifier"); Param_Info1(AribStdB24B37_Caption_conversion_type(Caption_conversion_type));
418 Skip_S1(2, "Undefined");
419 Get_S1 (3, DataIdentifier, "Closed caption data identifier"); Param_Info1(AribStdB24B37_Caption_DataIdentifier(DataIdentifier));
420 Info_S1(3, data_group_id, "Language identifier"); if (DataIdentifier) {Param_Info1(AribStdB24B37_data_group_id(data_group_id));}
421 BS_End();
422
423 if (DataIdentifier>6)
424 {
425 Skip_XX(245, "Dummy");
426 }
427 else if (DataIdentifier<4)
428 {
429 Skip_XX(245, "Exchange format data, not supported");
430 }
431 else
432 {
433 Element_Begin1("Short form data");
434 int8u LEN, Label_01, Label_3A, Data_Length;
435 Get_B1(LEN, "LEN");
436 Element_Begin1("display timing");
437 Get_B1(Label_01, "Label (01)");
438 BS_Begin();
439 Skip_S1(6, "Undefined");
440 Skip_S1(2, "Data-type identifier");
441 Skip_S1(6, "Undefined");
442 Skip_S1(2, "Timing-type identifier");
443 Skip_S1(6, "Undefined");
444 Skip_S1(2, "Timing-direction identifier");
445 Skip_B5( "Display timing value");
446 BS_End();
447 Element_End0();
448 Element_Begin1("closed caption data");
449 Get_B1(Label_3A, "Label (3A)");
450 Get_B1(Data_Length, "Data Length");
451 #if defined(MEDIAINFO_MPEGTS_YES)
452 if (Parser==NULL)
453 {
454 Parser=new File_MpegTs;
455 ((File_MpegTs*)Parser)->FromAribStdB24B37=true;
456 Open_Buffer_Init(Parser);
457 }
458 if (FrameInfo.PTS==(int64u)-1)
459 FrameInfo.PTS=FrameInfo.DTS;
460 Parser->FrameInfo=FrameInfo;
461 Open_Buffer_Continue(Parser, Buffer+Buffer_Offset+(size_t)Element_Offset, 188);
462 Element_Offset+=188;
463 #else
464 Skip_XX(188, "TS data");
465 #endif
466 if (Data_Length==192)
467 {
468 Skip_B2( "Group-A CRC");
469 Skip_B2( "Group-B CRC");
470 }
471 else if (Data_Length>188)
472 Skip_XX(Data_Length-188, "Unknown");
473 Element_End0();
474 if (LEN>203)
475 Skip_XX(LEN-204, "User Data");
476 if (LEN<244)
477 Skip_XX(244-LEN, "Unused");
478 Skip_XX(Element_Size-Element_Offset-6, "Format data");
479 Element_End0();
480 }
481
482 Skip_B6( "ECC");
483 return;
484 }
485
486
487 if (ParseCcis)
488 {
489 int32u CCIS_code;
490 Get_C4 ( CCIS_code, "CCIS_code");
491 if (CCIS_code==0xFFFFFFFF)
492 {
493 Skip_XX(Element_Size, "?");
494 return;
495 }
496 Get_B1 ( Caption_conversion_type, "Caption_conversion_type"); Param_Info1(AribStdB24B37_Caption_conversion_type(Caption_conversion_type));
497 BS_Begin();
498 Info_S1(2, DRCS_conversion_type, "DRCS_conversion_type"); Param_Info1(AribStdB24B37_DRCS_conversion_type(DRCS_conversion_type));
499 Skip_S1(6, "reserved");
500 BS_End();
501 Skip_B2( "reserved");
502 Skip_B8( "reserved");
503 ParseCcis=false;
504 return;
505 }
506
507 Skip_B1( "Data_identifier");
508 Skip_B1( "Private_stream_id");
509 BS_Begin();
510 Skip_S1(4, "reserved");
511 Skip_S1(4, "PES_data_packet_header_length");
512 BS_End();
513}

548//---------------------------------------------------------------------------
549void File_AribStdB24B37::Data_Parse()
550{
551 //CRC
552 int16u CRC_CCIT_Xmodem=0x0000;
553 const int8u* CRC_CCIT_Xmodem_Buffer=Buffer+Buffer_Offset-Header_Size; //data_group_id position
554 while(CRC_CCIT_Xmodem_Buffer<Buffer+Buffer_Offset+(size_t)Element_Size) //from data_group_id to the end, CRC included
555 {
556 CRC_CCIT_Xmodem=(CRC_CCIT_Xmodem<<8) ^ AribStdB24B37_CRC_CCIT_Xmodem_Table[(CRC_CCIT_Xmodem>>8)^(*CRC_CCIT_Xmodem_Buffer)];
557 CRC_CCIT_Xmodem_Buffer++;
558 }
559 if (CRC_CCIT_Xmodem)
560 {
561 Skip_XX(Element_Size, "Data");
562 Trusted_IsNot("CRC error");
563 return;
564 }
565
566 Element_Size-=2;
567
568 switch (Element_Code)
569 {
570 case 0 :
571 caption_management(); break;
572 case 1 :
573 case 2 :
574 case 3 :
575 case 4 :
576 case 5 :
577 case 6 :
578 case 7 :
579 case 8 :
580 if (Streams.empty())
581 {
582 Skip_XX(Element_Size, "Waiting for caption_management");
583 break;
584 }
585 if (Element_Code>Streams.size())
586 {
587 Skip_XX(Element_Size, "Unknown service");
588 Trusted_IsNot("Invalid service number");
589 break;
590 }
591
592 Streams[(size_t)(Element_Code-1)].Line.clear();
593 caption_statement();
594 Streams[(size_t)(Element_Code-1)].Line.clear();
595 break;
596 default: Skip_XX(Element_Size, "Unknown");
597 }
598
599 Element_Size+=2;
600 Skip_B2( "CRC_16");
601}

607//---------------------------------------------------------------------------
608void File_AribStdB24B37::caption_management() //caption_management_data()
609{
610 //Parsing
611 int32u data_unit_loop_length;
612 int8u TMD, num_languages;
613 BS_Begin();
614 Get_S1 (2, TMD, "TMD"); Param_Info1(AribStdB24B37_TMD(TMD));
615 Skip_S1(6, "Reserved");
616 if (TMD==2)
617 {
618 Skip_S5(36, "OTM"); //Note: time offset (HHMMSSmmm in BCD format) is currently not supported
619 Skip_S5( 4, "Reserved");
620 }
621 BS_End();
622 Get_B1 (num_languages, "num_languages");
623
624 Streams.clear();
625 Streams.resize(num_languages);
626 for (int8u pos_languages=0; pos_languages<num_languages; pos_languages++)
627 {
628 int8u DMF_reception, Format;
629 string ISO_639_language_code;
630 Element_Begin1("language");
631 BS_Begin();
632 Skip_S1(3, "language_tag");
633 Skip_SB( "Reserved");
634 Get_S1 (2, DMF_reception, "DMF (reception)"); Param_Info1(AribStdB24B37_DMF_reception(DMF_reception)); //Note: Display Method is currently not supported (all is displayed immediatly)
635 Info_S1(2, DMF_recording, "DMF (recording)"); Param_Info1(AribStdB24B37_DMF_recording(DMF_recording));
636 BS_End();
637 if (DMF_reception==3)
638 Skip_B1( "DC");
639 Get_String(3, ISO_639_language_code, "ISO_639_language_code");
640 BS_Begin();
641 Get_S1 (4, Format, "Format"); Param_Info1(AribStdB24B37_format(Format));
642 Info_S1(2, TCS, "TCS"); Param_Info1(AribStdB24B37_TCS(TCS));
643 Info_S1(2, rollup_mode, "rollup_mode"); Param_Info1(AribStdB24B37_rollup_mode(rollup_mode));
644 BS_End();
645 Element_End0();
646
647 FILLING_BEGIN();
648 Streams[pos_languages].ISO_639_language_code=ISO_639_language_code;
649 Streams[pos_languages].DMF_reception=DMF_reception;
650 Streams[pos_languages].Format=Format;
651
652 // Special case
653 if (ISO_639_language_code=="por")
654 {
655 Streams[pos_languages].G[0]=GS_Alphanumeric; //GS_Kanji;
656 Streams[pos_languages].G[1]=GS_Alphanumeric;
657 Streams[pos_languages].G[2]=GS_Alphanumeric; //GS_Hiragana;
658 Streams[pos_languages].G[3]=GS_Alphanumeric; //GS_DRCS|GS_Macro;
659 Streams[pos_languages].G_Width[0]=2;
660 Streams[pos_languages].G_Width[1]=1;
661 Streams[pos_languages].G_Width[2]=1;
662 Streams[pos_languages].G_Width[3]=1;
663 }
664 FILLING_END();
665 }
666 Get_B3 (data_unit_loop_length, "data_unit_loop_length");
667 if (data_unit_loop_length)
668 Skip_XX(data_unit_loop_length, "data_unit");
669
670 FILLING_BEGIN();
671 if (!Status[IsAccepted])
672 Accept();
673 FILLING_END();
674}

676//---------------------------------------------------------------------------
677void File_AribStdB24B37::caption_statement() //caption_data()
678{
679 if (Streams[(size_t)(Element_Code-1)].ISO_639_language_code!="jpn")
680 {
681 Skip_XX(Element_Size-Element_Offset, "Data");
682 return; //Not supported (e.g. Portuguese from Brazil, ISDB-Tb)
683 }
684
685 //Parsing
686 int32u data_unit_loop_length;
687 int8u TMD;
688 BS_Begin();
689 Get_S1 (2, TMD, "TMD"); Param_Info1(AribStdB24B37_TMD(TMD));
690 Skip_S1(6, "Reserved");
691 if (TMD==2)
692 {
693 Skip_S5(36, "STM"); //Note: start time (HHMMSSmmm in BCD format) is currently not supported (and is applicable only if out of band PTS is not avaialble)
694 Skip_S5( 4, "Reserved");
695 }
696 BS_End();
697 Get_B3 (data_unit_loop_length, "data_unit_loop_length");
698
699 if (Element_Offset+data_unit_loop_length!=Element_Size)
700 {
701 Skip_XX(Element_Size-Element_Offset, "Problem");
702 return;
703 }
704
705 while (Element_Offset<Element_Size)
706 {
707 Element_Begin1("data_unit");
708 int8u unit_separator;
709 Get_B1 (unit_separator, "unit_separator"); // Should always be 0x1F?
710 if (unit_separator==0x1F)
711 {
712 int32u data_unit_size;
713 int8u data_unit_parameter;
714 Get_B1 (data_unit_parameter, "data_unit_parameter"); Param_Info1(AribStdB24B37_data_unit_parameter(data_unit_parameter));
715 Get_B3 (data_unit_size, "data_unit_size");
716 switch (data_unit_parameter)
717 {
718 case 0x20 : data_unit_data(Element_Offset+data_unit_size); break;
719 default : Skip_XX(data_unit_size, "(Not implemented)");
720 }
721 }
722 Element_End0();
723 }
724
725 Frame_Count++;
726 Frame_Count_NotParsedIncluded++;
727}

729//---------------------------------------------------------------------------
730void File_AribStdB24B37::data_unit_data(int64u End)
731{
732 Element_Begin1("data_unit_data");
733
734 //data_unit_data_byte
735 while (Element_Offset<End)
736 {
737 int8u header;
738 Peek_B1(header);
739 if (header&0x60) // Not C0 or C1
740 {
741 if ((header&0x7F)==0x20
742 || (header&0x7F)==0x7F)
743 {
744 Skip_C1 ( "Character");
745 Add((Char)header);
746 }
747 else if (header&0x80) // GR //TODO: buffer check //Note about Caption_conversion_type == Mobile closed caption: see B37 Figure 3-4 Code Ranges for Mobile Closed Caption
748 Character(Caption_conversion_type==4?GS_Kanji:Streams[(size_t)(Element_Code-1)].G[Streams[(size_t)(Element_Code-1)].GR],
749 Streams[(size_t)(Element_Code-1)].GR,
750 Buffer[Buffer_Offset+Element_Offset]&0x7F,
751 Buffer[Buffer_Offset+Element_Offset+1]&0x7F);
752 else // GL
753 {
754 Character(Caption_conversion_type==4?GS_DRCS:Streams[(size_t)(Element_Code-1)].G[Streams[(size_t)(Element_Code-1)].GL_SS?Streams[(size_t)(Element_Code-1)].GL_SS:Streams[(size_t)(Element_Code-1)].GL],
755 Streams[(size_t)(Element_Code-1)].GL_SS?Streams[(size_t)(Element_Code-1)].GL_SS:Streams[(size_t)(Element_Code-1)].GL,
756 Buffer[Buffer_Offset+Element_Offset],
757 Buffer[Buffer_Offset+Element_Offset+1]);
758 Streams[(size_t)(Element_Code-1)].GL_SS=0;
759 }
760 }
761 else
762 control_code(); // C0 or C1
763 }
764
765 Element_End0();
766}

771 // Encoding is JIS, but we have only Shift-JIS to Unicode algo, we transform JIS to Shift-JIS (Windows-31J)

773
774 if (Column<32)
775 return; // Problem

777 #ifdef __WINDOWS__

786 int CharSize=MultiByteToWideChar(932, 0, ShiftJIS, 2, Temp, 2); //932 = Shift-JIS (Windows-31J)
787 if (CharSize>0)
788 {
789 Temp[CharSize]=__T('\0');
790 Param_Info1(Ztring().From_Unicode(Temp));
791 Add (Ztring().From_Unicode(Temp));

793 #else // __WINDOWS__
794 //TODO
795 #endif // __WINDOWS__

849//---------------------------------------------------------------------------
850void File_AribStdB24B37::Character (int16u CharacterSet, int8u G_Value, int8u FirstByte, int8u SecondByte)
851{
852 int16u Value=(FirstByte<<8) | SecondByte;
853
854 //ARIB STD B24/B37 provide numbers with 2 number (Row and Colupmn) a shift of 32 (b00100000)
855 #define Compute(f,s) (((f+32)<<8) | (s+32))
856
857 switch (CharacterSet)
858 {
859 case GS_Kanji:
860 Skip_B2( "Character");
861 if (Value<=Compute(84, 6))
862 JIS(FirstByte, SecondByte);
863 else
864 switch (Value)
865 {
866 case Compute(92, 1): JIS( 35, 42); break;
867 case Compute(92, 2): JIS( 35, 43); break;
868 case Compute(92, 3): JIS( 35, 44); break;
869 case Compute(92, 4): JIS( 35, 45); break;
870 case Compute(93, 79): JIS( 40, 110); break;
871 case Compute(93, 88): Param_Info1(Ztring().From_UTF8("\xE2\x99\xAB")+__T(" (not exact)")); Add (Ztring().From_UTF8("\xE2\x99\xAB")); break; //Ending music note?
872 case Compute(93, 89): Param_Info1(Ztring().From_UTF8("\xE2\x99\xAB")+__T(" (not exact)")); Add (Ztring().From_UTF8("\xE2\x99\xAB")); break; //Opening music note?
873 case Compute(93, 90): Param_Info1(Ztring().From_UTF8("\xE2\x99\xAB")); Add (Ztring().From_UTF8("\xE2\x99\xAB")); break; //Music note
874 default: Param_Info1("(Unsupported)"); //empty in spec or not yet mapped
875 }
876 break;
877 case GS_Hiragana:
878 case GS_PropHiragana:
879 Skip_C1( "Character");
880 switch (FirstByte)
881 {
882 case 0x74:
883 case 0x75:
884 case 0x76:
885 Param_Info1("(Unsupported)");
886 break; //empty in spec
887 case 0x77: JIS( 33, 53); break;
888 case 0x78: JIS( 33, 54); break;
889 case 0x79: JIS( 33, 60); break;
890 case 0x7A: JIS( 33, 35); break;
891 case 0x7B: JIS( 33, 86); break;
892 case 0x7C: JIS( 33, 87); break;
893 case 0x7D: JIS( 33, 34); break;
894 case 0x7E: JIS( 33, 38); break;
895 default : JIS( 36, FirstByte);
896 }
897 break;
898 case GS_Katakana:
899 case GS_PropKatakana:
900 Skip_C1( "Character");
901 switch (FirstByte)
902 {
903 case 0x77: JIS( 33, 41); break;
904 case 0x78: JIS( 33, 42); break;
905 case 0x79: JIS( 33, 51); break;
906 case 0x7A: JIS( 33, 33); break;
907 case 0x7B: JIS( 33, 86); break;
908 case 0x7C: JIS( 33, 87); break;
909 case 0x7D: JIS( 33, 34); break;
910 case 0x7E: JIS( 33, 38); break;
911 default : JIS( 37, FirstByte); break;
912 }
913 break;
914 case GS_Alphanumeric:
915 case GS_PropAscii:
916 Skip_C1( "Character");
917 Add(FirstByte);
918 break;
919 case GS_DRCS|GS_Macro :
920 DefaultMacro();
921 break;
922 default : ;
923 switch (Streams[(size_t)(Element_Code-1)].G_Width[G_Value])
924 {
925 case 1 : Skip_C1("Character (unsupported)"); break;
926 case 2 : Skip_C2("Character (unsupported)"); break;
927 default: Skip_XX(Streams[(size_t)(Element_Code-1)].G_Width[G_Value], "Character (unsupported)");
928 }
929 }
930}

1357void File_AribStdB24B37::CSI()
1358{
1359 Element_Begin1("CSI - Extended Control Codes");
1360 Skip_B1( "control_code");
1361
1362 vector<int64u> Params;
1363 Params.push_back(0);
1364 size_t Pos=0;
1365
1366 while (Element_Offset+Pos<=Element_Size)
1367 {
1368 int8u Value=Buffer[Buffer_Offset+Element_Offset+Pos];
1369 Pos++;
1370 if (Value==0x3B)
1371 Params.push_back(0);
1372 else if (Value>=0x30 && Value<0x3A)
1373 {
1374 Params[Params.size()-1]*=10;
1375 Params[Params.size()-1]+=Value&0x0F;
1376 }
1377 else if (Value>=0x40 && Value<0x80)
1378 {
1379 Pos--;
1380 Skip_Local(Pos, "Values");
1381 Get_B1(Value, "Delimiter");
1382 switch (Value)
1383 {
1384 case 0x3B:
1385 Element_Info1("SRC - Raster Colour Designation");
1386 break;
1387 case 0x42:
1388 Element_Info1("GSM - Character deformation");
1389 break;
1390 case 0x53:
1391 Element_Info1("SWF - Set Writing Format");
1392 if (!Params.empty() && Params[0]<0x100)
1393 Streams[(size_t)(Element_Code-1)].Format=(int8u)Params[0];
1394 break;
1395 case 0x54:
1396 Element_Info1("CCC - Composite Character Composition");
1397 break;
1398 case 0x56:
1399 Element_Info1("SDF - Set Display Format");
1400 break;
1401 case 0x57:

1404 case 0x58:
