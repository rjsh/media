83//---------------------------------------------------------------------------
84void File_DvbSubtitle::Streams_Fill()
85{
86 Stream_Prepare(Stream_Text);
87 Fill(Stream_Text, 0, Text_Format, "DVB Subtitle");
88
89 for (std::map<int8u, subtitle_stream_data>::iterator subtitle_stream=subtitle_streams.begin(); subtitle_stream!=subtitle_streams.end(); ++subtitle_stream)
90 for (std::map<int16u, page_data>::iterator page=subtitle_stream->second.pages.begin(); page!=subtitle_stream->second.pages.end(); ++page)
91 for (std::map<int8u, region_data>::iterator region=page->second.regions.begin(); region!=page->second.regions.end(); ++region)
92 {
93 Fill(Stream_Text, 0, "subtitle_stream_id", subtitle_stream->first);
94 (*Stream_More)[Stream_Text][0](Ztring().From_Local("subtitle_stream_id"), Info_Options)=__T("N NI");
95 Fill(Stream_Text, 0, "page_id", page->first);
96 (*Stream_More)[Stream_Text][0](Ztring().From_Local("page_id"), Info_Options)=__T("N NI");
97 Fill(Stream_Text, 0, "region_id", region->first);
98 (*Stream_More)[Stream_Text][0](Ztring().From_Local("region_id"), Info_Options)=__T("N NI");
99 Fill(Stream_Text, 0, "region_horizontal_address", region->second.page_composition_segment?Ztring::ToZtring(region->second.region_horizontal_address):Ztring());
100 (*Stream_More)[Stream_Text][0](Ztring().From_Local("region_horizontal_address"), Info_Options)=__T("N NI");
101 Fill(Stream_Text, 0, "region_vertical_address", region->second.page_composition_segment?Ztring::ToZtring(region->second.region_vertical_address):Ztring());
102 (*Stream_More)[Stream_Text][0](Ztring().From_Local("region_vertical_address"), Info_Options)=__T("N NI");
103 Fill(Stream_Text, 0, "region_width", region->second.region_composition_segment?Ztring::ToZtring(region->second.region_width):Ztring());
104 (*Stream_More)[Stream_Text][0](Ztring().From_Local("region_width"), Info_Options)=__T("N NI");
105 Fill(Stream_Text, 0, "region_height", region->second.region_composition_segment?Ztring::ToZtring(region->second.region_height):Ztring());
106 (*Stream_More)[Stream_Text][0](Ztring().From_Local("region_height"), Info_Options)=__T("N NI");
107 Fill(Stream_Text, 0, "region_depth", region->second.region_composition_segment?Ztring::ToZtring(DvbSubtitle_region_depth[region->second.region_depth]):Ztring());
108 (*Stream_More)[Stream_Text][0](Ztring().From_Local("region_depth"), Info_Options)=__T("N NI");
109 }
110}

121//---------------------------------------------------------------------------
122bool File_DvbSubtitle::Synchronize()
123{
124 //Synchronizing
125 if (MustFindDvbHeader)
126 {
127 while(Buffer_Offset+3<=Buffer_Size)
128 {
129 if (Buffer[Buffer_Offset]==0x20
130 && Buffer[Buffer_Offset+1]==0x00
131 && (Buffer[Buffer_Offset+2]==0x0F
132 || Buffer[Buffer_Offset+1]==0xFF))
133 break;
134 Buffer_Offset++;
135 }
136
137 if (Buffer_Offset+3>Buffer_Size)
138 return false;
139
140 Accept();
141 }
142 else
143 {
144 while(Buffer_Offset<Buffer_Size)
145 {
146 if (Buffer[Buffer_Offset]==0x0F
147 || Buffer[Buffer_Offset]==0xFF)
148 break;
149 Buffer_Offset++;
150 }
151
152 if (Buffer_Offset>=Buffer_Size)
153 return false;
154 }
155
156 //Synched is OK
157 Synched=true;
158 return true;
159}

161//---------------------------------------------------------------------------
162bool File_DvbSubtitle::Synched_Test()
163{
164 if (MustFindDvbHeader)
165 {
166 //Must have enough buffer for having header
167 if (Buffer_Offset+1>Buffer_Size)
168 return false;
169
170 if (CC2(Buffer+Buffer_Offset)!=0x2000)
171 {
172 Synched=false;
173 return true;
174 }
175
176 //Displaying it
177 Element_Size=2;
178 Skip_B1( "data_identifier");
179 Get_B1 (subtitle_stream_id, "subtitle_stream_id");
180 Buffer_Offset+=2;
181 MustFindDvbHeader=false;
182 }
183
184 //Must have enough buffer for having header
185 if (Buffer_Offset+1>Buffer_Size)
186 return false;
187
188 //Quick test of synchro
189 if (Buffer[Buffer_Offset]!=0x0F
190 && Buffer[Buffer_Offset]!=0xFF)
191 {
192 Synched=false;
193 return true;
194 }
195
196 //We continue
197 return true;
198}

212#if MEDIAINFO_DEMUX
213bool File_DvbSubtitle::Demux_UnpacketizeContainer_Test()
214{
215 if (Demux_Offset==0)
216 {
217 Demux_Offset=Buffer_Offset;
218 }
219 while (Demux_Offset<Buffer_Size)
220 {
221 bool MustBreak;
222 switch (Buffer[Demux_Offset])
223 {
224 case 0xFF :
225 MustBreak=true; break; //0xFF is not in the demuxed frame
226 default : MustBreak=false;
227 }
228 if (MustBreak)
229 break; //while() loop
230
231 if (Demux_Offset+6>Buffer_Size)
232 return false; //No complete frame
233
234 int16u segment_length=BigEndian2int16u(Buffer+Demux_Offset+4);
235 Demux_Offset+=6+segment_length;
236
237 if (Demux_Offset>=Buffer_Size)
238 return false; //No complete frame
239 }
240
241 if (Demux_Offset>=Buffer_Size)
242 return false; //No complete frame
243
244 Demux_UnpacketizeContainer_Demux();
245
246 Demux_TotalBytes++; //0xFF is not demuxed
247
248 return true;
249}

256//---------------------------------------------------------------------------
257void File_DvbSubtitle::Header_Parse()
258{
259 //Parsing
260 int8u sync_byte;
261 Get_B1 (sync_byte, "sync_byte");
262 switch (sync_byte)
263 {
264 case 0xFF : //Stuffing
265 MustFindDvbHeader=true;
266
267 //Filling
268 Header_Fill_Code(0xFF, "end of PES data field marker");
269 Header_Fill_Size(1);
270 return;
271 default : ; //Normal (0x0F)
272 }
273
274 int16u segment_length;
275 int8u segment_type;
276 Get_B1 (segment_type, "segment_type");
277 Get_B2 (page_id, "page_id");
278 Get_B2 (segment_length, "segment_length");
279
280 //Filling
281 Header_Fill_Code(segment_type);
282 Header_Fill_Size(Element_Offset+segment_length);
283}
284
285//---------------------------------------------------------------------------
286void File_DvbSubtitle::Data_Parse()
287{
288 switch (Element_Code)
289 {
290 case 0x10 : page_composition_segment(); break;
291 case 0x11 : region_composition_segment(); break;
292 case 0x12 : CLUT_definition_segment(); break;
293 case 0x13 : object_data_segment(); break;
294 case 0x14 : display_definition_segment(); break;
295 case 0x80 : end_of_display_set_segment(); break;
296 case 0xFF : end_of_PES_data_field_marker(); return;
297 default :
298 if (Element_Code>=0x40 && Element_Code<=0x7F)
299 reserved_for_future_use();
300 else if (Element_Code>=0x81 && Element_Code<=0xEF)
301 private_data();
302 else if (Element_Size)
303 Skip_XX(Element_Size, "Unknown");
304 }
305}

311//---------------------------------------------------------------------------
312void File_DvbSubtitle::page_composition_segment()
313{
314 Element_Name("page composition segment");
315
316 //Parsing
317 Skip_B1( "page_time_out");
318 BS_Begin();
319 Skip_S1(4, "page_version_number");
320 Skip_S1(2, "page_state");
321 Skip_S1(2, "reserved");
322 BS_End();
323 while(Element_Offset<Element_Size)
324 {
325 Element_Begin1("Region");
326 int16u region_horizontal_address, region_vertical_address;
327 int8u region_id;
328 Get_B1 (region_id, "region_id");
329 Skip_B1( "reserved");
330 Get_B2 (region_horizontal_address, "region_horizontal_address");
331 Get_B2 (region_vertical_address, "region_vertical_address");
332 Element_End0();
333
334 FILLING_BEGIN();
335 subtitle_streams[subtitle_stream_id].pages[page_id].regions[region_id].page_composition_segment=true;
336 subtitle_streams[subtitle_stream_id].pages[page_id].regions[region_id].region_horizontal_address=region_horizontal_address;
337 subtitle_streams[subtitle_stream_id].pages[page_id].regions[region_id].region_vertical_address=region_vertical_address;
338 FILLING_END();
339 }
340}

342//---------------------------------------------------------------------------
343void File_DvbSubtitle::region_composition_segment()
344{
345 Element_Name("region composition segment");
346
347 //Parsing
348 int16u region_width, region_height;
349 int8u region_id, region_depth;
350 Get_B1 ( region_id, "region_id");
351 BS_Begin();
352 Skip_S1(4, "region_version_number");
353 Skip_S1(1, "region_fill_flag");
354 Skip_S1(3, "reserved");
355 BS_End();
356 Get_B2 ( region_width, "region_width");
357 Get_B2 ( region_height, "region_height");
358 BS_Begin();
359 Skip_S1(3, "region_level_of_compatibility");
360 Get_S1 (3, region_depth, "region_depth"); Param_Info2(DvbSubtitle_region_depth[region_depth], " bits");
361 Skip_S1(2, "reserved");
362 BS_End();
363 Skip_B1( "CLUT_id");
364 Skip_B1( "region_8-bit_pixel_code");
365 BS_Begin();
366 Skip_S1(4, "region_4-bit_pixel-code");
367 Skip_S1(2, "region_2-bit_pixel-code");
368 Skip_S1(2, "reserved");
369 BS_End();
370 while(Element_Offset<Element_Size)
371 {
372 Element_Begin1("Object");
373 int8u object_type;
374 Skip_B2( "object_id");
375 BS_Begin();
376 Get_S1 ( 2, object_type, "object_type");
377 Skip_S1( 2, "object_provider_flag");
378 Skip_S1(12, "object_horizontal_position");
379 Skip_S1( 4, "reserved");
380 Skip_S1(12, "object_vertical_position");
381 BS_End();
382 switch (object_type)
383 {
384 case 0x01 :
385 case 0x02 :
386 Skip_B2( "foreground_pixel_code");
387 Skip_B2( "background_pixel_code");
388 break;
389 default : ;
390 }
391 Element_End0();
392 }
393
394 FILLING_BEGIN();
395 subtitle_streams[subtitle_stream_id].pages[page_id].regions[region_id].region_composition_segment=true;
396 subtitle_streams[subtitle_stream_id].pages[page_id].regions[region_id].region_width=region_width;
397 subtitle_streams[subtitle_stream_id].pages[page_id].regions[region_id].region_height=region_height;
398 subtitle_streams[subtitle_stream_id].pages[page_id].regions[region_id].region_depth=region_depth;
399 FILLING_END();
400}
