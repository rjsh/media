52//---------------------------------------------------------------------------
53void File_TimedText::Header_Parse()
54{
55 //Parsing
56 int16u Size;
57 Get_B2 (Size, "Size");
58
59 //Filling
60 Header_Fill_Code(0, "Block");
61 Header_Fill_Size(Element_Offset+Size);
62
63 //TODO: if IsChapter, it may be UTF-16 (with BOM), it may also be followed by an encd atom (e.g. for UTF-8 00 00 00 0C 65 6E 63 64 00 00 01 00)
64}
65
66//---------------------------------------------------------------------------
67void File_TimedText::Data_Parse()
68{
69 //Parsing
70 Ztring Value;
71 Get_UTF8 (Element_Size, Value, "Value");
72
73 FILLING_BEGIN();
74 if (!Status[IsAccepted])
75 {
76 Accept();
77 #ifdef MEDIAINFO_MPEG4_YES
78 if (IsChapter)
79 {
80 Stream_Prepare(Stream_Menu);
81 }
82 else
83 #endif //MEDIAINFO_MPEG4_YES
84 {
85 Stream_Prepare(Stream_Text);
86 }
87 Fill(StreamKind_Last, StreamPos_Last, Fill_Parameter(StreamKind_Last, Generic_Format), "Timed Text");
88
89 #ifdef MEDIAINFO_MPEG4_YES
90 if (IsChapter)
91 Fill(StreamKind_Last, StreamPos_Last, Menu_Chapters_Pos_Begin, Count_Get(StreamKind_Last, StreamPos_Last), 10, true);
92 #endif //MEDIAINFO_MPEG4_YES
93
94 }
95 #ifdef MEDIAINFO_MPEG4_YES
96 if (IsChapter)
97 {
98 }
99 else
100 #endif //MEDIAINFO_MPEG4_YES
101 {
102 Finish();
103 }
104
105 #ifdef MEDIAINFO_MPEG4_YES
106 if (IsChapter && FrameInfo.DTS!=(int64u)-1 && Buffer_Offset==2)
107 {
108 Fill(Stream_Menu, StreamPos_Last, Ztring().Duration_From_Milliseconds(FrameInfo.DTS/1000000).To_UTF8().c_str(), Value);
109 Fill(Stream_Menu, StreamPos_Last, Menu_Chapters_Pos_End, Count_Get(Stream_Menu, StreamPos_Last), 10, true);
110 }
111 #endif //MEDIAINFO_MPEG4_YES
112 FILLING_END();
113
114 Element_Offset=Buffer_Size-Buffer_Offset; //Buffer can also contain atoms after the text, ignoring them
115}
116
117} //NameSpace
118
119#endif //MEDIAINFO_TIMEDTEXT_YES
