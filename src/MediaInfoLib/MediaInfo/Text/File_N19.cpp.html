223//---------------------------------------------------------------------------
224bool File_N19::FileHeader_Begin()
225{
226 //Element_Size
227 if (Buffer_Size<11)
228 return false; //Must wait for more data
229
230 if (Buffer[ 3]!=0x53
231 || Buffer[ 4]!=0x54
232 || Buffer[ 5]!=0x4C
233 || Buffer[ 8]!=0x2E
234 || Buffer[ 9]!=0x30
235 || Buffer[10]!=0x31) // "STLxx.01"
236 {
237 Reject("N19");
238 return false;
239 }
240
241 //Element_Size
242 if (Buffer_Size<1024)
243 return false; //Must wait for more data about GSI
244
245 //All should be OK...
246 return true;
247}

249//---------------------------------------------------------------------------
250void File_N19::FileHeader_Parse()
251{
252 Element_Name("General Subtitle Information");
253
254 //Parsing
255 Ztring OPT, RD, TNS, MNC, MNR, CO, EN;
256 string TCP;
257 int16u LC;
258 int8u TCS;
259 Info_C3 ( CPN, "CPN - Code Page Number"); Param_Info1(N19_CodePageNumber(CPN));
260 Get_C8 ( DFC, "DFC - Disk Format Code"); Param_Info1(N19_DiskFormatCode_FrameRate(DFC));
261 Info_C1 ( DSC, "DSC - Display Standard Code"); Param_Info1(N19_DisplayStandardCode(DSC));
262 Get_C2 ( CCT, "CCT - Character Code Table number"); Param_Info1(N19_CharacterCodeTable(CCT));
263 Get_C2 ( LC, "LC - Language Code"); Param_Info1(N19_LanguageCode(LC));
264 Get_Local (32, OPT, "OPT - Original Programme Title");
265 Skip_Local(32, "OET - Original Episode Title");
266 Skip_Local(32, "TPT - Translated Programme");
267 Skip_Local(32, "TET - Translated Episode");
268 Skip_Local(32, "TN - Translator's Name");
269 Skip_Local(32, "TCD - Translator's Contact Details");
270 Skip_Local(16, "SLR - Subtitle List Reference Code");
271 Skip_Local( 6, "CD - Creation Date");
272 Get_Local ( 6, RD, "RD - Revision Date");
273 Skip_C2 ( "RN - Revision number");
274 Skip_C5 ( "TNB - Total Number of Text and Timing Information (TTI) blocks");
275 Get_Local ( 5, TNS, "TNS - Total Number of Subtitles");
276 Skip_C3 ( "TNG - Total Number of Subtitle Groups");
277 Get_Local ( 2, MNC, "MNC - Maximum Number of Displayable Characters in any text row");
278 Get_Local ( 2, MNR, "MNR - Maximum Number of Displayable Rows");
279 Get_C1 ( TCS, "TCS - Time Code: Status");
280 Get_String( 8, TCP, "TCP - Time Code: Start-of-Programme");
281 Skip_Local( 8, "TCF - Time Code: First In-Cue");
282 Skip_C1 ( "TND - Total Number of Disks");
283 Skip_C1 ( "DSN - Disk Sequence Number");
284 Get_Local ( 3, CO, "CO - Country of Origin");
285 Skip_Local(32, "PUB - Publisher");
286 Get_Local (32, EN, "EN - Editor's Name");
287 Skip_Local(32, "ECD - Editor's Contact Details");
288 Skip_XX(75, "Spare Bytes");
289 Skip_XX(576, "UDA - User-Defined Area");
290
291 FILLING_BEGIN();
292 Accept("N19");
293
294 Fill(Stream_General, 0, General_Format, "N19");
295 Fill(Stream_General, 0, General_Title, OPT);
296 RD.insert(0, __T("20"));
297 RD.insert(4, __T("-"));
298 RD.insert(7, __T("-"));
299 Fill(Stream_General, 0, General_Recorded_Date, RD);
300 Fill(Stream_General, 0, General_Country, Ztring(CO).MakeLowerCase());
301 Fill(Stream_General, 0, General_DistributedBy, EN);
302
303 Stream_Prepare(Stream_Text);
304 Fill(Stream_Text, 0, Text_Format, "N19");
305 if (N19_DiskFormatCode_FrameRate(DFC))
306 {
307 Fill(Stream_Text, 0, "FrameRate", N19_DiskFormatCode_FrameRate(DFC));
308 if (TCS==0x31 && TCP.size()==8
309 && TCP[0]>='0' && TCP[0]<='9'
310 && TCP[1]>='0' && TCP[1]<='9'
311 && TCP[2]>='0' && TCP[2]<='6'
312 && TCP[3]>='0' && TCP[3]<='9'
313 && TCP[4]>='0' && TCP[4]<='6'
314 && TCP[5]>='0' && TCP[5]<='9'
315 && TCP[6]>='0' && TCP[6]<='2'
316 && TCP[7]>='0' && TCP[7]<='9')
317 {
318 int64u Delay=0;
319 Delay+=(((int32u)TCP[0])-'0')*10*60*60*1000;
320 Delay+=(((int32u)TCP[1])-'0')* 60*60*1000;
321 Delay+=(((int32u)TCP[2])-'0')* 10*60*1000;
322 Delay+=(((int32u)TCP[3])-'0')* 60*1000;
323 Delay+=(((int32u)TCP[4])-'0')* 10*1000;
324 Delay+=(((int32u)TCP[5])-'0')* 1000;
325 int8u Frames=0;
326 Frames+=(((int8u)TCP[6])-'0')*10;
327 Frames+=(((int8u)TCP[7])-'0');
328 Delay+=float64_int64s(Frames*1000/N19_DiskFormatCode_FrameRate(DFC));
329 //Fill(Stream_Text, 0, Text_Delay, Delay); //TODO is 0???
330 /*TCP.insert(':', 2);
331 TCP.insert(':', 5);
332 TCP.insert(':', 8);
333 Fill(Stream_Text, 0, "Delay/String4", TCP);*/
334 }
335 }
336 Fill(Stream_Text, 0, Text_Width, MNC.To_int32u());
337 Fill(Stream_Text, 0, Text_Height, MNR.To_int32u());
338 Fill(Stream_Text, 0, Text_Language, N19_LanguageCode(LC));
339
340 //Init
341 FirstFrame_TCI=(int64u)-1;
342 #if MEDIAINFO_DEMUX
343 Frame_Count=0;
344 TCO_Latest=(int64u)-1;
345 #endif //MEDIAINFO_DEMUX
346 FILLING_END();
347}

379//---------------------------------------------------------------------------
380void File_N19::Data_Parse()
381{
382 //Parsing
383 Ztring TF;
384 int32u TCI, TCO;
385 Skip_B1 ( "SGN - Subtitle Group Number");
386 Skip_B2 ( "SN - Subtitle Number");
387 Skip_B1 ( "EBN - Extension Block Number");
388 Skip_B1 ( "CS - Cumulative Status");
389 Get_B4 (TCI, "TCI - Time Code In");
390 TCI=((TCI>>24)&0xFF)*60*60*1000
391 + ((TCI>>16)&0xFF) *60*1000
392 + ((TCI>>8 )&0xFF) *1000
393 + float64_int64s((TCI &0xFF) *1000/N19_DiskFormatCode_FrameRate(DFC));
394 Param_Info1(Ztring().Duration_From_Milliseconds((int64u)TCI));
395 Get_B4 (TCO, "TCO - Time Code Out");
396 TCO=((TCO>>24)&0xFF)*60*60*1000
397 + ((TCO>>16)&0xFF) *60*1000
398 + ((TCO>>8 )&0xFF) *1000
399 + float64_int64s((TCO &0xFF) *1000/N19_DiskFormatCode_FrameRate(DFC));
400 Param_Info1(Ztring().Duration_From_Milliseconds((int64u)TCO));
401 Skip_B1 ( "VP - Vertical Position");
402 Skip_B1 ( "JC - Justification Code");
403 Skip_B1 ( "CF - Comment Flag");
404 switch (CCT)
405 {
406 case 0x3030 : //Latin ISO 6937-2
407 Get_ISO_6937_2(112, TF, "TF - Text Field");
408 break;
409 case 0x3031 : //Latin ISO 8859-5
410 Get_ISO_8859_5(112, TF, "TF - Text Field");
411 break;
412 default:
413 //Not yet supported, basic default
414 Get_ISO_8859_1(112, TF, "TF - Text Field");
415 }
416 #if MEDIAINFO_TRACE
417 for (size_t i = 0; i < TF.size(); ++i)
418 switch (TF[i])
419 {
420 case 0x8A: //EOL
421 TF[i] = EOL[0];
422 {
423 size_t j = 1;
424 while (EOL[j] != __T('\0'))
425 TF.insert(++i, 1, EOL[j++]);
426 };
427 break;
428 case 0x8F: // Padding
429 TF.erase(i--, 1);
430 break;
431 default:
432 if ( TF[i]< 0x20
433 || (TF[i]>=0x80 && TF[i]<0xA0))
434 TF.erase(i--, 1);
435 }
436 Param_Info1(TF);
437 #endif //MEDIAINFO_TRACE
438
439 FILLING_BEGIN();
440 if (FirstFrame_TCI==(int64u)-1)
441 {
442 FirstFrame_TCI=TCI;
443 Fill(Stream_Text, 0, Text_Delay, TCI);
444 Fill(Stream_Text, 0, Text_Delay_Source, "Container");
445 }
446 if (File_Offset+Buffer_Offset+Element_Size+128>File_Size)
447 {
448 Fill(Stream_Text, 0, Text_Duration, TCO-FirstFrame_TCI);
449 }
450 else if (Config->ParseSpeed<1.0)
451 //Jumping
452 GoToFromEnd(128, "N19");
453 FILLING_END();
454}
