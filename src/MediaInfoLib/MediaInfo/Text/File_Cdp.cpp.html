45const char* Cdp_cc_type(int8u cc_type)
46{
47 switch (cc_type)
48 {
49 case 0 : return "CEA-608 line 21 field 1 closed captions"; //closed caption 3 if this is second field
50 case 1 : return "CEA-608 line 21 field 2 closed captions"; //closed caption 4 if this is second field
51 case 2 : return "DTVCC Channel Packet Data";
52 case 3 : return "DTVCC Channel Packet Start";
53 default : return "";
54 }
55}

57//---------------------------------------------------------------------------
58float32 Cdp_cdp_frame_rate(int8u cdp_frame_rate)
59{
60 switch (cdp_frame_rate)
61 {
62 case 1 : return (float32)23.976;
63 case 2 : return (float32)24.000;
64 case 3 : return (float32)25.000;
65 case 4 : return (float32)29.970;
66 case 5 : return (float32)30.000;
67 case 6 : return (float32)50.000;
68 case 7 : return (float32)59.940;
69 case 8 : return (float32)60.000;
70 default : return (float32) 0.000;
71 }
72}

137//---------------------------------------------------------------------------
138void File_Cdp::Streams_Update()
139{
140 Clear(Stream_Text);
141
142 //Per stream
143 for (size_t Pos=0; Pos<Streams.size(); Pos++)
144 if (Streams[Pos] && Streams[Pos]->Parser && Streams[Pos]->Parser->Status[IsFilled] /*&& Streams[Pos]->Parser->Status[IsUpdated]*/ && Streams[Pos]->Parser->Count_Get(Stream_Text))
145 Streams_Update_PerStream(Pos);
146}
147
148//---------------------------------------------------------------------------
149void File_Cdp::Streams_Update_PerStream(size_t Pos)
150{
151 if (Streams[Pos]==NULL)
152 return;
153
154 Update(Streams[Pos]->Parser);
155
156 if (Streams[Pos]->Parser)
157 {
158 for (size_t Pos2=0; Pos2<Streams[Pos]->Parser->Count_Get(Stream_Text); Pos2++)
159 {
160 Stream_Prepare(Stream_Text);
161 Merge(*Streams[Pos]->Parser, Stream_Text, Pos2, StreamPos_Last);
162 if (WithAppleHeader)
163 Fill(Stream_Text, StreamPos_Last, "MuxingMode", "Final Cut");
164 Fill(Stream_Text, StreamPos_Last, "MuxingMode", "CDP");
165 if (cdp_frame_rate!=(int8u)-1)
166 Fill(Stream_Text, StreamPos_Last, Text_FrameRate, Cdp_cdp_frame_rate(cdp_frame_rate));
167 Fill(Stream_Text, StreamPos_Last, Text_ID, Streams[Pos]->Parser->Retrieve(Stream_Text, Pos2, Text_ID), true);
168
169 //cdp_length
170 if (cdp_length_Min<=cdp_length_Max)
171 {
172 Fill(Stream_Text, StreamPos_Last, "cdp_length_Min", cdp_length_Min, 10, true);
173 Fill(Stream_Text, StreamPos_Last, "cdp_length_Max", cdp_length_Max, 10, true);
174 (*Stream_More)[Stream_Text][StreamPos_Last](Ztring().From_Local("cdp_length_Min"), Info_Options)=__T("N NT");
175 (*Stream_More)[Stream_Text][StreamPos_Last](Ztring().From_Local("cdp_length_Max"), Info_Options)=__T("N NT");
176 }
177 }
178
179 Ztring LawRating=Streams[Pos]->Parser->Retrieve(Stream_General, 0, General_LawRating);
180 if (!LawRating.empty())
181 Fill(Stream_General, 0, General_LawRating, LawRating, true);
182 Ztring Title=Streams[Pos]->Parser->Retrieve(Stream_General, 0, General_Title);
183 if (!Title.empty() && Retrieve(Stream_General, 0, General_Title).empty())
184 Fill(Stream_General, 0, General_Title, Title);
185 }
186}

188//---------------------------------------------------------------------------
189void File_Cdp::Streams_Finish()
190{
191 Clear(Stream_Text);
192
193 //Per stream
194 for (size_t Pos=0; Pos<Streams.size(); Pos++)
195 if (Streams[Pos] && Streams[Pos]->Parser && Streams[Pos]->Parser->Status[IsAccepted] /*&& Streams[Pos]->Parser->Status[IsUpdated]*/)
196 {
197 Finish(Streams[Pos]->Parser);
198 Streams_Update_PerStream(Pos);
199 }
200}

219//---------------------------------------------------------------------------
220bool File_Cdp::Synchronize()
221{
222 //Synchronizing
223 while (Buffer_Offset+3<=Buffer_Size)
224 {
225 while (Buffer_Offset+3<=Buffer_Size && (Buffer[Buffer_Offset ]!=0x96
226 || Buffer[Buffer_Offset+1]!=0x69))
227 {
228 Buffer_Offset++;
229 while (Buffer_Offset<Buffer_Size && Buffer[Buffer_Offset]!=0x96)
230 Buffer_Offset++;
231 }
232
233 if (Buffer_Offset+3>Buffer_Size)
234 break;
235
236 //CRC
237 int8u CRC=0;
238 int8u cdp_length=Buffer[Buffer_Offset+2];
239
240 //Must have enough buffer for having header
241 if (Buffer_Offset+cdp_length>Buffer_Size)
242 return false;
243
244 const int8u* Buffer_Temp=Buffer+Buffer_Offset;
245 const int8u* Buffer_End=Buffer+Buffer_Offset+cdp_length;
246 while (Buffer_Temp<Buffer_End)
247 CRC+=*(Buffer_Temp++);
248 if (!CRC)
249 break;
250
251 Buffer_Offset++;
252 }
253
254 //Parsing last bytes if needed
255 if (Buffer_Offset+3>Buffer_Size)
256 {
257 if (Buffer_Offset+3==Buffer_Size && CC2(Buffer+Buffer_Offset)!=0x9669)
258 Buffer_Offset++;
259 if (Buffer_Offset+2==Buffer_Size && CC2(Buffer+Buffer_Offset)!=0x9669)
260 Buffer_Offset++;
261 if (Buffer_Offset+1==Buffer_Size && CC1(Buffer+Buffer_Offset)!=0x96)
262 Buffer_Offset++;
263 return false;
264 }
265
266 //Synched is OK
267 return true;
268}

270//---------------------------------------------------------------------------
271bool File_Cdp::Synched_Test()
272{
273 //Must have enough buffer for having header
274 if (Buffer_Offset+3>Buffer_Size)
275 return false;
276
277 //Quick test of synchro
278 if (Buffer[Buffer_Offset ]!=0x96
279 || Buffer[Buffer_Offset+1]!=0x69)
280 Synched=false;
281
282 if (Synched)
283 {
284 //CRC
285 int8u CRC=0;
286 int8u cdp_length=Buffer[Buffer_Offset+2];
287
288 //Must have enough buffer for having header
289 if (Buffer_Offset+cdp_length>Buffer_Size)
290 return false;
291
292 const int8u* Buffer_Temp=Buffer+Buffer_Offset;
293 const int8u* Buffer_End=Buffer+Buffer_Offset+cdp_length;
294 while (Buffer_Temp<Buffer_End)
295 CRC+=*(Buffer_Temp++);
296 if (CRC)
297 Synched=false;
298 }
299
300 //We continue
301 return true;
302}

308//---------------------------------------------------------------------------
309void File_Cdp::Header_Parse()
310{
311 //Parsing
312 int32u cdp_identifier_length; // For trace feature, we don't want to parse with classic Get() and cdp_identifier was already tested during Sync test
313 Peek_B3(cdp_identifier_length);
314
315 Header_Fill_Code(0, "CDP packet");
316 Header_Fill_Size(cdp_identifier_length&0xFF);
317}

319//---------------------------------------------------------------------------
320void File_Cdp::Data_Parse()
321{
322 if (!Status[IsAccepted])
323 Accept("CDP");
324
325 while(Element_Offset<Element_Size)
326 {
327 if (!IsSub)
328 Element_Begin1("CDP");
329
330 cdp_header();
331 int64u End=cdp_length;
332 if (End>Element_Size)
333 End=Element_Size;
334 while(Element_Offset<End)
335 {
336 int8u section_id;
337 Peek_L1(section_id);
338 switch (section_id)
339 {
340 case 0x71 : time_code_section(); break;
341 case 0x72 : ccdata_section(); break;
342 case 0x73 : ccsvcinfo_section(); break;
343 case 0x74 : cdp_footer(); break;
344 case 0xFF : Skip_B1("Padding?"); break;
345 default : if (section_id>=0x75 && section_id<=0xEF)
346 future_section();
347 else
348 Skip_XX(Element_Size-Element_Offset, "Unknown");
349 }
350 }
351
352 if (!IsSub)
353 Element_End1("CDP");
354 }
355
356 FILLING_BEGIN();
357 Frame_Count++;
358 if (!IsSub && Config->ParseSpeed<1.0 && Frame_Count>=300)
359 Finish();
360 FILLING_END();
361}

363//---------------------------------------------------------------------------
364void File_Cdp::Read_Buffer_Continue()
365{
366 if (Buffer_Size==0)
367 return;
368
369 if (WithAppleHeader)
370 {
371 int32u Size, Magic;
372 Get_B4 (Size, "Size");
373 Get_B4 (Magic, "Magic");
374
375 FILLING_BEGIN();
376 if (Magic!=0x63636470)
377 {
378 Reject("CDP");
379 return;
380 }
381 FILLING_END();
382 }
383}

389//---------------------------------------------------------------------------
390void File_Cdp::cdp_header()
391{
392 Element_Begin1("cdp_header");
393 int16u cdp_identifier;
394 Get_B2 ( cdp_identifier, "cdp_identifier");
395 Get_B1 ( cdp_length, "cdp_length");
396 BS_Begin();
397 Get_S1 (4, cdp_frame_rate, "cdp_frame_rate"); Param_Info1(Ztring::ToZtring(Cdp_cdp_frame_rate(cdp_frame_rate))+__T(" fps"));
398 Skip_S1(4, "Reserved");
399 Skip_SB( "time_code_present");
400 Skip_SB( "ccdata_present");
401 Skip_SB( "svcinfo_present");
402 Skip_SB( "svc_info_start");
403 Skip_SB( "svc_info_change");
404 Skip_SB( "svc_info_complete");
405 Skip_SB( "caption_service_active");
406 Skip_SB( "Reserved");
407 BS_End();
408 Skip_B2( "cdp_hdr_sequence_cntr");
409 Element_End0();
410
411 FILLING_BEGIN();
412 //cdp_length
413 if (cdp_length>cdp_length_Max)
414 cdp_length_Max=cdp_length;
415 if (cdp_length<cdp_length_Min)
416 cdp_length_Min=cdp_length;
417 FILLING_END();
418}

420//---------------------------------------------------------------------------
421void File_Cdp::time_code_section()
422{
423 Element_Begin1("time_code_section");
424 Skip_B1( "time_code_section_id");
425 BS_Begin();

434 Skip_S1(3, "tc_10sec");

444//---------------------------------------------------------------------------
445void File_Cdp::ccdata_section()
446{
447 //Parsing
448 int8u cc_count;
449 Element_Begin1("ccdata_section");
450 Skip_B1( "ccdata_id");
451 BS_Begin();
452 Mark_1();
453 Mark_1();
454 Mark_1();
455 Get_S1 (5, cc_count, "cc_count");
456 BS_End();
457 for (int8u Pos=0; Pos<cc_count; Pos++)
458 {
459 Element_Begin1("cc");
460 int8u cc_type;
461 bool cc_valid;
462 BS_Begin();
463 Mark_1();
464 Mark_1();
465 Mark_1();
466 Mark_1();
467 Mark_1();
468 Get_SB ( cc_valid, "cc_valid");
469 Get_S1 (2, cc_type, "cc_type"); Param_Info1(Cdp_cc_type(cc_type));
470 BS_End();
471
472 #if MEDIAINFO_ADVANCED
473 if (cc_type>=2 && !Streams[2] && Config->File_Eia708_DisplayEmptyStream_Get())
474 CreateStream(2);
475 #endif //MEDIAINFO_ADVANCED
476
477 if (cc_valid)
478 {
479 Element_Begin1("cc_data");
480 //Calculating the parser position
481 int8u Parser_Pos=cc_type==3?2:cc_type; //cc_type 2 and 3 are for the same text
482
483 //Stream creation
484 if (Streams[Parser_Pos]==NULL)
485 CreateStream(Parser_Pos);
486
487 //Parsing
488 #if MEDIAINFO_DEMUX
489 Element_Code=Parser_Pos;
490 Demux(Buffer+(size_t)(Buffer_Offset+Element_Offset), 2, ContentType_MainStream);
491 #endif //MEDIAINFO_DEMUX
492 if (!Streams[Parser_Pos]->Parser->Status[IsFinished])
493 {
494 if (Streams[Parser_Pos]->Parser->PTS_DTS_Needed)
495 {
496 Streams[Parser_Pos]->Parser->FrameInfo.PCR=FrameInfo.PCR;
497 Streams[Parser_Pos]->Parser->FrameInfo.PTS=FrameInfo.PTS;
498 Streams[Parser_Pos]->Parser->FrameInfo.DTS=FrameInfo.DTS;
499 }
500 if (Parser_Pos==2)
501 {
502 #if defined(MEDIAINFO_EIA708_YES)
503 ((File_Eia708*)Streams[2]->Parser)->cc_type=cc_type;
504 if (AspectRatio)
505 ((File_Eia708*)Streams[2]->Parser)->AspectRatio=AspectRatio;
506 #endif //defined(MEDIAINFO_EIA708_YES)
507 }
508 #if defined(MEDIAINFO_EIA608_YES) || defined(MEDIAINFO_EIA708_YES)
509 Streams[Parser_Pos]->Parser->ServiceDescriptors=ServiceDescriptors;
510 #endif
511 Open_Buffer_Continue(Streams[Parser_Pos]->Parser, Buffer+(size_t)(Buffer_Offset+Element_Offset), 2);
512 Element_Offset+=2;
513
514 //Filled
515 if (!Streams[Parser_Pos]->IsFilled && Streams[Parser_Pos]->Parser->Status[IsFilled])
516 {
517 if (Count_Get(Stream_General)==0)
518 Accept("CDP");
519 Streams_Count++;
520 if (Streams_Count==3)
521 Fill("CDP");
522 Streams[Parser_Pos]->IsFilled=true;
523 }
524 }
525 else
526 Skip_XX(2, "Data");
527 Element_End0();
528 }
529 else
530 Skip_XX(2, "Junk");
531 Element_End0();
532 }
533 Element_End0();
534
535 Frame_Count++;
536 Frame_Count_InThisBlock++;
537 if (Frame_Count_NotParsedIncluded!=(int64u)-1)
538 Frame_Count_NotParsedIncluded++;
539}

541//---------------------------------------------------------------------------
542void File_Cdp::ccsvcinfo_section()
543{
544 //Parsing
545 int8u svc_count;
546 Element_Begin1("ccsvcinfo_section");
547 Skip_B1( "ccsvcinfo_id");
548 BS_Begin();
549 Skip_SB( "reserved");
550 Skip_SB( "svc_info_start");
551 Skip_SB( "svc_info_change");
552 Skip_SB( "svc_info_complete");
553 Get_S1 (4, svc_count, "svc_count");
554 BS_End();
555 for (int8u Pos=0; Pos<svc_count; Pos++)
556 {
557 Element_Begin1("svc");
558 bool csn_size;
559 BS_Begin();
560 Skip_SB( "reserved");
561 Get_SB ( csn_size, "csn_size");
562 if (csn_size)
563 {
564 Skip_SB( "reserved");
565 Skip_S1(5, "caption_service_number");
566 }
567 else
568 Skip_S1(6, "caption_service_number");
569 BS_End();
570
571 //svc_data_byte - caption_service_descriptor
572 Element_Begin1("service");
573 string language;
574 int8u caption_service_number=0;
575 bool digital_cc, line21_field=false;
576 Get_String(3, language, "language");
577 BS_Begin();
578 Get_SB (digital_cc, "digital_cc");
579 Skip_SB( "reserved");
580 if (digital_cc) //line21
581 Get_S1 (6, caption_service_number, "caption_service_number");
582 else
583 {
584 Skip_S1(5, "reserved");
585 Get_SB ( line21_field, "line21_field");
586
587 //Coherency test
588 if (line21_field && svc_count==1)
589 line21_field=false; // Wrong info in the descriptor?
590 }
591 Skip_SB( "easy_reader");
592 Skip_SB( "wide_aspect_ratio");
593 Skip_S2(14, "reserved");
594 BS_End();
595 Element_End0();
596 Element_End0();
597
598 FILLING_BEGIN();
599 #if defined(MEDIAINFO_EIA608_YES) || defined(MEDIAINFO_EIA708_YES)
600 if (digital_cc)
601 {
602 #if defined(MEDIAINFO_EIA708_YES)
603 ServiceDescriptors->ServiceDescriptors708[caption_service_number].language=language;
604 #endif
605 }
606 else
607 {
608 #if defined(MEDIAINFO_EIA708_YES)
609 ServiceDescriptors->ServiceDescriptors608[line21_field?1:0].language=language;
610 #endif
611 }
612 #endif
613
614 //Stream creation
615 int8u Parser_Pos;
616 if (digital_cc) //line21
617 Parser_Pos = 2;
618 else
619 Parser_Pos= (line21_field ? 1 : 0); //cc_type 2 and 3 are for the same text
620 if (Streams[Parser_Pos]==NULL)
621 CreateStream(Parser_Pos);
622 FILLING_END();
623 }
624 Element_End0();
625}

653//---------------------------------------------------------------------------
654void File_Cdp::CreateStream(int8u Parser_Pos)
655{
656 if (Streams[Parser_Pos])
657 return; //Already done
658
659 //Parsing
660 #if MEDIAINFO_DEMUX
661 Element_Code=Parser_Pos;
662 #endif //MEDIAINFO_DEMUX
663
664 //Creation of the parser
665 Streams[Parser_Pos]=new stream;
666 #if defined(MEDIAINFO_EIA608_YES) || defined(MEDIAINFO_EIA708_YES)
667 if (Parser_Pos<2)
668 {
669 #if defined(MEDIAINFO_EIA608_YES)
670 Streams[Parser_Pos]->Parser=new File_Eia608();
671 ((File_Eia608*)Streams[Parser_Pos]->Parser)->cc_type=Parser_Pos;
672 #else //defined(MEDIAINFO_EIA608_YES)
673 Streams[Parser_Pos]->Parser=new File__Analyze();
674 #endif //defined(MEDIAINFO_EIA608_YES)
675 }
676 else
677 {
678 #if defined(MEDIAINFO_EIA708_YES)
679 Streams[Parser_Pos]->Parser=new File_Eia708();
680 #else //defined(MEDIAINFO_EIA708_YES)
681 Streams[Parser_Pos]->Parser=new File__Analyze();
682 #endif //defined(MEDIAINFO_EIA708_YES)
683 }
684 #else //defined(MEDIAINFO_EIA608_YES) || defined(MEDIAINFO_EIA708_YES)
685 Streams[Parser_Pos]->Parser=new File__Analyze();
686 #endif //defined(MEDIAINFO_EIA608_YES) || defined(MEDIAINFO_EIA708_YES)
687
688 //Init
689 Streams[Parser_Pos]->Parser->ServiceDescriptors=ServiceDescriptors;
690 Open_Buffer_Init(Streams[Parser_Pos]->Parser);
691 Streams[Parser_Pos]->Parser->Accept();
692}
