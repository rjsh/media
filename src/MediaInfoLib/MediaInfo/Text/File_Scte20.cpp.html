44const char* Scte20_field_number (int8u field_number)
45{
46 switch (field_number)
47 {
48 case 0 : return "Forbidden";
49 case 1 : return "1st display field";
50 case 2 : return "2nd display field";
51 case 3 : return "3rd display field";
52 default : return "";
53 }
54}

61File_Scte20::File_Scte20()

95//---------------------------------------------------------------------------
96void File_Scte20::Streams_Update()
97{
98 Clear(Stream_Text);
99
100 //Per stream
101 for (size_t Pos=0; Pos<Streams.size(); Pos++)
102 if (Streams[Pos] && Streams[Pos]->Parser && Streams[Pos]->Parser->Status[IsFilled] /*&& Streams[Pos]->Parser->Status[IsUpdated]*/ && Streams[Pos]->Parser->Count_Get(Stream_Text))
103 Streams_Update_PerStream(Pos);
104}
105
106//---------------------------------------------------------------------------
107void File_Scte20::Streams_Update_PerStream(size_t Pos)
108{
109 if (Streams[Pos]==NULL)
110 return;
111
112 Update(Streams[Pos]->Parser);
113
114 if (Streams[Pos]->Parser)
115 {
116 for (size_t Pos2=0; Pos2<Streams[Pos]->Parser->Count_Get(Stream_Text); Pos2++)
117 {
118 Stream_Prepare(Stream_Text);
119 Merge(*Streams[Pos]->Parser, Stream_Text, Pos2, StreamPos_Last);
120 Fill(Stream_Text, StreamPos_Last, "MuxingMode", "SCTE 20");
121 Fill(Stream_Text, StreamPos_Last, Text_ID, Streams[Pos]->Parser->Retrieve(Stream_Text, Pos2, Text_ID), true);
122 }
123
124 Ztring LawRating=Streams[Pos]->Parser->Retrieve(Stream_General, 0, General_LawRating);
125 if (!LawRating.empty())
126 Fill(Stream_General, 0, General_LawRating, LawRating, true);
127 Ztring Title=Streams[Pos]->Parser->Retrieve(Stream_General, 0, General_Title);
128 if (!Title.empty() && Retrieve(Stream_General, 0, General_Title).empty())
129 Fill(Stream_General, 0, General_Title, Title);
130 }
131}
132
133//---------------------------------------------------------------------------
134void File_Scte20::Streams_Finish()
135{
136 Clear(Stream_Text);
137
138 //Per stream
139 for (size_t Pos=0; Pos<Streams.size(); Pos++)
140 if (Streams[Pos] && Streams[Pos]->Parser && Streams[Pos]->Parser->Status[IsAccepted] /*&& Streams[Pos]->Parser->Status[IsUpdated]*/)
141 {
142 Finish(Streams[Pos]->Parser);
143 Streams_Update_PerStream(Pos);
144 }
145}

172//***************************************************************************
173// Buffer - Global
174//***************************************************************************
175
176static inline int8u ReverseBits(int8u c)
177{
178 // Input: bit order is 76543210
179 //Output: bit order is 01234567
180 c = (c & 0x0F) << 4 | (c & 0xF0) >> 4;
181 c = (c & 0x33) << 2 | (c & 0xCC) >> 2;
182 c = (c & 0x55) << 1 | (c & 0xAA) >> 1;
183 return c;
184}

186//---------------------------------------------------------------------------
187void File_Scte20::Read_Buffer_Continue()
188{
189 //Parsing
190 Element_Begin1("SCTE 20");
191 int8u cc_count;
192 bool vbi_data_flag;
193 BS_Begin();
194 Mark_1_NoTrustError();
195 Mark_0_NoTrustError();
196 Mark_0_NoTrustError();
197 Mark_0_NoTrustError();
198 Mark_0_NoTrustError();
199 Mark_0_NoTrustError();
200 Mark_0_NoTrustError();
201 Get_SB (vbi_data_flag, "vbi_data_flag");
202 if (vbi_data_flag)
203 {
204 Get_S1 (5, cc_count, "cc_count");
205 for (int8u Pos=0; Pos<cc_count; Pos++)
206 {
207 Element_Begin1("cc");
208 int8u cc_data[2];
209 int8u field_number, cc_data_1, cc_data_2;
210 Skip_S1(2, "cc_priority");
211 Get_S1 (2, field_number, "field_number"); Param_Info1(Scte20_field_number(field_number));
212 Skip_S1(5, "line_offset");
213 Get_S1 (8, cc_data_1, "cc_data_1");
214 cc_data[0]=ReverseBits(cc_data_1);
215 Param_Info1(Ztring::ToZtring(cc_data[0], 16));
216 Get_S1 (8, cc_data_2, "cc_data_2");
217 cc_data[1]=ReverseBits(cc_data_2);
218 Param_Info1(Ztring::ToZtring(cc_data[1], 16));
219 Mark_1_NoTrustError();
220 if (field_number!=0 && picture_structure!=(int8u)-1 && picture_structure!=0)
221 {
222 Element_Begin1("cc_data");
223
224 //Finding the corresponding cc_type (CEA-608 1st field or 2nd field)
225 int8u cc_type;
226 if (progressive_sequence)
227 cc_type=0;
228 else if (picture_structure!=3)
229 cc_type=picture_structure-1;
230 else if (field_number==2)
231 cc_type=top_field_first?1:0;
232 else //if (field_number==1 || field_number==3)
233 cc_type=top_field_first?0:1;
234
235 //Parsing
236 #if MEDIAINFO_DEMUX
237 Element_Code=cc_type;
238 #endif //MEDIAINFO_DEMUX
239 if (Streams[cc_type]==NULL)
240 Streams[cc_type]=new stream;
241 if (Streams[cc_type]->Parser==NULL)
242 {
243 #if defined(MEDIAINFO_EIA608_YES)
244 Streams[cc_type]->Parser=new File_Eia608();
245 ((File_Eia608*)Streams[cc_type]->Parser)->cc_type=cc_type;
246 #else //defined(MEDIAINFO_EIA608_YES)
247 Streams[cc_type]->Parser=new File__Analyze();
248 #endif //defined(MEDIAINFO_EIA608_YES)
249 Open_Buffer_Init(Streams[cc_type]->Parser);
250 }
251 Demux(cc_data, 2, ContentType_MainStream);
252 if (!Streams[cc_type]->Parser->Status[IsFinished])
253 {
254 //Parsing
255 if (Streams[cc_type]->Parser->PTS_DTS_Needed)
256 {
257 Streams[cc_type]->Parser->FrameInfo.PCR=FrameInfo.PCR;
258 Streams[cc_type]->Parser->FrameInfo.PTS=FrameInfo.PTS;
259 Streams[cc_type]->Parser->FrameInfo.DTS=FrameInfo.DTS;
260 }
261 Open_Buffer_Continue(Streams[cc_type]->Parser, cc_data, 2);
262 Element_Show();
263
264 //Filled
265 if (!Status[IsAccepted])
266 Accept("SCTE 20");
267 }
268 else
269 Skip_XX(2, "Data");
270 Element_End0();
271 }
272 Element_End0();
273 }
274 }
275 Skip_S1(4, "non_real_time_video_count");
276 BS_End();
277
278 if (Element_Size-Element_Offset)
279 Skip_XX(Element_Size-Element_Offset, "non_real_time_video + reserved");
280 Element_End0();
281 Element_Show();
282
283 FILLING_BEGIN();
284 //Filled
285 if (!Status[IsAccepted] && Eia608_DisplayEmptyStream)
286 {
287 Accept("SCTE 20");
288
289 //TODO: merge duplicated content
290 for (int8u cc_type=0; cc_type<2; cc_type++)
291 {
292 //Parsing
293 #if MEDIAINFO_DEMUX
294 Element_Code=cc_type;
295 #endif //MEDIAINFO_DEMUX
296 if (Streams[cc_type]==NULL)
297 Streams[cc_type]=new stream;
298 if (Streams[cc_type]->Parser==NULL)
299 {
300 #if defined(MEDIAINFO_EIA608_YES)
301 Streams[cc_type]->Parser=new File_Eia608();
302 ((File_Eia608*)Streams[cc_type]->Parser)->cc_type=cc_type;
303 #else //defined(MEDIAINFO_EIA608_YES)
304 Streams[cc_type]->Parser=new File__Analyze();
305 #endif //defined(MEDIAINFO_EIA608_YES)
306 Open_Buffer_Init(Streams[cc_type]->Parser);
307
308 Streams[cc_type]->Parser->Accept();
309 }
310 }
311 }
312 FILLING_END();
313}
