42Ztring Kate_Category(const Ztring &Category)
43{
44 //http://wiki.xiph.org/index.php/OggText#Categories_of_Text_Codecs
45 if (Category==__T("CC"))
46 return __T("Closed caption");
47 if (Category==__T("SUB"))
48 return __T("Subtitles");
49 if (Category==__T("TAD"))
50 return __T("Textual audio descriptions");
51 if (Category==__T("KTV"))
52 return __T("Karaoke");
53 if (Category==__T("TIK"))
54 return __T("Ticker text");
55 if (Category==__T("AR"))
56 return __T("Active regions");
57 if (Category==__T("NB"))
58 return __T("Semantic annotations");
59 if (Category==__T("META"))
60 return __T("Metadata, mostly machine-readable");
61 if (Category==__T("TRX"))
62 return __T("Transcript");
63 if (Category==__T("LRC"))
64 return __T("Lyrics");
65 if (Category==__T("LIN"))
66 return __T("Linguistic markup");
67 if (Category==__T("CUE"))
68 return __T("Cue points");
69
70 //From Kate
71 if (Category==__T("K-SLD-I"))
72 return __T("Slides, as images");
73 if (Category==__T("K-SLD-T"))
74 return __T("Slides, as text");
75 return Category;
76}

93//---------------------------------------------------------------------------
94void File_Kate::Identification()
95{
96 Element_Name("Identification");
97
98 //Parsing
99 Ztring Language, Category;
100 int16u Width, Height;
101 int8u VersionMajor, VersionMinor, NumHeaders, TextEncoding;
102 Skip_B1 ( "Signature");
103 Skip_Local(7, "Signature");
104 Skip_L1( "Reserved");
105 Get_L1 (VersionMajor, "version major");
106 Get_L1 (VersionMinor, "version minor");
107 Get_L1 (NumHeaders, "num headers");
108 Get_L1 (TextEncoding, "text encoding");
109 Skip_L1( "directionality");
110 Skip_L1( "Reserved");
111 Skip_L1( "granule shift");
112 Skip_L4( "Reserved");
113 Get_L2 (Width, "cw sh + canvas width");
114 Get_L2 (Height, "ch sh + canvas height");
115 /*
116 BS_Begin();
117 Skip_BS( 4, "cw sh");
118 Get_BS (12, Width, "canvas width");
119 Skip_BS( 4, "ch sh");
120 Get_BS (12, Height, "canvas height");
121 BS_End();
122 */
123 Skip_L4( "granule rate numerator");
124 Skip_L4( "granule rate denominator");
125 Get_UTF8(16, Language, "Language");
126 Get_UTF8(16, Category, "Category");
127
128 FILLING_BEGIN();
129 Accept("Kate");
130
131 Stream_Prepare(Stream_Text);
132 Fill(Stream_Text, 0, Text_Format, "Kate");
133 Fill(Stream_Text, 0, Text_Codec, "Kate");
134 Fill(Stream_Text, 0, Text_Language, Language);
135 Fill(Stream_Text, 0, Text_Language_More, Kate_Category(Category));
136
137 Finish("Kate");
138 FILLING_END();
139}
