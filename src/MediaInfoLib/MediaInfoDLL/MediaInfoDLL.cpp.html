48struct mi_output
49{
50 std::string Ansi; //One-Byte-sized characters
51 std::wstring Unicode; //Unicode characters
52};
53typedef std::map<void*, mi_output*> mi_outputs;
54
55struct mi_input
56{
57 Ztring Unicode[8]; //Unicode characters multiple times
58};
59typedef std::map<void*, mi_input*> mi_inputs;
60
61mi_outputs MI_Outputs;
62mi_inputs MI_Inputs;
63
64static CriticalSection Critical;
65static bool utf8=false;
66
67//---------------------------------------------------------------------------
68const char* WC2MB(void* Handle, const wchar_t* Text)
69{
70 //Coherancy
71 Critical.Enter();
72 mi_outputs::iterator MI_Output=MI_Outputs.find(Handle);
73 if (MI_Outputs.find(Handle)==MI_Outputs.end())
74 {
75 MI_Outputs[Handle]=new mi_output; //Generic Handle
76 MI_Output=MI_Outputs.find(Handle);
77 }
78 Critical.Leave();
79
80 //Adaptation
81 if (utf8)
82 MI_Output->second->Ansi=Ztring(Text).To_UTF8();
83 else
84 MI_Output->second->Ansi=Ztring(Text).To_Local();
85 return MI_Output->second->Ansi.c_str();
86}
87
88//---------------------------------------------------------------------------
89const wchar_t* MB2WC(void* Handle, size_t Pos, const char* Text)
90{
91 //Coherancy
92 Critical.Enter();
93 mi_inputs::iterator MI_Input=MI_Inputs.find(Handle);
94 if (MI_Input==MI_Inputs.end())
95 {
96 MI_Inputs[Handle]=new mi_input; //Generic Handle
97 MI_Input=MI_Inputs.find(Handle);
98 }
99 Critical.Leave();
100
101 //Adaptation
102 if (utf8)
103 return MI_Input->second->Unicode[Pos].From_UTF8(Text).c_str();
104 else
105 return MI_Input->second->Unicode[Pos].From_Local(Text).c_str();
106}

558void* __stdcall MediaInfo_New ()
559{
560 #ifdef MEDIAINFO_DEBUG
561 Debug_Open(false);
562 Debug+=", New, Build="; Debug+=__DATE__; Debug+=' '; Debug+=__TIME__;
563 Debug_Close();
564 #endif //MEDIAINFO_DEBUG
565
566 //First init
567 Critical.Enter();
568 if (MI_Outputs.find(NULL)==MI_Outputs.end())
569 {
570 MI_Outputs[NULL]=new mi_output; //Generic Handle
571 }
572 Critical.Leave();
573
574 //New
575 MediaInfo* Handle=NULL;
576 try
577 {
578 Handle=new MediaInfo;
579 }
580 catch(...)
581 {
582 MEDIAINFO_DEBUG2( "New",
583 Debug+="!!!Exception thrown!!!";)
584
585 delete Handle;
586 return NULL;
587 }
588
589 Critical.Enter();
590 MI_Outputs[Handle]=new mi_output;
591 Critical.Leave();
592
593 MEDIAINFO_DEBUG2( "New",
594 Debug+=", returns ";Debug+=Ztring::ToZtring((size_t)Handle).To_UTF8();)
595
596 return Handle;
597}

612void __stdcall MediaInfo_Delete (void* Handle)
613{
614 INTEGRITY_VOID( "Delete"
615 ,)
616
617 //Delete the object

619
620 //Delete strings
621 Critical.Enter();
622 delete MI_Outputs[Handle];
623 MI_Outputs.erase(Handle);

625 {

627 MI_Outputs.erase(NULL);

629 Critical.Leave();

632 )
